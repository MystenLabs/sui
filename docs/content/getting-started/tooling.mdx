export const SearchCtx = React.createContext({ query: '', setQuery: () => {} });

export const SearchProvider = ({ children }) => {
  const [query, setQuery] = React.useState('');
  const ref = React.useRef(null);
  const resultsRef = React.useRef(null);
  React.useEffect(() => {
    if (!ref.current) return;
    const container = ref.current;
    if (query) {
      Array.from(container.children).forEach((child) => {
        if (child.classList.contains('tool-search') || child === resultsRef.current) return;
        child.style.display = 'none';
      });
      if (!resultsRef.current) {
        const div = document.createElement('div');
        div.className = 'tool-grid';
        container.appendChild(div);
        resultsRef.current = div;
      }
      const results = resultsRef.current;
      results.style.display = '';
      results.innerHTML = '';
      const q = query.toLowerCase();
      container.querySelectorAll('.tool-card[data-search]').forEach((card) => {
        if (card.closest('.tool-grid') === results) return;
        const text = card.getAttribute('data-search').toLowerCase();
        if (text.includes(q)) {
          const clone = card.cloneNode(true);
          results.appendChild(clone);
        }
      });
      if (results.children.length === 0) {
        results.innerHTML = '<p style="grid-column:1/-1;text-align:center;color:var(--ifm-color-emphasis-600)">No tools found.</p>';
      }
    } else {
      Array.from(container.children).forEach((child) => {
        if (child === resultsRef.current) {
          child.style.display = 'none';
          return;
        }
        child.style.display = '';
      });
    }
  }, [query]);
  return (
    <SearchCtx.Provider value={{ query, setQuery }}>
      <div ref={ref}>
        {children}
      </div>
    </SearchCtx.Provider>
  );
};
