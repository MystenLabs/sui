4use crate::Benchmark;
5use rand::Rng;
6
7pub struct Server;
8
9#[anemo::async_trait]
10impl Benchmark for Server {
11    async fn send_bytes(
12        &self,
13        _request: anemo::Request<Vec<u8>>,
14    ) -> Result<anemo::Response<()>, anemo::rpc::Status> {
15        Ok(anemo::Response::new(()))
16    }
17
18    async fn request_bytes(
19        &self,
20        request: anemo::Request<u32>,
21    ) -> Result<anemo::Response<Vec<u8>>, anemo::rpc::Status> {
22        let rng = rand::thread_rng();
23        Ok(anemo::Response::new(
24            rng.sample_iter(rand::distributions::Standard)
25                .take(request.into_inner() as usize)
26                .collect(),
27        ))
28    }
29}
