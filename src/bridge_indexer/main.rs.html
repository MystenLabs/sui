47#[derive(Parser, Clone, Debug)]
48struct Args {
49    /// Path to a yaml config
50    #[clap(long, short)]
51    config_path: Option<PathBuf>,
52}

54#[tokio::main]
55async fn main() -> Result<()> {
56    let _guard = telemetry_subscribers::TelemetryConfig::new()
57        .with_env()
58        .init();
59
60    let args = Args::parse();
61
62    // load config
63    let config_path = if let Some(path) = args.config_path {
64        path
65    } else {
66        env::current_dir()
67            .expect("Couldn't get current directory")
68            .join("config.yaml")
69    };
70    let config = IndexerConfig::load(&config_path)?;
71
72    // Init metrics server
73    let metrics_address =
74        SocketAddr::new(IpAddr::V4(Ipv4Addr::new(0, 0, 0, 0)), config.metric_port);
75    let registry_service = start_prometheus_server(metrics_address);
76    let registry = registry_service.default_registry();
77    mysten_metrics::init_metrics(&registry);
78    info!("Metrics server started at port {}", config.metric_port);
79
80    let indexer_meterics = BridgeIndexerMetrics::new(&registry);
81    let ingestion_metrics = DataIngestionMetrics::new(&registry);
82    let bridge_metrics = Arc::new(BridgeMetrics::new(&registry));
83
84    let db_url = config.db_url.clone();
85    let pool = get_connection_pool(db_url.clone()).await;
86
87    let eth_client: Arc<EthClient<MeteredEthHttpProvier>> = Arc::new(
88        EthClient::<MeteredEthHttpProvier>::new(
89            &config.eth_rpc_url,
90            HashSet::from_iter(vec![]), // dummy
91            bridge_metrics.clone(),
92        )
93        .await?,
94    );
95    let eth_bridge_proxy_address = EthAddress::from_str(&config.eth_sui_bridge_contract_address)?;
96    let mut tasks = vec![];
97    // Start the eth subscription indexer
98    let eth_subscription_indexer = create_eth_subscription_indexer(
99        pool.clone(),
100        indexer_meterics.clone(),
101        &config,
102        eth_client.clone(),
103    )
104    .await?;
105    tasks.push(spawn_logged_monitored_task!(
106        eth_subscription_indexer.start()
107    ));
108
109    // Start the eth sync data source
110    let eth_sync_indexer = create_eth_sync_indexer(
111        pool.clone(),
112        indexer_meterics.clone(),
113        bridge_metrics.clone(),
114        &config,
115        eth_client,
116    )
117    .await?;
118    tasks.push(spawn_logged_monitored_task!(eth_sync_indexer.start()));
119
120    if !config.eth_only {
121        let indexer =
122            create_sui_indexer(pool, indexer_meterics, ingestion_metrics, &config).await?;
123        tasks.push(spawn_logged_monitored_task!(indexer.start()));
124    }
125
126    let sui_bridge_client =
127        Arc::new(SuiBridgeClient::new(&config.sui_rpc_url, bridge_metrics.clone()).await?);
128    start_watchdog(
129        config,
130        eth_bridge_proxy_address,
131        sui_bridge_client,
132        &registry,
133        bridge_metrics.clone(),
134    )
135    .await?;
136
137    // Wait for tasks in `tasks` to finish. Return when anyone of them returns an error.
138    futures::future::try_join_all(tasks).await?;
139    unreachable!("Indexer tasks finished unexpectedly");
140}

142async fn start_watchdog(
143    config: IndexerConfig,
144    eth_bridge_proxy_address: EthAddress,
145    sui_client: Arc<SuiBridgeClient>,
146    registry: &Registry,
147    bridge_metrics: Arc<BridgeMetrics>,
148) -> Result<()> {
149    let watchdog_metrics = WatchdogMetrics::new(registry);
150    let eth_provider =
151        Arc::new(new_metered_eth_provider(&config.eth_rpc_url, bridge_metrics.clone()).unwrap());
152    let (
153        _committee_address,
154        _limiter_address,
155        vault_address,
156        _config_address,
157        weth_address,
158        usdt_address,
159        wbtc_address,
160        lbtc_address,
161    ) = get_eth_contract_addresses(eth_bridge_proxy_address, &eth_provider).await?;
162
163    let eth_vault_balance = EthereumVaultBalance::new(
164        eth_provider.clone(),
165        vault_address,
166        weth_address,
167        VaultAsset::WETH,
168        watchdog_metrics.eth_vault_balance.clone(),
169    )
170    .await
171    .unwrap_or_else(|e| panic!("Failed to create eth vault balance: {}", e));
172
173    let usdt_vault_balance = EthereumVaultBalance::new(
174        eth_provider.clone(),
175        vault_address,
176        usdt_address,
177        VaultAsset::USDT,
178        watchdog_metrics.usdt_vault_balance.clone(),
179    )
180    .await
181    .unwrap_or_else(|e| panic!("Failed to create usdt vault balance: {}", e));
182
183    let wbtc_vault_balance = EthereumVaultBalance::new(
184        eth_provider.clone(),
185        vault_address,
186        wbtc_address,
187        VaultAsset::WBTC,
188        watchdog_metrics.wbtc_vault_balance.clone(),
189    )
190    .await
191    .unwrap_or_else(|e| panic!("Failed to create wbtc vault balance: {}", e));
192
193    let lbtc_vault_balance = if !lbtc_address.is_zero() {
194        Some(
195            EthereumVaultBalance::new(
196                eth_provider.clone(),
197                vault_address,
198                lbtc_address,
199                VaultAsset::LBTC,
200                watchdog_metrics.lbtc_vault_balance.clone(),
201            )
202            .await
203            .unwrap_or_else(|e| panic!("Failed to create lbtc vault balance: {}", e)),
204        )
205    } else {
206        None
207    };
208
209    let eth_bridge_status = EthBridgeStatus::new(
210        eth_provider,
211        eth_bridge_proxy_address,
212        watchdog_metrics.eth_bridge_paused.clone(),
213    );
214
215    let sui_bridge_status =
216        SuiBridgeStatus::new(sui_client, watchdog_metrics.sui_bridge_paused.clone());
217    let mut observables: Vec<Box<dyn Observable + Send + Sync>> = vec![
218        Box::new(eth_vault_balance),
219        Box::new(usdt_vault_balance),
220        Box::new(wbtc_vault_balance),
221        Box::new(eth_bridge_status),
222        Box::new(sui_bridge_status),
223    ];
224
225    // Add lbtc_vault_balance if it's available
226    if let Some(balance) = lbtc_vault_balance {
227        observables.push(Box::new(balance));
228    }
229
230    BridgeWatchDog::new(observables).run().await;
231
232    Ok(())
233}

235#[allow(unused)]
236async fn start_processing_sui_checkpoints_by_querying_txns(
237    sui_rpc_url: String,
238    db_url: String,
239    indexer_metrics: BridgeIndexerMetrics,
240) -> Result<Vec<JoinHandle<()>>> {
241    let pg_pool = get_connection_pool(db_url.clone()).await;
242    let (tx, rx) = channel(
243        100,
244        &mysten_metrics::get_metrics()
245            .unwrap()
246            .channel_inflight
247            .with_label_values(&["sui_transaction_processing_queue"]),
248    );
249    let mut handles = vec![];
250    let cursor = read_sui_progress_store(&pg_pool)
251        .await
252        .expect("Failed to read cursor from sui progress store");
253    let sui_client = SuiClientBuilder::default().build(sui_rpc_url).await?;
254    handles.push(spawn_logged_monitored_task!(
255        start_sui_tx_polling_task(sui_client, cursor, tx),
256        "start_sui_tx_polling_task"
257    ));
258    handles.push(spawn_logged_monitored_task!(
259        handle_sui_transactions_loop(pg_pool.clone(), rx, indexer_metrics.clone()),
260        "handle_sui_transcations_loop"
261    ));
262    Ok(handles)
263}
