38/// Block manager suspends incoming blocks until they are connected to the existing graph,
39/// returning newly connected blocks.
40/// TODO: As it is possible to have Byzantine validators who produce Blocks without valid causal
41/// history we need to make sure that BlockManager takes care of that and avoid OOM (Out Of Memory)
42/// situations.
43pub(crate) struct BlockManager {
44    context: Arc<Context>,
45    dag_state: Arc<RwLock<DagState>>,
46
47    /// Keeps all the suspended blocks. A suspended block is a block that is missing part of its causal history and thus
48    /// can't be immediately processed. A block will remain in this map until all its causal history has been successfully
49    /// processed.
50    suspended_blocks: BTreeMap<BlockRef, SuspendedBlock>,
51    /// A map that keeps all the blocks that we are missing (keys) and the corresponding blocks that reference the missing blocks
52    /// as ancestors and need them to get unsuspended. It is possible for a missing dependency (key) to be a suspended block, so
53    /// the block has been already fetched but it self is still missing some of its ancestors to be processed.
54    missing_ancestors: BTreeMap<BlockRef, BTreeSet<BlockRef>>,
55    /// Keeps all the blocks that we actually miss and haven't fetched them yet. That set will basically contain all the
56    /// keys from the `missing_ancestors` minus any keys that exist in `suspended_blocks`.
57    missing_blocks: BTreeSet<BlockRef>,
58    /// A vector that holds a tuple of (lowest_round, highest_round) of received blocks per authority.
59    /// This is used for metrics reporting purposes and resets during restarts.
60    received_block_rounds: Vec<Option<(Round, Round)>>,
61}
62
63impl BlockManager {
64    pub(crate) fn new(context: Arc<Context>, dag_state: Arc<RwLock<DagState>>) -> Self {
65        let committee_size = context.committee.size();
66        Self {
67            context,
68            dag_state,
69            suspended_blocks: BTreeMap::new(),
70            missing_ancestors: BTreeMap::new(),
71            missing_blocks: BTreeSet::new(),
72            received_block_rounds: vec![None; committee_size],
73        }
74    }

76    /// Tries to accept the provided blocks assuming that all their causal history exists. The method
77    /// returns all the blocks that have been successfully processed in round ascending order, that includes also previously
78    /// suspended blocks that have now been able to get accepted. Method also returns a set with the missing ancestor blocks.
79    #[tracing::instrument(skip_all)]
80    pub(crate) fn try_accept_blocks(
81        &mut self,
82        blocks: Vec<VerifiedBlock>,
83    ) -> (Vec<VerifiedBlock>, BTreeSet<BlockRef>) {
84        let _s = monitored_scope("BlockManager::try_accept_blocks");
85        self.try_accept_blocks_internal(blocks, false)
86    }

88    // Tries to accept blocks that have been committed. Returns all the blocks that have been accepted, both from the ones
89    // provided and any children blocks.
90    #[tracing::instrument(skip_all)]
91    pub(crate) fn try_accept_committed_blocks(
92        &mut self,
93        blocks: Vec<VerifiedBlock>,
94    ) -> Vec<VerifiedBlock> {
95        // Just accept the blocks
96        let _s = monitored_scope("BlockManager::try_accept_committed_blocks");
97        let (accepted_blocks, missing_blocks) = self.try_accept_blocks_internal(blocks, true);
98        assert!(
99            missing_blocks.is_empty(),
100            "No missing blocks should be returned for committed blocks"
101        );
102
103        accepted_blocks
104    }

106    /// Attempts to accept the provided blocks. When `committed = true` then the blocks are considered to be committed via certified commits and
107    /// are handled differently.
108    fn try_accept_blocks_internal(
109        &mut self,
110        mut blocks: Vec<VerifiedBlock>,
111        committed: bool,
112    ) -> (Vec<VerifiedBlock>, BTreeSet<BlockRef>) {
113        let _s = monitored_scope("BlockManager::try_accept_blocks_internal");
114
115        blocks.sort_by_key(|b| b.round());
116        if !blocks.is_empty() {
117            debug!(
118                "Trying to accept blocks: {}",
119                blocks.iter().map(|b| b.reference().to_string()).join(",")
120            );
121        }
122
123        let mut accepted_blocks = vec![];
124        let mut missing_blocks = BTreeSet::new();
125
126        for block in blocks {
127            self.update_block_received_metrics(&block);
128
129            // Try to accept the input block.
130            let block_ref = block.reference();
131
132            let mut blocks_to_accept = vec![];
133            if committed {
134                match self.try_accept_one_committed_block(block) {
135                    TryAcceptResult::Accepted(block) => {
136                        // As this is a committed block, then it's already accepted and there is no need to verify its timestamps.
137                        // Just add it to the accepted blocks list.
138                        accepted_blocks.push(block);
139                    }
140                    TryAcceptResult::Processed => continue,
141                    TryAcceptResult::Suspended(_) | TryAcceptResult::Skipped => panic!(
142                        "Did not expect to suspend or skip a committed block: {:?}",
143                        block_ref
144                    ),
145                };
146            } else {
147                match self.try_accept_one_block(block) {
148                    TryAcceptResult::Accepted(block) => {
149                        blocks_to_accept.push(block);
150                    }
151                    TryAcceptResult::Suspended(ancestors_to_fetch) => {
152                        debug!(
153                            "Missing ancestors to fetch for block {block_ref}: {}",
154                            ancestors_to_fetch.iter().map(|b| b.to_string()).join(",")
155                        );
156                        missing_blocks.extend(ancestors_to_fetch);
157                        continue;
158                    }
159                    TryAcceptResult::Processed | TryAcceptResult::Skipped => continue,
160                };
161            };
162
163            // If the block is accepted, try to unsuspend its children blocks if any.
164            let unsuspended_blocks = self.try_unsuspend_children_blocks(block_ref);
165            blocks_to_accept.extend(unsuspended_blocks);
166
167            // Insert the accepted blocks into DAG state so future blocks including them as
168            // ancestors do not get suspended.
169            self.dag_state
170                .write()
171                .accept_blocks(blocks_to_accept.clone());
172
173            accepted_blocks.extend(blocks_to_accept);
174        }
175
176        self.update_stats(missing_blocks.len() as u64);
177
178        // Figure out the new missing blocks
179        (accepted_blocks, missing_blocks)
180    }
181
182    fn try_accept_one_committed_block(&mut self, block: VerifiedBlock) -> TryAcceptResult {
183        if self.dag_state.read().contains_block(&block.reference()) {
184            return TryAcceptResult::Processed;
185        }
186
187        // Remove the block from missing and suspended blocks
188        self.missing_blocks.remove(&block.reference());
189
190        // If the block has been already fetched and parked as suspended block, then remove it. Also find all the references of missing
191        // ancestors to remove those as well. If we don't do that then it's possible once the missing ancestor is fetched to cause a panic
192        // when trying to unsuspend this children as it won't be found in the suspended blocks map.
193        if let Some(suspended_block) = self.suspended_blocks.remove(&block.reference()) {
194            suspended_block
195                .missing_ancestors
196                .iter()
197                .for_each(|ancestor| {
198                    if let Some(references) = self.missing_ancestors.get_mut(ancestor) {
199                        references.remove(&block.reference());
200                    }
201                });
202        }
203
204        // Accept this block before any unsuspended children blocks
205        self.dag_state.write().accept_blocks(vec![block.clone()]);
206
207        TryAcceptResult::Accepted(block)
208    }

210    /// Tries to find the provided block_refs in DagState and BlockManager,
211    /// and returns missing block refs.
212    pub(crate) fn try_find_blocks(&mut self, block_refs: Vec<BlockRef>) -> BTreeSet<BlockRef> {
213        let _s = monitored_scope("BlockManager::try_find_blocks");
214        let gc_round = self.dag_state.read().gc_round();
215
216        // No need to fetch blocks that are <= gc_round as they won't get processed anyways and they'll get skipped.
217        // So keep only the ones above.
218        let mut block_refs = block_refs
219            .into_iter()
220            .filter(|block_ref| block_ref.round > gc_round)
221            .collect::<Vec<_>>();
222
223        if block_refs.is_empty() {
224            return BTreeSet::new();
225        }
226
227        block_refs.sort_by_key(|b| b.round);
228
229        trace!(
230            "Trying to find blocks: {}",
231            block_refs.iter().map(|b| b.to_string()).join(",")
232        );
233
234        let mut missing_blocks = BTreeSet::new();
235
236        for (found, block_ref) in self
237            .dag_state
238            .read()
239            .contains_blocks(block_refs.clone())
240            .into_iter()
241            .zip(block_refs.iter())
242        {
243            if found || self.suspended_blocks.contains_key(block_ref) {
244                continue;
245            }
246            // Fetches the block if it is not in dag state or suspended.
247            missing_blocks.insert(*block_ref);
248            if self.missing_blocks.insert(*block_ref) {
249                // We want to report this as a missing ancestor even if there is no block that is actually references it right now. That will allow us
250                // to seamlessly GC the block later if needed.
251                self.missing_ancestors.entry(*block_ref).or_default();
252
253                let block_ref_hostname =
254                    &self.context.committee.authority(block_ref.author).hostname;
255                self.context
256                    .metrics
257                    .node_metrics
258                    .block_manager_missing_blocks_by_authority
259                    .with_label_values(&[block_ref_hostname])
260                    .inc();
261            }
262        }
263
264        let metrics = &self.context.metrics.node_metrics;
265        metrics
266            .missing_blocks_total
267            .inc_by(missing_blocks.len() as u64);
268        metrics
269            .block_manager_missing_blocks
270            .set(self.missing_blocks.len() as i64);
271
272        missing_blocks
273    }

275    /// Tries to accept the provided block. To accept a block its ancestors must have been already successfully accepted. If
276    /// block is accepted then Some result is returned. None is returned when either the block is suspended or the block
277    /// has been already accepted before.
278    fn try_accept_one_block(&mut self, block: VerifiedBlock) -> TryAcceptResult {
279        let block_ref = block.reference();
280        let mut missing_ancestors = BTreeSet::new();
281        let mut ancestors_to_fetch = BTreeSet::new();
282        let dag_state = self.dag_state.read();
283        let gc_round = dag_state.gc_round();
284
285        // If block has been already received and suspended, or already processed and stored, or is a genesis block, then skip it.
286        if self.suspended_blocks.contains_key(&block_ref) || dag_state.contains_block(&block_ref) {
287            return TryAcceptResult::Processed;
288        }
289
290        // If the block is <= gc_round, then we simply skip its processing as there is no meaning do any action on it or even store it.
291        if block.round() <= gc_round {
292            let hostname = self
293                .context
294                .committee
295                .authority(block.author())
296                .hostname
297                .as_str();
298            self.context
299                .metrics
300                .node_metrics
301                .block_manager_skipped_blocks
302                .with_label_values(&[hostname])
303                .inc();
304            return TryAcceptResult::Skipped;
305        }
306
307        // Keep only the ancestors that are greater than the GC round to check for their existence.
308        let ancestors = block
309            .ancestors()
310            .iter()
311            .filter(|ancestor| ancestor.round == GENESIS_ROUND || ancestor.round > gc_round)
312            .cloned()
313            .collect::<Vec<_>>();
314
315        // make sure that we have all the required ancestors in store
316        for (found, ancestor) in dag_state
317            .contains_blocks(ancestors.clone())
318            .into_iter()
319            .zip(ancestors.iter())
320        {
321            if !found {
322                missing_ancestors.insert(*ancestor);
323
324                // mark the block as having missing ancestors
325                self.missing_ancestors
326                    .entry(*ancestor)
327                    .or_default()
328                    .insert(block_ref);
329
330                let ancestor_hostname = &self.context.committee.authority(ancestor.author).hostname;
331                self.context
332                    .metrics
333                    .node_metrics
334                    .block_manager_missing_ancestors_by_authority
335                    .with_label_values(&[ancestor_hostname])
336                    .inc();
337
338                // Add the ancestor to the missing blocks set only if it doesn't already exist in the suspended blocks - meaning
339                // that we already have its payload.
340                if !self.suspended_blocks.contains_key(ancestor) {
341                    // Fetches the block if it is not in dag state or suspended.
342                    ancestors_to_fetch.insert(*ancestor);
343                    if self.missing_blocks.insert(*ancestor) {
344                        self.context
345                            .metrics
346                            .node_metrics
347                            .block_manager_missing_blocks_by_authority
348                            .with_label_values(&[ancestor_hostname])
349                            .inc();
350                    }
351                }
352            }
353        }
354
355        // Remove the block ref from the `missing_blocks` - if exists - since we now have received the block. The block
356        // might still get suspended, but we won't report it as missing in order to not re-fetch.
357        self.missing_blocks.remove(&block.reference());
358
359        if !missing_ancestors.is_empty() {
360            let hostname = self
361                .context
362                .committee
363                .authority(block.author())
364                .hostname
365                .as_str();
366            self.context
367                .metrics
368                .node_metrics
369                .block_suspensions
370                .with_label_values(&[hostname])
371                .inc();
372            self.suspended_blocks
373                .insert(block_ref, SuspendedBlock::new(block, missing_ancestors));
374            return TryAcceptResult::Suspended(ancestors_to_fetch);
375        }
376
377        TryAcceptResult::Accepted(block)
378    }

380    /// Given an accepted block `accepted_block` it attempts to accept all the suspended children blocks assuming such exist.
381    /// All the unsuspended / accepted blocks are returned as a vector in causal order.
382    fn try_unsuspend_children_blocks(&mut self, accepted_block: BlockRef) -> Vec<VerifiedBlock> {
383        let mut unsuspended_blocks = vec![];
384        let mut to_process_blocks = vec![accepted_block];
385
386        while let Some(block_ref) = to_process_blocks.pop() {
387            // And try to check if its direct children can be unsuspended
388            if let Some(block_refs_with_missing_deps) = self.missing_ancestors.remove(&block_ref) {
389                for r in block_refs_with_missing_deps {
390                    // For each dependency try to unsuspend it. If that's successful then we add it to the queue so
391                    // we can recursively try to unsuspend its children.
392                    if let Some(block) = self.try_unsuspend_block(&r, &block_ref) {
393                        to_process_blocks.push(block.block.reference());
394                        unsuspended_blocks.push(block);
395                    }
396                }
397            }
398        }
399
400        let now = Instant::now();
401
402        // Report the unsuspended blocks
403        for block in &unsuspended_blocks {
404            let hostname = self
405                .context
406                .committee
407                .authority(block.block.author())
408                .hostname
409                .as_str();
410            self.context
411                .metrics
412                .node_metrics
413                .block_unsuspensions
414                .with_label_values(&[hostname])
415                .inc();
416            self.context
417                .metrics
418                .node_metrics
419                .suspended_block_time
420                .with_label_values(&[hostname])
421                .observe(now.saturating_duration_since(block.timestamp).as_secs_f64());
422        }
423
424        unsuspended_blocks
425            .into_iter()
426            .map(|block| block.block)
427            .collect()
428    }

430    /// Attempts to unsuspend a block by checking its ancestors and removing the `accepted_dependency` by its local set.
431    /// If there is no missing dependency then this block can be unsuspended immediately and is removed from the `suspended_blocks` map.
432    fn try_unsuspend_block(
433        &mut self,
434        block_ref: &BlockRef,
435        accepted_dependency: &BlockRef,
436    ) -> Option<SuspendedBlock> {
437        let block = self
438            .suspended_blocks
439            .get_mut(block_ref)
440            .expect("Block should be in suspended map");
441
442        assert!(
443            block.missing_ancestors.remove(accepted_dependency),
444            "Block reference {} should be present in missing dependencies of {:?}",
445            block_ref,
446            block.block
447        );
448
449        if block.missing_ancestors.is_empty() {
450            // we have no missing dependency, so we unsuspend the block and return it
451            return self.suspended_blocks.remove(block_ref);
452        }
453        None
454    }

456    /// Tries to unsuspend any blocks for the latest gc round. If gc round hasn't changed then no blocks will be unsuspended due to
457    /// this action.
458    pub(crate) fn try_unsuspend_blocks_for_latest_gc_round(&mut self) {
459        let _s = monitored_scope("BlockManager::try_unsuspend_blocks_for_latest_gc_round");
460        let gc_round = self.dag_state.read().gc_round();
461        let mut blocks_unsuspended_below_gc_round = 0;
462        let mut blocks_gc_ed = 0;
463
464        while let Some((block_ref, _children_refs)) = self.missing_ancestors.first_key_value() {
465            // If the first block in the missing ancestors is higher than the gc_round, then we can't unsuspend it yet. So we just put it back
466            // and we terminate the iteration as any next entry will be of equal or higher round anyways.
467            if block_ref.round > gc_round {
468                return;
469            }
470
471            blocks_gc_ed += 1;
472
473            let hostname = self
474                .context
475                .committee
476                .authority(block_ref.author)
477                .hostname
478                .as_str();
479            self.context
480                .metrics
481                .node_metrics
482                .block_manager_gced_blocks
483                .with_label_values(&[hostname])
484                .inc();
485
486            assert!(
487                !self.suspended_blocks.contains_key(block_ref),
488                "Block should not be suspended, as we are causally GC'ing and no suspended block should exist for a missing ancestor."
489            );
490
491            // Also remove it from the missing list - we don't want to keep looking for it.
492            self.missing_blocks.remove(block_ref);
493
494            // Find all the children blocks that have a dependency on this one and try to unsuspend them
495            let unsuspended_blocks = self.try_unsuspend_children_blocks(*block_ref);
496
497            unsuspended_blocks.iter().for_each(|block| {
498                if block.round() <= gc_round {
499                    blocks_unsuspended_below_gc_round += 1;
500                }
501            });
502
503            // Now accept the unsuspended blocks
504            self.dag_state
505                .write()
506                .accept_blocks(unsuspended_blocks.clone());
507
508            for block in unsuspended_blocks {
509                let hostname = self
510                    .context
511                    .committee
512                    .authority(block.author())
513                    .hostname
514                    .as_str();
515                self.context
516                    .metrics
517                    .node_metrics
518                    .block_manager_gc_unsuspended_blocks
519                    .with_label_values(&[hostname])
520                    .inc();
521            }
522        }
523
524        debug!(
525            "Total {} blocks unsuspended and total blocks {} gc'ed <= gc_round {}",
526            blocks_unsuspended_below_gc_round, blocks_gc_ed, gc_round
527        );
528    }

530    /// Returns all the blocks that are currently missing and needed in order to accept suspended
531    /// blocks.
532    pub(crate) fn missing_blocks(&self) -> BTreeSet<BlockRef> {
533        self.missing_blocks.clone()
534    }

536    fn update_stats(&mut self, missing_blocks: u64) {
537        let metrics = &self.context.metrics.node_metrics;
538        metrics.missing_blocks_total.inc_by(missing_blocks);
539        metrics
540            .block_manager_suspended_blocks
541            .set(self.suspended_blocks.len() as i64);
542        metrics
543            .block_manager_missing_ancestors
544            .set(self.missing_ancestors.len() as i64);
545        metrics
546            .block_manager_missing_blocks
547            .set(self.missing_blocks.len() as i64);
548    }
549
550    fn update_block_received_metrics(&mut self, block: &VerifiedBlock) {
551        let (min_round, max_round) =
552            if let Some((curr_min, curr_max)) = self.received_block_rounds[block.author()] {
553                (curr_min.min(block.round()), curr_max.max(block.round()))
554            } else {
555                (block.round(), block.round())
556            };
557        self.received_block_rounds[block.author()] = Some((min_round, max_round));
558
559        let hostname = &self.context.committee.authority(block.author()).hostname;
560        self.context
561            .metrics
562            .node_metrics
563            .lowest_verified_authority_round
564            .with_label_values(&[hostname])
565            .set(min_round.into());
566        self.context
567            .metrics
568            .node_metrics
569            .highest_verified_authority_round
570            .with_label_values(&[hostname])
571            .set(max_round.into());
572    }

577        self.suspended_blocks.is_empty()
578            && self.missing_ancestors.is_empty()
579            && self.missing_blocks.is_empty()

589// Result of trying to accept one block.
590enum TryAcceptResult {
591    // The block is accepted. Wraps the block itself.
592    Accepted(VerifiedBlock),
593    // The block is suspended. Wraps ancestors to be fetched.
594    Suspended(BTreeSet<BlockRef>),
595    // The block has been processed before and already exists in BlockManager (and is suspended) or
596    // in DagState (so has been already accepted). No further processing has been done at this point.
597    Processed,
598    // When a received block is <= gc_round, then we simply skip its processing as there is no meaning
599    // do any action on it or even store it.
600    Skipped,
601}
