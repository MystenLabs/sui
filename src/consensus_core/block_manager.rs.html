39/// Block manager suspends incoming blocks until they are connected to the existing graph,
40/// returning newly connected blocks.
41/// TODO: As it is possible to have Byzantine validators who produce Blocks without valid causal
42/// history we need to make sure that BlockManager takes care of that and avoid OOM (Out Of Memory)
43/// situations.
44pub(crate) struct BlockManager {
45    context: Arc<Context>,
46    dag_state: Arc<RwLock<DagState>>,
47
48    /// Keeps all the suspended blocks. A suspended block is a block that is missing part of its causal history and thus
49    /// can't be immediately processed. A block will remain in this map until all its causal history has been successfully
50    /// processed.
51    suspended_blocks: BTreeMap<BlockRef, SuspendedBlock>,
52    /// A map that keeps all the blocks that we are missing (keys) and the corresponding blocks that reference the missing blocks
53    /// as ancestors and need them to get unsuspended. It is possible for a missing dependency (key) to be a suspended block, so
54    /// the block has been already fetched but it self is still missing some of its ancestors to be processed.
55    missing_ancestors: BTreeMap<BlockRef, BTreeSet<BlockRef>>,
56    /// Keeps all the blocks that we actually miss and haven't fetched them yet. That set will basically contain all the
57    /// keys from the `missing_ancestors` minus any keys that exist in `suspended_blocks`.
58    missing_blocks: BTreeSet<BlockRef>,
59    /// A vector that holds a tuple of (lowest_round, highest_round) of received blocks per authority.
60    /// This is used for metrics reporting purposes and resets during restarts.
61    received_block_rounds: Vec<Option<(Round, Round)>>,
62}
63
64impl BlockManager {
65    pub(crate) fn new(context: Arc<Context>, dag_state: Arc<RwLock<DagState>>) -> Self {
66        let committee_size = context.committee.size();
67        Self {
68            context,
69            dag_state,
70            suspended_blocks: BTreeMap::new(),
71            missing_ancestors: BTreeMap::new(),
72            missing_blocks: BTreeSet::new(),
73            received_block_rounds: vec![None; committee_size],
74        }
75    }

77    /// Tries to accept the provided blocks assuming that all their causal history exists. The method
78    /// returns all the blocks that have been successfully processed in round ascending order, that includes also previously
79    /// suspended blocks that have now been able to get accepted. Method also returns a set with the missing ancestor blocks.
80    #[tracing::instrument(skip_all)]
81    pub(crate) fn try_accept_blocks(
82        &mut self,
83        blocks: Vec<VerifiedBlock>,
84    ) -> (Vec<VerifiedBlock>, BTreeSet<BlockRef>) {
85        let _s = monitored_scope("BlockManager::try_accept_blocks");
86        self.try_accept_blocks_internal(blocks, false)
87    }

89    // Tries to accept blocks that have been committed. Returns all the blocks that have been accepted, both from the ones
90    // provided and any children blocks.
91    #[tracing::instrument(skip_all)]
92    pub(crate) fn try_accept_committed_blocks(
93        &mut self,
94        blocks: Vec<VerifiedBlock>,
95    ) -> Vec<VerifiedBlock> {
96        // Just accept the blocks
97        let _s = monitored_scope("BlockManager::try_accept_committed_blocks");
98        let (accepted_blocks, missing_blocks) = self.try_accept_blocks_internal(blocks, true);
99        assert!(
100            missing_blocks.is_empty(),
101            "No missing blocks should be returned for committed blocks"
102        );
103
104        accepted_blocks
105    }

107    /// Attempts to accept the provided blocks. When `committed = true` then the blocks are considered to be committed via certified commits and
108    /// are handled differently.
109    fn try_accept_blocks_internal(
110        &mut self,
111        mut blocks: Vec<VerifiedBlock>,
112        committed: bool,
113    ) -> (Vec<VerifiedBlock>, BTreeSet<BlockRef>) {
114        let _s = monitored_scope("BlockManager::try_accept_blocks_internal");
115
116        blocks.sort_by_key(|b| b.round());
117        if !blocks.is_empty() {
118            debug!(
119                "Trying to accept blocks: {}",
120                blocks.iter().map(|b| b.reference().to_string()).join(",")
121            );
122        }
123
124        let mut accepted_blocks = vec![];
125        let mut missing_blocks = BTreeSet::new();
126
127        for block in blocks {
128            self.update_block_received_metrics(&block);
129
130            // Try to accept the input block.
131            let block_ref = block.reference();
132
133            let mut blocks_to_accept = vec![];
134            if committed {
135                match self.try_accept_one_committed_block(block) {
136                    TryAcceptResult::Accepted(block) => {
137                        // As this is a committed block, then it's already accepted and there is no need to verify its timestamps.
138                        // Just add it to the accepted blocks list.
139                        accepted_blocks.push(block);
140                    }
141                    TryAcceptResult::Processed => continue,
142                    TryAcceptResult::Suspended(_) | TryAcceptResult::Skipped => panic!(
143                        "Did not expect to suspend or skip a committed block: {:?}",
144                        block_ref
145                    ),
146                };
147            } else {
148                match self.try_accept_one_block(block) {
149                    TryAcceptResult::Accepted(block) => {
150                        blocks_to_accept.push(block);
151                    }
152                    TryAcceptResult::Suspended(ancestors_to_fetch) => {
153                        debug!(
154                            "Missing ancestors to fetch for block {block_ref}: {}",
155                            ancestors_to_fetch.iter().map(|b| b.to_string()).join(",")
156                        );
157                        missing_blocks.extend(ancestors_to_fetch);
158                        continue;
159                    }
160                    TryAcceptResult::Processed | TryAcceptResult::Skipped => continue,
161                };
162            };
163
164            // If the block is accepted, try to unsuspend its children blocks if any.
165            let unsuspended_blocks = self.try_unsuspend_children_blocks(block_ref);
166            blocks_to_accept.extend(unsuspended_blocks);
167
168            // Insert the accepted blocks into DAG state so future blocks including them as
169            // ancestors do not get suspended.
170            self.dag_state
171                .write()
172                .accept_blocks(blocks_to_accept.clone());
173
174            accepted_blocks.extend(blocks_to_accept);
175        }
176
177        self.update_stats(missing_blocks.len() as u64);
178
179        // Figure out the new missing blocks
180        (accepted_blocks, missing_blocks)
181    }
182
183    fn try_accept_one_committed_block(&mut self, block: VerifiedBlock) -> TryAcceptResult {
184        if self.dag_state.read().contains_block(&block.reference()) {
185            return TryAcceptResult::Processed;
186        }
187
188        // Remove the block from missing and suspended blocks
189        self.missing_blocks.remove(&block.reference());
190
191        // If the block has been already fetched and parked as suspended block, then remove it. Also find all the references of missing
192        // ancestors to remove those as well. If we don't do that then it's possible once the missing ancestor is fetched to cause a panic
193        // when trying to unsuspend this children as it won't be found in the suspended blocks map.
194        if let Some(suspended_block) = self.suspended_blocks.remove(&block.reference()) {
195            suspended_block
196                .missing_ancestors
197                .iter()
198                .for_each(|ancestor| {
199                    if let Some(references) = self.missing_ancestors.get_mut(ancestor) {
200                        references.remove(&block.reference());
201                    }
202                });
203        }
204
205        // Accept this block before any unsuspended children blocks
206        self.dag_state.write().accept_blocks(vec![block.clone()]);
207
208        TryAcceptResult::Accepted(block)
209    }

211    /// Tries to find the provided block_refs in DagState and BlockManager,
212    /// and returns missing block refs.
213    pub(crate) fn try_find_blocks(&mut self, block_refs: Vec<BlockRef>) -> BTreeSet<BlockRef> {
214        let _s = monitored_scope("BlockManager::try_find_blocks");
215        let gc_round = self.dag_state.read().gc_round();
216
217        // No need to fetch blocks that are <= gc_round as they won't get processed anyways and they'll get skipped.
218        // So keep only the ones above.
219        let mut block_refs = block_refs
220            .into_iter()
221            .filter(|block_ref| block_ref.round > gc_round)
222            .collect::<Vec<_>>();
223
224        if block_refs.is_empty() {
225            return BTreeSet::new();
226        }
227
228        block_refs.sort_by_key(|b| b.round);
229
230        trace!(
231            "Trying to find blocks: {}",
232            block_refs.iter().map(|b| b.to_string()).join(",")
233        );
234
235        let mut missing_blocks = BTreeSet::new();
236
237        for (found, block_ref) in self
238            .dag_state
239            .read()
240            .contains_blocks(block_refs.clone())
241            .into_iter()
242            .zip_debug_eq(block_refs.iter())
243        {
244            if found || self.suspended_blocks.contains_key(block_ref) {
245                continue;
246            }
247            // Fetches the block if it is not in dag state or suspended.
248            missing_blocks.insert(*block_ref);
249            if self.missing_blocks.insert(*block_ref) {
250                // We want to report this as a missing ancestor even if there is no block that is actually references it right now. That will allow us
251                // to seamlessly GC the block later if needed.
252                self.missing_ancestors.entry(*block_ref).or_default();
253
254                let block_ref_hostname =
255                    &self.context.committee.authority(block_ref.author).hostname;
256                self.context
257                    .metrics
258                    .node_metrics
259                    .block_manager_missing_blocks_by_authority
260                    .with_label_values(&[block_ref_hostname])
261                    .inc();
262            }
263        }
264
265        let metrics = &self.context.metrics.node_metrics;
266        metrics
267            .missing_blocks_total
268            .inc_by(missing_blocks.len() as u64);
269        metrics
270            .block_manager_missing_blocks
271            .set(self.missing_blocks.len() as i64);
272
273        missing_blocks
274    }

276    /// Tries to accept the provided block. To accept a block its ancestors must have been already successfully accepted. If
277    /// block is accepted then Some result is returned. None is returned when either the block is suspended or the block
278    /// has been already accepted before.
279    fn try_accept_one_block(&mut self, block: VerifiedBlock) -> TryAcceptResult {
280        let block_ref = block.reference();
281        let mut missing_ancestors = BTreeSet::new();
282        let mut ancestors_to_fetch = BTreeSet::new();
283        let dag_state = self.dag_state.read();
284        let gc_round = dag_state.gc_round();
285
286        // If block has been already received and suspended, or already processed and stored, or is a genesis block, then skip it.
287        if self.suspended_blocks.contains_key(&block_ref) || dag_state.contains_block(&block_ref) {
288            return TryAcceptResult::Processed;
289        }
290
291        // If the block is <= gc_round, then we simply skip its processing as there is no meaning do any action on it or even store it.
292        if block.round() <= gc_round {
293            let hostname = self
294                .context
295                .committee
296                .authority(block.author())
297                .hostname
298                .as_str();
299            self.context
300                .metrics
301                .node_metrics
302                .block_manager_skipped_blocks
303                .with_label_values(&[hostname])
304                .inc();
305            return TryAcceptResult::Skipped;
306        }
307
308        // Keep only the ancestors that are greater than the GC round to check for their existence.
309        let ancestors = block
310            .ancestors()
311            .iter()
312            .filter(|ancestor| ancestor.round == GENESIS_ROUND || ancestor.round > gc_round)
313            .cloned()
314            .collect::<Vec<_>>();
315
316        // make sure that we have all the required ancestors in store
317        for (found, ancestor) in dag_state
318            .contains_blocks(ancestors.clone())
319            .into_iter()
320            .zip_debug_eq(ancestors.iter())
321        {
322            if !found {
323                missing_ancestors.insert(*ancestor);
324
325                // mark the block as having missing ancestors
326                self.missing_ancestors
327                    .entry(*ancestor)
328                    .or_default()
329                    .insert(block_ref);
330
331                let ancestor_hostname = &self.context.committee.authority(ancestor.author).hostname;
332                self.context
333                    .metrics
334                    .node_metrics
335                    .block_manager_missing_ancestors_by_authority
336                    .with_label_values(&[ancestor_hostname])
337                    .inc();
338
339                // Add the ancestor to the missing blocks set only if it doesn't already exist in the suspended blocks - meaning
340                // that we already have its payload.
341                if !self.suspended_blocks.contains_key(ancestor) {
342                    // Fetches the block if it is not in dag state or suspended.
343                    ancestors_to_fetch.insert(*ancestor);
344                    if self.missing_blocks.insert(*ancestor) {
345                        self.context
346                            .metrics
347                            .node_metrics
348                            .block_manager_missing_blocks_by_authority
349                            .with_label_values(&[ancestor_hostname])
350                            .inc();
351                    }
352                }
353            }
354        }
355
356        // Remove the block ref from the `missing_blocks` - if exists - since we now have received the block. The block
357        // might still get suspended, but we won't report it as missing in order to not re-fetch.
358        self.missing_blocks.remove(&block.reference());
359
360        if !missing_ancestors.is_empty() {
361            let hostname = self
362                .context
363                .committee
364                .authority(block.author())
365                .hostname
366                .as_str();
367            self.context
368                .metrics
369                .node_metrics
370                .block_suspensions
371                .with_label_values(&[hostname])
372                .inc();
373            self.suspended_blocks
374                .insert(block_ref, SuspendedBlock::new(block, missing_ancestors));
375            return TryAcceptResult::Suspended(ancestors_to_fetch);
376        }
377
378        TryAcceptResult::Accepted(block)
379    }

381    /// Given an accepted block `accepted_block` it attempts to accept all the suspended children blocks assuming such exist.
382    /// All the unsuspended / accepted blocks are returned as a vector in causal order.
383    fn try_unsuspend_children_blocks(&mut self, accepted_block: BlockRef) -> Vec<VerifiedBlock> {
384        let mut unsuspended_blocks = vec![];
385        let mut to_process_blocks = vec![accepted_block];
386
387        while let Some(block_ref) = to_process_blocks.pop() {
388            // And try to check if its direct children can be unsuspended
389            if let Some(block_refs_with_missing_deps) = self.missing_ancestors.remove(&block_ref) {
390                for r in block_refs_with_missing_deps {
391                    // For each dependency try to unsuspend it. If that's successful then we add it to the queue so
392                    // we can recursively try to unsuspend its children.
393                    if let Some(block) = self.try_unsuspend_block(&r, &block_ref) {
394                        to_process_blocks.push(block.block.reference());
395                        unsuspended_blocks.push(block);
396                    }
397                }
398            }
399        }
400
401        let now = Instant::now();
402
403        // Report the unsuspended blocks
404        for block in &unsuspended_blocks {
405            let hostname = self
406                .context
407                .committee
408                .authority(block.block.author())
409                .hostname
410                .as_str();
411            self.context
412                .metrics
413                .node_metrics
414                .block_unsuspensions
415                .with_label_values(&[hostname])
416                .inc();
417            self.context
418                .metrics
419                .node_metrics
420                .suspended_block_time
421                .with_label_values(&[hostname])
422                .observe(now.saturating_duration_since(block.timestamp).as_secs_f64());
423        }
424
425        unsuspended_blocks
426            .into_iter()
427            .map(|block| block.block)
428            .collect()
429    }

431    /// Attempts to unsuspend a block by checking its ancestors and removing the `accepted_dependency` by its local set.
432    /// If there is no missing dependency then this block can be unsuspended immediately and is removed from the `suspended_blocks` map.
433    fn try_unsuspend_block(
434        &mut self,
435        block_ref: &BlockRef,
436        accepted_dependency: &BlockRef,
437    ) -> Option<SuspendedBlock> {
438        let block = self
439            .suspended_blocks
440            .get_mut(block_ref)
441            .expect("Block should be in suspended map");
442
443        assert!(
444            block.missing_ancestors.remove(accepted_dependency),
445            "Block reference {} should be present in missing dependencies of {:?}",
446            block_ref,
447            block.block
448        );
449
450        if block.missing_ancestors.is_empty() {
451            // we have no missing dependency, so we unsuspend the block and return it
452            return self.suspended_blocks.remove(block_ref);
453        }
454        None
455    }

457    /// Tries to unsuspend any blocks for the latest gc round. If gc round hasn't changed then no blocks will be unsuspended due to
458    /// this action.
459    pub(crate) fn try_unsuspend_blocks_for_latest_gc_round(&mut self) {
460        let _s = monitored_scope("BlockManager::try_unsuspend_blocks_for_latest_gc_round");
461        let gc_round = self.dag_state.read().gc_round();
462        let mut blocks_unsuspended_below_gc_round = 0;
463        let mut blocks_gc_ed = 0;
464
465        while let Some((block_ref, _children_refs)) = self.missing_ancestors.first_key_value() {
466            // If the first block in the missing ancestors is higher than the gc_round, then we can't unsuspend it yet. So we just put it back
467            // and we terminate the iteration as any next entry will be of equal or higher round anyways.
468            if block_ref.round > gc_round {
469                return;
470            }
471
472            blocks_gc_ed += 1;
473
474            let hostname = self
475                .context
476                .committee
477                .authority(block_ref.author)
478                .hostname
479                .as_str();
480            self.context
481                .metrics
482                .node_metrics
483                .block_manager_gced_blocks
484                .with_label_values(&[hostname])
485                .inc();
486
487            assert!(
488                !self.suspended_blocks.contains_key(block_ref),
489                "Block should not be suspended, as we are causally GC'ing and no suspended block should exist for a missing ancestor."
490            );
491
492            // Also remove it from the missing list - we don't want to keep looking for it.
493            self.missing_blocks.remove(block_ref);
494
495            // Find all the children blocks that have a dependency on this one and try to unsuspend them
496            let unsuspended_blocks = self.try_unsuspend_children_blocks(*block_ref);
497
498            unsuspended_blocks.iter().for_each(|block| {
499                if block.round() <= gc_round {
500                    blocks_unsuspended_below_gc_round += 1;
501                }
502            });
503
504            // Now accept the unsuspended blocks
505            self.dag_state
506                .write()
507                .accept_blocks(unsuspended_blocks.clone());
508
509            for block in unsuspended_blocks {
510                let hostname = self
511                    .context
512                    .committee
513                    .authority(block.author())
514                    .hostname
515                    .as_str();
516                self.context
517                    .metrics
518                    .node_metrics
519                    .block_manager_gc_unsuspended_blocks
520                    .with_label_values(&[hostname])
521                    .inc();
522            }
523        }
524
525        debug!(
526            "Total {} blocks unsuspended and total blocks {} gc'ed <= gc_round {}",
527            blocks_unsuspended_below_gc_round, blocks_gc_ed, gc_round
528        );
529    }

531    /// Returns all the blocks that are currently missing and needed in order to accept suspended
532    /// blocks.
533    pub(crate) fn missing_blocks(&self) -> BTreeSet<BlockRef> {
534        self.missing_blocks.clone()
535    }

537    fn update_stats(&mut self, missing_blocks: u64) {
538        let metrics = &self.context.metrics.node_metrics;
539        metrics.missing_blocks_total.inc_by(missing_blocks);
540        metrics
541            .block_manager_suspended_blocks
542            .set(self.suspended_blocks.len() as i64);
543        metrics
544            .block_manager_missing_ancestors
545            .set(self.missing_ancestors.len() as i64);
546        metrics
547            .block_manager_missing_blocks
548            .set(self.missing_blocks.len() as i64);
549    }
550
551    fn update_block_received_metrics(&mut self, block: &VerifiedBlock) {
552        let (min_round, max_round) =
553            if let Some((curr_min, curr_max)) = self.received_block_rounds[block.author()] {
554                (curr_min.min(block.round()), curr_max.max(block.round()))
555            } else {
556                (block.round(), block.round())
557            };
558        self.received_block_rounds[block.author()] = Some((min_round, max_round));
559
560        let hostname = &self.context.committee.authority(block.author()).hostname;
561        self.context
562            .metrics
563            .node_metrics
564            .lowest_verified_authority_round
565            .with_label_values(&[hostname])
566            .set(min_round.into());
567        self.context
568            .metrics
569            .node_metrics
570            .highest_verified_authority_round
571            .with_label_values(&[hostname])
572            .set(max_round.into());
573    }

579            && self.missing_ancestors.is_empty()

590// Result of trying to accept one block.
591enum TryAcceptResult {
592    // The block is accepted. Wraps the block itself.
593    Accepted(VerifiedBlock),
594    // The block is suspended. Wraps ancestors to be fetched.
595    Suspended(BTreeSet<BlockRef>),
596    // The block has been processed before and already exists in BlockManager (and is suspended) or
597    // in DagState (so has been already accepted). No further processing has been done at this point.
598    Processed,
599    // When a received block is <= gc_round, then we simply skip its processing as there is no meaning
600    // do any action on it or even store it.
601    Skipped,
602}
