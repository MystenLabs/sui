24/// Role of CommitObserver
25/// - Called by core when try_commit() returns newly committed leaders.
26/// - The newly committed leaders are sent to commit observer and then commit observer
27///   gets subdags for each leader via the commit interpreter (linearizer)
28/// - The committed subdags are sent as consensus output via an unbounded tokio channel.
29///
30/// There is no flow control on sending output. Consensus backpressure is applied earlier
31/// at consensus input level, and on commit sync.
32///
33/// Commit is persisted in store before the CommittedSubDag is sent to the commit handler.
34/// When Sui recovers, it blocks until the commits it knows about are recovered. So consensus
35/// must be able to quickly recover the commits it has sent to Sui.
36pub(crate) struct CommitObserver {
37    context: Arc<Context>,
38    dag_state: Arc<RwLock<DagState>>,
39    /// Persistent storage for blocks, commits and other consensus data.
40    store: Arc<dyn Store>,
41    transaction_certifier: TransactionCertifier,
42    leader_schedule: Arc<LeaderSchedule>,
43    /// Component to deterministically collect subdags for committed leaders.
44    commit_interpreter: Linearizer,
45    /// Handle to an unbounded channel to send output commits.
46    commit_finalizer_handle: CommitFinalizerHandle,
47}
48
49impl CommitObserver {
50    pub(crate) async fn new(
51        context: Arc<Context>,
52        commit_consumer: CommitConsumerArgs,
53        dag_state: Arc<RwLock<DagState>>,
54        transaction_certifier: TransactionCertifier,
55        leader_schedule: Arc<LeaderSchedule>,
56    ) -> Self {
57        let store = dag_state.read().store();
58        let commit_interpreter = Linearizer::new(context.clone(), dag_state.clone());
59        let commit_finalizer_handle = CommitFinalizer::start(
60            context.clone(),
61            dag_state.clone(),
62            transaction_certifier.clone(),
63            commit_consumer.commit_sender.clone(),
64        );
65
66        let mut observer = Self {
67            context,
68            dag_state,
69            store,
70            transaction_certifier,
71            leader_schedule,
72            commit_interpreter,
73            commit_finalizer_handle,
74        };
75        observer.recover_and_send_commits(&commit_consumer).await;
76
77        // Recover blocks needed for future commits (and block proposals).
78        // Some blocks might have been recovered as committed blocks in recover_and_send_commits().
79        // They will just be ignored.
80        observer
81            .transaction_certifier
82            .recover_blocks_after_round(observer.dag_state.read().gc_round());
83
84        observer
85    }
86
87    /// Creates and returns a list of committed subdags containing committed blocks, from a sequence
88    /// of selected leader blocks, and whether they come from local committer or commit sync remotely.
89    ///
90    /// Also, buffers the commits to DagState and forwards committed subdags to commit finalizer.
91    pub(crate) fn handle_commit(
92        &mut self,
93        committed_leaders: Vec<VerifiedBlock>,
94        local: bool,
95    ) -> ConsensusResult<Vec<CommittedSubDag>> {
96        let _s = self
97            .context
98            .metrics
99            .node_metrics
100            .scope_processing_time
101            .with_label_values(&["CommitObserver::handle_commit"])
102            .start_timer();
103
104        let mut committed_sub_dags = self.commit_interpreter.handle_commit(committed_leaders);
105        self.report_metrics(&committed_sub_dags);
106
107        // Set if the commit is produced from local DAG, or received through commit sync.
108        for subdag in committed_sub_dags.iter_mut() {
109            subdag.decided_with_local_blocks = local;
110        }
111
112        // Send scores as part of the first sub dag, if the leader schedule has been updated.
113        let schedule_updated = self
114            .leader_schedule
115            .leader_schedule_updated(&self.dag_state);
116        if schedule_updated {
117            let reputation_scores_desc = self
118                .leader_schedule
119                .leader_swap_table
120                .read()
121                .reputation_scores_desc
122                .clone();
123            committed_sub_dags[0].reputation_scores_desc = reputation_scores_desc;
124        }
125
126        for commit in committed_sub_dags.iter() {
127            tracing::debug!(
128                "Sending commit {} leader {} to finalization and execution.",
129                commit.commit_ref,
130                commit.leader
131            );
132            tracing::trace!("Committed subdag: {:#?}", commit);
133            // Failures in sender.send() are assumed to be permanent
134            self.commit_finalizer_handle.send(commit.clone())?;
135        }
136
137        self.dag_state
138            .write()
139            .add_scoring_subdags(committed_sub_dags.clone());
140
141        Ok(committed_sub_dags)
142    }

144    async fn recover_and_send_commits(&mut self, commit_consumer: &CommitConsumerArgs) {
145        let now = Instant::now();
146
147        let replay_after_commit_index = commit_consumer.replay_after_commit_index;
148
149        let last_commit = self
150            .store
151            .read_last_commit()
152            .expect("Reading the last commit should not fail");
153        let Some(last_commit) = &last_commit else {
154            assert_eq!(
155                replay_after_commit_index, 0,
156                "Commit replay should start at the beginning if there is no commit history"
157            );
158            info!("Nothing to recover for commit observer - starting new epoch");
159            return;
160        };
161
162        let last_commit_index = last_commit.index();
163        if last_commit_index == replay_after_commit_index {
164            info!(
165                "Nothing to recover for commit observer - replay is requested immediately after last commit index {last_commit_index}"
166            );
167            return;
168        }
169        assert!(last_commit_index > replay_after_commit_index);
170
171        info!(
172            "Recovering commit observer in the range [{}..={last_commit_index}]",
173            replay_after_commit_index + 1,
174        );
175
176        // To avoid scanning too many commits at once and load in memory,
177        // we limit the batch size to 250 and iterate over.
178        const COMMIT_RECOVERY_BATCH_SIZE: u32 = if cfg!(test) { 3 } else { 250 };
179
180        let mut last_sent_commit_index = replay_after_commit_index;
181
182        // Make sure that there is no pending commits to be written to the store.
183        self.dag_state.read().ensure_commits_to_write_is_empty();
184
185        let mut seen_unfinalized_commit = false;
186        for start_index in (replay_after_commit_index + 1..=last_commit_index)
187            .step_by(COMMIT_RECOVERY_BATCH_SIZE as usize)
188        {
189            let end_index = start_index
190                .saturating_add(COMMIT_RECOVERY_BATCH_SIZE - 1)
191                .min(last_commit_index);
192
193            let unsent_commits = self
194                .store
195                .scan_commits((start_index..=end_index).into())
196                .expect("Scanning commits should not fail");
197            assert_eq!(
198                unsent_commits.len() as u32,
199                end_index.checked_sub(start_index).unwrap() + 1,
200                "Gap in scanned commits: start index: {start_index}, end index: {end_index}, commits: {:?}",
201                unsent_commits,
202            );
203
204            // Buffered unsent commits in DAG state which is required to contain them when they are flushed
205            // by CommitFinalizer.
206            self.dag_state
207                .write()
208                .recover_commits_to_write(unsent_commits.clone());
209
210            info!(
211                "Recovered {} unsent commits in range [{start_index}..={end_index}]",
212                unsent_commits.len()
213            );
214
215            // Resend all the committed subdags to the consensus output channel
216            // for all the commits above the last processed index.
217            for commit in unsent_commits.into_iter() {
218                // Commit index must be continuous.
219                last_sent_commit_index += 1;
220                assert_eq!(commit.index(), last_sent_commit_index);
221
222                // On recovery leader schedule will be updated with the current scores
223                // and the scores will be passed along with the last commit of this recovered batch sent to
224                // Sui so that the current scores are available for submission.
225                let reputation_scores = if commit.index() == last_commit_index {
226                    self.leader_schedule
227                        .leader_swap_table
228                        .read()
229                        .reputation_scores_desc
230                        .clone()
231                } else {
232                    vec![]
233                };
234
235                let committed_sub_dag = load_committed_subdag_from_store(
236                    self.store.as_ref(),
237                    commit,
238                    reputation_scores,
239                );
240
241                if !committed_sub_dag.recovered_rejected_transactions {
242                    // When the commit has no associated storage entry for rejected transactions,
243                    // even if an empty set, the commit is unfinalized.
244                    seen_unfinalized_commit = true;
245                }
246
247                if seen_unfinalized_commit {
248                    // After observing the first unfinalized commit, the rest of recovered commits should all be unfinalized.
249                    assert!(!committed_sub_dag.recovered_rejected_transactions);
250                    // All unfinalized commit cannot be assumed to be decided with local blocks, because they
251                    // might have been received through commit sync.
252                    assert!(!committed_sub_dag.decided_with_local_blocks);
253                    // All unfinalized commits need to be processed by the CommitFinalizer, making it necessary to
254                    // recover and vote on the blocks in this commit.
255                    self.transaction_certifier
256                        .recover_and_vote_on_blocks(committed_sub_dag.blocks.clone());
257                }
258
259                self.commit_finalizer_handle
260                    .send(committed_sub_dag)
261                    .unwrap();
262
263                self.context
264                    .metrics
265                    .node_metrics
266                    .commit_observer_last_recovered_commit_index
267                    .set(last_sent_commit_index as i64);
268
269                tokio::task::yield_now().await;
270            }
271        }
272
273        assert_eq!(
274            last_sent_commit_index, last_commit_index,
275            "We should have sent all commits up to the last commit {}",
276            last_commit_index
277        );
278
279        info!(
280            "Commit observer recovery [{}..={}] completed, took {:?}",
281            replay_after_commit_index + 1,
282            last_commit_index,
283            now.elapsed()
284        );
285    }
286
287    fn report_metrics(&self, committed: &[CommittedSubDag]) {
288        let metrics = &self.context.metrics.node_metrics;
289        let utc_now = self.context.clock.timestamp_utc_ms();
290
291        for commit in committed {
292            info!(
293                "Consensus commit {} with leader {} has {} blocks",
294                commit.commit_ref,
295                commit.leader,
296                commit.blocks.len()
297            );
298
299            metrics
300                .last_committed_leader_round
301                .set(commit.leader.round as i64);
302            metrics
303                .last_commit_index
304                .set(commit.commit_ref.index as i64);
305            metrics
306                .blocks_per_commit_count
307                .observe(commit.blocks.len() as f64);
308
309            for block in &commit.blocks {
310                let latency_ms = utc_now
311                    .checked_sub(block.timestamp_ms())
312                    .unwrap_or_default();
313                metrics
314                    .block_commit_latency
315                    .observe(Duration::from_millis(latency_ms).as_secs_f64());
316            }
317        }
318
319        self.context
320            .metrics
321            .node_metrics
322            .sub_dags_per_commit_count
323            .observe(committed.len() as f64);
324    }

345    async fn test_handle_commit() {
346        use crate::leader_schedule::LeaderSwapTable;
347
348        telemetry_subscribers::init_for_testing();
349        let num_authorities = 4;
350        let (context, _keys) = Context::new_for_test(num_authorities);
351        let context = Arc::new(context);
352
353        let mem_store = Arc::new(MemStore::new());
354        let dag_state = Arc::new(RwLock::new(DagState::new(
355            context.clone(),
356            mem_store.clone(),
357        )));
358        let last_processed_commit_index = 0;
359        let (commit_consumer, mut commit_receiver, _transaction_receiver) =
360            CommitConsumerArgs::new(0, last_processed_commit_index);
361        let (blocks_sender, _blocks_receiver) = unbounded_channel("consensus_block_output");
362        let transaction_certifier = TransactionCertifier::new(
363            context.clone(),
364            Arc::new(NoopBlockVerifier {}),
365            dag_state.clone(),
366            blocks_sender,
367        );
368        const NUM_OF_COMMITS_PER_SCHEDULE: u64 = 5;
369        let leader_schedule = Arc::new(
370            LeaderSchedule::new(context.clone(), LeaderSwapTable::default())
371                .with_num_commits_per_schedule(NUM_OF_COMMITS_PER_SCHEDULE),
372        );
373
374        let mut observer = CommitObserver::new(
375            context.clone(),
376            commit_consumer,
377            dag_state.clone(),
378            transaction_certifier.clone(),
379            leader_schedule.clone(),
380        )
381        .await;
382
383        // Populate fully connected test blocks for round 0 ~ 10, authorities 0 ~ 3.
384        let num_rounds = 10;
385        let mut builder = DagBuilder::new(context.clone());
386        builder
387            .layers(1..=num_rounds)
388            .build()
389            .persist_layers(dag_state.clone());
390        transaction_certifier.add_voted_blocks(
391            builder
392                .all_blocks()
393                .iter()
394                .map(|b| (b.clone(), vec![]))
395                .collect(),
396        );
397
398        let leaders = builder
399            .leader_blocks(1..=num_rounds)
400            .into_iter()
401            .map(Option::unwrap)
402            .collect::<Vec<_>>();
403
404        // Commit first 5 leaders.
405        let mut commits = observer
406            .handle_commit(leaders[0..5].to_vec(), true)
407            .unwrap();
408
409        // Trigger a leader schedule update.
410        leader_schedule.update_leader_schedule_v2(&dag_state);
411
412        // Commit the next 5 leaders.
413        commits.extend(observer.handle_commit(leaders[5..].to_vec(), true).unwrap());
414
415        // Check commits are returned by CommitObserver::handle_commit is accurate
416        let mut expected_stored_refs: Vec<BlockRef> = vec![];
417        for (idx, subdag) in commits.iter().enumerate() {
418            tracing::info!("{subdag:?}");
419            assert_eq!(subdag.leader, leaders[idx].reference());
420
421            // 5th subdag should contain the updated scores.
422            if idx == 5 {
423                let scores = vec![
424                    (AuthorityIndex::new_for_test(1), 9),
425                    (AuthorityIndex::new_for_test(3), 9),
426                    (AuthorityIndex::new_for_test(0), 9),
427                    (AuthorityIndex::new_for_test(2), 9),
428                ];
429                assert_eq!(subdag.reputation_scores_desc, scores);
430            } else {
431                assert!(subdag.reputation_scores_desc.is_empty());
432            }
433
434            let expected_ts = {
435                let block_refs = leaders[idx]
436                    .ancestors()
437                    .iter()
438                    .filter(|block_ref| block_ref.round == leaders[idx].round() - 1)
439                    .cloned()
440                    .collect::<Vec<_>>();
441                let blocks = dag_state
442                    .read()
443                    .get_blocks(&block_refs)
444                    .into_iter()
445                    .map(|block_opt| block_opt.expect("We should have all blocks in dag state."));
446                median_timestamp_by_stake(&context, blocks).unwrap()
447            };
448
449            let expected_ts = if idx == 0 {
450                expected_ts
451            } else {
452                expected_ts.max(commits[idx - 1].timestamp_ms)
453            };
454
455            assert_eq!(expected_ts, subdag.timestamp_ms);
456
457            if idx == 0 {
458                // First subdag includes the leader block plus all ancestor blocks
459                // of the leader minus the genesis round blocks
460                assert_eq!(subdag.blocks.len(), 1);
461            } else {
462                // Every subdag after will be missing the leader block from the previous
463                // committed subdag
464                assert_eq!(subdag.blocks.len(), num_authorities);
465            }
466            for block in subdag.blocks.iter() {
467                expected_stored_refs.push(block.reference());
468                assert!(block.round() <= leaders[idx].round());
469            }
470            assert_eq!(subdag.commit_ref.index, idx as CommitIndex + 1);
471        }
472
473        // Check commits sent over consensus output channel is accurate
474        let mut processed_subdag_index = 0;
475        while let Ok(Some(subdag)) = timeout(Duration::from_secs(1), commit_receiver.recv()).await {
476            assert_eq!(subdag, commits[processed_subdag_index]);
477            processed_subdag_index = subdag.commit_ref.index as usize;
478            if processed_subdag_index == leaders.len() {
479                break;
480            }
481        }
482        assert_eq!(processed_subdag_index, leaders.len());
483
484        verify_channel_empty(&mut commit_receiver).await;
485
486        // Check commits have been persisted to storage
487        let last_commit = mem_store.read_last_commit().unwrap().unwrap();
488        assert_eq!(
489            last_commit.index(),
490            commits.last().unwrap().commit_ref.index
491        );
492        let all_stored_commits = mem_store
493            .scan_commits((0..=CommitIndex::MAX).into())
494            .unwrap();
495        assert_eq!(all_stored_commits.len(), leaders.len());
496        let blocks_existence = mem_store.contains_blocks(&expected_stored_refs).unwrap();
497        assert!(blocks_existence.iter().all(|exists| *exists));
498    }

500    #[tokio::test]
501    async fn test_recover_and_send_commits() {
502        telemetry_subscribers::init_for_testing();
503        let num_authorities = 4;
504        let context = Arc::new(Context::new_for_test(num_authorities).0);
505        let mem_store = Arc::new(MemStore::new());
506        let dag_state = Arc::new(RwLock::new(DagState::new(
507            context.clone(),
508            mem_store.clone(),
509        )));
510        let (blocks_sender, _blocks_receiver) = unbounded_channel("consensus_block_output");
511        let transaction_certifier = TransactionCertifier::new(
512            context.clone(),
513            Arc::new(NoopBlockVerifier {}),
514            dag_state.clone(),
515            blocks_sender,
516        );
517        let last_processed_commit_index = 0;
518        let (commit_consumer, mut commit_receiver, _transaction_receiver) =
519            CommitConsumerArgs::new(0, last_processed_commit_index);
520        let leader_schedule = Arc::new(LeaderSchedule::from_store(
521            context.clone(),
522            dag_state.clone(),
523        ));
524
525        let mut observer = CommitObserver::new(
526            context.clone(),
527            commit_consumer,
528            dag_state.clone(),
529            transaction_certifier.clone(),
530            leader_schedule.clone(),
531        )
532        .await;
533
534        // Populate fully connected test blocks for round 0 ~ 10, authorities 0 ~ 3.
535        let num_rounds = 10;
536        let mut builder = DagBuilder::new(context.clone());
537        builder
538            .layers(1..=num_rounds)
539            .build()
540            .persist_layers(dag_state.clone());
541        transaction_certifier.add_voted_blocks(
542            builder
543                .all_blocks()
544                .iter()
545                .map(|b| (b.clone(), vec![]))
546                .collect(),
547        );
548
549        let leaders = builder
550            .leader_blocks(1..=num_rounds)
551            .into_iter()
552            .map(Option::unwrap)
553            .collect::<Vec<_>>();
554
555        // Commit first batch of leaders (2) and "receive" the subdags as the
556        // consumer of the consensus output channel.
557        let expected_last_processed_index: usize = 2;
558        let mut commits = observer
559            .handle_commit(leaders[..expected_last_processed_index].to_vec(), true)
560            .unwrap();
561
562        // Check commits sent over consensus output channel is accurate
563        let mut processed_subdag_index = 0;
564        while let Ok(Some(subdag)) = timeout(Duration::from_secs(1), commit_receiver.recv()).await {
565            tracing::info!("Processed {subdag}");
566            assert_eq!(subdag, commits[processed_subdag_index]);
567            assert_eq!(subdag.reputation_scores_desc, vec![]);
568            processed_subdag_index = subdag.commit_ref.index as usize;
569            if processed_subdag_index == expected_last_processed_index {
570                break;
571            }
572        }
573        assert_eq!(processed_subdag_index, expected_last_processed_index);
574
575        verify_channel_empty(&mut commit_receiver).await;
576
577        // Check last stored commit is correct
578        let last_commit = mem_store.read_last_commit().unwrap().unwrap();
579        assert_eq!(
580            last_commit.index(),
581            expected_last_processed_index as CommitIndex
582        );
583
584        // Handle next batch of leaders (10 - 2 = 8), these will be sent by consensus but not
585        // "processed" by consensus output channel. Simulating something happened on
586        // the consumer side where the commits were not persisted.
587        commits.append(
588            &mut observer
589                .handle_commit(leaders[expected_last_processed_index..].to_vec(), true)
590                .unwrap(),
591        );
592
593        let expected_last_sent_index = num_rounds as usize;
594        while let Ok(Some(subdag)) = timeout(Duration::from_secs(1), commit_receiver.recv()).await {
595            tracing::info!("{subdag} was sent but not processed by consumer");
596            assert_eq!(subdag, commits[processed_subdag_index]);
597            assert!(subdag.decided_with_local_blocks);
598            assert_eq!(subdag.reputation_scores_desc, vec![]);
599            processed_subdag_index = subdag.commit_ref.index as usize;
600            if processed_subdag_index == expected_last_sent_index {
601                break;
602            }
603        }
604        assert_eq!(processed_subdag_index, expected_last_sent_index);
605
606        verify_channel_empty(&mut commit_receiver).await;
607
608        // Check last stored commit is correct. We should persist the last commit
609        // that was sent over the channel regardless of how the consumer handled
610        // the commit on their end.
611        let last_commit = mem_store.read_last_commit().unwrap().unwrap();
612        assert_eq!(last_commit.index(), expected_last_sent_index as CommitIndex);
613
614        // Replay commits after index 2. And use last processed index by consumer at 10, which is the last persisted commit.
615        {
616            let replay_after_commit_index = 2;
617            let consumer_last_processed_commit_index = 10;
618            let dag_state = Arc::new(RwLock::new(DagState::new(
619                context.clone(),
620                mem_store.clone(),
621            )));
622            let (commit_consumer, mut commit_receiver, _transaction_receiver) =
623                CommitConsumerArgs::new(
624                    replay_after_commit_index,
625                    consumer_last_processed_commit_index,
626                );
627            let _observer = CommitObserver::new(
628                context.clone(),
629                commit_consumer,
630                dag_state.clone(),
631                transaction_certifier.clone(),
632                leader_schedule.clone(),
633            )
634            .await;
635
636            let mut processed_subdag_index = replay_after_commit_index;
637            while let Ok(Some(mut subdag)) =
638                timeout(Duration::from_secs(1), commit_receiver.recv()).await
639            {
640                tracing::info!("Received {subdag} on recovery");
641                assert_eq!(subdag.commit_ref.index, processed_subdag_index + 1);
642                assert!(subdag.recovered_rejected_transactions);
643
644                // Allow comparison with committed subdag before recovery.
645                subdag.recovered_rejected_transactions = false;
646                assert_eq!(subdag, commits[processed_subdag_index as usize]);
647
648                assert!(subdag.decided_with_local_blocks);
649                assert_eq!(subdag.reputation_scores_desc, vec![]);
650                processed_subdag_index = subdag.commit_ref.index;
651                if processed_subdag_index == consumer_last_processed_commit_index {
652                    break;
653                }
654            }
655            assert_eq!(processed_subdag_index, consumer_last_processed_commit_index);
656
657            verify_channel_empty(&mut commit_receiver).await;
658        }
659
660        // Replay commits from index 10, which is the last persisted commit.
661        {
662            let replay_after_commit_index = 10;
663            let consumer_last_processed_commit_index = 10;
664            let dag_state = Arc::new(RwLock::new(DagState::new(
665                context.clone(),
666                mem_store.clone(),
667            )));
668            // Re-create commit observer starting after index 10 which represents the
669            // last processed index from the consumer over consensus output channel
670            let (commit_consumer, mut commit_receiver, _transaction_receiver) =
671                CommitConsumerArgs::new(
672                    replay_after_commit_index,
673                    consumer_last_processed_commit_index,
674                );
675            let _observer = CommitObserver::new(
676                context.clone(),
677                commit_consumer,
678                dag_state.clone(),
679                transaction_certifier.clone(),
680                leader_schedule.clone(),
681            )
682            .await;
683
684            // No commits should be resubmitted as consensus store's last commit index
685            // is equal to replay after index by consumer
686            verify_channel_empty(&mut commit_receiver).await;
687        }
688
689        // Replay commits after index 2. And use last processed index by consumer at 4, less than the last persisted commit.
690        {
691            let replay_after_commit_index = 2;
692            let consumer_last_processed_commit_index = 4;
693            let dag_state = Arc::new(RwLock::new(DagState::new(
694                context.clone(),
695                mem_store.clone(),
696            )));
697            let (commit_consumer, mut commit_receiver, _transaction_receiver) =
698                CommitConsumerArgs::new(
699                    replay_after_commit_index,
700                    consumer_last_processed_commit_index,
701                );
702            let _observer = CommitObserver::new(
703                context.clone(),
704                commit_consumer,
705                dag_state.clone(),
706                transaction_certifier.clone(),
707                leader_schedule.clone(),
708            )
709            .await;
710
711            // Checks that commits up to expected_last_sent_index are recovered as finalized.
712            // The fact that they are finalized and have been recorded in the store.
713            let mut processed_subdag_index = replay_after_commit_index;
714            while let Ok(Some(subdag)) =
715                timeout(Duration::from_secs(1), commit_receiver.recv()).await
716            {
717                tracing::info!("Received {subdag} on recovery");
718                assert_eq!(subdag.commit_ref.index, processed_subdag_index + 1);
719                assert!(subdag.decided_with_local_blocks);
720                assert_eq!(subdag.reputation_scores_desc, vec![]);
721                processed_subdag_index = subdag.commit_ref.index;
722                if processed_subdag_index == expected_last_sent_index as CommitIndex {
723                    break;
724                }
725            }
726            assert_eq!(
727                processed_subdag_index,
728                expected_last_sent_index as CommitIndex
729            );
730
731            verify_channel_empty(&mut commit_receiver).await;
732        }
733
734        // Replay commits after index 2. And use last processed index by consumer at 20,
735        // which is greater than the last persisted commit.
736        // This allows removing from the store a suffix of commits which have not been part of certified checkpoints.
737        {
738            let replay_after_commit_index = 2;
739            let consumer_last_processed_commit_index = 20;
740            let dag_state = Arc::new(RwLock::new(DagState::new(
741                context.clone(),
742                mem_store.clone(),
743            )));
744            let (commit_consumer, mut commit_receiver, _transaction_receiver) =
745                CommitConsumerArgs::new(
746                    replay_after_commit_index,
747                    consumer_last_processed_commit_index,
748                );
749            let _observer = CommitObserver::new(
750                context.clone(),
751                commit_consumer,
752                dag_state.clone(),
753                transaction_certifier.clone(),
754                leader_schedule.clone(),
755            )
756            .await;
757
758            // Check commits sent over consensus output channel is accurate starting
759            // from last processed index of 2 and finishing at last sent index of 10.
760            let mut processed_subdag_index = replay_after_commit_index;
761            while let Ok(Some(mut subdag)) =
762                timeout(Duration::from_secs(1), commit_receiver.recv()).await
763            {
764                tracing::info!("Received {subdag} on recovery");
765                assert_eq!(subdag.commit_ref.index, processed_subdag_index + 1);
766                assert!(subdag.recovered_rejected_transactions);
767
768                // Allow comparison with committed subdag before recovery.
769                subdag.recovered_rejected_transactions = false;
770                assert_eq!(subdag, commits[processed_subdag_index as usize]);
771
772                assert!(subdag.decided_with_local_blocks);
773                assert_eq!(subdag.reputation_scores_desc, vec![]);
774                processed_subdag_index = subdag.commit_ref.index;
775                if processed_subdag_index == expected_last_sent_index as CommitIndex {
776                    break;
777                }
778            }
779            assert_eq!(
780                processed_subdag_index,
781                expected_last_sent_index as CommitIndex
782            );
783            assert_eq!(10, expected_last_sent_index);
784
785            verify_channel_empty(&mut commit_receiver).await;
786        }
787    }

789    /// After receiving all expected subdags, ensure channel is empty
790    async fn verify_channel_empty(receiver: &mut UnboundedReceiver<CommittedSubDag>) {
791        if let Ok(Some(_)) = timeout(Duration::from_secs(1), receiver.recv()).await {
792            panic!("Expected the consensus output channel to be empty, but found more subdags.")
793        }
794    }
