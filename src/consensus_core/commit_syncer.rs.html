4//! CommitSyncer implements efficient synchronization of committed data.
5//!
6//! During the operation of a committee of authorities for consensus, one or more authorities
7//! can fall behind the quorum in their received and accepted blocks. This can happen due to
8//! network disruptions, host crash, or other reasons. Authorities fell behind need to catch up to
9//! the quorum to be able to vote on the latest leaders. So efficient synchronization is necessary
10//! to minimize the impact of temporary disruptions and maintain smooth operations of the network.
11//!  
12//! CommitSyncer achieves efficient synchronization by relying on the following: when blocks
13//! are included in commits with >= 2f+1 certifiers by stake, these blocks must have passed
14//! verifications on some honest validators, so re-verifying them is unnecessary. In fact, the
15//! quorum certified commits themselves can be trusted to be sent to Sui directly, but for
16//! simplicity this is not done. Blocks from trusted commits still go through Core and committer.
17//!
18//! Another way CommitSyncer improves the efficiency of synchronization is parallel fetching:
19//! commits have a simple dependency graph (linear), so it is easy to fetch ranges of commits
20//! in parallel.
21//!
22//! Commit synchronization is an expensive operation, involving transferring large amount of data via
23//! the network. And it is not on the critical path of block processing. So the heuristics for
24//! synchronization, including triggers and retries, should be chosen to favor throughput and
25//! efficient resource usage, over faster reactions.
26

68// Handle to stop the CommitSyncer loop.
69pub(crate) struct CommitSyncerHandle {
70    schedule_task: JoinHandle<()>,
71    tx_shutdown: oneshot::Sender<()>,
72}
73
74impl CommitSyncerHandle {
75    pub(crate) async fn stop(self) {
76        let _ = self.tx_shutdown.send(());
77        // Do not abort schedule task, which waits for fetches to shut down.
78        if let Err(e) = self.schedule_task.await
79            && e.is_panic()
80        {
81            std::panic::resume_unwind(e.into_panic());
82        }
83    }
84}

86pub(crate) struct CommitSyncer<C: NetworkClient> {
87    // States shared by scheduler and fetch tasks.
88
89    // Shared components wrapper.
90    inner: Arc<Inner<C>>,
91
92    // States only used by the scheduler.
93
94    // Inflight requests to fetch commits from different authorities.
95    inflight_fetches: JoinSet<(u32, CertifiedCommits)>,
96    // Additional ranges of commits to fetch.
97    pending_fetches: BTreeSet<CommitRange>,
98    // Fetched commits and blocks by commit range.
99    fetched_ranges: BTreeMap<CommitRange, CertifiedCommits>,
100    // Highest commit index among inflight and pending fetches.
101    // Used to determine the start of new ranges to be fetched.
102    highest_scheduled_index: Option<CommitIndex>,
103    // Highest index among fetched commits, after commits and blocks are verified.
104    // Used for metrics.
105    highest_fetched_commit_index: CommitIndex,
106    // The commit index that is the max of highest local commit index and commit index inflight to Core.
107    // Used to determine if fetched blocks can be sent to Core without gaps.
108    synced_commit_index: CommitIndex,
109}

146    pub(crate) fn start(self) -> CommitSyncerHandle {
147        let (tx_shutdown, rx_shutdown) = oneshot::channel();
148        let schedule_task = spawn_logged_monitored_task!(self.schedule_loop(rx_shutdown,));
149        CommitSyncerHandle {
150            schedule_task,
151            tx_shutdown,
152        }
153    }

155    async fn schedule_loop(mut self, mut rx_shutdown: oneshot::Receiver<()>) {
156        let mut interval = tokio::time::interval(Duration::from_secs(2));
157        interval.set_missed_tick_behavior(MissedTickBehavior::Skip);
158
159        loop {
160            tokio::select! {
161                // Periodically, schedule new fetches if the node is falling behind.
162                _ = interval.tick() => {
163                    self.try_schedule_once();
164                }
165                // Handles results from fetch tasks.
166                Some(result) = self.inflight_fetches.join_next(), if !self.inflight_fetches.is_empty() => {
167                    if let Err(e) = result {
168                        if e.is_panic() {
169                            std::panic::resume_unwind(e.into_panic());
170                        }
171                        warn!("Fetch cancelled. CommitSyncer shutting down: {}", e);
172                        // If any fetch is cancelled or panicked, try to shutdown and exit the loop.
173                        self.inflight_fetches.shutdown().await;
174                        return;
175                    }
176                    let (target_end, commits) = result.unwrap();
177                    self.handle_fetch_result(target_end, commits).await;
178                }
179                _ = &mut rx_shutdown => {
180                    // Shutdown requested.
181                    info!("CommitSyncer shutting down ...");
182                    self.inflight_fetches.shutdown().await;
183                    return;
184                }
185            }
186
187            self.try_start_fetches();
188        }
189    }
190
191    fn try_schedule_once(&mut self) {
192        let quorum_commit_index = self.inner.commit_vote_monitor.quorum_commit_index();
193        let local_commit_index = self.inner.dag_state.read().last_commit_index();
194        let metrics = &self.inner.context.metrics.node_metrics;
195        metrics
196            .commit_sync_quorum_index
197            .set(quorum_commit_index as i64);
198        metrics
199            .commit_sync_local_index
200            .set(local_commit_index as i64);
201        let highest_handled_index = self.inner.commit_consumer_monitor.highest_handled_commit();
202        let highest_scheduled_index = self.highest_scheduled_index.unwrap_or(0);
203        // Update synced_commit_index periodically to make sure it is no smaller than
204        // local commit index.
205        self.synced_commit_index = self.synced_commit_index.max(local_commit_index);
206        let unhandled_commits_threshold = self.unhandled_commits_threshold();
207        info!(
208            "Checking to schedule fetches: synced_commit_index={}, highest_handled_index={}, highest_scheduled_index={}, quorum_commit_index={}, unhandled_commits_threshold={}",
209            self.synced_commit_index,
210            highest_handled_index,
211            highest_scheduled_index,
212            quorum_commit_index,
213            unhandled_commits_threshold,
214        );
215
216        // TODO: cleanup inflight fetches that are no longer needed.
217        let fetch_after_index = self
218            .synced_commit_index
219            .max(self.highest_scheduled_index.unwrap_or(0));
220        // When the node is falling behind, schedule pending fetches which will be executed on later.
221        for prev_end in (fetch_after_index..=quorum_commit_index)
222            .step_by(self.inner.context.parameters.commit_sync_batch_size as usize)
223        {
224            // Create range with inclusive start and end.
225            let range_start = prev_end + 1;
226            let range_end = prev_end + self.inner.context.parameters.commit_sync_batch_size;
227            // Commit range is not fetched when [range_start, range_end] contains less number of commits
228            // than the target batch size. This is to avoid the cost of processing more and smaller batches.
229            // Block broadcast, subscription and synchronization will help the node catchup.
230            if quorum_commit_index < range_end {
231                break;
232            }
233            // Pause scheduling new fetches when handling of commits is lagging.
234            if highest_handled_index + unhandled_commits_threshold < range_end {
235                warn!(
236                    "Skip scheduling new commit fetches: consensus handler is lagging. highest_handled_index={}, highest_scheduled_index={}",
237                    highest_handled_index, highest_scheduled_index
238                );
239                break;
240            }
241            self.pending_fetches
242                .insert((range_start..=range_end).into());
243            // quorum_commit_index should be non-decreasing, so highest_scheduled_index should not
244            // decrease either.
245            self.highest_scheduled_index = Some(range_end);
246        }
247    }

249    async fn handle_fetch_result(
250        &mut self,
251        target_end: CommitIndex,
252        certified_commits: CertifiedCommits,
253    ) {
254        assert!(!certified_commits.commits().is_empty());
255
256        let (total_blocks_fetched, total_blocks_size_bytes) = certified_commits
257            .commits()
258            .iter()
259            .fold((0, 0), |(blocks, bytes), c| {
260                (
261                    blocks + c.blocks().len(),
262                    bytes
263                        + c.blocks()
264                            .iter()
265                            .map(|b| b.serialized().len())
266                            .sum::<usize>() as u64,
267                )
268            });
269
270        let metrics = &self.inner.context.metrics.node_metrics;
271        metrics
272            .commit_sync_fetched_commits
273            .inc_by(certified_commits.commits().len() as u64);
274        metrics
275            .commit_sync_fetched_blocks
276            .inc_by(total_blocks_fetched as u64);
277        metrics
278            .commit_sync_total_fetched_blocks_size
279            .inc_by(total_blocks_size_bytes);
280
281        let (commit_start, commit_end) = (
282            certified_commits.commits().first().unwrap().index(),
283            certified_commits.commits().last().unwrap().index(),
284        );
285        self.highest_fetched_commit_index = self.highest_fetched_commit_index.max(commit_end);
286        metrics
287            .commit_sync_highest_fetched_index
288            .set(self.highest_fetched_commit_index as i64);
289
290        // Allow returning partial results, and try fetching the rest separately.
291        if commit_end < target_end {
292            self.pending_fetches
293                .insert((commit_end + 1..=target_end).into());
294        }
295        // Make sure synced_commit_index is up to date.
296        self.synced_commit_index = self
297            .synced_commit_index
298            .max(self.inner.dag_state.read().last_commit_index());
299        // Only add new blocks if at least some of them are not already synced.
300        if self.synced_commit_index < commit_end {
301            self.fetched_ranges
302                .insert((commit_start..=commit_end).into(), certified_commits);
303        }
304        // Try to process as many fetched blocks as possible.
305        while let Some((fetched_commit_range, _commits)) = self.fetched_ranges.first_key_value() {
306            // Only pop fetched_ranges if there is no gap with blocks already synced.
307            // Note: start, end and synced_commit_index are all inclusive.
308            let (fetched_commit_range, commits) =
309                if fetched_commit_range.start() <= self.synced_commit_index + 1 {
310                    self.fetched_ranges.pop_first().unwrap()
311                } else {
312                    // Found gap between earliest fetched block and latest synced block,
313                    // so not sending additional blocks to Core.
314                    metrics.commit_sync_gap_on_processing.inc();
315                    break;
316                };
317            // Avoid sending to Core a whole batch of already synced blocks.
318            if fetched_commit_range.end() <= self.synced_commit_index {
319                continue;
320            }
321
322            debug!(
323                "Fetched blocks for commit range {:?}: {}",
324                fetched_commit_range,
325                commits
326                    .commits()
327                    .iter()
328                    .flat_map(|c| c.blocks())
329                    .map(|b| b.reference().to_string())
330                    .join(","),
331            );
332
333            // If core thread cannot handle the incoming blocks, it is ok to block here
334            // to slow down the commit syncer.
335            match self
336                .inner
337                .core_thread_dispatcher
338                .add_certified_commits(commits)
339                .await
340            {
341                // Missing ancestors are possible from certification blocks, but
342                // it is unnecessary to try to sync their causal history. If they are required
343                // for the progress of the DAG, they will be included in a future commit.
344                Ok(missing) => {
345                    if !missing.is_empty() {
346                        info!(
347                            "Certification blocks have missing ancestors: {} for commit range {:?}",
348                            missing.iter().map(|b| b.to_string()).join(","),
349                            fetched_commit_range,
350                        );
351                    }
352                    for block_ref in missing {
353                        let hostname = &self
354                            .inner
355                            .context
356                            .committee
357                            .authority(block_ref.author)
358                            .hostname;
359                        metrics
360                            .commit_sync_fetch_missing_blocks
361                            .with_label_values(&[hostname])
362                            .inc();
363                    }
364                }
365                Err(e) => {
366                    info!("Failed to add blocks, shutting down: {}", e);
367                    return;
368                }
369            };
370
371            // Once commits and blocks are sent to Core, ratchet up synced_commit_index
372            self.synced_commit_index = self.synced_commit_index.max(fetched_commit_range.end());
373        }
374
375        metrics
376            .commit_sync_inflight_fetches
377            .set(self.inflight_fetches.len() as i64);
378        metrics
379            .commit_sync_pending_fetches
380            .set(self.pending_fetches.len() as i64);
381        metrics
382            .commit_sync_highest_synced_index
383            .set(self.synced_commit_index as i64);
384    }

386    fn try_start_fetches(&mut self) {
387        // Cap parallel fetches based on configured limit and committee size, to avoid overloading the network.
388        // Also when there are too many fetched blocks that cannot be sent to Core before an earlier fetch
389        // has not finished, reduce parallelism so the earlier fetch can retry on a better host and succeed.
390        let target_parallel_fetches = self
391            .inner
392            .context
393            .parameters
394            .commit_sync_parallel_fetches
395            .min(self.inner.context.committee.size() * 2 / 3)
396            .min(
397                self.inner
398                    .context
399                    .parameters
400                    .commit_sync_batches_ahead
401                    .saturating_sub(self.fetched_ranges.len()),
402            )
403            .max(1);
404        // Start new fetches if there are pending batches and available slots.
405        loop {
406            if self.inflight_fetches.len() >= target_parallel_fetches {
407                break;
408            }
409            let Some(commit_range) = self.pending_fetches.pop_first() else {
410                break;
411            };
412            self.inflight_fetches
413                .spawn(Self::fetch_loop(self.inner.clone(), commit_range));
414        }
415
416        let metrics = &self.inner.context.metrics.node_metrics;
417        metrics
418            .commit_sync_inflight_fetches
419            .set(self.inflight_fetches.len() as i64);
420        metrics
421            .commit_sync_pending_fetches
422            .set(self.pending_fetches.len() as i64);
423        metrics
424            .commit_sync_highest_synced_index
425            .set(self.synced_commit_index as i64);
426    }

428    // Retries fetching commits and blocks from available authorities, until a request succeeds
429    // where at least a prefix of the commit range is fetched.
430    // Returns the fetched commits and blocks referenced by the commits.
431    async fn fetch_loop(
432        inner: Arc<Inner<C>>,
433        commit_range: CommitRange,
434    ) -> (CommitIndex, CertifiedCommits) {
435        // Individual request base timeout.
436        const TIMEOUT: Duration = Duration::from_secs(10);
437        // Max per-request timeout will be base timeout times a multiplier.
438        // At the extreme, this means there will be 120s timeout to fetch max_blocks_per_fetch blocks.
439        const MAX_TIMEOUT_MULTIPLIER: u32 = 12;
440        // timeout * max number of targets should be reasonably small, so the
441        // system can adjust to slow network or large data sizes quickly.
442        const MAX_NUM_TARGETS: usize = 24;
443        let mut timeout_multiplier = 0;
444        let _timer = inner
445            .context
446            .metrics
447            .node_metrics
448            .commit_sync_fetch_loop_latency
449            .start_timer();
450        info!("Starting to fetch commits in {commit_range:?} ...",);
451        loop {
452            // Attempt to fetch commits and blocks through min(committee size, MAX_NUM_TARGETS) peers.
453            let mut target_authorities = inner
454                .context
455                .committee
456                .authorities()
457                .filter_map(|(i, _)| {
458                    if i != inner.context.own_index {
459                        Some(i)
460                    } else {
461                        None
462                    }
463                })
464                .collect_vec();
465            target_authorities.shuffle(&mut ThreadRng::default());
466            target_authorities.truncate(MAX_NUM_TARGETS);
467            // Increase timeout multiplier for each loop until MAX_TIMEOUT_MULTIPLIER.
468            timeout_multiplier = (timeout_multiplier + 1).min(MAX_TIMEOUT_MULTIPLIER);
469            let request_timeout = TIMEOUT * timeout_multiplier;
470            // Give enough overall timeout for fetching commits and blocks.
471            // - Timeout for fetching commits and commit certifying blocks.
472            // - Timeout for fetching blocks referenced by the commits.
473            // - Time spent on pipelining requests to fetch blocks.
474            // - Another headroom to allow fetch_once() to timeout gracefully if possible.
475            let fetch_timeout = request_timeout * 4;
476            // Try fetching from selected target authority.
477            for authority in target_authorities {
478                match tokio::time::timeout(
479                    fetch_timeout,
480                    Self::fetch_once(
481                        inner.clone(),
482                        authority,
483                        commit_range.clone(),
484                        request_timeout,
485                    ),
486                )
487                .await
488                {
489                    Ok(Ok(commits)) => {
490                        info!("Finished fetching commits in {commit_range:?}",);
491                        return (commit_range.end(), commits);
492                    }
493                    Ok(Err(e)) => {
494                        let hostname = inner
495                            .context
496                            .committee
497                            .authority(authority)
498                            .hostname
499                            .clone();
500                        warn!("Failed to fetch {commit_range:?} from {hostname}: {}", e);
501                        inner
502                            .context
503                            .metrics
504                            .node_metrics
505                            .commit_sync_fetch_once_errors
506                            .with_label_values(&[hostname.as_str(), e.name()])
507                            .inc();
508                    }
509                    Err(_) => {
510                        let hostname = inner
511                            .context
512                            .committee
513                            .authority(authority)
514                            .hostname
515                            .clone();
516                        warn!("Timed out fetching {commit_range:?} from {authority}",);
517                        inner
518                            .context
519                            .metrics
520                            .node_metrics
521                            .commit_sync_fetch_once_errors
522                            .with_label_values(&[hostname.as_str(), "FetchTimeout"])
523                            .inc();
524                    }
525                }
526            }
527            // Avoid busy looping, by waiting for a while before retrying.
528            sleep(TIMEOUT).await;
529        }
530    }

532    // Fetches commits and blocks from a single authority. At a high level, first the commits are
533    // fetched and verified. After that, blocks referenced in the certified commits are fetched
534    // and sent to Core for processing.
535    async fn fetch_once(
536        inner: Arc<Inner<C>>,
537        target_authority: AuthorityIndex,
538        commit_range: CommitRange,
539        timeout: Duration,
540    ) -> ConsensusResult<CertifiedCommits> {
541        let _timer = inner
542            .context
543            .metrics
544            .node_metrics
545            .commit_sync_fetch_once_latency
546            .start_timer();
547
548        // 1. Fetch commits in the commit range from the target authority.
549        let (serialized_commits, serialized_blocks) = inner
550            .network_client
551            .fetch_commits(target_authority, commit_range.clone(), timeout)
552            .await?;
553
554        // 2. Verify the response contains blocks that can certify the last returned commit,
555        // and the returned commits are chained by digests, so earlier commits are certified
556        // as well.
557        let (commits, vote_blocks) = Handle::current()
558            .spawn_blocking({
559                let inner = inner.clone();
560                move || {
561                    inner.verify_commits(
562                        target_authority,
563                        commit_range,
564                        serialized_commits,
565                        serialized_blocks,
566                    )
567                }
568            })
569            .await
570            .expect("Spawn blocking should not fail")?;
571
572        // 3. Fetch blocks referenced by the commits, from the same peer where commits are fetched.
573        let mut block_refs: Vec<_> = commits.iter().flat_map(|c| c.blocks()).cloned().collect();
574        block_refs.sort();
575        let num_chunks = block_refs
576            .len()
577            .div_ceil(inner.context.parameters.max_blocks_per_fetch)
578            as u32;
579        let mut requests: FuturesOrdered<_> = block_refs
580            .chunks(inner.context.parameters.max_blocks_per_fetch)
581            .enumerate()
582            .map(|(i, request_block_refs)| {
583                let inner = inner.clone();
584                async move {
585                    // 4. Send out pipelined fetch requests to avoid overloading the target authority.
586                    sleep(timeout * i as u32 / num_chunks).await;
587                    // TODO: add some retries.
588                    let serialized_blocks = inner
589                        .network_client
590                        .fetch_blocks(
591                            target_authority,
592                            request_block_refs.to_vec(),
593                            vec![],
594                            false,
595                            timeout,
596                        )
597                        .await?;
598                    // 5. Verify the same number of blocks are returned as requested.
599                    if request_block_refs.len() != serialized_blocks.len() {
600                        return Err(ConsensusError::UnexpectedNumberOfBlocksFetched {
601                            authority: target_authority,
602                            requested: request_block_refs.len(),
603                            received: serialized_blocks.len(),
604                        });
605                    }
606                    // 6. Verify returned blocks have valid formats.
607                    let signed_blocks = serialized_blocks
608                        .iter()
609                        .map(|serialized| {
610                            let block: SignedBlock = bcs::from_bytes(serialized)
611                                .map_err(ConsensusError::MalformedBlock)?;
612                            Ok(block)
613                        })
614                        .collect::<ConsensusResult<Vec<_>>>()?;
615                    // 7. Verify the returned blocks match the requested block refs.
616                    // If they do match, the returned blocks can be considered verified as well.
617                    let mut blocks = Vec::new();
618                    for ((requested_block_ref, signed_block), serialized) in request_block_refs
619                        .iter()
620                        .zip(signed_blocks.into_iter())
621                        .zip(serialized_blocks.into_iter())
622                    {
623                        let signed_block_digest = VerifiedBlock::compute_digest(&serialized);
624                        let received_block_ref = BlockRef::new(
625                            signed_block.round(),
626                            signed_block.author(),
627                            signed_block_digest,
628                        );
629                        if *requested_block_ref != received_block_ref {
630                            return Err(ConsensusError::UnexpectedBlockForCommit {
631                                peer: target_authority,
632                                requested: *requested_block_ref,
633                                received: received_block_ref,
634                            });
635                        }
636                        blocks.push(VerifiedBlock::new_verified(signed_block, serialized));
637                    }
638                    Ok(blocks)
639                }
640            })
641            .collect();
642
643        let mut fetched_blocks = BTreeMap::new();
644        while let Some(result) = requests.next().await {
645            for block in result? {
646                fetched_blocks.insert(block.reference(), block);
647            }
648        }
649
650        // 8. Check if the block timestamps are lower than current time - this is for metrics only.
651        for block in fetched_blocks.values().chain(vote_blocks.iter()) {
652            let now_ms = inner.context.clock.timestamp_utc_ms();
653            let forward_drift = block.timestamp_ms().saturating_sub(now_ms);
654            if forward_drift == 0 {
655                continue;
656            };
657            let peer_hostname = &inner.context.committee.authority(target_authority).hostname;
658            inner
659                .context
660                .metrics
661                .node_metrics
662                .block_timestamp_drift_ms
663                .with_label_values(&[peer_hostname.as_str(), "commit_syncer"])
664                .inc_by(forward_drift);
665        }
666
667        // 9. Now create certified commits by assigning the blocks to each commit.
668        let mut certified_commits = Vec::new();
669        for commit in &commits {
670            let blocks = commit
671                .blocks()
672                .iter()
673                .map(|block_ref| {
674                    fetched_blocks
675                        .remove(block_ref)
676                        .expect("Block should exist")
677                })
678                .collect::<Vec<_>>();
679            certified_commits.push(CertifiedCommit::new_certified(commit.clone(), blocks));
680        }
681
682        // 10. Add blocks in certified commits to the transaction certifier.
683        for commit in &certified_commits {
684            for block in commit.blocks() {
685                // Only account for reject votes in the block, since they may vote on uncommitted
686                // blocks or transactions. It is unnecessary to vote on the committed blocks
687                // themselves.
688                if inner.context.protocol_config.mysticeti_fastpath() {
689                    inner
690                        .transaction_certifier
691                        .add_voted_blocks(vec![(block.clone(), vec![])]);
692                }
693            }
694        }
695
696        // 11. Record commit votes from the fetched blocks.
697        for commit in &certified_commits {
698            for block in commit.blocks() {
699                inner.commit_vote_monitor.observe_block(block);
700            }
701        }
702        for block in &vote_blocks {
703            inner.commit_vote_monitor.observe_block(block);
704        }
705
706        // 12. Update round tracker from the fetched blocks. For fetched blocks,
707        // excluded_ancestors are not available so we use an empty vector.
708        {
709            let mut tracker = inner.round_tracker.write();
710            // Update from commit blocks
711            for commit in &certified_commits {
712                for block in commit.blocks() {
713                    tracker.update_from_verified_block(&ExtendedBlock {
714                        block: block.clone(),
715                        excluded_ancestors: vec![],
716                    });
717                }
718            }
719            // Update from vote blocks
720            for block in &vote_blocks {
721                tracker.update_from_verified_block(&ExtendedBlock {
722                    block: block.clone(),
723                    excluded_ancestors: vec![],
724                });
725            }
726        }
727
728        Ok(CertifiedCommits::new(certified_commits, vote_blocks))
729    }

774impl<C: NetworkClient> Inner<C> {
775    /// Verifies the commits and also certifies them using the provided vote blocks for the last commit. The
776    /// method returns the trusted commits and the votes as verified blocks.
777    fn verify_commits(
778        &self,
779        peer: AuthorityIndex,
780        commit_range: CommitRange,
781        serialized_commits: Vec<Bytes>,
782        serialized_vote_blocks: Vec<Bytes>,
783    ) -> ConsensusResult<(Vec<TrustedCommit>, Vec<VerifiedBlock>)> {
784        // Parse and verify commits.
785        let mut commits = Vec::new();
786        for serialized in &serialized_commits {
787            let commit: Commit =
788                bcs::from_bytes(serialized).map_err(ConsensusError::MalformedCommit)?;
789            let digest = TrustedCommit::compute_digest(serialized);
790            if commits.is_empty() {
791                // start is inclusive, so first commit must be at the start index.
792                if commit.index() != commit_range.start() {
793                    return Err(ConsensusError::UnexpectedStartCommit {
794                        peer,
795                        start: commit_range.start(),
796                        commit: Box::new(commit),
797                    });
798                }
799            } else {
800                // Verify next commit increments index and references the previous digest.
801                let (last_commit_digest, last_commit): &(CommitDigest, Commit) =
802                    commits.last().unwrap();
803                if commit.index() != last_commit.index() + 1
804                    || &commit.previous_digest() != last_commit_digest
805                {
806                    return Err(ConsensusError::UnexpectedCommitSequence {
807                        peer,
808                        prev_commit: Box::new(last_commit.clone()),
809                        curr_commit: Box::new(commit),
810                    });
811                }
812            }
813            // Do not process more commits past the end index.
814            if commit.index() > commit_range.end() {
815                break;
816            }
817            commits.push((digest, commit));
818        }
819        let Some((end_commit_digest, end_commit)) = commits.last() else {
820            return Err(ConsensusError::NoCommitReceived { peer });
821        };
822
823        // Parse and verify blocks. Then accumulate votes on the end commit.
824        let end_commit_ref = CommitRef::new(end_commit.index(), *end_commit_digest);
825        let mut stake_aggregator = StakeAggregator::<QuorumThreshold>::new();
826        let mut vote_blocks = Vec::new();
827        for serialized in serialized_vote_blocks {
828            let block: SignedBlock =
829                bcs::from_bytes(&serialized).map_err(ConsensusError::MalformedBlock)?;
830            // Only block signatures need to be verified, to verify commit votes.
831            // But the blocks will be sent to Core, so they need to be fully verified.
832            let (block, reject_transaction_votes) =
833                self.block_verifier.verify_and_vote(block, serialized)?;
834            if self.context.protocol_config.mysticeti_fastpath() {
835                self.transaction_certifier
836                    .add_voted_blocks(vec![(block.clone(), reject_transaction_votes)]);
837            }
838            for vote in block.commit_votes() {
839                if *vote == end_commit_ref {
840                    stake_aggregator.add(block.author(), &self.context.committee);
841                }
842            }
843            vote_blocks.push(block);
844        }
845
846        // Check if the end commit has enough votes.
847        if !stake_aggregator.reached_threshold(&self.context.committee) {
848            return Err(ConsensusError::NotEnoughCommitVotes {
849                stake: stake_aggregator.stake(),
850                peer,
851                commit: Box::new(end_commit.clone()),
852            });
853        }
854
855        let trusted_commits = commits
856            .into_iter()
857            .zip(serialized_commits)
858            .map(|((_d, c), s)| TrustedCommit::new_trusted(c, s))
859            .collect();
860        Ok((trusted_commits, vote_blocks))
861    }
862}
