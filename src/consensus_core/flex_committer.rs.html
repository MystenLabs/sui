33/// FlexCommitter supports committing multiple and varying number of leaders per round,
34/// based on DagState and the schedule for next commit leaders.
35pub(crate) struct FlexCommitter {
36    context: Arc<Context>,
37    dag_state: Arc<RwLock<DagState>>,
38    slot_decider: LeaderSlotDecider,
39    pending_commit_state: PendingCommitState,
40}
41
42impl FlexCommitter {
43    pub(crate) fn new(context: Arc<Context>, dag_state: Arc<RwLock<DagState>>) -> Self {
44        let slot_decider = LeaderSlotDecider::new(context.clone(), dag_state.clone());
45        Self {
46            context,
47            dag_state,
48            slot_decider,
49            pending_commit_state: PendingCommitState::default(),
50        }
51    }

53    /// Attempts to create a commit based on the current DagState.
54    pub(crate) fn try_commit(
55        &mut self,
56        next_commit_leaders: NextCommitLeaderSchedule,
57    ) -> Option<(TrustedCommit, CommittedSubDag)> {
58        self.maybe_refresh_pending_commit_state(next_commit_leaders);
59
60        self.try_direct_commit();
61        if let Some(commit_leader_round) = self.find_commit_leader_round() {
62            return self.build_commit(commit_leader_round);
63        }
64
65        self.try_indirect_commit();
66        if let Some(commit_leader_round) = self.find_commit_leader_round() {
67            return self.build_commit(commit_leader_round);
68        }
69
70        None
71    }
72
73    fn maybe_refresh_pending_commit_state(
74        &mut self,
75        next_commit_leaders: NextCommitLeaderSchedule,
76    ) {
77        // pending_commit_state.next_commit_leaders contain the schedule for next commit,
78        // and are associated with the other internal state.
79        if self
80            .pending_commit_state
81            .next_commit_leaders
82            .next_commit_index
83            == next_commit_leaders.next_commit_index
84        {
85            // Use cached pending commit state for the same commit index.
86            return;
87        }
88        assert!(
89            self.pending_commit_state
90                .next_commit_leaders
91                .next_commit_index
92                < next_commit_leaders.next_commit_index,
93            "next_commit_index should only move forward: {} vs {}",
94            self.pending_commit_state
95                .next_commit_leaders
96                .next_commit_index,
97            next_commit_leaders.next_commit_index
98        );
99        tracing::debug!(
100            "Refreshing pending commit state: old_commit_index={}, new_commit_index={}, min_next_leader_round={}, num_leaders={}",
101            self.pending_commit_state
102                .next_commit_leaders
103                .next_commit_index,
104            next_commit_leaders.next_commit_index,
105            next_commit_leaders.min_next_leader_round,
106            next_commit_leaders.num_leaders(),
107        );
108        self.pending_commit_state = PendingCommitState {
109            next_commit_leaders,
110            rounds: vec![],
111        };
112    }

114    /// Runs the direct commit rule on every leader slot pending to be decided.
115    fn try_direct_commit(&mut self) {
116        let min_next_leader_round = self
117            .pending_commit_state
118            .next_commit_leaders
119            .min_next_leader_round;
120        let highest_accepted_round = self.dag_state.read().highest_accepted_round();
121
122        for leader_round in min_next_leader_round..highest_accepted_round {
123            let round_state = self
124                .pending_commit_state
125                .get_or_create_round_state(leader_round);
126            let slots_to_decide = round_state
127                .leader_slots
128                .iter()
129                .filter_map(|s| {
130                    if matches!(s.leader_status, LeaderStatus::Undecided(_)) {
131                        Some(s.slot)
132                    } else {
133                        None
134                    }
135                })
136                .collect::<Vec<_>>();
137            for slot in slots_to_decide {
138                let status = self.slot_decider.try_direct_decide(slot);
139                if status.is_decided() {
140                    tracing::debug!("Direct decision for slot {slot}: {status}");
141                }
142                round_state.update_slot_decision(slot, status, Decision::Direct);
143            }
144        }
145    }
146
147    fn try_indirect_commit(&mut self) {
148        let min_next_leader_round = self
149            .pending_commit_state
150            .next_commit_leaders
151            .min_next_leader_round;
152        let highest_accepted_round = self.dag_state.read().highest_accepted_round();
153        // Decide leaders from highest to lowest round, because lower rounds need anchor blocks in
154        // higher rounds to be decided first.
155        for round in (min_next_leader_round
156            ..=(highest_accepted_round.saturating_sub(INDIRECT_COMMIT_DEPTH)))
157            .rev()
158        {
159            let Some(anchor_block) = self.find_anchor_block(round + INDIRECT_COMMIT_DEPTH) else {
160                // Skip when there is no anchor block for this round yet.
161                continue;
162            };
163            self.decide_with_anchor_block(anchor_block.clone(), round);
164        }
165    }
166
167    fn find_anchor_block(&self, start_round: Round) -> Option<VerifiedBlock> {
168        let start_index = start_round
169            - self
170                .pending_commit_state
171                .next_commit_leaders
172                .min_next_leader_round;
173        for index in start_index as usize..self.pending_commit_state.rounds.len() {
174            let round_state = &self.pending_commit_state.rounds[index];
175            for slot in &round_state.leader_slots {
176                match &slot.leader_status {
177                    // First committed block becomes the anchor block, when there is no undecided slot before it.
178                    // Every validator will select the same anchor block.
179                    // This rule of choosing anchor block minimizes the chance of encountering undecided slots before it.
180                    LeaderStatus::Commit(block) => return Some(block.clone()),
181                    // There cannot be an anchor block after an undecided.
182                    LeaderStatus::Undecided(_) => return None,
183                    // Continue searching for an anchor block after skipped slots.
184                    LeaderStatus::Skip(_) => {}
185                }
186            }
187        }
188        None
189    }
190
191    // Finds leader slots on `decision_round` and calls indirect commit rule with `anchor_block`
192    // to decide on their statuses.
193    fn decide_with_anchor_block(&mut self, anchor_block: VerifiedBlock, decision_round: Round) {
194        let round_state = self
195            .pending_commit_state
196            .get_or_create_round_state(decision_round);
197        if round_state.undecided_slots.is_empty() {
198            // Skip indirect commit when the round has already been fully decided.
199            return;
200        }
201
202        let slots: Vec<Slot> = round_state.leader_slots.iter().map(|s| s.slot).collect();
203        let statuses = self.slot_decider.try_indirect_decide(&anchor_block, &slots);
204        let anchor_ref = anchor_block.reference();
205        for (slot, status) in slots.into_iter().zip_eq(statuses) {
206            // Only log slots the indirect rule actually decides. Slots already decided
207            // by the direct rule are re-evaluated (and asserted consistent) but their
208            // decision is kept, so they should not appear as indirect decisions.
209            if round_state.undecided_slots.contains(&slot) {
210                tracing::debug!(
211                    "Indirect decision for slot {slot} using anchor {anchor_ref}: {status}"
212                );
213            }
214            round_state.update_slot_decision(slot, status, Decision::Indirect);
215        }
216    }

218    // The next commit leader round must have all slots decided in this and earlier rounds,
219    // and at least one slot in this round committed.
220    fn find_commit_leader_round(&self) -> Option<Round> {
221        for round_state in &self.pending_commit_state.rounds {
222            if !round_state.undecided_slots.is_empty() {
223                // Found undecided slot, so this and later rounds cannot be committed yet.
224                return None;
225            }
226            if round_state.num_committed > 0 {
227                // Found a fully decided round with at least one committed leader, so it can be committed.
228                return Some(round_state.round);
229            }
230            // This round is fully decided but has no committed leader, so it is skipped.
231        }
232        None
233    }
234
235    /// Builds a single commit with leaders from `commit_leader_round`.
236    ///
237    /// Traversal starts from all committed leaders in `commit_leader_round`,
238    /// read from `pending_commit_state`. Other committed blocks are selected
239    /// via DFS from the leaders. Only uncommitted blocks above GC round are selected.
240    ///
241    /// Committed blocks are ordered deterministically: by round ascending, then
242    /// by `hash(seed || block_digest)` within a round. The last block of the
243    /// sorted leader round becomes the commit's named `leader`.
244    fn build_commit(
245        &mut self,
246        commit_leader_round: Round,
247    ) -> Option<(TrustedCommit, CommittedSubDag)> {
248        let round_state = self
249            .pending_commit_state
250            .get_round_state(commit_leader_round);
251        let mut committed_leaders = Vec::new();
252        for s in &round_state.leader_slots {
253            // Record every decided leader slot in the commit round: the authority
254            // whose slot it is, how it was decided (direct/indirect), and whether
255            // it was committed or skipped.
256            let (authority, outcome) = match &s.leader_status {
257                LeaderStatus::Commit(block) => {
258                    committed_leaders.push(block.clone());
259                    (block.author(), "commit")
260                }
261                LeaderStatus::Skip(slot) => (slot.authority, "skip"),
262                LeaderStatus::Undecided(slot) => panic!("Unexpected undecided slot {}", slot),
263            };
264            let decision = match s.decision.expect("Decided slot must have a decision") {
265                Decision::Direct => "direct",
266                Decision::Indirect => "indirect",
267                Decision::Certified => "certified",
268            };
269            let commit_type = format!("{decision}-{outcome}");
270            let leader_host: &str = &self.context.committee.authority(authority).hostname;
271            self.context
272                .metrics
273                .node_metrics
274                .committed_leaders_total
275                .with_label_values(&[leader_host, &commit_type])
276                .inc();
277        }
278        assert!(!committed_leaders.is_empty(), "No committed leaders found");
279
280        let mut dag_state = self.dag_state.write();
281        let last_commit_digest = dag_state.last_commit_digest();
282        let gc_round = dag_state.gc_round();
283
284        // DFS from all committed leaders. For each block, select its uncommitted
285        // ancestors above gc_round, mark them committed, and continue until the
286        // frontier is empty.
287        for leader in &committed_leaders {
288            assert!(
289                dag_state.set_committed(&leader.reference()),
290                "Leader block {:?} attempted to be committed twice",
291                leader.reference()
292            );
293        }
294        let mut to_visit: Vec<VerifiedBlock> = committed_leaders.clone();
295        let mut to_commit: Vec<VerifiedBlock> = Vec::new();
296        while let Some(block) = to_visit.pop() {
297            to_commit.push(block.clone());
298            let uncommitted_ancestor_refs: Vec<BlockRef> = block
299                .ancestors()
300                .iter()
301                .copied()
302                .filter(|a| a.round > gc_round && !dag_state.is_committed(a))
303                .collect();
304            for ancestor_ref in uncommitted_ancestor_refs {
305                if !dag_state.set_committed(&ancestor_ref) {
306                    continue;
307                }
308                to_visit.push(dag_state.get_block(&ancestor_ref).unwrap());
309            }
310        }
311        assert!(
312            to_commit.iter().all(|b| b.round() > gc_round),
313            "No blocks at or below gc_round {gc_round} should be committed"
314        );
315        drop(dag_state);
316
317        // Sort the committed blocks deterministically, first by round ascending,
318        // then by deterministic hash specific to this commit and each block.
319        let seed = compute_sort_seed(&committed_leaders);
320        sort_committed_blocks(&mut to_commit, &seed);
321
322        // Named leader = the committed leader that ended up last among the
323        // committed leaders in the sorted sub-dag.
324        let leader_ref = to_commit
325            .last()
326            .map(|b| b.reference())
327            .expect("At least one committed leader must be in to_commit");
328        assert_eq!(leader_ref.round, commit_leader_round);
329
330        // Compute deterministic commit timestamp from leader parents.
331        let timestamp_ms =
332            calculate_commit_timestamp(&self.context, &self.dag_state.read(), &committed_leaders);
333
334        let commit = Commit::new(
335            self.pending_commit_state
336                .next_commit_leaders
337                .next_commit_index,
338            last_commit_digest,
339            timestamp_ms,
340            leader_ref,
341            to_commit.iter().map(|b| b.reference()).collect(),
342        );
343        let serialized = commit
344            .serialize()
345            .unwrap_or_else(|e| panic!("Failed to serialize commit: {e}"));
346        let trusted_commit = TrustedCommit::new_trusted(commit, serialized);
347        let sub_dag = CommittedSubDag::new(
348            leader_ref,
349            to_commit,
350            timestamp_ms,
351            trusted_commit.reference(),
352        );
353
354        tracing::debug!(
355            "Built commit {} for leader round {} with {} committed leaders and {} blocks. Scheduled min next leader round {}, num allowed leaders {}",
356            trusted_commit.reference(),
357            commit_leader_round,
358            committed_leaders.len(),
359            sub_dag.blocks.len(),
360            self.pending_commit_state
361                .next_commit_leaders
362                .min_next_leader_round,
363            self.pending_commit_state
364                .next_commit_leaders
365                .allowed_leaders
366                .len(),
367        );
368
369        Some((trusted_commit, sub_dag))
370    }
371
372    /// Builds a CommittedSubDag for a certified commit synced from a peer, and
373    /// marks every committed block in DagState. The commit is already
374    /// quorum-certified, so the commit rule is skipped here — unlike local
375    /// commits, which are produced by `build_commit`.
376    pub(crate) fn handle_certified_commit(&self, commit: &TrustedCommit) -> CommittedSubDag {
377        tracing::debug!(
378            "Handling certified commit {} with leader {} and {} blocks",
379            commit.reference(),
380            commit.leader(),
381            commit.blocks().len()
382        );
383        let mut dag_state = self.dag_state.write();
384        for block_ref in commit.blocks() {
385            assert!(
386                dag_state.set_committed(block_ref),
387                "Block {:?} attempted to be committed twice",
388                block_ref
389            );
390        }
391        let blocks: Vec<VerifiedBlock> = dag_state
392            .get_blocks(commit.blocks())
393            .into_iter()
394            .map(|b| b.expect("All blocks referenced by a trusted commit must be in DagState"))
395            .collect();
396        drop(dag_state);
397
398        let mut subdag = CommittedSubDag::new(
399            commit.leader(),
400            blocks,
401            commit.timestamp_ms(),
402            commit.reference(),
403        );
404        subdag.decided_with_local_blocks = false;
405        subdag
406    }

409/// Tracks intermediate state related to generating the next commit.
410/// This state resets when encountering a more advanced leader schedule.
411#[derive(Clone, Debug, Default)]
412struct PendingCommitState {
413    // Leader schedule for generating the upcoming commit.
414    next_commit_leaders: NextCommitLeaderSchedule,
415    // Intermediate state per round.
416    rounds: Vec<RoundState>,
417}
418
419impl PendingCommitState {
420    /// Returns the round' state and panics if it doesn't exist.
421    /// Use `get_or_create_round_state` to create it if needed.
422    fn get_round_state(&mut self, round: Round) -> &mut RoundState {
423        let index = round
424            .checked_sub(self.next_commit_leaders.min_next_leader_round)
425            .unwrap() as usize;
426        self.rounds.get_mut(index).unwrap()
427    }
428
429    /// Populate state up to round, if they don't exist yet. Then return the round's state.
430    fn get_or_create_round_state(&mut self, round: Round) -> &mut RoundState {
431        let add_from_round = self
432            .rounds
433            .last()
434            .map(|state| state.round + 1)
435            .unwrap_or(self.next_commit_leaders.min_next_leader_round);
436        // No-op if the round's state exists.
437        for r in add_from_round..=round {
438            let mut seed_bytes = [0u8; 32];
439            seed_bytes[28..].copy_from_slice(&r.to_le_bytes());
440            let mut rng = StdRng::from_seed(seed_bytes);
441            let mut leaders = self.next_commit_leaders.allowed_leaders.clone();
442            leaders.shuffle(&mut rng);
443            let leader_slots: Vec<LeaderSlot> = leaders
444                .into_iter()
445                .map(|leader| {
446                    let slot = Slot::new(r, leader);
447                    LeaderSlot {
448                        slot,
449                        leader_status: LeaderStatus::Undecided(slot),
450                        decision: None,
451                    }
452                })
453                .collect();
454            let undecided_slots: BTreeSet<Slot> = leader_slots.iter().map(|s| s.slot).collect();
455            self.rounds.push(RoundState {
456                round: r,
457                leader_slots,
458                undecided_slots,
459                num_committed: 0,
460            });
461        }
462
463        self.get_round_state(round)
464    }
465}

467/// Per-round intermediate state computed with commit rule and current leader schedule
468#[derive(Clone, Debug)]
469struct RoundState {
470    round: Round,
471    // Leader slots selected in the round, in order.
472    leader_slots: Vec<LeaderSlot>,
473    // Leader slots that haven't decided.
474    undecided_slots: BTreeSet<Slot>,
475    // Number of leader slots voted to be committed.
476    // They may not appear in an actual commit as leaders though.
477    num_committed: usize,
478}
479
480impl RoundState {
481    fn update_slot_decision(&mut self, slot: Slot, new_status: LeaderStatus, decision: Decision) {
482        let leader_slot = self
483            .leader_slots
484            .iter_mut()
485            .find(|s| s.slot == slot)
486            .unwrap_or_else(|| panic!("Slot {} must be in round {}", slot, self.round));
487        if matches!(new_status, LeaderStatus::Undecided(_)) {
488            if matches!(leader_slot.leader_status, LeaderStatus::Undecided(_)) {
489                // No-op if the status is still undecided.
490                return;
491            } else {
492                // Undeciding a previously decided slot should not happen.
493                panic!(
494                    "Cannot undecide slot {}: {:?} vs {:?}",
495                    slot, leader_slot.leader_status, new_status
496                );
497            }
498        }
499        if self.undecided_slots.remove(&slot) {
500            if matches!(new_status, LeaderStatus::Commit(_)) {
501                self.num_committed += 1;
502            }
503            leader_slot.leader_status = new_status;
504            // Record the rule that first decided this slot. A slot decided by the direct
505            // rule may be re-evaluated by the indirect rule in the same round, but the
506            // initial decision is kept (the else branch leaves `decision` untouched).
507            leader_slot.decision = Some(decision);
508        } else {
509            assert_eq!(
510                leader_slot.leader_status, new_status,
511                "Cannot update slot {} decision: {:?} vs {:?}",
512                slot, leader_slot.leader_status, new_status
513            );
514        }
515    }
516}

518#[derive(Clone, Debug)]
519struct LeaderSlot {
520    slot: Slot,
521    leader_status: LeaderStatus,
522    // Which commit rule decided the slot (direct or indirect), or None while undecided.
523    decision: Option<Decision>,
524}

526/// From a deterministic array of leaders, compute a deterministic digest used as seed for sort.
527fn compute_sort_seed(committed_leaders: &[VerifiedBlock]) -> [u8; DIGEST_LENGTH] {
528    let mut hasher = DefaultHashFunction::new();
529    for leader in committed_leaders {
530        hasher.update(leader.digest().as_ref());
531    }
532    hasher.finalize().into()
533}
534
535/// Per-block sort key:
536/// - Primary part: block round.
537/// - Secondary part: hash of seed and block digest.
538fn block_sort_key(
539    seed: &[u8; DIGEST_LENGTH],
540    block_ref: &BlockRef,
541) -> (Round, [u8; DIGEST_LENGTH]) {
542    let mut hasher = DefaultHashFunction::new();
543    hasher.update(seed);
544    hasher.update(block_ref.digest);
545    (block_ref.round, hasher.finalize().into())
546}
547
548fn sort_committed_blocks(blocks: &mut [VerifiedBlock], seed: &[u8; DIGEST_LENGTH]) {
549    blocks.sort_by_cached_key(|b| block_sort_key(seed, &b.reference()));
550}

552/// Takes the union of each committed leader's parent-round ancestors,
553/// and returns the median timestamp of parent blocks weighted by stake.
554/// Monotonicity is enforced against `last_commit_timestamp_ms`.
555fn calculate_commit_timestamp(
556    context: &Context,
557    dag_state: &DagState,
558    committed_leaders: &[VerifiedBlock],
559) -> BlockTimestampMs {
560    let leader_round = committed_leaders[0].round();
561    debug_assert!(committed_leaders.iter().all(|b| b.round() == leader_round));
562
563    // When there are enough leader stake, use leaders to compute the commit timestamp.
564    let total_leader_stake = committed_leaders
565        .iter()
566        .map(|b| context.committee.stake(b.author()))
567        .sum::<Stake>();
568    if total_leader_stake >= context.committee.certification_threshold() {
569        let ts = crate::linearizer::median_timestamp_by_stake(context, committed_leaders)
570            .unwrap_or_else(|e| panic!("Cannot compute commit timestamp: {e}"));
571        return ts.max(dag_state.last_commit_timestamp_ms());
572    }
573
574    let mut parent_refs: BTreeMap<AuthorityIndex, BlockRef> = BTreeMap::new();
575    for leader in committed_leaders {
576        for ancestor in leader.ancestors() {
577            if ancestor.round + 1 == leader_round {
578                parent_refs.insert(ancestor.author, *ancestor);
579            }
580        }
581    }
582    let parent_refs: Vec<BlockRef> = parent_refs.values().cloned().collect();
583    let block_opts = dag_state.get_blocks(&parent_refs);
584    let blocks = block_opts
585        .iter()
586        .map(|b| b.as_ref().expect("Parent block must be in dag state"));
587    let ts = crate::linearizer::median_timestamp_by_stake(context, blocks)
588        .unwrap_or_else(|e| panic!("Cannot compute commit timestamp: {e}"));
589    ts.max(dag_state.last_commit_timestamp_ms())
590}
