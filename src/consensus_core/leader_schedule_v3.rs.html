23const MAX_PENDING_COMMITS: usize = 3;

25/// Incremental, sliding-window leader scorer over recent commits.
26///
27/// On each new commit C, score contributions are computed and added to the
28/// running totals. When the running window is full, the oldest commit is evicted
29/// and its contributions are subtracted from the running totals.
30///
31/// Scoring rule for commit C:
32///
33/// When `C >= 4`, C-1, C-2 and C-3 exist as pending commits.
34/// Let `r` be the leader round of commit C-3.
35///
36/// For each authority A: consider A's block at round `r+1` (voting round) across commits
37/// from commit C-2 to commit with leader round > `r+1`.
38/// If there are multiple A blocks in this slot, `score[A] = 0`.
39///
40/// Collect the set of distinct authorities P, such that A's voting block votes to
41/// (has an ancestor link to) P's round `r` block (leader block) in commit C-3.
42/// Then `voted_for_stake[A] = sum_{P in distinct set} stake(P)`.
43///
44/// Collect the set of distinct authorities Q, such that a Q's round `r+2`
45/// block certifies (has an ancestor link to) A's voting block in round `r+1`,
46/// across commits from C-2 to commit with leader round > `r+2`.
47/// Then `certified_by_stake[A] = sum_{Q in distinct set} stake(Q)`.
48///
49/// The overall score for A is `score[A] = voted_for_stake[A] * certified_by_stake[A]`.
50pub(crate) struct LeaderScheduleV3 {
51    context: Arc<Context>,
52    // Total scores per authority in the running window, indexed by AuthorityIndex.
53    // Each scored commit contributes `voted_for_stake[A] * certified_by_stake[A]`
54    // (stake² units) per authority A, summed across the running window.
55    total_scores_per_authority: Vec<u64>,
56    // Total number of leader-round blocks across the running window.
57    // Used for metrics.
58    total_num_leaders: usize,
59    // Running sum of `Σ stake(leader_authors)` across the scored commits in
60    // the window. The normalized-score metric multiplies this by
61    // `committee.total_stake()` at report time to match the stake² units of
62    // `total_scores_per_authority`.
63    total_leader_stakes: u64,
64
65    // Holds at most the last three committed subdags, for computing leader scores.
66    // Score entries are produced once 3 commits have accumulated; on each
67    // subsequent `add_commit`, the oldest is evicted and a new commit appended.
68    pending_commits: VecDeque<CommittedSubDag>,
69    // One entry per scored commit in the running window.
70    // Used to subtract a commit's contributions from `total_scores_per_authority`
71    // on eviction. Bounded by `leader_schedule_window_size`.
72    scores_entries: VecDeque<ScoresEntry>,
73    // Leader schedule for the current commit interval. The commit index and
74    // minimum round move every commit; leader selection changes only at
75    // configured interval boundaries.
76    current_schedule: NextCommitLeaderSchedule,
77}

80    /// Constructs a `LeaderScheduleV3`.
81    /// Replays persisted commits from storage if available, to ensure
82    /// the running window and current scores are recovered exactly.
83    pub(crate) fn from_store(
84        context: Arc<Context>,
85        dag_state: Arc<RwLock<DagState>>,
86    ) -> LeaderScheduleV3 {
87        let committee_size = context.committee.size();
88        let mut leader_schedule = LeaderScheduleV3::new(context.clone(), vec![0; committee_size]);
89
90        let last_commit_index = dag_state.read().last_commit_index();
91        if last_commit_index == 0 {
92            // No commits to replay, return the new schedule with empty state.
93            return leader_schedule;
94        }
95        let replay_start = LeaderScheduleV3::replay_start(&context, last_commit_index);
96
97        let store = dag_state.read().store();
98        let commits = store
99            .scan_commits((replay_start..=last_commit_index).into())
100            .expect("Failed to scan commits from storage");
101        for commit in commits {
102            let subdag = load_committed_subdag_from_store(store.as_ref(), commit);
103            leader_schedule.add_commit(subdag);
104        }
105        leader_schedule
106    }

145    /// Replays commits from the start commit index to the `last_commit_index`,
146    /// so afterwards the leader schedule matches what live processing would have
147    /// built.
148    ///
149    /// This means the most recent schedule must be correctly recovered, by processing
150    /// enough (window_size + MAX_PENDING_COMMITS) commit before computing the current schedule.
151    /// Each leader schedule applies to commits within leader_schedule_update_interval.
152    /// A new schedule is computed when next commit index % update interval == 1.
153    fn replay_start(context: &Context, last_commit_index: CommitIndex) -> CommitIndex {
154        let window_size = context.protocol_config.leader_schedule_window_size();
155        let update_interval = context.protocol_config.leader_schedule_update_interval();
156        // The next commit index which requires a new leader schedule to be computed.
157        let last_schedule_update_index =
158            (last_commit_index / update_interval) * update_interval + 1;
159        // Replay must at least window_size + MAX_PENDING_COMMITS commits
160        // when computing the schedule for last update, or replay from commit 1.
161        last_schedule_update_index
162            .saturating_sub(window_size + MAX_PENDING_COMMITS as u32)
163            .max(1)
164    }

172    pub(crate) fn compute_next_commit_leader_schedule(&self) -> NextCommitLeaderSchedule {

174        NextCommitLeaderSchedule {

176            min_next_leader_round: self.min_next_leader_round(),
177            allowed_leaders,

181    /// Returns the set of authorities allowed to be leaders, based on current
182    /// running scores and threshold in config.
183    ///
184    /// Authorities are ranked by score descending. Lowest-score authorities are
185    /// excluded while their cumulative stake remains within `stake_threshold`%
186    /// of total stake.
187    pub(crate) fn select_allowed_leaders_with_fixed_config(&self) -> Vec<AuthorityIndex> {
188        // Create a vector of authorities and their scores.
189        let mut by_score: Vec<(AuthorityIndex, u64)> = self
190            .total_scores_per_authority
191            .iter()
192            .enumerate()
193            .map(|(i, score)| {
194                let authority_index = self
195                    .context
196                    .committee
197                    .to_authority_index(i)
198                    .expect("Should be a valid AuthorityIndex");
199                (authority_index, *score)
200            })
201            .collect();
202
203        // Shuffle deterministically.
204        let seed_bytes = self
205            .pending_commits
206            .back()
207            .map(|c| c.commit_ref.digest.into_inner())
208            .unwrap_or_else(|| {
209                // No pending commits yet — derive a per-epoch seed so the
210                // initialization shuffle differs across epochs instead of
211                // always defaulting to all-zeros.
212                let mut seed = [0u8; 32];
213                seed[..8].copy_from_slice(&self.context.epoch_start_timestamp_ms.to_le_bytes());
214                seed[8..16].copy_from_slice(&self.context.committee.epoch().to_le_bytes());
215                seed
216            });
217        let mut rng = StdRng::from_seed(seed_bytes);
218        by_score.shuffle(&mut rng);
219
220        // Use stable sort to produce deterministic order of authorities with the same score.
221        by_score.sort_by(|a, b| b.1.cmp(&a.1));
222
223        let mut accumulated_bad_stake = 0u64;
224        let cutoff = (self.context.protocol_config.bad_nodes_stake_threshold()
225            * self.context.committee.total_stake())
226            / 100;
227        while let Some((idx, _)) = by_score.last() {
228            let stake = self.context.committee.stake(*idx);
229            if accumulated_bad_stake + stake > cutoff {
230                break;
231            }
232            accumulated_bad_stake += stake;
233            by_score.pop();
234        }
235
236        by_score.into_iter().map(|(idx, _)| idx).collect()
237    }

239    /// On each `add_commit(C)`, if there are already 3 commits buffered in pending_commits,
240    /// a score entry is produced and added to the back of scores_entries.
241    /// C is always appended to `pending_commits`. The oldest entry is
242    /// evicted from `pending_commits` if full (more than 3 entries).
243    ///
244    /// Architectural contract: consecutive commits must have strictly
245    /// increasing leader rounds. Under the invariant, 3 commits are sufficient
246    /// to satisfy both the r+1 voting-block scan and the r+2 certifying-block
247    /// scan (some commit in [C-2, C-1, c] satisfies each `>` bound).
248    pub(crate) fn add_commit(&mut self, c: CommittedSubDag) {
249        // Ensure commits are added in order. On the very first call there is
250        // no prior commit to check against; once we have one, every subsequent
251        // commit must be consecutive.
252        if let Some(last) = self.pending_commits.back() {
253            assert_eq!(c.commit_ref.index, last.commit_ref.index + 1);
254        }
255
256        // Need C-3, C-2, C-1 in pending before producing a score entry.
257        // Until then, this is warmup (new instance, new epoch, or recovery).
258        if self.pending_commits.len() < MAX_PENDING_COMMITS {
259            self.pending_commits.push_back(c);
260            self.refresh_current_schedule();
261            self.report_metrics();
262            return;
263        }
264
265        let c_minus_3 = &self.pending_commits[0];
266        let c_minus_2 = &self.pending_commits[1];
267        let c_minus_1 = &self.pending_commits[2];
268        let leader_round = c_minus_3.leader.round;
269        let vote_round = leader_round + 1;
270        let certify_round = leader_round + 2;
271
272        // Scoring logic assumes strictly increasing leader rounds across consecutive commits.
273        assert!(
274            c_minus_3.leader.round < c_minus_2.leader.round
275                && c_minus_2.leader.round < c_minus_1.leader.round
276                && c_minus_1.leader.round < c.leader.round,
277            "consecutive commits must have strictly increasing leader rounds; \
278             got {}, {}, {}, {}",
279            c_minus_3.leader.round,
280            c_minus_2.leader.round,
281            c_minus_1.leader.round,
282            c.leader.round,
283        );
284
285        // Every block in leader round is a leader block. A commit may
286        // contain one or multiple leader blocks.
287        let leader_refs: BTreeSet<BlockRef> = c_minus_3
288            .blocks
289            .iter()
290            .filter(|b| b.round() == leader_round)
291            .map(|b| b.reference())
292            .collect();
293
294        // A failure here points at an malformed CommittedSubDag.
295        assert!(!leader_refs.is_empty());
296        assert!(
297            leader_refs.contains(&c_minus_3.leader),
298            "C-3's leader {} missing from its own leader blocks {:?}",
299            c_minus_3.leader,
300            leader_refs,
301        );
302
303        // Walk [C-2, C-1, C] up to and including the first commit whose
304        // leader round exceeds each bound. The cut off is chosen to exclude
305        // blocks that are produced late relative to the current highest round.
306        let voting_commits =
307            Self::scan_commits_until_leader_round_above(&self.pending_commits, &c, vote_round);
308        let certifying_commits =
309            Self::scan_commits_until_leader_round_above(&self.pending_commits, &c, certify_round);
310
311        // Group round r+1 voting blocks by author.
312        let mut voting_blocks_by_author: BTreeMap<AuthorityIndex, Vec<&VerifiedBlock>> =
313            BTreeMap::new();
314        for cmt in &voting_commits {
315            for b in &cmt.blocks {
316                if b.round() == vote_round {
317                    voting_blocks_by_author
318                        .entry(b.author())
319                        .or_default()
320                        .push(b);
321                }
322            }
323        }
324        let mut certifying_blocks: Vec<&VerifiedBlock> = Vec::new();
325        for cmt in &certifying_commits {
326            for b in &cmt.blocks {
327                if b.round() == certify_round {
328                    certifying_blocks.push(b);
329                }
330            }
331        }
332
333        let mut score_contributions = vec![0u64; self.context.committee.size()];
334        for (a, voting_blocks) in &voting_blocks_by_author {
335            // Equivocation: more than one r+1 block from A → score[A] = 0.
336            if voting_blocks.len() != 1 {
337                tracing::debug!(
338                    "Equivocation detected on authority {}: {:?}",
339                    a,
340                    voting_blocks,
341                );
342                continue;
343            }
344            let voting_block = voting_blocks[0];
345
346            // voted_for_stake[A]: distinct authorities whose leader block
347            // in C-3 is voted by A's voting block.
348            let mut leader_authorities = BTreeSet::<AuthorityIndex>::new();
349            for anc in voting_block.ancestors() {
350                if leader_refs.contains(anc) {
351                    leader_authorities.insert(anc.author);
352                }
353            }
354            let voted_for_stake: u64 = leader_authorities
355                .into_iter()
356                .map(|i| self.context.committee.stake(i))
357                .sum();
358            // Product would be 0 anyway; skip the certifying-block scan.
359            if voted_for_stake == 0 {
360                continue;
361            }
362
363            // certified_by_stake[A]: distinct authorities whose round r+2 block
364            // certifies A's voting block (has it as an ancestor).
365            let voting_block_ref = voting_block.reference();
366            let mut certifying_authorities = BTreeSet::<AuthorityIndex>::new();
367            for certifying_block in &certifying_blocks {
368                if certifying_block.ancestors().contains(&voting_block_ref) {
369                    certifying_authorities.insert(certifying_block.author());
370                }
371            }
372            let certified_by_stake: u64 = certifying_authorities
373                .into_iter()
374                .map(|i| self.context.committee.stake(i))
375                .sum();
376
377            score_contributions[a.value()] =
378                voted_for_stake.checked_mul(certified_by_stake).unwrap();
379        }
380
381        let num_leaders = leader_refs.len();
382        let leader_stakes: u64 = leader_refs
383            .iter()
384            .map(|block_ref| self.context.committee.stake(block_ref.author))
385            .sum();
386
387        while self.scores_entries.len() >= self.window_size() {
388            let evicted = self.scores_entries.pop_front().expect("non empty");
389            tracing::trace!(
390                "LeaderScheduleV3 evicting scored commit {} from running window",
391                evicted.commit_ref
392            );
393            for (i, contribution) in evicted.score_contributions.iter().enumerate() {
394                self.total_scores_per_authority[i] = self.total_scores_per_authority[i]
395                    .checked_sub(*contribution)
396                    .unwrap();
397            }
398            self.total_num_leaders = self
399                .total_num_leaders
400                .checked_sub(evicted.num_leaders)
401                .unwrap();
402            self.total_leader_stakes = self
403                .total_leader_stakes
404                .checked_sub(evicted.leader_stakes)
405                .unwrap();
406        }
407
408        for (i, delta) in score_contributions.iter().enumerate() {
409            self.total_scores_per_authority[i] = self.total_scores_per_authority[i]
410                .checked_add(*delta)
411                .unwrap();
412        }
413        self.total_num_leaders = self.total_num_leaders.checked_add(num_leaders).unwrap();
414        self.total_leader_stakes = self.total_leader_stakes.checked_add(leader_stakes).unwrap();
415        self.scores_entries.push_back(ScoresEntry {
416            commit_ref: c_minus_3.commit_ref,
417            score_contributions,
418            num_leaders,
419            leader_stakes,
420        });
421
422        // Rotate pending commits: drop C-3, keep C-2 and C-1, append c.
423        self.pending_commits.pop_front();
424        self.pending_commits.push_back(c);
425
426        self.refresh_current_schedule();
427        self.report_metrics();
428    }

430    fn refresh_current_schedule(&mut self) {
431        let interval = self
432            .context
433            .protocol_config
434            .leader_schedule_update_interval() as CommitIndex;
435        let on_boundary = self
436            .next_commit_index()
437            .saturating_sub(1)
438            .is_multiple_of(interval);
439        if on_boundary {
440            // Boundary: full recompute, including allowed_leaders.
441            self.current_schedule = self.compute_next_commit_leader_schedule();
442        } else {
443            // Off-boundary: only the per-commit fields move; skip the
444            // expensive allowed-leaders selection.
445            self.current_schedule.next_commit_index = self.next_commit_index();
446            self.current_schedule.min_next_leader_round = self.min_next_leader_round();
447        }
448    }

450    /// Walks commits [C-2, C-1, C] in order.
451    /// Returning every commit up to and **including** the first one whose
452    /// leader round is strictly greater than `upper`.
453    fn scan_commits_until_leader_round_above<'a>(
454        pending: &'a VecDeque<CommittedSubDag>,
455        c: &'a CommittedSubDag,
456        upper: Round,
457    ) -> Vec<&'a CommittedSubDag> {
458        let mut out = Vec::new();
459        for cmt in pending.iter().skip(1).chain(std::iter::once(c)) {
460            out.push(cmt);
461            if cmt.leader.round > upper {
462                break;
463            }
464        }
465        assert!(
466            out.last()
467                .map(|cmt| cmt.leader.round > upper)
468                .unwrap_or(false),
469            "scan must end on a commit with leader.round > {upper}; \
470             strictly-increasing-leader-rounds invariant violated upstream",
471        );
472        out
473    }

475    /// Snapshot of the running per-authority scores.
476    pub(crate) fn current_reputation_scores(&self) -> ReputationScores {
477        let interval_size = self
478            .context
479            .protocol_config
480            .leader_schedule_update_interval();
481        let interval_num = (self.current_schedule.next_commit_index - 1) / interval_size;
482        let commit_range = CommitRange::new(
483            (interval_num * interval_size + 1)..=(interval_num + 1) * interval_size,
484        );
485        ReputationScores::new(commit_range, self.total_scores_per_authority.clone())
486    }

488    fn report_metrics(&self) {
489        let metrics = &self.context.metrics.node_metrics;
490        for (i, score) in self.total_scores_per_authority.iter().enumerate() {
491            let authority_index = self
492                .context
493                .committee
494                .to_authority_index(i)
495                .expect("Should be a valid AuthorityIndex");
496            let hostname = &self.context.committee.authority(authority_index).hostname;
497            if hostname.is_empty() {
498                continue;
499            }
500            metrics
501                .leader_schedule_total_scores
502                .with_label_values(&[hostname])
503                .set(*score as i64);
504            // Multiply at report time so the field stays raw `Σ stake(leaders)`
505            // and the metric formula divides stake² (numerator) by stake²
506            // (denominator) — i.e. a fraction with no stake dimension.
507            let denominator = self
508                .total_leader_stakes
509                .checked_mul(self.context.committee.total_stake())
510                .unwrap();
511            let normalized = if denominator == 0 {
512                0.0
513            } else {
514                *score as f64 / denominator as f64
515            };
516            metrics
517                .leader_schedule_normalized_scores
518                .with_label_values(&[hostname])
519                .set(normalized);
520        }
521        if let Some(last) = self.scores_entries.back() {
522            metrics
523                .leader_schedule_last_num_leaders
524                .set(last.num_leaders as i64);
525            metrics
526                .leader_schedule_average_num_leaders
527                .set(self.average_num_leaders());
528        }
529    }

586struct ScoresEntry {
587    // CommitRef of the oldest pending commit at the time the entry was produced.
588    commit_ref: CommitRef,
589    // Per-authority incremental score indexed by AuthorityIndex.
590    // Retained so eviction can subtract from `total_scores_per_authority`.
591    score_contributions: Vec<u64>,
592    // Number of leader blocks in the commit.
593    // Retained so eviction can subtract from `total_num_leaders`.
594    num_leaders: usize,
595    // `Σ stake(leader_authors)` from this commit's leaders.
596    // Retained so eviction can subtract from `total_leader_stakes`.
597    leader_stakes: u64,
598}

654        assert!(last_index > 0);

657        for round in 2..=last_index {
658            let prev = &blocks_by_round[(round - 2) as usize];
659            let ancestors: Vec<_> = prev.iter().map(|b| b.reference()).collect();
660            blocks_by_round.push(vec![
661                make_block(round, 0, ancestors.clone()),

664        }
665

667            .map(|i| {
668                let blocks = blocks_by_round[(i - 1) as usize].clone();
669                let leader = blocks[0].reference();
670                make_commit(i, leader, blocks)
671            })
672            .collect()
673    }

711    #[tokio::test]
712    async fn test_add_commit_basics() {
713        // 4 commits with strictly increasing leader rounds (1, 2, 3, 4).
714        // Scoring logic needs C-3, C-2, C-1 in pending before producing the
715        // first score entry, so the first three commits are warmup.
716        // C1 (the C-3-equivalent when scoring on C4) carries multiple round-1
717        // leader blocks, exercising "leader_refs = all round-r blocks in C-3".
718        // Voting blocks live at round 2 in C2, certifying blocks at round 3 in
719        // C3, and C4 acts as the > r+2 sentinel.
720        let context = setup(4);
721        let mut schedule = LeaderScheduleV3::new(context.clone(), vec![0; 4]);
722
723        // C1: leader = L0_1 (auth 0); blocks include L1_1 too, so leader_refs
724        // for C-3 = {L0_1, L1_1}.
725        let l0_1 = make_block(1, 0, vec![]);
726        let l1_1 = make_block(1, 1, vec![]);
727        schedule.add_commit(make_commit(
728            1,
729            l0_1.reference(),
730            vec![l0_1.clone(), l1_1.clone()],
731        ));
732        assert_eq!(schedule.total_scores_per_authority(), &[0u64; 4]);
733        assert_eq!(schedule.scores_entries_len(), 0);
734        assert_eq!(schedule.pending_commits_len(), 1);
735
736        // C2: round-2 voting blocks for authorities 0 and 2.
737        //   v0 (auth 0) links L0_1 only.
738        //   v2 (auth 2) links L0_1 and L1_1.
739        let v0 = make_block(2, 0, vec![l0_1.reference()]);
740        let v2 = make_block(2, 2, vec![l0_1.reference(), l1_1.reference()]);
741        schedule.add_commit(make_commit(2, v0.reference(), vec![v0.clone(), v2.clone()]));
742        assert_eq!(schedule.total_scores_per_authority(), &[0u64; 4]);
743        assert_eq!(schedule.scores_entries_len(), 0);
744        assert_eq!(schedule.pending_commits_len(), 2);
745
746        // C3: round-3 certifying block q1 (auth 1) certifies v0 and v2.
747        let q1 = make_block(3, 1, vec![v0.reference(), v2.reference()]);
748        schedule.add_commit(make_commit(3, q1.reference(), vec![q1.clone()]));
749        assert_eq!(schedule.total_scores_per_authority(), &[0u64; 4]);
750        assert_eq!(schedule.scores_entries_len(), 0);
751        assert_eq!(schedule.pending_commits_len(), 3);
752
753        // C4: leader at round 4. Add a round-3 block q3 (auth 3) certifying v0
754        // only — exercises "C4 (the inclusive r+2 sentinel) is searched too".
755        let q3 = make_block(3, 3, vec![v0.reference()]);
756        let lead4 = make_block(4, 0, vec![q1.reference()]);
757        schedule.add_commit(make_commit(
758            4,
759            lead4.reference(),
760            vec![lead4.clone(), q3.clone()],
761        ));
762
763        // Expected:
764        //   A=0: voting={v0}, voted_for_stake = s0 (links L0_1).
765        //        certifiers: q1 has v0 ancestor → qs={1}; q3 has v0 ancestor → qs={1,3}.
766        //        certified_by_stake = s1 + s3.   score[0] = s0 * (s1 + s3).
767        //   A=2: voting={v2}, voted_for_stake = s0 + s1 (links L0_1, L1_1).
768        //        certifiers: q1 has v2 → qs={1}; q3 doesn't → qs stays {1}.
769        //        certified_by_stake = s1.        score[2] = (s0 + s1) * s1.
770        //   A=1, A=3: no round-2 voting block → 0.
771        let s0 = context.committee.stake(AuthorityIndex::new_for_test(0));
772        let s1 = context.committee.stake(AuthorityIndex::new_for_test(1));
773        let s3 = context.committee.stake(AuthorityIndex::new_for_test(3));
774        assert_eq!(
775            schedule.total_scores_per_authority(),
776            &[s0 * (s1 + s3), 0, (s0 + s1) * s1, 0]
777        );
778        assert_eq!(schedule.scores_entries_len(), 1);
779        assert_eq!(schedule.pending_commits_len(), 3);
780        // C-3 (commit 1) has 2 leader-round blocks (auths 0 and 1) → average
781        // num_leaders=2, single scored entry contributes (s0 + s1) to the
782        // running total leader-stake.
783        assert_eq!(schedule.average_num_leaders(), 2.0);
784        assert_eq!(schedule.total_leader_stakes(), s0 + s1);
785    }

787    #[tokio::test]
788    async fn test_eviction_subtracts_contributions() {
789        // Uniform chain: rounds 1..=8, two authorities (0 and 1) producing
790        // blocks at each round and linking to both of the previous round's
791        // blocks. With `window_size=3` and the new rule, the first score
792        // entry lands when commit 4 is added, the window fills at commit 6,
793        // and commit 7 evicts the earliest entry. Because every scored commit
794        // contributes an identical delta, post-eviction totals equal
795        // pre-eviction totals — proving eviction is subtracting correctly.
796        let context = setup_with_window_size(4, Some(3));
797        let mut schedule = LeaderScheduleV3::new(context.clone(), vec![0; 4]);
798
799        let mut blocks_by_round: Vec<Vec<VerifiedBlock>> = Vec::new();
800        blocks_by_round.push(vec![make_block(1, 0, vec![]), make_block(1, 1, vec![])]);
801        for round in 2..=7u32 {
802            let prev = &blocks_by_round[(round - 2) as usize];
803            let ancestors: Vec<_> = prev.iter().map(|b| b.reference()).collect();
804            blocks_by_round.push(vec![
805                make_block(round, 0, ancestors.clone()),
806                make_block(round, 1, ancestors),
807            ]);
808        }
809
810        let commit = |i: CommitIndex, brs: &[Vec<VerifiedBlock>]| -> CommittedSubDag {
811            let blocks = brs[(i - 1) as usize].clone();
812            let leader = blocks[0].reference();
813            make_commit(i, leader, blocks)
814        };
815
816        for i in 1..=6 {
817            schedule.add_commit(commit(i, &blocks_by_round));
818        }
819        let after_6 = schedule.total_scores_per_authority().to_vec();
820        // Window full: scored entries for C1, C2, C3.
821        assert_eq!(schedule.scores_entries_len(), 3);
822
823        // Per-commit contribution to authority A in this uniform chain:
824        //   voted_for_stake[A]    = s0 + s1 (A's r+1 voting block links both leaders),
825        //   certified_by_stake[A] = s0 + s1 (both r+2 blocks certify A's voting block).
826        // → per_commit = (s0 + s1)^2 for authorities 0 and 1; 0 for 2 and 3.
827        let s0 = context.committee.stake(AuthorityIndex::new_for_test(0));
828        let s1 = context.committee.stake(AuthorityIndex::new_for_test(1));
829        let per_commit = (s0 + s1) * (s0 + s1);
830        assert_eq!(after_6, vec![3 * per_commit, 3 * per_commit, 0, 0]);
831
832        schedule.add_commit(commit(7, &blocks_by_round));
833        assert_eq!(schedule.scores_entries_len(), 3);
834        // C1's entry was evicted and C4's entry added; identical deltas leave
835        // totals unchanged. Had eviction not subtracted, totals would be
836        // 4 * per_commit.
837        assert_eq!(schedule.total_scores_per_authority().to_vec(), after_6);
838
839        // Each scored C-3 has exactly 2 leader-round blocks from auths 0/1;
840        // average_num_leaders=2.0 and the running leader-stake total is
841        // 3 entries × (s0 + s1) per commit.
842        assert_eq!(schedule.average_num_leaders(), 2.0);
843        assert_eq!(schedule.total_leader_stakes(), 3 * (s0 + s1));
844    }
