28/// Minimum number of rounds there an anchor block can indirectly decide a leader block.
29pub(crate) const INDIRECT_COMMIT_DEPTH: Round = 2;
30
31/// Stateless decision logic for commit rule v3.
32pub(crate) struct LeaderSlotDecider {
33    context: Arc<Context>,
34    dag_state: Arc<RwLock<DagState>>,
35}
36
37impl LeaderSlotDecider {
38    pub(crate) fn new(context: Arc<Context>, dag_state: Arc<RwLock<DagState>>) -> Self {
39        Self { context, dag_state }
40    }
41
42    /// Evaluates if `slot` should be directly committed or skipped,
43    /// based on voting information from DagState.
44    ///
45    /// A block is directly committed if it has a quorum of commit votes,
46    /// and a block is directly skipped if it has a quorum of skip votes.
47    /// Misbehaviors are assumed to be under the fault tolerance threshold, but we also
48    /// try to detect if this assumption is broken, when it is convenient.
49    pub(crate) fn try_direct_decide(&self, slot: Slot) -> LeaderStatus {
50        let dag_state = self.dag_state.read();
51        let Some(voting_round_info) = dag_state.get_round_info(slot.round + 1) else {
52            // No blocks accepted in the next round yet, so there is no votes to the slot.
53            return LeaderStatus::Undecided(slot);
54        };
55        if !voting_round_info
56            .total_stake
57            .reached_threshold(&self.context.committee)
58        {
59            // Next round has insufficient stake to decide the slot.
60            return LeaderStatus::Undecided(slot);
61        }
62
63        let block_infos = dag_state.get_block_info_at_slot(slot);
64        let num_blocks = block_infos.len();
65        let mut num_skipped = 0;
66        for block_info in block_infos {
67            // Skip the leader block if it has a quorum of skip votes.
68            // Byzantine authorities can vote on both sides, but its votes count at most once per side
69            // against a decision block.
70            let mut reject_votes = StakeAggregator::<QuorumThreshold>::new();
71            // Add up votes skipping the block.
72            for block_ref in voting_round_info.blocks.difference(&block_info.children) {
73                reject_votes.add_unique(block_ref.author, &self.context.committee);
74            }
75            let mut block_rejected = false;
76            if reject_votes.reached_threshold(&self.context.committee) {
77                num_skipped += 1;
78                block_rejected = true;
79                // Continue checking if the block can be committed,
80                // to detect if there are too many misbehaving authorities.
81            }
82            // Commit the leader block if it has a quorum of commit votes.
83            if block_info
84                .children_stake
85                .reached_threshold(&self.context.committee)
86            {
87                assert!(
88                    !block_rejected,
89                    "Block {} cannot be both committed and skipped. Commit voters: {:?}, skip voters: {:?}",
90                    block_info.block.reference(),
91                    block_info.children_stake.authorities(),
92                    reject_votes.authorities(),
93                );
94                return LeaderStatus::Commit(block_info.block);
95            }
96        }
97
98        // When there is no block in the slot or all blocks in the slot are skipped,
99        // the whole slot is skipped.
100        // Even if new blocks arrive at the slot later, they are guaranteed to be
101        // skipped directly since a quorum already skipped them.
102        if num_skipped == num_blocks {
103            return LeaderStatus::Skip(slot);
104        }
105
106        // There are undecided blocks, so the slot is undecided.
107        LeaderStatus::Undecided(slot)
108    }
109
110    /// Evaluates every slot in `decision_slots` for indirect commit.
111    /// Returns one LeaderStatus per `decision_slots` entry, in the same order.
112    ///
113    /// Commit votes per decision block are counted from voting blocks reachable
114    /// via BFS from `anchor_block`. Then the commit votes on each decision block is
115    /// compared against `certification_threshold`.
116    ///
117    /// It is possible for a block to have both commit and skip certificates. But skip
118    /// votes and certificates are not tracked because they do not affect the decision
119    /// on the slot.
120    ///
121    /// When a slot has exactly one block with commit certificate, the slot is considered
122    /// committed. But if there are zero or multiple blocks with commit certificates,
123    /// the slot is considered skipped because there cannot have been a direct commit on
124    /// the slot.
125    ///
126    /// Each slot will be either committed or skipped, never undecided after applying the
127    /// indirect commit rule.
128    pub(crate) fn try_indirect_decide(
129        &self,
130        anchor_block: &VerifiedBlock,
131        decision_slots: &[Slot],
132    ) -> Vec<LeaderStatus> {
133        assert!(!decision_slots.is_empty());
134        let decision_round = decision_slots[0].round;
135        assert!(decision_slots.iter().all(|s| s.round == decision_round));
136        assert!(
137            decision_round + INDIRECT_COMMIT_DEPTH <= anchor_block.round(),
138            "Anchor block {} is too close to decision round {}",
139            anchor_block.reference(),
140            decision_round,
141        );
142
143        let dag_state = self.dag_state.read();
144
145        let decision_blocks: BTreeSet<BlockRef> = decision_slots
146            .iter()
147            .flat_map(|slot| {
148                dag_state
149                    .get_uncommitted_blocks_at_slot(*slot)
150                    .into_iter()
151                    .map(|block| block.reference())
152            })
153            .collect();
154
155        // BFS once from the anchor: collect votes from voting round blocks
156        // against decision round blocks.
157        let mut to_visit = VecDeque::new();
158        to_visit.push_back(anchor_block.clone());
159        let mut visited = BTreeSet::new();
160        visited.insert(anchor_block.reference());
161
162        // Count commit votes for decision blocks.
163        let mut votes = BTreeMap::<BlockRef, StakeAggregator<CertificationThreshold>>::new();
164        while let Some(block) = to_visit.pop_front() {
165            for ancestor in block.ancestors() {
166                // Set up next steps of BFS.
167                if ancestor.round > decision_round && visited.insert(*ancestor) {
168                    let ancestor_block = dag_state
169                        .get_block(ancestor)
170                        .unwrap_or_else(|| panic!("Block {} must exist", ancestor));
171                    to_visit.push_back(ancestor_block);
172                }
173                // Count only voting block to decision block votes.
174                if decision_blocks.contains(ancestor) && block.round() == decision_round + 1 {
175                    let commit_stake = votes.entry(*ancestor).or_default();
176                    commit_stake.add_unique(block.author(), &self.context.committee);
177                }
178            }
179        }
180
181        // Collect the blocks with commit certificates per slot.
182        let mut certified_slots = BTreeMap::<Slot, Vec<VerifiedBlock>>::new();
183        for (block_ref, commit_stake) in votes {
184            let slot = Slot {
185                round: block_ref.round,
186                authority: block_ref.author,
187            };
188            if commit_stake.reached_threshold(&self.context.committee) {
189                certified_slots.entry(slot).or_default().push(
190                    dag_state
191                        .get_block(&block_ref)
192                        .unwrap_or_else(|| panic!("Block {} must exist", block_ref)),
193                );
194            }
195        }
196
197        // Make the decision on each slot.
198        decision_slots
199            .iter()
200            .map(|slot| {
201                let mut certified_blocks = certified_slots.get(slot).cloned().unwrap_or_default();
202                // When there is no certified block at the slot, obviously the slot is skipped.
203                // When there are multiple certified blocks at the slot, there cannot have been a direct commit
204                // on the slot if there are <= f malicious stake. So it is also safe to skip the slot.
205                if certified_blocks.is_empty() || certified_blocks.len() > 1 {
206                    return LeaderStatus::Skip(*slot);
207                }
208                LeaderStatus::Commit(certified_blocks.pop().unwrap())
209            })
210            .collect()
211    }
212}
