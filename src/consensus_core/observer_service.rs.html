33/// Serves observer requests from observer or validator peers. It is the server-side
34/// counterpart to `ObserverNetworkClient`.
35pub(crate) struct ObserverService {
36    context: Arc<Context>,
37    core_dispatcher: Arc<dyn CoreThreadDispatcher>,
38    dag_state: Arc<RwLock<DagState>>,
39    rx_accepted_block_broadcast: broadcast::Receiver<VerifiedBlock>,
40    subscription_counter: Arc<SubscriptionCounter>,
41    block_verifier: Arc<dyn BlockVerifier>,
42    commit_vote_monitor: Arc<CommitVoteMonitor>,
43    transaction_vote_tracker: TransactionVoteTracker,
44    synchronizer: Arc<SynchronizerHandle>,
45    block_sync_service: Arc<BlockSyncService>,
46    randomness_signature_handler: Option<Arc<dyn RandomnessSignatureHandler>>,
47}
48
49impl ObserverService {
50    pub(crate) fn new(
51        context: Arc<Context>,
52        core_dispatcher: Arc<dyn CoreThreadDispatcher>,
53        dag_state: Arc<RwLock<DagState>>,
54        rx_accepted_block_broadcast: broadcast::Receiver<VerifiedBlock>,
55        block_verifier: Arc<dyn BlockVerifier>,
56        commit_vote_monitor: Arc<CommitVoteMonitor>,
57        transaction_vote_tracker: TransactionVoteTracker,
58        synchronizer: Arc<SynchronizerHandle>,
59        block_sync_service: Arc<BlockSyncService>,
60        randomness_signature_handler: Option<Arc<dyn RandomnessSignatureHandler>>,
61    ) -> Self {
62        let subscription_counter = Arc::new(SubscriptionCounter::new(context.clone()));
63        Self {
64            context,
65            core_dispatcher,
66            dag_state,
67            rx_accepted_block_broadcast,
68            subscription_counter,
69            block_verifier,
70            commit_vote_monitor,
71            transaction_vote_tracker,
72            synchronizer,
73            block_sync_service,
74            randomness_signature_handler,
75        }
76    }
77}

80impl ObserverNetworkService for ObserverService {
81    async fn handle_block(&self, peer: PeerId, block: Bytes) -> ConsensusResult<()> {
82        fail_point_async!("consensus-rpc-response");
83
84        // TODO: dedup block verifications, here and with fetched blocks.
85        let signed_block: SignedBlock =
86            bcs::from_bytes(&block).map_err(ConsensusError::MalformedBlock)?;
87
88        // Create owned strings for observer peer names to avoid borrowing issues
89        let observer_name;
90        let peer_name = match &peer {
91            PeerId::Validator(authority) => self
92                .context
93                .committee
94                .authority(*authority)
95                .hostname
96                .as_str(),
97            PeerId::Observer(node_id) => {
98                observer_name = format!("{:?}", node_id);
99                observer_name.as_str()
100            }
101        };
102
103        // Reject blocks failing parsing and validations.
104        // Of Observer nodes we don't care about the transaction votes.
105        let (verified_block, _reject_txn_votes) = self
106            .block_verifier
107            .verify_and_vote(signed_block, block)
108            .tap_err(|e| {
109                self.context
110                    .metrics
111                    .node_metrics
112                    .invalid_blocks
113                    .with_label_values(&[peer_name, "handle_send_block", e.name()])
114                    .inc();
115                tracing::info!("Invalid block from {}: {}", peer.clone(), e);
116            })?;
117
118        let block_author_hostname = &self
119            .context
120            .committee
121            .authority(verified_block.author())
122            .hostname;
123        let block_ref = verified_block.reference();
124        tracing::debug!("Received block {} via send block.", block_ref);
125
126        self.context
127            .metrics
128            .node_metrics
129            .verified_blocks
130            .with_label_values(&[block_author_hostname])
131            .inc();
132
133        let now = self.context.clock.timestamp_utc_ms();
134        let forward_time_drift =
135            Duration::from_millis(verified_block.timestamp_ms().saturating_sub(now));
136
137        self.context
138            .metrics
139            .node_metrics
140            .block_timestamp_drift_ms
141            .with_label_values(&[block_author_hostname.as_str(), "handle_send_block"])
142            .inc_by(forward_time_drift.as_millis() as u64);
143
144        // Observe the block for the commit votes. When local commit is lagging too much,
145        // commit sync loop will trigger fetching.
146        self.commit_vote_monitor.observe_block(&verified_block);
147
148        // Reject blocks when local commit index is lagging too far from quorum commit index,
149        // to avoid the memory overhead from suspended blocks.
150        //
151        // IMPORTANT: this must be done after observing votes from the block, otherwise
152        // observed quorum commit will no longer progress.
153        //
154        // Since the main issue with too many suspended blocks is memory usage not CPU,
155        // it is ok to reject after block verifications instead of before.
156        let last_commit_index = self.dag_state.read().last_commit_index();
157        let quorum_commit_index = self.commit_vote_monitor.quorum_commit_index();
158        if is_commit_lagging(
159            self.context.as_ref(),
160            last_commit_index,
161            quorum_commit_index,
162        ) {
163            self.context
164                .metrics
165                .node_metrics
166                .rejected_blocks
167                .with_label_values(&["commit_lagging"])
168                .inc();
169            tracing::debug!(
170                "Block {:?} is rejected because last commit index is lagging quorum commit index too much ({} < {})",
171                block_ref,
172                last_commit_index,
173                quorum_commit_index,
174            );
175            return Err(ConsensusError::BlockRejected {
176                block_ref,
177                reason: format!(
178                    "Last commit index is lagging quorum commit index too much ({} < {})",
179                    last_commit_index, quorum_commit_index,
180                ),
181            });
182        }
183
184        // Add the block to the transaction vote tracker. No "own" votes are recorded for observer nodes.
185        if self.context.protocol_config.transaction_voting_enabled() {
186            self.transaction_vote_tracker
187                .add_voted_blocks(vec![(verified_block.clone(), vec![])]);
188        }
189
190        // Send the block to Core to try accepting it into the DAG.
191        let missing_ancestors = self
192            .core_dispatcher
193            .add_blocks(vec![verified_block.clone()])
194            .await
195            .map_err(|_| ConsensusError::Shutdown)?;
196
197        // Schedule fetching missing ancestors from this peer in the background.
198        if !missing_ancestors.is_empty() {
199            self.context
200                .metrics
201                .node_metrics
202                .handler_received_block_missing_ancestors
203                .with_label_values(&[block_author_hostname])
204                .inc_by(missing_ancestors.len() as u64);
205
206            let synchronizer = self.synchronizer.clone();
207            mysten_metrics::spawn_monitored_task!(async move {
208                // This does not wait for the fetch request to complete.
209                // It only waits for synchronizer to queue the request to a peer.
210                // When this fails, it usually means the queue is full.
211                // The fetch will retry from other peers via live and periodic syncs.
212                if let Err(err) = synchronizer.fetch_blocks(missing_ancestors, peer).await {
213                    tracing::debug!("Failed to fetch missing ancestors via synchronizer: {err}");
214                }
215            });
216        }
217
218        Ok(())
219    }

221    async fn handle_stream_blocks(
222        &self,
223        peer: NodeId,
224        highest_round_per_authority: Vec<u64>,
225    ) -> ConsensusResult<ObserverBlockStream> {
226        if highest_round_per_authority.len() != self.context.committee.size() {
227            return Err(ConsensusError::InvalidSizeOfHighestAcceptedRounds(
228                highest_round_per_authority.len(),
229                self.context.committee.size(),
230            ));
231        }
232
233        // Collect all accepted blocks from DagState that the observer hasn't yet seen,
234        // sorted by round for consistent ordering.
235        let past_blocks = {
236            let dag_state = self.dag_state.read();
237            let mut past_blocks = Vec::new();
238
239            for (authority, _) in self.context.committee.authorities() {
240                let from_round = highest_round_per_authority[authority.value()] as u32 + 1;
241                past_blocks.extend(dag_state.get_cached_blocks(authority, from_round));
242            }
243
244            past_blocks.sort_unstable_by_key(|b| b.round());
245            past_blocks
246        };
247
248        let past_stream =
249            stream::iter(
250                past_blocks
251                    .into_iter()
252                    .map(move |block| ObserverStreamItem {
253                        blocks: vec![block.serialized().clone()],
254                        auxiliary_data: Default::default(),
255                    }),
256            );
257
258        const MAX_BLOCKS_PER_POLL: usize = 20;
259        let live_block_stream = BroadcastStream::<VerifiedBlock>::new(
260            PeerId::Observer(Box::new(peer)),
261            self.rx_accepted_block_broadcast.resubscribe(),
262            MAX_BLOCKS_PER_POLL,
263            self.subscription_counter.clone(),
264        )
265        .map(|blocks| ObserverStreamItem {
266            blocks: blocks
267                .into_iter()
268                .map(|block| block.serialized().clone())
269                .collect(),
270            auxiliary_data: Default::default(),
271        });
272
273        let block_stream = past_stream.chain(live_block_stream);
274
275        // Merge randomness signature broadcast into the block stream when a handler is available.
276        if let Some(handler) = &self.randomness_signature_handler {
277            const MAX_SIGNATURES_PER_POLL: usize = 20;
278            let sig_stream = BroadcastStream::new_untracked(
279                handler.subscribe_randomness_signatures(),
280                MAX_SIGNATURES_PER_POLL,
281            )
282            .map(|sigs| ObserverStreamItem {
283                blocks: vec![],
284                auxiliary_data: AuxiliaryData {
285                    randomness_signatures: sigs,
286                },
287            });
288            Ok(Box::pin(futures::stream::select(block_stream, sig_stream)))
289        } else {
290            Ok(Box::pin(block_stream))
291        }
292    }

294    async fn handle_fetch_blocks(
295        &self,
296        _peer: NodeId,
297        block_refs: Vec<BlockRef>,
298        fetch_after_rounds: Vec<Round>,
299        fetch_missing_ancestors: bool,
300    ) -> ConsensusResult<Vec<Bytes>> {
301        fail_point_async!("consensus-rpc-response");
302
303        // Delegate to BlockSyncService
304        self.block_sync_service
305            .fetch_blocks(block_refs, fetch_after_rounds, fetch_missing_ancestors)
306            .await
307    }
308
309    async fn handle_fetch_commits(
310        &self,
311        _peer: NodeId,
312        commit_range: CommitRange,
313    ) -> ConsensusResult<(Vec<TrustedCommit>, Vec<VerifiedBlock>)> {
314        fail_point_async!("consensus-rpc-response");
315
316        // Delegate to BlockSyncService
317        self.block_sync_service.fetch_commits(commit_range).await
318    }

344    #[tokio::test]
345    async fn test_observer_stream_receives_broadcast_blocks() {
346        telemetry_subscribers::init_for_testing();
347        let (context, keys) = Context::new_for_test(4);
348        let context = Arc::new(context);
349
350        let store = Arc::new(MemStore::new());
351        let dag_state = Arc::new(RwLock::new(DagState::new(context.clone(), store.clone())));
352
353        let (tx_accepted_block, rx_accepted_block) = broadcast::channel::<VerifiedBlock>(100);
354
355        // Create mock dependencies
356        let core_dispatcher = Arc::new(MockCoreThreadDispatcher::default());
357        let block_verifier = Arc::new(NoopBlockVerifier);
358        let commit_vote_monitor = Arc::new(CommitVoteMonitor::new(context.clone()));
359        let transaction_vote_tracker =
360            TransactionVoteTracker::new(context.clone(), block_verifier.clone(), dag_state.clone());
361
362        let block_sync_service = Arc::new(BlockSyncService::new(
363            context.clone(),
364            dag_state.clone(),
365            store.clone(),
366        ));
367        let observer_service = ObserverService::new(
368            context.clone(),
369            core_dispatcher,
370            dag_state,
371            rx_accepted_block,
372            block_verifier,
373            commit_vote_monitor,
374            transaction_vote_tracker,
375            create_mock_synchronizer(),
376            block_sync_service,
377            None,
378        );
379
380        // Observer starts with no blocks seen
381        let highest_round_per_authority = vec![0u64; context.committee.size()];
382        let peer = keys[0].0.public().clone();
383
384        let mut stream = observer_service
385            .handle_stream_blocks(peer, highest_round_per_authority)
386            .await
387            .unwrap();
388
389        // Broadcast three blocks
390        let block1 = VerifiedBlock::new_for_test(TestBlock::new(5, 0).build());
391        let block2 = VerifiedBlock::new_for_test(TestBlock::new(10, 1).build());
392        let block3 = VerifiedBlock::new_for_test(TestBlock::new(15, 2).build());
393
394        tx_accepted_block.send(block1.clone()).unwrap();
395        tx_accepted_block.send(block2.clone()).unwrap();
396        tx_accepted_block.send(block3.clone()).unwrap();
397
398        // Verify observer receives all three blocks in order.
399        // Collect all blocks from the batched stream.
400        let mut received_blocks = Vec::new();
401        while received_blocks.len() < 3 {
402            let item = stream.next().await.unwrap();
403            for block_bytes in item.blocks {
404                let signed: SignedBlock = bcs::from_bytes(&block_bytes).unwrap();
405                received_blocks.push(VerifiedBlock::new_verified(signed, block_bytes));
406            }
407        }
408        assert_eq!(received_blocks[0].round(), 5);
409        assert_eq!(received_blocks[0].author().value(), 0);
410        assert_eq!(received_blocks[1].round(), 10);
411        assert_eq!(received_blocks[1].author().value(), 1);
412        assert_eq!(received_blocks[2].round(), 15);
413        assert_eq!(received_blocks[2].author().value(), 2);
414    }
