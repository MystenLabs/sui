30/// ObserverSubscriber manages block stream subscriptions to peers (validators or other observers),
31/// taking care of retrying when subscription streams break. Blocks returned from peers are sent
32/// to the observer service for processing. The `ObserverSubscriber` can only subscribe to one peer at a time.
33pub(crate) struct ObserverSubscriber<C: ObserverNetworkClient, S: ObserverNetworkService> {
34    context: Arc<Context>,
35    network_client: Arc<C>,
36    observer_service: Arc<S>,
37    commit_vote_monitor: Arc<CommitVoteMonitor>,
38    dag_state: Arc<parking_lot::RwLock<DagState>>,
39    subscription: Arc<Mutex<Option<JoinHandle<()>>>>,
40    randomness_signature_handler: Option<Arc<dyn RandomnessSignatureHandler>>,
41}
42
43impl<C: ObserverNetworkClient, S: ObserverNetworkService> ObserverSubscriber<C, S> {
44    pub(crate) fn new(
45        context: Arc<Context>,
46        network_client: Arc<C>,
47        observer_service: Arc<S>,
48        commit_vote_monitor: Arc<CommitVoteMonitor>,
49        dag_state: Arc<parking_lot::RwLock<DagState>>,
50        randomness_signature_handler: Option<Arc<dyn RandomnessSignatureHandler>>,
51    ) -> Self {
52        Self {
53            context,
54            network_client,
55            observer_service,
56            commit_vote_monitor,
57            dag_state,
58            subscription: Arc::new(Mutex::new(None)),
59            randomness_signature_handler,
60        }
61    }
62
63    /// Subscribe to a peer (validator or observer) to receive block streams. The `ObserverSubscriber` can only subscribe to one peer at a time.
64    /// The method will abort the existing subscription (if any) and start a new one.
65    pub(crate) fn subscribe(&self, peer: PeerId) {
66        let context = self.context.clone();
67        let network_client = self.network_client.clone();
68        let observer_service = self.observer_service.clone();
69        let commit_vote_monitor = self.commit_vote_monitor.clone();
70        let dag_state = self.dag_state.clone();
71        let randomness_signature_handler = self.randomness_signature_handler.clone();
72
73        let mut subscription = self.subscription.lock();
74        if let Some(handle) = subscription.take() {
75            handle.abort();
76        }
77        *subscription = Some(spawn_monitored_task!(Self::subscription_loop(
78            context,
79            network_client,
80            observer_service,
81            commit_vote_monitor,
82            dag_state,
83            peer,
84            randomness_signature_handler,
85        )));
86    }
87
88    /// Stop the active subscription (if any).
89    pub(crate) fn stop(&self) {
90        let mut subscription = self.subscription.lock();
91        if let Some(handle) = subscription.take() {
92            handle.abort();
93        }
94    }

96    async fn subscription_loop(
97        context: Arc<Context>,
98        network_client: Arc<C>,
99        observer_service: Arc<S>,
100        commit_vote_monitor: Arc<CommitVoteMonitor>,
101        dag_state: Arc<parking_lot::RwLock<DagState>>,
102        peer: PeerId,
103        randomness_signature_handler: Option<Arc<dyn RandomnessSignatureHandler>>,
104    ) {
105        const IMMEDIATE_RETRIES: i64 = 3;
106        const MIN_TIMEOUT: Duration = Duration::from_millis(500);
107        let mut backoff = mysten_common::backoff::ExponentialBackoff::new(
108            Duration::from_millis(100),
109            Duration::from_secs(10),
110        );
111
112        let mut retries: i64 = 0;
113        'subscription: loop {
114            let mut delay = Duration::ZERO;
115            if retries > IMMEDIATE_RETRIES {
116                delay = backoff.next().unwrap();
117                debug!(
118                    "Delaying retry {} of peer {:?} subscription, in {} seconds",
119                    retries,
120                    peer.clone(),
121                    delay.as_secs_f32(),
122                );
123                sleep(delay).await;
124            } else if retries > 0 {
125                tokio::task::yield_now().await;
126            }
127            retries += 1;
128
129            // Recompute highest rounds from DagState before each connection attempt
130            // so reconnections resume from where we left off rather than re-fetching
131            // already-seen blocks.
132            let highest_round_per_authority = {
133                let ds = dag_state.read();
134                let mut rounds = vec![0u64; context.committee.size()];
135                for (authority, _) in context.committee.authorities() {
136                    rounds[authority.value()] =
137                        ds.get_last_block_for_authority(authority).round() as u64;
138                }
139                rounds
140            };
141
142            // Subscribe to stream blocks from the peer.
143            let request_timeout = MIN_TIMEOUT.max(delay);
144            let mut blocks = match network_client
145                .stream_blocks(peer.clone(), highest_round_per_authority, request_timeout)
146                .await
147            {
148                Ok(blocks) => {
149                    debug!("Subscribed to peer {:?} after {} attempts", peer, retries);
150                    blocks
151                }
152                Err(e) => {
153                    debug!("Failed to subscribe to blocks from peer {:?}: {}", peer, e);
154                    continue 'subscription;
155                }
156            };
157
158            // On-demand task spawning with atomic counter for tracking active tasks
159            let max_parallel_tasks = context.committee.size();
160            let active_tasks = Arc::new(AtomicUsize::new(0));
161            let mut tasks = JoinSet::new();
162
163            'stream: loop {
164                let _scope = monitored_scope("ObserverSubscriberStreamConsumer");
165
166                match blocks.next().await {
167                    Some(item) => {
168                        context
169                            .metrics
170                            .node_metrics
171                            .observer_subscribed_blocks_batch_size
172                            .observe(item.blocks.len() as f64);
173
174                        // During catch-up (commit lagging behind quorum), drop silently --
175                        // those rounds will arrive via checkpoint sync, same as lagging validators.
176                        if let Some(handler) = &randomness_signature_handler
177                            && !is_commit_lagging(
178                                &context,
179                                dag_state.read().last_commit_index(),
180                                commit_vote_monitor.quorum_commit_index(),
181                            )
182                        {
183                            for sig in item.auxiliary_data.randomness_signatures {
184                                handler.handle_randomness_signature(sig);
185                            }
186                        }
187
188                        for block in item.blocks {
189                            // Backpressure: wait if we've hit max parallelism
190                            while active_tasks.load(Ordering::Acquire) >= max_parallel_tasks {
191                                // Block until a task completes instead of busy-waiting
192                                match tasks.join_next().await {
193                                    Some(Ok(_)) => {
194                                        // Task completed successfully, counter already decremented
195                                    }
196                                    Some(Err(e)) => {
197                                        warn!("Task failed with error: {}", e);
198                                    }
199                                    None => {
200                                        // This shouldn't happen if our counter is accurate
201                                        warn!(
202                                            "No tasks to join but counter shows {} active tasks",
203                                            active_tasks.load(Ordering::Acquire)
204                                        );
205                                        break;
206                                    }
207                                }
208                            }
209
210                            // Spawn a new task to handle this block
211                            active_tasks.fetch_add(1, Ordering::AcqRel);
212                            let counter = active_tasks.clone();
213                            let observer_service = observer_service.clone();
214                            let peer_cloned = peer.clone();
215
216                            tasks.spawn(async move {
217                                let _scope =
218                                    monitored_scope("ObserverSubscriberTask::handle_block");
219
220                                let result = observer_service
221                                    .handle_block(peer_cloned.clone(), block)
222                                    .await;
223                                if let Err(e) = result {
224                                    match e {
225                                        ConsensusError::BlockRejected {
226                                            block_ref,
227                                            reason,
228                                        } => {
229                                            debug!(
230                                                "Failed to process block from peer for block {:?}: {}",
231                                                block_ref, reason
232                                            );
233                                        }
234                                        _ => {
235                                            info!("Received invalid block from peer: {}", e);
236                                        }
237                                    }
238                                }
239
240                                // Decrement counter when done
241                                counter.fetch_sub(1, Ordering::AcqRel);
242                            });
243                        }
244
245                        // Reset retries when a block is received and also reset the backoff.
246                        retries = 0;
247                        backoff.reset();
248                    }
249                    None => {
250                        debug!("Subscription to blocks from peer {:?} ended", peer);
251                        retries += 1;
252                        break 'stream;
253                    }
254                }
255            }
256
257            // Wait for all spawned tasks to complete
258            while let Some(result) = tasks.join_next().await {
259                match result {
260                    Ok(_) => {} // Task completed successfully
261                    Err(e) => warn!("Task failed during cleanup: {}", e),
262                }
263            }
264
265            // Sanity check: ensure all tasks have completed
266            let remaining = active_tasks.load(Ordering::Acquire);
267            if remaining > 0 {
268                warn!(
269                    "Warning: {} tasks still marked as active after cleanup",
270                    remaining
271                );
272            }
273        }
274    }

401    #[tokio::test(flavor = "current_thread", start_paused = true)]
402    async fn test_observer_subscriber_retries() {
403        telemetry_subscribers::init_for_testing();
404        let (context, _keys) = Context::new_for_test(4);
405        let context = Arc::new(context);
406        let observer_service = Arc::new(ObserverSubscriberTestService::new());
407        let network_client = Arc::new(ObserverSubscriberTestClient::new());
408        let store = Arc::new(MemStore::new());
409        let commit_vote_monitor = Arc::new(CommitVoteMonitor::new(context.clone()));
410        let dag_state = Arc::new(RwLock::new(DagState::new(context.clone(), store)));
411
412        let subscriber = ObserverSubscriber::new(
413            context.clone(),
414            network_client,
415            observer_service.clone(),
416            commit_vote_monitor,
417            dag_state,
418            None,
419        );
420
421        // Subscribe to a validator peer
422        let peer = PeerId::Validator(context.committee.to_authority_index(2).unwrap());
423        subscriber.subscribe(peer.clone());
424
425        // Wait for enough blocks to be received
426        for _ in 0..10 {
427            sleep(Duration::from_secs(1)).await;
428            let calls = observer_service.handle_block_calls.lock();
429            if calls.len() >= 100 {
430                break;
431            }
432        }
433
434        // Even if the stream ends after 10 blocks, the subscriber should retry and get enough
435        // blocks eventually.
436        let calls = observer_service.handle_block_calls.lock();
437        assert!(calls.len() >= 100);
438        for (p, block) in calls.iter() {
439            assert_eq!(*p, peer);
440            assert_eq!(*block, Bytes::from(vec![3u8; 8])); // Peer index 2 + 1 = 3
441        }
442    }

444    #[tokio::test]
445    async fn test_observer_subscriber_override() {
446        telemetry_subscribers::init_for_testing();
447        let (context, _keys) = Context::new_for_test(4);
448        let context = Arc::new(context);
449        let observer_service = Arc::new(ObserverSubscriberTestService::new());
450        let network_client = Arc::new(ObserverSubscriberTestClient::new());
451        let store = Arc::new(MemStore::new());
452        let commit_vote_monitor = Arc::new(CommitVoteMonitor::new(context.clone()));
453        let dag_state = Arc::new(RwLock::new(DagState::new(context.clone(), store)));
454
455        let subscriber = ObserverSubscriber::new(
456            context.clone(),
457            network_client,
458            observer_service.clone(),
459            commit_vote_monitor,
460            dag_state,
461            None,
462        );
463
464        // Subscribe to first peer (validator 0)
465        let peer1 = PeerId::Validator(context.committee.to_authority_index(0).unwrap());
466        subscriber.subscribe(peer1.clone());
467
468        // Wait for some blocks to be received from peer1
469        sleep(Duration::from_millis(50)).await;
470        {
471            let calls = observer_service.handle_block_calls.lock();
472            assert!(!calls.is_empty(), "Should have received blocks from peer1");
473            // Verify blocks are from peer1 (value = 0 + 1 = 1)
474            for (p, block) in calls.iter() {
475                assert_eq!(*p, peer1);
476                assert_eq!(*block, Bytes::from(vec![1u8; 8]));
477            }
478        }
479
480        // Clear the received blocks for clarity
481        observer_service.handle_block_calls.lock().clear();
482
483        // Subscribe to second peer (validator 2) - this should override the first subscription
484        let peer2 = PeerId::Validator(context.committee.to_authority_index(2).unwrap());
485        subscriber.subscribe(peer2.clone());
486
487        // Wait for blocks from the new peer
488        sleep(Duration::from_millis(100)).await;
489        {
490            let calls = observer_service.handle_block_calls.lock();
491            assert!(!calls.is_empty(), "Should have received blocks from peer2");
492            // Verify ALL blocks are from peer2 (value = 2 + 1 = 3), none from peer1
493            for (p, block) in calls.iter() {
494                assert_eq!(*p, peer2, "All blocks should be from peer2 after override");
495                assert_eq!(
496                    *block,
497                    Bytes::from(vec![3u8; 8]),
498                    "Block content should match peer2"
499                );
500            }
501        }
502
503        // Clear blocks again
504        let count_before_stop = observer_service.handle_block_calls.lock().len();
505
506        // Test that stop() still works
507        subscriber.stop();
508
509        // Wait and verify no new blocks are received after stop
510        sleep(Duration::from_millis(50)).await;
511        let count_after_stop = observer_service.handle_block_calls.lock().len();
512        assert_eq!(
513            count_before_stop, count_after_stop,
514            "No new blocks should be received after stop()"
515        );
516    }
