72// Keeps a mapping between the missing blocks that have been instructed to be fetched and the authorities
73// that are currently fetching them. For a block ref there is a maximum number of authorities that can
74// concurrently fetch it. The authority ids that are currently fetching a block are set on the corresponding
75// `BTreeSet` and basically they act as "locks".
76struct InflightBlocksMap {
77    inner: Mutex<HashMap<BlockRef, BTreeSet<AuthorityIndex>>>,
78}
79
80impl InflightBlocksMap {
81    fn new() -> Arc<Self> {
82        Arc::new(Self {
83            inner: Mutex::new(HashMap::new()),
84        })
85    }
86
87    /// Locks the blocks to be fetched for the assigned `peer_index`. We want to avoid re-fetching the
88    /// missing blocks from too many authorities at the same time, thus we limit the concurrency
89    /// per block by attempting to lock per block. If a block is already fetched by the maximum allowed
90    /// number of authorities, then the block ref will not be included in the returned set. The method
91    /// returns all the block refs that have been successfully locked and allowed to be fetched.
92    fn lock_blocks(
93        self: &Arc<Self>,
94        missing_block_refs: BTreeSet<BlockRef>,
95        peer: AuthorityIndex,
96    ) -> Option<BlocksGuard> {
97        let mut blocks = BTreeSet::new();
98        let mut inner = self.inner.lock();
99
100        for block_ref in missing_block_refs {
101            // check that the number of authorities that are already instructed to fetch the block is not
102            // higher than the allowed and the `peer_index` has not already been instructed to do that.
103            let authorities = inner.entry(block_ref).or_default();
104            if authorities.len() < MAX_AUTHORITIES_TO_FETCH_PER_BLOCK
105                && authorities.get(&peer).is_none()
106            {
107                assert!(authorities.insert(peer));
108                blocks.insert(block_ref);
109            }
110        }
111
112        if blocks.is_empty() {
113            None
114        } else {
115            Some(BlocksGuard {
116                map: self.clone(),
117                block_refs: blocks,
118                peer,
119            })
120        }
121    }
122
123    /// Unlocks the provided block references for the given `peer`. The unlocking is strict, meaning that
124    /// if this method is called for a specific block ref and peer more times than the corresponding lock
125    /// has been called, it will panic.
126    fn unlock_blocks(self: &Arc<Self>, block_refs: &BTreeSet<BlockRef>, peer: AuthorityIndex) {
127        // Now mark all the blocks as fetched from the map
128        let mut blocks_to_fetch = self.inner.lock();
129        for block_ref in block_refs {
130            let authorities = blocks_to_fetch
131                .get_mut(block_ref)
132                .expect("Should have found a non empty map");
133
134            assert!(authorities.remove(&peer), "Peer index should be present!");
135
136            // if the last one then just clean up
137            if authorities.is_empty() {
138                blocks_to_fetch.remove(block_ref);
139            }
140        }
141    }
142
143    /// Drops the provided `blocks_guard` which will force to unlock the blocks, and lock now again the
144    /// referenced block refs. The swap is best effort and there is no guarantee that the `peer` will
145    /// be able to acquire the new locks.
146    fn swap_locks(
147        self: &Arc<Self>,
148        blocks_guard: BlocksGuard,
149        peer: AuthorityIndex,
150    ) -> Option<BlocksGuard> {
151        let block_refs = blocks_guard.block_refs.clone();
152
153        // Explicitly drop the guard
154        drop(blocks_guard);
155
156        // Now create new guard
157        self.lock_blocks(block_refs, peer)
158    }
159
160    #[cfg(test)]
161    fn num_of_locked_blocks(self: &Arc<Self>) -> usize {
162        let inner = self.inner.lock();
163        inner.len()
164    }
165}

182impl SynchronizerHandle {
183    /// Explicitly asks from the synchronizer to fetch the blocks - provided the block_refs set - from
184    /// the peer authority.
185    pub(crate) async fn fetch_blocks(
186        &self,
187        missing_block_refs: BTreeSet<BlockRef>,
188        peer_index: AuthorityIndex,
189    ) -> ConsensusResult<()> {
190        let (sender, receiver) = oneshot::channel();
191        self.commands_sender
192            .send(Command::FetchBlocks {
193                missing_block_refs,
194                peer_index,
195                result: sender,
196            })
197            .await
198            .map_err(|_err| ConsensusError::Shutdown)?;
199        receiver.await.map_err(|_err| ConsensusError::Shutdown)?
200    }
201
202    pub(crate) async fn stop(&self) -> Result<(), JoinError> {
203        let mut tasks = self.tasks.lock().await;
204        tasks.abort_all();
205        while let Some(result) = tasks.join_next().await {
206            result?
207        }
208        Ok(())
209    }

212/// `Synchronizer` oversees live block synchronization, crucial for node progress. Live synchronization
213/// refers to the process of retrieving missing blocks, particularly those essential for advancing a node
214/// when data from only a few rounds is absent. If a node significantly lags behind the network,
215/// `commit_syncer` handles fetching missing blocks via a more efficient approach. `Synchronizer`
216/// aims for swift catch-up employing two mechanisms:
217///
218/// 1. Explicitly requesting missing blocks from designated authorities via the "block send" path.
219///    This includes attempting to fetch any missing ancestors necessary for processing a received block.
220///    Such requests prioritize the block author, maximizing the chance of prompt retrieval.
221///    A locking mechanism allows concurrent requests for missing blocks from up to two authorities
222///    simultaneously, enhancing the chances of timely retrieval. Notably, if additional missing blocks
223///    arise during block processing, requests to the same authority are deferred to the scheduler.
224///
225/// 2. Periodically requesting missing blocks via a scheduler. This primarily serves to retrieve
226///    missing blocks that were not ancestors of a received block via the "block send" path.
227///    The scheduler operates on either a fixed periodic basis or is triggered immediately
228///    after explicit fetches described in (1), ensuring continued block retrieval if gaps persist.
229///
230/// Additionally to the above, the synchronizer can synchronize and fetch the last own proposed block
231/// from the network peers as best effort approach to recover node from amnesia and avoid making the
232/// node equivocate.
233pub(crate) struct Synchronizer<C: NetworkClient, V: BlockVerifier, D: CoreThreadDispatcher> {
234    context: Arc<Context>,
235    commands_receiver: Receiver<Command>,
236    fetch_block_senders: BTreeMap<AuthorityIndex, Sender<BlocksGuard>>,
237    core_dispatcher: Arc<D>,
238    commit_vote_monitor: Arc<CommitVoteMonitor>,
239    dag_state: Arc<RwLock<DagState>>,
240    fetch_blocks_scheduler_task: JoinSet<()>,
241    fetch_own_last_block_task: JoinSet<()>,
242    network_client: Arc<C>,
243    block_verifier: Arc<V>,
244    transaction_certifier: TransactionCertifier,
245    round_tracker: Arc<RwLock<RoundTracker>>,
246    inflight_blocks_map: Arc<InflightBlocksMap>,
247    commands_sender: Sender<Command>,
248}

251    pub(crate) fn start(
252        network_client: Arc<C>,
253        context: Arc<Context>,
254        core_dispatcher: Arc<D>,
255        commit_vote_monitor: Arc<CommitVoteMonitor>,
256        block_verifier: Arc<V>,
257        transaction_certifier: TransactionCertifier,
258        round_tracker: Arc<RwLock<RoundTracker>>,
259        dag_state: Arc<RwLock<DagState>>,
260        sync_last_known_own_block: bool,
261    ) -> Arc<SynchronizerHandle> {
262        let (commands_sender, commands_receiver) =
263            channel("consensus_synchronizer_commands", 1_000);
264        let inflight_blocks_map = InflightBlocksMap::new();
265
266        // Spawn the tasks to fetch the blocks from the others
267        let mut fetch_block_senders = BTreeMap::new();
268        let mut tasks = JoinSet::new();
269        for (index, _) in context.committee.authorities() {
270            if index == context.own_index {
271                continue;
272            }
273            let (sender, receiver) =
274                channel("consensus_synchronizer_fetches", FETCH_BLOCKS_CONCURRENCY);
275            let fetch_blocks_from_authority_async = Self::fetch_blocks_from_authority(
276                index,
277                network_client.clone(),
278                block_verifier.clone(),
279                transaction_certifier.clone(),
280                commit_vote_monitor.clone(),
281                context.clone(),
282                core_dispatcher.clone(),
283                dag_state.clone(),
284                receiver,
285                commands_sender.clone(),
286                round_tracker.clone(),
287            );
288            tasks.spawn(monitored_future!(fetch_blocks_from_authority_async));
289            fetch_block_senders.insert(index, sender);
290        }
291
292        let commands_sender_clone = commands_sender.clone();
293
294        if sync_last_known_own_block {
295            commands_sender
296                .try_send(Command::FetchOwnLastBlock)
297                .expect("Failed to sync our last block");
298        }
299
300        // Spawn the task to listen to the requests & periodic runs
301        tasks.spawn(monitored_future!(async move {
302            let mut s = Self {
303                context,
304                commands_receiver,
305                fetch_block_senders,
306                core_dispatcher,
307                commit_vote_monitor,
308                fetch_blocks_scheduler_task: JoinSet::new(),
309                fetch_own_last_block_task: JoinSet::new(),
310                network_client,
311                block_verifier,
312                transaction_certifier,
313                inflight_blocks_map,
314                commands_sender: commands_sender_clone,
315                dag_state,
316                round_tracker,
317            };
318            s.run().await;
319        }));
320
321        Arc::new(SynchronizerHandle {
322            commands_sender,
323            tasks: tokio::sync::Mutex::new(tasks),
324        })
325    }

327    // The main loop to listen for the submitted commands.
328    async fn run(&mut self) {
329        // We want the synchronizer to run periodically every 200ms to fetch any missing blocks.
330        const PERIODIC_FETCH_INTERVAL: Duration = Duration::from_millis(200);
331        let scheduler_timeout = sleep_until(Instant::now() + PERIODIC_FETCH_INTERVAL);
332
333        tokio::pin!(scheduler_timeout);
334
335        loop {
336            tokio::select! {
337                Some(command) = self.commands_receiver.recv() => {
338                    match command {
339                        Command::FetchBlocks{ missing_block_refs, peer_index, result } => {
340                            if peer_index == self.context.own_index {
341                                error!("We should never attempt to fetch blocks from our own node");
342                                continue;
343                            }
344
345                            // Keep only the max allowed blocks to request. Additional missing blocks
346                            // will be fetched via periodic sync.
347                            // Fetch from the lowest to highest round, to ensure progress.
348                            let missing_block_refs = missing_block_refs
349                                .into_iter()
350                                .take(self.context.parameters.max_blocks_per_sync)
351                                .collect();
352
353                            let blocks_guard = self.inflight_blocks_map.lock_blocks(missing_block_refs, peer_index);
354                            let Some(blocks_guard) = blocks_guard else {
355                                result.send(Ok(())).ok();
356                                continue;
357                            };
358
359                            // We don't block if the corresponding peer task is saturated - but we rather drop the request. That's ok as the periodic
360                            // synchronization task will handle any still missing blocks in next run.
361                            let r = self
362                                .fetch_block_senders
363                                .get(&peer_index)
364                                .expect("Fatal error, sender should be present")
365                                .try_send(blocks_guard)
366                                .map_err(|err| {
367                                    match err {
368                                        TrySendError::Full(_) => {
369                                            let peer_hostname = &self.context.committee.authority(peer_index).hostname;
370                                            self.context
371                                                .metrics
372                                                .node_metrics
373                                                .synchronizer_skipped_fetch_requests
374                                                .with_label_values(&[peer_hostname])
375                                                .inc();
376                                            ConsensusError::SynchronizerSaturated(peer_index)
377                                        },
378                                        TrySendError::Closed(_) => ConsensusError::Shutdown
379                                    }
380                                });
381
382                            result.send(r).ok();
383                        }
384                        Command::FetchOwnLastBlock => {
385                            if self.fetch_own_last_block_task.is_empty() {
386                                self.start_fetch_own_last_block_task();
387                            }
388                        }
389                        Command::KickOffScheduler => {
390                            // just reset the scheduler timeout timer to run immediately if not already running.
391                            // If the scheduler is already running then just reduce the remaining time to run.
392                            let timeout = if self.fetch_blocks_scheduler_task.is_empty() {
393                                Instant::now()
394                            } else {
395                                Instant::now() + PERIODIC_FETCH_INTERVAL.checked_div(2).unwrap()
396                            };
397
398                            // only reset if it is earlier than the next deadline
399                            if timeout < scheduler_timeout.deadline() {
400                                scheduler_timeout.as_mut().reset(timeout);
401                            }
402                        }
403                    }
404                },
405                Some(result) = self.fetch_own_last_block_task.join_next(), if !self.fetch_own_last_block_task.is_empty() => {
406                    match result {
407                        Ok(()) => {},
408                        Err(e) => {
409                            if e.is_cancelled() {
410                            } else if e.is_panic() {
411                                std::panic::resume_unwind(e.into_panic());
412                            } else {
413                                panic!("fetch our last block task failed: {e}");
414                            }
415                        },
416                    };
417                },
418                Some(result) = self.fetch_blocks_scheduler_task.join_next(), if !self.fetch_blocks_scheduler_task.is_empty() => {
419                    match result {
420                        Ok(()) => {},
421                        Err(e) => {
422                            if e.is_cancelled() {
423                            } else if e.is_panic() {
424                                std::panic::resume_unwind(e.into_panic());
425                            } else {
426                                panic!("fetch blocks scheduler task failed: {e}");
427                            }
428                        },
429                    };
430                },
431                () = &mut scheduler_timeout => {
432                    // we want to start a new task only if the previous one has already finished.
433                    // TODO: consider starting backup fetches in parallel, when a fetch takes too long?
434                    if self.fetch_blocks_scheduler_task.is_empty()
435                        && let Err(err) = self.start_fetch_missing_blocks_task().await {
436                            debug!("Core is shutting down, synchronizer is shutting down: {err:?}");
437                            return;
438                        };
439
440                    scheduler_timeout
441                        .as_mut()
442                        .reset(Instant::now() + PERIODIC_FETCH_INTERVAL);
443                }
444            }
445        }
446    }

448    async fn fetch_blocks_from_authority(
449        peer_index: AuthorityIndex,
450        network_client: Arc<C>,
451        block_verifier: Arc<V>,
452        transaction_certifier: TransactionCertifier,
453        commit_vote_monitor: Arc<CommitVoteMonitor>,
454        context: Arc<Context>,
455        core_dispatcher: Arc<D>,
456        dag_state: Arc<RwLock<DagState>>,
457        mut receiver: Receiver<BlocksGuard>,
458        commands_sender: Sender<Command>,
459        round_tracker: Arc<RwLock<RoundTracker>>,
460    ) {
461        const MAX_RETRIES: u32 = 3;
462        let peer_hostname = &context.committee.authority(peer_index).hostname;
463        let mut requests = FuturesUnordered::new();
464
465        loop {
466            tokio::select! {
467                Some(blocks_guard) = receiver.recv(), if requests.len() < FETCH_BLOCKS_CONCURRENCY => {
468                    // get the highest accepted rounds
469                    let highest_rounds = Self::get_highest_accepted_rounds(dag_state.clone(), &context);
470
471                    requests.push(Self::fetch_blocks_request(network_client.clone(), peer_index, blocks_guard, highest_rounds, true, FETCH_REQUEST_TIMEOUT, 1))
472                },
473                Some((response, blocks_guard, retries, _peer, highest_rounds)) = requests.next() => {
474                    match response {
475                        Ok(blocks) => {
476                            if let Err(err) = Self::process_fetched_blocks(blocks,
477                                peer_index,
478                                blocks_guard,
479                                core_dispatcher.clone(),
480                                block_verifier.clone(),
481                                transaction_certifier.clone(),
482                                commit_vote_monitor.clone(),
483                                context.clone(),
484                                commands_sender.clone(),
485                                round_tracker.clone(),
486                                "live"
487                            ).await {
488                                warn!("Error while processing fetched blocks from peer {peer_index} {peer_hostname}: {err}");
489                                context.metrics.node_metrics.synchronizer_process_fetched_failures.with_label_values(&[peer_hostname.as_str(), "live"]).inc();
490                            }
491                        },
492                        Err(_) => {
493                            context.metrics.node_metrics.synchronizer_fetch_failures.with_label_values(&[peer_hostname.as_str(), "live"]).inc();
494                            if retries <= MAX_RETRIES {
495                                requests.push(Self::fetch_blocks_request(network_client.clone(), peer_index, blocks_guard, highest_rounds, true, FETCH_REQUEST_TIMEOUT, retries))
496                            } else {
497                                warn!("Max retries {retries} reached while trying to fetch blocks from peer {peer_index} {peer_hostname}.");
498                                // we don't necessarily need to do, but dropping the guard here to unlock the blocks
499                                drop(blocks_guard);
500                            }
501                        }
502                    }
503                },
504                else => {
505                    info!("Fetching blocks from authority {peer_index} task will now abort.");
506                    break;
507                }
508            }
509        }
510    }

512    /// Processes the requested raw fetched blocks from peer `peer_index`. If no error is returned then
513    /// the verified blocks are immediately sent to Core for processing.
514    async fn process_fetched_blocks(
515        mut serialized_blocks: Vec<Bytes>,
516        peer_index: AuthorityIndex,
517        requested_blocks_guard: BlocksGuard,
518        core_dispatcher: Arc<D>,
519        block_verifier: Arc<V>,
520        transaction_certifier: TransactionCertifier,
521        commit_vote_monitor: Arc<CommitVoteMonitor>,
522        context: Arc<Context>,
523        commands_sender: Sender<Command>,
524        round_tracker: Arc<RwLock<RoundTracker>>,
525        sync_method: &str,
526    ) -> ConsensusResult<()> {
527        if serialized_blocks.is_empty() {
528            return Ok(());
529        }
530
531        // Limit the number of the returned blocks processed.
532        serialized_blocks.truncate(context.parameters.max_blocks_per_sync);
533
534        // Verify all the fetched blocks
535        let blocks = Handle::current()
536            .spawn_blocking({
537                let block_verifier = block_verifier.clone();
538                let context = context.clone();
539                move || {
540                    Self::verify_blocks(
541                        serialized_blocks,
542                        block_verifier,
543                        transaction_certifier,
544                        &context,
545                        peer_index,
546                    )
547                }
548            })
549            .await
550            .expect("Spawn blocking should not fail")?;
551
552        // Record commit votes from the verified blocks.
553        for block in &blocks {
554            commit_vote_monitor.observe_block(block);
555        }
556
557        // Update round tracker from the verified blocks. For fetched blocks,
558        // excluded_ancestors are not available so we use an empty vector.
559        {
560            let mut tracker = round_tracker.write();
561            for block in &blocks {
562                tracker.update_from_verified_block(&ExtendedBlock {
563                    block: block.clone(),
564                    excluded_ancestors: vec![],
565                });
566            }
567        }
568
569        let metrics = &context.metrics.node_metrics;
570        let peer_hostname = &context.committee.authority(peer_index).hostname;
571        metrics
572            .synchronizer_fetched_blocks_by_peer
573            .with_label_values(&[peer_hostname.as_str(), sync_method])
574            .inc_by(blocks.len() as u64);
575        for block in &blocks {
576            let block_hostname = &context.committee.authority(block.author()).hostname;
577            metrics
578                .synchronizer_fetched_blocks_by_authority
579                .with_label_values(&[block_hostname.as_str(), sync_method])
580                .inc();
581        }
582
583        debug!(
584            "Synced {} missing blocks from peer {peer_index} {peer_hostname}: {}",
585            blocks.len(),
586            blocks.iter().map(|b| b.reference().to_string()).join(", "),
587        );
588
589        // Now send them to core for processing. Ignore the returned missing blocks as we don't want
590        // this mechanism to keep feedback looping on fetching more blocks. The periodic synchronization
591        // will take care of that.
592        let missing_blocks = core_dispatcher
593            .add_blocks(blocks)
594            .await
595            .map_err(|_| ConsensusError::Shutdown)?;
596
597        // now release all the locked blocks as they have been fetched, verified & processed
598        drop(requested_blocks_guard);
599
600        // kick off immediately the scheduled synchronizer
601        if !missing_blocks.is_empty() {
602            // do not block here, so we avoid any possible cycles.
603            if let Err(TrySendError::Full(_)) = commands_sender.try_send(Command::KickOffScheduler)
604            {
605                warn!("Commands channel is full")
606            }
607        }
608
609        context
610            .metrics
611            .node_metrics
612            .missing_blocks_after_fetch_total
613            .inc_by(missing_blocks.len() as u64);
614
615        Ok(())
616    }

618    fn get_highest_accepted_rounds(
619        dag_state: Arc<RwLock<DagState>>,
620        context: &Arc<Context>,
621    ) -> Vec<Round> {
622        let blocks = dag_state
623            .read()
624            .get_last_cached_block_per_authority(Round::MAX);
625        assert_eq!(blocks.len(), context.committee.size());
626
627        blocks
628            .into_iter()
629            .map(|(block, _)| block.round())
630            .collect::<Vec<_>>()
631    }

633    fn verify_blocks(
634        serialized_blocks: Vec<Bytes>,
635        block_verifier: Arc<V>,
636        transaction_certifier: TransactionCertifier,
637        context: &Context,
638        peer_index: AuthorityIndex,
639    ) -> ConsensusResult<Vec<VerifiedBlock>> {
640        let mut verified_blocks = Vec::new();
641        let mut voted_blocks = Vec::new();
642        for serialized_block in serialized_blocks {
643            let signed_block: SignedBlock =
644                bcs::from_bytes(&serialized_block).map_err(ConsensusError::MalformedBlock)?;
645
646            // TODO: cache received and verified block refs to avoid duplicated work.
647            let (verified_block, reject_txn_votes) = block_verifier
648                .verify_and_vote(signed_block, serialized_block)
649                .tap_err(|e| {
650                    let hostname = context.committee.authority(peer_index).hostname.clone();
651                    context
652                        .metrics
653                        .node_metrics
654                        .invalid_blocks
655                        .with_label_values(&[hostname.as_str(), "synchronizer", e.clone().name()])
656                        .inc();
657                    info!("Invalid block received from {}: {}", peer_index, e);
658                })?;
659
660            // TODO: improve efficiency, maybe suspend and continue processing the block asynchronously.
661            let now = context.clock.timestamp_utc_ms();
662            let drift = verified_block.timestamp_ms().saturating_sub(now);
663            if drift > 0 {
664                let peer_hostname = &context
665                    .committee
666                    .authority(verified_block.author())
667                    .hostname;
668                context
669                    .metrics
670                    .node_metrics
671                    .block_timestamp_drift_ms
672                    .with_label_values(&[peer_hostname.as_str(), "synchronizer"])
673                    .inc_by(drift);
674
675                trace!(
676                    "Synced block {} timestamp {} is in the future (now={}).",
677                    verified_block.reference(),
678                    verified_block.timestamp_ms(),
679                    now
680                );
681            }
682
683            verified_blocks.push(verified_block.clone());
684            voted_blocks.push((verified_block, reject_txn_votes));
685        }
686
687        if context.protocol_config.mysticeti_fastpath() {
688            transaction_certifier.add_voted_blocks(voted_blocks);
689        }
690
691        Ok(verified_blocks)
692    }

694    async fn fetch_blocks_request(
695        network_client: Arc<C>,
696        peer: AuthorityIndex,
697        blocks_guard: BlocksGuard,
698        highest_rounds: Vec<Round>,
699        breadth_first: bool,
700        request_timeout: Duration,
701        mut retries: u32,
702    ) -> (
703        ConsensusResult<Vec<Bytes>>,
704        BlocksGuard,
705        u32,
706        AuthorityIndex,
707        Vec<Round>,
708    ) {
709        let start = Instant::now();
710        let resp = timeout(
711            request_timeout,
712            network_client.fetch_blocks(
713                peer,
714                blocks_guard
715                    .block_refs
716                    .clone()
717                    .into_iter()
718                    .collect::<Vec<_>>(),
719                highest_rounds.clone().into_iter().collect::<Vec<_>>(),
720                breadth_first,
721                request_timeout,
722            ),
723        )
724        .await;
725
726        fail_point_async!("consensus-delay");
727
728        let resp = match resp {
729            Ok(Err(err)) => {
730                // Add a delay before retrying - if that is needed. If request has timed out then eventually
731                // this will be a no-op.
732                sleep_until(start + request_timeout).await;
733                retries += 1;
734                Err(err)
735            } // network error
736            Err(err) => {
737                // timeout
738                sleep_until(start + request_timeout).await;
739                retries += 1;
740                Err(ConsensusError::NetworkRequestTimeout(err.to_string()))
741            }
742            Ok(result) => result,
743        };
744        (resp, blocks_guard, retries, peer, highest_rounds)
745    }

747    fn start_fetch_own_last_block_task(&mut self) {
748        const FETCH_OWN_BLOCK_RETRY_DELAY: Duration = Duration::from_millis(1_000);
749        const MAX_RETRY_DELAY_STEP: Duration = Duration::from_millis(4_000);
750
751        let context = self.context.clone();
752        let dag_state = self.dag_state.clone();
753        let network_client = self.network_client.clone();
754        let block_verifier = self.block_verifier.clone();
755        let core_dispatcher = self.core_dispatcher.clone();
756
757        self.fetch_own_last_block_task
758            .spawn(monitored_future!(async move {
759                let _scope = monitored_scope("FetchOwnLastBlockTask");
760
761                let fetch_own_block = |authority_index: AuthorityIndex, fetch_own_block_delay: Duration| {
762                    let network_client_cloned = network_client.clone();
763                    let own_index = context.own_index;
764                    async move {
765                        sleep(fetch_own_block_delay).await;
766                        let r = network_client_cloned.fetch_latest_blocks(authority_index, vec![own_index], FETCH_REQUEST_TIMEOUT).await;
767                        (r, authority_index)
768                    }
769                };
770
771                let process_blocks = |blocks: Vec<Bytes>, authority_index: AuthorityIndex| -> ConsensusResult<Vec<VerifiedBlock>> {
772                    let mut result = Vec::new();
773                    for serialized_block in blocks {
774                        let signed_block = bcs::from_bytes(&serialized_block).map_err(ConsensusError::MalformedBlock)?;
775                        let (verified_block, _) = block_verifier.verify_and_vote(signed_block, serialized_block).tap_err(|err|{
776                            let hostname = context.committee.authority(authority_index).hostname.clone();
777                            context
778                                .metrics
779                                .node_metrics
780                                .invalid_blocks
781                                .with_label_values(&[hostname.as_str(), "synchronizer_own_block", err.clone().name()])
782                                .inc();
783                            warn!("Invalid block received from {}: {}", authority_index, err);
784                        })?;
785
786                        if verified_block.author() != context.own_index {
787                            return Err(ConsensusError::UnexpectedLastOwnBlock { index: authority_index, block_ref: verified_block.reference()});
788                        }
789                        result.push(verified_block);
790                    }
791                    Ok(result)
792                };
793
794                // Get the highest of all the results. Retry until at least `f+1` results have been gathered.
795                let mut highest_round;
796                let mut retries = 0;
797                let mut retry_delay_step = Duration::from_millis(500);
798                'main:loop {
799                    if context.committee.size() == 1 {
800                        highest_round = dag_state.read().get_last_proposed_block().round();
801                        info!("Only one node in the network, will not try fetching own last block from peers.");
802                        break 'main;
803                    }
804
805                    let mut total_stake = 0;
806                    highest_round = 0;
807
808                    // Ask all the other peers about our last block
809                    let mut results = FuturesUnordered::new();
810
811                    for (authority_index, _authority) in context.committee.authorities() {
812                        if authority_index != context.own_index {
813                            results.push(fetch_own_block(authority_index, Duration::from_millis(0)));
814                        }
815                    }
816
817                    // Gather the results but wait to timeout as well
818                    let timer = sleep_until(Instant::now() + context.parameters.sync_last_known_own_block_timeout);
819                    tokio::pin!(timer);
820
821                    'inner: loop {
822                        tokio::select! {
823                            result = results.next() => {
824                                let Some((result, authority_index)) = result else {
825                                    break 'inner;
826                                };
827                                match result {
828                                    Ok(result) => {
829                                        match process_blocks(result, authority_index) {
830                                            Ok(blocks) => {
831                                                let max_round = blocks.into_iter().map(|b|b.round()).max().unwrap_or(0);
832                                                highest_round = highest_round.max(max_round);
833
834                                                total_stake += context.committee.stake(authority_index);
835                                            },
836                                            Err(err) => {
837                                                warn!("Invalid result returned from {authority_index} while fetching last own block: {err}");
838                                            }
839                                        }
840                                    },
841                                    Err(err) => {
842                                        warn!("Error {err} while fetching our own block from peer {authority_index}. Will retry.");
843                                        results.push(fetch_own_block(authority_index, FETCH_OWN_BLOCK_RETRY_DELAY));
844                                    }
845                                }
846                            },
847                            () = &mut timer => {
848                                info!("Timeout while trying to sync our own last block from peers");
849                                break 'inner;
850                            }
851                        }
852                    }
853
854                    // Request at least f+1 stake to have replied back.
855                    if context.committee.reached_validity(total_stake) {
856                        info!("{} out of {} total stake returned acceptable results for our own last block with highest round {}, with {retries} retries.", total_stake, context.committee.total_stake(), highest_round);
857                        break 'main;
858                    }
859
860                    retries += 1;
861                    context.metrics.node_metrics.sync_last_known_own_block_retries.inc();
862                    warn!("Not enough stake: {} out of {} total stake returned acceptable results for our own last block with highest round {}. Will now retry {retries}.", total_stake, context.committee.total_stake(), highest_round);
863
864                    sleep(retry_delay_step).await;
865
866                    retry_delay_step = Duration::from_secs_f64(retry_delay_step.as_secs_f64() * 1.5);
867                    retry_delay_step = retry_delay_step.min(MAX_RETRY_DELAY_STEP);
868                }
869
870                // Update the Core with the highest detected round
871                context.metrics.node_metrics.last_known_own_block_round.set(highest_round as i64);
872
873                if let Err(err) = core_dispatcher.set_last_known_proposed_round(highest_round) {
874                    warn!("Error received while calling dispatcher, probably dispatcher is shutting down, will now exit: {err:?}");
875                }
876            }));
877    }

879    async fn start_fetch_missing_blocks_task(&mut self) -> ConsensusResult<()> {
880        if self.context.committee.size() == 1 {
881            trace!(
882                "Only one node in the network, will not try fetching missing blocks from peers."
883            );
884            return Ok(());
885        }
886
887        let missing_blocks = self
888            .core_dispatcher
889            .get_missing_blocks()
890            .await
891            .map_err(|_err| ConsensusError::Shutdown)?;
892
893        // No reason to kick off the scheduler if there are no missing blocks to fetch
894        if missing_blocks.is_empty() {
895            return Ok(());
896        }
897
898        let context = self.context.clone();
899        let network_client = self.network_client.clone();
900        let block_verifier = self.block_verifier.clone();
901        let transaction_certifier = self.transaction_certifier.clone();
902        let commit_vote_monitor = self.commit_vote_monitor.clone();
903        let core_dispatcher = self.core_dispatcher.clone();
904        let blocks_to_fetch = self.inflight_blocks_map.clone();
905        let commands_sender = self.commands_sender.clone();
906        let dag_state = self.dag_state.clone();
907        let round_tracker = self.round_tracker.clone();
908
909        // If we are commit lagging, then we don't want to enable the scheduler. As the node is sycnhronizing via the commit syncer, the certified commits
910        // will bring all the necessary blocks to run the commits. As the commits are certified, we are guaranteed that all the necessary causal history is present.
911        if self.is_commit_lagging() {
912            return Ok(());
913        }
914
915        self.fetch_blocks_scheduler_task
916            .spawn(monitored_future!(async move {
917                let _scope = monitored_scope("FetchMissingBlocksScheduler");
918                context
919                    .metrics
920                    .node_metrics
921                    .fetch_blocks_scheduler_inflight
922                    .inc();
923                let total_requested = missing_blocks.len();
924
925                fail_point_async!("consensus-delay");
926                // Fetch blocks from peers
927                let results = Self::fetch_blocks_from_authorities(
928                    context.clone(),
929                    blocks_to_fetch.clone(),
930                    network_client,
931                    missing_blocks,
932                    dag_state,
933                )
934                .await;
935                context
936                    .metrics
937                    .node_metrics
938                    .fetch_blocks_scheduler_inflight
939                    .dec();
940                if results.is_empty() {
941                    return;
942                }
943
944                // Now process the returned results
945                let mut total_fetched = 0;
946                for (blocks_guard, fetched_blocks, peer) in results {
947                    total_fetched += fetched_blocks.len();
948
949                    if let Err(err) = Self::process_fetched_blocks(
950                        fetched_blocks,
951                        peer,
952                        blocks_guard,
953                        core_dispatcher.clone(),
954                        block_verifier.clone(),
955                        transaction_certifier.clone(),
956                        commit_vote_monitor.clone(),
957                        context.clone(),
958                        commands_sender.clone(),
959                        round_tracker.clone(),
960                        "periodic",
961                    )
962                    .await
963                    {
964                        warn!(
965                            "Error occurred while processing fetched blocks from peer {peer}: {err}"
966                        );
967                        context
968                            .metrics
969                            .node_metrics
970                            .synchronizer_process_fetched_failures
971                            .with_label_values(&[
972                                context.committee.authority(peer).hostname.as_str(),
973                                "periodic",
974                            ])
975                            .inc();
976                    }
977                }
978
979                debug!(
980                    "Total blocks requested to fetch: {}, total fetched: {}",
981                    total_requested, total_fetched
982                );
983            }));
984        Ok(())
985    }

988        let last_commit_index = self.dag_state.read().last_commit_index();
989        let quorum_commit_index = self.commit_vote_monitor.quorum_commit_index();
990        let commit_threshold = last_commit_index

992        commit_threshold < quorum_commit_index

995    /// Fetches the `missing_blocks` from peers. Requests the same number of authorities with missing blocks from each peer.
996    /// Each response from peer can contain the requested blocks, and additional blocks from the last accepted round for
997    /// authorities with missing blocks.
998    /// Each element of the vector is a tuple which contains the requested missing block refs, the returned blocks and
999    /// the peer authority index.
1000    async fn fetch_blocks_from_authorities(
1001        context: Arc<Context>,
1002        inflight_blocks: Arc<InflightBlocksMap>,
1003        network_client: Arc<C>,
1004        missing_blocks: BTreeSet<BlockRef>,
1005        dag_state: Arc<RwLock<DagState>>,
1006    ) -> Vec<(BlocksGuard, Vec<Bytes>, AuthorityIndex)> {
1007        // Preliminary truncation of missing blocks to fetch. Since each peer can have different
1008        // number of missing blocks and the fetching is batched by peer, so keep more than max_blocks_per_sync
1009        // per peer on average.
1010        let missing_blocks = missing_blocks
1011            .into_iter()
1012            .take(2 * MAX_PERIODIC_SYNC_PEERS * context.parameters.max_blocks_per_sync)
1013            .collect::<Vec<_>>();
1014
1015        // Maps authorities to the missing blocks they have.
1016        let mut authorities = BTreeMap::<AuthorityIndex, Vec<BlockRef>>::new();
1017        for block_ref in &missing_blocks {
1018            authorities
1019                .entry(block_ref.author)
1020                .or_default()
1021                .push(*block_ref);
1022        }
1023        // Distribute the same number of authorities into each peer to sync.
1024        // When running this function, context.committee.size() is always greater than 1.
1025        let num_authorities_per_peer = authorities
1026            .len()
1027            .div_ceil((context.committee.size() - 1).min(MAX_PERIODIC_SYNC_PEERS));
1028
1029        // Update metrics related to missing blocks.
1030        let mut missing_blocks_per_authority = vec![0; context.committee.size()];
1031        for (authority, blocks) in &authorities {
1032            missing_blocks_per_authority[*authority] += blocks.len();
1033        }
1034        for (missing, (_, authority)) in missing_blocks_per_authority
1035            .into_iter()
1036            .zip(context.committee.authorities())
1037        {
1038            context
1039                .metrics
1040                .node_metrics
1041                .synchronizer_missing_blocks_by_authority
1042                .with_label_values(&[&authority.hostname])
1043                .inc_by(missing as u64);
1044            context
1045                .metrics
1046                .node_metrics
1047                .synchronizer_current_missing_blocks_by_authority
1048                .with_label_values(&[&authority.hostname])
1049                .set(missing as i64);
1050        }
1051
1052        let mut peers = context
1053            .committee
1054            .authorities()
1055            .filter_map(|(peer_index, _)| (peer_index != context.own_index).then_some(peer_index))
1056            .collect::<Vec<_>>();
1057
1058        // TODO: probably inject the RNG to allow unit testing - this is a work around for now.
1059        if cfg!(not(test)) {
1060            // Shuffle the peers
1061            peers.shuffle(&mut ThreadRng::default());
1062        }
1063
1064        let mut peers = peers.into_iter();
1065        let mut request_futures = FuturesUnordered::new();
1066
1067        let highest_rounds = Self::get_highest_accepted_rounds(dag_state, &context);
1068
1069        // Shuffle the authorities for each request.
1070        let mut authorities = authorities.into_values().collect::<Vec<_>>();
1071        if cfg!(not(test)) {
1072            // Shuffle the authorities
1073            authorities.shuffle(&mut ThreadRng::default());
1074        }
1075
1076        // Send the fetch requests
1077        for batch in authorities.chunks(num_authorities_per_peer) {
1078            let Some(peer) = peers.next() else {
1079                debug_fatal!("No more peers left to fetch blocks!");
1080                break;
1081            };
1082            let peer_hostname = &context.committee.authority(peer).hostname;
1083            // Fetch from the lowest round missing blocks to ensure progress.
1084            // This may reduce efficiency and increase the chance of duplicated data transfer in edge cases.
1085            let block_refs = batch
1086                .iter()
1087                .flatten()
1088                .cloned()
1089                .collect::<BTreeSet<_>>()
1090                .into_iter()
1091                .take(context.parameters.max_blocks_per_sync)
1092                .collect::<BTreeSet<_>>();
1093
1094            // lock the blocks to be fetched. If no lock can be acquired for any of the blocks then don't bother
1095            if let Some(blocks_guard) = inflight_blocks.lock_blocks(block_refs.clone(), peer) {
1096                info!(
1097                    "Periodic sync of {} missing blocks from peer {} {}: {}",
1098                    block_refs.len(),
1099                    peer,
1100                    peer_hostname,
1101                    block_refs
1102                        .iter()
1103                        .map(|b| b.to_string())
1104                        .collect::<Vec<_>>()
1105                        .join(", ")
1106                );
1107                request_futures.push(Self::fetch_blocks_request(
1108                    network_client.clone(),
1109                    peer,
1110                    blocks_guard,
1111                    highest_rounds.clone(),
1112                    false,
1113                    FETCH_REQUEST_TIMEOUT,
1114                    1,
1115                ));
1116            }
1117        }
1118
1119        let mut results = Vec::new();
1120        let fetcher_timeout = sleep(FETCH_FROM_PEERS_TIMEOUT);
1121
1122        tokio::pin!(fetcher_timeout);
1123
1124        loop {
1125            tokio::select! {
1126                Some((response, blocks_guard, _retries, peer_index, highest_rounds)) = request_futures.next() => {
1127                    let peer_hostname = &context.committee.authority(peer_index).hostname;
1128                    match response {
1129                        Ok(fetched_blocks) => {
1130                            results.push((blocks_guard, fetched_blocks, peer_index));
1131
1132                            // no more pending requests are left, just break the loop
1133                            if request_futures.is_empty() {
1134                                break;
1135                            }
1136                        },
1137                        Err(_) => {
1138                            context.metrics.node_metrics.synchronizer_fetch_failures.with_label_values(&[peer_hostname.as_str(), "periodic"]).inc();
1139                            // try again if there is any peer left
1140                            if let Some(next_peer) = peers.next() {
1141                                // do best effort to lock guards. If we can't lock then don't bother at this run.
1142                                if let Some(blocks_guard) = inflight_blocks.swap_locks(blocks_guard, next_peer) {
1143                                    info!(
1144                                        "Retrying syncing {} missing blocks from peer {}: {}",
1145                                        blocks_guard.block_refs.len(),
1146                                        peer_hostname,
1147                                        blocks_guard.block_refs
1148                                            .iter()
1149                                            .map(|b| b.to_string())
1150                                            .collect::<Vec<_>>()
1151                                            .join(", ")
1152                                    );
1153                                    request_futures.push(Self::fetch_blocks_request(
1154                                        network_client.clone(),
1155                                        next_peer,
1156                                        blocks_guard,
1157                                        highest_rounds,
1158                                        false,
1159                                        FETCH_REQUEST_TIMEOUT,
1160                                        1,
1161                                    ));
1162                                } else {
1163                                    debug!("Couldn't acquire locks to fetch blocks from peer {next_peer}.")
1164                                }
1165                            } else {
1166                                debug!("No more peers left to fetch blocks");
1167                            }
1168                        }
1169                    }
1170                },
1171                _ = &mut fetcher_timeout => {
1172                    debug!("Timed out while fetching missing blocks");
1173                    break;
1174                }
1175            }
1176        }
1177
1178        results
1179    }
1180}

1909        let (context, _) = Context::new_for_test(4);
1910        let context = Arc::new(context);
1911        let block_verifier = Arc::new(NoopBlockVerifier {});
1912        let core_dispatcher = Arc::new(MockCoreThreadDispatcher::default());
1913        let (blocks_sender, _blocks_receiver) =
1914            monitored_mpsc::unbounded_channel("consensus_block_output");
1915        let commit_vote_monitor = Arc::new(CommitVoteMonitor::new(context.clone()));
1916        let store = Arc::new(MemStore::new());
1917        let dag_state = Arc::new(RwLock::new(DagState::new(context.clone(), store)));
1918        let transaction_certifier = TransactionCertifier::new(
1919            context.clone(),
1920            block_verifier.clone(),
1921            dag_state.clone(),
1922            blocks_sender,
1923        );
1924        let round_tracker = Arc::new(RwLock::new(RoundTracker::new(context.clone(), vec![])));
1925        let (commands_sender, _commands_receiver) =
1926            monitored_mpsc::channel("consensus_synchronizer_commands", 1000);
1927
1928        // Create input test blocks:
1929        // - Authority 0 block at round 60.
1930        // - Authority 1 blocks from round 30 to 60.
1931        let mut expected_blocks = vec![VerifiedBlock::new_for_test(TestBlock::new(60, 0).build())];
1932        expected_blocks.extend(
1933            (30..=60).map(|round| VerifiedBlock::new_for_test(TestBlock::new(round, 1).build())),
1934        );
1935        assert_eq!(
1936            expected_blocks.len(),
1937            context.parameters.max_blocks_per_sync
1938        );
1939
1940        let expected_serialized_blocks = expected_blocks
1941            .iter()
1942            .map(|b| b.serialized().clone())
1943            .collect::<Vec<_>>();
1944
1945        let expected_block_refs = expected_blocks
1946            .iter()
1947            .map(|b| b.reference())
1948            .collect::<BTreeSet<_>>();
1949
1950        // GIVEN peer to fetch blocks from
1951        let peer_index = AuthorityIndex::new_for_test(2);
1952
1953        // Create blocks_guard
1954        let inflight_blocks_map = InflightBlocksMap::new();
1955        let blocks_guard = inflight_blocks_map
1956            .lock_blocks(expected_block_refs.clone(), peer_index)
1957            .expect("Failed to lock blocks");
1958
1959        assert_eq!(
1960            inflight_blocks_map.num_of_locked_blocks(),
1961            expected_block_refs.len()
1962        );
1963
1964        // Create a Synchronizer
1965        let result = Synchronizer::<
1966            MockNetworkClient,
1967            NoopBlockVerifier,
1968            MockCoreThreadDispatcher,
1969        >::process_fetched_blocks(
1970            expected_serialized_blocks,
1971            peer_index,
1972            blocks_guard, // The guard is consumed here
1973            core_dispatcher.clone(),
1974            block_verifier,
1975            transaction_certifier,
1976            commit_vote_monitor,
1977            context.clone(),
1978            commands_sender,
1979            round_tracker,
1980            "test",
1981        )
1982        .await;

1990            added_blocks

1992                .map(|b| b.reference())
1993                .collect::<BTreeSet<_>>(),
1994            expected_block_refs,

1997        // Check blocks were unlocked
1998        assert_eq!(inflight_blocks_map.num_of_locked_blocks(), 0);
