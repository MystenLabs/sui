75// Keeps a mapping between the missing blocks that have been instructed to be fetched and the peers
76// that are currently fetching them. For a block ref there is a maximum number of peers that can
77// concurrently fetch it. The peer ids that are currently fetching a block are set on the corresponding
78// `BTreeSet` and basically they act as "locks".
79struct InflightBlocksMap {
80    inner: Mutex<BTreeMap<BlockRef, BTreeSet<PeerId>>>,
81}
82
83impl InflightBlocksMap {
84    fn new() -> Arc<Self> {
85        Arc::new(Self {
86            inner: Mutex::new(BTreeMap::new()),
87        })
88    }
89
90    /// Locks the blocks to be fetched for the assigned `peer`. We want to avoid re-fetching the
91    /// missing blocks from too many peers at the same time, thus we limit the concurrency
92    /// per block by attempting to lock per block. If a block is already fetched by the maximum allowed
93    /// number of peers, then the block ref will not be included in the returned set. The method
94    /// returns all the block refs that have been successfully locked and allowed to be fetched.
95    fn lock_blocks(
96        self: &Arc<Self>,
97        missing_block_refs: BTreeSet<BlockRef>,
98        peer: PeerId,
99    ) -> Option<BlocksGuard> {
100        let mut blocks = BTreeSet::new();
101        let mut inner = self.inner.lock();
102
103        for block_ref in missing_block_refs {
104            // check that the number of peers that are already instructed to fetch the block is not
105            // higher than the allowed and the `peer` has not already been instructed to do that.
106            let peers = inner.entry(block_ref).or_default();
107            if peers.len() < MAX_AUTHORITIES_TO_FETCH_PER_BLOCK && peers.get(&peer).is_none() {
108                assert!(peers.insert(peer.clone()));
109                blocks.insert(block_ref);
110            }
111        }
112
113        if blocks.is_empty() {
114            None
115        } else {
116            Some(BlocksGuard {
117                map: self.clone(),
118                block_refs: blocks,
119                peer,
120            })
121        }
122    }
123
124    /// Unlocks the provided block references for the given `peer`. The unlocking is strict, meaning that
125    /// if this method is called for a specific block ref and peer more times than the corresponding lock
126    /// has been called, it will panic.
127    fn unlock_blocks(self: &Arc<Self>, block_refs: &BTreeSet<BlockRef>, peer: PeerId) {
128        // Now mark all the blocks as fetched from the map
129        let mut blocks_to_fetch = self.inner.lock();
130        for block_ref in block_refs {
131            let peers = blocks_to_fetch
132                .get_mut(block_ref)
133                .expect("Should have found a non empty map");
134
135            assert!(peers.remove(&peer), "Peer should be present!");
136
137            // if the last one then just clean up
138            if peers.is_empty() {
139                blocks_to_fetch.remove(block_ref);
140            }
141        }
142    }
143
144    /// Drops the provided `blocks_guard` which will force to unlock the blocks, and lock now again the
145    /// referenced block refs. The swap is best effort and there is no guarantee that the `peer` will
146    /// be able to acquire the new locks.
147    fn swap_locks(
148        self: &Arc<Self>,
149        blocks_guard: BlocksGuard,
150        peer: PeerId,
151    ) -> Option<BlocksGuard> {
152        let block_refs = blocks_guard.block_refs.clone();
153
154        // Explicitly drop the guard
155        drop(blocks_guard);
156
157        // Now create new guard
158        self.lock_blocks(block_refs, peer)
159    }
160
161    #[cfg(test)]
162    fn num_of_locked_blocks(self: &Arc<Self>) -> usize {
163        let inner = self.inner.lock();
164        inner.len()
165    }
166}

232/// `Synchronizer` oversees live block synchronization, crucial for node progress. Live synchronization
233/// refers to the process of retrieving missing blocks, particularly those essential for advancing a node
234/// when data from only a few rounds is absent. If a node significantly lags behind the network,
235/// `commit_syncer` handles fetching missing blocks via a more efficient approach. `Synchronizer`
236/// aims for swift catch-up employing two mechanisms:
237///
238/// 1. Explicitly requesting missing blocks from designated authorities via the "block send" path.
239///    This includes attempting to fetch any missing ancestors necessary for processing a received block.
240///    Such requests prioritize the block author, maximizing the chance of prompt retrieval.
241///    A locking mechanism allows concurrent requests for missing blocks from up to two authorities
242///    simultaneously, enhancing the chances of timely retrieval. Notably, if additional missing blocks
243///    arise during block processing, requests to the same authority are deferred to the scheduler.
244///
245/// 2. Periodically requesting missing blocks via a scheduler. This primarily serves to retrieve
246///    missing blocks that were not ancestors of a received block via the "block send" path.
247///    The scheduler operates on either a fixed periodic basis or is triggered immediately
248///    after explicit fetches described in (1), ensuring continued block retrieval if gaps persist.
249///
250/// Additionally to the above, the synchronizer can synchronize and fetch the last own proposed block
251/// from the network peers as best effort approach to recover node from amnesia and avoid making the
252/// node equivocate.
253pub(crate) struct Synchronizer<
254    V: BlockVerifier,
255    D: CoreThreadDispatcher,
256    VC: ValidatorNetworkClient,
257    OC: ObserverNetworkClient,
258> {
259    context: Arc<Context>,
260    commands_receiver: Receiver<Command>,
261    fetch_block_senders: BTreeMap<PeerId, Sender<BlocksGuard>>,
262    core_dispatcher: Arc<D>,
263    commit_vote_monitor: Arc<CommitVoteMonitor>,
264    dag_state: Arc<RwLock<DagState>>,
265    fetch_blocks_scheduler_task: JoinSet<()>,
266    fetch_own_last_block_task: JoinSet<()>,
267    network_client: Arc<SynchronizerClient<VC, OC>>,
268    block_verifier: Arc<V>,
269    transaction_vote_tracker: TransactionVoteTracker,
270    round_tracker: Arc<RwLock<RoundTracker>>,
271    inflight_blocks_map: Arc<InflightBlocksMap>,
272    commands_sender: Sender<Command>,
273    last_changed_commit_index: CommitIndex,
274    last_commit_change_time: Instant,
275    // When commit is not progressing, commit sync fails over to periodic sync for catchup.
276    commit_sync_failover: bool,
277    peers_pool: Arc<PeersPool>,
278}

287    pub(crate) fn start(
288        network_client: Arc<SynchronizerClient<VC, OC>>,
289        context: Arc<Context>,
290        core_dispatcher: Arc<D>,
291        commit_vote_monitor: Arc<CommitVoteMonitor>,
292        block_verifier: Arc<V>,
293        transaction_vote_tracker: TransactionVoteTracker,
294        round_tracker: Arc<RwLock<RoundTracker>>,
295        dag_state: Arc<RwLock<DagState>>,
296        peers_pool: Arc<PeersPool>,
297        sync_last_known_own_block: bool,
298    ) -> Arc<SynchronizerHandle> {
299        let (commands_sender, commands_receiver) =
300            channel("consensus_synchronizer_commands", 1_000);
301        let inflight_blocks_map = InflightBlocksMap::new();
302
303        // Spawn the tasks to fetch the blocks from the others
304        let mut fetch_block_senders = BTreeMap::new();
305        let mut tasks = JoinSet::new();
306
307        // Create fetch tasks for all known peers (validators and observers)
308        // TODO: refactor to update the sender tasks based on the registered/removed pool peers.
309        let known_peers = peers_pool.get_known_peers();
310        for peer in known_peers {
311            let (sender, receiver) =
312                channel("consensus_synchronizer_fetches", FETCH_BLOCKS_CONCURRENCY);
313            let fetch_blocks_from_peer_async = Self::fetch_blocks_from_peer(
314                peer.clone(),
315                network_client.clone(),
316                block_verifier.clone(),
317                transaction_vote_tracker.clone(),
318                commit_vote_monitor.clone(),
319                context.clone(),
320                core_dispatcher.clone(),
321                dag_state.clone(),
322                receiver,
323                commands_sender.clone(),
324                round_tracker.clone(),
325                peers_pool.clone(),
326            );
327            tasks.spawn(monitored_future!(fetch_blocks_from_peer_async));
328            fetch_block_senders.insert(peer, sender);
329        }
330
331        let commands_sender_clone = commands_sender.clone();
332
333        if sync_last_known_own_block {
334            commands_sender
335                .try_send(Command::FetchOwnLastBlock)
336                .expect("Failed to sync our last block");
337        }
338
339        // Spawn the task to listen to the requests & periodic runs
340        tasks.spawn(monitored_future!(async move {
341            let mut s = Self {
342                context,
343                commands_receiver,
344                fetch_block_senders,
345                core_dispatcher,
346                commit_vote_monitor,
347                fetch_blocks_scheduler_task: JoinSet::new(),
348                fetch_own_last_block_task: JoinSet::new(),
349                network_client,
350                block_verifier,
351                transaction_vote_tracker,
352                inflight_blocks_map,
353                commands_sender: commands_sender_clone,
354                dag_state,
355                round_tracker,
356                last_changed_commit_index: 0,
357                last_commit_change_time: Instant::now(),
358                commit_sync_failover: false,
359                peers_pool,
360            };
361            s.run().await;
362        }));
363
364        Arc::new(SynchronizerHandle {
365            commands_sender,
366            tasks: tokio::sync::Mutex::new(tasks),
367        })
368    }

370    // The main loop to listen for the submitted commands.
371    async fn run(&mut self) {
372        // We want the synchronizer to run periodically every 200ms to fetch any missing blocks.
373        const PERIODIC_FETCH_INTERVAL: Duration = Duration::from_millis(200);
374        let scheduler_timeout = sleep_until(Instant::now() + PERIODIC_FETCH_INTERVAL);
375
376        tokio::pin!(scheduler_timeout);
377
378        loop {
379            tokio::select! {
380                Some(command) = self.commands_receiver.recv() => {
381                    match command {
382                        Command::FetchBlocks{ missing_block_refs, peer, result } => {
383                            // Check if peer is available. This check also makes sure that we are not trying to fetch from ourselves.
384                            if !self.peers_pool.is_peer_known(&peer) {
385                                result.send(Err(ConsensusError::PeerUnavailable(format!("{:?}", peer)))).ok();
386                                continue;
387                            }
388
389                            // Keep only the max allowed blocks to request. Additional missing blocks
390                            // will be fetched via periodic sync.
391                            // Fetch from the lowest to highest round, to ensure progress.
392                            let missing_block_refs = missing_block_refs
393                                .into_iter()
394                                .take(self.context.parameters.max_blocks_per_sync)
395                                .collect();
396
397                            let blocks_guard = self.inflight_blocks_map.lock_blocks(missing_block_refs, peer.clone());
398                            let Some(blocks_guard) = blocks_guard else {
399                                result.send(Ok(())).ok();
400                                continue;
401                            };
402
403                            // We don't block if the corresponding peer task is saturated - but we rather drop the request. That's ok as the periodic
404                            // synchronization task will handle any still missing blocks in next run.
405                            let r = self
406                                .fetch_block_senders
407                                .get(&peer)
408                                .ok_or(ConsensusError::PeerNotFound(format!("Peer {} not found in fetch_block_senders", peer)))
409                                .and_then(|sender| {
410                                    sender
411                                        .try_send(blocks_guard)
412                                        .map_err(|err| {
413                                            match err {
414                                                TrySendError::Full(_) => {
415                                                    let peer_name = peer.labelname(&self.context);
416                                                    self.context
417                                                        .metrics
418                                                        .node_metrics
419                                                        .synchronizer_skipped_fetch_requests
420                                                        .with_label_values(&[peer_name])
421                                                        .inc();
422                                                    ConsensusError::SynchronizerSaturated(format!("{:?}", peer))
423                                                },
424                                                TrySendError::Closed(_) => ConsensusError::Shutdown
425                                            }
426                                        })
427                                });
428
429                            result.send(r).ok();
430                        }
431                        Command::FetchOwnLastBlock => {
432                            if self.fetch_own_last_block_task.is_empty() {
433                                self.start_fetch_own_last_block_task();
434                            }
435                        }
436                        Command::KickOffScheduler => {
437                            // just reset the scheduler timeout timer to run immediately if not already running.
438                            // If the scheduler is already running then just reduce the remaining time to run.
439                            let timeout = if self.fetch_blocks_scheduler_task.is_empty() {
440                                Instant::now()
441                            } else {
442                                Instant::now() + PERIODIC_FETCH_INTERVAL.checked_div(2).unwrap()
443                            };
444
445                            // only reset if it is earlier than the next deadline
446                            if timeout < scheduler_timeout.deadline() {
447                                scheduler_timeout.as_mut().reset(timeout);
448                            }
449                        }
450                        Command::Shutdown { result } => {
451                            self.shutdown_tasks().await;
452                            self.fetch_block_senders.clear();
453                            let _ = result.send(());
454                            return;
455                        }
456                    }
457                },
458                Some(result) = self.fetch_own_last_block_task.join_next(), if !self.fetch_own_last_block_task.is_empty() => {
459                    match result {
460                        Ok(()) => {},
461                        Err(e) => {
462                            if e.is_cancelled() {
463                            } else if e.is_panic() {
464                                std::panic::resume_unwind(e.into_panic());
465                            } else {
466                                panic!("fetch our last block task failed: {e}");
467                            }
468                        },
469                    };
470                },
471                Some(result) = self.fetch_blocks_scheduler_task.join_next(), if !self.fetch_blocks_scheduler_task.is_empty() => {
472                    match result {
473                        Ok(()) => {},
474                        Err(e) => {
475                            if e.is_cancelled() {
476                            } else if e.is_panic() {
477                                std::panic::resume_unwind(e.into_panic());
478                            } else {
479                                panic!("fetch blocks scheduler task failed: {e}");
480                            }
481                        },
482                    };
483                },
484                () = &mut scheduler_timeout => {
485                    // we want to start a new task only if the previous one has already finished.
486                    // TODO: consider starting backup fetches in parallel, when a fetch takes too long?
487                    if self.fetch_blocks_scheduler_task.is_empty()
488                        && let Err(err) = self.start_fetch_missing_blocks_task().await {
489                            debug!("Core is shutting down, synchronizer is shutting down: {err:?}");
490                            self.shutdown_tasks().await;
491                            return;
492                        };
493
494                    scheduler_timeout
495                        .as_mut()
496                        .reset(Instant::now() + PERIODIC_FETCH_INTERVAL);
497                }
498            }
499        }
500    }

509    async fn fetch_blocks_from_peer(
510        peer: PeerId,
511        network_client: Arc<SynchronizerClient<VC, OC>>,
512        block_verifier: Arc<V>,
513        transaction_vote_tracker: TransactionVoteTracker,
514        commit_vote_monitor: Arc<CommitVoteMonitor>,
515        context: Arc<Context>,
516        core_dispatcher: Arc<D>,
517        dag_state: Arc<RwLock<DagState>>,
518        mut receiver: Receiver<BlocksGuard>,
519        commands_sender: Sender<Command>,
520        round_tracker: Arc<RwLock<RoundTracker>>,
521        _peers_pool: Arc<PeersPool>,
522    ) {
523        const MAX_RETRIES: u32 = 3;
524        let mut requests = FuturesUnordered::new();
525
526        loop {
527            tokio::select! {
528                Some(blocks_guard) = receiver.recv(), if requests.len() < FETCH_BLOCKS_CONCURRENCY => {
529                    let fetch_after_rounds = Self::get_fetch_after_rounds(&context, dag_state.clone());
530
531                    requests.push(Self::fetch_blocks_request(network_client.clone(), peer.clone(), blocks_guard, fetch_after_rounds, true, FETCH_REQUEST_TIMEOUT, 1))
532                },
533                Some((response, blocks_guard, retries, _peer, fetch_after_rounds)) = requests.next() => {
534                    match response {
535                        Ok(blocks) => {
536                            if let Err(err) = Self::process_fetched_blocks(blocks,
537                                peer.clone(),
538                                blocks_guard,
539                                core_dispatcher.clone(),
540                                block_verifier.clone(),
541                                transaction_vote_tracker.clone(),
542                                commit_vote_monitor.clone(),
543                                context.clone(),
544                                commands_sender.clone(),
545                                round_tracker.clone(),
546                                "live"
547                            ).await {
548                                warn!("Error while processing fetched blocks from peer {}: {err}", peer.hostname(&context));
549                                context.metrics.node_metrics.synchronizer_process_fetched_failures.with_label_values(&[peer.labelname(&context).as_str(), "live"]).inc();
550                            }
551                        },
552                        Err(_) => {
553                            context.metrics.node_metrics.synchronizer_fetch_failures.with_label_values(&[peer.labelname(&context).as_str(), "live"]).inc();
554                            if retries <= MAX_RETRIES {
555                                requests.push(Self::fetch_blocks_request(network_client.clone(), peer.clone(), blocks_guard, fetch_after_rounds, true, FETCH_REQUEST_TIMEOUT, retries))
556                            } else {
557                                warn!("Max retries {retries} reached while trying to fetch blocks from peer {}.", peer.hostname(&context));
558                                // we don't necessarily need to do, but dropping the guard here to unlock the blocks
559                                drop(blocks_guard);
560                            }
561                        }
562                    }
563                },
564                else => {
565                    info!("Fetching blocks from peer {} task will now abort.", peer.hostname(&context));
566                    break;
567                }
568            }
569        }
570    }

572    /// Processes the requested raw fetched blocks from peer. If no error is returned then
573    /// the verified blocks are immediately sent to Core for processing.
574    async fn process_fetched_blocks(
575        mut serialized_blocks: Vec<Bytes>,
576        peer: PeerId,
577        requested_blocks_guard: BlocksGuard,
578        core_dispatcher: Arc<D>,
579        block_verifier: Arc<V>,
580        transaction_vote_tracker: TransactionVoteTracker,
581        commit_vote_monitor: Arc<CommitVoteMonitor>,
582        context: Arc<Context>,
583        commands_sender: Sender<Command>,
584        round_tracker: Arc<RwLock<RoundTracker>>,
585        sync_method: &str,
586    ) -> ConsensusResult<()> {
587        if serialized_blocks.is_empty() {
588            return Ok(());
589        }
590
591        // Limit the number of the returned blocks processed.
592        serialized_blocks.truncate(context.parameters.max_blocks_per_sync);
593
594        // Verify all the fetched blocks
595        let (blocks, voted_blocks) = Handle::current()
596            .spawn_blocking({
597                let block_verifier = block_verifier.clone();
598                let context = context.clone();
599                let peer = peer.clone();
600                move || Self::verify_blocks(serialized_blocks, block_verifier, &context, peer)
601            })
602            .await
603            .expect("Spawn blocking should not fail")?;
604
605        if context.protocol_config.transaction_voting_enabled() {
606            transaction_vote_tracker.add_voted_blocks(voted_blocks);
607        }
608
609        // Record commit votes from the verified blocks.
610        for block in &blocks {
611            commit_vote_monitor.observe_block(block);
612        }
613
614        // Update round tracker from the verified blocks. For fetched blocks,
615        // excluded_ancestors are not available so we use an empty vector.
616        {
617            let mut tracker = round_tracker.write();
618            for block in &blocks {
619                tracker.update_from_verified_block(&ExtendedBlock {
620                    block: block.clone(),
621                    excluded_ancestors: vec![],
622                });
623            }
624        }
625
626        let metrics = &context.metrics.node_metrics;
627        metrics
628            .synchronizer_fetched_blocks_by_peer
629            .with_label_values(&[peer.labelname(&context).as_str(), sync_method])
630            .inc_by(blocks.len() as u64);
631        for block in &blocks {
632            let block_hostname = &context.committee.authority(block.author()).hostname;
633            metrics
634                .synchronizer_fetched_blocks_by_authority
635                .with_label_values(&[block_hostname.as_str(), sync_method])
636                .inc();
637        }
638
639        debug!(
640            "Synced {} missing blocks from peer {:?}: {}",
641            blocks.len(),
642            peer,
643            blocks.iter().map(|b| b.reference().to_string()).join(", "),
644        );
645
646        // Now send them to core for processing. Ignore the returned missing blocks as we don't want
647        // this mechanism to keep feedback looping on fetching more blocks. The periodic synchronization
648        // will take care of that.
649        let missing_blocks = core_dispatcher
650            .add_blocks(blocks)
651            .await
652            .map_err(|_| ConsensusError::Shutdown)?;
653
654        // now release all the locked blocks as they have been fetched, verified & processed
655        drop(requested_blocks_guard);
656
657        // kick off immediately the scheduled synchronizer
658        if !missing_blocks.is_empty() {
659            // do not block here, so we avoid any possible cycles.
660            if let Err(TrySendError::Full(_)) = commands_sender.try_send(Command::KickOffScheduler)
661            {
662                warn!("Commands channel is full")
663            }
664        }
665
666        context
667            .metrics
668            .node_metrics
669            .missing_blocks_after_fetch_total
670            .inc_by(missing_blocks.len() as u64);
671
672        Ok(())
673    }

676        context: &Arc<Context>,
677        dag_state: Arc<RwLock<DagState>>,
678    ) -> Vec<Round> {
679        let (blocks, gc_round) = {

692    }

694    fn verify_blocks(
695        serialized_blocks: Vec<Bytes>,
696        block_verifier: Arc<V>,
697        context: &Context,
698        peer: PeerId,
699    ) -> ConsensusResult<(
700        Vec<VerifiedBlock>,
701        Vec<(VerifiedBlock, Vec<TransactionIndex>)>,
702    )> {
703        let mut verified_blocks = Vec::new();
704        let mut voted_blocks = Vec::new();
705        for serialized_block in serialized_blocks {
706            let signed_block: SignedBlock =
707                bcs::from_bytes(&serialized_block).map_err(ConsensusError::MalformedBlock)?;
708
709            // TODO: cache received and verified block refs to avoid duplicated work.
710            let (verified_block, reject_txn_votes) = block_verifier
711                .verify_and_vote(signed_block, serialized_block)
712                .tap_err(|e| {
713                    let peer_label = peer.labelname(context);
714                    context
715                        .metrics
716                        .node_metrics
717                        .invalid_blocks
718                        .with_label_values(&[peer_label.as_str(), "synchronizer", e.clone().name()])
719                        .inc();
720                    info!("Invalid block received from {}: {}", peer, e);
721                })?;
722
723            // TODO: improve efficiency, maybe suspend and continue processing the block asynchronously.
724            let now = context.clock.timestamp_utc_ms();
725            let drift = verified_block.timestamp_ms().saturating_sub(now);
726            if drift > 0 {
727                let peer_hostname = &context
728                    .committee
729                    .authority(verified_block.author())
730                    .hostname;
731                context
732                    .metrics
733                    .node_metrics
734                    .block_timestamp_drift_ms
735                    .with_label_values(&[peer_hostname.as_str(), "synchronizer"])
736                    .inc_by(drift);
737
738                trace!(
739                    "Synced block {} timestamp {} is in the future (now={}).",
740                    verified_block.reference(),
741                    verified_block.timestamp_ms(),
742                    now
743                );
744            }
745
746            verified_blocks.push(verified_block.clone());
747            voted_blocks.push((verified_block, reject_txn_votes));
748        }
749
750        Ok((verified_blocks, voted_blocks))
751    }

753    async fn fetch_blocks_request(
754        network_client: Arc<SynchronizerClient<VC, OC>>,
755        peer: PeerId,
756        blocks_guard: BlocksGuard,
757        fetch_after_rounds: Vec<Round>,
758        fetch_missing_ancestors: bool,
759        request_timeout: Duration,
760        mut retries: u32,
761    ) -> (
762        ConsensusResult<Vec<Bytes>>,
763        BlocksGuard,
764        u32,
765        PeerId,
766        Vec<Round>,
767    ) {
768        let start = Instant::now();
769        let resp = timeout(
770            request_timeout,
771            network_client.fetch_blocks(
772                peer.clone(),
773                blocks_guard
774                    .block_refs
775                    .clone()
776                    .into_iter()
777                    .collect::<Vec<_>>(),
778                fetch_after_rounds.clone().into_iter().collect::<Vec<_>>(),
779                fetch_missing_ancestors,
780                request_timeout,
781            ),
782        )
783        .await;
784
785        fail_point_async!("consensus-delay");
786
787        let resp = match resp {
788            Ok(Err(err)) => {
789                // Add a delay before retrying - if that is needed. If request has timed out then eventually
790                // this will be a no-op.
791                sleep_until(start + request_timeout).await;
792                retries += 1;
793                Err(err)
794            } // network error
795            Err(err) => {
796                // timeout
797                sleep_until(start + request_timeout).await;
798                retries += 1;
799                Err(ConsensusError::NetworkRequestTimeout(err.to_string()))
800            }
801            Ok(result) => result,
802        };
803        (resp, blocks_guard, retries, peer, fetch_after_rounds)
804    }

806    fn start_fetch_own_last_block_task(&mut self) {
807        const FETCH_OWN_BLOCK_RETRY_DELAY: Duration = Duration::from_millis(1_000);
808        const MAX_RETRY_DELAY_STEP: Duration = Duration::from_millis(4_000);
809
810        let context = self.context.clone();
811        let dag_state = self.dag_state.clone();
812        let network_client = self.network_client.clone();
813        let block_verifier = self.block_verifier.clone();
814        let core_dispatcher = self.core_dispatcher.clone();
815
816        self.fetch_own_last_block_task
817            .spawn(monitored_future!(async move {
818                let _scope = monitored_scope("FetchOwnLastBlockTask");
819
820                let fetch_own_block = |authority_index: AuthorityIndex, fetch_own_block_delay: Duration| {
821                    let network_client_cloned = network_client.clone();
822                    let own_index = context.own_index;
823                    async move {
824                        sleep(fetch_own_block_delay).await;
825                        let r = network_client_cloned.fetch_latest_blocks(authority_index, vec![own_index], FETCH_REQUEST_TIMEOUT).await;
826                        (r, authority_index)
827                    }
828                };
829
830
831                let process_blocks = |blocks: Vec<Bytes>, authority_index: AuthorityIndex| -> ConsensusResult<Vec<VerifiedBlock>> {
832                    let mut result = Vec::new();
833                    for serialized_block in blocks {
834                        let signed_block = bcs::from_bytes(&serialized_block).map_err(ConsensusError::MalformedBlock)?;
835                        let (verified_block, _) = block_verifier.verify_and_vote(signed_block, serialized_block).tap_err(|err|{
836                            let hostname = context.committee.authority(authority_index).hostname.clone();
837                            context
838                                .metrics
839                                .node_metrics
840                                .invalid_blocks
841                                .with_label_values(&[hostname.as_str(), "synchronizer_own_block", err.clone().name()])
842                                .inc();
843                            warn!("Invalid block received from {}: {}", authority_index, err);
844                        })?;
845
846                        if verified_block.author() != context.own_index {
847                            return Err(ConsensusError::UnexpectedLastOwnBlock { index: authority_index, block_ref: verified_block.reference()});
848                        }
849                        result.push(verified_block);
850                    }
851                    Ok(result)
852                };
853
854                // Get the highest of all the results. Retry until at least `f+1` results have been gathered.
855                let mut highest_round;
856                let mut retries = 0;
857                let mut retry_delay_step = Duration::from_millis(500);
858                'main:loop {
859                    if context.committee.size() == 1 {
860                        highest_round = dag_state.read().get_last_proposed_block().expect("Last proposed block should be returned on validators").round();
861                        info!("Only one node in the network, will not try fetching own last block from peers.");
862                        break 'main;
863                    }
864
865                    let mut total_stake = 0;
866                    highest_round = 0;
867
868                    // Ask all the other peers about our last block
869                    let mut results = FuturesUnordered::new();
870
871                    for (authority_index, _authority) in context.committee.authorities() {
872                        if authority_index != context.own_index {
873                            results.push(fetch_own_block(authority_index, Duration::from_millis(0)));
874                        }
875                    }
876
877                    // Gather the results but wait to timeout as well
878                    let timer = sleep_until(Instant::now() + context.parameters.sync_last_known_own_block_timeout);
879                    tokio::pin!(timer);
880
881                    'inner: loop {
882                        tokio::select! {
883                            result = results.next() => {
884                                let Some((result, authority_index)) = result else {
885                                    break 'inner;
886                                };
887                                match result {
888                                    Ok(result) => {
889                                        match process_blocks(result, authority_index) {
890                                            Ok(blocks) => {
891                                                let max_round = blocks.into_iter().map(|b|b.round()).max().unwrap_or(0);
892                                                highest_round = highest_round.max(max_round);
893
894                                                total_stake += context.committee.stake(authority_index);
895                                            },
896                                            Err(err) => {
897                                                warn!("Invalid result returned from {authority_index} while fetching last own block: {err}");
898                                            }
899                                        }
900                                    },
901                                    Err(err) => {
902                                        warn!("Error {err} while fetching our own block from peer {authority_index}. Will retry.");
903                                        results.push(fetch_own_block(authority_index, FETCH_OWN_BLOCK_RETRY_DELAY));
904                                    }
905                                }
906                            },
907                            () = &mut timer => {
908                                info!("Timeout while trying to sync our own last block from peers");
909                                break 'inner;
910                            }
911                        }
912                    }
913
914                    // Request at least f+1 stake to have replied back.
915                    if context.committee.reached_validity(total_stake) {
916                        info!("{} out of {} total stake returned acceptable results for our own last block with highest round {}, with {retries} retries.", total_stake, context.committee.total_stake(), highest_round);
917                        break 'main;
918                    }
919
920                    retries += 1;
921                    context.metrics.node_metrics.sync_last_known_own_block_retries.inc();
922                    warn!("Not enough stake: {} out of {} total stake returned acceptable results for our own last block with highest round {}. Will now retry {retries}.", total_stake, context.committee.total_stake(), highest_round);
923
924                    sleep(retry_delay_step).await;
925
926                    retry_delay_step = Duration::from_secs_f64(retry_delay_step.as_secs_f64() * 1.5);
927                    retry_delay_step = retry_delay_step.min(MAX_RETRY_DELAY_STEP);
928                }
929
930                // Update the Core with the highest detected round
931                context.metrics.node_metrics.last_known_own_block_round.set(highest_round as i64);
932
933                if let Err(err) = core_dispatcher.set_last_known_proposed_round(highest_round) {
934                    warn!("Error received while calling dispatcher, probably dispatcher is shutting down, will now exit: {err:?}");
935                }
936            }));
937    }

939    async fn start_fetch_missing_blocks_task(&mut self) -> ConsensusResult<()> {
940        if self.context.committee.size() == 1 {
941            trace!(
942                "Only one node in the network, will not try fetching missing blocks from peers."
943            );
944            return Ok(());
945        }
946
947        // If commit is lagging and commit sync is making progress, skip periodic sync.
948        // Commit syncer fetches certified commits with all necessary causal history.
949        // If commit sync is not making progress, periodic sync resumes as a fallback.
950        if !self.should_run_periodic_sync() {
951            return Ok(());
952        }
953
954        let context = self.context.clone();
955        let network_client = self.network_client.clone();
956        let block_verifier = self.block_verifier.clone();
957        let transaction_vote_tracker = self.transaction_vote_tracker.clone();
958        let commit_vote_monitor = self.commit_vote_monitor.clone();
959        let core_dispatcher = self.core_dispatcher.clone();
960        let blocks_to_fetch = self.inflight_blocks_map.clone();
961        let commands_sender = self.commands_sender.clone();
962        let dag_state = self.dag_state.clone();
963        let round_tracker = self.round_tracker.clone();
964        let peers_pool = self.peers_pool.clone();
965
966        let mut missing_blocks = self
967            .core_dispatcher
968            .get_missing_blocks()
969            .await
970            .map_err(|_err| ConsensusError::Shutdown)?;
971        if self.commit_sync_failover {
972            // Keep missing blocks to those that must be included in fetch request.
973            // Filtered out missing blocks that will eventually be fetched with fetch_after_rounds.
974            let fetch_after_rounds = Self::get_fetch_after_rounds(&context, dag_state.clone());
975            missing_blocks.retain(|block| block.round <= fetch_after_rounds[block.author.value()]);
976        } else if missing_blocks.is_empty() {
977            return Ok(());
978        }
979
980        self.fetch_blocks_scheduler_task
981            .spawn(monitored_future!(async move {
982                let _scope = monitored_scope("FetchMissingBlocksScheduler");
983                context
984                    .metrics
985                    .node_metrics
986                    .fetch_blocks_scheduler_inflight
987                    .inc();
988                let total_requested = missing_blocks.len();
989
990                let results = if missing_blocks.is_empty() {
991                    let _scope = monitored_scope("BlockSync::Periodic::HighestAcceptedRounds");
992                    // Fetch blocks from a random peer using highest accepted rounds (commit sync failover)
993                    Self::fetch_blocks_with_fetch_after_rounds(
994                        context.clone(),
995                        blocks_to_fetch.clone(),
996                        network_client,
997                        dag_state,
998                        peers_pool,
999                    )
1000                    .await
1001                } else {
1002                    let _scope = monitored_scope("BlockSync::Periodic::MissingBlocks");
1003                    // Fetch blocks from 1 to MAX_PERIODIC_SYNC_PEERS peers
1004                    Self::fetch_blocks_from_peers(
1005                        context.clone(),
1006                        blocks_to_fetch.clone(),
1007                        network_client,
1008                        missing_blocks,
1009                        dag_state,
1010                        peers_pool,
1011                    )
1012                    .await
1013                };
1014                context
1015                    .metrics
1016                    .node_metrics
1017                    .fetch_blocks_scheduler_inflight
1018                    .dec();
1019                if results.is_empty() {
1020                    return;
1021                }
1022
1023                fail_point_async!("consensus-delay");
1024
1025                // Now process the returned results
1026                let mut total_fetched = 0;
1027                for (blocks_guard, fetched_blocks, peer) in results {
1028                    total_fetched += fetched_blocks.len();
1029
1030                    if let Err(err) = Self::process_fetched_blocks(
1031                        fetched_blocks,
1032                        peer.clone(),
1033                        blocks_guard,
1034                        core_dispatcher.clone(),
1035                        block_verifier.clone(),
1036                        transaction_vote_tracker.clone(),
1037                        commit_vote_monitor.clone(),
1038                        context.clone(),
1039                        commands_sender.clone(),
1040                        round_tracker.clone(),
1041                        "periodic",
1042                    )
1043                    .await
1044                    {
1045                        warn!(
1046                            "Error occurred while processing fetched blocks from peer {:?}: {err}",
1047                            peer
1048                        );
1049                        let peer_name = peer.labelname(&context);
1050                        context
1051                            .metrics
1052                            .node_metrics
1053                            .synchronizer_process_fetched_failures
1054                            .with_label_values(&[peer_name.as_str(), "periodic"])
1055                            .inc();
1056                    }
1057                }
1058
1059                debug!(
1060                    "Total blocks requested to fetch: {}, total fetched: {}",
1061                    total_requested, total_fetched
1062                );
1063            }));
1064
1065        Ok(())
1066    }

1068    fn should_run_periodic_sync(&mut self) -> bool {
1069        let current_commit_index = self.dag_state.read().last_commit_index();
1070        let quorum_commit_index = self.commit_vote_monitor.quorum_commit_index();
1071        let now = Instant::now();
1072        let metrics = &self.context.metrics.node_metrics;
1073
1074        // Commit is not lagging.
1075        if !is_commit_lagging(
1076            self.context.as_ref(),
1077            current_commit_index,
1078            quorum_commit_index,
1079        ) {
1080            metrics
1081                .synchronizer_periodic_sync_decision
1082                .with_label_values(&["true", "default"])
1083                .inc();
1084            // Reset last commit state.
1085            self.last_changed_commit_index = current_commit_index;
1086            self.last_commit_change_time = now;
1087            self.commit_sync_failover = false;
1088            // Run periodic sync.
1089            return true;
1090        }
1091        // Commit is lagging.
1092
1093        // When in commit sync failover, check if enough progress has been made.
1094        if self.commit_sync_failover {
1095            if current_commit_index
1096                < self.last_changed_commit_index + self.context.parameters.commit_sync_batch_size
1097            {
1098                metrics
1099                    .synchronizer_periodic_sync_decision
1100                    .with_label_values(&["true", "commit_catchup::run"])
1101                    .inc();
1102                // Not enough progress has been made yet. Keep running periodic sync.
1103                // Do not update last commit state yet.
1104                // Run periodic sync.
1105                return true;
1106            } else {
1107                metrics
1108                    .synchronizer_periodic_sync_decision
1109                    .with_label_values(&["false", "commit_catchup::end"])
1110                    .inc();
1111                // Enough progress has been made. Disable commit sync failover and reset last commit state.
1112                self.last_changed_commit_index = current_commit_index;
1113                self.last_commit_change_time = now;
1114                self.commit_sync_failover = false;
1115                // Skip periodic sync because of commit lag.
1116                return false;
1117            }
1118        }
1119
1120        // The node is commit lagging and not in commit sync failover yet.
1121        if current_commit_index == self.last_changed_commit_index {
1122            // Enter commit sync failover if not enough progress has been made.
1123            if now.duration_since(self.last_commit_change_time) >= COMMIT_PROGRESS_TIMEOUT {
1124                metrics
1125                    .synchronizer_periodic_sync_decision
1126                    .with_label_values(&["true", "commit_catchup::start"])
1127                    .inc();
1128                self.commit_sync_failover = true;
1129                // Run periodic sync.
1130                return true;
1131            }
1132        } else {
1133            // IMPORTANT: Only update last commit state when commit index is changing.
1134            self.last_changed_commit_index = current_commit_index;
1135            self.last_commit_change_time = now;
1136        }
1137
1138        metrics
1139            .synchronizer_periodic_sync_decision
1140            .with_label_values(&["false", "commit_lag"])
1141            .inc();
1142        false
1143    }

1203    /// Fetches the `missing_blocks` from peers. Requests the same number of authorities with missing blocks from each peer.
1204    /// Each response from peer can contain the requested blocks, and additional blocks from the last accepted round for
1205    /// authorities with missing blocks.
1206    /// Each element of the vector is a tuple which contains the requested missing block refs, the returned blocks and
1207    /// the peer.
1208    async fn fetch_blocks_from_peers(
1209        context: Arc<Context>,
1210        inflight_blocks: Arc<InflightBlocksMap>,
1211        network_client: Arc<SynchronizerClient<VC, OC>>,
1212        missing_blocks: BTreeSet<BlockRef>,
1213        dag_state: Arc<RwLock<DagState>>,
1214        peers_pool: Arc<PeersPool>,
1215    ) -> Vec<(BlocksGuard, Vec<Bytes>, PeerId)> {
1216        // Preliminary truncation of missing blocks to fetch. Since each peer can have different
1217        // number of missing blocks and the fetching is batched by peer, so keep more than max_blocks_per_fetch
1218        // per peer on average.
1219        let missing_blocks = missing_blocks
1220            .into_iter()
1221            .take(2 * MAX_PERIODIC_SYNC_PEERS * context.parameters.max_blocks_per_fetch)
1222            .collect::<Vec<_>>();
1223
1224        // Maps authorities to the missing blocks they have.
1225        let mut authorities = BTreeMap::<AuthorityIndex, Vec<BlockRef>>::new();
1226        for block_ref in &missing_blocks {
1227            authorities
1228                .entry(block_ref.author)
1229                .or_default()
1230                .push(*block_ref);
1231        }
1232
1233        // Get known peers from the PeersPool
1234        let mut peers = peers_pool.get_known_peers();
1235
1236        // Distribute the same number of authorities into each peer to sync.
1237        // Use the number of known peers from the pool, capped at MAX_PERIODIC_SYNC_PEERS
1238        // TODO: in the future it would be possible, temporarily, for an Observer node to not have peers to fetch from.
1239        // We should change this assertion to allow for this case.
1240        assert!(!peers.is_empty(), "No known peers to fetch blocks from");
1241
1242        let num_authorities_per_peer = authorities
1243            .len()
1244            .div_ceil(peers.len().min(MAX_PERIODIC_SYNC_PEERS));
1245
1246        // Update metrics related to missing blocks.
1247        let mut missing_blocks_per_authority = vec![0; context.committee.size()];
1248        for (authority, blocks) in &authorities {
1249            missing_blocks_per_authority[*authority] += blocks.len();
1250        }
1251        for (missing, (_, authority)) in missing_blocks_per_authority
1252            .into_iter()
1253            .zip_debug_eq(context.committee.authorities())
1254        {
1255            context
1256                .metrics
1257                .node_metrics
1258                .synchronizer_missing_blocks_by_authority
1259                .with_label_values(&[&authority.hostname])
1260                .inc_by(missing as u64);
1261            context
1262                .metrics
1263                .node_metrics
1264                .synchronizer_current_missing_blocks_by_authority
1265                .with_label_values(&[&authority.hostname])
1266                .set(missing as i64);
1267        }
1268
1269        // TODO: probably inject the RNG to allow unit testing - this is a work around for now.
1270        if cfg!(not(test)) {
1271            // Shuffle the peers
1272            peers.shuffle(&mut ThreadRng::default());
1273        }
1274
1275        let mut peers = peers.into_iter();
1276        let mut request_futures = FuturesUnordered::new();
1277
1278        // Shuffle the authorities for each request.
1279        let mut authorities = authorities.into_values().collect::<Vec<_>>();
1280        if cfg!(not(test)) {
1281            // Shuffle the authorities
1282            authorities.shuffle(&mut ThreadRng::default());
1283        }
1284
1285        let fetch_after_rounds = Self::get_fetch_after_rounds(&context, dag_state.clone());
1286
1287        // Send the fetch requests
1288        for batch in authorities.chunks(num_authorities_per_peer) {
1289            let Some(peer) = peers.next() else {
1290                debug_fatal!("No more peers left to fetch blocks!");
1291                break;
1292            };
1293            let peer_name = peer.hostname(&context);
1294            // Fetch from the lowest round missing blocks to ensure progress.
1295            // This may reduce efficiency and increase the chance of duplicated data transfer in edge cases.
1296            let block_refs = batch
1297                .iter()
1298                .flatten()
1299                .cloned()
1300                .collect::<BTreeSet<_>>()
1301                .into_iter()
1302                .take(context.parameters.max_blocks_per_fetch)
1303                .collect::<BTreeSet<_>>();
1304
1305            // lock the blocks to be fetched. If no lock can be acquired for any of the blocks then don't bother
1306            if let Some(blocks_guard) =
1307                inflight_blocks.lock_blocks(block_refs.clone(), peer.clone())
1308            {
1309                info!(
1310                    "Periodic sync of {} missing blocks from peer {} {:?}: {}",
1311                    peer_name.as_str(),
1312                    block_refs.len(),
1313                    peer,
1314                    block_refs
1315                        .iter()
1316                        .map(|b| b.to_string())
1317                        .collect::<Vec<_>>()
1318                        .join(", ")
1319                );
1320                request_futures.push(Self::fetch_blocks_request(
1321                    network_client.clone(),
1322                    peer,
1323                    blocks_guard,
1324                    fetch_after_rounds.clone(),
1325                    false,
1326                    FETCH_REQUEST_TIMEOUT,
1327                    1,
1328                ));
1329            }
1330        }
1331
1332        let mut results = Vec::new();
1333        let fetcher_timeout = sleep(FETCH_FROM_PEERS_TIMEOUT);
1334
1335        tokio::pin!(fetcher_timeout);
1336
1337        loop {
1338            tokio::select! {
1339                Some((response, blocks_guard, _retries, peer, fetch_after_rounds)) = request_futures.next() => {
1340                    match response {
1341                        Ok(fetched_blocks) => {
1342                            results.push((blocks_guard, fetched_blocks, peer));
1343
1344                            // no more pending requests are left, just break the loop
1345                            if request_futures.is_empty() {
1346                                break;
1347                            }
1348                        },
1349                        Err(_) => {
1350                            let peer_name = peer.labelname(&context);
1351                            context.metrics.node_metrics.synchronizer_fetch_failures.with_label_values(&[peer_name.as_str(), "periodic"]).inc();
1352                            // try again if there is any peer left
1353                            if let Some(next_peer) = peers.next() {
1354                                // do best effort to lock guards. If we can't lock then don't bother at this run.
1355                                if let Some(blocks_guard) = inflight_blocks.swap_locks(blocks_guard, next_peer.clone()) {
1356                                    info!(
1357                                        "Retrying syncing {} missing blocks from peer {:?}: {}",
1358                                        blocks_guard.block_refs.len(),
1359                                        next_peer,
1360                                        blocks_guard.block_refs
1361                                            .iter()
1362                                            .map(|b| b.to_string())
1363                                            .collect::<Vec<_>>()
1364                                            .join(", ")
1365                                    );
1366                                    request_futures.push(Self::fetch_blocks_request(
1367                                        network_client.clone(),
1368                                        next_peer,
1369                                        blocks_guard,
1370                                        fetch_after_rounds,
1371                                        false,
1372                                        FETCH_REQUEST_TIMEOUT,
1373                                        1,
1374                                    ));
1375                                } else {
1376                                    debug!("Couldn't acquire locks to fetch blocks from peer {:?}.", next_peer)
1377                                }
1378                            } else {
1379                                debug!("No more peers left to fetch blocks");
1380                            }
1381                        }
1382                    }
1383                },
1384                _ = &mut fetcher_timeout => {
1385                    debug!("Timed out while fetching missing blocks");
1386                    break;
1387                }
1388            }
1389        }
1390
1391        results
1392    }
1393}
