12#[derive(Error, Debug)]
13pub(crate) enum Error {
14    #[error("Path attempts to access parent of root directory: {}", .0.display())]
15    ParentOfRoot(PathBuf),
16
17    #[error("Unexpected symlink: {}", .0.display())]
18    Symlink(PathBuf),
19}

21/// Normalize the representation of `path` by eliminating redundant `.` components and applying `..`
22/// components.  Does not access the filesystem (e.g. to resolve symlinks or test for file
23/// existence), unlike `std::fs::canonicalize`.
24///
25/// Fails if the normalized path attempts to access the parent of a root directory or volume
26/// prefix.  Returns the normalized path on success.
27pub(crate) fn normalize_path<P: AsRef<Path>>(path: P) -> Result<PathBuf> {
28    use std::path::Component as C;
29
30    let mut stack = vec![];
31    for component in path.as_ref().components() {
32        match component {
33            // Components that contribute to the path as-is.
34            verbatim @ (C::Prefix(_) | C::RootDir | C::Normal(_)) => stack.push(verbatim),
35
36            // Equivalent of a `.` path component -- can be ignored.
37            C::CurDir => { /* nop */ }
38
39            // Going up in the directory hierarchy, which may fail if that's not possible.
40            C::ParentDir => match stack.last() {
41                None | Some(C::ParentDir) => {
42                    stack.push(C::ParentDir);
43                }
44
45                Some(C::Normal(_)) => {
46                    stack.pop();
47                }
48
49                Some(C::CurDir) => {
50                    unreachable!("Component::CurDir never added to the stack");
51                }
52
53                Some(C::RootDir | C::Prefix(_)) => {
54                    bail!(Error::ParentOfRoot(path.as_ref().to_path_buf()))
55                }
56            },
57        }
58    }
59
60    Ok(stack.iter().collect())
61}

63/// Return the path to `dst` relative to `src`.  If `src` is a file, the path is relative to the
64/// directory that contains it, while if it is a directory, the path is relative to it.  Returns
65/// an error if either `src` or `dst` do not exist.
66pub(crate) fn path_relative_to<P, Q>(src: P, dst: Q) -> io::Result<PathBuf>
67where
68    P: AsRef<Path>,
69    Q: AsRef<Path>,
70{
71    use std::path::Component as C;
72
73    let mut src = fs::canonicalize(src)?;
74    let dst = fs::canonicalize(dst)?;
75
76    if src.is_file() {
77        src.pop();
78    }
79
80    let mut s_comps = src.components().peekable();
81    let mut d_comps = dst.components().peekable();
82
83    // (1). Strip matching prefix
84    while let (Some(s_comp), Some(d_comp)) = (s_comps.peek(), d_comps.peek()) {
85        if s_comp != d_comp {
86            break;
87        }
88        s_comps.next();
89        d_comps.next();
90    }
91    // (2) Push parent directory components (moving out of directories in `base`)
92    let mut stack = vec![];
93    for _ in s_comps {
94        stack.push(C::ParentDir)
95    }
96
97    // (3) Push extension directory components (moving into directories unique to `ext`)
98    for comp in d_comps {
99        stack.push(comp)
100    }
101
102    // (4) Check for base == ext case
103    if stack.is_empty() {
104        stack.push(C::CurDir)
105    }
106
107    Ok(stack.into_iter().collect())
108}

110/// Returns the shortest prefix of `path` that doesn't exist, or `None` if `path` already exists.
111pub(crate) fn shortest_new_prefix(path: impl AsRef<Path>) -> Option<PathBuf> {
112    if path.as_ref().exists() {
113        return None;
114    }
115
116    let mut path = path.as_ref().to_owned();
117    let mut parent = path.clone();
118    parent.pop();
119
120    // Invariant: parent == { path.pop(); path }
121    //         && !path.exists()
122    while !parent.exists() {
123        parent.pop();
124        path.pop();
125    }
126
127    Some(path)
128}

130/// Recursively copy the contents of `src` to `dst`.  Fails if `src` transitively contains a
131/// symlink.  Only copies paths that pass the `keep` predicate.
132pub(crate) fn deep_copy<P, Q, K>(src: P, dst: Q, keep: &mut K) -> Result<()>
133where
134    P: AsRef<Path>,
135    Q: AsRef<Path>,
136    K: FnMut(&Path) -> bool,
137{
138    let src = src.as_ref();
139    let dst = dst.as_ref();
140
141    if !keep(src) {
142        return Ok(());
143    }
144
145    if src.is_file() {
146        fs::create_dir_all(dst.parent().expect("files have parents"))?;
147        fs::copy(src, dst)?;
148        return Ok(());
149    }
150
151    if src.is_symlink() {
152        bail!(Error::Symlink(src.to_path_buf()));
153    }
154
155    for entry in fs::read_dir(src)? {
156        let entry = entry?;
157        deep_copy(
158            src.join(entry.file_name()),
159            dst.join(entry.file_name()),
160            keep,
161        )?
162    }
163
164    Ok(())
165}

300    #[test]
301    fn test_deep_copy() {
302        let tmp = tempdir().unwrap();
303        let src = tmp.path().join("src");
304        let dst = tmp.path().join("dst");
305
306        // Set-up some things to copy:
307        //
308        // src/foo:         bar
309        // src/baz/qux/quy: plugh
310        // src/baz/quz:     xyzzy
311
312        fs::create_dir_all(src.join("baz/qux")).unwrap();
313        fs::write(src.join("foo"), "bar").unwrap();
314        fs::write(src.join("baz/qux/quy"), "plugh").unwrap();
315        fs::write(src.join("baz/quz"), "xyzzy").unwrap();
316
317        let read = |path: &str| fs::read_to_string(dst.join(path)).unwrap();
318
319        // Copy without filtering
320        deep_copy(&src, dst.join("cpy-0"), &mut |_| true).unwrap();
321
322        assert_eq!(read("cpy-0/foo"), "bar");
323        assert_eq!(read("cpy-0/baz/qux/quy"), "plugh");
324        assert_eq!(read("cpy-0/baz/quz"), "xyzzy");
325
326        // Filter a file
327        deep_copy(&src, dst.join("cpy-1"), &mut |p| !p.ends_with("foo")).unwrap();
328
329        assert!(!dst.join("cpy-1/foo").exists());
330        assert_eq!(read("cpy-1/baz/qux/quy"), "plugh");
331        assert_eq!(read("cpy-1/baz/quz"), "xyzzy");
332
333        // Filter a directory
334        deep_copy(&src, dst.join("cpy-2"), &mut |p| !p.ends_with("baz")).unwrap();
335
336        assert_eq!(read("cpy-2/foo"), "bar");
337        assert!(!dst.join("cpy-2/baz").exists());
338
339        // Filtering a file gets rid of its (empty) parent
340        deep_copy(&src, dst.join("cpy-3"), &mut |p| !p.ends_with("quy")).unwrap();
341
342        // Because qux is now empty, it also doesn't exist in the copy, even though we only
343        // explicitly filtered `quy`.
344        assert_eq!(read("cpy-3/foo"), "bar");
345        assert!(!dst.join("cpy-3/baz/qux").exists());
346        assert_eq!(read("cpy-3/baz/quz"), "xyzzy");
347    }
