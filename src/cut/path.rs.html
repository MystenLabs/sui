12#[derive(Error, Debug)]
13pub(crate) enum Error {
14    #[error("Path attempts to access parent of root directory: {}", .0.display())]
15    ParentOfRoot(PathBuf),
16
17    #[error("Unexpected symlink: {}", .0.display())]
18    Symlink(PathBuf),
19}

21/// Normalize the representation of `path` by eliminating redundant `.` components and applying `..`
22/// components.  Does not access the filesystem (e.g. to resolve symlinks or test for file
23/// existence), unlike `std::fs::canonicalize`.
24///
25/// Fails if the normalized path attempts to access the parent of a root directory or volume
26/// prefix.  Returns the normalized path on success.
27pub(crate) fn normalize_path<P: AsRef<Path>>(path: P) -> Result<PathBuf> {
28    use std::path::Component as C;
29
30    let mut stack = vec![];
31    for component in path.as_ref().components() {
32        match component {
33            // Components that contribute to the path as-is.
34            verbatim @ (C::Prefix(_) | C::RootDir | C::Normal(_)) => stack.push(verbatim),
35
36            // Equivalent of a `.` path component -- can be ignored.
37            C::CurDir => { /* nop */ }
38
39            // Going up in the directory hierarchy, which may fail if that's not possible.
40            C::ParentDir => match stack.last() {
41                None | Some(C::ParentDir) => {
42                    stack.push(C::ParentDir);
43                }
44
45                Some(C::Normal(_)) => {
46                    stack.pop();
47                }
48
49                Some(C::CurDir) => {
50                    unreachable!("Component::CurDir never added to the stack");
51                }
52
53                Some(C::RootDir | C::Prefix(_)) => {
54                    bail!(Error::ParentOfRoot(path.as_ref().to_path_buf()))
55                }
56            },
57        }
58    }
59
60    Ok(stack.iter().collect())
61}

63/// Return the path to `dst` relative to `src`.  If `src` is a file, the path is relative to the
64/// directory that contains it, while if it is a directory, the path is relative to it.  Returns
65/// an error if either `src` or `dst` do not exist.
66pub(crate) fn path_relative_to<P, Q>(src: P, dst: Q) -> io::Result<PathBuf>
67where
68    P: AsRef<Path>,
69    Q: AsRef<Path>,
70{
71    use std::path::Component as C;
72
73    let mut src = fs::canonicalize(src)?;
74    let dst = fs::canonicalize(dst)?;
75
76    if src.is_file() {
77        src.pop();
78    }
79
80    let mut s_comps = src.components().peekable();
81    let mut d_comps = dst.components().peekable();
82
83    // (1). Strip matching prefix
84    loop {
85        let Some(s_comp) = s_comps.peek() else { break };
86        let Some(d_comp) = d_comps.peek() else { break };
87        if s_comp != d_comp {
88            break;
89        }
90
91        s_comps.next();
92        d_comps.next();
93    }
94
95    // (2) Push parent directory components (moving out of directories in `base`)
96    let mut stack = vec![];
97    for _ in s_comps {
98        stack.push(C::ParentDir)
99    }
100
101    // (3) Push extension directory components (moving into directories unique to `ext`)
102    for comp in d_comps {
103        stack.push(comp)
104    }
105
106    // (4) Check for base == ext case
107    if stack.is_empty() {
108        stack.push(C::CurDir)
109    }
110
111    Ok(stack.into_iter().collect())
112}

114/// Returns the shortest prefix of `path` that doesn't exist, or `None` if `path` already exists.
115pub(crate) fn shortest_new_prefix(path: impl AsRef<Path>) -> Option<PathBuf> {
116    if path.as_ref().exists() {
117        return None;
118    }
119
120    let mut path = path.as_ref().to_owned();
121    let mut parent = path.clone();
122    parent.pop();
123
124    // Invariant: parent == { path.pop(); path }
125    //         && !path.exists()
126    while !parent.exists() {
127        parent.pop();
128        path.pop();
129    }
130
131    Some(path)
132}

134/// Recursively copy the contents of `src` to `dst`.  Fails if `src` transitively contains a
135/// symlink.  Only copies paths that pass the `keep` predicate.
136pub(crate) fn deep_copy<P, Q, K>(src: P, dst: Q, keep: &mut K) -> Result<()>
137where
138    P: AsRef<Path>,
139    Q: AsRef<Path>,
140    K: FnMut(&Path) -> bool,
141{
142    let src = src.as_ref();
143    let dst = dst.as_ref();
144
145    if !keep(src) {
146        return Ok(());
147    }
148
149    if src.is_file() {
150        fs::create_dir_all(dst.parent().expect("files have parents"))?;
151        fs::copy(src, dst)?;
152        return Ok(());
153    }
154
155    if src.is_symlink() {
156        bail!(Error::Symlink(src.to_path_buf()));
157    }
158
159    for entry in fs::read_dir(src)? {
160        let entry = entry?;
161        deep_copy(
162            src.join(entry.file_name()),
163            dst.join(entry.file_name()),
164            keep,
165        )?
166    }
167
168    Ok(())
169}

304    #[test]
305    fn test_deep_copy() {
306        let tmp = tempdir().unwrap();
307        let src = tmp.path().join("src");
308        let dst = tmp.path().join("dst");
309
310        // Set-up some things to copy:
311        //
312        // src/foo:         bar
313        // src/baz/qux/quy: plugh
314        // src/baz/quz:     xyzzy
315
316        fs::create_dir_all(src.join("baz/qux")).unwrap();
317        fs::write(src.join("foo"), "bar").unwrap();
318        fs::write(src.join("baz/qux/quy"), "plugh").unwrap();
319        fs::write(src.join("baz/quz"), "xyzzy").unwrap();
320
321        let read = |path: &str| fs::read_to_string(dst.join(path)).unwrap();
322
323        // Copy without filtering
324        deep_copy(&src, dst.join("cpy-0"), &mut |_| true).unwrap();
325
326        assert_eq!(read("cpy-0/foo"), "bar");
327        assert_eq!(read("cpy-0/baz/qux/quy"), "plugh");
328        assert_eq!(read("cpy-0/baz/quz"), "xyzzy");
329
330        // Filter a file
331        deep_copy(&src, dst.join("cpy-1"), &mut |p| !p.ends_with("foo")).unwrap();
332
333        assert!(!dst.join("cpy-1/foo").exists());
334        assert_eq!(read("cpy-1/baz/qux/quy"), "plugh");
335        assert_eq!(read("cpy-1/baz/quz"), "xyzzy");
336
337        // Filter a directory
338        deep_copy(&src, dst.join("cpy-2"), &mut |p| !p.ends_with("baz")).unwrap();
339
340        assert_eq!(read("cpy-2/foo"), "bar");
341        assert!(!dst.join("cpy-2/baz").exists());
342
343        // Filtering a file gets rid of its (empty) parent
344        deep_copy(&src, dst.join("cpy-3"), &mut |p| !p.ends_with("quy")).unwrap();
345
346        // Because qux is now empty, it also doesn't exist in the copy, even though we only
347        // explicitly filtered `quy`.
348        assert_eq!(read("cpy-3/foo"), "bar");
349        assert!(!dst.join("cpy-3/baz/qux").exists());
350        assert_eq!(read("cpy-3/baz/quz"), "xyzzy");
351    }
