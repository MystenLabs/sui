4use anyhow::Result;
5use clap::*;
6use mysten_metrics::start_prometheus_server;
7use std::env;
8use std::net::IpAddr;
9use std::net::{Ipv4Addr, SocketAddr};
10use std::path::PathBuf;
11use std::sync::Arc;
12use sui_config::Config;
13use sui_data_ingestion_core::DataIngestionMetrics;
14use sui_deepbook_indexer::config::IndexerConfig;
15use sui_deepbook_indexer::metrics::DeepBookIndexerMetrics;
16use sui_deepbook_indexer::postgres_manager::get_connection_pool;
17use sui_deepbook_indexer::server::run_server;
18use sui_deepbook_indexer::sui_deepbook_indexer::PgDeepbookPersistent;
19use sui_deepbook_indexer::sui_deepbook_indexer::SuiDeepBookDataMapper;
20use sui_indexer_builder::indexer_builder::IndexerBuilder;
21use sui_indexer_builder::progress::{OutOfOrderSaveAfterDurationPolicy, ProgressSavingPolicy};
22use sui_indexer_builder::sui_datasource::SuiCheckpointDatasource;
23use sui_sdk::SuiClientBuilder;
24use sui_types::base_types::ObjectID;
25use tracing::info;
26
27#[derive(Parser, Clone, Debug)]
28struct Args {
29    /// Path to a yaml config
30    #[clap(long, short)]
31    config_path: Option<PathBuf>,
32}
33
34#[tokio::main]
35async fn main() -> Result<()> {
36    let _guard = telemetry_subscribers::TelemetryConfig::new()
37        .with_env()
38        .init();
39
40    let args = Args::parse();
41
42    // load config
43    let config_path = if let Some(path) = args.config_path {
44        path
45    } else {
46        env::current_dir()
47            .expect("Couldn't get current directory")
48            .join("config.yaml")
49    };
50    let config = IndexerConfig::load(&config_path)?;
51
52    // Init metrics server
53    let metrics_address =
54        SocketAddr::new(IpAddr::V4(Ipv4Addr::new(0, 0, 0, 0)), config.metric_port);
55    let registry_service = start_prometheus_server(metrics_address);
56    let registry = registry_service.default_registry();
57    mysten_metrics::init_metrics(&registry);
58    info!("Metrics server started at port {}", config.metric_port);
59
60    let indexer_meterics = DeepBookIndexerMetrics::new(&registry);
61    let ingestion_metrics = DataIngestionMetrics::new(&registry);
62
63    let db_url = config.db_url.clone();
64    let datastore = PgDeepbookPersistent::new(
65        get_connection_pool(db_url.clone()).await,
66        ProgressSavingPolicy::OutOfOrderSaveAfterDuration(OutOfOrderSaveAfterDurationPolicy::new(
67            tokio::time::Duration::from_secs(30),
68        )),
69    );
70
71    let sui_client = Arc::new(
72        SuiClientBuilder::default()
73            .build(config.sui_rpc_url.clone())
74            .await?,
75    );
76    let sui_checkpoint_datasource = SuiCheckpointDatasource::new(
77        config.remote_store_url,
78        sui_client,
79        config.concurrency as usize,
80        config
81            .checkpoints_path
82            .map(|p| p.into())
83            .unwrap_or(tempfile::tempdir()?.keep()),
84        config.deepbook_genesis_checkpoint,
85        ingestion_metrics.clone(),
86        Box::new(indexer_meterics.clone()),
87    );
88
89    let service_address =
90        SocketAddr::new(IpAddr::V4(Ipv4Addr::new(0, 0, 0, 0)), config.service_port);
91    run_server(service_address, datastore.clone());
92
93    let indexer = IndexerBuilder::new(
94        "SuiDeepBookIndexer",
95        sui_checkpoint_datasource,
96        SuiDeepBookDataMapper {
97            metrics: indexer_meterics.clone(),
98            package_id: ObjectID::from_hex_literal(&config.deepbook_package_id.clone())
99                .unwrap_or_else(|err| panic!("Failed to parse deepbook package ID: {}", err)),
100        },
101        datastore,
102    )
103    .build();
104    indexer.start().await?;
105
106    Ok(())
107}
