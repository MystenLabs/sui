4use clap::*;
5use std::str::FromStr;
6use std::sync::Arc;
7use sui_storage::http_key_value_store::*;
8use sui_storage::key_value_store::TransactionKeyValueStore;
9use sui_storage::key_value_store_metrics::KeyValueStoreMetrics;
10use sui_types::base_types::ObjectID;
11use sui_types::digests::{CheckpointDigest, TransactionDigest};
12use sui_types::messages_checkpoint::CheckpointSequenceNumber;
13
14#[derive(Parser)]
15#[command(rename_all = "kebab-case")]
16enum Command {
17    Fetch {
18        // default value of 'https://transactions.sui.io/'
19        #[arg(short, long, default_value = "https://transactions.sui.io/mainnet")]
20        base_url: String,
21
22        #[arg(short, long)]
23        digest: Vec<String>,
24
25        #[arg(short, long)]
26        seq: Vec<String>,
27
28        // must be either 'tx', 'fx','ob','events', or 'ckpt_contents'
29        // default value of 'tx'
30        #[arg(short, long, default_value = "tx")]
31        type_: String,
32    },
33
34    DecodeKey {
35        #[arg(short, long)]
36        url: String,
37    },
38}
39
40impl Command {
41    async fn execute(self) -> anyhow::Result<(), anyhow::Error> {
42        match self {
43            Command::Fetch {
44                base_url,
45                digest,
46                seq,
47                type_,
48            } => {
49                let metrics = KeyValueStoreMetrics::new_for_tests();
50                let http_kv = Arc::new(HttpKVStore::new(&base_url, 100, metrics).unwrap());
51                let kv = TransactionKeyValueStore::new(
52                    "http_kv",
53                    KeyValueStoreMetrics::new_for_tests(),
54                    http_kv,
55                );
56
57                let seqs: Vec<_> = seq
58                    .into_iter()
59                    .map(|s| {
60                        CheckpointSequenceNumber::from_str(&s)
61                            .expect("invalid checkpoint sequence number")
62                    })
63                    .collect();
64
65                // verify that type is valid
66                match type_.as_str() {
67                    "tx" | "fx" => {
68                        let digests: Vec<_> = digest
69                            .into_iter()
70                            .map(|digest| {
71                                TransactionDigest::from_str(&digest)
72                                    .expect("invalid transaction digest")
73                            })
74                            .collect();
75
76                        if type_ == "tx" {
77                            let tx = kv.multi_get_tx(&digests).await.unwrap();
78                            for (digest, tx) in digests.iter().zip(tx.iter()) {
79                                println!("fetched tx: {:?} {:?}", digest, tx);
80                            }
81                        } else {
82                            let fx = kv.multi_get_fx_by_tx_digest(&digests).await.unwrap();
83                            for (digest, fx) in digests.iter().zip(fx.iter()) {
84                                println!("fetched fx: {:?} {:?}", digest, fx);
85                            }
86                        }
87                    }
88
89                    "ckpt_contents" => {
90                        let ckpts = kv.multi_get_checkpoints(&[], &seqs, &[]).await.unwrap();
91
92                        for (seq, ckpt) in seqs.iter().zip(ckpts.1.iter()) {
93                            // populate digest before printing
94                            ckpt.as_ref().map(|c| c.digest());
95                            println!("fetched ckpt contents: {:?} {:?}", seq, ckpt);
96                        }
97                    }
98
99                    "ckpt_summary" => {
100                        let digests: Vec<_> = digest
101                            .into_iter()
102                            .map(|s| {
103                                CheckpointDigest::from_str(&s).expect("invalid checkpoint digest")
104                            })
105                            .collect();
106
107                        let ckpts = kv
108                            .multi_get_checkpoints(&seqs, &[], &digests)
109                            .await
110                            .unwrap();
111
112                        for (seq, ckpt) in seqs.iter().zip(ckpts.0.iter()) {
113                            // populate digest before printing
114                            ckpt.as_ref().map(|c| c.digest());
115                            println!("fetched ckpt summary: {:?} {:?}", seq, ckpt);
116                        }
117                        for (digest, ckpt) in digests.iter().zip(ckpts.2.iter()) {
118                            // populate digest before printing
119                            ckpt.as_ref().map(|c| c.digest());
120                            println!("fetched ckpt summary: {:?} {:?}", digest, ckpt);
121                        }
122                    }
123
124                    "ob" => {
125                        let object_id = ObjectID::from_str(&digest[0]).expect("invalid object id");
126                        let object = kv.get_object(object_id, seqs[0].into()).await.unwrap();
127                        println!("fetched object {:?}", object);
128                    }
129
130                    _ => {
131                        println!(
132                            "Invalid key type: {}. Must be one of 'tx', 'fx', or 'ev'.",
133                            type_
134                        );
135                        std::process::exit(1);
136                    }
137                }
138
139                Ok(())
140            }
141            Command::DecodeKey { url } => {
142                // url may look like
143                // https://transactions.sui.io/mainnet/jlkqmZbVuunngIyy2vjBOJSETrM56EH_kIc5wuLvDydN_x0GAAAAAA/ob
144                // extract the digest and type
145                let parts: Vec<_> = url.split('/').collect();
146
147                // its allowed to supply either the whole URL, or the last two pieces
148                if parts.len() < 2 {
149                    println!("Invalid URL: {}", url);
150                    std::process::exit(1);
151                }
152
153                let identifier = parts[parts.len() - 2];
154                let type_ = parts[parts.len() - 1];
155
156                let key = path_elements_to_key(identifier, type_)?;
157                println!("decoded key: {:?}", key);
158                Ok(())
159            }
160        }
161    }
162}

164#[derive(Parser)]
165#[command(
166    name = "http_kv_tool",
167    about = "Utilities for the HTTP key-value store",
168    rename_all = "kebab-case",
169    author,
170    version = "1"
171)]
172struct App {
173    #[command(subcommand)]
174    command: Command,
175}
176
177#[tokio::main]
178async fn main() {
179    let _guard = telemetry_subscribers::TelemetryConfig::new()
180        .with_env()
181        .init();
182
183    let app = App::parse();
184    app.command.execute().await.unwrap();
