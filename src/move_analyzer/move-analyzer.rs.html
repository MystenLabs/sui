4use clap::*;
5use move_analyzer::analyzer;
6use move_compiler::editions::Flavor;
7use sui_move_build::{SuiPackageHooks, implicit_deps};
8use sui_package_management::system_package_versions::latest_system_packages;
9
10#[cfg(target_os = "linux")]
11mod alloc_utils {
12    // On Linux, jemaclloc produces better results in terms of memory usage.
13    // Standard allocator does not work too well for cases when a lot of memory
14    // is allocated temporarily and then freed as tends to hold on to allocated
15    // memory rather than returning it to the OS right away.
16    use tikv_jemallocator::Jemalloc;
17    #[global_allocator]
18    static GLOBAL: Jemalloc = Jemalloc;
19
20    pub fn maybe_enable_jemalloc() {
21        match tikv_jemalloc_ctl::version::read() {
22            Ok(version) => eprintln!("jemalloc version = {}", version),
23            Err(error) => eprintln!("cannot read jemalloc version: {}", error),
24        }
25        // enable background purge threads
26        let _ = tikv_jemalloc_ctl::background_thread::write(true);
27        let _ = tikv_jemalloc_ctl::epoch::advance();
28    }
29}
30
31#[cfg(not(target_os = "linux"))]
32mod alloc_utils {
33    // We could use a jemalloc here as well but standard allocator
34    // on MacOS is better tuned even for the specific workload
35    // we are dealing with, and jemalloc on Windows is not well supported
36    // so we are not going to use it there either, particularly
37    // that the standard allocator on Windows is known to be well tuned as well.
38
39    pub fn maybe_enable_jemalloc() {
40        eprintln!("using standard allocator");
41    }
42}
43
44// Define the `GIT_REVISION` and `VERSION` consts
45bin_version::bin_version!();
46
47#[derive(Parser)]
48#[clap(
49    name = env!("CARGO_BIN_NAME"),
50    rename_all = "kebab-case",
51    author,
52    version = VERSION,
53)]
54struct App {}
55
56fn main() {
57    App::parse();
58
59    alloc_utils::maybe_enable_jemalloc();
60
61    let sui_implicit_deps = implicit_deps(latest_system_packages());
62    let flavor = Flavor::Sui;
63    let sui_pkg_hooks = Box::new(SuiPackageHooks);
64    analyzer::run(sui_implicit_deps, Some(flavor), Some(sui_pkg_hooks));
65}
