4use clap::*;
5use move_analyzer::analyzer;
6use move_compiler::editions::Flavor;
7use sui_package_alt::SuiFlavor;
8
9#[cfg(target_os = "linux")]
10mod alloc_utils {
11    // On Linux, jemaclloc produces better results in terms of memory usage.
12    // Standard allocator does not work too well for cases when a lot of memory
13    // is allocated temporarily and then freed as tends to hold on to allocated
14    // memory rather than returning it to the OS right away.
15    use tikv_jemallocator::Jemalloc;
16    #[global_allocator]
17    static GLOBAL: Jemalloc = Jemalloc;
18
19    pub fn maybe_enable_jemalloc() {
20        match tikv_jemalloc_ctl::version::read() {
21            Ok(version) => eprintln!("jemalloc version = {}", version),
22            Err(error) => eprintln!("cannot read jemalloc version: {}", error),
23        }
24        // enable background purge threads
25        let _ = tikv_jemalloc_ctl::background_thread::write(true);
26        let _ = tikv_jemalloc_ctl::epoch::advance();
27    }
28}
29
30#[cfg(not(target_os = "linux"))]
31mod alloc_utils {
32    // We could use a jemalloc here as well but standard allocator
33    // on MacOS is better tuned even for the specific workload
34    // we are dealing with, and jemalloc on Windows is not well supported
35    // so we are not going to use it there either, particularly
36    // that the standard allocator on Windows is known to be well tuned as well.
37
38    pub fn maybe_enable_jemalloc() {
39        eprintln!("using standard allocator");
40    }
41}
42
43// Define the `GIT_REVISION` and `VERSION` consts
44bin_version::bin_version!();
45
46#[derive(Parser)]
47#[clap(
48    name = env!("CARGO_BIN_NAME"),
49    rename_all = "kebab-case",
50    author,
51    version = VERSION,
52)]
53struct App {}
54
55fn main() {
56    App::parse();
57    alloc_utils::maybe_enable_jemalloc();
58    analyzer::run::<SuiFlavor>(Some(Flavor::Sui));
59}
