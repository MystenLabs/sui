4/// service health related utilities
5///
6use serde::Serialize;
7
8#[derive(Debug, Clone, Serialize, PartialEq, Eq)]
9#[serde(rename_all = "lowercase")]
10pub enum ServiceStatus {
11    Up,
12}
13
14#[derive(Debug, Clone, Serialize, PartialEq, Eq)]
15pub struct HealthResponse {
16    name: String,
17    version: String,
18    status: ServiceStatus,
19}
20
21impl HealthResponse {
22    pub fn new(package_name: &str, package_version: &str) -> Self {
23        Self {
24            name: package_name.to_owned(),
25            version: package_version.to_owned(),
26            status: ServiceStatus::Up,
27        }
28    }
29}
30
31#[cfg(test)]
32mod tests {
33    use super::*;
34
35    #[test]
36    fn health_response_new_works() {
37        let result = HealthResponse::new("myslib", "0.0.1");
38        assert_eq!(
39            result,
40            HealthResponse {
41                name: "myslib".to_owned(),
42                version: "0.0.1".to_owned(),
43                status: ServiceStatus::Up
44            }
45        );
46    }
47}
