4use prometheus::Registry;
5use std::net::SocketAddr;
6use std::net::{IpAddr, Ipv4Addr};
7
8pub const METRICS_HOST_PORT: u16 = 9184;
9
10/// This is an option if you need to use the underlying method
11pub use mysten_metrics::start_prometheus_server;
12
13/// Use the standard IP (0.0.0.0) and port (9184) to start a new
14/// prometheus server.
15///
16/// Use this function to get a registry and then register your
17/// own application metrics via
18///
19/// ```
20/// use prometheus::{register_int_counter_with_registry, IntCounter, Registry};
21///
22/// pub struct MyMetrics {
23///     pub requests: IntCounter,
24/// }
25///
26/// impl MyMetrics {
27///     pub fn new(registry: &Registry) -> Self {
28///         Self {
29///             requests: register_int_counter_with_registry!(
30///                 "total_requests",
31///                 "Total number of requests received by my service",
32///                 registry
33///             )
34///             .unwrap(),
35///         }
36///     }
37/// }
38///
39/// #[tokio::main]
40/// async fn main() {
41///      let prometheus_registry = mysten_service::metrics::start_basic_prometheus_server();
42///      let metrics = MyMetrics::new(&prometheus_registry);
43/// }
44/// ```
45pub fn start_basic_prometheus_server() -> Registry {
46    let addr = SocketAddr::new(IpAddr::V4(Ipv4Addr::new(0, 0, 0, 0)), METRICS_HOST_PORT);
47    mysten_metrics::start_prometheus_server(addr).default_registry()
48}
