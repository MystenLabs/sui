4use axum::extract::Request;
5use axum::middleware::Next;
6use axum::response::Response;
7use mysten_metrics::{add_server_timing, get_server_timing, with_new_server_timing};
8use simple_server_timing_header::Timer;
9
10pub async fn server_timing_middleware(request: Request, next: Next) -> Response {
11    with_new_server_timing(async move {
12        let mut response = next.run(request).await;
13        add_server_timing("finish_request");
14
15        if let Ok(header_value) = get_server_timing()
16            .expect("server timing is not set")
17            .lock()
18            .header_value()
19            .try_into()
20        {
21            response
22                .headers_mut()
23                .insert(Timer::header_key(), header_value);
24        }
25        response
26    })
27    .await
28}
