4//! A `Simulacrum` of Sui.

7//! a copy of a creature which then follows the player's commands and wishes. As such this crate

67/// Configuration for advancing epochs in the Simulacrum.
68///
69/// Controls which special end-of-epoch transactions are created during epoch transitions.
70#[derive(Debug, Clone, Default)]
71pub struct AdvanceEpochConfig {
72    /// Controls whether a `RandomStateCreate` end-of-epoch transaction is included
73    /// (to initialise on-chain randomness for the first time).
74    pub create_random_state: bool,
75    /// Controls whether to create authenticator state.
76    pub create_authenticator_state: bool,
77    /// Controls whether to expire authenticator state.
78    pub create_authenticator_state_expire: bool,
79    /// Controls whether to create deny list state.
80    pub create_deny_list_state: bool,
81    /// Controls whether to create bridge state.
82    pub create_bridge_state: bool,
83    /// Controls whether to create bridge committee.
84    pub create_bridge_committee: bool,
85    /// When specified, loads system packages from a framework snapshot for the given
86    /// protocol version and includes them in the epoch change transaction.
87    /// This provides test stability as snapshot packages don't change when the framework is updated.
88    /// If None, no system packages are included in the epoch change transaction.
89    pub system_packages_snapshot: Option<u64>,
90}

95/// A `Simulacrum` of Sui.
96///
97/// This type represents a simulated instantiation of a Sui blockchain that needs to be driven
98/// manually, that is time doesn't advance and checkpoints are not formed unless explicitly
99/// requested.
100///
101/// See [module level][mod] documentation for more details.
102///
103/// [mod]: index.html
104pub struct Simulacrum<R = OsRng, Store: SimulatorStore = InMemoryStore> {
105    rng: R,
106    keystore: KeyStore,
107    #[allow(unused)]
108    genesis: genesis::Genesis,
109    store: Store,
110    checkpoint_builder: MockCheckpointBuilder,
111
112    // Epoch specific data
113    epoch_state: EpochState,
114
115    // Other
116    deny_config: TransactionDenyConfig,
117    data_ingestion_path: Option<PathBuf>,
118    verifier_signing_config: VerifierSigningConfig,
119}

133    /// Create a new Simulacrum instance using the provided `rng`.
134    ///
135    /// This allows you to create a fully deterministic initial chainstate when a seeded rng is
136    /// used.
137    ///
138    /// ```
139    /// use simulacrum::Simulacrum;
140    /// use rand::{SeedableRng, rngs::StdRng};
141    ///
142    /// # fn main() {
143    /// let mut rng = StdRng::seed_from_u64(1);
144    /// let simulacrum = Simulacrum::new_with_rng(rng);
145    /// # }
146    /// ```
147    pub fn new_with_rng(mut rng: R) -> Self {
148        let config = ConfigBuilder::new_with_temp_dir()
149            .rng(&mut rng)
150            .with_chain_start_timestamp_ms(1)
151            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
152            .build();
153        Self::new_with_network_config_in_mem(&config, rng)
154    }

241    /// Execute a transaction, allowing empty-signature sender impersonation.
242    ///
243    /// Transactions with signatures are verified normally. Transactions with an empty signature
244    /// list are treated as an explicit impersonation request and skip user-signature verification,
245    /// while preserving the normal execution-time input checks.
246    ///
247    /// # Arguments
248    /// * `transaction` - The transaction to execute
249    ///
250    /// # Returns
251    /// The transaction effects and optional execution error
252    pub fn execute_transaction_impersonating(
253        &mut self,
254        transaction: Transaction,
255    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
256        let verified_transaction = if transaction.data().tx_signatures().is_empty() {
257            VerifiedTransaction::new_unchecked(transaction)
258        } else {
259            transaction
260                .try_into_verified_for_testing(self.epoch_state.epoch(), &VerifyParams::default())?
261        };
262
263        self.execute_transaction_impl(verified_transaction)
264    }
265
266    /// Attempts to execute the provided Transaction.
267    ///
268    /// The provided Transaction undergoes the same types of checks that a Validator does prior to
269    /// signing and executing in the production system. Some of these checks are as follows:
270    /// - User signature is valid
271    /// - Sender owns all OwnedObject inputs
272    /// - etc
273    ///
274    /// If the above checks are successful then the transaction is immediately executed, enqueued
275    /// to be included in the next checkpoint (the next time `create_checkpoint` is called) and the
276    /// corresponding TransactionEffects are returned.
277    pub fn execute_transaction(
278        &mut self,
279        transaction: Transaction,
280    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
281        let transaction = transaction
282            .try_into_verified_for_testing(self.epoch_state.epoch(), &VerifyParams::default())?;
283        self.execute_transaction_impl(transaction)
284    }
285
286    fn execute_transaction_impl(
287        &mut self,
288        transaction: VerifiedTransaction,
289    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
290        let (inner_temporary_store, _, effects, execution_error_opt) =
291            self.epoch_state.execute_transaction(
292                &self.store,
293                &self.deny_config,
294                &self.verifier_signing_config,
295                &transaction,
296            )?;
297
298        let InnerTemporaryStore {
299            written, events, ..
300        } = inner_temporary_store;
301
302        self.store.insert_executed_transaction(
303            transaction.clone(),
304            effects.clone(),
305            events,
306            written,
307        );
308
309        // Insert into checkpoint builder
310        self.checkpoint_builder
311            .push_transaction(transaction, effects.clone());
312        Ok((effects, execution_error_opt.err()))
313    }
314
315    fn execute_system_transaction(
316        &mut self,
317        transaction: Transaction,
318    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
319        let transaction = VerifiedTransaction::new_unchecked(transaction);
320        self.execute_transaction_impl(transaction)
321    }

323    /// Creates the next Checkpoint using the Transactions enqueued since the last checkpoint was
324    /// created.
325    pub fn create_checkpoint(&mut self) -> VerifiedCheckpoint {
326        if self.epoch_state.protocol_config().enable_accumulators() {
327            let (settlement_txns, checkpoint_height) = self
328                .checkpoint_builder
329                .get_settlement_txns(self.epoch_state.protocol_config());
330
331            // Execute settlement transactions and collect their effects
332            let mut settlement_effects = Vec::with_capacity(settlement_txns.len());
333            for txn in settlement_txns {
334                let (effects, _) = self
335                    .execute_system_transaction(txn)
336                    .expect("settlement txn cannot fail");
337                effects.status().unwrap();
338                settlement_effects.push(effects);
339            }
340
341            // Build and execute the barrier transaction using settlement effects
342            let barrier_tx = self
343                .checkpoint_builder
344                .get_barrier_tx(checkpoint_height, &settlement_effects);
345            self.execute_system_transaction(barrier_tx)
346                .expect("barrier txn cannot fail")
347                .0
348                .status()
349                .unwrap();
350        }
351
352        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
353        let (checkpoint, contents, _) = self
354            .checkpoint_builder
355            .build(&committee, self.store.get_clock().timestamp_ms());
356        self.store.insert_checkpoint(checkpoint.clone());
357        self.store.insert_checkpoint_contents(contents.clone());
358        self.process_data_ingestion(checkpoint.clone(), contents)
359            .unwrap();
360        checkpoint
361    }

363    /// Advances the clock by `duration`.
364    ///
365    /// This creates and executes a ConsensusCommitPrologue transaction which advances the chain
366    /// Clock by the provided duration.
367    pub fn advance_clock(&mut self, duration: std::time::Duration) -> TransactionEffects {
368        let epoch = self.epoch_state.epoch();
369        let round = self.epoch_state.next_consensus_round();
370        let timestamp_ms = self.store.get_clock().timestamp_ms() + duration.as_millis() as u64;
371
372        let consensus_commit_prologue_transaction =
373            VerifiedTransaction::new_consensus_commit_prologue_v3(
374                epoch,
375                round,
376                timestamp_ms,
377                ConsensusCommitDigest::default(),
378                ConsensusDeterminedVersionAssignments::empty_for_testing(),
379            );
380
381        self.execute_transaction(consensus_commit_prologue_transaction.into())
382            .expect("advancing the clock cannot fail")
383            .0
384    }

386    /// Advances the epoch.
387    ///
388    /// This creates and executes an EndOfEpoch transaction which advances the chain into the next
389    /// epoch. Since it is required to be the final transaction in an epoch, the final checkpoint in
390    /// the epoch is also created.
391    ///
392    /// The `config` parameter controls which special end-of-epoch transactions are created
393    /// as part of this epoch change.
394    ///
395    /// NOTE: This function does not currently support updating the protocol version
396    pub fn advance_epoch(&mut self, config: AdvanceEpochConfig) {
397        let next_epoch = self.epoch_state.epoch() + 1;
398        let next_epoch_protocol_version = self.epoch_state.protocol_version();
399        let gas_cost_summary = self.checkpoint_builder.epoch_rolling_gas_cost_summary();
400        let epoch_start_timestamp_ms = self.store.get_clock().timestamp_ms();
401
402        let next_epoch_system_package_bytes = if let Some(snapshot_version) =
403            config.system_packages_snapshot
404        {
405            let packages = match load_bytecode_snapshot(snapshot_version) {
406                Ok(snapshot_packages) => snapshot_packages,
407                Err(e) => {
408                    panic!("Failed to load bytecode snapshot for version {snapshot_version}: {e}");
409                }
410            };
411
412            packages
413                .into_iter()
414                .map(|pkg| (SequenceNumber::from(1u64), pkg.bytes, pkg.dependencies))
415                .collect()
416        } else {
417            vec![]
418        };
419
420        let mut kinds = vec![];
421
422        if config.create_random_state {
423            kinds.push(EndOfEpochTransactionKind::new_randomness_state_create());
424        }
425
426        if config.create_authenticator_state {
427            kinds.push(EndOfEpochTransactionKind::new_authenticator_state_create());
428        }
429
430        if config.create_authenticator_state_expire {
431            let current_epoch = self.epoch_state.epoch();
432            kinds.push(EndOfEpochTransactionKind::new_authenticator_state_expire(
433                current_epoch,
434                SequenceNumber::from(1),
435            ));
436        }
437
438        if config.create_deny_list_state {
439            kinds.push(EndOfEpochTransactionKind::new_deny_list_state_create());
440        }
441
442        if config.create_bridge_state {
443            // Use a default test chain identifier for bridge state creation
444            let chain_id = ChainIdentifier::default();
445            kinds.push(EndOfEpochTransactionKind::new_bridge_create(chain_id));
446        }
447
448        if config.create_bridge_committee {
449            // Use a default sequence number for bridge committee initialization
450            let bridge_version = SequenceNumber::from(1);
451            kinds.push(EndOfEpochTransactionKind::init_bridge_committee(
452                bridge_version,
453            ));
454        }
455
456        kinds.push(EndOfEpochTransactionKind::new_change_epoch(
457            next_epoch,
458            next_epoch_protocol_version,
459            gas_cost_summary.storage_cost,
460            gas_cost_summary.computation_cost,
461            gas_cost_summary.storage_rebate,
462            gas_cost_summary.non_refundable_storage_fee,
463            epoch_start_timestamp_ms,
464            next_epoch_system_package_bytes,
465        ));
466
467        let tx = VerifiedTransaction::new_end_of_epoch_transaction(kinds);
468        self.execute_transaction(tx.into())
469            .expect("advancing the epoch cannot fail");
470
471        let new_epoch_state = EpochState::new_with_protocol_config(
472            self.store.get_system_state(),
473            self.epoch_state.protocol_config().clone(),
474            self.epoch_state.chain_identifier(),
475        );
476        let end_of_epoch_data = EndOfEpochData {
477            next_epoch_committee: new_epoch_state.committee().voting_rights.clone(),
478            next_epoch_protocol_version,
479            epoch_commitments: vec![],
480        };
481        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
482        let (checkpoint, contents, _) = self.checkpoint_builder.build_end_of_epoch(
483            &committee,
484            self.store.get_clock().timestamp_ms(),
485            next_epoch,
486            end_of_epoch_data,
487        );
488
489        self.store.insert_checkpoint(checkpoint.clone());
490        self.store.insert_checkpoint_contents(contents.clone());
491        self.process_data_ingestion(checkpoint, contents).unwrap();
492        self.epoch_state = new_epoch_state;
493    }

533    /// Create a new account and credit it with `amount` gas units from a faucet account. Returns
534    /// the account, its keypair, and a reference to the gas object it was funded with.
535    ///
536    /// ```
537    /// use simulacrum::Simulacrum;
538    /// use sui_types::base_types::SuiAddress;
539    /// use sui_types::gas_coin::MIST_PER_SUI;
540    ///
541    /// # fn main() {
542    /// let mut simulacrum = Simulacrum::new();
543    /// let (account, kp, gas) = simulacrum.funded_account(MIST_PER_SUI).unwrap();
544    ///
545    /// // `account` is a fresh SuiAddress that owns a Coin<SUI> object with single SUI in it,
546    /// // referred to by `gas`.
547    /// // ...
548    /// # }
549    /// ```
550    pub fn funded_account(
551        &mut self,
552        amount: u64,
553    ) -> Result<(SuiAddress, AccountKeyPair, ObjectRef)> {
554        let (address, key) = get_account_key_pair();
555        let fx = self.request_gas(address, amount)?;
556        ensure!(fx.status().is_ok(), "Failed to request gas for account");
557
558        let gas = fx
559            .created()
560            .into_iter()
561            .find_map(|(oref, owner)| {
562                matches!(owner, Owner::AddressOwner(owner) if owner == address).then_some(oref)
563            })
564            .context("Could not find created object")?;
565
566        Ok((address, key, gas))
567    }

569    /// Request that `amount` Mist be sent to `address` from a faucet account.
570    ///
571    /// ```
572    /// use simulacrum::Simulacrum;
573    /// use sui_types::base_types::SuiAddress;
574    /// use sui_types::gas_coin::MIST_PER_SUI;
575    ///
576    /// # fn main() {
577    /// let mut simulacrum = Simulacrum::new();
578    /// let address = SuiAddress::generate(simulacrum.rng());
579    /// simulacrum.request_gas(address, MIST_PER_SUI).unwrap();
580    ///
581    /// // `account` now has a Coin<SUI> object with single SUI in it.
582    /// // ...
583    /// # }
584    /// ```
585    pub fn request_gas(&mut self, address: SuiAddress, amount: u64) -> Result<TransactionEffects> {
586        // For right now we'll just use the first account as the `faucet` account. We may want to
587        // explicitly cordon off the faucet account from the rest of the accounts though.
588        let (sender, key) = self.keystore().accounts().next().unwrap();
589        let object = self
590            .store()
591            .owned_objects(*sender)
592            .find(|object| {
593                object.is_gas_coin() && object.get_coin_value_unsafe() > amount + MIST_PER_SUI
594            })
595            .ok_or_else(|| {
596                anyhow!("unable to find a coin with enough to satisfy request for {amount} Mist")
597            })?;
598
599        let gas_data = sui_types::transaction::GasData {
600            payment: vec![object.compute_object_reference()],
601            owner: *sender,
602            price: self.reference_gas_price(),
603            budget: MIST_PER_SUI,
604        };
605
606        let pt = {
607            let mut builder =
608                sui_types::programmable_transaction_builder::ProgrammableTransactionBuilder::new();
609            builder.transfer_sui(address, Some(amount));
610            builder.finish()
611        };
612
613        let kind = sui_types::transaction::TransactionKind::ProgrammableTransaction(pt);
614        let tx_data =
615            sui_types::transaction::TransactionData::new_with_gas_data(kind, *sender, gas_data);
616        let tx = Transaction::from_data_and_signer(tx_data, vec![key]);
617
618        self.execute_transaction(tx).map(|x| x.0)
619    }

621    pub fn set_data_ingestion_path(&mut self, data_ingestion_path: PathBuf) {
622        self.data_ingestion_path = Some(data_ingestion_path);
623        let checkpoint = self.store.get_checkpoint_by_sequence_number(0).unwrap();
624        let contents = self
625            .store
626            .get_checkpoint_contents(&checkpoint.content_digest);
627        self.process_data_ingestion(checkpoint, contents.unwrap())
628            .unwrap();
629    }
630
631    pub fn override_next_checkpoint_number(&mut self, number: CheckpointSequenceNumber) {
632        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
633        self.checkpoint_builder
634            .override_next_checkpoint_number(number, &committee);
635    }
636
637    fn process_data_ingestion(
638        &self,
639        checkpoint: VerifiedCheckpoint,
640        checkpoint_contents: CheckpointContents,
641    ) -> anyhow::Result<()> {
642        if let Some(path) = &self.data_ingestion_path {
643            let sequence_number = checkpoint.sequence_number;
644            let checkpoint_data = self.get_checkpoint_data(checkpoint, checkpoint_contents)?;
645
646            let mask = FieldMask::from_paths([
647                rpc::v2::Checkpoint::path_builder().sequence_number(),
648                rpc::v2::Checkpoint::path_builder().summary().bcs().value(),
649                rpc::v2::Checkpoint::path_builder().signature().finish(),
650                rpc::v2::Checkpoint::path_builder().contents().bcs().value(),
651                rpc::v2::Checkpoint::path_builder()
652                    .transactions()
653                    .transaction()
654                    .bcs()
655                    .value(),
656                rpc::v2::Checkpoint::path_builder()
657                    .transactions()
658                    .effects()
659                    .bcs()
660                    .value(),
661                rpc::v2::Checkpoint::path_builder()
662                    .transactions()
663                    .effects()
664                    .unchanged_loaded_runtime_objects()
665                    .finish(),
666                rpc::v2::Checkpoint::path_builder()
667                    .transactions()
668                    .events()
669                    .bcs()
670                    .value(),
671                rpc::v2::Checkpoint::path_builder()
672                    .objects()
673                    .objects()
674                    .bcs()
675                    .value(),
676            ]);
677
678            let proto_checkpoint = rpc::v2::Checkpoint::merge_from(&checkpoint_data, &mask.into());
679            let proto_bytes = proto_checkpoint.encode_to_vec();
680            let compressed = zstd::encode_all(&proto_bytes[..], 3)?;
681
682            let file_name = format!("{}.binpb.zst", sequence_number);
683            std::fs::create_dir_all(path)?;
684            std::fs::write(path.join(file_name), compressed)?;
685        }
686        Ok(())
687    }

898impl Simulacrum {
899    /// Generate a random transfer transaction.
900    /// TODO: This is here today to make it easier to write tests. But we should utilize all the
901    /// existing code for generating transactions in sui-test-transaction-builder by defining a trait
902    /// that both WalletContext and Simulacrum implement. Then we can remove this function.
903    pub fn transfer_txn(&mut self, recipient: SuiAddress) -> (Transaction, u64) {
904        let (sender, key) = self.keystore().accounts().next().unwrap();
905        let sender = *sender;
906
907        let object = self
908            .store()
909            .owned_objects(sender)
910            .find(|object| object.is_gas_coin())
911            .unwrap();
912        let gas_coin = GasCoin::try_from(object).unwrap();
913        let transfer_amount = gas_coin.value() / 2;
914
915        let pt = {
916            let mut builder = ProgrammableTransactionBuilder::new();
917            builder.transfer_sui(recipient, Some(transfer_amount));
918            builder.finish()
919        };
920
921        let kind = TransactionKind::ProgrammableTransaction(pt);
922        let gas_data = GasData {
923            payment: vec![object.compute_object_reference()],
924            owner: sender,
925            price: self.reference_gas_price(),
926            budget: 1_000_000_000,
927        };
928        let tx_data = TransactionData::new_with_gas_data(kind, sender, gas_data);
929        let tx = Transaction::from_data_and_signer(tx_data, vec![key]);
930        (tx, transfer_amount)
931    }
