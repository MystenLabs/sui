62/// Configuration for advancing epochs in the Simulacrum.
63///
64/// Controls which special end-of-epoch transactions are created during epoch transitions.
65#[derive(Debug, Clone, Default)]
66pub struct AdvanceEpochConfig {
67    /// Controls whether a `RandomStateCreate` end-of-epoch transaction is included
68    /// (to initialise on-chain randomness for the first time).
69    pub create_random_state: bool,
70    /// Controls whether to create authenticator state.
71    pub create_authenticator_state: bool,
72    /// Controls whether to expire authenticator state.
73    pub create_authenticator_state_expire: bool,
74    /// Controls whether to create deny list state.
75    pub create_deny_list_state: bool,
76    /// Controls whether to create bridge state.
77    pub create_bridge_state: bool,
78    /// Controls whether to create bridge committee.
79    pub create_bridge_committee: bool,
80    /// When specified, loads system packages from a framework snapshot for the given
81    /// protocol version and includes them in the epoch change transaction.
82    /// This provides test stability as snapshot packages don't change when the framework is updated.
83    /// If None, no system packages are included in the epoch change transaction.
84    pub system_packages_snapshot: Option<u64>,
85}

90/// A `Simulacrum` of Sui.
91///
92/// This type represents a simulated instantiation of a Sui blockchain that needs to be driven
93/// manually, that is time doesn't advance and checkpoints are not formed unless explicitly
94/// requested.
95///
96/// See [module level][mod] documentation for more details.
97///
98/// [mod]: index.html
99pub struct Simulacrum<R = OsRng, Store: SimulatorStore = InMemoryStore> {
100    rng: R,
101    keystore: KeyStore,
102    #[allow(unused)]
103    genesis: genesis::Genesis,
104    store: Store,
105    checkpoint_builder: MockCheckpointBuilder,
106
107    // Epoch specific data
108    epoch_state: EpochState,
109
110    // Other
111    deny_config: TransactionDenyConfig,
112    data_ingestion_path: Option<PathBuf>,
113    verifier_signing_config: VerifierSigningConfig,
114}

128    /// Create a new Simulacrum instance using the provided `rng`.
129    ///
130    /// This allows you to create a fully deterministic initial chainstate when a seeded rng is
131    /// used.
132    ///
133    /// ```
134    /// use simulacrum::Simulacrum;
135    /// use rand::{SeedableRng, rngs::StdRng};
136    ///
137    /// # fn main() {
138    /// let mut rng = StdRng::seed_from_u64(1);
139    /// let simulacrum = Simulacrum::new_with_rng(rng);
140    /// # }
141    /// ```
142    pub fn new_with_rng(mut rng: R) -> Self {
143        let config = ConfigBuilder::new_with_temp_dir()
144            .rng(&mut rng)
145            .with_chain_start_timestamp_ms(1)
146            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
147            .build();
148        Self::new_with_network_config_in_mem(&config, rng)
149    }
150
151    /// Create a new Simulacrum instance with a specific protocol version.
152    pub fn new_with_protocol_version(mut rng: R, protocol_version: ProtocolVersion) -> Self {
153        let config = ConfigBuilder::new_with_temp_dir()
154            .rng(&mut rng)
155            .with_chain_start_timestamp_ms(1)
156            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
157            .with_protocol_version(protocol_version)
158            .build();
159        Self::new_with_network_config_in_mem(&config, rng)
160    }
161
162    pub fn new_with_protocol_version_and_accounts(
163        mut rng: R,
164        chain_start_timestamp_ms: u64,
165        protocol_version: ProtocolVersion,
166        account_configs: Vec<AccountConfig>,

168        let config = ConfigBuilder::new_with_temp_dir()

170            .with_chain_start_timestamp_ms(chain_start_timestamp_ms)
171            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
172            .with_protocol_version(protocol_version)
173            .with_accounts(account_configs)

175        Self::new_with_network_config_in_mem(&config, rng)

186        let keystore = KeyStore::from_network_config(config);
187        let checkpoint_builder = MockCheckpointBuilder::new(config.genesis.checkpoint());

190        let epoch_state = EpochState::new(genesis.sui_system_object());

193            rng,
194            keystore,
195            genesis: genesis.clone(),
196            store,
197            checkpoint_builder,

200            verifier_signing_config: VerifierSigningConfig::default(),
201            data_ingestion_path: None,
202        }

205    /// Attempts to execute the provided Transaction.
206    ///
207    /// The provided Transaction undergoes the same types of checks that a Validator does prior to
208    /// signing and executing in the production system. Some of these checks are as follows:
209    /// - User signature is valid
210    /// - Sender owns all OwnedObject inputs
211    /// - etc
212    ///
213    /// If the above checks are successful then the transaction is immediately executed, enqueued
214    /// to be included in the next checkpoint (the next time `create_checkpoint` is called) and the
215    /// corresponding TransactionEffects are returned.
216    pub fn execute_transaction(
217        &mut self,
218        transaction: Transaction,
219    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
220        let transaction = transaction
221            .try_into_verified_for_testing(self.epoch_state.epoch(), &VerifyParams::default())?;
222        self.execute_transaction_impl(transaction)
223    }
224
225    fn execute_transaction_impl(
226        &mut self,
227        transaction: VerifiedTransaction,
228    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
229        let (inner_temporary_store, _, effects, execution_error_opt) =
230            self.epoch_state.execute_transaction(
231                &self.store,
232                &self.deny_config,
233                &self.verifier_signing_config,
234                &transaction,
235            )?;
236
237        let InnerTemporaryStore {
238            written, events, ..
239        } = inner_temporary_store;
240
241        self.store.insert_executed_transaction(
242            transaction.clone(),
243            effects.clone(),
244            events,
245            written,
246        );
247
248        // Insert into checkpoint builder
249        self.checkpoint_builder
250            .push_transaction(transaction, effects.clone());
251        Ok((effects, execution_error_opt.err()))
252    }
253
254    fn execute_system_transaction(
255        &mut self,
256        transaction: Transaction,
257    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
258        let transaction = VerifiedTransaction::new_unchecked(transaction);
259        self.execute_transaction_impl(transaction)
260    }

262    /// Creates the next Checkpoint using the Transactions enqueued since the last checkpoint was
263    /// created.
264    pub fn create_checkpoint(&mut self) -> VerifiedCheckpoint {
265        if self.epoch_state.protocol_config().enable_accumulators() {
266            let settlement_txns = self
267                .checkpoint_builder
268                .get_settlement_txns(self.epoch_state.protocol_config());
269
270            // The final transaction (the barrier transaction) must be executed after
271            // all the settlement transactions.
272            for txn in settlement_txns {
273                self.execute_system_transaction(txn)
274                    .expect("settlement txn cannot fail")
275                    .0
276                    .status()
277                    .unwrap();
278            }
279        }
280
281        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
282        let (checkpoint, contents, _) = self
283            .checkpoint_builder
284            .build(&committee, self.store.get_clock().timestamp_ms());
285        self.store.insert_checkpoint(checkpoint.clone());
286        self.store.insert_checkpoint_contents(contents.clone());
287        self.process_data_ingestion(checkpoint.clone(), contents)
288            .unwrap();
289        checkpoint
290    }

292    /// Advances the clock by `duration`.
293    ///
294    /// This creates and executes a ConsensusCommitPrologue transaction which advances the chain
295    /// Clock by the provided duration.
296    pub fn advance_clock(&mut self, duration: std::time::Duration) -> TransactionEffects {
297        let epoch = self.epoch_state.epoch();
298        let round = self.epoch_state.next_consensus_round();
299        let timestamp_ms = self.store.get_clock().timestamp_ms() + duration.as_millis() as u64;
300
301        let consensus_commit_prologue_transaction =
302            VerifiedTransaction::new_consensus_commit_prologue_v3(
303                epoch,
304                round,
305                timestamp_ms,
306                ConsensusCommitDigest::default(),
307                ConsensusDeterminedVersionAssignments::empty_for_testing(),
308            );
309
310        self.execute_transaction(consensus_commit_prologue_transaction.into())
311            .expect("advancing the clock cannot fail")
312            .0
313    }

315    /// Advances the epoch.
316    ///
317    /// This creates and executes an EndOfEpoch transaction which advances the chain into the next
318    /// epoch. Since it is required to be the final transaction in an epoch, the final checkpoint in
319    /// the epoch is also created.
320    ///
321    /// The `config` parameter controls which special end-of-epoch transactions are created
322    /// as part of this epoch change.
323    ///
324    /// NOTE: This function does not currently support updating the protocol version
325    pub fn advance_epoch(&mut self, config: AdvanceEpochConfig) {
326        let next_epoch = self.epoch_state.epoch() + 1;
327        let next_epoch_protocol_version = self.epoch_state.protocol_version();
328        let gas_cost_summary = self.checkpoint_builder.epoch_rolling_gas_cost_summary();
329        let epoch_start_timestamp_ms = self.store.get_clock().timestamp_ms();
330
331        let next_epoch_system_package_bytes = if let Some(snapshot_version) =
332            config.system_packages_snapshot
333        {
334            let packages = match load_bytecode_snapshot(snapshot_version) {
335                Ok(snapshot_packages) => snapshot_packages,
336                Err(e) => {
337                    panic!("Failed to load bytecode snapshot for version {snapshot_version}: {e}");
338                }
339            };
340
341            packages
342                .into_iter()
343                .map(|pkg| (SequenceNumber::from(1u64), pkg.bytes, pkg.dependencies))
344                .collect()
345        } else {
346            vec![]
347        };
348
349        let mut kinds = vec![];
350
351        if config.create_random_state {
352            kinds.push(EndOfEpochTransactionKind::new_randomness_state_create());
353        }
354
355        if config.create_authenticator_state {
356            kinds.push(EndOfEpochTransactionKind::new_authenticator_state_create());
357        }
358
359        if config.create_authenticator_state_expire {
360            let current_epoch = self.epoch_state.epoch();
361            kinds.push(EndOfEpochTransactionKind::new_authenticator_state_expire(
362                current_epoch,
363                SequenceNumber::from(1),
364            ));
365        }
366
367        if config.create_deny_list_state {
368            kinds.push(EndOfEpochTransactionKind::new_deny_list_state_create());
369        }
370
371        if config.create_bridge_state {
372            // Use a default test chain identifier for bridge state creation
373            let chain_id = ChainIdentifier::default();
374            kinds.push(EndOfEpochTransactionKind::new_bridge_create(chain_id));
375        }
376
377        if config.create_bridge_committee {
378            // Use a default sequence number for bridge committee initialization
379            let bridge_version = SequenceNumber::from(1);
380            kinds.push(EndOfEpochTransactionKind::init_bridge_committee(
381                bridge_version,
382            ));
383        }
384
385        kinds.push(EndOfEpochTransactionKind::new_change_epoch(
386            next_epoch,
387            next_epoch_protocol_version,
388            gas_cost_summary.storage_cost,
389            gas_cost_summary.computation_cost,
390            gas_cost_summary.storage_rebate,
391            gas_cost_summary.non_refundable_storage_fee,
392            epoch_start_timestamp_ms,
393            next_epoch_system_package_bytes,
394        ));
395
396        let tx = VerifiedTransaction::new_end_of_epoch_transaction(kinds);
397        self.execute_transaction(tx.into())
398            .expect("advancing the epoch cannot fail");
399
400        let new_epoch_state = EpochState::new_with_protocol_config(
401            self.store.get_system_state(),
402            self.epoch_state.protocol_config().clone(),
403        );
404        let end_of_epoch_data = EndOfEpochData {
405            next_epoch_committee: new_epoch_state.committee().voting_rights.clone(),
406            next_epoch_protocol_version,
407            epoch_commitments: vec![],
408        };
409        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
410        let (checkpoint, contents, _) = self.checkpoint_builder.build_end_of_epoch(
411            &committee,
412            self.store.get_clock().timestamp_ms(),
413            next_epoch,
414            end_of_epoch_data,
415        );
416
417        self.store.insert_checkpoint(checkpoint.clone());
418        self.store.insert_checkpoint_contents(contents.clone());
419        self.process_data_ingestion(checkpoint, contents).unwrap();
420        self.epoch_state = new_epoch_state;
421    }

449    /// Create a new account and credit it with `amount` gas units from a faucet account. Returns
450    /// the account, its keypair, and a reference to the gas object it was funded with.
451    ///
452    /// ```
453    /// use simulacrum::Simulacrum;
454    /// use sui_types::base_types::SuiAddress;
455    /// use sui_types::gas_coin::MIST_PER_SUI;
456    ///
457    /// # fn main() {
458    /// let mut simulacrum = Simulacrum::new();
459    /// let (account, kp, gas) = simulacrum.funded_account(MIST_PER_SUI).unwrap();
460    ///
461    /// // `account` is a fresh SuiAddress that owns a Coin<SUI> object with single SUI in it,
462    /// // referred to by `gas`.
463    /// // ...
464    /// # }
465    /// ```
466    pub fn funded_account(
467        &mut self,
468        amount: u64,
469    ) -> Result<(SuiAddress, AccountKeyPair, ObjectRef)> {
470        let (address, key) = get_account_key_pair();
471        let fx = self.request_gas(address, amount)?;
472        ensure!(fx.status().is_ok(), "Failed to request gas for account");
473
474        let gas = fx
475            .created()
476            .into_iter()
477            .find_map(|(oref, owner)| {
478                matches!(owner, Owner::AddressOwner(owner) if owner == address).then_some(oref)
479            })
480            .context("Could not find created object")?;
481
482        Ok((address, key, gas))
483    }

485    /// Request that `amount` Mist be sent to `address` from a faucet account.
486    ///
487    /// ```
488    /// use simulacrum::Simulacrum;
489    /// use sui_types::base_types::SuiAddress;
490    /// use sui_types::gas_coin::MIST_PER_SUI;
491    ///
492    /// # fn main() {
493    /// let mut simulacrum = Simulacrum::new();
494    /// let address = SuiAddress::generate(simulacrum.rng());
495    /// simulacrum.request_gas(address, MIST_PER_SUI).unwrap();
496    ///
497    /// // `account` now has a Coin<SUI> object with single SUI in it.
498    /// // ...
499    /// # }
500    /// ```
501    pub fn request_gas(&mut self, address: SuiAddress, amount: u64) -> Result<TransactionEffects> {
502        // For right now we'll just use the first account as the `faucet` account. We may want to
503        // explicitly cordon off the faucet account from the rest of the accounts though.
504        let (sender, key) = self.keystore().accounts().next().unwrap();
505        let object = self
506            .store()
507            .owned_objects(*sender)
508            .find(|object| {
509                object.is_gas_coin() && object.get_coin_value_unsafe() > amount + MIST_PER_SUI
510            })
511            .ok_or_else(|| {
512                anyhow!("unable to find a coin with enough to satisfy request for {amount} Mist")
513            })?;
514
515        let gas_data = sui_types::transaction::GasData {
516            payment: vec![object.compute_object_reference()],
517            owner: *sender,
518            price: self.reference_gas_price(),
519            budget: MIST_PER_SUI,
520        };
521
522        let pt = {
523            let mut builder =
524                sui_types::programmable_transaction_builder::ProgrammableTransactionBuilder::new();
525            builder.transfer_sui(address, Some(amount));
526            builder.finish()
527        };
528
529        let kind = sui_types::transaction::TransactionKind::ProgrammableTransaction(pt);
530        let tx_data =
531            sui_types::transaction::TransactionData::new_with_gas_data(kind, *sender, gas_data);
532        let tx = Transaction::from_data_and_signer(tx_data, vec![key]);
533
534        self.execute_transaction(tx).map(|x| x.0)
535    }

537    pub fn set_data_ingestion_path(&mut self, data_ingestion_path: PathBuf) {
538        self.data_ingestion_path = Some(data_ingestion_path);
539        let checkpoint = self.store.get_checkpoint_by_sequence_number(0).unwrap();
540        let contents = self
541            .store
542            .get_checkpoint_contents(&checkpoint.content_digest);
543        self.process_data_ingestion(checkpoint, contents.unwrap())
544            .unwrap();
545    }

547    pub fn override_next_checkpoint_number(&mut self, number: CheckpointSequenceNumber) {
548        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
549        self.checkpoint_builder
550            .override_next_checkpoint_number(number, &committee);

552
553    fn process_data_ingestion(
554        &self,
555        checkpoint: VerifiedCheckpoint,
556        checkpoint_contents: CheckpointContents,
557    ) -> anyhow::Result<()> {
558        if let Some(path) = &self.data_ingestion_path {
559            let file_name = format!("{}.chk", checkpoint.sequence_number);
560            let checkpoint_data: sui_types::full_checkpoint_content::CheckpointData = self
561                .get_checkpoint_data(checkpoint, checkpoint_contents)?
562                .into();
563            std::fs::create_dir_all(path)?;
564            let blob = Blob::encode(&checkpoint_data, BlobEncoding::Bcs)?;
565            std::fs::write(path.join(file_name), blob.to_bytes())?;
566        }
567        Ok(())
568    }

744impl Simulacrum {
745    /// Generate a random transfer transaction.
746    /// TODO: This is here today to make it easier to write tests. But we should utilize all the
747    /// existing code for generating transactions in sui-test-transaction-builder by defining a trait
748    /// that both WalletContext and Simulacrum implement. Then we can remove this function.
749    pub fn transfer_txn(&mut self, recipient: SuiAddress) -> (Transaction, u64) {
750        let (sender, key) = self.keystore().accounts().next().unwrap();
751        let sender = *sender;
752
753        let object = self
754            .store()
755            .owned_objects(sender)
756            .find(|object| object.is_gas_coin())
757            .unwrap();
758        let gas_coin = GasCoin::try_from(&object).unwrap();
759        let transfer_amount = gas_coin.value() / 2;
760
761        let pt = {
762            let mut builder = ProgrammableTransactionBuilder::new();
763            builder.transfer_sui(recipient, Some(transfer_amount));
764            builder.finish()
765        };
766
767        let kind = TransactionKind::ProgrammableTransaction(pt);
768        let gas_data = GasData {
769            payment: vec![object.compute_object_reference()],
770            owner: sender,
771            price: self.reference_gas_price(),
772            budget: 1_000_000_000,
773        };
774        let tx_data = TransactionData::new_with_gas_data(kind, sender, gas_data);
775        let tx = Transaction::from_data_and_signer(tx_data, vec![key]);
776        (tx, transfer_amount)
777    }
778}

858    #[test]
859    fn transfer() {
860        let mut sim = Simulacrum::new();
861        let recipient = SuiAddress::random_for_testing_only();
862        let (tx, transfer_amount) = sim.transfer_txn(recipient);
863
864        let gas_id = tx.data().transaction_data().gas_data().payment[0].0;
865        let effects = sim.execute_transaction(tx).unwrap().0;
866        let gas_summary = effects.gas_cost_summary();
867        let gas_paid = gas_summary.net_gas_usage();
868
869        assert_eq!(
870            (transfer_amount as i64 - gas_paid) as u64,
871            store::SimulatorStore::get_object(sim.store(), &gas_id)
872                .and_then(|object| GasCoin::try_from(&object).ok())
873                .unwrap()
874                .value()
875        );
876
877        assert_eq!(
878            transfer_amount,
879            sim.store()
880                .owned_objects(recipient)
881                .next()
882                .and_then(|object| GasCoin::try_from(&object).ok())
883                .unwrap()
884                .value()
885        );
886
887        let checkpoint = sim.create_checkpoint();
888
889        assert_eq!(&checkpoint.epoch_rolling_gas_cost_summary, gas_summary);
890        if sim.epoch_state.protocol_config().enable_accumulators() {
891            assert_eq!(checkpoint.network_total_transactions, 3); // genesis + 1 user txn + 1 settlement txn
892        } else {
893            assert_eq!(checkpoint.network_total_transactions, 2); // genesis + 1 user txn
894        };
895    }
