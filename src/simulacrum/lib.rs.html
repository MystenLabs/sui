65/// Configuration for advancing epochs in the Simulacrum.
66///
67/// Controls which special end-of-epoch transactions are created during epoch transitions.
68#[derive(Debug, Clone, Default)]
69pub struct AdvanceEpochConfig {
70    /// Controls whether a `RandomStateCreate` end-of-epoch transaction is included
71    /// (to initialise on-chain randomness for the first time).
72    pub create_random_state: bool,
73    /// Controls whether to create authenticator state.
74    pub create_authenticator_state: bool,
75    /// Controls whether to expire authenticator state.
76    pub create_authenticator_state_expire: bool,
77    /// Controls whether to create deny list state.
78    pub create_deny_list_state: bool,
79    /// Controls whether to create bridge state.
80    pub create_bridge_state: bool,
81    /// Controls whether to create bridge committee.
82    pub create_bridge_committee: bool,
83    /// When specified, loads system packages from a framework snapshot for the given
84    /// protocol version and includes them in the epoch change transaction.
85    /// This provides test stability as snapshot packages don't change when the framework is updated.
86    /// If None, no system packages are included in the epoch change transaction.
87    pub system_packages_snapshot: Option<u64>,
88}

93/// A `Simulacrum` of Sui.
94///
95/// This type represents a simulated instantiation of a Sui blockchain that needs to be driven
96/// manually, that is time doesn't advance and checkpoints are not formed unless explicitly
97/// requested.
98///
99/// See [module level][mod] documentation for more details.
100///
101/// [mod]: index.html
102pub struct Simulacrum<R = OsRng, Store: SimulatorStore = InMemoryStore> {

111    epoch_state: EpochState,
112

114    deny_config: TransactionDenyConfig,
115    data_ingestion_path: Option<PathBuf>,
116    verifier_signing_config: VerifierSigningConfig,
117}

131    /// Create a new Simulacrum instance using the provided `rng`.
132    ///
133    /// This allows you to create a fully deterministic initial chainstate when a seeded rng is
134    /// used.
135    ///
136    /// ```
137    /// use simulacrum::Simulacrum;
138    /// use rand::{SeedableRng, rngs::StdRng};
139    ///
140    /// # fn main() {
141    /// let mut rng = StdRng::seed_from_u64(1);
142    /// let simulacrum = Simulacrum::new_with_rng(rng);
143    /// # }
144    /// ```
145    pub fn new_with_rng(mut rng: R) -> Self {
146        let config = ConfigBuilder::new_with_temp_dir()
147            .rng(&mut rng)
148            .with_chain_start_timestamp_ms(1)
149            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
150            .build();
151        Self::new_with_network_config_in_mem(&config, rng)
152    }

208    /// Attempts to execute the provided Transaction.
209    ///
210    /// The provided Transaction undergoes the same types of checks that a Validator does prior to
211    /// signing and executing in the production system. Some of these checks are as follows:
212    /// - User signature is valid
213    /// - Sender owns all OwnedObject inputs
214    /// - etc
215    ///
216    /// If the above checks are successful then the transaction is immediately executed, enqueued
217    /// to be included in the next checkpoint (the next time `create_checkpoint` is called) and the
218    /// corresponding TransactionEffects are returned.
219    pub fn execute_transaction(
220        &mut self,
221        transaction: Transaction,
222    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
223        let transaction = transaction
224            .try_into_verified_for_testing(self.epoch_state.epoch(), &VerifyParams::default())?;
225        self.execute_transaction_impl(transaction)
226    }
227
228    fn execute_transaction_impl(
229        &mut self,
230        transaction: VerifiedTransaction,
231    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
232        let (inner_temporary_store, _, effects, execution_error_opt) =
233            self.epoch_state.execute_transaction(
234                &self.store,
235                &self.deny_config,
236                &self.verifier_signing_config,
237                &transaction,
238            )?;
239
240        let InnerTemporaryStore {
241            written, events, ..
242        } = inner_temporary_store;
243
244        self.store.insert_executed_transaction(
245            transaction.clone(),
246            effects.clone(),
247            events,
248            written,
249        );
250
251        // Insert into checkpoint builder
252        self.checkpoint_builder
253            .push_transaction(transaction, effects.clone());
254        Ok((effects, execution_error_opt.err()))
255    }
256
257    fn execute_system_transaction(
258        &mut self,
259        transaction: Transaction,
260    ) -> anyhow::Result<(TransactionEffects, Option<ExecutionError>)> {
261        let transaction = VerifiedTransaction::new_unchecked(transaction);
262        self.execute_transaction_impl(transaction)
263    }

265    /// Creates the next Checkpoint using the Transactions enqueued since the last checkpoint was
266    /// created.
267    pub fn create_checkpoint(&mut self) -> VerifiedCheckpoint {
268        if self.epoch_state.protocol_config().enable_accumulators() {
269            let (settlement_txns, checkpoint_height) = self
270                .checkpoint_builder
271                .get_settlement_txns(self.epoch_state.protocol_config());
272
273            // Execute settlement transactions and collect their effects
274            let mut settlement_effects = Vec::with_capacity(settlement_txns.len());
275            for txn in settlement_txns {
276                let (effects, _) = self
277                    .execute_system_transaction(txn)
278                    .expect("settlement txn cannot fail");
279                effects.status().unwrap();
280                settlement_effects.push(effects);
281            }
282
283            // Build and execute the barrier transaction using settlement effects
284            let barrier_tx = self
285                .checkpoint_builder
286                .get_barrier_tx(checkpoint_height, &settlement_effects);
287            self.execute_system_transaction(barrier_tx)
288                .expect("barrier txn cannot fail")
289                .0
290                .status()
291                .unwrap();
292        }
293
294        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
295        let (checkpoint, contents, _) = self
296            .checkpoint_builder
297            .build(&committee, self.store.get_clock().timestamp_ms());
298        self.store.insert_checkpoint(checkpoint.clone());
299        self.store.insert_checkpoint_contents(contents.clone());
300        self.process_data_ingestion(checkpoint.clone(), contents)
301            .unwrap();
302        checkpoint
303    }

305    /// Advances the clock by `duration`.
306    ///
307    /// This creates and executes a ConsensusCommitPrologue transaction which advances the chain
308    /// Clock by the provided duration.
309    pub fn advance_clock(&mut self, duration: std::time::Duration) -> TransactionEffects {
310        let epoch = self.epoch_state.epoch();
311        let round = self.epoch_state.next_consensus_round();
312        let timestamp_ms = self.store.get_clock().timestamp_ms() + duration.as_millis() as u64;
313
314        let consensus_commit_prologue_transaction =
315            VerifiedTransaction::new_consensus_commit_prologue_v3(
316                epoch,
317                round,
318                timestamp_ms,
319                ConsensusCommitDigest::default(),
320                ConsensusDeterminedVersionAssignments::empty_for_testing(),
321            );
322
323        self.execute_transaction(consensus_commit_prologue_transaction.into())
324            .expect("advancing the clock cannot fail")
325            .0
326    }

328    /// Advances the epoch.
329    ///
330    /// This creates and executes an EndOfEpoch transaction which advances the chain into the next
331    /// epoch. Since it is required to be the final transaction in an epoch, the final checkpoint in
332    /// the epoch is also created.
333    ///
334    /// The `config` parameter controls which special end-of-epoch transactions are created
335    /// as part of this epoch change.
336    ///
337    /// NOTE: This function does not currently support updating the protocol version
338    pub fn advance_epoch(&mut self, config: AdvanceEpochConfig) {
339        let next_epoch = self.epoch_state.epoch() + 1;
340        let next_epoch_protocol_version = self.epoch_state.protocol_version();
341        let gas_cost_summary = self.checkpoint_builder.epoch_rolling_gas_cost_summary();
342        let epoch_start_timestamp_ms = self.store.get_clock().timestamp_ms();
343
344        let next_epoch_system_package_bytes = if let Some(snapshot_version) =
345            config.system_packages_snapshot
346        {
347            let packages = match load_bytecode_snapshot(snapshot_version) {
348                Ok(snapshot_packages) => snapshot_packages,
349                Err(e) => {
350                    panic!("Failed to load bytecode snapshot for version {snapshot_version}: {e}");
351                }
352            };
353
354            packages
355                .into_iter()
356                .map(|pkg| (SequenceNumber::from(1u64), pkg.bytes, pkg.dependencies))
357                .collect()
358        } else {
359            vec![]
360        };
361
362        let mut kinds = vec![];
363
364        if config.create_random_state {
365            kinds.push(EndOfEpochTransactionKind::new_randomness_state_create());
366        }
367
368        if config.create_authenticator_state {
369            kinds.push(EndOfEpochTransactionKind::new_authenticator_state_create());
370        }
371
372        if config.create_authenticator_state_expire {
373            let current_epoch = self.epoch_state.epoch();
374            kinds.push(EndOfEpochTransactionKind::new_authenticator_state_expire(
375                current_epoch,
376                SequenceNumber::from(1),
377            ));
378        }
379
380        if config.create_deny_list_state {
381            kinds.push(EndOfEpochTransactionKind::new_deny_list_state_create());
382        }
383
384        if config.create_bridge_state {
385            // Use a default test chain identifier for bridge state creation
386            let chain_id = ChainIdentifier::default();
387            kinds.push(EndOfEpochTransactionKind::new_bridge_create(chain_id));
388        }
389
390        if config.create_bridge_committee {
391            // Use a default sequence number for bridge committee initialization
392            let bridge_version = SequenceNumber::from(1);
393            kinds.push(EndOfEpochTransactionKind::init_bridge_committee(
394                bridge_version,
395            ));
396        }
397
398        kinds.push(EndOfEpochTransactionKind::new_change_epoch(
399            next_epoch,
400            next_epoch_protocol_version,
401            gas_cost_summary.storage_cost,
402            gas_cost_summary.computation_cost,
403            gas_cost_summary.storage_rebate,
404            gas_cost_summary.non_refundable_storage_fee,
405            epoch_start_timestamp_ms,
406            next_epoch_system_package_bytes,
407        ));
408
409        let tx = VerifiedTransaction::new_end_of_epoch_transaction(kinds);
410        self.execute_transaction(tx.into())
411            .expect("advancing the epoch cannot fail");
412
413        let new_epoch_state = EpochState::new_with_protocol_config(
414            self.store.get_system_state(),
415            self.epoch_state.protocol_config().clone(),
416        );
417        let end_of_epoch_data = EndOfEpochData {
418            next_epoch_committee: new_epoch_state.committee().voting_rights.clone(),
419            next_epoch_protocol_version,
420            epoch_commitments: vec![],
421        };
422        let committee = CommitteeWithKeys::new(&self.keystore, self.epoch_state.committee());
423        let (checkpoint, contents, _) = self.checkpoint_builder.build_end_of_epoch(
424            &committee,
425            self.store.get_clock().timestamp_ms(),
426            next_epoch,
427            end_of_epoch_data,
428        );
429
430        self.store.insert_checkpoint(checkpoint.clone());
431        self.store.insert_checkpoint_contents(contents.clone());
432        self.process_data_ingestion(checkpoint, contents).unwrap();
433        self.epoch_state = new_epoch_state;
434    }

462    /// Create a new account and credit it with `amount` gas units from a faucet account. Returns
463    /// the account, its keypair, and a reference to the gas object it was funded with.
464    ///
465    /// ```
466    /// use simulacrum::Simulacrum;
467    /// use sui_types::base_types::SuiAddress;
468    /// use sui_types::gas_coin::MIST_PER_SUI;
469    ///
470    /// # fn main() {
471    /// let mut simulacrum = Simulacrum::new();
472    /// let (account, kp, gas) = simulacrum.funded_account(MIST_PER_SUI).unwrap();
473    ///
474    /// // `account` is a fresh SuiAddress that owns a Coin<SUI> object with single SUI in it,
475    /// // referred to by `gas`.
476    /// // ...
477    /// # }
478    /// ```
479    pub fn funded_account(
480        &mut self,
481        amount: u64,
482    ) -> Result<(SuiAddress, AccountKeyPair, ObjectRef)> {
483        let (address, key) = get_account_key_pair();
484        let fx = self.request_gas(address, amount)?;
485        ensure!(fx.status().is_ok(), "Failed to request gas for account");
486
487        let gas = fx
488            .created()
489            .into_iter()
490            .find_map(|(oref, owner)| {
491                matches!(owner, Owner::AddressOwner(owner) if owner == address).then_some(oref)
492            })
493            .context("Could not find created object")?;
494
495        Ok((address, key, gas))
496    }

498    /// Request that `amount` Mist be sent to `address` from a faucet account.
499    ///
500    /// ```
501    /// use simulacrum::Simulacrum;
502    /// use sui_types::base_types::SuiAddress;
503    /// use sui_types::gas_coin::MIST_PER_SUI;
504    ///
505    /// # fn main() {
506    /// let mut simulacrum = Simulacrum::new();
507    /// let address = SuiAddress::generate(simulacrum.rng());
508    /// simulacrum.request_gas(address, MIST_PER_SUI).unwrap();
509    ///
510    /// // `account` now has a Coin<SUI> object with single SUI in it.
511    /// // ...
512    /// # }
513    /// ```
514    pub fn request_gas(&mut self, address: SuiAddress, amount: u64) -> Result<TransactionEffects> {
515        // For right now we'll just use the first account as the `faucet` account. We may want to
516        // explicitly cordon off the faucet account from the rest of the accounts though.
517        let (sender, key) = self.keystore().accounts().next().unwrap();
518        let object = self
519            .store()
520            .owned_objects(*sender)
521            .find(|object| {
522                object.is_gas_coin() && object.get_coin_value_unsafe() > amount + MIST_PER_SUI
523            })
524            .ok_or_else(|| {
525                anyhow!("unable to find a coin with enough to satisfy request for {amount} Mist")
526            })?;
527
528        let gas_data = sui_types::transaction::GasData {
529            payment: vec![object.compute_object_reference()],
530            owner: *sender,
531            price: self.reference_gas_price(),
532            budget: MIST_PER_SUI,
533        };
534
535        let pt = {
536            let mut builder =
537                sui_types::programmable_transaction_builder::ProgrammableTransactionBuilder::new();
538            builder.transfer_sui(address, Some(amount));
539            builder.finish()
540        };
541
542        let kind = sui_types::transaction::TransactionKind::ProgrammableTransaction(pt);
543        let tx_data =
544            sui_types::transaction::TransactionData::new_with_gas_data(kind, *sender, gas_data);
545        let tx = Transaction::from_data_and_signer(tx_data, vec![key]);
546
547        self.execute_transaction(tx).map(|x| x.0)
548    }

566    fn process_data_ingestion(
567        &self,
568        checkpoint: VerifiedCheckpoint,
569        checkpoint_contents: CheckpointContents,
570    ) -> anyhow::Result<()> {
571        if let Some(path) = &self.data_ingestion_path {
572            let sequence_number = checkpoint.sequence_number;
573            let checkpoint_data = self.get_checkpoint_data(checkpoint, checkpoint_contents)?;
574
575            let mask = FieldMask::from_paths([
576                rpc::v2::Checkpoint::path_builder().sequence_number(),
577                rpc::v2::Checkpoint::path_builder().summary().bcs().value(),
578                rpc::v2::Checkpoint::path_builder().signature().finish(),
579                rpc::v2::Checkpoint::path_builder().contents().bcs().value(),
580                rpc::v2::Checkpoint::path_builder()
581                    .transactions()
582                    .transaction()
583                    .bcs()
584                    .value(),
585                rpc::v2::Checkpoint::path_builder()
586                    .transactions()
587                    .effects()
588                    .bcs()
589                    .value(),
590                rpc::v2::Checkpoint::path_builder()
591                    .transactions()
592                    .effects()
593                    .unchanged_loaded_runtime_objects()
594                    .finish(),
595                rpc::v2::Checkpoint::path_builder()
596                    .transactions()
597                    .events()
598                    .bcs()
599                    .value(),
600                rpc::v2::Checkpoint::path_builder()
601                    .objects()
602                    .objects()
603                    .bcs()
604                    .value(),
605            ]);
606
607            let proto_checkpoint = rpc::v2::Checkpoint::merge_from(&checkpoint_data, &mask.into());
608            let proto_bytes = proto_checkpoint.encode_to_vec();
609            let compressed = zstd::encode_all(&proto_bytes[..], 3)?;
610
611            let file_name = format!("{}.binpb.zst", sequence_number);
612            std::fs::create_dir_all(path)?;
613            std::fs::write(path.join(file_name), compressed)?;
614        }
615        Ok(())
616    }

624impl<'a> CommitteeWithKeys<'a> {

799impl Simulacrum {
800    /// Generate a random transfer transaction.
801    /// TODO: This is here today to make it easier to write tests. But we should utilize all the
802    /// existing code for generating transactions in sui-test-transaction-builder by defining a trait
803    /// that both WalletContext and Simulacrum implement. Then we can remove this function.
804    pub fn transfer_txn(&mut self, recipient: SuiAddress) -> (Transaction, u64) {
805        let (sender, key) = self.keystore().accounts().next().unwrap();
806        let sender = *sender;
807
808        let object = self
809            .store()
810            .owned_objects(sender)
811            .find(|object| object.is_gas_coin())
812            .unwrap();
813        let gas_coin = GasCoin::try_from(&object).unwrap();
814        let transfer_amount = gas_coin.value() / 2;
815
816        let pt = {
817            let mut builder = ProgrammableTransactionBuilder::new();
818            builder.transfer_sui(recipient, Some(transfer_amount));
819            builder.finish()
820        };
821
822        let kind = TransactionKind::ProgrammableTransaction(pt);
823        let gas_data = GasData {
824            payment: vec![object.compute_object_reference()],
825            owner: sender,
826            price: self.reference_gas_price(),
827            budget: 1_000_000_000,
828        };
829        let tx_data = TransactionData::new_with_gas_data(kind, sender, gas_data);
830        let tx = Transaction::from_data_and_signer(tx_data, vec![key]);
831        (tx, transfer_amount)
832    }
833}

914    fn transfer() {
915        let mut sim = Simulacrum::new();
916        let recipient = SuiAddress::random_for_testing_only();
917        let (tx, transfer_amount) = sim.transfer_txn(recipient);
918
919        let gas_id = tx.data().transaction_data().gas_data().payment[0].0;
920        let effects = sim.execute_transaction(tx).unwrap().0;
921        let gas_summary = effects.gas_cost_summary();
922        let gas_paid = gas_summary.net_gas_usage();
923
924        assert_eq!(
925            (transfer_amount as i64 - gas_paid) as u64,
926            store::SimulatorStore::get_object(sim.store(), &gas_id)
927                .and_then(|object| GasCoin::try_from(&object).ok())
928                .unwrap()
929                .value()
930        );
931
932        assert_eq!(
933            transfer_amount,
934            sim.store()
935                .owned_objects(recipient)
936                .next()
937                .and_then(|object| GasCoin::try_from(&object).ok())
938                .unwrap()
939                .value()
940        );
941
942        let checkpoint = sim.create_checkpoint();
943
944        assert_eq!(&checkpoint.epoch_rolling_gas_cost_summary, gas_summary);
945        if sim.epoch_state.protocol_config().enable_accumulators() {
946            assert_eq!(checkpoint.network_total_transactions, 3); // genesis + 1 user txn + 1 settlement txn
947        } else {
948            assert_eq!(checkpoint.network_total_transactions, 2); // genesis + 1 user txn
949        };
