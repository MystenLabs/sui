4//! This module provides a function to render a Move abort status string into a more human-readable
5//! error message using the clever error rendering logic.
6//!
7//! The logic in this file is largely a stop-gap to provide Clever Error rendering in the CLI while
8//! it still uses the JSON-RPC API. The new GraphQL API already rendered Clever Errors on the server
9//! side in a much more robust and efficient way.
10//!
11//! Once the CLI is updated to use the GraphQL API, this file can be removed, and the GraphQL-based
12//! rendering logic for Clever Errors should be used instead.

27/// Take a Move abort status string and render it into a more human-readable error message using
28/// by parsing the string (as best we can) and seeing if the abort code is a Clever Error abort
29/// code. If it is, we attempt to render the error in a more huma-readable manner using the Read
30/// API and decoding the Clever Error encoding in the abort code.
31///
32/// This function is used to render Clever Errors for on-chain errors only within the Sui CLI. This
33/// function is _not_ used at all for off-chain errors or Move unit tests. You should only use this
34/// function within this crate.
35pub(crate) async fn render_clever_error_opt(
36    error_string: &str,
37    read_api: &ReadApi,
38) -> Option<String> {
39    let (address, module_name, function_name, instruction, abort_code, command_index) =
40        parse_abort_status_string(error_string).ok()?;
41
42    let error = 'error: {
43        let Some(error_bitset) = ErrorBitset::from_u64(abort_code) else {
44            break 'error format!(
45                "function '{}::{}::{}' at instruction {} with code {}",
46                address.to_canonical_display(true),
47                module_name,
48                function_name,
49                instruction,
50                abort_code
51            );
52        };
53
54        let line_number = error_bitset.line_number()?;
55
56        if error_bitset.constant_index().is_none() && error_bitset.identifier_index().is_none() {
57            break 'error format!(
58                "function '{}::{}::{}' at line {}",
59                address.to_canonical_display(true),
60                module_name,
61                function_name,
62                line_number
63            );
64        }
65
66        let SuiRawData::Package(package) = read_api
67            .get_object_with_options(
68                ObjectID::from_address(address),
69                SuiObjectDataOptions::bcs_lossless(),
70            )
71            .await
72            .ok()?
73            .into_object()
74            .ok()?
75            .bcs?
76        else {
77            return None;
78        };
79
80        let module = package.module_map.get(module_name.as_str())?;
81        let module =
82            CompiledModule::deserialize_with_config(module, &BinaryConfig::standard()).ok()?;
83
84        let error_identifier_constant = module
85            .constant_pool()
86            .get(error_bitset.identifier_index()? as usize)?;
87        let error_value_constant = module
88            .constant_pool()
89            .get(error_bitset.constant_index()? as usize)?;
90
91        if !matches!(&error_identifier_constant.type_, SignatureToken::Vector(x) if x.as_ref() == &SignatureToken::U8)
92        {
93            return None;
94        };
95
96        let error_identifier = bcs::from_bytes::<Vec<u8>>(&error_identifier_constant.data)
97            .ok()
98            .and_then(|x| String::from_utf8(x).ok())?;
99
100        let const_str = match try_render_constant(error_value_constant) {
101            RenderResult::NotRendered => {
102                format!("'{}'", Base64::encode(&error_value_constant.data))
103            }
104            RenderResult::AsString(s) => format!("'{s}'"),
105            RenderResult::AsValue(v_str) => v_str,
106        };
107
108        let code_str = match error_bitset.error_code() {
109            Some(code) => format!("error code {} --", code),
110            None => "".to_string(),
111        };
112
113        format!(
114            "function '{}::{}::{}' at line {}. Aborted with {}'{}' -- {}",
115            address.to_canonical_display(true),
116            module_name,
117            function_name,
118            line_number,
119            code_str,
120            error_identifier,
121            const_str
122        )
123    };
124
125    // Convert the command index into an ordinal.
126    let command = command_index + 1;
127    let suffix = match command % 10 {
128        1 => "st",
129        2 => "nd",
130        3 => "rd",
131        _ => "th",
132    };
133
134    Some(format!("{command}{suffix} command aborted within {error}"))
135}

137/// Parsing the error with a regex is not great, but it's the best we can do with the current
138/// JSON-RPC API since we only get error messages as strings. This function attempts to parse a
139/// Move abort status string into its different parts, and then parses it back into the structured
140/// format that we can then use to render a Clever Error.
141///
142/// If we are able to parse the string, we return a tuple with the address, module name, function
143/// name, instruction, abort code, and command index. If we are unable to parse the string, we
144/// return `Err`.
145///
146/// You should delete this function with glee once the CLI is updated to use the GraphQL API.
147fn parse_abort_status_string(
148    s: &str,
149) -> Result<(AccountAddress, Identifier, Identifier, u16, u64, u16), anyhow::Error> {
150    use regex::Regex;
151    let re = Regex::new(r#"MoveAbort.*address:\s*(.*?),.* name:.*Identifier\((.*?)\).*instruction:\s+(\d+),.*function_name:.*Some\((.*?)\).*},\s*(\d+).*in command\s*(\d+)"#).unwrap();
152    let Some(captures) = re.captures(s) else {
153        anyhow::bail!(
154            "Cannot parse abort status string: {} as a move abort string",
155            s
156        );
157    };
158
159    // Remove any escape characters from the string if present.
160    let clean_string = |s: &str| s.replace(['\\', '\"'], "");
161
162    let address = AccountAddress::from_hex(&captures[1])?;
163    let module_name = Identifier::new(clean_string(&captures[2]))?;
164    let instruction = captures[3].parse::<u16>()?;
165    let function_name = Identifier::new(clean_string(&captures[4]))?;
166    let abort_code = captures[5].parse::<u64>()?;
167    let command_index = captures[6].parse::<u16>()?;
168    Ok((
169        address,
170        module_name,
171        function_name,
172        instruction,
173        abort_code,
174        command_index,
175    ))
176}
