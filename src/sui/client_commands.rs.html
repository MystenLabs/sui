665/// Arguments related to setting gas data, apart from payment coins.
666#[derive(Args, Debug, Default)]
667pub struct GasDataArgs {
668    /// An optional gas budget for this transaction (in MIST). If gas budget is not provided, the
669    /// tool will first perform a dry run to estimate the gas cost, and then it will execute the
670    /// transaction. Please note that this incurs a small cost in performance due to the additional
671    /// dry run call.
672    #[arg(long)]
673    pub gas_budget: Option<u64>,
674    /// An optional gas price for this transaction (in MIST). If gas price is not provided, the
675    /// tool will use the current reference gas price from RPC.
676    ///
677    /// Transactions with a gas price lower than the reference will not be signed by enough
678    /// validators to execute. Transactions accessing congested shared objects are prioritized by
679    /// gas price, so setting a higher gas price higher than the reference can ensure the
680    /// transaction accesses the shared object sooner.
681    #[arg(long)]
682    pub gas_price: Option<u64>,
683    /// An optional field to specify a gas sponsor address. If provided, the gas owner is set to
684    /// this address, rather than the transaction's sender.
685    ///
686    /// Note that if the CLI does not have access to the sponsor's keys, it will not be able to
687    /// sign and execute transactions that have a sponsor set.
688    #[arg(long)]

1975/// Process the `--skip-dependency-verification` and `--verify-dependencies` flags for a publish or
1976/// upgrade command. Prints deprecation warnings as appropriate and returns true if the
1977/// dependencies should be verified
1978fn check_dep_verification_flags(
1979    skip_dependency_verification: bool,
1980    verify_dependencies: bool,
1981) -> anyhow::Result<bool> {
1982    match (skip_dependency_verification, verify_dependencies) {
1983        (true, true) => bail!(
1984            "[error]: --skip-dependency-verification and --verify-deps are mutually exclusive"
1985        ),
1986
1987        (false, false) => {
1988            eprintln!(
1989                "{}: Dependency sources are no longer verified automatically during publication and upgrade. \
1990                You can pass the `--verify-deps` option if you would like to verify them as part of publication or upgrade.",
1991                "[Note]".bold().yellow()
1992            );
1993            Ok(verify_dependencies)
1994        }
1995
1996        (true, false) => {
1997            eprintln!(
1998                "{}: Dependency sources are no longer verified automatically during publication and upgrade, \
1999                so the `--skip-dependency-verification` flag is no longer necessary.",
2000                "[Warning]".bold().yellow()
2001            );
2002            Ok(verify_dependencies)
2003        }
2004
2005        (false, true) => Ok(verify_dependencies),
2006    }
2007}

2030pub(crate) async fn upgrade_package(
2031    read_api: &ReadApi,
2032    build_config: MoveBuildConfig,
2033    package_path: &Path,
2034    upgrade_capability: ObjectID,
2035    with_unpublished_dependencies: bool,
2036    skip_dependency_verification: bool,
2037    env_alias: Option<String>,
2038) -> Result<(u8, CompiledPackage), anyhow::Error> {
2039    let mut compiled_package = compile_package(
2040        read_api,
2041        build_config,
2042        package_path,
2043        with_unpublished_dependencies,
2044        skip_dependency_verification,
2045    )
2046    .await?;
2047
2048    pkg_tree_shake(
2049        read_api,
2050        with_unpublished_dependencies,
2051        &mut compiled_package,
2052    )
2053    .await?;
2054
2055    compiled_package.published_at.as_ref().map_err(|e| match e {
2056        PublishedAtError::NotPresent => {
2057            anyhow!("No 'published-at' field in Move.toml or 'published-id' in Move.lock for package to be upgraded.")
2058        }
2059        PublishedAtError::Invalid(v) => anyhow!(
2060            "Invalid 'published-at' field in Move.toml or 'published-id' in Move.lock of package to be upgraded. \
2061                         Expected an on-chain address, but found: {v:?}"
2062        ),
2063        PublishedAtError::Conflict {
2064            id_lock,
2065            id_manifest,
2066        } => {
2067            let env_alias = format!("(currently {})", env_alias.unwrap_or_default());
2068            anyhow!(
2069                "Conflicting published package address: `Move.toml` contains published-at address \
2070                 {id_manifest} but `Move.lock` file contains published-at address {id_lock}. \
2071                 You may want to:
2072 - delete the published-at address in the `Move.toml` if the `Move.lock` address is correct; OR
2073 - update the `Move.lock` address using the `sui manage-package` command to be the same as the `Move.toml`; OR
2074 - check that your `sui active-env` {env_alias} corresponds to the chain on which the package is published (i.e., devnet, testnet, mainnet); OR
2075 - contact the maintainer if this package is a dependency and request resolving the conflict."
2076            )
2077        }
2078    })?;
2079
2080    let resp = read_api
2081        .get_object_with_options(
2082            upgrade_capability,
2083            SuiObjectDataOptions::default().with_bcs().with_owner(),
2084        )
2085        .await?;
2086
2087    let Some(data) = resp.data else {
2088        return Err(anyhow!(
2089            "Could not find upgrade capability at {upgrade_capability}"
2090        ));
2091    };
2092
2093    let upgrade_cap: UpgradeCap = data
2094        .bcs
2095        .ok_or_else(|| anyhow!("Fetch upgrade capability object but no data was returned"))?
2096        .try_as_move()
2097        .ok_or_else(|| anyhow!("Upgrade capability is not a Move Object"))?
2098        .deserialize()?;
2099    // We keep the existing policy -- no fancy policies or changing the upgrade
2100    // policy at the moment. To change the policy you can call a Move function in the
2101    // `package` module to change this policy.
2102    let upgrade_policy = upgrade_cap.policy;
2103
2104    Ok((upgrade_policy, compiled_package))
2105}

2174        protocol_config.attributes.get("move_binary_format_version")

2200        return Err(SuiErrorKind::ModulePublishFailure {

2202                "Modules must all have 0x0 as their addresses. \
2203                     Violated by module {:?}",
2204                already_published.self_id(),

2206        }
2207        .into());

2228                     --skip-dependency-verification flag."
2229                ),
2230            }
2231            .into());
2232        } else {
2233            eprintln!(
2234                "{}",
2235                "Successfully verified dependencies on-chain against source."
2236                    .bold()
2237                    .green(),
2238            );
2239        }
2240    } else {
2241        eprintln!("{}", "Skipping dependency verification".bold().yellow());
2242    }
2243

2245        .get_package_bytes(with_unpublished_dependencies)

2251        .into());
2252    }

2255        .package

2257        .build_flags

2260            error: format!("Failed to update Move.lock toolchain version: {e}"),
2261        })?;
2262
2263    Ok(compiled_package)
2264}

3521/// Filter out a package's dependencies which are not referenced in the source code. The algorithm
3522/// finds the immediate dependencies of this package, and the original ids of each transitive
3523/// dependencies for all these immediate package dependencies. For packages that are not referenced
3524/// in the source code, they will be filtered out from the list of dependencies.
3525pub(crate) async fn pkg_tree_shake(
3526    read_api: &ReadApi,
3527    with_unpublished_dependencies: bool,
3528    compiled_package: &mut CompiledPackage,
3529) -> Result<(), anyhow::Error> {
3530    // these are packages that are immediate dependencies of the root package
3531    let immediate_dep_packages =
3532        compiled_package.find_immediate_deps_pkgs_to_keep(with_unpublished_dependencies)?;
3533
3534    // for every immediate dependency package, we need to use its linkage table to determine its
3535    // transitive dependencies and ensure that we keep the required packages, so fetch those tables
3536    let trans_deps_orig_ids = trans_deps_original_ids(read_api, &immediate_dep_packages).await?;
3537    let pkg_name_to_orig_id: BTreeMap<_, _> = compiled_package
3538        .package
3539        .deps_compiled_units
3540        .iter()
3541        .map(|(pkg_name, module)| (*pkg_name, ObjectID::from(module.unit.address.into_inner())))
3542        .collect();
3543
3544    // for every published package in the original list of published dependencies, get its original
3545    // id and then check if that id exists in the linkage table. If it does, then we need to keep
3546    // this package. Similarly, all immediate dep packages must stay
3547    compiled_package.dependency_ids.published.retain(|pkg, _| {
3548        immediate_dep_packages.contains_key(pkg)
3549            || pkg_name_to_orig_id
3550                .get(pkg)
