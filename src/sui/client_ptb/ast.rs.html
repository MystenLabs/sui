140/// A parsed PTB command consisting of the command and the parsed arguments to the command.
141#[derive(Debug, Clone)]
142pub enum ParsedPTBCommand {
143    TransferObjects(Spanned<Vec<Spanned<Argument>>>, Spanned<Argument>),
144    SplitCoins(Spanned<Argument>, Spanned<Vec<Spanned<Argument>>>),
145    MergeCoins(Spanned<Argument>, Spanned<Vec<Spanned<Argument>>>),
146    MakeMoveVec(Spanned<ParsedType>, Spanned<Vec<Spanned<Argument>>>),
147    MoveCall(
148        Spanned<ModuleAccess>,
149        Option<Spanned<Vec<ParsedType>>>,
150        Vec<Spanned<Argument>>,
151    ),
152    Assign(Spanned<String>, Option<Spanned<Argument>>),
153    Publish(Spanned<String>),
154    Upgrade(Spanned<String>, Spanned<Argument>),
155    WarnShadows,
156    Preview,
157}

159/// An enum representing the parsed arguments of a PTB command.
160#[derive(Debug, Clone)]
161pub enum Argument {
162    Bool(bool),
163    U8(u8),
164    U16(u16),
165    U32(u32),
166    U64(u64),
167    U128(u128),
168    U256(move_core_types::u256::U256),
169    InferredNum(move_core_types::u256::U256),
170    Gas,
171    Identifier(String),
172    VariableAccess(Spanned<String>, Vec<Spanned<String>>),
173    Address(NumericalAddress),
174    String(String),
175    Vector(Vec<Spanned<Argument>>),
176    Option(Spanned<Option<Box<Argument>>>),
177}

179impl Argument {
180    /// Resolve an `Argument` into a `MoveValue` if possible. Errors if the `Argument` is not
181    /// convertible to a `MoveValue` of the provided `tag`.
182    pub fn checked_to_pure_move_value(&self, loc: Span, tag: &TypeTag) -> PTBResult<MoveValue> {
183        Ok(match (self, tag) {
184            (Argument::Bool(b), TypeTag::Bool) => MoveValue::Bool(*b),
185            (Argument::U8(u), TypeTag::U8) => MoveValue::U8(*u),
186            (Argument::U16(u), TypeTag::U16) => MoveValue::U16(*u),
187            (Argument::U32(u), TypeTag::U32) => MoveValue::U32(*u),
188            (Argument::U64(u), TypeTag::U64) => MoveValue::U64(*u),
189            (Argument::U128(u), TypeTag::U128) => MoveValue::U128(*u),
190            (Argument::U256(u), TypeTag::U256) => MoveValue::U256(*u),
191            // Inferred numbers, unless they need to be used as a specific type default to u64.
192            (Argument::InferredNum(u), tag) => Self::cast_inferrred_num(*u, tag, loc)?,
193            (Argument::Address(a), TypeTag::Address) => MoveValue::Address(a.into_inner()),
194            (Argument::Vector(vs), TypeTag::Vector(ty)) => MoveValue::Vector(
195                vs.iter()
196                    .map(|sp!(loc, v)| v.checked_to_pure_move_value(*loc, ty))
197                    .collect::<PTBResult<Vec<_>>>()
198                    .map_err(|e| {
199                        e.with_help("Literal vectors cannot contain object values.".to_string())
200                    })?,
201            ),
202            (Argument::String(s), TypeTag::Vector(ty)) if **ty == TypeTag::U8 => {
203                MoveValue::Vector(s.bytes().map(MoveValue::U8).collect::<Vec<_>>())
204            }
205            (Argument::String(s), TypeTag::Struct(stag))
206                if {
207                    let resolved = (
208                        &stag.address,
209                        stag.module.as_ident_str(),
210                        stag.name.as_ident_str(),
211                    );
212                    resolved == RESOLVED_ASCII_STR || resolved == RESOLVED_UTF8_STR
213                } =>
214            {
215                MoveValue::Vector(s.bytes().map(MoveValue::U8).collect::<Vec<_>>())
216            }
217            (Argument::Address(a), TypeTag::Struct(stag))
218                if (
219                    &stag.address,
220                    stag.module.as_ident_str(),
221                    stag.name.as_ident_str(),
222                ) == RESOLVED_SUI_ID =>
223            {
224                MoveValue::Address(a.into_inner())
225            }
226            (Argument::Option(sp!(loc, o)), TypeTag::Vector(ty)) => {
227                if let Some(v) = o {
228                    let v = v
229                        .as_ref()
230                        .checked_to_pure_move_value(*loc, ty)
231                        .map_err(|e| {
232                            e.with_help(
233                                "Literal option values cannot contain object values.".to_string(),
234                            )
235                        })?;
236                    MoveValue::Vector(vec![v])
237                } else {
238                    MoveValue::Vector(vec![])
239                }
240            }
241            (Argument::Option(sp!(loc, o)), TypeTag::Struct(stag))
242                if (
243                    &stag.address,
244                    stag.module.as_ident_str(),
245                    stag.name.as_ident_str(),
246                ) == RESOLVED_STD_OPTION
247                    && stag.type_params.len() == 1 =>
248            {
249                if let Some(v) = o {
250                    let v = v
251                        .as_ref()
252                        .checked_to_pure_move_value(*loc, &stag.type_params[0])
253                        .map_err(|e| {
254                            e.with_help(
255                                "Literal option values cannot contain object values.".to_string(),
256                            )
257                        })?;
258                    MoveValue::Vector(vec![v])
259                } else {
260                    MoveValue::Vector(vec![])
261                }
262            }
263            (Argument::Identifier(_) | Argument::VariableAccess(_, _) | Argument::Gas, _) => {
264                error!(loc, "Unable to convert '{self}' to non-object value.")
265            }
266            (arg, tag) => error!(loc, "Unable to serialize '{arg}' as a {tag} value"),
267        })
268    }

270    /// Resolve an `Argument` into a `MoveValue` if possible. Errors if the `Argument` is not
271    /// convertible to a `MoveValue`.
272    pub fn to_pure_move_value(&self, loc: Span) -> PTBResult<MoveValue> {
273        Ok(match self {
274            Argument::Bool(b) => MoveValue::Bool(*b),
275            Argument::U8(u) => MoveValue::U8(*u),
276            Argument::U16(u) => MoveValue::U16(*u),
277            Argument::U32(u) => MoveValue::U32(*u),
278            Argument::U64(u) => MoveValue::U64(*u),
279            Argument::U128(u) => MoveValue::U128(*u),
280            Argument::U256(u) => MoveValue::U256(*u),
281            // Inferred numbers, unless they need to be used as a specific type default to u64.
282            Argument::InferredNum(u) => Self::cast_inferrred_num(*u, &TypeTag::U64, loc)?,
283            Argument::Address(a) => MoveValue::Address(a.into_inner()),
284            Argument::Vector(vs) => MoveValue::Vector(
285                vs.iter()
286                    .map(|sp!(loc, v)| v.to_pure_move_value(*loc))
287                    .collect::<PTBResult<Vec<_>>>()
288                    .map_err(|e| {
289                        e.with_help("Literal vectors cannot contain object values.".to_string())
290                    })?,
291            ),
292            Argument::String(s) => {
293                MoveValue::Vector(s.bytes().map(MoveValue::U8).collect::<Vec<_>>())
294            }
295            Argument::Option(sp!(loc, o)) => {
296                if let Some(v) = o {
297                    let v = v.as_ref().to_pure_move_value(*loc).map_err(|e| {
298                        e.with_help(
299                            "Literal option values cannot contain object values.".to_string(),
300                        )
301                    })?;
302                    MoveValue::Vector(vec![v])
303                } else {
304                    MoveValue::Vector(vec![])
305                }
306            }
307            Argument::Identifier(_) | Argument::VariableAccess(_, _) | Argument::Gas => {
308                error!(loc, "Unable to convert '{self}' to non-object value.")
309            }
310        })
311    }

313    fn cast_inferrred_num(
314        val: move_core_types::u256::U256,
315        tag: &TypeTag,
316        loc: Span,
317    ) -> PTBResult<MoveValue> {
318        match tag {
319            TypeTag::U8 => u8::try_from(val)
320                .map(MoveValue::U8)
321                .map_err(|_| err!(loc, "Value {val} is too large to be a u8 value")),
322            TypeTag::U16 => u16::try_from(val)
323                .map(MoveValue::U16)
324                .map_err(|_| err!(loc, "Value {val} is too large to be a u16 value")),
325            TypeTag::U32 => u32::try_from(val)
326                .map(MoveValue::U32)
327                .map_err(|_| err!(loc, "Value {val} is too large to be a u32 value")),
328            TypeTag::U64 => u64::try_from(val)
329                .map(MoveValue::U64)
330                .map_err(|_| err!(loc, "Value {val} is too large to be a u64 value")),
331            TypeTag::U128 => u128::try_from(val)
332                .map(MoveValue::U128)
333                .map_err(|_| err!(loc, "Value {val} is too large to be a u128 value")),
334            TypeTag::U256 => Ok(MoveValue::U256(val)),
335            _ => error!(loc, "Expected an integer type but got {tag} for '{val}'"),
336        }
337    }
338}

340impl fmt::Display for Argument {
341    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
342        match self {
343            Argument::Bool(b) => write!(f, "{b}"),
344            Argument::U8(u) => write!(f, "{u}u8"),
345            Argument::U16(u) => write!(f, "{u}u16"),
346            Argument::U32(u) => write!(f, "{u}u32"),
347            Argument::U64(u) => write!(f, "{u}u64"),
348            Argument::U128(u) => write!(f, "{u}u128"),
349            Argument::U256(u) => write!(f, "{u}u256"),
350            Argument::InferredNum(u) => write!(f, "{u}"),
351            Argument::Gas => write!(f, "gas"),
352            Argument::Identifier(i) => write!(f, "{i}"),
353            Argument::VariableAccess(sp!(_, head), accesses) => {
354                write!(f, "{}", head)?;
355                for sp!(_, access) in accesses {
356                    write!(f, ".{}", access)?;
357                }
358                Ok(())
359            }
360            Argument::Address(a) => write!(f, "@{a}"),
361            Argument::String(s) => write!(f, "{s:?}"),
362            Argument::Vector(v) => {
363                write!(f, "vector[")?;
364                let mut prefix = "";
365                for sp!(_, arg) in v.iter() {
366                    write!(f, "{prefix}")?;
367                    write!(f, "{arg}")?;
368                    prefix = ", ";
369                }
370                write!(f, "]")
371            }
372            Argument::Option(sp!(_, o)) => match o {
373                Some(v) => write!(f, "some({v})"),
374                None => write!(f, "none"),
375            },
376        }
377    }
378}

380fn delimited_list<T: fmt::Display>(
381    f: &mut std::fmt::Formatter<'_>,
382    sep: &str,
383    items: impl IntoIterator<Item = T>,
384) -> std::fmt::Result {
385    let mut prefix = "";
386    for item in items {
387        write!(f, "{}{}", prefix, item)?;
388        prefix = sep;
389    }
390    Ok(())
391}

393impl fmt::Display for ParsedPTBCommand {
394    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
395        match self {
396            ParsedPTBCommand::TransferObjects(args, arg) => {
397                write!(f, "{TRANSFER_OBJECTS} [")?;
398                delimited_list(f, ", ", args.value.iter().map(|x| &x.value))?;
399                write!(f, "]")?;
400                write!(f, " {}", arg.value)
401            }
402            ParsedPTBCommand::SplitCoins(arg, args) => {
403                write!(f, "{SPLIT_COINS} {} [", arg.value)?;
404                delimited_list(f, ", ", args.value.iter().map(|x| &x.value))?;
405                write!(f, "]")
406            }
407            ParsedPTBCommand::MergeCoins(arg, args) => {
408                write!(f, "{MERGE_COINS} {} [", arg.value)?;
409                delimited_list(f, ", ", args.value.iter().map(|x| &x.value))?;
410                write!(f, "]")
411            }
412            ParsedPTBCommand::Assign(arg, arg_opt) => write!(
413                f,
414                "{ASSIGN} {}{}",
415                arg.value,
416                if let Some(arg) = arg_opt {
417                    format!(" {}", arg.value)
418                } else {
419                    "".to_string()
420                }
421            ),
422            ParsedPTBCommand::Publish(s) => write!(f, "{PUBLISH} {}", s.value),
423            ParsedPTBCommand::Upgrade(s, a) => write!(f, "{UPGRADE} {} {}", s.value, a.value),
424            ParsedPTBCommand::WarnShadows => write!(f, "{WARN_SHADOWS}"),
425            ParsedPTBCommand::Preview => write!(f, "{PREVIEW}"),
426            ParsedPTBCommand::MakeMoveVec(ty, args) => {
427                write!(f, "{MAKE_MOVE_VEC} <",)?;
428                write!(f, "{}", TyDisplay(&ty.value))?;
429                write!(f, "> [")?;
430                delimited_list(f, ", ", args.value.iter().map(|x| &x.value))?;
431                write!(f, "]")
432            }
433            ParsedPTBCommand::MoveCall(
434                sp!(
435                    _,
436                    ModuleAccess {
437                        address,
438                        module_name,
439                        function_name
440                    }
441                ),
442                tys,
443                args,
444            ) => {
445                let type_args = |f: &mut std::fmt::Formatter| match tys {
446                    Some(tys) => {
447                        write!(f, "<")?;
448                        delimited_list(f, ", ", tys.value.iter().map(TyDisplay))?;
449                        write!(f, ">")
450                    }
451                    None => Ok(()),
452                };
453                write!(
454                    f,
455                    "{MOVE_CALL} {}::{}::{}",
456                    address.value, module_name.value, function_name.value
457                )?;
458                type_args(f)?;
459
460                if !args.is_empty() {
461                    write!(f, " ")?;
462                }
463
464                delimited_list(f, " ", args.iter().map(|x| x.value.to_string()))
465            }
466        }
467    }
468}

472impl fmt::Display for TyDisplay<'_> {
473    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
474        use ParsedType::*;
475        match self.0 {
476            Address => write!(f, "address"),
477            U8 => write!(f, "u8"),
478            U16 => write!(f, "u16"),
479            U32 => write!(f, "u32"),
480            U64 => write!(f, "u64"),
481            U128 => write!(f, "u128"),
482            U256 => write!(f, "u256"),
483            Bool => write!(f, "bool"),
484            Signer => write!(f, "signer"),
485            Vector(ty) => write!(f, "vector<{}>", TyDisplay(ty)),
486            Struct(ParsedStructType {
487                fq_name:
488                    ParsedFqName {
489                        module: ParsedModuleId { address, name },
490                        name: struct_name,
491                    },
492                type_args,
493            }) => {
494                write!(f, "{address}::{name}::{struct_name}")?;
495                if type_args.is_empty() {
496                    Ok(())
497                } else {
498                    write!(f, "<")?;
499                    delimited_list(f, ", ", type_args.iter().map(TyDisplay))?;
500                    write!(f, ">")
501                }
502            }
503        }
504    }
