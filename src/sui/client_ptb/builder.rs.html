51// ===========================================================================
52// Object Resolution
53// ===========================================================================
54// We need to resolve the same argument in different ways depending on the context in which that
55// argument is used. For example, if we are using an object ID as an argument to a move call that
56// expects an object argument in that position, the object ID should be resolved to an object,
57// whereas if we are using the same object ID as an argument to a pure value, it should be resolved
58// to a pure value (e.g., the object ID itself). The different ways of resolving this is the
59// purpose of the `Resolver` trait -- different contexts will implement this trait in different
60// ways.
61
62/// A resolver is used to resolve arguments to a PTB. Depending on the context, we may resolve
63/// object IDs in different ways -- e.g., in a pure context they should be resolved to a pure
64/// value, whereas in an object context they should be resolved to the appropriate object argument.
65#[async_trait]
66trait Resolver<'a>: Send {
67    /// Resolve a pure value. This should almost always resolve to a pure value.
68    async fn pure(
69        &mut self,
70        builder: &mut PTBBuilder<'a>,
71        loc: Span,
72        argument: PTBArg,
73    ) -> PTBResult<Tx::Argument> {
74        let value = argument.to_pure_move_value(loc)?;
75        builder.ptb.pure(value).map_err(|e| err!(loc, "{e}"))
76    }
77
78    async fn resolve_object_id(
79        &mut self,
80        builder: &mut PTBBuilder<'a>,
81        loc: Span,
82        obj_id: ObjectID,
83    ) -> PTBResult<Tx::Argument>;
84
85    fn re_resolve(&self) -> bool {
86        false
87    }
88}
89
90/// A resolver that resolves object IDs to object arguments.
91/// * If `is_receiving` is true, then the object argument will be resolved to a receiving object
92///   argument.
93/// * If `is_mut` is true, then the object argument will be resolved to a mutable object argument.
94struct ToObject {
95    is_receiving: bool,
96    is_mut: bool,
97}
98
99impl Default for ToObject {
100    fn default() -> Self {
101        Self {
102            is_receiving: false,
103            is_mut: true,
104        }
105    }
106}
107
108impl ToObject {
109    fn new(is_receiving: bool, is_mut: bool) -> Self {
110        Self {
111            is_receiving,
112            is_mut,
113        }
114    }
115}

117#[async_trait]
118impl<'a> Resolver<'a> for ToObject {
119    async fn resolve_object_id(
120        &mut self,
121        builder: &mut PTBBuilder<'a>,
122        loc: Span,
123        obj_id: ObjectID,
124    ) -> PTBResult<Tx::Argument> {
125        // Get the object from the reader to get metadata about the object.
126        let obj = builder.get_object(obj_id, loc).await?;
127        let owner = obj
128            .owner
129            .clone()
130            .ok_or_else(|| err!(loc, "Unable to get owner info for object {obj_id}"))?;
131        let object_ref = obj.object_ref();
132        // Depending on the ownership of the object, we resolve it to different types of object
133        // arguments for the transaction.
134        let obj_arg = match owner {
135            Owner::AddressOwner(_) if self.is_receiving => ObjectArg::Receiving(object_ref),
136            Owner::Immutable | Owner::AddressOwner(_) => ObjectArg::ImmOrOwnedObject(object_ref),
137            Owner::Shared {
138                initial_shared_version,
139            }
140            | Owner::ConsensusAddressOwner {
141                start_version: initial_shared_version,
142                ..
143            } => ObjectArg::SharedObject {
144                id: object_ref.0,
145                initial_shared_version,
146                mutability: if self.is_mut {
147                    Tx::SharedObjectMutability::Mutable
148                } else {
149                    Tx::SharedObjectMutability::Immutable
150                },
151            },
152            Owner::ObjectOwner(_) => {
153                error!(loc => help: {
154                    "{obj_id} is an object-owned object, you can only use immutable, shared, or owned objects here."
155                }, "Cannot use an object-owned object as an argument")
156            }
157        };
158        // Insert the correct object arg that we built above into the transaction.
159        builder.ptb.obj(obj_arg).map_err(|e| err!(loc, "{e}"))
160    }
161
162    // We always re-resolve object IDs to object arguments if we need it mutably -- we could have
163    // added it earlier as an immutable argument.
164    fn re_resolve(&self) -> bool {
165        self.is_mut
166    }
167}

212/// The PTBBuilder struct is the main workhorse that transforms a sequence of `ParsedPTBCommand`s
213/// into an actual PTB that can be run. The main things to keep in mind are that this contains:
214/// - A way to handle identifiers -- note that we "lazily" resolve identifiers to arguments, so
215///   that the first usage of the identifier determines what it is resolved to. If an identifier is
216///   used in multiple positions at different resolutions (e.g., in one place as an object argument,
217///   and in another as a pure value), this will result in an error. This error can be avoided by
218///   creating another identifier for the second usage.
219/// - A way to resolve arguments -- this is done by calling `resolve` on a `PTBArg` and passing in
220///   appropriate context. The context is used to determine how to resolve the argument -- e.g., if
221///   an object ID should be resolved to a pure value or an object argument.
222/// - A way to bind the result of a command to an identifier.
223pub struct PTBBuilder<'a> {
224    /// A map from identifiers to address or move pkg. This is used to support address resolution
225    /// and also supports external address sources (e.g., keystore, mvr).
226    addresses: BTreeMap<String, AddressData>,
227    /// A map from identifiers to the file scopes in which they were declared. This is used
228    /// for reporting shadowing warnings.
229    identifiers: BTreeMap<String, Vec<Span>>,
230    /// The arguments that we need to resolve. This is a map from identifiers to the argument
231    /// values -- they haven't been resolved to a transaction argument yet.
232    arguments_to_resolve: BTreeMap<String, ArgWithHistory>,
233    /// The arguments that we have resolved. This is a map from identifiers to the actual
234    /// transaction arguments.
235    resolved_arguments: BTreeMap<String, Tx::Argument>,
236    /// Read API for reading objects from chain. Needed for object resolution.
237    reader: &'a ReadApi,
238    /// The last command that we have added. This is used to support assignment commands.
239    last_command: Option<Tx::Argument>,
240    /// The actual PTB that we are building up.
241    ptb: ProgrammableTransactionBuilder,
242    /// The list of errors that we have built up while processing commands. We do not report errors
243    /// eagerly but instead wait until we have processed all commands to report any errors.
244    errors: Vec<PTBError>,
245}
246
247#[derive(Debug, Clone, PartialEq, Eq)]
248enum ResolvedAccess {
249    ResultAccess(u16),
250    DottedString(String),
251}
252
253/// Hold a PTB argument, always remembering its most recent state even if it's already been
254/// resolved.
255#[derive(Debug)]
256enum ArgWithHistory {
257    Resolved(Spanned<PTBArg>),
258    Unresolved(Spanned<PTBArg>),
259}
260
261impl ArgWithHistory {
262    fn get_unresolved(&self) -> Option<&Spanned<PTBArg>> {
263        match self {
264            ArgWithHistory::Resolved(_) => None,
265            ArgWithHistory::Unresolved(x) => Some(x),
266        }
267    }
268
269    fn get(&self) -> &Spanned<PTBArg> {
270        match self {
271            ArgWithHistory::Resolved(x) => x,
272            ArgWithHistory::Unresolved(x) => x,
273        }
274    }
275
276    fn resolve(&mut self) {
277        *self = match self {
278            ArgWithHistory::Resolved(x) => ArgWithHistory::Resolved(x.clone()),
279            ArgWithHistory::Unresolved(x) => ArgWithHistory::Resolved(x.clone()),
280        }
281    }
282
283    fn is_resolved(&self) -> bool {
284        matches!(self, ArgWithHistory::Resolved(_))
285    }
286}

302    /// Finalize a PTB. If there were errors during the construction of the PTB these are returned
303    /// now. Otherwise, the PTB is finalized and returned.
304    /// If the warn_on_shadowing flag was set, then we will print warnings for any shadowed
305    /// variables that we encountered during the building of the PTB.
306    pub fn finish(
307        self,
308        warn_on_shadowing: bool,
309    ) -> (
310        Result<Tx::ProgrammableTransaction, Vec<PTBError>>,
311        Vec<PTBError>,
312    ) {
313        let mut warnings = vec![];
314        if warn_on_shadowing {
315            for (ident, commands) in self.identifiers.iter() {
316                if commands.len() == 1 {
317                    continue;
318                }
319
320                for (i, command_loc) in commands.iter().enumerate() {
321                    if i == 0 {
322                        warnings.push(PTBError {
323                            message: format!("Variable '{}' first declared here", ident),
324                            span: *command_loc,
325                            help: None,
326                            severity: Severity::Warning,
327                        });
328                    } else {
329                        warnings.push(PTBError {
330                            message: format!(
331                                "Variable '{}' used again here (shadowed) for the {} time.",
332                                ident, to_ordinal_contraction(i + 1)
333                            ),
334                            span: *command_loc,
335                            help: Some("You can either rename this variable, or do not \
336                                       pass the `warn-shadows` flag to ignore these types of errors.".to_string()),
337                            severity: Severity::Warning,
338                        });
339                    }
340                }
341            }
342        }
343
344        if !self.errors.is_empty() {
345            return (Err(self.errors), warnings);
346        }
347
348        let ptb = self.ptb.finish();
349        (Ok(ptb), warnings)
350    }

352    pub async fn build(
353        mut self,
354        program: Program,
355    ) -> (
356        Result<Tx::ProgrammableTransaction, Vec<PTBError>>,
357        Vec<PTBError>,
358    ) {
359        for command in program.commands.into_iter() {
360            self.handle_command(command).await;
361        }
362        self.finish(program.warn_shadows_set)
363    }

365    /// Add a single PTB command to the PTB that we are building up.
366    /// Errors are added to the `errors` field of the PTBBuilder.
367    async fn handle_command(&mut self, sp!(span, command): Spanned<ParsedPTBCommand>) {
368        if let Err(e) = self.handle_command_(span, command).await {

383    /// Declare a possible address binding. This is used to support address resolution. If the
384    /// `possible_addr` is not an address, then this is a no-op.
385    fn declare_possible_address_binding(&mut self, ident: String, possible_addr: &Spanned<PTBArg>) {
386        match possible_addr.value {
387            PTBArg::Address(addr) => {
388                self.addresses
389                    .insert(ident, AddressData::AccountAddress(addr.into_inner()));
390            }
391            PTBArg::Identifier(ref i) => {
392                // We do a one-hop resolution here to see if we can resolve the identifier to an
393                // externally-bound address (i.e., one coming in through the initial environment).
394                // This will also handle direct aliasing of addresses throughout the ptb.
395                // Note that we don't do this recursively so no need to worry about loops/cycles.
396                if let Some(addr) = self.addresses.get(i)
397                    && let Some(a) = addr.address()
398                {
399                    self.addresses.insert(ident, AddressData::AccountAddress(a));
400                }
401            }
402            // If we encounter a dotted string e.g., "foo.0" or "sui.io" or something like that
403            // this see if we can find an address for it in the environment and bind to it.
404            PTBArg::VariableAccess(ref head, ref fields) => {
405                let key = format!(
406                    "{}.{}",
407                    head.value,
408                    fields
409                        .iter()
410                        .map(|f| f.value.clone())
411                        .collect::<Vec<_>>()
412                        .join(".")
413                );
414                if let Some(addr) = self.addresses.get(&key) {
415                    self.addresses.insert(ident, addr.clone());
416                }
417            }
418            _ => (),
419        }
420    }

431    /// Resolves the argument to the move call based on the type information of the function being
432    /// called.
433    async fn resolve_move_call_arg(
434        &mut self,
435        view: &CompiledModule,
436        ty_args: &[TypeTag],
437        sp!(loc, arg): Spanned<PTBArg>,
438        param: &SignatureToken,
439    ) -> PTBResult<Tx::Argument> {
440        let layout = primitive_type(view, ty_args, param);
441
442        // If it's a primitive value, see if we've already resolved this argument. Otherwise, we
443        // need to resolve it.
444        if let Some(layout) = layout {
445            return self
446                .resolve(loc.wrap(arg), ToPure::new_from_layout(layout))
447                .await;
448        }
449
450        // Otherwise it's ambiguous what the value should be, and we need to turn to the signature
451        // to determine it.
452        let mut is_receiving = false;
453        // A value is mutable by default.
454        let mut is_mutable = true;
455
456        // traverse the types in the signature to see if the argument is an object argument or not,
457        // and also determine if it's a receiving argument or not.
458        for tok in param.preorder_traversal() {
459            match tok {
460                SignatureToken::Datatype(..) | SignatureToken::DatatypeInstantiation(..) => {
461                    is_receiving |= is_receiving_argument(view, tok);
462                }
463                SignatureToken::TypeParameter(idx) => {
464                    if *idx as usize >= ty_args.len() {
465                        error!(loc, "Not enough type parameters supplied for Move call");
466                    }
467                }
468                SignatureToken::Reference(_) => {
469                    is_mutable = false;
470                }
471                SignatureToken::MutableReference(_) => {
472                    // Not strictly needed, but for clarity
473                    is_mutable = true;
474                }
475                SignatureToken::Bool
476                | SignatureToken::U8
477                | SignatureToken::U64
478                | SignatureToken::U128
479                | SignatureToken::Address
480                | SignatureToken::Signer
481                | SignatureToken::Vector(_)
482                | SignatureToken::U16
483                | SignatureToken::U32
484                | SignatureToken::U256 => {
485                    is_mutable = false;
486                }
487            }
488        }
489
490        // Note: need to re-resolve an argument possibly since it may be used immutably first, and
491        // then mutably.
492        self.resolve(loc.wrap(arg), ToObject::new(is_receiving, is_mutable))
493            .await
494    }

496    /// Resolve the arguments to a Move call based on the type information about the function
497    /// being called.
498    async fn resolve_move_call_args(
499        &mut self,
500        package: MovePackage,
501        sp!(mloc, module_name): &Spanned<Identifier>,
502        sp!(floc, function_name): &Spanned<Identifier>,
503        ty_args: &[TypeTag],
504        args: Vec<Spanned<PTBArg>>,
505        package_name_loc: Span,
506    ) -> PTBResult<Vec<Tx::Argument>> {
507        let module = package
508            .deserialize_module(module_name, &BinaryConfig::standard())
509            .map_err(|e| {
510                let help_message = if package.serialized_module_map().is_empty() {
511                    Some("No modules found in this package".to_string())
512                } else {
513                    display_did_you_mean(find_did_you_means(
514                        module_name.as_str(),
515                        package.serialized_module_map().keys().map(|x| x.as_str()),
516                    ))
517                };
518                let e = err!(*mloc, "{e}");
519                if let Some(help_message) = help_message {
520                    e.with_help(help_message)
521                } else {
522                    e
523                }
524            })?;
525        let fdef = module
526            .function_defs
527            .iter()
528            .find(|fdef| {
529                module.identifier_at(module.function_handle_at(fdef.function).name)
530                    == function_name.as_ident_str()
531            })
532            .ok_or_else(|| {
533                let e = err!(
534                    *floc,
535                    "Could not resolve function '{}' in module '{}'",
536                    function_name,
537                    module_name
538                );
539                if let Some(help_message) = display_did_you_mean(find_did_you_means(
540                    function_name.as_str(),
541                    module.function_defs.iter().map(|fdef| {
542                        module
543                            .identifier_at(module.function_handle_at(fdef.function).name)
544                            .as_str()
545                    }),
546                )) {
547                    e.with_help(help_message)
548                } else {
549                    e
550                }
551            })?;
552        let function_signature = module.function_handle_at(fdef.function);
553        let parameters: Vec<_> = module
554            .signature_at(function_signature.parameters)
555            .0
556            .clone()
557            .into_iter()
558            .filter(|tok| matches!(TxContext::kind(&module, tok), TxContextKind::None))
559            .collect();
560
561        if parameters.len() != args.len() {
562            let loc = if args.is_empty() {
563                package_name_loc.widen(*mloc).widen(*floc)
564            } else {
565                args[0].span.widen_opt(args.last().map(|x| x.span))
566            };
567            error!(
568                loc,
569                "Expected {} argument{}, but got {}",
570                parameters.len(),
571                if parameters.len() == 1 { "" } else { "s" },
572                args.len()
573            );
574        }
575
576        let mut call_args = vec![];
577        for (param, arg) in parameters.iter().zip(args.into_iter()) {
578            let call_arg = self
579                .resolve_move_call_arg(&module, ty_args, arg, param)
580                .await?;
581            call_args.push(call_arg);
582        }
583        Ok(call_args)
584    }

586    fn resolve_variable_access(
587        &self,
588        head: &Spanned<String>,
589        fields: Vec<Spanned<String>>,
590    ) -> Spanned<ResolvedAccess> {
591        if fields.len() == 1 {
592            // Get the span and value of the field zero'th field. Safe since we just checked the
593            // length above. Since the length is 1, we know that the field is non-empty.
594            let sp!(field_loc, field) = &fields[0];
595            if let Ok(n) = field.parse::<u16>() {
596                return field_loc.wrap(ResolvedAccess::ResultAccess(n));
597            }
598        }
599        let tl_loc = head.span.widen_opt(fields.last().map(|x| x.span));
600        tl_loc.wrap(ResolvedAccess::DottedString(format!(
601            "{}.{}",
602            head.value,
603            fields
604                .into_iter()
605                .map(|f| f.value)
606                .collect::<Vec<_>>()
607                .join(".")
608        )))
609    }

611    /// Resolve an argument based on the argument value, and the `resolver` that is passed in.
612    #[async_recursion]
613    async fn resolve(
614        &mut self,
615        sp!(arg_loc, arg): Spanned<PTBArg>,
616        mut ctx: impl Resolver<'a> + 'async_recursion,
617    ) -> PTBResult<Tx::Argument> {
618        match arg {
619            a @ (PTBArg::Bool(_)
620            | PTBArg::U8(_)
621            | PTBArg::U16(_)
622            | PTBArg::U32(_)
623            | PTBArg::U64(_)
624            | PTBArg::U128(_)
625            | PTBArg::U256(_)
626            | PTBArg::InferredNum(_)
627            | PTBArg::String(_)
628            | PTBArg::Option(_)
629            | PTBArg::Vector(_)) => ctx.pure(self, arg_loc, a).await,
630            PTBArg::Gas => Ok(Tx::Argument::GasCoin),
631            // NB: the ordering of these lines is important so that shadowing is properly
632            // supported.
633            // If we encounter an identifier that we have not already resolved, then we resolve the
634            // value and return it.
635            PTBArg::Identifier(i)
636                if self
637                    .arguments_to_resolve
638                    .get(&i)
639                    .is_some_and(|arg_hist| !arg_hist.is_resolved()) =>
640            {
641                let arg_hist = self.arguments_to_resolve.get(&i).unwrap();
642                let arg = arg_hist.get().clone();
643                let resolved = self.resolve(arg, ctx).await?;
644                self.arguments_to_resolve.get_mut(&i).unwrap().resolve();
645                self.resolved_arguments.insert(i, resolved);
646                Ok(resolved)
647            }
648            // If the identifier does not need to be resolved, but has already been resolved, then
649            // we return the resolved value.
650            PTBArg::Identifier(i) if self.resolved_arguments.contains_key(&i) => {
651                if ctx.re_resolve() && self.arguments_to_resolve.contains_key(&i) {
652                    self.resolve(self.arguments_to_resolve[&i].get().clone(), ctx)
653                        .await
654                } else {
655                    Ok(self.resolved_arguments[&i])
656                }
657            }
658            // Lastly -- look to see if this is an address that has been either declared in scope,
659            // or that is coming from an external source (e.g., the keystore).
660            PTBArg::Identifier(i) if self.addresses.contains_key(&i) => {
661                // We now have a location for this address (which may have come from the keystore
662                // so we didnt' have an address for it before), so we tag it with its first usage
663                // location put it in the arguments to resolve and resolve away.
664                let addr = self.addresses[&i]
665                    .address()
666                    .ok_or_else(|| err!(arg_loc, "Expected an address"))?;
667                let arg = arg_loc.wrap(PTBArg::Address(NumericalAddress::new(
668                    addr.into_bytes(),
669                    NumberFormat::Hex,
670                )));
671                self.arguments_to_resolve
672                    .insert(i.clone(), ArgWithHistory::Unresolved(arg.clone()));
673                self.resolve(arg_loc.wrap(PTBArg::Identifier(i)), ctx).await
674            }
675            PTBArg::Address(addr) => {
676                let object_id = ObjectID::from_address(addr.into_inner());
677                ctx.resolve_object_id(self, arg_loc, object_id).await
678            }
679            PTBArg::VariableAccess(head, fields) => {
680                // Since keystore aliases can contain dots, we need to resolve these/disambiguate
681                // them as best as possible here.
682                // First: See if structurally we know whether this is a dotted access or a
683                // string(alias) containing dot(s). If there is more than one dotted access, or
684                // if the field(s) are not all numbers, then we assume it's a alias.
685                match self.resolve_variable_access(&head, fields) {
686                    sp!(l, ResolvedAccess::DottedString(string)) => {
687                        self.resolve(l.wrap(PTBArg::Identifier(string)), ctx).await
688                    }
689                    sp!(_, ResolvedAccess::ResultAccess(access)) => {
690                        match self.resolved_arguments.get(&head.value) {
691                            Some(Tx::Argument::Result(u)) => {
692                                Ok(Tx::Argument::NestedResult(*u, access))
693                            }
694                            // Tried to access into a nested result, input, or gascoin
695                            Some(
696                                x @ (Tx::Argument::NestedResult(..)
697                                | Tx::Argument::Input(..)
698                                | Tx::Argument::GasCoin),
699                            ) => {
700                                error!(
701                                    arg_loc,
702                                    "Tried to access a nested result, input, or gascoin {}: {}",
703                                    head.value,
704                                    x,
705                                );
706                            }
707                            // Unable to resolve, so now see if we can resolve it to an alias, i.e.,
708                            // handle a alias that looks something like `foo.0`
709                            None => {
710                                let formatted_access = format!("{}.{}", head.value, access);
711                                if !self.addresses.contains_key(&formatted_access)
712                                    && !self.identifiers.contains_key(&formatted_access)
713                                {
714                                    match self.did_you_mean_identifier(&head.value) {
715                                        Some(similars) => {
716                                            error!(
717                                                head.span => help: { "{}", similars },
718                                                "Tried to access an unresolved identifier: {}", head.value
719                                            );
720                                        }
721                                        None => {
722                                            error!(
723                                                head.span,
724                                                "Tried to access an unresolved identifier: {}",
725                                                head.value
726                                            );
727                                        }
728                                    }
729                                }
730                                self.resolve(
731                                    arg_loc.wrap(PTBArg::Identifier(formatted_access.clone())),
732                                    ctx,
733                                )
734                                .await
735                            }
736                        }
737                    }
738                }
739            }
740            // Unable to resolve an identifer to anything at this point -- error and see if we can
741            // find a similar identifier to suggest.
742            PTBArg::Identifier(i) => match self.did_you_mean_identifier(&i) {
743                Some(similars) => {
744                    error!(arg_loc => help: { "{}", similars }, "Unresolved identifier: '{}'", i)
745                }
746                None => error!(arg_loc, "Unresolved identifier: '{}'", i),
747            },
748        }
749    }

780    /// Add a single PTB command to the PTB that we are building up. This is the workhorse of it
781    /// all.
782    async fn handle_command_(
783        &mut self,
784        cmd_span: Span,
785        command: ParsedPTBCommand,
786    ) -> PTBResult<()> {
787        // let sp!(cmd_span, tok) = &command.name;
788        match command {
789            ParsedPTBCommand::TransferObjects(obj_args, to_address) => {
790                let to_arg = self
791                    .resolve(to_address, ToPure::new(TypeTag::Address))
792                    .await?;
793                let mut transfer_args = vec![];
794                for o in obj_args.value.into_iter() {
795                    let arg = self.resolve(o, ToObject::default()).await?;
796                    transfer_args.push(arg);
797                }
798                self.last_command = Some(
799                    self.ptb
800                        .command(Tx::Command::TransferObjects(transfer_args, to_arg)),
801                );
802            }
803            ParsedPTBCommand::Assign(sp!(ident_loc, i), None) => {
804                let Some(prev_ptb_arg) = self.last_command.take() else {
805                    error!(
806                        ident_loc => help: {
807                           "This is most likely because the previous command did not \
808                           produce a result. E.g., '{ASSIGN}' or '{GAS_BUDGET}' commands do not produce results."
809
810                        },
811                        "Cannot assign a value to this variable."
812                    );
813                };
814                self.declare_identifier(i.clone(), ident_loc);
815                self.resolved_arguments.insert(i, prev_ptb_arg);
816            }
817            ParsedPTBCommand::Assign(sp!(ident_loc, i), Some(arg_w_loc)) => {
818                self.declare_identifier(i.clone(), ident_loc);
819                self.declare_possible_address_binding(i.clone(), &arg_w_loc);
820                self.arguments_to_resolve
821                    .insert(i, ArgWithHistory::Unresolved(arg_w_loc));
822            }
823            ParsedPTBCommand::MakeMoveVec(sp!(ty_loc, ty_arg), sp!(_, args)) => {
824                let ty_arg = into_type_tag(&self.addresses, ty_arg, &resolve_address)
825                    .map_err(|e| err!(ty_loc, "{e}"))?;
826                let mut vec_args: Vec<Tx::Argument> = vec![];
827                if is_primitive_type_tag(&ty_arg) {
828                    for arg in args.into_iter() {
829                        let arg = self.resolve(arg, ToPure::new(ty_arg.clone())).await?;
830                        vec_args.push(arg);
831                    }
832                } else {
833                    for arg in args.into_iter() {
834                        let arg = self.resolve(arg, ToObject::default()).await?;
835                        vec_args.push(arg);
836                    }
837                }
838                let res = self
839                    .ptb
840                    .command(Tx::Command::make_move_vec(Some(ty_arg), vec_args));
841                self.last_command = Some(res);
842            }
843            ParsedPTBCommand::SplitCoins(pre_coin, sp!(_, amounts)) => {
844                let coin = self.resolve(pre_coin, ToObject::default()).await?;
845                let mut args = vec![];
846                for arg in amounts.into_iter() {
847                    let arg = self.resolve(arg, ToPure::new(TypeTag::U64)).await?;
848                    args.push(arg);
849                }
850                let res = self.ptb.command(Tx::Command::SplitCoins(coin, args));
851                self.last_command = Some(res);
852            }
853            ParsedPTBCommand::MergeCoins(pre_coin, sp!(_, coins)) => {
854                let coin = self.resolve(pre_coin, ToObject::default()).await?;
855                let mut args = vec![];
856                for arg in coins.into_iter() {
857                    let arg = self.resolve(arg, ToObject::default()).await?;
858                    args.push(arg);
859                }
860                let res = self.ptb.command(Tx::Command::MergeCoins(coin, args));
861                self.last_command = Some(res);
862            }
863            ParsedPTBCommand::MoveCall(
864                sp!(
865                    mod_access_loc,
866                    PTBModuleAccess {
867                        address,
868                        module_name,
869                        function_name,
870                    }
871                ),
872                in_ty_args,
873                args,
874            ) => {
875                let resolved_address = try_resolve_parsed_address(&self.addresses, &address)
876                    .map_err(|e| {
877                    let e = err!(address.span, "{e}");
878                    if let ParsedAddress::Named(name) = address.value {
879                        e.with_help(
880                            format!("This is most likely because the named address '{name}' is not in scope. \
881                                     You can either bind a variable to the address that you want to use or use the address in the command."))
882                    } else {
883                        e
884                    }
885                })?;
886
887                let mut ty_args = vec![];
888                if let Some(sp!(ty_loc, in_ty_args)) = in_ty_args {
889                    for t in in_ty_args.into_iter() {
890                        ty_args.push(
891                            into_type_tag(&self.addresses, t, &resolve_address)
892                                .map_err(|e| err!(ty_loc, "{e}"))?,
893                        )
894                    }
895                }
896
897                let package_id = ObjectID::from_address(resolved_address);
898                let package = self.resolve_to_package(package_id, address.span).await?;
899                let args = self
900                    .resolve_move_call_args(
901                        package,
902                        &module_name,
903                        &function_name,
904                        &ty_args,
905                        args,
906                        mod_access_loc,
907                    )
908                    .await?;
909                let res = self.ptb.command(Tx::Command::move_call(
910                    package_id,
911                    module_name.value,
912                    function_name.value,
913                    ty_args,
914                    args,
915                ));
916                self.last_command = Some(res);
917            }
918            ParsedPTBCommand::Publish(sp!(pkg_loc, package_path)) => {
919                let chain_id = self.reader.get_chain_identifier().await.ok();
920                let build_config = MoveBuildConfig::default();
921                let package_path = Path::new(&package_path);
922                let build_config = resolve_lock_file_path(build_config.clone(), Some(package_path))
923                    .map_err(|e| err!(pkg_loc, "{e}"))?;
924                let previous_id = if let Some(ref chain_id) = chain_id {
925                    sui_package_management::set_package_id(
926                        package_path,
927                        build_config.install_dir.clone(),
928                        chain_id,
929                        AccountAddress::ZERO,
930                    )
931                    .map_err(|e| err!(pkg_loc, "{e}"))?
932                } else {
933                    None
934                };
935                // Restore original ID, then check result.
936                if let (Some(chain_id), Some(previous_id)) = (chain_id, previous_id) {
937                    let _ = sui_package_management::set_package_id(
938                        package_path,
939                        build_config.install_dir.clone(),
940                        &chain_id,
941                        previous_id,
942                    )
943                    .map_err(|e| err!(pkg_loc, "{e}"))?;
944                }
945                let compiled_package = compile_package(
946                    self.reader,
947                    build_config.clone(),
948                    package_path,
949                    false, /* with_unpublished_dependencies */
950                    false, /* skip_dependency_verification */
951                )
952                .await
953                .map_err(|e| err!(pkg_loc, "{e}"))?;
954
955                let compiled_modules = compiled_package.get_package_bytes(false);
956
957                let res = self.ptb.publish_upgradeable(
958                    compiled_modules,
959                    compiled_package.get_published_dependencies_ids(),
960                );
961                self.last_command = Some(res);
962            }
963            // Update this command to not do as many things. It should result in a single command.
964            ParsedPTBCommand::Upgrade(sp!(path_loc, package_path), mut arg) => {
965                if let sp!(loc, PTBArg::Identifier(id)) = arg {
966                    arg = self
967                        .arguments_to_resolve
968                        .get(&id)
969                        .and_then(|x| x.get_unresolved())
970                        .ok_or_else(|| err!(loc, "Unable to find object ID argument"))?
971                        .clone();
972                }
973                let (cap_loc, upgrade_cap_id) = match arg {
974                    sp!(loc, PTBArg::Address(id)) => (loc, id),
975                    sp!(loc, _) => {
976                        error!(loc, "Expected upgrade capability object ID");
977                    }
978                };
979
980                let upgrade_cap_arg = self
981                    .resolve(
982                        cap_loc.wrap(PTBArg::Address(upgrade_cap_id)),
983                        ToObject::default(),
984                    )
985                    .await?;
986
987                let chain_id = self.reader.get_chain_identifier().await.ok();
988                let build_config = MoveBuildConfig::default();
989                let package_path = Path::new(&package_path);
990                let build_config = resolve_lock_file_path(build_config.clone(), Some(package_path))
991                    .map_err(|e| err!(path_loc, "{e}"))?;
992                let previous_id = if let Some(ref chain_id) = chain_id {
993                    sui_package_management::set_package_id(
994                        package_path,
995                        build_config.install_dir.clone(),
996                        chain_id,
997                        AccountAddress::ZERO,
998                    )
999                    .map_err(|e| err!(path_loc, "{e}"))?
1000                } else {
1001                    None
1002                };
1003                // Restore original ID, then check result.
1004                if let (Some(chain_id), Some(previous_id)) = (chain_id, previous_id) {
1005                    let _ = sui_package_management::set_package_id(
1006                        package_path,
1007                        build_config.install_dir.clone(),
1008                        &chain_id,
1009                        previous_id,
1010                    )
1011                    .map_err(|e| err!(path_loc, "{e}"))?;
1012                }
1013
1014                let (upgrade_policy, compiled_package) = upgrade_package(
1015                    self.reader,
1016                    build_config.clone(),
1017                    package_path,
1018                    ObjectID::from_address(upgrade_cap_id.into_inner()),
1019                    false, /* with_unpublished_dependencies */
1020                    false, /* skip_dependency_verification */
1021                    None,
1022                )
1023                .await
1024                .map_err(|e| err!(path_loc, "{e}"))?;
1025
1026                let package_digest = compiled_package.get_package_digest(false);
1027                let package_id = compiled_package
1028                    .published_at
1029                    .as_ref()
1030                    .map_err(|e| err!(path_loc, "{e}"))?;
1031                let compiled_modules = compiled_package.get_package_bytes(false);
1032                // let (package_id, compiled_modules, dependencies, package_digest, upgrade_policy, _) =
1033                //     upgrade_result.map_err(|e| err!(path_loc, "{e}"))?;
1034
1035                let upgrade_arg = self
1036                    .ptb
1037                    .pure(upgrade_policy)
1038                    .map_err(|e| err!(cmd_span, "{e}"))?;
1039                let digest_arg = self
1040                    .ptb
1041                    // .to_vec() is necessary to get the length prefix
1042                    .pure(package_digest.to_vec())
1043                    .map_err(|e| err!(cmd_span, "{e}"))?;
1044                let upgrade_ticket = self.ptb.command(Tx::Command::move_call(
1045                    SUI_FRAMEWORK_PACKAGE_ID,
1046                    ident_str!("package").to_owned(),
1047                    ident_str!("authorize_upgrade").to_owned(),
1048                    vec![],
1049                    vec![upgrade_cap_arg, upgrade_arg, digest_arg],
1050                ));
1051                let upgrade_receipt = self.ptb.upgrade(
1052                    *package_id,
1053                    upgrade_ticket,
1054                    compiled_package
1055                        .dependency_ids
1056                        .published
1057                        .into_values()
1058                        .collect(),
1059                    compiled_modules,
1060                );
1061                let res = self.ptb.command(Tx::Command::move_call(
1062                    SUI_FRAMEWORK_PACKAGE_ID,
1063                    ident_str!("package").to_owned(),
1064                    ident_str!("commit_upgrade").to_owned(),
1065                    vec![],
1066                    vec![upgrade_cap_arg, upgrade_receipt],
1067                ));
1068                self.last_command = Some(res);
1069            }
1070            ParsedPTBCommand::WarnShadows => {}
1071            ParsedPTBCommand::Preview => {}
1072        }
1073        Ok(())
1074    }
1075}

1080pub fn to_ordinal_contraction(num: usize) -> String {
1081    let suffix = match num % 100 {
1082        // exceptions
1083        11..=13 => "th",
1084        _ => match num % 10 {
1085            1 => "st",
1086            2 => "nd",
1087            3 => "rd",
1088            _ => "th",
1089        },
1090    };
1091    format!("{}{}", num, suffix)
1092}

1094pub(crate) fn find_did_you_means<'a>(
1095    needle: &str,
1096    haystack: impl IntoIterator<Item = &'a str>,
1097) -> Vec<&'a str> {
1098    let mut results = Vec::new();
1099    let mut best_distance = usize::MAX;
1100
1101    for item in haystack {
1102        let distance = edit_distance(needle, item);
1103
1104        match distance.cmp(&best_distance) {
1105            std::cmp::Ordering::Less => {
1106                best_distance = distance;
1107                results.clear();
1108                results.push(item);
1109            }
1110            std::cmp::Ordering::Equal => {
1111                results.push(item);
1112            }
1113            std::cmp::Ordering::Greater => {}
1114        }
1115    }
1116
1117    results
1118}

1120pub(crate) fn display_did_you_mean<S: AsRef<str> + std::fmt::Display>(
1121    possibles: Vec<S>,
1122) -> Option<String> {
1123    if possibles.is_empty() {
1124        return None;
1125    }
1126
1127    let mut strs = vec![];
1128
1129    let preposition = if possibles.len() == 1 {
1130        "Did you mean "
1131    } else {
1132        "Did you mean one of "
1133    };
1134
1135    let len = possibles.len();
1136    for (i, possible) in possibles.into_iter().enumerate() {
1137        if i == len - 1 && len > 1 {
1138            strs.push(format!("or '{}'", possible));
1139        } else {
1140            strs.push(format!("'{}'", possible));
1141        }
1142    }
1143
1144    Some(format!("{preposition}{}?", strs.join(", ")))
1145}

1147// This lint is disabled because it's not good and doesn't look at what you're actually
1148// iterating over. This seems to be a common problem with this lint.
1149// See e.g., https://github.com/rust-lang/rust-clippy/issues/6075
1150#[allow(clippy::needless_range_loop)]
1151fn edit_distance(a: &str, b: &str) -> usize {
1152    let mut cache = vec![vec![0; b.len() + 1]; a.len() + 1];
1153
1154    for i in 0..=a.len() {
1155        cache[i][0] = i;
1156    }
1157
1158    for j in 0..=b.len() {
1159        cache[0][j] = j;
1160    }
1161
1162    for (i, char_a) in a.chars().enumerate().map(|(i, c)| (i + 1, c)) {
1163        for (j, char_b) in b.chars().enumerate().map(|(j, c)| (j + 1, c)) {
1164            if char_a == char_b {
1165                cache[i][j] = cache[i - 1][j - 1];
1166            } else {
1167                let insert = cache[i][j - 1];
1168                let delete = cache[i - 1][j];
1169                let replace = cache[i - 1][j - 1];
1170
1171                cache[i][j] = 1 + std::cmp::min(insert, std::cmp::min(delete, replace));
1172            }
1173        }
1174    }
1175
1176    cache[a.len()][b.len()]
1177}

1179/// Check if it's a MVR name. MVR names must contain a / in them so as we already
1180/// parsed the name, we can just check if it contains a / in it. A keystore alias cannot
1181/// contain a slash, and similarly a numerical address or named address cannot have a / in it.
1182pub fn is_mvr_name(name: &str) -> bool {
1183    name.contains('/') && (name.contains("@") || name.contains(".sui"))
1184}

1186pub fn into_struct_tag(
1187    addresses: &BTreeMap<String, AddressData>,
1188    parsed_struct_type: ParsedStructType,
1189    mapping: &(impl Fn(&str) -> Option<AccountAddress> + std::marker::Sync),
1190) -> anyhow::Result<StructTag> {
1191    let fq_name = parsed_struct_type.fq_name;
1192    let type_args = parsed_struct_type.type_args;
1193
1194    let address = match fq_name.module.address {
1195        ParsedAddress::Named(name) if is_mvr_name(&name) => {
1196            let addr = addresses.get(&name);
1197            let Some(AddressData::MovePackage((pkg, type_origin_map))) = addr else {
1198                return Err(anyhow::anyhow!(
1199                    "Expected that {name} is an actual Move package"
1200                ));
1201            };
1202            let Some(address) =
1203                type_origin_map.get(&(fq_name.module.name.clone(), fq_name.name.clone()))
1204            else {
1205                return Err(anyhow::anyhow!(
1206                    "Cannot find the type {} in module {} in package {}({})",
1207                    fq_name.name,
1208                    fq_name.module.name,
1209                    name,
1210                    pkg.id()
1211                ));
1212            };
1213            AccountAddress::from_bytes(address.into_bytes()).map_err(|_| {
1214                anyhow::anyhow!("Cannot convert ObjectID {address} into AccountAddress")
1215            })
1216        }
1217        _ => fq_name.module.address.into_account_address(mapping),
1218    }?;
1219
1220    Ok(StructTag {
1221        address,
1222        module: Identifier::new(fq_name.module.name)?,
1223        name: Identifier::new(fq_name.name)?,
1224        type_params: type_args
1225            .into_iter()
1226            .map(|t| into_type_tag(addresses, t, mapping))
1227            .collect::<anyhow::Result<_>>()?,
1228    })
1229}

1231/// Given a parsed type, resolve it to a TypeTag. This custom function makes usage of the custom
1232/// into struct tag function that makes use of the already resolved MVR data, if needed.
1233pub fn into_type_tag(
1234    addresses: &BTreeMap<String, AddressData>,
1235    parsed_type: ParsedType,
1236    mapping: &(impl Fn(&str) -> Option<AccountAddress> + std::marker::Sync),
1237) -> anyhow::Result<TypeTag> {
1238    Ok(match parsed_type {
1239        ParsedType::U8 => TypeTag::U8,
1240        ParsedType::U16 => TypeTag::U16,
1241        ParsedType::U32 => TypeTag::U32,
1242        ParsedType::U64 => TypeTag::U64,
1243        ParsedType::U128 => TypeTag::U128,
1244        ParsedType::U256 => TypeTag::U256,
1245        ParsedType::Bool => TypeTag::Bool,
1246        ParsedType::Address => TypeTag::Address,
1247        ParsedType::Signer => TypeTag::Signer,
1248        ParsedType::Vector(inner) => {
1249            TypeTag::Vector(Box::new(into_type_tag(addresses, *inner, mapping)?))
1250        }
1251        ParsedType::Struct(s) => TypeTag::Struct(Box::new(into_struct_tag(addresses, s, mapping)?)),
1252    })
1253}

1255/// Try to resolve a parsed address by using the already known mvr data and the keystore (addresses
1256/// ) or the known package names (e.g., sui, deepbook, etc) by calling `resolve_address`.
1257fn try_resolve_parsed_address(
1258    addresses: &BTreeMap<String, AddressData>,
1259    address: &Spanned<ParsedAddress>,
1260) -> Result<AccountAddress> {
1261    match &address.value {
1262        ParsedAddress::Named(name) => {
1263            let address = if let Some(addr) = addresses.get(name) {
1264                addr.address()
1265            } else {
1266                resolve_address(name)
1267            };
1268
1269            address.ok_or_else(|| anyhow!("Unbound named address: '{name}'"))
1270        }
1271
1272        ParsedAddress::Numerical(x) => Ok(x.into_inner()),
1273    }
1274}

1276/// Try to resolve an ObjectID to a MovePackage
1277pub async fn resolve_package(
1278    reader: &ReadApi,
1279    package_id: ObjectID,
1280    loc: Span,
1281) -> Result<MovePackage, PTBError> {
1282    let object = reader
1283        .get_object_with_options(package_id, SuiObjectDataOptions::bcs_lossless())
1284        .await
1285        .map_err(|e| err!(loc, "{e}"))?
1286        .into_object()
1287        .map_err(|e| err!(loc, "{e}"))?;
1288
1289    let Some(SuiRawData::Package(package)) = object.bcs else {
1290        error!(
1291            loc,
1292            "BCS field in object '{}' is missing or not a package.", package_id
1293        );
1294    };
1295
1296    MovePackage::new(
1297        package.id,
1298        package.version,
1299        package.module_map,
1300        // This package came from on-chain and the tool runs locally, so don't worry about
1301        // trying to enforce the package size limit.
1302        u64::MAX,
1303        package.type_origin_table,
1304        package.linkage_table,
1305    )
1306    .map_err(|e| err!(loc, "{e}"))
1307}
