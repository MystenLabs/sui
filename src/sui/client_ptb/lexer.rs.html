11pub struct Lexer<'l, I: Iterator<Item = &'l str>> {
12    pub buf: &'l str,
13    pub tokens: I,
14    pub offset: usize,
15    pub done: Option<Spanned<Lexeme<'l>>>,
16}
17
18impl<'l, I: Iterator<Item = &'l str>> Lexer<'l, I> {
19    pub fn new(mut tokens: I) -> Option<Self> {
20        let buf = tokens.next()?;
21
22        Some(Self {
23            buf,
24            tokens,
25            offset: 0,
26            done: None,
27        })
28    }
29
30    /// Returns the next character in the current shell token, along with the byte offset it ends
31    /// at, or None if the current shell token is empty.
32    fn next_char_boundary(&self) -> Option<(usize, char)> {
33        let mut chars = self.buf.char_indices();
34        let (_, c) = chars.next()?;
35        let ix = chars.next().map_or(self.buf.len(), |(ix, _)| ix);
36        Some((ix, c))
37    }
38
39    /// Repeatedly consume whitespace, stopping only if you hit a non-whitespace character, or the
40    /// end of the shell token stream.
41    fn eat_whitespace(&mut self) {
42        loop {
43            if let Some((ix, c)) = self.next_char_boundary() {
44                if c.is_whitespace() {
45                    self.buf = &self.buf[ix..];
46                    self.offset += ix;
47                } else {
48                    break;
49                }
50            } else if let Some(next) = self.tokens.next() {
51                self.offset += 1; // +1 for the space between tokens
52                self.buf = next;
53            } else {
54                break;
55            };
56        }
57    }
58
59    /// Checks whether the current shell token starts with the prefix `patt`, and consumes it if so,
60    /// returning a spanned slice of the consumed prefix.
61    fn eat_prefix(&mut self, patt: &str) -> Option<Spanned<&'l str>> {
62        let start = self.offset;
63
64        let rest = self.buf.strip_prefix(patt)?;
65
66        let len = self.buf.len() - rest.len();
67        let value = &self.buf[..len];
68        self.offset += len;
69        self.buf = rest;
70
71        let span = Span {
72            start,
73            end: self.offset,
74        };
75        Some(Spanned { span, value })
76    }
77
78    /// Checks whether the current shell token starts with at least one character that satisfies
79    /// `pred`. Consumes all such characters from the front of the shell token, returning a spanned
80    /// slice of the consumed prefix.
81    fn eat_while(&mut self, pred: impl FnMut(char) -> bool) -> Option<Spanned<&'l str>> {
82        let start = self.offset;
83
84        let rest = self.buf.trim_start_matches(pred);
85        if self.buf == rest {
86            return None;
87        };
88
89        let len = self.buf.len() - rest.len();
90        let value = &self.buf[..len];
91        self.offset += len;
92        self.buf = rest;
93
94        let span = Span {
95            start,
96            end: self.offset,
97        };
98        Some(Spanned { span, value })
99    }
100
101    /// Consume the whole next shell token (assumes the current shell token has already been
102    /// consumed).
103    fn eat_token(&mut self) -> Option<Spanned<&'l str>> {
104        debug_assert!(self.buf.is_empty());
105        let start = self.offset + 1;
106        let value = self.tokens.next()?;
107        self.offset += value.len() + 1;
108
109        let span = Span {
110            start,
111            end: self.offset,
112        };
113        Some(Spanned { span, value })
114    }
115
116    /// Look at the next character in the current shell token without consuming it, if it exists.
117    fn peek(&self) -> Option<Spanned<&'l str>> {
118        let start = self.offset;
119        let (ix, _) = self.next_char_boundary()?;
120
121        let value = &self.buf[..ix];
122        let span = Span {
123            start,
124            end: start + ix,
125        };
126        Some(Spanned { span, value })
127    }
128
129    /// Consume the next character in the current shell token, assuming there is one.
130    fn bump(&mut self) {
131        if let Some((ix, _)) = self.next_char_boundary() {
132            self.buf = &self.buf[ix..];
133            self.offset += ix;
134        }
135    }
136
137    /// Tokenize a string at the prefix of the current shell token. `start` is the spanned slice
138    /// containing the initial quote character, which also specifies the terminating quote
139    /// character.
140    ///
141    /// A string that is not terminated in the same shell token it was started in is tokenized as an
142    /// `UnfinishedString`, even if it would have been terminated in a following shell token.
143    fn string(&mut self, start: Spanned<&'l str>) -> Spanned<Lexeme<'l>> {
144        self.bump();
145        let sp!(sp, quote) = start;
146
147        let mut escaped = false;
148        let content = self
149            .eat_while(|c| {
150                if escaped {
151                    escaped = false;
152                    true
153                } else if c == '\\' {
154                    escaped = true;
155                    true
156                } else {
157                    !quote.starts_with(c)
158                }
159            })
160            .unwrap_or(Spanned {
161                span: Span {
162                    start: sp.end,
163                    end: sp.end,
164                },
165                value: "",
166            })
167            .widen(start);
168
169        let Some(end) = self.eat_prefix(quote) else {
170            let error = content.map(|src| Lexeme(T::UnfinishedString, src));
171            self.done = Some(error);
172            return error;
173        };
174
175        content.widen(end).map(|src| Lexeme(T::String, src))
176    }
177
178    /// Signal that `c` is an unexpected token, and trigger the lexer's error flag, to prevent
179    /// further iteration.
180    fn unexpected(&mut self, c: Spanned<&'l str>) -> Spanned<Lexeme<'l>> {
181        let error = c.map(|src| Lexeme(T::Unexpected, src));
182        self.done = Some(error);
183        error
184    }
185
186    /// Signal that the lexer has experienced an unexpected, early end-of-file, and trigger the
187    /// lexer's error flag, to prevent further iteration.
188    fn done(&mut self, token: T) -> Spanned<Lexeme<'l>> {
189        let error = self.offset().wrap(Lexeme(token, ""));
190        self.done = Some(error);
191        error
192    }
193
194    /// Span pointing to the current offset in the input.
195    fn offset(&self) -> Span {
196        Span {
197            start: self.offset,
198            end: self.offset,
199        }
200    }
201}
202
203impl<'l, I: Iterator<Item = &'l str>> Iterator for Lexer<'l, I> {
204    type Item = Spanned<Lexeme<'l>>;
205
206    fn next(&mut self) -> Option<Self::Item> {
207        // Lexer has been expended, repeatedly return the terminal token.
208        if let Some(done) = self.done {
209            return Some(done);
210        }
211
212        self.eat_whitespace();
213
214        let Some(c) = self.peek() else {
215            return Some(self.done(T::Eof));
216        };
217
218        macro_rules! token {
219            ($t:expr) => {{
220                self.bump();
221                c.map(|src| Lexeme($t, src))
222            }};
223        }
224
225        Some(match c {
226            // Single character tokens
227            sp!(_, ",") => token!(T::Comma),
228            sp!(_, "[") => token!(T::LBracket),
229            sp!(_, "]") => token!(T::RBracket),
230            sp!(_, "(") => token!(T::LParen),
231            sp!(_, ")") => token!(T::RParen),
232            sp!(_, "<") => token!(T::LAngle),
233            sp!(_, ">") => token!(T::RAngle),
234            sp!(_, "@") => token!(T::At),
235            sp!(_, ".") => token!(T::Dot),
236            sp!(_, "/") => token!(T::ForwardSlash),
237
238            sp!(_, "'" | "\"") => self.string(c),
239
240            sp!(_, ":") => 'colon: {
241                let Some(sp) = self.eat_prefix("::") else {
242                    break 'colon self.unexpected(c);
243                };
244
245                sp.map(|src| Lexeme(T::ColonColon, src))
246            }
247
248            sp!(_, c) if c.chars().next().is_some_and(is_ident_start) => {
249                let Some(ident) = self.eat_while(is_ident_continue) else {
250                    unreachable!("is_ident_start implies is_ident_continue");
251                };
252
253                ident.map(|src| Lexeme(T::Ident, src))
254            }
255
256            sp!(_, "0") => 'zero: {
257                let Some(prefix) = self.eat_prefix("0x") else {
258                    break 'zero token!(T::Number);
259                };
260
261                let Some(digits) = self.eat_while(is_hex_continue) else {
262                    break 'zero self.unexpected(prefix);
263                };
264
265                digits.widen(prefix).map(|src| Lexeme(T::HexNumber, src))
266            }
267
268            sp!(_, n) if n.chars().next().is_some_and(is_number_start) => {
269                let Some(num) = self.eat_while(is_number_continue) else {
270                    unreachable!("is_number_start implies is_number_continue");
271                };
272
273                num.map(|src| Lexeme(T::Number, src))
274            }
275
276            sp!(_, "-") => 'command: {
277                self.bump();
278                let Some(next) = self.peek() else {
279                    break 'command self.unexpected(c);
280                };
281
282                match next {
283                    sp!(_, "-") => {
284                        self.bump();
285                    }
286                    sp!(_, flag) if is_flag(flag.chars().next().unwrap()) => {
287                        self.bump();
288                        break 'command next.widen(c).map(|src| Lexeme(T::Flag, src));
289                    }
290                    sp!(_, _) => break 'command self.unexpected(next),
291                }
292
293                let Some(ident) = self.eat_while(is_ident_continue) else {
294                    break 'command self.unexpected(c);
295                };
296
297                match ident {
298                    sp!(_, "publish") => {
299                        if let Some(next) = self.peek() {
300                            break 'command self.unexpected(next);
301                        }
302
303                        let Some(file) = self.eat_token() else {
304                            break 'command self.done(T::EarlyEof);
305                        };
306
307                        file.widen(c).map(|src| Lexeme(T::Publish, src))
308                    }
309
310                    sp!(_, "upgrade") => {
311                        if let Some(next) = self.peek() {
312                            break 'command self.unexpected(next);
313                        }
314
315                        let Some(file) = self.eat_token() else {
316                            break 'command self.done(T::EarlyEof);
317                        };
318
319                        file.widen(c).map(|src| Lexeme(T::Upgrade, src))
320                    }
321
322                    sp!(_, _) => ident.widen(c).map(|src| Lexeme(T::Command, src)),
323                }
324            }
325
326            sp!(_, _) => self.unexpected(c),
327        })
328    }
329}
330
331fn is_flag(c: char) -> bool {
332    c.is_ascii_alphanumeric()
333}
334
335fn is_ident_start(c: char) -> bool {
336    c.is_ascii_alphabetic() || c == '_'
337}
338
339fn is_ident_continue(c: char) -> bool {
340    c.is_ascii_alphanumeric() || c == '_' || c == '-'
341}
342
343fn is_number_start(c: char) -> bool {
344    c.is_ascii_digit()
345}
346
347fn is_number_continue(c: char) -> bool {
348    c.is_ascii_digit() || c == '_'
349}
350
351fn is_hex_continue(c: char) -> bool {
352    c.is_ascii_hexdigit() || c == '_'
353}

359    /// Tokenize the input up to and including the first terminal token.
360    fn lex(input: Vec<&str>) -> Vec<Spanned<Lexeme<'_>>> {
361        let mut lexer = Lexer::new(input.into_iter()).unwrap();
362        let mut lexemes: Vec<_> = (&mut lexer)
363            .take_while(|sp!(_, lex)| !lex.is_terminal())
364            .collect();
365        lexemes.push(lexer.next().unwrap());
366        lexemes
367    }
