28/// Parse a program
29pub struct ProgramParser<'a, I: Iterator<Item = &'a str>> {
30    tokens: Peekable<Lexer<'a, I>>,
31    state: ProgramParsingState,
32}
33
34struct ProgramParsingState {
35    parsed: Vec<Spanned<ParsedPTBCommand>>,
36    errors: Vec<PTBError>,
37    mvr_names_with_span: BTreeMap<String, Span>,
38    preview_set: bool,
39    summary_set: bool,
40    warn_shadows_set: bool,
41    serialize_unsigned_set: bool,
42    serialize_signed_set: bool,
43    json_set: bool,
44    tx_digest_set: bool,
45    dry_run_set: bool,
46    dev_inspect_set: bool,
47    gas_object_ids: Option<Vec<Spanned<ObjectID>>>,
48    gas_budget: Option<Spanned<u64>>,
49    gas_price: Option<Spanned<u64>>,
50    gas_sponsor: Option<Spanned<NumericalAddress>>,
51    sender: Option<Spanned<NumericalAddress>>,
52}

54macro_rules! mvr_ident {

56        Token::Ident | Token::Number | Token::HexNumber

60impl<'a, I: Iterator<Item = &'a str>> ProgramParser<'a, I> {
61    /// Create a PTB program parser from a sequence of string.
62    pub fn new(tokens: I) -> PTBResult<Self> {
63        let Some(tokens) = Lexer::new(tokens) else {
64            error!(Span { start: 0, end: 0 }, "No tokens")
65        };
66        Ok(Self {
67            tokens: tokens.peekable(),
68            state: ProgramParsingState {
69                parsed: Vec::new(),
70                errors: Vec::new(),
71                mvr_names_with_span: BTreeMap::new(),
72                preview_set: false,
73                summary_set: false,
74                warn_shadows_set: false,
75                serialize_unsigned_set: false,
76                serialize_signed_set: false,
77                json_set: false,
78                tx_digest_set: false,
79                dry_run_set: false,
80                dev_inspect_set: false,
81                gas_object_ids: None,
82                gas_budget: None,
83                gas_price: None,
84                gas_sponsor: None,
85                sender: None,
86            },
87        })
88    }
89
90    /// Parse the sequence of strings into a PTB program. We continue to parse even if an error is
91    /// raised, and return the errors at the end. If no errors are raised, we return the parsed PTB
92    /// program along with the metadata that we have parsed (e.g., json output, summary).
93    pub fn parse(mut self) -> Result<ParsedProgram, Vec<PTBError>> {
94        use Lexeme as L;
95        use Token as T;
96
97        while let Some(sp!(sp, lexeme)) = self.tokens.next() {
98            macro_rules! try_ {
99                ($expr: expr) => {
100                    match $expr {
101                        Ok(arg) => arg,
102                        Err(err) => {
103                            self.state.errors.push(err);
104                            self.fast_forward_to_next_command();
105                            continue;
106                        }
107                    }
108                };
109            }
110
111            macro_rules! command {
112                ($args:expr) => {{
113                    let sp!(sp_args, value) = try_!($args);
114                    let cmd = sp.widen(sp_args).wrap(value);
115                    self.state.parsed.push(cmd);
116                }};
117            }
118
119            macro_rules! flag {
120                ($flag:ident) => {{
121                    self.state.$flag = true;
122                }};
123            }
124
125            match lexeme {
126                L(T::Command, A::SERIALIZE_UNSIGNED) => flag!(serialize_unsigned_set),
127                L(T::Command, A::SERIALIZE_SIGNED) => flag!(serialize_signed_set),
128                L(T::Command, A::SUMMARY) => flag!(summary_set),
129                L(T::Command, A::JSON) => flag!(json_set),
130                L(T::Command, A::TX_DIGEST) => flag!(tx_digest_set),
131                L(T::Command, A::DRY_RUN) => flag!(dry_run_set),
132                L(T::Command, A::DEV_INSPECT) => flag!(dev_inspect_set),
133                L(T::Command, A::PREVIEW) => flag!(preview_set),
134                L(T::Command, A::WARN_SHADOWS) => flag!(warn_shadows_set),
135                L(T::Command, A::GAS_COIN) => {
136                    let coins = try_!(self.parse_gas_coins());
137                    self.state.gas_object_ids = Some(coins);
138                }
139                L(T::Command, A::GAS_SPONSOR) => {
140                    let sponsor = try_!(self.parse_address_literal());
141                    self.state.gas_sponsor = Some(sponsor);
142                }
143
144                L(T::Command, A::GAS_BUDGET) => {
145                    let budget = try_!(self.parse_gas_denomination()).widen_span(sp);
146                    if let Some(other) = self.state.gas_budget.replace(budget) {
147                        self.state.errors.extend([
148                            err!(
149                                other.span,
150                                "Multiple gas budgets found. Gas budget first set here.",
151                            ),
152                            err!(
153                                budget.span => help: { "PTBs must have exactly one gas budget set." },
154                                "Budget set again here."
155                            ),
156                        ]);
157                        self.fast_forward_to_next_command();
158                    }
159                }
160
161                L(T::Command, A::GAS_PRICE) => {
162                    let price = try_!(self.parse_gas_denomination()).widen_span(sp);
163                    if let Some(other) = self.state.gas_price.replace(price) {
164                        self.state.errors.extend([
165                            err!(
166                                other.span,
167                                "Multiple gas prices found. Gas price first set here.",
168                            ),
169                            err!(
170                                price.span => help: { "PTBs must have at most one gas price set." },
171                                "Price set again here."
172                            ),
173                        ]);
174                        self.fast_forward_to_next_command();
175                    }
176                }
177
178                L(T::Command, A::SENDER) => {
179                    let sender = try_!(self.parse_address_literal());
180                    self.state.sender = Some(sender);
181                }
182
183                L(T::Command, A::TRANSFER_OBJECTS) => command!(self.parse_transfer_objects()),
184                L(T::Command, A::SPLIT_COINS) => command!(self.parse_split_coins()),
185                L(T::Command, A::MERGE_COINS) => command!(self.parse_merge_coins()),
186                L(T::Command, A::ASSIGN) => command!(self.parse_assign()),
187                L(T::Command, A::MAKE_MOVE_VEC) => command!(self.parse_make_move_vec()),
188                L(T::Command, A::MOVE_CALL) => command!(self.parse_move_call()),
189
190                L(T::Publish, src) => command!({
191                    let src = sp.wrap(src.to_owned());
192                    Ok(sp.wrap(ParsedPTBCommand::Publish(src)))
193                }),
194
195                L(T::Upgrade, src) => command!({
196                    let src = sp.wrap(src.to_owned());
197                    let cap = try_!(self.parse_argument());
198                    Ok(cap.span.wrap(ParsedPTBCommand::Upgrade(src, cap)))
199                }),
200
201                L(T::Command, s) => {
202                    let possibles = find_did_you_means(s, COMMANDS.iter().copied())
203                        .into_iter()
204                        .map(|s| format!("--{s}"))
205                        .collect();
206                    let err = if let Some(suggestion) = display_did_you_mean(possibles) {
207                        err!(
208                            sp => help: { "{suggestion}" },
209                            "Unknown {lexeme}",
210                        )
211                    } else {
212                        err!(sp, "Unknown {lexeme}")
213                    };
214                    self.state.errors.push(err);
215                    self.fast_forward_to_next_command();
216                }
217
218                L(T::Eof, _) => break,
219
220                unexpected => {
221                    let err = err!(
222                        sp => help: { "Expected to find a command here" },
223                        "Unexpected {unexpected}",
224                    );
225
226                    self.state.errors.push(err);
227                    if unexpected.is_terminal() {
228                        break;
229                    } else {
230                        self.fast_forward_to_next_command();
231                    }
232                }
233            }
234        }
235
236        let sp!(sp, tok) = self.peek();
237
238        if !tok.is_terminal() {
239            self.state
240                .errors
241                .push(err!(sp, "Trailing {tok} found after the last command",));
242        }
243
244        if self.state.errors.is_empty() {
245            Ok((
246                A::Program {
247                    commands: self.state.parsed,
248                    warn_shadows_set: self.state.warn_shadows_set,
249                },
250                A::ProgramMetadata {
251                    preview_set: self.state.preview_set,
252                    summary_set: self.state.summary_set,
253                    serialize_unsigned_set: self.state.serialize_unsigned_set,
254                    serialize_signed_set: self.state.serialize_signed_set,
255                    gas_object_ids: self.state.gas_object_ids,
256                    json_set: self.state.json_set,
257                    tx_digest_set: self.state.tx_digest_set,
258                    dry_run_set: self.state.dry_run_set,
259                    dev_inspect_set: self.state.dev_inspect_set,
260                    gas_budget: self.state.gas_budget,
261                    gas_price: self.state.gas_price,
262                    gas_sponsor: self.state.gas_sponsor,
263                    mvr_names: self.state.mvr_names_with_span,
264                    sender: self.state.sender,
265                },
266            ))
267        } else {
268            Err(self.state.errors)
269        }
270    }
271}

273/// Iterator convenience methods over tokens
274impl<'a, I: Iterator<Item = &'a str>> ProgramParser<'a, I> {
275    /// Advance the iterator and return the next lexeme. If the next lexeme's token is not the
276    /// expected one, return an error, and don't advance the token stream.
277    fn expect(&mut self, expected: Token) -> PTBResult<Spanned<Lexeme<'a>>> {
278        let result @ sp!(sp, lexeme@Lexeme(token, _)) = self.peek();
279        Ok(if token == expected {
280            self.bump();
281            result
282        } else {
283            error!(sp, "Expected {expected} but found {lexeme}");
284        })
285    }
286
287    /// Peek at the next token without advancing the iterator.
288    fn peek(&mut self) -> Spanned<Lexeme<'a>> {
289        *self
290            .tokens
291            .peek()
292            .expect("Lexer returns an infinite stream")
293    }
294
295    /// Unconditionally advance the next token. It is always safe to do this, because the underlying
296    /// token stream is "infinite" (the lexer will repeatedly return its terminal token).
297    fn bump(&mut self) {
298        self.tokens.next();
299    }
300
301    /// Fast forward to the next command token (if any).
302    fn fast_forward_to_next_command(&mut self) {
303        loop {
304            let sp!(_, lexeme) = self.peek();
305            if lexeme.is_command_end() {
306                break;
307            }
308            self.bump();
309        }
310    }
311}

313/// Methods for parsing commands
314impl<'a, I: Iterator<Item = &'a str>> ProgramParser<'a, I> {
315    /// Parse a transfer-objects command.
316    /// The expected format is: `--transfer-objects [<from>, ...] <to>`
317    fn parse_transfer_objects(&mut self) -> PTBResult<Spanned<ParsedPTBCommand>> {
318        let transfer_froms = self.parse_array()?;
319        let transfer_to = self.parse_argument()?;
320        let sp = transfer_to.span.widen(transfer_froms.span);
321        Ok(sp.wrap(ParsedPTBCommand::TransferObjects(
322            transfer_froms,
323            transfer_to,
324        )))
325    }
326
327    /// Parse a split-coins command.
328    /// The expected format is: `--split-coins <coin> [<amount>, ...]`
329    fn parse_split_coins(&mut self) -> PTBResult<Spanned<ParsedPTBCommand>> {
330        let split_from = self.parse_argument()?;
331        let splits = self.parse_array()?;
332        let sp = split_from.span.widen(splits.span);
333        Ok(sp.wrap(ParsedPTBCommand::SplitCoins(split_from, splits)))
334    }
335
336    /// Parse a merge-coins command.
337    /// The expected format is: `--merge-coins <coin> [<coin1>, ...]`
338    fn parse_merge_coins(&mut self) -> PTBResult<Spanned<ParsedPTBCommand>> {
339        let merge_into = self.parse_argument()?;
340        let coins = self.parse_array()?;
341        let sp = merge_into.span.widen(coins.span);
342        Ok(sp.wrap(ParsedPTBCommand::MergeCoins(merge_into, coins)))
343    }
344
345    /// Parse an assign command.
346    /// The expected format is: `--assign <variable> (<value>)?`
347    fn parse_assign(&mut self) -> PTBResult<Spanned<ParsedPTBCommand>> {
348        use Lexeme as L;
349        let sp!(sp, L(_, contents)) = self.expect(Token::Ident)?;
350        if is_keyword(contents) {
351            error!(sp => help: {
352                "Variable names cannot be {}.",
353                all_keywords()
354            },
355            "Expected a variable name but found reserved word '{contents}'.");
356        }
357
358        let ident = sp.wrap(contents.to_owned());
359
360        Ok(if self.peek().value.is_command_end() {
361            ident.span.wrap(ParsedPTBCommand::Assign(ident, None))
362        } else {
363            let value = self.parse_argument()?;
364            let sp = ident.span.widen(value.span);
365            sp.wrap(ParsedPTBCommand::Assign(ident, Some(value)))
366        })
367    }
368
369    /// Parse a make-move-vec command
370    /// The expected format is: `--make-move-vec <type> [<elem>, ...]`
371    fn parse_make_move_vec(&mut self) -> PTBResult<Spanned<ParsedPTBCommand>> {
372        use Token as T;
373
374        let sp!(start_sp, _) = self.expect(T::LAngle)?;
375        let type_ = self.parse_type()?;
376        self.expect(T::RAngle)?;
377
378        let elems = self.parse_array()?;
379
380        let sp = start_sp.widen(elems.span);
381        Ok(sp.wrap(ParsedPTBCommand::MakeMoveVec(type_, elems)))
382    }
383
384    /// Parse a move-call command
385    /// The expected format is: `--move-call <address>::<module>::<name> (<<type>, ...>)? <arg> ...`
386    fn parse_move_call(&mut self) -> PTBResult<Spanned<ParsedPTBCommand>> {
387        use Lexeme as L;
388        use Token as T;
389
390        let function = self.parse_module_access()?;
391        let mut end_sp = function.span;
392
393        let ty_args = if let sp!(_, L(T::LAngle, _)) = self.peek() {
394            let type_args = self.parse_type_args()?;
395            end_sp = type_args.span;
396            Some(type_args)
397        } else {
398            None
399        };
400
401        let mut args = vec![];
402        while !self.peek().value.is_command_end() {
403            let arg = self.parse_argument()?;
404            end_sp = arg.span;
405            args.push(arg);
406        }
407
408        let sp = function.span.widen(end_sp);
409        Ok(sp.wrap(ParsedPTBCommand::MoveCall(function, ty_args, args)))
410    }
411
412    /// Parse a quantity of gas, as a numeric literal that is or can be inferred to be a u64.
413    fn parse_gas_denomination(&mut self) -> PTBResult<Spanned<u64>> {
414        Ok(match self.parse_argument()? {
415            sp!(sp, Argument::U64(u)) => sp.wrap(u),
416            sp!(sp, Argument::InferredNum(n)) => {
417                sp.wrap(u64::try_from(n).map_err(|_| err!(sp, "Value does not fit within a u64"))?)
418            }
419            sp!(sp, _) => error!(sp, "Expected a u64 value"),
420        })
421    }
422
423    /// Parse the gas payment
424    /// The expected format is: `--gas-coin <address> [<address> ...]`
425    fn parse_gas_coins(&mut self) -> PTBResult<Vec<Spanned<ObjectID>>> {
426        // Need at least one gas coin.
427        let mut coins = vec![self.parse_object_id_literal()?];
428        while matches!(self.peek(), sp!(_, Lexeme(Token::At, _))) {
429            coins.push(self.parse_object_id_literal()?)
430        }
431
432        Ok(coins)
433    }
434}

436/// Methods for parsing arguments and types in commands
437impl<'a, I: Iterator<Item = &'a str>> ProgramParser<'a, I> {
438    /// Parse a single PTB argument from the beginning of the token stream.
439    fn parse_argument(&mut self) -> PTBResult<Spanned<Argument>> {
440        use Argument as V;
441        use Lexeme as L;
442        use Token as T;
443
444        let sp!(sp, lexeme) = self.peek();
445        Ok(match lexeme {
446            L(T::Ident, "true") => {
447                self.bump();
448                sp.wrap(V::Bool(true))
449            }
450
451            L(T::Ident, "false") => {
452                self.bump();
453                sp.wrap(V::Bool(false))
454            }
455
456            L(T::Ident, A::GAS) => {
457                self.bump();
458                sp.wrap(V::Gas)
459            }
460
461            L(T::Number | T::HexNumber, number) => {
462                let number = if lexeme.0 == T::HexNumber {
463                    format!("0x{number}")
464                } else {
465                    number.to_owned()
466                };
467
468                self.bump();
469                self.parse_number(sp.wrap(&number))?
470            }
471
472            L(T::At, _) => self.parse_address_literal()?.map(V::Address).widen_span(sp),
473
474            L(T::Ident, A::NONE) => {
475                self.bump();
476                sp.wrap(V::Option(sp.wrap(None)))
477            }
478
479            L(T::Ident, A::SOME) => {
480                self.bump();
481                self.expect(T::LParen)?;
482                let sp!(arg_sp, arg) = self.parse_argument()?;
483                let sp!(end_sp, _) = self.expect(T::RParen)?;
484
485                let sp = sp.widen(end_sp);
486                sp.wrap(V::Option(arg_sp.wrap(Some(Box::new(arg)))))
487            }
488
489            L(T::Ident, A::VECTOR) => {
490                self.bump();
491                self.parse_array()?.map(V::Vector).widen_span(sp)
492            }
493
494            L(T::Ident, _) => self.parse_variable()?,
495
496            L(T::String, contents) => {
497                self.bump();
498                sp.wrap(V::String(contents.to_owned()))
499            }
500
501            unexpected => error!(
502                sp => help: { "Expected an argument here" },
503                "Unexpected {unexpected}",
504            ),
505        })
506    }
507
508    /// Parse a type.
509    fn parse_type(&mut self) -> PTBResult<Spanned<ParsedType>> {
510        use Lexeme as L;
511        use Token as T;
512
513        let sp!(sp, lexeme) = self.peek();
514
515        macro_rules! type_ {
516            ($ty: expr) => {{
517                self.bump();
518                sp.wrap($ty)
519            }};
520        }
521
522        Ok(match lexeme {
523            L(T::Ident, A::U8) => type_!(ParsedType::U8),
524            L(T::Ident, A::U16) => type_!(ParsedType::U16),
525            L(T::Ident, A::U32) => type_!(ParsedType::U32),
526            L(T::Ident, A::U64) => type_!(ParsedType::U64),
527            L(T::Ident, A::U128) => type_!(ParsedType::U128),
528            L(T::Ident, A::U256) => type_!(ParsedType::U256),
529            L(T::Ident, A::BOOL) => type_!(ParsedType::Bool),
530            L(T::Ident, A::ADDRESS) => type_!(ParsedType::Address),
531
532            L(T::Ident, A::VECTOR) => {
533                self.bump();
534                self.expect(T::LAngle)?;
535                let sp!(_, ty) = self.parse_type()?;
536                let sp!(end_sp, _) = self.expect(T::RAngle)?;
537
538                let sp = sp.widen(end_sp);
539                sp.wrap(ParsedType::Vector(Box::new(ty)))
540            }
541
542            L(T::At | T::Ident | T::Number | T::HexNumber, _) => 'fq: {
543                let sp!(_, module_access) = self.parse_module_access()?;
544                let sp!(_, address) = module_access.address;
545                let sp!(_, module_name) = module_access.module_name;
546                let sp!(fun_sp, function_name) = module_access.function_name;
547
548                let module = ParsedModuleId {
549                    address,
550                    name: module_name.to_string(),
551                };
552
553                let name = function_name.to_string();
554                let fq_name = ParsedFqName { module, name };
555
556                let sp!(_, L(T::LAngle, _)) = self.peek() else {
557                    let sp = sp.widen(fun_sp);
558                    break 'fq sp.wrap(ParsedType::Struct(ParsedStructType {
559                        fq_name,
560                        type_args: vec![],
561                    }));
562                };
563
564                let sp!(tys_sp, type_args) = self.parse_type_args()?;
565
566                let sp = sp.widen(tys_sp);
567                sp.wrap(ParsedType::Struct(ParsedStructType { fq_name, type_args }))
568            }
569
570            unexpected => error!(
571                sp => help: { "Expected a type here" },
572                "Unexpected {unexpected}",
573            ),
574        })
575    }
576
577    /// Parse a fully-qualified name, corresponding to accessing a function or type from a module.
578    fn parse_module_access(&mut self) -> PTBResult<Spanned<ModuleAccess>> {
579        use Lexeme as L;
580        use Token as T;
581
582        let address = self.parse_address()?;
583
584        self.expect(T::ColonColon)?;
585        let sp!(mod_sp, L(_, module_name)) = self.expect(T::Ident)?;
586        let module_name = Identifier::new(module_name)
587            .map_err(|_| err!(mod_sp, "Invalid module name {module_name:?}"))?;
588
589        self.expect(T::ColonColon)?;
590        let sp!(fun_sp, L(_, function_name)) = self.expect(T::Ident)?;
591        let function_name = Identifier::new(function_name)
592            .map_err(|_| err!(fun_sp, "Invalid function name {function_name:?}"))?;
593
594        let sp = address.span.widen(fun_sp);
595        Ok(sp.wrap(ModuleAccess {
596            address,
597            module_name: mod_sp.wrap(module_name),
598            function_name: fun_sp.wrap(function_name),
599        }))
600    }
601
602    /// Parse a list of type arguments, surrounded by angle brackets, and separated by commas.
603    fn parse_type_args(&mut self) -> PTBResult<Spanned<Vec<ParsedType>>> {
604        use Lexeme as L;
605        use Token as T;
606
607        let sp!(start_sp, _) = self.expect(T::LAngle)?;
608
609        let mut type_args = vec![];
610        loop {
611            type_args.push(self.parse_type()?.value);
612
613            let sp!(sp, lexeme) = self.peek();
614            match lexeme {
615                L(T::Comma, _) => self.bump(),
616                L(T::RAngle, _) => break,
617                unexpected => error!(
618                    sp => help: { "Expected {} or {}", T::Comma, T::RAngle },
619                    "Unexpected {unexpected}",
620                ),
621            }
622        }
623
624        let sp!(end_sp, _) = self.expect(T::RAngle)?;
625        Ok(start_sp.widen(end_sp).wrap(type_args))
626    }
627
628    /// Parse a variable access (a non-empty chain of identifiers, separated by '.')
629    fn parse_variable(&mut self) -> PTBResult<Spanned<Argument>> {
630        use Lexeme as L;
631        use Token as T;
632
633        let sp!(start_sp, L(_, ident)) = self.expect(T::Ident)?;
634        let ident = start_sp.wrap(ident.to_owned());
635
636        let sp!(_, L(T::Dot, _)) = self.peek() else {
637            return Ok(start_sp.wrap(Argument::Identifier(ident.value)));
638        };
639
640        self.bump();
641        let mut fields = vec![];
642        loop {
643            // A field can be any non-terminal token (identifier, number, etc).
644            let sp!(sp, lexeme@L(_, field)) = self.peek();
645            if lexeme.is_terminal() {
646                error!(sp, "Expected a field name after '.'");
647            }
648
649            self.bump();
650            fields.push(sp.wrap(field.to_owned()));
651
652            if let sp!(_, L(T::Dot, _)) = self.peek() {
653                self.bump();
654            } else {
655                break;
656            }
657        }
658
659        let end_sp = fields.last().map(|f| f.span).unwrap_or(start_sp);
660        let sp = start_sp.widen(end_sp);
661        Ok(sp.wrap(Argument::VariableAccess(ident, fields)))
662    }
663
664    /// Parse a decimal or hexadecimal number, optionally followed by a type suffix.
665    fn parse_number(&mut self, contents: Spanned<&str>) -> PTBResult<Spanned<Argument>> {
666        use Argument as V;
667        use Lexeme as L;
668        use Token as T;
669
670        let sp!(sp, suffix) = self.peek();
671
672        macro_rules! parse_num {
673            ($fn: ident, $ty: expr) => {{
674                self.bump();
675                let sp = sp.widen(contents.span);
676                match $fn(contents.value) {
677                    Ok((value, _)) => sp.wrap($ty(value)),
678                    Err(e) => error!(sp, "{e}"),
679                }
680            }};
681        }
682
683        Ok(match suffix {
684            L(T::Ident, A::U8) => parse_num!(parse_u8, V::U8),
685            L(T::Ident, A::U16) => parse_num!(parse_u16, V::U16),
686            L(T::Ident, A::U32) => parse_num!(parse_u32, V::U32),
687            L(T::Ident, A::U64) => parse_num!(parse_u64, V::U64),
688            L(T::Ident, A::U128) => parse_num!(parse_u128, V::U128),
689            L(T::Ident, A::U256) => parse_num!(parse_u256, V::U256),
690
691            // If there's no suffix, parse as `InferredNum`, and don't consume the peeked character.
692            _ => match parse_u256(contents.value) {
693                Ok((value, _)) => contents.span.wrap(V::InferredNum(value)),
694                Err(_) => error!(contents.span, "Invalid integer literal"),
695            },
696        })
697    }

699    fn parse_mvr_address(
700        &mut self,
701        prefix: Option<Spanned<&str>>,
702    ) -> PTBResult<Spanned<ParsedAddress>> {
703        use Lexeme as L;
704        use Token as T;
705
706        let mut address = String::new();
707        let start_sp = prefix.map(|p| p.span).unwrap_or_else(|| self.peek().span);
708
709        // Step 1: parse before `/` in the MVR name
710        match prefix {
711            // No @ prefix
712            Some(sp!(_, prefix)) => {
713                address.push_str(prefix);
714                // parse .sui
715                self.expect(T::Dot)?;
716                address.push('.');
717                match self.peek() {
718                    sp!(sp, L(T::Ident, s)) => {
719                        if s != "sui" {
720                            error!(sp => help: { "Expected 'sui' extension" },
721                            "Expected a valid MVR top level domain"
722                            );
723                        }
724                        self.bump();
725                        address.push_str(s);
726                    }
727                    sp!(sp, l) => {
728                        error!(
729                            sp,
730                            "Expected a valid MVR domain extension ('.sui') but got {}", l
731                        )
732                    }
733                }
734            }
735            // @ prefixed MVR name
736            None => {
737                self.expect(T::At)?;
738                address.push('@');
739                match self.peek() {
740                    sp!(_, L(mvr_ident!(), nm)) => {
741                        self.bump();
742                        address.push_str(nm);
743                    }
744                    sp!(sp, l) => {
745                        error!(sp, "Expected a MVR domain name but got {l}")
746                    }
747                }
748            }
749        }
750
751        self.expect(T::ForwardSlash).map_err(|e| {
752            err!(e.span => help: {
753                "MVR packages must contain a slash and be in the format '(@<domain>|<domain>.sui)/<package>[/<version>]'"
754            }, "Invalid MVR package reference")
755        })?;
756        address.push('/');
757
758        // Step 2: post slash -- package name
759        match self.peek() {
760            sp!(_, L(mvr_ident!(), s)) => {
761                address.push_str(s);
762                self.bump();
763            }
764            sp!(sp, l) => {
765                error!(sp, "Expected a valid MVR package name, but got {l}")
766            }
767        }
768
769        // Step 3: post package name -- possible version
770        if let sp!(_, L(T::ForwardSlash, _)) = self.peek() {
771            address.push('/');
772            self.bump();
773            match self.peek() {
774                sp!(_, L(T::Number, s)) => {
775                    address.push_str(s);
776                    self.bump();
777                }
778                sp!(sp, l) => {
779                    error!(sp, "Expected a valid MVR version, but got {l}")
780                }
781            }
782        }
783        let end_sp = self.peek().span;
784
785        self.state
786            .mvr_names_with_span
787            .entry(address.clone())
788            .or_insert_with(|| start_sp.widen(end_sp));
789        Ok(start_sp.widen(end_sp).wrap(ParsedAddress::Named(address)))
790    }

792    /// Parse a numerical, named address, or mvr address.
793    fn parse_address(&mut self) -> PTBResult<Spanned<ParsedAddress>> {
794        use Lexeme as L;
795        use Token as T;
796
797        let sp!(sp, lexeme) = self.peek();
798        let addr = match lexeme {
799            L(T::At, _) => self.parse_mvr_address(None)?.value,
800            L(t @ (mvr_ident!()), contents) => {
801                self.bump();
802                if let sp!(s, L(T::Dot, _)) = self.peek() {
803                    self.parse_mvr_address(Some(Spanned {
804                        span: s,
805                        value: contents,
806                    }))?
807                    .value
808                } else {
809                    match t {
810                        T::Ident => ParsedAddress::Named(contents.to_owned()),
811                        T::Number => {
812                            let number = contents.to_string();
813                            NumericalAddress::parse_str(&number)
814                                .map_err(|e| err!(sp, "Failed to parse address {number:?}: {e}"))
815                                .map(ParsedAddress::Numerical)?
816                        }
817                        T::HexNumber => {
818                            let number = format!("0x{contents}");
819                            NumericalAddress::parse_str(&number)
820                                .map_err(|e| err!(sp, "Failed to parse address {number:?}: {e}"))
821                                .map(ParsedAddress::Numerical)?
822                        }
823                        _ => unreachable!(),
824                    }
825                }
826            }
827            unexpected => error!(
828                sp => help: {
829                    "Value addresses can either be a variable in-scope, or a numerical address, \
830                     e.g., 0xc0ffee"
831                },
832                "Unexpected {unexpected}",
833            ),
834        };
835
836        Ok(sp.wrap(addr))
837    }

839    /// Parse a numeric address literal (must be prefixed by an `@` symbol).
840    fn parse_address_literal(&mut self) -> PTBResult<Spanned<NumericalAddress>> {
841        let sp!(sp, _) = self.expect(Token::At).map_err(|e| {
842            err!(e.span => help: {
843                "Addresses or object IDs require the character '@' in front"
844            }, "Expected an address")
845        })?;
846
847        Ok(match self.parse_address()?.widen_span(sp) {
848            sp!(sp, ParsedAddress::Numerical(n)) => sp.wrap(n),
849            sp!(sp, ParsedAddress::Named(n)) => error!(
850                sp,
851                "Expected a numerical address but got a named address '{n}'",
852            ),
853        })
854    }
855
856    /// Parse a numeric address literal (must be prefixed by an `@` symbol) as an ObjectID.
857    fn parse_object_id_literal(&mut self) -> PTBResult<Spanned<ObjectID>> {
858        Ok(self.parse_address_literal()?.map(|a| a.into_inner().into()))
859    }

861    // Parse an array of arguments. Each element of the array is separated by a comma.
862    fn parse_array(&mut self) -> PTBResult<Spanned<Vec<Spanned<Argument>>>> {
863        use Lexeme as L;
864        use Token as T;
865        let sp!(start_sp, _) = self.expect(T::LBracket)?;
866
867        let mut values = vec![];
868        loop {
869            let sp!(sp, lexeme) = self.peek();
870            if lexeme.is_terminal() {
871                error!(
872                    sp => help: { "Expected an array here" },
873                    "Unexpected {lexeme}"
874                );
875            } else if let L(T::RBracket, _) = lexeme {
876                break;
877            }
878
879            values.push(self.parse_argument()?);
880
881            let sp!(sp, lexeme) = self.peek();
882            match lexeme {
883                L(T::RBracket, _) => break,
884                L(T::Comma, _) => self.bump(),
885                unexpected => error!(
886                    sp => help: { "Expected {} or {}", T::RBracket, T::Comma },
887                    "Unexpected {unexpected}",
888                ),
889            }
890        }
891
892        let sp!(end_sp, _) = self.expect(T::RBracket)?;
893        Ok(start_sp.widen(end_sp).wrap(values))
894    }
