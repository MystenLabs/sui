57/// An enum holding either an account address or a move package information. The Move Package data
58/// is used to resolve the origin id for a typetag, in the `PTBBuilder`.
59#[derive(Clone, Debug)]
60pub enum AddressData {
61    /// The account address
62    AccountAddress(AccountAddress),
63    /// The MovePackage itself and the TypeOrigin table in a (module, type) -> origin ID map.
64    MovePackage((MovePackage, BTreeMap<(String, String), ObjectID>)),
65}
66
67impl AddressData {
68    /// Return the account address
69    pub fn address(&self) -> Option<AccountAddress> {
70        match self {
71            AddressData::AccountAddress(account_address) => Some(*account_address),
72            AddressData::MovePackage((pkg, _)) => {
73                AccountAddress::from_bytes(pkg.id().into_bytes()).ok()
74            }
75        }
76    }
77}

79impl PTB {
80    /// Parses and executes the PTB with the sender as the current active address
81    pub async fn execute(self, context: &mut WalletContext) -> Result<(), Error> {
82        if self.args.is_empty() {
83            ptb_description().print_help().unwrap();
84            return Ok(());
85        }
86        let source_string = to_source_string(self.args.clone());
87
88        // Tokenize once to detect help flags
89        let tokens = self.args.iter().map(|s| s.as_str());
90        for sp!(_, lexeme) in Lexer::new(tokens.clone()).into_iter().flatten() {
91            match lexeme {
92                Lexeme(Token::Command, "help") => return Ok(ptb_description().print_long_help()?),
93                Lexeme(Token::Flag, "h") => return Ok(ptb_description().print_help()?),
94                lexeme if lexeme.is_terminal() => break,
95                _ => continue,
96            }
97        }
98
99        // Tokenize and parse to get the program
100        let (program, program_metadata) = match ProgramParser::new(tokens)
101            .map_err(|e| vec![e])
102            .and_then(|parser| parser.parse())
103        {
104            Err(errors) => {
105                let suffix = if errors.len() > 1 { "s" } else { "" };
106                let rendered = build_error_reports(&source_string, errors);
107                eprintln!("Encountered error{suffix} when parsing PTB:");
108                for e in rendered.iter() {
109                    eprintln!("{:?}", e);
110                }
111                anyhow::bail!("Could not build PTB due to previous error{suffix}");
112            }
113            Ok(parsed) => parsed,
114        };
115
116        ensure!(
117            !program_metadata.serialize_unsigned_set || !program_metadata.serialize_signed_set,
118            "Cannot specify both flags: --serialize-unsigned-transaction and --serialize-signed-transaction."
119        );
120
121        if program_metadata.preview_set {
122            println!(
123                "{}",
124                PTBPreview {
125                    program: &program,
126                    program_metadata: &program_metadata
127                }
128            );
129            return Ok(());
130        }
131
132        let mut client = context.grpc_client()?;
133
134        let mut starting_addresses: BTreeMap<String, AddressData> = context
135            .config
136            .keystore
137            .addresses_with_alias()
138            .into_iter()
139            .map(|(sa, alias)| {
140                (
141                    alias.alias.clone(),
142                    AddressData::AccountAddress(AccountAddress::from(*sa)),
143                )
144            })
145            .collect();
146
147        let mvr_names = program_metadata.mvr_names.clone();
148        let mvr_resolver = MvrResolver {
149            names: program_metadata.mvr_names.into_keys().collect(),
150        };
151        if mvr_resolver.should_resolve() {
152            let resolved = mvr_resolver.resolve_names(&client).await?;
153            let mut mvr_data: BTreeMap<String, AddressData> = BTreeMap::new();
154            for (name, package_id) in resolved.resolution {
155                let span = mvr_names.get(&name).unwrap_or(&Span { start: 0, end: 0 });
156                let pkg = resolve_package(&mut client, package_id.package_id, *span).await?;
157                let type_origin_id_map = pkg.type_origin_map();
158                mvr_data.insert(name, AddressData::MovePackage((pkg, type_origin_id_map)));
159            }
160
161            starting_addresses.extend(mvr_data);
162        }
163
164        let (res, warnings) =
165            Self::build_ptb(program, starting_addresses, client.clone(), context).await;
166
167        // Render warnings
168        if !warnings.is_empty() {
169            let suffix = if warnings.len() > 1 { "s" } else { "" };
170            eprintln!("Warning{suffix} produced when building PTB:");
171            let rendered = build_error_reports(&source_string, warnings);
172            for e in rendered.iter() {
173                eprintln!("{:?}", e);
174            }
175        }
176        let ptb = match res {
177            Err(errors) => {
178                let suffix = if errors.len() > 1 { "s" } else { "" };
179                eprintln!("Encountered error{suffix} when building PTB:");
180                let rendered = build_error_reports(&source_string, errors);
181                for e in rendered.iter() {
182                    eprintln!("{:?}", e);
183                }
184                anyhow::bail!("Could not build PTB due to previous error{suffix}");
185            }
186            Ok(x) => x,
187        };
188
189        let gas: Vec<_> = program_metadata
190            .gas_object_ids
191            .into_iter()
192            .flatten()
193            .map(|x| x.value)
194            .collect();
195
196        // the sender is the gas object if gas is provided, otherwise the active address
197        let sender = if let Some(sender) = program_metadata.sender {
198            sender.value.into_inner().into()
199        } else {
200            context.infer_sender(&gas).await?
201        };
202
203        // build the tx kind
204        let tx_kind = TransactionKind::ProgrammableTransaction(ProgrammableTransaction {
205            inputs: ptb.inputs,
206            commands: ptb.commands,
207        });
208
209        let gas_data = GasDataArgs {
210            gas_budget: program_metadata.gas_budget.map(|x| x.value),
211            gas_price: program_metadata.gas_price.map(|x| x.value),
212            gas_sponsor: program_metadata
213                .gas_sponsor
214                .map(|x| x.value.into_inner().into()),
215        };
216
217        let processing = TxProcessingArgs {
218            tx_digest: program_metadata.tx_digest_set,
219            dry_run: program_metadata.dry_run_set,
220            dev_inspect: program_metadata.dev_inspect_set,
221            serialize_unsigned_transaction: program_metadata.serialize_unsigned_set,
222            serialize_signed_transaction: program_metadata.serialize_signed_set,
223            sender: program_metadata.sender.map(|x| x.value.into_inner().into()),
224        };
225
226        let gas_payment = client.transaction_builder().input_refs(&gas).await?;
227
228        let transaction_response = dry_run_or_execute_or_serialize(
229            sender,
230            tx_kind,
231            context,
232            gas_payment,
233            gas_data,
234            processing,
235        )
236        .await?;
237
238        let transaction_response = match transaction_response {
239            SuiClientCommandResult::ComputeTransactionDigest(_)
240            | SuiClientCommandResult::DryRun(_)
241            | SuiClientCommandResult::SerializedUnsignedTransaction(_)
242            | SuiClientCommandResult::SerializedSignedTransaction(_) => {
243                println!("{transaction_response}");
244                return Ok(());
245            }
246            SuiClientCommandResult::TransactionBlock(response) => response,
247            SuiClientCommandResult::DevInspect(response) => {
248                println!("{}", Pretty(&response));
249                return Ok(());
250            }
251            _ => anyhow::bail!("Internal error, unexpected response from PTB execution."),
252        };
253
254        if transaction_response.effects.status().is_err() {
255            return Err(anyhow!(
256                "PTB execution {:?}. Transaction digest is: {}",
257                transaction_response.effects.status(),
258                transaction_response.effects.transaction_digest()
259            ));
260        }
261
262        let summary = {
263            let effects = &transaction_response.effects;
264            Summary {
265                digest: transaction_response.transaction.digest(),
266                status: effects.status().clone(),
267                gas_cost: effects.gas_cost_summary().clone(),
268            }
269        };
270
271        let command_result = SuiClientCommandResult::TransactionBlock(transaction_response);
272
273        if program_metadata.json_set {
274            let json_string = if program_metadata.summary_set {
275                serde_json::to_string_pretty(&serde_json::json!(summary))
276                    .map_err(|_| anyhow!("Cannot serialize PTB result to json"))?
277            } else {
278                format!("{:?}", command_result)
279            };
280            println!("{}", json_string);
281        } else if program_metadata.summary_set {
282            println!("{}", Pretty(&summary));
283        } else {
284            println!("{}", command_result);
285        }
286
287        Ok(())
288    }

290    // Also used in testing, thus public
291    pub async fn build_ptb(
292        program: Program,
293        starting_addresses: BTreeMap<String, AddressData>,
294        reader: Client,
295        wallet: &WalletContext,
296    ) -> (
297        Result<ProgrammableTransaction, Vec<PTBError>>,
298        Vec<PTBError>,
299    ) {
300        let builder = PTBBuilder::new(starting_addresses, reader, wallet);
301        builder.build(program).await
302    }

304    // Also used in testing, thus public
305    pub fn parse_ptb_commands(args: Vec<String>) -> Result<ParsedProgram, Vec<PTBError>> {
306        ProgramParser::new(args.iter().map(|s| s.as_str()))
307            .map_err(|e| vec![e])
308            .and_then(|parser| parser.parse())
309    }

312/// Convert a vector of shell tokens into a single string, with each shell token separated by a
313/// space with each command starting on a new line.
314/// NB: we add a space to the end of the source string to ensure that for unexpected EOF
315/// errors we have a location to point to.
316pub fn to_source_string(strings: Vec<String>) -> String {
317    let mut strings = strings.into_iter();
318    let mut string = String::new();
319
320    let Some(first) = strings.next() else {
321        return string;
322    };
323    string.push_str(&first);
324
325    for s in strings {
326        if s.starts_with("--") {
327            string.push('\n');
328            string.push_str(&s);
329        } else {
330            string.push(' ');
331            string.push_str(&s);
332        }
333    }
334    string.push(' ');
335    string
336}
