55/// An enum holding either an account address or a move package information. The Move Package data
56/// is used to resolve the origin id for a typetag, in the `PTBBuilder`.
57#[derive(Clone, Debug)]
58pub enum AddressData {
59    /// The account address
60    AccountAddress(AccountAddress),
61    /// The MovePackage itself and the TypeOrigin table in a (module, type) -> origin ID map.
62    MovePackage((MovePackage, BTreeMap<(String, String), ObjectID>)),
63}
64
65impl AddressData {
66    /// Return the account address
67    pub fn address(&self) -> Option<AccountAddress> {
68        match self {
69            AddressData::AccountAddress(account_address) => Some(*account_address),
70            AddressData::MovePackage((pkg, _)) => {
71                AccountAddress::from_bytes(pkg.id().into_bytes()).ok()
72            }
73        }
74    }
75}

77impl PTB {
78    /// Parses and executes the PTB with the sender as the current active address
79    pub async fn execute(self, context: &mut WalletContext) -> Result<(), Error> {
80        if self.args.is_empty() {
81            ptb_description().print_help().unwrap();
82            return Ok(());
83        }
84        let source_string = to_source_string(self.args.clone());
85
86        // Tokenize once to detect help flags
87        let tokens = self.args.iter().map(|s| s.as_str());
88        for sp!(_, lexeme) in Lexer::new(tokens.clone()).into_iter().flatten() {
89            match lexeme {
90                Lexeme(Token::Command, "help") => return Ok(ptb_description().print_long_help()?),
91                Lexeme(Token::Flag, "h") => return Ok(ptb_description().print_help()?),
92                lexeme if lexeme.is_terminal() => break,
93                _ => continue,
94            }
95        }
96
97        // Tokenize and parse to get the program
98        let (program, program_metadata) = match ProgramParser::new(tokens)
99            .map_err(|e| vec![e])
100            .and_then(|parser| parser.parse())
101        {
102            Err(errors) => {
103                let suffix = if errors.len() > 1 { "s" } else { "" };
104                let rendered = build_error_reports(&source_string, errors);
105                eprintln!("Encountered error{suffix} when parsing PTB:");
106                for e in rendered.iter() {
107                    eprintln!("{:?}", e);
108                }
109                anyhow::bail!("Could not build PTB due to previous error{suffix}");
110            }
111            Ok(parsed) => parsed,
112        };
113
114        ensure!(
115            !program_metadata.serialize_unsigned_set || !program_metadata.serialize_signed_set,
116            "Cannot specify both flags: --serialize-unsigned-transaction and --serialize-signed-transaction."
117        );
118
119        if program_metadata.preview_set {
120            println!(
121                "{}",
122                PTBPreview {
123                    program: &program,
124                    program_metadata: &program_metadata
125                }
126            );
127            return Ok(());
128        }
129
130        let client = context.get_client().await?;
131
132        let mut starting_addresses: BTreeMap<String, AddressData> = context
133            .config
134            .keystore
135            .addresses_with_alias()
136            .into_iter()
137            .map(|(sa, alias)| {
138                (
139                    alias.alias.clone(),
140                    AddressData::AccountAddress(AccountAddress::from(*sa)),
141                )
142            })
143            .collect();
144
145        let mvr_names = program_metadata.mvr_names.clone();
146        let mvr_resolver = MvrResolver {
147            names: program_metadata.mvr_names.into_keys().collect(),
148        };
149        if mvr_resolver.should_resolve() {
150            let resolved = mvr_resolver.resolve_names(client.read_api()).await?;
151            let mut mvr_data: BTreeMap<String, AddressData> = BTreeMap::new();
152            for (name, package_id) in resolved.resolution {
153                let span = mvr_names.get(&name).unwrap_or(&Span { start: 0, end: 0 });
154                let pkg = resolve_package(client.read_api(), package_id.package_id, *span).await?;
155                let type_origin_id_map = pkg.type_origin_map();
156                mvr_data.insert(name, AddressData::MovePackage((pkg, type_origin_id_map)));
157            }
158
159            starting_addresses.extend(mvr_data);
160        }
161
162        let (res, warnings) = Self::build_ptb(program, starting_addresses, client.clone()).await;
163
164        // Render warnings
165        if !warnings.is_empty() {
166            let suffix = if warnings.len() > 1 { "s" } else { "" };
167            eprintln!("Warning{suffix} produced when building PTB:");
168            let rendered = build_error_reports(&source_string, warnings);
169            for e in rendered.iter() {
170                eprintln!("{:?}", e);
171            }
172        }
173        let ptb = match res {
174            Err(errors) => {
175                let suffix = if errors.len() > 1 { "s" } else { "" };
176                eprintln!("Encountered error{suffix} when building PTB:");
177                let rendered = build_error_reports(&source_string, errors);
178                for e in rendered.iter() {
179                    eprintln!("{:?}", e);
180                }
181                anyhow::bail!("Could not build PTB due to previous error{suffix}");
182            }
183            Ok(x) => x,
184        };
185
186        let gas: Vec<_> = program_metadata
187            .gas_object_ids
188            .into_iter()
189            .flatten()
190            .map(|x| x.value)
191            .collect();
192
193        // the sender is the gas object if gas is provided, otherwise the active address
194        let sender = if let Some(sender) = program_metadata.sender {
195            sender.value.into_inner().into()
196        } else {
197            context.infer_sender(&gas).await?
198        };
199
200        // build the tx kind
201        let tx_kind = TransactionKind::ProgrammableTransaction(ProgrammableTransaction {
202            inputs: ptb.inputs,
203            commands: ptb.commands,
204        });
205
206        let gas_data = GasDataArgs {
207            gas_budget: program_metadata.gas_budget.map(|x| x.value),
208            gas_price: program_metadata.gas_price.map(|x| x.value),
209            gas_sponsor: program_metadata
210                .gas_sponsor
211                .map(|x| x.value.into_inner().into()),
212        };
213
214        let processing = TxProcessingArgs {
215            tx_digest: program_metadata.tx_digest_set,
216            dry_run: program_metadata.dry_run_set,
217            dev_inspect: program_metadata.dev_inspect_set,
218            serialize_unsigned_transaction: program_metadata.serialize_unsigned_set,
219            serialize_signed_transaction: program_metadata.serialize_signed_set,
220            sender: program_metadata.sender.map(|x| x.value.into_inner().into()),
221        };
222
223        let gas_payment = client.transaction_builder().input_refs(&gas).await?;
224
225        let transaction_response = dry_run_or_execute_or_serialize(
226            sender,
227            tx_kind,
228            context,
229            gas_payment,
230            gas_data,
231            processing,
232        )
233        .await?;
234
235        let transaction_response = match transaction_response {
236            SuiClientCommandResult::ComputeTransactionDigest(_)
237            | SuiClientCommandResult::DryRun(_)
238            | SuiClientCommandResult::SerializedUnsignedTransaction(_)
239            | SuiClientCommandResult::SerializedSignedTransaction(_) => {
240                println!("{transaction_response}");
241                return Ok(());
242            }
243            SuiClientCommandResult::TransactionBlock(response) => response,
244            SuiClientCommandResult::DevInspect(response) => {
245                println!("{}", Pretty(&response));
246                return Ok(());
247            }
248            _ => anyhow::bail!("Internal error, unexpected response from PTB execution."),
249        };
250
251        if let Some(effects) = transaction_response.effects.as_ref()
252            && effects.status().is_err()
253        {
254            return Err(anyhow!(
255                "PTB execution {}. Transaction digest is: {}",
256                Pretty(effects.status()),
257                effects.transaction_digest()
258            ));
259        }
260
261        let summary = {
262            let effects = transaction_response.effects.as_ref().ok_or_else(|| {
263                anyhow!("Internal error: no transaction effects after PTB was executed.")
264            })?;
265            Summary {
266                digest: transaction_response.digest,
267                status: effects.status().clone(),
268                gas_cost: effects.gas_cost_summary().clone(),
269            }
270        };
271
272        if program_metadata.json_set {
273            let json_string = if program_metadata.summary_set {
274                serde_json::to_string_pretty(&serde_json::json!(summary))
275                    .map_err(|_| anyhow!("Cannot serialize PTB result to json"))?
276            } else {
277                serde_json::to_string_pretty(&serde_json::json!(transaction_response))
278                    .map_err(|_| anyhow!("Cannot serialize PTB result to json"))?
279            };
280            println!("{}", json_string);
281        } else if program_metadata.summary_set {
282            println!("{}", Pretty(&summary));
283        } else {
284            println!("{}", transaction_response);
285        }
286
287        Ok(())
288    }

290    // Also used in testing, thus public
291    pub async fn build_ptb(
292        program: Program,
293        starting_addresses: BTreeMap<String, AddressData>,
294        client: SuiClient,
295    ) -> (
296        Result<ProgrammableTransaction, Vec<PTBError>>,
297        Vec<PTBError>,
298    ) {
299        let builder = PTBBuilder::new(starting_addresses, client.read_api());
300        builder.build(program).await
301    }

303    // Also used in testing, thus public
304    pub fn parse_ptb_commands(args: Vec<String>) -> Result<ParsedProgram, Vec<PTBError>> {
305        ProgramParser::new(args.iter().map(|s| s.as_str()))
306            .map_err(|e| vec![e])
307            .and_then(|parser| parser.parse())
308    }

311/// Convert a vector of shell tokens into a single string, with each shell token separated by a
312/// space with each command starting on a new line.
313/// NB: we add a space to the end of the source string to ensure that for unexpected EOF
314/// errors we have a location to point to.
315pub fn to_source_string(strings: Vec<String>) -> String {
316    let mut strings = strings.into_iter();
317    let mut string = String::new();
318
319    let Some(first) = strings.next() else {
320        return string;
321    };
322    string.push_str(&first);
323
324    for s in strings {
325        if s.starts_with("--") {
326            string.push('\n');
327            string.push_str(&s);
328        } else {
329            string.push(' ');
330            string.push_str(&s);
331        }
332    }
333    string.push(' ');
334    string
335}
