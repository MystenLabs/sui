4use std::fmt;
5
6#[derive(Clone, Copy, Debug)]
7pub struct Lexeme<'t>(
8    /// The kind of lexeme.
9    pub Token,
10    /// Slice from source that identifies this lexeme (among other instances of this token).
11    pub &'t str,
12);
13
14#[derive(Clone, Copy, PartialEq, Eq, Debug)]
15pub enum Token {
16    /// --[a-zA-Z0-9_-]+
17    Command,
18    /// -[a-zA-Z0-9]
19    Flag,
20    /// [a-zA-Z_][a-zA-Z0-9_-]*
21    Ident,
22    /// [1-9][0-9_]*
23    Number,
24    /// 0x[0-9a-fA-F][0-9a-fA-F_]*
25    HexNumber,
26    /// "..." | '...'
27    String,
28    /// ::
29    ColonColon,
30    /// ,
31    Comma,
32    /// [
33    LBracket,
34    /// ]
35    RBracket,
36    /// (
37    LParen,
38    /// )
39    RParen,
40    /// <
41    LAngle,
42    /// >
43    RAngle,
44    /// @
45    At,
46    /// .
47    Dot,
48    /// /
49    ForwardSlash,
50
51    /// End of input.
52    Eof,
53
54    /// Special tokens for unexpected lexer states that the parser should error on.
55    Unexpected,
56    UnfinishedString,
57    EarlyEof,
58
59    // The following tokens are special -- they consume multiple shell tokens, to ensure we capture
60    // the path for a publish or an upgrade command.
61    /// --publish \<shell-token\>
62    Publish,
63    /// --upgraded \<shell-token\>
64    Upgrade,
65}
66
67impl Lexeme<'_> {
68    /// Returns true if this lexeme corresponds to a special error token.
69    pub fn is_error(&self) -> bool {
70        use Token as T;
71        matches!(self.0, T::Unexpected | T::UnfinishedString | T::EarlyEof)
72    }
73
74    /// Returns true if this is the kind of lexeme that finishes the token stream.
75    pub fn is_terminal(&self) -> bool {
76        self.is_error() || self.0 == Token::Eof
77    }
78
79    /// Returns true if this lexeme signifies the end of the current command.
80    pub fn is_command_end(&self) -> bool {
81        self.is_terminal() || [Token::Command, Token::Publish, Token::Upgrade].contains(&self.0)
82    }
83}
84
85impl fmt::Display for Lexeme<'_> {
86    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
87        use Token as T;
88
89        match self.0 {
90            T::Command => write!(f, "command '--{}'", self.1),
91            T::Flag => write!(f, "flag '-{}'", self.1),
92            T::Ident => write!(f, "identifier '{}'", self.1),
93            T::Number => write!(f, "number '{}'", self.1),
94            T::HexNumber => write!(f, "hexadecimal number '0x{}'", self.1),
95            T::String => write!(f, "string {:?}", self.1),
96            T::ColonColon => write!(f, "'::'"),
97            T::Comma => write!(f, "','"),
98            T::LBracket => write!(f, "'['"),
99            T::RBracket => write!(f, "']'"),
100            T::LParen => write!(f, "'('"),
101            T::RParen => write!(f, "')'"),
102            T::LAngle => write!(f, "'<'"),
103            T::RAngle => write!(f, "'>'"),
104            T::At => write!(f, "'@'"),
105            T::Dot => write!(f, "'.'"),
106            T::ForwardSlash => write!(f, "'/'"),
107            T::Unexpected => write!(f, "input {:?}", self.1),
108            T::UnfinishedString => write!(f, "unfinished string {:?}", format!("{}...", self.1)),
109            T::EarlyEof | T::Eof => write!(f, "end of input"),
110            T::Publish => write!(f, "command '--publish {:?}'", self.1),
111            T::Upgrade => write!(f, "command '--upgrade {:?}'", self.1),
112        }
113    }
114}
115
116impl fmt::Display for Token {
117    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
118        use Token as T;
119        match self {
120            T::Command => write!(f, "a command"),
121            T::Flag => write!(f, "a flag"),
122            T::Ident => write!(f, "an identifier"),
123            T::Number => write!(f, "a number"),
124            T::HexNumber => write!(f, "a hexadecimal number"),
125            T::String => write!(f, "a string"),
126            T::ColonColon => write!(f, "'::'"),
127            T::Comma => write!(f, "','"),
128            T::LBracket => write!(f, "'['"),
129            T::RBracket => write!(f, "']'"),
130            T::LParen => write!(f, "'('"),
131            T::RParen => write!(f, "')'"),
132            T::LAngle => write!(f, "'<'"),
133            T::RAngle => write!(f, "'>'"),
134            T::At => write!(f, "'@'"),
135            T::Dot => write!(f, "'.'"),
136            T::ForwardSlash => write!(f, "'/'"),
137            T::Eof => write!(f, "end of input"),
138            T::Unexpected => write!(f, "unexpected input"),
139            T::UnfinishedString => write!(f, "an unfinished string"),
140            T::EarlyEof => write!(f, "unexpected end of input"),
141            T::Publish => write!(f, "a '--publish' command"),
142            T::Upgrade => write!(f, "an '--upgrade' command"),
143        }
144    }
145}
