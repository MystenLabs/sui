4use crate::{client_commands::estimate_gas_budget_from_gas_cost, displays::Pretty};
5use std::fmt::{Display, Formatter};
6use sui_json_rpc_types::{
7    DryRunTransactionBlockResponse, ObjectChange, SuiTransactionBlockDataAPI,
8    SuiTransactionBlockEffectsAPI,
9};
10use tabled::{
11    builder::Builder as TableBuilder,
12    settings::{Panel as TablePanel, Style as TableStyle, style::HorizontalLine},
13};
14impl Display for Pretty<'_, DryRunTransactionBlockResponse> {
15    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
16        let Pretty(response) = self;
17
18        writeln!(
19            f,
20            "Dry run completed, execution status: {}",
21            response.effects.status()
22        )?;
23
24        let mut builder = TableBuilder::default();
25        builder.push_record(vec![format!("{}", response.input)]);
26        let mut table = builder.build();
27        table.with(TablePanel::header("Dry Run Transaction Data"));
28        table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
29            1,
30            TableStyle::modern().get_horizontal(),
31        )]));
32        writeln!(f, "{}", table)?;
33        writeln!(f, "{}", response.effects)?;
34        write!(f, "{}", response.events)?;
35
36        if response.object_changes.is_empty() {
37            writeln!(f, "╭─────────────────────────────╮")?;
38            writeln!(f, "│ No object changes           │")?;
39            writeln!(f, "╰─────────────────────────────╯")?;
40        } else {
41            let mut builder = TableBuilder::default();
42            let (
43                mut created,
44                mut deleted,
45                mut mutated,
46                mut published,
47                mut transferred,
48                mut wrapped,
49            ) = (vec![], vec![], vec![], vec![], vec![], vec![]);
50            for obj in &response.object_changes {
51                match obj {
52                    ObjectChange::Created { .. } => created.push(obj),
53                    ObjectChange::Deleted { .. } => deleted.push(obj),
54                    ObjectChange::Mutated { .. } => mutated.push(obj),
55                    ObjectChange::Published { .. } => published.push(obj),
56                    ObjectChange::Transferred { .. } => transferred.push(obj),
57                    ObjectChange::Wrapped { .. } => wrapped.push(obj),
58                };
59            }
60
61            write_obj_changes(created, "Created", &mut builder)?;
62            write_obj_changes(deleted, "Deleted", &mut builder)?;
63            write_obj_changes(mutated, "Mutated", &mut builder)?;
64            write_obj_changes(published, "Published", &mut builder)?;
65            write_obj_changes(transferred, "Transferred", &mut builder)?;
66            write_obj_changes(wrapped, "Wrapped", &mut builder)?;
67
68            let mut table = builder.build();
69            table.with(TablePanel::header("Object Changes"));
70            table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
71                1,
72                TableStyle::modern().get_horizontal(),
73            )]));
74            writeln!(f, "{}", table)?;
75        }
76        if response.balance_changes.is_empty() {
77            writeln!(f, "╭─────────────────────────────╮")?;
78            writeln!(f, "│ No balance changes          │")?;
79            writeln!(f, "╰─────────────────────────────╯")?;
80        } else {
81            let mut builder = TableBuilder::default();
82            for balance in &response.balance_changes {
83                builder.push_record(vec![format!("{}", balance)]);
84            }
85            let mut table = builder.build();
86            table.with(TablePanel::header("Balance Changes"));
87            table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
88                1,
89                TableStyle::modern().get_horizontal(),
90            )]));
91            writeln!(f, "{}", table)?;
92        }
93        writeln!(
94            f,
95            "Dry run completed, execution status: {}",
96            response.effects.status()
97        )?;
98        writeln!(
99            f,
100            "Estimated gas cost (includes a small buffer): {} MIST",
101            estimate_gas_budget_from_gas_cost(
102                response.effects.gas_cost_summary(),
103                response.input.gas_data().price
104            )
105        )?;
106
107        if let Some(err) = &response.execution_error_source {
108            writeln!(f, "Execution error: {}", err)?;
109        }
110
111        Ok(())
112    }
113}
114
115fn write_obj_changes<T: Display>(
116    values: Vec<T>,
117    output_string: &str,
118    builder: &mut TableBuilder,
119) -> std::fmt::Result {
120    if !values.is_empty() {
121        builder.push_record(vec![format!("{} Objects: ", output_string)]);
122        for obj in values {
123            builder.push_record(vec![format!("{}", obj)]);
124        }
125    }
126    Ok(())
127}
