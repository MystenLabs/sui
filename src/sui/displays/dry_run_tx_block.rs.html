4use crate::{client_commands::estimate_gas_budget_from_gas_cost, displays::Pretty};
5use std::fmt::{Display, Formatter};
6use sui_json_rpc_types::{
7    DryRunTransactionBlockResponse, ObjectChange, SuiTransactionBlockDataAPI,
8    SuiTransactionBlockEffectsAPI,
9};
10use tabled::{
11    builder::Builder as TableBuilder,
12    settings::{Panel as TablePanel, Style as TableStyle, style::HorizontalLine},
13};
14
15impl Display for Pretty<'_, DryRunTransactionBlockResponse> {
16    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
17        let Pretty(response) = self;
18
19        writeln!(
20            f,
21            "Dry run completed, execution status: {}",
22            response.effects.status()
23        )?;
24
25        let mut builder = TableBuilder::default();
26        builder.push_record(vec![format!("{}", response.input)]);
27        let mut table = builder.build();
28        table.with(TablePanel::header("Dry Run Transaction Data"));
29        table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
30            1,
31            TableStyle::modern().get_horizontal(),
32        )]));
33        writeln!(f, "{}", table)?;
34        writeln!(f, "{}", response.effects)?;
35        write!(f, "{}", response.events)?;
36
37        if response.object_changes.is_empty() {
38            writeln!(f, "╭─────────────────────────────╮")?;
39            writeln!(f, "│ No object changes           │")?;
40            writeln!(f, "╰─────────────────────────────╯")?;
41        } else {
42            let mut builder = TableBuilder::default();
43            let (
44                mut created,
45                mut deleted,
46                mut mutated,
47                mut published,
48                mut transferred,
49                mut wrapped,
50            ) = (vec![], vec![], vec![], vec![], vec![], vec![]);
51            for obj in &response.object_changes {
52                match obj {
53                    ObjectChange::Created { .. } => created.push(obj),
54                    ObjectChange::Deleted { .. } => deleted.push(obj),
55                    ObjectChange::Mutated { .. } => mutated.push(obj),
56                    ObjectChange::Published { .. } => published.push(obj),
57                    ObjectChange::Transferred { .. } => transferred.push(obj),
58                    ObjectChange::Wrapped { .. } => wrapped.push(obj),
59                };
60            }
61
62            write_obj_changes(created, "Created", &mut builder)?;
63            write_obj_changes(deleted, "Deleted", &mut builder)?;
64            write_obj_changes(mutated, "Mutated", &mut builder)?;
65            write_obj_changes(published, "Published", &mut builder)?;
66            write_obj_changes(transferred, "Transferred", &mut builder)?;
67            write_obj_changes(wrapped, "Wrapped", &mut builder)?;
68
69            let mut table = builder.build();
70            table.with(TablePanel::header("Object Changes"));
71            table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
72                1,
73                TableStyle::modern().get_horizontal(),
74            )]));
75            writeln!(f, "{}", table)?;
76        }
77        if response.balance_changes.is_empty() {
78            writeln!(f, "╭─────────────────────────────╮")?;
79            writeln!(f, "│ No balance changes          │")?;
80            writeln!(f, "╰─────────────────────────────╯")?;
81        } else {
82            let mut builder = TableBuilder::default();
83            for balance in &response.balance_changes {
84                builder.push_record(vec![format!("{}", balance)]);
85            }
86            let mut table = builder.build();
87            table.with(TablePanel::header("Balance Changes"));
88            table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
89                1,
90                TableStyle::modern().get_horizontal(),
91            )]));
92            writeln!(f, "{}", table)?;
93        }
94        writeln!(
95            f,
96            "Dry run completed, execution status: {}",
97            response.effects.status()
98        )?;
99        writeln!(
100            f,
101            "Estimated gas cost (includes a small buffer): {} MIST",
102            estimate_gas_budget_from_gas_cost(
103                response.effects.gas_cost_summary(),
104                response.input.gas_data().price
105            )
106        )?;
107
108        if let Some(err) = &response.execution_error_source {
109            writeln!(f, "Execution error: {}", err)?;
110        }
111
112        Ok(())
113    }
114}
115
116fn write_obj_changes<T: Display>(
117    values: Vec<T>,
118    output_string: &str,
119    builder: &mut TableBuilder,
120) -> std::fmt::Result {
121    if !values.is_empty() {
122        builder.push_record(vec![format!("{} Objects: ", output_string)]);
123        for obj in values {
124            builder.push_record(vec![format!("{}", obj)]);
125        }
126    }
127    Ok(())
128}
