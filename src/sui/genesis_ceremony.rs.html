27#[derive(Parser)]
28pub struct Ceremony {
29    #[clap(long)]
30    path: Option<PathBuf>,
31
32    #[clap(long)]
33    protocol_version: Option<u64>,
34
35    #[clap(subcommand)]
36    command: CeremonyCommand,
37}
38
39impl Ceremony {
40    pub fn run(self) -> Result<()> {
41        run(self)
42    }
43}
44
45#[derive(Parser)]
46pub enum CeremonyCommand {
47    Init,
48
49    ValidateState,
50
51    AddValidator {
52        #[clap(long)]
53        name: String,
54        #[clap(long)]
55        validator_key_file: PathBuf,
56        #[clap(long)]
57        worker_key_file: PathBuf,
58        #[clap(long)]
59        account_key_file: PathBuf,
60        #[clap(long)]
61        network_key_file: PathBuf,
62        #[clap(long)]
63        network_address: Multiaddr,
64        #[clap(long)]
65        p2p_address: Multiaddr,
66        #[clap(long)]
67        narwhal_primary_address: Multiaddr,
68        #[clap(long)]
69        narwhal_worker_address: Multiaddr,
70        #[clap(long)]
71        description: String,
72        #[clap(long)]
73        image_url: String,
74        #[clap(long)]
75        project_url: String,
76    },
77
78    ListValidators,
79
80    BuildUnsignedCheckpoint,
81
82    ExamineGenesisCheckpoint,
83
84    VerifyAndSign {
85        #[clap(long)]
86        key_file: PathBuf,
87    },
88
89    Finalize,
90}

92pub fn run(cmd: Ceremony) -> Result<()> {
93    let dir = if let Some(path) = cmd.path {
94        path
95    } else {
96        std::env::current_dir()?
97    };
98    let dir = Utf8PathBuf::try_from(dir)?;
99
100    let protocol_version = cmd
101        .protocol_version
102        .map(ProtocolVersion::new)
103        .unwrap_or(ProtocolVersion::MAX);
104
105    match cmd.command {
106        CeremonyCommand::Init => {
107            let builder = Builder::new().with_protocol_version(protocol_version);
108            builder.save(dir)?;
109        }
110
111        CeremonyCommand::ValidateState => {
112            let builder = Builder::load(&dir)?;
113            builder.validate()?;
114        }
115
116        CeremonyCommand::AddValidator {
117            name,
118            validator_key_file,
119            worker_key_file,
120            account_key_file,
121            network_key_file,
122            network_address,
123            p2p_address,
124            narwhal_primary_address,
125            narwhal_worker_address,
126            description,
127            image_url,
128            project_url,
129        } => {
130            let mut builder = Builder::load(&dir)?;
131            let keypair: AuthorityKeyPair = read_authority_keypair_from_file(validator_key_file)?;
132            let account_keypair: SuiKeyPair = read_keypair_from_file(account_key_file)?;
133            let worker_keypair: NetworkKeyPair = read_network_keypair_from_file(worker_key_file)?;
134            let network_keypair: NetworkKeyPair = read_network_keypair_from_file(network_key_file)?;
135            let pop = generate_proof_of_possession(&keypair, (&account_keypair.public()).into());
136            builder = builder.add_validator(
137                sui_genesis_builder::validator_info::ValidatorInfo {
138                    name,
139                    protocol_key: keypair.public().into(),
140                    worker_key: worker_keypair.public().clone(),
141                    account_address: SuiAddress::from(&account_keypair.public()),
142                    network_key: network_keypair.public().clone(),
143                    gas_price: sui_config::node::DEFAULT_VALIDATOR_GAS_PRICE,
144                    commission_rate: sui_config::node::DEFAULT_COMMISSION_RATE,
145                    network_address,
146                    p2p_address,
147                    narwhal_primary_address,
148                    narwhal_worker_address,
149                    description,
150                    image_url,
151                    project_url,
152                },
153                pop,
154            );
155            builder.save(dir)?;
156        }
157
158        CeremonyCommand::ListValidators => {
159            let builder = Builder::load(&dir)?;
160
161            let mut writer = csv::Writer::from_writer(std::io::stdout());
162
163            writer.write_record(["validator-name", "account-address"])?;
164
165            let mut validators = builder
166                .validators()
167                .values()
168                .map(|v| {
169                    (
170                        v.info.name().to_lowercase(),
171                        v.info.account_address.to_string(),
172                    )
173                })
174                .collect::<Vec<_>>();
175
176            validators.sort_by_key(|v| v.0.clone());
177
178            for (name, address) in validators {
179                writer.write_record([&name, &address])?;
180            }
181        }
182
183        CeremonyCommand::BuildUnsignedCheckpoint => {
184            let mut builder = Builder::load(&dir)?;
185            let UnsignedGenesis { checkpoint, .. } = builder.build_unsigned_genesis_checkpoint();
186            println!(
187                "Successfully built unsigned checkpoint: {}",
188                checkpoint.digest()
189            );
190
191            builder.save(dir)?;
192        }
193
194        CeremonyCommand::ExamineGenesisCheckpoint => {
195            let builder = Builder::load(&dir)?;
196
197            let Some(unsigned_genesis) = builder.unsigned_genesis_checkpoint() else {
198                return Err(anyhow::anyhow!(
199                    "Unable to examine genesis checkpoint; it hasn't been built yet"
200                ));
201            };
202
203            examine_genesis_checkpoint(unsigned_genesis);
204        }
205
206        CeremonyCommand::VerifyAndSign { key_file } => {
207            let keypair: AuthorityKeyPair = read_authority_keypair_from_file(key_file)?;
208
209            let mut builder = Builder::load(&dir)?;
210
211            check_protocol_version(&builder, protocol_version)?;
212
213            // Don't sign unless the unsigned checkpoint has already been created
214            if builder.unsigned_genesis_checkpoint().is_none() {
215                return Err(anyhow::anyhow!(
216                    "Unable to verify and sign genesis checkpoint; it hasn't been built yet"
217                ));
218            }
219
220            builder = builder.add_validator_signature(&keypair);
221            let UnsignedGenesis { checkpoint, .. } = builder.unsigned_genesis_checkpoint().unwrap();
222            builder.save(dir)?;
223
224            println!(
225                "Successfully verified and signed genesis checkpoint: {}",
226                checkpoint.digest()
227            );
228        }
229
230        CeremonyCommand::Finalize => {
231            let builder = Builder::load(&dir)?;
232            check_protocol_version(&builder, protocol_version)?;
233
234            let genesis = builder.build();
235
236            genesis.save(dir.join(SUI_GENESIS_FILENAME))?;
237
238            println!("Successfully built {SUI_GENESIS_FILENAME}");
239            println!(
240                "{SUI_GENESIS_FILENAME} blake2b-256: {}",
241                Hex::encode(genesis.hash())
242            );
243        }
244    }
245
246    Ok(())
247}
248
249fn check_protocol_version(builder: &Builder, protocol_version: ProtocolVersion) -> Result<()> {
250    // It is entirely possible for the user to sign a genesis blob with an unknown
251    // protocol version, but if this happens there is almost certainly some confusion
252    // (e.g. using a `sui` binary built at the wrong commit).
253    if builder.protocol_version() != protocol_version {
254        return Err(anyhow::anyhow!(
255            "Serialized protocol version does not match local --protocol-version argument. ({:?} vs {:?})",
256            builder.protocol_version(),
257            protocol_version
258        ));
259    }
260    Ok(())
261}

263#[cfg(test)]
264mod test {
265    use super::*;
266    use anyhow::Result;
267    use sui_config::local_ip_utils;
268    use sui_genesis_builder::validator_info::ValidatorInfo;
269    use sui_keys::keypair_file::{write_authority_keypair_to_file, write_keypair_to_file};
270    use sui_macros::nondeterministic;
271    use sui_types::crypto::{AccountKeyPair, AuthorityKeyPair, SuiKeyPair, get_key_pair_from_rng};
272
273    #[test]
274    #[cfg_attr(msim, ignore)]
275    fn ceremony() -> Result<()> {
276        let dir = nondeterministic!(tempfile::TempDir::new().unwrap());
277
278        let validators = (0..10)
279            .map(|i| {
280                let keypair: AuthorityKeyPair = get_key_pair_from_rng(&mut rand::rngs::OsRng).1;
281                let worker_keypair: NetworkKeyPair =
282                    get_key_pair_from_rng(&mut rand::rngs::OsRng).1;
283                let network_keypair: NetworkKeyPair =
284                    get_key_pair_from_rng(&mut rand::rngs::OsRng).1;
285                let account_keypair: AccountKeyPair =
286                    get_key_pair_from_rng(&mut rand::rngs::OsRng).1;
287                let info = ValidatorInfo {
288                    name: format!("validator-{i}"),
289                    protocol_key: keypair.public().into(),
290                    worker_key: worker_keypair.public().clone(),
291                    account_address: SuiAddress::from(account_keypair.public()),
292                    network_key: network_keypair.public().clone(),
293                    gas_price: sui_config::node::DEFAULT_VALIDATOR_GAS_PRICE,
294                    commission_rate: sui_config::node::DEFAULT_COMMISSION_RATE,
295                    network_address: local_ip_utils::new_local_tcp_address_for_testing(),
296                    p2p_address: local_ip_utils::new_local_udp_address_for_testing(),
297                    narwhal_primary_address: local_ip_utils::new_local_udp_address_for_testing(),
298                    narwhal_worker_address: local_ip_utils::new_local_udp_address_for_testing(),
299                    description: String::new(),
300                    image_url: String::new(),
301                    project_url: String::new(),
302                };
303                let key_file = dir.path().join(format!("{}-0.key", info.name));
304                write_authority_keypair_to_file(&keypair, &key_file).unwrap();
305
306                let worker_key_file = dir.path().join(format!("{}.key", info.name));
307                write_keypair_to_file(&SuiKeyPair::Ed25519(worker_keypair), &worker_key_file)
308                    .unwrap();
309
310                let network_key_file = dir.path().join(format!("{}-1.key", info.name));
311                write_keypair_to_file(&SuiKeyPair::Ed25519(network_keypair), &network_key_file)
312                    .unwrap();
313
314                let account_key_file = dir.path().join(format!("{}-2.key", info.name));
315                write_keypair_to_file(&SuiKeyPair::Ed25519(account_keypair), &account_key_file)
316                    .unwrap();
317
318                (
319                    key_file,
320                    worker_key_file,
321                    network_key_file,
322                    account_key_file,
323                    info,
324                )
325            })
326            .collect::<Vec<_>>();
327
328        // Initialize
329        let command = Ceremony {
330            path: Some(dir.path().into()),
331            protocol_version: None,
332            command: CeremonyCommand::Init,
333        };
334        command.run()?;
335
336        // Add the validators
337        for (key_file, worker_key_file, network_key_file, account_key_file, validator) in
338            &validators
339        {
340            let command = Ceremony {
341                path: Some(dir.path().into()),
342                protocol_version: None,
343                command: CeremonyCommand::AddValidator {
344                    name: validator.name().to_owned(),
345                    validator_key_file: key_file.into(),
346                    worker_key_file: worker_key_file.into(),
347                    network_key_file: network_key_file.into(),
348                    account_key_file: account_key_file.into(),
349                    network_address: validator.network_address().to_owned(),
350                    p2p_address: validator.p2p_address().to_owned(),
351                    narwhal_primary_address: validator.narwhal_primary_address.clone(),
352                    narwhal_worker_address: validator.narwhal_worker_address.clone(),
353                    description: String::new(),
354                    image_url: String::new(),
355                    project_url: String::new(),
356                },
357            };
358            command.run()?;
359
360            Ceremony {
361                path: Some(dir.path().into()),
362                protocol_version: None,
363                command: CeremonyCommand::ValidateState,
364            }
365            .run()?;
366        }
367
368        // Build the unsigned checkpoint
369        let command = Ceremony {
370            path: Some(dir.path().into()),
371            protocol_version: None,
372            command: CeremonyCommand::BuildUnsignedCheckpoint,
373        };
374        command.run()?;
375
376        // Have all the validators verify and sign genesis
377        for (key, _worker_key, _network_key, _account_key, _validator) in &validators {
378            let command = Ceremony {
379                path: Some(dir.path().into()),
380                protocol_version: None,
381                command: CeremonyCommand::VerifyAndSign {
382                    key_file: key.into(),
383                },
384            };
385            command.run()?;
386
387            Ceremony {
388                path: Some(dir.path().into()),
389                protocol_version: None,
390                command: CeremonyCommand::ValidateState,
391            }
392            .run()?;
393        }
394
395        // Finalize the Ceremony and build the Genesis object
396        let command = Ceremony {
397            path: Some(dir.path().into()),
398            protocol_version: None,
399            command: CeremonyCommand::Finalize,
400        };
401        command.run()?;
402
403        Ok(())
404    }
