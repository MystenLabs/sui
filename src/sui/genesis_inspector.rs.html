17const STR_ALL: &str = "All";
18const STR_EXIT: &str = "Exit";
19const STR_SUI: &str = "Sui";
20const STR_STAKED_SUI: &str = "StakedSui";
21const STR_PACKAGE: &str = "Package";
22const STR_COIN_METADATA: &str = "CoinMetadata";
23const STR_OTHER: &str = "Other";
24const STR_SUI_DISTRIBUTION: &str = "Sui Distribution";
25const STR_OBJECTS: &str = "Objects";
26const STR_VALIDATORS: &str = "Validators";

28#[allow(clippy::or_fun_call)]
29pub(crate) fn examine_genesis_checkpoint(genesis: UnsignedGenesis) {
30    let system_object = genesis
31        .sui_system_object()
32        .into_genesis_version_for_tooling();
33
34    // Prepare Validator info
35    let validator_set = &system_object.validators.active_validators;
36    let validator_map = validator_set
37        .iter()
38        .map(|v| (v.verified_metadata().name.as_str(), v))
39        .collect::<BTreeMap<_, _>>();
40    let validator_pool_id_map = validator_set
41        .iter()
42        .map(|v| (v.staking_pool.id, v))
43        .collect::<BTreeMap<_, _>>();
44
45    let mut validator_options: Vec<_> = validator_map.keys().copied().collect();
46    validator_options.extend_from_slice(&[STR_ALL, STR_EXIT]);
47    println!("Total Number of Validators: {}", validator_set.len());
48
49    // Prepare Sui distribution info
50    let mut sui_distribution = BTreeMap::new();
51    let entry = sui_distribution
52        .entry("Sui System".to_string())
53        .or_insert(BTreeMap::new());
54    entry.insert(
55        "Storage Fund".to_string(),
56        (
57            STR_SUI,
58            system_object.storage_fund.non_refundable_balance.value(),
59        ),
60    );
61    entry.insert(
62        "Stake Subsidy".to_string(),
63        (STR_SUI, system_object.stake_subsidy.balance.value()),
64    );
65
66    // Prepare Object Info
67    let mut owner_map = BTreeMap::new();
68    let mut package_map = BTreeMap::new();
69    let mut sui_map = BTreeMap::new();
70    let mut staked_sui_map = BTreeMap::new();
71    let mut coin_metadata_map = BTreeMap::new();
72    let mut other_object_map = BTreeMap::new();
73
74    for object in genesis.objects() {
75        let object_id = object.id();
76        let object_id_str = object_id.to_string();
77        assert_eq!(object.storage_rebate, 0);
78        owner_map.insert(object.id(), object.owner.clone());
79
80        match &object.data {
81            sui_types::object::Data::Move(move_object) => {
82                if let Ok(gas) = GasCoin::try_from(object) {
83                    let entry = sui_distribution
84                        .entry(object.owner.to_string())
85                        .or_default();
86                    entry.insert(object_id_str.clone(), (STR_SUI, gas.value()));
87                    sui_map.insert(object.id(), gas);
88                } else if let Ok(coin_metadata) = CoinMetadata::try_from(object) {
89                    coin_metadata_map.insert(object.id(), coin_metadata);
90                } else if let Ok(staked_sui) = StakedSui::try_from(object) {
91                    let entry = sui_distribution
92                        .entry(object.owner.to_string())
93                        .or_default();
94                    entry.insert(object_id_str, (STR_STAKED_SUI, staked_sui.principal()));
95                    // Assert pool id is associated with a knonw validator.
96                    let validator = validator_pool_id_map.get(&staked_sui.pool_id()).unwrap();
97                    assert_eq!(validator.staking_pool.id, staked_sui.pool_id());
98
99                    staked_sui_map.insert(object.id(), staked_sui);
100                } else {
101                    other_object_map.insert(object.id(), move_object);
102                }
103            }
104            sui_types::object::Data::Package(p) => {
105                package_map.insert(object.id(), p);
106            }
107        }
108    }
109    println!(
110        "Total Number of Objects/Pacakges: {}",
111        genesis.objects().len()
112    );
113
114    // Always check the Total Supply
115    examine_total_supply(&sui_distribution, false);
116
117    // Main loop for inspection
118    let main_options: Vec<&str> = vec![STR_SUI_DISTRIBUTION, STR_VALIDATORS, STR_OBJECTS, STR_EXIT];
119    loop {
120        let ans = Select::new(
121            "Select one main category to examine ('Exit' to exit the program):",
122            main_options.clone(),
123        )
124        .prompt();
125        match ans {
126            Ok(name) if name == STR_SUI_DISTRIBUTION => {
127                examine_total_supply(&sui_distribution, true)
128            }
129            Ok(name) if name == STR_VALIDATORS => {
130                examine_validators(&validator_options, &validator_map);
131            }
132            Ok(name) if name == STR_OBJECTS => {
133                println!("Examine Objects (total: {})", genesis.objects().len());
134                examine_object(
135                    &owner_map,
136                    &validator_pool_id_map,
137                    &package_map,
138                    &sui_map,
139                    &staked_sui_map,
140                    &coin_metadata_map,
141                    &other_object_map,
142                );
143            }
144            Ok(name) if name == STR_EXIT => break,
145            Ok(_) => (),
146            Err(err) => {
147                println!("Error: {err}");
148                break;
149            }
150        }
151    }
152}

154#[allow(clippy::ptr_arg)]
155fn examine_validators(
156    validator_options: &Vec<&str>,
157    validator_map: &BTreeMap<&str, &SuiValidatorGenesis>,
158) {
159    loop {
160        let ans = Select::new("Select one validator to examine ('All' to display all Validators, 'Exit' to return to Main):", validator_options.clone()).prompt();
161        match ans {
162            Ok(name) if name == STR_ALL => {
163                for validator in validator_map.values() {
164                    display_validator(validator);
165                }
166            }
167            Ok(name) if name == STR_EXIT => break,
168            Ok(name) => {
169                let validator = validator_map.get(name).unwrap();
170                display_validator(validator);
171            }
172            Err(err) => {
173                println!("Error: {err}");
174                break;
175            }
176        }
177    }
178    print_divider("Validator");
179}

181fn examine_object(
182    owner_map: &BTreeMap<ObjectID, Owner>,
183    validator_pool_id_map: &BTreeMap<ObjectID, &SuiValidatorGenesis>,
184    package_map: &BTreeMap<ObjectID, &MovePackage>,
185    sui_map: &BTreeMap<ObjectID, GasCoin>,
186    staked_sui_map: &BTreeMap<ObjectID, StakedSui>,
187    coin_metadata_map: &BTreeMap<ObjectID, CoinMetadata>,
188    other_object_map: &BTreeMap<ObjectID, &MoveObject>,
189) {
190    let object_options: Vec<&str> = vec![
191        STR_SUI,
192        STR_STAKED_SUI,
193        STR_COIN_METADATA,
194        STR_PACKAGE,
195        STR_OTHER,
196        STR_EXIT,
197    ];
198    loop {
199        let ans = Select::new(
200            "Select one object category to examine ('Exit' to return to Main):",
201            object_options.clone(),
202        )
203        .prompt();
204        match ans {
205            Ok(name) if name == STR_EXIT => break,
206            Ok(name) if name == STR_SUI => {
207                for gas_coin in sui_map.values() {
208                    display_sui(gas_coin, owner_map);
209                }
210                print_divider("Sui");
211            }
212            Ok(name) if name == STR_STAKED_SUI => {
213                for staked_sui_coin in staked_sui_map.values() {
214                    display_staked_sui(staked_sui_coin, validator_pool_id_map, owner_map);
215                }
216                print_divider(STR_STAKED_SUI);
217            }
218            Ok(name) if name == STR_PACKAGE => {
219                for package in package_map.values() {
220                    println!("Package ID: {}", package.id());
221                    println!("Version: {}", package.version());
222                    println!("Modules: {:?}\n", package.serialized_module_map().keys());
223                }
224                print_divider("Package");
225            }
226            Ok(name) if name == STR_OTHER => {
227                for other_obj in other_object_map.values() {
228                    println!("{:#?}", other_obj.type_());
229                    println!("{:?}", other_obj.version());
230                    println!("Has Public Transfer: {}\n", other_obj.has_public_transfer());
231                }
232                print_divider("Other");
233            }
234            Ok(name) if name == STR_COIN_METADATA => {
235                for coin_metadata in coin_metadata_map.values() {
236                    println!("{:#?}\n", coin_metadata);
237                }
238                print_divider("CoinMetadata");
239            }
240            Ok(_) => (),
241            Err(err) => {
242                println!("Error: {err}");
243                break;
244            }
245        }
246    }
247    print_divider("Object");
248}

250fn examine_total_supply(
251    sui_distribution: &BTreeMap<String, BTreeMap<String, (&str, u64)>>,
252    print: bool,
253) {
254    let mut total_sui = 0;
255    let mut total_staked_sui = 0;
256    for (owner, coins) in sui_distribution {
257        let mut amount_sum = 0;
258        for (owner, value) in coins.values() {
259            amount_sum += value;
260            if *owner == STR_STAKED_SUI {
261                total_staked_sui += value;
262            }
263        }
264        total_sui += amount_sum;
265        if print {
266            println!("Owner {:?}", owner);
267            println!(
268                "Total Amount of Sui/StakedSui Owned: {amount_sum} MIST or {} SUI:",
269                amount_sum / MIST_PER_SUI
270            );
271            println!("{:#?}\n", coins);
272        }
273    }
274    assert_eq!(total_sui, TOTAL_SUPPLY_MIST);
275    // Always print this.
276    println!(
277        "Total Supply of Sui: {total_sui} MIST or {} SUI",
278        total_sui / MIST_PER_SUI
279    );
280    println!(
281        "Total Amount of StakedSui: {total_staked_sui} MIST or {} SUI\n",
282        total_staked_sui / MIST_PER_SUI
283    );
284    if print {
285        print_divider("Sui Distribution");
286    }
287}

289fn display_validator(validator: &SuiValidatorGenesis) {
290    let metadata = validator.verified_metadata();
291    println!("Validator name: {}", metadata.name);
292    println!("{:#?}", metadata);
293    println!("Voting Power: {}", validator.voting_power);
294    println!("Gas Price: {}", validator.gas_price);
295    println!("Next Epoch Gas Price: {}", validator.next_epoch_gas_price);
296    println!("Commission Rate: {}", validator.commission_rate);
297    println!(
298        "Next Epoch Commission Rate: {}",
299        validator.next_epoch_commission_rate
300    );
301    println!("Next Epoch Stake: {}", validator.next_epoch_stake);
302    println!("Staking Pool ID: {}", validator.staking_pool.id);
303    println!(
304        "Staking Pool Activation Epoch: {:?}",
305        validator.staking_pool.activation_epoch
306    );
307    println!(
308        "Staking Pool Deactivation Epoch: {:?}",
309        validator.staking_pool.deactivation_epoch
310    );
311    println!(
312        "Staking Pool Sui Balance: {:?}",
313        validator.staking_pool.sui_balance
314    );
315    println!(
316        "Rewards Pool: {}",
317        validator.staking_pool.rewards_pool.value()
318    );
319    println!(
320        "Pool Token Balance: {}",
321        validator.staking_pool.pool_token_balance
322    );
323    println!(
324        "Pending Delegation: {}",
325        validator.staking_pool.pending_stake
326    );
327    println!(
328        "Pending Total Sui Withdraw: {}",
329        validator.staking_pool.pending_total_sui_withdraw
330    );
331    println!(
332        "Pendign Pool Token Withdraw: {}",
333        validator.staking_pool.pending_pool_token_withdraw
334    );
335    println!(
336        "Exchange Rates ID: {}",
337        validator.staking_pool.exchange_rates.id
338    );
339    println!(
340        "Exchange Rates Size: {}",
341        validator.staking_pool.exchange_rates.size
342    );
343    print_divider(&metadata.name);
344}

346fn display_sui(gas_coin: &GasCoin, owner_map: &BTreeMap<ObjectID, Owner>) {
347    println!("ID: {}", gas_coin.id());
348    println!("Balance: {}", gas_coin.value());
349    println!("Owner: {}\n", owner_map.get(gas_coin.id()).unwrap());
350}

352fn display_staked_sui(
353    staked_sui: &StakedSui,
354    validator_pool_id_map: &BTreeMap<ObjectID, &SuiValidatorGenesis>,
355    owner_map: &BTreeMap<ObjectID, Owner>,
356) {
357    let validator = validator_pool_id_map.get(&staked_sui.pool_id()).unwrap();
358    println!("{:#?}", staked_sui);
359    println!(
360        "Staked to Validator: {}",
361        validator.verified_metadata().name
362    );
363    println!("Owner: {}\n", owner_map.get(&staked_sui.id()).unwrap());
364}

366fn print_divider(title: &str) {
367    let title = format!("End of {title}");
368    let divider_length = 80;
369    let left_divider_length = 10;
370    assert!(title.len() <= divider_length - left_divider_length * 2);
371    let divider_op = "-";
372    let divider = divider_op.repeat(divider_length);
373    let left_divider = divider_op.repeat(left_divider_length);
374    let margin_length = (divider_length - left_divider_length * 2 - title.len()) / 2;
375    let margin = " ".repeat(margin_length);
376    println!();
377    println!("{divider}");
378    println!("{left_divider}{margin}{title}{margin}{left_divider}");
379    println!("{divider}");
380    println!();
