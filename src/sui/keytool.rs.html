482impl KeyToolCommand {
483    pub async fn execute(self, keystore: &mut Keystore) -> Result<CommandOutput, anyhow::Error> {
484        Ok(match self {
485            KeyToolCommand::Alias {
486                old_alias,
487                new_alias,
488            } => {
489                let new_alias = keystore
490                    .update_alias(&old_alias, new_alias.as_deref())
491                    .await?;
492                CommandOutput::Alias(AliasUpdate {
493                    old_alias,
494                    new_alias,
495                })
496            }
497            KeyToolCommand::Convert { value } => {
498                let result = convert_private_key_to_bech32(value)?;
499                CommandOutput::Convert(result)
500            }
501
502            KeyToolCommand::DecodeMultiSig {
503                multisig,
504                tx_bytes,
505                cur_epoch,
506            } => {
507                let pks = multisig.get_pk().pubkeys();
508                let sigs = multisig.get_sigs();
509                let bitmap = multisig.get_indices()?;
510                let address = SuiAddress::from(multisig.get_pk());
511
512                let pub_keys = pks
513                    .iter()
514                    .map(|(pk, w)| MultiSigOutput {
515                        address: (pk).into(),
516                        public_base64_key: pk.encode_base64(),
517                        weight: *w,
518                    })
519                    .collect::<Vec<MultiSigOutput>>();
520
521                let threshold = *multisig.get_pk().threshold() as usize;
522
523                let mut output = DecodedMultiSigOutput {
524                    multisig_address: address,
525                    participating_keys_signatures: vec![],
526                    pub_keys,
527                    threshold,
528                    sig_verify_result: "".to_string(),
529                };
530
531                for (sig, i) in sigs.iter().zip(bitmap) {
532                    let (pk, w) = pks
533                        .get(i as usize)
534                        .ok_or(anyhow!("Invalid public keys index".to_string()))?;
535                    output.participating_keys_signatures.push(DecodedMultiSig {
536                        public_base64_key: pk.encode_base64().clone(),
537                        sig_base64: Base64::encode(sig.as_ref()),
538                        weight: w.to_string(),
539                    })
540                }
541
542                if tx_bytes.is_some() {
543                    let tx_bytes = Base64::decode(&tx_bytes.unwrap())
544                        .map_err(|e| anyhow!("Invalid base64 tx bytes: {:?}", e))?;
545                    let tx_data: TransactionData = bcs::from_bytes(&tx_bytes)?;
546                    let s = GenericSignature::MultiSig(multisig);
547                    let res = s.verify_authenticator(
548                        &IntentMessage::new(Intent::sui_transaction(), tx_data),
549                        address,
550                        cur_epoch,
551                        &VerifyParams::default(),
552                        Arc::new(VerifiedDigestCache::new_empty()),
553                    );
554
555                    match res {
556                        Ok(()) => output.sig_verify_result = "OK".to_string(),
557                        Err(e) => output.sig_verify_result = format!("{:?}", e),
558                    };
559                };
560
561                CommandOutput::DecodeMultiSig(output)
562            }
563
564            KeyToolCommand::DecodeOrVerifyTx {
565                tx_bytes,
566                sig,
567                cur_epoch,
568            } => {
569                let tx_bytes = Base64::decode(&tx_bytes)
570                    .map_err(|e| anyhow!("Invalid base64 key: {:?}", e))?;
571                let tx_data: TransactionData = bcs::from_bytes(&tx_bytes)?;
572                match sig {
573                    None => CommandOutput::DecodeOrVerifyTx(DecodeOrVerifyTxOutput {
574                        tx: tx_data,
575                        result: None,
576                    }),
577                    Some(s) => {
578                        let res = s.verify_authenticator(
579                            &IntentMessage::new(Intent::sui_transaction(), tx_data.clone()),
580                            tx_data.sender(),
581                            cur_epoch,
582                            &VerifyParams::default(),
583                            Arc::new(VerifiedDigestCache::new_empty()),
584                        );
585                        CommandOutput::DecodeOrVerifyTx(DecodeOrVerifyTxOutput {
586                            tx: tx_data,
587                            result: Some(res),
588                        })
589                    }
590                }
591            }
592            KeyToolCommand::Generate {
593                key_scheme,
594                derivation_path,
595                word_length,
596            } => match key_scheme {
597                SignatureScheme::BLS12381 => {
598                    let (sui_address, kp) = get_authority_key_pair();
599                    let file_name = format!("bls-{sui_address}.key");
600                    write_authority_keypair_to_file(&kp, file_name)?;
601                    CommandOutput::Generate(Key {
602                        alias: None,
603                        sui_address,
604                        public_base64_key: kp.public().encode_base64(),

1205                max_epoch,
1206                jwt_randomness,
1207                kp_bigint,
1208                ephemeral_key_identifier,
1209                network,
1210                test_multisig,
1211                sign_with_sk,
1212            } => {
1213                let tx_digest = perform_zk_login_test_tx(
1214                    &parsed_token,
1215                    max_epoch,
1216                    &jwt_randomness,
1217                    &kp_bigint,
1218                    ephemeral_key_identifier,
1219                    keystore,
1220                    &network,
1221                    test_multisig,
1222                    sign_with_sk,
1223                )
1224                .await?;
1225                CommandOutput::ZkLoginSignAndExecuteTx(ZkLoginSignAndExecuteTx { tx_digest })
1226            }
1227
1228            KeyToolCommand::ZkLoginSigVerify {
1229                sig,
1230                bytes,
1231                intent_scope,
1232                cur_epoch,
1233                network,
1234            } => {
1235                match GenericSignature::from_bytes(
1236                    &Base64::decode(&sig).map_err(|e| anyhow!("Invalid base64 sig: {:?}", e))?,
1237                )? {
1238                    GenericSignature::ZkLoginAuthenticator(zk) => {
1239                        if bytes.is_none() || cur_epoch.is_none() {
1240                            return Ok(CommandOutput::ZkLoginSigVerify(ZkLoginSigVerifyResponse {
1241                                data: None,
1242                                parsed: serde_json::to_string(&zk)?,
1243                                iss: zk.inputs.get_iss().to_owned(),
1244                                kid: zk.inputs.get_kid().to_owned(),
1245                                address_seed: zk.inputs.get_address_seed().to_string(),
1246                                res: None,
1247                                jwks: None,
1248                            }));
1249                        }
1250
1251                        let client = reqwest::Client::new();
1252                        let provider = OIDCProvider::from_iss(zk.get_iss())
1253                            .map_err(|_| anyhow!("Invalid iss"))?;
1254                        let jwks = fetch_jwks(&provider, &client, true).await?;
1255                        let parsed: ImHashMap<JwkId, JWK> = jwks.clone().into_iter().collect();
1256                        let env = match network.as_str() {
1257                            "devnet" | "localnet" => ZkLoginEnv::Test,
1258                            "mainnet" | "testnet" => ZkLoginEnv::Prod,
1259                            _ => return Err(anyhow!("Invalid network")),
1260                        };
1261                        let verify_params =
1262                            VerifyParams::new(parsed, vec![], env, true, true, true, Some(2), true);
1263
1264                        let (serialized, res) = match IntentScope::try_from(intent_scope)
1265                            .map_err(|_| anyhow!("Invalid scope"))?
1266                        {
1267                            IntentScope::TransactionData => {
1268                                let tx_data: TransactionData = bcs::from_bytes(
1269                                    &Base64::decode(&bytes.unwrap())
1270                                        .map_err(|e| anyhow!("Invalid base64 tx data: {:?}", e))?,
1271                                )?;
1272
1273                                let sig = GenericSignature::ZkLoginAuthenticator(zk.clone());
1274                                let res = sig.verify_authenticator(
1275                                    &IntentMessage::new(Intent::sui_transaction(), tx_data.clone()),
1276                                    tx_data.execution_parts().1,
1277                                    cur_epoch.unwrap(),
1278                                    &verify_params,
1279                                    Arc::new(VerifiedDigestCache::new_empty()),
1280                                );
1281                                (serde_json::to_string(&tx_data)?, res)
1282                            }
1283                            IntentScope::PersonalMessage => {
1284                                let data = PersonalMessage {
1285                                    message: Base64::decode(&bytes.unwrap()).map_err(|e| {
1286                                        anyhow!("Invalid base64 personal message data: {:?}", e)
1287                                    })?,
1288                                };
1289
1290                                let sig = GenericSignature::ZkLoginAuthenticator(zk.clone());
1291                                let res = sig.verify_authenticator(
1292                                    &IntentMessage::new(Intent::personal_message(), data.clone()),
1293                                    (&zk).try_into()?,
1294                                    cur_epoch.unwrap(),
1295                                    &verify_params,
1296                                    Arc::new(VerifiedDigestCache::new_empty()),
1297                                );
1298                                (serde_json::to_string(&data)?, res)
1299                            }
1300                            _ => return Err(anyhow!("Invalid intent scope")),
1301                        };
1302                        CommandOutput::ZkLoginSigVerify(ZkLoginSigVerifyResponse {
1303                            data: Some(serialized),
1304                            parsed: serde_json::to_string(&zk)?,
1305                            iss: zk.inputs.get_iss().to_owned(),
1306                            kid: zk.inputs.get_kid().to_owned(),
1307                            address_seed: zk.inputs.get_address_seed().to_string(),
1308                            jwks: Some(serde_json::to_string(&jwks)?),
1309                            res: Some(res),
1310                        })
1311                    }
1312                    _ => CommandOutput::Error("Not a zkLogin signature".to_string()),
1313                }
1314            }
1315        })
1316    }
1317}

1385                table.with(style);

1419/// Converts legacy formatted private key to 33 bytes bech32 encoded private key or vice versa.
1420/// It can handle:
1421/// 1) Hex encoded 32 byte private key (assumes scheme is Ed25519), this is the legacy wallet format
1422/// 2) Base64 encoded 32 bytes private key (assumes scheme is Ed25519)
1423/// 3) Base64 encoded 33 bytes private key with flag.
1424/// 4) Bech32 encoded 33 bytes private key with flag.
1425fn convert_private_key_to_bech32(value: String) -> Result<ConvertOutput, anyhow::Error> {
1426    let skp = match SuiKeyPair::decode(&value) {
1427        Ok(s) => s,
1428        Err(_) => match Hex::decode(&value) {
1429            Ok(decoded) => {
1430                if decoded.len() != 32 {
1431                    return Err(anyhow!(format!(
1432                        "Invalid private key length, expected 32 but got {}",
1433                        decoded.len()
1434                    )));
1435                }
1436                SuiKeyPair::Ed25519(Ed25519KeyPair::from_bytes(&decoded)?)
1437            }
1438            Err(_) => match SuiKeyPair::decode_base64(&value) {
1439                Ok(skp) => skp,
1440                Err(_) => match Ed25519KeyPair::decode_base64(&value) {
1441                    Ok(kp) => SuiKeyPair::Ed25519(kp),
1442                    Err(_) => return Err(anyhow!("Invalid private key encoding")),
1443                },
1444            },
1445        },
1446    };
1447
1448    Ok(ConvertOutput {
1449        bech32_with_flag: skp.encode().map_err(|_| anyhow!("Cannot encode keypair"))?,
1450        base64_with_flag: skp.encode_base64(),
1451        hex_without_flag: Hex::encode(&skp.to_bytes()[1..]),
1452        scheme: skp.public().scheme().to_string(),
1453    })
1454}
