4use anyhow::Error;
5use serde::{Deserialize, Serialize};
6use serde_json::json;
7use std::collections::{BTreeMap, BTreeSet};
8use sui_protocol_config::Chain;
9use sui_rpc_api::Client;
10use sui_types::base_types::ObjectID;
11
12const MVR_RESOLVER_MAINNET_URL: &str = "https://mainnet.mvr.mystenlabs.com";
13const MVR_RESOLVER_TESTNET_URL: &str = "https://testnet.mvr.mystenlabs.com";
14
15#[derive(Debug, Serialize)]
16pub struct MvrResolver {
17    pub names: BTreeSet<String>,
18}
19
20#[derive(Debug, Serialize, Deserialize)]
21pub struct ResolvedNames {
22    pub resolution: BTreeMap<String, PackageId>,
23}
24
25#[derive(Debug, Serialize, Deserialize)]
26pub struct PackageId {
27    pub package_id: ObjectID,
28}
29
30#[derive(Debug, Serialize, Deserialize)]
31pub struct NamesRequest {
32    pub names: BTreeSet<String>,
33}
34
35impl MvrResolver {
36    /// Call this function before calling resolve_names to avoid making a call to the resolver if
37    /// not needed.
38    pub fn should_resolve(&self) -> bool {
39        !self.names.is_empty()
40    }
41
42    /// Given a set of MVR names, resolve them to their corresponding package IDs. Note that this
43    /// API will error if the resolved list length does not match with the given input.
44    pub async fn resolve_names(&self, client: &Client) -> Result<ResolvedNames, Error> {
45        if self.names.is_empty() {
46            return Ok(ResolvedNames {
47                resolution: BTreeMap::new(),
48            });
49        }
50
51        let request = reqwest::Client::new();
52        let (url, chain) = mvr_req_url(client).await?;
53        let json_body = json!(NamesRequest {
54            names: self.names.clone()
55        });
56        let response = request
57            .post(format!("{url}/v1/resolution/bulk"))
58            .header("Content-Type", "application/json")
59            .json(&json_body)
60            .send()
61            .await?;
62
63        let resolved_addresses: ResolvedNames = response.json().await?;
64
65        anyhow::ensure!(
66            resolved_addresses.resolution.len() == self.names.len(),
67            "Could not find package id for {} for {chain} enviroment",
68            self.names
69                .difference(
70                    &resolved_addresses
71                        .resolution
72                        .keys()
73                        .cloned()
74                        .collect::<BTreeSet<_>>()
75                )
76                .cloned()
77                .collect::<Vec<_>>()
78                .join(", ")
79        );
80
81        Ok(resolved_addresses)
82    }
83}
84
85/// Based on the chain id of the current set environment, return the correct MVR URL to use for
86/// resolution.
87async fn mvr_req_url(client: &Client) -> Result<(&'static str, &'static str), Error> {
88    let chain = client.get_chain_identifier().await?;
89    let chain = chain.chain();
90    match chain {
91        Chain::Mainnet => Ok((MVR_RESOLVER_MAINNET_URL, "mainnet")),
92        Chain::Testnet => Ok((MVR_RESOLVER_TESTNET_URL, "testnet")),
93        Chain::Unknown => {
94            anyhow::bail!("Unsupported chain identifier: {:?}", chain);
95        }
96    }
97}
