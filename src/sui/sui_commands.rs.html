405impl SuiCommand {
406    pub async fn execute(self) -> Result<(), anyhow::Error> {
407        move_package::package_hooks::register_package_hooks(Box::new(SuiPackageHooks));
408        match self {
409            SuiCommand::Network {
410                config,
411                dump_addresses,
412            } => {
413                let config_path = config.unwrap_or(sui_config_dir()?.join(SUI_NETWORK_CONFIG));
414                let config: NetworkConfig = PersistedConfig::read(&config_path).map_err(|err| {
415                    err.context(format!(
416                        "Cannot open Sui network config file at {:?}",
417                        config_path
418                    ))
419                })?;
420
421                if dump_addresses {
422                    for validator in config.validator_configs() {
423                        println!(
424                            "{} - {}",
425                            validator.network_address(),
426                            validator.protocol_key_pair().public(),
427                        );
428                    }
429                }
430                Ok(())
431            }
432            SuiCommand::Start {
433                config_dir,
434                force_regenesis,
435                with_faucet,
436                rpc_args,
437                fullnode_rpc_port,
438                data_ingestion_dir,
439                no_full_node,
440                epoch_duration_ms,
441                committee_size,
442            } => {
443                start(
444                    config_dir.clone(),
445                    with_faucet,
446                    rpc_args,
447                    force_regenesis,
448                    epoch_duration_ms,
449                    fullnode_rpc_port,
450                    data_ingestion_dir,
451                    no_full_node,
452                    committee_size,
453                )
454                .await?;
455
456                Ok(())
457            }
458            SuiCommand::Genesis {
459                working_dir,
460                force,
461                from_config,
462                write_config,
463                epoch_duration_ms,
464                benchmark_ips,
465                with_faucet,
466                committee_size,
467            } => {
468                genesis(
469                    from_config,
470                    write_config,
471                    working_dir,
472                    force,
473                    epoch_duration_ms,
474                    benchmark_ips,
475                    with_faucet,
476                    committee_size,
477                )
478                .await
479            }
480            SuiCommand::GenesisCeremony(cmd) => run(cmd),
481            SuiCommand::KeyTool {
482                keystore_path,
483                json,
484                cmd,
485            } => {
486                let keystore_path =
487                    keystore_path.unwrap_or(sui_config_dir()?.join(SUI_KEYSTORE_FILENAME));
488                let mut keystore =
489                    Keystore::from(FileBasedKeystore::load_or_create(&keystore_path)?);
490                cmd.execute(&mut keystore).await?.print(!json);
491                Ok(())
492            }
493            SuiCommand::Client {
494                config,
495                cmd,
496                json,
497                accept_defaults,
498            } => {
499                let config_path = config
500                    .config
501                    .unwrap_or(sui_config_dir()?.join(SUI_CLIENT_CONFIG));
502                prompt_if_no_config(&config_path, accept_defaults).await?;
503                if let Some(cmd) = cmd {
504                    let mut context = WalletContext::new(&config_path)?;
505                    if let Some(env_override) = config.env {
506                        context = context.with_env_override(env_override);
507                    }
508                    if let Ok(client) = context.get_client().await
509                        && let Err(e) = client.check_api_version()
510                    {
511                        eprintln!("{}", format!("[warning] {e}").yellow().bold());
512                    }
513                    cmd.execute(&mut context).await?.print(!json);
514                } else {
515                    // Print help
516                    let mut app: Command = SuiCommand::command();
517                    app.build();
518                    app.find_subcommand_mut("client").unwrap().print_help()?;
519                }
520                Ok(())
521            }
522            SuiCommand::Validator {
523                config,
524                cmd,
525                json,
526                accept_defaults,
527            } => {
528                let config_path = config.unwrap_or(sui_config_dir()?.join(SUI_CLIENT_CONFIG));
529                prompt_if_no_config(&config_path, accept_defaults).await?;
530                let mut context = WalletContext::new(&config_path)?;
531                if let Some(cmd) = cmd {
532                    if let Ok(client) = context.get_client().await
533                        && let Err(e) = client.check_api_version()
534                    {
535                        eprintln!("{}", format!("[warning] {e}").yellow().bold());
536                    }
537                    cmd.execute(&mut context).await?.print(!json);
538                } else {
539                    // Print help
540                    let mut app: Command = SuiCommand::command();
541                    app.build();
542                    app.find_subcommand_mut("validator").unwrap().print_help()?;
543                }
544                Ok(())
545            }
546            SuiCommand::Move {
547                package_path,
548                build_config,
549                mut cmd,
550                config: client_config,
551            } => {
552                match cmd {
553                    sui_move::Command::Summary(mut s) if s.package_id.is_some() => {
554                        let (_, client) = get_chain_id_and_client(
555                            client_config,
556                            "sui move summary --package-id <object_id>",
557                        )
558                        .await?;
559                        let Some(client) = client else {
560                            bail!(
561                                "`sui move summary --package-id <object_id>` requires a configured network"
562                            );
563                        };
564
565                        let read_api = client.read_api();
566
567                        // If they didn't run with `--bytecode` correct this for them but warn them
568                        // to let them know that we are changing it.
569                        if !s.summary.bytecode {
570                            eprintln!("{}",
571                                "[warning] `sui move summary --package-id <object_id>` only supports bytecode summaries. \
572                                 Falling back to producing a bytecode-based summary. To not get this warning you can run with `--bytecode`".yellow().bold()
573                            );
574                            s.summary.bytecode = true;
575                        }
576                        let root_package_id = s
577                            .package_id
578                            .as_ref()
579                            .expect("Safe since we checked in the match statement");
580
581                        // Create a tempdir to download the package bytes to, and then download the
582                        // packages bytes there.
583                        let package_bytes_location = tempdir()?;
584                        let path = package_bytes_location.path();
585                        let package_metadata =
586                            download_package_and_deps_under(read_api, path, *root_package_id)
587                                .await?;
588
589                        // Now produce the summary, pointing at the tempdir containing the package
590                        // bytes.
591                        execute_move_command(
592                            Some(path),
593                            build_config,
594                            sui_move::Command::Summary(s),
595                            Some(sui_move::CommandMeta::Summary(package_metadata)),
596                        )?;
597                        return Ok(());
598                    }
599                    sui_move::Command::Build(build) if build.dump_bytecode_as_base64 => {
600                        // `sui move build` does not ordinarily require a network connection.
601                        // The exception is when --dump-bytecode-as-base64 is specified: In this
602                        // case, we should resolve the correct addresses for the respective chain
603                        // (e.g., testnet, mainnet) from the Move.lock under automated address management.
604                        // In addition, tree shaking also requires a network as it needs to fetch
605                        // on-chain linkage table of package dependencies.
606                        let (chain_id, client) = if build.ignore_chain {
607                            // for tests it's useful to ignore the chain id!
608                            (None, None)
609                        } else {
610                            get_chain_id_and_client(
611                                client_config,
612                                "sui move build --dump-bytecode-as-base64",
613                            )
614                            .await?
615                        };
616
617                        let rerooted_path = move_cli::base::reroot_path(package_path.as_deref())?;
618                        let mut build_config =
619                            resolve_lock_file_path(build_config, Some(&rerooted_path))?;
620
621                        let previous_id = if let Some(ref chain_id) = chain_id {
622                            sui_package_management::set_package_id(
623                                &rerooted_path,
624                                build_config.install_dir.clone(),
625                                chain_id,
626                                AccountAddress::ZERO,
627                            )?
628                        } else {
629                            None
630                        };
631
632                        if let Some(client) = &client {
633                            let protocol_config =
634                                client.read_api().get_protocol_config(None).await?;
635                            build_config.implicit_dependencies =
636                                implicit_deps_for_protocol_version(
637                                    protocol_config.protocol_version,
638                                )?;
639                        } else {
640                            build_config.implicit_dependencies =
641                                implicit_deps(latest_system_packages());
642                        }
643
644                        let mut pkg = SuiBuildConfig {
645                            config: build_config.clone(),
646                            run_bytecode_verifier: true,
647                            print_diags_to_stderr: true,
648                            chain_id: chain_id.clone(),
649                        }
650                        .build(&rerooted_path)?;
651
652                        // Restore original ID, then check result.
653                        if let (Some(chain_id), Some(previous_id)) = (chain_id, previous_id) {
654                            let _ = sui_package_management::set_package_id(
655                                &rerooted_path,
656                                build_config.install_dir.clone(),
657                                &chain_id,
658                                previous_id,
659                            )?;
660                        }
661
662                        let with_unpublished_deps = build.with_unpublished_dependencies;
663
664                        check_conflicting_addresses(&pkg.dependency_ids.conflicting, true)?;
665                        check_invalid_dependencies(&pkg.dependency_ids.invalid)?;
666                        if !with_unpublished_deps {
667                            check_unpublished_dependencies(&pkg.dependency_ids.unpublished)?;
668                        }
669
670                        if let Some(client) = client {
671                            pkg_tree_shake(client.read_api(), with_unpublished_deps, &mut pkg)
672                                .await?;
673                        }
674
675                        println!(
676                            "{}",
677                            json!({
678                                "modules": pkg.get_package_base64(with_unpublished_deps),
679                                "dependencies": pkg.get_dependency_storage_package_ids(),
680                                "digest": pkg.get_package_digest(with_unpublished_deps),
681                            })
682                        );
683                        return Ok(());
684                    }
685                    _ => (),
686                };
687
688                // If a specific environment is specified for the build command we set the chain ID
689                // to the one that is specified.
690                if client_config.env.is_some() && matches!(cmd, sui_move::Command::Build(_)) {
691                    let (chain_id, _) =
692                        get_chain_id_and_client(client_config, "sui move build").await?;
693
694                    let sui_move::Command::Build(build_config) = &mut cmd else {
695                        unreachable!("We checked for Build above, so this should never happen");
696                    };
697
698                    build_config.chain_id = chain_id;
699                }
700
701                execute_move_command(package_path.as_deref(), build_config, cmd, None)
702            }
703            SuiCommand::BridgeInitialize {
704                network_config,
705                client_config,
706                bridge_committee_config_path,
707            } => {
708                // Load the config of the Sui authority.
709                let network_config_path = network_config
710                    .clone()
711                    .unwrap_or(sui_config_dir()?.join(SUI_NETWORK_CONFIG));
712                let network_config: NetworkConfig = PersistedConfig::read(&network_config_path)
713                    .map_err(|err| {
714                        err.context(format!(
715                            "Cannot open Sui network config file at {:?}",
716                            network_config_path
717                        ))
718                    })?;
719                let bridge_committee_config: BridgeCommitteeConfig =
720                    PersistedConfig::read(&bridge_committee_config_path).map_err(|err| {
721                        err.context(format!(
722                            "Cannot open Bridge Committee config file at {:?}",
723                            bridge_committee_config_path
724                        ))
725                    })?;
726
727                let config_path =
728                    client_config.unwrap_or(sui_config_dir()?.join(SUI_CLIENT_CONFIG));
729                let mut context = WalletContext::new(&config_path)?;
730                if let Ok(client) = context.get_client().await
731                    && let Err(e) = client.check_api_version()
732                {
733                    eprintln!("{}", format!("[warning] {e}").yellow().bold());
734                }
735                let rgp = context.get_reference_gas_price().await?;
736                let rpc_url = &context.get_active_env()?.rpc;
737                println!("rpc_url: {}", rpc_url);
738                let bridge_metrics = Arc::new(BridgeMetrics::new_for_testing());
739                let sui_bridge_client = SuiBridgeClient::new(rpc_url, bridge_metrics).await?;
740                let bridge_arg = sui_bridge_client
741                    .get_mutable_bridge_object_arg_must_succeed()
742                    .await;
743                assert_eq!(
744                    network_config.validator_configs().len(),
745                    bridge_committee_config
746                        .bridge_authority_port_and_key_path
747                        .len()
748                );
749                for node_config in network_config.validator_configs() {
750                    let account_kp = node_config.account_key_pair.keypair();
751                    context.add_account(None, account_kp.copy()).await;
752                }
753
754                let context = context;
755                let mut tasks = vec![];
756                for (node_config, (port, key_path)) in network_config
757                    .validator_configs()
758                    .iter()
759                    .zip(bridge_committee_config.bridge_authority_port_and_key_path)
760                {
761                    let account_kp = node_config.account_key_pair.keypair();
762                    let sui_address = SuiAddress::from(&account_kp.public());
763                    let gas_obj_ref = context
764                        .get_one_gas_object_owned_by_address(sui_address)
765                        .await?
766                        .expect("Validator does not own any gas objects");
767                    let kp = match read_key(&key_path, true)? {
768                        SuiKeyPair::Secp256k1(key) => key,
769                        _ => unreachable!("we required secp256k1 key in `read_key`"),
770                    };
771
772                    // build registration tx
773                    let tx = build_committee_register_transaction(
774                        sui_address,
775                        &gas_obj_ref,
776                        bridge_arg,
777                        kp.public().as_bytes().to_vec(),
778                        &format!("http://127.0.0.1:{port}"),
779                        rgp,
780                        1000000000,
781                    )
782                    .unwrap();
783                    let signed_tx = context.sign_transaction(&tx).await;
784                    tasks.push(context.execute_transaction_must_succeed(signed_tx));
785                }
786                future::join_all(tasks).await;
787                Ok(())
788            }
789            SuiCommand::FireDrill { fire_drill } => run_fire_drill(fire_drill).await,
790            SuiCommand::Analyzer => {
791                let sui_implicit_deps = implicit_deps(latest_system_packages());
792                let flavor = Flavor::Sui;
793                let sui_pkg_hooks = Box::new(SuiPackageHooks);
794                analyzer::run(sui_implicit_deps, Some(flavor), Some(sui_pkg_hooks));
795                Ok(())
796            }
797            SuiCommand::AnalyzeTrace {
798                path,
799                output_dir,
800                command,
801            } => command.execute(path, output_dir).await,
802            SuiCommand::ReplayTransaction {
803                config,
804                replay_config,
805            } => {
806                let config_path = config
807                    .config
808                    .unwrap_or(sui_config_dir()?.join(SUI_CLIENT_CONFIG));
809                prompt_if_no_config(&config_path, /* accept_defaults */ false).await?;
810                let mut context = WalletContext::new(&config_path)?;
811                if let Some(env_override) = config.env {
812                    context = context.with_env_override(env_override);
813                }
814
815                let node = get_replay_node(&context).await?;
816                let file_config = SR2::load_config_file()?;
817                let stable_config = SR2::merge_configs(replay_config, file_config);
818                let experimental_config = SR2::ReplayConfigExperimental {
819                    node,
820                    ..Default::default()
821                };
822
823                let artifact_path =
824                    SR2::handle_replay_config(&stable_config, &experimental_config, USER_AGENT)
825                        .await?;
826
827                if let Some(digest) = &stable_config.digest {
828                    SR2::print_effects_or_fork(
829                        digest,
830                        &artifact_path,
831                        stable_config.show_effects,
832                        &mut std::io::stdout(),
833                    )?;
834                }
835
836                Ok(())
837            }
838        }
839    }
840}

842/// Starts a local network with the given configuration.
843async fn start(
844    config: Option<PathBuf>,
845    with_faucet: Option<String>,
846    rpc_args: RpcArgs,
847    force_regenesis: bool,
848    epoch_duration_ms: Option<u64>,
849    fullnode_rpc_port: u16,
850    mut data_ingestion_dir: Option<PathBuf>,
851    no_full_node: bool,
852    committee_size: Option<usize>,
853) -> Result<(), anyhow::Error> {
854    if force_regenesis {
855        ensure!(
856            config.is_none(),
857            "Cannot pass `--force-regenesis` and `--network.config` at the same time."
858        );
859    }
860
861    let RpcArgs {
862        with_indexer,
863        mut with_consistent_store,
864        with_graphql,
865    } = rpc_args;
866
867    // Automatically enable consistent store if GraphQL is enabled
868    if with_graphql.is_some() && with_consistent_store.is_none() {
869        with_consistent_store = Some("0.0.0.0:9124".to_string());
870    }
871
872    // Automatically enable indexer if GraphQL is enabled
873    // If with_indexer is None but with_graphql is Some, default to temporary database (Some(None))
874    let with_indexer = match (with_indexer, with_graphql.is_some()) {
875        (Some(db_url), _) => Some(db_url),
876        (None, true) => Some(None),
877        (None, false) => None,
878    };
879
880    if with_indexer.is_some() {
881        ensure!(
882            !no_full_node,
883            "Cannot start the indexer without a fullnode."
884        );
885    }
886
887    if epoch_duration_ms.is_some() && genesis_blob_exists(config.clone()) && !force_regenesis {
888        bail!(
889            "Epoch duration can only be set when passing the `--force-regenesis` flag, or when \
890            there is no genesis configuration in the default Sui configuration folder or the given \
891            network.config argument.",
892        );
893    }
894
895    let mut swarm_builder = Swarm::builder();
896
897    // If this is set, then no data will be persisted between runs, and a new genesis will be
898    // generated each run.
899    let config_dir = if force_regenesis {
900        let committee_size = match committee_size {
901            Some(x) => NonZeroUsize::new(x),
902            None => NonZeroUsize::new(1),
903        }
904        .ok_or_else(|| anyhow!("Committee size must be at least 1."))?;
905        swarm_builder = swarm_builder.committee_size(committee_size);
906        let genesis_config = GenesisConfig::custom_genesis(1, 100);
907        swarm_builder = swarm_builder.with_genesis_config(genesis_config);
908        let epoch_duration_ms = epoch_duration_ms.unwrap_or(DEFAULT_EPOCH_DURATION_MS);
909        swarm_builder = swarm_builder.with_epoch_duration_ms(epoch_duration_ms);
910        mysten_common::tempdir()?.keep()
911    } else {
912        // If the config path looks like a YAML file, it is treated as if it is the network.yaml
913        // overriding the network.yaml found in the sui config directry. Otherwise it is treated as
914        // the sui config directory for backwards compatibility with `sui-test-validator`.
915        let (network_config_path, sui_config_path) = match config {
916            Some(config)
917                if config.is_file()
918                    && config
919                        .extension()
920                        .is_some_and(|e| e == "yml" || e == "yaml") =>
921            {
922                if committee_size.is_some() {
923                    eprintln!(
924                        "{}",
925                        "[warning] The committee-size arg wil be ignored as a network \
926                            configuration already exists. To change the committee-size, you'll \
927                            have to adjust the network configuration file or regenerate a genesis \
928                            with the desired committee size. See `sui genesis --help` for more \
929                            information."
930                            .yellow()
931                            .bold()
932                    );
933                }
934                (config, sui_config_dir()?)
935            }
936
937            Some(config) => {
938                if committee_size.is_some() {
939                    eprintln!(
940                        "{}",
941                        "[warning] The committee-size arg wil be ignored as a network \
942                            configuration already exists. To change the committee-size, you'll \
943                            have to adjust the network configuration file or regenerate a genesis \
944                            with the desired committee size. See `sui genesis --help` for more \
945                            information."
946                            .yellow()
947                            .bold()
948                    );
949                }
950                (config.join(SUI_NETWORK_CONFIG), config)
951            }
952
953            None => {
954                let sui_config = sui_config_dir()?;
955                let network_config = sui_config.join(SUI_NETWORK_CONFIG);
956
957                if !network_config.exists() {
958                    genesis(
959                        None,
960                        None,
961                        None,
962                        false,
963                        epoch_duration_ms,
964                        None,
965                        false,
966                        committee_size,
967                    )
968                    .await
969                    .map_err(|_| {
970                        anyhow!(
971                            "Cannot run genesis with non-empty Sui config directory: {}.\n\n\
972                                If you are trying to run a local network without persisting the \
973                                data (so a new genesis that is randomly generated and will not be \
974                                saved once the network is shut down), use --force-regenesis flag.\n\
975                                If you are trying to persist the network data and start from a new \
976                                genesis, use sui genesis --help to see how to generate a new \
977                                genesis.",
978                            sui_config.display(),
979                        )
980                    })?;
981                } else if committee_size.is_some() {
982                    eprintln!(
983                        "{}",
984                        "[warning] The committee-size arg wil be ignored as a network \
985                            configuration already exists. To change the committee-size, you'll \
986                            have to adjust the network configuration file or regenerate a genesis \
987                            with the desired committee size. See `sui genesis --help` for more \
988                            information."
989                            .yellow()
990                            .bold()
991                    );
992                }
993
994                (network_config, sui_config)
995            }
996        };
997
998        // Load the config of the Sui authority.
999        let network_config: NetworkConfig =
1000            PersistedConfig::read(&network_config_path).map_err(|err| {
1001                err.context(format!(
1002                    "Cannot open Sui network config file at {:?}",
1003                    network_config_path
1004                ))
1005            })?;
1006
1007        swarm_builder = swarm_builder
1008            .dir(sui_config_path.clone())
1009            .with_network_config(network_config);
1010
1011        sui_config_path
1012    };
1013
1014    // the indexer requires to set the fullnode's data ingestion directory
1015    // note that this overrides the default configuration that is set when running the genesis
1016    // command, which sets data_ingestion_dir to None.
1017    if with_indexer.is_some() && data_ingestion_dir.is_none() {
1018        data_ingestion_dir = Some(mysten_common::tempdir()?.keep())
1019    }
1020
1021    if let Some(ref dir) = data_ingestion_dir {
1022        swarm_builder = swarm_builder.with_data_ingestion_dir(dir.clone());
1023    }
1024
1025    let mut fullnode_rpc_address = sui_config::node::default_json_rpc_address();
1026    fullnode_rpc_address.set_port(fullnode_rpc_port);
1027
1028    if no_full_node {
1029        swarm_builder = swarm_builder.with_fullnode_count(0);
1030    } else {
1031        let rpc_config = sui_config::RpcConfig {
1032            enable_indexing: Some(true),
1033            ..Default::default()
1034        };
1035
1036        swarm_builder = swarm_builder
1037            .with_fullnode_count(1)
1038            .with_fullnode_rpc_addr(fullnode_rpc_address)
1039            .with_fullnode_rpc_config(rpc_config);
1040    }
1041
1042    let mut swarm = swarm_builder.build();
1043    swarm.launch().await?;
1044    // Let nodes connect to one another
1045    tokio::time::sleep(tokio::time::Duration::from_secs(2)).await;
1046    info!("Cluster started");
1047
1048    let fullnode_rpc_url = format!("http://{fullnode_rpc_address}");
1049    info!("Fullnode RPC URL: {fullnode_rpc_url}");
1050
1051    let prometheus_registry = Registry::new();
1052    let cancel = CancellationToken::new();
1053    let mut rpc_services = vec![];
1054
1055    // Set-up the database for the indexer, if needed
1056    let (_database, database_url) = match with_indexer {
1057        None => (None, None),
1058
1059        // Temporary (local) database in config directory
1060        Some(None) => {
1061            let pg_dir = config_dir.join("indexer");
1062            let port = get_available_port();
1063
1064            info!("Starting local PostgreSQL database at {pg_dir:?} on port {port}");
1065
1066            let db = if pg_dir.exists() {
1067                LocalDatabase::new(pg_dir, port)
1068                    .context("Failed to start local PostgreSQL database")?
1069            } else {
1070                LocalDatabase::new_initdb(pg_dir, port)
1071                    .context("Failed to initialize and start local PostgreSQL database")?
1072            };
1073
1074            let url = db.url().clone();
1075            info!("Starting indexer with local database");
1076            (Some(db), Some(url))
1077        }
1078
1079        // Use provided database URL
1080        Some(Some(url)) => {
1081            info!("Starting indexer with provided database");
1082            (None, Some(url))
1083        }
1084    };
1085
1086    let pipelines = if let Some(ref db_url) = database_url {
1087        let client_args = ClientArgs {
1088            local_ingestion_path: data_ingestion_dir.clone(),
1089            ..Default::default()
1090        };
1091
1092        let indexer = setup_indexer(
1093            db_url.clone(),
1094            DbArgs::default(),
1095            IndexerArgs::default(),
1096            client_args,
1097            IndexerConfig::for_test(),
1098            None,
1099            &prometheus_registry,
1100            cancel.child_token(),
1101        )
1102        .await
1103        .context("Failed to setup indexer")?;
1104
1105        let pipelines = indexer.pipelines().map(|s| s.to_string()).collect();
1106        let handle = indexer.run().await.context("Failed to start indexer")?;
1107        rpc_services.push(handle);
1108
1109        info!("Indexer started with pipelines: {pipelines:?}");
1110        pipelines
1111    } else {
1112        vec![]
1113    };
1114
1115    let consistent_store_url = if let Some(input) = with_consistent_store {
1116        let address = parse_host_port(input, DEFAULT_CONSISTENT_STORE_PORT)
1117            .context("Invalid consistent store host and port")?;
1118
1119        let client_args = ClientArgs {
1120            local_ingestion_path: data_ingestion_dir.clone(),
1121            ..Default::default()
1122        };
1123
1124        let consistent_args = ConsistentArgs {
1125            rpc_listen_address: address,
1126            ..Default::default()
1127        };
1128
1129        let handle = start_consistent_store(
1130            config_dir.join("consistent_store"),
1131            IndexerArgs::default(),
1132            client_args,
1133            consistent_args,
1134            "0.0.0",
1135            ConsistentConfig::for_test(),
1136            &prometheus_registry,
1137            cancel.child_token(),
1138        )
1139        .await
1140        .context("Failed to start Consistent Store")?;
1141        rpc_services.push(handle);
1142
1143        info!("Consistent Store started at {address}");
1144        Some(format!("http://{address}"))
1145    } else {
1146        None
1147    };
1148
1149    if let Some(input) = with_graphql {
1150        let address = parse_host_port(input, DEFAULT_GRAPHQL_PORT)
1151            .context("Invalid graphql host and port")?;
1152
1153        info!("Starting the GraphQL service at {address}");
1154
1155        let graphql_args = GraphQlArgs {
1156            rpc_listen_address: address,
1157            no_ide: false,
1158        };
1159
1160        let consistent_reader_args = ConsistentReaderArgs {
1161            consistent_store_url: consistent_store_url
1162                .as_ref()
1163                .map(|url| Url::parse(url))
1164                .transpose()
1165                .context("Failed to parse consistent store URL")?,
1166            ..Default::default()
1167        };
1168
1169        let fullnode_args = FullnodeArgs {
1170            fullnode_rpc_url: Some(fullnode_rpc_url.clone()),
1171        };
1172
1173        let mut graphql_config = GraphQlConfig::default();
1174        graphql_config.zklogin.env = sui_indexer_alt_graphql::config::ZkLoginEnv::Test;
1175
1176        let handle = start_graphql(
1177            database_url.clone(),
1178            None,
1179            fullnode_args,
1180            DbArgs::default(),
1181            BigtableArgs::default(),
1182            consistent_reader_args,
1183            graphql_args,
1184            SystemPackageTaskArgs::default(),
1185            "0.0.0",
1186            graphql_config,
1187            pipelines,
1188            &prometheus_registry,
1189            cancel.child_token(),
1190        )
1191        .await
1192        .context("Failed to start GraphQL server")?;
1193        rpc_services.push(handle);
1194
1195        info!("GraphQL started at {address}");
1196    }
1197
1198    if let Some(input) = with_faucet {
1199        let faucet_address = parse_host_port(input, DEFAULT_FAUCET_PORT)
1200            .map_err(|_| anyhow!("Invalid faucet host and port"))?;
1201        info!("Starting the faucet service at {faucet_address}");
1202
1203        let host_ip = match faucet_address {
1204            SocketAddr::V4(addr) => *addr.ip(),
1205            _ => bail!("Faucet configuration requires an IPv4 address"),
1206        };
1207
1208        let config = FaucetConfig {
1209            host_ip,
1210            port: faucet_address.port(),
1211            amount: DEFAULT_FAUCET_MIST_AMOUNT,
1212            ..Default::default()
1213        };
1214
1215        if force_regenesis {
1216            let kp = swarm.config_mut().account_keys.swap_remove(0);
1217            let keystore_path = config_dir.join(SUI_KEYSTORE_FILENAME);
1218            let mut keystore =
1219                Keystore::from(FileBasedKeystore::load_or_create(&keystore_path).unwrap());
1220            let address: SuiAddress = kp.public().into();
1221            keystore
1222                .import(None, SuiKeyPair::Ed25519(kp))
1223                .await
1224                .unwrap();
1225            SuiClientConfig {
1226                keystore,
1227                external_keys: None,
1228                envs: vec![SuiEnv {
1229                    alias: "localnet".to_string(),
1230                    rpc: fullnode_rpc_url,
1231                    ws: None,
1232                    basic_auth: None,
1233                    chain_id: None,
1234                }],
1235                active_address: Some(address),
1236                active_env: Some("localnet".to_string()),
1237            }
1238            .persisted(config_dir.join(SUI_CLIENT_CONFIG).as_path())
1239            .save()
1240            .unwrap();
1241        }
1242
1243        let local_faucet = LocalFaucet::new(
1244            create_wallet_context(config.wallet_client_timeout_secs, config_dir.clone())?,
1245            config.clone(),
1246        )
1247        .await?;
1248
1249        let app_state = Arc::new(AppState {
1250            faucet: local_faucet,
1251            config,
1252        });
1253
1254        start_faucet(app_state).await?;
1255    }
1256
1257    // Run health check loop until Ctrl+C or failure
1258    let mut interval = interval(Duration::from_secs(3));
1259    let mut unhealthy = 0;
1260
1261    loop {
1262        tokio::select! {
1263            _ = tokio::signal::ctrl_c() => {
1264                info!("Received Ctrl+C, shutting down...");
1265                break;
1266            }
1267            _ = interval.tick() => {}
1268        }
1269
1270        for node in swarm.validator_nodes() {
1271            let Err(e) = node.health_check(true).await else {
1272                unhealthy = 0;
1273                break;
1274            };
1275
1276            unhealthy += 1;
1277            if unhealthy > 3 {
1278                return Err(e.into());
1279            }
1280        }
1281    }
1282
1283    // Trigger cancellation to shut down all RPC services, and wait for all services to exit
1284    // cleanly.
1285    cancel.cancel();
1286    // TODO (amnn): The indexer can take some time to shut down if the database it is talking to
1287    // stops responding. Re-enable graceful shutdown once cancel handling is revamped across the
1288    // framework.
1289    // future::join_all(rpc_services).await;
1290    Ok(())
1291}

1293async fn genesis(
1294    from_config: Option<PathBuf>,
1295    write_config: Option<PathBuf>,
1296    working_dir: Option<PathBuf>,
1297    force: bool,
1298    epoch_duration_ms: Option<u64>,
1299    benchmark_ips: Option<Vec<String>>,
1300    with_faucet: bool,
1301    committee_size: Option<usize>,
1302) -> Result<(), anyhow::Error> {
1303    let sui_config_dir = &match working_dir {
1304        // if a directory is specified, it must exist (it
1305        // will not be created)
1306        Some(v) => v,
1307        // create default Sui config dir if not specified
1308        // on the command line and if it does not exist
1309        // yet
1310        None => {
1311            let config_path = sui_config_dir()?;
1312            fs::create_dir_all(&config_path)?;
1313            config_path
1314        }
1315    };
1316
1317    // if Sui config dir is not empty then either clean it
1318    // up (if --force/-f option was specified or report an
1319    // error
1320    let dir = sui_config_dir.read_dir().map_err(|err| {
1321        anyhow!(err).context(format!("Cannot open Sui config dir {:?}", sui_config_dir))
1322    })?;
1323    let files = dir.collect::<Result<Vec<_>, _>>()?;
1324
1325    let client_path = sui_config_dir.join(SUI_CLIENT_CONFIG);
1326    let keystore_path = sui_config_dir.join(SUI_KEYSTORE_FILENAME);
1327
1328    if write_config.is_none() && !files.is_empty() {
1329        if force {
1330            // check old keystore and client.yaml is compatible
1331            let is_compatible = FileBasedKeystore::load_or_create(&keystore_path).is_ok()
1332                && PersistedConfig::<SuiClientConfig>::read(&client_path).is_ok();
1333            // Keep keystore and client.yaml if they are compatible
1334            if is_compatible {
1335                for file in files {
1336                    let path = file.path();
1337                    if path != client_path && path != keystore_path {
1338                        if path.is_file() {
1339                            fs::remove_file(path)
1340                        } else {
1341                            fs::remove_dir_all(path)
1342                        }
1343                        .map_err(|err| {
1344                            anyhow!(err).context(format!("Cannot remove file {:?}", file.path()))
1345                        })?;
1346                    }
1347                }
1348            } else {
1349                fs::remove_dir_all(sui_config_dir).map_err(|err| {
1350                    anyhow!(err)
1351                        .context(format!("Cannot remove Sui config dir {:?}", sui_config_dir))
1352                })?;
1353                fs::create_dir(sui_config_dir).map_err(|err| {
1354                    anyhow!(err)
1355                        .context(format!("Cannot create Sui config dir {:?}", sui_config_dir))
1356                })?;
1357            }
1358        } else if files.len() != 2 || !client_path.exists() || !keystore_path.exists() {
1359            bail!(
1360                "Cannot run genesis with non-empty Sui config directory {}, please use the --force/-f option to remove the existing configuration",
1361                sui_config_dir.to_str().unwrap()
1362            );
1363        }
1364    }
1365
1366    let network_path = sui_config_dir.join(SUI_NETWORK_CONFIG);
1367    let genesis_path = sui_config_dir.join(SUI_GENESIS_FILENAME);
1368
1369    let mut genesis_conf = match from_config {
1370        Some(path) => PersistedConfig::read(&path)?,
1371        None => {
1372            if let Some(ips) = benchmark_ips {
1373                // Make a keystore containing the key for the genesis gas object.
1374                let path = sui_config_dir.join(SUI_BENCHMARK_GENESIS_GAS_KEYSTORE_FILENAME);
1375                let mut keystore = FileBasedKeystore::load_or_create(&path)?;
1376                for gas_key in GenesisConfig::benchmark_gas_keys(ips.len()) {
1377                    keystore.import(None, gas_key).await?;
1378                }
1379                keystore.save().await?;
1380
1381                // Make a new genesis config from the provided ip addresses.
1382                GenesisConfig::new_for_benchmarks(&ips)
1383            } else if keystore_path.exists() {
1384                let existing_keys = FileBasedKeystore::load_or_create(&keystore_path)?.addresses();
1385                GenesisConfig::for_local_testing_with_addresses(existing_keys)
1386            } else {
1387                GenesisConfig::for_local_testing()
1388            }
1389        }
1390    };
1391
1392    // Adds an extra faucet account to the genesis
1393    if with_faucet {
1394        info!("Adding faucet account in genesis config...");
1395        genesis_conf = genesis_conf.add_faucet_account();
1396    }
1397
1398    if let Some(path) = write_config {
1399        let persisted = genesis_conf.persisted(&path);
1400        persisted.save()?;
1401        return Ok(());
1402    }
1403
1404    let validator_info = genesis_conf.validator_config_info.take();
1405    let ssfn_info = genesis_conf.ssfn_config_info.take();

1408    if let Some(epoch_duration_ms) = epoch_duration_ms {
1409        genesis_conf.parameters.epoch_duration_ms = epoch_duration_ms;
1410    }
1411    let committee_size = match committee_size {
1412        Some(x) => NonZeroUsize::new(x),
1413        None => NonZeroUsize::new(1),
1414    }
1415    .ok_or_else(|| anyhow!("Committee size must be at least 1."))?;
1416
1417    let mut network_config = if let Some(validators) = validator_info {
1418        builder
1419            .with_genesis_config(genesis_conf)
1420            .with_validators(validators)
1421            .build()
1422    } else {
1423        builder
1424            .committee_size(committee_size)
1425            .with_genesis_config(genesis_conf)
1426            .build()
1427    };
1428
1429    let mut keystore = FileBasedKeystore::load_or_create(&keystore_path)?;
1430    for key in &network_config.account_keys {
1431        keystore
1432            .import(None, SuiKeyPair::Ed25519(key.copy()))
1433            .await?;
1434    }
1435    let active_address = keystore.addresses().pop();
1436
1437    network_config.genesis.save(&genesis_path)?;
1438    for validator in &mut network_config.validator_configs {
1439        validator.genesis = sui_config::node::Genesis::new_from_file(&genesis_path);
1440    }
1441
1442    info!("Network genesis completed.");
1443    network_config.save(&network_path)?;
1444    info!("Network config file is stored in {:?}.", network_path);
1445
1446    info!("Client keystore is stored in {:?}.", keystore_path);
1447
1448    let fullnode_config = FullnodeConfigBuilder::new()
1449        .with_config_directory(FULL_NODE_DB_PATH.into())
1450        .with_rpc_addr(sui_config::node::default_json_rpc_address())
1451        .build(&mut OsRng, &network_config);
1452
1453    fullnode_config.save(sui_config_dir.join(SUI_FULLNODE_CONFIG))?;
1454    let mut ssfn_nodes = vec![];
1455    if let Some(ssfn_info) = ssfn_info {
1456        for (i, ssfn) in ssfn_info.into_iter().enumerate() {
1457            let path =
1458                sui_config_dir.join(sui_config::ssfn_config_file(ssfn.p2p_address.clone(), i));
1459            // join base fullnode config with each SsfnGenesisConfig entry
1460            let ssfn_config = FullnodeConfigBuilder::new()
1461                .with_config_directory(FULL_NODE_DB_PATH.into())
1462                .with_p2p_external_address(ssfn.p2p_address)
1463                .with_network_key_pair(ssfn.network_key_pair)
1464                .with_p2p_listen_address("0.0.0.0:8084".parse().unwrap())
1465                .with_db_path(PathBuf::from("/opt/sui/db/authorities_db/full_node_db"))
1466                .with_network_address("/ip4/0.0.0.0/tcp/8080/http".parse().unwrap())
1467                .with_metrics_address("0.0.0.0:9184".parse().unwrap())
1468                .with_admin_interface_port(1337)
1469                .with_json_rpc_address("0.0.0.0:9000".parse().unwrap())
1470                .with_genesis(Genesis::new_from_file("/opt/sui/config/genesis.blob"))
1471                .build(&mut OsRng, &network_config);
1472            ssfn_nodes.push(ssfn_config.clone());
1473            ssfn_config.save(path)?;
1474        }
1475
1476        let ssfn_seed_peers: Vec<SeedPeer> = ssfn_nodes
1477            .iter()
1478            .map(|config| SeedPeer {
1479                peer_id: Some(anemo::PeerId(
1480                    config.network_key_pair().public().0.to_bytes(),
1481                )),
1482                address: config.p2p_config.external_address.clone().unwrap(),
1483            })
1484            .collect();
1485
1486        for (i, mut validator) in network_config
1487            .into_validator_configs()
1488            .into_iter()
1489            .enumerate()
1490        {
1491            let path = sui_config_dir.join(sui_config::validator_config_file(
1492                validator.network_address.clone(),
1493                i,
1494            ));
1495            let mut val_p2p = validator.p2p_config.clone();
1496            val_p2p.seed_peers = ssfn_seed_peers.clone();
1497            validator.p2p_config = val_p2p;
1498            validator.save(path)?;
1499        }
1500    } else {
1501        for (i, validator) in network_config
1502            .into_validator_configs()
1503            .into_iter()
1504            .enumerate()
1505        {
1506            let path = sui_config_dir.join(sui_config::validator_config_file(
1507                validator.network_address.clone(),
1508                i,
1509            ));
1510            validator.save(path)?;
1511        }
1512    }
1513
1514    let mut client_config = if client_path.exists() {
1515        PersistedConfig::read(&client_path)?
1516    } else {
1517        SuiClientConfig::new(keystore.into())
1518    };
1519
1520    if client_config.active_address.is_none() {
1521        client_config.active_address = active_address;
1522    }
1523
1524    // On windows, using 0.0.0.0 will usually yield in an networking error. This localnet ip
1525    // address must bind to 127.0.0.1 if the default 0.0.0.0 is used.
1526    let localnet_ip =
1527        if fullnode_config.json_rpc_address.ip() == IpAddr::V4(Ipv4Addr::new(0, 0, 0, 0)) {
1528            "127.0.0.1".to_string()
1529        } else {
1530            fullnode_config.json_rpc_address.ip().to_string()
1531        };
1532    client_config.add_env(SuiEnv {
1533        alias: "localnet".to_string(),
1534        rpc: format!(
1535            "http://{}:{}",
1536            localnet_ip,
1537            fullnode_config.json_rpc_address.port()
1538        ),
1539        ws: None,
1540        basic_auth: None,
1541        chain_id: None,
1542    });
1543    client_config.add_env(SuiEnv::devnet());
1544
1545    if client_config.active_env.is_none() {
1546        client_config.active_env = client_config.envs.first().map(|env| env.alias.clone());
1547    }
1548
1549    client_config.save(&client_path)?;
1550    info!("Client config file is stored in {:?}.", client_path);
1551
1552    Ok(())
1553}

1739/// Download the package's modules and its dependencies to the specified path.
1740async fn download_package_and_deps_under(
1741    read_api: &ReadApi,
1742    path: &Path,
1743    package_id: ObjectID,
1744) -> anyhow::Result<PackageSummaryMetadata> {
1745    let mut dependencies = BTreeMap::new();
1746    let mut linkage = BTreeMap::new();
1747    let mut type_origins = BTreeMap::new();
1748
1749    let root_package = resolve_package(read_api, package_id).await?;
1750    for (original_id, pkg_info) in root_package.linkage_table().iter() {
1751        let package = resolve_package(read_api, pkg_info.upgraded_id).await?;
1752        let relative_package_path = package
1753            .id()
1754            .deref()
1755            .to_canonical_string(/* with_prefix */ true);
1756
1757        let package_path = path.join(&relative_package_path);
1758        fs::create_dir_all(&package_path)?;
1759        for (m_name, module) in package.serialized_module_map() {
1760            let mut file = fs::File::create(
1761                package_path
1762                    .join(m_name)
1763                    .with_extension(MOVE_COMPILED_EXTENSION),
1764            )?;
1765            file.write_all(module)?;
1766        }
1767
1768        dependencies.insert(*original_id, PathBuf::from(relative_package_path));
1769        linkage.insert(*original_id, pkg_info.clone());
1770        type_origins.insert(*original_id, package.type_origin_table().clone());
1771    }
1772
1773    let package_path = path.join(
1774        root_package
1775            .id()
1776            .deref()
1777            .to_canonical_string(/* with_prefix */ true),
1778    );
1779    fs::create_dir_all(&package_path)?;
1780    for (m_name, module) in root_package.serialized_module_map() {
1781        let file_path = package_path
1782            .join(m_name)
1783            .with_extension(MOVE_COMPILED_EXTENSION);
1784        let mut file = fs::File::create(&file_path)?;
1785        file.write_all(module).with_context(|| {
1786            format!(
1787                "Unable to write module {m_name} for package {} to {}",
1788                root_package
1789                    .id()
1790                    .deref()
1791                    .to_canonical_string(/* with_prefix */ true),
1792                file_path.display(),
1793            )
1794        })?;
1795    }
1796
1797    Ok(PackageSummaryMetadata {
1798        root_package_id: Some(root_package.id()),
1799        root_package_original_id: Some(root_package.original_package_id()),
1800        root_package_version: Some(root_package.version().value()),
1801        type_origins: Some(type_origins),
1802        dependencies: Some(dependencies),
1803        linkage: Some(linkage),
1804    })
1805}

1807/// Parse the input string into a SocketAddr, with a default port if none is provided.
1808pub fn parse_host_port(
1809    input: String,
1810    default_port_if_missing: u16,
1811) -> Result<SocketAddr, AddrParseError> {
1812    let default_host = "0.0.0.0";
1813    let mut input = input;
1814    if input.contains("localhost") {
1815        input = input.replace("localhost", "127.0.0.1");
1816    }
1817    if input.contains(':') {
1818        input.parse::<SocketAddr>()
1819    } else if input.contains('.') {
1820        format!("{input}:{default_port_if_missing}").parse::<SocketAddr>()
1821    } else if input.is_empty() {
1822        format!("{default_host}:{default_port_if_missing}").parse::<SocketAddr>()
1823    } else if !input.is_empty() {
1824        format!("{default_host}:{input}").parse::<SocketAddr>()
1825    } else {
1826        format!("{default_host}:{default_port_if_missing}").parse::<SocketAddr>()
1827    }
1828}

1830/// Get the replay node representing a specific chain (e.g., testnet, mainnet, or custom)
1831/// from a given wallet context contining chain identifier string.
1832pub async fn get_replay_node(context: &WalletContext) -> Result<SR2::Node, anyhow::Error> {
1833    let chain_id = context
1834        .get_client()
1835        .await?
1836        .read_api()
1837        .get_chain_identifier()
1838        .await?;
1839    let err_msg = format!(
1840        "'{chain_id}' chain identifier is not supported for replay -- only testnet and mainnet are supported currently"
1841    );
1842    let chain_id = ChainIdentifier::from_chain_short_id(&chain_id)
1843        .ok_or_else(|| anyhow::anyhow!(err_msg.clone()))?;
1844    Ok(match chain_id.chain() {
1845        Chain::Mainnet => SR2::Node::Mainnet,
1846        Chain::Testnet => SR2::Node::Testnet,
