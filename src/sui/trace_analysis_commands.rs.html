4use anyhow::anyhow;
5use clap::*;
6use move_trace_format::format::MoveTraceReader;
7use move_vm_profiler::trace_converter::{GasProfiler, ProfilerConfig};
8use std::path::PathBuf;
9
10#[derive(Parser)]
11#[clap(rename_all = "kebab-case")]
12pub enum AnalyzeTraceCommand {
13    /// Generate a gas profile for the trace file compatible with the `speedscope.app` profiler.
14    GasProfile {
15        /// Whether function names should be fully qualified with their module and package
16        /// addresses or if only the function name should be used.
17        #[arg(long, short)]
18        use_long_function_name: bool,
19    },
20}
21
22impl AnalyzeTraceCommand {
23    pub async fn execute(
24        self,
25        path: PathBuf,
26        output_dir: Option<PathBuf>,
27    ) -> Result<(), anyhow::Error> {
28        let trace_file = std::fs::File::open(&path).map_err(|e| {
29            anyhow!(
30                "Failed to open trace file at {}: {e}",
31                path.to_string_lossy()
32            )
33        })?;
34        let trace_reader = MoveTraceReader::new(trace_file)
35            .map_err(|e| anyhow!("Failed to read trace file: {e}"))?;
36
37        match self {
38            AnalyzeTraceCommand::GasProfile {
39                use_long_function_name,
40            } => {
41                let mut profiler = GasProfiler::init(
42                    ProfilerConfig {
43                        output_dir,
44                        use_long_function_name,
45                    },
46                    path.to_string_lossy().to_string(),
47                );
48                profiler.generate_from_trace(trace_reader);
49                profiler.save_profile();
50            }
51        }
52
53        Ok(())
54    }
55}
