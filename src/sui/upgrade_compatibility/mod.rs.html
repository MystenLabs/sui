165/// Check if an `UpgradeCompatibilityModeError` variant breaks 'compatible' upgrades.
166fn breaks_compatibility(
167    error: &UpgradeCompatibilityModeError,
168    compatability: &Compatibility,
169) -> bool {
170    match error {
171        UpgradeCompatibilityModeError::StructAbilityMismatch { .. }
172        | UpgradeCompatibilityModeError::StructTypeParamMismatch { .. }
173        | UpgradeCompatibilityModeError::EnumAbilityMismatch { .. }
174        | UpgradeCompatibilityModeError::EnumTypeParamMismatch { .. }
175        | UpgradeCompatibilityModeError::FunctionMissingPublic { .. }
176        | UpgradeCompatibilityModeError::FunctionLostPublicVisibility { .. } => true,
177
178        UpgradeCompatibilityModeError::StructFieldMismatch { .. }
179        | UpgradeCompatibilityModeError::EnumVariantMissing { .. }
180        | UpgradeCompatibilityModeError::EnumVariantMismatch { .. } => {
181            compatability.check_datatype_layout
182        }
183
184        UpgradeCompatibilityModeError::StructMissing { .. }
185        | UpgradeCompatibilityModeError::EnumMissing { .. } => true,
186
187        UpgradeCompatibilityModeError::FunctionSignatureMismatch { old_function, .. } => {
188            if old_function.visibility == Visibility::Public {
189                return true;
190            }
191            if old_function.is_entry {
192                compatability.check_private_entry_linking
193            } else {
194                false
195            }
196        }
197
198        UpgradeCompatibilityModeError::FunctionMissingEntry { .. }
199        | UpgradeCompatibilityModeError::FunctionEntryCompatibility { .. } => {
200            compatability.check_private_entry_linking
201        }
202        UpgradeCompatibilityModeError::EnumNewVariant { .. } => compatability.check_datatype_layout,
203
204        UpgradeCompatibilityModeError::FileFormatVersionDowngrade { .. }
205        | UpgradeCompatibilityModeError::StructNew { .. }
206        | UpgradeCompatibilityModeError::StructChange { .. }
207        | UpgradeCompatibilityModeError::EnumNew { .. }
208        | UpgradeCompatibilityModeError::EnumChange { .. }
209        | UpgradeCompatibilityModeError::FunctionNew { .. }
210        | UpgradeCompatibilityModeError::FunctionChange { .. }
211        | UpgradeCompatibilityModeError::FunctionMissing { .. }
212        | UpgradeCompatibilityModeError::FriendNew
213        | UpgradeCompatibilityModeError::FriendMissing => false,
214    }
215}

217/// Check if an `UpgradeCompatibilityModeError` variant breaks inclusion checks for 'additive' (Subset)
218/// or 'dependency only' (Equal) upgrades.
219fn breaks_inclusion_check(
220    error: &UpgradeCompatibilityModeError,
221    inclusion_check: &InclusionCheck,
222) -> bool {
223    match error {
224        UpgradeCompatibilityModeError::StructNew { .. } => {
225            matches!(inclusion_check, InclusionCheck::Equal)
226        }
227        UpgradeCompatibilityModeError::EnumNew { .. } => {
228            matches!(inclusion_check, InclusionCheck::Equal)
229        }
230        UpgradeCompatibilityModeError::FunctionNew { .. } => {
231            matches!(inclusion_check, InclusionCheck::Equal)
232        }
233
234        UpgradeCompatibilityModeError::StructMissing { .. }
235        | UpgradeCompatibilityModeError::StructAbilityMismatch { .. }
236        | UpgradeCompatibilityModeError::StructTypeParamMismatch { .. }
237        | UpgradeCompatibilityModeError::StructFieldMismatch { .. }
238        | UpgradeCompatibilityModeError::EnumMissing { .. }
239        | UpgradeCompatibilityModeError::EnumAbilityMismatch { .. }
240        | UpgradeCompatibilityModeError::EnumTypeParamMismatch { .. }
241        | UpgradeCompatibilityModeError::EnumNewVariant { .. }
242        | UpgradeCompatibilityModeError::EnumVariantMissing { .. }
243        | UpgradeCompatibilityModeError::EnumVariantMismatch { .. }
244        | UpgradeCompatibilityModeError::FunctionMissingPublic { .. }
245        | UpgradeCompatibilityModeError::FunctionMissingEntry { .. }
246        | UpgradeCompatibilityModeError::FunctionSignatureMismatch { .. }
247        | UpgradeCompatibilityModeError::FunctionLostPublicVisibility { .. }
248        | UpgradeCompatibilityModeError::FunctionEntryCompatibility { .. }
249        | UpgradeCompatibilityModeError::FileFormatVersionDowngrade { .. }
250        | UpgradeCompatibilityModeError::StructChange { .. }
251        | UpgradeCompatibilityModeError::EnumChange { .. }
252        | UpgradeCompatibilityModeError::FunctionChange { .. }
253        | UpgradeCompatibilityModeError::FunctionMissing { .. }
254        | UpgradeCompatibilityModeError::FriendNew
255        | UpgradeCompatibilityModeError::FriendMissing => true,
256    }
257}

570/// Creates an index to allow looking up the table index of a struct, enum, or function by its identifier.
571fn table_index(compiled_module: &CompiledModule) -> IdentifierTableLookup {
572    // for each in compiled module
573    let struct_identifier_to_index: BTreeMap<Identifier, TableIndex> = compiled_module
574        .struct_defs()
575        .iter()
576        .enumerate()
577        .map(|(i, d)| {
578            // get the identifier of the struct
579            let s_id = compiled_module
580                .identifier_at(compiled_module.datatype_handle_at(d.struct_handle).name);
581            (s_id.to_owned(), i as TableIndex)
582        })
583        .collect();
584
585    let enum_identifier_to_index: BTreeMap<Identifier, TableIndex> = compiled_module
586        .enum_defs()
587        .iter()
588        .enumerate()
589        .map(|(i, d)| {
590            let e_id = compiled_module
591                .identifier_at(compiled_module.datatype_handle_at(d.enum_handle).name);
592            (e_id.to_owned(), i as TableIndex)
593        })
594        .collect();
595
596    let function_identifier_to_index: BTreeMap<Identifier, TableIndex> = compiled_module
597        .function_defs()
598        .iter()
599        .enumerate()
600        .map(|(i, d)| {
601            let f_id =
602                compiled_module.identifier_at(compiled_module.function_handle_at(d.function).name);
603            (f_id.to_owned(), i as TableIndex)
604        })
605        .collect();
606
607    IdentifierTableLookup {
608        struct_identifier_to_index,
609        enum_identifier_to_index,
610        function_identifier_to_index,
611    }
612}

684/// Check the upgrade compatibility of a new package with an existing on-chain package.
685pub(crate) async fn check_compatibility(
686    mut client: Client,
687    package_id: ObjectID,
688    new_package: CompiledPackage,
689    package_path: PathBuf,
690    upgrade_policy: u8,
691    protocol_config: ProtocolConfig,
692) -> Result<(), Error> {
693    let existing_obj = client.get_object(package_id).await?;
694    let existing_package = existing_obj
695        .data
696        .try_as_package()
697        .ok_or_else(|| anyhow!("Object found when package expected"))?;
698
699    let existing_modules = existing_package
700        .serialized_module_map()
701        .iter()
702        .map(|m| CompiledModule::deserialize_with_config(m.1, &protocol_config.binary_config(None)))
703        .collect::<Result<Vec<_>, _>>()
704        .context("Unable to get existing package")?;
705
706    let policy =
707        UpgradePolicy::try_from(upgrade_policy).map_err(|_| anyhow!("Invalid upgrade policy"))?;
708
709    compare_packages(
710        *existing_package.original_package_id(),
711        existing_modules,
712        new_package,
713        package_path,
714        policy,
715    )
716}

718/// Collect all the errors into a single error message.
719fn compare_packages(
720    package_id: AccountAddress,
721    existing_modules: Vec<CompiledModule>,
722    mut new_package: CompiledPackage,
723    package_path: PathBuf,
724    policy: UpgradePolicy,
725) -> Result<(), Error> {
726    // create a map from the new modules
727    let mut new_modules_map: HashMap<Identifier, CompiledModule> = new_package
728        .get_modules()
729        .map(|m| (m.self_id().name().to_owned(), m.clone()))
730        .collect();
731
732    let lookup: HashMap<Identifier, IdentifierTableLookup> = new_modules_map
733        .values()
734        .map(|m| (m.self_id().name().to_owned(), table_index(m)))
735        .collect();
736
737    let mut diags = Diagnostics::new();
738
739    // add move toml
740    let move_toml_path = package_path.join("Move.toml");
741    let move_toml_contents = Arc::from(
742        fs::read_to_string(&move_toml_path)
743            .context("Unable to read Move.toml")?
744            .to_string(),
745    );
746    let move_toml_hash = FileHash::new(&move_toml_contents);
747
748    new_package.package.file_map.add(
749        FileHash::new(&move_toml_contents),
750        FileName::from(move_toml_path.to_string_lossy()),
751        Arc::clone(&move_toml_contents),
752    );
753
754    for existing_module in existing_modules {
755        let name = existing_module.self_id().name().to_owned();
756        match new_modules_map.get_mut(&name) {
757            Some(new_module) => {
758                let new_module_address_idx = new_module.self_handle().address;
759                let addrs = &mut new_module.address_identifiers;
760                if let Some(address_mut) = addrs.get_mut(new_module_address_idx.0 as usize)
761                    && *address_mut == AccountAddress::ZERO
762                {
763                    // if the new module address is zero, set it to the on-chain address
764                    *address_mut = package_id;
765                }
766
767                let compiled_unit_with_source = new_package
768                    .package
769                    .get_module_by_name_from_root(name.as_str())
770                    .context("Unable to get module")?;
771
772                let module_diags = modules_into_diags(
773                    &existing_module,
774                    new_module,
775                    compiled_unit_with_source,
776                    &lookup[&name],
777                    policy,
778                )?;
779
780                diags.extend(module_diags);
781            }
782            None => {
783                diags.extend(missing_module_diag(
784                    &name,
785                    &move_toml_hash,
786                    &move_toml_contents,
787                )?);
788            }
789        }
790    }
791
792    if diags.is_empty() {
793        Ok(())
794    } else {
795        // Sort diagnostics to ensure consistent error ordering across platforms
796        // Diagnostic implements Ord, so sorting will be deterministic
797        let mut sorted_vec = diags.into_vec();
798        sorted_vec.sort();
799        let sorted_diags: Diagnostics = sorted_vec.into_iter().collect();
800
801        Err(anyhow!(
802            "{}\nUpgrade failed, this package requires changes to be compatible with the existing package. \
803            Its upgrade policy is set to '{}'. Use --skip-verify-compatibility to bypass this check locally.",
804            String::from_utf8(report_diagnostics_to_buffer(
805                &new_package.package.file_map,
806                sorted_diags,
807                use_colors()
808            ))
809            .context("Unable to convert buffer to string")?,
810            match policy {
811                UpgradePolicy::Compatible => "compatible",
812                UpgradePolicy::Additive => "additive",
813                UpgradePolicy::DepOnly => "dependency only",
814            }
815        ))
816    }
817}

819/// Runs the compatibility check between two modules and flattens the errors into a single error message.
820fn modules_into_diags(
821    existing_module: &CompiledModule,
822    new_module: &CompiledModule,
823    compiled_unit_with_source: &CompiledUnitWithSource,
824    lookup: &IdentifierTableLookup,
825    policy: UpgradePolicy,
826) -> Result<Diagnostics, Error> {
827    let pool = &mut normalized::RcPool::new();
828    let diags_list = match policy {
829        UpgradePolicy::DepOnly => InclusionCheck::Equal.check_with_mode::<CliInclusionCheckMode>(
830            &Module::new(pool, existing_module, /* include code */ true),
831            &Module::new(pool, new_module, /* include code */ true),
832        ),
833        UpgradePolicy::Additive => InclusionCheck::Subset.check_with_mode::<CliInclusionCheckMode>(
834            &Module::new(pool, existing_module, /* include code */ true),
835            &Module::new(pool, new_module, /* include code */ true),
836        ),
837        _ => Compatibility::upgrade_check().check_with_mode::<CliCompatibilityMode>(
838            &Module::new(pool, existing_module, /* include code */ true),
839            &Module::new(pool, new_module, /* include code */ true),
840        ),
841    }
842    .err()
843    .unwrap_or_default()
844    .into_iter()
845    .map(|e| {
846        compatibility_diag_from_error(
847            &e,
848            compiled_unit_with_source,
849            matches!(policy, UpgradePolicy::Compatible),
850            lookup,
851        )
852    })
853    .collect::<Result<Vec<_>, _>>()?;
854
855    let mut diags = Diagnostics::new();
856
857    for diag in diags_list {
858        diags.extend(diag);
859    }
860
861    Ok(diags)
862}

1052/// Returns a diagnostic when a module is missing from a package.
1053fn missing_module_diag(
1054    module_name: &Identifier,
1055    move_toml_hash: &FileHash,
1056    move_toml_contents: &Arc<str>,
1057) -> Result<Diagnostics, Error> {
1058    const PACKAGE_TABLE: &str = "[package]";
1059    let mut diags = Diagnostics::new();
1060
1061    let start: usize = move_toml_contents.find(PACKAGE_TABLE).unwrap_or_default();
1062    // default to the end of the package table definition
1063    // get the third newline after the start of the package table declaration if it exists
1064    let end = move_toml_contents[start..]
1065        .match_indices('\n')
1066        .take(3)
1067        .last()
1068        .map(|(idx, _)| start + idx)
1069        .unwrap_or(start + PACKAGE_TABLE.len());
1070
1071    let loc = Loc::new(*move_toml_hash, start as ByteIndex, end as ByteIndex);
1072
1073    diags.add(Diagnostic::new(
1074        Declarations::ModuleMissing,
1075        (loc, format!("Package is missing module '{module_name}'")),
1076        Vec::<(Loc, String)>::new(),
1077        vec![
1078            "Modules which are part package cannot be removed during an upgrade.".to_string(),
1079            format!("Add missing module '{module_name}' back to the package."),
1080        ],
1081    ));
1082
1083    Ok(diags)
1084}

1087fn missing_definition_diag(
1088    declaration_kind: &str,

1091    compiled_unit_with_source: &CompiledUnitWithSource,

1094
1095    // capitalize the first letter

1097        .chars()
1098        .enumerate()
1099        .map(|(i, c)| {
1100            if i == 0 {
1101                c.to_uppercase().next().unwrap_or(c)
1102            } else {
1103                c
1104            }
1105        })
1106        .collect::<String>();
1107
1108    let module_name = compiled_unit_with_source.unit.name.as_str();
1109    let loc = compiled_unit_with_source
1110        .unit
1111        .source_map
1112        .definition_location;

1114    let (code, reason_notes) = if public_visibility_related_error {
1115        (
1116            Declarations::PublicMissing,
1117            vec![
1118                format!(
1119                    "{capital_declaration_kind}s are part of a module's public interface \
1120                     and cannot be removed or changed during a 'compatible' upgrade.",
1121                ),
1122                format!(
1123                    "Add missing {declaration_kind} '{identifier_name}' \
1124                     back to the module '{module_name}'.",
1125                ),
1126            ],
1127        )
1128    } else {
1129        (
1130            Declarations::Missing,
1131            vec![
1132                format!(
1133                    "{capital_declaration_kind}s cannot be removed or changed during an 'additive' or \
1134                    'dependency only' upgrade.",
1135                ),
1136                format!(
1137                    "Add missing {declaration_kind} '{identifier_name}' \
1138                     back to the module '{module_name}'.",
1139                ),

1143
1144    diags.add(Diagnostic::new(
1145        code,

1155}

1199/// Returns diagnostics for each signature mismatch in the given function.
1200/// Start by checking the lengths of the parameters and returns and add a diagnostic if they are different.
1201/// If the lengths are the same check each parameter piece wise and add a diagnostic for each mismatch.
1202fn function_signature_mismatch_diag(
1203    function_name: &Identifier,
1204    old_function: &Rc<Function>,
1205    new_function: &Rc<Function>,
1206    public_visibility_related_error: bool,
1207    compiled_unit_with_source: &CompiledUnitWithSource,
1208    lookup: &IdentifierTableLookup,
1209) -> Result<Diagnostics, Error> {
1210    let mut diags = Diagnostics::new();
1211
1212    let func_index = lookup
1213        .function_identifier_to_index
1214        .get(function_name)
1215        .context("Unable to get function index")?;
1216
1217    let func_sourcemap = compiled_unit_with_source
1218        .unit
1219        .source_map
1220        .get_function_source_map(FunctionDefinitionIndex::new(*func_index))
1221        .context("Unable to get function source map")?;
1222
1223    let def_loc = func_sourcemap.definition_location;
1224
1225    let reason = if public_visibility_related_error {
1226        "Functions are part of a module's public interface \
1227        and cannot be removed or changed during a 'compatible' upgrade."
1228    } else {
1229        "Functions cannot be removed or changed during an 'additive' or \
1230        'dependency only' upgrade."
1231    };
1232
1233    // handle function arguments
1234    if old_function.parameters.len() != new_function.parameters.len() {
1235        diags.add(Diagnostic::new(
1236            Functions_::SignatureMismatch,
1237            (
1238                def_loc,
1239                format!(
1240                    "Expected {} {}, found {}",
1241                    old_function.parameters.len(),
1242                    singular_or_plural(old_function.parameters.len(), "parameter", "parameters"),
1243                    new_function.parameters.len(),
1244                ),
1245            ),
1246            Vec::<(Loc, String)>::new(),
1247            vec![
1248                reason.to_string(),
1249                format!(
1250                    "Restore the original function's {} for \
1251                    function '{function_name}', expected {} {}.",
1252                    singular_or_plural(old_function.parameters.len(), "parameter", "parameters"),
1253                    old_function.parameters.len(),
1254                    singular_or_plural(old_function.parameters.len(), "parameter", "parameters"),
1255                ),
1256            ],
1257        ));
1258    } else if old_function.parameters != new_function.parameters {
1259        for ((i, old_param), new_param) in old_function
1260            .parameters
1261            .iter()
1262            .enumerate()
1263            .zip(new_function.parameters.iter())
1264        {
1265            if old_param != new_param {
1266                let param_loc = func_sourcemap
1267                    .parameters
1268                    .get(i)
1269                    .context("Unable to get parameter location")?
1270                    .1;
1271
1272                let mut secondary = Vec::new();
1273
1274                let old_param = format_param(
1275                    old_param,
1276                    func_sourcemap.type_parameters.clone(),
1277                    &mut secondary,
1278                )?;
1279                let new_param = format_param(
1280                    new_param,
1281                    func_sourcemap.type_parameters.clone(),
1282                    &mut Vec::new(),
1283                )?;
1284
1285                let label = format!("Unexpected parameter '{new_param}', expected '{old_param}'");
1286
1287                diags.add(Diagnostic::new(
1288                    Functions_::SignatureMismatch,
1289                    (param_loc, label),
1290                    secondary,
1291                    vec![
1292                        reason.to_string(),
1293                        format!(
1294                            "Restore the original function's {} \
1295                            for function '{function_name}'.",
1296                            singular_or_plural(
1297                                old_function.parameters.len(),
1298                                "parameter",
1299                                "parameters"
1300                            )
1301                        ),
1302                    ],
1303                ));
1304            }
1305        }
1306    }
1307    // type parameters are a vector of AbilitySet and therefore cannot share the same logic as structs and enums
1308    if old_function.type_parameters.len() != new_function.type_parameters.len() {
1309        diags.add(Diagnostic::new(
1310            Declarations::TypeParamMismatch,
1311            (
1312                def_loc,
1313                format!(
1314                    "Expected {} type {}, found {}",
1315                    old_function.type_parameters.len(),
1316                    singular_or_plural(
1317                        old_function.type_parameters.len(),
1318                        "parameter",
1319                        "parameters"
1320                    ),
1321                    new_function.type_parameters.len()
1322                ),
1323            ),
1324            Vec::<(Loc, String)>::new(),
1325            vec![
1326                reason.to_string(),
1327                format!(
1328                    "Restore the original function's type {} for \
1329                    function '{function_name}', expected {} type {}.",
1330                    singular_or_plural(
1331                        old_function.type_parameters.len(),
1332                        "parameter",
1333                        "parameters"
1334                    ),
1335                    old_function.type_parameters.len(),
1336                    singular_or_plural(
1337                        old_function.type_parameters.len(),
1338                        "parameter",
1339                        "parameters"
1340                    ),
1341                ),
1342            ],
1343        ));
1344    } else if old_function.type_parameters != new_function.type_parameters {
1345        for ((i, old_type_param), new_type_param) in old_function
1346            .type_parameters
1347            .iter()
1348            .enumerate()
1349            .zip(new_function.type_parameters.iter())
1350        {
1351            if old_type_param != new_type_param {
1352                let type_param_loc = func_sourcemap
1353                    .type_parameters
1354                    .get(i)
1355                    .context("Unable to get type parameter location")?
1356                    .1;
1357
1358                diags.add(Diagnostic::new(
1359                    Declarations::TypeParamMismatch,
1360                    (
1361                        type_param_loc,
1362                        format!(
1363                            "Unexpected type parameter {}, expected {}",
1364                            format_list(
1365                                new_type_param
1366                                    .into_iter()
1367                                    .map(|t| format!("'{:?}'", t).to_lowercase()),
1368                                Some(("constraint", "constraints"))
1369                            ),
1370                            format_list(
1371                                old_type_param
1372                                    .into_iter()
1373                                    .map(|t| format!("'{:?}'", t).to_lowercase()),
1374                                None
1375                            ),
1376                        ),
1377                    ),
1378                    Vec::<(Loc, String)>::new(),
1379                    vec![
1380                        reason.to_string(),
1381                        format!(
1382                            "Restore the original function's type {} \
1383                            for function '{function_name}'.",
1384                            singular_or_plural(
1385                                old_function.type_parameters.len(),
1386                                "parameter",
1387                                "parameters"
1388                            )
1389                        ),
1390                    ],
1391                ));
1392            }
1393        }
1394    }
1395
1396    // handle return
1397    if old_function.return_.len() != new_function.return_.len() {
1398        diags.add(Diagnostic::new(
1399            Functions_::SignatureMismatch,
1400            (
1401                def_loc,
1402                format!(
1403                    "Expected {} return {}, found {}",
1404                    old_function.return_.len(),
1405                    singular_or_plural(old_function.return_.len(), "type", "types"),
1406                    new_function.return_.len()
1407                ),
1408            ),
1409            Vec::<(Loc, String)>::new(),
1410            vec![
1411                reason.to_string(),

1661/// Returns diagnostics for each field mismatch in the given struct.
1662/// Start by checking the lengths of the fields and return a diagnostic if they are different.
1663/// If the lengths are the same check each field piece wise and return a diagnostic for each mismatch.
1664fn struct_field_mismatch_diag(
1665    struct_name: &Identifier,
1666    old_struct: &Rc<Struct>,
1667    new_struct: &Rc<Struct>,
1668    public_visibility_related_error: bool,
1669    compiled_unit_with_source: &CompiledUnitWithSource,
1670    lookup: &IdentifierTableLookup,
1671) -> Result<Diagnostics, Error> {
1672    let mut diags = Diagnostics::new();
1673
1674    let struct_index = lookup
1675        .struct_identifier_to_index
1676        .get(struct_name)
1677        .context("Unable to get struct index")?;
1678
1679    let struct_sourcemap = compiled_unit_with_source
1680        .unit
1681        .source_map
1682        .get_struct_source_map(StructDefinitionIndex::new(*struct_index))

1692        .filter(|f| !is_dummy_field(f))

1694        .collect();

1710    .to_string();

1731            ],
1732        ));

1734        .iter()
1735        .zip(&new_fields)
1736        .all(|(old_field, new_field)| old_field.equivalent(new_field))

1738        for (i, (old_field, new_field)) in old_fields.iter().zip(&new_fields).enumerate() {
1739            if !old_field.equivalent(new_field) {
1740                let field_loc = struct_sourcemap
1741                    .fields
1742                    .get(i)
1743                    .context("Unable to get field location")?;
1744
1745                let (code, label) = field_mismatch_message(
1746                    old_field,
1747                    new_field,
1748                    struct_sourcemap.type_parameters.clone(),
1749                )?;
1750
1751                diags.add(Diagnostic::new(
1752                    code,
1753                    (*field_loc, label),
1754                    vec![(def_loc, "Struct definition".to_string())],
1755                    vec![
1756                        reason.to_string(),
1757                        format!(
1758                            "Restore the original struct's {} for \
1759                            struct '{struct_name}' including the ordering.",
1760                            singular_or_plural(old_fields.len(), "field", "fields")
1761                        ),
1762                    ],
1763                ));
1764            }
1765        }
1766    }
1767
1768    Ok(diags)
1769}

2361/// Returns a diagnostic for a type parameter mismatch.
2362fn type_parameter_diag(
2363    declaration_kind: &str,
2364    name: &Identifier,
2365    old_type_parameters: &[DatatypeTyParameter],
2366    new_type_parameters: &[DatatypeTyParameter],
2367    public_visibility_related_error: bool, // give a different code for errors which are public visibility related
2368    def_loc: Loc,
2369    type_parameter_locs: &[SourceName],
2370) -> Result<Diagnostics, Error> {
2371    let mut diags = Diagnostics::new();
2372
2373    // capitalize the first letter
2374    let capital_declaration_kind = declaration_kind
2375        .chars()
2376        .enumerate()
2377        .map(|(i, c)| {
2378            if i == 0 {
2379                c.to_uppercase().next().unwrap_or(c)
2380            } else {
2381                c
2382            }
2383        })
2384        .collect::<String>();
2385
2386    if old_type_parameters.len() != new_type_parameters.len() {
2387        diags.add(Diagnostic::new(
2388            Declarations::TypeParamMismatch,
2389            (
2390                def_loc,
2391                format!(
2392                    "Incorrect number of type parameters: expected {}, found {}",
2393                    old_type_parameters.len(),
2394                    new_type_parameters.len()
2395                ),
2396            ),
2397            Vec::<(Loc, String)>::new(),
2398            vec![
2399                if public_visibility_related_error {
2400                    format!(
2401                        "{capital_declaration_kind}s are part of a module's public interface \
2402                        and cannot be removed or changed during a 'compatible' upgrade.",
2403                    )
2404                } else {
2405                    format!(
2406                        "{capital_declaration_kind}s cannot be removed or changed during an 'additive' or \
2407                        'dependency only' upgrade."
2408                    )
2409                },
2410                format!(
2411                    "Restore the original {declaration_kind}'s type {} \
2412                    for {declaration_kind} '{name}' including the ordering.",
2413                    singular_or_plural(old_type_parameters.len(), "parameter", "parameters"),
2414                ),
2415            ],
2416        ));
2417    } else if old_type_parameters != new_type_parameters {
2418        for (i, (old_type_param, new_type_param)) in old_type_parameters
2419            .iter()
2420            .zip(new_type_parameters.iter())
2421            .enumerate()
2422        {
2423            let type_param_loc = type_parameter_locs
2424                .get(i)
2425                .context("Unable to get type parameter location")?;
2426
2427            if let Some((label, fix_note)) =
2428                type_param_constraint_labels(old_type_param.constraints, new_type_param.constraints)
2429            {
2430                diags.add(Diagnostic::new(
2431                    Declarations::TypeParamMismatch,
2432                    (type_param_loc.1, label),
2433                    vec![(def_loc, format!("{capital_declaration_kind} definition"))],
2434                    vec![
2435                        format!(
2436                            "{capital_declaration_kind}s are part of a module's public interface \
2437                            and cannot be removed or changed during an upgrade."
2438                        ),
2439                        fix_note,
2440                    ],
2441                ));
2442            }
2443
2444            if let Some((label, fix_note)) =
2445                type_param_phantom_labels(old_type_param.is_phantom, new_type_param.is_phantom)
2446            {
2447                diags.add(Diagnostic::new(
2448                    Declarations::TypeParamMismatch,
2449                    (type_param_loc.1, label),
2450                    vec![(def_loc, format!("{capital_declaration_kind} definition"))],
2451                    vec![
2452                        format!(
2453                            "{capital_declaration_kind}s are part of a module's public interface \
2454                            and cannot be removed or changed during an upgrade."
2455                        ),
2456                        fix_note,
2457                    ],
2458                ));
2459            }
2460        }
2461    }
2462    Ok(diags)
2463}
