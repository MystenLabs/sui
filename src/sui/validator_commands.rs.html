261fn make_key_files(
262    file_name: PathBuf,
263    is_protocol_key: bool,
264    key: Option<SuiKeyPair>,
265) -> Result<()> {
266    if file_name.exists() {
267        println!("Use existing {:?} key file.", file_name);
268        return Ok(());
269    } else if is_protocol_key {
270        let (_, keypair) = get_authority_key_pair();
271        write_authority_keypair_to_file(&keypair, file_name.clone())?;
272        println!("Generated new key file: {:?}.", file_name);
273    } else {
274        let kp = match key {
275            Some(key) => {
276                println!(
277                    "Generated new key file {:?} based on sui.keystore file.",
278                    file_name
279                );
280                key
281            }
282            None => {
283                let (_, kp, _, _) = generate_new_key(SignatureScheme::ED25519, None, None)?;
284                println!("Generated new key file: {:?}.", file_name);
285                kp
286            }
287        };
288        write_keypair_to_file(&kp, &file_name)?;
289    }
290    Ok(())
291}

293impl SuiValidatorCommand {
294    pub async fn execute(
295        self,
296        context: &mut WalletContext,
297    ) -> Result<SuiValidatorCommandResponse, anyhow::Error> {
298        let sui_address = context.active_address()?;
299
300        Ok(match self {
301            SuiValidatorCommand::MakeValidatorInfo {
302                name,
303                description,
304                image_url,
305                project_url,
306                host_name,
307                gas_price,
308            } => {
309                let dir = std::env::current_dir()?;
310                let protocol_key_file_name = dir.join("protocol.key");
311                let account_key = match context.config.keystore.export(&sui_address)? {
312                    SuiKeyPair::Ed25519(account_key) => SuiKeyPair::Ed25519(account_key.copy()),
313                    _ => panic!(
314                        "Other account key types supported yet, please use Ed25519 keys for now."
315                    ),
316                };
317                let account_key_file_name = dir.join("account.key");
318                let network_key_file_name = dir.join("network.key");
319                let worker_key_file_name = dir.join("worker.key");
320                make_key_files(protocol_key_file_name.clone(), true, None)?;
321                make_key_files(account_key_file_name.clone(), false, Some(account_key))?;
322                make_key_files(network_key_file_name.clone(), false, None)?;
323                make_key_files(worker_key_file_name.clone(), false, None)?;
324
325                let keypair: AuthorityKeyPair =
326                    read_authority_keypair_from_file(protocol_key_file_name)?;
327                let account_keypair: SuiKeyPair = read_keypair_from_file(account_key_file_name)?;
328                let worker_keypair: NetworkKeyPair =
329                    read_network_keypair_from_file(worker_key_file_name)?;
330                let network_keypair: NetworkKeyPair =
331                    read_network_keypair_from_file(network_key_file_name)?;
332                let pop =
333                    generate_proof_of_possession(&keypair, (&account_keypair.public()).into());
334                let validator_info = GenesisValidatorInfo {
335                    info: sui_genesis_builder::validator_info::ValidatorInfo {
336                        name,
337                        protocol_key: keypair.public().into(),
338                        worker_key: worker_keypair.public().clone(),
339                        account_address: SuiAddress::from(&account_keypair.public()),
340                        network_key: network_keypair.public().clone(),
341                        gas_price,
342                        commission_rate: sui_config::node::DEFAULT_COMMISSION_RATE,
343                        network_address: Multiaddr::try_from(format!(
344                            "/dns/{}/tcp/8080/http",
345                            host_name
346                        ))?,
347                        p2p_address: Multiaddr::try_from(format!("/dns/{}/udp/8084", host_name))?,
348                        narwhal_primary_address: Multiaddr::try_from(format!(
349                            "/dns/{}/udp/8081",
350                            host_name
351                        ))?,
352                        narwhal_worker_address: Multiaddr::try_from(format!(
353                            "/dns/{}/udp/8082",
354                            host_name
355                        ))?,
356                        description,
357                        image_url,
358                        project_url,
359                    },
360                    proof_of_possession: pop,
361                };
362                // TODO set key files permission
363                let validator_info_file_name = dir.join("validator.info");
364                let validator_info_bytes = serde_yaml::to_string(&validator_info)?;
365                fs::write(validator_info_file_name.clone(), validator_info_bytes)?;
366                println!(
367                    "Generated validator info file: {:?}.",
368                    validator_info_file_name
369                );
370                SuiValidatorCommandResponse::MakeValidatorInfo
371            }
372            SuiValidatorCommand::BecomeCandidate { file, tx_args } => {
373                let gas_budget = tx_args.gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
374                let validator_info_bytes = fs::read(file)?;
375                // Note: we should probably rename the struct or evolve it accordingly.
376                let validator_info: GenesisValidatorInfo =
377                    serde_yaml::from_slice(&validator_info_bytes)?;
378                let validator = validator_info.info;
379
380                let args = vec![
381                    CallArg::Pure(
382                        bcs::to_bytes(&AuthorityPublicKeyBytes::from_bytes(
383                            validator.protocol_key().as_bytes(),
384                        )?)
385                        .unwrap(),
386                    ),
387                    CallArg::Pure(
388                        bcs::to_bytes(&validator.network_key().as_bytes().to_vec()).unwrap(),
389                    ),
390                    CallArg::Pure(
391                        bcs::to_bytes(&validator.worker_key().as_bytes().to_vec()).unwrap(),
392                    ),
393                    CallArg::Pure(
394                        bcs::to_bytes(&validator_info.proof_of_possession.as_ref().to_vec())
395                            .unwrap(),
396                    ),
397                    CallArg::Pure(
398                        bcs::to_bytes(&validator.name().to_owned().into_bytes()).unwrap(),
399                    ),
400                    CallArg::Pure(
401                        bcs::to_bytes(&validator.description.clone().into_bytes()).unwrap(),
402                    ),
403                    CallArg::Pure(
404                        bcs::to_bytes(&validator.image_url.clone().into_bytes()).unwrap(),
405                    ),
406                    CallArg::Pure(
407                        bcs::to_bytes(&validator.project_url.clone().into_bytes()).unwrap(),
408                    ),
409                    CallArg::Pure(bcs::to_bytes(validator.network_address()).unwrap()),
410                    CallArg::Pure(bcs::to_bytes(validator.p2p_address()).unwrap()),
411                    CallArg::Pure(bcs::to_bytes(validator.narwhal_primary_address()).unwrap()),
412                    CallArg::Pure(bcs::to_bytes(validator.narwhal_worker_address()).unwrap()),
413                    CallArg::Pure(bcs::to_bytes(&validator.gas_price()).unwrap()),
414                    CallArg::Pure(bcs::to_bytes(&validator.commission_rate()).unwrap()),
415                ];
416                let (response, serialized_unsigned_transaction) = call_0x5(
417                    context,
418                    "request_add_validator_candidate",
419                    args,
420                    gas_budget,
421                    tx_args.serialize_unsigned_transaction,
422                )
423                .await?;
424                SuiValidatorCommandResponse::BecomeCandidate {
425                    response,
426                    serialized_unsigned_transaction,
427                }
428            }
429
430            SuiValidatorCommand::JoinCommittee { tx_args } => {
431                let gas_budget = tx_args.gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
432                let (response, serialized_unsigned_transaction) = call_0x5(
433                    context,
434                    "request_add_validator",
435                    vec![],
436                    gas_budget,
437                    tx_args.serialize_unsigned_transaction,
438                )
439                .await?;
440                SuiValidatorCommandResponse::JoinCommittee {
441                    response,
442                    serialized_unsigned_transaction,
443                }
444            }
445
446            SuiValidatorCommand::LeaveCommittee { tx_args } => {
447                // Only an active validator can leave committee.
448                let _status =
449                    check_status(context, HashSet::from([ValidatorStatus::Active])).await?;
450                let gas_budget = tx_args.gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
451                let (response, serialized_unsigned_transaction) = call_0x5(
452                    context,
453                    "request_remove_validator",
454                    vec![],
455                    gas_budget,
456                    tx_args.serialize_unsigned_transaction,
457                )
458                .await?;
459                SuiValidatorCommandResponse::LeaveCommittee {
460                    response,
461                    serialized_unsigned_transaction,
462                }
463            }
464
465            SuiValidatorCommand::DisplayMetadata {
466                validator_address,
467                json,
468            } => {
469                let validator_address = validator_address.unwrap_or(context.active_address()?);
470                // Default display with json serialization for better UX.
471                let sui_client = context.grpc_client()?;
472                display_metadata(&sui_client, validator_address, json.unwrap_or(true)).await?;
473                SuiValidatorCommandResponse::DisplayMetadata
474            }
475
476            SuiValidatorCommand::UpdateMetadata { metadata, tx_args } => {
477                let gas_budget = tx_args.gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
478                let (response, serialized_unsigned_transaction) = update_metadata(
479                    context,
480                    metadata,
481                    gas_budget,
482                    tx_args.serialize_unsigned_transaction,
483                )
484                .await?;
485                SuiValidatorCommandResponse::UpdateMetadata {
486                    response,
487                    serialized_unsigned_transaction,
488                }
489            }
490
491            SuiValidatorCommand::UpdateGasPrice {
492                operation_cap_id,
493                gas_price,
494                tx_args,
495            } => {
496                let gas_budget = tx_args.gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
497                let (response, serialized_unsigned_transaction) = update_gas_price(
498                    context,
499                    operation_cap_id,
500                    gas_price,
501                    gas_budget,
502                    tx_args.serialize_unsigned_transaction,
503                )
504                .await?;
505                SuiValidatorCommandResponse::UpdateGasPrice {
506                    response,
507                    serialized_unsigned_transaction,
508                }
509            }
510
511            SuiValidatorCommand::ReportValidator {
512                operation_cap_id,
513                reportee_address,
514                undo_report,
515                tx_args,
516            } => {
517                let gas_budget = tx_args.gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
518                let undo_report = undo_report.unwrap_or(false);
519                let (response, serialized_unsigned_transaction) = report_validator(
520                    context,
521                    reportee_address,
522                    operation_cap_id,
523                    undo_report,
524                    gas_budget,
525                    tx_args.serialize_unsigned_transaction,
526                )
527                .await?;
528                SuiValidatorCommandResponse::ReportValidator {
529                    response,
530                    serialized_unsigned_transaction,
531                }
532            }
533
534            SuiValidatorCommand::SerializePayloadForPoP {
535                account_address,
536                protocol_public_key,
537            } => {
538                let mut msg: Vec<u8> = Vec::new();
539                msg.extend_from_slice(protocol_public_key.as_bytes());
540                msg.extend_from_slice(account_address.as_ref());
541                let mut intent_msg_bytes = bcs::to_bytes(&IntentMessage::new(
542                    Intent::sui_app(IntentScope::ProofOfPossession),
543                    msg,
544                ))
545                .expect("Message serialization should not fail");
546                DEFAULT_EPOCH_ID.write(&mut intent_msg_bytes);
547                SuiValidatorCommandResponse::SerializedPayload(Base64::encode(&intent_msg_bytes))
548            }
549
550            SuiValidatorCommand::DisplayGasPriceUpdateRawTxn {
551                sender_address,
552                operation_cap_id,
553                new_gas_price,
554                gas_budget,
555            } => {
556                let gas_budget = gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
557                let (_status, _summary, cap_obj_ref) =
558                    get_cap_object_ref(context, Some(operation_cap_id)).await?;
559
560                let args = vec![
561                    CallArg::Object(ObjectArg::ImmOrOwnedObject(cap_obj_ref)),
562                    CallArg::Pure(bcs::to_bytes(&new_gas_price).unwrap()),
563                ];
564                let data = construct_unsigned_0x5_txn(
565                    context,
566                    sender_address,
567                    "request_set_gas_price",
568                    args,
569                    gas_budget,
570                )
571                .await?;
572                let serialized_data = Base64::encode(bcs::to_bytes(&data)?);
573                SuiValidatorCommandResponse::DisplayGasPriceUpdateRawTxn {
574                    data,
575                    serialized_data,
576                }
577            }
578            SuiValidatorCommand::RegisterBridgeCommittee {
579                bridge_authority_key_path,
580                bridge_authority_url,
581                print_unsigned_transaction_only,
582                validator_address,
583                gas_budget,
584            } => {
585                let parsed_url =
586                    Url::parse(&bridge_authority_url).map_err(|e: ParseError| anyhow!(e))?;
587                if parsed_url.scheme() != "http" && parsed_url.scheme() != "https" {
588                    anyhow::bail!(
589                        "URL scheme has to be http or https: {}",
590                        parsed_url.scheme()
591                    );
592                }
593                // Read bridge keypair
594                let ecdsa_keypair = match read_key(&bridge_authority_key_path, true)? {
595                    SuiKeyPair::Secp256k1(key) => key,
596                    _ => unreachable!("we required secp256k1 key in `read_key`"),
597                };
598                let address = check_address(
599                    context.active_address()?,
600                    validator_address,
601                    print_unsigned_transaction_only,
602                )?;
603                // Make sure the address is a validator
604                let sui_client = context.grpc_client()?;
605                let active_validators = sui_client
606                    .get_system_state(None)
607                    .await?
608                    .validators()
609                    .active_validators()
610                    .to_owned();
611                if !active_validators
612                    .into_iter()
613                    .any(|s| s.address() == address.to_string())
614                {
615                    bail!("Address {} is not in the committee", address);
616                }
617                println!(
618                    "Starting bridge committee registration for Sui validator: {address}, with bridge public key: {} and url: {}",
619                    ecdsa_keypair.public, bridge_authority_url
620                );
621                let sui_rpc_url = &context.get_active_env().unwrap().rpc;
622                let bridge_metrics = Arc::new(BridgeMetrics::new_for_testing());
623                let bridge_client = SuiBridgeClient::new(sui_rpc_url, bridge_metrics).await?;
624                let bridge = bridge_client
625                    .get_mutable_bridge_object_arg_must_succeed()
626                    .await;
627
628                let gas_budget = gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
629                let (_, gas) = context
630                    .gas_for_owner_budget(address, gas_budget, Default::default())
631                    .await?;
632
633                let gas_price = context.get_reference_gas_price().await?;
634                let tx_data = build_committee_register_transaction(
635                    address,
636                    &gas.compute_object_reference(),
637                    bridge,
638                    ecdsa_keypair.public().as_bytes().to_vec(),
639                    &bridge_authority_url,
640                    gas_price,
641                    gas_budget,
642                )
643                .map_err(|e| anyhow!("{e:?}"))?;
644                if print_unsigned_transaction_only {
645                    let serialized_data = Base64::encode(bcs::to_bytes(&tx_data)?);
646                    SuiValidatorCommandResponse::RegisterBridgeCommittee {
647                        execution_response: None,
648                        serialized_unsigned_transaction: Some(serialized_data),
649                    }
650                } else {
651                    let tx = context.sign_transaction(&tx_data).await;
652                    let response = context.execute_transaction_must_succeed(tx).await;
653                    println!(
654                        "Committee registration successful. Transaction digest: {}",
655                        response.transaction.digest()
656                    );
657                    SuiValidatorCommandResponse::RegisterBridgeCommittee {
658                        execution_response: Some(response),
659                        serialized_unsigned_transaction: None,
660                    }
661                }
662            }
663            SuiValidatorCommand::UpdateBridgeCommitteeNodeUrl {
664                bridge_authority_url,
665                print_unsigned_transaction_only,
666                validator_address,
667                gas_budget,
668            } => {
669                let parsed_url =
670                    Url::parse(&bridge_authority_url).map_err(|e: ParseError| anyhow!(e))?;
671                if parsed_url.scheme() != "http" && parsed_url.scheme() != "https" {
672                    anyhow::bail!(
673                        "URL scheme has to be http or https: {}",
674                        parsed_url.scheme()
675                    );
676                }
677                // Make sure the address is member of the committee
678                let address = check_address(
679                    context.active_address()?,
680                    validator_address,
681                    print_unsigned_transaction_only,
682                )?;
683                let sui_rpc_url = &context.get_active_env().unwrap().rpc;
684                let bridge_metrics = Arc::new(BridgeMetrics::new_for_testing());
685                let bridge_client = SuiBridgeClient::new(sui_rpc_url, bridge_metrics).await?;
686                let committee_members = bridge_client
687                    .get_bridge_summary()
688                    .await
689                    .map_err(|e| anyhow!("{e:?}"))?
690                    .committee
691                    .members;
692                if !committee_members
693                    .into_iter()
694                    .any(|(_, m)| m.sui_address == address)
695                {
696                    bail!("Address {} is not in the committee", address);
697                }
698                println!(
699                    "Updating bridge committee node URL for Sui validator: {address}, url: {}",
700                    bridge_authority_url
701                );
702
703                let bridge = bridge_client
704                    .get_mutable_bridge_object_arg_must_succeed()
705                    .await;
706
707                let gas_budget = gas_budget.unwrap_or(DEFAULT_GAS_BUDGET);
708                let (_, gas) = context
709                    .gas_for_owner_budget(address, gas_budget, Default::default())
710                    .await?;
711
712                let gas_price = context.get_reference_gas_price().await?;
713                let tx_data = build_committee_update_url_transaction(
714                    address,
715                    &gas.compute_object_reference(),
716                    bridge,
717                    &bridge_authority_url,
718                    gas_price,
719                    gas_budget,
720                )
721                .map_err(|e| anyhow!("{e:?}"))?;
722                if print_unsigned_transaction_only {
723                    let serialized_data = Base64::encode(bcs::to_bytes(&tx_data)?);
724                    SuiValidatorCommandResponse::UpdateBridgeCommitteeURL {
725                        execution_response: None,
726                        serialized_unsigned_transaction: Some(serialized_data),
727                    }
728                } else {
729                    let tx = context.sign_transaction(&tx_data).await;
730                    let response = context.execute_transaction_must_succeed(tx).await;
731                    println!(
732                        "Update Bridge validator node URL successful. Transaction digest: {}",
733                        response.transaction.digest()
734                    );
735                    SuiValidatorCommandResponse::UpdateBridgeCommitteeURL {
736                        execution_response: Some(response),
737                        serialized_unsigned_transaction: None,
738                    }
739                }
740            }
741        })
742    }

745fn check_address(
746    active_address: SuiAddress,
747    validator_address: Option<SuiAddress>,
748    print_unsigned_transaction_only: bool,
749) -> Result<SuiAddress, anyhow::Error> {
750    if !print_unsigned_transaction_only {
751        if let Some(validator_address) = validator_address
752            && validator_address != active_address
753        {
754            bail!(
755                "`--validator-address` must be the same as the current active address: {}",
756                active_address
757            );
758        }
759        Ok(active_address)
760    } else {
761        validator_address
762            .ok_or_else(|| anyhow!("--validator-address must be provided when `print_unsigned_transaction_only` is true"))
763    }
764}

766async fn get_cap_object_ref(
767    context: &mut WalletContext,
768    operation_cap_id: Option<ObjectID>,
769) -> Result<(ValidatorStatus, proto::Validator, ObjectRef)> {
770    let mut sui_client = context.grpc_client()?;
771    if let Some(operation_cap_id) = operation_cap_id {
772        let (status, summary) =
773            get_validator_summary_from_cap_id(&sui_client, operation_cap_id).await?;
774        let cap_obj_ref = sui_client
775            .get_object(summary.operation_cap_id().parse()?)
776            .await?
777            .compute_object_reference();
778        Ok::<(ValidatorStatus, proto::Validator, ObjectRef), anyhow::Error>((
779            status,
780            summary,
781            cap_obj_ref,
782        ))
783    } else {
784        // Sender is Reporter Validator itself.
785        let validator_address = context.active_address()?;
786        let (status, summary) = get_validator_summary(&sui_client, validator_address)
787            .await?
788            .ok_or_else(|| anyhow::anyhow!("{} is not a validator.", validator_address))?;
789        // TODO we should allow validator to perform this operation even though the Cap is not at hand.
790        // But for now we need to make sure the cap is owned by the sender.
791        let cap_object_id = summary.operation_cap_id();
792        let resp = sui_client.get_object(cap_object_id.parse()?).await?;
793        // Safe to unwrap as we ask with `with_owner`.
794        let owner = resp.owner().to_owned();
795        let cap_obj_ref = resp.compute_object_reference();
796        if owner != Owner::AddressOwner(context.active_address()?) {
797            anyhow::bail!(
798                "OperationCap {} is not owned by the sender address {} but {:?}",
799                summary.operation_cap_id(),
800                validator_address,
801                owner
802            );
803        }
804        Ok((status, summary, cap_obj_ref))
805    }
806}

871async fn get_validator_summary_from_cap_id(
872    client: &Client,
873    operation_cap_id: ObjectID,
874) -> anyhow::Result<(ValidatorStatus, proto::Validator)> {
875    let resp = client.clone().get_object(operation_cap_id).await?;
876    let bcs = resp.data.try_as_move().ok_or_else(|| {
877        anyhow::anyhow!(
878            "Object {} does not exist or does not return bcs bytes",
879            operation_cap_id
880        )
881    })?;
882    let cap =
883        bcs::from_bytes::<UnverifiedValidatorOperationCapV1>(bcs.contents()).map_err(|e| {
884            anyhow::anyhow!(
885                "Can't convert bcs bytes of object {} to UnverifiedValidatorOperationCapV1: {}",
886                operation_cap_id,
887                e,
888            )
889        })?;
890    let validator_address = cap.authorizer_validator_address;
891    let (status, summary) = get_validator_summary(client, validator_address)
892        .await?
893        .ok_or_else(|| anyhow::anyhow!("{} is not a validator", validator_address))?;
894    if summary.operation_cap_id() != operation_cap_id.to_string() {
895        anyhow::bail!(
896            "Validator {}'s current operation cap id is {}",
897            validator_address,
898            summary.operation_cap_id()
899        );
900    }
901    Ok((status, summary))
902}

930async fn call_0x5(
931    context: &mut WalletContext,
932    function: &'static str,
933    call_args: Vec<CallArg>,
934    gas_budget: u64,
935    serialize_unsigned_transaction: bool,
936) -> anyhow::Result<(Option<ExecutedTransaction>, Option<String>)> {
937    let sender = context.active_address()?;
938    let tx_data =
939        construct_unsigned_0x5_txn(context, sender, function, call_args, gas_budget).await?;
940    if serialize_unsigned_transaction {
941        let serialized_data = Base64::encode(bcs::to_bytes(&tx_data)?);
942        return Ok((None, Some(serialized_data)));
943    }
944    let signature = context
945        .config
946        .keystore
947        .sign_secure(&sender, &tx_data, Intent::sui_transaction())
948        .await?;
949    let transaction = Transaction::from_data(tx_data, vec![signature]);
950    let response = context
951        .grpc_client()?
952        .execute_transaction_and_wait_for_checkpoint(&transaction)
953        .await?;
954    Ok((Some(response), None))
955}

1033pub fn write_transaction_response(response: &ExecutedTransaction) -> Result<String, fmt::Error> {
1034    // we requested with for full_content, so the following content should be available.
1035    let success = response.effects.status().is_ok();
1036    let lines = vec![
1037        String::from("----- Transaction Digest ----"),
1038        response.transaction.digest().to_string(),
1039        String::from("\n----- Transaction Data ----"),
1040        serde_json::to_string_pretty(&response.transaction).unwrap(),
1041        String::from("----- Transaction Effects ----"),
1042        serde_json::to_string_pretty(&response.effects).unwrap(),
1043    ];
1044    let mut writer = String::new();
1045    for line in lines {
1046        let colorized_line = if success { line.green() } else { line.red() };
1047        writeln!(writer, "{}", colorized_line)?;
1048    }
1049    Ok(writer)
1050}

1090pub async fn get_validator_summary(
1091    client: &Client,
1092    validator_address: SuiAddress,
1093) -> anyhow::Result<Option<(ValidatorStatus, proto::Validator)>> {
1094    let system_state = client.get_system_state(None).await?;
1095    let mut status = None;
1096    let mut active_validators = system_state
1097        .validators()
1098        .active_validators()
1099        .iter()
1100        .map(|s| (s.address().to_owned(), s))
1101        .collect::<BTreeMap<_, _>>();
1102    let validator_info = if active_validators.contains_key(&validator_address.to_string()) {
1103        status = Some(ValidatorStatus::Active);
1104        Some(
1105            active_validators
1106                .remove(&validator_address.to_string())
1107                .unwrap()
1108                .to_owned(),
1109        )
1110    } else {
1111        // Check panding validators
1112        get_pending_candidate_summary(
1113            validator_address,
1114            client,
1115            system_state
1116                .validators()
1117                .pending_active_validators()
1118                .id()
1119                .parse()
1120                .unwrap(),
1121        )
1122        .await?
1123        .tap_some(|_s| status = Some(ValidatorStatus::Pending))
1124
1125        // TODO also check candidate and inactive valdiators
1126    };
1127    if validator_info.is_none() {
1128        return Ok(None);
1129    }
1130    // status is safe unwrap because it has to be Some when the code recahes here
1131    // validator_info is safe to unwrap because of the above check
1132    Ok(Some((status.unwrap(), validator_info.unwrap())))
1133}

1157async fn get_pending_candidate_summary(
1158    validator_address: SuiAddress,
1159    sui_client: &Client,
1160    pending_active_validators_id: ObjectID,
1161) -> anyhow::Result<Option<proto::Validator>> {
1162    let pending_validators = sui_client
1163        .get_dynamic_fields(pending_active_validators_id, None, None)
1164        .await?;
1165    for resp in pending_validators.dynamic_fields() {
1166        // We always expect an objectId from the response as one of data/error should be included.
1167        let object_id = resp.field_id();

1169            anyhow::anyhow!(
1170                "Can't convert bcs bytes of object {} to ValidatorV1: {}",

1172                e,
1173            )
1174        })?;
1175        if field.verified_metadata().sui_address == validator_address {
1176            return Ok(Some(field.into()));
1177        }
1178    }
1179    Ok(None)
1180}

1290            .await
1291        }
1292        MetadataUpdate::PrimaryAddress { primary_address } => {
1293            primary_address.to_anemo_address().map_err(|_| {
1294                anyhow!("Invalid primary address, it must look like `/[ip4,ip6,dns]/.../udp/port`")
1295            })?;
1296            let _status = check_status(context, HashSet::from([Pending, Active])).await?;
1297            let args = vec![CallArg::Pure(bcs::to_bytes(&primary_address).unwrap())];
1298            call_0x5(
1299                context,
1300                "update_validator_next_epoch_primary_address",
1301                args,
1302                gas_budget,
1303                serialize_unsigned_transaction,
1304            )
1305            .await
1306        }
1307        MetadataUpdate::WorkerAddress { worker_address } => {
1308            worker_address.to_anemo_address().map_err(|_| {
1309                anyhow!("Invalid worker address, it must look like `/[ip4,ip6,dns]/.../udp/port`")
1310            })?;
1311            // Only an active validator can leave committee.
1312            let _status = check_status(context, HashSet::from([Pending, Active])).await?;
1313            let args = vec![CallArg::Pure(bcs::to_bytes(&worker_address).unwrap())];
1314            call_0x5(
1315                context,
1316                "update_validator_next_epoch_worker_address",
1317                args,
1318                gas_budget,
1319                serialize_unsigned_transaction,
1320            )
1321            .await
1322        }
1323        MetadataUpdate::P2pAddress { p2p_address } => {
1324            p2p_address.to_anemo_address().map_err(|_| {
1325                anyhow!("Invalid p2p address, it must look like `/[ip4,ip6,dns]/.../udp/port`")
1326            })?;
1327            let _status = check_status(context, HashSet::from([Pending, Active])).await?;
1328            let args = vec![CallArg::Pure(bcs::to_bytes(&p2p_address).unwrap())];
1329            call_0x5(
1330                context,
1331                "update_validator_next_epoch_p2p_address",
1332                args,
1333                gas_budget,
1334                serialize_unsigned_transaction,
1335            )
1336            .await
1337        }
1338        MetadataUpdate::NetworkPubKey { file } => {
1339            let _status = check_status(context, HashSet::from([Pending, Active])).await?;
1340            let network_pub_key: NetworkPublicKey =
1341                read_network_keypair_from_file(file)?.public().clone();
1342            let args = vec![CallArg::Pure(
1343                bcs::to_bytes(&network_pub_key.as_bytes().to_vec()).unwrap(),
1344            )];
1345            call_0x5(
1346                context,
1347                "update_validator_next_epoch_network_pubkey",
1348                args,
1349                gas_budget,
1350                serialize_unsigned_transaction,
1351            )
1352            .await
1353        }
1354        MetadataUpdate::WorkerPubKey { file } => {
1355            let _status = check_status(context, HashSet::from([Pending, Active])).await?;
1356            let worker_pub_key: NetworkPublicKey =
1357                read_network_keypair_from_file(file)?.public().clone();
1358            let args = vec![CallArg::Pure(
1359                bcs::to_bytes(&worker_pub_key.as_bytes().to_vec()).unwrap(),
1360            )];
1361            call_0x5(
1362                context,
1363                "update_validator_next_epoch_worker_pubkey",
1364                args,
1365                gas_budget,
1366                serialize_unsigned_transaction,
1367            )
1368            .await
1369        }
1370        MetadataUpdate::ProtocolPubKey { file } => {
1371            let _status = check_status(context, HashSet::from([Pending, Active])).await?;
1372            let sui_address = context.active_address()?;
1373            let protocol_key_pair: AuthorityKeyPair = read_authority_keypair_from_file(file)?;
1374            let protocol_pub_key: AuthorityPublicKey = protocol_key_pair.public().clone();
1375            let pop = generate_proof_of_possession(&protocol_key_pair, sui_address);
1376            let args = vec![
1377                CallArg::Pure(
1378                    bcs::to_bytes(&AuthorityPublicKeyBytes::from_bytes(
1379                        protocol_pub_key.as_bytes(),
1380                    )?)
1381                    .unwrap(),
1382                ),
1383                CallArg::Pure(bcs::to_bytes(&pop.as_ref().to_vec()).unwrap()),
1384            ];
1385            call_0x5(
1386                context,
1387                "update_validator_next_epoch_protocol_pubkey",
1388                args,
1389                gas_budget,
1390                serialize_unsigned_transaction,
1391            )
1392            .await
1393        }
1394    }
1395}
1396
1397async fn check_status(
1398    context: &mut WalletContext,
1399    allowed_status: HashSet<ValidatorStatus>,
1400) -> Result<ValidatorStatus> {
1401    let sui_client = context.grpc_client()?;
1402    let validator_address = context.active_address()?;
1403    let summary = get_validator_summary(&sui_client, validator_address).await?;
1404    if summary.is_none() {
1405        bail!("{validator_address} is not a Validator.");
1406    }
1407    let (status, _summary) = summary.unwrap();
1408    if allowed_status.contains(&status) {
1409        return Ok(status);
1410    }
1411    bail!(
1412        "Validator {validator_address} is {:?}, this operation is not supported in this tool or prohibited.",
1413        status
1414    )
1415}
