4use move_binary_format::errors::PartialVMResult;
5use move_bytecode_verifier_meter::{Meter, Scope};
6use serde::Serialize;
7
8/// A meter that accumulates all the scopes that it sees, without enforcing a limit.
9#[derive(Debug)]
10pub(crate) struct AccumulatingMeter {
11    pkg_acc: Accumulator,
12    mod_acc: Accumulator,
13    fun_acc: Accumulator,
14}
15
16/// Ticks and child scopes recorded for an individual scope.
17#[derive(Clone, Debug, Serialize)]
18pub struct Accumulator {
19    pub name: String,
20    #[serde(skip)]
21    pub scope: Scope,
22    pub ticks: u128,
23    pub children: Vec<Accumulator>,
24}
25
26impl AccumulatingMeter {
27    pub fn new() -> Self {
28        Self {
29            pkg_acc: Accumulator::new("<unknown>", Scope::Package),
30            mod_acc: Accumulator::new("<unknown>", Scope::Module),
31            fun_acc: Accumulator::new("<unknown>", Scope::Function),
32        }
33    }
34
35    pub fn accumulator(&self, scope: Scope) -> &Accumulator {
36        match scope {
37            Scope::Transaction => unreachable!("transaction scope is not supported"),
38            Scope::Package => &self.pkg_acc,
39            Scope::Module => &self.mod_acc,
40            Scope::Function => &self.fun_acc,
41        }
42    }
43
44    pub fn accumulator_mut(&mut self, scope: Scope) -> &mut Accumulator {
45        match scope {
46            Scope::Transaction => unreachable!("transaction scope is not supported"),
47            Scope::Package => &mut self.pkg_acc,
48            Scope::Module => &mut self.mod_acc,
49            Scope::Function => &mut self.fun_acc,
50        }
51    }
52}
53
54impl Accumulator {
55    fn new(name: &str, scope: Scope) -> Self {
56        Self {
57            name: name.to_string(),
58            scope,
59            ticks: 0,
60            children: vec![],
61        }
62    }
63
64    /// Find the max ticks spent verifying `scope`s within this scope (including itself).
65    pub fn max_ticks(&self, scope: Scope) -> u128 {
66        let mut accs = vec![self];
67
68        let mut curr = 0u128;
69        while let Some(acc) = accs.pop() {
70            if acc.scope == scope {
71                curr = curr.max(acc.ticks);
72            }
73
74            accs.extend(acc.children.iter());
75        }
76
77        curr
78    }
79}
80
81impl Meter for AccumulatingMeter {
82    fn enter_scope(&mut self, name: &str, scope: Scope) {
83        *self.accumulator_mut(scope) = Accumulator::new(name, scope);
84    }
85
86    fn transfer(&mut self, from: Scope, to: Scope, factor: f32) -> PartialVMResult<()> {
87        let from_acc = self.accumulator(from).clone();
88        let to_acc = self.accumulator_mut(to);
89
90        to_acc.ticks += (from_acc.ticks as f32 * factor) as u128;
91        to_acc.children.push(from_acc);
92        Ok(())
93    }
94
95    fn add(&mut self, scope: Scope, units: u128) -> PartialVMResult<()> {
96        self.accumulator_mut(scope).ticks += units;
97        Ok(())
98    }
99}
