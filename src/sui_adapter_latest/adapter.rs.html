41    pub fn new_move_vm(
42        natives: NativeFunctionTable,
43        protocol_config: &ProtocolConfig,
44    ) -> Result<MoveVM, SuiError> {
45        MoveVM::new_with_config(
46            natives,
47            VMConfig {
48                verifier: protocol_config.verifier_config(/* signing_limits */ None),
49                max_binary_format_version: protocol_config.move_binary_format_version(),
50                runtime_limits_config: VMRuntimeLimitsConfig {
51                    vector_len_max: protocol_config.max_move_vector_len(),
52                    max_value_nest_depth: protocol_config.max_move_value_depth_as_option(),
53                    hardened_otw_check: protocol_config.hardened_otw_check(),
54                },
55                enable_invariant_violation_check_in_swap_loc: !protocol_config
56                    .disable_invariant_violation_check_in_swap_loc(),
57                check_no_extraneous_bytes_during_deserialization: protocol_config
58                    .no_extraneous_module_bytes(),
59                // Don't augment errors with execution state on-chain
60                error_execution_state: false,
61                binary_config: protocol_config.binary_config(None),
62                rethrow_serialization_type_layout_errors: protocol_config
63                    .rethrow_serialization_type_layout_errors(),
64                max_type_to_layout_nodes: protocol_config.max_type_to_layout_nodes_as_option(),
65                variant_nodes: protocol_config.variant_nodes(),
66                deprecate_global_storage_ops_during_deserialization: protocol_config
67                    .deprecate_global_storage_ops_during_deserialization(),
68            },
69        )
70        .map_err(|_| SuiErrorKind::ExecutionInvariantViolation.into())
71    }

73    pub fn new_native_extensions<'r>(
74        child_resolver: &'r dyn ChildObjectResolver,
75        input_objects: BTreeMap<ObjectID, object_runtime::InputObject>,
76        is_metered: bool,
77        protocol_config: &'r ProtocolConfig,
78        metrics: Arc<LimitsMetrics>,
79        tx_context: Rc<RefCell<TxContext>>,
80    ) -> NativeContextExtensions<'r> {
81        let current_epoch_id: EpochId = tx_context.borrow().epoch();
82        let mut extensions = NativeContextExtensions::default();
83        extensions.add(ObjectRuntime::new(
84            child_resolver,
85            input_objects,
86            is_metered,
87            protocol_config,
88            metrics,
89            current_epoch_id,
90        ));
91        extensions.add(NativesCostTable::from_protocol_config(protocol_config));
92        extensions.add(TransactionContext::new(tx_context));
93        extensions
94    }

96    /// Given a list of `modules` and an `object_id`, mutate each module's self ID (which must be
97    /// 0x0) to be `object_id`.
98    pub fn substitute_package_id(
99        modules: &mut [CompiledModule],
100        object_id: ObjectID,
101    ) -> Result<(), ExecutionError> {
102        let new_address = AccountAddress::from(object_id);
103
104        for module in modules.iter_mut() {
105            let self_handle = module.self_handle().clone();
106            let self_address_idx = self_handle.address;
107
108            let addrs = &mut module.address_identifiers;
109            let Some(address_mut) = addrs.get_mut(self_address_idx.0 as usize) else {
110                let name = module.identifier_at(self_handle.name);
111                return Err(ExecutionError::new_with_source(
112                    ExecutionErrorKind::PublishErrorNonZeroAddress,
113                    format!("Publishing module {name} with invalid address index"),
114                ));
115            };
116
117            if *address_mut != AccountAddress::ZERO {
118                let name = module.identifier_at(self_handle.name);
119                return Err(ExecutionError::new_with_source(
120                    ExecutionErrorKind::PublishErrorNonZeroAddress,
121                    format!("Publishing module {name} with non-zero address is not allowed"),
122                ));
123            };
124
125            *address_mut = new_address;
126        }
127
128        Ok(())
129    }

138    /// Run the bytecode verifier with a meter limit
139    ///
140    /// This function only fails if the verification does not complete within the limit.  If the
141    /// modules fail to verify but verification completes within the meter limit, the function
142    /// succeeds.
143    #[instrument(level = "trace", skip_all)]
144    pub fn run_metered_move_bytecode_verifier(
145        modules: &[CompiledModule],
146        verifier_config: &VerifierConfig,
147        meter: &mut (impl Meter + ?Sized),
148        metrics: &Arc<BytecodeVerifierMetrics>,
149    ) -> Result<(), SuiError> {
150        // run the Move verifier
151        for module in modules.iter() {
152            let per_module_meter_verifier_timer = metrics
153                .verifier_runtime_per_module_success_latency
154                .start_timer();
155
156            if let Err(e) = verify_module_timeout_only(module, verifier_config, meter) {
157                // We only checked that the failure was due to timeout
158                // Discard success timer, but record timeout/failure timer
159                metrics
160                    .verifier_runtime_per_module_timeout_latency
161                    .observe(per_module_meter_verifier_timer.stop_and_discard());
162                metrics
163                    .verifier_timeout_metrics
164                    .with_label_values(&[
165                        BytecodeVerifierMetrics::OVERALL_TAG,
166                        BytecodeVerifierMetrics::TIMEOUT_TAG,
167                    ])
168                    .inc();
169
170                return Err(e);
171            };
172
173            // Save the success timer
174            per_module_meter_verifier_timer.stop_and_record();
175            metrics
176                .verifier_timeout_metrics
177                .with_label_values(&[
178                    BytecodeVerifierMetrics::OVERALL_TAG,
179                    BytecodeVerifierMetrics::SUCCESS_TAG,
180                ])
181                .inc();
182        }
183
184        Ok(())
185    }

187    /// Run both the Move verifier and the Sui verifier, checking just for timeouts. Returns Ok(())
188    /// if the verifier completes within the module meter limit and the ticks are successfully
189    /// transfered to the package limit (regardless of whether verification succeeds or not).
190    fn verify_module_timeout_only(
191        module: &CompiledModule,
192        verifier_config: &VerifierConfig,
193        meter: &mut (impl Meter + ?Sized),
194    ) -> Result<(), SuiError> {
195        meter.enter_scope(module.self_id().name().as_str(), Scope::Module);
196
197        if let Err(e) = verify_module_with_config_metered(verifier_config, module, meter) {
198            // Check that the status indicates metering timeout.
199            if check_for_verifier_timeout(&e.major_status()) {
200                return Err(SuiErrorKind::ModuleVerificationFailure {
201                    error: format!("Verification timed out: {}", e),
202                }
203                .into());
204            }
205        } else if let Err(err) = sui_verify_module_metered_check_timeout_only(
206            module,
207            &BTreeMap::new(),
208            meter,
209            verifier_config,
210        ) {
211            return Err(err.into());
212        }
213
214        if meter.transfer(Scope::Module, Scope::Package, 1.0).is_err() {
215            return Err(SuiErrorKind::ModuleVerificationFailure {
216                error: "Verification timed out".to_string(),
217            }
218            .into());
219        }
220
221        Ok(())
222    }
