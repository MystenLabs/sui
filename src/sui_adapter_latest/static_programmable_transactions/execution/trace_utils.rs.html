27/// Inserts Move call start event into the trace. As is the case for all other public functions in this module,
28/// its body is (and must be) enclosed in an if statement checking if tracing is enabled.
29pub fn trace_move_call_start(trace_builder_opt: &mut Option<MoveTraceBuilder>) {
30    if let Some(trace_builder) = trace_builder_opt {
31        trace_builder.push_event(TraceEvent::External(Box::new(serde_json::json!(
32            PTBEvent::MoveCallStart
33        ))));
34    }
35}

47/// Inserts transfer event into the trace. As is the case for all other public functions in this module,
48/// its body is (and must be) enclosed in an if statement checking if tracing is enabled.
49pub fn trace_transfer(
50    context: &mut Context,
51    trace_builder_opt: &mut Option<MoveTraceBuilder>,
52    values: &[CtxValue],
53    tys: &[Type],
54) -> Result<(), ExecutionError> {
55    if let Some(trace_builder) = trace_builder_opt {
56        let mut to_transfer = vec![];
57        for (idx, (v, ty)) in values.iter().zip(tys).enumerate() {
58            let tag = adapter_type_to_type_tag_with_refs(ty)?;
59            let layout = annotated_type_layout_for_adapter_ty(context, ty)?;
60            let value = serializable_move_value_from_ctx_value(v, &layout)?;
61            to_transfer.push(ExtMoveValue::Single {
62                name: format!("obj{idx}"),
63                info: ExtMoveValueInfo { type_: tag, value },
64            });
65        }
66        trace_builder.push_event(TraceEvent::External(Box::new(serde_json::json!(
67            PTBEvent::ExternalEvent(ExternalEvent {
68                description: "TransferObjects: obj0...objN => ()".to_string(),
69                name: "Transfer".to_string(),
70                values: to_transfer,
71            })
72        ))));
73    }
74    Ok(())
75}

77/// Inserts PTB summary event into the trace. As is the case for all other public functions in this module,
78/// its body is (and must be) enclosed in an if statement checking if tracing is enabled.
79pub fn trace_ptb_summary(
80    context: &mut Context,
81    trace_builder_opt: &mut Option<MoveTraceBuilder>,
82    commands: &Commands,
83) -> Result<(), ExecutionError> {
84    if let Some(trace_builder) = trace_builder_opt {
85        let events = commands
86            .iter()
87            .map(|c| match &c.value.command {
88                Command__::MoveCall(move_call) => {
89                    let pkg = move_call
90                        .function
91                        .storage_id
92                        .address()
93                        .to_canonical_string(/*with_prefix*/ true);
94                    let module = move_call.function.storage_id.name().to_string();
95                    let function = move_call.function.name.to_string();
96                    Ok(vec![PTBCommandInfo::MoveCall {
97                        pkg,
98                        module,
99                        function,
100                    }])
101                }
102                Command__::TransferObjects(..) => Ok(vec![PTBCommandInfo::ExternalEvent(
103                    "TransferObjects".to_string(),
104                )]),
105                Command__::SplitCoins(..) => Ok(vec![PTBCommandInfo::ExternalEvent(
106                    "SplitCoins".to_string(),
107                )]),
108                Command__::MergeCoins(..) => Ok(vec![PTBCommandInfo::ExternalEvent(
109                    "MergeCoins".to_string(),
110                )]),
111                Command__::Publish(module_bytes, _, _) => {
112                    let mut events = vec![];
113                    events.push(PTBCommandInfo::ExternalEvent("Publish".to_string()));
114                    // Not ideal but it only runs when tracing is enabled so overhead
115                    // should be insignificant
116                    let modules = context.deserialize_modules(module_bytes, false)?;
117                    events.extend(modules.into_iter().find_map(|m| {
118                        for fdef in &m.function_defs {
119                            let fhandle = m.function_handle_at(fdef.function);
120                            let fname = m.identifier_at(fhandle.name);
121                            if fname == INIT_FN_NAME {
122                                return Some(PTBCommandInfo::MoveCall {
123                                    pkg: m.address().to_string(),
124                                    module: m.name().to_string(),
125                                    function: INIT_FN_NAME.to_string(),
126                                });
127                            }
128                        }
129                        None
130                    }));
131                    Ok(events)
132                }
133                Command__::MakeMoveVec(..) => Ok(vec![PTBCommandInfo::ExternalEvent(
134                    "MakeMoveVec".to_string(),
135                )]),
136                Command__::Upgrade(..) => {
137                    Ok(vec![PTBCommandInfo::ExternalEvent("Upgrade".to_string())])
138                }
139            })
140            .collect::<Result<Vec<Vec<PTBCommandInfo>>, ExecutionError>>()?
141            .into_iter()
142            .flatten()
143            .collect();
144        trace_builder.push_event(TraceEvent::External(Box::new(serde_json::json!(
145            PTBEvent::Summary(SummaryEvent {
146                name: "PTBSummary".to_string(),
147                events,
148            })
149        ))));
150    }
151
152    Ok(())
153}

155/// Inserts split coins event into the trace. As is the case for all other public functions in this module,
156/// its body is (and must be) enclosed in an if statement checking if tracing is enabled.
157pub fn trace_split_coins(
158    context: &mut Context,
159    trace_builder_opt: &mut Option<MoveTraceBuilder>,
160    coin_type: &Type,
161    input_coin: Vec<ExtMoveValueInfo>,
162    split_coin_values: &[CtxValue],
163    total_split_value: u64,
164) -> Result<(), ExecutionError> {
165    if let Some(trace_builder) = trace_builder_opt {
166        let type_tag_with_refs = adapter_type_to_type_tag_with_refs(coin_type)?;
167        let layout = annotated_type_layout_for_adapter_ty(context, coin_type)?;
168        let mut split_coin_move_values = vec![];
169        for coin_val in split_coin_values {
170            let coin_val = serializable_move_value_from_ctx_value(coin_val, &layout)?;
171            split_coin_move_values.push(coin_val);
172        }
173
174        let Some([mut input]): Option<[ExtMoveValueInfo; 1]> = input_coin.try_into().ok() else {
175            invariant_violation!("Expected exactly one input coin for tracing `SplitCoins`");
176        };
177
178        update_coin_balance(&mut input, |current_balance| {
179            current_balance.saturating_sub(total_split_value)
180        })?;
181
182        trace_builder.push_event(TraceEvent::External(Box::new(serde_json::json!(
183            PTBEvent::ExternalEvent(ExternalEvent {
184                description: "SplitCoins: input => result".to_string(),
185                name: "SplitCoins".to_string(),
186                values: vec![
187                    ExtMoveValue::Single {
188                        name: "input".to_string(),
189                        info: input
190                    },
191                    ExtMoveValue::Vector {
192                        name: "result".to_string(),
193                        type_: type_tag_with_refs.clone(),
194                        value: split_coin_move_values
195                    },
196                ],
197            })
198        ))));
199    }
200    Ok(())
201}

203/// Inserts merge coins event into the trace. As is the case for all other public functions in this module,
204/// its body is (and must be) enclosed in an if statement checking if tracing is enabled.
205pub fn trace_merge_coins(
206    _context: &mut Context,
207    trace_builder_opt: &mut Option<MoveTraceBuilder>,
208    _coin_type: &Type,
209    mut trace_values: Vec<ExtMoveValueInfo>,
210    total_merged_value: u64,
211) -> Result<(), ExecutionError> {
212    if let Some(trace_builder) = trace_builder_opt {
213        if trace_values.is_empty() {
214            invariant_violation!("Missing destination coin for tracing `MergeCoins`");
215        }
216
217        let target_coin_starting_state = trace_values.remove(0);
218        let mut target_coin_resulting_state = target_coin_starting_state.clone();
219        update_coin_balance(&mut target_coin_resulting_state, |current_balance| {
220            current_balance.saturating_add(total_merged_value)
221        })?;
222
223        let mut values = vec![ExtMoveValue::Single {
224            name: "merge_target".to_string(),
225            info: target_coin_starting_state,
226        }];
227        for (idx, input_value) in trace_values.into_iter().enumerate() {
228            values.push(ExtMoveValue::Single {
229                name: format!("coin{idx}"),
230                info: input_value,
231            });
232        }
233        values.push(ExtMoveValue::Single {
234            name: "merge_result".to_string(),
235            info: target_coin_resulting_state,
236        });
237        trace_builder.push_event(TraceEvent::External(Box::new(serde_json::json!(
238            PTBEvent::ExternalEvent(ExternalEvent {
239                description: "MergeCoins: merge_target, coin0...coinN => mergeresult".to_string(),
240                name: "MergeCoins".to_string(),
241                values,
242            })
243        ))));
244    }
245    Ok(())
246}

255) -> Result<(), ExecutionError> {

264            PTBEvent::ExternalEvent(ExternalEvent {
265                description: "MakeMoveVec: vector".to_string(),
266                name: "MakeMoveVec".to_string(),
267                values: vec![ExtMoveValue::Vector {
268                    name: "vector".to_string(),
269                    type_: type_tag_with_refs,
270                    value: values,
271                }],
272            })
273        ))));
274    }
275    Ok(())
276}

320/// Adds `ExtMoveValueInfo` to the mutable vector passed as an argument.
321/// As is the case for all other public functions in this module,
322/// its body is (and must be) enclosed in an if statement checking if tracing is enabled.
323pub fn add_move_value_info_from_ctx_value(
324    context: &mut Context,
325    trace_builder_opt: &mut Option<MoveTraceBuilder>,
326    move_values: &mut Vec<ExtMoveValueInfo>,
327    type_: &Type,
328    value: &CtxValue,
329) -> Result<(), ExecutionError> {
330    if trace_builder_opt.is_some()
331        && let Some(move_value_info) = move_value_info_from_ctx_value(context, type_, value)?
332    {
333        move_values.push(move_value_info);
334    }
335    Ok(())
336}

338/// Creates `ExtMoveValueInfo` from `Value`.
339fn move_value_info_from_ctx_value(
340    context: &mut Context,
341    type_: &Type,
342    value: &CtxValue,
343) -> Result<Option<ExtMoveValueInfo>, ExecutionError> {
344    let layout = annotated_type_layout_for_adapter_ty(context, type_)?;
345    let value = serializable_move_value_from_ctx_value(value, &layout)?;
346    let type_ = adapter_type_to_type_tag_with_refs(type_)?;
347    Ok(Some(ExtMoveValueInfo { type_, value }))
348}

350fn annotated_type_layout_for_adapter_ty(
351    context: &mut Context,
352    type_: &Type,
353) -> Result<A::MoveTypeLayout, ExecutionError> {
354    let ty = context
355        .env
356        .load_vm_type_argument_from_adapter_type(0, type_)?;
357    context
358        .env
359        .vm
360        .get_runtime()
361        .type_to_fully_annotated_layout(&ty)
362        .map_err(|e| {
363            make_invariant_violation!(
364                "Failed to get annotated type layout for adapter type: {}",
365                e
366            )
367        })
368}

370/// Creates a `SerializableMoveValue` (a Move value for the trace format) from a `CtxValue` and a
371/// provided annotated layout for that value.
372fn serializable_move_value_from_ctx_value(
373    value: &CtxValue,
374    annotated_layout: &A::MoveTypeLayout,
375) -> Result<SerializableMoveValue, ExecutionError> {
376    VMValue::as_annotated_move_value_for_tracing_only(
377        value.inner_for_tracing().inner_for_tracing(),
378        annotated_layout,
379    )
380    .ok_or_else(|| {
381        make_invariant_violation!(
382            "Failed to convert Move value to `SerializableMoveValue` for tracing"
383        )
384    })
385    .map(SerializableMoveValue::from)
386}

388fn update_coin_balance(
389    coin: &mut ExtMoveValueInfo,
390    balance_update: impl Fn(u64) -> u64,
391) -> Result<(), ExecutionError> {
392    use SerializableMoveValue as SMV;
393    let SMV::Struct(SimplifiedMoveStruct { fields, .. }) = &mut coin.value else {
394        invariant_violation!("Expected coin to be a struct");
395    };
396
397    let [_, (_, balance_value)] = fields.as_mut_slice() else {
398        invariant_violation!("Expected coin struct to have two fields");
399    };
400
401    let SMV::Struct(SimplifiedMoveStruct { fields, .. }) = balance_value else {
402        invariant_violation!("Expected balance field to be a struct");
403    };
404
405    let [(_, SMV::U64(current_balance))] = fields.as_mut_slice() else {
406        invariant_violation!("Expected balance struct to have a single u64 field");
407    };
408
409    *current_balance = balance_update(*current_balance);
410
411    Ok(())
412}

414/// Converts a type to type tag format used in tracing.
415fn adapter_type_to_type_tag_with_refs(type_: &Type) -> Result<TypeTagWithRefs, ExecutionError> {
416    Ok(match type_ {
417        Type::Reference(mutable, inner_ty) => {
418            let ref_type = if *mutable { RefType::Mut } else { RefType::Imm };
419            let inner_type: TypeTag = Type::try_into((**inner_ty).clone()).map_err(|e| {
420                make_invariant_violation!(
421                    "Failed to convert adapter type to type tag for tracing: {}",
422                    e
423                )
424            })?;
425            TypeTagWithRefs {
426                type_: inner_type,
427                ref_type: Some(ref_type),
428            }
429        }
430        ty => {
431            let type_: TypeTag = Type::try_into(ty.clone()).map_err(|e| {
432                make_invariant_violation!(
433                    "Failed to convert adapter type to type tag for tracing: {}",
434                    e
435                )
436            })?;
437
438            TypeTagWithRefs {
439                type_,
440                ref_type: None,
441            }
442        }
443    })
