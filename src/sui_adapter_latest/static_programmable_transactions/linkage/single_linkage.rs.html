28/// Replace each command's per-call linkage with a single linkage shared by the whole transaction.
29///
30/// Done in two passes:
31///   1. Fold every command's package and type-argument constraints into one `ResolutionTable`,
32///      unifying as we go (an error here means the commands cannot agree on a single set of
33///      package versions).
34///      - Top level functions are pinned `exact`, while their dependencies are
35///        pinned `exact` or `at_least` based on the visibility of the top-level function.
36///        Type-argument packages are always `at_least`.
37///      - Publishes and upgrades introduce their own constraints to the linkage, but only if
38///        they have an `init` function (otherwise they do not contribute to the linkage). See
39///        comments on each of the command arms for details on this.
40///   2. Write the resulting unified linkage back into every `MoveCall`.
41///
42/// Because all calls end up sharing one linkage, every package version selection is consistent
43/// across the transaction.
44pub fn refine_to_single_linkage<E: ExecutionErrorTrait>(
45    txn: &mut Transaction,
46    linkage_analysis: &LinkageAnalyzer,
47    package_store: &dyn PackageStore,
48    protocol_config: &ProtocolConfig,
49) -> Result<(), E> {
50    let mut base_linkage = linkage_analysis
51        .config()
52        .resolution_table_with_native_packages::<E>(package_store)?;
53
54    for (i, command) in txn.commands.iter().enumerate() {
55        analyze_command::<E>(command, &mut base_linkage, package_store, protocol_config)
56            .map_err(|e| e.with_command_index(i))?;
57    }
58    let resolved_linkage =
59        ExecutableLinkage::new(ResolvedLinkage::from_resolution_table(base_linkage));
60
61    for (i, command) in txn.commands.iter_mut().enumerate() {
62        write_back_linkage::<E>(command, &resolved_linkage).map_err(|e| e.with_command_index(i))?;
63    }
64
65    Ok(())
66}

68/// Fold a single command's contribution into the shared `resolution_table` (pass 1). Only commands
69/// that pull packages into the runtime linkage contribute; the rest are no-ops.
70fn analyze_command<E: ExecutionErrorTrait>(
71    command: &Command,
72    resolution_table: &mut ResolutionTable,
73    store: &dyn PackageStore,
74    protocol_config: &ProtocolConfig,
75) -> Result<(), E> {
76    match command {
77        Command::MoveCall(move_call) => {
78            add_call_to_table::<E>(resolution_table, &move_call.function, store)?;
79        }
80        Command::Publish(PackagePayload::Serialized(_), ..) => {
81            invariant_violation!("Unexpected serialized package payload in linkage analysis")
82        }
83        Command::Publish(
84            PackagePayload::Deserialized(DeserializedPackage {
85                deserialized_modules,
86                ..
87            }),
88            _,
89            resolved_linkage,
90        ) => {
91            // A publish only affects the transaction's linkage if the package has an `init`
92            // function: `init` runs as part of the publish, so its dependencies must be resolvable
93            // in this transaction. Without an `init` the freshly published package is not called
94            // and contributes nothing.
95            //
96            // NB: We presuppose here that if there is a function with the name "init" in the
97            // modules being published, then it is the init function for the package.
98            //
99            // If for some reason it is not (i.e., does not conform to `init` function signature
100            // requirements), the entry points verifier will the publish later, and the transaction
101            // as a whole will error.
102            //
103            // `modules` is guaranteed to be non-empty by the `deserialize_modules` function.
104            if deserialized_modules.iter().any(module_has_init) {
105                for resolved in resolved_linkage.linkage.values() {
106                    add_and_unify(resolved, store, resolution_table, VersionConstraint::exact)?;
107                }
108            }
109        }
110        Command::Upgrade(payload, _, current_package_id, _, resolved_linkage) => {
111            if !protocol_config.enable_init_on_upgrade() {
112                return Ok(());
113            }
114
115            let current_pkg = get_package(current_package_id, store)?;
116
117            assert_invariant!(
118                protocol_config.enable_unified_linkage(),
119                "Unified linkage must be enabled before init on upgrade is supported"
120            );
121
122            let new_modules = match payload {
123                PackagePayload::Serialized(_) => {
124                    invariant_violation!(
125                        "Unexpected serialized package payload in linkage analysis"
126                    )
127                }
128                PackagePayload::Deserialized(DeserializedPackage {
129                    deserialized_modules,
130                    ..
131                }) => deserialized_modules,
132            };
133
134            // Whether each module already present in the current package defines an `init`.
135            let current_module_inits = current_pkg
136                .modules()
137                .iter()
138                .map(|(module_id, module)| {
139                    (
140                        module_id.name().as_str(),
141                        module_has_init(module.compiled_module()),
142                    )
143                })
144                .collect::<BTreeMap<_, _>>();
145
146            // reject upgrades where an existing module adds an `init`.
147            reject_existing_module_added_init::<E>(&current_module_inits, new_modules)?;
148
149            // only newly-introduced modules with an `init` contribute to the linkage.
150            if has_new_module_init(&current_module_inits, new_modules) {
151                add_upgrade_init_linkage_to_table::<E>(
152                    resolution_table,
153                    current_package_id,
154                    resolved_linkage,
155                    store,
156                )?;
157            }
158        }
159        Command::MakeMoveVec(Some(ty), _) => {
160            add_type_packages::<E>(resolution_table, std::iter::once(ty), store)?;
161        }
162        Command::MakeMoveVec(None, _) => (),
163        Command::TransferObjects(_, _) | Command::SplitCoins(_, _) | Command::MergeCoins(_, _) => {}
164    };
165    Ok(())
166}

168/// Reject an upgrade in which a module that already exists in the current package (and did not
169/// previously define an `init`) introduces one.
170fn reject_existing_module_added_init<E: ExecutionErrorTrait>(
171    current_module_inits: &BTreeMap<&str, bool>,
172    new_modules: &[CompiledModule],
173) -> Result<(), E> {
174    for new_module in new_modules {
175        let module_name = new_module
176            .identifier_at(new_module.self_handle().name)
177            .as_str();
178        if current_module_inits.get(module_name) == Some(&false) && module_has_init(new_module) {
179            return Err(<E>::from_kind(ExecutionErrorKind::PackageUpgradeError {
180                upgrade_error: PackageUpgradeError::IncompatibleUpgrade,
181            }));
182        }
183    }
184    Ok(())
185}

187/// Return true if the upgrade introduces at least one new module (absent from the current package)
188/// that defines an `init` function. Existing modules never count (rejected by `reject_existing_module_added_init`).
189fn has_new_module_init(
190    current_module_inits: &BTreeMap<&str, bool>,
191    new_modules: &[CompiledModule],
192) -> bool {
193    new_modules.iter().any(|new_module| {
194        let module_name = new_module
195            .identifier_at(new_module.self_handle().name)
196            .as_str();
197        current_module_inits.get(module_name).is_none() && module_has_init(new_module)
198    })
199}
200
201fn module_has_init(module: &CompiledModule) -> bool {
202    module.function_defs().iter().any(|func_def| {
203        let handle = module.function_handle_at(func_def.function);
204        module.identifier_at(handle.name) == INIT_FN_NAME
205    })
206}

208/// Add the linkage constraints introduced by an upgrade, there are two cases based on whether the
209/// upgraded package already participates in the transaction-wide (Lumpy) linkage:
210///
211/// - If the upgraded package's original id is not already in the resolution table, the upgrade
212///   is treated like a fresh publish-with-init: every entry of its resolved linkage is added as an
213///   `exact` constraint.
214/// - If the upgraded package's original id is in the resolution table, then for any `(original_id,
215///   version_id)` as defined in the `Upgrade` command either:
216///   a. It is not in the existing Lumpy linkage, and a `original_id -> exact(version_id)` constraint is introduced; or
217///   b. It is in the existing Lumpy linkage, in which case Lumpy[original_id].id must equal `version_id`.
218fn add_upgrade_init_linkage_to_table<E: ExecutionErrorTrait>(
219    resolution_table: &mut ResolutionTable,
220    current_package_id: &ObjectID,
221    resolved_linkage: &ResolvedLinkage,
222    store: &dyn PackageStore,
223) -> Result<(), E> {
224    let current_pkg = get_package(current_package_id, store)?;
225    let pkg_original_id: ObjectID = current_pkg.original_id().into();
226
227    if !resolution_table
228        .resolution_table
229        .contains_key(&pkg_original_id)
230    {
231        for resolved in resolved_linkage.linkage.values() {
232            add_and_unify(resolved, store, resolution_table, VersionConstraint::exact)?;
233        }
234        return Ok(());
235    }
236
237    for (original_id, version_id) in &resolved_linkage.linkage {
238        match resolution_table.resolution_table.get(original_id) {
239            None => {
240                add_and_unify(
241                    version_id,
242                    store,
243                    resolution_table,
244                    VersionConstraint::exact,
245                )?;
246            }
247            Some(existing) if existing.object_id() == *version_id => (),
248            Some(existing) => {
249                return Err(E::new_with_source(
250                    ExecutionErrorKind::InvalidLinkage,
251                    format!(
252                        "upgrade init linkage conflicts with transaction linkage: package \
253                         {original_id} resolves to {} in transaction linkage, but upgrade \
254                         linkage requires {version_id}",
255                        existing.object_id(),
256                    ),
257                ));
258            }
259        }
260    }
261
262    Ok(())
263}

265/// Add a `MoveCall`'s target package and type-argument packages to the resolution table.
266///
267/// The called package itself is pinned `exact` (we must run exactly the version being called). Its
268/// dependencies are constrained by the callee's visibility: a public entrypoint is a stable ABI,
269/// so its dependencies may be upgraded (`at_least`); a private/`friend` entrypoint is not, so they
270/// are pinned `exact`. Type-argument packages are always `at_least`, since types resolve upwards
271/// to later versions. This mirrors `LinkageAnalyzer::compute_call_linkage_`.
272fn add_call_to_table<E: ExecutionErrorTrait>(
273    resolution_table: &mut ResolutionTable,
274    function: &LoadedFunction,
275    store: &dyn PackageStore,
276) -> Result<(), E> {
277    let dep_resolution_fn = match function.visibility {
278        Visibility::Public => VersionConstraint::at_least,
279        Visibility::Private | Visibility::Friend => VersionConstraint::exact,
280    };
281    let package: ObjectID = (*function.version_mid.address()).into();
282    add_package::<E>(
283        &package,
284        store,
285        resolution_table,
286        VersionConstraint::exact,
287        dep_resolution_fn,
288    )?;
289    add_type_packages::<E>(resolution_table, function.type_arguments.iter(), store)
290}

292/// Resolve every package mentioned by `types`. Types resolve upwards to later versions, so the
293/// package and its deps are both `at_least`.
294fn add_type_packages<'a, E: ExecutionErrorTrait>(
295    resolution_table: &mut ResolutionTable,
296    types: impl IntoIterator<Item = &'a Type>,
297    store: &dyn PackageStore,
298) -> Result<(), E> {
299    for type_defining_id in types.into_iter().flat_map(|ty| ty.all_addresses()) {
300        add_package::<E>(
301            &ObjectID::from(type_defining_id),
302            store,
303            resolution_table,
304            VersionConstraint::at_least,
305            VersionConstraint::at_least,
306        )?;
307    }
308    Ok(())
309}

311/// Add a package and its transitive dependencies to the resolution table. The package itself
312/// gets `self_resolution_fn`'s constraint; every transitive dep (per the package's linkage
313/// table) gets `dep_resolution_fn`'s constraint.
314fn add_package<E: ExecutionErrorTrait>(
315    object_id: &ObjectID,
316    store: &dyn PackageStore,
317    resolution_table: &mut ResolutionTable,
318    self_resolution_fn: fn(&VerifiedPackage) -> Option<VersionConstraint>,
319    dep_resolution_fn: fn(&VerifiedPackage) -> Option<VersionConstraint>,
320) -> Result<(), E> {
321    let pkg = get_package(object_id, store)?;
322    let transitive_deps = resolution_table
323        .config
324        .linkage_table(&pkg)
325        .into_values()
326        .map(ObjectID::from);
327    add_and_unify(object_id, store, resolution_table, self_resolution_fn)?;
328    for dep_id in transitive_deps {
329        add_and_unify(&dep_id, store, resolution_table, dep_resolution_fn)?;
330    }
331    Ok(())
332}

334/// Overwrite each `MoveCall`'s per-call linkage with the unified transaction-wide linkage (pass 2).
335/// Only `MoveCall`s carry an executable linkage; the other commands need no write-back.
336fn write_back_linkage<E: ExecutionErrorTrait>(
337    command: &mut Command,
338    ptb_linkage: &ExecutableLinkage,
339) -> Result<(), E> {
340    match command {
341        Command::MoveCall(move_call) => {
342            let previous_linkage = &move_call.function.linkage;
343            // Stronger than the length check above: every package the per-call linkage resolved
344            // must still be present in the per-component linkage. Unification only ever adds
345            // packages (the key set is a union across member calls), so a dropped key signals a
346            // bug in how component constraints were folded together.
347            //
348            // Since `linkage`'s keys are a set, this check also implies that
349            // `previous_linkage.0.linkage.len() <= ptb_linkage.0.linkage.len()`.
350            assert_invariant!(
351                previous_linkage
352                    .0
353                    .linkage
354                    .keys()
355                    .all(|k| ptb_linkage.0.linkage.contains_key(k)),
356                "single linkage drops a package that the per-call linkage of MoveCall had resolved"
357            );
358            debug_assert!(
359                previous_linkage.0.linkage.len() <= ptb_linkage.0.linkage.len(),
360                "single linkage has fewer candidates than the per-call linkage of MoveCall"
361            );
362            move_call.function.linkage = ptb_linkage.clone();
363        }
364        Command::TransferObjects(_, _)
365        | Command::SplitCoins(_, _)
366        | Command::MergeCoins(_, _)
367        | Command::MakeMoveVec(_, _)
368        | Command::Publish(_, _, _)
369        | Command::Upgrade(_, _, _, _, _) => (),
370    };
371    Ok(())
