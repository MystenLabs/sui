21pub fn transaction<Mode: ExecutionMode>(
22    meter: &mut TranslationMeter<'_, '_>,
23    env: &Env,
24    tx_context: &TxContext,
25    // which inputs are withdrawals that need to be converted to coins, must
26    // be the same length as the inputs
27    withdrawal_compatibility_inputs: Option<Vec<bool>>,
28    pt: P::ProgrammableTransaction,
29) -> Result<L::Transaction, ExecutionError> {
30    metering::pre_translation::meter(meter, &pt)?;
31    let P::ProgrammableTransaction { inputs, commands } = pt;
32    // withdrawal_compatibility_inputs specified ==> the protocol config flag is set
33    assert_invariant!(
34        withdrawal_compatibility_inputs.is_none()
35            || env
36                .protocol_config
37                .convert_withdrawal_compatibility_ptb_arguments(),
38        "if withdrawal compatibility must be specified, then the flag is set in the protocol config"
39    );
40    let withdrawal_compatibility_inputs =
41        withdrawal_compatibility_inputs.unwrap_or_else(|| vec![false; inputs.len()]);
42    assert_invariant!(
43        inputs.len() == withdrawal_compatibility_inputs.len(),
44        "withdrawal compatibility inputs must be the same length as the inputs"
45    );
46    let inputs = withdrawal_compatibility_inputs
47        .into_iter()
48        .zip(inputs)
49        .map(|(is_withdrawal_compatibility_input, arg)| {
50            input::<Mode>(env, tx_context, is_withdrawal_compatibility_input, arg)
51        })
52        .collect::<Result<Vec<_>, _>>()?;
53    let commands = commands
54        .into_iter()
55        .enumerate()
56        .map(|(idx, cmd)| command(env, cmd).map_err(|e| e.with_command_index(idx)))
57        .collect::<Result<Vec<_>, _>>()?;
58    let loaded_tx = L::Transaction { inputs, commands };
59    metering::loading::meter(meter, &loaded_tx)?;
60    Ok(loaded_tx)
61}

63fn input<Mode: ExecutionMode>(
64    env: &Env,
65    tx_context: &TxContext,
66    // True iff this is a withdrawal that needs to be converted to a coin
67    is_withdrawal_compatibility_input: bool,
68    arg: CallArg,
69) -> Result<(L::InputArg, L::InputType), ExecutionError> {
70    // is_withdrawal_compatibility_input ==> FundsWithdrawal
71    assert_invariant!(
72        !is_withdrawal_compatibility_input || matches!(arg, CallArg::FundsWithdrawal(_)),
73        "withdrawal compatibility inputs must be FundsWithdrawal"
74    );
75    Ok(match arg {
76        CallArg::Pure(bytes) => (L::InputArg::Pure(bytes), L::InputType::Bytes),
77        CallArg::Object(ObjectArg::Receiving(oref)) => {
78            (L::InputArg::Receiving(oref), L::InputType::Bytes)
79        }
80        CallArg::Object(ObjectArg::ImmOrOwnedObject(oref)) => {
81            let id = &oref.0;
82            let obj = env.read_object(id)?;
83            let Some(ty) = obj.type_() else {
84                invariant_violation!("Object {:?} has does not have a Move type", id);
85            };
86            let tag: StructTag = ty.clone().into();
87            let ty = env.load_type_from_struct(&tag)?;
88            let arg = match obj.owner {
89                Owner::AddressOwner(_) => L::ObjectArg::OwnedObject(oref),
90                Owner::Immutable => L::ObjectArg::ImmObject(oref),
91                Owner::ObjectOwner(_)
92                | Owner::Shared { .. }
93                | Owner::ConsensusAddressOwner { .. } => {
94                    assert_invariant!(
95                        Mode::allow_arbitrary_values(),
96                        "Unexpected owner for ImmOrOwnedObject: {:?}",
97                        obj.owner,
98                    );
99                    L::ObjectArg::OwnedObject(oref)
100                }
101            };
102            (L::InputArg::Object(arg), L::InputType::Fixed(ty))
103        }
104        CallArg::Object(ObjectArg::SharedObject {
105            id,
106            initial_shared_version,
107            mutability,
108        }) => {
109            let obj = env.read_object(&id)?;
110            let Some(ty) = obj.type_() else {
111                invariant_violation!("Object {:?} has does not have a Move type", id);
112            };
113            let tag: StructTag = ty.clone().into();
114            let ty = env.load_type_from_struct(&tag)?;
115            let kind = match obj.owner {
116                Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
117                    assert_invariant!(
118                        Mode::allow_arbitrary_values(),
119                        "Unexpected owner for SharedObject: {:?}",
120                        obj.owner
121                    );
122                    L::SharedObjectKind::Party
123                }
124                Owner::Shared { .. } => L::SharedObjectKind::Legacy,
125                Owner::ConsensusAddressOwner { .. } => L::SharedObjectKind::Party,
126            };
127            (
128                L::InputArg::Object(L::ObjectArg::SharedObject {
129                    id,
130                    initial_shared_version,
131                    mutability: object_mutability(mutability),
132                    kind,
133                }),
134                L::InputType::Fixed(ty),
135            )
136        }
137        CallArg::FundsWithdrawal(f) => {
138            assert_invariant!(
139                env.protocol_config.enable_accumulators(),
140                "Withdrawals should be rejected at signing if accumulators are not enabled"
141            );
142            let FundsWithdrawalArg {
143                reservation,
144                type_arg,
145                withdraw_from,
146            } = f;
147            let amount = match reservation {
148                P::Reservation::MaxAmountU64(u) => U256::from(u),
149                // TODO when types other than u64 are supported, we must check that this is a
150                // valid amount for the type
151            };
152            let funds_ty = match type_arg {
153                P::WithdrawalTypeArg::Balance(inner) => {
154                    let inner = env.load_type_tag(0, &inner)?;
155                    env.balance_type(inner)?
156                }
157            };
158            let ty = env.withdrawal_type(funds_ty.clone())?;
159            let owner: AccountAddress = match withdraw_from {
160                P::WithdrawFrom::Sender => tx_context.sender().into(),
161                P::WithdrawFrom::Sponsor => tx_context
162                    .sponsor()
163                    .ok_or_else(|| {
164                        make_invariant_violation!(
165                            "A sponsor withdrawal requires a sponsor and should have been \
166                            checked at signing"
167                        )
168                    })?
169                    .into(),
170            };
171            (
172                L::InputArg::FundsWithdrawal(L::FundsWithdrawalArg {
173                    from_compatibility_object: is_withdrawal_compatibility_input,
174                    amount,
175                    ty: ty.clone(),
176                    owner,
177                }),
178                L::InputType::Fixed(ty),
179            )
180        }
181    })
182}

184fn object_mutability(mutability: SharedObjectMutability) -> L::ObjectMutability {
185    match mutability {
186        SharedObjectMutability::Mutable => L::ObjectMutability::Mutable,
187        SharedObjectMutability::NonExclusiveWrite => L::ObjectMutability::NonExclusiveWrite,
188        SharedObjectMutability::Immutable => L::ObjectMutability::Immutable,
189    }
190}

192fn command(env: &Env, command: P::Command) -> Result<L::Command, ExecutionError> {
193    Ok(match command {
194        P::Command::MoveCall(pmc) => {
195            let resolved_linkage = env
196                .linkage_analysis
197                .compute_call_linkage(&pmc, env.linkable_store)?;
198            let P::ProgrammableMoveCall {
199                package,
200                module,
201                function: name,
202                type_arguments: ptype_arguments,
203                arguments,
204            } = *pmc;
205            let linkage = RootedLinkage::new(*package, resolved_linkage);
206            let type_arguments = ptype_arguments
207                .into_iter()
208                .enumerate()
209                .map(|(idx, ty)| env.load_type_input(idx, ty))
210                .collect::<Result<Vec<_>, _>>()?;
211            let function = env.load_function(package, module, name, type_arguments, linkage)?;
212            L::Command::MoveCall(Box::new(L::MoveCall {
213                function,
214                arguments,
215            }))
216        }
217        P::Command::MakeMoveVec(ptype_argument, arguments) => {
218            let type_argument = ptype_argument
219                .map(|ty| env.load_type_input(0, ty))
220                .transpose()?;
221            L::Command::MakeMoveVec(type_argument, arguments)
222        }
223        P::Command::TransferObjects(objects, address) => {
224            L::Command::TransferObjects(objects, address)
225        }
226        P::Command::SplitCoins(coin, amounts) => L::Command::SplitCoins(coin, amounts),
227        P::Command::MergeCoins(target, coins) => L::Command::MergeCoins(target, coins),
228        P::Command::Publish(items, object_ids) => {
229            let resolved_linkage = env
230                .linkage_analysis
231                .compute_publication_linkage(&object_ids, env.linkable_store)?;
232            L::Command::Publish(items, object_ids, resolved_linkage)
233        }
234        P::Command::Upgrade(items, object_ids, object_id, argument) => {
235            let resolved_linkage = env
236                .linkage_analysis
237                .compute_publication_linkage(&object_ids, env.linkable_store)?;
238            L::Command::Upgrade(items, object_ids, object_id, argument, resolved_linkage)
239        }
240    })
241}
