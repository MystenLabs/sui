35pub fn execute<Mode: ExecutionMode>(
36    protocol_config: &ProtocolConfig,
37    metrics: Arc<LimitsMetrics>,
38    vm: &MoveVM,
39    state_view: &mut dyn ExecutionState,
40    package_store: &dyn BackingPackageStore,
41    tx_context: Rc<RefCell<TxContext>>,
42    gas_charger: &mut GasCharger,
43    // which inputs are withdrawals that need to be converted to coins
44    withdrawal_compatibility_inputs: Option<Vec<bool>>,
45    txn: ProgrammableTransaction,
46    trace_builder_opt: &mut Option<MoveTraceBuilder>,
47) -> ResultWithTimings<Mode::ExecutionResults, ExecutionError> {
48    let package_store = CachedPackageStore::new(Box::new(package_store));
49    let linkage_analysis =
50        LinkageAnalyzer::new::<Mode>(protocol_config).map_err(|e| (e, vec![]))?;
51
52    let mut env = Env::new(
53        protocol_config,
54        vm,
55        state_view,
56        &package_store,
57        &linkage_analysis,
58    );
59    let mut translation_meter =
60        translation_meter::TranslationMeter::new(protocol_config, gas_charger);
61
62    let txn = {
63        let tx_context_ref = tx_context.borrow();
64        loading::translate::transaction::<Mode>(
65            &mut translation_meter,
66            &env,
67            &tx_context_ref,
68            withdrawal_compatibility_inputs,
69            txn,
70        )
71        .map_err(|e| (e, vec![]))?
72    };
73    let txn = typing::translate_and_verify::<Mode>(&mut translation_meter, &env, txn)
74        .map_err(|e| (e, vec![]))?;
75    execution::interpreter::execute::<Mode>(
76        &mut env,
77        metrics,
78        tx_context,
79        gas_charger,
80        txn,
81        trace_builder_opt,
82    )
83}
