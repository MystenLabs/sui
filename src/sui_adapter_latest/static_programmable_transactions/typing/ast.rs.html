158// Non borrowing usage of locations, moving or copying
159#[derive(Clone, Debug)]
160pub enum Usage {
161    Move(Location),
162    Copy {
163        location: Location,
164        /// Was this location borrowed at the time of copying?
165        /// Initially empty and populated by `memory_safety`
166        borrowed: OnceCell<bool>,
167    },
168}
169
170pub type Argument = Spanned<Argument_>;
171pub type Argument_ = (Argument__, Type);
172
173#[derive(Clone, Debug)]
174pub enum Argument__ {
175    /// Move or copy a value
176    Use(Usage),
177    /// Borrow a value, i.e. `&x` or `&mut x`
178    Borrow(/* mut */ bool, Location),
179    /// Read a value from a reference, i.e. `*&x`
180    Read(Usage),
181    /// Freeze a mutable reference, making an `&t` from `&mut t`
182    Freeze(Usage),
183}

189impl Transaction {
190    pub fn types(&self) -> impl Iterator<Item = &Type> {
191        let pure_types = self.pure.iter().map(|p| &p.ty);
192        let object_types = self.objects.iter().map(|o| &o.ty);
193        let receiving_types = self.receiving.iter().map(|r| &r.ty);
194        let command_types = self.commands.iter().flat_map(command_types);
195        pure_types
196            .chain(object_types)
197            .chain(receiving_types)
198            .chain(command_types)
199    }
200}
201
202impl Usage {
203    pub fn new_move(location: Location) -> Usage {
204        Usage::Move(location)
205    }
206
207    pub fn new_copy(location: Location) -> Usage {
208        Usage::Copy {
209            location,
210            borrowed: OnceCell::new(),
211        }
212    }
213
214    pub fn location(&self) -> Location {
215        match self {
216            Usage::Move(location) => *location,
217            Usage::Copy { location, .. } => *location,
218        }
219    }
220}
221
222impl Argument__ {
223    pub fn new_move(location: Location) -> Self {
224        Self::Use(Usage::new_move(location))
225    }
226
227    pub fn new_copy(location: Location) -> Self {
228        Self::Use(Usage::new_copy(location))
229    }
230
231    pub fn location(&self) -> Location {
232        match self {
233            Self::Use(usage) | Self::Read(usage) => usage.location(),
234            Self::Borrow(_, location) => *location,
235            Self::Freeze(usage) => usage.location(),
236        }
237    }
238}
239
240impl Command__ {
241    pub fn arguments(&self) -> Box<dyn Iterator<Item = &Argument> + '_> {
242        match self {
243            Command__::MoveCall(mc) => Box::new(mc.arguments.iter()),
244            Command__::TransferObjects(objs, addr) => {
245                Box::new(objs.iter().chain(std::iter::once(addr)))
246            }
247            Command__::SplitCoins(_, coin, amounts) => {
248                Box::new(std::iter::once(coin).chain(amounts))
249            }
250            Command__::MergeCoins(_, target, sources) => {
251                Box::new(std::iter::once(target).chain(sources))
252            }
253            Command__::MakeMoveVec(_, elems) => Box::new(elems.iter()),
254            Command__::Publish(_, _, _) => Box::new(std::iter::empty()),
255            Command__::Upgrade(_, _, _, arg, _) => Box::new(std::iter::once(arg)),
256        }
257    }
258
259    pub fn types(&self) -> Box<dyn Iterator<Item = &Type> + '_> {
260        match self {
261            Command__::TransferObjects(args, arg) => {
262                Box::new(std::iter::once(arg).chain(args.iter()).map(argument_type))
263            }
264            Command__::SplitCoins(ty, arg, args) | Command__::MergeCoins(ty, arg, args) => {
265                Box::new(
266                    std::iter::once(arg)
267                        .chain(args.iter())
268                        .map(argument_type)
269                        .chain(std::iter::once(ty)),
270                )
271            }
272            Command__::MakeMoveVec(ty, args) => {
273                Box::new(args.iter().map(argument_type).chain(std::iter::once(ty)))
274            }
275            Command__::MoveCall(call) => Box::new(
276                call.arguments
277                    .iter()
278                    .map(argument_type)
279                    .chain(call.function.type_arguments.iter())
280                    .chain(call.function.signature.parameters.iter())
281                    .chain(call.function.signature.return_.iter()),
282            ),
283            Command__::Upgrade(_, _, _, arg, _) => {
284                Box::new(std::iter::once(arg).map(argument_type))
285            }
286            Command__::Publish(_, _, _) => Box::new(std::iter::empty()),
287        }
288    }
289
290    pub fn arguments_len(&self) -> usize {
291        let n = match self {
292            Command__::MoveCall(mc) => mc.arguments.len(),
293            Command__::TransferObjects(objs, _) => objs.len().saturating_add(1),
294            Command__::SplitCoins(_, _, amounts) => amounts.len().saturating_add(1),
295            Command__::MergeCoins(_, _, sources) => sources.len().saturating_add(1),
296            Command__::MakeMoveVec(_, elems) => elems.len(),
297            Command__::Publish(_, _, _) => 0,
298            Command__::Upgrade(_, _, _, _, _) => 1,
299        };
300        debug_assert_eq!(self.arguments().count(), n);
301        n
302    }
303}

309pub fn command_types(cmd: &Command) -> impl Iterator<Item = &Type> {
310    let result_types = cmd.value.result_type.iter();
311    let command_types = cmd.value.command.types();
312    result_types.chain(command_types)
313}

323impl TryFrom<Type> for VectorSpecialization {
324    type Error = &'static str;
325
326    fn try_from(value: Type) -> Result<Self, Self::Error> {
327        Ok(match value {
328            Type::U8 => VectorSpecialization::U8,
329            Type::U16 => VectorSpecialization::U16,
330            Type::U32 => VectorSpecialization::U32,
331            Type::U64 => VectorSpecialization::U64,
332            Type::U128 => VectorSpecialization::U128,
333            Type::U256 => VectorSpecialization::U256,
334            Type::Address => VectorSpecialization::Address,
335            Type::Bool => VectorSpecialization::Bool,
336            Type::Signer | Type::Vector(_) | Type::Datatype(_) => VectorSpecialization::Container,
337            Type::Reference(_, _) => return Err("unexpected reference in vector specialization"),
338        })
339    }
