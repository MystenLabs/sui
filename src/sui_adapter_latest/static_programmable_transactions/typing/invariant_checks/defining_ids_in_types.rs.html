4use crate::{
5    data_store::PackageStore,
6    sp,
7    static_programmable_transactions::{env, spanned::Spanned, typing::ast as T},
8};
9use sui_types::error::ExecutionError;
10
11pub fn verify(env: &env::Env, tt: &T::Transaction) -> Result<(), ExecutionError> {
12    let check_type = |ty| ensure_type_defining_id_based(env, ty);
13    let check_arg = |sp!(_, (_, ty)): &Spanned<_>| ensure_type_defining_id_based(env, ty);
14
15    // Verify all types in inputs are defining-id based.
16    tt.objects
17        .iter()
18        .try_for_each(|object_input| check_type(&object_input.ty))?;
19    tt.pure
20        .iter()
21        .try_for_each(|pure_input| check_type(&pure_input.ty))?;
22    tt.receiving
23        .iter()
24        .try_for_each(|receiving_input| check_type(&receiving_input.ty))?;
25
26    // Verify all types in commands are defining-id based.
27    tt.commands.iter().try_for_each(|sp!(_, c)| {
28        c.result_type.iter().try_for_each(check_type)?;
29        match &c.command {
30            T::Command__::Publish(_, _, _) => Ok(()),
31            T::Command__::Upgrade(_, _, _, sp!(_, (_, ty)), _) => check_type(ty),
32            T::Command__::SplitCoins(ty, sp!(_, (_, coin_ty)), amounts) => {
33                check_type(ty)?;
34                check_type(coin_ty)?;
35                amounts.iter().try_for_each(check_arg)
36            }
37            T::Command__::MergeCoins(ty, sp!(_, (_, target_ty)), coins) => {
38                check_type(ty)?;
39                check_type(target_ty)?;
40                coins.iter().try_for_each(check_arg)
41            }
42            T::Command__::MakeMoveVec(ty, args) => {
43                check_type(ty)?;
44                args.iter().try_for_each(check_arg)
45            }
46            T::Command__::TransferObjects(objs, sp!(_, (_, recipient_ty))) => {
47                objs.iter().try_for_each(check_arg)?;
48                check_type(recipient_ty)
49            }
50            T::Command__::MoveCall(move_call) => {
51                move_call
52                    .function
53                    .type_arguments
54                    .iter()
55                    .try_for_each(check_type)?;
56                move_call.arguments.iter().try_for_each(check_arg)
57            }
58        }
59    })
60}
61
62fn ensure_type_defining_id_based(env: &env::Env, ty: &T::Type) -> Result<(), ExecutionError> {
63    match ty {
64        T::Type::Bool
65        | T::Type::U8
66        | T::Type::U16
67        | T::Type::U32
68        | T::Type::U64
69        | T::Type::U128
70        | T::Type::U256
71        | T::Type::Address
72        | T::Type::Signer => Ok(()),
73        T::Type::Reference(_, ty) => ensure_type_defining_id_based(env, ty),
74        T::Type::Vector(vector) => ensure_type_defining_id_based(env, &vector.element_type),
75        T::Type::Datatype(datatype) => {
76            // Resolve the type to its defining ID and ensure it matches the module address that is
77            // already written down as the defining ID.
78            //
79            // If we fail to resolve the type that's an invariant violation as we should be able to
80            // load the package that defined this type otherwise we should have failed at
81            // load/typing time.
82            let Ok(Some(resolved_id)) = env.linkable_store.resolve_type_to_defining_id(
83                (*datatype.module.address()).into(),
84                datatype.module.name(),
85                &datatype.name,
86            ) else {
87                invariant_violation!("[defining_ids_in_types] Unable to resolve Type {ty:?}",);
88            };
89
90            if *resolved_id != *datatype.module.address() {
91                invariant_violation!(
92                    "[defining_ids_in_types] Type {ty:?} has a different defining ID {} than expected: {resolved_id}",
93                    datatype.module.address()
94                );
95            }
96
97            datatype
98                .type_arguments
99                .iter()
100                .try_for_each(|arg| ensure_type_defining_id_based(env, arg))
101        }
102    }
103}
