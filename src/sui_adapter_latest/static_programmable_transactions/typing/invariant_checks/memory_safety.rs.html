12/// A dot-star like extension, but with a unique identifier. Deltas can be compared between
13/// different Deltas of the same command, otherwise they behave like .* in the regex based
14/// implementation. This means that it represents an arbitrary field extension of the reference
15/// in question. However, due to invariants within reference safety, for mutable references these
16/// extensions cannot with other references from the same command.
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
18struct Delta {
19    command: u16,
20    result: u16,
21}
22
23#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
24enum RootLocation {
25    /// The result of a command, specifically a `MoveCall`, without any input references.
26    /// These calls will always abort, but we still must track them.
27    Unknown {
28        command: u16,
29    },
30    Known(T::Location),
31}
32
33/// A path points to an abstract memory location, rooted in an input or command result. Any
34/// extension is the result from a reference being returned from a command.
35#[derive(Debug, Clone, PartialEq, Eq, Hash)]
36struct Path {
37    root: RootLocation,
38    extensions: Vec<Delta>,
39}
40
41#[derive(Debug)]
42struct PathSet(IndexSet<Path>);
43
44#[derive(Debug)]
45enum Value {
46    NonRef,
47    Ref { is_mut: bool, paths: Rc<PathSet> },
48}
49
50#[derive(Debug)]
51struct Location {
52    // A singleton set pointing to the location itself
53    self_path: Rc<PathSet>,
54    value: Option<Value>,
55}
56
57#[derive(Debug)]
58struct Context {
59    tx_context: Location,
60    gas: Location,
61    object_inputs: Vec<Location>,
62    withdrawal_inputs: Vec<Location>,
63    pure_inputs: Vec<Location>,
64    receiving_inputs: Vec<Location>,
65    results: Vec<Vec<Location>>,
66    // Temporary set of locations borrowed by arguments seen thus far for the current command.
67    // Used exclusively for checking the validity copy/move.
68    arg_roots: IndexSet<T::Location>,
69}
70
71enum PathComparison {
72    /// `self` is a strict prefix of `other`
73    Prefix,
74    /// `self` and `other` are the same path
75    Aliases,
76    /// `self` extends `other`
77    Extends,
78    /// `self` and `other` point to distinct regions of memory. They might however be rooted
79    /// in the same parent region, i.e. the same `root` location or a prefix of the same
80    /// `extensions`
81    Disjoint,
82}
83
84impl Path {
85    fn initial(location: T::Location) -> Self {
86        Self {
87            root: RootLocation::Known(location),
88            extensions: vec![],
89        }
90    }
91
92    /// See `PathComparison` for the meaning of the return value
93    fn compare(&self, other: &Self) -> PathComparison {
94        if self.root != other.root {
95            return PathComparison::Disjoint;
96        };
97        let mut self_extensions = self.extensions.iter();
98        let mut other_extensions = other.extensions.iter();
99        loop {
100            match (self_extensions.next(), other_extensions.next()) {
101                (Some(self_ext), Some(other_ext)) => {
102                    if self_ext.command != other_ext.command {
103                        // Cannot compare `Delta` from distinct commands, as such we do assume
104                        // the possibility that `self` extends `other`
105                        return PathComparison::Extends;
106                    }
107                    if self_ext.result != other_ext.result {
108                        // If the command is the same, but the result is different, we know that
109                        // they must be disjoint. Or they are immutable references, in which case
110                        // we do not care.
111                        return PathComparison::Disjoint;
112                    }
113                }
114                (None, Some(_)) => return PathComparison::Prefix,
115                (Some(_), None) => return PathComparison::Extends,
116                (None, None) => return PathComparison::Aliases,
117            }
118        }
119    }
120
121    /// Create a new `Path` that extends the current path with the given `extension`.
122    fn extend(&self, extension: Delta) -> Self {
123        let mut new_extensions = self.extensions.clone();
124        new_extensions.push(extension);
125        Self {
126            root: self.root,
127            extensions: new_extensions,
128        }
129    }
130}
131
132impl PathSet {
133    /// Should be used only for `call` for creating initial path sets.
134    fn empty() -> Self {
135        Self(IndexSet::new())
136    }
137
138    fn initial(location: T::Location) -> Self {
139        Self(IndexSet::from([Path::initial(location)]))
140    }
141
142    fn unknown_root(command: u16) -> Self {
143        Self(IndexSet::from([Path {
144            root: RootLocation::Unknown { command },
145            extensions: vec![],
146        }]))
147    }
148
149    fn is_empty(&self) -> bool {
150        self.0.is_empty()
151    }
152
153    /// Returns true if any path in `self` `Extends` with any path in `other`.
154    /// Excludes `Aliases` if `ignore_aliases` is true.
155    fn extends(&self, other: &Self, ignore_aliases: bool) -> bool {
156        self.0.iter().any(|self_path| {
157            other
158                .0
159                .iter()
160                .any(|other_path| match self_path.compare(other_path) {
161                    PathComparison::Prefix | PathComparison::Disjoint => false,
162                    PathComparison::Aliases => !ignore_aliases,
163                    PathComparison::Extends => true,
164                })
165        })
166    }
167
168    /// Returns true if all paths in `self` are `Disjoint` with all paths in `other`.
169    fn is_disjoint(&self, other: &Self) -> bool {
170        self.0.iter().all(|self_path| {
171            other
172                .0
173                .iter()
174                .all(|other_path| match self_path.compare(other_path) {
175                    PathComparison::Disjoint => true,
176                    PathComparison::Prefix | PathComparison::Aliases | PathComparison::Extends => {
177                        false
178                    }
179                })
180        })
181    }
182
183    /// Insert all paths from `other` into `self`.
184    fn union(&mut self, other: &PathSet) {
185        // We might be able to optimize this slightly by not including paths that are extensions
186        // of existing paths
187        self.0.extend(other.0.iter().cloned());
188    }
189
190    /// Create a new `PathSet` where all paths in `self` are extended with the given `extension`.
191    fn extend(&self, extension: Delta) -> Self {
192        let mut new_paths = IndexSet::with_capacity(self.0.len());
193        for path in &self.0 {
194            new_paths.insert(path.extend(extension));
195        }
196        Self(new_paths)
197    }
198}

200impl Value {
201    /// Create a new reference value
202    fn ref_(is_mut: bool, paths: PathSet) -> anyhow::Result<Value> {
203        anyhow::ensure!(
204            !paths.is_empty(),
205            "Cannot create a reference with an empty path set"
206        );
207        Ok(Value::Ref {
208            is_mut,
209            paths: Rc::new(paths),
210        })
211    }
212
213    fn copy(&self) -> Value {
214        match self {
215            Value::NonRef => Value::NonRef,
216            Value::Ref { is_mut, paths } => Value::Ref {
217                is_mut: *is_mut,
218                paths: paths.clone(),
219            },
220        }
221    }
222
223    fn freeze(&mut self) -> anyhow::Result<Value> {
224        let copied = self.copy();
225        match copied {
226            Value::NonRef => {
227                anyhow::bail!("Cannot freeze a non-reference value")
228            }
229            Value::Ref { is_mut, paths } => {
230                anyhow::ensure!(is_mut, "Cannot freeze an immutable reference");
231                Ok(Value::Ref {
232                    is_mut: false,
233                    paths,
234                })
235            }
236        }
237    }
238}

240impl Location {
241    fn non_ref(location: T::Location) -> Self {
242        Self {
243            self_path: Rc::new(PathSet::initial(location)),
244            value: Some(Value::NonRef),
245        }
246    }
247
248    fn copy_value(&self) -> anyhow::Result<Value> {
249        let Some(value) = self.value.as_ref() else {
250            anyhow::bail!("Use of invalid memory location")
251        };
252        Ok(value.copy())
253    }
254
255    fn move_value(&mut self) -> anyhow::Result<Value> {
256        let Some(value) = self.value.take() else {
257            anyhow::bail!("Use of invalid memory location")
258        };
259        Ok(value)
260    }
261
262    fn use_(&mut self, usage: &T::Usage) -> anyhow::Result<Value> {
263        match usage {
264            T::Usage::Move(_) => self.move_value(),
265            T::Usage::Copy { .. } => self.copy_value(),
266        }
267    }
268
269    fn borrow(&mut self, is_mut: bool) -> anyhow::Result<Value> {
270        let Some(value) = self.value.as_ref() else {
271            anyhow::bail!("Borrow of invalid memory location")
272        };
273        match value {
274            Value::Ref { .. } => {
275                anyhow::bail!("Cannot borrow a reference")
276            }
277            Value::NonRef => {
278                anyhow::ensure!(
279                    !self.self_path.is_empty(),
280                    "Cannot have an empty location to borrow from"
281                );
282                // a new reference that borrows from this location
283                Ok(Value::Ref {
284                    is_mut,
285                    paths: self.self_path.clone(),
286                })
287            }
288        }
289    }
290}

292impl Context {
293    fn new(txn: &T::Transaction) -> anyhow::Result<Self> {
294        let T::Transaction {
295            bytes: _,
296            objects,
297            withdrawals,
298            pure,
299            receiving,
300            withdrawal_compatibility_conversions: _,
301            commands: _,
302        } = txn;
303        let tx_context = Location::non_ref(T::Location::TxContext);
304        let gas = Location::non_ref(T::Location::GasCoin);
305        let object_inputs = (0..objects.len())
306            .map(|i| {
307                Ok(Location::non_ref(T::Location::ObjectInput(checked_as!(
308                    i, u16
309                )?)))
310            })
311            .collect::<Result<_, ExecutionError>>()?;
312        let withdrawal_inputs = (0..withdrawals.len())
313            .map(|i| {
314                Ok(Location::non_ref(T::Location::WithdrawalInput(
315                    checked_as!(i, u16)?,
316                )))
317            })
318            .collect::<Result<_, ExecutionError>>()?;
319        let pure_inputs = (0..pure.len())
320            .map(|i| {
321                Ok(Location::non_ref(T::Location::PureInput(checked_as!(
322                    i, u16
323                )?)))
324            })
325            .collect::<Result<_, ExecutionError>>()?;
326        let receiving_inputs = (0..receiving.len())
327            .map(|i| {
328                Ok(Location::non_ref(T::Location::ReceivingInput(checked_as!(
329                    i, u16
330                )?)))
331            })
332            .collect::<Result<_, ExecutionError>>()?;
333        Ok(Self {
334            tx_context,
335            gas,
336            object_inputs,
337            withdrawal_inputs,
338            pure_inputs,
339            receiving_inputs,
340            results: vec![],
341            arg_roots: IndexSet::new(),
342        })
343    }
344
345    fn current_command(&self) -> anyhow::Result<u16> {
346        Ok(checked_as!(self.results.len(), u16)?)
347    }
348
349    fn add_result_values(
350        &mut self,
351        results: impl IntoIterator<Item = Option<Value>>,
352    ) -> anyhow::Result<()> {
353        let command = self.current_command()?;
354        self.results.push(
355            results
356                .into_iter()
357                .enumerate()
358                .map(|(i, v)| {
359                    Ok(Location {
360                        self_path: Rc::new(PathSet::initial(T::Location::Result(
361                            command,
362                            checked_as!(i, u16)?,
363                        ))),
364                        value: v,
365                    })
366                })
367                .collect::<Result<_, ExecutionError>>()?,
368        );
369        Ok(())
370    }

372    fn location(&self, loc: T::Location) -> anyhow::Result<&Location> {
373        Ok(match loc {
374            T::Location::TxContext => &self.tx_context,
375            T::Location::GasCoin => &self.gas,
376            T::Location::ObjectInput(i) => self
377                .object_inputs
378                .get(i as usize)
379                .ok_or_else(|| anyhow::anyhow!("Object input index out of bounds {i}"))?,
380            T::Location::WithdrawalInput(i) => self
381                .withdrawal_inputs
382                .get(i as usize)
383                .ok_or_else(|| anyhow::anyhow!("Withdrawal input index out of bounds {i}"))?,
384            T::Location::PureInput(i) => self
385                .pure_inputs
386                .get(i as usize)
387                .ok_or_else(|| anyhow::anyhow!("Pure input index out of bounds {i}"))?,
388            T::Location::ReceivingInput(i) => self
389                .receiving_inputs
390                .get(i as usize)
391                .ok_or_else(|| anyhow::anyhow!("Receiving input index out of bounds {i}"))?,
392            T::Location::Result(i, j) => self
393                .results
394                .get(i as usize)
395                .and_then(|r| r.get(j as usize))
396                .ok_or_else(|| anyhow::anyhow!("Result index out of bounds ({i},{j})"))?,
397        })
398    }

408            T::Location::WithdrawalInput(i) => self
409                .withdrawal_inputs
410                .get_mut(i as usize)
411                .ok_or_else(|| anyhow::anyhow!("Withdrawal input index out of bounds {i}"))?,
412            T::Location::PureInput(i) => self
413                .pure_inputs
414                .get_mut(i as usize)
415                .ok_or_else(|| anyhow::anyhow!("Pure input index out of bounds {i}"))?,
416            T::Location::ReceivingInput(i) => self
417                .receiving_inputs
418                .get_mut(i as usize)
419                .ok_or_else(|| anyhow::anyhow!("Receiving input index out of bounds {i}"))?,
420            T::Location::Result(i, j) => self
421                .results
422                .get_mut(i as usize)
423                .and_then(|r| r.get_mut(j as usize))
424                .ok_or_else(|| anyhow::anyhow!("Result index out of bounds ({i},{j})"))?,
425        })

428    fn check_usage(&self, usage: &T::Usage, location: &Location) -> anyhow::Result<()> {
429        // by marking "ignore alias" as `false`, we will also check for `Alias` paths, i.e. paths
430        // that point to the location itself without any extensions.
431        let is_borrowed = self.any_extends(&location.self_path, /* ignore alias */ false)
432            || self.arg_roots.contains(&usage.location());
433        match usage {
434            T::Usage::Move(_) => {
435                anyhow::ensure!(!is_borrowed, "Cannot move a value that is borrowed");
436            }
437            T::Usage::Copy { borrowed, .. } => {
438                let Some(borrowed) = borrowed.get().copied() else {
439                    anyhow::bail!("Borrowed flag not set for copy usage");
440                };
441                anyhow::ensure!(
442                    borrowed == is_borrowed,
443                    "Borrowed flag mismatch for copy usage: expected {borrowed}, got {is_borrowed} \
444                    location {:?} for in command {}",
445                    location.self_path,
446                    self.current_command()?
447                );
448            }
449        }
450        Ok(())
451    }
452
453    fn argument(&mut self, sp!(_, (arg, _)): &T::Argument) -> anyhow::Result<Value> {
454        let location = self.location(arg.location())?;
455        match arg {
456            T::Argument__::Use(usage)
457            | T::Argument__::Freeze(usage)
458            | T::Argument__::Read(usage) => self.check_usage(usage, location)?,
459            T::Argument__::Borrow(_, _) => (),
460        };
461        let location = self.location_mut(arg.location())?;
462        let value = match arg {
463            T::Argument__::Use(usage) => location.use_(usage)?,
464            T::Argument__::Freeze(usage) => location.use_(usage)?.freeze()?,
465            T::Argument__::Borrow(is_mut, _) => location.borrow(*is_mut)?,
466            T::Argument__::Read(usage) => {
467                location.use_(usage)?;
468                Value::NonRef
469            }
470        };
471        if let Value::Ref { paths, .. } = &value {
472            for p in &paths.0 {
473                match p.root {
474                    RootLocation::Unknown { .. } => (),
475                    RootLocation::Known(location) => {
476                        self.arg_roots.insert(location);
477                    }
478                }
479            }
480        }
481        Ok(value)
482    }

484    fn arguments(&mut self, args: &[T::Argument]) -> anyhow::Result<Vec<Value>> {
485        args.iter()
486            .map(|arg| self.argument(arg))
487            .collect::<anyhow::Result<Vec<_>>>()
488    }

490    fn all_references(&self) -> impl Iterator<Item = Rc<PathSet>> {
491        let Self {
492            tx_context,
493            gas,
494            object_inputs,
495            withdrawal_inputs,
496            pure_inputs,
497            receiving_inputs,
498            results,
499            arg_roots: _,
500        } = self;
501        std::iter::once(tx_context)
502            .chain(std::iter::once(gas))
503            .chain(object_inputs)
504            .chain(withdrawal_inputs)
505            .chain(pure_inputs)
506            .chain(receiving_inputs)
507            .chain(results.iter().flatten())
508            .filter_map(|v| -> Option<Rc<PathSet>> {
509                match v.value.as_ref() {
510                    Some(Value::Ref { paths, .. }) => Some(paths.clone()),
511                    Some(Value::NonRef) | None => None,
512                }
513            })
514    }

516    /// Returns true if any of the references in a given `T::Location` extends a path in `paths`.
517    /// Excludes `Aliases` if `ignore_aliases` is true.
518    fn any_extends(&self, paths: &PathSet, ignore_aliases: bool) -> bool {
519        self.all_references()
520            .any(|other| other.extends(paths, ignore_aliases))
521    }

524/// Verifies memory safety of a transaction. This is a re-implementation of `verify::memory_safety`
525/// using an alternative approach given the newness of the Regex based borrow graph in that
526/// implementation.
527/// This is a set based approach were each reference is represent as a set of paths. A path
528/// is has a root (basically a `T::Location` plus some edge case massaging) and a list of extensions
529/// resulting from Move function calls. Each one of those Move calls gets a `Delta` extension for
530/// each return value. The `Delta` is like the ".*" in the regex based implementation but where it
531/// carries a sense of identity. This identity allows for invariants from the return values of the
532/// Move call to be leveraged. For example, mutable references returned from a call cannot overlap.
533/// If we just used ".*", we would not be able to express this invariant without some sense of
534/// identity for the reference itself (which is what is going on in the Regex based implementation).
535/// This implementation stems from research work for the Move borrow checker, but would normally
536/// not be expressive enough in the presence of control flow. Luckily, PTBs do not have control flow
537/// so we can use this approach as a safety net for the Regex based implementation until that
538/// code is sufficiently. tested and hardened.
539/// Checks the following
540/// - Values are not used after being moved
541/// - Reference safety is upheld (no dangling references)
542pub fn verify(_env: &Env, txn: &T::Transaction) -> Result<(), ExecutionError> {
543    verify_(txn).map_err(|e| make_invariant_violation!("{}. Transaction {:?}", e, txn))
544}
545
546fn verify_(txn: &T::Transaction) -> anyhow::Result<()> {
547    let mut context = Context::new(txn)?;
548    let T::Transaction {
549        bytes: _,
550        objects: _,
551        withdrawals: _,
552        pure: _,
553        receiving: _,
554        withdrawal_compatibility_conversions: _,
555        commands,
556    } = txn;
557    for c in commands {
558        command(&mut context, c)?;
559    }
560    Ok(())
561}
562
563fn command(context: &mut Context, c: &T::Command) -> anyhow::Result<()> {
564    // process the command
565    debug_assert!(context.arg_roots.is_empty());
566    let results = command_(context, c)?;
567    // drop unused result values by marking them as `None`
568    assert_invariant!(
569        results.len() == c.value.drop_values.len(),
570        "result length mismatch. expected {}, got {}",
571        c.value.drop_values.len(),
572        results.len()
573    );
574    context.add_result_values(
575        results
576            .into_iter()
577            .zip(c.value.drop_values.iter().copied())
578            .map(|(v, drop)| if drop { None } else { Some(v) }),
579    )?;
580    context.arg_roots.clear();
581    Ok(())
582}
583
584fn command_(context: &mut Context, sp!(_, c): &T::Command) -> anyhow::Result<Vec<Value>> {
585    let result_tys = &c.result_type;
586    let results = match &c.command {
587        T::Command__::MoveCall(move_call) => {
588            let T::MoveCall {
589                function,
590                arguments,
591            } = &**move_call;
592            let arg_values = context.arguments(arguments)?;
593            call(context, &function.signature, arg_values)?
594        }
595        T::Command__::TransferObjects(objs, recipient) => {
596            context.arguments(objs)?;
597            context.argument(recipient)?;
598            non_ref_results(result_tys)?
599        }
600        T::Command__::SplitCoins(_, coin, amounts) => {
601            context.arguments(amounts)?;
602            let coin_value = context.argument(coin)?;
603            write_ref(context, coin_value)?;
604            non_ref_results(result_tys)?
605        }
606        T::Command__::MergeCoins(_, target, coins) => {
607            context.arguments(coins)?;
608            let target_value = context.argument(target)?;
609            write_ref(context, target_value)?;
610            non_ref_results(result_tys)?
611        }
612        T::Command__::MakeMoveVec(_, arguments) => {
613            context.arguments(arguments)?;
614            non_ref_results(result_tys)?
615        }
616        T::Command__::Publish(_, _, _) => non_ref_results(result_tys)?,
617        T::Command__::Upgrade(_, _, _, ticket, _) => {
618            context.argument(ticket)?;
619            non_ref_results(result_tys)?
620        }
621    };
622    assert_invariant!(
623        result_tys.len() == results.len(),
624        "result length mismatch. Expected {}, got {}",
625        result_tys.len(),
626        results.len()
627    );
628    Ok(results)
629}

631fn write_ref(context: &Context, value: Value) -> anyhow::Result<()> {
632    match value {
633        Value::NonRef => {
634            anyhow::bail!("Cannot write to a non-reference value");
635        }
636
637        Value::Ref { is_mut: false, .. } => {
638            anyhow::bail!("Cannot write to an immutable reference");
639        }
640        Value::Ref {
641            is_mut: true,
642            paths,
643        } => {
644            anyhow::ensure!(
645                !context.any_extends(&paths, /* ignore alias */ true),
646                "Cannot write to a mutable reference that has extensions"
647            );
648            Ok(())
649        }
650    }
651}

653fn call(
654    context: &mut Context,
655    signature: &T::LoadedFunctionInstantiation,
656    arguments: Vec<Value>,
657) -> anyhow::Result<Vec<Value>> {
658    let return_ = &signature.return_;
659    let mut all_paths: PathSet = PathSet::empty();
660    let mut imm_paths: PathSet = PathSet::empty();
661    let mut mut_paths: PathSet = PathSet::empty();
662    for arg in arguments {
663        match arg {
664            Value::NonRef => (),
665            Value::Ref {
666                is_mut: true,
667                paths,
668            } => {
669                // Allow alias conflicts with references not passed as arguments
670                anyhow::ensure!(
671                    !context.any_extends(&paths, /* ignore alias */ true),
672                    "Cannot transfer a mutable ref with extensions"
673                );
674                // All mutable argument references must be disjoint from all other references
675                anyhow::ensure!(mut_paths.is_disjoint(&paths), "Double mutable borrow");
676                all_paths.union(&paths);
677                mut_paths.union(&paths);
678            }
679            Value::Ref {
680                is_mut: false,
681                paths,
682            } => {
683                all_paths.union(&paths);
684                imm_paths.union(&paths);
685            }
686        }
687    }
688    // All mutable references must be disjoint from all immutable references
689    anyhow::ensure!(
690        imm_paths.is_disjoint(&mut_paths),
691        "Mutable and immutable borrows cannot overlap"
692    );
693    let command = context.current_command()?;
694    let mut_paths = if mut_paths.is_empty() {
695        PathSet::unknown_root(command)
696    } else {
697        mut_paths
698    };
699    let all_paths = if all_paths.is_empty() {
700        PathSet::unknown_root(command)
701    } else {
702        all_paths
703    };
704    return_
705        .iter()
706        .enumerate()
707        .map(|(i, ty)| {
708            let delta = Delta {
709                command,
710                result: checked_as!(i, u16)?,
711            };
712            match ty {
713                T::Type::Reference(/* is mut */ true, _) => {
714                    Value::ref_(true, mut_paths.extend(delta))
715                }
716                T::Type::Reference(/* is mut */ false, _) => {
717                    Value::ref_(false, all_paths.extend(delta))
718                }
719                _ => Ok(Value::NonRef),
720            }
721        })
722        .collect::<anyhow::Result<Vec<_>>>()
723}
724
725fn non_ref_results(results: &[T::Type]) -> anyhow::Result<Vec<Value>> {
726    results
727        .iter()
728        .map(|t| {
729            anyhow::ensure!(
730                !matches!(t, T::Type::Reference(_, _)),
731                "attempted to create a non-reference result from a reference type",
732            );
733            Ok(Value::NonRef)
734        })
735        .collect()
736}

742impl Path {
743    #[cfg(debug_assertions)]
744    #[allow(unused)]
745    fn print(&self) {
746        print!("{:?}", self.root);
747        for ext in &self.extensions {
748            let Delta { command, result } = ext;
749            print!(".d{}_{}", command, result);
750        }
751        println!(",");
752    }
753}
754
755impl PathSet {
756    #[cfg(debug_assertions)]
757    #[allow(unused)]
758    fn print(&self) {
759        println!("{{");
760        for path in &self.0 {
761            path.print();
762        }
763        println!("}}");
764    }
765}
766
767impl Value {
768    #[cfg(debug_assertions)]
769    #[allow(unused)]
770    fn print(&self) {
771        match self {
772            Value::NonRef => print!("NonRef"),
773            Value::Ref { is_mut, paths } => {
774                if *is_mut {
775                    print!("mut ");
776                } else {
777                    print!("imm ");
778                }
779                paths.print();
780            }
781        }
782    }
783}

785impl Location {

787    #[allow(unused)]

789        print!("{{ self_path: ");

791        print!(", value: ");
792        if let Some(value) = &self.value {
793            value.print();
794        } else {
795            println!("_");
796        }
797        println!("}}");
798    }
799}
