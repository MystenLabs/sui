17struct Context<'txn> {
18    objects: Vec<&'txn T::Type>,
19    pure: Vec<&'txn T::Type>,
20    receiving: Vec<&'txn T::Type>,
21    result_types: Vec<&'txn [T::Type]>,
22}
23
24impl<'txn> Context<'txn> {
25    fn new(txn: &'txn T::Transaction) -> Self {
26        Self {
27            objects: txn.objects.iter().map(|o| &o.ty).collect(),
28            pure: txn.pure.iter().map(|p| &p.ty).collect(),
29            receiving: txn.receiving.iter().map(|r| &r.ty).collect(),
30            result_types: txn
31                .commands
32                .iter()
33                .map(|sp!(_, c)| c.result_type.as_slice())
34                .collect(),
35        }
36    }
37}

39/// Verifies the correctness of the typing on the AST
40/// - All object inputs have key
41/// - All pure inputs are valid types
42/// - All receiving inputs types have key
43/// - All commands are well formed with correct argument/result types
44/// - All dropped result values have the `drop` ability
45pub fn verify<Mode: ExecutionMode>(env: &Env, txn: &T::Transaction) -> Result<(), ExecutionError> {
46    verify_::<Mode>(env, txn).map_err(|e| make_invariant_violation!("{}. Transaction {:?}", e, txn))
47}
48
49fn verify_<Mode: ExecutionMode>(env: &Env, txn: &T::Transaction) -> anyhow::Result<()> {
50    let context = Context::new(txn);
51    let T::Transaction {
52        bytes: _,
53        objects,
54        pure,
55        receiving,
56        commands,
57    } = txn;
58    for obj in objects {
59        object_input(obj)?;
60    }
61    for p in pure {
62        pure_input::<Mode>(p)?;
63    }
64    for r in receiving {
65        receiving_input(r)?;
66    }
67    for c in commands {
68        command::<Mode>(env, &context, c)?;
69    }
70    Ok(())
71}
72
73fn object_input(obj: &T::ObjectInput) -> anyhow::Result<()> {
74    anyhow::ensure!(obj.ty.abilities().has_key(), "object type must have key");
75    Ok(())
76}
77
78fn pure_input<Mode: ExecutionMode>(p: &T::PureInput) -> anyhow::Result<()> {
79    if !Mode::allow_arbitrary_values() {
80        anyhow::ensure!(
81            input_arguments::is_valid_pure_type(&p.ty)?,
82            "pure type must be valid"
83        );
84    }
85    Ok(())
86}
87
88fn receiving_input(r: &T::ReceivingInput) -> anyhow::Result<()> {
89    anyhow::ensure!(
90        input_arguments::is_valid_receiving(&r.ty),
91        "receiving type must be valid"
92    );
93    Ok(())
94}

96fn command<Mode: ExecutionMode>(
97    env: &Env,
98    context: &Context,
99    sp!(_, c): &T::Command,
100) -> anyhow::Result<()> {
101    let result_tys = &c.result_type;
102    match &c.command {
103        T::Command__::MoveCall(move_call) => {
104            let T::MoveCall {
105                function,
106                arguments,
107            } = &**move_call;
108            let L::LoadedFunction { signature, .. } = function;
109            let L::LoadedFunctionInstantiation {
110                parameters,
111                return_,
112            } = signature;
113            anyhow::ensure!(
114                arguments.len() == parameters.len(),
115                "arity mismatch. Expected {}, got {}",
116                parameters.len(),
117                arguments.len()
118            );
119            for (arg, param) in arguments.iter().zip(parameters) {
120                argument(env, context, arg, param)?;
121            }
122            anyhow::ensure!(
123                return_.len() == result_tys.len(),
124                "result arity mismatch. Expected {}, got {}",
125                return_.len(),
126                result_tys.len()
127            );
128            for (actual, expected) in return_.iter().zip(result_tys) {
129                anyhow::ensure!(
130                    actual == expected,
131                    "return type mismatch. Expected {expected:?}, got {actual:?}"
132                );
133            }
134        }
135        T::Command__::TransferObjects(objs, recipient) => {
136            for obj in objs {
137                let ty = &obj.value.1;
138                anyhow::ensure!(
139                    ty.abilities().has_key(),
140                    "transfer object type must have key, got {ty:?}"
141                );
142                argument(env, context, obj, ty)?;
143            }
144            argument(env, context, recipient, &T::Type::Address)?;
145            anyhow::ensure!(
146                result_tys.is_empty(),
147                "transfer objects should not return any value, got {result_tys:?}"
148            );
149        }
150        T::Command__::SplitCoins(ty_coin, coin, amounts) => {
151            let T::Type::Datatype(dt) = ty_coin else {
152                anyhow::bail!("split coins should have a coin type, got {ty_coin:?}");
153            };
154            let resolved = dt.qualified_ident();
155            anyhow::ensure!(
156                resolved == RESOLVED_COIN_STRUCT,
157                "split coins should have a coin type, got {resolved:?}"
158            );
159            argument(
160                env,
161                context,
162                coin,
163                &T::Type::Reference(true, Rc::new(ty_coin.clone())),
164            )?;
165            for amount in amounts {
166                argument(env, context, amount, &T::Type::U64)?;
167            }
168            anyhow::ensure!(
169                amounts.len() == result_tys.len(),
170                "split coins should return as many values as amounts, expected {} got {}",
171                amounts.len(),
172                result_tys.len()
173            );
174            anyhow::ensure!(
175                result_tys.iter().all(|t| t == ty_coin),
176                "split coins should return coin<{ty_coin:?}>, got {result_tys:?}"
177            );
178        }
179        T::Command__::MergeCoins(ty_coin, target, coins) => {
180            let T::Type::Datatype(dt) = ty_coin else {
181                anyhow::bail!("split coins should have a coin type, got {ty_coin:?}");
182            };
183            let resolved = dt.qualified_ident();
184            anyhow::ensure!(
185                resolved == RESOLVED_COIN_STRUCT,
186                "split coins should have a coin type, got {resolved:?}"
187            );
188            argument(
189                env,
190                context,
191                target,
192                &T::Type::Reference(true, Rc::new(ty_coin.clone())),
193            )?;
194            for coin in coins {
195                argument(env, context, coin, ty_coin)?;
196            }
197            anyhow::ensure!(
198                result_tys.is_empty(),
199                "merge coins should not return any value, got {result_tys:?}"
200            );
201        }
202        T::Command__::MakeMoveVec(t, args) => {
203            for arg in args {
204                argument(env, context, arg, t)?;
205            }
206            anyhow::ensure!(
207                result_tys.len() == 1,
208                "make move vec should return exactly one vector"
209            );
210            let T::Type::Vector(inner) = &result_tys[0] else {
211                anyhow::bail!("make move vec should return a vector type, got {result_tys:?}");
212            };
213            anyhow::ensure!(
214                t == &inner.element_type,
215                "make move vec should return vector<{t:?}>, got {result_tys:?}"
216            );
217        }
218        T::Command__::Publish(_, _, _) => {
219            if Mode::packages_are_predefined() {
220                anyhow::ensure!(
221                    result_tys.is_empty(),
222                    "publish should not return upgrade cap for predefined packages"
223                );
224            } else {
225                anyhow::ensure!(
226                    result_tys.len() == 1,
227                    "publish should return exactly one upgrade cap"
228                );
229                let cap = &env.upgrade_cap_type()?;
230                anyhow::ensure!(
231                    cap == &result_tys[0],
232                    "publish should return {cap:?}, got {result_tys:?}",
233                );
234            }
235        }
236        T::Command__::Upgrade(_, _, _, arg, _) => {
237            argument(env, context, arg, &env.upgrade_ticket_type()?)?;
238            let receipt = &env.upgrade_receipt_type()?;
239            anyhow::ensure!(
240                result_tys.len() == 1,
241                "upgrade should return exactly one receipt"
242            );
243            anyhow::ensure!(
244                receipt == &result_tys[0],
245                "upgrade should return {receipt:?}, got {result_tys:?}"
246            );
247        }
248    }
249    assert_invariant!(
250        c.drop_values.len() == result_tys.len(),
251        "drop values should match result types, expected {} got {}",
252        c.drop_values.len(),
253        result_tys.len()
254    );
255    for (drop_value, result_ty) in c.drop_values.iter().copied().zip(result_tys) {
256        // drop value ==> `ty: drop`
257        assert_invariant!(
258            !drop_value || result_ty.abilities().has_drop(),
259            "result was marked for drop but does not have the `drop` ability"
260        );
261    }
262    Ok(())
263}

265fn argument(
266    env: &Env,
267    context: &Context,
268    sp!(_, (arg__, ty)): &T::Argument,
269    param: &T::Type,
270) -> anyhow::Result<()> {
271    anyhow::ensure!(
272        ty == param,
273        "argument type mismatch. Expected {param:?}, got {ty:?}"
274    );
275    let (actual, expected) = match arg__ {
276        T::Argument__::Use(u) => (usage(env, context, u)?, param),
277        T::Argument__::Read(u) => {
278            let actual = match usage(env, context, u)? {
279                T::Type::Reference(_, inner) => (*inner).clone(),
280                _ => {
281                    anyhow::bail!("should never ReadRef a non-reference type, got {ty:?}");
282                }
283            };
284            (actual, param)
285        }
286        T::Argument__::Freeze(u) => {
287            let actual = match usage(env, context, u)? {
288                T::Type::Reference(true, inner) => T::Type::Reference(false, inner.clone()),
289                T::Type::Reference(false, _) => {
290                    anyhow::bail!("should never FreezeRef an immutable reference")
291                }
292                ty => {
293                    anyhow::bail!("should never Freeze a non-reference type, got {ty:?}");
294                }
295            };
296            (actual, param)
297        }
298        T::Argument__::Borrow(is_mut, l) => {
299            let T::Type::Reference(param_mut, expected) = param else {
300                anyhow::bail!("expected a reference type for borrowed location, got {param:?}");
301            };
302            anyhow::ensure!(
303                *param_mut == *is_mut,
304                "borrowed location mutability mismatch. Expected {param_mut}, got {is_mut}"
305            );
306            let actual = location(env, context, *l)?;
307            (actual, &**expected)
308        }
309    };
310    // check actual == expected
311    anyhow::ensure!(
312        &actual == expected,
313        "argument type mismatch. Expected {expected:?}, got {actual:?}"
314    );
315    // check copy usage
316    match arg__ {
317        T::Argument__::Use(T::Usage::Copy { .. }) | T::Argument__::Read(_) => {
318            anyhow::ensure!(
319                param.abilities().has_copy(),
320                "expected type does not have copy, {expected:?}"
321            );
322        }
323        T::Argument__::Use(T::Usage::Move(_))
324        | T::Argument__::Freeze(_)
325        | T::Argument__::Borrow(_, _) => (),
326    }
327    Ok(())
328}

330fn usage(env: &Env, context: &Context, u: &T::Usage) -> anyhow::Result<T::Type> {
331    match u {
332        T::Usage::Move(l)
333        | T::Usage::Copy {
334            location: l,
335            borrowed: _,
336        } => location(env, context, *l),
337    }
338}
339
340fn location(env: &Env, context: &Context, l: T::Location) -> anyhow::Result<T::Type> {
341    Ok(match l {
342        T::Location::TxContext => env.tx_context_type()?,
343        T::Location::GasCoin => env.gas_coin_type()?,
344        T::Location::ObjectInput(i) => context
345            .objects
346            .get(i as usize)
347            .copied()
348            .ok_or_else(|| anyhow::anyhow!("object input {i} out of bounds"))?
349            .clone(),
350        T::Location::PureInput(i) => context
351            .pure
352            .get(i as usize)
353            .copied()
354            .ok_or_else(|| anyhow::anyhow!("pure input {i} out of bounds"))?
355            .clone(),
356        T::Location::ReceivingInput(i) => context
357            .receiving
358            .get(i as usize)
359            .copied()
360            .ok_or_else(|| anyhow::anyhow!("receiving input {i} out of bounds"))?
361            .clone(),
362        T::Location::Result(i, j) => context
363            .result_types
364            .get(i as usize)
365            .and_then(|v| v.get(j as usize))
366            .ok_or_else(|| anyhow::anyhow!("result ({i}, {j}) out of bounds",))?
367            .clone(),
368    })
