24struct Context<'txn> {
25    objects: Vec<&'txn T::Type>,
26    withdrawals: Vec<&'txn T::Type>,
27    pure: Vec<&'txn T::Type>,
28    receiving: Vec<&'txn T::Type>,
29    result_types: Vec<&'txn [T::Type]>,
30}
31
32impl<'txn> Context<'txn> {
33    fn new(txn: &'txn T::Transaction) -> Self {
34        Self {
35            objects: txn.objects.iter().map(|o| &o.ty).collect(),
36            withdrawals: txn.withdrawals.iter().map(|w| &w.ty).collect(),
37            pure: txn.pure.iter().map(|p| &p.ty).collect(),
38            receiving: txn.receiving.iter().map(|r| &r.ty).collect(),
39            result_types: txn
40                .commands
41                .iter()
42                .map(|sp!(_, c)| c.result_type.as_slice())
43                .collect(),
44        }
45    }
46}

48/// Verifies the correctness of the typing on the AST
49/// - All object inputs have key
50/// - All pure inputs are valid types
51/// - All receiving inputs types have key
52/// - All commands are well formed with correct argument/result types
53/// - All dropped result values have the `drop` ability
54pub fn verify<Mode: ExecutionMode>(env: &Env, txn: &T::Transaction) -> Result<(), ExecutionError> {
55    verify_::<Mode>(env, txn).map_err(|e| make_invariant_violation!("{}. Transaction {:?}", e, txn))
56}
57
58fn verify_<Mode: ExecutionMode>(env: &Env, txn: &T::Transaction) -> anyhow::Result<()> {
59    let context = Context::new(txn);
60    let T::Transaction {
61        bytes: _,
62        objects,
63        withdrawals,
64        pure,
65        receiving,
66        withdrawal_compatibility_conversions,
67        commands,
68    } = txn;
69    for obj in objects {
70        object_input(obj)?;
71    }
72    for w in withdrawals {
73        withdrawal_input(&w.ty)?;
74    }
75    for p in pure {
76        pure_input::<Mode>(p)?;
77    }
78    for r in receiving {
79        receiving_input(r)?;
80    }
81    for c in commands {
82        command::<Mode>(env, &context, c)?;
83    }
84    for (withdrawal, conversion) in withdrawal_compatibility_conversions {
85        withdrawal_compatibility_conversion(env, &context, *withdrawal, conversion)?;
86    }
87    Ok(())
88}

90fn object_input(obj: &T::ObjectInput) -> anyhow::Result<()> {
91    anyhow::ensure!(obj.ty.abilities().has_key(), "object type must have key");
92    Ok(())
93}
94
95fn withdrawal_input(ty: &T::Type) -> anyhow::Result<()> {
96    anyhow::ensure!(ty.abilities().has_drop(), "withdrawal type must have drop");
97    let T::Type::Datatype(dt) = ty else {
98        anyhow::bail!("withdrawal input must be a datatype, got {ty:?}");
99    };
100    anyhow::ensure!(
101        dt.type_arguments.len() == 1,
102        "withdrawal input must have exactly one type argument, got {}",
103        dt.type_arguments.len()
104    );
105    anyhow::ensure!(
106        dt.qualified_ident() == RESOLVED_WITHDRAWAL_STRUCT,
107        "withdrawal input must be sui::funds_accumulator::Withdrawal, got {:?}",
108        dt.qualified_ident()
109    );
110    Ok(())
111}

113fn pure_input<Mode: ExecutionMode>(p: &T::PureInput) -> anyhow::Result<()> {
114    if !Mode::allow_arbitrary_values() {
115        anyhow::ensure!(
116            input_arguments::is_valid_pure_type(&p.ty)?,
117            "pure type must be valid"
118        );
119    }
120    Ok(())
121}
122
123fn receiving_input(r: &T::ReceivingInput) -> anyhow::Result<()> {
124    anyhow::ensure!(
125        input_arguments::is_valid_receiving(&r.ty),
126        "receiving type must be valid"
127    );
128    Ok(())
129}

131fn command<Mode: ExecutionMode>(
132    env: &Env,
133    context: &Context,
134    sp!(_, c): &T::Command,
135) -> anyhow::Result<()> {
136    let result_tys = &c.result_type;
137    match &c.command {
138        T::Command__::MoveCall(move_call) => {
139            let T::MoveCall {
140                function,
141                arguments,
142            } = &**move_call;
143            let L::LoadedFunction { signature, .. } = function;
144            let L::LoadedFunctionInstantiation {
145                parameters,
146                return_,
147            } = signature;
148            anyhow::ensure!(
149                arguments.len() == parameters.len(),
150                "arity mismatch. Expected {}, got {}",
151                parameters.len(),
152                arguments.len()
153            );
154            for (arg, param) in arguments.iter().zip(parameters) {
155                argument(env, context, arg, param)?;
156            }
157            anyhow::ensure!(
158                return_.len() == result_tys.len(),
159                "result arity mismatch. Expected {}, got {}",
160                return_.len(),
161                result_tys.len()
162            );
163            for (actual, expected) in return_.iter().zip(result_tys) {
164                anyhow::ensure!(
165                    actual == expected,
166                    "return type mismatch. Expected {expected:?}, got {actual:?}"
167                );
168            }
169        }
170        T::Command__::TransferObjects(objs, recipient) => {
171            for obj in objs {
172                let ty = &obj.value.1;
173                anyhow::ensure!(
174                    ty.abilities().has_key(),
175                    "transfer object type must have key, got {ty:?}"
176                );
177                argument(env, context, obj, ty)?;
178            }
179            argument(env, context, recipient, &T::Type::Address)?;
180            anyhow::ensure!(
181                result_tys.is_empty(),
182                "transfer objects should not return any value, got {result_tys:?}"
183            );
184        }
185        T::Command__::SplitCoins(ty_coin, coin, amounts) => {
186            let T::Type::Datatype(dt) = ty_coin else {
187                anyhow::bail!("split coins should have a coin type, got {ty_coin:?}");
188            };
189            let resolved = dt.qualified_ident();
190            anyhow::ensure!(
191                resolved == RESOLVED_COIN_STRUCT,
192                "split coins should have a coin type, got {resolved:?}"
193            );
194            argument(
195                env,
196                context,
197                coin,
198                &T::Type::Reference(true, Rc::new(ty_coin.clone())),
199            )?;
200            for amount in amounts {
201                argument(env, context, amount, &T::Type::U64)?;
202            }
203            anyhow::ensure!(
204                amounts.len() == result_tys.len(),
205                "split coins should return as many values as amounts, expected {} got {}",
206                amounts.len(),
207                result_tys.len()
208            );
209            anyhow::ensure!(
210                result_tys.iter().all(|t| t == ty_coin),
211                "split coins should return coin<{ty_coin:?}>, got {result_tys:?}"
212            );
213        }
214        T::Command__::MergeCoins(ty_coin, target, coins) => {
215            let T::Type::Datatype(dt) = ty_coin else {
216                anyhow::bail!("split coins should have a coin type, got {ty_coin:?}");
217            };
218            let resolved = dt.qualified_ident();
219            anyhow::ensure!(
220                resolved == RESOLVED_COIN_STRUCT,
221                "split coins should have a coin type, got {resolved:?}"
222            );
223            argument(
224                env,
225                context,
226                target,
227                &T::Type::Reference(true, Rc::new(ty_coin.clone())),
228            )?;
229            for coin in coins {
230                argument(env, context, coin, ty_coin)?;
231            }
232            anyhow::ensure!(
233                result_tys.is_empty(),
234                "merge coins should not return any value, got {result_tys:?}"
235            );
236        }
237        T::Command__::MakeMoveVec(t, args) => {
238            for arg in args {
239                argument(env, context, arg, t)?;
240            }
241            anyhow::ensure!(
242                result_tys.len() == 1,
243                "make move vec should return exactly one vector"
244            );
245            let T::Type::Vector(inner) = result_tys.first().unwrap() else {
246                anyhow::bail!("make move vec should return a vector type, got {result_tys:?}");
247            };
248            anyhow::ensure!(
249                t == &inner.element_type,
250                "make move vec should return vector<{t:?}>, got {result_tys:?}"
251            );
252        }
253        T::Command__::Publish(_, _, _) => {
254            if Mode::packages_are_predefined() {
255                anyhow::ensure!(
256                    result_tys.is_empty(),
257                    "publish should not return upgrade cap for predefined packages"
258                );
259            } else {
260                anyhow::ensure!(
261                    result_tys.len() == 1,
262                    "publish should return exactly one upgrade cap"
263                );
264                let cap = &env.upgrade_cap_type()?;
265                anyhow::ensure!(
266                    cap == result_tys.first().unwrap(),
267                    "publish should return {cap:?}, got {result_tys:?}",
268                );
269            }
270        }
271        T::Command__::Upgrade(_, _, _, arg, _) => {
272            argument(env, context, arg, &env.upgrade_ticket_type()?)?;
273            let receipt = &env.upgrade_receipt_type()?;
274            anyhow::ensure!(
275                result_tys.len() == 1,
276                "upgrade should return exactly one receipt"
277            );
278            anyhow::ensure!(
279                receipt == result_tys.first().unwrap(),
280                "upgrade should return {receipt:?}, got {result_tys:?}"
281            );
282        }
283    }
284    assert_invariant!(
285        c.drop_values.len() == result_tys.len(),
286        "drop values should match result types, expected {} got {}",
287        c.drop_values.len(),
288        result_tys.len()
289    );
290    for (drop_value, result_ty) in c.drop_values.iter().copied().zip(result_tys) {
291        // drop value ==> `ty: drop`
292        assert_invariant!(
293            !drop_value || result_ty.abilities().has_drop(),
294            "result was marked for drop but does not have the `drop` ability"
295        );
296    }
297    Ok(())
298}

300fn argument(
301    env: &Env,
302    context: &Context,
303    sp!(_, (arg__, ty)): &T::Argument,
304    param: &T::Type,
305) -> anyhow::Result<()> {
306    anyhow::ensure!(
307        ty == param,
308        "argument type mismatch. Expected {param:?}, got {ty:?}"
309    );
310    let (actual, expected) = match arg__ {
311        T::Argument__::Use(u) => (usage(env, context, u)?, param),
312        T::Argument__::Read(u) => {
313            let actual = match usage(env, context, u)? {
314                T::Type::Reference(_, inner) => (*inner).clone(),
315                _ => {
316                    anyhow::bail!("should never ReadRef a non-reference type, got {ty:?}");
317                }
318            };
319            (actual, param)
320        }
321        T::Argument__::Freeze(u) => {
322            let actual = match usage(env, context, u)? {
323                T::Type::Reference(true, inner) => T::Type::Reference(false, inner.clone()),
324                T::Type::Reference(false, _) => {
325                    anyhow::bail!("should never FreezeRef an immutable reference")
326                }
327                ty => {
328                    anyhow::bail!("should never Freeze a non-reference type, got {ty:?}");
329                }
330            };
331            (actual, param)
332        }
333        T::Argument__::Borrow(is_mut, l) => {
334            let T::Type::Reference(param_mut, expected) = param else {
335                anyhow::bail!("expected a reference type for borrowed location, got {param:?}");
336            };
337            anyhow::ensure!(
338                *param_mut == *is_mut,
339                "borrowed location mutability mismatch. Expected {param_mut}, got {is_mut}"
340            );
341            let actual = location(env, context, *l)?;
342            (actual, &**expected)
343        }
344    };
345    // check actual == expected
346    anyhow::ensure!(
347        &actual == expected,
348        "argument type mismatch. Expected {expected:?}, got {actual:?}"
349    );
350    // check copy usage
351    match arg__ {
352        T::Argument__::Use(T::Usage::Copy { .. }) | T::Argument__::Read(_) => {
353            anyhow::ensure!(
354                param.abilities().has_copy(),
355                "expected type does not have copy, {expected:?}"
356            );
357        }
358        T::Argument__::Use(T::Usage::Move(_))
359        | T::Argument__::Freeze(_)
360        | T::Argument__::Borrow(_, _) => (),
361    }
362    Ok(())
363}

365fn usage(env: &Env, context: &Context, u: &T::Usage) -> anyhow::Result<T::Type> {
366    match u {
367        T::Usage::Move(l)
368        | T::Usage::Copy {
369            location: l,
370            borrowed: _,
371        } => location(env, context, *l),
372    }
373}
374
375fn location(env: &Env, context: &Context, l: T::Location) -> anyhow::Result<T::Type> {
376    Ok(match l {
377        T::Location::TxContext => env.tx_context_type()?,
378        T::Location::GasCoin => env.gas_coin_type()?,
379        T::Location::ObjectInput(i) => context
380            .objects
381            .get(i as usize)
382            .copied()
383            .ok_or_else(|| anyhow::anyhow!("object input {i} out of bounds"))?
384            .clone(),
385        T::Location::WithdrawalInput(i) => context
386            .withdrawals
387            .get(i as usize)
388            .copied()
389            .ok_or_else(|| anyhow::anyhow!("withdrawal input {i} out of bounds"))?
390            .clone(),
391        T::Location::PureInput(i) => context
392            .pure
393            .get(i as usize)
394            .copied()
395            .ok_or_else(|| anyhow::anyhow!("pure input {i} out of bounds"))?
396            .clone(),
397        T::Location::ReceivingInput(i) => context
398            .receiving
399            .get(i as usize)
400            .copied()
401            .ok_or_else(|| anyhow::anyhow!("receiving input {i} out of bounds"))?
402            .clone(),
403        T::Location::Result(i, j) => context
404            .result_types
405            .get(i as usize)
406            .and_then(|v| v.get(j as usize))
407            .ok_or_else(|| anyhow::anyhow!("result ({i}, {j}) out of bounds",))?
408            .clone(),
409    })
410}

412fn withdrawal_compatibility_conversion(
413    env: &Env,
414    context: &Context,
415    withdrawal_location: T::Location,
416    conv: &T::WithdrawalCompatibilityConversion,
417) -> anyhow::Result<()> {
418    let T::WithdrawalCompatibilityConversion {
419        owner,
420        conversion_result,
421    } = conv;
422    // checker owner is a pure input of type address
423    anyhow::ensure!(
424        matches!(owner, T::Location::PureInput(_)),
425        "withdrawal compatibility conversion owner should be a pure input"
426    );
427    anyhow::ensure!(
428        location(env, context, *owner)? == T::Type::Address,
429        "withdrawal compatibility conversion owner type should be address"
430    );
431    // check the conversion result type is coin
432    let conversion_location = T::Location::Result(*conversion_result, 0);
433    let conversion_ty = location(env, context, conversion_location)?;
434    let Some(coin_inner) = coin_inner_type(&conversion_ty) else {
435        anyhow::bail!("conversion result should be a coin type");
436    };
437    // check the withdrawal location is a withdrawal input of Withdarawal<Balance<coin_inner>>
438    anyhow::ensure!(
439        matches!(withdrawal_location, T::Location::WithdrawalInput(_)),
440        "withdrawal should be a withdrawal input"
441    );
442    let withdrawal_ty = location(env, context, withdrawal_location)?;
443    let Some(withdrawal_inner_ty) = withdrawal_inner_type(&withdrawal_ty) else {
444        anyhow::bail!("withdrawal input should be a withdrawal type");
445    };
446    let Some(withdrawal_balance_inner) = balance_inner_type(withdrawal_inner_ty) else {
447        anyhow::bail!("withdrawal inner type should be a balance type");
448    };
449    anyhow::ensure!(
450        withdrawal_balance_inner == coin_inner,
451        "withdrawal balance inner type should match conversion coin inner type"
452    );
453    Ok(())
