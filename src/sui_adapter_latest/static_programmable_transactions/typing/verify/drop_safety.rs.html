10/// Refines usage of values so that the last `Copy` of a value is a `Move` if it is not borrowed
11/// After, it verifies the following
12/// - No results without `drop` are unused (all unused non-input values have `drop`)
13pub fn refine_and_verify<Mode: ExecutionMode>(
14    env: &Env,
15    ast: &mut T::Transaction,
16) -> Result<(), ExecutionError> {
17    refine::transaction(env, ast)?;
18    verify::transaction::<Mode>(env, ast)?;
19    Ok(())
20}

41    struct Context {
42        // All locations that were used
43        used: BTreeSet<T::Location>,
44        // All locations that were used via a Move. This is a subset of `used`, but `used` does
45        // not track how each value was used in each instance. For simplicity, this is kept as a
46        // separate set.
47        moved: BTreeSet<T::Location>,
48    }
49
50    impl Context {
51        fn new() -> Self {
52            Self {
53                used: BTreeSet::new(),
54                moved: BTreeSet::new(),
55            }
56        }
57    }

59    /// After memory safety, we can switch the last usage of a `Copy` to a `Move` if it is not
60    /// borrowed at the time of the last usage.
61    pub fn transaction(env: &Env, ast: &mut T::Transaction) -> Result<(), ExecutionError> {
62        let mut context = Context::new();
63        for c in ast.commands.iter_mut().rev() {
64            command(&mut context, c);
65        }
66        return_unused_withdrawal_conversions(env, ast, &context.moved)
67    }
68
69    fn command(context: &mut Context, sp!(_, c): &mut T::Command) {
70        match &mut c.command {
71            T::Command__::MoveCall(mc) => arguments(context, &mut mc.arguments),
72            T::Command__::TransferObjects(objects, recipient) => {
73                argument(context, recipient);
74                arguments(context, objects);
75            }
76            T::Command__::SplitCoins(_, coin, amounts) => {
77                arguments(context, amounts);
78                argument(context, coin);
79            }
80            T::Command__::MergeCoins(_, target, coins) => {
81                arguments(context, coins);
82                argument(context, target);
83            }
84            T::Command__::MakeMoveVec(_, xs) => arguments(context, xs),
85            T::Command__::Publish(_, _, _) => (),
86            T::Command__::Upgrade(_, _, _, x, _) => argument(context, x),
87        }
88    }
89
90    fn arguments(context: &mut Context, args: &mut [T::Argument]) {
91        for arg in args.iter_mut().rev() {
92            argument(context, arg)
93        }
94    }
95
96    fn argument(context: &mut Context, arg: &mut T::Argument) {
97        let usage = match &mut arg.value.0 {
98            T::Argument__::Use(u) | T::Argument__::Read(u) | T::Argument__::Freeze(u) => u,
99            T::Argument__::Borrow(_, loc) => {
100                // mark location as used
101                context.used.insert(*loc);
102                return;
103            }
104        };
105        match &usage {
106            T::Usage::Move(loc) => {
107                // mark location as used
108                context.used.insert(*loc);
109                context.moved.insert(*loc);
110            }
111            T::Usage::Copy { location, borrowed } => {
112                // we are at the last usage of a reference result if it was not yet added to the set
113                let location = *location;
114                let last_usage = context.used.insert(location);
115                if last_usage && !borrowed.get().unwrap() {
116                    // if it was the last usage, we need to change the Copy to a Move
117                    *usage = T::Usage::Move(location);
118                    context.moved.insert(location);
119                }
120            }
121        }
122    }

124    /// For any withdrawal conversion where the value was not moved, send it back to the original
125    /// owner
126    fn return_unused_withdrawal_conversions(
127        env: &Env,
128        ast: &mut T::Transaction,
129        moved_locations: &BTreeSet<T::Location>,
130    ) -> Result<(), ExecutionError> {
131        // withdrawal conversions not empty ==> accumulators enabled
132        assert_invariant!(
133            ast.withdrawal_compatibility_conversions.is_empty()
134                || env.protocol_config.enable_accumulators(),
135            "Withdrawal conversions should be empty if accumulators are not enabled"
136        );
137        for conversion_info in
138            ast.withdrawal_compatibility_conversions
139                .values()
140                .filter(|conversion| {
141                    // The conversion result is always a single return value, as such we can specify
142                    // a secondary index of `0` without worrying about other results.
143                    let conversion_location = T::Location::Result(conversion.conversion_result, 0);
144                    !moved_locations.contains(&conversion_location)
145                })
146        {
147            let Some(cur_command) = ast.commands.len().checked_sub(1) else {
148                invariant_violation!("cannot be zero commands with a conversion")
149            };
150            let cur_command = checked_as!(cur_command, u16)?;
151            let T::WithdrawalCompatibilityConversion {
152                owner,
153                conversion_result,
154            } = *conversion_info;
155            let Some(conversion_command) = ast.commands.get(conversion_result as usize) else {
156                invariant_violation!("conversion result should be a valid command index")
157            };
158            assert_invariant!(
159                conversion_command.value.result_type.len() == 1,
160                "conversion should have one result"
161            );
162            let T::Location::PureInput(owner_pure_idx) = owner else {
163                invariant_violation!("owner should be a pure input")
164            };
165            assert_invariant!(
166                ast.pure.len() > owner_pure_idx as usize,
167                "owner pure input index out of bounds"
168            );
169            assert_invariant!(
170                ast.pure.get(owner_pure_idx as usize).unwrap().ty == T::Type::Address,
171                "owner pure input should be an address"
172            );
173            let Some(conversion_ty) = conversion_command.value.result_type.first() else {
174                invariant_violation!("conversion should have a result type")
175            };
176            let Some(inner_ty) = coin_inner_type(conversion_ty) else {
177                invariant_violation!("conversion result should be a coin type")
178            };
179            let move_result_ = T::Argument__::new_move(T::Location::Result(conversion_result, 0));
180            let move_result = sp(cur_command, (move_result_, conversion_ty.clone()));
181            let owner_ty = Type::Address;
182            let owner_arg_ = T::Argument__::new_move(owner);
183            let owner_arg = sp(cur_command, (owner_arg_, owner_ty));
184            let return_command__ = T::Command__::MoveCall(Box::new(T::MoveCall {
185                function: env.load_framework_function(
186                    COIN_MODULE_NAME,
187                    SEND_FUNDS_FUNC_NAME,
188                    vec![inner_ty.clone()],
189                )?,
190                arguments: vec![move_result, owner_arg],
191            }));
192            let return_command = sp(
193                cur_command,
194                T::Command_ {
195                    command: return_command__,
196                    result_type: vec![],
197                    drop_values: vec![],
198                    consumed_shared_objects: vec![],
199                },
200            );
201            ast.commands.push(return_command);
202        }
203        Ok(())
204    }

218    #[must_use]
219    struct Value;

221    struct Context {
222        tx_context: Option<Value>,
223        gas_coin: Option<Value>,
224        objects: Vec<Option<Value>>,
225        withdrawals: Vec<Option<Value>>,
226        pure: Vec<Option<Value>>,
227        receiving: Vec<Option<Value>>,
228        results: Vec<Vec<Option<Value>>>,
229    }
230
231    impl Context {
232        fn new(ast: &T::Transaction) -> Result<Self, ExecutionError> {
233            let objects = ast.objects.iter().map(|_| Some(Value)).collect::<Vec<_>>();
234            let withdrawals = ast
235                .withdrawals
236                .iter()
237                .map(|_| Some(Value))
238                .collect::<Vec<_>>();
239            let pure = ast.pure.iter().map(|_| Some(Value)).collect::<Vec<_>>();
240            let receiving = ast
241                .receiving
242                .iter()
243                .map(|_| Some(Value))
244                .collect::<Vec<_>>();
245            Ok(Self {
246                tx_context: Some(Value),
247                gas_coin: Some(Value),
248                objects,
249                withdrawals,
250                pure,
251                receiving,
252                results: Vec::with_capacity(ast.commands.len()),
253            })
254        }
255
256        fn location(&mut self, l: T::Location) -> Result<&mut Option<Value>, ExecutionError> {
257            Ok(match l {
258                T::Location::TxContext => &mut self.tx_context,
259                T::Location::GasCoin => &mut self.gas_coin,
260                T::Location::ObjectInput(i) => self.objects.safe_get_mut(i as usize)?,
261                T::Location::WithdrawalInput(i) => self.withdrawals.safe_get_mut(i as usize)?,
262                T::Location::PureInput(i) => self.pure.safe_get_mut(i as usize)?,
263                T::Location::ReceivingInput(i) => self.receiving.safe_get_mut(i as usize)?,
264                T::Location::Result(i, j) => self
265                    .results
266                    .safe_get_mut(i as usize)?
267                    .safe_get_mut(j as usize)?,
268            })
269        }
270    }

272    /// Checks the following
273    /// - All unused result values have `drop`
274    pub fn transaction<Mode: ExecutionMode>(
275        _env: &Env,
276        ast: &T::Transaction,
277    ) -> Result<(), ExecutionError> {
278        let mut context = Context::new(ast)?;
279        let commands = &ast.commands;
280        for c in commands {
281            let result =
282                command(&mut context, c).map_err(|e| e.with_command_index(c.idx as usize))?;
283            assert_invariant!(
284                result.len() == c.value.result_type.len(),
285                "result length mismatch"
286            );
287            // drop unused result values
288            assert_invariant!(
289                result.len() == c.value.drop_values.len(),
290                "drop values length mismatch"
291            );
292            let result_values = result
293                .into_iter()
294                .zip(c.value.drop_values.iter().copied())
295                .map(|(v, drop)| {
296                    if !drop {
297                        Some(v)
298                    } else {
299                        consume_value(v);
300                        None
301                    }
302                })
303                .collect();
304            context.results.push(result_values);
305        }
306
307        let Context {
308            tx_context,
309            gas_coin,
310            objects,
311            withdrawals,
312            pure,
313            receiving,
314            results,
315        } = context;
316        consume_value_opt(gas_coin);
317        // TODO do we want to check inputs in the dev inspect case?
318        consume_value_opts(objects);
319        consume_value_opts(withdrawals);
320        consume_value_opts(pure);
321        consume_value_opts(receiving);
322        assert_invariant!(results.len() == commands.len(), "result length mismatch");
323        for (i, (result, c)) in results.into_iter().zip(&ast.commands).enumerate() {
324            let tys = &c.value.result_type;
325            assert_invariant!(result.len() == tys.len(), "result length mismatch");
326            for (j, (vopt, ty)) in result.into_iter().zip(tys).enumerate() {
327                drop_value_opt::<Mode>((i, j), vopt, ty)?;
328            }
329        }
330        assert_invariant!(tx_context.is_some(), "tx_context should never be moved");
331        Ok(())
332    }

334    fn command(
335        context: &mut Context,
336        sp!(_, c): &T::Command,
337    ) -> Result<Vec<Value>, ExecutionError> {
338        let result_tys = &c.result_type;
339        Ok(match &c.command {
340            T::Command__::MoveCall(mc) => {
341                let T::MoveCall {
342                    function,
343                    arguments: args,
344                } = &**mc;
345                let return_ = &function.signature.return_;
346                let arg_values = arguments(context, args)?;
347                consume_values(arg_values);
348                (0..return_.len()).map(|_| Value).collect()
349            }
350            T::Command__::TransferObjects(objects, recipient) => {
351                let object_values = arguments(context, objects)?;
352                let recipient_value = argument(context, recipient)?;
353                consume_values(object_values);
354                consume_value(recipient_value);
355                vec![]
356            }
357            T::Command__::SplitCoins(_, coin, amounts) => {
358                let coin_value = argument(context, coin)?;
359                let amount_values = arguments(context, amounts)?;
360                consume_values(amount_values);
361                consume_value(coin_value);
362                (0..amounts.len()).map(|_| Value).collect()
363            }
364            T::Command__::MergeCoins(_, target, coins) => {
365                let target_value = argument(context, target)?;
366                let coin_values = arguments(context, coins)?;
367                consume_values(coin_values);
368                consume_value(target_value);
369                vec![]
370            }
371            T::Command__::MakeMoveVec(_, xs) => {
372                let vs = arguments(context, xs)?;
373                consume_values(vs);
374                vec![Value]
375            }
376            T::Command__::Publish(_, _, _) => result_tys.iter().map(|_| Value).collect(),
377            T::Command__::Upgrade(_, _, _, x, _) => {
378                let v = argument(context, x)?;
379                consume_value(v);
380                vec![Value]
381            }
382        })
383    }

385    fn consume_values(_: Vec<Value>) {}

393    fn drop_value_opt<Mode: ExecutionMode>(
394        idx: (usize, usize),
395        value: Option<Value>,
396        ty: &Type,
397    ) -> Result<(), ExecutionError> {
398        match value {
399            Some(v) => drop_value::<Mode>(idx, v, ty),
400            None => Ok(()),
401        }
402    }
403
404    fn drop_value<Mode: ExecutionMode>(
405        (i, j): (usize, usize),
406        value: Value,
407        ty: &Type,
408    ) -> Result<(), ExecutionError> {
409        let abilities = ty.abilities();
410        if !abilities.has_drop() && !Mode::allow_arbitrary_values() {
411            let msg = if abilities.has_copy() {
412                "The value has copy, but not drop. \
413                Its last usage must be by-value so it can be taken."
414            } else {
415                "Unused value without drop"
416            };
417            return Err(ExecutionError::new_with_source(
418                ExecutionErrorKind::UnusedValueWithoutDrop {
419                    result_idx: checked_as!(i, u16)?,
420                    secondary_idx: checked_as!(j, u16)?,
421                },
422                msg,
423            ));
424        }
425        consume_value(value);
426        Ok(())
427    }

429    fn arguments(context: &mut Context, xs: &[T::Argument]) -> Result<Vec<Value>, ExecutionError> {
430        xs.iter().map(|x| argument(context, x)).collect()
431    }
432
433    fn argument(context: &mut Context, sp!(_, x): &T::Argument) -> Result<Value, ExecutionError> {
434        match &x.0 {
435            T::Argument__::Use(T::Usage::Move(location)) => move_value(context, *location),
436            T::Argument__::Use(T::Usage::Copy { location, .. }) => copy_value(context, *location),
437            T::Argument__::Borrow(_, location) => borrow_location(context, *location),
438            T::Argument__::Read(usage) => read_ref(context, usage),
439            T::Argument__::Freeze(usage) => freeze_ref(context, usage),
440        }
441    }
442
443    fn move_value(context: &mut Context, l: T::Location) -> Result<Value, ExecutionError> {
444        let Some(value) = context.location(l)?.take() else {
445            invariant_violation!("memory safety should have failed")
446        };
447        Ok(value)
448    }
449
450    fn copy_value(context: &mut Context, l: T::Location) -> Result<Value, ExecutionError> {
451        assert_invariant!(
452            context.location(l)?.is_some(),
453            "memory safety should have failed"
454        );
455        Ok(Value)
456    }
457
458    fn borrow_location(context: &mut Context, l: T::Location) -> Result<Value, ExecutionError> {
459        assert_invariant!(
460            context.location(l)?.is_some(),
461            "memory safety should have failed"
462        );
463        Ok(Value)
464    }
465
466    fn read_ref(context: &mut Context, u: &T::Usage) -> Result<Value, ExecutionError> {
467        let value = match u {
468            T::Usage::Move(l) => move_value(context, *l)?,
469            T::Usage::Copy { location, .. } => copy_value(context, *location)?,
470        };
471        consume_value(value);
472        Ok(Value)
473    }
474
475    fn freeze_ref(context: &mut Context, u: &T::Usage) -> Result<Value, ExecutionError> {
476        let value = match u {
477            T::Usage::Move(l) => move_value(context, *l)?,
478            T::Usage::Copy { location, .. } => copy_value(context, *location)?,
479        };
480        consume_value(value);
481        Ok(Value)
482    }
