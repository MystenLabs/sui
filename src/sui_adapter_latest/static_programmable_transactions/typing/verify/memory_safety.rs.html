23#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug)]
24struct Location(T::Location);
25
26type Graph = move_regex_borrow_graph::collections::Graph<(), Location>;
27type Paths = move_regex_borrow_graph::collections::Paths<(), Location>;
28
29#[must_use]
30enum Value {
31    Ref(Ref),
32    NonRef,
33}
34
35struct Context {
36    graph: Graph,
37    local_root: Ref,
38    tx_context: Option<Value>,
39    gas_coin: Option<Value>,
40    objects: Vec<Option<Value>>,
41    withdrawals: Vec<Option<Value>>,
42    pure: Vec<Option<Value>>,
43    receiving: Vec<Option<Value>>,
44    results: Vec<Vec<Option<Value>>>,
45}
46
47impl Value {
48    fn is_ref(&self) -> bool {
49        match self {
50            Value::Ref(_) => true,
51            Value::NonRef => false,
52        }
53    }
54
55    fn is_non_ref(&self) -> bool {
56        match self {
57            Value::Ref(_) => false,
58            Value::NonRef => true,
59        }
60    }
61
62    fn to_ref(&self) -> Option<Ref> {
63        match self {
64            Value::Ref(r) => Some(*r),
65            Value::NonRef => None,
66        }
67    }
68}

70impl Context {
71    fn new(ast: &T::Transaction) -> Result<Self, ExecutionError> {
72        let objects = ast.objects.iter().map(|_| Some(Value::NonRef)).collect();
73        let withdrawals = ast
74            .withdrawals
75            .iter()
76            .map(|_| Some(Value::NonRef))
77            .collect::<Vec<_>>();
78        let pure = ast
79            .pure
80            .iter()
81            .map(|_| Some(Value::NonRef))
82            .collect::<Vec<_>>();
83        let receiving = ast
84            .receiving
85            .iter()
86            .map(|_| Some(Value::NonRef))
87            .collect::<Vec<_>>();
88        let (mut graph, _locals) = Graph::new::<()>([]).map_err(graph_err)?;
89        let local_root = graph
90            .extend_by_epsilon((), std::iter::empty(), /* is_mut */ true)
91            .map_err(graph_err)?;
92        Ok(Self {
93            graph,
94            local_root,
95            tx_context: Some(Value::NonRef),
96            gas_coin: Some(Value::NonRef),
97            objects,
98            withdrawals,
99            pure,
100            receiving,
101            results: Vec::with_capacity(ast.commands.len()),
102        })
103    }
104
105    fn location(&mut self, l: T::Location) -> Result<&mut Option<Value>, ExecutionError> {
106        Ok(match l {
107            T::Location::TxContext => &mut self.tx_context,
108            T::Location::GasCoin => &mut self.gas_coin,
109            T::Location::ObjectInput(i) => self.objects.safe_get_mut(i as usize)?,
110            T::Location::WithdrawalInput(i) => self.withdrawals.safe_get_mut(i as usize)?,
111            T::Location::PureInput(i) => self.pure.safe_get_mut(i as usize)?,
112            T::Location::ReceivingInput(i) => self.receiving.safe_get_mut(i as usize)?,
113            T::Location::Result(i, j) => self
114                .results
115                .safe_get_mut(i as usize)?
116                .safe_get_mut(j as usize)?,
117        })
118    }
119
120    fn is_mutable(&self, r: Ref) -> Result<bool, ExecutionError> {
121        self.graph.is_mutable(r).map_err(graph_err)
122    }
123
124    fn borrowed_by(&self, r: Ref) -> Result<BTreeMap<Ref, Paths>, ExecutionError> {
125        self.graph.borrowed_by(r).map_err(graph_err)
126    }
127
128    /// Used for checking if a location is borrowed
129    /// Used for updating the borrowed marker in Copy, and for correctness of Move
130    fn is_location_borrowed(&self, l: T::Location) -> Result<bool, ExecutionError> {
131        let borrowed_by = self.borrowed_by(self.local_root)?;
132        Ok(borrowed_by
133            .iter()
134            .any(|(_, paths)| paths.iter().any(|path| path.starts_with(&Location(l)))))
135    }
136
137    fn release(&mut self, r: Ref) -> Result<(), ExecutionError> {
138        self.graph.release(r).map_err(graph_err)
139    }
140
141    fn extend_by_epsilon(&mut self, r: Ref, is_mut: bool) -> Result<Ref, ExecutionError> {
142        let new_r = self
143            .graph
144            .extend_by_epsilon((), std::iter::once(r), is_mut)
145            .map_err(graph_err)?;
146        Ok(new_r)
147    }
148
149    fn extend_by_label(
150        &mut self,
151        r: Ref,
152        is_mut: bool,
153        extension: T::Location,
154    ) -> Result<Ref, ExecutionError> {
155        let new_r = self
156            .graph
157            .extend_by_label((), std::iter::once(r), is_mut, Location(extension))
158            .map_err(graph_err)?;
159        Ok(new_r)
160    }
161
162    fn extend_by_dot_star_for_call(
163        &mut self,
164        sources: &BTreeSet<Ref>,
165        mutabilities: Vec<bool>,
166    ) -> Result<Vec<Ref>, ExecutionError> {
167        let new_refs = self
168            .graph
169            .extend_by_dot_star_for_call((), sources.iter().copied(), mutabilities)
170            .map_err(graph_err)?;
171        Ok(new_refs)
172    }
173
174    // Writable if
175    // No imm equal
176    // No extensions
177    fn is_writable(&self, r: Ref) -> Result<bool, ExecutionError> {
178        debug_assert!(self.is_mutable(r)?);
179        Ok(self
180            .borrowed_by(r)?
181            .values()
182            .all(|paths| paths.iter().all(|path| path.is_epsilon())))
183    }
184
185    // is in reference not able to be used in a call or return
186    fn find_non_transferrable(&self, refs: &BTreeSet<Ref>) -> Result<Option<Ref>, ExecutionError> {
187        let borrows = refs
188            .iter()
189            .copied()
190            .map(|r| Ok((r, self.borrowed_by(r)?)))
191            .collect::<Result<BTreeMap<_, _>, ExecutionError>>()?;
192        let mut_refs = refs
193            .iter()
194            .copied()
195            .filter_map(|r| match self.is_mutable(r) {
196                Ok(true) => Some(Ok(r)),
197                Ok(false) => None,
198                Err(e) => Some(Err(e)),
199            })
200            .collect::<Result<BTreeSet<_>, ExecutionError>>()?;
201        for (r, borrowed_by) in borrows {
202            let is_mut = mut_refs.contains(&r);
203            for (borrower, paths) in borrowed_by {
204                if !is_mut {
205                    if mut_refs.contains(&borrower) {
206                        // If the ref is imm, but is borrowed by a mut ref in the set
207                        // the mut ref is not transferrable
208                        // In other words, the mut ref is an extension of the imm ref
209                        return Ok(Some(borrower));
210                    }
211                } else {
212                    for path in paths {
213                        if !path.is_epsilon() || refs.contains(&borrower) {
214                            // If the ref is mut, it cannot have any non-epsilon extensions
215                            // If extension is epsilon (an alias), it cannot be in the transfer set
216                            return Ok(Some(r));
217                        }
218                    }
219                }
220            }
221        }
222        Ok(None)
223    }
224}

226/// Checks the following
227/// - Values are not used after being moved
228/// - Reference safety is upheld (no dangling references)
229pub fn verify(_env: &Env, ast: &T::Transaction) -> Result<(), ExecutionError> {
230    let mut context = Context::new(ast)?;
231    let commands = &ast.commands;
232    for c in commands {
233        let result = command(&mut context, c).map_err(|e| e.with_command_index(c.idx as usize))?;
234        assert_invariant!(
235            result.len() == c.value.result_type.len(),
236            "result length mismatch for command. {c:?}"
237        );
238        // drop unused result values
239        assert_invariant!(
240            result.len() == c.value.drop_values.len(),
241            "drop values length mismatch for command. {c:?}"
242        );
243        let result_values = result
244            .into_iter()
245            .zip(c.value.drop_values.iter().copied())
246            .map(|(v, drop)| {
247                Ok(if !drop {
248                    Some(v)
249                } else {
250                    consume_value(&mut context, v)?;
251                    None
252                })
253            })
254            .collect::<Result<Vec<_>, ExecutionError>>()?;
255        context.results.push(result_values);
256    }
257
258    let Context {
259        gas_coin,
260        objects,
261        pure,
262        receiving,
263        results,
264        ..
265    } = &mut context;
266    let gas_coin = gas_coin.take();
267    let objects = std::mem::take(objects);
268    let pure = std::mem::take(pure);
269    let receiving = std::mem::take(receiving);
270    let results = std::mem::take(results);
271    consume_value_opt(&mut context, gas_coin)?;
272    for vopt in objects.into_iter().chain(pure).chain(receiving) {
273        consume_value_opt(&mut context, vopt)?;
274    }
275    for result in results {
276        for vopt in result {
277            consume_value_opt(&mut context, vopt)?;
278        }
279    }
280
281    assert_invariant!(
282        context.borrowed_by(context.local_root)?.is_empty(),
283        "reference to local root not released"
284    );
285    context.release(context.local_root)?;
286    assert_invariant!(context.graph.abstract_size() == 0, "reference not released");
287    assert_invariant!(
288        context.tx_context.is_some(),
289        "tx_context should never be moved"
290    );
291
292    Ok(())
293}

295fn command(context: &mut Context, sp!(_, c): &T::Command) -> Result<Vec<Value>, ExecutionError> {
296    let result_tys = &c.result_type;
297    Ok(match &c.command {
298        T::Command__::MoveCall(mc) => {
299            let T::MoveCall {
300                function,
301                arguments: args,
302            } = &**mc;
303            let arg_values = arguments(context, args)?;
304            call(context, arg_values, &function.signature)?
305        }
306        T::Command__::TransferObjects(objects, recipient) => {
307            let object_values = arguments(context, objects)?;
308            let recipient_value = argument(context, recipient)?;
309            consume_values(context, object_values)?;
310            consume_value(context, recipient_value)?;
311            vec![]
312        }
313        T::Command__::SplitCoins(_, coin, amounts) => {
314            let coin_value = argument(context, coin)?;
315            let amount_values = arguments(context, amounts)?;
316            consume_values(context, amount_values)?;
317            write_ref(context, 0, coin_value)?;
318            (0..amounts.len()).map(|_| Value::NonRef).collect()
319        }
320        T::Command__::MergeCoins(_, target, coins) => {
321            let target_value = argument(context, target)?;
322            let coin_values = arguments(context, coins)?;
323            consume_values(context, coin_values)?;
324            write_ref(context, 0, target_value)?;
325            vec![]
326        }
327        T::Command__::MakeMoveVec(_, xs) => {
328            let vs = arguments(context, xs)?;
329            consume_values(context, vs)?;
330            vec![Value::NonRef]
331        }
332        T::Command__::Publish(_, _, _) => result_tys.iter().map(|_| Value::NonRef).collect(),
333        T::Command__::Upgrade(_, _, _, x, _) => {
334            let v = argument(context, x)?;
335            consume_value(context, v)?;
336            vec![Value::NonRef]
337        }
338    })
339}

345fn consume_values(context: &mut Context, values: Vec<Value>) -> Result<(), ExecutionError> {
346    for v in values {
347        consume_value(context, v)?;
348    }
349    Ok(())
350}
351
352fn consume_value_opt(context: &mut Context, value: Option<Value>) -> Result<(), ExecutionError> {
353    match value {
354        Some(v) => consume_value(context, v),
355        None => Ok(()),
356    }
357}
358
359fn consume_value(context: &mut Context, value: Value) -> Result<(), ExecutionError> {
360    match value {
361        Value::NonRef => Ok(()),
362        Value::Ref(r) => {
363            context.release(r)?;
364            Ok(())
365        }
366    }
367}
368
369fn arguments(context: &mut Context, xs: &[T::Argument]) -> Result<Vec<Value>, ExecutionError> {
370    xs.iter().map(|x| argument(context, x)).collect()
371}
372
373fn argument(context: &mut Context, x: &T::Argument) -> Result<Value, ExecutionError> {
374    match &x.value.0 {
375        T::Argument__::Use(T::Usage::Move(location)) => move_value(context, x.idx, *location),
376        T::Argument__::Use(T::Usage::Copy { location, borrowed }) => {
377            copy_value(context, x.idx, *location, borrowed)
378        }
379        T::Argument__::Borrow(is_mut, location) => {
380            borrow_location(context, x.idx, *is_mut, *location)
381        }
382        T::Argument__::Read(usage) => read_ref(context, x.idx, usage),
383        T::Argument__::Freeze(usage) => freeze_ref(context, x.idx, usage),
384    }
385}
386
387fn move_value(
388    context: &mut Context,
389    arg_idx: u16,
390    l: T::Location,
391) -> Result<Value, ExecutionError> {
392    if context.is_location_borrowed(l)? {
393        // TODO more specific error
394        return Err(command_argument_error(
395            CommandArgumentError::CannotMoveBorrowedValue,
396            arg_idx as usize,
397        ));
398    }
399    let Some(value) = context.location(l)?.take() else {
400        return Err(command_argument_error(
401            CommandArgumentError::ArgumentWithoutValue,
402            arg_idx as usize,
403        ));
404    };
405    Ok(value)
406}
407
408fn copy_value(
409    context: &mut Context,
410    arg_idx: u16,
411    l: T::Location,
412    borrowed: &OnceCell<bool>,
413) -> Result<Value, ExecutionError> {
414    let is_borrowed = context.is_location_borrowed(l)?;
415    borrowed
416        .set(is_borrowed)
417        .map_err(|_| make_invariant_violation!("Copy's borrowed marker should not yet be set"))?;
418
419    let Some(value) = context.location(l)? else {
420        // TODO more specific error
421        return Err(command_argument_error(
422            CommandArgumentError::ArgumentWithoutValue,
423            arg_idx as usize,
424        ));
425    };
426    Ok(match value {
427        Value::Ref(r) => {
428            let r = *r;
429            let is_mut = context.is_mutable(r)?;
430            let new_r = context.extend_by_epsilon(r, is_mut)?;
431            Value::Ref(new_r)
432        }
433        Value::NonRef => Value::NonRef,
434    })
435}

437fn borrow_location(
438    context: &mut Context,
439    arg_idx: u16,
440    is_mut: bool,
441    l: T::Location,
442) -> Result<Value, ExecutionError> {
443    // check that the location has a value
444    let Some(value) = context.location(l)? else {
445        // TODO more specific error
446        return Err(command_argument_error(
447            CommandArgumentError::ArgumentWithoutValue,
448            arg_idx as usize,
449        ));
450    };
451    assert_invariant!(
452        value.is_non_ref(),
453        "type checking should guarantee no borrowing of references"
454    );
455    let new_r = context.extend_by_label(context.local_root, is_mut, l)?;
456    Ok(Value::Ref(new_r))
457}

459/// Creates an alias to the reference, but one that is immutable
460fn freeze_ref(context: &mut Context, arg_idx: u16, u: &T::Usage) -> Result<Value, ExecutionError> {
461    let value = match u {
462        T::Usage::Move(l) => move_value(context, arg_idx, *l)?,
463        T::Usage::Copy { location, borrowed } => copy_value(context, arg_idx, *location, borrowed)?,
464    };
465    let Some(r) = value.to_ref() else {
466        invariant_violation!("type checking should guarantee FreezeRef is used on only references")
467    };
468    let new_r = context.extend_by_epsilon(r, /* is_mut */ false)?;
469    consume_value(context, value)?;
470    Ok(Value::Ref(new_r))
471}
472
473fn read_ref(context: &mut Context, arg_idx: u16, u: &T::Usage) -> Result<Value, ExecutionError> {
474    let value = match u {
475        T::Usage::Move(l) => move_value(context, arg_idx, *l)?,
476        T::Usage::Copy { location, borrowed } => copy_value(context, arg_idx, *location, borrowed)?,
477    };
478    assert_invariant!(
479        value.is_ref(),
480        "type checking should guarantee ReadRef is used on only references"
481    );
482    consume_value(context, value)?;
483    Ok(Value::NonRef)
484}
485
486fn write_ref(context: &mut Context, arg_idx: usize, value: Value) -> Result<(), ExecutionError> {
487    let Value::Ref(r) = value else {
488        invariant_violation!("type checking should guarantee WriteRef is used on only references");
489    };
490
491    if !context.is_writable(r)? {
492        // TODO more specific error
493        return Err(command_argument_error(
494            CommandArgumentError::CannotWriteToExtendedReference,
495            arg_idx,
496        ));
497    }
498    consume_value(context, value)?;
499    Ok(())
500}

502fn call(
503    context: &mut Context,
504    arg_values: Vec<Value>,
505    signature: &T::LoadedFunctionInstantiation,
506) -> Result<Vec<Value>, ExecutionError> {
507    let sources = arg_values
508        .iter()
509        .filter_map(|v| v.to_ref())
510        .collect::<BTreeSet<_>>();
511    if let Some(v) = context.find_non_transferrable(&sources)? {
512        let mut_idx = arg_values
513            .iter()
514            .zip(&signature.parameters)
515            .enumerate()
516            .find(|(_, (x, ty))| x.to_ref() == Some(v) && matches!(ty, Type::Reference(true, _)));
517
518        let Some((idx, _)) = mut_idx else {
519            invariant_violation!("non transferrable value was not found in arguments");
520        };
521        return Err(command_argument_error(
522            CommandArgumentError::InvalidReferenceArgument,
523            idx,
524        ));
525    }
526    let mutabilities = signature
527        .return_
528        .iter()
529        .filter_map(|ty| match ty {
530            Type::Reference(is_mut, _) => Some(*is_mut),
531            _ => None,
532        })
533        .collect::<Vec<_>>();
534    let mutabilities_len = mutabilities.len();
535    let mut return_references = context.extend_by_dot_star_for_call(&sources, mutabilities)?;
536    assert_invariant!(
537        return_references.len() == mutabilities_len,
538        "return_references should have the same length as mutabilities"
539    );
540
541    let mut return_values: Vec<_> = signature
542        .return_
543        .iter()
544        .rev()
545        .map(|ty| {
546            Ok(match ty {
547                Type::Reference(_is_mut, _) => {
548                    let Some(new_ref) = return_references.pop() else {
549                        invariant_violation!("return_references has less references than return_");
550                    };
551                    debug_assert_eq!(context.is_mutable(new_ref)?, *_is_mut);
552                    Value::Ref(new_ref)
553                }
554                _ => Value::NonRef,
555            })
556        })
557        .collect::<Result<Vec<_>, ExecutionError>>()?;
558    return_values.reverse();
559    assert_invariant!(
560        return_references.is_empty(),
561        "return_references has more references than return_"
562    );
563    consume_values(context, arg_values)?;
564    Ok(return_values)
565}
566
567fn graph_err(e: move_regex_borrow_graph::InvariantViolation) -> ExecutionError {
568    ExecutionError::invariant_violation(format!("Borrow graph invariant violation: {}", e.0))
569}

571impl fmt::Display for Location {
572    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
573        match self.0 {
574            T::Location::TxContext => write!(f, "TxContext"),
575            T::Location::GasCoin => write!(f, "GasCoin"),
576            T::Location::ObjectInput(idx) => write!(f, "ObjectInput({idx})"),
577            T::Location::WithdrawalInput(idx) => write!(f, "WithdrawalInput({idx})"),
578            T::Location::PureInput(idx) => write!(f, "PureInput({idx})"),
579            T::Location::ReceivingInput(idx) => write!(f, "ReceivingInput({idx})"),
580            T::Location::Result(i, j) => write!(f, "Result({i}, {j})"),
581        }
582    }
