18/// Marks if a clique is hot or not
19#[derive(Clone, Copy, Debug)]
20enum Temperature {
21    /// The clique is always hot--since it's heat was not the result of a hot potato value
22    AlwaysHot,
23    /// The clique is hot because it contains hot potato values--tracked via the `usize` count
24    Count(usize),
25}
26
27/// The data for a clique, which tracks the what values/locations are used together. Once a value
28/// is used with another in a `Command`, their cliques are merged forever.
29enum Clique {
30    Merged(CliqueID),
31    Root(Temperature),
32}
33
34type CliqueID = usize;
35
36#[must_use]
37enum Value {
38    // We don't want to entangle the TxContext with cliques
39    TxContext,
40    Normal {
41        /// The clique this value belongs to
42        clique: CliqueID,
43        /// If the value contributes to the hot-value count in its clique
44        heats: bool,
45    },
46}
47
48/// A list of all cliques. The size is bound by the number of inputs + number of commands
49struct Cliques(Vec<Clique>);
50
51struct Context {
52    cliques: Cliques,
53    tx_context: Option<Value>,
54    gas_coin: Option<Value>,
55    objects: Vec<Option<Value>>,
56    withdrawals: Vec<Option<Value>>,
57    pure: Vec<Option<Value>>,
58    receiving: Vec<Option<Value>>,
59    results: Vec<Vec<Option<Value>>>,
60}
61
62impl Temperature {
63    fn add(self, other: Temperature) -> Result<Temperature, ExecutionError> {
64        Ok(match (self, other) {
65            (Temperature::AlwaysHot, _) | (_, Temperature::AlwaysHot) => Temperature::AlwaysHot,
66            (Temperature::Count(a), Temperature::Count(b)) => Temperature::Count(
67                a.checked_add(b)
68                    .ok_or_else(|| make_invariant_violation!("Hot count overflow"))?,
69            ),
70        })
71    }
72
73    fn sub(self, b: usize) -> Result<Temperature, ExecutionError> {
74        Ok(match self {
75            Temperature::AlwaysHot => Temperature::AlwaysHot,
76            Temperature::Count(a) => Temperature::Count(
77                a.checked_sub(b)
78                    .ok_or_else(|| make_invariant_violation!("Hot count cannot go negative"))?,
79            ),
80        })
81    }
82
83    /// Returns true if the clique is always hot or has a positive hot count
84    fn is_hot(&self) -> bool {
85        match self {
86            Temperature::AlwaysHot => true,
87            Temperature::Count(c) => *c > 0,
88        }
89    }
90}

92impl Cliques {
93    /// Creates a new empty set of cliques
94    fn new() -> Self {
95        Self(vec![])
96    }
97
98    /// Creates a new clique and returns its ID
99    fn next(&mut self) -> CliqueID {
100        let id = self.0.len();
101        self.0.push(Clique::Root(Temperature::Count(0)));
102        id
103    }
104
105    /// Returns the root of the clique (resolving any merges/forwards)
106    fn root(&self, id: CliqueID) -> Result<CliqueID, ExecutionError> {
107        let mut visited = BTreeSet::from([id]);
108        let mut cur = id;
109        loop {
110            match self.0.safe_get(cur)? {
111                Clique::Root(_) => return Ok(cur),
112                Clique::Merged(next) => {
113                    let newly_visited = visited.insert(*next);
114                    if !newly_visited {
115                        invariant_violation!("Clique merge cycle detected");
116                    }
117                    cur = *next
118                }
119            }
120        }
121    }
122
123    /// Returns the temperature of the clique (at the root)
124    fn temp(&self, id: CliqueID) -> Result<Temperature, ExecutionError> {
125        let root = self.root(id)?;
126        let Clique::Root(temp) = *self.0.safe_get(root)? else {
127            invariant_violation!("Clique {root} should be a root");
128        };
129        Ok(temp)
130    }
131
132    /// Returns a mutable reference to the temperature of the clique (at the root)
133    fn temp_mut(&mut self, id: CliqueID) -> Result<&mut Temperature, ExecutionError> {
134        let root = self.root(id)?;
135        let Clique::Root(temp) = self.0.safe_get_mut(root)? else {
136            invariant_violation!("Clique {root} should be a root");
137        };
138        Ok(temp)
139    }
140
141    /// Modifies the temperature of this clique (at the root) via `f`, whose first parameter is the
142    /// current temperature
143    fn modify_temp(
144        &mut self,
145        id: CliqueID,
146        f: impl FnOnce(Temperature) -> Result<Temperature, ExecutionError>,
147    ) -> Result<(), ExecutionError> {
148        let temp = self.temp_mut(id)?;
149        *temp = f(*temp)?;
150        Ok(())
151    }
152
153    /// Merges the given cliques into one clique
154    fn merge(&mut self, clique_ids: BTreeSet<CliqueID>) -> Result<CliqueID, ExecutionError> {
155        let roots: BTreeSet<CliqueID> = clique_ids
156            .iter()
157            .map(|&id| self.root(id))
158            .collect::<Result<_, _>>()?;
159        Ok(match roots.len() {
160            0 => self.next(),
161            1 => *roots.iter().next().unwrap(),
162            _ => {
163                let merged = self.next();
164                let mut merged_temp = Temperature::Count(0);
165                for &root in &roots {
166                    let temp = self.temp(root)?;
167                    *self.0.safe_get_mut(root)? = Clique::Merged(merged);
168                    merged_temp = merged_temp.add(temp)?;
169                }
170                *self.0.safe_get_mut(merged)? = Clique::Root(merged_temp);
171                // For efficiency, forward all the non-roots to the merged root
172                // (bypassing the old root)
173                for id in clique_ids {
174                    *self.0.safe_get_mut(id)? = Clique::Merged(merged);
175                }
176                merged
177            }
178        })
179    }
180
181    /// Creates a new value in its own clique (not hot)
182    fn input_value(&mut self) -> Value {
183        let clique = self.next();
184        Value::Normal {
185            clique,
186            heats: false,
187        }
188    }
189
190    /// Creates a new value in the given `clique` and bumps the hot count if `heats` is true
191    fn new_value(&mut self, clique: CliqueID, heats: bool) -> Result<Value, ExecutionError> {
192        if heats {
193            self.modify_temp(clique, |t| t.add(Temperature::Count(1)))?;
194        }
195        Ok(Value::Normal { clique, heats })
196    }
197
198    /// Releases a value, decrementing the hot count of its clique if it `heats`.
199    fn release_value(&mut self, value: Value) -> Result<Option<CliqueID>, ExecutionError> {
200        let (clique, heats) = match value {
201            Value::TxContext => return Ok(None),
202            Value::Normal { clique, heats } => (clique, heats),
203        };
204        if heats {
205            self.modify_temp(clique, |t| t.sub(1))?;
206        }
207        Ok(Some(clique))
208    }
209
210    /// Returns true if the clique is hot, always hot or a positive hot count
211    fn is_hot(&self, clique: CliqueID) -> Result<bool, ExecutionError> {
212        Ok(self.temp(clique)?.is_hot())
213    }
214
215    /// Marks the given clique as always hot
216    fn mark_always_hot(&mut self, clique: CliqueID) -> Result<(), ExecutionError> {
217        self.modify_temp(clique, |_| Ok(Temperature::AlwaysHot))
218    }
219}

221impl Context {
222    fn new(txn: &T::Transaction) -> Self {
223        let mut cliques = Cliques::new();
224        let tx_context = Some(Value::TxContext);
225        let gas_coin = Some(cliques.input_value());
226        let objects = (0..txn.objects.len())
227            .map(|_| Some(cliques.input_value()))
228            .collect();
229        let withdrawals = (0..txn.withdrawals.len())
230            .map(|_| Some(cliques.input_value()))
231            .collect();
232        let pure = (0..txn.pure.len())
233            .map(|_| Some(cliques.input_value()))
234            .collect();
235        let receiving = (0..txn.receiving.len())
236            .map(|_| Some(cliques.input_value()))
237            .collect();
238        Self {
239            tx_context,
240            cliques,
241            gas_coin,
242            objects,
243            withdrawals,
244            pure,
245            receiving,
246            results: vec![],
247        }
248    }
249
250    // Checks if all values are released and that all hot counts are zero
251    fn finish(self) -> Result<(), ExecutionError> {
252        let Context {
253            mut cliques,
254            tx_context,
255            gas_coin,
256            objects,
257            withdrawals,
258            pure,
259            receiving,
260            results,
261        } = self;
262
263        // Check that if all values are released, the hot counts are zero
264        let all_values = tx_context
265            .into_iter()
266            .chain(gas_coin)
267            .chain(objects.into_iter().flatten())
268            .chain(withdrawals.into_iter().flatten())
269            .chain(pure.into_iter().flatten())
270            .chain(receiving.into_iter().flatten())
271            .chain(results.into_iter().flatten().flatten());
272        let mut clique_ids: BTreeSet<CliqueID> = BTreeSet::new();
273        for value in all_values {
274            let clique = match &value {
275                Value::TxContext => continue,
276                Value::Normal { clique, .. } => clique,
277            };
278            clique_ids.insert(*clique);
279            cliques.release_value(value)?;
280        }
281        for id in clique_ids {
282            match cliques.temp(id)? {
283                Temperature::AlwaysHot => (),
284                Temperature::Count(c) => {
285                    assert_invariant!(c == 0, "All hot counts should be zero at end")
286                }
287            }
288        }
289        Ok(())
290    }
291
292    fn location(&mut self, location: &T::Location) -> Result<&mut Option<Value>, ExecutionError> {
293        Ok(match location {
294            T::Location::GasCoin => &mut self.gas_coin,
295            T::Location::ObjectInput(i) => self.objects.safe_get_mut(*i as usize)?,
296            T::Location::WithdrawalInput(i) => self.withdrawals.safe_get_mut(*i as usize)?,
297            T::Location::PureInput(i) => self.pure.safe_get_mut(*i as usize)?,
298            T::Location::ReceivingInput(i) => self.receiving.safe_get_mut(*i as usize)?,
299            T::Location::Result(i, j) => self
300                .results
301                .safe_get_mut(*i as usize)?
302                .safe_get_mut(*j as usize)?,
303            T::Location::TxContext => &mut self.tx_context,
304        })
305    }
306
307    fn usage(&mut self, usage: &T::Usage) -> Result<Value, ExecutionError> {
308        match usage {
309            T::Usage::Move(location) => {
310                let Some(value) = self.location(location)?.take() else {
311                    invariant_violation!("Move of moved value");
312                };
313                Ok(value)
314            }
315            T::Usage::Copy { location, .. } => {
316                let Some(location) = self.location(location)?.as_ref() else {
317                    invariant_violation!("Copy of moved value");
318                };
319                let (clique, heats) = match location {
320                    Value::TxContext => {
321                        invariant_violation!("Cannot copy TxContext");
322                    }
323                    Value::Normal { clique, heats } => (*clique, *heats),
324                };
325                self.cliques.new_value(clique, heats)
326            }
327        }
328    }
329
330    fn argument(&mut self, sp!(_, (arg, _ty)): &T::Argument) -> Result<Value, ExecutionError> {
331        Ok(match arg {
332            T::Argument__::Use(usage) => self.usage(usage)?,
333            T::Argument__::Read(usage) | T::Argument__::Freeze(usage) => {
334                // This is equivalent to just the `usage` but we go through the steps of
335                // creating a new value and releasing the old one for "correctness" and clarity
336                let value = self.usage(usage)?;
337                let (clique, heats) = match &value {
338                    Value::TxContext => {
339                        invariant_violation!("Cannot read or freeze TxContext");
340                    }
341                    Value::Normal { clique, heats } => (*clique, *heats),
342                };
343                let new_value = self.cliques.new_value(clique, heats)?;
344                self.cliques.release_value(value)?;
345                new_value
346            }
347            T::Argument__::Borrow(_, location) => {
348                let Some(location) = self.location(location)?.as_ref() else {
349                    invariant_violation!("Borrow of moved value");
350                };
351                let (clique, heats) = match location {
352                    Value::TxContext => {
353                        // no clique/heat for TxContext
354                        return Ok(Value::TxContext);
355                    }
356                    Value::Normal { clique, heats } => (*clique, *heats),
357                };
358                // Create a new value (representing the reference to this value)
359                // that is in the same clique and has the same heat
360                self.cliques.new_value(clique, heats)?
361            }
362        })
363    }
364}

366/// Checks entry taint rules. An `entry` function cannot have any arguments in a "hot" clique.
367/// We define cliques and "hot" with the following rules:
368/// - Each input value starts as a node in the graph, as such in its own clique.
369/// - When a value is used in a command, we "entangle" all of the values and the outputs.
370///   Conceptually, we add an edge between all of the inputs/outputs and any other nodes in the
371///   inputs's cliques. As such, we don't really need to track the graph as a whole, just what
372///   clique each value is in.
373/// - A value is "heats" its clique if it is a "hot potato" value, i.e. a value whose type
374///   does not have `drop` and does not have `store`.
375/// - A clique is the "hot" in one of two ways:
376///   - If it contains 1 or more "hot potato" values
377///   - If it is marked as "always hot" if a command consumes a shared object by-value
378/// - A non-public `entry` function cannot have any inputs that are in a `hot` clique.
379/// - Note that command inputs are released before checking the rules, so an `entry` function can
380///   consume a hot potato value if it is the last "heating" value in its clique.
381pub fn verify<Mode: ExecutionMode>(env: &Env, txn: &T::Transaction) -> Result<(), ExecutionError> {
382    let mut context = Context::new(txn);
383    for c in &txn.commands {
384        let result_values = command::<Mode>(env, &mut context, c)
385            .map_err(|e| e.with_command_index(c.idx as usize))?;
386        assert_invariant!(
387            result_values.len() == c.value.result_type.len(),
388            "result length mismatch"
389        );
390        context.results.push(result_values);
391    }
392    context.finish()?;
393    Ok(())
394}

396fn command<Mode: ExecutionMode>(
397    env: &Env,
398    context: &mut Context,
399    sp!(_, c): &T::Command,
400) -> Result<Vec<Option<Value>>, ExecutionError> {
401    let T::Command_ {
402        command,
403        result_type,
404        drop_values,
405        consumed_shared_objects,
406    } = c;
407    let argument_cliques = arguments(env, context, command.arguments())?;
408    match command {
409        T::Command__::MoveCall(call) => move_call::<Mode>(env, context, call, &argument_cliques)?,
410        T::Command__::TransferObjects(_, _)
411        | T::Command__::SplitCoins(_, _, _)
412        | T::Command__::MergeCoins(_, _, _)
413        | T::Command__::MakeMoveVec(_, _)
414        | T::Command__::Publish(_, _, _)
415        | T::Command__::Upgrade(_, _, _, _, _) => (),
416    }
417    let merged_clique = context
418        .cliques
419        .merge(argument_cliques.into_iter().map(|(_, c)| c).collect())?;
420    let consumes_shared_objects = !consumed_shared_objects.is_empty();
421    if consumes_shared_objects {
422        context.cliques.mark_always_hot(merged_clique)?;
423    }
424    assert_invariant!(
425        drop_values.len() == result_type.len(),
426        "drop_values length mismatch"
427    );
428    result_type
429        .iter()
430        .zip(drop_values)
431        .map(|(ty, dropped)| {
432            Ok(if *dropped {
433                None
434            } else {
435                let heats = is_hot_potato_return_type(ty);
436                Some(context.cliques.new_value(merged_clique, heats)?)
437            })
438        })
439        .collect()
440}

442/// Returns the index of the first hot argument, if any
443fn arguments<'a>(
444    env: &Env,
445    context: &mut Context,
446    args: impl IntoIterator<Item = &'a T::Argument>,
447) -> Result<Vec<(u16, CliqueID)>, ExecutionError> {
448    let mut arguments = vec![];
449    for arg in args {
450        if let Some(clique) = argument(env, context, arg)? {
451            arguments.push((arg.idx, clique));
452        }
453    }
454    Ok(arguments)
455}

457fn argument(
458    _env: &Env,
459    context: &mut Context,
460    arg: &T::Argument,
461) -> Result<Option<CliqueID>, ExecutionError> {
462    let value = context.argument(arg)?;
463    context.cliques.release_value(value)
464}

466/// Checks a move call for
467/// - valid signature (no references in return type)
468/// - valid visibility
469/// - private generics rules
470/// - if entry, no hot arguments
471///
472/// Returns true iff any return type is a hot potato
473fn move_call<Mode: ExecutionMode>(
474    env: &Env,
475    context: &mut Context,
476    call: &T::MoveCall,
477    argument_cliques: &[(u16, CliqueID)],
478) -> Result<(), ExecutionError> {
479    let T::MoveCall {
480        function,
481        arguments: _,
482    } = call;
483    let module = env.module_definition(&function.runtime_id, &function.linkage)?;
484    let module: &CompiledModule = module.as_ref();
485    let Some((_index, fdef)) = module.find_function_def_by_name(function.name.as_str()) else {
486        invariant_violation!(
487            "Could not resolve function '{}' in module {}. \
488            This should have been checked when linking",
489            &function.name,
490            module.self_id(),
491        );
492    };
493    let visibility = fdef.visibility;
494    let is_entry = fdef.is_entry;
495    // check rules around hot arguments and entry functions
496    let is_non_public = if env.protocol_config.restrict_hot_or_not_entry_functions() {
497        !matches!(visibility, Visibility::Public)
498    } else {
499        matches!(visibility, Visibility::Private)
500    };
501    if is_entry && is_non_public && !Mode::allow_arbitrary_values() {
502        let mut hot_argument: Option<u16> = None;
503        for (idx, clique) in argument_cliques {
504            if context.cliques.is_hot(*clique)? {
505                hot_argument = Some(*idx);
506                break;
507            }
508        }
509        if let Some(idx) = hot_argument {
510            return Err(command_argument_error(
511                CommandArgumentError::InvalidArgumentToPrivateEntryFunction,
512                idx as usize,
513            ));
514        }
515    }
516    Ok(())
517}

519// is missing both drop and store
520fn is_hot_potato_return_type(ty: &T::Type) -> bool {
521    let abilities = ty.abilities();
522    !abilities.has_drop() && !abilities.has_store()
