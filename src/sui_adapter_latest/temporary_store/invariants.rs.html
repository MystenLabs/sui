4//! Post-execution system-invariant checks for transaction execution.
5//!
6//! All of the defense-in-depth checks that verify the finalized execution results satisfy system
7//! invariants live here, behind the [`InvariantChecker`] API: that the transaction neither mints
8//! nor burns SUI, that balance-accumulator withdrawals stay authorized, and that every modified
9//! object traces back to an authenticated owner. [`TemporaryStore`] owns one `InvariantChecker`,
10//! forwards the little bookkeeping it accumulates during execution to it (see
11//! [`record_ptb_event_range`]), and defers to it for the checks. Keeping it in its own module
12//! keeps the invariant accounting -- which reaches across gas, accumulator events, settlement SUI,
13//! per-object storage rebates, and object ownership -- out of the main store code.
14//!
15//! Note these are invariant *assertions* (a failure means a bug, and aborts or panics), distinct
16//! from transaction-validation guards like `TemporaryStore::check_accumulator_amounts_representable`
17//! that can legitimately reject a well-formed transaction.
18//!
19//! [`record_ptb_event_range`]: InvariantChecker::record_ptb_event_range
20

45/// The per-transaction inputs the invariant checks need that are derived from the raw transaction
46/// (rather than accumulated during execution). Built once, up front, by
47/// [`InvariantChecker::set_transaction_inputs`] so the checks read them off the store instead of
48/// receiving them as threaded arguments.
49#[derive(Default)]
50struct InvariantInputs {
51    /// Per-`(address, type)` funds-accumulator reservation budget authorized by this transaction.
52    /// Sources: PTB `FundsWithdrawalArg`s (sender/sponsor as owner), gas paid entirely from an
53    /// address balance, and gas-data coin-reservation digests. Consumed by
54    /// `check_address_balance_changes` and `check_ownership_invariants`.
55    input_reservations: BTreeMap<(SuiAddress, TypeTag), u64>,
56    /// For the advance-epoch transaction, `(epoch_fees minted, epoch_rebates burned)`; `None`
57    /// for every other transaction. Needed by `check_sui_conserved_expensive`, which must account
58    /// for the SUI the epoch change mints and burns.
59    advance_epoch_gas_summary: Option<(u64, u64)>,
60    /// The genesis transaction mints the initial SUI supply and so is exempt from conservation.
61    is_genesis: bool,
62}
63
64/// Holds the invariant-check-only bookkeeping accumulated during execution and exposes the
65/// post-execution system-invariant checks (SUI conservation, balance-accumulator authorization,
66/// object ownership) as its API.
67///
68/// It owns the transaction-derived inputs the checks need ([`InvariantInputs`], set once up
69/// front) and the PTB-emitted accumulator-event ranges accumulated during execution; the rest of
70/// the data the checks need (modified objects, accumulator events, settlement SUI, gas summary,
71/// ...) is read from the owning [`TemporaryStore`] passed to each check method.
72pub(crate) struct InvariantChecker {
73    /// Transaction-derived inputs, populated by [`Self::set_transaction_inputs`] before execution.
74    inputs: InvariantInputs,
75    /// Index ranges into `execution_results.accumulator_events` for events emitted from PTB
76    /// (Move) execution. Each `record_ptb_event_range` call appends a contiguous range
77    /// bracketing the merge. Any index outside these ranges was emitted by the runtime outside
78    /// of PTB execution (currently only `gas_charger`'s `add_accumulator_event`). Consumed by
79    /// `check_address_balance_changes` to gate non-PTB-emitted events behind input reservations.
80    /// Cleared on `drop_writes` since the underlying events are also cleared. A `Vec` is
81    /// sufficient because real transactions run the PTB at most a handful of times.
82    ptb_emitted_accumulator_event_ranges: Vec<Range<usize>>,
83}

85impl InvariantChecker {
86    pub(crate) fn new() -> Self {
87        Self {
88            inputs: InvariantInputs::default(),
89            ptb_emitted_accumulator_event_ranges: Vec::new(),
90        }
91    }

93    /// Derive and cache the per-transaction invariant-check inputs from the raw transaction. Must be
94    /// called once, before execution, and *after* any gas-smash filtering of `gas_data`, since the
95    /// reservation budget reads the final gas payment list.
96    pub(crate) fn set_transaction_inputs(
97        &mut self,
98        transaction_kind: &TransactionKind,
99        gas_data: &GasData,
100        transaction_signer: SuiAddress,
101    ) {
102        self.inputs = InvariantInputs {
103            input_reservations: compute_input_reservations(
104                transaction_kind,
105                gas_data,
106                transaction_signer,
107            ),
108            advance_epoch_gas_summary: transaction_kind.get_advance_epoch_tx_gas_summary(),
109            is_genesis: matches!(transaction_kind, TransactionKind::Genesis(_)),
110        };
111    }

113    /// The funds-accumulator reservation budget for this transaction. Also consumed by
114    /// `TemporaryStore::check_ownership_invariants`, which shares the same authorization model.
115    pub(crate) fn input_reservations(&self) -> &BTreeMap<(SuiAddress, TypeTag), u64> {
116        &self.inputs.input_reservations
117    }
118
119    /// Drop the PTB-emitted ranges. Called from `TemporaryStore::drop_writes` since the
120    /// underlying accumulator events the ranges point into are cleared at the same time.
121    pub(crate) fn clear(&mut self) {
122        self.ptb_emitted_accumulator_event_ranges.clear();
123    }
124
125    /// Record that the accumulator events in `[start, end)` (indices into
126    /// `execution_results.accumulator_events`) were emitted by PTB (Move) execution. `start`/`end`
127    /// are the lengths of the accumulator-event vec before/after a `merge_results`. The
128    /// address-balance change invariant uses this set to distinguish trusted PTB-emitted events
129    /// from runtime-emitted ones.
130    pub(crate) fn record_ptb_event_range(&mut self, start: usize, end: usize) {
131        debug_assert!(
132            start <= end,
133            "merge_results should not shrink accumulator_events"
134        );
135        let (start, end) = (start.min(end), start.max(end));
136        let range = start..end;
137        match self.ptb_emitted_accumulator_event_ranges.last_mut() {
138            // Coalesce with the previous PTB range if no runtime events were added in between.
139            Some(last) if last.end == range.start => last.end = range.end,
140            _ => self.ptb_emitted_accumulator_event_ranges.push(range),
141        }
142    }

144    /// Check that this transaction neither creates nor destroys SUI. This should hold for all txes
145    /// except the epoch change tx, which mints staking rewards equal to the gas fees burned in the
146    /// previous epoch.  Specifically, this checks two key invariants about storage
147    /// fees and storage rebate:
148    ///
149    /// 1. all SUI in storage rebate fields of input objects should flow either to the transaction
150    ///    storage rebate, or the transaction non-refundable storage rebate
151    /// 2. all SUI charged for storage should flow into the storage rebate field of some output
152    ///    object
153    ///
154    /// This function is intended to be called *after* we have charged for
155    /// gas + applied the storage rebate to the gas object, but *before* we
156    /// have updated object versions.
157    pub(crate) fn check_sui_conserved(
158        &self,
159        store: &TemporaryStore<'_>,
160        simple_conservation_checks: bool,
161        gas_summary: &GasCostSummary,
162    ) -> Result<(), ExecutionError> {
163        if !simple_conservation_checks {
164            return Ok(());
165        }
166        // total amount of SUI in storage rebate of input objects
167        let mut total_input_rebate = 0;
168        // total amount of SUI in storage rebate of output objects
169        let mut total_output_rebate = 0;
170        for (_id, input, output) in get_modified_objects(store) {
171            if let Some(input) = input {
172                total_input_rebate += input.storage_rebate;
173            }
174            if let Some(object) = output {
175                total_output_rebate += object.storage_rebate;
176            }
177        }
178
179        if gas_summary.storage_cost == 0 {
180            // this condition is usually true when the transaction went OOG and no
181            // gas is left for storage charges.
182            // The storage cost has to be there at least for the gas coin which
183            // will not be deleted even when going to 0.
184            // However if the storage cost is 0 and if there is any object touched
185            // or deleted the value in input must be equal to the output plus rebate and
186            // non refundable.
187            // Rebate and non refundable will be positive when there are object deleted
188            // (gas smashing being the primary and possibly only example).
189            // A more typical condition is for all storage charges in summary to be 0 and
190            // then input and output must be the same value
191            if total_input_rebate
192                != total_output_rebate
193                    + gas_summary.storage_rebate
194                    + gas_summary.non_refundable_storage_fee
195            {
196                return Err(ExecutionError::invariant_violation(format!(
197                    "SUI conservation failed -- no storage charges in gas summary \
198                        and total storage input rebate {} not equal  \
199                        to total storage output rebate {}",
200                    total_input_rebate, total_output_rebate,
201                )));
202            }
203        } else {
204            // all SUI in storage rebate fields of input objects should flow either to
205            // the transaction storage rebate, or the non-refundable storage rebate pool
206            if total_input_rebate
207                != gas_summary.storage_rebate + gas_summary.non_refundable_storage_fee
208            {
209                return Err(ExecutionError::invariant_violation(format!(
210                    "SUI conservation failed -- {} SUI in storage rebate field of input objects, \
211                        {} SUI in tx storage rebate or tx non-refundable storage rebate",
212                    total_input_rebate, gas_summary.non_refundable_storage_fee,
213                )));
214            }
215
216            // all SUI charged for storage should flow into the storage rebate field
217            // of some output object
218            if gas_summary.storage_cost != total_output_rebate {
219                return Err(ExecutionError::invariant_violation(format!(
220                    "SUI conservation failed -- {} SUI charged for storage, \
221                        {} SUI in storage rebate field of output objects",
222                    gas_summary.storage_cost, total_output_rebate
223                )));
224            }
225        }
226        Ok(())
227    }

229    /// Defense-in-depth invariant on funds-accumulator events. Per `(address, type)`:
230    /// - If the pair is in `input_reservations`, net withdrawal <= budget.
231    /// - Else if the PTB emitted a Split at this key, we assume there must be an object withdrawal.
232    ///   As such, any net change is acceptable.
233    /// - Else if the PTB emitted only Merges at this key, we can assume there might not be an
234    ///   object withdrawal. In any case, the net balance at the end of the transaction should be
235    ///   non-negative, since there could be additional withdrawals from gas, but they should
236    ///   not exceed the deposits.
237    /// - Else, any event is unauthorized.
238    ///
239    /// Currently the only funds-accumulator type is `Balance<T>`, so the check is scoped to
240    /// those events. As more accumulator shapes are added the filter and the integer
241    /// arithmetic in this method will need to grow with them.
242    ///
243    /// PTB-emitted events are identified via `ptb_emitted_accumulator_event_ranges`, populated
244    /// at `record_execution_results` time. They are trusted because Move enforces `&mut UID`
245    /// and the native checks the actual balance.
246    pub(crate) fn check_address_balance_changes(
247        &self,
248        store: &TemporaryStore<'_>,
249    ) -> Result<(), ExecutionError> {
250        use sui_types::balance::Balance;
251
252        let input_reservations = &self.inputs.input_reservations;
253        let mut actual_changes: BTreeMap<(SuiAddress, TypeTag), i128> = BTreeMap::new();
254        let mut has_ptb_withdrawals: BTreeSet<(SuiAddress, TypeTag)> = BTreeSet::new();
255        let mut has_ptb_deposits: BTreeSet<(SuiAddress, TypeTag)> = BTreeSet::new();
256        for (idx, event) in store
257            .execution_results
258            .accumulator_events
259            .iter()
260            .enumerate()
261        {
262            // Filter on the value shape first: only `Integer` carries the funds-flow we care
263            // about. Other shapes (e.g. `EventDigest` for event-stream heads) belong to
264            // non-Balance accumulators and are out of scope here. If we ever see an `Integer`
265            // value at a non-`Balance<T>` type, the accounting invariants below don't apply
266            // -- debug_fatal so that case is surfaced instead of silently accepted.
267            let amount = match event.write.value {
268                AccumulatorValue::Integer(amount) => amount as i128,
269                AccumulatorValue::IntegerTuple(_, _) | AccumulatorValue::EventDigest(_) => {
270                    assert_invariant!(
271                        !sui_types::balance::Balance::is_balance_type(&event.write.address.ty),
272                        "Non-integer accumulator changes should not be balances"
273                    );
274                    continue;
275                }
276            };
277            if !Balance::is_balance_type(&event.write.address.ty) {
278                debug_fatal!(
279                    "Integer accumulator value at non-Balance type: {:?}",
280                    event.write.address.ty
281                );
282                continue;
283            }
284            let is_ptb_emitted = self
285                .ptb_emitted_accumulator_event_ranges
286                .iter()
287                .any(|range| range.contains(&idx));
288            let key = (event.write.address.address, event.write.address.ty.clone());
289            let change = match event.write.operation {
290                AccumulatorOperation::Split => {
291                    if is_ptb_emitted {
292                        has_ptb_withdrawals.insert(key.clone());
293                    }
294                    -amount
295                }
296                AccumulatorOperation::Merge => {
297                    if is_ptb_emitted {
298                        has_ptb_deposits.insert(key.clone());
299                    }
300                    amount
301                }
302            };
303            *actual_changes.entry(key.clone()).or_insert(0) += change;
304        }
305
306        for (key, actual) in actual_changes {
307            let (address, type_tag) = &key;
308            if let Some(budget) = input_reservations.get(&key).copied() {
309                let net_withdrawn = -actual.min(0) as u128;
310                assert_invariant!(
311                    net_withdrawn <= budget as u128,
312                    "Balance accumulator withdrawal exceeds reservation budget at address \
313                    {address} for type {type_tag}: net Split {net_withdrawn}, budget {budget}"
314                );
315            } else if has_ptb_withdrawals.contains(&key) {
316                // Move authorized the PTB Split against the on-chain balance, so any
317                // resulting net (including a net withdrawal beyond any PTB Merges here)
318                // is trusted.
319            } else if has_ptb_deposits.contains(&key) {
320                // PTB only deposited at this key. As such, the final net change must be
321                // non-negative, since there was no authorization for any withdrawal.
322                // We cannot compare this value to the sum of the PTB deposits due to intricacies
323                // with gas charging and storage rebate.
324                assert_invariant!(
325                    actual >= 0,
326                    "PTB-emitted Balance accumulator deposits do not cover the runtime \
327                    withdrawal at address {address} for type {type_tag}: net change {actual}"
328                );
329            } else {
330                invariant_violation!(
331                    "Unauthorized runtime Balance accumulator event at address {address} for \
332                    type {type_tag}: net change {actual} (no input reservation, no PTB-emitted \
333                    events)"
334                );
335            }
336        }
337
338        Ok(())
339    }

341    /// Check that this transaction neither creates nor destroys SUI.
342    /// This more expensive check will check a third invariant on top of the 2 performed
343    /// by `check_sui_conserved` above:
344    ///
345    /// * all SUI in input objects (including coins etc in the Move part of an object) should flow
346    ///   either to an output object, or be burned as part of computation fees or non-refundable
347    ///   storage rebate
348    ///
349    /// This function is intended to be called *after* we have charged for gas + applied the
350    /// storage rebate to the gas object, but *before* we have updated object versions. The
351    /// advance epoch transaction would mint `epoch_fees` amount of SUI, and burn `epoch_rebates`
352    /// amount of SUI. We need these information for this check.
353    pub(crate) fn check_sui_conserved_expensive(
354        &self,
355        store: &TemporaryStore<'_>,
356        gas_summary: &GasCostSummary,
357        layout_resolver: &mut impl LayoutResolver,
358    ) -> Result<(), ExecutionError> {
359        let advance_epoch_gas_summary = self.inputs.advance_epoch_gas_summary;
360        // Accumulate in u128. The per-object SUI totals are bounded by the real supply, but the
361        // accumulator-event terms below are not: an object-sourced withdrawal/deposit (backing
362        // verified only at settlement) can contribute up to u64::MAX on each side, and a transaction
363        // can stack several across distinct keys, so a u64 running total could overflow. These
364        // amounts net out, so a u128 sum stays exact and conservation is decided correctly.
365        // total amount of SUI in input objects, including both coins and storage rebates
366        let mut total_input_sui: u128 = 0;
367        // total amount of SUI in output objects, including both coins and storage rebates
368        let mut total_output_sui: u128 = 0;
369
370        // settlement input/output sui is used by the settlement transactions to account for
371        // Sui that has been gathered from the accumulator writes of transactions which it is
372        // settling.
373        total_input_sui += store.execution_results.settlement_input_sui as u128;
374        total_output_sui += store.execution_results.settlement_output_sui as u128;
375
376        for (id, input, output) in get_modified_objects(store) {
377            if let Some(input) = input {
378                total_input_sui +=
379                    get_input_sui(store, &id, input.version, layout_resolver)? as u128;
380            }
381            if let Some(object) = output {
382                total_output_sui += object.get_total_sui(layout_resolver).map_err(|e| {
383                    make_invariant_violation!(
384                        "Failed looking up output SUI in SUI conservation checking for \
385                         mutated type {:?}: {e:#?}",
386                        object.struct_tag(),
387                    )
388                })? as u128;
389            }
390        }
391
392        for event in &store.execution_results.accumulator_events {
393            let (input, output) = event.total_sui_in_event();
394            total_input_sui += input as u128;
395            total_output_sui += output as u128;
396        }
397
398        // note: storage_cost flows into the storage_rebate field of the output objects, which is
399        // why it is not accounted for here.
400        // similarly, all of the storage_rebate *except* the storage_fund_rebate_inflow
401        // gets credited to the gas coin both computation costs and storage rebate inflow are
402        total_output_sui +=
403            gas_summary.computation_cost as u128 + gas_summary.non_refundable_storage_fee as u128;
404        if let Some((epoch_fees, epoch_rebates)) = advance_epoch_gas_summary {
405            total_input_sui += epoch_fees as u128;
406            total_output_sui += epoch_rebates as u128;
407        }
408        if total_input_sui != total_output_sui {
409            return Err(ExecutionError::invariant_violation(format!(
410                "SUI conservation failed: input={}, output={}, \
411                    this transaction either mints or burns SUI",
412                total_input_sui, total_output_sui,
413            )));
414        }
415        Ok(())
416    }
417}

419type ModifiedObjectInfo<'a> = (
420    ObjectID,
421    // old object metadata, including version, digest, owner, and storage rebate.
422    Option<DynamicallyLoadedObjectMetadata>,
423    Option<&'a Object>,
424);
425
426/// Return the list of all modified objects, for each object, returns
427/// - Object ID,
428/// - Input: If the object existed prior to this transaction, include their version and storage_rebate,
429/// - Output: If a new version of the object is written, include the new object.
430fn get_modified_objects<'a>(store: &'a TemporaryStore<'_>) -> Vec<ModifiedObjectInfo<'a>> {
431    store
432        .execution_results
433        .modified_objects
434        .iter()
435        .map(|id| {
436            let metadata = store.get_object_modified_at(id);
437            let output = store.execution_results.written_objects.get(id);
438            (*id, metadata, output)
439        })
440        .chain(
441            store
442                .execution_results
443                .written_objects
444                .iter()
445                .filter_map(|(id, object)| {
446                    if store.execution_results.modified_objects.contains(id) {
447                        None
448                    } else {
449                        Some((*id, None, Some(object)))
450                    }
451                }),
452        )
453        .collect()
454}
455
456fn get_input_sui(
457    store: &TemporaryStore<'_>,
458    id: &ObjectID,
459    expected_version: SequenceNumber,
460    layout_resolver: &mut impl LayoutResolver,
461) -> Result<u64, ExecutionError> {
462    if let Some(obj) = store.input_objects.get(id) {
463        // the assumption here is that if it is in the input objects must be the right one
464        if obj.version() != expected_version {
465            invariant_violation!(
466                "Version mismatching when resolving input object to check conservation--\
467                 expected {}, got {}",
468                expected_version,
469                obj.version(),
470            );
471        }
472        obj.get_total_sui(layout_resolver).map_err(|e| {
473            make_invariant_violation!(
474                "Failed looking up input SUI in SUI conservation checking for input with \
475                     type {:?}: {e:#?}",
476                obj.struct_tag(),
477            )
478        })
479    } else {
480        // not in input objects, must be a dynamic field
481        let Some(obj) = store.store.get_object_by_key(id, expected_version) else {
482            invariant_violation!(
483                "Failed looking up dynamic field {id} in SUI conservation checking"
484            );
485        };
486        obj.get_total_sui(layout_resolver).map_err(|e| {
487            make_invariant_violation!(
488                "Failed looking up input SUI in SUI conservation checking for type \
489                     {:?}: {e:#?}",
490                obj.struct_tag(),
491            )
492        })
493    }
494}

496/// Compute the per-`(address, type)` funds-accumulator reservation budget authorized by the
497/// transaction. Today every funds accumulator is a `Balance<T>`, but the `(address, TypeTag)`
498/// keying lets this generalize as more accumulator types are added. Sources:
499/// - PTB `FundsWithdrawalArg`s for any supported accumulator type (sender or sponsor as owner).
500/// - Gas paid entirely from address balance (credits `(gas_owner, Balance<SUI>)`).
501/// - Gas-data entries with coin-reservation digests (also credit `(gas_owner, Balance<SUI>)`).
502fn compute_input_reservations(
503    transaction_kind: &TransactionKind,
504    gas_data: &GasData,
505    transaction_signer: SuiAddress,
506) -> BTreeMap<(SuiAddress, TypeTag), u64> {
507    use sui_types::balance::Balance;
508    use sui_types::gas_coin::GAS;
509    use sui_types::transaction::{Reservation, WithdrawFrom, is_gas_paid_from_address_balance};
510
511    let mut reservations: BTreeMap<(SuiAddress, TypeTag), u64> = BTreeMap::new();
512    let sui_balance_type = Balance::type_tag(GAS::type_tag());
513
514    for arg in transaction_kind.get_funds_withdrawals() {
515        let owner = match arg.withdraw_from {
516            WithdrawFrom::Sender => transaction_signer,
517            WithdrawFrom::Sponsor => gas_data.owner,
518        };
519        let Reservation::MaxAmountU64(reservation) = arg.reservation;
520        *reservations
521            .entry((owner, arg.type_arg.to_type_tag()))
522            .or_insert(0) += reservation;
523    }
524
525    if is_gas_paid_from_address_balance(gas_data, transaction_kind) {
526        *reservations
527            .entry((gas_data.owner, sui_balance_type.clone()))
528            .or_insert(0) += gas_data.budget;
529    }
530
531    for entry in &gas_data.payment {
532        if let Ok(parsed) = ParsedDigest::try_from(entry.2) {
533            *reservations
534                .entry((gas_data.owner, sui_balance_type.clone()))
535                .or_insert(0) += parsed.reservation_amount();
536        }
537    }
538
539    reservations
540}

542impl InvariantChecker {
543    /// Run the SUI-conservation and balance-accumulator invariant checks against the
544    /// (already-finalized, gas-charged) `store`. Read-only: the caller (the execution engine's
545    /// `run_conservation_checks`) owns any recovery that mutates state.
546    ///
547    /// Returns `Ok(())` when the checks are not applicable: the genesis transaction mints the SUI
548    /// supply, and dev-inspect mode is allowed to violate conservation. The transaction-derived
549    /// inputs the checks need (reservation budget, advance-epoch mint/burn, genesis flag) were
550    /// cached up front by [`Self::set_transaction_inputs`].
551    pub(crate) fn check_conservation_invariants<Mode: ExecutionMode>(
552        &self,
553        store: &TemporaryStore<'_>,
554        move_vm: &Arc<MoveRuntime>,
555        enable_expensive_checks: bool,
556        cost_summary: &GasCostSummary,
557    ) -> Result<(), ExecutionError> {
558        if self.inputs.is_genesis || Mode::skip_conservation_checks() {
559            return Ok(());
560        }
561        let simple_conservation_checks = store.protocol_config().simple_conservation_checks();
562        self.check_sui_conserved(store, simple_conservation_checks, cost_summary)
563            .and_then(|()| {
564                if enable_expensive_checks {
565                    let mut layout_resolver =
566                        TypeLayoutResolver::new(move_vm, store.protocol_config(), Box::new(store));
567                    self.check_sui_conserved_expensive(store, cost_summary, &mut layout_resolver)
568                } else {
569                    Ok(())
570                }
571            })
572            .and_then(|()| self.check_address_balance_changes(store))
573    }

575    // check that every object read is owned directly or indirectly by sender, sponsor,
576    // or a shared object input
577    pub(crate) fn check_ownership_invariants(
578        &self,
579        store: &TemporaryStore<'_>,
580        sender: &SuiAddress,
581        sponsor: &Option<SuiAddress>,
582        gas_charger: &GasCharger,
583        mutable_inputs: &HashSet<ObjectID>,
584        is_epoch_change: bool,
585    ) -> SuiResult<()> {
586        // The funds-accumulator reservation budget is shared with the conservation checks; see
587        // `Self::check_address_balance_changes`.
588        let input_reservations = self.input_reservations();
589        let gas_objs: HashSet<&ObjectID> = gas_charger.used_coins().map(|g| &g.0).collect();
590        let gas_owner = sponsor.as_ref().unwrap_or(sender);
591
592        // mark input objects as authenticated
593        let objects_authenticated_for_mutation: HashSet<SuiAddress> = store
594            .input_objects
595            .iter()
596            .filter_map(|(id, obj)| {
597                match &obj.owner {
598                    Owner::AddressOwner(a) => {
599                        if gas_objs.contains(id) {
600                            // gas object must be owned by sender or sponsor
601                            assert!(
602                                a == gas_owner,
603                                "Gas object must be owned by sender or sponsor"
604                            );
605                        } else {
606                            assert!(sender == a, "Input object must be owned by sender");
607                        }
608                        Some(id)
609                    }
610                    Owner::Shared { .. } | Owner::ConsensusAddressOwner { .. } => Some(id),
611                    Owner::Immutable => {
612                        // object is authenticated, but it cannot own other objects,
613                        // so we should not add it to `authenticated_objs`
614                        // However, we would definitely want to add immutable objects
615                        // to the set of authenticated roots if we were doing runtime
616                        // checks inside the VM instead of after-the-fact in the temporary
617                        // store. Here, we choose not to add them because this will catch a
618                        // bug where we mutate or delete an object that belongs to an immutable
619                        // object (though it will show up somewhat opaquely as an authentication
620                        // failure), whereas adding the immutable object to the roots will prevent
621                        // us from catching this.
622                        None
623                    }
624                    Owner::Party { permissions, .. } => {
625                        let sender_permissions = permissions.permissions_for(sender);
626                        let sponsor_permissions = sponsor
627                            .as_ref()
628                            .map(|s| permissions.permissions_for(s))
629                            .unwrap_or(ObjectPermissions::NONE);
630                        (sender_permissions | sponsor_permissions)
631                            .can_use_mutably()
632                            .then_some(id)
633                    }
634                    Owner::ObjectOwner(_parent) => {
635                        unreachable!(
636                            "Input objects must be address owned, shared, consensus, or immutable"
637                        )
638                    }
639                }
640            })
641            .filter(|id| {
642                // remove any non-mutable inputs. This will remove deleted or readonly shared
643                // objects
644                mutable_inputs.contains(id)
645            })
646            .copied()
647            // Add any object IDs generated in the object runtime during execution to the
648            // authenticated set (i.e., new (non-package) objects, and possibly ephemeral UIDs).
649            .chain(store.generated_runtime_ids.iter().copied())
650            .map(SuiAddress::from)
651            .collect();
652
653        // Add sender and sponsor (if present) to authenticated set
654        let mut authenticated_for_mutation = {
655            assert!(
656                !objects_authenticated_for_mutation.contains(sender),
657                "Sender cannot be an object"
658            );
659            assert!(
660                sponsor
661                    .is_none_or(|sponsor| !objects_authenticated_for_mutation.contains(&sponsor)),
662                "Sponsor cannot be an object"
663            );
664            let mut s = objects_authenticated_for_mutation.clone();
665            s.insert(*sender);
666            if let Some(sponsor) = sponsor {
667                s.insert(*sponsor);
668            }
669            s
670        };
671
672        // check all modified objects are authenticated
673        let mut objects_to_authenticate = store
674            .execution_results
675            .modified_objects
676            .iter()
677            .copied()
678            .collect::<Vec<_>>();
679
680        while let Some(to_authenticate) = objects_to_authenticate.pop() {
681            if authenticated_for_mutation.contains(&to_authenticate.into()) {
682                // object has already been authenticated
683                continue;
684            }
685
686            let parent = if let Some(container_id) =
687                store.wrapped_object_containers.get(&to_authenticate)
688            {
689                // It's a wrapped object, so check that the container is authenticated
690                *container_id
691            } else {
692                // It's non-wrapped, so check the owner -- we can load the object from the
693                // store.
694                let Some(old_obj) = store.store.get_object(&to_authenticate) else {
695                    panic!(
696                        "Failed to load object {to_authenticate:?}.\n \
697                         If it cannot be loaded, we would expect it to be in the wrapped object map: {:#?}",
698                        &store.wrapped_object_containers
699                    )
700                };
701
702                match &old_obj.owner {
703                    // We mutated a dynamic field, we can continue to trace this back to verify
704                    // proper ownership.
705                    Owner::ObjectOwner(parent) => ObjectID::from(*parent),
706                    // We mutated an address owned or sequenced address owned object -- one of two cases apply:
707                    // 1) the object is owned by an object or address in the authenticated set,
708                    // 2) the object is owned by some other address, in which case we should
709                    //    continue to trace this back.
710                    Owner::AddressOwner(parent)
711                    | Owner::ConsensusAddressOwner { owner: parent, .. } => {
712                        // For Receiving<_> objects, the address owner is actually an object.
713                        // If it was actually an address, we should have caught it as an input and
714                        // it would already have been in authenticated_for_mutation
715                        ObjectID::from(*parent)
716                    }
717                    // We mutated a shared object -- we checked if this object was in the
718                    // authenticated set at the top of this loop and it wasn't so this is a failure.
719                    owner @ Owner::Shared { .. } | owner @ Owner::Party { .. } => {
720                        panic!(
721                            "Unauthenticated root at {to_authenticate:?} with owner {owner:?}\n\
722                             Potentially covering objects in: {authenticated_for_mutation:#?}"
723                        );
724                    }
725
726                    Owner::Immutable => {
727                        assert!(
728                            is_epoch_change,
729                            "Immutable objects cannot be written, except for \
730                             Sui Framework/Move stdlib upgrades at epoch change boundaries"
731                        );
732                        // Note: this assumes that the only immutable objects an epoch change
733                        // tx can update are system packages,
734                        // but in principle we could allow others.
735                        assert!(
736                            is_system_package(to_authenticate),
737                            "Only system packages can be upgraded"
738                        );
739                        continue;
740                    }
741                }
742            };
743
744            // we now assume the object is authenticated and check the parent
745            authenticated_for_mutation.insert(to_authenticate.into());
746            objects_to_authenticate.push(parent);
747        }
748
749        // Check that all funds accumulator splits are authorized
750        let sui_balance_type =
751            sui_types::balance::Balance::type_tag(sui_types::gas_coin::GAS::type_tag());
752        let gas_payment_address_balance =
753            gas_charger
754                .gas_payment_location()
755                .and_then(|location| match location {
756                    PaymentLocation::Coin(_) => None,
757                    PaymentLocation::AddressBalance(address) => Some(address),
758                });
759        let mut funds_net_changes: BTreeMap<(SuiAddress, TypeTag), i128> = BTreeMap::new();
760        for event in store.execution_results.accumulator_events.iter() {
761            let amount = match event.write.value {
762                AccumulatorValue::Integer(a) => a as i128,
763                AccumulatorValue::IntegerTuple(_, _) | AccumulatorValue::EventDigest(_) => {
764                    assert!(
765                        !sui_types::balance::Balance::is_balance_type(&event.write.address.ty),
766                        "Non-integer accumulator changes should not be balances"
767                    );
768                    continue;
769                }
770            };
771            let signed = match event.write.operation {
772                AccumulatorOperation::Split => -amount,
773                AccumulatorOperation::Merge => amount,
774            };
775            let address = event.write.address.address;
776            let type_tag = &event.write.address.ty;
777            let key = (address, type_tag.clone());
778            *funds_net_changes.entry(key.clone()).or_insert(0) += signed;
779            // Authorized if it is:
780            // - A merge/deposit (anyone can deposit)
781            // - A withdrawal
782            //   - with a corresponding input reservation
783            //   - from an object authenticated for mutation
784            //   - for the gas payment (potentially from a GasCoin send_funds transfer)
785            let authorized = match event.write.operation {
786                AccumulatorOperation::Merge => true,
787                AccumulatorOperation::Split => {
788                    input_reservations.contains_key(&key)
789                        || objects_authenticated_for_mutation.contains(&address)
790                        || (*type_tag == sui_balance_type
791                            && gas_payment_address_balance
792                                .is_some_and(|gas_addr| gas_addr == address))
793                }
794            };
795            assert!(
796                authorized,
797                "Unauthenticated funds-accumulator Split at address {address} for type \
798                 {type_tag}: no input reservation, address is not an authenticated object, and \
799                 it is not the final gas payment address balance"
800            );
801        }
802
803        // For all net negative changes (net withdrawals), the net changes _must_ be less than the
804        // reservation amount, or it must be from an object. This excludes the final gas payment
805        // address since the case where that withdrawal is allowed should be a net positive (or
806        // zero) since it occurs only in the case where the gas coin is transferred via send_funds
807        for (key, change) in funds_net_changes {
808            // skip if deposit or for an object
809            if change >= 0 || objects_authenticated_for_mutation.contains(&key.0) {
810                continue;
811            }
812            let reservation = input_reservations.get(&key).copied().unwrap_or(0) as u128;
813            let withdrawn = change.unsigned_abs();
814            assert!(
815                withdrawn <= reservation,
816                "Net withdrawal of {withdrawn} for {key:?} exceeds input reservation of \
817                 {reservation}"
818            );
819        }
820
821        Ok(())
822    }
