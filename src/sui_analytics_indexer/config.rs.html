94/// Main configuration for an analytics indexer job.
95#[derive(Debug, Clone, Serialize, Deserialize)]
96pub struct IndexerConfig {
97    /// The url of the metrics client to connect to.
98    #[serde(default = "default_client_metric_host")]
99    pub client_metric_host: String,
100    /// The port of the metrics client to connect to.
101    #[serde(default = "default_client_metric_port")]
102    pub client_metric_port: u16,
103    /// Output object store configuration
104    pub output_store: OutputStoreConfig,
105    /// Remote store URL.
106    pub remote_store_url: Option<String>,
107    /// Optional streaming URL for real-time indexing
108    pub streaming_url: Option<String>,
109    /// RPC API URL for request/reply from full node.
110    pub rpc_api_url: String,
111    /// Optional RPC username
112    pub rpc_username: Option<String>,
113    /// Optional RPC password
114    pub rpc_password: Option<String>,
115    /// Optional working directory for temporary files (defaults to system temp dir)
116    pub work_dir: Option<PathBuf>,
117    /// Optional local ingestion path for reading checkpoints from disk instead of remote
118    pub local_ingestion_path: Option<PathBuf>,
119    pub sf_account_identifier: Option<String>,
120    pub sf_warehouse: Option<String>,
121    pub sf_database: Option<String>,
122    pub sf_schema: Option<String>,
123    pub sf_username: Option<String>,
124    pub sf_role: Option<String>,
125    pub sf_password_file: Option<String>,
126
127    /// Migration mode identifier. When set, the indexer operates in migration mode:
128    /// - Overwrites existing files matching target checkpoint ranges
129    /// - Uses conditional PUT with etag to prevent concurrent modification
130    /// - Uses per-file metadata to track migration progress separately from main pipeline
131    #[serde(default)]
132    pub migration_id: Option<String>,
133
134    /// File format for output files (csv or parquet).
135    #[serde(default = "default_file_format")]
136    pub file_format: FileFormat,
137
138    #[serde(rename = "pipelines")]
139    pub pipeline_configs: Vec<PipelineConfig>,
140
141    #[serde(default)]
142    pub ingestion: IngestionConfig,
143
144    #[serde(default)]
145    pub sequential: SequentialConfig,
146
147    pub first_checkpoint: Option<u64>,
148    pub last_checkpoint: Option<u64>,
149
150    /// Maximum serialized files waiting in upload queue per pipeline.
151    /// When the queue is full, serialization blocks until uploads complete.
152    #[serde(default = "default_max_pending_uploads")]
153    pub max_pending_uploads: usize,
154
155    /// Maximum concurrent serialization tasks per pipeline.
156    /// Limits CPU usage from parallel parquet/csv encoding.
157    #[serde(default = "default_max_concurrent_serialization")]
158    pub max_concurrent_serialization: usize,
159
160    /// Minimum interval between watermark writes to object store (seconds).
161    /// Watermarks are updated after file uploads; this rate-limits those writes.
162    /// Default: 60 seconds.
163    #[serde(default = "default_watermark_update_interval_secs")]
164    pub watermark_update_interval_secs: u64,
165}
166
167impl IndexerConfig {
168    /// Validate the indexer configuration.
169    ///
170    /// Checks for:
171    /// - Duplicate pipeline types (each pipeline can only be configured once)
172    /// - Individual pipeline config validity (e.g., batch_size required in live mode)
173    pub fn validate(&self) -> anyhow::Result<()> {
174        // Check for duplicate pipeline types
175        let mut seen = std::collections::HashSet::new();
176        for config in &self.pipeline_configs {
177            let name = config.pipeline.name();
178            if !seen.insert(name) {
179                anyhow::bail!(
180                    "Duplicate pipeline type '{}' in config. Each pipeline type can only be configured once.",
181                    name
182                );
183            }
184        }
185
186        // Validate individual pipeline configs
187        let is_migration_mode = self.migration_id.is_some();
188        for config in &self.pipeline_configs {
189            config.validate(is_migration_mode)?;
190        }
191
192        Ok(())
193    }
194
195    pub fn pipeline_configs(&self) -> &[PipelineConfig] {
196        &self.pipeline_configs
197    }
198
199    pub fn get_pipeline_config(&self, name: &str) -> Option<&PipelineConfig> {
200        self.pipeline_configs
201            .iter()
202            .find(|p| p.pipeline.name() == name)
203    }
204}

206/// Batch size configuration for when to write files.
207#[derive(Debug, Clone, Copy, Serialize, Deserialize)]
208#[serde(rename_all = "snake_case")]
209pub enum BatchSizeConfig {
210    /// Write a file after accumulating this many checkpoints.
211    Checkpoints(usize),
212    /// Write a file after accumulating this many rows.
213    Rows(usize),
214}
215
216impl BatchSizeConfig {
217    /// Validate the batch size configuration.
218    pub fn validate(&self) -> anyhow::Result<()> {
219        match self {
220            BatchSizeConfig::Checkpoints(0) => {
221                anyhow::bail!("batch_size.checkpoints must be > 0")
222            }
223            BatchSizeConfig::Rows(0) => {
224                anyhow::bail!("batch_size.rows must be > 0")
225            }
226            _ => Ok(()),
227        }
228    }
229}

231/// Configuration for a single analytics task/pipeline.
232#[derive(Debug, Clone, Serialize, Deserialize)]
233pub struct PipelineConfig {
234    /// Type of data to write i.e. checkpoint, object, transaction, etc
235    pub pipeline: Pipeline,
236    /// File format to use (csv or parquet)
237    #[serde(default = "default_file_format")]
238    pub file_format: FileFormat,
239    pub package_id_filter: Option<String>,
240    /// Snowflake table to monitor
241    pub sf_table_id: Option<String>,
242    /// Snowflake column containing checkpoint numbers
243    pub sf_checkpoint_col_id: Option<String>,
244    /// Whether to report max checkpoint from Snowflake table
245    #[serde(default)]
246    pub report_sf_max_table_checkpoint: bool,
247    /// Batch size configuration - determines when to write files.
248    /// Required for live mode (when top-level migration_id is None).
249    /// Ignored in migration mode (file boundaries come from existing files).
250    #[serde(default)]
251    pub batch_size: Option<BatchSizeConfig>,
252    /// Override the output path prefix. Defaults to the pipeline name.
253    #[serde(default)]
254    pub output_prefix: Option<String>,
255    /// Force a batch cut after this many seconds, even if size thresholds aren't met.
256    /// Default: 600 (10 minutes).
257    #[serde(default = "default_force_batch_cut_after_secs")]
258    pub force_batch_cut_after_secs: u64,
259}
260
261impl PipelineConfig {
262    /// Validate the configuration.
263    ///
264    /// Returns an error if batch_size is required but not set, or if batch_size is invalid.
265    /// In migration mode, batch_size is not required since file boundaries
266    /// come from existing files.
267    pub fn validate(&self, is_migration_mode: bool) -> anyhow::Result<()> {
268        if !is_migration_mode {
269            match &self.batch_size {
270                None => anyhow::bail!(
271                    "batch_size is required for pipeline '{}' (not in migration mode)",
272                    self.pipeline
273                ),
274                Some(batch_size) => batch_size.validate()?,
275            }
276        }
277        Ok(())
278    }
279
280    /// Get the output path prefix for this pipeline.
281    ///
282    /// Returns the configured `output_prefix` if set, otherwise the pipeline's default path.
283    pub fn output_prefix(&self) -> &str {
284        self.output_prefix
285            .as_deref()
286            .unwrap_or_else(|| self.pipeline.default_path())
287    }
