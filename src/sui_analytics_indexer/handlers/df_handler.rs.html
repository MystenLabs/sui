29#[derive(Clone)]
30pub struct DynamicFieldHandler {
31    package_cache: Arc<PackageCache>,
32}
33
34impl DynamicFieldHandler {
35    pub fn new(package_cache: Arc<PackageCache>) -> Self {
36        Self { package_cache }
37    }
38    async fn process_dynamic_field(
39        &self,
40        epoch: u64,
41        checkpoint: u64,
42        timestamp_ms: u64,
43        object: &Object,
44        all_written_objects: &HashMap<ObjectID, Object>,
45    ) -> Result<Option<DynamicFieldEntry>> {
46        let move_obj_opt = object.data.try_as_move();
47        let Some(move_object) = move_obj_opt else {
48            return Ok(None);
49        };
50        if !move_object.type_().is_dynamic_field() {
51            return Ok(None);
52        }
53
54        let layout = self
55            .package_cache
56            .resolver_for_epoch(epoch)
57            .type_layout(move_object.type_().clone().into())
58            .await?;
59        let object_id = object.id();
60
61        let field = DFV::FieldVisitor::deserialize(move_object.contents(), &layout)?;
62
63        let type_ = field.kind;
64        let name_type: TypeTag = field.name_layout.into();
65        let bcs_name = field.name_bytes.to_owned();
66
67        let name_value = BoundedVisitor::deserialize_value(field.name_bytes, field.name_layout)
68            .tap_err(|e| {
69                warn!("{e}");
70            })?;
71        let name = DynamicFieldName {
72            type_: name_type,
73            value: SuiMoveValue::from(name_value).to_json_value(),
74        };
75        let name_json = serde_json::to_string(&name)?;
76        let (_owner_type, owner_id) = owner_to_owner_info(&object.owner);
77        let Some(parent_id) = owner_id else {
78            return Ok(None);
79        };
80        let entry = match type_ {
81            DynamicFieldType::DynamicField => DynamicFieldEntry {
82                parent_object_id: parent_id.to_string(),
83                transaction_digest: object.previous_transaction.base58_encode(),
84                checkpoint,
85                epoch,
86                timestamp_ms,
87                name: name_json,
88                bcs_name: Base64::encode(bcs_name),
89                type_,
90                object_id: object.id().to_string(),
91                version: object.version().value(),
92                digest: object.digest().to_string(),
93                object_type: move_object.clone().into_type().into_type_params()[1]
94                    .to_canonical_string(/* with_prefix */ true),
95            },
96            DynamicFieldType::DynamicObject => {
97                let object = all_written_objects.get(&object_id).ok_or(
98                    IndexerError::UncategorizedError(anyhow::anyhow!(
99                        "Failed to find object_id {:?} when trying to create dynamic field info",
100                        object_id
101                    )),
102                )?;
103                let version = object.version().value();
104                let digest = object.digest().to_string();
105                let object_type = object.data.type_().unwrap().clone();
106                DynamicFieldEntry {
107                    parent_object_id: parent_id.to_string(),
108                    transaction_digest: object.previous_transaction.base58_encode(),
109                    checkpoint,
110                    epoch,
111                    timestamp_ms,
112                    name: name_json,
113                    bcs_name: Base64::encode(bcs_name),
114                    type_,
115                    object_id: object.id().to_string(),
116                    digest,
117                    version,
118                    object_type: object_type.to_canonical_string(true),
119                }
120            }
121        };
122        Ok(Some(entry))
123    }
124}

126#[async_trait::async_trait]
127impl AnalyticsHandler<DynamicFieldEntry> for DynamicFieldHandler {
128    async fn process_checkpoint(
129        &self,
130        checkpoint_data: &Arc<CheckpointData>,
131    ) -> Result<Box<dyn Iterator<Item = DynamicFieldEntry> + Send + Sync>> {
132        wait_for_cache(checkpoint_data, &self.package_cache).await;
133        process_transactions(checkpoint_data.clone(), Arc::new(self.clone())).await
134    }
135
136    fn file_type(&self) -> Result<FileType> {
137        Ok(FileType::DynamicField)
138    }
139
140    fn name(&self) -> &'static str {
141        "dynamic_field"
142    }
143}

145#[async_trait::async_trait]
146impl TransactionProcessor<DynamicFieldEntry> for DynamicFieldHandler {
147    async fn process_transaction(
148        &self,
149        tx_idx: usize,
150        checkpoint: &CheckpointData,
151    ) -> Result<Box<dyn Iterator<Item = DynamicFieldEntry> + Send + Sync>> {
152        let checkpoint_transaction = &checkpoint.transactions[tx_idx];
153        for object in checkpoint_transaction.output_objects.iter() {
154            self.package_cache.update(object)?;
155        }
156
157        let all_objects: HashMap<_, _> = checkpoint_transaction
158            .output_objects
159            .iter()
160            .map(|x| (x.id(), x.clone()))
161            .collect();
162
163        let mut entries = Vec::new();
164        for object in checkpoint_transaction.output_objects.iter() {
165            if let Some(entry) = self
166                .process_dynamic_field(
167                    checkpoint.checkpoint_summary.epoch,
168                    checkpoint.checkpoint_summary.sequence_number,
169                    checkpoint.checkpoint_summary.timestamp_ms,
170                    object,
171                    &all_objects,
172                )
173                .await?
174            {
175                entries.push(entry);
176            }
177        }
178
179        Ok(Box::new(entries.into_iter()))
180    }
