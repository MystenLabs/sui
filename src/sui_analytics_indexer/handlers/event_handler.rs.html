19#[derive(Clone)]
20pub struct EventHandler {
21    package_cache: Arc<PackageCache>,
22}
23
24impl EventHandler {
25    pub fn new(package_cache: Arc<PackageCache>) -> Self {
26        Self { package_cache }
27    }
28}
29
30#[async_trait::async_trait]
31impl AnalyticsHandler<EventEntry> for EventHandler {
32    async fn process_checkpoint(
33        &self,
34        checkpoint_data: &Arc<CheckpointData>,
35    ) -> Result<Box<dyn Iterator<Item = EventEntry> + Send + Sync>> {
36        wait_for_cache(checkpoint_data, &self.package_cache).await;
37        process_transactions(checkpoint_data.clone(), Arc::new(self.clone())).await
38    }
39
40    fn file_type(&self) -> Result<FileType> {
41        Ok(FileType::Event)
42    }
43
44    fn name(&self) -> &'static str {
45        "event"
46    }
47}
48
49#[async_trait::async_trait]
50impl TransactionProcessor<EventEntry> for EventHandler {
51    async fn process_transaction(
52        &self,
53        tx_idx: usize,
54        checkpoint: &CheckpointData,
55    ) -> Result<Box<dyn Iterator<Item = EventEntry> + Send + Sync>> {
56        let transaction = &checkpoint.transactions[tx_idx];
57        if let Some(events) = &transaction.events {
58            let epoch = checkpoint.checkpoint_summary.epoch;
59            let checkpoint_seq = checkpoint.checkpoint_summary.sequence_number;
60            let timestamp_ms = checkpoint.checkpoint_summary.timestamp_ms;
61            let digest = transaction.transaction.digest();
62
63            let mut entries = Vec::new();
64            for (idx, event) in events.data.iter().enumerate() {
65                let Event {
66                    package_id,
67                    transaction_module,
68                    sender,
69                    type_,
70                    contents,
71                } = event;
72                let layout = self
73                    .package_cache
74                    .resolver_for_epoch(epoch)
75                    .type_layout(move_core_types::language_storage::TypeTag::Struct(
76                        Box::new(type_.clone()),
77                    ))
78                    .await?;
79                let move_value = MoveValue::simple_deserialize(contents, &layout)?;
80                let (_, event_json) = type_and_fields_from_move_event_data(move_value)?;
81                let entry = EventEntry {
82                    transaction_digest: digest.base58_encode(),
83                    event_index: idx as u64,
84                    checkpoint: checkpoint_seq,
85                    epoch,
86                    timestamp_ms,
87                    sender: sender.to_string(),
88                    package: package_id.to_string(),
89                    module: transaction_module.to_string(),
90                    event_type: type_.to_string(),
91                    bcs: "".to_string(),
92                    bcs_length: contents.len() as u64,
93                    event_json: event_json.to_string(),
94                };
95
96                entries.push(entry);
97            }
98            Ok(Box::new(entries.into_iter()))
99        } else {
100            Ok(Box::new(std::iter::empty()))
101        }
102    }
103}
