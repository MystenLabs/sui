26#[derive(Clone)]
27pub struct ObjectHandler {
28    package_filter: Option<ObjectID>,
29    metrics: AnalyticsMetrics,
30    package_cache: Arc<PackageCache>,
31}

33impl ObjectHandler {
34    pub fn new(
35        package_cache: Arc<PackageCache>,
36        package_filter: &Option<String>,
37        metrics: AnalyticsMetrics,
38    ) -> Self {
39        Self {
40            package_filter: package_filter
41                .clone()
42                .map(|x| ObjectID::from_hex_literal(&x).unwrap()),
43            metrics,
44            package_cache,
45        }
46    }
47
48    async fn check_type_hierarchy(
49        &self,
50        type_tag: &TypeTag,
51        original_package_id: ObjectID,
52    ) -> Result<bool> {
53        use std::collections::BTreeSet;
54        use tokio::task::JoinSet;
55
56        // Collect all package IDs using stack-based traversal
57        let mut types = vec![type_tag];
58        let mut package_ids = BTreeSet::new();
59
60        while let Some(type_) = types.pop() {
61            match type_ {
62                TypeTag::Struct(s) => {
63                    package_ids.insert(s.address);
64                    types.extend(s.type_params.iter());
65                }
66                TypeTag::Vector(inner) => types.push(inner.as_ref()),
67                _ => {} // Primitive types can't match a package ID
68            }
69        }
70
71        // Resolve original package IDs in parallel
72        let mut original_ids = JoinSet::new();
73
74        for id in package_ids {
75            let package_cache = self.package_cache.clone();
76            original_ids.spawn(async move { package_cache.get_original_package_id(id).await });
77        }
78
79        // Check if any resolved ID matches our target
80        while let Some(result) = original_ids.join_next().await {
81            if result?? == original_package_id {
82                return Ok(true);
83            }
84        }
85
86        Ok(false)
87    }

89    // Object data. Only called if there are objects in the transaction.
90    // Responsible to build the live object table.
91    async fn process_object(
92        &self,
93        epoch: u64,
94        checkpoint: u64,
95        timestamp_ms: u64,
96        object: &Object,
97        object_status_tracker: &ObjectStatusTracker,
98    ) -> Result<Option<ObjectEntry>> {
99        let move_obj_opt = object.data.try_as_move();
100        let has_public_transfer = move_obj_opt
101            .map(|o| o.has_public_transfer())
102            .unwrap_or(false);
103        let move_struct = if let Some((tag, contents)) = object
104            .struct_tag()
105            .and_then(|tag| object.data.try_as_move().map(|mo| (tag, mo.contents())))
106        {
107            match get_move_struct(
108                &tag,
109                contents,
110                &self.package_cache.resolver_for_epoch(epoch),
111            )
112            .await
113            {
114                Ok(move_struct) => Some(move_struct),
115                Err(err)
116                    if err
117                        .downcast_ref::<sui_types::object::bounded_visitor::Error>()
118                        .filter(|e| {
119                            matches!(e, sui_types::object::bounded_visitor::Error::OutOfBudget)
120                        })
121                        .is_some() =>
122                {
123                    self.metrics
124                        .total_too_large_to_deserialize
125                        .with_label_values(&[NAME])
126                        .inc();
127                    tracing::warn!(
128                        "Skipping struct with type {} because it was too large.",
129                        tag
130                    );
131                    None
132                }
133                Err(err) => return Err(err),
134            }
135        } else {
136            None
137        };
138        let (struct_tag, sui_move_struct) = if let Some(move_struct) = move_struct {
139            match move_struct.into() {
140                SuiMoveStruct::WithTypes { type_, fields } => {
141                    (Some(type_), Some(SuiMoveStruct::WithFields(fields)))
142                }
143                fields => (object.struct_tag(), Some(fields)),
144            }
145        } else {
146            (None, None)
147        };
148
149        let object_type = move_obj_opt.map(|o| o.type_());
150
151        let is_match = if let Some(package_id) = self.package_filter {
152            if let Some(object_type) = object_type {
153                let original_package_id = self
154                    .package_cache
155                    .get_original_package_id(package_id.into())
156                    .await?;
157
158                // Check if any type parameter matches the package filter
159                let type_tag: TypeTag = object_type.clone().into();
160                self.check_type_hierarchy(&type_tag, original_package_id)
161                    .await?
162            } else {
163                false
164            }
165        } else {
166            true
167        };
168
169        if !is_match {
170            return Ok(None);
171        }
172
173        let object_id = object.id();
174        let entry = ObjectEntry {
175            object_id: object_id.to_string(),
176            digest: object.digest().to_string(),
177            version: object.version().value(),
178            type_: object_type.map(|t| t.to_string()),
179            checkpoint,
180            epoch,
181            timestamp_ms,
182            owner_type: Some(get_owner_type(object)),
183            owner_address: get_owner_address(object),
184            is_consensus: get_is_consensus(object),
185            object_status: object_status_tracker
186                .get_object_status(&object_id)
187                .expect("Object must be in output objects"),
188            initial_shared_version: initial_shared_version(object),
189            previous_transaction: object.previous_transaction.base58_encode(),
190            has_public_transfer,
191            storage_rebate: Some(object.storage_rebate),
192            bcs: "".to_string(),
193            bcs_length: bcs::to_bytes(object).unwrap().len() as u64,
194            coin_type: object.coin_type_maybe().map(|t| t.to_string()),
195            coin_balance: if object.coin_type_maybe().is_some() {
196                Some(object.get_coin_value_unsafe())
197            } else {
198                None
199            },
200            struct_tag: struct_tag.map(|x| x.to_string()),
201            object_json: sui_move_struct.map(|x| x.to_json_value().to_string()),
202        };
203        Ok(Some(entry))
204    }
205}

207#[async_trait::async_trait]
208impl AnalyticsHandler<ObjectEntry> for ObjectHandler {
209    async fn process_checkpoint(
210        &self,
211        checkpoint_data: &Arc<CheckpointData>,
212    ) -> Result<Box<dyn Iterator<Item = ObjectEntry> + Send + Sync>> {
213        wait_for_cache(checkpoint_data, &self.package_cache).await;
214        process_transactions(checkpoint_data.clone(), Arc::new(self.clone())).await
215    }
216
217    fn file_type(&self) -> Result<FileType> {
218        Ok(FileType::Object)
219    }
220
221    fn name(&self) -> &'static str {
222        NAME
223    }
224}

226#[async_trait::async_trait]
227impl TransactionProcessor<ObjectEntry> for ObjectHandler {
228    async fn process_transaction(
229        &self,
230        tx_idx: usize,
231        checkpoint_data: &CheckpointData,
232    ) -> Result<Box<dyn Iterator<Item = ObjectEntry> + Send + Sync>> {
233        let checkpoint_transaction = &checkpoint_data.transactions[tx_idx];
234
235        for object in checkpoint_transaction.output_objects.iter() {
236            self.package_cache.update(object)?;
237        }
238
239        let epoch = checkpoint_data.checkpoint_summary.epoch;
240        let checkpoint = checkpoint_data.checkpoint_summary.sequence_number;
241        let timestamp_ms = checkpoint_data.checkpoint_summary.timestamp_ms;
242        let effects = &checkpoint_transaction.effects;
243
244        let object_status_tracker = ObjectStatusTracker::new(effects);
245        let mut vec = Vec::new();
246        for object in checkpoint_transaction.output_objects.iter() {
247            if let Some(object_entry) = self
248                .process_object(
249                    epoch,
250                    checkpoint,
251                    timestamp_ms,
252                    object,
253                    &object_status_tracker,
254                )
255                .await?
256            {
257                vec.push(object_entry);
258            }
259        }
260        for (object_ref, _) in effects.all_removed_objects().iter() {
261            let object_entry = ObjectEntry {
262                object_id: object_ref.0.to_string(),
263                digest: object_ref.2.to_string(),
264                version: u64::from(object_ref.1),
265                type_: None,
266                checkpoint,
267                epoch,
268                timestamp_ms,
269                owner_type: None,
270                owner_address: None,
271                object_status: ObjectStatus::Deleted,
272                initial_shared_version: None,
273                previous_transaction: checkpoint_transaction.transaction.digest().base58_encode(),
274                has_public_transfer: false,
275                is_consensus: false,
276                storage_rebate: None,
277                bcs: "".to_string(),
278                coin_type: None,
279                coin_balance: None,
280                struct_tag: None,
281                object_json: None,
282                bcs_length: 0,
283            };
284            vec.push(object_entry);
285        }
286        Ok(Box::new(vec.into_iter()))
287    }
288}

300    fn create_struct_tag(
301        addr: &str,
302        module: &str,
303        name: &str,
304        type_params: Vec<TypeTag>,
305    ) -> TypeTag {
306        TypeTag::Struct(Box::new(StructTag {
307            address: AccountAddress::from_str(addr).unwrap(),
308            module: Identifier::new(module).unwrap(),
309            name: Identifier::new(name).unwrap(),
310            type_params,
311        }))
312    }

314    #[tokio::test]
315    async fn test_check_type_hierarchy() {
316        let temp_dir = tempfile::tempdir().unwrap();
317        let registry = Registry::new();
318        let metrics = AnalyticsMetrics::new(&registry);
319        let package_cache = Arc::new(PackageCache::new(temp_dir.path(), "http://localhost:9000"));
320
321        // Create handler with the necessary context
322        let handler = ObjectHandler::new(package_cache, &Some("0xabc".to_string()), metrics);
323
324        // 1. Direct match
325        let type_tag = create_struct_tag("0xabc", "module", "Type", vec![]);
326        assert!(
327            handler
328                .check_type_hierarchy(&type_tag, ObjectID::from_hex_literal("0xabc").unwrap())
329                .await
330                .unwrap()
331        );
332
333        // 2. Match in type parameter
334        let inner_type = create_struct_tag("0xabc", "module", "Inner", vec![]);
335        let type_tag = create_struct_tag("0xcde", "module", "Type", vec![inner_type]);
336        assert!(
337            handler
338                .check_type_hierarchy(&type_tag, ObjectID::from_hex_literal("0xabc").unwrap())
339                .await
340                .unwrap()
341        );
342
343        // 3. Match in nested vector
344        let inner_type = create_struct_tag("0xabc", "module", "Inner", vec![]);
345        let vector_type = TypeTag::Vector(Box::new(inner_type));
346        let type_tag = create_struct_tag("0xcde", "module", "Type", vec![vector_type]);
347        assert!(
348            handler
349                .check_type_hierarchy(&type_tag, ObjectID::from_hex_literal("0xabc").unwrap())
350                .await
351                .unwrap()
352        );
353
354        // 4. No match
355        let type_tag = create_struct_tag("0xcde", "module", "Type", vec![]);
356        assert!(
357            !handler
358                .check_type_hierarchy(&type_tag, ObjectID::from_hex_literal("0xabc").unwrap())
359                .await
360                .unwrap()
361        );
362
363        // 5. Primitive type
364        let type_tag = TypeTag::U64;
365        assert!(
366            !handler
367                .check_type_hierarchy(&type_tag, ObjectID::from_hex_literal("0xabc").unwrap())
368                .await
369                .unwrap()
370        );
371    }
