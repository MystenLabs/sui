48#[async_trait::async_trait]
49impl TransactionProcessor<TransactionEntry> for TransactionHandler {
50    async fn process_transaction(
51        &self,
52        tx_idx: usize,
53        checkpoint: &CheckpointData,
54    ) -> Result<Box<dyn Iterator<Item = TransactionEntry> + Send + Sync>> {
55        let transaction = &checkpoint.transactions[tx_idx];
56        let epoch = checkpoint.checkpoint_summary.epoch;
57        let checkpoint_seq = checkpoint.checkpoint_summary.sequence_number;
58        let timestamp_ms = checkpoint.checkpoint_summary.timestamp_ms;
59
60        let transaction_positions = compute_transaction_positions(&checkpoint.checkpoint_contents);
61
62        let checkpoint_transaction = transaction;
63        let effects = &transaction.effects;
64
65        let transaction = &checkpoint_transaction.transaction;
66        let txn_data = transaction.transaction_data();
67        let gas_object = effects.gas_object();
68        let gas_summary = effects.gas_cost_summary();
69        let move_calls_vec = txn_data.move_calls();
70        let packages: BTreeSet<_> = move_calls_vec
71            .iter()
72            .map(|(package, _, _)| package.to_canonical_string(/* with_prefix */ false))
73            .collect();
74        let packages = packages
75            .iter()
76            .map(|s| s.as_str())
77            .collect::<Vec<_>>()
78            .join("-");
79        let transaction_digest = transaction.digest().base58_encode();
80        let events_digest = checkpoint_transaction
81            .events
82            .as_ref()
83            .map(|events| events.digest().base58_encode());
84
85        let transaction_position = *transaction_positions
86            .get(transaction.digest())
87            .expect("Expect transaction to exist in checkpoint_contents.")
88            as u64;
89
90        let transaction_data_bcs_length = bcs::to_bytes(&txn_data).unwrap().len() as u64;
91        let effects_bcs_length = bcs::to_bytes(&checkpoint_transaction.effects)
92            .unwrap()
93            .len() as u64;
94        let events_bcs_length = checkpoint_transaction
95            .events
96            .as_ref()
97            .map(|events| bcs::to_bytes(events).unwrap().len() as u64)
98            .unwrap_or(0);
99        let signatures_bcs_length =
100            bcs::to_bytes(&checkpoint_transaction.transaction.data().tx_signatures())
101                .unwrap()
102                .len() as u64;
103
104        let mut transfers: u64 = 0;
105        let mut split_coins: u64 = 0;
106        let mut merge_coins: u64 = 0;
107        let mut publish: u64 = 0;
108        let mut upgrade: u64 = 0;
109        let mut others: u64 = 0;
110        let mut move_calls_count = 0;
111        let move_calls = move_calls_vec.len() as u64;
112
113        let is_sponsored_tx = txn_data.is_sponsored_tx();
114        let is_system_txn = txn_data.is_system_tx();
115        if !is_system_txn {
116            let kind = txn_data.kind();
117            if let TransactionKind::ProgrammableTransaction(pt) = txn_data.kind() {
118                for cmd in &pt.commands {
119                    match cmd {
120                        Command::MoveCall(_) => move_calls_count += 1,
121                        Command::TransferObjects(_, _) => transfers += 1,
122                        Command::SplitCoins(_, _) => split_coins += 1,
123                        Command::MergeCoins(_, _) => merge_coins += 1,
124                        Command::Publish(_, _) => publish += 1,
125                        Command::Upgrade(_, _, _, _) => upgrade += 1,
126                        _ => others += 1,
127                    }
128                }
129            } else {
130                error!(
131                    "Transaction kind [{kind}] is not programmable transaction and not a system transaction"
132                );
133            }
134            if move_calls_count != move_calls {
135                error!(
136                    "Mismatch in move calls count: commands {move_calls_count} != {move_calls} calls"
137                );
138            }
139        }
140        let transaction_json = serde_json::to_string(&transaction)?;
141        let effects_json = serde_json::to_string(&checkpoint_transaction.effects)?;
142        let entry = TransactionEntry {
143            transaction_digest,
144            checkpoint: checkpoint_seq,
145            epoch,
146            timestamp_ms,
147
148            sender: txn_data.sender().to_string(),
149            transaction_kind: txn_data.kind().name().to_owned(),
150            is_system_txn,
151            is_sponsored_tx,
152            transaction_count: txn_data.kind().num_commands() as u64,
153            execution_success: effects.status().is_ok(),
154            input: txn_data
155                .input_objects()
156                .expect("Input objects must be valid")
157                .len() as u64,
158            shared_input: txn_data.shared_input_objects().len() as u64,
159            gas_coins: txn_data.gas().len() as u64,
160            created: effects.created().len() as u64,
161            mutated: (effects.mutated().len() + effects.unwrapped().len()) as u64,
162            deleted: (effects.deleted().len()
163                + effects.unwrapped_then_deleted().len()
164                + effects.wrapped().len()) as u64,
165            transfers,
166            split_coins,
167            merge_coins,
168            publish,
169            upgrade,
170            others,
171            move_calls,
172            packages,
173            gas_owner: txn_data.gas_owner().to_string(),
174            gas_object_id: gas_object.0.0.to_string(),
175            gas_object_sequence: gas_object.0.1.value(),
176            gas_object_digest: gas_object.0.2.to_string(),
177            gas_budget: txn_data.gas_budget(),
178            total_gas_cost: gas_summary.net_gas_usage(),
179            computation_cost: gas_summary.computation_cost,
180            storage_cost: gas_summary.storage_cost,
181            storage_rebate: gas_summary.storage_rebate,
182            non_refundable_storage_fee: gas_summary.non_refundable_storage_fee,
183            gas_price: txn_data.gas_price(),
184            has_zklogin_sig: transaction.has_zklogin_sig(),
185            has_upgraded_multisig: transaction.has_upgraded_multisig(),
186            transaction_json: Some(transaction_json),
187            effects_json: Some(effects_json),
188            transaction_position,
189            events_digest,
190            raw_transaction: "".to_string(),
191            transaction_data_bcs_length,
192            effects_bcs_length,
193            events_bcs_length,
194            signatures_bcs_length,
195        };
196
197        Ok(Box::new(std::iter::once(entry)))
198    }
199}

201fn compute_transaction_positions(
202    checkpoint_contents: &CheckpointContents,
203) -> HashMap<TransactionDigest, usize> {
204    let mut digest_to_position: HashMap<TransactionDigest, usize> = HashMap::new();
205
206    for (position, execution_digest) in checkpoint_contents.iter().enumerate() {
207        digest_to_position.insert(execution_digest.transaction, position);
208    }
209
210    digest_to_position
211}

213#[cfg(test)]
214mod tests {
215    use crate::handlers::AnalyticsHandler;
216    use crate::handlers::transaction_handler::TransactionHandler;
217    use simulacrum::Simulacrum;
218    use std::sync::Arc;
219    use sui_types::base_types::SuiAddress;
220    use sui_types::storage::ReadStore;
221
222    #[tokio::test]
223    pub async fn test_transaction_handler() -> anyhow::Result<()> {
224        let mut sim = Simulacrum::new();
225
226        // Execute a simple transaction.
227        let transfer_recipient = SuiAddress::random_for_testing_only();
228        let (transaction, _) = sim.transfer_txn(transfer_recipient);
229        let (_effects, err) = sim.execute_transaction(transaction.clone()).unwrap();
230        assert!(err.is_none());
231
232        // Create a checkpoint which should include the transaction we executed.
233        let checkpoint = sim.create_checkpoint();
234        let checkpoint_data: sui_types::full_checkpoint_content::CheckpointData = sim
235            .get_checkpoint_data(
236                checkpoint.clone(),
237                sim.get_checkpoint_contents_by_digest(&checkpoint.content_digest)
238                    .unwrap(),
239            )?
240            .into();
241        let txn_handler = TransactionHandler::new();
242        let transaction_entries: Vec<_> = txn_handler
243            .process_checkpoint(&Arc::new(checkpoint_data))
244            .await?
245            .collect();
246        assert_eq!(transaction_entries.len(), 1);
247        let db_txn = transaction_entries.first().unwrap();
248
249        // Check that the transaction was stored correctly.
250        assert_eq!(db_txn.transaction_digest, transaction.digest().to_string());
251        assert_eq!(
252            db_txn.transaction_data_bcs_length,
253            bcs::to_bytes(&transaction.transaction_data())
254                .unwrap()
255                .len() as u64
256        );
257        assert_eq!(db_txn.epoch, checkpoint.epoch);
258        assert_eq!(db_txn.timestamp_ms, checkpoint.timestamp_ms);
259        assert_eq!(db_txn.checkpoint, checkpoint.sequence_number);
260        assert_eq!(db_txn.transaction_position, 0);
261        Ok(())
262    }
