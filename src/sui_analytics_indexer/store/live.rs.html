22/// Live mode - derives watermarks from file names.
23///
24/// Used for normal streaming ingestion where files are written with checkpoint
25/// ranges in their names, and watermarks are derived from those file names.
26#[derive(Clone)]
27pub struct LiveStore {
28    object_store: Arc<dyn ObjectStore>,
29}

41    /// Determine the watermark by scanning file names in the object store.
42    ///
43    /// 1. Find epoch directories under `{pipeline}/epoch_*`
44    /// 2. Get the latest epoch (max epoch number)
45    /// 3. Find files in that epoch and extract checkpoint ranges from file names
46    /// 4. Return the maximum `end` value as the watermark
47    pub(crate) async fn committer_watermark(
48        &self,
49        pipeline: &str,
50    ) -> anyhow::Result<Option<CommitterWatermark>> {
51        let prefix = ObjectPath::from(pipeline);
52        let epoch_dirs = find_all_dirs_with_epoch_prefix(&self.object_store, Some(&prefix)).await?;
53
54        // Get latest epoch
55        let Some((&epoch, epoch_path)) = epoch_dirs.last_key_value() else {
56            return Ok(None); // No data yet
57        };
58
59        // Find files in latest epoch: {pipeline}/epoch_N/{start}_{end}.{format}
60        let checkpoint_ranges =
61            find_all_files_with_epoch_prefix(&self.object_store, Some(epoch_path)).await?;
62
63        // Watermark = max(end) across all files
64        let checkpoint_hi = checkpoint_ranges.iter().map(|r| r.end).max().unwrap_or(0);
65
66        // Need checkpoint_hi - 1 since ranges are exclusive and watermark is inclusive
67        if checkpoint_hi == 0 {
68            return Ok(None);
69        }
70
71        info!(
72            pipeline,
73            epoch,
74            checkpoint = checkpoint_hi - 1,
75            "Determined watermark from bucket iteration"
76        );
77
78        Ok(Some(CommitterWatermark {
79            epoch_hi_inclusive: epoch,
80            checkpoint_hi_inclusive: checkpoint_hi - 1, // Convert exclusive end to inclusive
81            tx_hi: 0,                                   // Not stored in filenames
82            timestamp_ms_hi_inclusive: 0,               // Not stored in filenames
83        }))
84    }

96    /// Split a batch of checkpoints into files based on epoch boundaries, batch size, and time.
97    ///
98    /// Cuts at:
99    /// - Epoch boundaries (files don't span epochs)
100    /// - Batch size thresholds (rows or checkpoint count)
101    /// - Time threshold (if `max_batch_duration_secs` is configured)
102    pub(crate) fn split_framework_batch_into_files(
103        &self,
104        pipeline_config: &PipelineConfig,
105        rows_by_checkpoint: &[CheckpointRows],
106        mut pending_batch: Batch,
107    ) -> (Batch, Vec<Batch>) {
108        let batch_size = pipeline_config
109            .batch_size
110            .as_ref()
111            .expect("batch_size not configured for pipeline");
112
113        let max_duration = Duration::from_secs(pipeline_config.force_batch_cut_after_secs);
114
115        let mut complete_batches: Vec<Batch> = Vec::new();
116
117        for checkpoint_rows in rows_by_checkpoint {
118            // Cut at epoch boundary
119            if pending_batch
120                .epoch()
121                .is_some_and(|e| e != checkpoint_rows.epoch)
122            {
123                assert!(
124                    !pending_batch.checkpoints_rows.is_empty(),
125                    "invalid state: epoch set but rows empty"
126                );
127                complete_batches.push(pending_batch);
128                pending_batch = Batch::default();
129            }
130
131            match *batch_size {
132                BatchSizeConfig::Rows(n) => {
133                    // Flush BEFORE adding so checkpoint rows stay together
134                    if pending_batch.row_count() >= n {
135                        complete_batches.push(pending_batch);
136                        pending_batch = Batch::default();
137                    }
138                    pending_batch.add(checkpoint_rows.clone());
139                }
140                BatchSizeConfig::Checkpoints(n) => {
141                    pending_batch.add(checkpoint_rows.clone());
142                    // Flush AFTER adding
143                    if pending_batch.checkpoint_count() == n {
144                        complete_batches.push(pending_batch);
145                        pending_batch = Batch::default();
146                    }
147                }
148            }
149
150            // Time-based flush (only if batch has data)
151            if pending_batch.checkpoint_count() > 0 && pending_batch.elapsed() >= max_duration {
152                complete_batches.push(pending_batch);
153                pending_batch = Batch::default();
154            }
155        }
156
157        (pending_batch, complete_batches)
158    }

238    async fn read_stored_chain_id(
239        object_store: &Arc<dyn object_store::ObjectStore>,
240        pipeline_task: &str,
241    ) -> Vec<u8> {
242        let path = ObjectPath::from(format!("_metadata/chain_id/{pipeline_task}"));

338    /// Simulate the framework's per-pipeline `init_watermark` calls in registration
339    /// order, populating the initial_watermarks cache before the eviction pipeline fires.
340    async fn prime_pipelines(conn: &mut crate::store::AnalyticsConnection<'_>, pipelines: &[&str]) {
341        for p in pipelines {
342            let _ = conn.committer_watermark(p).await.unwrap();
343        }
344    }
