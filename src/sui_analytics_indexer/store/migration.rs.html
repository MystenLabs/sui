36/// Error type for watermark updates.
37#[derive(Error, Debug)]
38pub enum WatermarkUpdateError {
39    /// Precondition failure - concurrent writer detected. This is fatal.
40    #[error("Concurrent writer detected on watermark {path}: {message}")]
41    ConcurrentWriter { path: String, message: String },
42
43    /// Transient error - can be retried.
44    #[error("Transient error updating watermark: {0}")]
45    Transient(#[from] anyhow::Error),
46}

59/// Migration mode - uses explicit watermark files and conditional PUT.
60///
61/// Used for rewriting existing files (e.g., adding new columns to parquet files).
62/// Tracks progress separately via watermark files and uses conditional PUT to
63/// ensure we are overwriting the files we expect.
64#[derive(Clone)]
65pub struct MigrationStore {
66    object_store: Arc<dyn ObjectStore>,
67    /// Migration identifier.
68    migration_id: String,
69    /// Pipeline -> FileRangeIndex (target ranges).
70    file_ranges: Arc<RwLock<HashMap<String, FileRangeIndex>>>,
71    /// Pipeline -> (etag, version) for conditional PUT on watermark files.
72    watermark_versions: Arc<RwLock<HashMap<String, VersionInfo>>>,
73    /// Pre-computed per-pipeline adjusted starting checkpoints.
74    /// Set during pre-loading in build_analytics_indexer.
75    adjusted_start_checkpoints: Arc<RwLock<HashMap<String, u64>>>,
76}

78/// Entry for a single file range in the index.

80pub struct FileRangeEntry {

84    pub end: u64,

112    /// Load file ranges and find the starting/ending checkpoints for migration.
113    ///
114    /// This snaps `first_checkpoint` to file boundaries:
115    /// - If checkpoint is inside a file → snap to file start
116    /// - If checkpoint is in a gap → snap to next file start
117    /// - If no files at or after checkpoint → error
118    ///
119    /// This also snaps `last_checkpoint` to file boundaries:
120    /// - If checkpoint is inside a file → snap to file end (exclusive)
121    /// - If checkpoint is in a gap → snap to previous file end (exclusive)
122    /// - If no files at or before checkpoint → error
123    ///
124    /// Returns (adjusted_first, adjusted_last) where:
125    /// - adjusted_first is the minimum adjusted start across all pipelines
126    /// - adjusted_last is the maximum adjusted end across all pipelines
127    ///
128    /// Each pipeline is specified as (pipeline_name, output_prefix) where:
129    /// - pipeline_name is used as the key in file_ranges map (for lookup by handlers)
130    /// - output_prefix is the path prefix in the object store where files are located
131    pub async fn find_checkpoint_range(
132        &self,
133        pipelines: impl Iterator<Item = (&str, &str)>,
134        first_checkpoint: Option<u64>,
135        last_checkpoint: Option<u64>,
136    ) -> Result<(Option<u64>, Option<u64>)> {
137        let mut file_ranges = HashMap::new();
138        let mut adjusted_starts = HashMap::new();
139        let mut min_adjusted_start: Option<u64> = None;
140        let mut max_adjusted_end: Option<u64> = None;
141
142        for (pipeline_name, output_prefix) in pipelines {
143            // Load file ranges from the output_prefix path in the object store
144            let index =
145                FileRangeIndex::load_from_store(&self.object_store, output_prefix, None).await?;
146
147            // Snap starting checkpoint to file boundary.
148            // If first_checkpoint is not specified, snap from 0 to find the first available file.
149            let first_cp = first_checkpoint.unwrap_or(0);
150            let adjusted = index.snap_to_boundary(first_cp).ok_or_else(|| {
151                anyhow!(
152                    "No files at or after checkpoint {} for pipeline '{}'. Nothing to migrate.",
153                    first_cp,
154                    pipeline_name
155                )
156            })?;
157            // Key by pipeline_name for lookup by handlers
158            adjusted_starts.insert(pipeline_name.to_string(), adjusted);
159            min_adjusted_start = Some(min_adjusted_start.map_or(adjusted, |m| m.min(adjusted)));
160
161            info!(
162                pipeline = pipeline_name,
163                output_prefix,
164                requested_checkpoint = first_cp,
165                adjusted_checkpoint = adjusted,
166                "Snapped first_checkpoint to file boundary"
167            );
168
169            // Compute adjusted ending checkpoint if last_checkpoint specified
170            if let Some(last_cp) = last_checkpoint {
171                let adjusted = index.snap_end_to_boundary(last_cp).ok_or_else(|| {
172                    anyhow!(
173                        "No files at or before checkpoint {} for pipeline '{}'. Nothing to migrate.",
174                        last_cp,
175                        pipeline_name
176                    )
177                })?;
178                // Use max across pipelines so all pipelines get all their data
179                max_adjusted_end = Some(max_adjusted_end.map_or(adjusted, |m| m.max(adjusted)));
180
181                info!(
182                    pipeline = pipeline_name,
183                    requested_checkpoint = last_cp,
184                    adjusted_checkpoint = adjusted,
185                    "Snapped last_checkpoint to file boundary"
186                );
187            }
188
189            // Key by pipeline_name for lookup by handlers
190            file_ranges.insert(pipeline_name.to_string(), index);
191        }
192
193        // Store the loaded data
194        debug!(
195            pipeline_keys = ?file_ranges.keys().collect::<Vec<_>>(),
196            "Loaded file ranges for migration"
197        );
198        *self.file_ranges.write().unwrap() = file_ranges;
199        *self.adjusted_start_checkpoints.write().unwrap() = adjusted_starts;
200
201        // Return adjusted checkpoints.
202        // min_adjusted_start is always set (we snap from 0 if first_checkpoint not specified).
203        // last_checkpoint is inclusive in framework (uses ..=), and snap_end_to_boundary
204        // returns end - 1 which is also inclusive, so they match.
205        Ok((min_adjusted_start, max_adjusted_end.or(last_checkpoint)))
206    }

216    /// Read watermark from metadata file and cache its etag/version.
217    pub(crate) async fn committer_watermark(
218        &self,
219        pipeline: &str,
220    ) -> anyhow::Result<Option<CommitterWatermark>> {
221        let path = migration_watermark_path(pipeline, &self.migration_id);
222        match self.object_store.get(&path).await {
223            Ok(result) => {
224                // Capture etag and version for conditional PUT
225                let e_tag = result.meta.e_tag.clone();
226                let version = result.meta.version.clone();
227                self.watermark_versions
228                    .write()
229                    .unwrap()
230                    .insert(pipeline.to_string(), (e_tag, version));
231
232                let bytes = result.bytes().await?;
233                let watermark: MigrationWatermark = serde_json::from_slice(&bytes)
234                    .context("Failed to parse migration watermark from object store")?;
235                info!(
236                    pipeline,
237                    migration_id = self.migration_id,
238                    epoch = watermark.epoch_hi_inclusive,
239                    checkpoint = watermark.checkpoint_hi_inclusive,
240                    "Migration mode: found progress from watermark file"
241                );
242                Ok(Some(CommitterWatermark {
243                    epoch_hi_inclusive: watermark.epoch_hi_inclusive,
244                    checkpoint_hi_inclusive: watermark.checkpoint_hi_inclusive,
245                    tx_hi: 0,
246                    timestamp_ms_hi_inclusive: 0,
247                }))
248            }
249            Err(ObjectStoreError::NotFound { .. }) => Ok(None),
250            Err(e) => Err(e.into()),
251        }
252    }

254    /// Initialize migration mode for a pipeline.
255    ///
256    /// Reads existing watermark file if present, and ensures file ranges are loaded.
257    ///
258    /// Returns the last processed checkpoint if a watermark exists, or None if starting fresh.
259    /// The watermark file is created/updated by `update_watermark` after file uploads.
260    pub(crate) async fn init_watermark(
261        &self,
262        pipeline: &str,
263        _default_next_checkpoint: u64,
264    ) -> anyhow::Result<Option<u64>> {
265        // Check existing watermark
266        let (checkpoint_hi, epoch_hi) =
267            if let Some(watermark) = self.committer_watermark(pipeline).await? {
268                (
269                    Some(watermark.checkpoint_hi_inclusive),
270                    Some(watermark.epoch_hi_inclusive),
271                )
272            } else {
273                // No existing watermark - framework will use default_next_checkpoint
274                // Watermark will be created by update_watermark after first file upload
275                (None, None)
276            };
277
278        // Load file ranges if not already pre-loaded
279        if !self.file_ranges.read().unwrap().contains_key(pipeline) {
280            let index =
281                FileRangeIndex::load_from_store(&self.object_store, pipeline, epoch_hi).await?;
282            self.file_ranges
283                .write()
284                .unwrap()
285                .insert(pipeline.to_string(), index);
286        }
287
288        Ok(checkpoint_hi)
289    }

291    /// Update watermark for a single pipeline after successful file upload.
292    ///
293    /// Called by the upload worker after each file is successfully uploaded.
294    /// This provides incremental progress tracking for crash recovery.
295    ///
296    /// Returns `WatermarkUpdateError::ConcurrentWriter` on precondition failure (fatal),
297    /// or `WatermarkUpdateError::Transient` on other errors (can be retried).
298    pub(crate) async fn update_watermark(
299        &self,
300        pipeline: &str,
301        epoch_hi_inclusive: u64,
302        checkpoint_hi_inclusive: u64,
303    ) -> std::result::Result<(), WatermarkUpdateError> {
304        let path = migration_watermark_path(pipeline, &self.migration_id);
305        let json = serde_json::to_vec(&MigrationWatermark {
306            checkpoint_hi_inclusive,
307            epoch_hi_inclusive,
308        })
309        .map_err(|e| WatermarkUpdateError::Transient(e.into()))?;
310
311        // Look up cached etag/version for conditional PUT
312        let (e_tag, version) = self
313            .watermark_versions
314            .read()
315            .unwrap()
316            .get(pipeline)
317            .cloned()
318            .unwrap_or((None, None));
319
320        let mode = if e_tag.is_some() || version.is_some() {
321            PutMode::Update(UpdateVersion { e_tag, version })
322        } else {
323            PutMode::Create
324        };
325
326        let result = self
327            .object_store
328            .put_opts(
329                &path,
330                json.into(),
331                PutOptions {
332                    mode,
333                    ..Default::default()
334                },
335            )
336            .await
337            .map_err(|e| match e {
338                ObjectStoreError::Precondition { path, source } => {
339                    WatermarkUpdateError::ConcurrentWriter {
340                        path: path.to_string(),
341                        message: source.to_string(),
342                    }
343                }
344                other => WatermarkUpdateError::Transient(other.into()),
345            })?;
346
347        // Update cached etag/version
348        self.watermark_versions
349            .write()
350            .unwrap()
351            .insert(pipeline.to_string(), (result.e_tag, result.version));
352
353        tracing::debug!(
354            pipeline,
355            migration_id = self.migration_id,
356            checkpoint = checkpoint_hi_inclusive,
357            epoch = epoch_hi_inclusive,
358            "Updated migration watermark"
359        );
360
361        Ok(())
362    }

364    /// Split a batch of checkpoints into files based on existing file boundaries.
365    ///
366    /// In migration mode, we match the boundaries of existing files to ensure
367    /// we can use conditional PUT with the correct e_tag/version.
368    ///
369    /// The watermark indicates the highest checkpoint processed by the framework,
370    /// including empty checkpoints. It's used to detect when the final file is
371    /// complete (when there's no "next row" to trigger boundary detection).
372    pub(crate) fn split_framework_batch_into_files(
373        &self,
374        pipeline_config: &PipelineConfig,
375        batch_from_framework: &[CheckpointRows],
376        mut pending_batch: Batch,
377        watermark: &CommitterWatermark,
378    ) -> (Batch, Vec<Batch>) {
379        let mut complete_batches: Vec<Batch> = Vec::new();
380        let pipeline = pipeline_config.pipeline.name();
381
382        let ranges = self
383            .file_ranges
384            .read()
385            .unwrap()
386            .get(pipeline)
387            .cloned()
388            .expect("migration ranges not loaded for pipeline");
389
390        for checkpoint_rows in batch_from_framework {
391            let cp = checkpoint_rows.checkpoint;
392
393            // Check if we've crossed a file boundary BEFORE adding this checkpoint.
394            // This handles sparse checkpoints - the framework only sends checkpoints with rows,
395            // so we might skip from e.g. 6500 to 6600, missing the exact boundary at 6543.
396            if let Some(first) = pending_batch.first_checkpoint()
397                && let Some(entry) = ranges.find_containing(first)
398                && cp >= entry.end
399            {
400                debug!(
401                    pipeline,
402                    current_cp = cp,
403                    file_start = entry.start,
404                    file_end = entry.end,
405                    "Crossed file boundary - completing batch"
406                );
407                pending_batch.explicit_range = Some(entry.start..entry.end);
408                complete_batches.push(pending_batch);
409                pending_batch = Batch::default();
410            }
411
412            pending_batch.add(checkpoint_rows.clone());
413
414            // Validate that checkpoints with rows are in a known file range
415            if let Some(first) = pending_batch.first_checkpoint()
416                && ranges.find_containing(first).is_none()
417                && !checkpoint_rows.is_empty()
418            {
419                panic!(
420                    "Migration error: checkpoint {} has {} rows but is not in any existing file range for pipeline '{}'. \
421                     File ranges cover checkpoints {:?} to {:?}.",
422                    first,
423                    checkpoint_rows.len(),
424                    pipeline,
425                    ranges.first_checkpoint(),
426                    ranges.last_checkpoint_exclusive(),
427                );
428            }
429        }
430
431        // Check if the pending batch's file is complete based on watermark.
432        // This handles the final batch when there's no "next row" to trigger
433        // the boundary detection above.
434        if let Some(first) = pending_batch.first_checkpoint()
435            && let Some(entry) = ranges.find_containing(first)
436            && watermark.checkpoint_hi_inclusive >= entry.end - 1
437        {
438            debug!(
439                pipeline,
440                watermark_cp = watermark.checkpoint_hi_inclusive,
441                file_start = entry.start,
442                file_end = entry.end,
443                "File complete per watermark - completing batch"
444            );
445            pending_batch.explicit_range = Some(entry.start..entry.end);
446            complete_batches.push(pending_batch);
447            pending_batch = Batch::default();
448        }
449
450        (pending_batch, complete_batches)
451    }

453    /// Write a file to the object store with conditional update.
454    ///
455    /// Verifies the checkpoint range matches an existing file in the index,
456    /// then fetches current metadata via HEAD for conditional PUT.
457    ///
458    /// Errors if the file doesn't exist (migration mode requires existing files to replace).
459    pub(crate) async fn write_to_object_store(
460        &self,
461        pipeline: &str,
462        path: &ObjectPath,
463        checkpoint_range: &Range<u64>,
464        payload: PutPayload,
465    ) -> anyhow::Result<()> {
466        // Verify this range exists in our index
467        {
468            let ranges = self.file_ranges.read().unwrap();
469            let pipeline_ranges = ranges.get(pipeline).expect("migration ranges not loaded");
470
471            let entry = pipeline_ranges
472                .find_containing(checkpoint_range.start)
473                .ok_or_else(|| {
474                    anyhow!(
475                        "No file in index for checkpoint range {:?} - migration requires existing files",
476                        checkpoint_range
477                    )
478                })?;
479
480            // Verify the range matches exactly
481            anyhow::ensure!(
482                entry.start == checkpoint_range.start,
483                "checkpoint range start mismatch: expected {}, got {}",
484                entry.start,
485                checkpoint_range.start
486            );
487            anyhow::ensure!(
488                entry.end == checkpoint_range.end,
489                "checkpoint range end mismatch: expected {}, got {}",
490                entry.end,
491                checkpoint_range.end
492            );
493        }
494
495        // Fetch current metadata for conditional PUT (GCS requires version/generation)
496        let meta = self.object_store.head(path).await.map_err(|e| {
497            anyhow!(
498                "Failed to get metadata for {} (file should exist for migration): {}",
499                path,
500                e
501            )
502        })?;
503
504        self.put_conditional(
505            path,
506            payload,
507            meta.e_tag.as_deref(),
508            meta.version.as_deref(),
509        )
510        .await
511    }
512
513    /// Put a file with conditional update for migration mode.
514    ///
515    /// Uses `PutMode::Update` with etag/version for atomic replacement to prevent
516    /// concurrent modification. GCS requires version (generation), other stores use e_tag.
517    async fn put_conditional(
518        &self,
519        path: &ObjectPath,
520        payload: PutPayload,
521        expected_etag: Option<&str>,
522        expected_version: Option<&str>,
523    ) -> anyhow::Result<()> {
524        let mode = if expected_version.is_some() || expected_etag.is_some() {
525            PutMode::Update(UpdateVersion {
526                e_tag: expected_etag.map(String::from),
527                version: expected_version.map(String::from),
528            })
529        } else {
530            PutMode::Create
531        };
532
533        self.object_store
534            .put_opts(
535                path,
536                payload,
537                PutOptions {
538                    mode,
539                    ..Default::default()
540                },
541            )
542            .await
543            .map_err(|e| match e {
544                ObjectStoreError::Precondition { path, source } => {
545                    anyhow!(
546                        "Concurrent writer detected - etag mismatch for {}: {}",
547                        path,
548                        source
549                    )
550                }
551                ObjectStoreError::AlreadyExists { path, source } => {
552                    anyhow!(
553                        "File already exists (expected for conditional create): {}: {}",
554                        path,
555                        source
556                    )
557                }
558                _ => e.into(),
559            })?;
560        Ok(())
561    }
562}

564impl FileRangeIndex {
565    /// Create a new empty index.
566    pub fn new() -> Self {
567        Self::default()
568    }

570    /// Insert a file range entry.
571    pub fn insert(&mut self, entry: FileRangeEntry) {
572        self.ranges.insert(entry.start, entry);
573    }
574
575    /// Get the number of file ranges.
576    pub fn len(&self) -> usize {
577        self.ranges.len()
578    }
579
580    /// Check if the index is empty.
581    pub fn is_empty(&self) -> bool {
582        self.ranges.is_empty()
583    }
584
585    /// Find the file range that contains the given checkpoint.
586    pub fn find_containing(&self, checkpoint: u64) -> Option<&FileRangeEntry> {
587        // Find the largest start <= checkpoint
588        self.ranges
589            .range(..=checkpoint)
590            .next_back()
591            .filter(|(_, entry)| checkpoint < entry.end)
592            .map(|(_, entry)| entry)
593    }
594
595    /// Find the next file boundary at or after the given checkpoint.
596    pub fn find_next_boundary(&self, checkpoint: u64) -> Option<&FileRangeEntry> {
597        self.ranges
598            .range(checkpoint..)
599            .next()
600            .map(|(_, entry)| entry)
601    }
602
603    /// Snap a start checkpoint to file boundaries for migration.
604    ///
605    /// - If checkpoint is inside a file → returns file start
606    /// - If checkpoint is in a gap → returns next file start
607    /// - If no files at or after checkpoint → returns None (error case)
608    pub fn snap_to_boundary(&self, checkpoint: u64) -> Option<u64> {
609        // Check if checkpoint is inside a file
610        if let Some(entry) = self.find_containing(checkpoint) {
611            return Some(entry.start);
612        }
613        // Check for next file after checkpoint
614        if let Some(entry) = self.find_next_boundary(checkpoint) {
615            return Some(entry.start);
616        }
617        // No files at or after checkpoint
618        None
619    }
620
621    /// Snap an end checkpoint to file boundaries for migration.
622    ///
623    /// Returns the last checkpoint (inclusive) that should be processed to complete
624    /// the file containing the given checkpoint:
625    /// - If checkpoint is inside a file → returns file's last checkpoint (end - 1)
626    /// - If checkpoint is in a gap → returns previous file's last checkpoint (end - 1)
627    /// - If no files at or before checkpoint → returns None (error case)
628    pub fn snap_end_to_boundary(&self, checkpoint: u64) -> Option<u64> {
629        // Check if checkpoint is inside a file
630        if let Some(entry) = self.find_containing(checkpoint) {
631            // Return last checkpoint in file (end is exclusive, so end - 1 is last inclusive)
632            return Some(entry.end - 1);
633        }
634        // Checkpoint is in a gap - find the previous file's end
635        // Look for the largest start < checkpoint
636        if let Some((_, entry)) = self.ranges.range(..checkpoint).next_back() {
637            return Some(entry.end - 1);
638        }
639        // No files at or before checkpoint
640        None
641    }
642
643    /// Get the first checkpoint across all ranges.
644    pub fn first_checkpoint(&self) -> Option<u64> {
645        self.ranges.keys().next().copied()
646    }
647
648    /// Get the last checkpoint across all ranges (exclusive).
649    pub fn last_checkpoint_exclusive(&self) -> Option<u64> {
650        self.ranges.values().map(|e| e.end).max()
651    }

653    /// Load file range index from object store.
654    ///
655    /// This lists all files in the pipeline directory to build the index of
656    /// target ranges for migration. Progress is tracked via a separate watermark file.
657    ///
658    /// If `min_epoch` is provided, only epochs >= min_epoch are scanned, which
659    /// reduces startup time when resuming a migration.
660    pub async fn load_from_store(
661        store: &Arc<dyn ObjectStore>,
662        pipeline: &str,
663        min_epoch: Option<u64>,
664    ) -> Result<Self> {
665        let mut index = Self::new();
666
667        // Find all epoch directories under {pipeline}/epoch_*
668        let prefix = ObjectPath::from(pipeline);
669        let epoch_dirs = find_all_dirs_with_epoch_prefix(store, Some(&prefix)).await?;
670
671        let skipped_epochs = min_epoch
672            .map(|min| epoch_dirs.range(..min).count())
673            .unwrap_or(0);
674
675        for (epoch, epoch_path) in epoch_dirs {
676            // Skip epochs before the watermark
677            if let Some(min) = min_epoch
678                && epoch < min
679            {
680                continue;
681            }
682
683            // List files in this epoch directory
684            let list_result = store.list_with_delimiter(Some(&epoch_path)).await?;
685
686            for obj in list_result.objects {
687                // Parse checkpoint range from filename: {start}_{end}.{format}
688                let Some(filename) = obj.location.filename() else {
689                    continue;
690                };
691                let Some(range) = super::parse_checkpoint_range(filename) else {
692                    continue;
693                };
694
695                index.insert(FileRangeEntry {
696                    start: range.start,
697                    end: range.end,
698                    epoch,
699                });
700            }
701        }
702
703        info!(
704            pipeline,
705            num_files = index.len(),
706            skipped_epochs,
707            min_epoch,
708            first_checkpoint = ?index.first_checkpoint(),
709            last_checkpoint = ?index.last_checkpoint_exclusive(),
710            "Loaded existing file ranges"
711        );
712
713        Ok(index)
714    }
715}

717/// Construct the path for a migration watermark file.
718///
719/// Format: `_metadata/watermarks/{pipeline}@migration_{migration_id}.json`
720pub(crate) fn migration_watermark_path(pipeline: &str, migration_id: &str) -> ObjectPath {
721    ObjectPath::from(format!(
722        "_metadata/watermarks/{}@migration_{}.json",
723        pipeline, migration_id
724    ))
725}

846        let mut index = FileRangeIndex::new();
847        // Files: 0-100, 200-300 (gap at 100-200)
848        index.insert(FileRangeEntry {
849            start: 0,
850            end: 100,
851            epoch: 0,
852        });
853        index.insert(FileRangeEntry {
854            start: 200,
855            end: 300,
856            epoch: 1,
857        });
858
859        // Checkpoint inside first file → snap to file start
860        assert_eq!(index.snap_to_boundary(50), Some(0));
861        assert_eq!(index.snap_to_boundary(0), Some(0));
862        assert_eq!(index.snap_to_boundary(99), Some(0));
863
864        // Checkpoint in gap → snap to next file start
865        assert_eq!(index.snap_to_boundary(100), Some(200));
866        assert_eq!(index.snap_to_boundary(150), Some(200));
867        assert_eq!(index.snap_to_boundary(199), Some(200));
868
869        // Checkpoint inside second file → snap to file start
870        assert_eq!(index.snap_to_boundary(200), Some(200));
871        assert_eq!(index.snap_to_boundary(250), Some(200));
872        assert_eq!(index.snap_to_boundary(299), Some(200));
873
874        // Checkpoint beyond all files → None (error case)
875        assert_eq!(index.snap_to_boundary(300), None);
876        assert_eq!(index.snap_to_boundary(1000), None);
877    }

881        let mut index = FileRangeIndex::new();
882        // Files: 0-100, 200-300 (gap at 100-200)
883        // File ranges are [start, end) so file 0-100 contains checkpoints 0-99

886            end: 100,
887            epoch: 0,
888        });
889        index.insert(FileRangeEntry {

892            epoch: 1,
893        });
894
895        // Checkpoint inside first file → snap to last checkpoint in file (99)
896        assert_eq!(index.snap_end_to_boundary(0), Some(99));
897        assert_eq!(index.snap_end_to_boundary(50), Some(99));
898        assert_eq!(index.snap_end_to_boundary(99), Some(99));
899
900        // Checkpoint in gap → snap to previous file's last checkpoint (99)
901        assert_eq!(index.snap_end_to_boundary(100), Some(99));
902        assert_eq!(index.snap_end_to_boundary(150), Some(99));
903        assert_eq!(index.snap_end_to_boundary(199), Some(99));

905        // Checkpoint inside second file → snap to last checkpoint in file (299)
906        assert_eq!(index.snap_end_to_boundary(200), Some(299));
907        assert_eq!(index.snap_end_to_boundary(250), Some(299));
908        assert_eq!(index.snap_end_to_boundary(299), Some(299));
909
910        // Checkpoint at or beyond second file end → snap to last checkpoint (299)
911        assert_eq!(index.snap_end_to_boundary(300), Some(299));
