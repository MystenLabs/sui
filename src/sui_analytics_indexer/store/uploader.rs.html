4//! Async upload worker for analytics files.
5//!
6//! This module provides a background worker that handles file serialization and upload
7//! asynchronously. The worker is split into two concurrent tasks:
8//! - Serialization task: receives batches, serializes in parallel, sends to uploader
9//! - Upload task: receives serialized files, reorders them, uploads sequentially
10//!
11//! This separation allows serialization to continue while uploads are in progress.
12

42/// Initial backoff delay for retries.
43const INITIAL_BACKOFF: Duration = Duration::from_millis(100);
44/// Maximum backoff delay for retries.
45const MAX_BACKOFF: Duration = Duration::from_secs(60 * 5);
46
47/// Helper for exponential backoff with jitter.
48pub(crate) struct Backoff {
49    current: Duration,
50}
51
52impl Backoff {
53    pub(crate) fn new() -> Self {
54        Self {
55            current: INITIAL_BACKOFF,
56        }
57    }
58
59    /// Sleep for the current backoff duration, then increase it.
60    pub(crate) async fn sleep_and_advance(&mut self) {
61        tokio::time::sleep(self.current).await;
62        self.current = (self.current * 2).min(MAX_BACKOFF);
63    }
64
65    /// Get the current backoff duration (for logging).
66    pub(crate) fn current_ms(&self) -> u128 {
67        self.current.as_millis()
68    }
69}

89/// Spawn an upload worker for a pipeline.
90///
91/// Returns the sender for queueing files and the worker's JoinHandle for shutdown.
92pub fn spawn_uploader(
93    pipeline_name: String,
94    output_prefix: String,
95    mode: StoreMode,
96    metrics: Metrics,
97    config: &IndexerConfig,
98) -> (mpsc::Sender<PendingFileUpload>, JoinHandle<()>) {
99    // Size 1: backpressure to handler, limits unserialized batches in memory
100    let (tx, rx) = mpsc::channel(1);
101
102    // Channel between serialization and upload tasks.
103    // When full, serialization blocks until uploads complete.
104    let (upload_tx, upload_rx) = mpsc::channel(config.max_pending_uploads);
105
106    let dispatcher = Dispatcher::new(
107        rx,
108        upload_tx,
109        pipeline_name.clone(),
110        config.max_concurrent_serialization,
111    );
112
113    let uploader = SequentialUploader::new(
114        upload_rx,
115        pipeline_name,
116        output_prefix,
117        mode,
118        metrics,
119        Duration::from_secs(config.watermark_update_interval_secs),
120    );
121
122    // Spawn both tasks, join them together
123    let worker_handle = tokio::spawn(async move {
124        let dispatcher_handle = tokio::spawn(dispatcher.run());
125        let uploader_handle = tokio::spawn(uploader.run());
126
127        // Wait for both to complete
128        let _ = dispatcher_handle.await;
129        let _ = uploader_handle.await;
130    });
131
132    (tx, worker_handle)
133}

135/// Dispatches serialization tasks and forwards results to the uploader.
136/// Serialization runs in parallel on the blocking thread pool.
137/// Ordering is handled by the uploader via sequence numbers.
138struct Dispatcher {
139    rx: mpsc::Receiver<PendingFileUpload>,
140    upload_tx: mpsc::Sender<SerializedFile>,
141    /// Next sequence number to assign (used by uploader for ordering)
142    next_seq: u64,
143    /// Serialization tasks in flight
144    serializing: FuturesUnordered<JoinHandle<Result<SerializedFile>>>,
145    /// Pipeline name (for logging)
146    pipeline: String,
147    /// Maximum concurrent serialization tasks
148    max_concurrent_serialization: usize,
149}
150
151impl Dispatcher {
152    fn new(
153        rx: mpsc::Receiver<PendingFileUpload>,
154        upload_tx: mpsc::Sender<SerializedFile>,
155        pipeline: String,
156        max_concurrent_serialization: usize,
157    ) -> Self {
158        Self {
159            rx,
160            upload_tx,
161            next_seq: 0,
162            serializing: FuturesUnordered::new(),
163            pipeline,
164            max_concurrent_serialization,
165        }
166    }
167
168    async fn run(mut self) {
169        debug!(pipeline = %self.pipeline, "Dispatcher starting");
170
171        loop {
172            let serializing_inflight = self.serializing.len();
173
174            tokio::select! {
175                // Receive new batch to serialize (only if below concurrency limit)
176                Some(pending) = self.rx.recv(), if serializing_inflight < self.max_concurrent_serialization => {
177                    self.spawn_serialization(pending);
178                }
179
180                // Serialization task completed - send to uploader
181                Some(join_result) = self.serializing.next(), if !self.serializing.is_empty() => {
182                    if self.forward_to_uploader(join_result).await.is_err() {
183                        return;
184                    }
185                }
186
187                // Channel closed and no more serialization work
188                else => {
189                    if self.serializing.is_empty() {
190                        break;
191                    }
192                }
193            }
194        }
195
196        // Drain remaining serialization tasks
197        while let Some(join_result) = self.serializing.next().await {
198            if let Ok(Ok(serialized)) = join_result {
199                let _ = self.upload_tx.send(serialized).await;
200            }
201        }
202
203        debug!(pipeline = %self.pipeline, "Dispatcher finished");
204        // upload_tx drops here, signaling uploader to finish
205    }
206
207    fn spawn_serialization(&mut self, pending: PendingFileUpload) {
208        let seq = self.next_seq;
209        self.next_seq += 1;
210
211        let pipeline = self.pipeline.clone();
212
213        let handle = tokio::task::spawn_blocking(move || {
214            let bytes = serialize_rows(
215                &pending.checkpoints_rows,
216                pending.schema,
217                pending.file_format,
218            )?;
219            debug!(
220                pipeline = %pipeline,
221                seq,
222                checkpoint_range = ?pending.checkpoint_range,
223                bytes = bytes.len(),
224                "Serialized file"
225            );
226            Ok(SerializedFile {
227                seq,
228                epoch: pending.epoch,
229                checkpoint_range: pending.checkpoint_range,
230                file_format: pending.file_format,
231                bytes,
232            })
233        });
234        self.serializing.push(handle);
235    }
236
237    /// Forward a completed serialization result to the uploader.
238    /// Returns Err(()) if the dispatcher should stop.
239    async fn forward_to_uploader(
240        &mut self,
241        join_result: Result<Result<SerializedFile, anyhow::Error>, tokio::task::JoinError>,
242    ) -> Result<(), ()> {
243        let serialized = match join_result {
244            Ok(Ok(file)) => file,
245            Ok(Err(err)) => {
246                error!(pipeline = %self.pipeline, %err, "Serialization failed, stopping");
247                return Err(());
248            }
249            Err(err) => {
250                error!(pipeline = %self.pipeline, %err, "Serialization task panicked, stopping");
251                return Err(());
252            }
253        };
254        if self.upload_tx.send(serialized).await.is_err() {
255            error!(pipeline = %self.pipeline, "Upload channel closed, stopping");
256            return Err(());
257        }
258        Ok(())
259    }
260}

262/// Worker that receives serialized files, reorders them, and uploads sequentially.
263struct SequentialUploader {
264    rx: mpsc::Receiver<SerializedFile>,
265    /// Files received out of order, waiting to be uploaded
266    pending: BTreeMap<u64, SerializedFile>,
267    /// Next sequence number to upload
268    next_upload_seq: u64,
269    pipeline_name: String,
270    output_prefix: String,
271    mode: StoreMode,
272    metrics: Metrics,
273    /// Last time we wrote watermark to object store (for rate limiting)
274    last_watermark_update: Option<std::time::Instant>,
275    /// Minimum interval between watermark writes
276    watermark_update_interval: Duration,
277    /// Latest uploaded watermark (epoch, checkpoint_hi_inclusive).
278    latest_watermark: Option<(EpochId, u64)>,
279}
280
281impl SequentialUploader {
282    fn new(
283        rx: mpsc::Receiver<SerializedFile>,
284        pipeline_name: String,
285        output_prefix: String,
286        mode: StoreMode,
287        metrics: Metrics,
288        watermark_update_interval: Duration,
289    ) -> Self {
290        Self {
291            rx,
292            pending: BTreeMap::new(),
293            next_upload_seq: 0,
294            pipeline_name,
295            output_prefix,
296            mode,
297            metrics,
298            last_watermark_update: None,
299            watermark_update_interval,
300            latest_watermark: None,
301        }
302    }
303
304    async fn run(mut self) {
305        debug!(pipeline = %self.pipeline_name, "Upload worker starting");
306
307        // Receive files and upload in sequence order
308        while let Some(file) = self.rx.recv().await {
309            self.pending.insert(file.seq, file);
310            self.drain_and_upload().await;
311        }
312
313        // Drain any remaining pending files
314        self.drain_and_upload().await;
315
316        // Final watermark flush on shutdown
317        if let Some((epoch, checkpoint_hi)) = self.latest_watermark {
318            self.update_watermark_with_retry(epoch, checkpoint_hi).await;
319            info!(
320                pipeline = %self.pipeline_name,
321                epoch,
322                checkpoint_hi,
323                "Flushed watermark on shutdown"
324            );
325        }
326
327        debug!(pipeline = %self.pipeline_name, "Upload worker finished");
328    }
329
330    async fn drain_and_upload(&mut self) {
331        // Upload files in sequence order
332        while let Some(file) = self.pending.remove(&self.next_upload_seq) {
333            self.upload_with_retry(&file).await;
334            self.next_upload_seq += 1;
335        }
336    }
337
338    async fn upload_with_retry(&mut self, file: &SerializedFile) {
339        let mut backoff = Backoff::new();
340
341        let path = construct_object_store_path(
342            &self.output_prefix,
343            file.epoch,
344            &file.checkpoint_range,
345            file.file_format,
346        );
347
348        loop {
349            match self
350                .do_upload(&path, &file.checkpoint_range, file.bytes.clone())
351                .await
352            {
353                Ok(()) => {
354                    let checkpoint_hi = file.checkpoint_range.end - 1;
355
356                    record_file_metrics(&self.metrics, &self.pipeline_name, file.bytes.len());
357                    self.metrics
358                        .latest_uploaded_checkpoint
359                        .with_label_values(&[&self.pipeline_name])
360                        .set(checkpoint_hi as i64);
361                    self.metrics
362                        .latest_uploaded_epoch
363                        .with_label_values(&[&self.pipeline_name])
364                        .set(file.epoch as i64);
365
366                    info!(
367                        pipeline = %self.pipeline_name,
368                        epoch = file.epoch,
369                        checkpoint_range = ?file.checkpoint_range,
370                        bytes = file.bytes.len(),
371                        "Uploaded file"
372                    );
373
374                    // Always track latest watermark in memory (for shutdown flush)
375                    self.latest_watermark = Some((file.epoch, checkpoint_hi));
376
377                    // Rate-limit writes to object store
378                    let should_update = self
379                        .last_watermark_update
380                        .map(|last| last.elapsed() >= self.watermark_update_interval)
381                        .unwrap_or(true);
382
383                    if should_update {
384                        self.update_watermark_with_retry(file.epoch, checkpoint_hi)
385                            .await;
386                        self.last_watermark_update = Some(std::time::Instant::now());
387                    }
388
389                    return;
390                }
391                Err(e) => {
392                    warn!(
393                        pipeline = %self.pipeline_name,
394                        checkpoint_range = ?file.checkpoint_range,
395                        error = %e,
396                        backoff_ms = backoff.current_ms(),
397                        "Upload failed, retrying"
398                    );
399                    backoff.sleep_and_advance().await;
400                }
401            }
402        }
403    }
404
405    async fn do_upload(
406        &self,
407        path: &ObjectPath,
408        checkpoint_range: &Range<u64>,
409        bytes: Bytes,
410    ) -> Result<()> {
411        self.mode
412            .write_to_object_store(
413                &self.output_prefix,
414                path,
415                checkpoint_range,
416                PutPayload::from(bytes),
417            )
418            .await
419    }
420
421    /// Update watermark with retry on transient errors, panic on concurrent writer.
422    async fn update_watermark_with_retry(&self, epoch: EpochId, checkpoint_hi: u64) {
423        let mut backoff = Backoff::new();
424
425        loop {
426            match self
427                .mode
428                .update_watermark_after_upload(&self.output_prefix, epoch, checkpoint_hi)
429                .await
430            {
431                Ok(()) => return,
432                Err(WatermarkUpdateError::ConcurrentWriter { path, message }) => {
433                    panic!(
434                        "Concurrent writer detected on watermark {}: {}. \
435                         Only one instance should run at a time per pipeline.",
436                        path, message
437                    );
438                }
439                Err(WatermarkUpdateError::Transient(e)) => {
440                    warn!(
441                        pipeline = %self.pipeline_name,
442                        epoch,
443                        checkpoint_hi,
444                        error = %e,
445                        backoff_ms = backoff.current_ms(),
446                        "Transient error updating watermark, retrying"
447                    );
448                    backoff.sleep_and_advance().await;
449                }
450            }
451        }
452    }
453}

455/// Serialize rows grouped by checkpoint to the appropriate file format.
456fn serialize_rows(
457    checkpoints: &[CheckpointRows],
458    schema: &[&str],
459    format: FileFormat,
460) -> Result<Bytes> {
461    match format {
462        FileFormat::Csv => {
463            let mut writer = CsvWriter::new()?;
464            for checkpoint in checkpoints {
465                writer.write(checkpoint)?;
466            }
467            writer
468                .flush()
469                .map(|opt| opt.unwrap_or_default())
470                .map(Bytes::from)
471        }
472        FileFormat::Parquet => {
473            let mut writer = ParquetWriter::new()?;
474            for checkpoint in checkpoints {
475                writer.write(checkpoint)?;
476            }
477            writer
478                .flush(schema)
479                .map(|opt| opt.unwrap_or_default())
480                .map(Bytes::from)
481        }
482    }
