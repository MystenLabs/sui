22type StrBuilder = GenericStringBuilder<i32>;
23
24enum ColumnBuilder {
25    U64(UInt64Builder),
26    I64(Int64Builder),
27    Bool(BooleanBuilder),
28    Str(StrBuilder),
29}
30
31impl ColumnBuilder {
32    fn as_any_builder(&mut self) -> &mut dyn ArrayBuilder {
33        match self {
34            Self::U64(b) => b,
35            Self::I64(b) => b,
36            Self::Bool(b) => b,
37            Self::Str(b) => b,
38        }
39    }
40
41    fn finish(self) -> ArrayRef {
42        match self {
43            Self::U64(mut b) => Arc::new(b.finish()),
44            Self::I64(mut b) => Arc::new(b.finish()),
45            Self::Bool(mut b) => Arc::new(b.finish()),
46            Self::Str(mut b) => Arc::new(b.finish()),
47        }
48    }
49}
50
51/// Writes table entries to Parquet format in memory.
52pub struct ParquetWriter {
53    builders: Vec<ColumnBuilder>,
54}
55
56impl ParquetWriter {
57    pub fn new() -> Result<Self> {
58        Ok(Self { builders: vec![] })
59    }
60
61    /// Writes the given rows to the Parquet buffer.
62    ///
63    /// Uses `RowSchema::get_column()` for dynamic column access via trait objects.
64    pub fn write(&mut self, checkpoint: &CheckpointRows) -> Result<()> {
65        if checkpoint.is_empty() {
66            return Ok(());
67        }
68
69        for row in checkpoint.iter() {
70            // Lazily sample the first row to infer the schema
71            if self.builders.is_empty() {
72                for col_idx in 0..row.column_count() {
73                    let value = row.get_column(col_idx)?;
74                    self.builders.push(match value {
75                        ColumnValue::U64(_) | ColumnValue::OptionU64(_) => {
76                            ColumnBuilder::U64(UInt64Builder::new())
77                        }
78                        ColumnValue::I64(_) => ColumnBuilder::I64(Int64Builder::new()),
79                        ColumnValue::Bool(_) => ColumnBuilder::Bool(BooleanBuilder::new()),
80                        ColumnValue::Str(_) | ColumnValue::OptionStr(_) => {
81                            ColumnBuilder::Str(StrBuilder::new())
82                        }
83                    });
84                }
85            }
86
87            for col_idx in 0..row.column_count() {
88                let value = row.get_column(col_idx)?;
89                match (&mut self.builders[col_idx], value) {
90                    (ColumnBuilder::U64(b), ColumnValue::U64(v)) => b.append_value(v),
91                    (ColumnBuilder::I64(b), ColumnValue::I64(v)) => b.append_value(v),
92                    (ColumnBuilder::Bool(b), ColumnValue::Bool(v)) => b.append_value(v),
93                    (ColumnBuilder::Str(b), ColumnValue::Str(v)) => b.append_value(v.as_ref()),
94
95                    (ColumnBuilder::U64(b), ColumnValue::OptionU64(opt)) => match opt {
96                        Some(v) => b.append_value(v),
97                        None => b.append_null(),
98                    },
99                    (ColumnBuilder::Str(b), ColumnValue::OptionStr(opt)) => match opt {
100                        Some(v) => b.append_value(v.as_ref()),
101                        None => b.append_null(),
102                    },
103
104                    _ => return Err(anyhow!("type mismatch on column {}", col_idx)),
105                }
106            }
107        }
108
109        Ok(())
110    }
111
112    /// Flushes accumulated rows to an in-memory Parquet buffer.
113    ///
114    /// Takes schema as a parameter since `schema()` is not object-safe.
115    pub fn flush(&mut self, schema: &[&str]) -> Result<Option<Vec<u8>>> {
116        // Nothing to flush if builders aren't initialized or are empty
117        if self.builders.is_empty()
118            || self
119                .builders
120                .iter_mut()
121                .all(|b| b.as_any_builder().is_empty())
122        {
123            return Ok(None);
124        }
125
126        // Turn builders into Arrow arrays.
127        let arrays: Vec<ArrayRef> = std::mem::take(&mut self.builders)
128            .into_iter()
129            .map(|b| b.finish())
130            .collect();
131
132        let batch = RecordBatch::try_from_iter(schema.iter().zip(arrays))?;
133
134        let properties = WriterProperties::builder()
135            .set_compression(Compression::SNAPPY)
136            .build();
137
138        // Write to in-memory buffer
139        let mut buffer = Vec::new();
140        let mut writer = ArrowWriter::try_new(&mut buffer, batch.schema(), Some(properties))?;
141        writer.write(&batch)?;
142        writer.close()?;
143
144        Ok(Some(buffer))
145    }
146}
