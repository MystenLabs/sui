4use proc_macro::TokenStream;
5
6use quote::quote;
7use syn::Data;
8use syn::DeriveInput;
9use syn::Fields;
10use syn::parse_macro_input;
11
12#[proc_macro_derive(SerializeRow)]
13pub fn schema_derive(input: TokenStream) -> TokenStream {
14    let input = parse_macro_input!(input as DeriveInput);
15    let struct_name = &input.ident;
16    let (schema, getter_implementation, field_count) = match &input.data {
17        Data::Struct(data_struct) => match &data_struct.fields {
18            Fields::Named(fields) => {
19                let field_count = fields.named.len();
20                let (schema_iter, getter_iter): (Vec<_>, Vec<_>) = fields
21                    .named
22                    .iter()
23                    .enumerate()
24                    .map(|(idx, field)| {
25                        let field_name = field.ident.as_ref().unwrap().to_string();
26                        (
27                            format!("\"{}\"", field_name),
28                            format!(
29                                "if idx == {} {{ return Ok((&self.{}).into()); }}",
30                                idx, field_name
31                            ),
32                        )
33                    })
34                    .unzip();
35                (schema_iter.join(", "), getter_iter.join("\n"), field_count)
36            }
37            _ => panic!("not supported struct for row serialization"),
38        },
39        _ => panic!("not supported struct for row serialization"),
40    };
41    let schema_tokens: proc_macro2::TokenStream = schema.parse().unwrap();
42    let getter_implementation_tokens: proc_macro2::TokenStream =
43        getter_implementation.parse().unwrap();
44
45    quote! {
46        impl crate::schema::RowSchema for #struct_name {
47            fn schema() -> &'static [&'static str] {
48                &[#schema_tokens]
49            }
50
51            fn column_count(&self) -> usize {
52                #field_count
53            }
54
55            fn get_column(&self, idx: usize) -> Result<crate::schema::ColumnValue<'_>, crate::schema::ColumnError> {
56                #getter_implementation_tokens
57                Err(crate::schema::ColumnError::InvalidIndex(idx))
58            }
59        }
60    }
61    .into()
62}
