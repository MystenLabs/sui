53pub async fn run_bridge_node(
54    config: BridgeNodeConfig,
55    metadata: BridgeNodePublicMetadata,
56    prometheus_registry: prometheus::Registry,
57) -> anyhow::Result<JoinHandle<()>> {
58    init_all_struct_tags();
59    let metrics = Arc::new(BridgeMetrics::new(&prometheus_registry));
60    let watchdog_config = config.watchdog_config.clone();
61    let (server_config, client_config) = config.validate(metrics.clone()).await?;
62    let sui_chain_identifier = server_config
63        .sui_client
64        .get_chain_identifier()
65        .await
66        .map_err(|e| anyhow::anyhow!("Failed to get sui chain identifier: {:?}", e))?;
67    let eth_chain_identifier = server_config
68        .eth_client
69        .get_chain_id()
70        .await
71        .map_err(|e| anyhow::anyhow!("Failed to get eth chain identifier: {:?}", e))?;
72    prometheus_registry
73        .register(mysten_metrics::bridge_uptime_metric(
74            "bridge",
75            metadata.version,
76            &sui_chain_identifier,
77            &eth_chain_identifier.to_string(),
78            client_config.is_some(),
79        ))
80        .unwrap();
81
82    let committee = Arc::new(
83        server_config
84            .sui_client
85            .get_bridge_committee()
86            .await
87            .expect("Failed to get committee"),
88    );
89    let mut handles = vec![];
90
91    // Start watchdog
92    let eth_provider = server_config.eth_client.provider();
93    let eth_bridge_proxy_address = server_config.eth_bridge_proxy_address;
94    let sui_client = server_config.sui_client.clone();
95    handles.push(spawn_logged_monitored_task!(start_watchdog(
96        watchdog_config,
97        &prometheus_registry,
98        eth_provider,
99        eth_bridge_proxy_address,
100        sui_client
101    )));
102
103    // Update voting right metrics
104    // Before reconfiguration happens we only set it once when the node starts
105    let sui_system = server_config
106        .sui_client
107        .sui_client()
108        .governance_api()
109        .get_latest_sui_system_state()
110        .await?;
111
112    // Start Client
113    if let Some(client_config) = client_config {
114        let committee_keys_to_names =
115            Arc::new(get_validator_names_by_pub_keys(&committee, &sui_system).await);
116        let client_components = start_client_components(
117            client_config,
118            committee.clone(),
119            committee_keys_to_names,
120            metrics.clone(),
121        )
122        .await?;
123        handles.extend(client_components);
124    }
125
126    let committee_name_mapping = get_committee_voting_power_by_name(&committee, &sui_system).await;
127    for (name, voting_power) in committee_name_mapping.into_iter() {
128        metrics
129            .current_bridge_voting_rights
130            .with_label_values(&[name.as_str()])
131            .set(voting_power as i64);
132    }
133
134    // Start Server
135    let socket_address = SocketAddr::new(
136        IpAddr::V4(Ipv4Addr::new(0, 0, 0, 0)),
137        server_config.server_listen_port,
138    );
139    Ok(run_server(
140        &socket_address,
141        BridgeRequestHandler::new(
142            server_config.key,
143            server_config.sui_client,
144            server_config.eth_client,
145            server_config.approved_governance_actions,
146            metrics.clone(),
147        ),
148        metrics,
149        Arc::new(metadata),
150    ))
151}

153async fn start_watchdog(
154    watchdog_config: Option<WatchdogConfig>,
155    registry: &prometheus::Registry,
156    eth_provider: Arc<Provider<MeteredEthHttpProvier>>,
157    eth_bridge_proxy_address: EthAddress,
158    sui_client: Arc<SuiBridgeClient>,
159) {
160    let watchdog_metrics = WatchdogMetrics::new(registry);
161    let (
162        _committee_address,
163        _limiter_address,
164        vault_address,
165        _config_address,
166        weth_address,
167        usdt_address,
168        wbtc_address,
169        lbtc_address,
170    ) = get_eth_contract_addresses(eth_bridge_proxy_address, &eth_provider)
171        .await
172        .unwrap_or_else(|e| panic!("get_eth_contract_addresses should not fail: {}", e));
173
174    let eth_vault_balance = EthereumVaultBalance::new(
175        eth_provider.clone(),
176        vault_address,
177        weth_address,
178        VaultAsset::WETH,
179        watchdog_metrics.eth_vault_balance.clone(),
180    )
181    .await
182    .unwrap_or_else(|e| panic!("Failed to create eth vault balance: {}", e));
183
184    let usdt_vault_balance = EthereumVaultBalance::new(
185        eth_provider.clone(),
186        vault_address,
187        usdt_address,
188        VaultAsset::USDT,
189        watchdog_metrics.usdt_vault_balance.clone(),
190    )
191    .await
192    .unwrap_or_else(|e| panic!("Failed to create usdt vault balance: {}", e));
193
194    let wbtc_vault_balance = EthereumVaultBalance::new(
195        eth_provider.clone(),
196        vault_address,
197        wbtc_address,
198        VaultAsset::WBTC,
199        watchdog_metrics.wbtc_vault_balance.clone(),
200    )
201    .await
202    .unwrap_or_else(|e| panic!("Failed to create wbtc vault balance: {}", e));
203
204    let lbtc_vault_balance = if !lbtc_address.is_zero() {
205        Some(
206            EthereumVaultBalance::new(
207                eth_provider.clone(),
208                vault_address,
209                lbtc_address,
210                VaultAsset::LBTC,
211                watchdog_metrics.lbtc_vault_balance.clone(),
212            )
213            .await
214            .unwrap_or_else(|e| panic!("Failed to create lbtc vault balance: {}", e)),
215        )
216    } else {
217        None
218    };
219
220    let eth_bridge_status = EthBridgeStatus::new(
221        eth_provider,
222        eth_bridge_proxy_address,
223        watchdog_metrics.eth_bridge_paused.clone(),
224    );
225
226    let sui_bridge_status = SuiBridgeStatus::new(
227        sui_client.clone(),
228        watchdog_metrics.sui_bridge_paused.clone(),
229    );
230
231    let mut observables: Vec<Box<dyn Observable + Send + Sync>> = vec![
232        Box::new(eth_vault_balance),
233        Box::new(usdt_vault_balance),
234        Box::new(wbtc_vault_balance),
235        Box::new(eth_bridge_status),
236        Box::new(sui_bridge_status),
237    ];
238
239    // Add lbtc_vault_balance if it's available
240    if let Some(balance) = lbtc_vault_balance {
241        observables.push(Box::new(balance));
242    }
243
244    if let Some(watchdog_config) = watchdog_config
245        && !watchdog_config.total_supplies.is_empty()
246    {
247        let total_supplies = TotalSupplies::new(
248            Arc::new(sui_client.sui_client().clone()),
249            watchdog_config.total_supplies,
250            watchdog_metrics.total_supplies.clone(),
251        );
252        observables.push(Box::new(total_supplies));
253    }
254
255    BridgeWatchDog::new(observables).run().await
256}

259async fn start_client_components(
260    client_config: BridgeClientConfig,
261    committee: Arc<BridgeCommittee>,
262    committee_keys_to_names: Arc<BTreeMap<BridgeAuthorityPublicKeyBytes, String>>,
263    metrics: Arc<BridgeMetrics>,
264) -> anyhow::Result<Vec<JoinHandle<()>>> {
265    let store: std::sync::Arc<BridgeOrchestratorTables> =
266        BridgeOrchestratorTables::new(&client_config.db_path.join("client"));
267    let sui_modules_to_watch = get_sui_modules_to_watch(
268        &store,
269        client_config.sui_bridge_module_last_processed_event_id_override,
270    );
271    let eth_contracts_to_watch = get_eth_contracts_to_watch(
272        &store,
273        &client_config.eth_contracts,
274        client_config.eth_contracts_start_block_fallback,
275        client_config.eth_contracts_start_block_override,
276    );
277
278    let sui_client = client_config.sui_client.clone();
279
280    let mut all_handles = vec![];
281    let (task_handles, eth_events_rx, _) =
282        EthSyncer::new(client_config.eth_client.clone(), eth_contracts_to_watch)
283            .run(metrics.clone())
284            .await
285            .expect("Failed to start eth syncer");
286    all_handles.extend(task_handles);
287
288    let (task_handles, sui_events_rx) = SuiSyncer::new(
289        client_config.sui_client,
290        sui_modules_to_watch,
291        metrics.clone(),
292    )
293    .run(Duration::from_secs(2))
294    .await
295    .expect("Failed to start sui syncer");
296    all_handles.extend(task_handles);
297
298    let bridge_auth_agg = Arc::new(ArcSwap::from(Arc::new(BridgeAuthorityAggregator::new(
299        committee,
300        metrics.clone(),
301        committee_keys_to_names,
302    ))));
303    // TODO: should we use one query instead of two?
304    let sui_token_type_tags = sui_client.get_token_id_map().await.unwrap();
305    let is_bridge_paused = sui_client.is_bridge_paused().await.unwrap();
306
307    let (bridge_pause_tx, bridge_pause_rx) = tokio::sync::watch::channel(is_bridge_paused);
308
309    let (sui_monitor_tx, sui_monitor_rx) = mysten_metrics::metered_channel::channel(
310        10000,
311        &mysten_metrics::get_metrics()
312            .unwrap()
313            .channel_inflight
314            .with_label_values(&["sui_monitor_queue"]),
315    );
316    let (eth_monitor_tx, eth_monitor_rx) = mysten_metrics::metered_channel::channel(
317        10000,
318        &mysten_metrics::get_metrics()
319            .unwrap()
320            .channel_inflight
321            .with_label_values(&["eth_monitor_queue"]),
322    );
323
324    let sui_token_type_tags = Arc::new(ArcSwap::from(Arc::new(sui_token_type_tags)));
325    let bridge_action_executor = BridgeActionExecutor::new(
326        sui_client.clone(),
327        bridge_auth_agg.clone(),
328        store.clone(),
329        client_config.key,
330        client_config.sui_address,
331        client_config.gas_object_ref.0,
332        sui_token_type_tags.clone(),
333        bridge_pause_rx,
334        metrics.clone(),
335    )
336    .await;
337
338    let monitor = BridgeMonitor::new(
339        sui_client.clone(),
340        sui_monitor_rx,
341        eth_monitor_rx,
342        bridge_auth_agg.clone(),
343        bridge_pause_tx,
344        sui_token_type_tags,
345        metrics.clone(),
346    );
347    all_handles.push(spawn_logged_monitored_task!(monitor.run()));
348
349    let orchestrator = BridgeOrchestrator::new(
350        sui_client,
351        sui_events_rx,
352        eth_events_rx,
353        store.clone(),
354        sui_monitor_tx,
355        eth_monitor_tx,
356        metrics,
357    );
358
359    all_handles.extend(orchestrator.run(bridge_action_executor).await);
360    Ok(all_handles)
361}

363fn get_sui_modules_to_watch(
364    store: &std::sync::Arc<BridgeOrchestratorTables>,
365    sui_bridge_module_last_processed_event_id_override: Option<EventID>,
366) -> HashMap<Identifier, Option<EventID>> {
367    let sui_bridge_modules = vec![
368        BRIDGE_MODULE_NAME.to_owned(),
369        BRIDGE_COMMITTEE_MODULE_NAME.to_owned(),
370        BRIDGE_TREASURY_MODULE_NAME.to_owned(),
371        BRIDGE_LIMITER_MODULE_NAME.to_owned(),
372    ];
373    if let Some(cursor) = sui_bridge_module_last_processed_event_id_override {
374        info!("Overriding cursor for sui bridge modules to {:?}", cursor);
375        return HashMap::from_iter(
376            sui_bridge_modules
377                .iter()
378                .map(|module| (module.clone(), Some(cursor))),
379        );
380    }
381
382    let sui_bridge_module_stored_cursor = store
383        .get_sui_event_cursors(&sui_bridge_modules)
384        .expect("Failed to get eth sui event cursors from storage");
385    let mut sui_modules_to_watch = HashMap::new();
386    for (module_identifier, cursor) in sui_bridge_modules
387        .iter()
388        .zip(sui_bridge_module_stored_cursor)
389    {
390        if cursor.is_none() {
391            info!(
392                "No cursor found for sui bridge module {} in storage or config override, query start from the beginning.",
393                module_identifier
394            );
395        }
396        sui_modules_to_watch.insert(module_identifier.clone(), cursor);
397    }
398    sui_modules_to_watch
399}

401fn get_eth_contracts_to_watch(
402    store: &std::sync::Arc<BridgeOrchestratorTables>,
403    eth_contracts: &[EthAddress],
404    eth_contracts_start_block_fallback: u64,
405    eth_contracts_start_block_override: Option<u64>,
406) -> HashMap<EthAddress, u64> {
407    let stored_eth_cursors = store
408        .get_eth_event_cursors(eth_contracts)
409        .expect("Failed to get eth event cursors from storage");
410    let mut eth_contracts_to_watch = HashMap::new();
411    for (contract, stored_cursor) in eth_contracts.iter().zip(stored_eth_cursors) {
412        // start block precedence:
413        // eth_contracts_start_block_override > stored cursor > eth_contracts_start_block_fallback
414        match (eth_contracts_start_block_override, stored_cursor) {
415            (Some(override_), _) => {
416                eth_contracts_to_watch.insert(*contract, override_);
417                info!(
418                    "Overriding cursor for eth bridge contract {} to {}. Stored cursor: {:?}",
419                    contract, override_, stored_cursor
420                );
421            }
422            (None, Some(stored_cursor)) => {
423                // +1: The stored value is the last block that was processed, so we start from the next block.
424                eth_contracts_to_watch.insert(*contract, stored_cursor + 1);
425            }
426            (None, None) => {
427                // If no cursor is found, start from the fallback block.
428                eth_contracts_to_watch.insert(*contract, eth_contracts_start_block_fallback);
429            }
430        }
431    }
432    eth_contracts_to_watch
433}
