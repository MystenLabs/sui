77#[async_trait]
78impl Datasource<RawEthData> for EthSubscriptionDatasource {
79    async fn start_data_retrieval(
80        &self,
81        task: Task,
82        data_sender: DataSender<RawEthData>,
83    ) -> Result<JoinHandle<Result<(), Error>>, Error> {
84        assert!(
85            task.is_live_task,
86            "EthSubscriptionDatasource only supports live tasks"
87        );
88        let filter = Filter::new()
89            .address(self.addresses.clone())
90            .from_block(task.start_checkpoint)
91            .to_block(task.target_checkpoint);
92
93        let eth_ws_url = self.eth_ws_url.clone();
94        let task_name = task.task_name.clone();
95        let task_name_clone = task_name.clone();
96        let progress_metric = self
97            .metrics
98            .get_tasks_latest_retrieved_checkpoints()
99            .with_label_values(&[task.name_prefix(), task.type_str()]);
100        let handle = spawn_monitored_task!(async move {
101            let eth_ws_client = Provider::<Ws>::connect(&eth_ws_url).await.tap_err(|e| {
102                tracing::error!("Failed to connect to websocket: {:?}", e);
103            })?;
104
105            let mut log_stream = eth_ws_client.subscribe_logs(&filter).await.tap_err(|e| {
106                tracing::error!("Failed to subscribe logs: {:?}", e);
107            })?;
108            // Check latest block height every 5 sec
109            let mut interval = tokio::time::interval(Duration::from_secs(5));
110            interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
111            loop {
112                select! {
113                    log = log_stream.next() => {
114                        if let Some(log) = log {
115                            Self::handle_log(&task_name_clone, log, &eth_ws_client, &data_sender).await;
116                        } else {
117                            panic!("EthSubscriptionDatasource log stream ended unexpectedly");
118                        }
119                    }
120                    _ = interval.tick() => {
121                        let Ok(Ok(block_num)) = retry_with_max_elapsed_time!(
122                            eth_ws_client.get_block_number(),
123                            Duration::from_secs(30000)
124                        ) else {
125                            tracing::error!("Failed to get block number");
126                            continue;
127                        };
128                        progress_metric.set(block_num.as_u64() as i64);
129                    }
130                }
131            }
132        });
133        Ok(handle)
134    }
135
136    async fn get_live_task_starting_checkpoint(&self) -> Result<u64, Error> {
137        self.eth_client
138            .get_last_finalized_block_id()
139            .await
140            .map_err(|e| anyhow::anyhow!("Failed to get last finalized block id: {:?}", e))
141    }
142
143    fn get_genesis_height(&self) -> u64 {
144        self.genesis_block
145    }
146
147    fn metric_provider(&self) -> &dyn IndexerMetricProvider {
148        self.metrics.as_ref()
149    }
150}

152impl EthSubscriptionDatasource {
153    async fn handle_log(
154        task_name: &str,
155        log: Log,
156        eth_ws_client: &Provider<Ws>,
157        data_sender: &DataSender<RawEthData>,
158    ) {
159        tracing::info!(
160            task_name,
161            "EthSubscriptionDatasource retrieved log: {:?}",
162            log
163        );
164        // TODO: enable a shared cache for blocks that can be used by both the subscription and finalized sync
165        let mut cached_blocks: HashMap<u64, Block<H256>> = HashMap::new();
166        let raw_log = RawEthLog {
167            block_number: log
168                .block_number
169                .ok_or(BridgeError::ProviderError(
170                    "Provider returns log without block_number".into(),
171                ))
172                .unwrap()
173                .as_u64(),
174            tx_hash: log
175                .transaction_hash
176                .ok_or(BridgeError::ProviderError(
177                    "Provider returns log without transaction_hash".into(),
178                ))
179                .unwrap(),
180            log,
181        };
182
183        let block_number = raw_log.block_number();
184
185        let block = if let Some(cached_block) = cached_blocks.get(&block_number) {
186            cached_block.clone()
187        } else {
188            let Ok(Ok(Some(block))) = retry_with_max_elapsed_time!(
189                eth_ws_client.get_block(block_number),
190                Duration::from_secs(30000)
191            ) else {
192                panic!("Unable to get block from provider");
193            };
194
195            cached_blocks.insert(block_number, block.clone());
196            block
197        };
198
199        let Ok(Ok(Some(transaction))) = retry_with_max_elapsed_time!(
200            eth_ws_client.get_transaction(raw_log.tx_hash),
201            Duration::from_secs(30000)
202        ) else {
203            panic!("Unable to get transaction from provider");
204        };
205        tracing::info!(
206            task_name,
207            "Sending data from EthSubscriptionDatasource: {:?}",
208            (raw_log.tx_hash, block_number)
209        );
210        let raw_eth_data = vec![RawEthData {
211            log: raw_log,
212            block,
213            transaction,
214            is_finalized: false,
215        }];
216        data_sender
217            .send((block_number, raw_eth_data))
218            .await
219            .unwrap_or_else(|e| {
220                tracing::error!(
221                    task_name,
222                    "Failed to send data from EthSubscriptionDatasource: {:?}",
223                    e
224                );
225            });
226    }
227}

321async fn loop_retrieve_and_process_live_finalized_logs(
322    task: Task,
323    client: Arc<EthClient<MeteredEthHttpProvier>>,
324    provider: Arc<Provider<Http>>,
325    addresses: Vec<EthAddress>,
326    data_sender: DataSender<RawEthData>,
327    bridge_metrics: Arc<BridgeMetrics>,
328    progress_metric: IntGauge,
329) -> Result<(), Error> {
330    let task_name = task.task_name.clone();
331    let starting_checkpoint = task.start_checkpoint;
332    let eth_contracts_to_watch = HashMap::from_iter(
333        addresses
334            .iter()
335            .map(|address| (*address, starting_checkpoint)),
336    );
337    let (_, mut eth_events_rx, _) = EthSyncer::new(client.clone(), eth_contracts_to_watch)
338        .run(bridge_metrics.clone())
339        .await
340        .expect("Failed to start eth syncer");
341
342    // EthSyncer sends items even when there is no matching events.
343    // We leverge this to update the progress metric.
344    while let Some((_, block, logs)) = eth_events_rx.recv().await {
345        let raw_logs: Vec<RawEthLog> = logs
346            .into_iter()
347            .map(|log| RawEthLog {
348                block_number: block,
349                tx_hash: log.tx_hash,
350                log: log.log,
351            })
352            .collect();
353
354        process_logs(
355            &task_name,
356            raw_logs,
357            provider.clone(),
358            data_sender.clone(),
359            block,
360            true,
361        )
362        .await
363        .expect("Failed to process logs");
364        progress_metric.set(block as i64);
365    }
366
367    panic!("Eth finalized syncer live task stopped unexpectedly");
368}

370async fn loop_retrieve_and_process_log_range(
371    task: Task,
372    client: Arc<EthClient<MeteredEthHttpProvier>>,
373    provider: Arc<Provider<Http>>,
374    addresses: Vec<EthAddress>,
375    data_sender: DataSender<RawEthData>,
376    progress_metric: IntGauge,
377) -> Result<(), Error> {
378    let task_name = task.task_name.clone();
379    let starting_checkpoint = task.start_checkpoint;
380    let target_checkpoint = task.target_checkpoint;
381    let mut all_logs = Vec::new();
382    let mut current_start = starting_checkpoint;
383
384    while current_start <= target_checkpoint {
385        // Calculate the end of the current chunk
386        let current_end = (current_start + MAX_LOG_QUERY_RANGE - 1).min(target_checkpoint);
387
388        // Retry the request for the current chunk
389        let Ok(Ok(logs)) = retry_with_max_elapsed_time!(
390            client.get_raw_events_in_range(addresses.clone(), current_start, current_end),
391            Duration::from_secs(30000)
392        ) else {
393            panic!(
394                "Unable to get logs from provider for range {} to {}",
395                current_start, current_end
396            );
397        };
398
399        // Add the logs from this chunk to the total
400        all_logs.extend(logs);
401
402        // Update the start for the next chunk
403        current_start = current_end + 1;
404    }
405
406    process_logs(
407        &task_name,
408        all_logs,
409        provider.clone(),
410        data_sender.clone(),
411        target_checkpoint,
412        true,
413    )
414    .await
415    .tap_ok(|_| {
416        tracing::info!(task_name, "Finished processing range");
417    })
418    .tap_err(|e| {
419        tracing::error!(task_name, "Failed to process logs: {:?}", e);
420    })
421    .expect("Process logs should not fail");
422    progress_metric.set(target_checkpoint as i64);
423    Ok::<_, Error>(())
424}

426async fn process_logs(
427    task_name: &str,
428    logs: Vec<RawEthLog>,
429    provider: Arc<Provider<Http>>,
430    data_sender: DataSender<RawEthData>,
431    block_height: u64,
432    is_finalized: bool,
433) -> Result<(), Error> {
434    let mut data = Vec::new();
435    let mut cached_blocks: HashMap<u64, Block<H256>> = HashMap::new();
436
437    for log in logs {
438        let block = if let Some(cached_block) = cached_blocks.get(&log.block_number) {
439            cached_block.clone()
440        } else {
441            // TODO: add block query parallelism
442            let Ok(Ok(Some(block))) = retry_with_max_elapsed_time!(
443                provider.get_block(log.block_number),
444                Duration::from_secs(30000)
445            ) else {
446                panic!("Unable to get block from provider");
447            };
448
449            cached_blocks.insert(log.block_number, block.clone());
450            block
451        };
452
453        let Ok(Ok(Some(transaction))) = retry_with_max_elapsed_time!(
454            provider.get_transaction(log.tx_hash),
455            Duration::from_secs(30000)
456        ) else {
457            panic!("Unable to get transaction from provider");
458        };
459
460        data.push(RawEthData {
461            log,
462            block,
463            transaction,
464            is_finalized,
465        });
466    }
467    let tx_hashes = data
468        .iter()
469        .map(|data| (data.log.tx_hash, data.block.number.map(|n| n.as_u64())))
470        .collect::<Vec<(H256, Option<u64>)>>();
471    tracing::info!(
472        task_name,
473        "Sending data from EthFinalizedSyncDatasource: {:?}",
474        tx_hashes
475    );
476    data_sender
477        .send((block_height, data))
478        .await
479        .expect("Failed to send data");
480    Ok::<_, Error>(())
481}

488impl DataMapper<RawEthData, ProcessedTxnData> for EthDataMapper {
489    fn map(
490        &self,
491        RawEthData {
492            log,
493            block,
494            transaction,
495            is_finalized,
496        }: RawEthData,
497    ) -> Result<Vec<ProcessedTxnData>, Error> {
498        let eth_bridge_event = EthBridgeEvent::try_from_log(log.log());
499        if eth_bridge_event.is_none() {
500            return Ok(vec![]);
501        }
502        self.metrics.total_eth_bridge_transactions.inc();
503        let bridge_event = eth_bridge_event.unwrap();
504        let timestamp_ms = block.timestamp.as_u64() * 1000;
505        let gas = transaction.gas;
506        let mut processed_txn_data = Vec::new();
507        let txn_sender = transaction.from.as_bytes().to_vec();
508        let txn_hash = transaction.hash.as_bytes().to_vec();
509
510        match bridge_event {
511            EthBridgeEvent::EthSuiBridgeEvents(bridge_event) => match &bridge_event {
512                EthSuiBridgeEvents::TokensDepositedFilter(bridge_event) => {
513                    info!(
514                        "Observed Eth Deposit at block: {}, tx_hash: {}",
515                        log.block_number(),
516                        log.tx_hash
517                    );
518                    self.metrics.total_eth_token_deposited.inc();
519                    processed_txn_data.push(ProcessedTxnData::TokenTransfer(TokenTransfer {
520                        chain_id: bridge_event.source_chain_id,
521                        nonce: bridge_event.nonce,
522                        block_height: log.block_number(),
523                        timestamp_ms,
524                        txn_hash: txn_hash.clone(),
525                        txn_sender: txn_sender.clone(),
526                        status: TokenTransferStatus::Deposited,
527                        gas_usage: gas.as_u64() as i64,
528                        data_source: BridgeDataSource::ETH,
529                        is_finalized,
530                        data: Some(TokenTransferData {
531                            sender_address: txn_sender.clone(),
532                            destination_chain: bridge_event.destination_chain_id,
533                            recipient_address: bridge_event.recipient_address.to_vec(),
534                            token_id: bridge_event.token_id,
535                            amount: bridge_event.sui_adjusted_amount,
536                            is_finalized,
537                        }),
538                    }));
539                }
540                EthSuiBridgeEvents::TokensClaimedFilter(bridge_event) => {
541                    info!(
542                        "Observed Eth Claim at block: {}, tx_hash: {}",
543                        log.block_number(),
544                        log.tx_hash
545                    );
546                    self.metrics.total_eth_token_transfer_claimed.inc();
547                    processed_txn_data.push(ProcessedTxnData::TokenTransfer(TokenTransfer {
548                        chain_id: bridge_event.source_chain_id,
549                        nonce: bridge_event.nonce,
550                        block_height: log.block_number(),
551                        timestamp_ms,
552                        txn_hash: txn_hash.clone(),
553                        txn_sender: txn_sender.clone(),
554                        status: TokenTransferStatus::Claimed,
555                        gas_usage: gas.as_u64() as i64,
556                        data_source: BridgeDataSource::ETH,
557                        data: None,
558                        is_finalized,
559                    }));
560                }
561                EthSuiBridgeEvents::EmergencyOperationFilter(f) => {
562                    info!(
563                        "Observed Eth Emergency Operation at block: {}, tx_hash: {}",
564                        log.block_number(),
565                        log.tx_hash
566                    );
567                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
568                        nonce: Some(f.nonce),
569                        data_source: BridgeDataSource::ETH,
570                        tx_digest: txn_hash.clone(),
571                        sender: txn_sender.clone(),
572                        timestamp_ms,
573                        action: GovernanceActionType::EmergencyOperation,
574                        data: serde_json::to_value(bridge_event)?,
575                    }));
576                }
577                EthSuiBridgeEvents::ContractUpgradedFilter(f) => {
578                    info!(
579                        "Observed Eth SuiBridge Upgrade at block: {}, tx_hash: {}",
580                        log.block_number(),
581                        log.tx_hash
582                    );
583
584                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
585                        nonce: Some(f.nonce.as_u64()),
586                        data_source: BridgeDataSource::ETH,
587                        tx_digest: txn_hash.clone(),
588                        sender: txn_sender.clone(),
589                        timestamp_ms,
590                        action: GovernanceActionType::UpgradeEVMContract,
591                        data: serde_json::to_value(bridge_event)?,
592                    }));
593                }
594
595                EthSuiBridgeEvents::InitializedFilter(_)
596                | EthSuiBridgeEvents::PausedFilter(_)
597                | EthSuiBridgeEvents::UnpausedFilter(_)
598                | EthSuiBridgeEvents::UpgradedFilter(_) => {
599                    warn!("Unexpected event {bridge_event:?}.")
600                }
601            },
602            EthBridgeEvent::EthBridgeCommitteeEvents(bridge_event) => match &bridge_event {
603                EthBridgeCommitteeEvents::BlocklistUpdatedFilter(_) => {
604                    info!(
605                        "Observed Eth Blocklist Update at block: {}, tx_hash: {}",
606                        log.block_number(),
607                        log.tx_hash
608                    );
609
610                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
611                        nonce: None,
612                        data_source: BridgeDataSource::ETH,
613                        tx_digest: txn_hash.clone(),
614                        sender: txn_sender.clone(),
615                        timestamp_ms,
616                        action: GovernanceActionType::UpdateCommitteeBlocklist,
617                        data: serde_json::to_value(bridge_event)?,
618                    }));
619                }
620                EthBridgeCommitteeEvents::BlocklistUpdatedV2Filter(f) => {
621                    info!(
622                        "Observed Eth Blocklist Update at block: {}, tx_hash: {}",
623                        log.block_number(),
624                        log.tx_hash
625                    );
626
627                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
628                        nonce: Some(f.nonce),
629                        data_source: BridgeDataSource::ETH,
630                        tx_digest: txn_hash.clone(),
631                        sender: txn_sender.clone(),
632                        timestamp_ms,
633                        action: GovernanceActionType::UpdateCommitteeBlocklist,
634                        data: serde_json::to_value(bridge_event)?,
635                    }));
636                }
637                EthBridgeCommitteeEvents::ContractUpgradedFilter(f) => {
638                    info!(
639                        "Observed Eth BridgeCommittee Upgrade at block: {}, tx_hash: {}",
640                        log.block_number(),
641                        log.tx_hash
642                    );
643
644                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
645                        nonce: Some(f.nonce.as_u64()),
646                        data_source: BridgeDataSource::ETH,
647                        tx_digest: txn_hash.clone(),
648                        sender: txn_sender.clone(),
649                        timestamp_ms,
650                        action: GovernanceActionType::UpgradeEVMContract,
651                        data: serde_json::to_value(bridge_event)?,
652                    }));
653                }
654                EthBridgeCommitteeEvents::InitializedFilter(_)
655                | EthBridgeCommitteeEvents::UpgradedFilter(_) => {
656                    warn!("Unexpected event {bridge_event:?}.")
657                }
658            },
659            EthBridgeEvent::EthBridgeLimiterEvents(bridge_event) => match &bridge_event {
660                EthBridgeLimiterEvents::LimitUpdatedFilter(_) => {
661                    info!(
662                        "Observed Eth BridgeLimiter Update at block: {}, tx_hash: {}",
663                        log.block_number(),
664                        log.tx_hash
665                    );
666
667                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
668                        nonce: None,
669                        data_source: BridgeDataSource::ETH,
670                        tx_digest: txn_hash.clone(),
671                        sender: txn_sender.clone(),
672                        timestamp_ms,
673                        action: GovernanceActionType::UpdateBridgeLimit,
674                        data: serde_json::to_value(bridge_event)?,
675                    }));
676                }
677                EthBridgeLimiterEvents::LimitUpdatedV2Filter(f) => {
678                    info!(
679                        "Observed Eth BridgeLimiter Update at block: {}, tx_hash: {}",
680                        log.block_number(),
681                        log.tx_hash
682                    );
683
