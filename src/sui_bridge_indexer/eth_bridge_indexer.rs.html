76#[async_trait]
77impl Datasource<RawEthData> for EthSubscriptionDatasource {
78    async fn start_data_retrieval(
79        &self,
80        task: Task,
81        data_sender: DataSender<RawEthData>,
82    ) -> Result<JoinHandle<Result<(), Error>>, Error> {
83        assert!(
84            task.is_live_task,
85            "EthSubscriptionDatasource only supports live tasks"
86        );
87        let filter = Filter::new()
88            .address(self.addresses.clone())
89            .from_block(task.start_checkpoint)
90            .to_block(task.target_checkpoint);
91
92        let eth_ws_url = self.eth_ws_url.clone();
93        let task_name = task.task_name.clone();
94        let task_name_clone = task_name.clone();
95        let progress_metric = self
96            .metrics
97            .get_tasks_latest_retrieved_checkpoints()
98            .with_label_values(&[task.name_prefix(), task.type_str()]);
99        let handle = spawn_monitored_task!(async move {
100            let eth_ws_provider = get_eth_ws_provider(&eth_ws_url).await?;
101
102            let subscription = eth_ws_provider.subscribe_logs(&filter).await.tap_err(|e| {
103                tracing::error!("Failed to subscribe logs: {:?}", e);
104            })?;
105            let mut log_stream = subscription.into_stream();
106            // Check latest block height every 5 sec
107            let mut interval = tokio::time::interval(Duration::from_secs(5));
108            interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
109            loop {
110                select! {
111                    log = log_stream.next() => {
112                        if let Some(log) = log {
113                            Self::handle_log(&task_name_clone, log, eth_ws_provider.clone(), &data_sender).await;
114                        } else {
115                            panic!("EthSubscriptionDatasource log stream ended unexpectedly");
116                        }
117                    }
118                    _ = interval.tick() => {
119                        let Ok(Ok(block_num)) = retry_with_max_elapsed_time!(
120                            eth_ws_provider.get_block_number(),
121                            Duration::from_secs(30000)
122                        ) else {
123                            tracing::error!("Failed to get block number");
124                            continue;
125                        };
126                        progress_metric.set(block_num as i64);
127                    }
128                }
129            }
130        });
131        Ok(handle)
132    }
133
134    async fn get_live_task_starting_checkpoint(&self) -> Result<u64, Error> {
135        self.eth_client
136            .get_last_finalized_block_id()
137            .await
138            .map_err(|e| anyhow::anyhow!("Failed to get last finalized block id: {:?}", e))
139    }
140
141    fn get_genesis_height(&self) -> u64 {
142        self.genesis_block
143    }
144
145    fn metric_provider(&self) -> &dyn IndexerMetricProvider {
146        self.metrics.as_ref()
147    }

150impl EthSubscriptionDatasource {
151    async fn handle_log(
152        task_name: &str,
153        log: Log,
154        eth_ws_provider: EthWsProvider,
155        data_sender: &DataSender<RawEthData>,
156    ) {
157        tracing::info!(
158            task_name,
159            "EthSubscriptionDatasource retrieved log: {:?}",
160            log
161        );
162        // TODO: enable a shared cache for blocks that can be used by both the subscription and finalized sync
163        let mut cached_blocks: HashMap<u64, Block<B256>> = HashMap::new();
164        let raw_log = RawEthLog {
165            block_number: log
166                .block_number
167                .ok_or(BridgeError::ProviderError(
168                    "Provider returns log without block_number".into(),
169                ))
170                .unwrap(),
171            tx_hash: log
172                .transaction_hash
173                .ok_or(BridgeError::ProviderError(
174                    "Provider returns log without transaction_hash".into(),
175                ))
176                .unwrap(),
177            log,
178        };
179
180        let block_number = raw_log.block_number();
181
182        let block = if let Some(cached_block) = cached_blocks.get(&block_number) {
183            cached_block.clone()
184        } else {
185            let Ok(Ok(Some(block))) = retry_with_max_elapsed_time!(
186                eth_ws_provider.get_block_by_number(block_number.into()),
187                Duration::from_secs(30000)
188            ) else {
189                panic!("Unable to get block from provider");
190            };
191
192            let block = block.map_transactions(|tx| tx.block_hash.unwrap_or_default());
193            cached_blocks.insert(block_number, block.clone());
194            block
195        };
196
197        let Ok(Ok(Some(transaction))) = retry_with_max_elapsed_time!(
198            eth_ws_provider.get_transaction_by_hash(raw_log.tx_hash),
199            Duration::from_secs(30000)
200        ) else {
201            panic!("Unable to get transaction from provider");
202        };
203        tracing::info!(
204            task_name,
205            "Sending data from EthSubscriptionDatasource: {:?}",
206            (raw_log.tx_hash, block_number)
207        );
208        let raw_eth_data = vec![RawEthData {
209            log: raw_log,
210            block,
211            transaction,
212            is_finalized: false,
213        }];
214        data_sender
215            .send((block_number, raw_eth_data))
216            .await
217            .unwrap_or_else(|e| {
218                tracing::error!(
219                    task_name,
220                    "Failed to send data from EthSubscriptionDatasource: {:?}",
221                    e
222                );
223            });
224    }

316async fn loop_retrieve_and_process_live_finalized_logs(
317    task: Task,
318    client: Arc<EthClient>,
319    provider: EthProvider,
320    addresses: Vec<EthAddress>,
321    data_sender: DataSender<RawEthData>,
322    bridge_metrics: Arc<BridgeMetrics>,
323    progress_metric: IntGauge,
324) -> Result<(), Error> {
325    let task_name = task.task_name.clone();
326    let starting_checkpoint = task.start_checkpoint;
327    let eth_contracts_to_watch = HashMap::from_iter(
328        addresses
329            .iter()
330            .map(|address| (*address, starting_checkpoint)),
331    );
332    let (_, mut eth_events_rx, _) = EthSyncer::new(client.clone(), eth_contracts_to_watch)
333        .run(bridge_metrics.clone())
334        .await
335        .expect("Failed to start eth syncer");
336
337    // EthSyncer sends items even when there is no matching events.
338    // We leverge this to update the progress metric.
339    while let Some((_, block, logs)) = eth_events_rx.recv().await {
340        let raw_logs: Vec<RawEthLog> = logs
341            .into_iter()
342            .map(|log| RawEthLog {
343                block_number: block,
344                tx_hash: log.tx_hash,
345                log: log.log,
346            })
347            .collect();
348
349        process_logs(
350            &task_name,
351            raw_logs,
352            provider.clone(),
353            data_sender.clone(),
354            block,
355            true,
356        )
357        .await
358        .expect("Failed to process logs");
359        progress_metric.set(block as i64);
360    }
361
362    panic!("Eth finalized syncer live task stopped unexpectedly");
363}

365async fn loop_retrieve_and_process_log_range(
366    task: Task,
367    client: Arc<EthClient>,
368    provider: EthProvider,
369    addresses: Vec<EthAddress>,
370    data_sender: DataSender<RawEthData>,
371    progress_metric: IntGauge,
372) -> Result<(), Error> {
373    let task_name = task.task_name.clone();
374    let starting_checkpoint = task.start_checkpoint;
375    let target_checkpoint = task.target_checkpoint;
376    let mut all_logs = Vec::new();
377    let mut current_start = starting_checkpoint;
378
379    while current_start <= target_checkpoint {
380        // Calculate the end of the current chunk
381        let current_end = (current_start + MAX_LOG_QUERY_RANGE - 1).min(target_checkpoint);
382
383        // Retry the request for the current chunk
384        let Ok(Ok(logs)) = retry_with_max_elapsed_time!(
385            client.get_raw_events_in_range(addresses.clone(), current_start, current_end),
386            Duration::from_secs(30000)
387        ) else {
388            panic!(
389                "Unable to get logs from provider for range {} to {}",
390                current_start, current_end
391            );
392        };
393
394        // Add the logs from this chunk to the total
395        all_logs.extend(logs);
396
397        // Update the start for the next chunk
398        current_start = current_end + 1;
399    }
400
401    process_logs(
402        &task_name,
403        all_logs,
404        provider.clone(),
405        data_sender.clone(),
406        target_checkpoint,
407        true,
408    )
409    .await
410    .tap_ok(|_| {
411        tracing::info!(task_name, "Finished processing range");
412    })
413    .tap_err(|e| {
414        tracing::error!(task_name, "Failed to process logs: {:?}", e);
415    })
416    .expect("Process logs should not fail");
417    progress_metric.set(target_checkpoint as i64);
418    Ok::<_, Error>(())
419}

421async fn process_logs(
422    task_name: &str,
423    logs: Vec<RawEthLog>,
424    provider: EthProvider,
425    data_sender: DataSender<RawEthData>,
426    block_height: u64,
427    is_finalized: bool,
428) -> Result<(), Error> {
429    let mut data = Vec::new();
430    let mut cached_blocks: HashMap<u64, Block<B256>> = HashMap::new();
431
432    for log in logs {
433        let block = if let Some(cached_block) = cached_blocks.get(&log.block_number) {
434            cached_block.clone()
435        } else {
436            // TODO: add block query parallelism
437            let Ok(Ok(Some(block))) = retry_with_max_elapsed_time!(
438                provider.get_block_by_number(log.block_number.into()),
439                Duration::from_secs(30000)
440            ) else {
441                panic!("Unable to get block from provider");
442            };
443
444            let block = block.map_transactions(|tx| tx.block_hash.unwrap_or_default());
445            cached_blocks.insert(log.block_number, block.clone());
446            block
447        };
448
449        let Ok(Ok(Some(transaction))) = retry_with_max_elapsed_time!(
450            provider.get_transaction_by_hash(log.tx_hash),
451            Duration::from_secs(30000)
452        ) else {
453            panic!("Unable to get transaction from provider");
454        };
455
456        data.push(RawEthData {
457            log,
458            block,
459            transaction,
460            is_finalized,
461        });
462    }
463    let tx_hashes = data
464        .iter()
465        .map(|data| (data.log.tx_hash, data.block.number()))
466        .collect::<Vec<(B256, u64)>>();
467    tracing::info!(
468        task_name,
469        "Sending data from EthFinalizedSyncDatasource: {:?}",
470        tx_hashes
471    );
472    data_sender
473        .send((block_height, data))
474        .await
475        .expect("Failed to send data");
476    Ok::<_, Error>(())
477}

484impl DataMapper<RawEthData, ProcessedTxnData> for EthDataMapper {
485    fn map(
486        &self,
487        RawEthData {
488            log,
489            block,
490            transaction,
491            is_finalized,
492        }: RawEthData,
493    ) -> Result<Vec<ProcessedTxnData>, Error> {
494        let eth_bridge_event = EthBridgeEvent::try_from_log(log.log());
495        if eth_bridge_event.is_none() {
496            return Ok(vec![]);
497        }
498        self.metrics.total_eth_bridge_transactions.inc();
499        let bridge_event = eth_bridge_event.unwrap();
500        let timestamp_ms = block.header.timestamp * 1000;
501        let gas = alloy::consensus::Transaction::gas_limit(&transaction);
502        let mut processed_txn_data = Vec::new();
503        let txn_sender = transaction.inner.signer().to_vec();
504        let txn_hash = transaction.inner.hash().to_vec();
505
506        match bridge_event {
507            EthBridgeEvent::EthSuiBridgeEvents(bridge_event) => match &bridge_event {
508                EthSuiBridgeEvents::TokensDeposited(bridge_event) => {
509                    info!(
510                        "Observed Eth Deposit at block: {}, tx_hash: {}",
511                        log.block_number(),
512                        log.tx_hash
513                    );
514                    self.metrics.total_eth_token_deposited.inc();
515                    processed_txn_data.push(ProcessedTxnData::TokenTransfer(TokenTransfer {
516                        chain_id: bridge_event.sourceChainID,
517                        nonce: bridge_event.nonce,
518                        block_height: log.block_number(),
519                        timestamp_ms,
520                        txn_hash: txn_hash.clone(),
521                        txn_sender: txn_sender.clone(),
522                        status: TokenTransferStatus::Deposited,
523                        gas_usage: gas as i64,
524                        data_source: BridgeDataSource::ETH,
525                        is_finalized,
526                        data: Some(TokenTransferData {
527                            sender_address: txn_sender.clone(),
528                            destination_chain: bridge_event.destinationChainID,
529                            recipient_address: bridge_event.recipientAddress.to_vec(),
530                            token_id: bridge_event.tokenID,
531                            amount: bridge_event.suiAdjustedAmount,
532                            is_finalized,
533                        }),
534                    }));
535                }
536                EthSuiBridgeEvents::TokensDepositedV2(bridge_event) => {
537                    info!(
538                        "Observed Eth Deposit at block: {}, tx_hash: {}",
539                        log.block_number(),
540                        log.tx_hash
541                    );
542                    self.metrics.total_eth_token_deposited.inc();
543                    processed_txn_data.push(ProcessedTxnData::TokenTransfer(TokenTransfer {
544                        chain_id: bridge_event.sourceChainID,
545                        nonce: bridge_event.nonce,
546                        block_height: log.block_number(),
547                        timestamp_ms,
548                        txn_hash: txn_hash.clone(),
549                        txn_sender: txn_sender.clone(),
550                        status: TokenTransferStatus::Deposited,
551                        gas_usage: gas as i64,
552                        data_source: BridgeDataSource::ETH,
553                        is_finalized,
554                        data: Some(TokenTransferData {
555                            sender_address: txn_sender.clone(),
556                            destination_chain: bridge_event.destinationChainID,
557                            recipient_address: bridge_event.recipientAddress.to_vec(),
558                            token_id: bridge_event.tokenID,
559                            amount: bridge_event.suiAdjustedAmount,
560                            is_finalized,
561                        }),
562                    }));
563                }
564                EthSuiBridgeEvents::TokensClaimed(bridge_event) => {
565                    info!(
566                        "Observed Eth Claim at block: {}, tx_hash: {}",
567                        log.block_number(),
568                        log.tx_hash
569                    );
570                    self.metrics.total_eth_token_transfer_claimed.inc();
571                    processed_txn_data.push(ProcessedTxnData::TokenTransfer(TokenTransfer {
572                        chain_id: bridge_event.sourceChainID,
573                        nonce: bridge_event.nonce,
574                        block_height: log.block_number(),
575                        timestamp_ms,
576                        txn_hash: txn_hash.clone(),
577                        txn_sender: txn_sender.clone(),
578                        status: TokenTransferStatus::Claimed,
579                        gas_usage: gas as i64,
580                        data_source: BridgeDataSource::ETH,
581                        data: None,
582                        is_finalized,
583                    }));
584                }
585                EthSuiBridgeEvents::EmergencyOperation(f) => {
586                    info!(
587                        "Observed Eth Emergency Operation at block: {}, tx_hash: {}",
588                        log.block_number(),
589                        log.tx_hash
590                    );
591                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
592                        nonce: Some(f.nonce),
593                        data_source: BridgeDataSource::ETH,
594                        tx_digest: txn_hash.clone(),
595                        sender: txn_sender.clone(),
596                        timestamp_ms,
597                        action: GovernanceActionType::EmergencyOperation,
598                        data: serde_json::to_value(bridge_event)?,
599                    }));
600                }
601                EthSuiBridgeEvents::ContractUpgraded(f) => {
602                    info!(
603                        "Observed Eth SuiBridge Upgrade at block: {}, tx_hash: {}",
604                        log.block_number(),
605                        log.tx_hash
606                    );
607
608                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
609                        nonce: Some(f.nonce.to::<u64>()),
610                        data_source: BridgeDataSource::ETH,
611                        tx_digest: txn_hash.clone(),
612                        sender: txn_sender.clone(),
613                        timestamp_ms,
614                        action: GovernanceActionType::UpgradeEVMContract,
615                        data: serde_json::to_value(bridge_event)?,
616                    }));
617                }
618
619                EthSuiBridgeEvents::Initialized(_)
620                | EthSuiBridgeEvents::Paused(_)
621                | EthSuiBridgeEvents::Unpaused(_)
622                | EthSuiBridgeEvents::Upgraded(_) => {
623                    warn!("Unexpected event {bridge_event:?}.")
624                }
625            },
626            EthBridgeEvent::EthBridgeCommitteeEvents(bridge_event) => match &bridge_event {
627                EthBridgeCommitteeEvents::BlocklistUpdated(_) => {
628                    info!(
629                        "Observed Eth Blocklist Update at block: {}, tx_hash: {}",
630                        log.block_number(),
631                        log.tx_hash
632                    );
633
634                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
635                        nonce: None,
636                        data_source: BridgeDataSource::ETH,
637                        tx_digest: txn_hash.clone(),
638                        sender: txn_sender.clone(),
639                        timestamp_ms,
640                        action: GovernanceActionType::UpdateCommitteeBlocklist,
641                        data: serde_json::to_value(bridge_event)?,
642                    }));
643                }
644                EthBridgeCommitteeEvents::BlocklistUpdatedV2(f) => {
645                    info!(
646                        "Observed Eth Blocklist Update at block: {}, tx_hash: {}",
647                        log.block_number(),
648                        log.tx_hash
649                    );
650
651                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
652                        nonce: Some(f.nonce),
653                        data_source: BridgeDataSource::ETH,
654                        tx_digest: txn_hash.clone(),
655                        sender: txn_sender.clone(),
656                        timestamp_ms,
657                        action: GovernanceActionType::UpdateCommitteeBlocklist,
658                        data: serde_json::to_value(bridge_event)?,
659                    }));
660                }
661                EthBridgeCommitteeEvents::ContractUpgraded(f) => {
662                    info!(
663                        "Observed Eth BridgeCommittee Upgrade at block: {}, tx_hash: {}",
664                        log.block_number(),
665                        log.tx_hash
666                    );
667
668                    processed_txn_data.push(ProcessedTxnData::GovernanceAction(GovernanceAction {
669                        nonce: Some(f.nonce.to::<u64>()),
670                        data_source: BridgeDataSource::ETH,
671                        tx_digest: txn_hash.clone(),
672                        sender: txn_sender.clone(),
673                        timestamp_ms,
674                        action: GovernanceActionType::UpgradeEVMContract,
675                        data: serde_json::to_value(bridge_event)?,
676                    }));
677                }
678                EthBridgeCommitteeEvents::Initialized(_)
679                | EthBridgeCommitteeEvents::Upgraded(_) => {
680                    warn!("Unexpected event {bridge_event:?}.")
681                }
682            },
683            EthBridgeEvent::EthBridgeLimiterEvents(bridge_event) => match &bridge_event {
684                EthBridgeLimiterEvents::LimitUpdated(_) => {
