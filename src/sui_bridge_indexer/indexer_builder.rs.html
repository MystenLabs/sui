76    name: String,

85    pub async fn start<T, R>(mut self) -> Result<(), Error>
86    where
87        D: Datasource<T> + 'static,
88        M: DataMapper<T, R> + 'static,
89        P: Persistent<R> + 'static,
90        T: Send,
91    {
92        let task_name = self.name.clone();
93        // Update tasks first
94        self.update_tasks()
95            .await
96            .tap_err(|e| {
97                tracing::error!(task_name, "Failed to update tasks: {:?}", e);
98            })
99            .tap_ok(|_| {
100                tracing::info!(task_name, "Tasks updated.");
101            })?;
102
103        // get ongoing tasks from storage
104        let ongoing_tasks = self
105            .storage
106            .get_ongoing_tasks(&self.name)
107            .await
108            .tap_err(|e| {
109                tracing::error!(task_name, "Failed to get updated tasks: {:?}", e);
110            })
111            .tap_ok(|tasks| {
112                tracing::info!(task_name, "Got updated tasks: {:?}", tasks);
113            })?;
114
115        // Start latest checkpoint worker
116        // Tasks are ordered in checkpoint descending order, realtime update task always come first
117        // tasks won't be empty here, ok to unwrap.
118        let live_task_future = match ongoing_tasks.live_task() {
119            Some(live_task) if !self.disable_live_task => {
120                let live_task_future = self.datasource.start_ingestion_task(
121                    live_task,
122                    self.storage.clone(),
123                    self.data_mapper.clone(),
124                );
125                Some(live_task_future)
126            }
127            _ => None,
128        };
129
130        let backfill_tasks = ongoing_tasks.backfill_tasks_ordered_desc();
131        let storage_clone = self.storage.clone();
132        let data_mapper_clone = self.data_mapper.clone();
133        let datasource_clone = self.datasource.clone();
134
135        let handle = spawn_monitored_task!(async {
136            // Execute tasks one by one
137            for backfill_task in backfill_tasks {
138                if backfill_task.start_checkpoint < backfill_task.target_checkpoint {
139                    datasource_clone
140                        .start_ingestion_task(
141                            backfill_task,
142                            storage_clone.clone(),
143                            data_mapper_clone.clone(),
144                        )
145                        .await
146                        .expect("Backfill task failed");
147                }
148            }
149        });
150
151        if let Some(live_task_future) = live_task_future {
152            live_task_future.await?;
153        }
154
155        tokio::try_join!(handle)?;
156
157        Ok(())
158    }

160    async fn update_tasks<T, R>(&mut self) -> Result<(), Error>
161    where
162        P: Persistent<R>,
163        D: Datasource<T>,
164        T: Send,
165    {
166        let ongoing_tasks = self.storage.get_ongoing_tasks(&self.name).await?;
167        let largest_checkpoint = self
168            .storage
169            .get_largest_indexed_checkpoint(&self.name)
170            .await?;
171        let live_task_from_checkpoint = self.datasource.get_live_task_starting_checkpoint().await?;
172
173        // Create and update live task if needed
174        // for live task, we always start from `live_task_from_checkpoint`.
175        // What if there are older tasks with larger height? It's very
176        // unlikely, and even if it happens, we just reprocess the range.
177        // This simplifies the logic of determining task boundaries.
178        if !self.disable_live_task {
179            match ongoing_tasks.live_task() {
180                None => {
181                    self.storage
182                        .register_live_task(
183                            format!("{} - Live", self.name),
184                            live_task_from_checkpoint,
185                        )
186                        .await
187                        .tap_ok(|_| {
188                            tracing::info!(
189                                task_name = self.name.as_str(),
190                                "Created live task from {}",
191                                live_task_from_checkpoint,
192                            );
193                        })
194                        .tap_err(|e| {
195                            tracing::error!(
196                                "Failed to register live task ({}-MAX): {:?}",
197                                live_task_from_checkpoint,
198                                e
199                            );
200                        })?;
201                }
202                Some(mut live_task) => {
203                    // We still check this because in the case of slow
204                    // block generation (e.g. Ethereum), it's possible we will
205                    // stay on the same block for a bit.
206                    if live_task_from_checkpoint != live_task.start_checkpoint {
207                        let old_checkpoint = live_task.start_checkpoint;
208                        live_task.start_checkpoint = live_task_from_checkpoint;
209                        self.storage
210                            .update_task(live_task)
211                            .await
212                            .tap_ok(|_| {
213                                tracing::info!(
214                                    task_name = self.name.as_str(),
215                                    "Updated live task starting point from {} to {}",
216                                    old_checkpoint,
217                                    live_task_from_checkpoint,
218                                );
219                            })
220                            .tap_err(|e| {
221                                tracing::error!(
222                                    "Failed to update live task to ({}-MAX): {:?}",
223                                    live_task_from_checkpoint,
224                                    e
225                                );
226                            })?;
227                    }
228                }
229            }
230        }
231
232        // 2, if there is a gap between `largest_checkpoint` and `live_task_from_checkpoint`,
233        // create backfill task [largest_checkpoint + 1, live_task_from_checkpoint - 1]
234
235        // TODO: when there is a hole, we create one task for the hole, but ideally we should
236        // honor the partition size and create as needed.
237        let from_checkpoint = largest_checkpoint
238            .map(|cp| cp + 1)
239            .unwrap_or(self.datasource.get_genesis_height());
240        if from_checkpoint < live_task_from_checkpoint {
241            self.create_backfill_tasks(from_checkpoint, live_task_from_checkpoint - 1)
242                .await
243                .tap_ok(|_| {
244                    tracing::info!(
245                        task_name = self.name.as_str(),
246                        "Created backfill tasks ({}-{})",
247                        from_checkpoint,
248                        live_task_from_checkpoint - 1
249                    );
250                })
251                .tap_err(|e| {
252                    tracing::error!(
253                        task_name = self.name.as_str(),
254                        "Failed to create backfill tasks ({}-{}): {:?}",
255                        from_checkpoint,
256                        live_task_from_checkpoint - 1,
257                        e
258                    );
259                })?;
260        }
261        Ok(())
262    }

264    // Create backfill tasks according to backfill strategy
265    async fn create_backfill_tasks<R>(&mut self, mut from_cp: u64, to_cp: u64) -> Result<(), Error>
266    where
267        P: Persistent<R>,
268    {
269        match self.backfill_strategy {
270            BackfillStrategy::Simple => {
271                self.storage
272                    .register_task(
273                        format!("{} - backfill - {from_cp}:{to_cp}", self.name),
274                        from_cp,
275                        to_cp,
276                    )
277                    .await
278            }
279            BackfillStrategy::Partitioned { task_size } => {
280                // TODO: register all tasks in one DB write
281                while from_cp < to_cp {
282                    let target_cp = min(from_cp + task_size - 1, to_cp);
283                    self.storage
284                        .register_task(
285                            format!("{} - backfill - {from_cp}:{target_cp}", self.name),
286                            from_cp,
287                            target_cp,
288                        )
289                        .await?;
290                    from_cp = target_cp + 1;
291                }
292                Ok(())
293            }
294            BackfillStrategy::Disabled => Ok(()),
295        }
296    }

360#[async_trait]
361pub trait Datasource<T: Send>: Sync + Send {
362    async fn start_ingestion_task<M, P, R>(
363        &self,
364        task: Task,
365        mut storage: P,
366        data_mapper: M,
367    ) -> Result<(), Error>
368    where
369        M: DataMapper<T, R>,
370        P: Persistent<R>,
371    {
372        let task_name = task.task_name.clone();
373        let task_name_prefix = task.name_prefix();
374        let task_type_label = task.type_str();
375        let starting_checkpoint = task.start_checkpoint;
376        let target_checkpoint = task.target_checkpoint;
377        let ingestion_batch_size = std::env::var("INGESTION_BATCH_SIZE")
378            .unwrap_or(INGESTION_BATCH_SIZE.to_string())
379            .parse::<usize>()
380            .unwrap();
381        let checkpoint_channel_size = std::env::var("RETRIEVED_CHECKPOINT_CHANNEL_SIZE")
382            .unwrap_or(RETRIEVED_CHECKPOINT_CHANNEL_SIZE.to_string())
383            .parse::<usize>()
384            .unwrap();
385        tracing::info!(
386            task_name,
387            ingestion_batch_size,
388            checkpoint_channel_size,
389            "Starting ingestion task ({}-{})",
390            starting_checkpoint,
391            target_checkpoint,
392        );
393        let (data_sender, data_rx) = metered_channel::channel(
394            checkpoint_channel_size,
395            &mysten_metrics::get_metrics()
396                .unwrap()
397                .channel_inflight
398                // This metric works now when there is only 1 backfill task running per task name.
399                // It will be unusable when there are parallel backfill tasks per task name.
400                .with_label_values(&[&format!("{}-{}", task_name_prefix, task_type_label)]),
401        );
402        let is_live_task = task.is_live_task;
403        let _live_tasks_tracker = if is_live_task {
404            Some(LiveTasksTracker::new(
405                self.metric_provider()
406                    .get_inflight_live_tasks_metrics()
407                    .clone(),
408                &task_name,
409            ))
410        } else {
411            None
412        };
413        let join_handle = self.start_data_retrieval(task.clone(), data_sender).await?;
414        let processed_checkpoints_metrics = self
415            .metric_provider()
416            .get_tasks_processed_checkpoints_metric()
417            .with_label_values(&[task_name_prefix, task_type_label]);
418        // track remaining checkpoints per task, except for live task
419        let remaining_checkpoints_metric = if !is_live_task {
420            let remaining = self
421                .metric_provider()
422                .get_tasks_remaining_checkpoints_metric()
423                .with_label_values(&[task_name_prefix]);
424            remaining.set((target_checkpoint - starting_checkpoint + 1) as i64);
425            Some(remaining)
426        } else {
427            None
428        };
429
430        let mut stream = mysten_metrics::metered_channel::ReceiverStream::new(data_rx)
431            .ready_chunks(ingestion_batch_size);
432        let mut last_saved_checkpoint = None;
433        loop {
434            let batch_option = stream.next().await;
435            if batch_option.is_none() {
436                tracing::error!(task_name, "Data stream ended unexpectedly");
437                break;
438            }
439            let batch = batch_option.unwrap();
440            let mut max_height = 0;
441            let mut heights = vec![];
442            let mut data = vec![];
443            for (height, d) in batch {
444                // Filter out data with height > target_checkpoint, in case data source returns any
445                if height > target_checkpoint {
446                    tracing::warn!(
447                        task_name,
448                        height,
449                        "Received data with height > target_checkpoint, skipping."
450                    );
451                    continue;
452                }
453                max_height = std::cmp::max(max_height, height);
454                heights.push(height);
455                data.extend(d);
456            }
457            tracing::debug!(
458                task_name,
459                max_height,
460                "Ingestion task received {} blocks.",
461                heights.len(),
462            );
463            let timer = tokio::time::Instant::now();
464
465            if !data.is_empty() {
466                let timer = tokio::time::Instant::now();
467                let processed_data = data.into_iter().try_fold(vec![], |mut result, d| {
468                    result.append(&mut data_mapper.map(d)?);
469                    Ok::<Vec<_>, Error>(result)
470                })?;
471                tracing::debug!(
472                    task_name,
473                    max_height,
474                    "Data mapper processed {} blocks in {}ms.",
475                    heights.len(),
476                    timer.elapsed().as_millis(),
477                );
478                let timer = tokio::time::Instant::now();
479                // TODO: batch write data
480                // TODO: we might be able to write data and progress in a single transaction.
481                storage.write(processed_data).await?;
482                tracing::debug!(
483                    task_name,
484                    max_height,
485                    "Processed data ({} blocks) was wrote to storage in {}ms.",
486                    heights.len(),
487                    timer.elapsed().as_millis(),
488                );
489            }
490            last_saved_checkpoint = storage.save_progress(&task, &heights).await?;
491            tracing::debug!(
492                task_name,
493                max_height,
494                last_saved_checkpoint,
495                "Ingestion task processed {} blocks in {}ms",
496                heights.len(),
497                timer.elapsed().as_millis(),
498            );
499            processed_checkpoints_metrics.inc_by(heights.len() as u64);
500            if let Some(m) = &remaining_checkpoints_metric {
501                // Note this is only approximate as the data may come in out of order
502                m.set(std::cmp::max(
503                    target_checkpoint as i64 - max_height as i64,
504                    0,
505                ));
506            }
507            // If we have reached the target checkpoint, exit proactively
508            if let Some(cp) = last_saved_checkpoint
509                && cp >= target_checkpoint
510            {
511                // Task is done
512                break;
513            }
514        }
515        if is_live_task {
516            // Live task should never exit, except in unit tests
517            tracing::error!(task_name, "Live task exiting unexpectedly");
518        } else if let Some(last_saved_checkpoint) = last_saved_checkpoint {
519            if last_saved_checkpoint < target_checkpoint {
520                tracing::error!(
521                    task_name,
522                    last_saved_checkpoint,
523                    "Task exiting before reaching target checkpoint",
524                );
525            } else {
526                tracing::info!(task_name, "Backfill task is done, exiting");
527            }
528        } else {
529            tracing::error!(
530                task_name,
531                "Task exiting unexpectedly with no progress saved"
532            );
533        }
534        join_handle.abort();
535        if let Some(m) = &remaining_checkpoints_metric {
536            m.set(0)
537        }
538        join_handle.await?.tap_err(|err| {
539            tracing::error!(task_name, "Data retrieval task failed: {:?}", err);
540        })
541    }

721    #[async_trait]
722    impl<T: Send + Sync> IndexerProgressStore for InMemoryPersistent<T> {
723        async fn load_progress(&self, task_name: String) -> anyhow::Result<u64> {
724            Ok(self
725                .progress_store
726                .lock()
727                .await
728                .get(&task_name)
729                .unwrap()
730                .start_checkpoint)
731        }
732
733        async fn save_progress(
734            &mut self,
735            task: &Task,
736            checkpoint_numbers: &[u64],
737        ) -> anyhow::Result<Option<u64>> {
738            let checkpoint_number = *checkpoint_numbers.last().unwrap();
739            self.progress_store
740                .lock()
741                .await
742                .get_mut(&task.task_name)
743                .unwrap()
744                .start_checkpoint = checkpoint_number;
745            Ok(Some(checkpoint_number))
746        }
747
748        async fn get_ongoing_tasks(&self, task_prefix: &str) -> Result<Tasks, Error> {
749            let tasks = self
750                .progress_store
751                .lock()
752                .await
753                .values()
754                .filter(|task| task.task_name.starts_with(task_prefix))
755                .filter(|task| task.start_checkpoint.lt(&task.target_checkpoint))
756                .cloned()
757                .collect::<Vec<_>>();
758            Tasks::new(tasks)
759        }
760
761        async fn get_largest_indexed_checkpoint(
762            &self,
763            task_prefix: &str,
764        ) -> Result<Option<u64>, Error> {
765            let checkpoint = self
766                .progress_store
767                .lock()
768                .await
769                .values()
770                .filter(|task| task.task_name.starts_with(task_prefix))
771                .filter(|task| task.target_checkpoint.eq(&(i64::MAX as u64)))
772                .last()
773                .map(|t| t.start_checkpoint);
774
775            if checkpoint.is_some() {
776                Ok(checkpoint)
777            } else {
778                self.get_largest_backfill_task_target_checkpoint(task_prefix)
779                    .await
780            }
781        }
782
783        async fn register_task(
784            &mut self,
785            task_name: String,
786            checkpoint: u64,
787            target_checkpoint: u64,
788        ) -> Result<(), Error> {
789            let existing = self.progress_store.lock().await.insert(
790                task_name.clone(),
791                Task {
792                    task_name: task_name.clone(),
793                    start_checkpoint: checkpoint,
794                    target_checkpoint,
795                    timestamp: SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis() as u64,
796                    is_live_task: false,
797                },
798            );
799            if existing.is_some() {
800                return Err(anyhow!("Task {task_name} already exists"));
801            }
802            Ok(())
803        }
804
805        async fn register_live_task(
806            &mut self,
807            task_name: String,
808            checkpoint: u64,
809        ) -> Result<(), Error> {
810            let existing = self.progress_store.lock().await.insert(
811                task_name.clone(),
812                Task {
813                    task_name: task_name.clone(),
814                    start_checkpoint: checkpoint,
815                    target_checkpoint: LIVE_TASK_TARGET_CHECKPOINT as u64,
816                    timestamp: SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis() as u64,
817                    is_live_task: true,
818                },
819            );
820            if existing.is_some() {
821                return Err(anyhow!("Task {task_name} already exists"));
822            }
823            Ok(())
824        }
825
826        async fn update_task(&mut self, task: Task) -> Result<(), Error> {
827            self.progress_store
828                .lock()
829                .await
830                .insert(task.task_name.clone(), task);
831            Ok(())
832        }
