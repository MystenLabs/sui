4use std::time::Duration;
5use sui_json_rpc_types::SuiTransactionBlockResponseOptions;
6use sui_json_rpc_types::SuiTransactionBlockResponseQuery;
7use sui_json_rpc_types::TransactionFilter;
8use sui_sdk::SuiClient;
9use sui_types::SUI_BRIDGE_OBJECT_ID;
10use sui_types::digests::TransactionDigest;
11
12use sui_bridge::retry_with_max_elapsed_time;
13use tracing::{error, info};
14
15use crate::types::RetrievedTransaction;
16
17const QUERY_DURATION: Duration = Duration::from_secs(1);
18const SLEEP_DURATION: Duration = Duration::from_secs(5);
19
20pub async fn start_sui_tx_polling_task(
21    sui_client: SuiClient,
22    mut cursor: Option<TransactionDigest>,
23    tx: mysten_metrics::metered_channel::Sender<(
24        Vec<RetrievedTransaction>,
25        Option<TransactionDigest>,
26    )>,
27) {
28    info!("Starting SUI transaction polling task from {:?}", cursor);
29    loop {
30        let Ok(Ok(results)) = retry_with_max_elapsed_time!(
31            sui_client.read_api().query_transaction_blocks(
32                SuiTransactionBlockResponseQuery {
33                    filter: Some(TransactionFilter::InputObject(SUI_BRIDGE_OBJECT_ID)),
34                    options: Some(SuiTransactionBlockResponseOptions::full_content()),
35                },
36                cursor,
37                None,
38                false,
39            ),
40            Duration::from_secs(600)
41        ) else {
42            error!("Failed to query bridge transactions after retry");
43            continue;
44        };
45        info!("Retrieved {} bridge transactions", results.data.len());
46        let txes = match results
47            .data
48            .into_iter()
49            .map(RetrievedTransaction::try_from)
50            .collect::<anyhow::Result<Vec<_>>>()
51        {
52            Ok(data) => data,
53            Err(e) => {
54                // TOOD: Sometimes fullnode does not return checkpoint strangely. We retry instead of
55                // panicking.
56                error!(
57                    "Failed to convert retrieved transactions to sanitized format: {}",
58                    e
59                );
60                tokio::time::sleep(SLEEP_DURATION).await;
61                continue;
62            }
63        };
64        if txes.is_empty() {
65            // When there is no more new data, we are caught up, no need to stress the fullnode
66            tokio::time::sleep(QUERY_DURATION).await;
67            continue;
68        }
69        tx.send((txes, results.next_cursor))
70            .await
71            .expect("Failed to send transaction block to process");
72        cursor = results.next_cursor;
73    }
74}
