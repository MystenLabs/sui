4use crate::handlers::is_bridge_txn;
5use async_trait::async_trait;
6use diesel_async::RunQueryDsl;
7use std::sync::Arc;
8use sui_bridge_schema::models::SuiErrorTransactions;
9use sui_bridge_schema::schema::sui_error_transactions;
10use sui_indexer_alt_framework::pipeline::Processor;
11use sui_indexer_alt_framework::postgres::Connection;
12use sui_indexer_alt_framework::postgres::handler::Handler;
13use sui_indexer_alt_framework::types::effects::TransactionEffectsAPI;
14use sui_indexer_alt_framework::types::execution_status::ExecutionStatus;
15use sui_indexer_alt_framework::types::full_checkpoint_content::Checkpoint;
16use sui_indexer_alt_framework::types::transaction::TransactionDataAPI;
17
18pub struct ErrorTransactionHandler;
19
20#[async_trait]
21impl Processor for ErrorTransactionHandler {
22    const NAME: &'static str = "error_transactions";
23    type Value = SuiErrorTransactions;
24
25    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
26        let timestamp_ms = checkpoint.summary.timestamp_ms as i64;
27        let mut results = vec![];
28
29        for tx in &checkpoint.transactions {
30            if !is_bridge_txn(tx) {
31                continue;
32            }
33            if let ExecutionStatus::Failure { error, command } = tx.effects.status() {
34                results.push(SuiErrorTransactions {
35                    txn_digest: tx.transaction.digest().inner().to_vec(),
36                    timestamp_ms,
37                    failure_status: error.to_string(),
38                    cmd_idx: command.map(|idx| idx as i64),
39                    sender_address: tx.transaction.sender().to_vec(),
40                })
41            }
42        }
43        Ok(results)
44    }
45}
46
47#[async_trait]
48impl Handler for ErrorTransactionHandler {
49    async fn commit<'a>(
50        values: &[Self::Value],
51        conn: &mut Connection<'a>,
52    ) -> anyhow::Result<usize> {
53        Ok(diesel::insert_into(sui_error_transactions::table)
54            .values(values)
55            .on_conflict_do_nothing()
56            .execute(conn)
57            .await?)
58    }
59}
