33pub struct GovernanceActionHandler {

43impl GovernanceActionHandler {
44    pub fn new(metrics: Arc<BridgeIndexerMetrics>) -> Self {
45        Self {
46            update_limit_event_type: struct_tag!(BRIDGE_ADDRESS, LIMITER, UPDATE_ROUTE_LIMIT_EVENT),
47            emergency_op_event_type: struct_tag!(BRIDGE_ADDRESS, BRIDGE, EMERGENCY_OP_EVENT),
48            blocklist_event_type: struct_tag!(BRIDGE_ADDRESS, COMMITTEE, BLOCKLIST_VALIDATOR_EVENT),
49            token_reg_event_type: struct_tag!(BRIDGE_ADDRESS, TREASURY, TOKEN_REGISTRATION_EVENT),
50            update_price_event_type: struct_tag!(
51                BRIDGE_ADDRESS,
52                TREASURY,
53                UPDATE_TOKEN_PRICE_EVENT
54            ),
55            new_token_event_type: struct_tag!(BRIDGE_ADDRESS, TREASURY, NEW_TOKEN_EVENT),
56            metrics,
57        }
58    }
59}

70#[async_trait]
71impl Processor for GovernanceActionHandler {
72    const NAME: &'static str = "governance_action";
73    type Value = GovernanceAction;
74
75    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
76        let timestamp_ms = checkpoint.summary.timestamp_ms as i64;
77
78        let mut results = vec![];
79
80        for tx in &checkpoint.transactions {
81            if !is_bridge_txn(tx) {
82                continue;
83            }
84            let sender_address = tx.transaction.sender();
85
86            for ev in tx.events.iter().flat_map(|e| &e.data) {
87                use sui_bridge_schema::models::GovernanceActionType::*;
88
89                let (action, data) = match &ev.type_ {
90                    t if t == &self.update_limit_event_type => {
91                        info!(?ev, "Observed Sui Route Limit Update");
92                        let event: UpdateRouteLimitEvent = bcs::from_bytes(&ev.contents)?;
93
94                        // Critical bridge limit update metrics
95                        self.metrics
96                            .governance_actions_total
97                            .with_label_values(&["update_bridge_limit", "sui"])
98                            .inc();
99                        self.metrics
100                            .bridge_events_total
101                            .with_label_values(&["update_route_limit", "sui"])
102                            .inc();
103
104                        (UpdateBridgeLimit, serde_json::to_value(event)?)
105                    }
106                    t if t == &self.emergency_op_event_type => {
107                        info!(?ev, "Observed Sui Emergency Op");
108                        let event: EmergencyOpEvent = bcs::from_bytes(&ev.contents)?;
109
110                        // Critical security event - emergency bridge pause/unpause
111                        self.metrics
112                            .bridge_emergency_events_total
113                            .with_label_values(&["emergency_operation", "critical"])
114                            .inc();
115                        self.metrics
116                            .governance_actions_total
117                            .with_label_values(&["emergency_operation", "sui"])
118                            .inc();
119                        self.metrics
120                            .bridge_pause_status
121                            .with_label_values(&["bridge_main"])
122                            .set(if event.frozen { 0 } else { 1 });
123
124                        (EmergencyOperation, serde_json::to_value(event)?)
125                    }
126                    t if t == &self.blocklist_event_type => {
127                        info!(?ev, "Observed Sui Blocklist Validator");
128                        let event: MoveBlocklistValidatorEvent = bcs::from_bytes(&ev.contents)?;
129                        (UpdateCommitteeBlocklist, serde_json::to_value(event)?)
130                    }
131                    t if t == &self.token_reg_event_type => {
132                        info!(?ev, "Observed Sui Token Registration");
133                        let event: MoveTokenRegistrationEvent = bcs::from_bytes(&ev.contents)?;
134                        (AddSuiTokens, serde_json::to_value(event)?)
135                    }
136                    t if t == &self.update_price_event_type => {
137                        info!(?ev, "Observed Sui Token Price Update");
138                        let event: UpdateTokenPriceEvent = bcs::from_bytes(&ev.contents)?;
139                        (UpdateTokenPrices, serde_json::to_value(event)?)
140                    }
141                    t if t == &self.new_token_event_type => {
142                        info!(?ev, "Observed Sui New token event");
143                        let event: MoveNewTokenEvent = bcs::from_bytes(&ev.contents)?;
144                        (AddSuiTokens, serde_json::to_value(event)?)
145                    }
146                    _ => continue,
147                };
148
149                results.push(GovernanceAction {
150                    nonce: None,
151                    data_source: BridgeDataSource::SUI,
152                    txn_digest: tx.transaction.digest().inner().to_vec(),
153                    sender_address: sender_address.to_vec(),
154                    timestamp_ms,
155                    action,
156                    data,
157                });
158            }
159        }
160        Ok(results)
161    }
162}

164#[async_trait]
165impl Handler for GovernanceActionHandler {
166    async fn commit<'a>(
167        values: &[Self::Value],
168        conn: &mut Connection<'a>,
169    ) -> anyhow::Result<usize> {

171            .values(values)
172            .on_conflict_do_nothing()
173            .execute(conn)
174            .await?)
175    }
