43impl GovernanceActionHandler {
44    pub fn new(metrics: Arc<BridgeIndexerMetrics>) -> Self {
45        Self {
46            update_limit_event_type: struct_tag!(BRIDGE_ADDRESS, LIMITER, UPDATE_ROUTE_LIMIT_EVENT),
47            emergency_op_event_type: struct_tag!(BRIDGE_ADDRESS, BRIDGE, EMERGENCY_OP_EVENT),
48            blocklist_event_type: struct_tag!(BRIDGE_ADDRESS, COMMITTEE, BLOCKLIST_VALIDATOR_EVENT),
49            token_reg_event_type: struct_tag!(BRIDGE_ADDRESS, TREASURY, TOKEN_REGISTRATION_EVENT),
50            update_price_event_type: struct_tag!(
51                BRIDGE_ADDRESS,
52                TREASURY,
53                UPDATE_TOKEN_PRICE_EVENT
54            ),
55            new_token_event_type: struct_tag!(BRIDGE_ADDRESS, TREASURY, NEW_TOKEN_EVENT),
56            metrics,
57        }
58    }
59}

70#[async_trait]
71impl Processor for GovernanceActionHandler {
72    const NAME: &'static str = "governance_action";
73    type Value = GovernanceAction;
74
75    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> anyhow::Result<Vec<Self::Value>> {
76        let timestamp_ms = checkpoint.checkpoint_summary.timestamp_ms as i64;
77
78        let mut results = vec![];
79
80        for tx in &checkpoint.transactions {
81            if !is_bridge_txn(tx) {
82                continue;
83            }
84            let txn_digest = tx.transaction.digest().inner();
85            let sender_address = tx.transaction.sender_address();
86
87            for ev in tx.events.iter().flat_map(|e| &e.data) {
88                use sui_bridge_schema::models::GovernanceActionType::*;
89
90                let (action, data) = match &ev.type_ {
91                    t if t == &self.update_limit_event_type => {
92                        info!(?ev, "Observed Sui Route Limit Update");
93                        let event: UpdateRouteLimitEvent = bcs::from_bytes(&ev.contents)?;
94
95                        // Critical bridge limit update metrics
96                        self.metrics
97                            .governance_actions_total
98                            .with_label_values(&["update_bridge_limit", "sui"])
99                            .inc();
100                        self.metrics
101                            .bridge_events_total
102                            .with_label_values(&["update_route_limit", "sui"])
103                            .inc();
104
105                        (UpdateBridgeLimit, serde_json::to_value(event)?)
106                    }
107                    t if t == &self.emergency_op_event_type => {
108                        info!(?ev, "Observed Sui Emergency Op");
109                        let event: EmergencyOpEvent = bcs::from_bytes(&ev.contents)?;
110
111                        // Critical security event - emergency bridge pause/unpause
112                        self.metrics
113                            .bridge_emergency_events_total
114                            .with_label_values(&["emergency_operation", "critical"])
115                            .inc();
116                        self.metrics
117                            .governance_actions_total
118                            .with_label_values(&["emergency_operation", "sui"])
119                            .inc();
120                        self.metrics
121                            .bridge_pause_status
122                            .with_label_values(&["bridge_main"])
123                            .set(if event.frozen { 0 } else { 1 });
124
125                        (EmergencyOperation, serde_json::to_value(event)?)
126                    }
127                    t if t == &self.blocklist_event_type => {
128                        info!(?ev, "Observed Sui Blocklist Validator");
129                        let event: MoveBlocklistValidatorEvent = bcs::from_bytes(&ev.contents)?;
130                        (UpdateCommitteeBlocklist, serde_json::to_value(event)?)
131                    }
132                    t if t == &self.token_reg_event_type => {
133                        info!(?ev, "Observed Sui Token Registration");
134                        let event: MoveTokenRegistrationEvent = bcs::from_bytes(&ev.contents)?;
135                        (AddSuiTokens, serde_json::to_value(event)?)
136                    }
137                    t if t == &self.update_price_event_type => {
138                        info!(?ev, "Observed Sui Token Price Update");
139                        let event: UpdateTokenPriceEvent = bcs::from_bytes(&ev.contents)?;
140                        (UpdateTokenPrices, serde_json::to_value(event)?)
141                    }
142                    t if t == &self.new_token_event_type => {
143                        info!(?ev, "Observed Sui New token event");
144                        let event: MoveNewTokenEvent = bcs::from_bytes(&ev.contents)?;
145                        (AddSuiTokens, serde_json::to_value(event)?)
146                    }
147                    _ => continue,
148                };
149
150                results.push(GovernanceAction {
151                    nonce: None,
152                    data_source: BridgeDataSource::SUI,
153                    txn_digest: txn_digest.to_vec(),
154                    sender_address: sender_address.to_vec(),
155                    timestamp_ms,
156                    action,
157                    data,
158                });
159            }
160        }
161        Ok(results)
162    }
163}

165#[async_trait]
166impl Handler for GovernanceActionHandler {
167    type Store = Db;
168
169    async fn commit<'a>(
170        values: &[Self::Value],
171        conn: &mut <Self::Store as Store>::Connection<'a>,
172    ) -> anyhow::Result<usize> {
173        Ok(diesel::insert_into(schema::governance_actions::table)
174            .values(values)
175            .on_conflict_do_nothing()
176            .execute(conn)
177            .await?)
178    }
