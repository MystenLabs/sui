26pub struct TokenTransferHandler {
27    deposited_event_type: StructTag,
28    approved_event_type: StructTag,
29    claimed_event_type: StructTag,
30    metrics: Arc<BridgeIndexerMetrics>,
31}
32
33impl TokenTransferHandler {
34    pub fn new(metrics: Arc<BridgeIndexerMetrics>) -> Self {
35        Self {
36            deposited_event_type: struct_tag!(BRIDGE_ADDRESS, BRIDGE, TOKEN_DEPOSITED_EVENT),
37            approved_event_type: struct_tag!(BRIDGE_ADDRESS, BRIDGE, TOKEN_TRANSFER_APPROVED),
38            claimed_event_type: struct_tag!(BRIDGE_ADDRESS, BRIDGE, TOKEN_TRANSFER_CLAIMED),
39            metrics,
40        }
41    }
42}
43
44impl Default for TokenTransferHandler {
45    fn default() -> Self {
46        // For compatibility with existing code that doesn't pass metrics
47        use prometheus::Registry;
48        let registry = Registry::new();
49        let metrics = BridgeIndexerMetrics::new(&registry);
50        Self::new(metrics)
51    }
52}

54#[async_trait]
55impl Processor for TokenTransferHandler {
56    const NAME: &'static str = "token_transfer";
57    type Value = TokenTransfer;
58
59    async fn process(
60        &self,
61        checkpoint: &Arc<CheckpointData>,
62    ) -> Result<Vec<Self::Value>, anyhow::Error> {
63        let timestamp_ms = checkpoint.checkpoint_summary.timestamp_ms as i64;
64        let block_height = checkpoint.checkpoint_summary.sequence_number as i64;
65
66        let mut results = vec![];
67
68        for tx in &checkpoint.transactions {
69            if !is_bridge_txn(tx) {
70                continue;
71            }
72            for ev in tx.events.iter().flat_map(|e| &e.data) {
73                let (chain_id, nonce) = if self.deposited_event_type == ev.type_ {
74                    info!("Observed Sui Deposit {:?}", ev);
75                    let event: MoveTokenDepositedEvent = bcs::from_bytes(&ev.contents)?;
76
77                    // Bridge-specific metrics for token deposits
78                    self.metrics
79                        .bridge_events_total
80                        .with_label_values(&["token_deposited", "sui"])
81                        .inc();
82                    self.metrics
83                        .token_transfers_total
84                        .with_label_values(&[
85                            "sui_to_eth",
86                            "deposited",
87                            &event.token_type.to_string(),
88                        ])
89                        .inc();
90                    self.metrics
91                        .token_transfer_gas_used
92                        .with_label_values(&["sui_to_eth", "true"])
93                        .inc_by(tx.effects.gas_cost_summary().net_gas_usage() as u64);
94
95                    (event.source_chain, event.seq_num)
96                } else if self.approved_event_type == ev.type_ {
97                    info!("Observed Sui Approval {:?}", ev);
98                    let event: MoveTokenTransferApproved = bcs::from_bytes(&ev.contents)?;
99
100                    // Bridge committee approval metrics
101                    self.metrics
102                        .bridge_events_total
103                        .with_label_values(&["transfer_approved", "sui"])
104                        .inc();
105                    self.metrics
106                        .token_transfers_total
107                        .with_label_values(&["eth_to_sui", "approved", "unknown"])
108                        .inc();
109
110                    (
111                        event.message_key.source_chain,
112                        event.message_key.bridge_seq_num,
113                    )
114                } else if self.claimed_event_type == ev.type_ {
115                    info!("Observed Sui Claim {:?}", ev);
116                    let event: MoveTokenTransferClaimed = bcs::from_bytes(&ev.contents)?;
117
118                    // Bridge transfer completion metrics
119                    self.metrics
120                        .bridge_events_total
121                        .with_label_values(&["transfer_claimed", "sui"])
122                        .inc();
123                    self.metrics
124                        .token_transfers_total
125                        .with_label_values(&["eth_to_sui", "claimed", "unknown"])
126                        .inc();
127
128                    (
129                        event.message_key.source_chain,
130                        event.message_key.bridge_seq_num,
131                    )
132                } else {
133                    return Ok(results);
134                };
135
136                results.push(TokenTransfer {
137                    chain_id: chain_id as i32,
138                    nonce: nonce as i64,
139                    block_height,
140                    timestamp_ms,
141                    status: TokenTransferStatus::Deposited,
142                    data_source: BridgeDataSource::SUI,
143                    is_finalized: true,
144                    txn_hash: tx.transaction.digest().inner().to_vec(),
145                    txn_sender: tx.transaction.sender_address().to_vec(),
146                    gas_usage: tx.effects.gas_cost_summary().net_gas_usage(),
147                });
148            }
149        }
150        Ok(results)
151    }
152}

154#[async_trait]
155impl Handler for TokenTransferHandler {
156    type Store = Db;
157    async fn commit<'a>(
158        values: &[Self::Value],
159        conn: &mut <Self::Store as Store>::Connection<'a>,
160    ) -> anyhow::Result<usize> {
161        Ok(diesel::insert_into(token_transfer::table)
162            .values(values)
163            .on_conflict_do_nothing()
164            .execute(conn)
165            .await?)
166    }
