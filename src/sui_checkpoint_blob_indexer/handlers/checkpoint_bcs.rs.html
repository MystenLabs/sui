4use std::sync::Arc;
5
6use bytes::Bytes;
7use object_store::path::Path as ObjectPath;
8use sui_indexer_alt_framework::pipeline::Processor;
9use sui_indexer_alt_framework::pipeline::concurrent::BatchStatus;
10use sui_indexer_alt_framework::pipeline::concurrent::Handler;
11use sui_indexer_alt_framework::store::Store;
12use sui_indexer_alt_object_store::ObjectStore;
13use sui_storage::blob::{Blob, BlobEncoding};
14use sui_types::full_checkpoint_content::{Checkpoint, CheckpointData};
15
16pub struct BcsCheckpoint {
17    pub sequence_number: u64,
18    pub bcs_bytes: Bytes,
19}
20
21pub struct CheckpointBcsPipeline;
22
23#[async_trait::async_trait]
24impl Processor for CheckpointBcsPipeline {
25    const NAME: &'static str = "checkpoint_bcs";
26    type Value = BcsCheckpoint;
27
28    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
29        let sequence_number = checkpoint.summary.sequence_number;
30        let checkpoint_data = CheckpointData::from(Checkpoint::clone(checkpoint));
31        let bcs_bytes = Bytes::from(Blob::encode(&checkpoint_data, BlobEncoding::Bcs)?.to_bytes());
32
33        Ok(vec![BcsCheckpoint {
34            sequence_number,
35            bcs_bytes,
36        }])
37    }
38}
39
40#[async_trait::async_trait]
41impl Handler for CheckpointBcsPipeline {
42    type Store = ObjectStore;
43    type Batch = Option<Self::Value>;
44
45    fn batch(
46        &self,
47        batch: &mut Self::Batch,
48        values: &mut std::vec::IntoIter<Self::Value>,
49    ) -> BatchStatus {
50        if batch.is_none() && values.len() > 0 {
51            *batch = values.next();
52            BatchStatus::Ready
53        } else {
54            BatchStatus::Pending
55        }
56    }
57
58    async fn commit<'a>(
59        &self,
60        batch: &Self::Batch,
61        conn: &mut <Self::Store as Store>::Connection<'a>,
62    ) -> anyhow::Result<usize> {
63        let Some(blob) = batch else {
64            return Ok(0);
65        };
66
67        let path = format!("{}.chk", blob.sequence_number);
68        conn.object_store()
69            .put(&ObjectPath::from(path), blob.bcs_bytes.clone().into())
70            .await?;
71        Ok(1)
72    }
73}
