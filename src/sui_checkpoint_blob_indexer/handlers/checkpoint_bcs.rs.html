6use bytes::Bytes;

18
19pub struct BcsCheckpoint {
20    pub sequence_number: u64,
21    pub bcs_bytes: Bytes,
22}
23
24pub struct CheckpointBcsPipeline;
25
26#[async_trait::async_trait]
27impl Processor for CheckpointBcsPipeline {
28    const NAME: &'static str = "checkpoint_bcs";
29    type Value = BcsCheckpoint;
30
31    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
32        let sequence_number = checkpoint.summary.sequence_number;
33        let checkpoint_data = CheckpointData::from(Checkpoint::clone(checkpoint));
34        let bcs_bytes = Bytes::from(Blob::encode(&checkpoint_data, BlobEncoding::Bcs)?.to_bytes());
35
36        Ok(vec![BcsCheckpoint {
37            sequence_number,
38            bcs_bytes,
39        }])
40    }
41}
42
43#[async_trait::async_trait]
44impl Handler for CheckpointBcsPipeline {
45    type Store = ObjectStore;
46    type Batch = Option<Self::Value>;
47
48    fn batch(
49        &self,
50        batch: &mut Self::Batch,
51        values: &mut std::vec::IntoIter<Self::Value>,
52    ) -> BatchStatus {
53        if batch.is_none() && values.len() > 0 {
54            *batch = values.next();
55            BatchStatus::Ready
56        } else {
57            BatchStatus::Pending
58        }
59    }
60
61    async fn commit<'a>(
62        &self,
63        batch: &Self::Batch,
64        conn: &mut <Self::Store as Store>::Connection<'a>,
65    ) -> anyhow::Result<usize> {
66        let Some(blob) = batch else {
67            return Ok(0);
68        };
69
70        let path = format!("{}.chk", blob.sequence_number);
71        conn.object_store()
72            .put(&ObjectPath::from(path), blob.bcs_bytes.clone().into())
73            .await?;
74        Ok(1)
75    }
76}
