24pub struct CheckpointBlob {
25    pub sequence_number: u64,
26    pub proto_bytes: Bytes,
27}
28
29pub struct CheckpointBlobPipeline {
30    pub compression_level: Option<i32>,
31}
32
33#[async_trait::async_trait]
34impl Processor for CheckpointBlobPipeline {
35    const NAME: &'static str = "checkpoint_blob";
36    type Value = CheckpointBlob;
37
38    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
39        static MASK: LazyLock<sui_rpc::field::FieldMaskTree> = LazyLock::new(|| {
40            FieldMask::from_paths([
41                rpc::v2::Checkpoint::path_builder().sequence_number(),
42                rpc::v2::Checkpoint::path_builder().summary().bcs().value(),
43                rpc::v2::Checkpoint::path_builder().signature().finish(),
44                rpc::v2::Checkpoint::path_builder().contents().bcs().value(),
45                rpc::v2::Checkpoint::path_builder()
46                    .transactions()
47                    .transaction()
48                    .bcs()
49                    .value(),
50                rpc::v2::Checkpoint::path_builder()
51                    .transactions()
52                    .effects()
53                    .bcs()
54                    .value(),
55                rpc::v2::Checkpoint::path_builder()
56                    .transactions()
57                    .effects()
58                    .unchanged_loaded_runtime_objects()
59                    .finish(),
60                rpc::v2::Checkpoint::path_builder()
61                    .transactions()
62                    .events()
63                    .bcs()
64                    .value(),
65                rpc::v2::Checkpoint::path_builder()
66                    .objects()
67                    .objects()
68                    .bcs()
69                    .value(),
70            ])
71            .into()
72        });
73
74        let sequence_number = checkpoint.summary.sequence_number;
75        let proto_checkpoint = rpc::v2::Checkpoint::merge_from(checkpoint.as_ref(), &MASK);
76        let proto_bytes = Bytes::from(proto_checkpoint.encode_to_vec());
77
78        Ok(vec![CheckpointBlob {
79            sequence_number,
80            proto_bytes,
81        }])
82    }
83}
84
85#[async_trait::async_trait]
86impl Handler for CheckpointBlobPipeline {
87    type Store = ObjectStore;
88    type Batch = Option<Self::Value>;
89
90    fn batch(
91        &self,
92        batch: &mut Self::Batch,
93        values: &mut std::vec::IntoIter<Self::Value>,
94    ) -> BatchStatus {
95        if batch.is_none() && values.len() > 0 {
96            *batch = values.next();
97            BatchStatus::Ready
98        } else {
99            BatchStatus::Pending
100        }
101    }
102
103    async fn commit<'a>(
104        &self,
105        batch: &Self::Batch,
106        conn: &mut <Self::Store as Store>::Connection<'a>,
107    ) -> anyhow::Result<usize> {
108        let Some(blob) = batch else {
109            return Ok(0);
110        };
111
112        let mut path = format!("{}.binpb", blob.sequence_number);
113        let data: Bytes = if let Some(level) = self.compression_level {
114            path = format!("{}.zst", path);
115            tokio::task::spawn_blocking({
116                let bytes = blob.proto_bytes.clone();
117                move || {
118                    let compressed = BytesMut::new();
119                    let mut writer = compressed.writer();
120                    let mut encoder = zstd::Encoder::new(&mut writer, level)?;
121                    std::io::copy(&mut &bytes[..], &mut encoder)?;
122                    encoder.finish()?;
123                    Ok::<Bytes, std::io::Error>(writer.into_inner().freeze())
124                }
125            })
126            .await??
127        } else {
128            blob.proto_bytes.clone()
129        };
130
131        conn.object_store()
132            .put(&ObjectPath::from(path), data.into())
133            .await?;
134        Ok(1)
135    }
136}
