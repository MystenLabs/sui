4use move_vm_config::verifier::MeterConfig;
5use serde::{Deserialize, Serialize};
6
7// Default values for verifier signing config.
8pub const DEFAULT_MAX_PER_FUN_METER_UNITS: usize = 2_200_000;
9pub const DEFAULT_MAX_PER_MOD_METER_UNITS: usize = 2_200_000;
10pub const DEFAULT_MAX_PER_PKG_METER_UNITS: usize = 2_200_000;
11
12pub const DEFAULT_MAX_BACK_EDGES_PER_FUNCTION: usize = 10_000;
13pub const DEFAULT_MAX_BACK_EDGES_PER_MODULE: usize = 10_000;
14
15/// This holds limits that are only set and used by the verifier during signing _only_. There are
16/// additional limits in the `MeterConfig` and `VerifierConfig` that are used during both signing
17/// and execution, however those limits cannot be set here and must be protocol versioned.
18#[derive(Clone, Debug, Default, Deserialize, Serialize)]
19#[serde(rename_all = "kebab-case")]
20pub struct VerifierSigningConfig {
21    #[serde(default)]
22    max_per_fun_meter_units: Option<usize>,
23    #[serde(default)]
24    max_per_mod_meter_units: Option<usize>,
25    #[serde(default)]
26    max_per_pkg_meter_units: Option<usize>,
27
28    #[serde(default)]
29    max_back_edges_per_function: Option<usize>,
30    #[serde(default)]
31    max_back_edges_per_module: Option<usize>,
32}
33
34impl VerifierSigningConfig {
35    pub fn max_per_fun_meter_units(&self) -> usize {
36        self.max_per_fun_meter_units
37            .unwrap_or(DEFAULT_MAX_PER_FUN_METER_UNITS)
38    }
39
40    pub fn max_per_mod_meter_units(&self) -> usize {
41        self.max_per_mod_meter_units
42            .unwrap_or(DEFAULT_MAX_PER_MOD_METER_UNITS)
43    }
44
45    pub fn max_per_pkg_meter_units(&self) -> usize {
46        self.max_per_pkg_meter_units
47            .unwrap_or(DEFAULT_MAX_PER_PKG_METER_UNITS)
48    }
49
50    pub fn max_back_edges_per_function(&self) -> usize {
51        self.max_back_edges_per_function
52            .unwrap_or(DEFAULT_MAX_BACK_EDGES_PER_FUNCTION)
53    }
54
55    pub fn max_back_edges_per_module(&self) -> usize {
56        self.max_back_edges_per_module
57            .unwrap_or(DEFAULT_MAX_BACK_EDGES_PER_MODULE)
58    }
59
60    /// Return sign-time only limit for back edges for the verifier.
61    pub fn limits_for_signing(&self) -> (usize, usize) {
62        (
63            self.max_back_edges_per_function(),
64            self.max_back_edges_per_module(),
65        )
66    }
67
68    /// MeterConfig for metering packages during signing. It is NOT stable between binaries and
69    /// cannot used during execution.
70    pub fn meter_config_for_signing(&self) -> MeterConfig {
71        MeterConfig {
72            max_per_fun_meter_units: Some(self.max_per_fun_meter_units() as u128),
73            max_per_mod_meter_units: Some(self.max_per_mod_meter_units() as u128),
74            max_per_pkg_meter_units: Some(self.max_per_pkg_meter_units() as u128),
75        }
76    }
77}
