4//! Internal thread-local scratch buffer used by encoding-side call
5//! sites.
6//!
7//! Most operations on a typed column-family handle encode at least one
8//! key, and writes encode a value alongside. RocksDB's read and write
9//! APIs copy or consume the supplied byte slices synchronously, so the
10//! scratch buffer can be reused across operations on the same thread.
11
12use std::cell::RefCell;
13
14/// Maximum capacity retained on the thread-local buffer between calls.
15/// One-off encodes of values larger than this shrink back to this
16/// bound on Drop so they do not pin a large allocation indefinitely.
17const MAX_RETAIN_BYTES: usize = 1 << 20;
18
19thread_local! {
20    static ENCODE_BUF: RefCell<Vec<u8>> = const { RefCell::new(Vec::new()) };
21}
22
23/// Run `f` with access to the thread-local scratch buffer.
24///
25/// The buffer is cleared on entry and shrunk back to
26/// [`MAX_RETAIN_BYTES`] on exit if it grew past that bound. Multiple
27/// values can be encoded into the same buffer in sequence: record the
28/// length after each `encode_into` call to obtain non-overlapping
29/// subslice offsets, then take immutable subslices once all encodes
30/// finish.
31pub(crate) fn with_encode_buf<R>(f: impl FnOnce(&mut Vec<u8>) -> R) -> R {
32    ENCODE_BUF.with_borrow_mut(|buf| {
33        buf.clear();
34        let result = f(buf);
35        // Clear before shrinking: `Vec::shrink_to` cannot reduce
36        // capacity below the current length.
37        buf.clear();
38        if buf.capacity() > MAX_RETAIN_BYTES {
39            buf.shrink_to(MAX_RETAIN_BYTES);
40        }
41        result
42    })
43}

45#[cfg(test)]
46mod tests {
47    use super::*;
48
49    #[test]
50    fn buffer_is_cleared_on_entry() {
51        with_encode_buf(|buf| {
52            buf.extend_from_slice(&[1, 2, 3]);
53        });
54        with_encode_buf(|buf| {
55            assert!(buf.is_empty());
56        });
57    }
58
59    #[test]
60    fn capacity_is_shrunk_after_oversized_encode() {
61        with_encode_buf(|buf| {
62            buf.resize(MAX_RETAIN_BYTES * 2, 0);
63        });
64        with_encode_buf(|buf| {
65            assert!(buf.capacity() <= MAX_RETAIN_BYTES);
66        });
67    }
68
69    #[test]
70    fn small_capacities_are_retained_for_reuse() {
71        with_encode_buf(|buf| {
72            buf.reserve(1024);
73            buf.extend_from_slice(&[7; 16]);
74        });
75        with_encode_buf(|buf| {
76            assert!(buf.capacity() >= 1024);
77            assert!(buf.is_empty());
78        });
79    }
80
81    #[test]
82    fn returns_value_from_closure() {
83        let result = with_encode_buf(|buf| {
84            buf.extend_from_slice(b"hello");
85            buf.len()
86        });
87        assert_eq!(result, 5);
88    }
89}
