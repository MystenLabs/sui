4//! The auto-registered bookkeeping schema, [`FrameworkSchema`],
5//! plus the on-disk types it persists.
6//!
7//! Three column families are owned by the framework rather than by
8//! any consumer's schema, so the crate registers them automatically
9//! on every [`Db::open`](crate::Db::open) and exposes them as a
10//! cheaply-constructible [`FrameworkSchema`] handle:
11//!
12//! - `__restore` (`PipelineTaskKey → RestoreState`) — per-pipeline
13//!   restore progress; consumed by external restore drivers to
14//!   resume from the last-committed per-shard cursor.
15//! - `__watermark` (`PipelineTaskKey → Watermark`) — per-pipeline
16//!   committer watermark; used by tip-mode drivers to learn what
17//!   checkpoint each pipeline resumes from.
18//! - `__chain_id` (`PipelineTaskKey → ChainId`) — per-pipeline
19//!   chain identifier; used by tip-mode drivers to refuse
20//!   checkpoints from a different chain than the pipeline was
21//!   originally bound to.
22//!
23//! The double-underscore prefix marks each CF as crate-internal so
24//! user schemas avoid colliding on the name.
25//!
26//! # Access
27//!
28//! Hold a [`Db`] (or [`Snapshot`](crate::Snapshot)) and call
29//! [`framework`](Db::framework) /
30//! [`framework`](crate::Snapshot::framework) to obtain a
31//! `FrameworkSchema<&Db>` / `FrameworkSchema<&Snapshot>`. Both
32//! return values are zero-`Arc`-bump — three [`DbMap`]s borrowing
33//! the same reader — and scoped to the borrow that produced them.
34//!
35//! For an owned handle (e.g. to hold inside a longer-lived
36//! [`Store`](crate::Store)-like struct, or to use with [`Batch`](crate::Batch)'s
37//! typed writes), construct with
38//! [`FrameworkSchema::new(db.clone())`](FrameworkSchema::new).
39

72/// Typed `pipeline_task` key used by the framework's internal CFs.
73///
74/// Encoded as raw UTF-8 bytes — the framework CFs are internal, so
75/// we pick the simplest representation. Decoding produces an owned
76/// `String`.
77#[derive(Debug, Clone, PartialEq, Eq)]
78pub struct PipelineTaskKey(pub String);
79
80impl PipelineTaskKey {
81    /// Build a key from any string-ish input.
82    pub fn new(name: impl Into<String>) -> Self {
83        Self(name.into())
84    }
85}
86
87impl Encode for PipelineTaskKey {
88    fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
89        buf.put_slice(self.0.as_bytes());
90        Ok(())
91    }
92}
93
94impl Decode for PipelineTaskKey {
95    fn decode<B: Buf>(buf: &mut B) -> Result<Self, DecodeError> {
96        let mut bytes = vec![0u8; buf.remaining()];
97        buf.copy_to_slice(&mut bytes);
98        let s = String::from_utf8(bytes)
99            .map_err(|e| DecodeError::with_source("PipelineTaskKey not valid UTF-8", e))?;
100        Ok(Self(s))
101    }
102}

104/// Per-pipeline restore progress, persisted in the `__restore` CF.
105///
106/// Re-export of the generated protobuf message. Drivers transition
107/// a pipeline:
108/// 1. `None` (no entry) → `InProgress` when restore begins.
109/// 2. `InProgress` → `InProgress` with per-shard cursors advanced
110///    atomically with each chunk's data writes.
111/// 3. `InProgress` → `Complete` when every shard's stream has
112///    been fully consumed.
113///
114/// Tip indexing for a pipeline must wait until its state reaches
115/// `Complete`. Drivers check this on startup.
116pub use crate::proto::sui::db::v1alpha::RestoreState;
117pub use crate::proto::sui::db::v1alpha::restore_state;
118
119impl Encode for RestoreState {
120    fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
121        crate::protobuf::encode_into(self, buf)
122    }
123}
124
125impl Decode for RestoreState {
126    fn decode<B: Buf>(buf: &mut B) -> Result<Self, DecodeError> {
127        crate::protobuf::decode(buf)
128    }
129}

131/// Per-pipeline committer watermark, persisted in the `__watermark` CF.
132///
133/// Re-export of the generated protobuf message. Holds the highest
134/// checkpoint each pipeline has committed plus the corresponding
135/// epoch / tx / timestamp. Tip-mode drivers advance this
136/// atomically with each pipeline's data writes, and read it on
137/// restart to decide where to resume.
138pub use crate::proto::sui::db::v1alpha::Watermark;
139
140impl Watermark {
141    /// Build a watermark for `checkpoint`, leaving every other
142    /// field at zero. Useful in tests and for callers that only
143    /// care about the checkpoint axis (snapshot eviction order,
144    /// `Db::at_snapshot` keying).
145    pub fn for_checkpoint(checkpoint: u64) -> Self {
146        Self {
147            checkpoint_hi_inclusive: checkpoint,
148            ..Self::default()
149        }
150    }
151}
152
153impl Encode for Watermark {
154    fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
155        crate::protobuf::encode_into(self, buf)
156    }
157}
158
159impl Decode for Watermark {
160    fn decode<B: Buf>(buf: &mut B) -> Result<Self, DecodeError> {
161        crate::protobuf::decode(buf)
162    }
163}

165/// Typed chain-identifier value (`[u8; 32]`) persisted in the
166/// `__chain_id` CF.
167///
168/// Tip-mode drivers store the chain id the pipeline was first
169/// bound to so the framework can refuse checkpoints from a
170/// different chain.
171#[derive(Debug, Clone, Copy, PartialEq, Eq)]
172pub struct ChainId(pub [u8; 32]);
173
174impl Encode for ChainId {
175    fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
176        buf.put_slice(&self.0);
177        Ok(())
178    }
179}
180
181impl Decode for ChainId {
182    fn decode<B: Buf>(buf: &mut B) -> Result<Self, DecodeError> {
183        if buf.remaining() != 32 {
184            return Err(DecodeError::msg("ChainId wire size mismatch"));
185        }
186        let mut id = [0u8; 32];
187        buf.copy_to_slice(&mut id);
188        Ok(Self(id))
189    }
190}

192/// Typed handles into the framework's auto-registered column
193/// families.
194///
195/// The three CFs (`__restore`, `__watermark`, `__chain_id`)
196/// are registered automatically by
197/// [`Db::open`](crate::Db::open), so this schema does not need to
198/// be declared by user schemas. Obtain a borrowed handle via
199/// [`Db::framework`] or [`Snapshot::framework`](crate::Snapshot::framework);
200/// construct an owned one via [`FrameworkSchema::new`].
201///
202/// `R` defaults to [`Db`] for symmetry with [`DbMap`].
203pub struct FrameworkSchema<R: Reader + Clone = Db> {
204    /// Per-pipeline [`RestoreState`] entries. See the `__restore`
205    /// CF.
206    pub restore: DbMap<PipelineTaskKey, RestoreState, R>,
207    /// Per-pipeline [`Watermark`] entries. See the `__watermark`
208    /// CF.
209    pub watermarks: DbMap<PipelineTaskKey, Watermark, R>,
210    /// Per-pipeline [`ChainId`] entries. See the `__chain_id` CF.
211    pub chain_ids: DbMap<PipelineTaskKey, ChainId, R>,
212}
213
214impl<R: Reader + Clone> FrameworkSchema<R> {
215    /// Construct a `FrameworkSchema` bound to `reader`.
216    ///
217    /// The constructor is infallible because the three framework
218    /// CFs are auto-registered by [`Db::open`](crate::Db::open), so
219    /// the CF-existence check that [`DbMap::new`] performs is
220    /// redundant here. Each field clones `reader` once.
221    pub fn new(reader: R) -> Self {
222        Self {
223            restore: DbMap::new_unchecked(reader.clone(), RESTORE_CF),
224            watermarks: DbMap::new_unchecked(reader.clone(), WATERMARK_CF),
225            chain_ids: DbMap::new_unchecked(reader, CHAIN_ID_CF),
226        }
227    }
228}
229
230impl<R: Reader + Clone> std::fmt::Debug for FrameworkSchema<R> {
231    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
232        f.debug_struct("FrameworkSchema").finish_non_exhaustive()
233    }
234}

385    #[test]
386    fn snapshot_framework_reads_pre_snapshot_state() {
387        // The borrowed-snapshot accessor returns a FrameworkSchema
388        // whose reads see the captured snapshot state.
389        let (_dir, db) = open();
390        let key = PipelineTaskKey::new("p");
391        let w = Watermark {
392            checkpoint_hi_inclusive: 10,
393            ..Watermark::default()

395        let fw = FrameworkSchema::new(db.clone());
396        let mut batch = db.batch();
397        batch.put(&fw.watermarks, &key, &w).unwrap();
398        batch.commit().unwrap();
399
400        db.take_snapshot(Watermark::for_checkpoint(1));
401
402        let w2 = Watermark {
403            checkpoint_hi_inclusive: 999,
404            ..Watermark::default()
405        };
406        let mut batch = db.batch();
407        batch.put(&fw.watermarks, &key, &w2).unwrap();
408        batch.commit().unwrap();
409
410        let snap = db.at_snapshot(1).unwrap();
411        let fw_snap = snap.framework();
412        assert_eq!(fw_snap.watermarks.get(&key).unwrap(), Some(w));
413    }
