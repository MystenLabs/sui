4//! Typed iteration over a column family.
5//!
6//! Two iterator types are exposed: [`Iter`] for forward iteration in
7//! lexicographic key order, and [`RevIter`] for reverse iteration.
8//! Both implement [`Iterator`] yielding
9//! `Result<(K, V), Error>` so that decode failures are reported as
10//! per-item errors rather than poisoning the entire scan; the
11//! iterator stops yielding after the first error.
12//!
13//! Construct an iterator via [`DbMap::iter`](crate::DbMap::iter),
14//! [`iter_rev`](crate::DbMap::iter_rev),
15//! [`iter_prefix`](crate::DbMap::iter_prefix), or
16//! [`iter_rev_prefix`](crate::DbMap::iter_rev_prefix).
17//!
18//! # Range and prefix iteration
19//!
20//! `iter` and `iter_rev` take a [`RangeBounds<K>`](std::ops::RangeBounds)
21//! and produce iterators bounded by it. Use `..` for an unbounded
22//! scan, `start..end` (or any combination of `Included` / `Excluded`
23//! bounds) for a typed range.
24//!
25//! `iter_prefix` and `iter_rev_prefix` accept a separate prefix type
26//! whose [`Encode`] form is treated as a byte prefix of the full
27//! key's encoded form. The crate does *not* statically verify that
28//! the prefix encoding really is a prefix of the key encoding; that
29//! is a property of the schema's chosen encoding scheme. In practice,
30//! schemas that encode compound keys with a prefix-preserving
31//! representation (for example, big-endian fixed-int tuples) and
32//! pass tuple prefixes get the right behavior.
33//!
34//! Internally, both range and prefix iteration set RocksDB's
35//! `iterate_lower_bound` and `iterate_upper_bound` to the
36//! corresponding byte sequences, so the underlying scan stops cleanly
37//! at the bound without per-item filtering on the Rust side. The
38//! same byte-level helper backs all four entry points.
39//!
40//! # Cursor methods
41//!
42//! [`Iter`] and [`RevIter`] expose lower-level cursor methods —
43//! [`seek`](Iter::seek), [`skip_past`](Iter::skip_past),
44//! [`raw_key`](Iter::raw_key), [`raw_value`](Iter::raw_value), and
45//! [`valid`](Iter::valid) — for advanced traversal patterns (for
46//! example, byte-cursor pagination that does not want to round-trip
47//! through `K`'s decoder).
48

60/// Forward iterator over a column family in lexicographic key order.
61///
62/// Constructed by [`DbMap::iter`](crate::DbMap::iter) or
63/// [`DbMap::iter_prefix`](crate::DbMap::iter_prefix).
64pub struct Iter<'d, K, V> {
65    inner: Option<DBRawIterator<'d>>,
66    _phantom: PhantomData<fn() -> (K, V)>,
67}

69/// Reverse iterator over a column family in reverse lexicographic
70/// key order.
71///
72/// Constructed by [`DbMap::iter_rev`](crate::DbMap::iter_rev) or
73/// [`DbMap::iter_rev_prefix`](crate::DbMap::iter_rev_prefix).
74pub struct RevIter<'d, K, V> {
75    inner: Option<DBRawIterator<'d>>,
76    _phantom: PhantomData<fn() -> (K, V)>,

79impl<'d, K, V> Iter<'d, K, V> {
80    pub(crate) fn new(inner: DBRawIterator<'d>) -> Self {
81        Self {
82            inner: Some(inner),
83            _phantom: PhantomData,
84        }
85    }
86
87    /// An iterator that yields nothing. Used by callers that
88    /// determine, before issuing any RocksDB call, that the requested
89    /// range is provably empty (for example, an `Excluded(MAX)` lower
90    /// bound).
91    pub(crate) fn empty() -> Self {
92        Self {
93            inner: None,
94            _phantom: PhantomData,
95        }
96    }
97
98    /// Move the cursor to the first key greater than or equal to
99    /// `probe`.
100    ///
101    /// `probe` is interpreted as raw bytes; it does not have to be
102    /// produced from `K`. Useful for byte-cursor pagination.
103    pub fn seek(&mut self, probe: impl AsRef<[u8]>) {
104        if let Some(inner) = &mut self.inner {
105            inner.seek(probe);
106        }
107    }
108
109    /// Skip past every key whose bytes start with `prefix`. The
110    /// cursor lands on the first key that does not start with
111    /// `prefix`, or becomes invalid if no such key exists.
112    pub fn skip_past(&mut self, prefix: impl AsRef<[u8]>) {
113        if let Some(inner) = &mut self.inner {
114            match next_prefix(prefix.as_ref()) {
115                Some(end) => inner.seek(&end),
116                None => {
117                    // No successor: position past the end so the
118                    // iterator becomes invalid.
119                    inner.seek_to_last();
120                    inner.next();
121                }
122            }
123        }
124    }
125
126    /// Returns the raw bytes of the next key the iterator will yield,
127    /// or `None` if the cursor is invalid.
128    pub fn raw_key(&self) -> Option<&[u8]> {
129        self.inner.as_ref().and_then(|i| i.key())
130    }
131
132    /// Returns the raw bytes of the next value the iterator will
133    /// yield, or `None` if the cursor is invalid.
134    pub fn raw_value(&self) -> Option<&[u8]> {
135        self.inner.as_ref().and_then(|i| i.value())
136    }
137
138    /// Returns whether the cursor currently points at a key.
139    pub fn valid(&self) -> bool {
140        self.inner.as_ref().is_some_and(|i| i.valid())
141    }
142}

144impl<'d, K, V> RevIter<'d, K, V> {
145    pub(crate) fn new(inner: DBRawIterator<'d>) -> Self {
146        Self {
147            inner: Some(inner),
148            _phantom: PhantomData,
149        }
150    }
151
152    /// An iterator that yields nothing. See [`Iter::empty`].
153    pub(crate) fn empty() -> Self {
154        Self {
155            inner: None,
156            _phantom: PhantomData,
157        }
158    }
159
160    /// Move the cursor to the last key less than or equal to
161    /// `probe`.
162    ///
163    /// `probe` is interpreted as raw bytes; it does not have to be
164    /// produced from `K`.
165    pub fn seek(&mut self, probe: impl AsRef<[u8]>) {
166        if let Some(inner) = &mut self.inner {
167            inner.seek_for_prev(probe);
168        }
169    }
170
171    /// Skip past every key whose bytes start with `prefix` (in
172    /// reverse). The cursor lands on the last key strictly before
173    /// the prefix range, or becomes invalid if no such key exists.
174    pub fn skip_past(&mut self, prefix: impl AsRef<[u8]>) {
175        if let Some(inner) = &mut self.inner {
176            let end = prefix.as_ref();
177            inner.seek_for_prev(end);
178            // The only way to land on a prefix-matching key here is
179            // to land on `end` itself (the prefix as an exact key).
180            // Step back once if so.
181            if inner.key() == Some(end) {
182                inner.prev();
183            }
184        }
185    }
186
187    /// Returns the raw bytes of the next key the iterator will yield,
188    /// or `None` if the cursor is invalid.
189    pub fn raw_key(&self) -> Option<&[u8]> {
190        self.inner.as_ref().and_then(|i| i.key())
191    }
192
193    /// Returns the raw bytes of the next value the iterator will
194    /// yield, or `None` if the cursor is invalid.
195    pub fn raw_value(&self) -> Option<&[u8]> {
196        self.inner.as_ref().and_then(|i| i.value())
197    }
198
199    /// Returns whether the cursor currently points at a key.
200    pub fn valid(&self) -> bool {
201        self.inner.as_ref().is_some_and(|i| i.valid())
202    }
203}

229/// Compute the lexicographic successor of `prefix`, treating it as a
230/// variable-length byte string.
231///
232/// Returns `Some(succ)` such that `succ` is the smallest byte string
233/// strictly greater than `prefix` in lexicographic order. Returns
234/// `None` if `prefix` is composed entirely of `0xFF` bytes (and so
235/// has no successor in this ordering), in which case the prefix
236/// extends to the end of the key space.
237pub(crate) fn next_prefix(prefix: &[u8]) -> Option<Vec<u8>> {
238    let mut next = prefix.to_vec();
239    while let Some(last) = next.last_mut() {
240        if *last < 0xFF {
241            *last += 1;
242            return Some(next);
243        }
244        next.pop();
245    }
246    None
247}

249/// Encoded byte bounds for an iterator.
250///
251/// `Empty` means the requested range is provably empty (no key can
252/// match) — produced today by an `Excluded(k)` lower bound whose
253/// encoded `k` is non-empty all-`0xFF` (no successor exists). Callers
254/// short-circuit to an empty iterator on this variant.
255///
256/// `Range(lower, upper)` is the usual half-open
257/// `[lower, upper)` byte interval; either side may be `None` to leave
258/// that side unbounded.
259pub(crate) enum ByteBounds {
260    Empty,
261    Range(Option<Vec<u8>>, Option<Vec<u8>>),
262}
263
264/// Convert a typed [`RangeBounds<J>`] into encoded byte bounds
265/// suitable for RocksDB's `iterate_lower_bound` and
266/// `iterate_upper_bound` (lower inclusive, upper exclusive).
267///
268/// `Excluded(start)` for the lower bound is implemented by encoding
269/// `start` and taking its lex successor. If `start` encodes to a
270/// non-empty all-`0xFF` byte string the successor does not exist in
271/// the same byte length; the result is reported as
272/// [`ByteBounds::Empty`] so iteration yields nothing. (For empty
273/// encodings, where `next_prefix` also returns `None`, the bound is
274/// silently dropped — `Excluded([])` is degenerate input.)
275///
276/// `Included(end)` for the upper bound takes the lex successor as
277/// the exclusive upper. If `end` encodes to all-`0xFF` (no
278/// successor), the upper bound is dropped — for fixed-length
279/// encodings this is exact; variable-length encodings may iterate
280/// extra rows whose encoded keys extend the all-`0xFF` prefix.
281pub(crate) fn range_to_byte_bounds<J, R>(range: &R) -> Result<ByteBounds, EncodeError>
282where
283    J: Encode,
284    R: RangeBounds<J>,
285{
286    let lower = match range.start_bound() {
287        Bound::Included(k) => Some(k.encode()?),
288        Bound::Excluded(k) => {
289            let encoded = k.encode()?;
290            match next_prefix(&encoded) {
291                Some(succ) => Some(succ),
292                // Empty encoding has no successor but the bound is
293                // degenerate; drop it rather than declaring the
294                // whole range empty.
295                None if encoded.is_empty() => None,
296                // Non-empty all-0xFF: provably no key satisfies the
297                // bound (assuming fixed-length encoding); short-
298                // circuit to an empty iteration.
299                None => return Ok(ByteBounds::Empty),
300            }
301        }
302        Bound::Unbounded => None,
303    };
304    let upper = match range.end_bound() {
305        Bound::Excluded(k) => Some(k.encode()?),
306        Bound::Included(k) => next_prefix(&k.encode()?),
307        Bound::Unbounded => None,
308    };
309    Ok(ByteBounds::Range(lower, upper))
310}

312/// Convert a typed prefix into encoded byte bounds.
313///
314/// Lower bound is the encoded prefix; upper bound is the prefix's
315/// lex successor (or `None` if the prefix is empty or all `0xFF`).
316pub(crate) fn prefix_to_byte_bounds<P>(prefix: &P) -> Result<ByteBounds, EncodeError>
317where
318    P: Encode,
319{
320    let lower_bytes = prefix.encode()?;
321    let upper = next_prefix(&lower_bytes);
322    Ok(ByteBounds::Range(Some(lower_bytes), upper))
323}

325/// Shared `next` body for both directions. `forward = true` advances
326/// via `DBRawIterator::next` after yielding; `forward = false`
327/// advances via `prev`.
328fn next_step<K, V>(
329    slot: &mut Option<DBRawIterator<'_>>,
330    forward: bool,
331) -> Option<Result<(K, V), Error>>
332where
333    K: Decode,
334    V: Decode,
335{
336    let inner = slot.as_mut()?;
337    if !inner.valid() {
338        // Either the iterator is exhausted or it errored. `status`
339        // distinguishes the two.
340        let err = inner.status().err();
341        *slot = None;
342        return err.map(|e| Err(Error::Rocksdb(e)));
343    }
344
345    // The iterator reports valid, so key/value must be present;
346    // surface a defensive error rather than silently terminating if
347    // RocksDB ever violates this invariant.
348    let (Some(key_bytes), Some(value_bytes)) = (inner.key(), inner.value()) else {
349        *slot = None;
350        return Some(Err(Error::Internal(
351            "raw iterator returned None for key or value while marked valid",
352        )));
353    };
354
355    let item = match (
356        K::decode(&mut &key_bytes[..]),
357        V::decode(&mut &value_bytes[..]),
358    ) {
359        (Ok(k), Ok(v)) => Ok((k, v)),
360        (Err(e), _) | (_, Err(e)) => {
361            *slot = None;
362            return Some(Err(Error::Decode(e)));
363        }
364    };
365
366    if forward {
367        inner.next();
368    } else {
369        inner.prev();
370    }
371    Some(item)
372}

417    fn range(b: ByteBounds) -> (Option<Vec<u8>>, Option<Vec<u8>>) {
418        match b {
419            ByteBounds::Range(lo, hi) => (lo, hi),
420            ByteBounds::Empty => panic!("expected ByteBounds::Range, got Empty"),
421        }
