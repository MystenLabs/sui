4//! Typed column-family handles.
5//!
6//! [`DbMap<K, V, R>`] is the primary read and write surface in the
7//! crate. Each instance is tied to a single column family on a
8//! [`Db`] handle (carried by the reader), to a key type and a value
9//! type that implement the crate's encoding traits, and to a
10//! [`Reader`] that pins the consistency context.
11//!
12//! # The reader parameter
13//!
14//! `R` defaults to [`Db`], so today's call sites
15//! (`DbMap::new(db, "items")`, schemas opened via
16//! [`Db::open`](crate::Db::open)) work unchanged. To re-bind a map
17//! at a captured snapshot, call [`DbMap::at`]; for the whole-schema
18//! equivalent see
19//! [`SchemaAtSnapshot::at`](crate::SchemaAtSnapshot::at). To
20//! construct a borrowed-reader handle (no `Arc` bump), pass `&db`
21//! or `&snap` to [`DbMap::new`] — the reader type is inferred from
22//! the argument.
23//!
24//! Schemas are typically structs of `DbMap` fields parameterized by
25//! `R`; see the [`Schema`](crate::Schema) trait for the construction
26//! pattern.
27//!
28//! # Reading
29//!
30//! Two read methods are exposed for point lookups:
31//!
32//! - [`DbMap::get`] decodes the value into an owned `V` and is the
33//!   common path. Internally it routes through RocksDB's
34//!   `get_pinned_cf` to avoid the redundant copy that `DB::get`
35//!   performs into a fresh `Vec<u8>`.
36//! - [`DbMap::get_raw`] returns the raw value bytes as a
37//!   [`bytes::Bytes`]. The `Bytes` is backed zero-copy by the
38//!   RocksDB block cache where possible (cache hits) and by an
39//!   internal copy where not (memtable hits, merge results,
40//!   wide-column values). The handle co-owns the [`Db`] handle (via
41//!   the reader) so it is sound to hold the `Bytes` past the borrow
42//!   it was read through.
43//!
44//! Both methods have batched counterparts ([`DbMap::multi_get`] and
45//! [`DbMap::multi_get_raw`]) that issue a single RocksDB
46//! `batched_multi_get_cf` call rather than N independent reads.
47//!
48//! # A note on pinned reads
49//!
50//! Each outstanding `Bytes` clone of a block-cache-backed value pins
51//! an LRU handle in RocksDB's block cache. Long-lived or unbounded
52//! pins can drive `block_cache.pinned-usage` past
53//! `block_cache.capacity()`. Hold the `Bytes` for as short a scope as
54//! the application allows, especially on high-fanout query paths.

78/// A typed handle to a single column family on a [`Db`], bound to a
79/// [`Reader`] that pins its consistency context.
80///
81/// Construct a live-tip handle with [`DbMap::new`], typically inside a
82/// schema's [`Schema::open`](crate::Schema::open) implementation. Re-bind
83/// at a snapshot via [`DbMap::at`].
84///
85/// # Examples
86///
87/// ```
88/// use bytes::Buf;
89/// use bytes::BufMut;
90///
91/// use sui_consistent_store::Db;
92/// use sui_consistent_store::DbMap;
93/// use sui_consistent_store::DbOptions;
94/// use sui_consistent_store::Decode;
95/// use sui_consistent_store::Encode;
96/// use sui_consistent_store::Reader;
97/// use sui_consistent_store::Schema;
98/// use sui_consistent_store::error::DecodeError;
99/// use sui_consistent_store::error::EncodeError;
100/// use sui_consistent_store::error::OpenError;
101///
102/// #[derive(Debug, PartialEq, Eq)]
103/// struct U64Be(u64);
104///
105/// impl Encode for U64Be {
106///     fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
107///         buf.put_slice(&self.0.to_be_bytes());
108///         Ok(())
109///     }
110/// }
111///
112/// impl Decode for U64Be {
113///     fn decode<B: Buf>(buf: &mut B) -> Result<Self, DecodeError> {
114///         if buf.remaining() != 8 {
115///             return Err(DecodeError::msg("expected 8 bytes"));
116///         }
117///         Ok(Self(buf.get_u64()))
118///     }
119/// }
120///
121/// struct MySchema<R: Reader = Db> {
122///     items: DbMap<U64Be, U64Be, R>,
123/// }
124///
125/// impl Schema for MySchema {
126///     fn cfs(opts: &sui_consistent_store::CfOptionsResolver) -> Vec<sui_consistent_store::CfDescriptor> {
127///         vec![sui_consistent_store::CfDescriptor::new("items", opts.options("items"))]
128///     }
129///
130///     fn open(db: &Db) -> Result<Self, OpenError> {
131///         Ok(Self {
132///             items: DbMap::new(db.clone(), "items")?,
133///         })
134///     }
135/// }
136///
137/// let dir = tempfile::tempdir().unwrap();
138/// let (_db, schema) = Db::open::<MySchema>(dir.path(), DbOptions::default()).unwrap();
139/// // Nothing was inserted, so a lookup returns None.
140/// assert!(schema.items.get(&U64Be(1)).unwrap().is_none());
141/// ```
142#[derive(Debug)]
143pub struct DbMap<K, V, R: Reader = Db> {
144    reader: R,
145    cf_name: &'static str,
146    _data: PhantomData<fn(K) -> V>,
147}

149/// Owner struct used to tie a `DBPinnableSlice` to a [`Db`] handle
150/// so the slice can be wrapped in `bytes::Bytes::from_owner`. The
151/// `'static` lifetime on the slice is justified by the [`Db`]
152/// co-owner; see `pinned_to_bytes` for the safety argument.
153///
154/// Field declaration order is load-bearing: `slice` must drop before
155/// `_db` so the cleanup function the pin runs on `Drop` (typically a
156/// block-cache `Cache::Release`) executes while the underlying DB
157/// allocation is still alive.
158struct PinnedOwner {
159    slice: DBPinnableSlice<'static>,
160    _db: Db,
161}

163impl<K, V, R: Reader> DbMap<K, V, R> {
164    /// Like [`DbMap::new`], but skips the existence check on
165    /// `cf_name`. Crate-internal; used to construct typed handles
166    /// against column families known to be registered (auto-created
167    /// by [`Db::open`]), avoiding a redundant
168    /// [`cf_handle`](Db::cf_handle) lookup per construction.
169    pub(crate) fn new_unchecked(reader: R, cf_name: &'static str) -> Self {
170        Self {
171            reader,
172            cf_name,
173            _data: PhantomData,
174        }
175    }
176
177    /// Construct a typed handle for the column family named `cf_name`
178    /// on `reader`'s database.
179    ///
180    /// `reader` controls the consistency context. Common cases:
181    ///
182    /// - `DbMap::new(db, "items")` — owned [`Db`] (live-tip,
183    ///   one `Arc` bump).
184    /// - `DbMap::new(&db, "items")` — borrowed [`Db`] (live-tip,
185    ///   no `Arc` bump; the returned handle borrows `db`).
186    /// - `DbMap::new(snap, "items")` — owned [`Snapshot`]
187    ///   (snapshot-bound). Usually obtained via
188    ///   [`DbMap::at`] from an existing handle so the cross-`Db`
189    ///   safety check fires.
190    /// - `DbMap::new(&snap, "items")` — borrowed
191    ///   [`Snapshot`] (snapshot-bound, no `Arc` bump).
192    ///
193    /// Returns an [`OpenError`] if the named column family is not
194    /// registered on `reader.db()`. Construction is the only place
195    /// where a missing column family is reported as an open-time
196    /// error; subsequent operations report it as
197    /// [`crate::error::Error::MissingColumnFamily`].
198    pub fn new(reader: R, cf_name: &'static str) -> Result<Self, OpenError> {
199        if reader.db().cf_handle(cf_name).is_none() {
200            return Err(OpenError::msg(format!(
201                "column family `{cf_name}` is not registered",
202            )));
203        }
204        Ok(Self {
205            reader,
206            cf_name,
207            _data: PhantomData,
208        })
209    }
210
211    /// Re-bind this handle at a captured snapshot.
212    ///
213    /// Returns a new [`DbMap`] whose reader is
214    /// [`Snapshot`], so subsequent reads see the
215    /// database state at the snapshot's checkpoint regardless of
216    /// writes that occurred after
217    /// [`Db::take_snapshot`](crate::Db::take_snapshot) was called.
218    /// The returned handle owns a clone of `snap` (two `Arc`
219    /// bumps), so it is self-contained and can outlive the
220    /// originating `Snapshot`. The column-family name is a
221    /// [`&'static str`](prim@str), so re-binding does not allocate.
222    ///
223    /// # Panics
224    ///
225    /// Panics if `snap` was taken on a different [`Db`] than this
226    /// handle is bound to. Cross-`Db` re-binding is a programmer
227    /// error: the snapshot's column family is unrelated to this
228    /// handle's, and silently reading from the wrong CF (or hitting
229    /// a `MissingColumnFamily` error one read later) would mask the
230    /// underlying bug.
231    pub fn at(&self, snap: &Snapshot) -> DbMap<K, V, Snapshot> {
232        assert!(
233            self.reader.db().ptr_eq(snap.db()),
234            "snapshot was taken on a different Db than this DbMap is bound to",
235        );
236        DbMap {
237            reader: snap.clone(),
238            cf_name: self.cf_name,
239            _data: PhantomData,
240        }
241    }
242
243    /// Borrowed counterpart to [`DbMap::at`]: re-bind this handle
244    /// at a borrowed [`Snapshot`] rather than cloning it.
245    ///
246    /// Returns a new [`DbMap`] whose reader is `&'a Snapshot`,
247    /// tied to the lifetime of `snap`. The snapshot's two `Arc`s
248    /// are not bumped, and the column-family name is a
249    /// [`&'static str`](prim@str) (no allocation), so the re-bind
250    /// is essentially free.
251    ///
252    /// # Panics
253    ///
254    /// Panics if `snap` was taken on a different [`Db`] than this
255    /// handle is bound to; see [`DbMap::at`] for the rationale.
256    pub fn at_ref<'a>(&self, snap: &'a Snapshot) -> DbMap<K, V, &'a Snapshot> {
257        assert!(
258            self.reader.db().ptr_eq(snap.db()),
259            "snapshot was taken on a different Db than this DbMap is bound to",
260        );
261        DbMap {
262            reader: snap,
263            cf_name: self.cf_name,
264            _data: PhantomData,
265        }
266    }
267
268    fn cf(&self) -> Result<Arc<rocksdb::BoundColumnFamily<'_>>, Error> {
269        self.reader
270            .db()
271            .cf_handle(self.cf_name)
272            .ok_or_else(|| Error::MissingColumnFamily(self.cf_name.to_string()))
273    }
274
275    /// The shared [`Db`] handle this `DbMap` is bound to. Used by
276    /// `Batch` to look up the same column family the handle points
277    /// at.
278    pub(crate) fn db(&self) -> &Db {
279        self.reader.db()
280    }
281
282    /// The name of the column family this handle is bound to.
283    pub(crate) fn cf_name(&self) -> &'static str {
284        self.cf_name
285    }
286}

288impl<K, V, R: Reader> DbMap<K, V, R>
289where
290    K: Encode,
291    V: Decode,
292{
293    /// Read and decode the value for `key`.
294    ///
295    /// Returns `Ok(None)` if the key is not present; otherwise
296    /// returns the decoded value. Internally uses RocksDB's pinned
297    /// read path so that block-cache hits avoid the extra heap
298    /// allocation that `DB::get` would do.
299    pub fn get(&self, key: &K) -> Result<Option<V>, Error> {
300        let opts = self.reader.read_options();
301        let cf = self.cf()?;
302        with_encode_buf(|buf| {
303            key.encode_into(buf)?;
304            let pinned =
305                self.reader
306                    .db()
307                    .rocksdb()
308                    .get_pinned_cf_opt(&cf, buf.as_slice(), &opts)?;
309            match pinned {
310                Some(slice) => Ok(Some(V::decode(&mut &slice[..])?)),
311                None => Ok(None),
312            }
313        })
314    }
315
316    /// Batched counterpart to [`get`](Self::get).
317    ///
318    /// Issues one RocksDB `batched_multi_get_cf` for all the keys
319    /// rather than `keys.len()` separate reads. The outer `Result`
320    /// captures failures in the encoding step (which abort the whole
321    /// batch); each inner `Result` captures per-key read or decode
322    /// failures and is reported alongside successful results in the
323    /// returned vector.
324    pub fn multi_get<'k, I>(&self, keys: I) -> Result<Vec<Result<Option<V>, Error>>, Error>
325    where
326        I: IntoIterator<Item = &'k K>,
327        K: 'k,
328    {
329        Ok(self
330            .multi_get_pinned(keys)?
331            .into_iter()
332            .map(|r| match r {
333                Ok(Some(slice)) => V::decode(&mut &slice[..]).map(Some).map_err(Error::Decode),
334                Ok(None) => Ok(None),
335                Err(e) => Err(Error::Rocksdb(e)),
336            })
337            .collect())
338    }
339}

341impl<K, V, R: Reader> DbMap<K, V, R>
342where
343    K: Encode,
344{
345    /// Read the raw bytes for `key`.
346    ///
347    /// Returns `Ok(None)` if the key is not present. The returned
348    /// [`Bytes`] is backed zero-copy by the RocksDB block cache when
349    /// the read hits a cached block; otherwise it backs onto a small
350    /// internal copy made by RocksDB's pinned-read machinery. Either
351    /// way, the `Bytes` co-owns the underlying [`Db`] handle so it
352    /// can outlive any borrow this method was called through.
353    pub fn get_raw(&self, key: &K) -> Result<Option<Bytes>, Error> {
354        let opts = self.reader.read_options();
355        let cf = self.cf()?;
356        with_encode_buf(|buf| {
357            key.encode_into(buf)?;
358            let pinned =
359                self.reader
360                    .db()
361                    .rocksdb()
362                    .get_pinned_cf_opt(&cf, buf.as_slice(), &opts)?;
363            Ok(pinned.map(|slice| pinned_to_bytes(self.reader.db().clone(), slice)))
364        })
365    }
366
367    /// Batched counterpart to [`get_raw`](Self::get_raw).
368    ///
369    /// Issues one RocksDB `batched_multi_get_cf` for all the keys.
370    /// The outer `Result` captures encoding failures; the inner
371    /// `Result` captures per-key read failures.
372    pub fn multi_get_raw<'k, I>(&self, keys: I) -> Result<Vec<Result<Option<Bytes>, Error>>, Error>
373    where
374        I: IntoIterator<Item = &'k K>,
375        K: 'k,
376    {
377        Ok(self
378            .multi_get_pinned(keys)?
379            .into_iter()
380            .map(|r| match r {
381                Ok(Some(slice)) => Ok(Some(pinned_to_bytes(self.reader.db().clone(), slice))),
382                Ok(None) => Ok(None),
383                Err(e) => Err(Error::Rocksdb(e)),
384            })
385            .collect())
386    }
387
388    /// Internal helper: encode `keys` and issue a single RocksDB
389    /// `batched_multi_get_cf`, returning the raw pinned-slice
390    /// results. Both [`multi_get`](Self::multi_get) and
391    /// [`multi_get_raw`](Self::multi_get_raw) funnel through here
392    /// and just differ in how they map each result.
393    fn multi_get_pinned<'k, 's, I>(
394        &'s self,
395        keys: I,
396    ) -> Result<Vec<Result<Option<DBPinnableSlice<'s>>, rocksdb::Error>>, Error>
397    where
398        I: IntoIterator<Item = &'k K>,
399        K: 'k,
400    {
401        let keys: Vec<&K> = keys.into_iter().collect();
402        let opts = self.reader.read_options();
403        let cf = self.cf()?;
404
405        with_encode_buf(|buf| {
406            let mut offsets = Vec::with_capacity(keys.len() + 1);
407            offsets.push(0usize);
408            for key in &keys {
409                key.encode_into(buf)?;
410                offsets.push(buf.len());
411            }
412
413            let bytes = buf.as_slice();
414            let key_slices: Vec<&[u8]> = offsets
415                .windows(2)
416                .map(|window| &bytes[window[0]..window[1]])
417                .collect();
418
419            Ok(self
420                .reader
421                .db()
422                .rocksdb()
423                .batched_multi_get_cf_opt(&cf, key_slices, false, &opts))
424        })
425    }
426
427    /// Test whether `key` is present in the column family.
428    ///
429    /// Returns `Ok(true)` if a value exists at `key`, `Ok(false)`
430    /// otherwise. The implementation calls RocksDB's
431    /// `key_may_exist_cf` first; when that returns `false` (a
432    /// definitive negative answer), the call returns `Ok(false)`
433    /// without a pinned read. Otherwise it confirms with a pinned
434    /// `get_pinned_cf`.
435    ///
436    /// `key_may_exist_cf` is only useful when the column family has
437    /// a bloom filter configured on its
438    /// [`rocksdb::BlockBasedOptions`]; otherwise it always returns
439    /// `true` and `contains_key` does the same work as `get_raw`
440    /// plus an extra check. Schemas on hot
441    /// `contains_key` paths should set
442    /// [`set_bloom_filter`](rocksdb::BlockBasedOptions::set_bloom_filter)
443    /// on the CF's options.
444    pub fn contains_key(&self, key: &K) -> Result<bool, Error> {
445        let opts = self.reader.read_options();
446        let cf = self.cf()?;
447        with_encode_buf(|buf| -> Result<bool, Error> {
448            key.encode_into(buf)?;
449            // Bloom-filter fast path. A `false` here is definitive;
450            // a `true` is "may exist" and needs confirming.
451            if !self
452                .reader
453                .db()
454                .rocksdb()
455                .key_may_exist_cf_opt(&cf, buf.as_slice(), &opts)
456            {
457                return Ok(false);
458            }
459            let pinned =
460                self.reader
461                    .db()
462                    .rocksdb()
463                    .get_pinned_cf_opt(&cf, buf.as_slice(), &opts)?;
464            Ok(pinned.is_some())
465        })
466    }
467
468    /// Batched counterpart to [`contains_key`](Self::contains_key).
469    ///
470    /// Issues one RocksDB `batched_multi_get_cf` for all the keys.
471    /// Per-key encoding errors abort the entire batch; per-key read
472    /// failures surface as the outer `Err`.
473    pub fn multi_contains_keys<'k, I>(&self, keys: I) -> Result<Vec<bool>, Error>
474    where
475        I: IntoIterator<Item = &'k K>,
476        K: 'k,
477    {
478        let raw = self.multi_get_raw(keys)?;
479        raw.into_iter()
480            .map(|r| r.map(|opt| opt.is_some()))
481            .collect()
482    }
483}

485impl<K, V, R: Reader> DbMap<K, V, R>
486where
487    K: Encode + Decode,
488    V: Decode,
489{
490    /// Iterate forward over the subset of entries whose keys fall
491    /// within `range`, in lexicographic key order.
492    ///
493    /// Pass `..` for an unbounded scan, `start..end` for a half-open
494    /// range, `start..=end` for inclusive on both ends, or any other
495    /// [`RangeBounds<K>`] combination. To bound by a different type
496    /// (for example, a prefix-typed bound on a compound key), use
497    /// [`iter_with`](Self::iter_with). Decode failures are reported
498    /// as a per-item `Err`; the iterator stops yielding after the
499    /// first error.
500    pub fn iter(&self, range: impl RangeBounds<K>) -> Result<Iter<'_, K, V>, Error> {
501        self.iter_forward(range_to_byte_bounds(&range)?)
502    }
503
504    /// Iterate forward over the subset of entries whose keys, when
505    /// encoded, begin with `prefix`'s encoding.
506    ///
507    /// The prefix may be a different type than the full key, so long
508    /// as the schema's encoding choice makes the prefix's encoded
509    /// bytes a real byte prefix of every full-key encoding it should
510    /// match. The crate documents this contract but does not verify
511    /// it; see the [module-level docs](crate::iter) for an
512    /// explanation and a worked example.
513    pub fn iter_prefix(&self, prefix: &impl Encode) -> Result<Iter<'_, K, V>, Error> {
514        self.iter_forward(prefix_to_byte_bounds(prefix)?)
515    }
516
517    /// Iterate forward over the subset of entries whose keys, when
518    /// encoded, begin with `prefix`'s encoding and sort at or after
519    /// `from`'s encoding.
520    ///
521    /// Like [`iter_prefix`](Self::iter_prefix) but with the lower bound
522    /// raised to `from` (inclusive) -- typically a pagination cursor
523    /// whose encoding extends `prefix`. The upper bound stays the
524    /// prefix's lex successor, so the scan still stops cleanly at the
525    /// end of the prefix range without per-item filtering. The same
526    /// prefix-encoding contract as [`iter_prefix`](Self::iter_prefix)
527    /// applies: the schema's encoding must make `from`'s bytes
528    /// well-ordered against the full keys.
529    ///
530    /// `from` must itself lie within the prefix: its encoded bytes
531    /// must start with `prefix`'s encoding. This is enforced --
532    /// [`Error::CursorOutsidePrefix`] otherwise -- because cursors
533    /// typically derive from client-supplied page tokens, and a
534    /// cursor sorting before the prefix would otherwise widen the
535    /// scan below the prefix's start and leak foreign rows into the
536    /// result. (A cursor past the prefix's last *row* is fine -- it
537    /// still starts with the prefix and simply yields nothing.)
538    pub fn iter_prefix_from<P: Encode, F: Encode>(
539        &self,
540        prefix: &P,
541        from: &F,
542    ) -> Result<Iter<'_, K, V>, Error> {
543        let ByteBounds::Range(lower, upper) = prefix_to_byte_bounds(prefix)? else {
544            return Ok(Iter::empty());
545        };
546        let from_bytes = from.encode()?;
547        if !lower.is_some_and(|p| from_bytes.starts_with(&p)) {
548            return Err(Error::CursorOutsidePrefix);
549        }
550        self.iter_forward(ByteBounds::Range(Some(from_bytes), upper))
551    }
552
553    /// Iterate in reverse over the subset of entries whose keys fall
554    /// within `range`.
555    pub fn iter_rev(&self, range: impl RangeBounds<K>) -> Result<RevIter<'_, K, V>, Error> {
556        self.iter_reverse(range_to_byte_bounds(&range)?)
557    }
558
559    /// Iterate in reverse over the subset of entries whose keys,
560    /// when encoded, begin with `prefix`'s encoding.
561    pub fn iter_rev_prefix(&self, prefix: &impl Encode) -> Result<RevIter<'_, K, V>, Error> {
562        self.iter_reverse(prefix_to_byte_bounds(prefix)?)
563    }
564
565    /// Iterate forward over entries whose keys, when encoded, fall
566    /// within the encoded bytes of `range`'s `J`-typed bounds.
567    ///
568    /// Like [`iter`](Self::iter), but parameterized over an
569    /// independent `J: Encode`. Useful when the schema's key type
570    /// is a compound (e.g. `(owner, type, id)`) and the caller
571    /// wants to bound the iteration by a prefix-typed value (e.g. an
572    /// `owner`-typed range). The schema's encoding must make `J`'s
573    /// encoded bytes well-ordered against `K`'s encoded keys; the
574    /// crate documents this contract but does not verify it.
575    ///
576    /// Use turbofish to disambiguate `J` when passing an unbounded
577    /// range: `map.iter_with::<MyPrefix>(..)`.
578    pub fn iter_with<J: Encode>(
579        &self,
580        range: impl RangeBounds<J>,
581    ) -> Result<Iter<'_, K, V>, Error> {
582        self.iter_forward(range_to_byte_bounds(&range)?)
583    }
584
585    /// Reverse counterpart to [`iter_with`](Self::iter_with).
586    pub fn iter_rev_with<J: Encode>(
587        &self,
588        range: impl RangeBounds<J>,
589    ) -> Result<RevIter<'_, K, V>, Error> {
590        self.iter_reverse(range_to_byte_bounds(&range)?)
591    }
592
593    /// Internal helper: forward iteration with caller-supplied byte
594    /// bounds. Both range and prefix entry points funnel through
595    /// here. A [`ByteBounds::Empty`] short-circuits to an empty
596    /// iterator without touching RocksDB.
597    ///
598    /// `seek_to_first` honors `iterate_lower_bound` in modern
599    /// RocksDB, landing at the smallest key at or above the bound
600    /// — equivalent to `seek(lower)`. Calling `seek` explicitly
601    /// would require cloning the bound (RocksDB's
602    /// `set_iterate_lower_bound` consumes the `Vec<u8>`), so we
603    /// use the bound-aware `seek_to_first`.
604    fn iter_forward<'s>(&'s self, bounds: ByteBounds) -> Result<Iter<'s, K, V>, Error> {
605        let ByteBounds::Range(lower, upper) = bounds else {
606            return Ok(Iter::empty());
607        };
608        let mut opts = self.reader.read_options();
609        if let Some(l) = lower {
610            opts.set_iterate_lower_bound(l);
611        }
612        if let Some(u) = upper {
613            opts.set_iterate_upper_bound(u);
614        }
615        let cf = self.cf()?;
616        let mut raw = self.reader.db().rocksdb().raw_iterator_cf_opt(&cf, opts);
617        raw.seek_to_first();
618        Ok(Iter::new(raw))
619    }
620
621    /// Internal helper: reverse iteration with caller-supplied byte
622    /// bounds. The dual of [`iter_forward`](Self::iter_forward).
623    ///
624    /// Symmetric to forward: `seek_to_last` honors
625    /// `iterate_upper_bound` in modern RocksDB, landing at the
626    /// largest key strictly less than the bound. Calling
627    /// `seek_for_prev` explicitly would require cloning the bound,
628    /// so we use the bound-aware `seek_to_last`.
629    fn iter_reverse<'s>(&'s self, bounds: ByteBounds) -> Result<RevIter<'s, K, V>, Error> {
630        let ByteBounds::Range(lower, upper) = bounds else {
631            return Ok(RevIter::empty());
632        };
633        let mut opts = self.reader.read_options();
634        if let Some(l) = lower {
635            opts.set_iterate_lower_bound(l);
636        }
637        if let Some(u) = upper {
638            opts.set_iterate_upper_bound(u);
639        }
640        let cf = self.cf()?;
641        let mut raw = self.reader.db().rocksdb().raw_iterator_cf_opt(&cf, opts);
642        raw.seek_to_last();
643        Ok(RevIter::new(raw))
644    }
645}

653/// Wrap a `DBPinnableSlice` plus its co-owned [`Db`] handle in a
654/// `bytes::Bytes` so callers do not have to reason about RocksDB
655/// lifetimes themselves.
656///
657/// The `'a` lifetime on `DBPinnableSlice<'a>` is a `PhantomData<&'a
658/// DB>` annotation only. The actual backing memory is either a
659/// reference-counted block in the RocksDB block cache (kept alive
660/// until the slice's `Drop` runs the registered cleanup) or a copied
661/// buffer owned by the C++ `PinnableSlice` itself (freed when the
662/// slice drops). Neither path requires a live `&DB` borrow; both
663/// require only that the underlying DB allocation outlive the slice,
664/// which the co-owned [`Db`] handle guarantees. Drop order in
665/// `PinnedOwner` (slice first, `Db` second) ensures the cleanup
666/// runs before the [`Db`] handle's last reference goes away.
667fn pinned_to_bytes(db: Db, slice: DBPinnableSlice<'_>) -> Bytes {
668    // SAFETY: see the function-level doc comment. The lifetime is a
669    // conservative phantom annotation; the actual memory pin is
670    // sustained by the [`Db`] co-owner held in `PinnedOwner`.
671    let slice: DBPinnableSlice<'static> = unsafe { mem::transmute(slice) };
672    Bytes::from_owner(PinnedOwner { slice, _db: db })
673}
