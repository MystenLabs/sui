4//! A canonical [`prometheus::core::Collector`] that exposes the
5//! RocksDB integer properties surfaced by
6//! [`Db::cf_metrics`](crate::Db::cf_metrics) as labeled
7//! `IntGauge`s, one set of gauges per registered column family.
8//!
9//! # Registering
10//!
11//! Register on a [`prometheus::Registry`] the same way you would
12//! any other collector:
13//!
14//! ```
15//! use prometheus::Registry;
16//! use sui_consistent_store::Db;
17//! use sui_consistent_store::DbOptions;
18//! use sui_consistent_store::Schema;
19//! use sui_consistent_store::metrics::ColumnFamilyStatsCollector;
20//!
21//! #[derive(Debug)]
22//! struct EmptySchema;
23//!
24//! impl Schema for EmptySchema {
25//!     fn cfs(_: &sui_consistent_store::CfOptionsResolver) -> Vec<sui_consistent_store::CfDescriptor> {
26//!         vec![]
27//!     }
28//!     fn open(_: &Db) -> Result<Self, sui_consistent_store::error::OpenError> {
29//!         Ok(Self)
30//!     }
31//! }
32//!
33//! let dir = tempfile::tempdir().unwrap();
34//! let (db, _schema) = Db::open::<EmptySchema>(dir.path(), DbOptions::default()).unwrap();
35//! let registry = Registry::new();
36//! let collector = ColumnFamilyStatsCollector::new(None, &db);
37//! registry.register(Box::new(collector)).unwrap();
38//! ```
39//!
40//! # Liveness
41//!
42//! The collector holds a [`DbRef`] (weak handle), so
43//! registering it does not keep the underlying database open. If
44//! every strong [`Db`] handle has been dropped by the
45//! time Prometheus scrapes, every gauge is reset to `-1` so
46//! dashboards see the database is gone rather than stale-looking
47//! values.
48//!
49//! # Sampling cadence
50//!
51//! Sampling happens on Prometheus scrape: every
52//! `collect()` walks every column family in the underlying [`Db`]
53//! and reads each property. This is cheap (each property is one
54//! RocksDB API call) but not free, so a tight scrape interval over
55//! many CFs has a measurable cost.

67/// Default prefix applied to every metric name when the caller
68/// passes `None` to [`ColumnFamilyStatsCollector::new`].
69const DEFAULT_PREFIX: &str = "rocksdb";
70
71/// A [`prometheus::core::Collector`] that emits per-column-family
72/// RocksDB integer properties as labeled `IntGauge`s.
73///
74/// Each metric has a single `cf_name` label whose value is the
75/// column-family name; every CF the [`Db`] knew about at
76/// construction (both schema-declared and framework-internal) is
77/// sampled on every collect. The collector holds a
78/// [`DbRef`] (weak handle), so registering it does not keep the
79/// database alive — once every strong [`Db`] has been dropped,
80/// subsequent scrapes report every gauge as
81/// [`METRICS_ERROR`](crate::RocksMetrics).
82pub struct ColumnFamilyStatsCollector {
83    db: DbRef,
84    /// Snapshot of `db.cf_names()` taken at construction time so
85    /// the collector can emit one series per CF even after the
86    /// underlying [`Db`] has been dropped. CFs are fixed once a
87    /// database is open, so the snapshot does not go stale.
88    cf_names: Vec<&'static str>,
89    gauges: Gauges,
90}

94    /// every metric name (default: `"rocksdb"`). Holds a weak

205impl Collector for ColumnFamilyStatsCollector {
206    fn desc(&self) -> Vec<&Desc> {
207        self.gauges.descs()
208    }
209
210    fn collect(&self) -> Vec<MetricFamily> {
211        // If the database has been closed, fall back to a default
212        // (METRICS_ERROR-filled) sample for every CF so dashboards
213        // see the database is gone instead of stale values.
214        let maybe_db = self.db.upgrade();
215        for &cf_name in &self.cf_names {
216            let m = match &maybe_db {
217                Some(db) => db.cf_metrics(cf_name),
218                None => RocksMetrics::default(),
219            };
220            self.populate(cf_name, &m);
221        }
222        self.gauges.collect_all()
223    }
224}

340    fn descs(&self) -> Vec<&Desc> {
341        let mut out = vec![];
342        for g in self.all() {
343            out.extend(g.desc());
344        }
345        out
346    }

348    fn collect_all(&self) -> Vec<MetricFamily> {
349        let mut out = vec![];
350        for g in self.all() {

352        }

419    #[test]
420    fn cf_names_includes_schema_and_framework_cfs() {
421        let (_dir, db) = open_db();
422        let names: Vec<&str> = db.cf_names().to_vec();
423        assert!(names.contains(&"alpha"));
424        assert!(names.contains(&"beta"));
425        assert!(names.contains(&"__restore"));
426        assert!(names.contains(&"__watermark"));
427        assert!(names.contains(&"__chain_id"));
428        // RocksDB's mandatory `default` CF is opened but not tracked.
429        assert!(!names.contains(&"default"));
430    }

432    #[test]
433    fn collector_registers_and_collects_for_every_cf() {
434        let (_dir, db) = open_db();
435        let registry = Registry::new();
436        let collector = ColumnFamilyStatsCollector::new(None, &db);
437        registry.register(Box::new(collector)).unwrap();
438
439        // Gather metric families and confirm at least one
440        // `cf_name="alpha"` series shows up across the families.
441        let families = registry.gather();
442        let mut saw_alpha = false;
443        let mut saw_beta = false;
444        for family in &families {
445            for metric in family.get_metric() {
446                for label in metric.get_label() {
447                    if label.name() == "cf_name" {
448                        match label.value() {
449                            "alpha" => saw_alpha = true,
450                            "beta" => saw_beta = true,
451                            _ => {}
452                        }
453                    }
454                }
455            }
456        }
457        assert!(saw_alpha, "alpha cf metrics should be emitted");
458        assert!(saw_beta, "beta cf metrics should be emitted");
459    }

477    #[test]
478    fn collector_does_not_keep_db_alive() {
479        // Build a collector, drop the strong Db handle, then
480        // confirm the underlying `DbRef` can no longer upgrade —
481        // the collector's weak handle must not pin the Db.
482        let (_dir, db) = open_db();
483        let collector = ColumnFamilyStatsCollector::new(None, &db);
484        let weak = db.downgrade();
485        drop(db);
486        drop(collector);
487        assert!(
488            weak.upgrade().is_none(),
489            "Db should drop once every strong handle is released",
490        );
491    }

493    #[test]
494    fn collect_after_db_drop_emits_metrics_error_sentinel() {
495        // After the Db is dropped, every gauge must read
496        // METRICS_ERROR (-1) so dashboards see "unavailable"
497        // rather than the last live values.
498        let (_dir, db) = open_db();
499        let cf_names: Vec<&'static str> = db.cf_names().to_vec();
500        let collector = ColumnFamilyStatsCollector::new(None, &db);
501        drop(db);
502        let families = collector.collect();
503
504        let mut series_seen = 0usize;
505        for family in &families {
506            for metric in family.get_metric() {
507                let mut matched_cf = false;
508                for label in metric.get_label() {
509                    if label.name() == "cf_name" && cf_names.iter().any(|cf| *cf == label.value()) {
510                        matched_cf = true;
511                    }
512                }
513                if !matched_cf {
514                    continue;
515                }
516                series_seen += 1;
517                assert_eq!(
518                    metric.get_gauge().value() as i64,
519                    -1,
520                    "expected METRICS_ERROR sentinel, got {}",
521                    metric.get_gauge().value(),
522                );
523            }
524        }
525        assert!(
526            series_seen > 0,
527            "collector should still emit one series per registered CF",
528        );
529    }
