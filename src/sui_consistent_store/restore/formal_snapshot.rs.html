4//! Built-in [`RestoreSource`] that streams live objects from a
5//! Sui formal snapshot.
6//!
7//! Each formal snapshot is a per-epoch directory of `.obj` /
8//! `.ref` files indexed by `(bucket, partition)` pairs. This
9//! source maps each *bucket* to one driver shard so per-bucket
10//! `.obj` files run as one sequential stream and the driver
11//! iterates buckets in parallel.
12//!
13//! # Cursor encoding
14//!
15//! Big-endian `u32` of the last successfully committed partition
16//! index within the shard's bucket. `stream(shard_id, Some(c))`
17//! yields only partitions whose index is `> c`. The first chunk's
18//! cursor is the partition's own index (so after committing
19//! partition `0`, the persisted cursor is `0` and a re-run
20//! resumes at partition `1`).
21//!
22//! # Where the watermark comes from
23//!
24//! The snapshot store records files by `epoch`; the driver needs
25//! a checkpoint sequence number for
26//! [`target_checkpoint`](super::RestoreSource::target_checkpoint).
27//! On construction we fetch the `end-of-epoch` checkpoint via the
28//! supplied `remote_store_url` to anchor the snapshot at the
29//! correct sequence number.
30

103    pub azure: Option<String>,

105    /// Fetch formal snapshot from a generic HTTP endpoint.
106    #[arg(long, group = "source")]

110    /// path to the snapshot root directory.
111    #[arg(long, group = "source")]
112    pub local: Option<PathBuf>,

115    /// epoch → end-of-epoch checkpoint mapping so the snapshot
116    /// can be anchored at the right checkpoint sequence number.

137/// Built-in [`RestoreSource`] backed by a Sui formal snapshot.
138///
139/// Construct via [`FormalSnapshot::new`]; the connection to the
140/// underlying object store and the watermark anchoring round
141/// trip happen during construction so subsequent
142/// [`stream`](RestoreSource::stream) calls are pure file I/O.
143pub struct FormalSnapshot {
144    /// Underlying storage backend (S3/GCS/Azure/HTTP/local).
145    source: Arc<dyn Storage + Send + Sync + 'static>,
146
147    /// Path of the epoch's subdirectory within `source`.
148    epoch_dir: String,
149
150    /// Watermark this snapshot is anchored at (epoch, checkpoint,
151    /// tx count, timestamp). The driver writes this into
152    /// `__watermark` so tip indexing resumes from the right
153    /// place once restore finishes.
154    target_watermark: Watermark,
155
156    /// Chain identifier the snapshot belongs to (digest of the
157    /// chain's genesis checkpoint). Pinned into `__chain_id`
158    /// on finalise so tip indexing rejects checkpoints from a
159    /// different chain.
160    target_chain_id: ChainId,
161
162    /// Every `.obj` partition across all buckets, flattened and
163    /// sorted by `(bucket, partition)`. Each entry is one driver
164    /// shard, so `shard_id` indexes directly into this vector and
165    /// the shard count equals the partition count. Exposing one
166    /// shard per partition (rather than one per bucket) lets the
167    /// driver fetch partitions concurrently and makes its
168    /// `restore_shards_done` gauge track partition-level progress —
169    /// snapshots commonly ship a single bucket, which would
170    /// otherwise be one giant sequential shard.
171    partitions: Vec<FileMetadata>,
172
173    metrics: Arc<FormalSnapshotMetrics>,
174}
175
176impl FormalSnapshot {
177    /// Connect to the snapshot store described by `args` /
178    /// `connection_args`, fetch the epoch manifest, resolve the
179    /// anchor checkpoint, and group partitions by bucket.
180    pub async fn new(
181        args: FormalSnapshotArgs,
182        connection_args: StorageConnectionArgs,
183        metrics: Arc<FormalSnapshotMetrics>,
184    ) -> anyhow::Result<Self> {
185        let store = connect_storage(&args, connection_args.clone())?;
186
187        let (epoch, epoch_dir) = resolve_epoch(&args, &store).await?;
188        info!(epoch, epoch_dir, "Connected to valid formal snapshot");
189
190        let (target_watermark, target_chain_id) =
191            anchor_watermark_and_chain_id(&args.remote_store_url, epoch).await?;
192        info!(
193            epoch,
194            target_checkpoint = target_watermark.checkpoint_hi_inclusive,
195            "Anchored snapshot at end-of-epoch checkpoint",
196        );
197
198        let manifest_bytes = store
199            .get(format!("{epoch_dir}/MANIFEST").into())
200            .await
201            .context("Failed to fetch epoch manifest")?;
202        let manifest = EpochManifest::read(&manifest_bytes)?;
203
204        // Collect every `.obj` file (one partition each), dropping
205        // `.ref` entries (those carry historical-version metadata,
206        // not live objects). Sort by `(bucket, partition)` so the
207        // shard ordering is deterministic across runs, which keeps
208        // resume cursors stable.
209        let mut partitions: Vec<FileMetadata> = manifest
210            .metadata()
211            .iter()
212            .filter(|m| matches!(m.file_type, FileType::Object))
213            .cloned()
214            .collect();
215        partitions.sort_by_key(|m| (m.bucket, m.partition));
216
217        metrics.total_partitions.set(partitions.len() as i64);
218        info!(
219            shards = partitions.len(),
220            partitions = partitions.len(),
221            "Discovered formal-snapshot partitions",
222        );
223
224        Ok(Self {
225            source: store,
226            epoch_dir,
227            target_watermark,
228            target_chain_id,
229            partitions,
230            metrics,
231        })
232    }
233
234    /// Encode a partition index as a 4-byte big-endian cursor. Each
235    /// shard is a single partition, so the cursor is only ever
236    /// present (the partition's chunk committed) or absent (not yet);
237    /// the encoded value is recorded for debugging the `__restore`
238    /// state rather than for resuming mid-partition.
239    fn encode_cursor(partition: u32) -> Bytes {
240        Bytes::copy_from_slice(&partition.to_be_bytes())
241    }
242}

244#[async_trait]
245impl RestoreSource for FormalSnapshot {
246    fn target_checkpoint(&self) -> u64 {
247        self.target_watermark.checkpoint_hi_inclusive
248    }
249
250    fn target_watermark(&self) -> Watermark {
251        self.target_watermark
252    }
253
254    fn target_chain_id(&self) -> ChainId {
255        self.target_chain_id
256    }
257
258    fn shards(&self) -> u32 {
259        self.partitions.len() as u32
260    }
261
262    fn stream(
263        &self,
264        shard_id: u32,
265        cursor: Option<Bytes>,
266    ) -> BoxStream<'_, anyhow::Result<RestoreChunk>> {
267        let idx = shard_id as usize;
268        if idx >= self.partitions.len() {
269            let shard_id_for_msg = shard_id;
270            let shards = self.partitions.len();
271            return stream::once(async move {
272                bail!(
273                    "formal-snapshot shard_id {shard_id_for_msg} out of range \
274                     (have {shards} shards)",
275                )
276            })
277            .boxed();
278        }
279
280        // A shard is a single partition fetched and committed as one
281        // chunk. A non-empty cursor means that chunk was already
282        // committed (the driver persists the cursor atomically with
283        // the data), so there is nothing left to stream — re-yielding
284        // would double-apply additive index writes such as `balance`.
285        if cursor.is_some() {
286            return stream::empty().boxed();
287        }
288
289        let meta = self.partitions[idx].clone();
290        let metrics = self.metrics.clone();
291        let source = self.source.clone();
292        let epoch_dir = self.epoch_dir.clone();
293
294        stream::once(async move {
295            let path = format!("{}/{}_{}.obj", epoch_dir, meta.bucket, meta.partition);
296            let live = fetch_objects(source.as_ref(), &path, &meta, metrics.as_ref()).await?;
297            Ok(RestoreChunk {
298                objects: live.objects,
299                cursor: FormalSnapshot::encode_cursor(meta.partition),
300            })
301        })
302        .boxed()
303    }
304}

306/// Connect to the underlying object store described by `args`.
307fn connect_storage(
308    args: &FormalSnapshotArgs,
309    connection_args: StorageConnectionArgs,
310) -> anyhow::Result<Arc<dyn Storage + Send + Sync + 'static>> {
311    if let Some(bucket) = &args.s3 {
312        info!(bucket, "S3 storage");
313        return Ok(Arc::new(
314            AmazonS3Builder::from_env()
315                .with_client_options(connection_args.into())
316                .with_imdsv1_fallback()
317                .with_bucket_name(bucket)
318                .build()?,
319        ));
320    }
321    if let Some(bucket) = &args.gcs {
322        info!(bucket, "GCS storage");
323        return Ok(Arc::new(
324            GoogleCloudStorageBuilder::from_env()
325                .with_client_options(connection_args.into())
326                .with_bucket_name(bucket)
327                .build()?,
328        ));
329    }
330    if let Some(container) = &args.azure {
331        info!(container, "Azure storage");
332        return Ok(Arc::new(
333            MicrosoftAzureBuilder::from_env()
334                .with_client_options(connection_args.into())
335                .with_container_name(container)
336                .build()?,
337        ));
338    }
339    if let Some(endpoint) = &args.http {
340        info!(endpoint = %endpoint, "HTTP storage");
341        return Ok(Arc::new(HttpStorage::new(
342            endpoint.clone(),
343            connection_args,
344        )?));
345    }
346    if let Some(path) = &args.local {
347        info!(path = %path.display(), "Local storage");
348        return Ok(Arc::new(LocalFileSystem::new_with_prefix(path)?));
349    }
350    bail!("No formal snapshot source provided");
351}

353/// Resolve `(epoch, epoch_dir)` honouring the `--path` /
354/// `--epoch` overrides. When neither is set we consult the root
355/// `MANIFEST` for the latest epoch and check its `_SUCCESS`
356/// marker.
357async fn resolve_epoch(
358    args: &FormalSnapshotArgs,
359    store: &Arc<dyn Storage + Send + Sync + 'static>,
360) -> anyhow::Result<(u64, String)> {
361    if let Some(path) = &args.path {
362        let epoch = args
363            .epoch
364            .expect("clap requires --epoch when --path is set");
365        return Ok((epoch, format!("{path}/epoch_{epoch}")));
366    }
367
368    let root_manifest = RootManifest::read(
369        store
370            .get("MANIFEST".into())
371            .await
372            .context("Failed to fetch root manifest")?
373            .as_ref(),
374    )?;
375
376    let epoch = args
377        .epoch
378        .or_else(|| root_manifest.latest())
379        .context("No epochs available in the snapshot store")?;
380
381    ensure!(
382        root_manifest.contains(epoch),
383        "Requested epoch {epoch} is not available in the snapshot store",
384    );
385
386    let epoch_dir = format!("epoch_{epoch}");
387    let is_complete = store
388        .get(format!("{epoch_dir}/_SUCCESS").into())
389        .await
390        .is_ok();
391    ensure!(is_complete, "Snapshot for epoch {epoch} is not complete");
392
393    Ok((epoch, epoch_dir))
394}

396/// Hit the remote checkpoint store at `remote_store_url` to
397/// resolve `epoch` to its end-of-epoch checkpoint and return a
398/// fully populated [`Watermark`] alongside the chain id
399/// (digest of checkpoint 0).
400async fn anchor_watermark_and_chain_id(
401    remote_store_url: &Url,
402    epoch: u64,
403) -> anyhow::Result<(Watermark, ChainId)> {
404    let client = StoreIngestionClient::new(
405        HttpBuilder::new()
406            .with_url(remote_store_url.to_string())
407            .with_client_options(ClientOptions::new().with_allow_http(true))
408            .build()
409            .map(Arc::new)
410            .context("Failed to connect to remote checkpoint store")?,
411        None,
412    );
413
414    let end_of_epoch_checkpoints: Vec<u64> = client
415        .end_of_epoch_checkpoints()
416        .await
417        .context("Failed to fetch end-of-epoch checkpoints")?;
418
419    let checkpoint = end_of_epoch_checkpoints
420        .get(epoch as usize)
421        .copied()
422        .with_context(|| format!("Cannot find end-of-epoch checkpoint for epoch {epoch}"))?;
423
424    let Checkpoint { summary, .. } = client
425        .checkpoint(checkpoint)
426        .await
427        .context("Failed to fetch end-of-epoch checkpoint")?;
428
429    ensure!(
430        summary.epoch == epoch,
431        "End-of-epoch checkpoint {checkpoint} does not belong to epoch {epoch}",
432    );
433
434    let chain_identifier = client
435        .chain_id()
436        .await
437        .context("Failed to fetch chain id from remote checkpoint store")?;
438
439    let watermark = Watermark {
440        epoch_hi_inclusive: summary.epoch,
441        checkpoint_hi_inclusive: summary.sequence_number,
442        tx_hi: summary.network_total_transactions,
443        timestamp_ms_hi_inclusive: summary.timestamp_ms,
444    };
445    Ok((watermark, ChainId(*chain_identifier.as_bytes())))
446}

448/// Fetch one `.obj` file with exponential backoff, decode it,
449/// and tick the relevant metrics. Retries forever (matching the
450/// previous `Restorer`'s behaviour) — the caller is responsible
451/// for setting upstream timeouts if it needs to bound total
452/// wait time.
453async fn fetch_objects(
454    source: &(dyn Storage + Send + Sync + 'static),
455    path: &str,
456    meta: &FileMetadata,
457    metrics: &FormalSnapshotMetrics,
458) -> anyhow::Result<LiveObjects> {
459    let _guard = metrics.objects_fetch_latency.start_timer();
460
461    let attempts = AtomicUsize::new(1);
462    let request = || async {
463        let attempt = attempts.fetch_add(1, Ordering::Relaxed);
464
465        let future = async {
466            let bytes = source
467                .get(path.to_string().into())
468                .await
469                .with_context(|| format!("Failed to fetch {path}"))?;
470            let bytes_len = bytes.len();
471            let live = LiveObjects::read(bytes.as_ref(), meta)
472                .with_context(|| format!("Failed to decode {path}"))?;
473            Ok::<_, anyhow::Error>((bytes_len, live))
474        };
475
476        match with_slow_future_monitor(future, SLOW_FETCH_THRESHOLD, || {
477            warn!(
478                attempt,
479                bucket = meta.bucket,
480                partition = meta.partition,
481                "Fetch slow",
482            );
483        })
484        .await
485        {
486            Ok((bytes, live)) => {
487                metrics.total_bytes_fetched.inc_by(bytes as u64);
488                metrics.total_partitions_fetched.inc();
489                Ok(live)
490            }
491            Err(e) => {
492                warn!(
493                    attempt,
494                    bucket = meta.bucket,
495                    partition = meta.partition,
496                    "Fetch error: {e:#}",
497                );
498                metrics.total_objects_fetch_retries.inc();
499                Err(BE::transient(e))
500            }
501        }
502    };
503
504    let backoff = ExponentialBackoff {
505        max_interval: MAX_RETRY_INTERVAL,
506        max_elapsed_time: None,
507        ..Default::default()
508    };
509
510    backoff::future::retry(backoff, request).await
511}
