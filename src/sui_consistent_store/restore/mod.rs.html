4//! Bulk-load pipeline shape and driver.
5//!
6//! `sui-consistent-store` separates bulk-load (one-shot, from a
7//! [`RestoreSource`]) from tip indexing (continuous, framework-driven).
8//! Each `Restore` pipeline is bounded `: Processor` so that one
9//! pipeline shares one identity (`Processor::NAME`) and one
10//! configuration type across both phases, but the tip impl is
11//! independent — `Restore` does not touch tip data.
12//!
13//! # Method roles
14//!
15//! - [`Restore::restore`] takes a single live object and stages
16//!   writes derived from it onto the shared per-shard [`Batch`].
17//!   The driver parallelises across pipelines and partitions but
18//!   commits each (pipeline, partition) atomically.
19//!
20//! Per-object writes that target merge-CFs combine via the
21//! registered merge operator across shards (so cross-shard sums,
22//! counters, set unions all converge correctly). Writes to put-CFs
23//! have last-write-wins semantics; sources that need a different
24//! reconciliation strategy across shards should partition their
25//! input so that conflicting writes never hit the same key.
26//!
27//! # Why `anyhow::Error`
28//!
29//! Matches the indexer-alt framework's `Processor`/`Handler`
30//! signatures, so a single pipeline can return one error type from
31//! both its restore and tip methods. Pipeline authors who only
32//! call [`Batch`] methods (which return [`crate::error::Error`])
33//! get free conversion via `?`.
34

55/// A pipeline that can be bulk-loaded from a stream of live
56/// objects.
57///
58/// Implementations are typically unit structs whose trait methods
59/// are essentially functions; `&self` is supplied for symmetry with
60/// [`Processor::process`], so per-instance
61/// configuration (a config struct loaded from CLI, a logger handle)
62/// can live on the implementing type and be shared by both the
63/// processor and restore paths.
64pub trait Restore: Processor {
65    /// The portion of the schema this pipeline writes to during
66    /// restore. Typically a struct of [`DbMap`](crate::DbMap)
67    /// fields scoped to the CFs this pipeline owns. The driver
68    /// supplies an `&Self::Schema` on every [`restore`](Self::restore)
69    /// call.
70    type Schema: Send + Sync;
71
72    /// Stage writes derived from `object` onto `batch`.
73    ///
74    /// Called from worker tasks under a restore driver. Each
75    /// worker owns its own `batch`; objects may arrive in any
76    /// order within a worker's slice. The pipeline encodes each
77    /// object's resulting rows directly via the [`Batch`] API
78    /// (typed `put`, `delete`, `merge` against
79    /// [`DbMap`](crate::DbMap) handles in `schema`).
80    ///
81    /// Cross-shard collisions are RocksDB's concern: two shards
82    /// that both touch the same key each emit their own ops, and
83    /// the registered merge operator (for merge-CFs) or
84    /// last-write semantics (for put-CFs) reconcile them.
85    fn restore(
86        &self,
87        schema: &Self::Schema,
88        object: &Object,
89        batch: &mut Batch,
90    ) -> anyhow::Result<()>;
91}

93#[cfg(test)]
94mod tests {
95    //! End-to-end exercise of [`Restore::restore`] against an
96    //! in-memory `Db`. No restore drivers here — the test calls
97    //! [`Restore::restore`] directly to verify the trait shape
98    //! compiles and integrates with the [`Batch`] write path.
99
100    use sui_types::base_types::ObjectID;
101    use sui_types::object::Object;
102
103    use super::*;
104    use crate::restore::test_pipeline::ObjectIdKey;
105    use crate::restore::test_pipeline::ObjectVersionPipeline;
106    use crate::restore::test_pipeline::U64Be;
107    use crate::restore::test_pipeline::open;
108
109    #[test]
110    fn restore_writes_each_object_directly() {
111        let (_dir, db, schema) = open();
112        let pipeline = ObjectVersionPipeline;
113
114        let o1 = Object::immutable_with_id_for_testing(ObjectID::from_single_byte(1));
115        let o2 = Object::immutable_with_id_for_testing(ObjectID::from_single_byte(2));
116
117        let mut batch = db.batch();
118        pipeline.restore(&schema, &o1, &mut batch).unwrap();
119        pipeline.restore(&schema, &o2, &mut batch).unwrap();
120        batch.commit().unwrap();
121
122        assert_eq!(
123            schema.versions.get(&ObjectIdKey::new(o1.id())).unwrap(),
124            Some(U64Be(o1.version().value())),
125        );
126        assert_eq!(
127            schema.versions.get(&ObjectIdKey::new(o2.id())).unwrap(),
128            Some(U64Be(o2.version().value())),
129        );
130    }
131
132    #[test]
133    fn empty_batch_commit_writes_nothing() {
134        let (_dir, db, schema) = open();
135        let batch = db.batch();
136        batch.commit().unwrap();
137        let rows = schema.versions.iter(..).unwrap().count();
138        assert_eq!(rows, 0);
139    }
