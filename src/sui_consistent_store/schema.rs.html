4//! The [`Schema`] trait used to register column families with the
5//! database and to construct typed handles into them, plus the
6//! [`SchemaAtSnapshot`] companion trait used to re-bind a schema at
7//! a captured snapshot.
8//!
9//! Schemas are hand-written Rust structs whose fields are typed
10//! handles into individual column families ([`DbMap<K, V, R>`](crate::DbMap)).
11//! The struct is parameterized by a [`Reader`](crate::Reader)
12//! (defaulted to [`Db`]) so the same schema body serves both the
13//! live tip and snapshot-bound projections.
14//!
15//! [`Schema`] is implemented for the live variant (`MySchema<Db>`)
16//! and pairs the static set of column families a schema requires
17//! ([`Schema::cfs`]) with the constructor that builds the schema
18//! struct from an opened database ([`Schema::open`]).
19//!
20//! [`SchemaAtSnapshot`] is a separate trait the schema author opts
21//! into; it declares a `MySchema<Snapshot>` projection and a
22//! one-line constructor that re-binds each field via
23//! [`DbMap::at`](crate::DbMap::at).
24//!
25//! # Examples
26//!
27//! ```
28//! use sui_consistent_store::CfDescriptor;
29//! use sui_consistent_store::Db;
30//! use sui_consistent_store::DbMap;
31//! use sui_consistent_store::DbOptions;
32//! use sui_consistent_store::Reader;
33//! use sui_consistent_store::Schema;
34//! use sui_consistent_store::SchemaAtSnapshot;
35//! use sui_consistent_store::Snapshot;
36//! use sui_consistent_store::error::OpenError;
37//!
38//! struct MySchema<R: Reader = Db> {
39//!     _reader: std::marker::PhantomData<R>,
40//!     _db: Db,
41//! }
42//!
43//! impl Schema for MySchema {
44//!     fn cfs(opts: &sui_consistent_store::CfOptionsResolver) -> Vec<CfDescriptor> {
45//!         vec![CfDescriptor::new("my_cf", opts.options("my_cf"))]
46//!     }
47//!
48//!     fn open(db: &Db) -> Result<Self, OpenError> {
49//!         Ok(Self {
50//!             _reader: std::marker::PhantomData,
51//!             _db: db.clone(),
52//!         })
53//!     }
54//! }
55//!
56//! impl SchemaAtSnapshot for MySchema {
57//!     type At = MySchema<Snapshot>;
58//!     fn at(&self, _snap: &Snapshot) -> Self::At {
59//!         MySchema {
60//!             _reader: std::marker::PhantomData,
61//!             _db: self._db.clone(),
62//!         }
63//!     }
64//! }
65//!
66//! let dir = tempfile::tempdir().unwrap();
67//! let (_db, _schema) = Db::open::<MySchema>(dir.path(), DbOptions::default()).unwrap();
68//! ```

75/// Declares the column families a database needs and constructs the
76/// typed handle struct against an opened database at the live tip.
77///
78/// Implementations are typically hand-written structs parameterized
79/// by a [`Reader`](crate::Reader) (defaulted to [`Db`]) whose fields
80/// are typed column-family handles. The trait itself is implemented
81/// only for the [`Db`]-bound variant of the schema; snapshot-bound
82/// variants are constructed by re-binding, not by re-opening.
83///
84/// The trait has two responsibilities:
85///
86/// - [`cfs`](Self::cfs) returns the column families this schema
87///   requires, with per-CF [`rocksdb::Options`]. It is called once at
88///   open time, with a [`CfOptionsResolver`] as input. The
89///   order of entries does not matter.
90/// - [`open`](Self::open) constructs the schema struct against an
91///   already-opened database. Each column family named by `cfs()` is
92///   guaranteed to exist on the database before this is called.
93pub trait Schema: Sized {
94    /// The column families this schema requires.
95    ///
96    /// Each entry is a [`CfDescriptor`] carrying a column-family
97    /// name (a `&'static str` so the schema's CF set is fixed at
98    /// compile time) and its [`rocksdb::Options`] applied at create
99    /// time. Implementations obtain the resolved per-CF options from
100    /// `opts` via [`CfOptionsResolver::options`] and layer only
101    /// correctness-bearing per-CF tweaks (merge operators, compaction
102    /// filters) on top — performance knobs such as compression, write
103    /// buffers, and write-stall thresholds are resolved from
104    /// configuration inside the resolver.
105    ///
106    /// RocksDB's mandatory `default` column family is opened
107    /// automatically and need not be included here, though including
108    /// it is harmless.
109    ///
110    /// The framework-internal column families (`__restore`,
111    /// `__watermark`, `__chain_id`) are reserved by the crate; a
112    /// schema that declares one is rejected by
113    /// [`Db::open`](crate::Db::open).
114    fn cfs(opts: &CfOptionsResolver) -> Vec<CfDescriptor>;
115
116    /// Construct the schema struct against `db`.
117    ///
118    /// Implementations typically clone the supplied [`Db`] handle
119    /// into each column-family handle they construct (a `Db` clone
120    /// is one `Arc` bump). The default implementation in user
121    /// schemas is usually a one-line `Self::new(db.clone())` that
122    /// delegates to inherent methods on the schema struct.
123    fn open(db: &Db) -> Result<Self, OpenError>;
124}

126/// Describes one column family in a [`Schema`].
127///
128/// Construct via [`CfDescriptor::new`].
129///
130/// # Examples
131///
132/// ```
133/// use sui_consistent_store::CfDescriptor;
134///
135/// let opts = rocksdb::Options::default();
136/// let owners = CfDescriptor::new("owners", opts);
137/// ```
138pub struct CfDescriptor {
139    /// Column-family name.
140    pub name: &'static str,
141    /// Per-CF RocksDB options applied at create time.
142    pub options: rocksdb::Options,
143}
144
145impl std::fmt::Debug for CfDescriptor {
146    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
147        // `rocksdb::Options` does not implement Debug, so summarize.
148        f.debug_struct("CfDescriptor")
149            .field("name", &self.name)
150            .finish_non_exhaustive()
151    }
152}
153
154impl CfDescriptor {
155    /// Construct a descriptor.
156    pub fn new(name: &'static str, options: rocksdb::Options) -> Self {
157        Self { name, options }
158    }
159}

161/// Re-binds a [`Schema`] at a captured [`Snapshot`].
162///
163/// The schema author declares the projection's body type as `At`
164/// and writes a one-line constructor that re-binds each field via
165/// [`DbMap::at`](crate::DbMap::at). The trait is independent of
166/// [`Schema`] so authors who never need snapshot-bound reads can
167/// skip the impl entirely.
168///
169/// # Cost
170///
171/// Each call to [`at`](Self::at) constructs a fresh schema struct
172/// containing a [`DbMap<_, _, Snapshot>`](crate::DbMap) per field.
173/// Each per-field re-bind clones the column-family name (a
174/// `Box<str>` allocation) and clones the [`Snapshot`] (two `Arc`
175/// bumps). For an N-CF schema this is N allocations and 2N `Arc`
176/// bumps per re-bind. Re-bind once per request handler and read
177/// many times against the same projection.
178pub trait SchemaAtSnapshot {
179    /// The projected schema body — typically `MySchema<Snapshot>`
180    /// when the schema is parameterized by a [`Reader`](crate::Reader).
181    ///
182    /// Because [`Snapshot`] is an owned, lifetime-free reader, the
183    /// projection is self-contained: it can be stored in a struct
184    /// or moved into a spawned task without dragging a borrow on
185    /// the originating [`Snapshot`] value.
186    type At;
187
188    /// Re-bind this schema at `snap`.
189    ///
190    /// The returned projection's reads see the database state
191    /// captured by the snapshot, regardless of writes that occur
192    /// after [`Db::take_snapshot`](crate::Db::take_snapshot) was
193    /// called. The projection owns clones of `snap` (one per field)
194    /// and is independent of `self` after construction.
195    fn at(&self, snap: &Snapshot) -> Self::At;
196}
