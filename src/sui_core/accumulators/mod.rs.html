32/// Merged value is the value stored inside accumulator objects.
33/// Each mergeable Move type will map to a single variant as its representation.
34///
35/// For instance, Balance<T> stores a single u64 value, so it will map to SumU128.
36/// A clawback Balance<T> will map to SumU128U128 since it also needs to represent
37/// the amount of the balance that has been frozen.
38#[derive(Debug, Copy, Clone)]
39enum MergedValue {
40    SumU128(u128),
41    SumU128U128(u128, u128),
42    /// Merkle root of events in this checkpoint and event count.
43    EventDigest(/* merkle root */ Digest, /* event count */ u64),
44}
45
46enum ClassifiedType {
47    Balance,
48    Unknown,
49}
50
51impl ClassifiedType {
52    fn classify(ty: &TypeTag) -> Self {
53        let TypeTag::Struct(struct_tag) = ty else {
54            return Self::Unknown;
55        };
56
57        if struct_tag.address == SUI_FRAMEWORK_ADDRESS
58            && struct_tag.module.as_ident_str() == BALANCE_MODULE_NAME
59            && struct_tag.name.as_ident_str() == BALANCE_STRUCT_NAME
60        {
61            return Self::Balance;
62        }
63
64        Self::Unknown
65    }
66}

68impl MergedValue {
69    fn add_move_call(
70        merge: Self,
71        split: Self,
72        root: Argument,
73        address: &AccumulatorAddress,
74        builder: &mut ProgrammableTransactionBuilder,
75    ) {
76        let ty = ClassifiedType::classify(&address.ty);
77        let address_arg = builder.pure(address.address).unwrap();
78        let checkpoint_seq = 0u64; /* TODO: replace with actual checkpoint sequence number */
79
80        match (ty, merge, split) {
81            (
82                ClassifiedType::Balance,
83                MergedValue::SumU128(merge_amount),
84                MergedValue::SumU128(split_amount),
85            ) => {
86                // Net out the merge and split amounts.
87                let (merge_amount, split_amount) = if merge_amount >= split_amount {
88                    (merge_amount - split_amount, 0)
89                } else {
90                    (0, split_amount - merge_amount)
91                };
92
93                if merge_amount != 0 || split_amount != 0 {
94                    let merge_amount = builder.pure(merge_amount).unwrap();
95                    let split_amount = builder.pure(split_amount).unwrap();
96                    builder.programmable_move_call(
97                        SUI_FRAMEWORK_PACKAGE_ID,
98                        ACCUMULATOR_SETTLEMENT_MODULE.into(),
99                        ACCUMULATOR_ROOT_SETTLE_U128_FUNC.into(),
100                        vec![address.ty.clone()],
101                        vec![root, address_arg, merge_amount, split_amount],
102                    );
103                }
104            }
105            (_, MergedValue::SumU128U128(_v1, _v2), MergedValue::SumU128U128(_w1, _w2)) => todo!(),
106            (_, MergedValue::EventDigest(digest, event_count), MergedValue::EventDigest(_, _)) => {
107                let args = vec![
108                    root,
109                    builder.pure(address.address).unwrap(),
110                    builder
111                        .pure(U256::from_le_bytes(&digest.into_inner()))
112                        .unwrap(),
113                    builder.pure(event_count).unwrap(),
114                    builder.pure(checkpoint_seq).unwrap(),
115                ];
116                builder.programmable_move_call(
117                    SUI_FRAMEWORK_PACKAGE_ID,
118                    ACCUMULATOR_SETTLEMENT_MODULE.into(),
119                    sui_types::accumulator_root::ACCUMULATOR_ROOT_SETTLEMENT_SETTLE_EVENTS_FUNC
120                        .into(),
121                    vec![],
122                    args,
123                );
124            }
125            _ => fatal!("invalid merge {:?} {:?}", merge, split),
126        }
127    }
128}

130impl From<MergedValueIntermediate> for MergedValue {
131    fn from(value: MergedValueIntermediate) -> Self {
132        match value {
133            MergedValueIntermediate::SumU128(v) => MergedValue::SumU128(v),
134            MergedValueIntermediate::SumU128U128(v1, v2) => MergedValue::SumU128U128(v1, v2),
135            MergedValueIntermediate::Events(events) => {
136                let event_count = events.len() as u64;
137                MergedValue::EventDigest(build_event_merkle_root(&events), event_count)
138            }
139        }
140    }

143/// MergedValueIntermediate is an intermediate / in-memory representation of the for
144/// accumulators. It is used to store the merged result of all accumulator writes in a single
145/// checkpoint.
146///
147/// This pattern is not necessary for fully commutative operations, since those could use MergedValue directly.
148///
149/// However, this supports the commutative-merge + non-commutative-update pattern, which will be used by event
150/// streams. In this pattern, everything within a checkpoint is merged commutatively, and then a single
151/// non-commutative update is applied to the accumulator at the end of the checkpoint.
152#[derive(Debug, Clone)]
153enum MergedValueIntermediate {
154    SumU128(u128),
155    SumU128U128(u128, u128),
156    Events(Vec<EventCommitment>),
157}
158
159impl MergedValueIntermediate {
160    // Create a zero value with the appropriate type for the accumulator value.
161    fn zero(value: &AccumulatorValue) -> Self {
162        match value {
163            AccumulatorValue::Integer(_) => Self::SumU128(0),
164            AccumulatorValue::IntegerTuple(_, _) => Self::SumU128U128(0, 0),
165            AccumulatorValue::EventDigest(_, _) => Self::Events(vec![]),
166        }
167    }
168
169    fn accumulate_into(
170        &mut self,
171        value: AccumulatorValue,
172        checkpoint_seq: u64,
173        transaction_idx: u64,
174    ) {
175        match (self, value) {
176            (Self::SumU128(v1), AccumulatorValue::Integer(v2)) => *v1 += v2 as u128,
177            (Self::SumU128U128(v1, v2), AccumulatorValue::IntegerTuple(w1, w2)) => {
178                *v1 += w1 as u128;
179                *v2 += w2 as u128;
180            }
181            (Self::Events(commitments), AccumulatorValue::EventDigest(event_idx, digest)) => {
182                commitments.push(EventCommitment::new(
183                    checkpoint_seq,
184                    transaction_idx,
185                    event_idx,
186                    digest,
187                ));
188            }
189            _ => {
190                fatal!("invalid merge");
191            }
192        }
193    }
194}

196struct Update {
197    merge: MergedValueIntermediate,
198    split: MergedValueIntermediate,
199    // Track input and output SUI for each update. Necessary so that when we construct
200    // a settlement transaction from a collection of Updates, they can accurately
201    // track the net SUI flows.
202    input_sui: u64,
203    output_sui: u64,
204}
205
206pub(crate) struct AccumulatorSettlementTxBuilder {
207    // updates is iterated over, must be a BTreeMap
208    updates: BTreeMap<AccumulatorObjId, Update>,
209    // addresses is only used for lookups.
210    addresses: HashMap<AccumulatorObjId, AccumulatorAddress>,
211}
212
213impl AccumulatorSettlementTxBuilder {
214    pub fn new(
215        cache: Option<&dyn TransactionCacheRead>,
216        ckpt_effects: &[TransactionEffects],
217        tx_index_offset: u64,
218    ) -> Self {
219        let checkpoint_seq = 0u64; /* TODO: replace with actual checkpoint sequence number */
220
221        let mut updates = BTreeMap::<_, _>::new();
222
223        let mut addresses = HashMap::<_, _>::new();
224
225        for (tx_index, effect) in ckpt_effects.iter().enumerate() {
226            let tx = effect.transaction_digest();
227            // TransactionEffectsAPI::accumulator_events() uses a linear scan of all
228            // object changes and allocates a new vector. In the common case (on validators),
229            // we still have still have the original vector in the writeback cache, so
230            // we can avoid the unnecessary work by just taking it from the cache.
231            let events = match cache.and_then(|c| c.take_accumulator_events(tx)) {
232                Some(events) => events,
233                None => effect.accumulator_events(),
234            };
235
236            for event in events {
237                // The input to the settlement is the sum of the outputs of accumulator events (i.e. deposits).
238                // and the output of the settlement is the sum of the inputs (i.e. withdraws).
239                let (event_input_sui, event_output_sui) = event.total_sui_in_event();
240
241                let AccumulatorEvent {
242                    accumulator_obj,
243                    write:
244                        AccumulatorWriteV1 {
245                            operation,
246                            value,
247                            address,
248                        },
249                } = event;
250
251                if let Some(prev) = addresses.insert(accumulator_obj, address.clone()) {
252                    debug_assert_eq!(prev, address);
253                }
254
255                let entry = updates.entry(accumulator_obj).or_insert_with(|| {
256                    let zero = MergedValueIntermediate::zero(&value);
257                    Update {
258                        merge: zero.clone(),
259                        split: zero,
260                        input_sui: 0,
261                        output_sui: 0,
262                    }
263                });
264
265                // The output of the event is the input of the settlement, and vice versa.
266                entry.input_sui += event_output_sui;
267                entry.output_sui += event_input_sui;
268
269                match operation {
270                    AccumulatorOperation::Merge => {
271                        entry.merge.accumulate_into(
272                            value,
273                            checkpoint_seq,
274                            tx_index as u64 + tx_index_offset,
275                        );
276                    }
277                    AccumulatorOperation::Split => {
278                        entry.split.accumulate_into(
279                            value,
280                            checkpoint_seq,
281                            tx_index as u64 + tx_index_offset,
282                        );
283                    }
284                }
285            }
286        }
287
288        Self { updates, addresses }
289    }

291    pub fn num_updates(&self) -> usize {
292        self.updates.len()
293    }

295    /// Returns a unified map of accumulator changes for all accounts.
296    /// The accumulator change for each account is merged from the merge and split operations.
297    pub fn collect_accumulator_changes(&self) -> BTreeMap<AccumulatorObjId, i128> {
298        self.updates
299            .iter()
300            .filter_map(|(object_id, update)| match (&update.merge, &update.split) {
301                (
302                    MergedValueIntermediate::SumU128(merge),
303                    MergedValueIntermediate::SumU128(split),
304                ) => Some((*object_id, *merge as i128 - *split as i128)),
305                _ => None,
306            })
307            .collect()
308    }

310    // TODO(address-balances): This currently only creates a single accumulator update transaction.
311    // To support multiple accumulator update transactions, we need to:
312    // - have each transaction take the accumulator root as a "non-exclusive mutable" input
313    // - each transaction writes out a set of fields that are disjoint from the others.
314    // - a barrier transaction must be added to advance the version of the accumulator root object.
315    //   The barrier transaction doesn't do any field writes. This is necessary in order to provide
316    //   a consistent view of the system accumulator state. When the version of the accumulator
317    //   root object is advanced, we know that all accumulator state updates prior to that version
318    //   have been applied.
319    pub fn build_tx(
320        self,
321        protocol_config: &ProtocolConfig,
322        epoch: u64,
323        accumulator_root_obj_initial_shared_version: SequenceNumber,
324        checkpoint_height: u64,
325    ) -> (
326        Vec<TransactionKind>, /* settlements */
327        TransactionKind,      /* barrier */
328    ) {
329        let Self { updates, addresses } = self;
330
331        let mut pending_updates = Vec::new();
332        let mut settlements = Vec::new();
333
334        let build_one_settlement_txn = |idx: u64, updates: &mut Vec<(AccumulatorObjId, Update)>| {
335            let (total_input_sui, total_output_sui) =
336                updates
337                    .iter()
338                    .fold((0, 0), |(acc_input, acc_output), (_, update)| {
339                        (acc_input + update.input_sui, acc_output + update.output_sui)
340                    });
341
342            Self::build_one_settlement_txn(
343                &addresses,
344                epoch,
345                idx,
346                checkpoint_height,
347                accumulator_root_obj_initial_shared_version,
348                updates.drain(..),
349                total_input_sui,
350                total_output_sui,
351            )
352        };
353
354        for (obj, update) in updates.into_iter() {
355            pending_updates.push((obj, update));
356
357            if pending_updates.len()
358                == protocol_config
359                    .max_updates_per_settlement_txn_as_option()
360                    .unwrap_or(u32::MAX) as usize
361            {
362                settlements.push(build_one_settlement_txn(
363                    settlements.len() as u64,
364                    // pending_updates will be drained and can be re-used
365                    &mut pending_updates,
366                ));
367            }
368        }
369
370        if !pending_updates.is_empty() {
371            settlements.push(build_one_settlement_txn(
372                settlements.len() as u64,
373                &mut pending_updates,
374            ));
375        }
376
377        // Now construct the barrier transaction
378        let mut builder = ProgrammableTransactionBuilder::new();
379        let root = builder
380            .input(CallArg::Object(ObjectArg::SharedObject {
381                id: SUI_ACCUMULATOR_ROOT_OBJECT_ID,
382                initial_shared_version: accumulator_root_obj_initial_shared_version,
383                mutability: SharedObjectMutability::Mutable,
384            }))
385            .unwrap();
386
387        Self::add_prologue(
388            &mut builder,
389            root,
390            epoch,
391            checkpoint_height,
392            settlements.len() as u64,
393            0u64,
394            0u64,
395        );
396
397        let barrier = TransactionKind::ProgrammableSystemTransaction(builder.finish());
398
399        (settlements, barrier)
400    }
401
402    fn add_prologue(
403        builder: &mut ProgrammableTransactionBuilder,
404        root: Argument,
405        epoch: u64,
406        checkpoint_height: u64,
407        idx: u64,
408        total_input_sui: u64,
409        total_output_sui: u64,
410    ) {
411        let epoch_arg = builder.pure(epoch).unwrap();
412        let checkpoint_height_arg = builder.pure(checkpoint_height).unwrap();
413        let idx_arg = builder.pure(idx).unwrap();
414        let total_input_sui_arg = builder.pure(total_input_sui).unwrap();
415        let total_output_sui_arg = builder.pure(total_output_sui).unwrap();
416
417        builder.programmable_move_call(
418            SUI_FRAMEWORK_PACKAGE_ID,
419            ACCUMULATOR_SETTLEMENT_MODULE.into(),
420            ACCUMULATOR_ROOT_SETTLEMENT_PROLOGUE_FUNC.into(),
421            vec![],
422            vec![
423                root,
424                epoch_arg,
425                checkpoint_height_arg,
426                idx_arg,
427                total_input_sui_arg,
428                total_output_sui_arg,
429            ],
430        );
431    }
432
433    fn build_one_settlement_txn(
434        addresses: &HashMap<AccumulatorObjId, AccumulatorAddress>,
435        epoch: u64,
436        idx: u64,
437        checkpoint_height: u64,
438        accumulator_root_obj_initial_shared_version: SequenceNumber,
439        updates: impl Iterator<Item = (AccumulatorObjId, Update)>,
440        total_input_sui: u64,
441        total_output_sui: u64,
442    ) -> TransactionKind {
443        let mut builder = ProgrammableTransactionBuilder::new();
444
445        let root = builder
446            .input(CallArg::Object(ObjectArg::SharedObject {
447                id: SUI_ACCUMULATOR_ROOT_OBJECT_ID,
448                initial_shared_version: accumulator_root_obj_initial_shared_version,
449                mutability: SharedObjectMutability::NonExclusiveWrite,
450            }))
451            .unwrap();
452
453        Self::add_prologue(
454            &mut builder,
455            root,
456            epoch,
457            checkpoint_height,
458            idx,
459            total_input_sui,
460            total_output_sui,
461        );
462
463        for (accumulator_obj, update) in updates {
464            let Update { merge, split, .. } = update;
465            let address = addresses.get(&accumulator_obj).unwrap();
466            let merged_value = MergedValue::from(merge);
467            let split_value = MergedValue::from(split);
468            MergedValue::add_move_call(merged_value, split_value, root, address, &mut builder);
469        }
470
471        TransactionKind::ProgrammableSystemTransaction(builder.finish())
472    }
