37/// Note that there is no need to have a separate InsufficientFunds variant.
38/// If the funds are insufficient, the execution would still have to abort and rely on
39/// a rescheduling to be able to execute again.
40pub enum ObjectFundsWithdrawStatus {
41    SufficientFunds,
42    // The receiver will be notified when the funds are determined to be sufficient or insufficient.
43    // The bool is true if the funds are sufficient, false if the funds are insufficient.
44    Pending(oneshot::Receiver<FundsWithdrawStatus>),
45}
46
47pub struct ObjectFundsChecker {
48    /// Watchers to keep track the last settled accumulator version.
49    /// This is updated whenever the settlement barrier transaction is executed.
50    last_settled_version_sender: watch::Sender<SequenceNumber>,
51    last_settled_version_receiver: watch::Receiver<SequenceNumber>,
52    inner: RwLock<Inner>,
53}
54
55#[derive(Default)]
56struct Inner {
57    /// Tracks the amount of pending unsettled withdraws for each account at each accumulator version.
58    /// When we check object funds sufficiency, we read the balance bounded by the withdraw accumulator version.
59    /// Balance are updated only by settlement transactions, not when we withdraw funds.
60    /// Hence when we are checking object funds, on top of the settled balance, we also need to account for
61    /// the amount of withdraws from the same consensus commit (that all reads from the same accumulator version).
62    unsettled_withdraws: BTreeMap<AccumulatorObjId, BTreeMap<SequenceNumber, u128>>,
63    /// Tracks the accounts that have pending withdraws at each accumulator version.
64    /// This information is not required for functional correctness, but needed to garbage collect
65    /// unused entries in unsettled_withdraws that are now fully committed. Without doing so unsettled_withdraws
66    /// may grow unbounded.
67    unsettled_accounts: BTreeMap<SequenceNumber, BTreeSet<AccumulatorObjId>>,
68}
69
70impl ObjectFundsChecker {
71    pub fn new(starting_accumulator_version: SequenceNumber) -> Self {
72        let (last_settled_version_sender, last_settled_version_receiver) =
73            watch::channel(starting_accumulator_version);
74        Self {
75            last_settled_version_sender,
76            last_settled_version_receiver,
77            inner: RwLock::new(Inner::default()),
78        }
79    }

81    #[instrument(level = "debug", skip_all, fields(tx_digest = ?certificate.digest()))]
82    pub fn should_commit_object_funds_withdraws(
83        &self,
84        certificate: &VerifiedExecutableTransaction,
85        execution_status: &ExecutionStatus,
86        accumulator_running_max_withdraws: &BTreeMap<AccumulatorObjId, u128>,
87        execution_env: &ExecutionEnv,
88        funds_read: &Arc<dyn AccountFundsRead>,
89        execution_scheduler: &Arc<ExecutionScheduler>,
90        epoch_store: &Arc<AuthorityPerEpochStore>,
91    ) -> bool {
92        if execution_status.is_err() {
93            // This transaction already failed. It does not matter any more
94            // whether it has sufficient object funds or not.
95            debug!("Transaction failed, committing effects");
96            return true;
97        }
98        let address_funds_reservations: BTreeSet<_> = certificate
99            .transaction_data()
100            .process_funds_withdrawals_for_execution(epoch_store.get_chain_identifier())
101            .into_keys()
102            .collect();
103        // All withdraws will show up as accumulator events with integer values.
104        // Among them, addresses that do not have funds reservations are object
105        // withdraws.
106        let object_withdraws: BTreeMap<_, _> = accumulator_running_max_withdraws
107            .clone()
108            .into_iter()
109            .filter(|(account, _)| !address_funds_reservations.contains(account))
110            .collect();
111        // If there are no object withdraws, we can skip checking object funds.
112        if object_withdraws.is_empty() {
113            debug!("No object withdraws, committing effects");
114            return true;
115        }
116        let Some(accumulator_version) = execution_env.assigned_versions.accumulator_version else {
117            // Fastpath transactions that perform object funds withdraws
118            // must wait for consensus to assign the accumulator version.
119            // We cannot optimize the scheduling by processing fastpath object withdraws
120            // sooner because these may get reverted, and we don't want them
121            // pollute the scheduler tracking state.
122            // TODO: We could however optimize execution by caching
123            // the execution state to avoid re-execution.
124            return false;
125        };
126        match self.check_object_funds(object_withdraws, accumulator_version, funds_read.as_ref()) {
127            // Sufficient funds, we can go ahead and commit the execution results as it is.
128            ObjectFundsWithdrawStatus::SufficientFunds => {
129                debug!("Object funds sufficient, committing effects");
130                true
131            }
132            // Currently insufficient funds. We need to wait until it reach a deterministic state
133            // before we can determine if it is really insufficient (to include potential deposits)
134            // At that time we will have to re-enqueue the transaction for execution again.
135            // Re-enqueue is handled here so the caller does not need to worry about it.
136            ObjectFundsWithdrawStatus::Pending(receiver) => {
137                let scheduler = execution_scheduler.clone();
138                let cert = certificate.clone();
139                let mut execution_env = execution_env.clone();
140                let epoch_store = epoch_store.clone();
141                tokio::task::spawn(async move {
142                    // It is possible that checkpoint executor finished executing
143                    // the current epoch and went ahead with epoch change asynchronously,
144                    // while this is still waiting.
145                    let _ = epoch_store
146                        .within_alive_epoch(async move {
147                            let tx_digest = cert.digest();
148                            match receiver.await {
149                                Ok(FundsWithdrawStatus::MaybeSufficient) => {
150                                    // The withdraw state is now deterministically known,
151                                    // so we can enqueue the transaction again and it will check again
152                                    // whether it is sufficient or not in the next execution.
153                                    // TODO: We should be able to optimize this by avoiding re-execution.
154                                    debug!(?tx_digest, "Object funds possibly sufficient");
155                                }
156                                Ok(FundsWithdrawStatus::Insufficient) => {
157                                    // Re-enqueue with insufficient funds status, so it will be executed
158                                    // in the next execution and fail through early error.
159                                    // FIXME: We need to also track the amount of gas that was used,
160                                    // so that we could charge properly in the next execution when we
161                                    // go through early error. Otherwise we would undercharge.
162                                    execution_env = execution_env.with_insufficient_funds();
163                                    debug!(?tx_digest, "Object funds insufficient");
164                                }
165                                Err(e) => {
166                                    tracing::error!(
167                                        "Error receiving funds withdraw status: {:?}",
168                                        e
169                                    );
170                                }
171                            }
172                            scheduler.send_transaction_for_execution(
173                                &cert,
174                                execution_env,
175                                // TODO: Should the enqueue_time be the original enqueue time
176                                // of this transaction?
177                                Instant::now(),
178                            );
179                        })
180                        .await;
181                });
182                false
183            }
184        }
185    }

187    fn check_object_funds(
188        &self,
189        object_withdraws: BTreeMap<AccumulatorObjId, u128>,
190        accumulator_version: SequenceNumber,
191        funds_read: &dyn AccountFundsRead,
192    ) -> ObjectFundsWithdrawStatus {
193        let last_settled_version = *self.last_settled_version_receiver.borrow();
194        if accumulator_version <= last_settled_version {
195            // If the version we are withdrawing from is already settled, we have all the information
196            // we need to determine if the funds are sufficient or not.
197            if self.try_withdraw(funds_read, &object_withdraws, accumulator_version) {
198                return ObjectFundsWithdrawStatus::SufficientFunds;
199            } else {
200                let (sender, receiver) = oneshot::channel();
201                // unwrap is safe because the receiver is defined right above.
202                sender.send(FundsWithdrawStatus::Insufficient).unwrap();
203                return ObjectFundsWithdrawStatus::Pending(receiver);
204            }
205        }
206
207        // Spawn a task to wait for the last settled version to become accumulator_version,
208        // before we could check again.
209        let last_settled_version_sender = self.last_settled_version_sender.clone();
210        let (sender, receiver) = oneshot::channel();
211        tokio::spawn(async move {
212            let mut version_receiver = last_settled_version_sender.subscribe();
213            // The wait is guaranteed to be notified because we update version after executing each settlement transaction,
214            // and every settlement transaction must eventually be executed.
215            let res = version_receiver
216                .wait_for(|v| *v >= accumulator_version)
217                .await;
218            if res.is_err() {
219                // This shouldn't happen, but just to be safe.
220                tracing::error!("Last settled accumulator version receiver channel closed");
221                return;
222            }
223            // We notify the waiter that the funds are now deterministically known,
224            // but we don't need to check here whether they are sufficient or not.
225            // Next time during execution we will check again.
226            let _ = sender.send(FundsWithdrawStatus::MaybeSufficient);
227        });
228        ObjectFundsWithdrawStatus::Pending(receiver)
229    }

231    fn try_withdraw(
232        &self,
233        funds_read: &dyn AccountFundsRead,
234        object_withdraws: &BTreeMap<AccumulatorObjId, u128>,
235        accumulator_version: SequenceNumber,
236    ) -> bool {
237        for (obj_id, amount) in object_withdraws {
238            let funds = funds_read.get_account_amount_at_version(obj_id, accumulator_version);
239            // Reading inner without a top-level lock is safe because no two transactions can be withdrawing
240            // from the same account at the same time.
241            let unsettled_withdraw = self
242                .inner
243                .read()
244                .unsettled_withdraws
245                .get(obj_id)
246                .and_then(|withdraws| withdraws.get(&accumulator_version))
247                .copied()
248                .unwrap_or_default();
249            debug!(
250                ?obj_id,
251                ?funds,
252                ?accumulator_version,
253                ?unsettled_withdraw,
254                ?amount,
255                "Trying to withdraw"
256            );
257            assert!(funds >= unsettled_withdraw);
258            if funds - unsettled_withdraw < *amount {
259                return false;
260            }
261        }
262        let mut inner = self.inner.write();
263        for (obj_id, amount) in object_withdraws {
264            let entry = inner
265                .unsettled_withdraws
266                .entry(*obj_id)
267                .or_default()
268                .entry(accumulator_version)
269                .or_default();
270            debug!(?obj_id, ?amount, ?entry, "Updating unsettled withdraws");
271            *entry = entry.checked_add(*amount).unwrap();
272
273            inner
274                .unsettled_accounts
275                .entry(accumulator_version)
276                .or_default()
277                .insert(*obj_id);
278        }
279        true
280    }
281
282    pub fn settle_accumulator_version(&self, next_accumulator_version: SequenceNumber) {
283        // unwrap is safe because a receiver is always alive as part of self.
284        self.last_settled_version_sender
285            .send(next_accumulator_version)
286            .unwrap();
287    }
288
289    pub fn commit_effects<'a>(
290        &self,
291        committed_effects: impl Iterator<Item = &'a TransactionEffects>,
292    ) {
293        let committed_accumulator_versions = committed_effects
294            .filter_map(|effects| {
295                effects.object_changes().into_iter().find_map(|o| {
296                    if o.id == SUI_ACCUMULATOR_ROOT_OBJECT_ID {
297                        o.input_version
298                    } else {
299                        None
300                    }
301                })
302            })
303            .collect::<Vec<_>>();
304        self.commit_accumulator_versions(committed_accumulator_versions);
305    }
306
307    fn commit_accumulator_versions(&self, committed_accumulator_versions: Vec<SequenceNumber>) {
308        let mut inner = self.inner.write();
309        for accumulator_version in committed_accumulator_versions {
310            let accounts = inner
311                .unsettled_accounts
312                .remove(&accumulator_version)
313                .unwrap_or_default();
314            for account in accounts {
315                let withdraws = inner.unsettled_withdraws.get_mut(&account);
316                if let Some(withdraws) = withdraws {
317                    withdraws.remove(&accumulator_version);
318                    if withdraws.is_empty() {
319                        inner.unsettled_withdraws.remove(&account);
320                    }
321                }
322            }
323        }
324    }
