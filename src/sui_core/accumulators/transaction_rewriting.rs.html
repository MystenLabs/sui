4use sui_types::base_types::{SequenceNumber, SuiAddress};
5use sui_types::coin_reservation::{CoinReservationResolverTrait, ParsedObjectRefWithdrawal};
6use sui_types::digests::ChainIdentifier;
7use sui_types::error::UserInputResult;
8use sui_types::transaction::{CallArg, ObjectArg, ProgrammableTransaction, TransactionKind};
9
10/// Rewrites coin reservation inputs (fake coins encoded as masked ObjectRefs) into
11/// FundsWithdrawalArgs so the executor can resolve them as balance withdrawals.
12///
13/// Returns `Ok(Some(rewritten))` where each bool flags whether that input was rewritten,
14/// `Ok(None)` if nothing was rewritten, or `Err` if a reservation cannot be resolved.
15///
16/// `accumulator_version` selects the accumulator version for MVCC lookup during checkpoint
17/// replay (read before any settlement modifies it); pass `None` for the latest version.
18pub fn rewrite_transaction_for_coin_reservations(
19    chain_identifier: ChainIdentifier,
20    coin_reservation_resolver: &dyn CoinReservationResolverTrait,
21    sender: SuiAddress,
22    transaction_kind: &mut TransactionKind,
23    accumulator_version: Option<SequenceNumber>,
24) -> UserInputResult<Option<Vec<bool>>> {
25    match transaction_kind {
26        TransactionKind::ProgrammableTransaction(pt) => {
27            rewrite_programmable_transaction_for_coin_reservations(
28                chain_identifier,
29                coin_reservation_resolver,
30                sender,
31                pt,
32                accumulator_version,
33            )
34        }
35        _ => Ok(None),
36    }
37}
38
39fn rewrite_programmable_transaction_for_coin_reservations(
40    chain_identifier: ChainIdentifier,
41    coin_reservation_resolver: &dyn CoinReservationResolverTrait,
42    sender: SuiAddress,
43    pt: &mut ProgrammableTransaction,
44    accumulator_version: Option<SequenceNumber>,
45) -> UserInputResult<Option<Vec<bool>>> {
46    if pt.coin_reservation_obj_refs().count() == 0 {
47        return Ok(None);
48    }
49
50    let mut rewritten_inputs = Vec::with_capacity(pt.inputs.len());
51    for input in pt.inputs.iter_mut() {
52        if let CallArg::Object(ObjectArg::ImmOrOwnedObject(object_ref)) = input
53            && let Some(parsed) = ParsedObjectRefWithdrawal::parse(object_ref, chain_identifier)
54        {
55            rewritten_inputs.push(true);
56
57            let withdraw = coin_reservation_resolver.resolve_funds_withdrawal(
58                sender,
59                parsed,
60                accumulator_version,
61            )?;
62            *input = CallArg::FundsWithdrawal(withdraw);
63        } else {
64            rewritten_inputs.push(false);
65        }
66    }
67
68    Ok(Some(rewritten_inputs))
69}
