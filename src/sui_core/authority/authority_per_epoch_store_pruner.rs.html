3use crate::authority::authority_per_epoch_store::EPOCH_DB_PREFIX;
4use itertools::Itertools;
5use std::fs;
6use std::path::PathBuf;
7use std::time::Duration;
8use sui_config::node::AuthorityStorePruningConfig;
9use tokio::sync::oneshot;
10use tracing::log::{info, warn};
11use typed_store::rocks::safe_drop_db;
12
13pub struct AuthorityPerEpochStorePruner {
14    _cancel_handle: oneshot::Sender<()>,
15}
16
17impl AuthorityPerEpochStorePruner {
18    pub fn new(parent_path: PathBuf, config: &AuthorityStorePruningConfig) -> Self {
19        let (_cancel_handle, mut recv) = tokio::sync::oneshot::channel();
20        let num_latest_epoch_dbs_to_retain = config.num_latest_epoch_dbs_to_retain;
21        if num_latest_epoch_dbs_to_retain == 0 || num_latest_epoch_dbs_to_retain == usize::MAX {
22            info!("Skipping pruning of epoch tables as we want to retain all versions");
23            return Self { _cancel_handle };
24        }
25        let mut prune_interval =
26            tokio::time::interval(Duration::from_secs(config.epoch_db_pruning_period_secs));
27        tokio::task::spawn(async move {
28            loop {
29                tokio::select! {
30                    _ = prune_interval.tick() => {
31                        info!("Starting pruning of epoch tables");
32                        match Self::prune_old_directories(&parent_path, num_latest_epoch_dbs_to_retain).await {
33                            Ok(pruned_count) => info!("Finished pruning old epoch databases. Pruned {} dbs", pruned_count),
34                            Err(err) => warn!("Error while removing old epoch databases {:?}", err),
35                        }
36                    }
37                    _ = &mut recv => break,
38                }
39            }
40        });
41        Self { _cancel_handle }
42    }
43
44    async fn prune_old_directories(
45        parent_path: &PathBuf,
46        num_latest_epoch_dbs_to_retain: usize,
47    ) -> Result<usize, anyhow::Error> {
48        let mut candidates = vec![];
49        let directories = fs::read_dir(parent_path)?.collect::<Result<Vec<_>, _>>()?;
50        for directory in directories {
51            let path = directory.path();
52            if let Some(filename) = directory.file_name().to_str()
53                && let Ok(epoch) = filename.split_at(EPOCH_DB_PREFIX.len()).1.parse::<u64>()
54            {
55                candidates.push((epoch, path));
56            }
57        }
58        let mut pruned = 0;
59        let mut gc_tasks = vec![];
60        if num_latest_epoch_dbs_to_retain < candidates.len() {
61            let to_prune = candidates.len() - num_latest_epoch_dbs_to_retain;
62            for (_, path) in candidates.into_iter().sorted().take(to_prune) {
63                info!("Dropping epoch directory {:?}", path);
64                pruned += 1;
65                gc_tasks.push(safe_drop_db(
66                    path.join("recovery_log"),
67                    Duration::from_secs(30),
68                ));
69                gc_tasks.push(safe_drop_db(path, Duration::from_secs(30)));
70            }
71        }
72        futures::future::join_all(gc_tasks)
73            .await
74            .into_iter()
75            .collect::<Result<Vec<_>, _>>()?;
76        Ok(pruned)
77    }
78}

80#[cfg(test)]
81mod tests {
82    use crate::authority::authority_per_epoch_store_pruner::AuthorityPerEpochStorePruner;
83    use std::fs;
84
85    #[tokio::test]
86    async fn test_basic_epoch_pruner() {
87        let parent_directory = tempfile::tempdir().unwrap().keep();
88        let directories: Vec<_> = vec!["epoch_0", "epoch_1", "epoch_3", "epoch_4"]
89            .into_iter()
90            .map(|name| parent_directory.join(name))
91            .collect();
92        for directory in &directories {
93            fs::create_dir(directory).expect("failed to create directory");
94        }
95
96        let pruned = AuthorityPerEpochStorePruner::prune_old_directories(&parent_directory, 2)
97            .await
98            .unwrap();
99        assert_eq!(pruned, 2);
100        assert_eq!(
101            directories
102                .into_iter()
103                .map(|f| fs::metadata(f).is_ok())
104                .collect::<Vec<_>>(),
105            vec![false, false, true, true]
106        );
107    }
