13// Versioning process:
14//
15// Object storage versioning is done lazily (at read time) - therefore we must always preserve the
16// code for reading the very first storage version. For all versions, a migration function
17//
18//   f(V_n) -> V_(n+1)
19//
20// must be defined. This way we can iteratively migrate the very oldest version to the very newest
21// version at any point in the future.
22//
23// To change the format of the object table value types (StoreObject and StoreMoveObject), use the
24// following process:
25// - Add a new variant to the enum to store the new version type.
26// - Extend the `migrate` functions to migrate from the previous version to the new version.
27// - Change `From<Object> for StoreObjectPair` to create the newest version only.
28//
29// Additionally, the first time we version these formats, we will need to:
30// - Add a check in the `TryFrom<StoreObjectPair> for Object` to see if the object that was just
31//   read is the latest version.
32// - If it is not, use the migration function (as explained above) to migrate it to the next
33//   version.
34// - Repeat until we have arrive at the current version.
35
36/// Enum wrapper for versioning
37#[derive(Eq, PartialEq, Debug, Clone, Deserialize, Serialize, Hash)]
38pub enum StoreObjectWrapper {
39    V1(StoreObjectV1),
40}
41
42// always points to latest version.
43pub type StoreObject = StoreObjectV1;
44
45impl StoreObjectWrapper {
46    pub fn migrate(self) -> Self {
47        // TODO: when there are multiple versions, we must iteratively migrate from version N to
48        // N+1 until we arrive at the latest version
49        self
50    }
51
52    // Always returns the most recent version. Older versions are migrated to the latest version at
53    // read time, so there is never a need to access older versions.
54    pub fn inner(&self) -> &StoreObject {
55        match self {
56            Self::V1(v1) => v1,
57
58            // can remove #[allow] when there are multiple versions
59            #[allow(unreachable_patterns)]
60            _ => panic!("object should have been migrated to latest version at read time"),
61        }
62    }
63    pub fn into_inner(self) -> StoreObject {
64        match self {
65            Self::V1(v1) => v1,
66
67            // can remove #[allow] when there are multiple versions
68            #[allow(unreachable_patterns)]
69            _ => panic!("object should have been migrated to latest version at read time"),
70        }
71    }
72}
73
74impl From<StoreObject> for StoreObjectWrapper {
75    fn from(o: StoreObject) -> Self {
76        StoreObjectWrapper::V1(o)
77    }
78}

80#[derive(Eq, PartialEq, Debug, Clone, Deserialize, Serialize, Hash)]
81pub enum StoreObjectV1 {
82    Value(Box<StoreObjectValue>),
83    Deleted,
84    Wrapped,
85}

87/// Forked version of [`sui_types::object::Object`]
88/// Used for efficient storing of move objects in the database
89#[derive(Eq, PartialEq, Debug, Clone, Deserialize, Serialize, Hash)]
90pub struct StoreObjectValue {
91    pub data: StoreData,
92    pub owner: Owner,
93    pub previous_transaction: TransactionDigest,
94    pub storage_rebate: u64,
95}
96
97/// Forked version of [`sui_types::object::Data`]
98/// Adds extra enum value `IndirectObject`, which represents a reference to an object stored separately
99#[derive(Eq, PartialEq, Debug, Clone, Deserialize, Serialize, Hash)]
100pub enum StoreData {
101    Move(MoveObject),
102    Package(MovePackage),
103    IndirectObjectDeprecated,
104    Coin(u64),
105}

107pub fn get_store_object(object: Object) -> StoreObjectWrapper {
108    let object = object.into_inner();
109    let data = match object.data {
110        Data::Package(package) => StoreData::Package(package),
111        Data::Move(move_obj) => {
112            if move_obj.type_().is_gas_coin() {
113                StoreData::Coin(
114                    Coin::from_bcs_bytes(move_obj.contents())
115                        .expect("failed to deserialize coin")
116                        .balance
117                        .value(),
118                )
119            } else {
120                StoreData::Move(move_obj)
121            }
122        }
123    };
124    let store_object = StoreObjectValue {
125        data,
126        owner: object.owner,
127        previous_transaction: object.previous_transaction,
128        storage_rebate: object.storage_rebate,
129    };
130    StoreObject::Value(Box::new(store_object)).into()
131}

133pub(crate) fn try_construct_object(
134    object_key: &ObjectKey,
135    store_object: StoreObjectValue,
136) -> Result<Object, SuiError> {
137    let data = match store_object.data {
138        StoreData::Move(object) => Data::Move(object),
139        StoreData::Package(package) => Data::Package(package),
140        StoreData::Coin(balance) => unsafe {
141            Data::Move(MoveObject::new_from_execution_with_limit(
142                MoveObjectType::gas_coin(),
143                true,
144                object_key.1,
145                bcs::to_bytes(&(object_key.0, balance)).expect("serialization failed"),
146                u64::MAX,
147            )?)
148        },
149        _ => {
150            return Err(SuiErrorKind::Storage(
151                "corrupted field: inconsistent object representation".to_string(),
152            )
153            .into());
154        }
155    };
156
157    Ok(ObjectInner {
158        data,
159        owner: store_object.owner,
160        previous_transaction: store_object.previous_transaction,
161        storage_rebate: store_object.storage_rebate,
162    }
163    .into())
164}
