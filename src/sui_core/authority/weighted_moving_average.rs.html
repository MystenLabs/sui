4use std::collections::VecDeque;
5use std::fmt;
6
7#[derive(Clone)]
8struct Sample {
9    value: u64,
10    weight: u64,
11    insertion_total: u64, // cumulative weight at insertion time
12}
13
14impl fmt::Debug for Sample {
15    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
16        write!(f, "({:.2}, {:.2})", self.value, self.weight)
17    }
18}
19
20impl fmt::Display for Sample {
21    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
22        write!(f, "({:.2}, {:.2})", self.value, self.weight)
23    }
24}
25
26/// Weighted moving average of a stream of samples.
27///
28/// This is not only computing a weighted average, it is also attempting to prevent
29/// low weight samples from evicting higher weight samples. The idea is that we do
30/// not want the measurement to quickly become skewed due to the arrival of several
31/// low weight samples.
32#[derive(Debug, Clone)]
33pub struct WeightedMovingAverage {
34    queue: VecDeque<Sample>,
35    total_weight: u64, // running cumulative weight of all added samples
36    max_size: usize,   // maximum number of samples to store
37    default: u64,
38}
39
40impl WeightedMovingAverage {
41    /// Create a new WeightedMovingAverage with a fixed maximum number of samples.
42    pub fn new(default: u64, max_size: usize) -> Self {
43        Self {
44            queue: VecDeque::new(),
45            total_weight: 0,
46            max_size,
47            default,
48        }
49    }
50
51    fn is_full(&self) -> bool {
52        self.queue.len() >= self.max_size
53    }
54
55    pub fn add_sample(&mut self, value: u64, weight: u64) {
56        self.total_weight += weight;
57        let sample = Sample {
58            value,
59            weight,
60            insertion_total: self.total_weight,
61        };
62
63        // See if there are old samples to evict.
64        while self.is_full() && self.try_evict_oldest_sample().is_some() {}
65
66        // If we are still full, find a sample with weight less than the new sample
67        // and replace it with the new sample.
68        if self.is_full() {
69            // Find the old sample with weight less than the new sample
70            if let Some(min_idx) = self
71                .queue
72                .iter()
73                .enumerate()
74                .find(|(_, s)| s.weight < sample.weight)
75                .map(|(i, _)| i)
76            {
77                self.queue.remove(min_idx);
78                self.queue.push_back(sample);
79            }
80
81            // The sample is dropped, but it is has still incremented the total weight
82            // so eventually the oldest sample will become old enough to be evicted and
83            // make room for new measurements.
84        } else {
85            self.queue.push_back(sample);
86        }
87    }
88
89    // If the oldest sample is old enough, we evict it. "old enough" means that
90    // the total weight added since its insertion is max_size * weight.
91    // The intuition is that this preserves the property that if all samples had equal weight,
92    // you would have to add max_size samples to the queue before the oldest sample is evicted.
93    //
94    // For example, for a queue of size 10, and a sample of weight 10, you must add a total
95    // of 100 weight to the queue before that sample becomes old enough to be evicted.
96    fn try_evict_oldest_sample(&mut self) -> Option<Sample> {
97        let old_sample = self.queue.front()?;
98
99        let weight_since_insertion = self.total_weight - old_sample.insertion_total;
100        let threshold = old_sample.weight * self.max_size as u64;
101
102        if weight_since_insertion >= threshold {
103            self.queue.pop_front()
104        } else {
105            None
106        }
107    }
108
109    /// Compute the weighted average over the samples currently in the window.
110    pub fn get_weighted_average(&self) -> u64 {
111        if self.queue.is_empty() {
112            return self.default;
113        }
114        let (weighted_sum, total_w) = self.queue.iter().fold((0u64, 0u64), |(sum, w_sum), s| {
115            (sum + s.value * s.weight, w_sum + s.weight)
116        });
117        weighted_sum / total_w
118    }
119}

121#[cfg(test)]
122mod tests {
123    use super::*;
124
125    #[test]
126    fn test_weighted_moving_average_equal_weights() {
127        let mut wma = WeightedMovingAverage::new(0, 10);
128        wma.add_sample(100, 100);
129        assert_eq!(wma.get_weighted_average(), 100);
130
131        // Add 9 samples with value 1.0 and weight 100.0
132        for _ in 0..9 {
133            wma.add_sample(1, 100);
134        }
135        assert_eq!(wma.get_weighted_average(), 10);
136
137        // one more sample with weight 100 pushes the oldest sample out
138        wma.add_sample(1, 100);
139        assert_eq!(wma.get_weighted_average(), 1);
140    }
141
142    #[test]
143    fn test_old_sample_is_evicted() {
144        let mut wma = WeightedMovingAverage::new(0, 10);
145        wma.add_sample(10, 10);
146        assert_eq!(wma.get_weighted_average(), 10);
147
148        // verify that after adding 99 samples of value 1, weight 1,
149        // the original sample is still in the average. Not enough
150        // weight has been added to evict it.
151        for _ in 0..99 {
152            wma.add_sample(1, 1);
153        }
154        assert_eq!(wma.get_weighted_average(), 5);
155
156        // One more sample now means that there has been 10 x the capacity
157        // in weight added, so the oldest sample is evicted and the moving
158        // average is now 1.
159        wma.add_sample(1, 1);
160        assert_eq!(wma.get_weighted_average(), 1);
161    }
162
163    #[test]
164    fn test_lowest_sample_is_replaced() {
165        let mut wma = WeightedMovingAverage::new(0, 10);
166        wma.add_sample(10, 10);
167        assert_eq!(wma.get_weighted_average(), 10);
168
169        for _ in 0..10 {
170            wma.add_sample(1, 1);
171        }
172        assert_eq!(wma.get_weighted_average(), 5);
173
174        // A weight 5 sample is enough to evict one of the weight 1 samples,
175        // which moves the average up to 6.
176        wma.add_sample(10, 5);
177        assert_eq!(wma.get_weighted_average(), 6);
178    }
179}
