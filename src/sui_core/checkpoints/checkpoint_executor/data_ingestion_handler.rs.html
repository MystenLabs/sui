16pub(crate) fn store_checkpoint_locally(
17    path: impl AsRef<Path>,
18    checkpoint_data: &CheckpointData,
19) -> SuiResult {
20    let path = path.as_ref();
21    let file_name = format!("{}.chk", checkpoint_data.checkpoint_summary.sequence_number);
22
23    std::fs::create_dir_all(path).map_err(|err| {
24        SuiErrorKind::FileIOError(format!(
25            "failed to save full checkpoint content locally {:?}",
26            err
27        ))
28    })?;
29
30    Blob::encode(&checkpoint_data, BlobEncoding::Bcs)
31        .map_err(|_| SuiErrorKind::TransactionSerializationError {
32            error: "failed to serialize full checkpoint content".to_string(),
33        }) // Map the first error
34        .and_then(|blob| {
35            std::fs::write(path.join(file_name), blob.to_bytes()).map_err(|_| {
36                SuiErrorKind::FileIOError(
37                    "failed to save full checkpoint content locally".to_string(),
38                )
39            })
40        })?;
41
42    Ok(())
43}

45pub(crate) fn load_checkpoint(
46    ckpt_data: &CheckpointExecutionData,
47    ckpt_tx_data: &CheckpointTransactionData,
48    object_store: &dyn ObjectStore,
49    transaction_cache_reader: &dyn TransactionCacheRead,
50) -> SuiResult<Checkpoint> {
51    let event_tx_digests = ckpt_tx_data
52        .effects
53        .iter()
54        .flat_map(|fx| fx.events_digest().map(|_| fx.transaction_digest()).copied())
55        .collect::<Vec<_>>();
56
57    let mut events = transaction_cache_reader
58        .multi_get_events(&event_tx_digests)
59        .into_iter()
60        .zip(event_tx_digests)
61        .map(|(maybe_event, tx_digest)| {
62            maybe_event
63                .ok_or(SuiErrorKind::TransactionEventsNotFound { digest: tx_digest }.into())
64                .map(|event| (tx_digest, event))
65        })
66        .collect::<SuiResult<HashMap<_, _>>>()?;
67
68    let mut transactions = Vec::with_capacity(ckpt_tx_data.transactions.len());
69    for (tx, fx) in ckpt_tx_data
70        .transactions
71        .iter()
72        .zip(ckpt_tx_data.effects.iter())
73    {
74        let events = fx.events_digest().map(|_event_digest| {
75            events
76                .remove(fx.transaction_digest())
77                .expect("event was already checked to be present")
78        });
79
80        let transaction = ExecutedTransaction {
81            transaction: tx.transaction_data().clone(),
82            signatures: tx.tx_signatures().to_vec(),
83            effects: fx.clone(),
84            events,
85            unchanged_loaded_runtime_objects: transaction_cache_reader
86                .get_unchanged_loaded_runtime_objects(tx.digest())
87                // We don't write empty sets to the DB to save space, so if this load went through
88                // the writeback cache to the DB itself it wouldn't find an entry.
89                .unwrap_or_default(),
90        };
91        transactions.push(transaction);
92    }
93
94    let object_set = {
95        let refs = transactions
96            .iter()
97            .flat_map(|tx| {
98                sui_types::storage::get_transaction_object_set(
99                    &tx.transaction,
100                    &tx.effects,
101                    &tx.unchanged_loaded_runtime_objects,
102                )
103            })
104            .collect::<BTreeSet<_>>()
105            .into_iter()
106            .collect::<Vec<_>>();
107
108        let objects = object_store.multi_get_objects_by_key(&refs);
109
110        let mut object_set = ObjectSet::default();
111        for (idx, object) in objects.into_iter().enumerate() {
112            object_set.insert(object.ok_or_else(|| {
113                sui_types::storage::error::Error::custom(format!(
114                    "unabled to load object {:?}",
115                    refs[idx]
116                ))
117            })?);
118        }
119        object_set
120    };
121    let checkpoint = Checkpoint {
122        summary: ckpt_data.checkpoint.clone().into(),
123        contents: ckpt_data.checkpoint_contents.clone(),
124        transactions,
125        object_set,
126    };
127    Ok(checkpoint)
128}
