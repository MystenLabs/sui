177    /// AdditionalConsensusState tracks any in-memory state that is retained by ConsensusHandler
178    /// between consensus commits. Because of crash recovery, using such data is inherently risky.
179    /// In order to do this safely, we must store data from a fixed number of previous commits.
180    /// Then, at start-up, that same fixed number of already processed commits is replayed to
181    /// reconstruct the state.
182    ///
183    /// To make sure that bugs in this process appear immediately, we record the digest of this
184    /// state in ConsensusCommitPrologue, so that any deviation causes an immediate fork.
185    #[derive(Serialize, Deserialize)]
186    pub(super) struct AdditionalConsensusState {
187        commit_interval_observer: CommitIntervalObserver,
188    }

199        /// Update all internal state based on the new commit
200        pub(crate) fn observe_commit(
201            &mut self,
202            protocol_config: &ProtocolConfig,
203            epoch_start_time: u64,
204            consensus_commit: &impl ConsensusCommitAPI,
205        ) -> ConsensusCommitInfo {
206            self.commit_interval_observer
207                .observe_commit_time(consensus_commit);
208
209            let estimated_commit_period = self
210                .commit_interval_observer
211                .commit_interval_estimate()
212                .unwrap_or(Duration::from_millis(
213                    protocol_config.min_checkpoint_interval_ms(),
214                ));
215
216            info!("estimated commit rate: {:?}", estimated_commit_period);
217
218            self.commit_info_impl(
219                epoch_start_time,
220                protocol_config,
221                consensus_commit,
222                Some(estimated_commit_period),
223            )
224        }

226        pub(crate) fn stateless_commit_info(
227            &self,
228            epoch_store: &AuthorityPerEpochStore,
229            consensus_commit: &impl ConsensusCommitAPI,
230        ) -> ConsensusCommitInfo {
231            let protocol_config = epoch_store.protocol_config();
232            let epoch_start_time = epoch_store.epoch_start_config().epoch_start_timestamp_ms();
233            self.commit_info_impl(epoch_start_time, protocol_config, consensus_commit, None)
234        }
235
236        fn commit_info_impl(
237            &self,
238            epoch_start_time: u64,
239            protocol_config: &ProtocolConfig,
240            consensus_commit: &impl ConsensusCommitAPI,
241            estimated_commit_period: Option<Duration>,
242        ) -> ConsensusCommitInfo {
243            let leader_author = consensus_commit.leader_author_index();
244            let timestamp = consensus_commit.commit_timestamp_ms();
245
246            let timestamp = if timestamp < epoch_start_time {
247                error!(
248                    "Unexpected commit timestamp {timestamp} less then epoch start time {epoch_start_time}, author {leader_author:?}"
249                );
250                epoch_start_time
251            } else {
252                timestamp
253            };
254
255            ConsensusCommitInfo {
256                _phantom: PhantomData,
257                round: consensus_commit.leader_round(),
258                timestamp,
259                leader_author,
260                sub_dag_index: consensus_commit.commit_sub_dag_index(),
261                consensus_commit_digest: consensus_commit.consensus_digest(protocol_config),
262                additional_state_digest: Some(self.digest()),
263                estimated_commit_period,
264                skip_consensus_commit_prologue_in_test: false,
265            }
266        }
267
268        /// Get the digest of the current state.
269        fn digest(&self) -> AdditionalConsensusStateDigest {
270            let mut hash = DefaultHash::new();
271            bcs::serialize_into(&mut hash, self).unwrap();
272            AdditionalConsensusStateDigest::new(hash.finalize().into())
273        }

394        pub fn create_consensus_commit_prologue_transaction(
395            &self,
396            epoch: u64,
397            protocol_config: &ProtocolConfig,
398            cancelled_txn_version_assignment: Vec<(
399                TransactionDigest,
400                Vec<(ConsensusObjectSequenceKey, SequenceNumber)>,
401            )>,
402            commit_info: &ConsensusCommitInfo,
403            indirect_state_observer: IndirectStateObserver,
404        ) -> VerifiedExecutableTransaction {
405            let version_assignments = if protocol_config
406                .record_consensus_determined_version_assignments_in_prologue_v2()
407            {
408                Some(
409                    ConsensusDeterminedVersionAssignments::CancelledTransactionsV2(
410                        cancelled_txn_version_assignment,
411                    ),
412                )
413            } else if protocol_config.record_consensus_determined_version_assignments_in_prologue()
414            {
415                Some(
416                    ConsensusDeterminedVersionAssignments::CancelledTransactions(
417                        cancelled_txn_version_assignment
418                            .into_iter()
419                            .map(|(tx_digest, versions)| {
420                                (
421                                    tx_digest,
422                                    versions.into_iter().map(|(id, v)| (id.0, v)).collect(),
423                                )
424                            })
425                            .collect(),
426                    ),
427                )
428            } else {
429                None
430            };
431
432            if protocol_config.record_additional_state_digest_in_prologue() {
433                let additional_state_digest =
434                    if protocol_config.additional_consensus_digest_indirect_state() {
435                        let d1 = commit_info.additional_state_digest();
436                        indirect_state_observer.fold_with(d1)
437                    } else {
438                        commit_info.additional_state_digest()
439                    };
440
441                self.consensus_commit_prologue_v4_transaction(
442                    epoch,
443                    version_assignments.unwrap(),
444                    additional_state_digest,
445                )
446            } else if let Some(version_assignments) = version_assignments {
447                self.consensus_commit_prologue_v3_transaction(epoch, version_assignments)
448            } else if protocol_config.include_consensus_digest_in_prologue() {
449                self.consensus_commit_prologue_v2_transaction(epoch)
450            } else {
451                self.consensus_commit_prologue_transaction(epoch)
452            }
453        }

611        checkpoint_service: Arc<C>,

613        consensus_adapter: Arc<ConsensusAdapter>,

620        traffic_controller: Option<Arc<TrafficController>>,

623        let mut last_consensus_stats = epoch_store
624            .get_last_consensus_stats()
625            .expect("Should be able to read last consensus index");
626        // stats is empty at the beginning of epoch.
627        if !last_consensus_stats.stats.is_initialized() {
628            last_consensus_stats.stats = ConsensusStats::new(committee.size());
629        }
630        let execution_scheduler_sender =
631            ExecutionSchedulerSender::start(execution_scheduler, epoch_store.clone());
632        let commit_rate_estimate_window_size = epoch_store
633            .protocol_config()
634            .get_consensus_commit_rate_estimation_window_size();
635        Self {
636            epoch_store,
637            last_consensus_stats,
638            checkpoint_service,
639            cache_reader,
640            low_scoring_authorities,
641            committee,
642            metrics,
643            processed_cache: LruCache::new(
644                NonZeroUsize::new(randomize_cache_capacity_in_tests(PROCESSED_CACHE_CAP)).unwrap(),
645            ),
646            execution_scheduler_sender,
647            consensus_adapter,
648            throughput_calculator,
649            additional_consensus_state: AdditionalConsensusState::new(
650                commit_rate_estimate_window_size,
651            ),
652            backpressure_subscriber,
653            traffic_controller,
654        }

695    fn init_randomness<'a, 'epoch>(
696        &'a mut self,
697        epoch_store: &'epoch AuthorityPerEpochStore,
698        commit_info: &'a ConsensusCommitInfo,
699    ) -> Option<tokio::sync::MutexGuard<'epoch, RandomnessManager>> {
700        // DONE(commit-handler-rewrite): load randomness manager, get random round
701        let mut randomness_manager = epoch_store.randomness_manager.get().map(|rm| {
702            rm.try_lock()
703                .expect("should only ever be called from the commit handler thread")
704        });
705
706        let mut dkg_failed = false;
707        let randomness_round = if epoch_store.randomness_state_enabled() {
708            let randomness_manager = randomness_manager
709                .as_mut()
710                .expect("randomness manager should exist if randomness is enabled");
711            match randomness_manager.dkg_status() {
712                DkgStatus::Pending => None,
713                DkgStatus::Failed => {
714                    dkg_failed = true;
715                    None
716                }
717                DkgStatus::Successful => {
718                    // DONE(commit-handler-rewrite): do not reserve randomness if !should_accept_tx()
719                    // Generate randomness for this commit if DKG is successful and we are still
720                    // accepting certs.
721                    if self.initial_reconfig_state.should_accept_tx() {
722                        randomness_manager
723                            // TODO: make infallible
724                            .reserve_next_randomness(commit_info.timestamp, &mut self.output)
725                            .expect("epoch ended")
726                    } else {
727                        None
728                    }
729                }
730            }
731        } else {
732            None
733        };
734
735        if randomness_round.is_some() {
736            assert!(!dkg_failed); // invariant check
737        }
738
739        self.randomness_round = randomness_round;
740        self.dkg_failed = dkg_failed;
741
742        randomness_manager
743    }

769    #[instrument(level = "debug", skip_all, fields(epoch = self.epoch_store.epoch(), round = consensus_commit.leader_round()))]
770    async fn handle_consensus_commit_v2(&mut self, consensus_commit: impl ConsensusCommitAPI) {
771        let protocol_config = self.epoch_store.protocol_config();
772
773        // Assert all protocol config settings for which we don't support old behavior.
774        assert!(protocol_config.ignore_execution_time_observations_after_certs_closed());
775        assert!(protocol_config.record_time_estimate_processed());
776        assert!(protocol_config.prepend_prologue_tx_in_consensus_commit_in_checkpoints());
777        assert!(protocol_config.consensus_checkpoint_signature_key_includes_digest());
778        assert!(protocol_config.authority_capabilities_v2());
779        assert!(protocol_config.cancel_for_failed_dkg_early());
780
781        // This may block until one of two conditions happens:
782        // - Number of uncommitted transactions in the writeback cache goes below the
783        //   backpressure threshold.
784        // - The highest executed checkpoint catches up to the highest certified checkpoint.
785        self.backpressure_subscriber.await_no_backpressure().await;
786
787        let epoch = self.epoch_store.epoch();
788
789        let _scope = monitored_scope("ConsensusCommitHandler::handle_consensus_commit");
790
791        let last_committed_round = self.last_consensus_stats.index.last_committed_round;
792
793        // DONE(commit-handler-rewrite): Update last_committed_round stats
794        if let Some(consensus_tx_status_cache) = self.epoch_store.consensus_tx_status_cache.as_ref()
795        {
796            consensus_tx_status_cache
797                .update_last_committed_leader_round(last_committed_round as u32)
798                .await;
799        }
800        if let Some(tx_reject_reason_cache) = self.epoch_store.tx_reject_reason_cache.as_ref() {
801            tx_reject_reason_cache.set_last_committed_leader_round(last_committed_round as u32);
802        }
803
804        // DONE(commit-handler-rewrite): this will be unconditionally enabled in the rewrite
805        let commit_info = self.additional_consensus_state.observe_commit(
806            protocol_config,
807            self.epoch_store
808                .epoch_start_config()
809                .epoch_start_timestamp_ms(),
810            &consensus_commit,
811        );
812        assert!(commit_info.round > last_committed_round);
813
814        // DONE(commit-handler-rewrite): already not needed
815
816        // DONE(commit-handler-rewrite): gather commit metadata
817        let (timestamp, leader_author, commit_sub_dag_index) =
818            self.gather_commit_metadata(&consensus_commit);
819
820        info!(
821            %consensus_commit,
822            "Received consensus output"
823        );
824
825        // DONE(commit-handler-rewrite): update stats & ExecutionIndices
826        self.last_consensus_stats.index = ExecutionIndices {
827            last_committed_round: commit_info.round,
828            sub_dag_index: commit_sub_dag_index,
829            transaction_index: 0_u64,
830        };
831
832        // DONE(commit-handler-rewrite): update low scoring authorities
833        update_low_scoring_authorities(
834            self.low_scoring_authorities.clone(),
835            self.epoch_store.committee(),
836            &self.committee,
837            consensus_commit.reputation_score_sorted_desc(),
838            &self.metrics,
839            protocol_config.consensus_bad_nodes_stake_threshold(),
840        );
841
842        // DONE(commit-handler-rewrite): update metrics
843        self.metrics
844            .consensus_committed_subdags
845            .with_label_values(&[&leader_author.to_string()])
846            .inc();
847
848        let mut state = CommitHandlerState {
849            output: ConsensusCommitOutput::new(commit_info.round),
850            dkg_failed: false,
851            randomness_round: None,
852            indirect_state_observer: Some(IndirectStateObserver::new()),
853            initial_reconfig_state: self
854                .epoch_store
855                .get_reconfig_state_read_lock_guard()
856                .clone(),
857        };
858
859        // DONE(commit-handler-rewrite): update transaction status (rejected/finalized) and update metrics
860        let transactions = self.filter_consensus_txns(
861            state.initial_reconfig_state.clone(),
862            &commit_info,
863            &consensus_commit,
864        );
865        // DONE(commit-handler-rewrite): de-duplicate transactions
866        let transactions = self.deduplicate_consensus_txns(&mut state, &commit_info, transactions);
867
868        let mut randomness_manager = state.init_randomness(&self.epoch_store, &commit_info);
869
870        // DONE(commit-handler-rewrite): Split transactions into different types for processing.
871        let CommitHandlerInput {
872            user_transactions,
873            capability_notifications,
874            execution_time_observations,
875            checkpoint_signature_messages,
876            randomness_dkg_messages,
877            randomness_dkg_confirmations,
878            end_of_publish_transactions,
879            new_jwks,
880        } = self.build_commit_handler_input(transactions);
881
882        self.process_jwks(&mut state, &commit_info, new_jwks);
883        self.process_capability_notifications(capability_notifications);
884        self.process_execution_time_observations(&mut state, execution_time_observations);
885        self.process_checkpoint_signature_messages(checkpoint_signature_messages);
886
887        self.process_dkg_updates(
888            &mut state,
889            &commit_info,
890            randomness_manager.as_deref_mut(),
891            randomness_dkg_messages,
892            randomness_dkg_confirmations,
893        )
894        .await;
895
896        let mut execution_time_estimator = self
897            .epoch_store
898            .execution_time_estimator
899            .try_lock()
900            .expect("should only ever be called from the commit handler thread");
901
902        // DONE(commit-handler-rewrite): load and activate previous round's jwks
903        let authenticator_state_update_transaction =
904            self.create_authenticator_state_update(last_committed_round, &commit_info);
905
906        let (schedulables, randomness_schedulables, assigned_versions) = self.process_transactions(
907            &mut state,
908            execution_time_estimator.as_mut(),
909            &commit_info,
910            authenticator_state_update_transaction,
911            user_transactions,
912        );
913
914        let (should_accept_tx, lock, final_round) =
915            self.handle_eop(&mut state, end_of_publish_transactions);
916
917        let make_checkpoint = should_accept_tx || final_round;
918        if !make_checkpoint {
919            // No need for any further processing
920            // DONE(commit-handler-rewrite): do not insert commit prologue if !should_accept_tx()
921            // DONE(commit-handler-rewrite): commit prologue should not be added to roots after tx processing is closed
922            return;
923        }
924
925        // DONE(commit-handler-rewrite): record execution time observations for next epoch
926        // If this is the final round, record execution time observations for storage in the
927        // end-of-epoch tx.
928        if final_round {
929            self.record_end_of_epoch_execution_time_observations(execution_time_estimator);
930        }
931
932        self.create_pending_checkpoints(
933            &mut state,
934            &commit_info,
935            &schedulables,
936            &randomness_schedulables,
937            final_round,
938        );
939
940        let notifications = state.get_notifications();
941
942        // DONE(commit-handler-rewrite): record consensus_stats
943        state
944            .output
945            .record_consensus_commit_stats(self.last_consensus_stats.clone());
946
947        // DONE(commit-handler-rewrite): propogate deferral deletion to consensus output cache
948        self.record_deferral_deletion(&mut state);
949
950        // DONE(commit-handler-rewrite): send consensus output to quarantine
951        self.epoch_store
952            .consensus_quarantine
953            .write()
954            .push_consensus_output(state.output, &self.epoch_store)
955            .expect("push_consensus_output should not fail");
956
957        // DONE(commit-handler-rewrite): notify checkpoint service
958        // Only after batch is written, notify checkpoint service to start building any new
959        // pending checkpoints.
960        debug!(
961            ?commit_info.round,
962            "Notifying checkpoint service about new pending checkpoint(s)",
963        );
964        self.checkpoint_service
965            .notify_checkpoint()
966            .expect("failed to notify checkpoint service");
967
968        // DONE(commit-handler-rewrite): Once commit processing is recorded, kick off randomness generation.
969        if let Some(randomness_round) = state.randomness_round {
970            randomness_manager
971                .as_ref()
972                .expect("randomness manager should exist if randomness round is provided")
973                .generate_randomness(epoch, randomness_round);
974        }
975
976        // DONE(commit-handler-rewrite): notify waiters that consensus transactions have been processed
977        self.epoch_store.process_notifications(notifications.iter());
978
979        // DONE(commit-handler-rewrite): log end of epoch
980        // pass lock by value to ensure that it is held until this point
981        self.log_final_round(lock, final_round);
982
983        // DONE(commit-handler-rewrite): update throughput calculator
984        // update the calculated throughput
985        self.throughput_calculator
986            .add_transactions(timestamp, schedulables.len() as u64);
987
988        // DONE(commit-handler-rewrite): fail points
989        fail_point_if!("correlated-crash-after-consensus-commit-boundary", || {
990            let key = [commit_sub_dag_index, epoch];
991            if sui_simulator::random::deterministic_probability_once(&key, 0.01) {
992                sui_simulator::task::kill_current_node(None);
993            }
994        });
995
996        fail_point!("crash"); // for tests that produce random crashes
997
998        // DONE(commit-handler-rewrite): enqueue transactions
999        let mut schedulables = schedulables;
1000        schedulables.extend(randomness_schedulables);
1001        self.execution_scheduler_sender.send(
1002            schedulables,
1003            assigned_versions,
1004            SchedulingSource::NonFastPath,
1005        );
1006
1007        // DONE(commit-handler-rewrite): Check if we should send EndOfPublish after processing consensus commit
1008        self.send_end_of_publish_if_needed().await;
1009    }

1012        &self,
1013        state: &mut CommitHandlerState,
1014        end_of_publish_transactions: Vec<AuthorityName>,
1015    ) -> (bool, Option<RwLockWriteGuard<'_, ReconfigState>>, bool) {
1016        let collected_eop =
1017            self.process_end_of_publish_transactions(state, end_of_publish_transactions);
1018        if collected_eop {
1019            // DONE(commit-handler-rewrite): [ssm] after 2f+1 EOPs, transition to RejectAllCerts
1020            // DONE(commit-handler-rewrite): [ssm] check if epoch is over
1021            let (lock, final_round) = self.advance_eop_state_machine(state);
1022            // DONE(commit-handler-rewrite): check tx acceptance state
1023            (lock.should_accept_tx(), Some(lock), final_round)
1024        } else {
1025            (true, None, false)
1026        }
1027    }

1067    fn create_pending_checkpoints(
1068        &self,
1069        state: &mut CommitHandlerState,
1070        commit_info: &ConsensusCommitInfo,
1071        schedulables: &[Schedulable],
1072        randomness_schedulables: &[Schedulable],
1073        final_round: bool,
1074    ) {
1075        // DONE(commit-handler-rewrite): Create pending checkpoints if we are still accepting tx.
1076        let checkpoint_height = self
1077            .epoch_store
1078            .calculate_pending_checkpoint_height(commit_info.round);
1079
1080        // DONE(commit-handler-rewrite): [ssm] write pending randomness checkpoint if we have a new randomness round, OR dkg failed while there are pending randomness txns
1081
1082        // Determine whether to write pending checkpoint for user tx with randomness.
1083        // - If randomness is not generated for this commit, we will skip the
1084        //   checkpoint with the associated height. Therefore checkpoint heights may
1085        //   not be contiguous.
1086        // - Exception: if DKG fails, we always need to write out a PendingCheckpoint
1087        //   for randomness tx that are canceled.
1088        let should_write_random_checkpoint = state.randomness_round.is_some()
1089            || (state.dkg_failed && !randomness_schedulables.is_empty());
1090
1091        let pending_checkpoint = PendingCheckpoint {
1092            // DONE(commit-handler-rewrite): compute checkpoint roots (this should be done at the end)
1093            roots: schedulables.iter().map(|s| s.key()).collect(),
1094            details: PendingCheckpointInfo {
1095                timestamp_ms: commit_info.timestamp,
1096                last_of_epoch: final_round && !should_write_random_checkpoint,
1097                checkpoint_height,
1098            },
1099        };
1100        self.epoch_store
1101            .write_pending_checkpoint(&mut state.output, &pending_checkpoint)
1102            .expect("failed to write pending checkpoint");
1103
1104        if should_write_random_checkpoint {
1105            let pending_checkpoint = PendingCheckpoint {
1106                roots: randomness_schedulables.iter().map(|s| s.key()).collect(),
1107                details: PendingCheckpointInfo {
1108                    timestamp_ms: commit_info.timestamp,
1109                    last_of_epoch: final_round,
1110                    checkpoint_height: checkpoint_height + 1,
1111                },
1112            };
1113            self.epoch_store
1114                .write_pending_checkpoint(&mut state.output, &pending_checkpoint)
1115                .expect("failed to write pending checkpoint");
1116        }
1117    }

1119    fn process_transactions(
1120        &self,
1121        state: &mut CommitHandlerState,
1122        execution_time_estimator: Option<&mut ExecutionTimeEstimator>,
1123        commit_info: &ConsensusCommitInfo,
1124        authenticator_state_update_transaction: Option<VerifiedExecutableTransaction>,
1125        user_transactions: Vec<VerifiedExecutableTransaction>,
1126    ) -> (Vec<Schedulable>, Vec<Schedulable>, AssignedTxAndVersions) {
1127        let protocol_config = self.epoch_store.protocol_config();
1128        let epoch = self.epoch_store.epoch();
1129
1130        // Get the ordered set of all transactions to process, which includes deferred and
1131        // newly arrived transactions.
1132        let (ordered_txns, ordered_randomness_txns, previously_deferred_tx_digests) =
1133            self.merge_and_reorder_transactions(state, commit_info, user_transactions);
1134
1135        // DONE(commit-handler-rewrite): initialize congestion trackers
1136        let mut shared_object_congestion_tracker =
1137            self.init_congestion_tracker(commit_info, false, &ordered_txns);
1138        let mut shared_object_using_randomness_congestion_tracker =
1139            self.init_congestion_tracker(commit_info, true, &ordered_randomness_txns);
1140
1141        // DONE(commit-handler-rewrite): add randomness state update transaction
1142        let randomness_state_update_transaction = state
1143            .randomness_round
1144            .map(|round| Schedulable::RandomnessStateUpdate(epoch, round));
1145        debug!(
1146            "Randomness state update transaction: {:?}",
1147            randomness_state_update_transaction
1148                .as_ref()
1149                .map(|t| t.key())
1150        );
1151
1152        let mut transactions_to_schedule = Vec::with_capacity(ordered_txns.len());
1153        let mut randomness_transactions_to_schedule =
1154            Vec::with_capacity(ordered_randomness_txns.len());
1155        let mut deferred_txns = BTreeMap::new();
1156        let mut cancelled_txns = BTreeMap::new();
1157
1158        for transaction in ordered_txns {
1159            self.handle_deferral_and_cancellation(
1160                state,
1161                &mut cancelled_txns,
1162                &mut deferred_txns,
1163                &mut transactions_to_schedule,
1164                protocol_config,
1165                commit_info,
1166                transaction,
1167                &mut shared_object_congestion_tracker,
1168                &previously_deferred_tx_digests,
1169                execution_time_estimator.as_deref(),
1170            );
1171        }
1172
1173        for transaction in ordered_randomness_txns {
1174            // DONE(commit-handler-rewrite): cancel randomness-using txns if DKG failed
1175            if state.dkg_failed {
1176                debug!(
1177                    "Canceling randomness-using transaction {:?} because DKG failed",
1178                    transaction.digest(),
1179                );
1180                cancelled_txns.insert(
1181                    *transaction.digest(),
1182                    CancelConsensusCertificateReason::DkgFailed,
1183                );
1184                // DONE(commit-handler-rewrite): cancelled txns must be scheduled for execution
1185                randomness_transactions_to_schedule.push(transaction);
1186                continue;
1187            }
1188            self.handle_deferral_and_cancellation(
1189                state,
1190                &mut cancelled_txns,
1191                &mut deferred_txns,
1192                &mut randomness_transactions_to_schedule,
1193                protocol_config,
1194                commit_info,
1195                transaction,
1196                // DONE(commit-handler-rewrite): use correct congestion tracker for randomness vs non-randomness
1197                &mut shared_object_using_randomness_congestion_tracker,
1198                &previously_deferred_tx_digests,
1199                execution_time_estimator.as_deref(),
1200            );
1201        }
1202
1203        // DONE(commit-handler-rewrite): add deferred transactions to consensus output
1204        let mut total_deferred_txns = 0;
1205        {
1206            let mut deferred_transactions = self
1207                .epoch_store
1208                .consensus_output_cache
1209                .deferred_transactions_v2
1210                .lock();
1211            for (key, txns) in deferred_txns.into_iter() {
1212                total_deferred_txns += txns.len();
1213                deferred_transactions.insert(key, txns.clone());
1214                state.output.defer_transactions_v2(key, txns);
1215            }
1216        }
1217
1218        // DONE(commit-handler-rewrite): update metrics
1219        self.metrics
1220            .consensus_handler_deferred_transactions
1221            .inc_by(total_deferred_txns as u64);
1222        self.metrics
1223            .consensus_handler_cancelled_transactions
1224            .inc_by(cancelled_txns.len() as u64);
1225        self.metrics
1226            .consensus_handler_max_object_costs
1227            .with_label_values(&["regular_commit"])
1228            .set(shared_object_congestion_tracker.max_cost() as i64);
1229        self.metrics
1230            .consensus_handler_max_object_costs
1231            .with_label_values(&["randomness_commit"])
1232            .set(shared_object_using_randomness_congestion_tracker.max_cost() as i64);
1233
1234        // DONE(commit-handler-rewrite): gather object debts, send them to ExecutionTimeObserver
1235        let object_debts = shared_object_congestion_tracker.accumulated_debts(commit_info);
1236        let randomness_object_debts =
1237            shared_object_using_randomness_congestion_tracker.accumulated_debts(commit_info);
1238        if let Some(tx_object_debts) = self.epoch_store.tx_object_debts.get()
1239            && let Err(e) = tx_object_debts.try_send(
1240                object_debts
1241                    .iter()
1242                    .chain(randomness_object_debts.iter())
1243                    .map(|(id, _)| *id)
1244                    .collect(),
1245            )
1246        {
1247            info!("failed to send updated object debts to ExecutionTimeObserver: {e:?}");
1248        }
1249
1250        // DONE(commit-handler-rewrite): commit object debts to output
1251        state
1252            .output
1253            .set_congestion_control_object_debts(object_debts);
1254        state
1255            .output
1256            .set_congestion_control_randomness_object_debts(randomness_object_debts);
1257
1258        // DONE(commit-handler-rewrite): create accumulator settlement transactions
1259        let mut settlement = None;
1260        let mut randomness_settlement = None;
1261        if self.epoch_store.accumulators_enabled() {
1262            let checkpoint_height = self
1263                .epoch_store
1264                .calculate_pending_checkpoint_height(commit_info.round);
1265
1266            settlement = Some(Schedulable::AccumulatorSettlement(epoch, checkpoint_height));
1267
1268            if state.randomness_round.is_some() {
1269                randomness_settlement = Some(Schedulable::AccumulatorSettlement(
1270                    epoch,
1271                    checkpoint_height + 1,
1272                ));
1273            }
1274        }
1275
1276        // DONE(commit-handler-rewrite): add the consensus commit prologue transaction and root
1277        let consensus_commit_prologue = self.add_consensus_commit_prologue_transaction(
1278            state,
1279            commit_info,
1280            transactions_to_schedule
1281                .iter()
1282                .map(Schedulable::Transaction),
1283            &cancelled_txns,
1284        );
1285
1286        let schedulables: Vec<_> = itertools::chain!(
1287            consensus_commit_prologue.into_iter(),
1288            authenticator_state_update_transaction.into_iter(),
1289            transactions_to_schedule.into_iter(),
1290        )
1291        .map(Schedulable::Transaction)
1292        .chain(settlement)
1293        .collect();
1294
1295        let randomness_schedulables: Vec<_> = randomness_state_update_transaction
1296            .into_iter()
1297            .chain(
1298                randomness_transactions_to_schedule
1299                    .into_iter()
1300                    .map(Schedulable::Transaction),
1301            )
1302            .chain(randomness_settlement)
1303            .collect();
1304
1305        // DONE(commit-handler-rewrite): assign shared object versions
1306        let assigned_versions = self
1307            .epoch_store
1308            .process_consensus_transaction_shared_object_versions(
1309                self.cache_reader.as_ref(),
1310                schedulables.iter(),
1311                randomness_schedulables.iter(),
1312                &cancelled_txns,
1313                &mut state.output,
1314            )
1315            .expect("failed to assign shared object versions");
1316
1317        // DONE(commit-handler-rewrite): store all user signatures for use by checkpoint builder
1318        self.epoch_store
1319            .process_user_signatures(schedulables.iter().chain(randomness_schedulables.iter()));
1320
1321        (schedulables, randomness_schedulables, assigned_versions)
1322    }

1324    // Adds the consensus commit prologue transaction to the beginning of input `transactions` to update
1325    // the system clock used in all transactions in the current consensus commit.
1326    // Returns the root of the consensus commit prologue transaction if it was added to the input.
1327    fn add_consensus_commit_prologue_transaction<'a>(
1328        &'a self,
1329        state: &'a mut CommitHandlerState,
1330        commit_info: &'a ConsensusCommitInfo,
1331        schedulables: impl Iterator<Item = Schedulable<&'a VerifiedExecutableTransaction>>,
1332        cancelled_txns: &'a BTreeMap<TransactionDigest, CancelConsensusCertificateReason>,
1333    ) -> Option<VerifiedExecutableTransaction> {
1334        {
1335            if commit_info.skip_consensus_commit_prologue_in_test {
1336                return None;
1337            }
1338        }
1339
1340        let mut version_assignment = Vec::new();
1341        let mut shared_input_next_version = HashMap::new();
1342        for txn in schedulables {
1343            let key = txn.key();
1344            match key.as_digest().and_then(|d| cancelled_txns.get(d)) {
1345                Some(CancelConsensusCertificateReason::CongestionOnObjects(_))
1346                | Some(CancelConsensusCertificateReason::DkgFailed) => {
1347                    assert_reachable!("cancelled transactions");
1348                    let assigned_versions = SharedObjVerManager::assign_versions_for_certificate(
1349                        &self.epoch_store,
1350                        &txn,
1351                        &mut shared_input_next_version,
1352                        cancelled_txns,
1353                    );
1354                    version_assignment.push((
1355                        *key.unwrap_digest(),
1356                        assigned_versions.shared_object_versions,
1357                    ));
1358                }
1359                None => {}
1360            }
1361        }
1362
1363        fail_point_arg!(
1364            "additional_cancelled_txns_for_tests",
1365            |additional_cancelled_txns: Vec<(
1366                TransactionDigest,
1367                Vec<(ConsensusObjectSequenceKey, SequenceNumber)>
1368            )>| {
1369                version_assignment.extend(additional_cancelled_txns);
1370            }
1371        );
1372
1373        let transaction = commit_info.create_consensus_commit_prologue_transaction(
1374            self.epoch_store.epoch(),
1375            self.epoch_store.protocol_config(),
1376            version_assignment,
1377            commit_info,
1378            state.indirect_state_observer.take().unwrap(),
1379        );
1380        Some(transaction)
1381    }

1383    fn handle_deferral_and_cancellation(
1384        &self,
1385        state: &mut CommitHandlerState,
1386        cancelled_txns: &mut BTreeMap<TransactionDigest, CancelConsensusCertificateReason>,
1387        deferred_txns: &mut BTreeMap<DeferralKey, Vec<VerifiedExecutableTransaction>>,
1388        scheduled_txns: &mut Vec<VerifiedExecutableTransaction>,
1389        protocol_config: &ProtocolConfig,
1390        commit_info: &ConsensusCommitInfo,
1391        transaction: VerifiedExecutableTransaction,
1392        shared_object_congestion_tracker: &mut SharedObjectCongestionTracker,
1393        previously_deferred_tx_digests: &HashMap<TransactionDigest, DeferralKey>,
1394        execution_time_estimator: Option<&ExecutionTimeEstimator>,
1395    ) {
1396        // DONE(commit-handler-rewrite): check if transaction should be deferred
1397        let tx_cost = shared_object_congestion_tracker.get_tx_cost(
1398            execution_time_estimator,
1399            &transaction,
1400            state.indirect_state_observer.as_mut().unwrap(),
1401        );
1402
1403        let deferral_info = self.epoch_store.should_defer(
1404            tx_cost,
1405            &transaction,
1406            commit_info,
1407            state.dkg_failed,
1408            state.randomness_round.is_some(),
1409            previously_deferred_tx_digests,
1410            shared_object_congestion_tracker,
1411        );
1412
1413        if let Some((deferral_key, deferral_reason)) = deferral_info {
1414            debug!(
1415                "Deferring consensus certificate for transaction {:?} until {:?}",
1416                transaction.digest(),
1417                deferral_key
1418            );
1419
1420            match deferral_reason {
1421                // DONE(commit-handler-rewrite): Always defer transaction due to randomness not ready.
1422                DeferralReason::RandomnessNotReady => {
1423                    deferred_txns
1424                        .entry(deferral_key)
1425                        .or_default()
1426                        .push(transaction);
1427                }
1428                DeferralReason::SharedObjectCongestion(congested_objects) => {
1429                    self.metrics.consensus_handler_congested_transactions.inc();
1430                    // DONE(commit-handler-rewrite): when deferral limit is exceeded, cancel the transaction
1431                    if transaction_deferral_within_limit(
1432                        &deferral_key,
1433                        protocol_config.max_deferral_rounds_for_congestion_control(),
1434                    ) {
1435                        deferred_txns
1436                            .entry(deferral_key)
1437                            .or_default()
1438                            .push(transaction);
1439                    } else {
1440                        assert_sometimes!(
1441                            transaction.transaction_data().uses_randomness(),
1442                            "cancelled randomness-using transaction"
1443                        );
1444                        assert_sometimes!(
1445                            !transaction.transaction_data().uses_randomness(),
1446                            "cancelled non-randomness-using transaction"
1447                        );
1448
1449                        // Cancel the transaction that has been deferred for too long.
1450                        debug!(
1451                            "Cancelling consensus transaction {:?} with deferral key {:?} due to congestion on objects {:?}",
1452                            transaction.digest(),
1453                            deferral_key,
1454                            congested_objects
1455                        );
1456                        cancelled_txns.insert(
1457                            *transaction.digest(),
1458                            CancelConsensusCertificateReason::CongestionOnObjects(
1459                                congested_objects,
1460                            ),
1461                        );
1462                        // DONE(commit-handler-rewrite): cancelled txns must be scheduled for execution
1463                        scheduled_txns.push(transaction);
1464                    }
1465                }
1466            }
1467        } else {
1468            // DONE(commit-handler-rewrite): update object execution cost for all scheduled transactions
1469            // This certificate will be scheduled. Update object execution cost.
1470            shared_object_congestion_tracker.bump_object_execution_cost(tx_cost, &transaction);
1471            scheduled_txns.push(transaction);
1472        }
1473    }

1475    fn merge_and_reorder_transactions(
1476        &self,
1477        state: &mut CommitHandlerState,
1478        commit_info: &ConsensusCommitInfo,
1479        user_transactions: Vec<VerifiedExecutableTransaction>,
1480    ) -> (
1481        Vec<VerifiedExecutableTransaction>,
1482        Vec<VerifiedExecutableTransaction>,
1483        HashMap<TransactionDigest, DeferralKey>,
1484    ) {
1485        let protocol_config = self.epoch_store.protocol_config();
1486
1487        // DONE(commit-handler-rewrite): deferred transactions have higher priority than new txns (if gas price is equal)
1488        let (mut txns, mut randomness_txns, previously_deferred_tx_digests) =
1489            self.load_deferred_transactions(state, commit_info);
1490
1491        txns.reserve(user_transactions.len());
1492        randomness_txns.reserve(user_transactions.len());
1493
1494        // There may be randomness transactions in `txns`, which were deferred due to congestion.
1495        // They must be placed back into `randomness_txns`.
1496        let mut txns: Vec<_> = txns
1497            .into_iter()
1498            .filter_map(|tx| {
1499                if tx.transaction_data().uses_randomness() {
1500                    randomness_txns.push(tx);
1501                    None
1502                } else {
1503                    Some(tx)
1504                }
1505            })
1506            .collect();
1507
1508        for txn in user_transactions {
1509            if txn.transaction_data().uses_randomness() {
1510                randomness_txns.push(txn);
1511            } else {
1512                txns.push(txn);
1513            }
1514        }
1515
1516        // DONE(commit-handler-rewrite): reorder transactions by gas price
1517        PostConsensusTxReorder::reorder_v2(
1518            &mut txns,
1519            protocol_config.consensus_transaction_ordering(),
1520        );
1521        PostConsensusTxReorder::reorder_v2(
1522            &mut randomness_txns,
1523            protocol_config.consensus_transaction_ordering(),
1524        );
1525
1526        (txns, randomness_txns, previously_deferred_tx_digests)
1527    }

1801    /// Returns true if we have collected a quorum of end of publish messages (either in this round or a previous round).
1802    fn process_end_of_publish_transactions(

1804        state: &mut CommitHandlerState,
1805        end_of_publish_transactions: Vec<AuthorityName>,
1806    ) -> bool {
1807        let mut eop_aggregator = self.epoch_store.end_of_publish.try_lock().expect(
1808            "No contention on end_of_publish as it is only accessed from consensus handler",
1809        );

1812            return true;
1813        }
1814
1815        if end_of_publish_transactions.is_empty() {
1816            return false;
1817        }
1818
1819        for authority in end_of_publish_transactions {
1820            info!("Received EndOfPublish from {:?}", authority.concise());

1827                .is_quorum_reached()

1830                    "Collected enough end_of_publish messages with last message from validator {:?}",

1840    /// After we have collected 2f+1 EndOfPublish messages, we call this function every round until the epoch
1841    /// ends.
1842    fn advance_eop_state_machine(
1843        &self,
1844        state: &mut CommitHandlerState,
1845    ) -> (
1846        RwLockWriteGuard<'_, ReconfigState>,
1847        bool, // true if final round
1848    ) {
1849        let mut reconfig_state = self.epoch_store.get_reconfig_state_write_lock_guard();
1850        let start_state_is_reject_all_tx = reconfig_state.is_reject_all_tx();
1851
1852        reconfig_state.close_all_certs();
1853
1854        let commit_has_deferred_txns = state.output.has_deferred_transactions();
1855        let previous_commits_have_deferred_txns =
1856            !self.epoch_store.deferred_transactions_empty_v2();
1857
1858        // DONE(commit-handler-rewrite): [ssm] if we are rejecting all certs, AND there are no deferred transactions to process, transition to RejectAllTx
1859        if !commit_has_deferred_txns && !previous_commits_have_deferred_txns {
1860            if !start_state_is_reject_all_tx {
1861                info!("Transitioning to RejectAllTx");
1862            }
1863            reconfig_state.close_all_tx();
1864        } else {
1865            debug!(
1866                "Blocking end of epoch on deferred transactions, from previous commits?={}, from this commit?={}",
1867                previous_commits_have_deferred_txns, commit_has_deferred_txns,
1868            );
1869        }
1870
1871        state.output.store_reconfig_state(reconfig_state.clone());
1872
1873        // DONE(commit-handler-rewrite): [ssm] only return final_round=true on the first round where we transition to RejectAllTx
1874        if !start_state_is_reject_all_tx && reconfig_state.is_reject_all_tx() {
1875            (reconfig_state, true)
1876        } else {
1877            (reconfig_state, false)
1878        }
1879    }

2478                if !initial_reconfig_state.should_accept_consensus_certs() {
2479                    // DONE(commit-handler-rewrite): ignore txns due to !should_accept_consensus_certs(), unless they were previously deferred
2480                    // (Note: we no longer need to worry about the previously deferred condition, since we are only
2481                    // processing newly-received transactions at this time).
2482                    match &parsed.transaction.kind {
2483                        ConsensusTransactionKind::UserTransaction(_)
2484                        | ConsensusTransactionKind::CertifiedTransaction(_)
2485                        // deprecated and ignore later, but added for exhaustive match
2486                        | ConsensusTransactionKind::CapabilityNotification(_)
2487                        // DONE(commit-handler-rewrite): [ssm] ignore capability notifications if !should_accept_consensus_certs()
2488                        | ConsensusTransactionKind::CapabilityNotificationV2(_)
2489                        // DONE(commit-handler-rewrite): [ssm] ignore end of publish messages if !should_accept_consensus_certs()
2490                        | ConsensusTransactionKind::EndOfPublish(_)
2491                        // DONE(commit-handler-rewrite): [ssm] ignore execution time observations if !should_accept_consensus_certs()
2492                        // Note: we no longer have to check protocol_config.ignore_execution_time_observations_after_certs_closed()
2493                        | ConsensusTransactionKind::ExecutionTimeObservation(_)
2494                        // DONE(commit-handler-rewrite): [ssm] ignore jwk votes if !should_accept_consensus_certs()
2495                        | ConsensusTransactionKind::NewJWKFetched(_, _, _) => {
2496                            debug!(
2497                                "Ignoring consensus transaction {:?} because of end of epoch",
2498                                parsed.transaction.key()
2499                            );
2500                            continue;
2501                        }
2502
2503                        // These are the message types that are still processed even if !should_accept_consensus_certs()
2504                        ConsensusTransactionKind::CheckpointSignature(_)
2505                        | ConsensusTransactionKind::CheckpointSignatureV2(_)
2506                        | ConsensusTransactionKind::RandomnessStateUpdate(_, _)
2507                        | ConsensusTransactionKind::RandomnessDkgMessage(_, _)
2508                        | ConsensusTransactionKind::RandomnessDkgConfirmation(_, _) => ()
2509                    }
2510                }
2511
2512                if !initial_reconfig_state.should_accept_tx() {
2513                    match &parsed.transaction.kind {
2514                        // DONE(commit-handler-rewrite): [ssm] ignore dkg confirmation if !should_accept_tx()
2515                        ConsensusTransactionKind::RandomnessDkgConfirmation(_, _)
2516                        // DONE(commit-handler-rewrite): [ssm] ignore dkg message if !should_accept_tx()
2517                        | ConsensusTransactionKind::RandomnessDkgMessage(_, _) => continue,
2518                        _ => {},
2519                    }
2520                }
2521
2522                // DONE(commit-handler-rewrite): ignore mfp user transaction if mfp is disabled
2523                if parsed.transaction.is_mfp_transaction()
2524                    && !self.epoch_store.protocol_config().mysticeti_fastpath()
2525                {
2526                    debug!(
2527                        "Ignoring MFP transaction {:?} because MFP is disabled",
2528                        parsed.transaction.key()
2529                    );
2530                    continue;
2531                }
2532
2533                // DONE(commit-handler-rewrite): ignore certs from wrong epoch
2534                if let ConsensusTransactionKind::CertifiedTransaction(certificate) =
2535                    &parsed.transaction.kind
2536                    && certificate.epoch() != epoch
2537                {
2538                    debug!(
2539                        "Certificate epoch ({:?}) doesn't match the current epoch ({:?})",
2540                        certificate.epoch(),
2541                        epoch
2542                    );
2543                    continue;
2544                }
2545
2546                // Handle deprecated messages
2547                match &parsed.transaction.kind {
2548                    // DONE(commit-handler-rewrite): [ssm] can ignore capabilities v1 in rewrite
2549                    ConsensusTransactionKind::CapabilityNotification(_)
2550                    | ConsensusTransactionKind::RandomnessStateUpdate(_, _)
2551                    | ConsensusTransactionKind::CheckpointSignature(_) => {
2552                        debug_fatal!(
2553                            "BUG: saw deprecated tx {:?}for commit round {}",
2554                            parsed.transaction.key(),
2555                            commit_info.round
2556                        );
2557                        continue;
2558                    }
2559                    _ => {}
2560                }
2561
2562                // DONE(commit-handler-rewrite): ignore transactions sent by validators that have already sent EOP
2563                if matches!(
2564                    &parsed.transaction.kind,
2565                    ConsensusTransactionKind::UserTransaction(_)
2566                        | ConsensusTransactionKind::CertifiedTransaction(_)
2567                ) {
2568                    let author_name = self
2569                        .epoch_store
2570                        .committee()
2571                        .authority_by_index(author as u32)
2572                        .unwrap();

2574                        .epoch_store
2575                        .has_received_end_of_publish_from(author_name)
2576                    {
2577                        // In some edge cases, consensus might resend previously seen certificate after EndOfPublish
2578                        // An honest validator should not send a new transaction after EndOfPublish. Whether the
2579                        // transaction is duplicate or not, we filter it out here.
2580                        warn!(
2581                            "Ignoring consensus transaction {:?} from authority {:?}, which already sent EndOfPublish message to consensus",
2582                            author_name.concise(),
2583                            parsed.transaction.key(),
2584                        );
2585                        continue;
2586                    }
2587                }
2588
2589                let transaction = SequencedConsensusTransactionKind::External(parsed.transaction);
2590                transactions.push((transaction, author as u32));
2591            }
2592        }
2593
2594        // TODO(commit-handler-rewrite): update per validator metrics
2595        // DONE(commit-handler-rewrite): update per validator metrics
2596        // TODO and DONE are in same place because this code is shared between old and new commit handler
2597        for (i, authority) in self.committee.authorities() {
2598            let hostname = &authority.hostname;
2599            self.metrics
2600                .consensus_committed_messages
2601                .with_label_values(&[hostname])
2602                .set(self.last_consensus_stats.stats.get_num_messages(i.value()) as i64);
2603            self.metrics
2604                .consensus_committed_user_transactions
2605                .with_label_values(&[hostname])
2606                .set(
2607                    self.last_consensus_stats
2608                        .stats
2609                        .get_num_user_transactions(i.value()) as i64,
2610                );
2611            self.metrics
2612                .consensus_finalized_user_transactions
2613                .with_label_values(&[hostname])
2614                .add(num_finalized_user_transactions[i.value()] as i64);
2615            self.metrics
2616                .consensus_rejected_user_transactions
2617                .with_label_values(&[hostname])
2618                .add(num_rejected_user_transactions[i.value()] as i64);
2619        }
2620
2621        transactions
2622    }

2624    fn deduplicate_consensus_txns(
2625        &mut self,
2626        state: &mut CommitHandlerState,
2627        commit_info: &ConsensusCommitInfo,
2628        transactions: Vec<(SequencedConsensusTransactionKind, u32)>,
2629    ) -> Vec<VerifiedSequencedConsensusTransaction> {
2630        // We need a set here as well, since the processed_cache is a LRU cache and can drop
2631        // entries while we're iterating over the sequenced transactions.
2632        let mut processed_set = HashSet::new();
2633
2634        let mut all_transactions = Vec::new();
2635
2636        // All of these TODOs are handled here in the new code, whereas in the old code, they were
2637        // each handled separately. The key thing to see is that all messages are marked as processed
2638        // here, except for ones that are filtered out earlier (e.g. due to !should_accept_consensus_certs()).
2639
2640        // DONE(commit-handler-rewrite): record_consensus_message_processed() must be called for deferred txns
2641        // DONE(commit-handler-rewrite): cancelled txns must be recorded as processed
2642        // DONE(commit-handler-rewrite): consensus messages must be recorded as processed
2643        // DONE(commit-handler-rewrite): randomness messages must be recorded as processed
2644        for (seq, (transaction, cert_origin)) in transactions.into_iter().enumerate() {
2645            // TODO(consensus-handler-rewrite): the seq + 1 is probably not necessary, because we do not create a
2646            // SequencedConsensusTransaction for commit prologue any more.
2647            // In process_consensus_transactions_and_commit_boundary(), we will add a system consensus commit
2648            // prologue transaction, which will be the first transaction in this consensus commit batch.
2649            // Therefore, the transaction sequence number starts from 1 here.
2650            let current_tx_index = ExecutionIndices {
2651                last_committed_round: commit_info.round,
2652                sub_dag_index: commit_info.sub_dag_index,
2653                transaction_index: (seq + 1) as u64,
2654            };
2655
2656            self.last_consensus_stats.index = current_tx_index;
2657
2658            let certificate_author = *self
2659                .epoch_store
2660                .committee()
2661                .authority_by_index(cert_origin)
2662                .unwrap();
2663
2664            let sequenced_transaction = SequencedConsensusTransaction {
2665                certificate_author_index: cert_origin,
2666                certificate_author,
2667                consensus_index: current_tx_index,
2668                transaction,
2669            };
2670
2671            let Some(verified_transaction) = self
2672                .epoch_store
2673                .verify_consensus_transaction(sequenced_transaction)
2674            else {
2675                continue;
2676            };
2677
2678            let key = verified_transaction.0.key();
2679            let in_set = !processed_set.insert(key.clone());
2680            let in_cache = self.processed_cache.put(key.clone(), ()).is_some();
2681
2682            if in_set || in_cache {
2683                self.metrics.skipped_consensus_txns_cache_hit.inc();
2684                continue;
2685            }
2686            if self
2687                .epoch_store
2688                .is_consensus_message_processed(&key)
2689                .expect("db error")
2690            {
2691                self.metrics.skipped_consensus_txns.inc();
2692                continue;
2693            }
2694
2695            state.output.record_consensus_message_processed(key);
2696
2697            all_transactions.push(verified_transaction);
2698        }
2699
2700        all_transactions
2701    }

2703    fn build_commit_handler_input(
2704        &self,
2705        transactions: Vec<VerifiedSequencedConsensusTransaction>,
2706    ) -> CommitHandlerInput {
2707        let epoch = self.epoch_store.epoch();
2708        let mut commit_handler_input = CommitHandlerInput::default();
2709
2710        for VerifiedSequencedConsensusTransaction(transaction) in transactions.into_iter() {
2711            match transaction.transaction {
2712                SequencedConsensusTransactionKind::External(consensus_transaction) => {
2713                    match consensus_transaction.kind {
2714                        // === User transactions ===
2715                        ConsensusTransactionKind::CertifiedTransaction(cert) => {
2716                            // Safe because signatures are verified when consensus called into SuiTxValidator::validate_batch.
2717                            let cert = VerifiedCertificate::new_unchecked(*cert);
2718                            let transaction =
2719                                VerifiedExecutableTransaction::new_from_certificate(cert);
2720                            commit_handler_input.user_transactions.push(transaction);
2721                        }
2722                        ConsensusTransactionKind::UserTransaction(tx) => {
2723                            // Safe because transactions are certified by consensus.
2724                            let tx = VerifiedTransaction::new_unchecked(*tx);
2725                            // TODO(fastpath): accept position in consensus, after plumbing consensus round, authority index, and transaction index here.
2726                            let transaction =
2727                                VerifiedExecutableTransaction::new_from_consensus(tx, epoch);
2728                            commit_handler_input.user_transactions.push(transaction);
2729                        }
2730
2731                        // === State machines ===
2732                        ConsensusTransactionKind::EndOfPublish(authority_public_key_bytes) => {
2733                            commit_handler_input
2734                                .end_of_publish_transactions
2735                                .push(authority_public_key_bytes);
2736                        }
2737                        ConsensusTransactionKind::NewJWKFetched(
2738                            authority_public_key_bytes,
2739                            jwk_id,
2740                            jwk,
2741                        ) => {
2742                            commit_handler_input.new_jwks.push((
2743                                authority_public_key_bytes,
2744                                jwk_id,
2745                                jwk,
2746                            ));
2747                        }
2748                        ConsensusTransactionKind::RandomnessDkgMessage(
2749                            authority_public_key_bytes,
2750                            items,
2751                        ) => {
2752                            commit_handler_input
2753                                .randomness_dkg_messages
2754                                .push((authority_public_key_bytes, items));
2755                        }
2756                        ConsensusTransactionKind::RandomnessDkgConfirmation(
2757                            authority_public_key_bytes,
2758                            items,
2759                        ) => {
2760                            commit_handler_input
2761                                .randomness_dkg_confirmations
2762                                .push((authority_public_key_bytes, items));
2763                        }
2764                        ConsensusTransactionKind::CapabilityNotificationV2(
2765                            authority_capabilities_v2,
2766                        ) => {
2767                            commit_handler_input
2768                                .capability_notifications
2769                                .push(authority_capabilities_v2);
2770                        }
2771                        ConsensusTransactionKind::ExecutionTimeObservation(
2772                            execution_time_observation,
2773                        ) => {
2774                            commit_handler_input
2775                                .execution_time_observations
2776                                .push(execution_time_observation);
2777                        }
2778                        ConsensusTransactionKind::CheckpointSignatureV2(
2779                            checkpoint_signature_message,
2780                        ) => {
2781                            commit_handler_input
2782                                .checkpoint_signature_messages
2783                                .push(*checkpoint_signature_message);
2784                        }
2785
2786                        // Deprecated messages, filtered earlier by filter_consensus_txns()
2787                        ConsensusTransactionKind::CheckpointSignature(_)
2788                        | ConsensusTransactionKind::RandomnessStateUpdate(_, _)
2789                        | ConsensusTransactionKind::CapabilityNotification(_) => {
2790                            unreachable!("filtered earlier")
2791                        }
2792                    }
2793                }
2794                // TODO: I think we can delete this, it was only used to inject randomness state update into the tx stream.
2795                SequencedConsensusTransactionKind::System(_verified_envelope) => unreachable!(),
2796            }
2797        }
2798
2799        commit_handler_input
2800    }

2838    ) -> Self {

2856        mut recv: monitored_mpsc::UnboundedReceiver<(
2857            Vec<Schedulable>,
2858            AssignedTxAndVersions,
2859            SchedulingSource,
2860        )>,
2861        execution_scheduler: Arc<ExecutionScheduler>,
2862        epoch_store: Arc<AuthorityPerEpochStore>,
2863    ) {

2870                    let key = txn.key();
2871                    (
2872                        txn,
2873                        ExecutionEnv::new()
2874                            .with_scheduling_source(scheduling_source)
2875                            .with_assigned_versions(
2876                                assigned_versions.get(&key).cloned().unwrap_or_default(),
2877                            ),
2878                    )
2879                })
2880                .collect();
2881            execution_scheduler.enqueue(txns, &epoch_store);
2882        }
2883    }

2886/// Manages the lifetime of tasks handling the commits and transactions output by consensus.
2887pub(crate) struct MysticetiConsensusHandler {
2888    tasks: JoinSet<()>,
2889}
2890
2891impl MysticetiConsensusHandler {
2892    pub(crate) fn new(
2893        last_processed_commit_at_startup: CommitIndex,
2894        mut consensus_handler: ConsensusHandler<CheckpointService>,
2895        consensus_block_handler: ConsensusBlockHandler,
2896        mut commit_receiver: UnboundedReceiver<consensus_core::CommittedSubDag>,

2899    ) -> Self {

2909                        .handle_consensus_commit(consensus_commit)

2916            tasks.spawn(monitored_future!(async move {
2917                while let Some(blocks) = block_receiver.recv().await {
2918                    consensus_block_handler
2919                        .handle_certified_blocks(blocks)

2921                }
2922            }));
