15pub(crate) struct ParsedTransaction {
16    // Transaction from consensus output.
17    pub(crate) transaction: ConsensusTransaction,
18    // Whether the transaction was rejected in voting.
19    pub(crate) rejected: bool,
20    // Bytes length of the serialized transaction
21    pub(crate) serialized_len: usize,
22}
23
24pub(crate) trait ConsensusCommitAPI: Display {
25    /// Returns the ref of consensus output.
26    fn commit_ref(&self) -> CommitRef;
27
28    fn reputation_score_sorted_desc(&self) -> Option<Vec<(AuthorityIndex, u64)>>;
29    fn leader_round(&self) -> u64;
30    fn leader_author_index(&self) -> AuthorityIndex;
31
32    /// Returns epoch UNIX timestamp in milliseconds
33    fn commit_timestamp_ms(&self) -> u64;
34
35    /// Returns a unique global index for each committed sub-dag.
36    fn commit_sub_dag_index(&self) -> u64;
37
38    /// Returns all accepted and rejected transactions per block in the commit in deterministic order.
39    fn transactions(&self) -> Vec<(BlockRef, Vec<ParsedTransaction>)>;
40
41    /// Returns a debug string of all rejected transactions.
42    fn rejected_transactions_digest(&self) -> Digest;
43    fn rejected_transactions_debug_string(&self) -> String;
44}
45
46impl ConsensusCommitAPI for consensus_core::CommittedSubDag {
47    fn commit_ref(&self) -> CommitRef {
48        self.commit_ref
49    }
50
51    fn reputation_score_sorted_desc(&self) -> Option<Vec<(AuthorityIndex, u64)>> {
52        if !self.reputation_scores_desc.is_empty() {
53            Some(
54                self.reputation_scores_desc
55                    .iter()
56                    .map(|(id, score)| (id.value() as AuthorityIndex, *score))
57                    .collect(),
58            )
59        } else {
60            None
61        }
62    }
63
64    fn leader_round(&self) -> u64 {
65        self.leader.round as u64
66    }
67
68    fn leader_author_index(&self) -> AuthorityIndex {
69        self.leader.author.value() as AuthorityIndex
70    }
71
72    fn commit_timestamp_ms(&self) -> u64 {
73        // TODO: Enforce ordered timestamp in Mysticeti.
74        self.timestamp_ms
75    }
76
77    fn commit_sub_dag_index(&self) -> u64 {
78        self.commit_ref.index.into()
79    }
80
81    fn transactions(&self) -> Vec<(BlockRef, Vec<ParsedTransaction>)> {
82        let no_transaction = vec![];
83        self.blocks
84            .iter()
85            .map(|block| {
86                let rejected_transactions = self
87                    .rejected_transactions_by_block
88                    .get(&block.reference())
89                    .unwrap_or(&no_transaction);
90                (
91                    block.reference(),
92                    parse_block_transactions(
93                        block,
94                        rejected_transactions,
95                        self.always_accept_system_transactions,
96                    ),
97                )
98            })
99            .collect()
100    }
101
102    fn rejected_transactions_digest(&self) -> Digest {
103        let bytes = bcs::to_bytes(&self.rejected_transactions_by_block).unwrap();
104        let mut hasher = sui_types::crypto::DefaultHash::new();
105        hasher.update(bytes);
106        hasher.finalize().digest.into()
107    }
108
109    fn rejected_transactions_debug_string(&self) -> String {
110        let str = self
111            .rejected_transactions_by_block
112            .iter()
113            .map(|(block_ref, rejected_transactions)| {
114                format!(
115                    "{block_ref}: [{}]",
116                    rejected_transactions
117                        .iter()
118                        .map(|tx| tx.to_string())
119                        .join(",")
120                )
121            })
122            .join(", ");
123        let digest = self.rejected_transactions_digest();
124        format!("({digest}): [{str}]")
125    }
126}

128pub(crate) fn parse_block_transactions(
129    block: &VerifiedBlock,
130    rejected_transactions: &[TransactionIndex],
131    always_accept_system_transactions: bool,
132) -> Vec<ParsedTransaction> {
133    let round = block.round();
134    let authority = block.author().value() as AuthorityIndex;
135
136    // rejected_transactions contains sorted indices and can be checked more efficiently.
137    // But for simplicity, check rejection status from a BTreeSet.
138    let rejected_transaction_indices = BTreeSet::from_iter(rejected_transactions.iter().cloned());
139    block
140        .transactions()
141        .iter().enumerate()
142        .map(|(index, tx)| {
143            let transaction = match bcs::from_bytes::<ConsensusTransaction>(tx.data()) {
144                Ok(transaction) => transaction,
145                Err(err) => {
146                    panic!("Failed to deserialize sequenced consensus transaction(this should not happen) {err} from {authority} at {round}");
147                },
148            };
149            let rejected = rejected_transaction_indices.contains(&(index as TransactionIndex)) && (transaction.is_user_transaction() || !always_accept_system_transactions);
150            ParsedTransaction {
151                transaction,
152                rejected,
153                serialized_len: tx.data().len(),
154            }
155        })
156        .collect()
157}
