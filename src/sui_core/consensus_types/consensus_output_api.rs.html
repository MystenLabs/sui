13pub(crate) struct ParsedTransaction {
14    // Transaction from consensus output.
15    pub(crate) transaction: ConsensusTransaction,
16    // Whether the transaction was rejected in voting.
17    pub(crate) rejected: bool,
18    // Bytes length of the serialized transaction
19    pub(crate) serialized_len: usize,
20}
21
22pub(crate) trait ConsensusCommitAPI: Display {
23    fn reputation_score_sorted_desc(&self) -> Option<Vec<(AuthorityIndex, u64)>>;
24    fn leader_round(&self) -> u64;
25    fn leader_author_index(&self) -> AuthorityIndex;
26
27    /// Returns epoch UNIX timestamp in milliseconds
28    fn commit_timestamp_ms(&self) -> u64;
29
30    /// Returns a unique global index for each committed sub-dag.
31    fn commit_sub_dag_index(&self) -> u64;
32
33    /// Returns all accepted and rejected transactions per block in the commit in deterministic order.
34    fn transactions(&self) -> Vec<(BlockRef, Vec<ParsedTransaction>)>;
35
36    /// Returns the digest of consensus output.
37    fn consensus_digest(&self, protocol_config: &ProtocolConfig) -> ConsensusCommitDigest;
38}
39
40impl ConsensusCommitAPI for consensus_core::CommittedSubDag {
41    fn reputation_score_sorted_desc(&self) -> Option<Vec<(AuthorityIndex, u64)>> {
42        if !self.reputation_scores_desc.is_empty() {
43            Some(
44                self.reputation_scores_desc
45                    .iter()
46                    .map(|(id, score)| (id.value() as AuthorityIndex, *score))
47                    .collect(),
48            )
49        } else {
50            None
51        }
52    }
53
54    fn leader_round(&self) -> u64 {
55        self.leader.round as u64
56    }
57
58    fn leader_author_index(&self) -> AuthorityIndex {
59        self.leader.author.value() as AuthorityIndex
60    }
61
62    fn commit_timestamp_ms(&self) -> u64 {
63        // TODO: Enforce ordered timestamp in Mysticeti.
64        self.timestamp_ms
65    }
66
67    fn commit_sub_dag_index(&self) -> u64 {
68        self.commit_ref.index.into()
69    }
70
71    fn transactions(&self) -> Vec<(BlockRef, Vec<ParsedTransaction>)> {
72        let no_transaction = vec![];
73        self.blocks
74            .iter()
75            .map(|block| {
76                let rejected_transactions = self
77                    .rejected_transactions_by_block
78                    .get(&block.reference())
79                    .unwrap_or(&no_transaction);
80                (
81                    block.reference(),
82                    parse_block_transactions(block, rejected_transactions),
83                )
84            })
85            .collect()
86    }
87
88    fn consensus_digest(&self, protocol_config: &ProtocolConfig) -> ConsensusCommitDigest {
89        if protocol_config.mysticeti_use_committed_subdag_digest() {
90            // We port CommitDigest, a consensus space object, into ConsensusCommitDigest, a sui-core space object.
91            // We assume they always have the same format.
92            static_assertions::assert_eq_size!(ConsensusCommitDigest, CommitDigest);
93            ConsensusCommitDigest::new(self.commit_ref.digest.into_inner())
94        } else {
95            ConsensusCommitDigest::default()
96        }
97    }
98}

100pub(crate) fn parse_block_transactions(
101    block: &VerifiedBlock,
102    rejected_transactions: &[TransactionIndex],
103) -> Vec<ParsedTransaction> {
104    let round = block.round();
105    let authority = block.author().value() as AuthorityIndex;
106
107    let mut rejected_idx = 0;
108    block
109        .transactions()
110        .iter().enumerate()
111        .map(|(index, tx)| {
112            let transaction = match bcs::from_bytes::<ConsensusTransaction>(tx.data()) {
113                Ok(transaction) => transaction,
114                Err(err) => {
115                    panic!("Failed to deserialize sequenced consensus transaction(this should not happen) {err} from {authority} at {round}");
116                },
117            };
118            let rejected = if rejected_idx < rejected_transactions.len() {
119                match (index as TransactionIndex).cmp(&rejected_transactions[rejected_idx]) {
120                    Ordering::Less => {
121                        false
122                    },
123                    Ordering::Equal => {
124                        rejected_idx += 1;
125                        true
126                    },
127                    Ordering::Greater => {
128                        panic!("Rejected transaction indices are not in order. Block {block:?}, rejected transactions: {rejected_transactions:?}");
129                    },
130                }
131            } else {
132                false
133            };
134            ParsedTransaction {
135                transaction,
136                rejected,
137                serialized_len: tx.data().len(),
138            }
139        })
140        .collect()
141}
