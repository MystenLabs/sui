40/// Validates transactions from consensus and votes on whether to execute the transactions
41/// based on their validity and the current state of the authority.
42#[derive(Clone)]
43pub struct SuiTxValidator {
44    authority_state: Arc<AuthorityState>,
45    epoch_store: Arc<AuthorityPerEpochStore>,
46    consensus_overload_checker: Arc<dyn ConsensusOverloadChecker>,
47    checkpoint_service: Arc<dyn CheckpointServiceNotify + Send + Sync>,
48    metrics: Arc<SuiTxValidatorMetrics>,
49}
50
51impl SuiTxValidator {
52    pub fn new(
53        authority_state: Arc<AuthorityState>,
54        epoch_store: Arc<AuthorityPerEpochStore>,
55        checkpoint_service: Arc<dyn CheckpointServiceNotify + Send + Sync>,
56        metrics: Arc<SuiTxValidatorMetrics>,
57    ) -> Self {
58        info!(
59            "SuiTxValidator constructed for epoch {}",
60            epoch_store.epoch()
61        );
62        // Intentionally do not check consensus overload, because this is validating transactions already in consensus.
63        let consensus_overload_checker = Arc::new(NoopConsensusOverloadChecker {});
64        Self {
65            authority_state,
66            epoch_store,
67            consensus_overload_checker,
68            checkpoint_service,
69            metrics,
70        }
71    }

73    fn validate_transactions(&self, txs: &[ConsensusTransactionKind]) -> Result<(), SuiError> {
74        let epoch_store = &self.epoch_store;
75        // cert_batch is always empty since CertifiedTransaction is rejected
76        let cert_batch: Vec<&CertifiedTransaction> = Vec::new();
77        let mut ckpt_messages = Vec::new();
78        let mut ckpt_batch = Vec::new();
79        for tx in txs.iter() {
80            match tx {
81                ConsensusTransactionKind::CertifiedTransaction(_) => {
82                    return Err(SuiErrorKind::UnexpectedMessage(
83                        "CertifiedTransaction cannot be used when preconsensus locking is disabled"
84                            .to_string(),
85                    )
86                    .into());
87                }
88                ConsensusTransactionKind::CheckpointSignature(_) => {
89                    return Err(SuiErrorKind::UnexpectedMessage(
90                        "CheckpointSignature V1 is no longer supported".to_string(),
91                    )
92                    .into());
93                }
94                ConsensusTransactionKind::CheckpointSignatureV2(signature) => {
95                    ckpt_messages.push(signature.as_ref());
96                    ckpt_batch.push(&signature.summary);
97                }
98                ConsensusTransactionKind::RandomnessDkgMessage(_, bytes) => {
99                    if bytes.len() > dkg_v1::DKG_MESSAGES_MAX_SIZE {
100                        warn!("batch verification error: DKG Message too large");
101                        return Err(SuiErrorKind::InvalidDkgMessageSize.into());
102                    }
103                }
104                ConsensusTransactionKind::RandomnessDkgConfirmation(_, bytes) => {
105                    if bytes.len() > dkg_v1::DKG_MESSAGES_MAX_SIZE {
106                        warn!("batch verification error: DKG Confirmation too large");
107                        return Err(SuiErrorKind::InvalidDkgMessageSize.into());
108                    }
109                }
110
111                ConsensusTransactionKind::CapabilityNotification(_) => {
112                    return Err(SuiErrorKind::UnexpectedMessage(
113                        "CapabilityNotification V1 is no longer supported".to_string(),
114                    )
115                    .into());
116                }
117
118                ConsensusTransactionKind::RandomnessStateUpdate(_, _) => {
119                    return Err(SuiErrorKind::UnexpectedMessage(
120                        "RandomnessStateUpdate is no longer supported".to_string(),
121                    )
122                    .into());
123                }
124
125                ConsensusTransactionKind::EndOfPublish(_)
126                | ConsensusTransactionKind::NewJWKFetched(_, _, _)
127                | ConsensusTransactionKind::CapabilityNotificationV2(_) => {}
128
129                ConsensusTransactionKind::UserTransaction(_) => {
130                    return Err(SuiErrorKind::UnexpectedMessage(
131                        "ConsensusTransactionKind::UserTransaction cannot be used when address aliases is enabled or preconsensus locking is disabled".to_string(),
132                    )
133                    .into());
134                }
135
136                ConsensusTransactionKind::UserTransactionV2(tx) => {
137                    if epoch_store.protocol_config().address_aliases() {
138                        let has_aliases = if epoch_store
139                            .protocol_config()
140                            .fix_checkpoint_signature_mapping()
141                        {
142                            tx.aliases().is_some()
143                        } else {
144                            tx.aliases_v1().is_some()
145                        };
146                        if !has_aliases {
147                            return Err(SuiErrorKind::UnexpectedMessage(
148                                "ConsensusTransactionKind::UserTransactionV2 must contain an aliases claim".to_string(),
149                            )
150                            .into());
151                        }
152                    }
153
154                    if let Some(aliases) = tx.aliases() {
155                        let num_sigs = tx.tx().tx_signatures().len();
156                        for (sig_idx, _) in aliases.iter() {
157                            if (*sig_idx as usize) >= num_sigs {
158                                return Err(SuiErrorKind::UnexpectedMessage(format!(
159                                    "UserTransactionV2 alias contains out-of-bounds signature index {sig_idx} (transaction has {num_sigs} signatures)",
160                                )).into());
161                            }
162                        }
163                    }
164
165                    // TODO(fastpath): move deterministic verifications of user transactions here.
166                }
167
168                ConsensusTransactionKind::ExecutionTimeObservation(obs) => {
169                    // TODO: Use a separate limit for this that may truncate shared observations.
170                    if obs.estimates.len()
171                        > epoch_store
172                            .protocol_config()
173                            .max_programmable_tx_commands()
174                            .try_into()
175                            .unwrap()
176                    {
177                        return Err(SuiErrorKind::UnexpectedMessage(format!(
178                            "ExecutionTimeObservation contains too many estimates: {}",
179                            obs.estimates.len()
180                        ))
181                        .into());
182                    }
183                }
184            }
185        }
186
187        // verify the certificate signatures as a batch
188        let cert_count = cert_batch.len();
189        let ckpt_count = ckpt_batch.len();
190
191        epoch_store
192            .signature_verifier
193            .verify_certs_and_checkpoints(cert_batch, ckpt_batch)
194            .tap_err(|e| warn!("batch verification error: {}", e))?;
195
196        // All checkpoint sigs have been verified, forward them to the checkpoint service
197        for ckpt in ckpt_messages {
198            self.checkpoint_service.notify_checkpoint_signature(ckpt)?;
199        }
200
201        self.metrics
202            .certificate_signatures_verified
203            .inc_by(cert_count as u64);
204        self.metrics
205            .checkpoint_signatures_verified
206            .inc_by(ckpt_count as u64);
207        Ok(())
208    }

210    #[instrument(level = "debug", skip_all, fields(block_ref))]
211    fn vote_transactions(
212        &self,
213        block_ref: &BlockRef,
214        txs: Vec<ConsensusTransactionKind>,
215    ) -> Vec<TransactionIndex> {
216        let epoch_store = &self.epoch_store;
217        if !epoch_store.protocol_config().mysticeti_fastpath() {
218            return vec![];
219        }
220
221        let mut reject_txn_votes = Vec::new();
222        for (i, tx) in txs.into_iter().enumerate() {
223            let tx: PlainTransactionWithClaims = match tx {
224                ConsensusTransactionKind::UserTransaction(tx) => {
225                    TransactionWithClaims::no_aliases(*tx)
226                }
227                ConsensusTransactionKind::UserTransactionV2(tx) => *tx,
228                _ => continue,
229            };
230
231            let tx_digest = *tx.tx().digest();
232            if let Err(error) = self.vote_transaction(epoch_store, tx) {
233                debug!(?tx_digest, "Voting to reject transaction: {error}");
234                self.metrics
235                    .transaction_reject_votes
236                    .with_label_values(&[error.to_variant_name()])
237                    .inc();
238                reject_txn_votes.push(i as TransactionIndex);
239                // Cache the rejection vote reason (error) for the transaction
240                epoch_store.set_rejection_vote_reason(
241                    ConsensusPosition {
242                        epoch: epoch_store.epoch(),
243                        block: *block_ref,
244                        index: i as TransactionIndex,
245                    },
246                    &error,
247                );
248            } else {
249                debug!(?tx_digest, "Voting to accept transaction");
250            }
251        }
252
253        reject_txn_votes
254    }

256    #[instrument(level = "debug", skip_all, err(level = "debug"), fields(tx_digest = ?tx.tx().digest()))]
257    fn vote_transaction(
258        &self,
259        epoch_store: &Arc<AuthorityPerEpochStore>,
260        tx: PlainTransactionWithClaims,
261    ) -> SuiResult<()> {
262        // Extract claims before consuming the transaction
263        let aliases_v2 = tx.aliases();
264        let aliases_v1 = tx.aliases_v1();
265        let claimed_immutable_ids = tx.get_immutable_objects();
266        let inner_tx = tx.into_tx();
267
268        // Currently validity_check() and verify_transaction() are not required to be consistent across validators,
269        // so they do not run in validate_transactions(). They can run there once we confirm it is safe.
270        inner_tx.validity_check(&epoch_store.tx_validity_check_context())?;
271
272        self.authority_state.check_system_overload(
273            &*self.consensus_overload_checker,
274            inner_tx.data(),
275            self.authority_state.check_system_overload_at_signing(),
276        )?;
277
278        #[allow(unused_mut)]
279        let mut fail_point_always_report_aliases_changed = false;
280        fail_point_arg!(
281            "consensus-validator-always-report-aliases-changed",
282            |for_validators: Vec<AuthorityName>| {
283                if for_validators.contains(&self.authority_state.name) {
284                    // always report aliases changed in simtests
285                    fail_point_always_report_aliases_changed = true;
286                }
287            }
288        );
289
290        let verified_tx = epoch_store.verify_transaction_with_current_aliases(inner_tx)?;
291
292        // aliases must have data when address_aliases() is enabled.
293        if epoch_store.protocol_config().address_aliases() {
294            let aliases_match = if epoch_store
295                .protocol_config()
296                .fix_checkpoint_signature_mapping()
297            {
298                // V2 format comparison
299                let Some(claimed_v2) = aliases_v2 else {
300                    return Err(
301                        SuiErrorKind::InvalidRequest("missing address alias claim".into()).into(),
302                    );
303                };
304                *verified_tx.aliases() == claimed_v2
305            } else {
306                // V1 format comparison: derive V1 from verified_tx and compare
307                let Some(claimed_v1) = aliases_v1 else {
308                    return Err(
309                        SuiErrorKind::InvalidRequest("missing address alias claim".into()).into(),
310                    );
311                };
312                let computed_v1: Vec<_> = verified_tx
313                    .tx()
314                    .data()
315                    .intent_message()
316                    .value
317                    .required_signers()
318                    .into_iter()
319                    .zip_eq(verified_tx.aliases().iter().map(|(_, seq)| *seq))
320                    .collect();
321                let computed_v1 =
322                    NonEmpty::from_vec(computed_v1).expect("must have at least one signer");
323                computed_v1 == claimed_v1
324            };
325
326            if !aliases_match || fail_point_always_report_aliases_changed {
327                return Err(SuiErrorKind::AliasesChanged.into());
328            }
329        }
330
331        let inner_tx = verified_tx.into_tx();
332        self.authority_state
333            .handle_vote_transaction(epoch_store, inner_tx.clone())?;
334
335        if !claimed_immutable_ids.is_empty() {
336            let owned_object_refs: HashSet<ObjectRef> = inner_tx
337                .data()
338                .transaction_data()
339                .input_objects()?
340                .iter()
341                .filter_map(|obj| match obj {
342                    InputObjectKind::ImmOrOwnedMoveObject(obj_ref) => Some(*obj_ref),
343                    _ => None,
344                })
345                .collect();
346            self.verify_immutable_object_claims(&claimed_immutable_ids, owned_object_refs)?;
347        }
348
349        Ok(())
350    }
351
352    /// Verify immutable object claims are complete and accurate.
353    /// This ensures claimed_ids exactly matches the set of immutable objects in owned_object_refs.
354    /// This is stricter than general voting because the claim directly controls locking behavior.
355    fn verify_immutable_object_claims(
356        &self,
357        claimed_ids: &[ObjectID],
358        owned_object_refs: HashSet<ObjectRef>,
359    ) -> SuiResult<()> {
360        // Build map from object_id to input ref for version/digest verification
361        let input_refs_by_id: HashMap<ObjectID, ObjectRef> = owned_object_refs
362            .iter()
363            .map(|obj_ref| (obj_ref.0, *obj_ref))
364            .collect();
365
366        // First check: all claimed object IDs must be among the input object IDs
367        for claimed_id in claimed_ids {
368            if !input_refs_by_id.contains_key(claimed_id) {
369                return Err(SuiErrorKind::ImmutableObjectClaimNotFoundInInput {
370                    object_id: *claimed_id,
371                }
372                .into());
373            }
374        }
375
376        // Fetch all input objects and collect the actual immutable ones,
377        // verifying existence and version/digest match
378        let input_ids: Vec<ObjectID> = input_refs_by_id.keys().copied().collect();
379        let objects = self
380            .authority_state
381            .get_object_cache_reader()
382            .get_objects(&input_ids);
383
384        let claimed_immutable_ids = claimed_ids.iter().cloned().collect::<BTreeSet<_>>();
385        let mut found_immutable_ids = BTreeSet::new();
386
387        for (obj_opt, object_id) in objects.into_iter().zip(input_ids.iter()) {
388            let input_ref = input_refs_by_id.get(object_id).unwrap();
389            match obj_opt {
390                Some(o) => {
391                    // The object read here might drift from the one read earlier in validate_owned_object_versions(),
392                    // so re-check if input reference still matches actual object.
393                    let actual_ref = o.compute_object_reference();
394                    if actual_ref != *input_ref {
395                        return Err(SuiErrorKind::UserInputError {
396                            error: UserInputError::ObjectVersionUnavailableForConsumption {
397                                provided_obj_ref: *input_ref,
398                                current_version: actual_ref.1,
399                            },
400                        }
401                        .into());
402                    }
403                    if o.is_immutable() {
404                        found_immutable_ids.insert(*object_id);
405                    }
406                }
407                None => {
408                    // Object not found - we can't verify the claim, so we must reject.
409                    // This branch should not happen because owned input objects are already validated to exist.
410                    return Err(SuiErrorKind::UserInputError {
411                        error: UserInputError::ObjectNotFound {
412                            object_id: *object_id,
413                            version: Some(input_ref.1),
414                        },
415                    }
416                    .into());
417                }
418            }
419        }
420
421        // Compare claimed_ids with actual immutable objects - must match exactly
422        if let Some(claimed_id) = claimed_immutable_ids
423            .difference(&found_immutable_ids)
424            .next()
425        {
426            let input_ref = input_refs_by_id.get(claimed_id).unwrap();
427            return Err(SuiErrorKind::InvalidImmutableObjectClaim {
428                claimed_object_id: *claimed_id,
429                found_object_ref: *input_ref,
430            }
431            .into());
432        }
433        if let Some(found_id) = found_immutable_ids
434            .difference(&claimed_immutable_ids)
435            .next()
436        {
437            return Err(SuiErrorKind::ImmutableObjectNotClaimed {
438                object_id: *found_id,
439            }
440            .into());
441        }
442
443        Ok(())
444    }
445}

447fn tx_kind_from_bytes(tx: &[u8]) -> Result<ConsensusTransactionKind, ValidationError> {
448    bcs::from_bytes::<ConsensusTransaction>(tx)
449        .map_err(|e| {
450            ValidationError::InvalidTransaction(format!(
451                "Failed to parse transaction bytes: {:?}",
452                e
453            ))
454        })
455        .map(|tx| tx.kind)
456}
457
458impl TransactionVerifier for SuiTxValidator {
459    fn verify_batch(&self, batch: &[&[u8]]) -> Result<(), ValidationError> {
460        let _scope = monitored_scope("ValidateBatch");
461
462        let txs: Vec<_> = batch
463            .iter()
464            .map(|tx| tx_kind_from_bytes(tx))
465            .collect::<Result<Vec<_>, _>>()?;
466
467        self.validate_transactions(&txs)
468            .map_err(|e| ValidationError::InvalidTransaction(e.to_string()))
469    }
470
471    fn verify_and_vote_batch(
472        &self,
473        block_ref: &BlockRef,
474        batch: &[&[u8]],
475    ) -> Result<Vec<TransactionIndex>, ValidationError> {
476        let _scope = monitored_scope("VerifyAndVoteBatch");
477
478        let txs: Vec<_> = batch
479            .iter()
480            .map(|tx| tx_kind_from_bytes(tx))
481            .collect::<Result<Vec<_>, _>>()?;
482
483        self.validate_transactions(&txs)
484            .map_err(|e| ValidationError::InvalidTransaction(e.to_string()))?;
485
486        Ok(self.vote_transactions(block_ref, txs))
487    }
488}
