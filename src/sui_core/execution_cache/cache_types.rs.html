20    /// Entry is not in the cache and may or may not exist in the store
21    Miss,

24/// CachedVersionMap is a map from version to value, with the additional contraints:
25/// - The key (SequenceNumber) must be monotonically increasing for each insert. If
26///   a key is inserted that is less than the previous key, it results in an assertion
27///   failure.
28/// - Similarly, only the item with the least key can be removed.
29/// - The intent of these constraints is to ensure that there are never gaps in the collection,
30///   so that membership in the map can be tested by comparing to both the highest and lowest
31///   (first and last) entries.
32#[derive(Debug)]
33pub struct CachedVersionMap<V> {
34    values: VecDeque<(SequenceNumber, V)>,
35}
36
37impl<V> Default for CachedVersionMap<V> {
38    fn default() -> Self {
39        Self {
40            values: VecDeque::new(),
41        }
42    }
43}
44
45impl<V> CachedVersionMap<V> {
46    pub fn is_empty(&self) -> bool {
47        self.values.is_empty()
48    }
49
50    pub fn insert(&mut self, version: SequenceNumber, value: V) {
51        if !self.values.is_empty() {
52            let back = self.values.back().unwrap().0;
53            if back >= version {
54                fatal!(
55                    "version must be monotonically increasing ({} < {})",
56                    back,
57                    version
58                );
59            }
60        }
61        self.values.push_back((version, value));
62    }
63
64    pub fn all_versions_lt_or_eq_descending<'a>(
65        &'a self,
66        version: &'a SequenceNumber,
67    ) -> impl Iterator<Item = &'a (SequenceNumber, V)> {
68        self.values.iter().rev().filter(move |(v, _)| v <= version)
69    }
70
71    pub fn get(&self, version: &SequenceNumber) -> Option<&V> {
72        for (v, value) in self.values.iter().rev() {
73            match v.cmp(version) {
74                Ordering::Less => return None,
75                Ordering::Equal => return Some(value),
76                Ordering::Greater => (),
77            }
78        }
79
80        None
81    }
82
83    pub fn get_prior_to(&self, version: &SequenceNumber) -> Option<(SequenceNumber, &V)> {
84        for (v, value) in self.values.iter().rev() {
85            if v < version {
86                return Some((*v, value));
87            }
88        }
89
90        None
91    }
92
93    /// returns the newest (highest) version in the map
94    pub fn get_highest(&self) -> Option<&(SequenceNumber, V)> {
95        self.values.back()
96    }
97
98    /// returns the oldest (lowest) version in the map
99    pub fn get_least(&self) -> Option<&(SequenceNumber, V)> {
100        self.values.front()
101    }
102
103    // pop items from the front of the collection until the size is <= limit
104    pub fn truncate_to(&mut self, limit: usize) {
105        while self.values.len() > limit {
106            self.values.pop_front();
107        }
108    }
109
110    // remove the value if it is the first element in values.
111    pub fn pop_oldest(&mut self, version: &SequenceNumber) -> Option<V> {
112        let oldest = self.values.pop_front()?;
113        // if this assert fails it indicates we are committing transaction data out
114        // of causal order
115        assert_eq!(oldest.0, *version, "version must be the oldest in the map");
116        Some(oldest.1)
117    }
118}

120// an iterator adapter that asserts that the wrapped iterator yields elements in order
121#[allow(dead_code)]
122pub(super) struct AssertOrdered<I: Iterator> {
123    iter: I,
124    last: Option<I::Item>,
125}
126
127impl<I: Iterator> AssertOrdered<I> {
128    #[allow(dead_code)]
129    fn new(iter: I) -> Self {
130        Self { iter, last: None }
131    }
132}
133
134impl<I: IntoIterator> From<I> for AssertOrdered<I::IntoIter> {
135    fn from(iter: I) -> Self {
136        Self::new(iter.into_iter())
137    }
138}
139
140impl<I: Iterator> Iterator for AssertOrdered<I>
141where
142    I::Item: Ord + Copy,
143{
144    type Item = I::Item;
145
146    fn next(&mut self) -> Option<Self::Item> {
147        let next = self.iter.next();
148        if let Some(next) = next {
149            if let Some(last) = &self.last {
150                assert!(*last < next, "iterator must yield elements in order");
151            }
152            self.last = Some(next);
153        }
154        next
155    }
156}

164pub struct MonotonicCache<K, V> {
165    cache: MokaCache<K, Arc<Mutex<V>>>,
166    // When inserting a possibly stale value, we prove that it is not stale by
167    // ensuring that no fresh value was inserted since we began reading the value
168    // we are inserting. We do this by hashing the key to an element in this array,
169    // reading the current value, and then passing that value to insert(). If the
170    // value is out of date, then there may have been an intervening write, so we
171    // discard the insert attempt.
172    key_generation: Vec<AtomicU64>,
173}
174
175#[derive(Copy, Clone)]
176pub enum Ticket {
177    // Read tickets are used when caching the result of a read from the db.
178    // They are only valid if the generation number matches the current generation.
179    // Used to ensure that no write occurred while we were reading from the db.
180    Read(u64),
181    // Write tickets are always valid. Used when caching writes, which cannot be stale.
182    Write,
183}
184
185// key_generation should be big enough to make false positives unlikely. If, on
186// average, there is one millisecond between acquiring the ticket and calling insert(),
187// then even at 1 million inserts per second, there will be 1000 inserts between acquiring
188// the ticket and calling insert(), so about 1/16th of the entries will be invalidated,
189// so valid inserts will succeed with probability 15/16.
190const KEY_GENERATION_SIZE: usize = 1024 * 16;
191
192impl<K, V> MonotonicCache<K, V>
193where
194    K: Hash + Eq + Send + Sync + Copy + std::fmt::Debug + 'static,
195    V: IsNewer + Clone + Send + Sync + 'static,
196{
197    pub fn new(cache_size: u64) -> Self {
198        Self {
199            cache: MokaCache::builder(8).max_capacity(cache_size).build(),
200            key_generation: (0..KEY_GENERATION_SIZE)
201                .map(|_| AtomicU64::new(0))
202                .collect(),
203        }
204    }
205
206    pub fn get(&self, key: &K) -> Option<Arc<Mutex<V>>> {
207        self.cache.get(key)
208    }
209
210    fn generation(&self, key: &K) -> &AtomicU64 {
211        let mut state = DefaultHasher::new();
212        key.hash(&mut state);
213        let hash = state.finish();
214        &self.key_generation[(hash % KEY_GENERATION_SIZE as u64) as usize]
215    }
216
217    /// Get a ticket for caching the result of a read operation. The ticket will be
218    /// expired if a writer writes a new version of the value.
219    /// The caller must obtain the ticket BEFORE checking the dirty set and db. By
220    /// obeying this rule, the caller can be sure that if their ticket remains valid
221    /// at insert time, they either are inserting the most recent value, or a concurrent
222    /// writer will shortly overwrite their value.
223    pub fn get_ticket_for_read(&self, key: &K) -> Ticket {
224        let r#gen = self.generation(key);
225        Ticket::Read(r#gen.load(std::sync::atomic::Ordering::Acquire))
226    }
227
228    // Update the cache with guaranteed monotonicity. That is, if there are N
229    // calls to the this function from N threads, the write with the newest value will
230    // win the race regardless of what ordering the writes occur in.
231    //
232    // Caller should log the insert with trace! and increment the appropriate metric.
233    pub fn insert(&self, key: &K, value: V, ticket: Ticket) -> Result<(), ()> {
234        let r#gen = self.generation(key);
235
236        // invalidate other readers as early as possible. If a reader acquires a
237        // new ticket after this point, then it will read the new value from
238        // the dirty set (or db).
239        if matches!(ticket, Ticket::Write) {
240            r#gen.fetch_add(1, std::sync::atomic::Ordering::Release);
241        }
242
243        let check_ticket = || -> Result<(), ()> {
244            match ticket {
245                Ticket::Read(ticket) => {
246                    if ticket != r#gen.load(std::sync::atomic::Ordering::Acquire) {
247                        return Err(());
248                    }
249                    Ok(())
250                }
251                Ticket::Write => Ok(()),
252            }
253        };
254
255        // Warning: tricky code!
256        let entry = self
257            .cache
258            .entry(*key)
259            // Suppose there is a reader (who has an old version) and a writer (who has
260            // the newest version by definition) both trying to insert when the cache has
261            // no entry. Here are the possible outcomes:
262            //
263            // 1. Race in `or_optionally_insert_with`:
264            //    1. Reader wins race, ticket is valid, and reader inserts old version.
265            //       Writer will overwrite the old version after the !is_fresh check.
266            //    2. Writer wins race. Reader will enter is_fresh check, lock entry, and
267            //       find that its ticket is expired.
268            //
269            // 2. No race on `or_optionally_insert_with`:
270            //    1. Reader inserts first (via `or_optionally_insert_with`), writer enters !is_fresh
271            //       check and overwrites entry.
272            //       1. There are two sub-cases here because the reader's entry could be evicted,
273            //          but in either case the writer obviously overwrites it.
274            //    2. Writer inserts first (via `or_optionally_insert_with`), invalidates ticket.
275            //       Then, two cases can follow:
276            //       1. Reader skips `or_optionally_insert_with` (because entry is present), enters
277            //          !is_fresh check, and does not insert because its ticket is expired.
278            //       2. The writer's cache entry is evicted already, so reader enters
279            //          `or_optionally_insert_with`. The ticket is expired so we do not insert.
280            //
281            // The other cases are where there is already an entry. In this case neither reader
282            // nor writer will enter `or_optionally_insert_with` callback. Instead they will both enter
283            // the !is_fresh check and lock the entry:
284            // 1. If the reader locks first, it will insert its old version. Then the writer
285            //    will lock and overwrite it with the newer version.
286            // 2. If the writer locks first, it will have already expired the ticket, and the
287            //    reader will not insert anything.
288            //
289            // There may also be more than one concurrent reader. However, the only way the two
290            // readers can have different versions is if there is concurrently a writer that wrote
291            // a new version. In this case all stale readers will fail the ticket check, and only
292            // up-to-date readers will remain. So we cannot have a bad insert caused by two readers
293            // racing to insert, both with valid tickets.
294            .or_optionally_insert_with(|| {
295                check_ticket().ok()?;
296                Some(Arc::new(Mutex::new(value.clone())))
297            })
298            // Note: Ticket::Write cannot expire, but an insert can still fail, in the case where
299            // a writer and reader are racing to call `or_optionally_insert_with`, the reader wins,
300            // but then fails to insert because its ticket is expired. Then no entry at all is inserted.
301            .ok_or(())?;
302
303        // !is_fresh means we did not insert a new entry in or_optionally_insert_with above.
304        if !entry.is_fresh() {
305            let mut entry = entry.value().lock();
306            check_ticket()?;
307
308            // Ticket expiry should make this assert impossible.
309            if entry.is_newer_than(&value) {
310                debug_fatal!("entry is newer than value {:?}", key);
311            } else {
312                *entry = value;
313            }
314        }
315
316        Ok(())
317    }
318
319    pub fn invalidate(&self, key: &K) {
320        self.cache.invalidate(key);
321    }
322
323    #[cfg(test)]
324    pub fn contains_key(&self, key: &K) -> bool {
325        self.cache.contains_key(key)
326    }
327
328    pub fn invalidate_all(&self) {
329        self.cache.invalidate_all();
330    }
331
332    pub fn is_empty(&self) -> bool {
333        self.cache.iter().next().is_none()
334    }
335}
