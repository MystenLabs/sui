21const QUEUEING_DELAY_SAMPLING_RATIO: f64 = 0.05;
22
23/// When a notification that a new pending transaction is received we activate
24/// processing the transaction in a loop.
25pub async fn execution_process(
26    authority_state: Weak<AuthorityState>,
27    mut rx_ready_certificates: UnboundedReceiver<PendingCertificate>,
28    mut rx_execution_shutdown: oneshot::Receiver<()>,
29) {
30    info!("Starting pending certificates execution process.");
31
32    // Rate limit concurrent executions to # of cpus.
33    let limit = Arc::new(Semaphore::new(num_cpus::get()));
34
35    // Loop whenever there is a signal that a new transactions is ready to process.
36    loop {
37        let _scope = monitored_scope("ExecutionDriver::loop");
38
39        let certificate;
40        let execution_env;
41        let txn_ready_time;
42        let _executing_guard;
43        tokio::select! {
44            result = rx_ready_certificates.recv() => {
45                if let Some(pending_cert) = result {
46                    certificate = pending_cert.certificate;
47                    execution_env = pending_cert.execution_env;
48                    txn_ready_time = pending_cert.stats.ready_time.unwrap();
49                    _executing_guard = pending_cert.executing_guard;
50                } else {
51                    // Should only happen after the AuthorityState has shut down and tx_ready_certificate
52                    // has been dropped by ExecutionScheduler.
53                    info!("No more certificate will be received. Exiting executor ...");
54                    return;
55                };
56            }
57            _ = &mut rx_execution_shutdown => {
58                info!("Shutdown signal received. Exiting executor ...");
59                return;
60            }
61        };
62
63        let authority = if let Some(authority) = authority_state.upgrade() {
64            authority
65        } else {
66            // Terminate the execution if authority has already shutdown, even if there can be more
67            // items in rx_ready_certificates.
68            info!("Authority state has shutdown. Exiting ...");
69            return;
70        };
71        authority.metrics.execution_driver_dispatch_queue.dec();
72
73        // TODO: Ideally execution_driver should own a copy of epoch store and recreate each epoch.
74        let epoch_store = authority.load_epoch_store_one_call_per_task();
75
76        let digest = *certificate.digest();
77        trace!(?digest, "Pending certificate execution activated.");
78
79        if epoch_store.epoch() != certificate.epoch() {
80            info!(
81                ?digest,
82                cur_epoch = epoch_store.epoch(),
83                cert_epoch = certificate.epoch(),
84                "Ignoring certificate from previous epoch."
85            );
86            continue;
87        }
88
89        let limit = limit.clone();
90        // hold semaphore permit until task completes. unwrap ok because we never close
91        // the semaphore in this context.
92        let permit = limit.acquire_owned().await.unwrap();
93
94        if get_rng().gen_range(0.0..1.0) < QUEUEING_DELAY_SAMPLING_RATIO {
95            authority
96                .metrics
97                .execution_queueing_latency
98                .report(txn_ready_time.elapsed());
99            if let Some(latency) = authority.metrics.execution_queueing_latency.latency() {
100                authority
101                    .metrics
102                    .execution_queueing_delay_s
103                    .observe(latency.as_secs_f64());
104            }
105        }
106
107        authority.metrics.execution_rate_tracker.lock().record();
108
109        // Certificate execution can take significant time, so run it in a separate task.
110        let epoch_store_clone = epoch_store.clone();
111        spawn_monitored_task!(epoch_store.within_alive_epoch(async move {
112            let _scope = monitored_scope("ExecutionDriver::task");
113            let _guard = permit;
114            if authority.is_tx_already_executed(&digest) {
115                return;
116            }
117
118            fail_point_async!("transaction_execution_delay");
119
120            match authority.try_execute_immediately(
121                &certificate,
122                execution_env,
123                &epoch_store_clone,
124            ).await {
125                ExecutionOutput::Success(_) => {
126                    authority
127                        .metrics
128                        .execution_driver_executed_transactions
129                        .inc();
130                }
131                ExecutionOutput::EpochEnded => {
132                    warn!("Could not execute transaction {digest:?} because validator is halted at epoch end. certificate={certificate:?}");
133                }
134                ExecutionOutput::Fatal(e) => {
135                    fatal!("Failed to execute certified transaction {digest:?}! error={e} certificate={certificate:?}");
136                }
137                ExecutionOutput::RetryLater => {
138                    // Transaction will be retried later and auto-rescheduled, so we ignore it here
139                    authority
140                        .metrics
141                        .execution_driver_paused_transactions
142                        .inc();
143                }
144            }
145        }.instrument(error_span!("execution_driver", tx_digest = ?digest))));
146    }
147}
