22pub(crate) struct EagerBalanceWithdrawScheduler {
23    balance_read: Arc<dyn AccountBalanceRead>,
24    inner_state: Arc<Mutex<InnerState>>,
25}
26
27struct InnerState {
28    /// For each address balance account that we have seen withdraws through `schedule_withdraws`,
29    /// we track the current state of that account, and only remove it from the map after
30    /// we have settled all withdraws for that account.
31    tracked_accounts: HashMap<AccumulatorObjId, AccountState>,
32    /// Tracks all the acddress balance accounts that have a withdraw transaction tracked,
33    /// mapping from the accumulator version that the withdraw transaction reads from.
34    /// If a withdraw transaction needs to withdraw from account O at version V,
35    /// we must process and settle that withdraw transaction whenever we settle all transactions
36    /// scheduled for version V.
37    pending_settlements: HashMap<SequenceNumber, BTreeSet<AccumulatorObjId>>,
38    /// The current version of the accumulator object.
39    accumulator_version: SequenceNumber,
40}
41
42struct AccountState {
43    account_id: AccumulatorObjId,
44    /// The amount of balance that has been reserved for this account, for each accumulator version.
45    /// This is tracked so that we could add them back to the account balance when we settle the withdraws.
46    reserved_balance: HashMap<SequenceNumber, u128>,
47    /// Withdraws that could not yet be scheduled due to insufficient balance, and
48    /// hence have not reserved any balance yet. We track them so that we could schedule them
49    /// anytime we may have sufficient balance.
50    pending_reservations: VecDeque<Arc<PendingWithdraw>>,
51    /// The lower bound of the current balance of this account.
52    /// It is the amount of guaranteed balance that we could withdraw from this account at this point.
53    /// This is maintained as the most recent settled balance, subtracted by the reserved balance.
54    balance_lower_bound: u128,
55    /// The version of the accumulator object when this account object was read
56    /// into the scheduler. We need to know this because when we read an account object
57    /// during scheduling, we always read the latest version, which can have data
58    /// races with executions that are bumping the accumulator version at the same time.
59    /// It is possible that we read a version of the account object that we have not
60    /// yet processed the settle_balances call for that version. When we settle latter,
61    /// we need to ensure that we won't update this account object if we are already at the latest version.
62    init_version: Option<SequenceNumber>,
63}
64
65struct PendingWithdraw {
66    accumulator_version: SequenceNumber,
67    tx_digest: TransactionDigest,
68    sender: Mutex<Option<Sender<ScheduleResult>>>,
69    pending: Mutex<BTreeMap<AccumulatorObjId, u64>>,
70}
71
72enum TryReserveResult {
73    SufficientBalance,
74    InsufficientBalance,
75    Pending,
76}

78impl EagerBalanceWithdrawScheduler {
79    pub fn new(
80        balance_read: Arc<dyn AccountBalanceRead>,
81        starting_accumulator_version: SequenceNumber,
82    ) -> Arc<Self> {
83        Arc::new(Self {
84            balance_read,
85            inner_state: Arc::new(Mutex::new(InnerState {
86                tracked_accounts: HashMap::new(),
87                pending_settlements: HashMap::new(),
88                accumulator_version: starting_accumulator_version,
89            })),
90        })
91    }
92}

94#[async_trait::async_trait]
95impl BalanceWithdrawSchedulerTrait for EagerBalanceWithdrawScheduler {
96    #[instrument(level = "debug", skip_all, fields(withdraw_accumulator_version = ?withdraws.accumulator_version.value()))]
97    async fn schedule_withdraws(&self, withdraws: WithdrawReservations) {
98        let mut inner_state = self.inner_state.lock();
99        let cur_accumulator_version = inner_state.accumulator_version;
100        if withdraws.accumulator_version < cur_accumulator_version {
101            // This accumulator version is already settled.
102            // There is no need to schedule the withdraws.
103            withdraws.notify_skip_schedule();
104            return;
105        }
106
107        let all_accounts = withdraws.all_accounts();
108        let mut new_accounts = Vec::new();
109        for account_id in all_accounts {
110            if !inner_state.tracked_accounts.contains_key(&account_id) {
111                let (balance, version) = match self
112                    .balance_read
113                    .get_latest_account_balance(&account_id)
114                {
115                    Some((balance, version)) => {
116                        if version > withdraws.accumulator_version {
117                            // This account object is already at the next version, indicating
118                            // that a settlement transaction touching this account object has already been executed.
119                            // It doesn't mean all settlement transactions from the same commit batch have been executed,
120                            // but we are at a minimum in the process of executing them.
121                            // This means that all withdraw transactions in this commit have already been executed.
122                            // Hence we can skip scheduling the withdraws.
123                            debug!(
124                                ?account_id,
125                                "Accumulator account object is already at version {:?}, but the withdraws are at version {:?}",
126                                version,
127                                withdraws.accumulator_version
128                            );
129                            withdraws.notify_skip_schedule();
130                            return;
131                        }
132                        (balance, Some(version))
133                    }
134                    None => (0, None),
135                };
136                new_accounts.push(AccountState::new(account_id, balance, version));
137            }
138        }
139        for account_state in new_accounts {
140            inner_state
141                .tracked_accounts
142                .insert(account_state.account_id, account_state);
143        }
144
145        for (withdraw, sender) in withdraws.withdraws.into_iter().zip(withdraws.senders) {
146            let accounts = withdraw.reservations.keys().cloned().collect::<Vec<_>>();
147            inner_state
148                .pending_settlements
149                .entry(withdraws.accumulator_version)
150                .or_default()
151                .extend(&accounts);
152            let pending_withdraw =
153                PendingWithdraw::new(withdraws.accumulator_version, withdraw, sender);
154            for account_id in accounts {
155                let account_state = inner_state.tracked_accounts.get_mut(&account_id).unwrap();
156                let has_blocking_reservations = !account_state.pending_reservations.is_empty();
157                let result = account_state.try_reserve(
158                    cur_accumulator_version,
159                    &pending_withdraw,
160                    has_blocking_reservations,
161                );
162                if matches!(result, TryReserveResult::Pending) {
163                    account_state
164                        .pending_reservations
165                        .push_back(pending_withdraw.clone());
166                }
167            }
168        }
169    }
170
171    async fn settle_balances(&self, settlement: BalanceSettlement) {
172        let next_accumulator_version = settlement.next_accumulator_version;
173        let mut inner_state = self.inner_state.lock();
174        assert_eq!(
175            next_accumulator_version,
176            inner_state.accumulator_version.next()
177        );
178        let cleanup_version = inner_state.accumulator_version;
179        inner_state.accumulator_version.increment();
180        debug!(
181            "Bumping accumulator version from {:?} to {:?}",
182            cleanup_version.value(),
183            next_accumulator_version.value(),
184        );
185
186        // A settlement bumps the accumulator version from `cleanup_version` to `last_settled_version`.
187        // We must processing the following types of accounts:
188        // 1. Accounts that had withdraws scheduled at version `cleanup_version`. These withdraws
189        // are now settled and we know their exact balance changes.
190        // 2. Accounts that had withdraws scheduled at version `last_settled_version`. These withdraws
191        // if not yet scheduled, can now be deterministically scheduled, since
192        // we have the final state before them.
193        // 3. Accounts that have balance changes through settle_balances, this can include
194        // both withdraws and deposits.
195        let mut affected_accounts = inner_state
196            .pending_settlements
197            .remove(&cleanup_version)
198            .unwrap_or_default();
199        // Since we just bumped the accumulator version, any withdraws scheduled that depend
200        // on this version can now be deterministically scheduled.
201        if let Some(current_version_accounts) = inner_state
202            .pending_settlements
203            .get(&next_accumulator_version)
204        {
205            affected_accounts.extend(current_version_accounts);
206        }
207        affected_accounts.extend(settlement.balance_changes.keys().cloned());
208
209        debug!(
210            "Processing withdraws affecting accounts: {:?}",
211            affected_accounts,
212        );
213        for object_id in affected_accounts {
214            let Some(account_state) = inner_state.tracked_accounts.get_mut(&object_id) else {
215                continue;
216            };
217            let already_settled = match account_state.init_version {
218                Some(init_version) if init_version >= next_accumulator_version => {
219                    debug!(
220                        ?init_version,
221                        ?next_accumulator_version,
222                        account_id = ?object_id,
223                        "Account state is already at latest version, skip settling",
224                    );
225                    true
226                }
227                _ => false,
228            };
229            debug!(
230                account_id = ?object_id,
231                "Settling account",
232            );
233            if !already_settled {
234                let reserved = account_state
235                    .reserved_balance
236                    .remove(&cleanup_version)
237                    .unwrap_or_default() as i128;
238                let settled = settlement
239                    .balance_changes
240                    .get(&object_id)
241                    .copied()
242                    .unwrap_or_default();
243                // Withdraw amounts must be bounded by reservations.
244                let net = u128::try_from(reserved.checked_add(settled).unwrap()).unwrap();
245                account_state.balance_lower_bound += net;
246                debug!(
247                    account_id = ?object_id,
248                    "Reserved balance: {:?}, settled balance: {:?}, new min guaranteed balance: {:?}",
249                    reserved, settled, account_state.balance_lower_bound,
250                );
251            }
252
253            while !account_state.pending_reservations.is_empty() {
254                let pending_withdraw = account_state.pending_reservations.pop_front().unwrap();
255                assert!(pending_withdraw.accumulator_version >= next_accumulator_version);
256
257                let result =
258                    account_state.try_reserve(next_accumulator_version, &pending_withdraw, false);
259                if matches!(result, TryReserveResult::Pending) {
260                    account_state
261                        .pending_reservations
262                        .push_front(pending_withdraw);
263                    break;
264                }
265            }
266
267            if account_state.is_empty() {
268                debug!(
269                    account_id = ?object_id,
270                    "Removing account state since it is empty",
271                );
272                inner_state.tracked_accounts.remove(&object_id);
273            }
274            // TODO: Debug invariant check on account state and accumulator version match.
275        }
276        debug!("Settled accumulator version {:?}", next_accumulator_version);
277    }
278
279    fn close_epoch(&self) {
280        debug!("Closing epoch in EagerBalanceWithdrawScheduler",);
281        let inner_state = self.inner_state.lock();
282        assert!(inner_state.pending_settlements.is_empty());
283        assert!(inner_state.tracked_accounts.is_empty());
284    }
285
286    #[cfg(test)]
287    fn get_current_accumulator_version(&self) -> SequenceNumber {
288        let inner_state = self.inner_state.lock();
289        inner_state.accumulator_version
290    }
291}

293impl AccountState {
294    fn new(
295        account_id: AccumulatorObjId,
296        balance: u128,
297        last_read_version: Option<SequenceNumber>,
298    ) -> Self {
299        debug!(
300            account_id = ?account_id.inner(),
301            "New account state tracked with initial balance {:?}, read at version {:?}",
302            balance,
303            last_read_version,
304        );
305        Self {
306            account_id,
307            reserved_balance: HashMap::new(),
308            pending_reservations: VecDeque::new(),
309            balance_lower_bound: balance,
310            init_version: last_read_version,
311        }
312    }
313
314    #[instrument(
315        level = "debug",
316        skip_all,
317        fields(
318            cur_accumulator_version =? cur_accumulator_version.value(),
319            account_id = ?self.account_id.inner(),
320            tx_digest = ?pending_withdraw.tx_digest,
321            ?has_blocking_reservations
322        )
323    )]
324    fn try_reserve(
325        &mut self,
326        cur_accumulator_version: SequenceNumber,
327        pending_withdraw: &Arc<PendingWithdraw>,
328        has_blocking_reservations: bool,
329    ) -> TryReserveResult {
330        let to_reserve = pending_withdraw.pending_amount(&self.account_id);
331        debug!(
332            "Trying to reserve {}, min_guaranteed_balance: {}, pending_reservations size: {}",
333            to_reserve,
334            self.balance_lower_bound,
335            self.pending_reservations.len(),
336        );
337        let insufficient_balance = to_reserve > self.balance_lower_bound;
338        if insufficient_balance || has_blocking_reservations {
339            if cur_accumulator_version == pending_withdraw.accumulator_version {
340                assert!(!has_blocking_reservations);
341                pending_withdraw.notify_insufficient_balance();
342                TryReserveResult::InsufficientBalance
343            } else {
344                debug!("Adding to pending reservations since we cannot schedule it yet.");
345                TryReserveResult::Pending
346            }
347        } else {
348            self.commit_reservation(pending_withdraw);
349            TryReserveResult::SufficientBalance
350        }
351    }
352
353    fn commit_reservation(&mut self, pending_withdraw: &Arc<PendingWithdraw>) {
354        let mut pending = pending_withdraw.pending.lock();
355        let to_reserve = pending.remove(&self.account_id).unwrap() as u128;
356        assert!(self.balance_lower_bound >= to_reserve);
357        self.balance_lower_bound -= to_reserve;
358        *self
359            .reserved_balance
360            .entry(pending_withdraw.accumulator_version)
361            .or_default() += to_reserve;
362        debug!(
363            "Successfully reserved {} for account. New min guaranteed balance: {}",
364            to_reserve, self.balance_lower_bound
365        );
366        if pending.is_empty() {
367            debug!("Successfully reserved all accounts for withdraw transaction");
368            let sender = pending_withdraw.sender.lock().take().unwrap();
369            let _ = sender.send(ScheduleResult {
370                tx_digest: pending_withdraw.tx_digest,
371                status: ScheduleStatus::SufficientBalance,
372            });
373        }
374    }
375
376    fn is_empty(&self) -> bool {
377        self.reserved_balance.is_empty() && self.pending_reservations.is_empty()
378    }
379}
380
381impl PendingWithdraw {
382    fn new(
383        accumulator_version: SequenceNumber,
384        withdraw: TxBalanceWithdraw,
385        sender: Sender<ScheduleResult>,
386    ) -> Arc<Self> {
387        Arc::new(Self {
388            accumulator_version,
389            tx_digest: withdraw.tx_digest,
390            sender: Mutex::new(Some(sender)),
391            pending: Mutex::new(withdraw.reservations),
392        })
393    }
394
395    fn pending_amount(&self, account_id: &AccumulatorObjId) -> u128 {
396        self.pending.lock().get(account_id).copied().unwrap() as u128
397    }
398
399    fn notify_insufficient_balance(&self) {
400        let mut sender_guard = self.sender.lock();
401        // sender may be None because this pending withdraw may have multiple
402        // insufficient accounts, and when processing the first one, the sender
403        // is already taken.
404        if let Some(sender) = sender_guard.take() {
405            debug!("Insufficient balance for withdraw");
406            let _ = sender.send(ScheduleResult {
407                tx_digest: self.tx_digest,
408                status: ScheduleStatus::InsufficientBalance,
409            });
410        }
411    }
