16/// A naive implementation of the balance withdraw scheduler that does not attempt to optimize the scheduling.
17/// For each withdraw reservation, it will always wait until the dependent accumulator object is available,
18/// and then check if the balance is sufficient.
19/// This implementation is simple and easy to understand, but it is not efficient.
20/// It is only used to unblock further development of the balance withdraw scheduler.
21pub(crate) struct NaiveBalanceWithdrawScheduler {
22    balance_read: Arc<dyn AccountBalanceRead>,
23    accumulator_version_sender: watch::Sender<SequenceNumber>,
24    // We must keep a receiver alive to make sure sends go through and can update the last settled version.
25    accumulator_version_receiver: watch::Receiver<SequenceNumber>,
26}
27
28impl NaiveBalanceWithdrawScheduler {
29    pub fn new(
30        balance_read: Arc<dyn AccountBalanceRead>,
31        cur_accumulator_version: SequenceNumber,
32    ) -> Arc<Self> {
33        let (accumulator_version_sender, accumulator_version_receiver) =
34            watch::channel(cur_accumulator_version);
35        Arc::new(Self {
36            balance_read,
37            accumulator_version_sender,
38            accumulator_version_receiver,
39        })
40    }
41}
42
43#[async_trait::async_trait]
44impl BalanceWithdrawSchedulerTrait for NaiveBalanceWithdrawScheduler {
45    #[instrument(level = "debug", skip_all, fields(withdraw_accumulator_version = ?withdraws.accumulator_version.value()))]
46    async fn schedule_withdraws(&self, withdraws: WithdrawReservations) {
47        let mut receiver = self.accumulator_version_sender.subscribe();
48        while *receiver.borrow_and_update() < withdraws.accumulator_version {
49            debug!("Waiting for the dependent accumulator version to be settled");
50            if receiver.changed().await.is_err() {
51                return;
52            }
53        }
54        if *receiver.borrow() > withdraws.accumulator_version {
55            // This accumulator version is already settled.
56            // There is no need to schedule the withdraws.
57            withdraws.notify_skip_schedule();
58            return;
59        }
60
61        // Map from each account ID that we have seen so far to the current
62        // remaining balance for reservation.
63        let mut cur_balances = BTreeMap::new();
64        for account_id in withdraws.all_accounts() {
65            // Load the current balance for the account.
66            // It is possible that executions through checkpoint executor have already settled this version,
67            // and hence the current version is greater than the withdraws' accumulator version.
68            // In this case, we simply skip processing the withdraws and notify the caller that the withdraws are already settled.
69            // This is safe because these transactions have already been executed.
70            let balance = match self.balance_read.get_latest_account_balance(&account_id) {
71                Some((balance, cur_version)) => {
72                    if cur_version > withdraws.accumulator_version {
73                        // This account object is already at the next version, indicating
74                        // that a settlement transaction touching this account object has already been executed.
75                        // It doesn't mean all settlement transactions from the same commit batch have been executed,
76                        // but we are at a minimum in the process of executing them.
77                        // This means that all withdraw transactions in this commit have already been executed.
78                        // Hence we can skip scheduling the withdraws.
79                        debug!(
80                            ?account_id,
81                            "Accumulator account object is already at version {:?}, but the withdraws are at version {:?}",
82                            cur_version,
83                            withdraws.accumulator_version
84                        );
85                        withdraws.notify_skip_schedule();
86                        return;
87                    }
88                    balance
89                }
90                None => 0,
91            };
92            cur_balances.insert(account_id, balance);
93        }
94        for (withdraw, sender) in withdraws.withdraws.into_iter().zip(withdraws.senders) {
95            // Try to reserve all withdraws in this transaction.
96            // Note that this is not atomic, so it is possible that we reserve some withdraws and not others.
97            // This is intentional to be semantically consistent with the eager scheduler.
98            let mut success = true;
99            for (object_id, reservation) in &withdraw.reservations {
100                let entry = cur_balances.get_mut(object_id).unwrap();
101                if *entry < *reservation as u128 {
102                    debug!(
103                        tx_digest =? withdraw.tx_digest,
104                        "Insufficient balance for {:?}. Requested: {:?}, Available: {:?}",
105                        object_id, reservation, entry
106                    );
107                    success = false;
108                } else {
109                    debug!(
110                        tx_digest =? withdraw.tx_digest,
111                        "Successfully reserved {:?} for account {:?}",
112                        reservation, object_id
113                    );
114                    *entry -= *reservation as u128;
115                }
116            }
117            if success {
118                debug!(
119                    tx_digest =? withdraw.tx_digest,
120                    "Successfully reserved all withdraws"
121                );
122                let _ = sender.send(ScheduleResult {
123                    tx_digest: withdraw.tx_digest,
124                    status: ScheduleStatus::SufficientBalance,
125                });
126            } else {
127                let _ = sender.send(ScheduleResult {
128                    tx_digest: withdraw.tx_digest,
129                    status: ScheduleStatus::InsufficientBalance,
130                });
131            }
132        }
133    }
134
135    // We don't use the balance changes information in the naive scheduler.
136    // Instead, the withdraw scheduling always read the balance state fro storage.
137    async fn settle_balances(&self, settlement: BalanceSettlement) {
138        let cur_accumulator_version = *self.accumulator_version_receiver.borrow();
139        let next_version = cur_accumulator_version.next();
140        debug!(
141            settled_accumulator_version =? cur_accumulator_version.value(),
142            next_accumulator_version =? next_version.value(),
143            "Settling balances",
144        );
145        assert_eq!(next_version, settlement.next_accumulator_version);
146        let _ = self.accumulator_version_sender.send(next_version);
147    }
148
149    fn close_epoch(&self) {
150        debug!("Closing epoch in NaiveBalanceWithdrawScheduler");
151    }
152
153    #[cfg(test)]
154    fn get_current_accumulator_version(&self) -> SequenceNumber {
155        *self.accumulator_version_receiver.borrow()
156    }
157}
