20#[async_trait::async_trait]
21pub(crate) trait BalanceWithdrawSchedulerTrait: Send + Sync {
22    async fn schedule_withdraws(&self, withdraws: WithdrawReservations);
23    async fn settle_balances(&self, settlement: BalanceSettlement);
24    fn close_epoch(&self);
25    #[cfg(test)]
26    fn get_current_accumulator_version(&self) -> SequenceNumber;
27}
28
29pub(crate) struct WithdrawReservations {
30    pub accumulator_version: SequenceNumber,
31    pub withdraws: Vec<TxBalanceWithdraw>,
32    pub senders: Vec<oneshot::Sender<ScheduleResult>>,
33}
34
35#[derive(Clone)]
36pub(crate) struct BalanceWithdrawScheduler {
37    innards: BTreeMap<String, Arc<dyn BalanceWithdrawSchedulerTrait>>,
38    /// Use channels to process withdraws and settlements asynchronously without blocking the caller.
39    withdraw_sender: UnboundedSender<WithdrawReservations>,
40    settlement_sender: UnboundedSender<BalanceSettlement>,
41}
42
43impl WithdrawReservations {
44    pub fn new(
45        accumulator_version: SequenceNumber,
46        withdraws: Vec<TxBalanceWithdraw>,
47    ) -> (Self, FuturesUnordered<oneshot::Receiver<ScheduleResult>>) {
48        let (senders, receivers) = (0..withdraws.len())
49            .map(|_| {
50                let (sender, receiver) = oneshot::channel();
51                (sender, receiver)
52            })
53            .unzip();
54        (
55            Self {
56                accumulator_version,
57                withdraws,
58                senders,
59            },
60            receivers,
61        )
62    }
63
64    pub fn notify_skip_schedule(self) {
65        debug!(
66            "Withdraws at accumulator version {:?} are already settled",
67            self.accumulator_version
68        );
69        for (withdraw, sender) in self.withdraws.into_iter().zip(self.senders) {
70            let _ = sender.send(ScheduleResult {
71                tx_digest: withdraw.tx_digest,
72                status: ScheduleStatus::SkipSchedule,
73            });
74        }
75    }
76
77    pub fn all_accounts(&self) -> BTreeSet<AccumulatorObjId> {
78        self.withdraws
79            .iter()
80            .flat_map(|withdraw| withdraw.reservations.keys())
81            .cloned()
82            .collect::<BTreeSet<_>>()
83    }
84}
85
86impl BalanceWithdrawScheduler {
87    pub fn new(
88        balance_read: Arc<dyn AccountBalanceRead>,
89        starting_accumulator_version: SequenceNumber,
90    ) -> Self {
91        // TODO: Currently, scheduling will be as slow as the slowest scheduler (i.e., the naive scheduler).
92        // Once we ensure that the eager scheduler is correct, it will become the only scheduler needed here,
93        // and we would only run the naive scheduler in tests for correctness verification.
94        let innards = BTreeMap::from([
95            (
96                "naive".to_string(),
97                NaiveBalanceWithdrawScheduler::new(
98                    balance_read.clone(),
99                    starting_accumulator_version,
100                ) as Arc<dyn BalanceWithdrawSchedulerTrait>,
101            ),
102            (
103                "eager".to_string(),
104                EagerBalanceWithdrawScheduler::new(balance_read, starting_accumulator_version)
105                    as Arc<dyn BalanceWithdrawSchedulerTrait>,
106            ),
107        ]);
108        let (withdraw_sender, withdraw_receiver) =
109            unbounded_channel("withdraw_scheduler_withdraws");
110        let (settlement_sender, settlement_receiver) =
111            unbounded_channel("withdraw_scheduler_settlements");
112        let scheduler = Self {
113            innards,
114            withdraw_sender,
115            settlement_sender,
116        };
117        tokio::spawn(scheduler.clone().process_withdraw_task(withdraw_receiver));
118        tokio::spawn(
119            scheduler
120                .clone()
121                .process_settlement_task(settlement_receiver),
122        );
123        scheduler
124    }
125
126    /// This function will be called at most once per consensus commit batch that all reads the same root accumulator version.
127    /// If a consensus commit batch does not contain any withdraw reservations, it can skip calling this function.
128    /// It must be called sequentially in order to correctly schedule withdraws.
129    pub fn schedule_withdraws(
130        &self,
131        accumulator_version: SequenceNumber,
132        withdraws: Vec<TxBalanceWithdraw>,
133    ) -> FuturesUnordered<oneshot::Receiver<ScheduleResult>> {
134        // TODO: Add debug assertion that withdraws are scheduled in order.
135        let (reservations, receivers) = WithdrawReservations::new(accumulator_version, withdraws);
136        if let Err(err) = self.withdraw_sender.send(reservations) {
137            tracing::error!("Failed to send withdraw reservations: {:?}", err);
138        }
139        receivers
140    }
141
142    /// This function is called whenever a settlement transaction is executed.
143    /// It is only called from checkpoint builder, once for each accumulator version, in order.
144    pub fn settle_balances(&self, settlement: BalanceSettlement) {
145        if let Err(err) = self.settlement_sender.send(settlement) {
146            tracing::error!("Failed to send balance settlement: {}", err);
147        }
148    }
149
150    pub fn close_epoch(&self) {
151        for (name, scheduler) in &self.innards {
152            debug!("Closing epoch for scheduler: {}", name);
153            scheduler.close_epoch();
154        }
155    }
156
157    #[cfg(test)]
158    pub fn get_current_accumulator_version(&self) -> SequenceNumber {
159        let versions: Vec<_> = self
160            .innards
161            .iter()
162            .map(|(name, scheduler)| (name, scheduler.get_current_accumulator_version()))
163            .collect();
164
165        let first_version = versions[0].1;
166        for (name, version) in versions.iter().skip(1) {
167            assert_eq!(
168                *version, first_version,
169                "Scheduler '{}' has version {:?}, but expected {:?}",
170                name, version, first_version
171            );
172        }
173
174        first_version
175    }
176
177    async fn process_withdraw_task(
178        self,
179        mut withdraw_receiver: UnboundedReceiver<WithdrawReservations>,
180    ) {
181        while let Some(event) = withdraw_receiver.recv().await {
182            debug!(
183                withdraw_accumulator_version =? event.accumulator_version.value(),
184                "Processing withdraws: {:?}",
185                event.withdraws,
186            );
187
188            let accumulator_version = event.accumulator_version;
189            let withdraws = event.withdraws.clone();
190            let original_senders = event.senders;
191
192            // Create intermediate channels for each scheduler
193            let mut scheduler_receivers = BTreeMap::new();
194            for (name, scheduler) in &self.innards {
195                let (intermediate_senders, receivers): (Vec<_>, Vec<_>) =
196                    withdraws.iter().map(|_| oneshot::channel()).unzip();
197                scheduler_receivers.insert(name.clone(), receivers);
198
199                let reservations = WithdrawReservations {
200                    accumulator_version,
201                    withdraws: withdraws.clone(),
202                    senders: intermediate_senders,
203                };
204                debug!("Scheduling withdraws on scheduler: {}", name);
205                scheduler.schedule_withdraws(reservations).await;
206            }
207
208            // Collect results from all schedulers
209            for (tx_idx, original_sender) in original_senders.into_iter().enumerate() {
210                let mut results = Vec::new();
211                for (name, receivers) in &mut scheduler_receivers {
212                    if let Some(receiver) = receivers.get_mut(tx_idx) {
213                        match receiver.await {
214                            Ok(result) => {
215                                debug!(
216                                    scheduler = %name,
217                                    tx_digest =? result.tx_digest,
218                                    status =? result.status,
219                                    "Received scheduling result"
220                                );
221                                results.push(result);
222                            }
223                            Err(_) => {
224                                tracing::error!(
225                                    "Failed to receive result from scheduler: {}",
226                                    name
227                                );
228                            }
229                        }
230                    }
231                }
232
233                // Aggregate results according to the rules
234                let final_result = Self::aggregate_results(results);
235                let _ = original_sender.send(final_result);
236            }
237        }
238        tracing::info!("Balance withdraw receiver closed");
239    }
240
241    fn aggregate_results(results: Vec<ScheduleResult>) -> ScheduleResult {
242        assert!(!results.is_empty(), "Must have at least one result");
243
244        let tx_digest = results[0].tx_digest;
245        let non_skip_results: Vec<_> = results
246            .iter()
247            .filter(|r| r.status != ScheduleStatus::SkipSchedule)
248            .collect();
249
250        if non_skip_results.is_empty() {
251            // All schedulers returned SkipSchedule
252            return ScheduleResult {
253                tx_digest,
254                status: ScheduleStatus::SkipSchedule,
255            };
256        }
257
258        // Verify all non-skip results agree
259        let first_status = non_skip_results[0].status;
260        for result in &non_skip_results {
261            assert_eq!(
262                result.status, first_status,
263                "Schedulers returned different results for tx {:?}: expected {:?}, got {:?}",
264                tx_digest, first_status, result.status
265            );
266        }
267
268        ScheduleResult {
269            tx_digest,
270            status: first_status,
271        }
272    }
273
274    async fn process_settlement_task(
275        self,
276        mut settlement_receiver: UnboundedReceiver<BalanceSettlement>,
277    ) {
278        while let Some(settlement) = settlement_receiver.recv().await {
279            debug!(
280                next_accumulator_version =? settlement.next_accumulator_version.value(),
281                "Settling balance changes: {:?}",
282                settlement.balance_changes,
283            );
284            for (name, scheduler) in &self.innards {
285                debug!("Settling balances on scheduler: {}", name);
286                scheduler.settle_balances(settlement.clone()).await;
287            }
288        }
289        tracing::info!("Balance settlement receiver closed");
290    }
291}
