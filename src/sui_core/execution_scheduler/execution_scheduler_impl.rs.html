46/// Utility struct for collecting barrier dependencies
47pub(crate) struct BarrierDependencyBuilder {
48    dep_state: BTreeMap<ObjectID, BTreeSet<TransactionDigest>>,
49}
50
51impl BarrierDependencyBuilder {
52    pub fn new() -> Self {
53        Self {
54            dep_state: Default::default(),
55        }
56    }
57
58    /// process_tx must be called for each transaction in scheduling order. If the
59    /// transaction has a non-exclusive write to an object, the transaction digest is
60    /// stored to become a dependency of the eventual barrier transaction. If a
61    /// transaction has an exclusive write to an object, all pending non-exclusive write
62    /// transactions for that object are added to the barrier dependencies.
63    pub fn process_tx(
64        &mut self,
65        tx_digest: TransactionDigest,
66        tx: &TransactionData,
67    ) -> BTreeSet<TransactionDigest> {
68        let mut barrier_deps = BTreeSet::new();
69        for SharedInputObject { id, mutability, .. } in tx.kind().shared_input_objects() {
70            match mutability {
71                SharedObjectMutability::NonExclusiveWrite => {
72                    self.dep_state.entry(id).or_default().insert(tx_digest);
73                }
74                SharedObjectMutability::Mutable => {
75                    // If there were preceding non-exclusive writes to this object id, this
76                    // transaction is a barrier and must wait for them to finish.
77                    if let Some(deps) = self.dep_state.remove(&id) {
78                        barrier_deps.extend(deps);
79                    }
80                }
81                SharedObjectMutability::Immutable => (),
82            }
83        }
84        barrier_deps
85    }
86}

100struct PendingGuard<'a> {
101    scheduler: &'a ExecutionScheduler,
102    cert: &'a VerifiedExecutableTransaction,
103}

105impl<'a> PendingGuard<'a> {
106    pub fn new(scheduler: &'a ExecutionScheduler, cert: &'a VerifiedExecutableTransaction) -> Self {
107        scheduler
108            .metrics
109            .transaction_manager_num_pending_certificates
110            .inc();
111        scheduler
112            .overload_tracker
113            .add_pending_certificate(cert.data());
114        Self { scheduler, cert }
115    }
116}

118impl Drop for PendingGuard<'_> {
119    fn drop(&mut self) {
120        self.scheduler
121            .metrics
122            .transaction_manager_num_pending_certificates
123            .dec();
124        self.scheduler
125            .overload_tracker
126            .remove_pending_certificate(self.cert.data());
127    }
128}

168    fn initialize_funds_withdraw_scheduler(
169        epoch_store: &Arc<AuthorityPerEpochStore>,
170        object_cache_read: &Arc<dyn ObjectCacheRead>,
171        account_funds_read: Arc<dyn AccountFundsRead>,
172        scheduler_type: FundsWithdrawSchedulerType,
173        address_funds_scheduler_metrics: &Arc<AddressFundsSchedulerMetrics>,
174    ) -> Option<FundsWithdrawScheduler> {
175        let withdraw_scheduler_enabled =
176            epoch_store.is_validator() && epoch_store.accumulators_enabled();
177        if !withdraw_scheduler_enabled {
178            return None;
179        }
180        let starting_accumulator_version = object_cache_read
181            .get_object(&SUI_ACCUMULATOR_ROOT_OBJECT_ID)
182            .expect("Accumulator root object must be present if funds accumulator is enabled")
183            .version();
184        let address_funds_withdraw_scheduler = FundsWithdrawScheduler::new(
185            account_funds_read.clone(),
186            starting_accumulator_version,
187            scheduler_type,
188            address_funds_scheduler_metrics.clone(),
189        );
190
191        Some(address_funds_withdraw_scheduler)
192    }

194    #[instrument(level = "debug", skip_all, fields(tx_digest = ?cert.digest()))]
195    async fn schedule_transaction(
196        self,
197        cert: VerifiedExecutableTransaction,
198        execution_env: ExecutionEnv,
199        epoch_store: &Arc<AuthorityPerEpochStore>,
200    ) {
201        let enqueue_time = Instant::now();
202        let tx_digest = cert.digest();
203        let digests = [*tx_digest];
204
205        let tx_data = cert.transaction_data();
206        let input_object_kinds = tx_data
207            .input_objects()
208            .expect("input_objects() cannot fail");
209        let input_object_keys: Vec<_> = epoch_store
210            .get_input_object_keys(
211                &cert.key(),
212                &input_object_kinds,
213                &execution_env.assigned_versions,
214            )
215            .into_iter()
216            .collect();
217
218        let receiving_object_keys: HashSet<_> = tx_data
219            .receiving_objects()
220            .into_iter()
221            .map(|entry| {
222                InputKey::VersionedObject {
223                    // TODO: Add support for receiving ConsensusV2 objects. For now this assumes fastpath.
224                    id: FullObjectID::new(entry.0, None),
225                    version: entry.1,
226                }
227            })
228            .collect();
229        let input_and_receiving_keys = [
230            input_object_keys,
231            receiving_object_keys.iter().cloned().collect(),
232        ]
233        .concat();
234
235        let epoch = epoch_store.epoch();
236        debug!(
237            ?tx_digest,
238            "Scheduled transaction, waiting for input objects: {:?}", input_and_receiving_keys,
239        );
240
241        let availability = self
242            .object_cache_read
243            .multi_input_objects_available_cache_only(&input_and_receiving_keys);
244        // Most of the times, the transaction's input objects are already available.
245        // We can check the availability of the input objects first, and only wait for the
246        // missing input objects if necessary.
247        let missing_input_keys: Vec<_> = input_and_receiving_keys
248            .into_iter()
249            .zip(availability)
250            .filter_map(|(key, available)| if !available { Some(key) } else { None })
251            .collect();
252
253        let has_missing_barrier_dependencies = self
254            .transaction_cache_read
255            .multi_get_executed_effects_digests(&execution_env.barrier_dependencies)
256            .into_iter()
257            .any(|r| r.is_none());
258
259        if missing_input_keys.is_empty() && !has_missing_barrier_dependencies {
260            self.metrics
261                .transaction_manager_num_enqueued_certificates
262                .with_label_values(&["ready"])
263                .inc();
264            debug!(?tx_digest, "Input objects already available");
265            self.send_transaction_for_execution(&cert, execution_env, enqueue_time);
266            return;
267        }
268
269        let _pending_guard = PendingGuard::new(&self, &cert);
270        self.metrics
271            .transaction_manager_num_enqueued_certificates
272            .with_label_values(&["pending"])
273            .inc();
274
275        if !execution_env.barrier_dependencies.is_empty() {
276            debug!(
277                "waiting for barrier dependencies to be executed: {:?}",
278                execution_env.barrier_dependencies
279            );
280            self.transaction_cache_read
281                .notify_read_executed_effects_digests(
282                    "wait_for_barrier_dependencies",
283                    &execution_env.barrier_dependencies,
284                )
285                .await;
286        }
287
288        tokio::select! {
289            _ = self.object_cache_read
290                .notify_read_input_objects(&missing_input_keys, &receiving_object_keys, epoch)
291                => {
292                    self.metrics
293                        .transaction_manager_transaction_queue_age_s
294                        .observe(enqueue_time.elapsed().as_secs_f64());
295                    debug!(?tx_digest, "Input objects available");
296                    // TODO: Eventually we could fold execution_driver into the scheduler.
297                    self.send_transaction_for_execution(
298                        &cert,
299                        execution_env,
300                        enqueue_time,
301                    );
302                }
303            _ = self.transaction_cache_read.notify_read_executed_effects_digests(
304                "ExecutionScheduler::notify_read_executed_effects_digests",
305                &digests,
306            ) => {
307                debug!(?tx_digest, "Transaction already executed");
308            }
309        };
310    }

312    pub fn send_transaction_for_execution(
313        &self,
314        cert: &VerifiedExecutableTransaction,
315        execution_env: ExecutionEnv,
316        enqueue_time: Instant,
317    ) {
318        let pending_cert = PendingCertificate {
319            certificate: cert.clone(),
320            execution_env,
321            stats: PendingCertificateStats {
322                enqueue_time,
323                ready_time: Some(Instant::now()),
324            },
325            executing_guard: Some(ExecutingGuard::new(
326                self.metrics
327                    .transaction_manager_num_executing_certificates
328                    .clone(),
329            )),
330        };
331        let _ = self.tx_ready_certificates.send(pending_cert);
332    }

334    fn schedule_funds_withdraws(
335        &self,
336        certs: Vec<(VerifiedExecutableTransaction, ExecutionEnv)>,
337        epoch_store: &Arc<AuthorityPerEpochStore>,
338    ) {
339        if certs.is_empty() {
340            return;
341        }
342        let mut withdraws = BTreeMap::new();
343        let mut prev_version = None;
344        for (cert, env) in &certs {
345            let tx_withdraws = cert
346                .transaction_data()
347                .process_funds_withdrawals_for_execution(epoch_store.get_chain_identifier());
348            assert!(!tx_withdraws.is_empty());
349            let accumulator_version = env
350                .assigned_versions
351                .accumulator_version
352                .expect("accumulator_version must be set when there are withdraws");
353            if let Some(prev_version) = prev_version {
354                // Transactions must be in order.
355                assert!(prev_version <= accumulator_version);
356            }
357            prev_version = Some(accumulator_version);
358            let tx_digest = *cert.digest();
359            withdraws
360                .entry(accumulator_version)
361                .or_insert(Vec::new())
362                .push(TxFundsWithdraw {
363                    tx_digest,
364                    reservations: tx_withdraws,
365                });
366        }
367        let mut receivers = FuturesUnordered::new();
368        {
369            let guard = self.address_funds_withdraw_scheduler.lock();
370            let withdraw_scheduler = guard
371                .as_ref()
372                .expect("Funds withdraw scheduler must be enabled if there are withdraws");
373            for (version, tx_withdraws) in withdraws {
374                receivers.extend(withdraw_scheduler.schedule_withdraws(WithdrawReservations {
375                    accumulator_version: version,
376                    withdraws: tx_withdraws,
377                }));
378            }
379            // guard will be dropped here
380        }
381        let scheduler = self.clone();
382        let epoch_store = epoch_store.clone();
383        spawn_monitored_task!(epoch_store.clone().within_alive_epoch(async move {
384            let mut cert_map = HashMap::new();
385            for (cert, env) in certs {
386                cert_map.insert(*cert.digest(), (cert, env));
387            }
388            while let Some(result) = receivers.next().await {
389                match result {
390                    Ok((tx_digest, status)) => match status {
391                        ScheduleStatus::InsufficientFunds => {
392                            assert_reachable!("tx cancelled, insufficient funds");
393                            debug!(
394                                ?tx_digest,
395                                "Funds withdraw scheduling result: Insufficient funds"
396                            );
397                            let (cert, env) = cert_map.remove(&tx_digest).expect("cert must exist");
398                            let env = env.with_insufficient_funds();
399                            scheduler.enqueue_transactions(vec![(cert, env)], &epoch_store);
400                        }
401                        ScheduleStatus::SufficientFunds => {
402                            assert_reachable!("tx scheduled, sufficient funds");
403                            debug!(?tx_digest, "Funds withdraw scheduling result: Success");
404                            let (cert, env) = cert_map.remove(&tx_digest).expect("cert must exist");
405                            scheduler.enqueue_transactions(vec![(cert, env)], &epoch_store);
406                        }
407                        ScheduleStatus::SkipSchedule => {
408                            assert_reachable!("tx withdrawal scheduling skipped");
409                            debug!(?tx_digest, "Skip scheduling funds withdraw");
410                        }
411                    },
412                    Err(e) => {
413                        error!("Withdraw scheduler stopped: {:?}", e);
414                    }
415                }
416            }
417        }));
418    }

420    fn schedule_tx_keys(
421        &self,
422        tx_with_keys: Vec<(TransactionKey, ExecutionEnv)>,
423        epoch_store: &Arc<AuthorityPerEpochStore>,
424    ) {
425        if tx_with_keys.is_empty() {
426            return;
427        }
428
429        let scheduler = self.clone();
430        let epoch_store = epoch_store.clone();
431        spawn_monitored_task!(epoch_store.clone().within_alive_epoch(async move {
432            let tx_keys: Vec<_> = tx_with_keys.iter().map(|(key, _)| key).cloned().collect();
433            let digests = epoch_store
434                .notify_read_tx_key_to_digest(&tx_keys)
435                .await
436                .expect("db error");
437            let transactions = scheduler
438                .transaction_cache_read
439                .multi_get_transaction_blocks(&digests)
440                .into_iter()
441                .map(|tx| {
442                    let tx = tx.expect("tx must exist").as_ref().clone();
443                    VerifiedExecutableTransaction::new_system(tx, epoch_store.epoch())
444                })
445                .zip(tx_with_keys.into_iter().map(|(_, env)| env))
446                .collect::<Vec<_>>();
447            scheduler.enqueue_transactions(transactions, &epoch_store);
448        }));
449    }

451    /// When we schedule a certificate, it should be impossible for it to have been executed in a
452    /// previous epoch.
453    #[cfg(debug_assertions)]
454    fn assert_cert_not_executed_previous_epochs(&self, cert: &VerifiedExecutableTransaction) {
455        let epoch = cert.epoch();
456        let digest = *cert.digest();
457        let digests = [digest];
458        let executed = self
459            .transaction_cache_read
460            .multi_get_executed_effects(&digests)
461            .pop()
462            .unwrap();
463        // Due to pruning, we may not always have an executed effects for the certificate
464        // even if it was executed. So this is a best-effort check.
465        if let Some(executed) = executed {
466            use sui_types::effects::TransactionEffectsAPI;
467
468            assert_eq!(
469                executed.executed_epoch(),
470                epoch,
471                "Transaction {} was executed in epoch {}, but scheduled again in epoch {}",
472                digest,
473                executed.executed_epoch(),
474                epoch
475            );
476        }
477    }

480impl ExecutionScheduler {
481    pub fn enqueue(
482        &self,
483        certs: Vec<(Schedulable, ExecutionEnv)>,
484        epoch_store: &Arc<AuthorityPerEpochStore>,
485    ) {
486        // schedule all transactions immediately
487        let mut ordinary_txns = Vec::with_capacity(certs.len());
488        let mut tx_with_keys = Vec::new();
489        let mut tx_with_withdraws = Vec::new();
490
491        for (schedulable, env) in certs {
492            match schedulable {
493                Schedulable::Transaction(tx) => {
494                    if tx.transaction_data().has_funds_withdrawals() {
495                        tx_with_withdraws.push((tx, env));
496                    } else {
497                        ordinary_txns.push((tx, env));
498                    }
499                }
500                s @ Schedulable::RandomnessStateUpdate(..) => {
501                    tx_with_keys.push((s.key(), env));
502                }
503                Schedulable::AccumulatorSettlement(_, _) => {
504                    unreachable!("handled by SettlementScheduler");
505                }
506                Schedulable::ConsensusCommitPrologue(_, _, _) => {
507                    // we only use Schedulable::ConsensusCommitPrologue as a temporary placeholder
508                    // during version assignment, by the time we schedule transactions it should be
509                    // converted to Schedulable::Transaction
510                    unreachable!("Schedulable::ConsensusCommitPrologue should not be enqueued");
511                }
512            }
513        }
514
515        self.enqueue_transactions(ordinary_txns, epoch_store);
516        self.schedule_tx_keys(tx_with_keys, epoch_store);
517        self.schedule_funds_withdraws(tx_with_withdraws, epoch_store);
518    }
519
520    pub fn enqueue_transactions(
521        &self,
522        certs: Vec<(VerifiedExecutableTransaction, ExecutionEnv)>,
523        epoch_store: &Arc<AuthorityPerEpochStore>,
524    ) {
525        // Filter out certificates from wrong epoch.
526        let certs: Vec<_> = certs
527            .into_iter()
528            .filter_map(|cert| {
529                if cert.0.epoch() == epoch_store.epoch() {
530                    #[cfg(debug_assertions)]
531                    self.assert_cert_not_executed_previous_epochs(&cert.0);
532
533                    Some(cert)
534                } else {
535                    debug_fatal!(
536                        "We should never enqueue certificate from wrong epoch. Expected={} Certificate={:?}",
537                        epoch_store.epoch(),
538                        cert.0.epoch()
539                    );
540                    None
541                }
542            })
543            .collect();
544        let digests: Vec<_> = certs.iter().map(|(cert, _)| *cert.digest()).collect();
545        let executed = self
546            .transaction_cache_read
547            .multi_get_executed_effects_digests(&digests);
548        let mut already_executed_certs_num = 0;
549        let pending_certs =
550            certs
551                .into_iter()
552                .zip(executed)
553                .filter_map(|((cert, execution_env), executed)| {
554                    if executed.is_none() {
555                        Some((cert, execution_env))
556                    } else {
557                        already_executed_certs_num += 1;
558                        None
559                    }
560                });
561
562        for (cert, execution_env) in pending_certs {
563            let scheduler = self.clone();
564            let epoch_store = epoch_store.clone();
565            spawn_monitored_task!(
566                epoch_store.within_alive_epoch(scheduler.schedule_transaction(
567                    cert,
568                    execution_env,
569                    &epoch_store,
570                ))
571            );
572        }
573
574        self.metrics
575            .transaction_manager_num_enqueued_certificates
576            .with_label_values(&["already_executed"])
577            .inc_by(already_executed_certs_num);
578    }

581        self.address_funds_withdraw_scheduler

583            .as_ref()

588    /// Reconfigure internal state at epoch start. This resets the funds withdraw scheduler
589    /// to the current accumulator root object version.
590    pub fn reconfigure(
591        &self,
592        new_epoch_store: &Arc<AuthorityPerEpochStore>,
593        account_funds_read: &Arc<dyn AccountFundsRead>,
594    ) {
595        let address_funds_withdraw_scheduler = Self::initialize_funds_withdraw_scheduler(
596            new_epoch_store,
597            &self.object_cache_read,
598            account_funds_read.clone(),
599            self.funds_withdraw_scheduler_type,
600            &self.address_funds_scheduler_metrics,
601        );
602        let mut guard = self.address_funds_withdraw_scheduler.lock();
603        if let Some(old_scheduler) = guard.as_ref() {
604            old_scheduler.close_epoch();
605        }
606        *guard = address_funds_withdraw_scheduler;
607        drop(guard);
608    }

669    #[allow(clippy::disallowed_methods)] // allow unbounded_channel()
670    fn make_execution_scheduler(
671        state: &AuthorityState,
672    ) -> (ExecutionScheduler, UnboundedReceiver<PendingCertificate>) {
673        // Create a new execution scheduler instead of reusing the authority's, to examine
674        // execution_scheduler output from rx_ready_certificates.
675        let (tx_ready_certificates, rx_ready_certificates) = unbounded_channel();
676        let registry = prometheus::Registry::new();
677        let execution_scheduler = ExecutionScheduler::new(
678            state.get_object_cache_reader().clone(),
679            state.get_account_funds_read().clone(),
680            state.get_transaction_cache_reader().clone(),
681            tx_ready_certificates,
682            &state.epoch_store_for_testing(),
683            FundsWithdrawSchedulerType::default(),
684            state.metrics.clone(),
685            &registry,
686        );
687
688        (execution_scheduler, rx_ready_certificates)
689    }

691    fn make_transaction(gas_object: Object, input: Vec<CallArg>) -> VerifiedExecutableTransaction {
692        // Use fake module, function, package and gas prices since they are irrelevant for testing
693        // execution scheduler.
694        let rgp = 100;
695        let (sender, keypair) = deterministic_random_account_key();
696        let transaction =
697            TestTransactionBuilder::new(sender, gas_object.compute_object_reference(), rgp)
698                .move_call(SUI_FRAMEWORK_PACKAGE_ID, "counter", "assert_value", input)
699                .build_and_sign(&keypair);
700        VerifiedExecutableTransaction::new_system(
701            VerifiedTransaction::new_unchecked(transaction),
702            0,
703        )
704    }

706    #[tokio::test(flavor = "current_thread", start_paused = true)]
707    async fn execution_scheduler_basics() {
708        // Initialize an authority state.
709        let (owner, _keypair) = deterministic_random_account_key();
710        let gas_objects: Vec<Object> = (0..10)
711            .map(|_| {
712                let gas_object_id = ObjectID::random();
713                Object::with_id_owner_for_testing(gas_object_id, owner)
714            })
715            .collect();
716        let state = init_state_with_objects(gas_objects.clone()).await;
717
718        // Create a new execution scheduler instead of reusing the authority's, to examine
719        // execution_scheduler output from rx_ready_certificates.
720        let (execution_scheduler, mut rx_ready_certificates) = make_execution_scheduler(&state);
721        // scheduler should output no transaction.
722        assert!(
723            rx_ready_certificates
724                .try_recv()
725                .is_err_and(|err| err == TryRecvError::Empty)
726        );
727        // scheduler should be empty at the beginning.
728        assert_eq!(execution_scheduler.num_pending_certificates(), 0);
729
730        // Enqueue empty vec should not crash.
731        execution_scheduler.enqueue_transactions(vec![], &state.epoch_store_for_testing());
732        // scheduler should output no transaction.
733        assert!(
734            rx_ready_certificates
735                .try_recv()
736                .is_err_and(|err| err == TryRecvError::Empty)
737        );
738
739        // Enqueue a transaction with existing gas object, empty input.
740        let transaction = make_transaction(gas_objects[0].clone(), vec![]);
741        let tx_start_time = Instant::now();
742        execution_scheduler.enqueue_transactions(
743            vec![(transaction.clone(), ExecutionEnv::new())],
744            &state.epoch_store_for_testing(),
745        );
746        // scheduler should output the transaction eventually.
747        let pending_certificate = rx_ready_certificates.recv().await.unwrap();
748
749        // Tests that pending certificate stats are recorded properly.
750        assert!(pending_certificate.stats.enqueue_time >= tx_start_time);
751        assert!(
752            pending_certificate.stats.ready_time.unwrap() >= pending_certificate.stats.enqueue_time
753        );
754
755        assert_eq!(execution_scheduler.num_pending_certificates(), 1);
756
757        // Predent we have just executed the transaction.
758        drop(pending_certificate);
759
760        // scheduler should be empty.
761        execution_scheduler.check_empty_for_testing();
762
763        // Enqueue a transaction with a new gas object, empty input.
764        let gas_object_new = Object::with_id_owner_version_for_testing(
765            ObjectID::random(),
766            0.into(),
767            Owner::AddressOwner(owner),
768        );
769        let transaction = make_transaction(gas_object_new.clone(), vec![]);
770        let tx_start_time = Instant::now();
771        execution_scheduler.enqueue_transactions(
772            vec![(transaction.clone(), ExecutionEnv::new())],
773            &state.epoch_store_for_testing(),
774        );
775        // scheduler should output no transaction yet.
776        sleep(Duration::from_secs(1)).await;
777        assert!(
778            rx_ready_certificates
779                .try_recv()
780                .is_err_and(|err| err == TryRecvError::Empty)
781        );
782
783        assert_eq!(execution_scheduler.num_pending_certificates(), 1);
784
785        // Duplicated enqueue is allowed.
786        execution_scheduler.enqueue_transactions(
787            vec![(transaction.clone(), ExecutionEnv::new())],
788            &state.epoch_store_for_testing(),
789        );
790        sleep(Duration::from_secs(1)).await;
791        assert!(
792            rx_ready_certificates
793                .try_recv()
794                .is_err_and(|err| err == TryRecvError::Empty)
795        );
796
797        assert_eq!(execution_scheduler.num_pending_certificates(), 2);
798
799        // Notify scheduler about availability of the gas object.
800        state
801            .get_cache_writer()
802            .write_object_entry_for_test(gas_object_new);
803        // scheduler should output the transaction eventually.
804        // We will see both the original and the duplicated transaction.
805        let pending_certificate = rx_ready_certificates.recv().await.unwrap();
806        let pending_certificate2 = rx_ready_certificates.recv().await.unwrap();
807        assert_eq!(
808            pending_certificate.certificate.digest(),
809            pending_certificate2.certificate.digest()
810        );
811
812        // Tests that pending certificate stats are recorded properly. The ready time should be
813        // 2 seconds apart from the enqueue time.
814        assert!(pending_certificate.stats.enqueue_time >= tx_start_time);
815        assert!(
816            pending_certificate.stats.ready_time.unwrap() - pending_certificate.stats.enqueue_time
817                >= Duration::from_secs(2)
818        );
819
820        // Predent we have just executed the transaction.
821        drop(pending_certificate);
822        drop(pending_certificate2);
823
824        // scheduler should be empty at the end.
825        execution_scheduler.check_empty_for_testing();
826    }

840        let (owner, _keypair) = deterministic_random_account_key();
841        let gas_objects: Vec<Object> = (0..10)
842            .map(|_| {
843                let gas_object_id = ObjectID::random();
844                Object::with_id_owner_for_testing(gas_object_id, owner)
845            })
846            .collect();

848        let initial_shared_version = shared_object.owner().start_version().unwrap();
849        let shared_object_2 = Object::shared_for_testing();
850        let initial_shared_version_2 = shared_object_2.owner().start_version().unwrap();
851
852        let state = init_state_with_objects(
853            [
854                gas_objects.clone(),
855                vec![shared_object.clone(), shared_object_2.clone()],

860
861        // Create a new execution scheduler instead of reusing the authority's, to examine
862        // execution_scheduler output from rx_ready_certificates.
863        let (execution_scheduler, mut rx_ready_certificates) = make_execution_scheduler(&state);
864        // scheduler should output no transaction.
865        assert!(rx_ready_certificates.try_recv().is_err());
866
867        // Enqueue two transactions with the same shared object input in read-only mode.
868        let shared_version = 1000.into();
869        let shared_object_arg_read = ObjectArg::SharedObject {
870            id: shared_object.id(),
871            initial_shared_version,
872            mutability: SharedObjectMutability::Immutable,
873        };
874        let transaction_read_0 = make_transaction(
875            gas_objects[0].clone(),
876            vec![CallArg::Object(shared_object_arg_read)],
877        );
878        let transaction_read_1 = make_transaction(
879            gas_objects[1].clone(),
880            vec![CallArg::Object(shared_object_arg_read)],
881        );
882        let tx_read_0_assigned_versions = vec![(
883            (
884                shared_object.id(),
885                shared_object.owner().start_version().unwrap(),
886            ),
887            shared_version,
888        )];
889        let tx_read_1_assigned_versions = vec![(
890            (
891                shared_object.id(),
892                shared_object.owner().start_version().unwrap(),
893            ),
894            shared_version,
895        )];
896
897        // Enqueue one transaction with the same shared object in mutable mode.
898        let shared_object_arg_default = ObjectArg::SharedObject {
899            id: shared_object.id(),
900            initial_shared_version,
901            mutability: SharedObjectMutability::Mutable,
902        };
903        let transaction_default = make_transaction(
904            gas_objects[2].clone(),
905            vec![CallArg::Object(shared_object_arg_default)],
906        );
907        let tx_default_assigned_versions = vec![(
908            (
909                shared_object.id(),
910                shared_object.owner().start_version().unwrap(),
911            ),
912            shared_version,
913        )];
914
915        // Enqueue one transaction with two readonly shared object inputs, `shared_object` and `shared_object_2`.
916        let shared_version_2 = 1000.into();
917        let shared_object_arg_read_2 = ObjectArg::SharedObject {
918            id: shared_object_2.id(),
919            initial_shared_version: initial_shared_version_2,
920            mutability: SharedObjectMutability::Immutable,
921        };
922        let transaction_read_2 = make_transaction(
923            gas_objects[3].clone(),
924            vec![
925                CallArg::Object(shared_object_arg_default),
926                CallArg::Object(shared_object_arg_read_2),
927            ],
928        );
929        let tx_read_2_assigned_versions = vec![
930            (
931                (
932                    shared_object.id(),
933                    shared_object.owner().start_version().unwrap(),
934                ),
935                shared_version,
936            ),
937            (
938                (
939                    shared_object_2.id(),
940                    shared_object_2.owner().start_version().unwrap(),
941                ),
942                shared_version_2,
943            ),
944        ];
945
946        execution_scheduler.enqueue_transactions(
947            vec![
948                (
949                    transaction_read_0.clone(),
950                    ExecutionEnv::new().with_assigned_versions(AssignedVersions::new(
951                        tx_read_0_assigned_versions,
952                        None,
953                    )),
954                ),
955                (
956                    transaction_read_1.clone(),
957                    ExecutionEnv::new().with_assigned_versions(AssignedVersions::new(
958                        tx_read_1_assigned_versions,
959                        None,
960                    )),
961                ),
962                (
963                    transaction_default.clone(),
964                    ExecutionEnv::new().with_assigned_versions(AssignedVersions::new(
965                        tx_default_assigned_versions,
966                        None,
967                    )),
968                ),
969                (
970                    transaction_read_2.clone(),
971                    ExecutionEnv::new().with_assigned_versions(AssignedVersions::new(
972                        tx_read_2_assigned_versions,
973                        None,
974                    )),
975                ),
976            ],
977            &state.epoch_store_for_testing(),
978        );
979
980        // scheduler should output no transaction yet.
981        sleep(Duration::from_secs(1)).await;
982        assert!(rx_ready_certificates.try_recv().is_err());
983
984        assert_eq!(execution_scheduler.num_pending_certificates(), 4);
985
986        // Notify scheduler about availability of the first shared object.
987        let mut new_shared_object = shared_object.clone();
988        new_shared_object
989            .data
990            .try_as_move_mut()
991            .unwrap()
992            .increment_version_to(shared_version_2);
993        state
994            .get_cache_writer()
995            .write_object_entry_for_test(new_shared_object);
996
997        // scheduler should output the 3 transactions that are only waiting for this object.
998        let tx_0 = rx_ready_certificates.recv().await.unwrap().certificate;
999        let tx_1 = rx_ready_certificates.recv().await.unwrap().certificate;
1000        let tx_2 = rx_ready_certificates.recv().await.unwrap().certificate;
1001        {
1002            let mut want_digests = vec![
1003                transaction_read_0.digest(),
1004                transaction_read_1.digest(),
1005                transaction_default.digest(),
1006            ];
1007            want_digests.sort();
1008            let mut got_digests = vec![tx_0.digest(), tx_1.digest(), tx_2.digest()];
1009            got_digests.sort();
1010            assert_eq!(want_digests, got_digests);
1011        }
1012
1013        sleep(Duration::from_secs(1)).await;
1014        assert!(rx_ready_certificates.try_recv().is_err());
1015
1016        assert_eq!(execution_scheduler.num_pending_certificates(), 1);
1017
1018        // Make shared_object_2 available.
1019        let mut new_shared_object_2 = shared_object_2.clone();
1020        new_shared_object_2
1021            .data
1022            .try_as_move_mut()
1023            .unwrap()
1024            .increment_version_to(shared_version_2);
1025        state
1026            .get_cache_writer()
1027            .write_object_entry_for_test(new_shared_object_2);
1028
1029        // Now, the transaction waiting for both shared objects can be executed.
1030        let tx_3 = rx_ready_certificates.recv().await.unwrap().certificate;
1031        assert_eq!(transaction_read_2.digest(), tx_3.digest());
1032
1033        sleep(Duration::from_secs(1)).await;
1034        assert!(rx_ready_certificates.try_recv().is_err());
1035
1036        execution_scheduler.check_empty_for_testing();
1037    }

1040    async fn execution_scheduler_receiving_notify_commit() {
1041        telemetry_subscribers::init_for_testing();
1042        // Initialize an authority state.
1043        let (owner, _keypair) = deterministic_random_account_key();
1044        let gas_objects: Vec<Object> = (0..10)
1045            .map(|_| {
1046                let gas_object_id = ObjectID::random();
1047                Object::with_id_owner_for_testing(gas_object_id, owner)
1048            })
1049            .collect();
1050        let state = init_state_with_objects(gas_objects.clone()).await;
1051
1052        // Create a new execution scheduler instead of reusing the authority's, to examine
1053        // execution_scheduler output from rx_ready_certificates.
1054        let (execution_scheduler, mut rx_ready_certificates) = make_execution_scheduler(&state);
1055        // scheduler should output no transaction.
1056        assert!(rx_ready_certificates.try_recv().is_err());
1057        // scheduler should be empty at the beginning.
1058        execution_scheduler.check_empty_for_testing();
1059
1060        let obj_id = ObjectID::random();
1061        let object_arguments: Vec<_> = (0..10)
1062            .map(|i| {
1063                let object = Object::with_id_owner_version_for_testing(
1064                    obj_id,
1065                    i.into(),
1066                    Owner::AddressOwner(owner),
1067                );
1068                // Every other transaction receives the object, and we create a run of multiple receives in
1069                // a row at the beginning to test that the scheduler doesn't get stuck in either configuration of:
1070                // ImmOrOwnedObject => Receiving,
1071                // Receiving => Receiving
1072                // Receiving => ImmOrOwnedObject
1073                // ImmOrOwnedObject => ImmOrOwnedObject is already tested as the default case on mainnet.
1074                let object_arg = if i % 2 == 0 || i == 3 {
1075                    ObjectArg::Receiving(object.compute_object_reference())
1076                } else {
1077                    ObjectArg::ImmOrOwnedObject(object.compute_object_reference())
1078                };
1079                let txn =
1080                    make_transaction(gas_objects[0].clone(), vec![CallArg::Object(object_arg)]);
1081                (object, txn)
1082            })
1083            .collect();
1084
1085        for (i, (_, txn)) in object_arguments.iter().enumerate() {
1086            // scheduler should output no transaction yet since waiting on receiving object or
1087            // ImmOrOwnedObject input.
1088            execution_scheduler.enqueue_transactions(
1089                vec![(txn.clone(), ExecutionEnv::new())],
1090                &state.epoch_store_for_testing(),
1091            );
1092            sleep(Duration::from_secs(1)).await;
1093            assert!(rx_ready_certificates.try_recv().is_err());
1094            assert_eq!(execution_scheduler.num_pending_certificates(), i + 1);
1095        }
1096
1097        // Now start to unravel the transactions by notifying that each subsequent
1098        // transaction has been processed.
1099        let len = object_arguments.len();
1100        for (i, (object, txn)) in object_arguments.into_iter().enumerate() {
1101            // Mark the object as available.
1102            // We should now eventually see the transaction as ready.
1103            state
1104                .get_cache_writer()
1105                .write_object_entry_for_test(object.clone());
1106
1107            // scheduler should output the transaction eventually now that the receiving object has become
1108            // available.
1109            rx_ready_certificates.recv().await.unwrap();
1110
1111            // Only one transaction at a time should become available though. So if we try to get
1112            // another one it should fail.
1113            sleep(Duration::from_secs(1)).await;
1114            assert!(rx_ready_certificates.try_recv().is_err());
1115
1116            // Notify the scheduler that the transaction has been processed.
1117            drop(txn);
1118
1119            // scheduler should now output another transaction to run since it the next version of that object
1120            // has become available.
1121            assert_eq!(execution_scheduler.num_pending_certificates(), len - i - 1);
1122        }
1123
1124        // After everything scheduler should be empty.
1125        execution_scheduler.check_empty_for_testing();
1126    }

1511    fn test_barrier_dependency_builder() {
1512        let make_transaction = |non_exclusive_writes: Vec<u32>, exclusive_writes: Vec<u32>| {
1513            assert!(
1514                non_exclusive_writes
1515                    .iter()
1516                    .all(|id| !exclusive_writes.contains(id))
1517            );
1518            assert!(
1519                exclusive_writes
1520                    .iter()
1521                    .all(|id| !non_exclusive_writes.contains(id))
1522            );
1523
1524            let non_exclusive_writes = non_exclusive_writes
1525                .into_iter()
1526                .map(|id| ObjectID::from_single_byte(id as u8));
1527            let exclusive_writes = exclusive_writes
1528                .into_iter()
1529                .map(|id| ObjectID::from_single_byte(id as u8));
1530            let mut builder = ProgrammableTransactionBuilder::new();
1531            for non_exclusive_write in non_exclusive_writes {
1532                builder
1533                    .obj(ObjectArg::SharedObject {
1534                        id: non_exclusive_write,
1535                        initial_shared_version: SequenceNumber::new(),
1536                        mutability: SharedObjectMutability::NonExclusiveWrite,
1537                    })
1538                    .unwrap();
1539            }
1540
1541            for exclusive_write in exclusive_writes {
1542                builder
1543                    .obj(ObjectArg::SharedObject {
1544                        id: exclusive_write,
1545                        initial_shared_version: SequenceNumber::new(),
1546                        mutability: SharedObjectMutability::Mutable,
1547                    })
1548                    .unwrap();
1549            }
1550
1551            let tx = TransactionKind::ProgrammableTransaction(builder.finish());
1552            let tx_data =
1553                TransactionData::new(tx, SuiAddress::default(), random_object_ref(), 1, 1);
1554            Transaction::from_data_and_signer(tx_data, vec![])
1555        };
1556
1557        // One non-exclusive write, one exclusive write.
1558        {
1559            let mut barrier_dependency_builder = BarrierDependencyBuilder::new();
1560            let tx1 = make_transaction(vec![1], vec![]);
1561            let tx2 = make_transaction(vec![], vec![1]);
1562
1563            let tx1_deps =
1564                barrier_dependency_builder.process_tx(*tx1.digest(), tx1.transaction_data());
1565            let tx2_deps =
1566                barrier_dependency_builder.process_tx(*tx2.digest(), tx2.transaction_data());
1567            assert!(tx1_deps.is_empty());
1568            assert_eq!(Vec::from_iter(tx2_deps), vec![*tx1.digest()]);
1569        }
1570
1571        // One transaction has non-exclusive writes to two different objects, and then becomes
1572        // a dependency of two barriers
1573        {
1574            let mut barrier_dependency_builder = BarrierDependencyBuilder::new();
1575            let tx1 = make_transaction(vec![1, 2], vec![]);
1576            let tx2 = make_transaction(vec![], vec![1]);
1577            let tx3 = make_transaction(vec![], vec![2]);
1578
1579            let tx1_deps =
1580                barrier_dependency_builder.process_tx(*tx1.digest(), tx1.transaction_data());
1581            let tx2_deps =
1582                barrier_dependency_builder.process_tx(*tx2.digest(), tx2.transaction_data());
1583            let tx3_deps =
1584                barrier_dependency_builder.process_tx(*tx3.digest(), tx3.transaction_data());
1585            assert!(tx1_deps.is_empty());
1586            assert_eq!(Vec::from_iter(tx2_deps), vec![*tx1.digest()]);
1587            assert_eq!(Vec::from_iter(tx3_deps), vec![*tx1.digest()]);
1588        }
1589
1590        // Ensure multiple-object dependences are merged
1591        {
1592            let mut barrier_dependency_builder = BarrierDependencyBuilder::new();
1593            let tx1 = make_transaction(vec![1], vec![]);
1594            let tx2 = make_transaction(vec![2], vec![]);
1595            let tx3 = make_transaction(vec![], vec![1, 2]);
1596
1597            let tx1_deps =
1598                barrier_dependency_builder.process_tx(*tx1.digest(), tx1.transaction_data());
1599            let tx2_deps =
1600                barrier_dependency_builder.process_tx(*tx2.digest(), tx2.transaction_data());
1601            let tx3_deps =
1602                barrier_dependency_builder.process_tx(*tx3.digest(), tx3.transaction_data());
1603            assert!(tx1_deps.is_empty());
1604            assert!(tx2_deps.is_empty());

1606        }
1607
1608        // Ensure dependency state is cleared
1609        {
1610            let mut barrier_dependency_builder = BarrierDependencyBuilder::new();
1611            let tx1 = make_transaction(vec![1], vec![]);
1612            let tx2 = make_transaction(vec![], vec![1]);
1613            let tx3 = make_transaction(vec![], vec![1]);

1619            let tx3_deps =
1620                barrier_dependency_builder.process_tx(*tx3.digest(), tx3.transaction_data());
1621            assert!(tx1_deps.is_empty());
