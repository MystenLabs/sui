12pub(crate) struct AccountState {
13    account_id: AccumulatorObjId,
14    /// Last known balance of the account (that is settled and persisted).
15    last_updated_balance: u128,
16    /// The version where last_updated_balance was read from.
17    last_updated_version: SequenceNumber,
18    reserved_funds: ReservedFunds,
19    /// Withdraws that could not yet be scheduled due to insufficient funds, and
20    /// hence have not reserved any funds yet. We track them so that we could schedule them
21    /// anytime we may have sufficient funds.
22    pending_reservations: VecDeque<Arc<PendingWithdraw>>,
23}
24
25#[derive(Default)]
26struct ReservedFunds {
27    /// The amount of funds that has been reserved for this account, for each accumulator version.
28    /// This is tracked so that we could add them back to the account funds when we settle the withdraws.
29    /// Since we always reserve in order of accumulator version, this vector is natually ordered, by
30    /// SequenceNumber.
31    reserved_funds: VecDeque<(SequenceNumber, u128)>,
32    /// Sum of all amounts in reserved_funds, saved as an optimization to avoid summing it
33    /// every time.
34    total_reserved_funds_amount: u128,
35}
36
37impl AccountState {
38    pub fn new(
39        account_id: AccumulatorObjId,
40        init_balance: u128,
41        init_version: SequenceNumber,
42    ) -> Self {
43        debug!(
44            "New account {:?} tracked at version {:?} with balance {:?}",
45            account_id,
46            init_version.value(),
47            init_balance,
48        );
49        Self {
50            account_id,
51            last_updated_balance: init_balance,
52            last_updated_version: init_version,
53            reserved_funds: ReservedFunds::default(),
54            pending_reservations: VecDeque::new(),
55        }
56    }
57
58    pub fn try_reserve_new_withdraw(
59        &mut self,
60        new_withdraw: Arc<PendingWithdraw>,
61        last_settled_version: SequenceNumber,
62    ) -> bool {
63        self.pending_reservations.push_back(new_withdraw);
64        let len = self.pending_reservations.len();
65        // If there are existing blocking withdraws, we cannot schedule this withdraw either.
66        // Otherwise we schedule it immediately.
67        if len > 1 {
68            // The withdraws are scheduled in order of accumulator version, so the previous withdraw's version
69            // must be less or equal to the current withdraw's version.
70            assert!(
71                self.pending_reservations[len - 2].accumulator_version()
72                    <= self.pending_reservations[len - 1].accumulator_version()
73            );
74            false
75        } else {
76            self.try_reserve_front(last_settled_version)
77        }
78    }
79
80    /// Try to process the first withdraw in the pending_reservations queue.
81    /// Returns true if the processing was successful, i.e. we reached a deterministic
82    /// decision on whether that withdraw can be satisfied for this account.
83    fn try_reserve_front(&mut self, last_settled_version: SequenceNumber) -> bool {
84        let Some(pending_withdraw) = self.pending_reservations.pop_front() else {
85            return false;
86        };
87        assert!(
88            pending_withdraw.accumulator_version() >= self.last_updated_version,
89            "pending_withdraw.accumulator_version() = {:?}, self.last_updated_version = {:?}",
90            pending_withdraw.accumulator_version(),
91            self.last_updated_version
92        );
93        assert!(pending_withdraw.accumulator_version() >= last_settled_version);
94        let to_reserve = pending_withdraw.pending_amount(&self.account_id);
95        if self.reserved_funds.total_reserved_funds_amount() + to_reserve
96            <= self.last_updated_balance
97        {
98            debug!(
99                "Successfully reserved {:?} for account {:?} at version {:?}",
100                to_reserve,
101                self.account_id,
102                pending_withdraw.accumulator_version().value(),
103            );
104            self.reserved_funds
105                .add_reserved_fund(pending_withdraw.accumulator_version(), to_reserve);
106            pending_withdraw.remove_pending_account(&self.account_id);
107            return true;
108        }
109        if pending_withdraw.accumulator_version() == last_settled_version {
110            pending_withdraw.notify_insufficient_funds();
111            return true;
112        }
113        // Failed to reserve, put the pending withdraw back to the front of the queue.
114        assert_reachable!("withdraw must wait for settlement to be scheduled");
115        self.pending_reservations.push_front(pending_withdraw);
116        false
117    }
118
119    pub fn settle_funds(&mut self, settled: i128, next_version: SequenceNumber) {
120        debug!(
121            "Settling funds for account {:?} with amount {:?} at version {:?}",
122            self.account_id,
123            settled,
124            next_version.value(),
125        );
126        // If the next_version is less or equal to the last_updated_version,
127        // it means the state tracked for this account is already up to date.
128        // There is no need to update its balance.
129        if next_version > self.last_updated_version {
130            let new_balance = (self.last_updated_balance as i128)
131                .checked_add(settled)
132                .unwrap();
133            assert!(new_balance >= 0);
134            self.last_updated_balance = new_balance as u128;
135            self.last_updated_version = next_version;
136        }
137        self.reserved_funds.settle_fund(next_version);
138        while self.try_reserve_front(next_version) {}
139    }
140
141    pub fn is_empty(&self) -> bool {
142        self.pending_reservations.is_empty() && self.reserved_funds.reserved_funds.is_empty()
143    }
144}
145
146impl ReservedFunds {
147    fn total_reserved_funds_amount(&self) -> u128 {
148        self.total_reserved_funds_amount
149    }
150
151    fn add_reserved_fund(&mut self, accumulator_version: SequenceNumber, amount: u128) {
152        self.total_reserved_funds_amount += amount;
153
154        if let Some(entry) = self.reserved_funds.back_mut() {
155            if entry.0 == accumulator_version {
156                entry.1 = entry.1.checked_add(amount).unwrap();
157                return;
158            }
159            // Reservations are processed in order, so we must never see the version
160            // decrease.
161            assert!(entry.0 < accumulator_version);
162        }
163        self.reserved_funds.push_back((accumulator_version, amount));
164    }
165
166    fn settle_fund(&mut self, next_version: SequenceNumber) {
167        if let Some(entry) = self.reserved_funds.front().copied() {
168            if entry.0.next() == next_version {
169                self.reserved_funds.pop_front();
170                self.total_reserved_funds_amount = self
171                    .total_reserved_funds_amount
172                    .checked_sub(entry.1)
173                    .unwrap();
174            } else {
175                // If there exists reserved funds at even earlier versions, they must have
176                // already been settled in the past.
177                assert!(entry.0 >= next_version);
178            }
179        }
180    }
181}

194    fn make_account_id(byte: u8) -> AccumulatorObjId {

198    fn make_pending_withdraw(
199        account_id: AccumulatorObjId,
200        amount: u64,
201        version: SequenceNumber,
202    ) -> (Arc<PendingWithdraw>, oneshot::Receiver<ScheduleStatus>) {
203        let (tx, rx) = oneshot::channel();
204        let withdraw = TxFundsWithdraw {
205            tx_digest: TransactionDigest::random(),
206            reservations: BTreeMap::from([(account_id, amount)]),
207        };
208        (PendingWithdraw::new(version, withdraw, tx), rx)
209    }

405    async fn test_settle_funds_processes_pending_reservations() {
406        let account_id = make_account_id(1);
407        let mut state = AccountState::new(account_id, 100, SequenceNumber::from_u64(5));
408
409        // Add a withdraw that needs 200 (more than current balance)
410        let (pending, rx) = make_pending_withdraw(account_id, 200, SequenceNumber::from_u64(6));
411        // Try to reserve - should fail because insufficient funds
412        assert!(!state.try_reserve_new_withdraw(pending, SequenceNumber::from_u64(5)));
413        assert_eq!(state.pending_reservations.len(), 1);

415        // Settle with +150, bringing balance to 250 at version 6

417
418        // The pending reservation should have been automatically processed
419        assert!(state.pending_reservations.is_empty());
420        assert_eq!(
421            *state.reserved_funds.reserved_funds.front().unwrap(),
422            (SequenceNumber::from_u64(6), 200)
423        );
424
425        let result = rx.await.unwrap();
426        assert_eq!(result, ScheduleStatus::SufficientFunds);
427    }
