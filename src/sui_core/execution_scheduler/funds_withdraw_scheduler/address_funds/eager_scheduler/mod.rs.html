34pub(crate) struct EagerFundsWithdrawScheduler {
35    funds_read: Arc<dyn AccountFundsRead>,
36    inner_state: Arc<Mutex<InnerState>>,
37}
38
39struct InnerState {
40    /// For each address funds account that we have seen withdraws through `schedule_withdraws`,
41    /// we track the current state of that account, and only remove it from the map after
42    /// we have settled all withdraws for that account.
43    tracked_accounts: BTreeMap<AccumulatorObjId, AccountState>,
44    /// Tracks all the address funds accounts that have a withdraw transaction tracked,
45    /// mapping from the accumulator version that the withdraw transaction reads from.
46    /// If a withdraw transaction needs to withdraw from account O at version V,
47    /// we must process and settle that withdraw transaction whenever we settle all transactions
48    /// scheduled for version V.
49    pending_settlements: BTreeMap<SequenceNumber, BTreeSet<AccumulatorObjId>>,
50    /// The current version of the accumulator object known to the scheduler.
51    accumulator_version: SequenceNumber,
52}
53
54impl EagerFundsWithdrawScheduler {
55    pub fn new(
56        funds_read: Arc<dyn AccountFundsRead>,
57        starting_accumulator_version: SequenceNumber,
58    ) -> Arc<Self> {
59        Arc::new(Self {
60            funds_read,
61            inner_state: Arc::new(Mutex::new(InnerState {
62                tracked_accounts: BTreeMap::new(),
63                pending_settlements: BTreeMap::new(),
64                accumulator_version: starting_accumulator_version,
65            })),
66        })
67    }
68}

70#[async_trait::async_trait]
71impl FundsWithdrawSchedulerTrait for EagerFundsWithdrawScheduler {
72    fn schedule_withdraws(
73        &self,
74        reservations: WithdrawReservations,
75    ) -> BTreeMap<TransactionDigest, ScheduleResult> {
76        let mut inner_state = self.inner_state.lock();
77        let cur_accumulator_version = inner_state.accumulator_version;
78        if reservations.accumulator_version < cur_accumulator_version {
79            // This accumulator version is already settled.
80            // There is no need to schedule the withdraws.
81            return reservations.notify_skip_schedule();
82        }
83        let all_accounts = reservations.all_accounts();
84        let untracked_accounts = all_accounts
85            .iter()
86            .filter(|account_id| !inner_state.tracked_accounts.contains_key(account_id));
87        let mut init_balances = BTreeMap::new();
88        for account_id in untracked_accounts {
89            // TODO: We can warm up the cache prior to holding the lock.
90            let (balance, version) = self.funds_read.get_latest_account_amount(account_id);
91            if version > reservations.accumulator_version {
92                return reservations.notify_skip_schedule();
93            }
94            init_balances.insert(account_id, (balance, version));
95        }
96        let mut results = BTreeMap::new();
97        for withdraw in reservations.withdraws {
98            let (sender, mut receiver) = oneshot::channel();
99            let accounts = withdraw.reservations.keys().cloned().collect::<Vec<_>>();
100            let pending_withdraw =
101                PendingWithdraw::new(reservations.accumulator_version, withdraw, sender);
102            for account_id in accounts {
103                let entry = inner_state
104                    .tracked_accounts
105                    .entry(account_id)
106                    .or_insert_with(|| {
107                        let (balance, version) = init_balances.get(&account_id).cloned().unwrap();
108                        AccountState::new(account_id, balance, version)
109                    });
110                entry.try_reserve_new_withdraw(pending_withdraw.clone(), cur_accumulator_version);
111            }
112            if pending_withdraw.is_scheduled() {
113                let result = receiver.try_recv().unwrap();
114                results.insert(
115                    pending_withdraw.tx_digest(),
116                    ScheduleResult::ScheduleResult(result),
117                );
118            } else {
119                results.insert(
120                    pending_withdraw.tx_digest(),
121                    ScheduleResult::Pending(receiver),
122                );
123            }
124        }
125        let old = inner_state
126            .pending_settlements
127            .insert(reservations.accumulator_version, all_accounts);
128        assert!(old.is_none());
129        results
130    }
131
132    fn settle_funds(&self, settlement: FundsSettlement) {
133        let next_accumulator_version = settlement.next_accumulator_version;
134        let mut inner_state = self.inner_state.lock();
135        if next_accumulator_version <= inner_state.accumulator_version {
136            // This accumulator version is already settled.
137            // There is no need to settle the funds.
138            debug!(
139                next_accumulator_version =? next_accumulator_version.value(),
140                "Skipping settlement since it is already settled",
141            );
142            return;
143        }
144        assert_eq!(
145            next_accumulator_version,
146            inner_state.accumulator_version.next()
147        );
148        let cleanup_version = inner_state.accumulator_version;
149        inner_state.accumulator_version = next_accumulator_version;
150        debug!(
151            "Bumping accumulator version from {:?} to {:?}",
152            cleanup_version.value(),
153            next_accumulator_version.value(),
154        );
155
156        // A settlement bumps the accumulator version from `cleanup_version` to `next_accumulator_version`.
157        // We must process the following types of accounts:
158        // 1. Accounts that had withdraws scheduled at version `cleanup_version`. These withdraws
159        // are now settled and we know their exact funds changes.
160        // 2. Accounts that had withdraws scheduled at version `next_accumulator_version`. These withdraws
161        // if not yet scheduled, can now be deterministically scheduled, since
162        // we have the final state before them.
163        // 3. Accounts that have funds changes through settle_funds, this can include
164        // both withdraws and deposits.
165        let mut affected_accounts = inner_state
166            .pending_settlements
167            .remove(&cleanup_version)
168            .unwrap_or_default();
169        // Since we just bumped the accumulator version, any withdraws scheduled that depend
170        // on this version can now be deterministically scheduled.
171        if let Some(current_version_accounts) = inner_state
172            .pending_settlements
173            .get(&next_accumulator_version)
174        {
175            affected_accounts.extend(current_version_accounts);
176        }
177        affected_accounts.extend(settlement.funds_changes.keys().cloned());
178
179        for object_id in affected_accounts {
180            let Some(account_state) = inner_state.tracked_accounts.get_mut(&object_id) else {
181                continue;
182            };
183            account_state.settle_funds(
184                settlement
185                    .funds_changes
186                    .get(&object_id)
187                    .cloned()
188                    .unwrap_or_default(),
189                next_accumulator_version,
190            );
191
192            if account_state.is_empty() {
193                debug!(
194                    account_id = ?object_id,
195                    "Removing account state since it is empty",
196                );
197                inner_state.tracked_accounts.remove(&object_id);
198            }
199        }
200        debug!("Settled accumulator version {:?}", next_accumulator_version);
201    }
202
203    fn close_epoch(&self) {
204        debug!("Closing epoch in EagerFundsWithdrawScheduler");
205    }
206
207    fn num_tracked_accounts(&self) -> usize {
208        self.inner_state.lock().tracked_accounts.len()
209    }
210
211    #[cfg(test)]
212    fn get_current_accumulator_version(&self) -> SequenceNumber {
213        self.inner_state.lock().accumulator_version
214    }
