4use prometheus::{
5    Histogram, IntCounter, IntCounterVec, IntGauge, Registry, register_histogram_with_registry,
6    register_int_counter_vec_with_registry, register_int_counter_with_registry,
7    register_int_gauge_with_registry,
8};
9
10const SCHEDULING_LATENCY_SEC_BUCKETS: &[f64] = &[
11    0.001, 0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0, 30.0, 60.0,
12];
13
14pub(crate) struct AddressFundsSchedulerMetrics {
15    pub highest_scheduled_version: IntGauge,
16    pub highest_settled_version: IntGauge,
17    pub pending_schedules: IntGauge,
18    pub pending_schedule_latency: Histogram,
19    pub tracked_accounts: IntGauge,
20    pub total_scheduled: IntCounter,
21    pub total_settled: IntCounter,
22    pub schedule_result: IntCounterVec,
23}
24
25impl AddressFundsSchedulerMetrics {
26    pub fn new(registry: &Registry) -> Self {
27        Self {
28            highest_scheduled_version: register_int_gauge_with_registry!(
29                "address_funds_highest_scheduled_version",
30                "Highest accumulator version seen in schedule_withdraws",
31                registry,
32            )
33            .unwrap(),
34            highest_settled_version: register_int_gauge_with_registry!(
35                "address_funds_highest_settled_version",
36                "Highest accumulator version settled via settle_funds",
37                registry,
38            )
39            .unwrap(),
40            pending_schedules: register_int_gauge_with_registry!(
41                "address_funds_pending_schedules",
42                "In-flight pending schedules not yet resolved",
43                registry,
44            )
45            .unwrap(),
46            pending_schedule_latency: register_histogram_with_registry!(
47                "address_funds_pending_schedule_latency",
48                "Latency in seconds from pending schedule creation to resolution",
49                SCHEDULING_LATENCY_SEC_BUCKETS.to_vec(),
50                registry,
51            )
52            .unwrap(),
53            tracked_accounts: register_int_gauge_with_registry!(
54                "address_funds_tracked_accounts",
55                "Number of accounts being tracked by the eager scheduler",
56                registry,
57            )
58            .unwrap(),
59            total_scheduled: register_int_counter_with_registry!(
60                "address_funds_total_scheduled",
61                "Total withdrawal transactions scheduled",
62                registry,
63            )
64            .unwrap(),
65            total_settled: register_int_counter_with_registry!(
66                "address_funds_total_settled",
67                "Total settlement versions processed",
68                registry,
69            )
70            .unwrap(),
71            schedule_result: register_int_counter_vec_with_registry!(
72                "address_funds_schedule_result",
73                "Count of address funds scheduling decisions by outcome",
74                &["result"],
75                registry,
76            )
77            .unwrap(),
78        }
79    }
80}
