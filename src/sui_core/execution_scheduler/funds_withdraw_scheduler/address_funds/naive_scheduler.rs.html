18/// A naive implementation of the funds withdraw scheduler that does not attempt to optimize the scheduling.
19/// For each withdraw reservation, it will always wait until the dependent accumulator object is available,
20/// and then check if the funds are sufficient.
21/// This implementation is simple and easy to understand, but it is not efficient.
22/// It is only used to unblock further development of the funds withdraw scheduler.
23pub(crate) struct NaiveFundsWithdrawScheduler {
24    funds_read: Arc<dyn AccountFundsRead>,
25    accumulator_version_sender: watch::Sender<SequenceNumber>,
26    // We must keep a receiver alive to make sure sends go through and can update the last settled version.
27    accumulator_version_receiver: watch::Receiver<SequenceNumber>,
28}
29
30impl NaiveFundsWithdrawScheduler {
31    pub fn new(
32        funds_read: Arc<dyn AccountFundsRead>,
33        cur_accumulator_version: SequenceNumber,
34    ) -> Arc<Self> {
35        let (accumulator_version_sender, accumulator_version_receiver) =
36            watch::channel(cur_accumulator_version);
37        Arc::new(Self {
38            funds_read,
39            accumulator_version_sender,
40            accumulator_version_receiver,
41        })
42    }
43
44    fn process_withdraws(
45        funds_read: &dyn AccountFundsRead,
46        reservations: WithdrawReservations,
47    ) -> BTreeMap<TransactionDigest, ScheduleResult> {
48        // Map from each account ID that we have seen so far to the current
49        // remaining funds for reservation.
50        let mut cur_funds = BTreeMap::new();
51        let all_accounts = reservations.all_accounts();
52        for account_id in all_accounts {
53            // TODO: We can warm up the cache prior to holding the lock.
54            let (balance, version) = funds_read.get_latest_account_amount(&account_id);
55            if version > reservations.accumulator_version {
56                return reservations.notify_skip_schedule();
57            }
58            cur_funds.insert(account_id, balance);
59        }
60        reservations
61            .withdraws
62            .into_iter()
63            .map(|withdraw| {
64                // Try to reserve all withdraws in this transaction.
65                // Note that this is not atomic, so it is possible that we reserve some withdraws and not others.
66                // This is intentional to be semantically consistent with the eager scheduler.
67                let mut success = true;
68                for (object_id, reservation) in &withdraw.reservations {
69                    let entry = cur_funds.get_mut(object_id).unwrap();
70                    if *entry < *reservation as u128 {
71                        debug!(
72                            tx_digest =? withdraw.tx_digest,
73                            "Insufficient funds for {:?}. Requested: {:?}, Available: {:?}",
74                            object_id, reservation, entry
75                        );
76                        success = false;
77                    } else {
78                        debug!(
79                            tx_digest =? withdraw.tx_digest,
80                            "Successfully reserved {:?} for account {:?}",
81                            reservation, object_id
82                        );
83                        *entry -= *reservation as u128;
84                    }
85                }
86                let status = if success {
87                    debug!(
88                        tx_digest =? withdraw.tx_digest,
89                        "Successfully reserved all withdraws"
90                    );
91                    ScheduleStatus::SufficientFunds
92                } else {
93                    ScheduleStatus::InsufficientFunds
94                };
95                (withdraw.tx_digest, ScheduleResult::ScheduleResult(status))
96            })
97            .collect()
98    }
99}

101#[async_trait::async_trait]
102impl FundsWithdrawSchedulerTrait for NaiveFundsWithdrawScheduler {
103    #[instrument(level = "debug", skip_all, fields(withdraw_accumulator_version = ?reservations.accumulator_version.value()))]
104    fn schedule_withdraws(
105        &self,
106        reservations: WithdrawReservations,
107    ) -> BTreeMap<TransactionDigest, ScheduleResult> {
108        let mut receiver = self.accumulator_version_sender.subscribe();
109        let cur_version = *receiver.borrow_and_update();
110        if cur_version > reservations.accumulator_version {
111            return reservations.notify_skip_schedule();
112        }
113        if cur_version == reservations.accumulator_version {
114            return Self::process_withdraws(self.funds_read.as_ref(), reservations);
115        }
116
117        let (mut senders, receivers): (BTreeMap<_, _>, BTreeMap<_, _>) = reservations
118            .withdraws
119            .iter()
120            .map(|withdraw| {
121                let (sender, receiver) = oneshot::channel();
122                (
123                    (withdraw.tx_digest, sender),
124                    (withdraw.tx_digest, ScheduleResult::Pending(receiver)),
125                )
126            })
127            .unzip();
128
129        let funds_read = self.funds_read.clone();
130        tokio::spawn(async move {
131            while *receiver.borrow_and_update() < reservations.accumulator_version {
132                debug!("Waiting for the dependent accumulator version to be settled");
133                if receiver.changed().await.is_err() {
134                    tracing::error!(
135                        "Accumulator version receiver channel closed while waiting for accumulator version"
136                    );
137                    return;
138                }
139            }
140            if *receiver.borrow() > reservations.accumulator_version {
141                for sender in senders.into_values() {
142                    let _ = sender.send(ScheduleStatus::SkipSchedule);
143                }
144                return;
145            }
146            let results = Self::process_withdraws(funds_read.as_ref(), reservations);
147            for (tx_digest, result) in results {
148                let sender = senders.remove(&tx_digest).unwrap();
149                // unwrap is safe because process_withdraws always return a finalized result.
150                let _ = sender.send(result.unwrap_status());
151            }
152        });
153
154        receivers
155    }
156
157    // We don't use the funds changes information in the naive scheduler.
158    // Instead, the withdraw scheduling always reads the funds state from storage.
159    fn settle_funds(&self, settlement: FundsSettlement) {
160        let cur_accumulator_version = *self.accumulator_version_receiver.borrow();
161        let next_version = cur_accumulator_version.next();
162
163        if settlement.next_accumulator_version < next_version {
164            // This accumulator version is already settled.
165            // There is no need to settle the funds.
166            debug!(
167                next_accumulator_version =? settlement.next_accumulator_version.value(),
168                "Skipping settlement since it is already settled",
169            );
170            return;
171        }
172
173        debug!(
174            settled_accumulator_version =? cur_accumulator_version.value(),
175            next_accumulator_version =? next_version.value(),
176            "Settling funds",
177        );
178        assert_eq!(next_version, settlement.next_accumulator_version);
179        let _ = self.accumulator_version_sender.send(next_version);
180    }
181
182    fn close_epoch(&self) {
183        debug!("Closing epoch in NaiveFundsWithdrawScheduler");
184    }
185
186    fn num_tracked_accounts(&self) -> usize {
187        0
188    }
189
190    #[cfg(test)]
191    fn get_current_accumulator_version(&self) -> SequenceNumber {
192        *self.accumulator_version_receiver.borrow()
193    }
194}
