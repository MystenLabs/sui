21pub(crate) trait FundsWithdrawSchedulerTrait: Send + Sync {
22    fn schedule_withdraws(
23        &self,
24        reservations: WithdrawReservations,
25    ) -> BTreeMap<TransactionDigest, ScheduleResult>;
26    fn settle_funds(&self, settlement: FundsSettlement);
27    fn close_epoch(&self);
28    fn num_tracked_accounts(&self) -> usize;
29    #[cfg(test)]
30    fn get_current_accumulator_version(&self) -> SequenceNumber;
31}
32
33struct WithdrawEvent {
34    pub reservations: WithdrawReservations,
35    pub senders: BTreeMap<TransactionDigest, oneshot::Sender<(TransactionDigest, ScheduleStatus)>>,
36}
37
38#[derive(Clone)]
39pub(crate) struct FundsWithdrawScheduler {
40    scheduler: Arc<dyn FundsWithdrawSchedulerTrait>,
41    /// Use channels to process withdraws and settlements asynchronously without blocking the caller.
42    withdraw_sender: UnboundedSender<WithdrawEvent>,
43    settlement_sender: UnboundedSender<FundsSettlement>,
44}
45
46impl WithdrawEvent {
47    fn new(
48        reservations: WithdrawReservations,
49    ) -> (
50        Self,
51        FuturesUnordered<oneshot::Receiver<(TransactionDigest, ScheduleStatus)>>,
52    ) {
53        let (senders, receivers) = reservations
54            .withdraws
55            .iter()
56            .map(|withdraw| {
57                let (sender, receiver) = oneshot::channel();
58                ((withdraw.tx_digest, sender), receiver)
59            })
60            .unzip();
61        (
62            Self {
63                reservations,
64                senders,
65            },
66            receivers,
67        )
68    }
69}
70
71impl FundsWithdrawScheduler {
72    pub fn new(
73        funds_read: Arc<dyn AccountFundsRead>,
74        starting_accumulator_version: SequenceNumber,
75        scheduler_type: FundsWithdrawSchedulerType,
76        metrics: Arc<AddressFundsSchedulerMetrics>,
77    ) -> Self {
78        let scheduler: Arc<dyn FundsWithdrawSchedulerTrait> = match scheduler_type {
79            FundsWithdrawSchedulerType::Naive => {
80                NaiveFundsWithdrawScheduler::new(funds_read, starting_accumulator_version)
81            }
82            FundsWithdrawSchedulerType::Eager => {
83                EagerFundsWithdrawScheduler::new(funds_read, starting_accumulator_version)
84            }
85        };
86        let (withdraw_sender, withdraw_receiver) =
87            unbounded_channel("withdraw_scheduler_withdraws");
88        let (settlement_sender, settlement_receiver) =
89            unbounded_channel("withdraw_scheduler_settlements");
90        // Pass only the scheduler to the spawned tasks, not the senders. This ensures that when
91        // the FundsWithdrawScheduler is dropped (dropping the senders), the channels close and tasks exit.
92        tokio::spawn(Self::process_withdraw_task(
93            scheduler.clone(),
94            withdraw_receiver,
95            metrics.clone(),
96        ));
97        tokio::spawn(Self::process_settlement_task(
98            scheduler.clone(),
99            settlement_receiver,
100            metrics,
101        ));
102        Self {
103            scheduler,
104            withdraw_sender,
105            settlement_sender,
106        }
107    }
108
109    /// This function will be called at most once per consensus commit batch that all reads the same root accumulator version.
110    /// If a consensus commit batch does not contain any withdraw reservations, it can skip calling this function.
111    /// It must be called sequentially in order to correctly schedule withdraws.
112    pub fn schedule_withdraws(
113        &self,
114        withdraw_reservations: WithdrawReservations,
115    ) -> FuturesUnordered<oneshot::Receiver<(TransactionDigest, ScheduleStatus)>> {
116        // TODO: Add debug assertion that withdraws are scheduled in order.
117        let (event, receivers) = WithdrawEvent::new(withdraw_reservations);
118        if let Err(err) = self.withdraw_sender.send(event) {
119            tracing::error!("Failed to send withdraw reservations: {:?}", err);
120        }
121        receivers
122    }
123
124    /// This function is called whenever a settlement transaction is executed.
125    /// It is only called from checkpoint builder, once for each accumulator version, in order.
126    pub fn settle_funds(&self, settlement: FundsSettlement) {
127        if let Err(err) = self.settlement_sender.send(settlement) {
128            tracing::error!("Failed to send funds settlement: {}", err);
129        }
130    }
131
132    pub fn close_epoch(&self) {
133        debug!("Closing epoch for funds withdraw scheduler");
134        self.scheduler.close_epoch();
135    }
136
137    async fn process_withdraw_task(
138        scheduler: Arc<dyn FundsWithdrawSchedulerTrait>,
139        mut withdraw_receiver: UnboundedReceiver<WithdrawEvent>,
140        metrics: Arc<AddressFundsSchedulerMetrics>,
141    ) {
142        while let Some(event) = withdraw_receiver.recv().await {
143            let WithdrawEvent {
144                reservations,
145                mut senders,
146            } = event;
147            debug!(
148                withdraw_accumulator_version =? reservations.accumulator_version.value(),
149                "Processing withdraws: {:?}",
150                reservations.withdraws,
151            );
152
153            let num_txns = reservations.withdraws.len();
154            let accumulator_version = reservations.accumulator_version;
155            let results = scheduler.schedule_withdraws(reservations);
156
157            metrics
158                .highest_scheduled_version
159                .set(accumulator_version.value() as i64);
160            metrics.total_scheduled.inc_by(num_txns as u64);
161            metrics
162                .tracked_accounts
163                .set(scheduler.num_tracked_accounts() as i64);
164
165            for (tx_digest, result) in results {
166                let original_sender = senders.remove(&tx_digest).unwrap();
167                match result {
168                    ScheduleResult::ScheduleResult(status) => {
169                        debug!(?tx_digest, ?status, "Scheduling result");
170                        metrics
171                            .schedule_result
172                            .with_label_values(&[status_label(status)])
173                            .inc();
174                        let _ = original_sender.send((tx_digest, status));
175                    }
176                    ScheduleResult::Pending(receiver) => {
177                        metrics.pending_schedules.inc();
178                        let timer = metrics.pending_schedule_latency.start_timer();
179                        let pending_metrics = metrics.clone();
180                        tokio::spawn(async move {
181                            match receiver.await {
182                                Ok(status) => {
183                                    debug!(?tx_digest, ?status, "Scheduling result (pending)");
184                                    pending_metrics
185                                        .schedule_result
186                                        .with_label_values(&[status_label(status)])
187                                        .inc();
188                                    let _ = original_sender.send((tx_digest, status));
189                                }
190                                Err(_) => {
191                                    debug!(?tx_digest, "Failed to receive scheduling result");
192                                }
193                            }
194                            timer.observe_duration();
195                            pending_metrics.pending_schedules.dec();
196                        });
197                    }
198                }
199            }
200        }
201        tracing::info!("Funds withdraw receiver closed");
202    }
203
204    async fn process_settlement_task(
205        scheduler: Arc<dyn FundsWithdrawSchedulerTrait>,
206        mut settlement_receiver: UnboundedReceiver<FundsSettlement>,
207        metrics: Arc<AddressFundsSchedulerMetrics>,
208    ) {
209        while let Some(settlement) = settlement_receiver.recv().await {
210            debug!(
211                next_accumulator_version =? settlement.next_accumulator_version.value(),
212                "Settling funds changes: {:?}",
213                settlement.funds_changes,
214            );
215            let next_version = settlement.next_accumulator_version;
216            scheduler.settle_funds(settlement);
217
218            metrics
219                .highest_settled_version
220                .set(next_version.value() as i64);
221            metrics.total_settled.inc();
222            metrics
223                .tracked_accounts
224                .set(scheduler.num_tracked_accounts() as i64);
225        }
226        tracing::info!("Funds settlement receiver closed");
227    }
228}
229
230fn status_label(status: ScheduleStatus) -> &'static str {
231    match status {
232        ScheduleStatus::SufficientFunds => "sufficient",
233        ScheduleStatus::InsufficientFunds => "insufficient",
234        ScheduleStatus::SkipSchedule => "skip",
235    }
