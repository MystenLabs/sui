25pub struct MockConsensusClient {
26    tx_sender: mpsc::Sender<ConsensusTransaction>,
27    _consensus_handle: JoinHandle<()>,
28}
29
30pub enum ConsensusMode {
31    // ConsensusClient does absolutely nothing when receiving a transaction
32    Noop,
33    // ConsensusClient directly sequences the transaction into the store.
34    DirectSequencing,
35}
36
37impl MockConsensusClient {
38    pub fn new(validator: Weak<AuthorityState>, consensus_mode: ConsensusMode) -> Self {
39        let (tx_sender, tx_receiver) = mpsc::channel(1000000);
40        let _consensus_handle = Self::run(validator, tx_receiver, consensus_mode);
41        Self {
42            tx_sender,
43            _consensus_handle,
44        }
45    }
46
47    pub fn run(
48        validator: Weak<AuthorityState>,
49        tx_receiver: mpsc::Receiver<ConsensusTransaction>,
50        consensus_mode: ConsensusMode,
51    ) -> JoinHandle<()> {
52        tokio::spawn(async move { Self::run_impl(validator, tx_receiver, consensus_mode).await })
53    }
54
55    async fn run_impl(
56        validator: Weak<AuthorityState>,
57        mut tx_receiver: mpsc::Receiver<ConsensusTransaction>,
58        consensus_mode: ConsensusMode,
59    ) {
60        let checkpoint_service = Arc::new(CheckpointServiceNoop {});
61        let authority_metrics = Arc::new(AuthorityMetrics::new(&Registry::new()));
62        while let Some(tx) = tx_receiver.recv().await {
63            let Some(validator) = validator.upgrade() else {
64                debug!("validator shut down; exiting MockConsensusClient");
65                return;
66            };
67            let epoch_store = validator.epoch_store_for_testing();
68            let env = match consensus_mode {
69                ConsensusMode::Noop => ExecutionEnv::new(),
70                ConsensusMode::DirectSequencing => {
71                    let (_, assigned_versions) = epoch_store
72                        .process_consensus_transactions_for_tests(
73                            vec![SequencedConsensusTransaction::new_test(tx.clone())],
74                            &checkpoint_service,
75                            validator.get_object_cache_reader().as_ref(),
76                            &authority_metrics,
77                            true,
78                        )
79                        .await
80                        .unwrap();
81                    let assigned_versions = assigned_versions
82                        .0
83                        .into_iter()
84                        .next()
85                        .map(|(_, v)| v)
86                        .unwrap_or_default();
87                    ExecutionEnv::new().with_assigned_versions(assigned_versions)
88                }
89            };
90            match &tx.kind {
91                ConsensusTransactionKind::CertifiedTransaction(tx) => {
92                    if tx.is_consensus_tx() {
93                        validator.execution_scheduler().enqueue(
94                            vec![(
95                                VerifiedExecutableTransaction::new_from_certificate(
96                                    VerifiedCertificate::new_unchecked(*tx.clone()),
97                                )
98                                .into(),
99                                env,
100                            )],
101                            &epoch_store,
102                        );
103                    }
104                }
105                ConsensusTransactionKind::UserTransaction(tx) => {
106                    if tx.is_consensus_tx() {
107                        validator.execution_scheduler().enqueue(
108                            vec![(
109                                VerifiedExecutableTransaction::new_from_consensus(
110                                    VerifiedTransaction::new_unchecked(*tx.clone()),
111                                    0,
112                                )
113                                .into(),
114                                env,
115                            )],
116                            &epoch_store,
117                        );
118                    }
119                }
120                _ => {}
121            }
122        }
123    }
124
125    fn submit_impl(
126        &self,
127        transactions: &[ConsensusTransaction],
128    ) -> SuiResult<(Vec<ConsensusPosition>, BlockStatusReceiver)> {
129        // TODO: maybe support multi-transactions and remove this check
130        assert!(transactions.len() == 1);
131        let transaction = &transactions[0];
132        self.tx_sender
133            .try_send(transaction.clone())
134            .map_err(|_| SuiError::from("MockConsensusClient channel overflowed"))?;
135        // TODO(fastpath): Add some way to simulate consensus positions across blocks
136        Ok((
137            vec![ConsensusPosition {
138                epoch: EpochId::MIN,
139                block: BlockRef::MIN,
140                index: 0,
141            }],
142            with_block_status(consensus_core::BlockStatus::Sequenced(BlockRef::MIN)),
143        ))
144    }
145}
146
147impl SubmitToConsensus for MockConsensusClient {
148    fn submit_to_consensus(
149        &self,
150        transactions: &[ConsensusTransaction],
151        _epoch_store: &Arc<AuthorityPerEpochStore>,
152    ) -> SuiResult {
153        self.submit_impl(transactions).map(|_response| ())
154    }
155
156    fn submit_best_effort(
157        &self,
158        transaction: &ConsensusTransaction,
159        _epoch_store: &Arc<AuthorityPerEpochStore>,
160        _timeout: Duration,
161    ) -> SuiResult {
162        self.submit_impl(std::slice::from_ref(transaction))
163            .map(|_response| ())
164    }
165}

167#[async_trait::async_trait]
168impl ConsensusClient for MockConsensusClient {
169    async fn submit(
170        &self,
171        transactions: &[ConsensusTransaction],
172        _epoch_store: &Arc<AuthorityPerEpochStore>,
173    ) -> SuiResult<(Vec<ConsensusPosition>, BlockStatusReceiver)> {
174        self.submit_impl(transactions)
175    }
176}

178pub(crate) fn with_block_status(status: consensus_core::BlockStatus) -> BlockStatusReceiver {
179    let (tx, rx) = oneshot::channel();
180    tx.send(status).ok();
181    rx
