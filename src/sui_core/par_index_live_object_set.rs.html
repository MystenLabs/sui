4use crate::authority::AuthorityStore;
5use crate::authority::authority_store_tables::LiveObject;
6use std::time::Instant;
7use sui_types::base_types::ObjectID;
8use sui_types::object::Object;

10use tracing::info;

12/// Make `LiveObjectIndexer`s for parallel indexing of the live object set
13pub trait ParMakeLiveObjectIndexer: Sync {
14    type ObjectIndexer: LiveObjectIndexer;
15
16    fn make_live_object_indexer(&self) -> Self::ObjectIndexer;
17}
18
19/// Represents an instance of a indexer that operates on a subset of the live object set
20pub trait LiveObjectIndexer {
21    /// Called on each object in the range of the live object set this indexer task is responsible
22    /// for.
23    fn index_object(&mut self, object: Object) -> Result<(), StorageError>;
24
25    /// Called once the range of objects this indexer task is responsible for have been processed
26    /// by calling `index_object`.
27    fn finish(self) -> Result<(), StorageError>;
28}
29
30/// Utility for iterating over, and indexing, the live object set in parallel
31///
32/// This is done by dividing the addressable ObjectID space into smaller, disjoint sets and
33/// operating on each set in parallel in a separate thread. User's will need to implement the
34/// `ParMakeLiveObjectIndexer` trait which will be used to make N `LiveObjectIndexer`s which will
35/// then process one of the disjoint parts of the live object set.
36#[tracing::instrument(skip_all)]
37pub fn par_index_live_object_set<T: ParMakeLiveObjectIndexer>(
38    authority_store: &AuthorityStore,
39    make_indexer: &T,
40) -> Result<(), StorageError> {
41    info!("Indexing Live Object Set");
42    let start_time = Instant::now();
43    std::thread::scope(|s| -> Result<(), StorageError> {
44        let mut threads = Vec::new();
45        const BITS: u8 = 5;
46        for index in 0u8..(1 << BITS) {
47            threads.push(s.spawn(move || {
48                let object_indexer = make_indexer.make_live_object_indexer();
49                live_object_set_index_task(index, BITS, authority_store, object_indexer)
50            }));
51        }
52
53        // join threads
54        for thread in threads {
55            thread.join().unwrap()?;
56        }
57
58        Ok(())
59    })?;
60
61    info!(
62        "Indexing Live Object Set took {} seconds",
63        start_time.elapsed().as_secs()
64    );
65
66    Ok(())
67}
68
69#[tracing::instrument(skip(authority_store, object_indexer))]
70fn live_object_set_index_task<T: LiveObjectIndexer>(
71    task_id: u8,
72    bits: u8,
73    authority_store: &AuthorityStore,
74    mut object_indexer: T,
75) -> Result<(), StorageError> {
76    let mut id_bytes = [0; ObjectID::LENGTH];
77    id_bytes[0] = task_id << (8 - bits);
78    let start_id = ObjectID::new(id_bytes);
79
80    id_bytes[0] |= (1 << (8 - bits)) - 1;
81    for element in id_bytes.iter_mut().skip(1) {
82        *element = u8::MAX;
83    }
84    let end_id = ObjectID::new(id_bytes);
85
86    let mut object_scanned: u64 = 0;
87    for object in authority_store
88        .perpetual_tables
89        .range_iter_live_object_set(Some(start_id), Some(end_id), false)
90        .filter_map(LiveObject::to_normal)
91    {
92        object_scanned += 1;
93        if object_scanned.is_multiple_of(2_000_000) {
94            info!(
95                "[Index] Task {}: object scanned: {}",
96                task_id, object_scanned
97            );
98        }
99
100        object_indexer.index_object(object)?
101    }
102
103    object_indexer.finish()?;
104
105    Ok(())
106}
