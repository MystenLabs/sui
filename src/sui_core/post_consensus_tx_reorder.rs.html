4use crate::consensus_handler::{
5    SequencedConsensusTransactionKind, VerifiedSequencedConsensusTransaction,
6};
7use mysten_metrics::monitored_scope;
8use sui_protocol_config::ConsensusTransactionOrdering;
9use sui_types::{
10    executable_transaction::VerifiedExecutableTransaction,
11    messages_consensus::{ConsensusTransaction, ConsensusTransactionKind},
12    transaction::TransactionDataAPI as _,
13};
14
15pub struct PostConsensusTxReorder {}
16
17impl PostConsensusTxReorder {
18    pub fn reorder(
19        transactions: &mut [VerifiedSequencedConsensusTransaction],
20        kind: ConsensusTransactionOrdering,
21    ) {
22        // TODO: make the reordering algorithm richer and depend on object hotness as well.
23        // Order transactions based on their gas prices. System transactions without gas price
24        // are put to the beginning of the sequenced_transactions vector.
25        match kind {
26            ConsensusTransactionOrdering::ByGasPrice => Self::order_by_gas_price(transactions),
27            ConsensusTransactionOrdering::None => (),
28        }
29    }
30
31    fn order_by_gas_price(transactions: &mut [VerifiedSequencedConsensusTransaction]) {
32        let _scope = monitored_scope("ConsensusCommitHandler::order_by_gas_price");
33        transactions.sort_by_key(|txn| {
34            // Reverse order, so that transactions with higher gas price are put to the beginning.
35            std::cmp::Reverse({
36                match &txn.0.transaction {
37                    SequencedConsensusTransactionKind::External(ConsensusTransaction {
38                        tracking_id: _,
39                        kind: ConsensusTransactionKind::CertifiedTransaction(cert),
40                    }) => cert.gas_price(),
41                    SequencedConsensusTransactionKind::External(ConsensusTransaction {
42                        tracking_id: _,
43                        kind: ConsensusTransactionKind::UserTransaction(txn),
44                    }) => txn.transaction_data().gas_price(),
45                    // Non-user transactions are considered to have gas price of MAX u64 and are
46                    // put to the beginning.
47                    _ => u64::MAX,
48                }
49            })
50        })
51    }
52
53    // TODO: Remove the above versions and rename these without _v2
54    pub fn reorder_v2(
55        transactions: &mut [VerifiedExecutableTransaction],
56        kind: ConsensusTransactionOrdering,
57    ) {
58        match kind {
59            ConsensusTransactionOrdering::ByGasPrice => Self::order_by_gas_price_v2(transactions),
60            ConsensusTransactionOrdering::None => (),
61        }
62    }
63
64    fn order_by_gas_price_v2(transactions: &mut [VerifiedExecutableTransaction]) {
65        let _scope = monitored_scope("ConsensusCommitHandler::order_by_gas_price");
66        transactions.sort_by_key(|tx| {
67            // Reverse order, so that transactions with higher gas price are put to the beginning.
68            std::cmp::Reverse(tx.transaction_data().gas_price())
69        });
70    }
71}
