61/// Get available memory, respecting cgroup limits in containerized environments
62fn get_available_memory() -> u64 {
63    // RefreshKind::nothing().with_memory() avoids collecting other, slower stats
64    let mut sys = System::new_with_specifics(
65        RefreshKind::nothing().with_memory(MemoryRefreshKind::everything()),
66    );
67    sys.refresh_memory();
68
69    // Check if we have cgroup limits
70    if let Some(cgroup_limits) = sys.cgroup_limits() {
71        let memory_limit = cgroup_limits.total_memory;
72        // cgroup_limits.total_memory is 0 when there's no limit
73        if memory_limit > 0 {
74            debug!("Using cgroup memory limit: {} bytes", memory_limit);
75            return memory_limit;
76        }
77    }
78
79    // Fall back to system memory if no cgroup limits found
80    // sysinfo 0.35 already reports bytes (not KiB like older versions)
81    let total_memory_bytes = sys.total_memory();
82    debug!("Using system memory: {} bytes", total_memory_bytes);
83    total_memory_bytes
84}

179impl BalanceIndexInfo {
180    fn invert(self) -> Self {
181        // Check for potential overflow when negating i128::MIN
182        assert!(
183            self.balance_delta != i128::MIN,
184            "Cannot invert balance_delta: would overflow i128"
185        );
186
187        Self {
188            balance_delta: -self.balance_delta,
189        }
190    }
191
192    fn merge_delta(&mut self, other: &Self) {
193        self.balance_delta += other.balance_delta;
194    }
195}

197impl From<BalanceIndexInfo> for sui_types::storage::BalanceInfo {
198    fn from(index_info: BalanceIndexInfo) -> Self {
199        // Note: We represent balance deltas as i128 to simplify merging positive and negative updates.
200        // Be aware: Move doesn’t enforce a one-time-witness (OTW) pattern when creating a Supply<T>.
201        // Anyone can call `sui::balance::create_supply` and mint unbounded supply, potentially pushing
202        // total balances over u64::MAX. To avoid crashing the indexer, we clamp the merged value instead
203        // of panicking on overflow. This has the unfortunate consequence of making bugs in the index
204        // harder to detect, but is a necessary trade-off to avoid creating a DOS attack vector.
205        let balance = index_info.balance_delta.clamp(0, u64::MAX as i128) as u64;
206        sui_types::storage::BalanceInfo { balance }
207    }
208}

234    if let Some(filter) = compaction_filter {
235        options.options.set_compaction_filter(
236            "events_by_stream",

238                Ok(decision) => decision,
239                Err(e) => {
240                    warn!(
241                        "Failed to parse event key during compaction: {}, key: {:?}",
242                        e, key
243                    );
244                    Decision::Remove
245                }
246            },
247        );
248    }
249    options

252fn balance_delta_merge_operator(
253    _key: &[u8],
254    existing_val: Option<&[u8]>,
255    operands: &MergeOperands,
256) -> Option<Vec<u8>> {
257    let mut result = existing_val
258        .map(|v| {
259            bcs::from_bytes::<BalanceIndexInfo>(v)
260                .expect("Failed to deserialize BalanceIndexInfo from RocksDB - data corruption.")
261        })
262        .unwrap_or_default();
263
264    for operand in operands.iter() {
265        let delta = bcs::from_bytes::<BalanceIndexInfo>(operand)
266            .expect("Failed to deserialize BalanceIndexInfo from RocksDB - data corruption.");
267        result.merge_delta(&delta);
268    }
269    Some(
270        bcs::to_bytes(&result)
271            .expect("Failed to deserialize BalanceIndexInfo from RocksDB - data corruption."),
272    )
273}

275fn balance_compaction_filter(_level: u32, _key: &[u8], value: &[u8]) -> Decision {
276    let balance_info = match bcs::from_bytes::<BalanceIndexInfo>(value) {
277        Ok(info) => info,
278        Err(_) => return Decision::Keep,
279    };
280
281    if balance_info.balance_delta == 0 {
282        Decision::Remove
283    } else {
284        Decision::Keep
285    }
286}

394/// Compaction filter for automatic pruning of old authenticated events during RocksDB compaction.
395#[derive(Clone)]
396pub struct EventsCompactionFilter {
397    pruning_watermark: Arc<std::sync::atomic::AtomicU64>,
398}
399
400impl EventsCompactionFilter {
401    pub fn new(pruning_watermark: Arc<std::sync::atomic::AtomicU64>) -> Self {
402        Self { pruning_watermark }
403    }
404
405    pub fn filter(&self, key: &[u8], _value: &[u8]) -> anyhow::Result<Decision> {
406        let event_key: EventIndexKey = bcs::from_bytes(key)?;
407        let watermark = self
408            .pruning_watermark
409            .load(std::sync::atomic::Ordering::Relaxed);
410
411        if event_key.checkpoint_seq <= watermark {
412            Ok(Decision::Remove)
413        } else {
414            Ok(Decision::Keep)
415        }
416    }
417}

500    // Check if the index watermark is behind the highets_executed watermark.
501    fn is_indexed_watermark_out_of_date(&self, checkpoint_store: &CheckpointStore) -> bool {
502        let highest_executed_checkpint = checkpoint_store
503            .get_highest_executed_checkpoint_seq_number()
504            .ok()
505            .flatten();
506        let watermark = self.watermark.get(&Watermark::Indexed).ok().flatten();

508    }

510    #[tracing::instrument(skip_all)]
511    fn init(
512        &mut self,
513        authority_store: &AuthorityStore,
514        checkpoint_store: &CheckpointStore,
515        _epoch_store: &AuthorityPerEpochStore,
516        _package_store: &Arc<dyn BackingPackageStore + Send + Sync>,
517        batch_size_limit: usize,
518        rpc_config: &sui_config::RpcConfig,
519    ) -> Result<(), StorageError> {
520        info!("Initializing RPC indexes");
521
522        let highest_executed_checkpint =
523            checkpoint_store.get_highest_executed_checkpoint_seq_number()?;
524        let lowest_available_checkpoint = checkpoint_store
525            .get_highest_pruned_checkpoint_seq_number()?
526            .map(|c| c.saturating_add(1))
527            .unwrap_or(0);
528        let lowest_available_checkpoint_objects = authority_store
529            .perpetual_tables
530            .get_highest_pruned_checkpoint()?
531            .map(|c| c.saturating_add(1))
532            .unwrap_or(0);
533        // Doing backfill requires processing objects so we have to restrict our backfill range
534        // to the range of checkpoints that we have objects for.
535        let lowest_available_checkpoint =
536            lowest_available_checkpoint.max(lowest_available_checkpoint_objects);
537
538        let checkpoint_range = highest_executed_checkpint.map(|highest_executed_checkpint| {
539            lowest_available_checkpoint..=highest_executed_checkpint
540        });
541
542        if let Some(checkpoint_range) = checkpoint_range {
543            self.index_existing_transactions(
544                authority_store,
545                checkpoint_store,
546                checkpoint_range,
547                rpc_config,
548            )?;
549        }
550
551        self.initialize_current_epoch(authority_store, checkpoint_store)?;
552
553        // Only index live objects if genesis checkpoint has been executed.
554        // If genesis hasn't been executed yet, the objects will be properly indexed
555        // as checkpoints are processed through the normal checkpoint execution path.
556        if highest_executed_checkpint.is_some() {
557            let coin_index = Mutex::new(HashMap::new());
558
559            let make_live_object_indexer = RpcParLiveObjectSetIndexer {
560                tables: self,
561                coin_index: &coin_index,
562                batch_size_limit,
563            };
564
565            crate::par_index_live_object_set::par_index_live_object_set(
566                authority_store,
567                &make_live_object_indexer,
568            )?;
569
570            self.coin.multi_insert(coin_index.into_inner().unwrap())?;
571        }
572
573        self.watermark.insert(
574            &Watermark::Indexed,
575            &highest_executed_checkpint.unwrap_or(0),
576        )?;
577
578        self.meta.insert(
579            &(),
580            &MetadataInfo {
581                version: CURRENT_DB_VERSION,
582            },
583        )?;
584
585        info!("Finished initializing RPC indexes");
586
587        Ok(())
588    }

590    #[tracing::instrument(skip(self, authority_store, checkpoint_store, rpc_config))]
591    fn index_existing_transactions(
592        &mut self,
593        authority_store: &AuthorityStore,
594        checkpoint_store: &CheckpointStore,
595        checkpoint_range: std::ops::RangeInclusive<u64>,
596        rpc_config: &sui_config::RpcConfig,
597    ) -> Result<(), StorageError> {
598        info!(
599            "Indexing {} checkpoints in range {checkpoint_range:?}",
600            checkpoint_range.size_hint().0
601        );
602        let start_time = Instant::now();
603
604        checkpoint_range.into_par_iter().try_for_each(|seq| {
605            let load_events = rpc_config.authenticated_events_indexing();
606            let checkpoint_data = sparse_checkpoint_data_for_backfill(
607                authority_store,
608                checkpoint_store,
609                seq,
610                load_events,
611            )?;
612
613            let mut batch = self.transactions.batch();
614
615            self.index_epoch(&checkpoint_data, &mut batch)?;
616            self.index_transactions(&checkpoint_data, &mut batch, load_events)?;
617
618            batch
619                .write_opt(&(bulk_ingestion_write_options()))
620                .map_err(StorageError::from)
621        })?;
622
623        info!(
624            "Indexing checkpoints took {} seconds",
625            start_time.elapsed().as_secs()
626        );
627        Ok(())
628    }

630    /// Prune data from this Index
631    fn prune(
632        &self,
633        pruned_checkpoint_watermark: u64,
634        checkpoint_contents_to_prune: &[CheckpointContents],
635    ) -> Result<(), TypedStoreError> {
636        let mut batch = self.transactions.batch();
637
638        let transactions_to_prune = checkpoint_contents_to_prune
639            .iter()
640            .flat_map(|contents| contents.iter().map(|digests| digests.transaction));
641
642        batch.delete_batch(&self.transactions, transactions_to_prune)?;
643        batch.insert_batch(
644            &self.watermark,
645            [(Watermark::Pruned, pruned_checkpoint_watermark)],
646        )?;
647
648        batch.write()
649    }

667            checkpoint,
668            &mut batch,
669            rpc_config.authenticated_events_indexing(),
670        )?;
671        self.index_objects(checkpoint, &mut batch)?;
672
673        batch.insert_batch(
674            &self.watermark,
675            [(
676                Watermark::Indexed,
677                checkpoint.checkpoint_summary.sequence_number,
678            )],
679        )?;
680
681        debug!(
682            checkpoint = checkpoint.checkpoint_summary.sequence_number,
683            "finished indexing checkpoint"
684        );
685
686        Ok(batch)
687    }

689    fn index_transaction_events(
690        &self,
691        tx: &sui_types::full_checkpoint_content::CheckpointTransaction,
692        checkpoint_seq: u64,
693        tx_idx: u32,
694        batch: &mut typed_store::rocks::DBBatch,
695    ) -> Result<(), StorageError> {
696        let acc_events = tx.effects.accumulator_events();
697        if acc_events.is_empty() {
698            return Ok(());
699        }
700
701        let mut entries: Vec<(EventIndexKey, ())> = Vec::new();
702        for acc in acc_events {
703            if let Some(stream_id) =
704                sui_types::accumulator_root::stream_id_from_accumulator_event(&acc)
705                && let AccumulatorValue::EventDigest(idx, _d) = acc.write.value
706            {
707                let key = EventIndexKey {
708                    stream_id,
709                    checkpoint_seq,
710                    transaction_idx: tx_idx,
711                    event_index: idx as u32,
712                };
713                entries.push((key, ()));
714            }
715        }
716
717        if !entries.is_empty() {
718            batch.insert_batch(&self.events_by_stream, entries)?;
719        }
720        Ok(())
721    }

723    fn index_epoch(
724        &self,
725        checkpoint: &CheckpointData,
726        batch: &mut typed_store::rocks::DBBatch,
727    ) -> Result<(), StorageError> {
728        let Some(epoch_info) = checkpoint.epoch_info()? else {
729            return Ok(());
730        };
731        if epoch_info.epoch > 0 {
732            let prev_epoch = epoch_info.epoch - 1;
733            let mut current_epoch = self.epochs.get(&prev_epoch)?.unwrap_or_default();
734            current_epoch.epoch = prev_epoch; // set this incase there wasn't an entry
735            current_epoch.end_timestamp_ms = epoch_info.start_timestamp_ms;
736            current_epoch.end_checkpoint = epoch_info.start_checkpoint.map(|sq| sq - 1);
737            batch.insert_batch(&self.epochs, [(prev_epoch, current_epoch)])?;
738        }
739        batch.insert_batch(&self.epochs, [(epoch_info.epoch, epoch_info)])?;
740        Ok(())
741    }

743    // After attempting to reindex past epochs, ensure that the current epoch is at least partially
744    // initalized
745    fn initialize_current_epoch(
746        &mut self,
747        authority_store: &AuthorityStore,
748        checkpoint_store: &CheckpointStore,
749    ) -> Result<(), StorageError> {
750        let Some(checkpoint) = checkpoint_store.get_highest_executed_checkpoint()? else {
751            return Ok(());
752        };
753
754        let system_state = sui_types::sui_system_state::get_sui_system_state(authority_store)
755            .map_err(|e| StorageError::custom(format!("Failed to find system state: {e}")))?;
756
757        let mut epoch = self.epochs.get(&checkpoint.epoch)?.unwrap_or_default();
758        epoch.epoch = checkpoint.epoch;
759
760        if epoch.protocol_version.is_none() {
761            epoch.protocol_version = Some(system_state.protocol_version());
762        }
763
764        if epoch.start_timestamp_ms.is_none() {
765            epoch.start_timestamp_ms = Some(system_state.epoch_start_timestamp_ms());
766        }
767
768        if epoch.reference_gas_price.is_none() {
769            epoch.reference_gas_price = Some(system_state.reference_gas_price());
770        }
771
772        if epoch.system_state.is_none() {
773            epoch.system_state = Some(system_state);
774        }
775
776        self.epochs.insert(&epoch.epoch, &epoch)?;
777
778        Ok(())
779    }

781    fn index_transactions(
782        &self,
783        checkpoint: &CheckpointData,
784        batch: &mut typed_store::rocks::DBBatch,
785        index_events: bool,
786    ) -> Result<(), StorageError> {
787        let cp = checkpoint.checkpoint_summary.sequence_number;
788
789        for (tx_idx, tx) in checkpoint.transactions.iter().enumerate() {
790            let info = TransactionInfo::new(
791                tx.transaction.transaction_data(),
792                &tx.effects,
793                &tx.input_objects,
794                &tx.output_objects,
795                cp,
796            );
797
798            let digest = tx.transaction.digest();
799            batch.insert_batch(&self.transactions, [(digest, info)])?;
800
801            if index_events {
802                self.index_transaction_events(tx, cp, tx_idx as u32, batch)?;
803            }
804        }
805
806        Ok(())
807    }

809    fn index_objects(
810        &self,
811        checkpoint: &CheckpointData,
812        batch: &mut typed_store::rocks::DBBatch,
813    ) -> Result<(), StorageError> {
814        let mut coin_index: HashMap<CoinIndexKey, CoinIndexInfo> = HashMap::new();
815        let mut balance_changes: HashMap<BalanceKey, BalanceIndexInfo> = HashMap::new();
816        let mut package_version_index: Vec<(PackageVersionKey, PackageVersionInfo)> = vec![];
817
818        for tx in &checkpoint.transactions {
819            // determine changes from removed objects
820            for removed_object in tx.removed_objects_pre_version() {
821                match removed_object.owner() {
822                    Owner::AddressOwner(owner) | Owner::ConsensusAddressOwner { owner, .. } => {
823                        Self::track_coin_balance_change(
824                            removed_object,
825                            owner,
826                            true,
827                            &mut balance_changes,
828                        )?;
829
830                        let owner_key = OwnerIndexKey::from_object(removed_object);
831                        batch.delete_batch(&self.owner, [owner_key])?;
832                    }
833                    Owner::ObjectOwner(object_id) => {
834                        batch.delete_batch(
835                            &self.dynamic_field,
836                            [DynamicFieldKey::new(*object_id, removed_object.id())],
837                        )?;
838                    }
839                    Owner::Shared { .. } | Owner::Immutable => {}
840                }
841            }
842
843            // determine changes from changed objects
844            for (object, old_object) in tx.changed_objects() {
845                if let Some(old_object) = old_object {
846                    match old_object.owner() {
847                        Owner::AddressOwner(owner) | Owner::ConsensusAddressOwner { owner, .. } => {
848                            Self::track_coin_balance_change(
849                                old_object,
850                                owner,
851                                true,
852                                &mut balance_changes,
853                            )?;
854
855                            let owner_key = OwnerIndexKey::from_object(old_object);
856                            batch.delete_batch(&self.owner, [owner_key])?;
857                        }
858
859                        Owner::ObjectOwner(object_id) => {
860                            if old_object.owner() != object.owner() {
861                                batch.delete_batch(
862                                    &self.dynamic_field,
863                                    [DynamicFieldKey::new(*object_id, old_object.id())],
864                                )?;
865                            }
866                        }
867
868                        Owner::Shared { .. } | Owner::Immutable => {}
869                    }
870                }
871
872                match object.owner() {
873                    Owner::AddressOwner(owner) | Owner::ConsensusAddressOwner { owner, .. } => {
874                        Self::track_coin_balance_change(
875                            object,
876                            owner,
877                            false,
878                            &mut balance_changes,
879                        )?;
880                        let owner_key = OwnerIndexKey::from_object(object);
881                        let owner_info = OwnerIndexInfo::new(object);
882                        batch.insert_batch(&self.owner, [(owner_key, owner_info)])?;
883                    }
884                    Owner::ObjectOwner(parent) => {
885                        if should_index_dynamic_field(object) {
886                            let field_key = DynamicFieldKey::new(*parent, object.id());
887                            batch.insert_batch(&self.dynamic_field, [(field_key, ())])?;
888                        }
889                    }
890                    Owner::Shared { .. } | Owner::Immutable => {}
891                }
892                if let Some((key, info)) = Self::extract_version_if_package(object) {
893                    package_version_index.push((key, info));
894                }
895            }
896
897            // coin indexing
898            //
899            // coin indexing relies on the fact that CoinMetadata and TreasuryCap are created in
900            // the same transaction so we don't need to worry about overriding any older value
901            // that may exist in the database (because there necessarily cannot be).
902            for (key, value) in tx.created_objects().flat_map(try_create_coin_index_info) {
903                use std::collections::hash_map::Entry;
904
905                match coin_index.entry(key) {
906                    Entry::Occupied(mut o) => {
907                        o.get_mut().merge(value);
908                    }
909                    Entry::Vacant(v) => {
910                        v.insert(value);
911                    }
912                }
913            }
914        }
915
916        batch.insert_batch(&self.coin, coin_index)?;
917        batch.partial_merge_batch(&self.balance, balance_changes)?;
918        batch.insert_batch(&self.package_version, package_version_index)?;
919
920        Ok(())
921    }

964        &self,
965        owner: SuiAddress,
966        object_type: Option<StructTag>,
967        cursor: Option<OwnerIndexKey>,
968    ) -> Result<
969        impl Iterator<Item = Result<(OwnerIndexKey, OwnerIndexInfo), TypedStoreError>> + '_,
970        TypedStoreError,
971    > {
972        // TODO can we figure out how to pass a raw byte array as a cursor?
973        let lower_bound = cursor.unwrap_or_else(|| OwnerIndexKey {
974            owner,
975            object_type: object_type
976                .clone()
977                .unwrap_or_else(|| "0x0::a::a".parse::<StructTag>().unwrap()),
978            inverted_balance: None,
979            object_id: ObjectID::ZERO,
980        });
981
982        Ok(self
983            .owner
984            .safe_iter_with_bounds(Some(lower_bound), None)
985            .take_while(move |item| {
986                // If there's an error let if flow through
987                let Ok((key, _)) = item else {
988                    return true;
989                };
990
991                // Only take if owner matches
992                key.owner == owner
993                    // and if an object type was supplied that the type matches
994                    && object_type
995                        .as_ref()
996                        .map(|ty| {
997                            ty.address == key.object_type.address
998                                && ty.module == key.object_type.module
999                                && ty.name == key.object_type.name
1000                                // If type_params are not provided then we match all params
1001                                && (ty.type_params.is_empty() ||
1002                                    // If they are provided the type params must match
1003                                    ty.type_params == key.object_type.type_params)
1004                        }).unwrap_or(true)
1005            }))
1006    }

1131    pub async fn new_with_options(
1132        dir: &Path,
1133        authority_store: &AuthorityStore,

1135        epoch_store: &AuthorityPerEpochStore,
1136        package_store: &Arc<dyn BackingPackageStore + Send + Sync>,
1137        index_options: IndexStoreOptions,
1138        rpc_config: sui_config::RpcConfig,
1139    ) -> Self {
1140        let path = Self::db_path(dir);
1141        let index_config = rpc_config.index_initialization_config();
1142
1143        let tables = {
1144            let tables = IndexStoreTables::open_with_index_options(&path, index_options.clone());
1145
1146            // If the index tables are uninitialized or on an older version then we need to
1147            // populate them
1148            if tables.needs_to_do_initialization(checkpoint_store) {
1149                let batch_size_limit;
1150
1151                let mut tables = {
1152                    drop(tables);
1153                    typed_store::rocks::safe_drop_db(path.clone(), Duration::from_secs(30))
1154                        .await
1155                        .expect("unable to destroy old rpc-index db");
1156
1157                    // Open the empty DB with `unordered_write`s enabled in order to get a ~3x
1158                    // speedup when indexing
1159                    let mut options = typed_store::rocksdb::Options::default();
1160                    options.set_unordered_write(true);
1161
1162                    // Allow CPU-intensive flushing operations to use all CPUs.
1163                    let max_background_jobs = if let Some(jobs) =
1164                        index_config.as_ref().and_then(|c| c.max_background_jobs)
1165                    {
1166                        debug!("Using config override for max_background_jobs: {}", jobs);
1167                        jobs
1168                    } else {
1169                        let jobs = num_cpus::get() as i32;
1170                        debug!(
1171                            "Calculated max_background_jobs: {} (based on CPU count)",
1172                            jobs
1173                        );
1174                        jobs
1175                    };
1176                    options.set_max_background_jobs(max_background_jobs);
1177
1178                    // We are disabling compaction for all column families below. This means we can
1179                    // also disable the backpressure that slows down writes when the number of L0
1180                    // files builds up since we will never compact them anyway.
1181                    options.set_level_zero_file_num_compaction_trigger(0);
1182                    options.set_level_zero_slowdown_writes_trigger(-1);
1183                    options.set_level_zero_stop_writes_trigger(i32::MAX);
1184
1185                    let total_memory_bytes = get_available_memory();
1186                    // This is an upper bound on the amount to of ram the memtables can use across
1187                    // all column families.
1188                    let db_buffer_size = if let Some(size) =
1189                        index_config.as_ref().and_then(|c| c.db_write_buffer_size)
1190                    {
1191                        debug!(
1192                            "Using config override for db_write_buffer_size: {} bytes",
1193                            size
1194                        );
1195                        size
1196                    } else {
1197                        // Default to 80% of system RAM
1198                        let size = (total_memory_bytes as f64 * 0.8) as usize;
1199                        debug!(
1200                            "Calculated db_write_buffer_size: {} bytes (80% of {} total bytes)",
1201                            size, total_memory_bytes
1202                        );
1203                        size
1204                    };
1205                    options.set_db_write_buffer_size(db_buffer_size);
1206
1207                    // Create column family specific options.
1208                    let mut table_config_map = BTreeMap::new();
1209
1210                    // Create options with compactions disabled and large write buffers.
1211                    // Each CF can use up to 25% of system RAM, but total is still limited by
1212                    // set_db_write_buffer_size configured above.
1213                    let mut cf_options = typed_store::rocks::default_db_options();
1214                    cf_options.options.set_disable_auto_compactions(true);
1215
1216                    let (buffer_size, buffer_count) = match (
1217                        index_config.as_ref().and_then(|c| c.cf_write_buffer_size),
1218                        index_config
1219                            .as_ref()
1220                            .and_then(|c| c.cf_max_write_buffer_number),
1221                    ) {
1222                        (Some(size), Some(count)) => {
1223                            debug!(
1224                                "Using config overrides - buffer_size: {} bytes, buffer_count: {}",
1225                                size, count
1226                            );
1227                            (size, count)
1228                        }
1229                        (None, None) => {
1230                            // Calculate buffer configuration: 25% of RAM split across buffers
1231                            let cf_memory_budget = (total_memory_bytes as f64 * 0.25) as usize;
1232                            debug!(
1233                                "Column family memory budget: {} bytes (25% of {} total bytes)",
1234                                cf_memory_budget, total_memory_bytes
1235                            );
1236                            const MIN_BUFFER_SIZE: usize = 64 * 1024 * 1024; // 64MB minimum
1237
1238                            // Target number of buffers based on CPU count
1239                            // More CPUs = more parallel flushing capability
1240                            let target_buffer_count = num_cpus::get().max(2);
1241
1242                            // Aim for CPU-based buffer count, but reduce if it would make buffers too small
1243                            //   For example:
1244                            // - 128GB RAM, 32 CPUs: 32GB per CF / 32 buffers = 1GB each
1245                            // - 16GB RAM, 8 CPUs: 4GB per CF / 8 buffers = 512MB each
1246                            // - 4GB RAM, 8 CPUs: 1GB per CF / 64MB min = ~16 buffers of 64MB each
1247                            let buffer_size =
1248                                (cf_memory_budget / target_buffer_count).max(MIN_BUFFER_SIZE);
1249                            let buffer_count = (cf_memory_budget / buffer_size)
1250                                .clamp(2, target_buffer_count)
1251                                as i32;
1252                            debug!(
1253                                "Calculated buffer_size: {} bytes, buffer_count: {} (based on {} CPUs)",
1254                                buffer_size, buffer_count, target_buffer_count
1255                            );
1256                            (buffer_size, buffer_count)
1257                        }
1258                        _ => {
1259                            panic!(
1260                                "indexing-cf-write-buffer-size and indexing-cf-max-write-buffer-number must both be specified or both be omitted"
1261                            );
1262                        }
1263                    };
1264
1265                    cf_options.options.set_write_buffer_size(buffer_size);
1266                    cf_options.options.set_max_write_buffer_number(buffer_count);
1267
1268                    // Calculate batch size limit: default to half the buffer size or 128MB, whichever is smaller
1269                    batch_size_limit = if let Some(limit) =
1270                        index_config.as_ref().and_then(|c| c.batch_size_limit)
1271                    {
1272                        debug!(
1273                            "Using config override for batch_size_limit: {} bytes",
1274                            limit
1275                        );
1276                        limit
1277                    } else {
1278                        let half_buffer = buffer_size / 2;
1279                        let default_limit = 1 << 27; // 128MB
1280                        let limit = half_buffer.min(default_limit);
1281                        debug!(
1282                            "Calculated batch_size_limit: {} bytes (min of half_buffer={} and default_limit={})",
1283                            limit, half_buffer, default_limit
1284                        );
1285                        limit
1286                    };
1287
1288                    // Apply cf_options to all tables
1289                    for (table_name, _) in IndexStoreTables::describe_tables() {
1290                        table_config_map.insert(table_name, cf_options.clone());
1291                    }
1292
1293                    // Override Balance options with the merge operator
1294                    let mut balance_options = cf_options.clone();
1295                    balance_options = balance_options.set_merge_operator_associative(
1296                        "balance_merge",
1297                        balance_delta_merge_operator,
1298                    );
1299                    table_config_map.insert("balance".to_string(), balance_options);
1300
1301                    table_config_map.insert(
1302                        "events_by_stream".to_string(),
1303                        events_table_options(index_options.events_compaction_filter.clone()),
1304                    );
1305
1306                    IndexStoreTables::open_with_options(
1307                        &path,
1308                        options,
1309                        Some(DBMapTableConfigMap::new(table_config_map)),
1310                    )
1311                };
1312
1313                tables
1314                    .init(
1315                        authority_store,
1316                        checkpoint_store,
1317                        epoch_store,
1318                        package_store,
1319                        batch_size_limit,
1320                        &rpc_config,
1321                    )
1322                    .expect("unable to initialize rpc index from live object set");
1323
1324                // Flush all data to disk before dropping tables.
1325                // This is critical because WAL is disabled during bulk indexing.
1326                // Note we only need to call flush on one table because all tables share the same
1327                // underlying database.
1328                tables
1329                    .meta
1330                    .flush()
1331                    .expect("Failed to flush RPC index tables to disk");
1332
1333                let weak_db = Arc::downgrade(&tables.meta.db);
1334                drop(tables);
1335
1336                let deadline = std::time::Instant::now() + std::time::Duration::from_secs(30);
1337                loop {
1338                    if weak_db.strong_count() == 0 {
1339                        break;
1340                    }
1341                    if std::time::Instant::now() > deadline {
1342                        panic!("unable to reopen DB after indexing");
1343                    }
1344                    tokio::time::sleep(std::time::Duration::from_millis(100)).await;
1345                }
1346
1347                // Reopen the DB with default options (eg without `unordered_write`s enabled)
1348                let reopened_tables =
1349                    IndexStoreTables::open_with_index_options(&path, index_options);
1350
1351                // Sanity check: verify the database version was persisted correctly
1352                let stored_version = reopened_tables
1353                    .meta
1354                    .get(&())
1355                    .expect("Failed to read metadata from reopened database")
1356                    .expect("Metadata not found in reopened database");
1357                assert_eq!(
1358                    stored_version.version, CURRENT_DB_VERSION,
1359                    "Database version mismatch after flush and reopen: expected {}, found {}",
1360                    CURRENT_DB_VERSION, stored_version.version
1361                );
1362
1363                reopened_tables
1364            } else {
1365                tables
1366            }
1367        };
1368
1369        Self {
1370            tables,
1371            pending_updates: Default::default(),
1372            rpc_config,
1373        }
1374    }

1397    ///
1398    /// Updates will not be committed to the database until `commit_update_for_checkpoint` is
1399    /// called.
1400    #[tracing::instrument(
1401        skip_all,
1402        fields(checkpoint = checkpoint.checkpoint_summary.sequence_number)
1403    )]
1404    pub fn index_checkpoint(&self, checkpoint: &CheckpointData, resolver: &mut dyn LayoutResolver) {
1405        let sequence_number = checkpoint.checkpoint_summary.sequence_number;
1406        let batch = self
1407            .tables
1408            .index_checkpoint(checkpoint, resolver, &self.rpc_config)
1409            .expect("db error");
1410
1411        self.pending_updates
1412            .lock()
1413            .unwrap()
1414            .insert(sequence_number, batch);
1415    }

1417    /// Commits the pending updates for the provided checkpoint number.
1418    ///
1419    /// Invariants:
1420    /// - `index_checkpoint` must have been called for the provided checkpoint
1421    /// - Callers of this function must ensure that it is called for each checkpoint in sequential
1422    ///   order. This will panic if the provided checkpoint does not match the expected next
1423    ///   checkpoint to commit.
1424    #[tracing::instrument(skip(self))]
1425    pub fn commit_update_for_checkpoint(&self, checkpoint: u64) -> Result<(), StorageError> {
1426        let next_batch = self.pending_updates.lock().unwrap().pop_first();
1427
1428        // Its expected that the next batch exists
1429        let (next_sequence_number, batch) = next_batch.unwrap();
1430        assert_eq!(
1431            checkpoint, next_sequence_number,
1432            "commit_update_for_checkpoint must be called in order"
1433        );
1434
1435        Ok(batch.write()?)
1436    }

1599struct RpcLiveObjectIndexer<'a> {
1600    tables: &'a IndexStoreTables,
1601    batch: typed_store::rocks::DBBatch,
1602    coin_index: &'a Mutex<HashMap<CoinIndexKey, CoinIndexInfo>>,
1603    balance_changes: HashMap<BalanceKey, BalanceIndexInfo>,
1604    batch_size_limit: usize,
1605}

1607impl<'a> ParMakeLiveObjectIndexer for RpcParLiveObjectSetIndexer<'a> {
1608    type ObjectIndexer = RpcLiveObjectIndexer<'a>;
1609
1610    fn make_live_object_indexer(&self) -> Self::ObjectIndexer {
1611        RpcLiveObjectIndexer {
1612            tables: self.tables,
1613            batch: self.tables.owner.batch(),
1614            coin_index: self.coin_index,
1615            balance_changes: HashMap::new(),
1616            batch_size_limit: self.batch_size_limit,
1617        }
1618    }
1619}
1620
1621impl LiveObjectIndexer for RpcLiveObjectIndexer<'_> {
1622    fn index_object(&mut self, object: Object) -> Result<(), StorageError> {
1623        match object.owner {
1624            // Owner Index
1625            Owner::AddressOwner(owner) | Owner::ConsensusAddressOwner { owner, .. } => {
1626                let owner_key = OwnerIndexKey::from_object(&object);
1627                let owner_info = OwnerIndexInfo::new(&object);
1628                self.batch
1629                    .insert_batch(&self.tables.owner, [(owner_key, owner_info)])?;
1630
1631                if let Some((coin_type, value)) = get_balance_and_type_if_coin(&object)? {
1632                    let balance_key = BalanceKey { owner, coin_type };
1633                    let balance_info = BalanceIndexInfo::from(value);
1634                    self.balance_changes
1635                        .entry(balance_key)
1636                        .or_default()
1637                        .merge_delta(&balance_info);
1638
1639                    if self.balance_changes.len() >= BALANCE_FLUSH_THRESHOLD {
1640                        self.batch.partial_merge_batch(
1641                            &self.tables.balance,
1642                            std::mem::take(&mut self.balance_changes),
1643                        )?;
1644                    }
1645                }
1646            }
1647
1648            // Dynamic Field Index
1649            Owner::ObjectOwner(parent) => {
1650                if should_index_dynamic_field(&object) {
1651                    let field_key = DynamicFieldKey::new(parent, object.id());
1652                    self.batch
1653                        .insert_batch(&self.tables.dynamic_field, [(field_key, ())])?;
1654                }
1655            }
1656
1657            Owner::Shared { .. } | Owner::Immutable => {}
1658        }
1659
1660        // Look for CoinMetadata<T> and TreasuryCap<T> objects
1661        if let Some((key, value)) = try_create_coin_index_info(&object) {
1662            use std::collections::hash_map::Entry;
1663
1664            match self.coin_index.lock().unwrap().entry(key) {
1665                Entry::Occupied(mut o) => {
1666                    o.get_mut().merge(value);
1667                }
1668                Entry::Vacant(v) => {
1669                    v.insert(value);
1670                }
1671            }
1672        }
1673
1674        if let Some((key, info)) = IndexStoreTables::extract_version_if_package(&object) {
1675            self.batch
1676                .insert_batch(&self.tables.package_version, [(key, info)])?;
1677        }
1678
1679        // If the batch size grows to greater than the limit then write out to the DB so that the
1680        // data we need to hold in memory doesn't grow unbounded.
1681        if self.batch.size_in_bytes() >= self.batch_size_limit {
1682            std::mem::replace(&mut self.batch, self.tables.owner.batch())
1683                .write_opt(&bulk_ingestion_write_options())?;
1684        }
1685
1686        Ok(())
1687    }
1688
1689    fn finish(mut self) -> Result<(), StorageError> {
1690        self.batch.partial_merge_batch(
1691            &self.tables.balance,
1692            std::mem::take(&mut self.balance_changes),
1693        )?;
1694        self.batch.write_opt(&bulk_ingestion_write_options())?;
1695        Ok(())
1696    }
1697}

1701fn sparse_checkpoint_data_for_backfill(
1702    authority_store: &AuthorityStore,
1703    checkpoint_store: &CheckpointStore,
1704    checkpoint: u64,
1705    load_events: bool,
1706) -> Result<CheckpointData, StorageError> {
1707    use sui_types::full_checkpoint_content::CheckpointTransaction;
1708
1709    let summary = checkpoint_store
1710        .get_checkpoint_by_sequence_number(checkpoint)?
1711        .ok_or_else(|| StorageError::missing(format!("missing checkpoint {checkpoint}")))?;
1712    let contents = checkpoint_store
1713        .get_checkpoint_contents(&summary.content_digest)?
1714        .ok_or_else(|| StorageError::missing(format!("missing checkpoint {checkpoint}")))?;
1715
1716    let transaction_digests = contents
1717        .iter()
1718        .map(|execution_digests| execution_digests.transaction)
1719        .collect::<Vec<_>>();
1720    let transactions = authority_store
1721        .multi_get_transaction_blocks(&transaction_digests)?
1722        .into_iter()
1723        .map(|maybe_transaction| {
1724            maybe_transaction.ok_or_else(|| StorageError::custom("missing transaction"))
1725        })
1726        .collect::<Result<Vec<_>, _>>()?;
1727
1728    let effects = authority_store
1729        .multi_get_executed_effects(&transaction_digests)?
1730        .into_iter()
1731        .map(|maybe_effects| maybe_effects.ok_or_else(|| StorageError::custom("missing effects")))
1732        .collect::<Result<Vec<_>, _>>()?;
1733
1734    let events = if load_events {
1735        authority_store
1736            .multi_get_events(&transaction_digests)
1737            .map_err(|e| StorageError::custom(e.to_string()))?
1738    } else {
1739        vec![None; transaction_digests.len()]
1740    };
1741
1742    let mut full_transactions = Vec::with_capacity(transactions.len());
1743    for ((tx, fx), ev) in transactions.into_iter().zip(effects).zip(events) {
1744        let input_objects =
1745            sui_types::storage::get_transaction_input_objects(authority_store, &fx)?;
1746        let output_objects =
1747            sui_types::storage::get_transaction_output_objects(authority_store, &fx)?;
1748
1749        let full_transaction = CheckpointTransaction {
1750            transaction: tx.into(),
1751            effects: fx,
1752            events: ev,
1753            input_objects,
1754            output_objects,
1755        };
1756
1757        full_transactions.push(full_transaction);
1758    }
1759
1760    let checkpoint_data = CheckpointData {
1761        checkpoint_summary: summary.into(),
1762        checkpoint_contents: contents,
1763        transactions: full_transactions,
1764    };
1765
1766    Ok(checkpoint_data)
1767}

1769fn get_balance_and_type_if_coin(object: &Object) -> Result<Option<(StructTag, u64)>, StorageError> {
1770    match Coin::extract_balance_if_coin(object) {
1771        Ok(Some((TypeTag::Struct(struct_tag), value))) => Ok(Some((*struct_tag, value))),
1772        Ok(Some(_)) => {
1773            debug!("Coin object {} has non-struct type tag", object.id());
1774            Ok(None)
1775        }
1776        Ok(None) => {

1778            Ok(None)
1779        }

1782            Err(StorageError::custom(format!(
1783                "Failed to deserialize coin object {}: {}",
1784                object.id(),
1785                e
1786            )))
1787        }
1788    }
