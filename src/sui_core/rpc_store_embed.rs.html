4//! Startup orchestration for the embedded `sui-rpc-store` indexer.
5//!
6//! When a fullnode is configured with
7//! [`RpcConfig::enable_indexing`](sui_config::RpcConfig::enable_indexing), it
8//! serves the rpc-api index surface from an embedded [`sui_rpc_store`]
9//! instance. This module owns the lifecycle of that instance:
10//!
11//! 1. Open the rpc-store database under the node's `db_path()`.
12//! 2. Compare its persisted per-pipeline watermarks against the
13//!    perpetual store's currently-available checkpoint range `[L, T]`
14//!    (`L` = lowest available, `T` = highest executed) and `decide`
15//!    what to do: resume as-is, (re)seed the history cohort, or
16//!    (re)restore the live cohort.
17//! 3. Bulk-load the live cohort from the perpetual store and seed the
18//!    history cohort when needed (blocking, before the node starts
19//!    executing).
20//! 4. Build the read handle the rpc-api serves through, hand the store
21//!    to the pruner, and spawn the tip-following indexer fed by the
22//!    perpetual store ([`PerpetualStoreIngestionClient`]) and the
23//!    checkpoint executor's broadcast stream
24//!    ([`BroadcastStreamingClient`]).
25//!
26//! The live cohort (live-object-derivable indexes) is restored to the
27//! tip and follows forward. The history cohort (ledger-history bitmaps,
28//! `tx_seq` maps, per-epoch metadata) is seeded to the lowest available
29//! checkpoint and backfilled upward; the synchronizer's dynamic cohort
30//! lets it catch up to the live frontier without stalling tip
31//! snapshots.

100/// Open the rpc-store database at `path`.
101#[cfg(not(msim))]
102async fn open_db(path: &std::path::Path) -> anyhow::Result<(Db, RpcStoreSchema)> {
103    Db::open::<RpcStoreSchema>(path, db_options())
104        .context("opening the embedded rpc-store database")
105}
106
107/// Open the rpc-store database at `path`, retrying a transient lock
108/// conflict.
109///
110/// A node restarted on the same `db_path` (the simtest restart path) can
111/// briefly observe the path as locked: the previous instance's RocksDB
112/// teardown frees its in-process lock registry entry inside RocksDB's
113/// native close, on its own threads, and that release is not synchronized
114/// with the drop of the last `Db` handle -- so even after every strong
115/// handle has dropped, the reopen can race the unfinished native teardown.
116/// Retrying is robust where a wait is not (the only operation that can
117/// observe "the path is openable again" is asking RocksDB to open it); a
118/// genuine, persistent failure surfaces once attempts are exhausted.
119/// Mirrors `typed_store::safe_drop_rocksdb`'s retry on the inverse
120/// (destroy-vs-teardown) race.
121///
122/// Simulation-only: under a real runtime `Node::stop` joins the node's
123/// thread, so the prior instance's teardown completes before the restart
124/// and the open never races.
125#[cfg(msim)]
126async fn open_db(path: &std::path::Path) -> anyhow::Result<(Db, RpcStoreSchema)> {
127    const OPEN_ATTEMPTS: usize = 60;
128    const OPEN_RETRY_DELAY: std::time::Duration = std::time::Duration::from_millis(100);
129
130    let mut attempt = 1;
131    loop {
132        match Db::open::<RpcStoreSchema>(path, db_options()) {
133            Ok(opened) => return Ok(opened),
134            Err(e) if attempt < OPEN_ATTEMPTS => {
135                tracing::warn!(
136                    attempt,
137                    "opening the embedded rpc-store database failed, retrying: {e:?}"
138                );
139                attempt += 1;
140                tokio::time::sleep(OPEN_RETRY_DELAY).await;
141            }
142            Err(e) => return Err(e).context("opening the embedded rpc-store database"),
143        }
144    }
145}

147/// What the startup orchestration does with the on-disk rpc-store.
148///
149/// The action chosen at startup is retained on [`EmbeddedRpcStore`] and
150/// exposed via [`EmbeddedRpcStore::bootstrap_action`] so tests (and
151/// future introspection surfaces) can tell whether a restart resumed the
152/// existing indexes or rebuilt them.
153#[derive(Debug, Clone, Copy, PartialEq, Eq)]
154pub enum Bootstrap {
155    /// The on-disk state resumes within the available range; open it
156    /// and follow the tip with no blocking work.
157    Resume,
158
159    /// The live cohort is fine, but the history cohort is missing or
160    /// has fallen below the available floor; (re)seed it in place
161    /// without disturbing the live cohort.
162    SeedHistory,
163
164    /// (Re)bulk-load the live cohort from the perpetual store, then
165    /// seed the history cohort. `clear` wipes the database first (for
166    /// out-of-range or wrong-chain data); otherwise the restore
167    /// resumes from any in-progress per-shard cursors.
168    Restore { clear: bool },
169}
170
171/// The persisted framework state [`decide`] consumes, read from the
172/// on-disk store at bootstrap.
173#[derive(Debug, Clone, Copy)]
174struct StoreState {
175    /// `Some(c)` when every [`LIVE_COHORT`] pipeline has a committed
176    /// watermark, where `c` is the lowest checkpoint tip indexing
177    /// would resume from across them
178    /// (`min(checkpoint_hi_inclusive) + 1`); `None` when any live
179    /// pipeline lacks a watermark.
180    live_resume: Option<u64>,
181
182    /// The same for the [`HISTORY_COHORT`], but a missing watermark
183    /// maps to `0`: an unwatermarked pipeline resumes at
184    /// `first_checkpoint`, which the embedded path leaves at its `0`
185    /// default, so the history cohort backfills from genesis.
186    history_resume: u64,
187
188    /// `Some(false)` when the database is bound to a different chain,
189    /// `Some(true)` when it matches, and `None` when no chain id has
190    /// been recorded yet.
191    chain_matches: Option<bool>,
192
193    /// A bulk restore crashed mid-run and its per-shard cursors are
194    /// resumable ([`sui_rpc_store::restore_in_progress`]).
195    restore_in_progress: bool,
196
197    /// The store is in the exact post-restore, pre-seed state that
198    /// [`seed_history_cohort`] repairs
199    /// ([`sui_rpc_store::history_seed_pending`]).
200    history_seed_pending: bool,
201}
202
203/// Decide what bootstrap action the embedded store needs from the
204/// persisted framework state and `lowest_available` — `L`, the lowest
205/// checkpoint the perpetual store can still serve.
206fn decide(state: StoreState, lowest_available: u64) -> Bootstrap {
207    // A database bound to another chain is unusable; wipe and rebuild.
208    if state.chain_matches == Some(false) {
209        return Bootstrap::Restore { clear: true };
210    }
211
212    let Some(live_resume) = state.live_resume else {
213        // No complete live cohort. Resume in place only when a bulk
214        // restore is actually mid-run (per-shard cursors on disk).
215        // Anything else that lacks a live watermark -- a fresh store,
216        // or a from-genesis run that crashed between the live
217        // pipelines' first commits -- must start from a clean slate:
218        // the bulk restore merges and puts on top of whatever rows
219        // exist, so running it over partially tip-indexed CFs would
220        // double-count balances and leave stale owner and type rows.
221        // (On a genuinely fresh store the clear is a no-op.)
222        return Bootstrap::Restore {
223            clear: !state.restore_in_progress,
224        };
225    };
226
227    // The live cohort's indexes reference checkpoints the perpetual
228    // store has since pruned; the bulk-loaded data is unusable.
229    if live_resume < lowest_available {
230        return Bootstrap::Restore { clear: true };
231    }
232
233    if state.history_resume < lowest_available {
234        // The history cohort sits below the available floor. Seeding
235        // it in place is sound only in the post-restore, pre-seed
236        // state (a crash between the restore's finalize and the
237        // history seed): there the floor pipelines' restored rows
238        // anchor reads at the target and nothing else has committed.
239        // Any other coverage below the floor sits behind a pruned,
240        // unfillable gap -- e.g. pruning advanced while indexing was
241        // disabled -- and seeding over it would advertise `[L, tip]`
242        // as served while stale sub-floor rows resolve
243        // checkpoint-pinned reads to pre-gap versions. Wipe and
244        // rebuild instead.
245        return if state.history_seed_pending {
246            Bootstrap::SeedHistory
247        } else {
248            Bootstrap::Restore { clear: true }
249        };
250    }
251
252    Bootstrap::Resume
253}

255/// A bootstrapped embedded rpc-store, ready to hand to the pruner and
256/// the rpc-api read path and to start tip indexing.
257pub struct EmbeddedRpcStore {
258    /// Shared store handle. Cloned for the pruner (via [`Self::store`])
259    /// and for the tip indexer.
260    store: RpcStore,
261
262    /// Read handle exposing the rpc-store's index surface to
263    /// `sui-rpc-api`.
264    reader: RpcStoreReader,
265
266    /// Local checkpoint source for the tip ingestion client.
267    ingestion_source: RocksDbStore,
268
269    chain_id: ChainIdentifier,
270
271    /// The bootstrap action [`decide`] selected for the on-disk store at
272    /// startup. Retained for introspection (see
273    /// [`Self::bootstrap_action`]); does not affect runtime behavior.
274    action: Bootstrap,
275
276    /// Service wrapping the background task that builds and runs the tip
277    /// indexer, populated by [`Self::spawn_indexer`]. Its single primary
278    /// task builds the indexer's inner [`Service`] and joins it; dropping
279    /// this service on node shutdown aborts that task, which in turn drops
280    /// and aborts the inner service's pipeline tasks. This is what keeps
281    /// the indexer from leaking (notably across e2e tests sharing a
282    /// process), so no explicit [`Drop`] is needed.
283    indexer_service: Option<Service>,
284}
285
286impl EmbeddedRpcStore {
287    /// Open the rpc-store, bring it in line with the perpetual store's
288    /// available range (restoring / seeding as needed), and build the
289    /// store and read handles.
290    ///
291    /// Blocks while restoring the live cohort. Call before the node
292    /// starts executing checkpoints, so the perpetual store's range is
293    /// stable for the duration of the restore.
294    pub async fn bootstrap(
295        config: &NodeConfig,
296        authority_store: &Arc<AuthorityStore>,
297        checkpoint_store: &Arc<CheckpointStore>,
298        ingestion_source: RocksDbStore,
299        chain_identifier: ChainIdentifier,
300        registry: &Registry,
301    ) -> anyhow::Result<Self> {
302        let perpetual = authority_store.perpetual_tables.clone();
303        let path = config.db_path().join(RPC_STORE_DIR);
304        let (db, schema) = open_db(&path).await?;
305        let schema = Arc::new(schema);
306
307        // Expose per-CF RocksDB stats (sizes, compaction backlog,
308        // write-stall state) for the store's database.
309        registry
310            .register(Box::new(ColumnFamilyStatsCollector::new(
311                Some(METRICS_PREFIX),
312                &db,
313            )))
314            .context("registering the embedded rpc-store RocksDB stats collector")?;
315
316        // The highest checkpoint whose transaction outputs are durably
317        // committed to the perpetual store. This is the live cohort's restore
318        // target: the bulk restore reads the live object set, so the target
319        // must match the checkpoint that set reflects. We use the perpetual
320        // store's `highest_committed` watermark (written atomically with the
321        // objects) rather than the checkpoint store's `highest_executed`
322        // (bumped separately afterward), so an unclean stop cannot leave the
323        // restore reading objects beyond its target and double-counting them
324        // against the forward indexer. `None` only on a node's very first boot
325        // (genesis is executed later in startup), in which case there is
326        // nothing to bulk-load and the indexer builds both cohorts from genesis
327        // as the node executes.
328        let highest_committed = perpetual
329            .get_highest_committed_checkpoint()
330            .context("reading highest committed checkpoint")?
331            // Fall back to the checkpoint store's executed watermark for a
332            // database written before the atomic `highest_committed` watermark
333            // existed: it has no stamp yet, so this preserves the prior restore
334            // target until the next committed checkpoint stamps the consistent
335            // one. In normal operation `highest_committed` is written before
336            // `highest_executed` is bumped, so it is never absent while the
337            // executed watermark is present.
338            .or(checkpoint_store
339                .get_highest_executed_checkpoint_seq_number()
340                .context("reading highest executed checkpoint")?);
341        let lowest_available = lowest_available_checkpoint(&perpetual, checkpoint_store)?;
342
343        let chain_id = ChainId(*chain_identifier.as_bytes());
344        let state = StoreState {
345            live_resume: cohort_resume(&db, LIVE_COHORT)?,
346            history_resume: cohort_resume(&db, HISTORY_COHORT)?.unwrap_or(0),
347            chain_matches: stored_chain_id(&db)?.map(|stored| stored == chain_id),
348            restore_in_progress: sui_rpc_store::restore_in_progress(&db)?,
349            history_seed_pending: sui_rpc_store::history_seed_pending(&db)?,
350        };
351
352        let action = decide(state, lowest_available);
353        info!(
354            ?action,
355            lowest_available,
356            ?highest_committed,
357            "bootstrapping embedded rpc-store",
358        );
359
360        match action {
361            Bootstrap::Resume => {}
362            Bootstrap::SeedHistory => {
363                seed_history(
364                    &db,
365                    &schema,
366                    &perpetual,
367                    checkpoint_store,
368                    lowest_available,
369                    chain_id,
370                )?;
371            }
372            Bootstrap::Restore { clear } => {
373                if clear {
374                    db.clear_all()
375                        .context("clearing the out-of-range embedded rpc-store")?;
376                }
377                // A synced node enabling the embedded store for the
378                // first time (or recovering an out-of-range one):
379                // bulk-load the live cohort, then seed the history cohort
380                // so it backfills `(L, T]`. When `highest_committed` is
381                // `None` (a fresh node, nothing committed yet) there is
382                // nothing to load -- every pipeline stays unwatermarked
383                // so the indexer builds both cohorts from genesis as
384                // checkpoints execute.
385                if let Some(target) = highest_committed {
386                    // At `L > 0` the history cohort backfills only
387                    // `(L, T]`, so the restore also bulk-loads the
388                    // `object_version_by_checkpoint` and
389                    // `package_versions` floor rows at `T` (the seed
390                    // below rewinds their watermarks to `L-1`). At
391                    // `L == 0` the backfill replays the whole chain
392                    // from genesis and rebuilds both CFs in full, so
393                    // those pipelines must not be restored at all --
394                    // the `__watermark = T` rows the restore would
395                    // stamp on them would make them skip `(0, T]`
396                    // entirely.
397                    let layer = if lowest_available > 0 {
398                        RestoreLayer::indexes_only()
399                    } else {
400                        RestoreLayer::live_only()
401                    };
402                    restore_live(
403                        db.clone(),
404                        schema.clone(),
405                        perpetual.clone(),
406                        target,
407                        chain_id,
408                        layer,
409                        registry,
410                    )
411                    .await?;
412                    // `L == 0` means genesis is still available, so the
413                    // history cohort backfills from checkpoint 0 with no
414                    // seed (every history pipeline -- including the two
415                    // the restore skipped -- is unwatermarked and
416                    // resumes at `first_checkpoint = 0`).
417                    if lowest_available > 0 {
418                        seed_history(
419                            &db,
420                            &schema,
421                            &perpetual,
422                            checkpoint_store,
423                            lowest_available,
424                            chain_id,
425                        )?;
426                    }
427                }
428            }
429        }
430
431        let store = sui_consistent_store::Store::new(db.clone(), schema.clone());
432        let reader = RpcStoreReader::new(db, schema);
433
434        Ok(Self {
435            store,
436            reader,
437            ingestion_source,
438            chain_id: chain_identifier,
439            action,
440            indexer_service: None,
441        })
442    }
443
444    /// The bootstrap action selected for the on-disk store at startup:
445    /// whether this run resumed the existing indexes, re-seeded the
446    /// history cohort, or rebuilt the live cohort. Read-only
447    /// introspection; primarily for tests.
448    pub fn bootstrap_action(&self) -> Bootstrap {
449        self.action
450    }
451
452    /// The highest checkpoint the live cohort has committed
453    /// (`min(checkpoint_hi_inclusive)` across its pipelines), i.e. how
454    /// far the live-object indexes have caught up to the tip. `None`
455    /// until every live pipeline has a watermark. Read-only
456    /// introspection; primarily for tests.
457    pub fn live_committed_checkpoint(&self) -> Option<u64> {
458        cohort_committed(self.store.db(), LIVE_COHORT)
459            .ok()
460            .flatten()
461    }
462
463    /// The highest checkpoint the history cohort has committed, i.e. how
464    /// far the ledger-history backfill has progressed. `None` until every
465    /// history pipeline has a watermark. Read-only introspection;
466    /// primarily for tests.
467    pub fn history_committed_checkpoint(&self) -> Option<u64> {
468        cohort_committed(self.store.db(), HISTORY_COHORT)
469            .ok()
470            .flatten()
471    }
472
473    /// A clone of the store handle, for the pruner's history-cohort
474    /// pruning ([`sui_rpc_store::prune_history_cohort`]).
475    pub fn store(&self) -> RpcStore {
476        self.store.clone()
477    }
478
479    /// A clone of the read handle, for the rpc-api read path
480    /// ([`crate::storage::RpcStoreReadStore`]).
481    pub fn reader(&self) -> RpcStoreReader {
482        self.reader.clone()
483    }
484
485    /// A callback reading the highest checkpoint the live cohort has
486    /// committed, for the subscription service's index gate (so a
487    /// checkpoint is not delivered to clients until its indexed state is
488    /// readable).
489    ///
490    /// Reads only the live cohort: the history cohort backfills
491    /// independently from the lowest available checkpoint, so gating on it
492    /// would hold back delivery on a restored node for the duration of the
493    /// backfill. On a node indexing from genesis the synchronizer keeps the
494    /// cohorts in lockstep, so the live cohort's progress implies the
495    /// history cohort's.
496    pub fn indexed_checkpoint_fn(&self) -> Arc<dyn Fn() -> Option<u64> + Send + Sync> {
497        let db = self.store.db().clone();
498        Arc::new(move || cohort_committed(&db, LIVE_COHORT).ok().flatten())
499    }
500
501    /// Spawn a background task that builds and runs the tip-following
502    /// indexer over the embedded store.
503    ///
504    /// The indexer is built on a background task -- rather than inline --
505    /// because the framework reads the starting tip via
506    /// `latest_checkpoint_number`, which on a node booting from genesis
507    /// blocks until the first checkpoint has been *executed*. At this
508    /// point genesis is only *synced* (the executor has not run yet), so
509    /// the read retries until the checkpoint executor catches up; that
510    /// executor only starts once `start_async` returns, so building
511    /// inline would deadlock node startup against a checkpoint that
512    /// cannot arrive until startup completes. (A resuming node has an
513    /// executed tip on disk and would not block, but the genesis case
514    /// forces the deferral unconditionally.) The follower catches up once
515    /// checkpoints begin to flow.
516    ///
517    /// The background task is held as a single-task [`Service`] so the
518    /// node owns the indexer's lifetime: dropping [`EmbeddedRpcStore`]
519    /// drops this service, which aborts the task and, with it, the inner
520    /// indexer service it is joining.
521    ///
522    /// A failure to *build* the indexer is fatal to the node
523    /// ([`mysten_common::fatal!`]): it will not heal on its own, and a
524    /// node whose index surface silently never advances is worse than a
525    /// crashed one.
526    ///
527    /// `checkpoint_sender` is the checkpoint executor's broadcast
528    /// stream; when present it drives a low-latency
529    /// [`BroadcastStreamingClient`], with the perpetual-store ingestion
530    /// client filling any gap. When absent (e.g. on a node that does
531    /// not run the rpc servers) the ingestion client polls the
532    /// perpetual store alone.
533    pub fn spawn_indexer(
534        &mut self,
535        checkpoint_sender: Option<broadcast::Sender<Arc<Checkpoint>>>,
536        registry: Registry,
537    ) {
538        let store = self.store.clone();
539        let ingestion_source = self.ingestion_source.clone();
540        let chain_id = self.chain_id;
541
542        let service = Service::new().spawn(async move {
543            // A node whose embedded indexer never starts would keep
544            // serving RPC from index CFs that never advance -- a frozen
545            // (or, on a fresh node, permanently empty) index surface
546            // with nothing but a log line to say why. Build failure is
547            // a configuration or database problem that will not heal on
548            // its own, so fail the node instead of limping.
549            let mut service = match build_indexer(
550                store,
551                ingestion_source,
552                chain_id,
553                checkpoint_sender,
554                &registry,
555            )
556            .await
557            {
558                Ok(service) => service,
559                Err(e) => {
560                    fatal!("failed to build the embedded rpc-store indexer: {e:#}");
561                }
562            };
563            // Hold the service for the task's lifetime; `join` only
564            // returns if an indexer task exits (it otherwise runs for the
565            // node's lifetime). Deliberately not fatal: an error here can
566            // be a teardown race with node shutdown (this task is aborted
567            // on drop, but the inner pipelines may observe the closing
568            // stores first), and a genuine mid-run death is surfaced by
569            // the health check as an ever-growing gap between the
570            // executed tip and the frozen live-index watermark.
571            if let Err(e) = service.join().await {
572                error!("the embedded rpc-store indexer exited with an error: {e:#}");
573            }
574            Ok(())
575        });
576        self.indexer_service = Some(service);
577    }
578}

580/// Build the tip-following indexer over `store`, register the embedded
581/// cohort pipelines, and run it. Returns the composed [`Service`]
582/// driving ingestion, the synchronizer, and the committers.
583async fn build_indexer(
584    store: RpcStore,
585    ingestion_source: RocksDbStore,
586    chain_id: ChainIdentifier,
587    checkpoint_sender: Option<broadcast::Sender<Arc<Checkpoint>>>,
588    registry: &Registry,
589) -> anyhow::Result<Service> {
590    let ingestion_metrics = IngestionMetrics::new(Some(METRICS_PREFIX), registry);
591    let ingestion_client = IngestionClient::from_trait(
592        Arc::new(PerpetualStoreIngestionClient::new(
593            ingestion_source.clone(),
594            chain_id,
595        )),
596        ingestion_metrics,
597    );
598    // The broadcast streaming client follows the tip with low latency; it
599    // reads the current tip from the same local store the ingestion
600    // client uses (so the framework's `peek()` resolves immediately even
601    // on an idle chain), and the ingestion client backfills any gap.
602    let streaming_client: Option<ArcStreamingClient> = checkpoint_sender.map(|sender| {
603        Arc::new(BroadcastStreamingClient::new(
604            sender,
605            chain_id,
606            ingestion_source,
607        )) as ArcStreamingClient
608    });
609
610    let mut indexer = Indexer::from_store(
611        store,
612        IndexerArgs::default(),
613        ingestion_client,
614        streaming_client,
615        ConsistencyConfig::default(),
616        // Pruning is driven by the validator's `AuthorityStorePruner`
617        // (history cohort only), not the rpc-store's own pruner.
618        None,
619        IngestionConfig::default(),
620        registry,
621    )
622    .await
623    .context("constructing the embedded rpc-store indexer")?;
624    indexer
625        .add_pipelines(PipelineLayer::embedded(), CommitterConfig::default())
626        .await
627        .context("registering embedded rpc-store pipelines")?;
628    indexer
629        .run()
630        .await
631        .context("starting the embedded rpc-store indexer")
632}

634/// The lowest checkpoint the perpetual store can still serve: one past
635/// the higher of the object-store and checkpoint-store pruned
636/// watermarks (both inclusive). `0` when nothing has been pruned.
637fn lowest_available_checkpoint(
638    perpetual: &AuthorityPerpetualTables,
639    checkpoint_store: &CheckpointStore,
640) -> anyhow::Result<u64> {
641    let object_pruned = perpetual
642        .get_highest_pruned_checkpoint()
643        .context("reading object store pruned watermark")?;
644    let checkpoint_pruned = checkpoint_store
645        .get_highest_pruned_checkpoint_seq_number()
646        .context("reading checkpoint store pruned watermark")?;
647    Ok(object_pruned
648        .into_iter()
649        .chain(checkpoint_pruned)
650        .max()
651        .map(|pruned| pruned + 1)
652        .unwrap_or(0))
653}
654
655/// The highest checkpoint every pipeline in `cohort` has committed
656/// (`min(checkpoint_hi_inclusive)`). `None` if any pipeline in the
657/// cohort has no committed watermark.
658fn cohort_committed(db: &Db, cohort: &[&str]) -> anyhow::Result<Option<u64>> {
659    let framework = db.framework();
660    let mut min_hi: Option<u64> = None;
661    for name in cohort {
662        let key = PipelineTaskKey::new(*name);
663        let Some(watermark) = framework
664            .watermarks
665            .get(&key)
666            .with_context(|| format!("reading watermark for {name}"))?
667        else {
668            return Ok(None);
669        };
670        min_hi = Some(match min_hi {
671            Some(hi) => hi.min(watermark.checkpoint_hi_inclusive),
672            None => watermark.checkpoint_hi_inclusive,
673        });
674    }
675    Ok(min_hi)
676}
677
678/// The lowest checkpoint tip indexing would resume from across a
679/// cohort: `min(checkpoint_hi_inclusive) + 1`. `None` if any pipeline
680/// in the cohort has no committed watermark.
681fn cohort_resume(db: &Db, cohort: &[&str]) -> anyhow::Result<Option<u64>> {
682    Ok(cohort_committed(db, cohort)?.map(|hi| hi + 1))
683}

685/// The chain id the database is bound to, read from the first pipeline
686/// that has one recorded. All pipelines pin the same chain, so any one
687/// is representative.
688fn stored_chain_id(db: &Db) -> anyhow::Result<Option<ChainId>> {
689    let framework = db.framework();
690    for name in LIVE_COHORT.iter().chain(HISTORY_COHORT) {
691        let key = PipelineTaskKey::new(*name);
692        if let Some(id) = framework
693            .chain_ids
694            .get(&key)
695            .with_context(|| format!("reading chain id for {name}"))?
696        {
697            return Ok(Some(id));
698        }
699    }
700    Ok(None)
701}

703/// Bulk-load the live cohort from the perpetual store up to
704/// `target_checkpoint`, blocking until the restore completes.
705///
706/// `layer` selects whether the `object_version_by_checkpoint` and
707/// `package_versions` floor rows are also bulk-loaded
708/// ([`RestoreLayer::indexes_only`], for `L > 0`) or left entirely to
709/// the from-genesis backfill ([`RestoreLayer::live_only`], for
710/// `L == 0`).
711async fn restore_live(
712    db: Db,
713    schema: Arc<RpcStoreSchema>,
714    perpetual: Arc<AuthorityPerpetualTables>,
715    target_checkpoint: u64,
716    chain_id: ChainId,
717    layer: RestoreLayer,
718    registry: &Registry,
719) -> anyhow::Result<()> {
720    let source = PerpetualStoreRestoreSource::new(perpetual, target_checkpoint, chain_id);
721    let metrics = RestoreMetrics::new(Some(METRICS_PREFIX), registry);
722    let mut service = restore_indexes(
723        db,
724        schema,
725        source,
726        RestoreDriverConfig::default(),
727        layer,
728        metrics,
729    )
730    .context("starting the live-cohort restore")?;
731    service
732        .join()
733        .await
734        .context("restoring the live cohort from the perpetual store")?;
735    Ok(())
736}

738/// Seed the history cohort to `L - 1` so the backfill resumes at the
739/// lowest available checkpoint `L`. The seed watermark's `tx_hi`,
740/// epoch, and timestamp come from checkpoint `L - 1`'s summary, so the
741/// seeded pruning floor lines up with the first checkpoint the backfill
742/// will index.
743fn seed_history(
744    db: &Db,
745    schema: &RpcStoreSchema,
746    perpetual: &AuthorityPerpetualTables,
747    checkpoint_store: &CheckpointStore,
748    lowest_available: u64,
749    chain_id: ChainId,
750) -> anyhow::Result<()> {
751    debug_assert!(lowest_available > 0, "seed_history requires L > 0");
752    let anchor = lowest_available - 1;
753    let checkpoint = checkpoint_store
754        .get_checkpoint_by_sequence_number(anchor)
755        .context("reading the history seed-anchor checkpoint")?
756        .with_context(|| format!("history seed-anchor checkpoint {anchor} is unavailable"))?;
757    let summary = checkpoint.data();
758    let watermark = Watermark {
759        epoch_hi_inclusive: summary.epoch,
760        checkpoint_hi_inclusive: anchor,
761        tx_hi: summary.network_total_transactions,
762        timestamp_ms_hi_inclusive: summary.timestamp_ms,
763    };
764    seed_history_cohort(
765        db,
766        schema,
767        watermark,
768        chain_id,
769        Some(perpetual as &dyn ObjectStore),
770    )
771    .context("seeding the history cohort")
772}

774#[cfg(test)]
775mod tests {
776    use super::*;
777
778    /// A [`StoreState`] with neither a mid-run restore nor a pending
779    /// history seed — the shape of every store that was not stopped
780    /// inside the bootstrap's own restore/seed sequence.
781    fn state(
782        live_resume: Option<u64>,
783        history_resume: u64,
784        chain_matches: Option<bool>,
785    ) -> StoreState {
786        StoreState {
787            live_resume,
788            history_resume,
789            chain_matches,
790            restore_in_progress: false,
791            history_seed_pending: false,
792        }
793    }
794
795    // `L = 0` (nothing pruned): an unseeded history cohort backfills
796    // from genesis, so a complete live cohort is enough to resume.
797    #[test]
798    fn resumes_from_genesis_when_nothing_pruned() {
799        assert_eq!(decide(state(Some(10), 0, Some(true)), 0), Bootstrap::Resume);
800        // History never seeded (resume 0) is fine at L = 0.
801        assert_eq!(decide(state(Some(10), 0, None), 0), Bootstrap::Resume);
802    }
803
804    // Both cohorts resume at or above the available floor.
805    #[test]
806    fn resumes_when_in_range() {
807        assert_eq!(
808            decide(state(Some(100), 100, Some(true)), 100),
809            Bootstrap::Resume
810        );
811        assert_eq!(
812            decide(state(Some(200), 100, Some(true)), 100),
813            Bootstrap::Resume
814        );
815    }
816
817    // A bulk restore crashed mid-run: resume it in place from its
818    // per-shard cursors rather than discarding the partial progress.
819    #[test]
820    fn resumes_restore_in_progress_without_clearing() {
821        assert_eq!(
822            decide(
823                StoreState {
824                    restore_in_progress: true,
825                    ..state(None, 0, None)
826                },
827                0,
828            ),
829            Bootstrap::Restore { clear: false }
830        );
831        assert_eq!(
832            decide(
833                StoreState {
834                    restore_in_progress: true,
835                    ..state(None, 50, Some(true))
836                },
837                100,
838            ),
839            Bootstrap::Restore { clear: false }
840        );
841    }
842
843    // No live watermark and no restore in progress: a fresh store (the
844    // clear is a no-op) or a from-genesis run that crashed between the
845    // live pipelines' first commits (whose partial rows the bulk
846    // restore would otherwise merge on top of, double-counting
847    // balances). Either way, start from a clean slate.
848    #[test]
849    fn clears_when_live_unwatermarked_without_restore_in_progress() {
850        assert_eq!(
851            decide(state(None, 0, None), 0),
852            Bootstrap::Restore { clear: true }
853        );
854        assert_eq!(
855            decide(state(None, 50, Some(true)), 100),
856            Bootstrap::Restore { clear: true }
857        );
858    }
859
860    // The live cohort references checkpoints the perpetual store has
861    // pruned away: wipe and rebuild.
862    #[test]
863    fn clears_and_restores_when_live_out_of_range() {
864        assert_eq!(
865            decide(state(Some(50), 200, Some(true)), 100),
866            Bootstrap::Restore { clear: true }
867        );
868    }
869
870    // A database bound to a different chain is always wiped.
871    #[test]
872    fn clears_and_restores_on_chain_mismatch() {
873        assert_eq!(
874            decide(state(Some(200), 200, Some(false)), 100),
875            Bootstrap::Restore { clear: true }
876        );
877        // Chain mismatch dominates even an otherwise-resumable state.
878        assert_eq!(
879            decide(state(Some(200), 200, Some(false)), 0),
880            Bootstrap::Restore { clear: true }
881        );
882    }
883
884    // The crash window between the restore's finalize and the history
885    // seed: the floor pipelines hold the restore watermark, nothing
886    // else in the history cohort has committed, and re-running the
887    // seed alone repairs the store.
888    #[test]
889    fn seeds_history_only_in_post_restore_state() {
890        assert_eq!(
891            decide(
892                StoreState {
893                    history_seed_pending: true,
894                    ..state(Some(200), 0, Some(true))
895                },
896                100,
897            ),
898            Bootstrap::SeedHistory
899        );
900        // History exactly at the floor resumes.
901        assert_eq!(
902            decide(state(Some(200), 100, Some(true)), 100),
903            Bootstrap::Resume
904        );
905    }
906
907    // Committed history coverage below the available floor (e.g.
908    // pruning advanced while indexing was disabled): the gap up to the
909    // floor is pruned and unfillable, so seeding in place would leave
910    // stale sub-floor rows poisoning checkpoint-pinned reads above the
911    // floor. Wipe and rebuild instead.
912    #[test]
913    fn clears_when_history_coverage_behind_floor() {
914        // The whole history cohort committed below the floor.
915        assert_eq!(
916            decide(state(Some(200), 50, Some(true)), 100),
917            Bootstrap::Restore { clear: true }
918        );
919        // Partially unwatermarked history (resume 0) that is not the
920        // post-restore state — e.g. a live_only bootstrap whose
921        // history cohort never committed before pruning advanced.
922        assert_eq!(
923            decide(state(Some(200), 0, Some(true)), 100),
924            Bootstrap::Restore { clear: true }
925        );
926    }
