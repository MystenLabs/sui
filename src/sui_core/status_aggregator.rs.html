4use std::{collections::BTreeMap, sync::Arc};
5
6use mysten_common::debug_fatal;
7use sui_types::{
8    base_types::AuthorityName,
9    committee::{Committee, StakeUnit},
10};
11
12/// Aggregates various types of statuses from different authorities,
13/// and the total stake of authorities that have inserted statuses.
14/// Only keeps the latest status for each authority.
15pub(crate) struct StatusAggregator<T> {
16    committee: Arc<Committee>,
17    total_votes: StakeUnit,
18    statuses: BTreeMap<AuthorityName, T>,
19}
20
21impl<T> StatusAggregator<T> {
22    pub(crate) fn new(committee: Arc<Committee>) -> Self {
23        Self {
24            committee,
25            total_votes: 0,
26            statuses: BTreeMap::new(),
27        }
28    }
29
30    /// Returns true if the status is inserted the first time for the authority.
31    pub(crate) fn insert(&mut self, authority: AuthorityName, status: T) -> bool {
32        let Some(index) = self.committee.authority_index(&authority) else {
33            debug_fatal!("Authority {} not found in committee", authority);
34            return false;
35        };
36        if self.statuses.insert(authority, status).is_some() {
37            return false;
38        }
39        self.total_votes += self.committee.stake_by_index(index).unwrap();
40        true
41    }
42
43    /// Returns the total stake of authorities that have inserted statuses.
44    pub(crate) fn total_votes(&self) -> StakeUnit {
45        self.total_votes
46    }
47
48    /// Returns all authorities that have inserted statuses.
49    pub(crate) fn authorities(&self) -> Vec<AuthorityName> {
50        self.statuses.keys().copied().collect()
51    }
52
53    /// Returns the status of each authority.
54    #[cfg(test)]
55    pub(crate) fn statuses(&self) -> &BTreeMap<AuthorityName, T> {
56        &self.statuses
57    }
58
59    /// Returns the status of each authority.
60    pub(crate) fn status_by_authority(&self) -> Vec<(AuthorityName, StakeUnit, T)>
61    where
62        T: Clone,
63    {
64        self.statuses
65            .iter()
66            .map(|(name, status)| {
67                (
68                    *name,
69                    self.committee
70                        .stake_by_index(self.committee.authority_index(name).unwrap())
71                        .unwrap(),
72                    status.clone(),
73                )
74            })
75            .collect()
76    }
77
78    pub(crate) fn reached_validity_threshold(&self) -> bool {
79        self.total_votes >= self.committee.validity_threshold()
80    }
81
82    pub(crate) fn reached_quorum_threshold(&self) -> bool {
83        self.total_votes >= self.committee.quorum_threshold()
84    }
85}

87#[cfg(test)]
88mod tests {
89    use super::*;
90
91    #[test]
92    fn test_insert() {
93        let (committee, _) = Committee::new_simple_test_committee();
94        let authorities: Vec<_> = committee.names().copied().collect();
95        let stake_per_authority = committee.stake_by_index(0).unwrap();
96        let mut aggregator = StatusAggregator::<&str>::new(Arc::new(committee));
97
98        // Insert a status for authority 0.
99        let initial_status = "initial";
100        assert!(aggregator.insert(authorities[0], initial_status));
101        assert_eq!(aggregator.total_votes(), stake_per_authority);
102        assert_eq!(
103            aggregator.statuses().get(&authorities[0]),
104            Some(&initial_status)
105        );
106
107        // Insert the same status for authority 0 again.
108        assert!(!aggregator.insert(authorities[0], initial_status));
109        assert_eq!(aggregator.total_votes(), stake_per_authority);
110
111        // Insert a different status for authority 0.
112        let different_status = "different";
113        assert!(!aggregator.insert(authorities[0], different_status));
114        assert_eq!(aggregator.total_votes(), stake_per_authority);
115        assert_eq!(
116            aggregator.statuses().get(&authorities[0]),
117            Some(&different_status)
118        );
119
120        // Does not reach validity threshold or quorum threshold.
121        assert!(!aggregator.reached_validity_threshold());
122        assert!(!aggregator.reached_quorum_threshold());
123
124        // Insert a new status for authority 1.
125        let new_status = "new";
126        assert!(aggregator.insert(authorities[1], new_status));
127        assert_eq!(aggregator.total_votes(), 2 * stake_per_authority);
128        assert_eq!(
129            aggregator.statuses().get(&authorities[1]),
130            Some(&new_status)
131        );
132
133        // Reaches validity threshold but not quorum threshold.
134        assert!(aggregator.reached_validity_threshold());
135        assert!(!aggregator.reached_quorum_threshold());
136
137        // Insert a new status for authority 2.
138        assert!(aggregator.insert(authorities[2], new_status));
139        assert_eq!(aggregator.total_votes(), 3 * stake_per_authority);
140        assert_eq!(
141            aggregator.statuses().get(&authorities[2]),
142            Some(&new_status)
143        );
144
145        // Reaches validity and quorum thresholds.
146        assert!(aggregator.reached_validity_threshold());
147        assert!(aggregator.reached_quorum_threshold());
148    }
149}
