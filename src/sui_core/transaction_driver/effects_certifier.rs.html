61    #[instrument(level = "error", skip_all, err(level = "debug"))]
62    pub(crate) async fn get_certified_finalized_effects<A>(
63        &self,
64        authority_aggregator: &Arc<AuthorityAggregator<A>>,
65        client_monitor: &Arc<ValidatorClientMonitor<A>>,
66        tx_digest: Option<TransactionDigest>,
67        tx_type: TxType,
68        // This keeps track of the current target for getting full effects.
69        mut current_target: AuthorityName,
70        // Guaranteed to be not the Rejected variant.
71        submit_txn_result: SubmitTxResult,
72        options: &SubmitTransactionOptions,
73    ) -> Result<QuorumTransactionResponse, TransactionDriverError>
74    where
75        A: AuthorityAPI + Send + Sync + 'static + Clone,
76    {
77        // When consensus position is provided, wait for finalized and fastpath outputs at the validators' side.
78        // Otherwise, only wait for finalized effects.
79        // Skip the first attempt to get full effects if it is already provided.
80        let (consensus_position, full_effects) = match submit_txn_result {
81            SubmitTxResult::Submitted { consensus_position } => (Some(consensus_position), None),
82            SubmitTxResult::Executed {
83                effects_digest,
84                details,
85                fast_path,
86            } => match details {
87                Some(details) => (None, Some((effects_digest, details, fast_path))),
88                // Details should always be set in correct responses.
89                // But if it is not set, continuing to get full effects and certify the digest are still correct.
90                None => (None, None),
91            },
92            SubmitTxResult::Rejected { error } => {
93                return Err(TransactionDriverError::ClientInternal {
94                    error: format!(
95                        "Unexpected submission error in get_certified_finalized_effects(): {}",
96                        error
97                    ),
98                });
99            }
100        };
101
102        let mut retrier =
103            RequestRetrier::new(authority_aggregator, client_monitor, tx_type, vec![]);
104        let ping_type = get_ping_type(&tx_digest, tx_type);
105
106        // Setting this to None at first because if the full effects are already provided,
107        // we do not need to record the latency. We track the time in this function instead of inside
108        // get_full_effects so that we could record differently depending on whether the result is byzantine.
109        let mut full_effects_start_time = None;
110        let (acknowledgments_result, mut full_effects_result) = join!(
111            self.wait_for_acknowledgments(
112                authority_aggregator,
113                client_monitor,
114                tx_digest,
115                tx_type,
116                consensus_position,
117                options,
118                current_target,
119            ),
120            async {
121                // No need to send a full effects request if it is already provided.
122                if let Some(full_effects) = full_effects {
123                    // In this branch, current_target is the authority providing the full effects,
124                    // so it is consistent. This is not used though because current_target is
125                    // only used with failed full effects query.
126                    return Ok(full_effects);
127                }
128                let (name, client) = retrier
129                    .next_target()
130                    .expect("there should be at least 1 target");
131                current_target = name;
132                full_effects_start_time = Some(Instant::now());
133                self.get_full_effects(client, tx_digest, tx_type, consensus_position, options)
134                    .await
135            },
136        );
137
138        // If the consensus position got rejected, effects certification will see the failure and gather
139        // error messages to explain the rejection.
140        let certified_digest = acknowledgments_result?;
141
142        // Retry until there is a valid full effects that matches the certified digest, or all targets
143        // have been attempted.
144        loop {
145            let display_name = authority_aggregator.get_display_name(&current_target);
146            match full_effects_result {
147                Ok((effects_digest, executed_data, _fast_path)) => {
148                    if effects_digest != certified_digest {
149                        tracing::warn!(
150                            ?current_target,
151                            "Full effects digest mismatch ({} vs certified {})",
152                            effects_digest,
153                            certified_digest
154                        );
155                        // This validator is byzantine, record the error.
156                        client_monitor.record_interaction_result(OperationFeedback {
157                            authority_name: current_target,
158                            display_name,
159                            operation: OperationType::Effects,
160                            ping_type,
161                            result: Err(()),
162                        });
163                    } else {
164                        if let Some(start_time) = full_effects_start_time {
165                            let latency = start_time.elapsed();
166                            client_monitor.record_interaction_result(OperationFeedback {
167                                authority_name: current_target,
168                                display_name,
169                                operation: OperationType::Effects,
170                                ping_type,
171                                result: Ok(latency),
172                            });
173                        }
174                        return Ok(
175                            self.get_quorum_transaction_response(effects_digest, executed_data)
176                        );
177                    }
178                }
179                Err(e) => {
180                    tracing::debug!(?current_target, "Failed to get full effects: {e}");
181                    client_monitor.record_interaction_result(OperationFeedback {
182                        authority_name: current_target,
183                        display_name,
184                        operation: OperationType::Effects,
185                        ping_type,
186                        result: Err(()),
187                    });
188                    // This emits an error when retrier gathers enough (f+1) non-retriable effects errors,
189                    // but the error should not happen after effects certification unless there are software bugs
190                    // or > f malicious validators.
191                    retrier.add_error(current_target, e)?;
192                }
193            };
194
195            tokio::task::yield_now().await;
196
197            // Retry getting full effects from the next target.
198
199            // This emits an error when retrier has no targets available.
200            let (name, client) = retrier.next_target()?;
201            current_target = name;
202            full_effects_start_time = Some(Instant::now());
203            full_effects_result = self
204                .get_full_effects(client, tx_digest, tx_type, consensus_position, options)
205                .await;
206        }
207    }

209    #[instrument(level = "debug", skip_all, err(level = "debug"), fields(tx_digest = ?tx_digest, consensus_position = ?consensus_position, ret_effects_digest = tracing::field::Empty))]
210    async fn get_full_effects<A>(
211        &self,
212        client: Arc<SafeClient<A>>,
213        tx_digest: Option<TransactionDigest>,
214        tx_type: TxType,
215        consensus_position: Option<ConsensusPosition>,
216        options: &SubmitTransactionOptions,
217    ) -> Result<(TransactionEffectsDigest, Box<ExecutedData>, bool), TransactionRequestError>
218    where
219        A: AuthorityAPI + Send + Sync + 'static + Clone,
220    {
221        let ping_type = get_ping_type(&tx_digest, tx_type);
222        let request = WaitForEffectsRequest {
223            transaction_digest: tx_digest,
224            consensus_position,
225            include_details: true,
226            ping_type,
227        };
228
229        match timeout(
230            WAIT_FOR_EFFECTS_TIMEOUT,
231            client.wait_for_effects(request, options.forwarded_client_addr),
232        )
233        .await
234        {
235            Ok(Ok(response)) => match response {
236                WaitForEffectsResponse::Executed {
237                    effects_digest,
238                    details,
239                    fast_path,
240                } => {
241                    if let Some(details) = details {
242                        tracing::Span::current()
243                            .record("ret_effects_digest", format!("{:?}", effects_digest));
244                        Ok((effects_digest, details, fast_path))
245                    } else {
246                        tracing::debug!("Execution data not found, retrying...");
247                        Err(TransactionRequestError::ValidatorInternal(
248                            "Execution data not found".to_string(),
249                        ))
250                    }
251                }
252                WaitForEffectsResponse::Rejected { error } => match error {
253                    Some(e) => Err(TransactionRequestError::RejectedAtValidator(e)),
254                    // Even though this response is not an error, returning an error which is required
255                    // by the function signature. This will get ignored by the caller as a retriable error.
256                    None => Err(TransactionRequestError::RejectedByConsensus),
257                },
258                WaitForEffectsResponse::Expired { epoch, round } => Err(
259                    TransactionRequestError::StatusExpired(epoch, round.unwrap_or(0)),
260                ),
261            },
262            Ok(Err(e)) => Err(TransactionRequestError::Aborted(e)),
263            Err(_) => Err(TransactionRequestError::TimedOutGettingFullEffectsAtValidator),
264        }
265    }

267    #[instrument(level = "debug", skip_all, err(level = "debug"), ret, fields(consensus_position = ?consensus_position))]
268    async fn wait_for_acknowledgments<A>(
269        &self,
270        authority_aggregator: &Arc<AuthorityAggregator<A>>,
271        client_monitor: &Arc<ValidatorClientMonitor<A>>,
272        tx_digest: Option<TransactionDigest>,
273        tx_type: TxType,
274        consensus_position: Option<ConsensusPosition>,
275        options: &SubmitTransactionOptions,
276        submitted_tx_to_validator: AuthorityName,
277    ) -> Result<TransactionEffectsDigest, TransactionDriverError>
278    where
279        A: AuthorityAPI + Send + Sync + 'static + Clone,
280    {
281        let ping_type = get_ping_type(&tx_digest, tx_type);
282        let ping_label = if tx_digest.is_none() { "true" } else { "false" };
283        self.metrics
284            .certified_effects_ack_attempts
285            .with_label_values(&[tx_type.as_str(), ping_label])
286            .inc();
287        let timer = tokio::time::Instant::now();
288        let clients = authority_aggregator
289            .authority_clients
290            .iter()
291            .collect::<Vec<_>>();
292        let committee = authority_aggregator.committee.clone();
293        let raw_request = RawWaitForEffectsRequest::try_from(WaitForEffectsRequest {
294            transaction_digest: tx_digest,
295            consensus_position,
296            include_details: false,
297            ping_type,
298        })
299        .unwrap();
300
301        // Broadcast requests for digest acknowledgments against all validators.
302        let mut futures = FuturesUnordered::new();
303        for (name, client) in clients {
304            let client = client.clone();
305            let name = *name;
306            let display_name = authority_aggregator.get_display_name(&name);
307
308            let raw_request = raw_request.clone();
309            let future = async move {
310                match timeout(
311                    WAIT_FOR_EFFECTS_TIMEOUT,
312                    self.wait_for_acknowledgment_rpc(
313                        name,
314                        display_name.clone(),
315                        &client,
316                        client_monitor,
317                        &raw_request,
318                        options,
319                    ),
320                )
321                .await
322                {
323                    Ok(result) => (name, result),
324                    Err(_) => {
325                        client_monitor.record_interaction_result(OperationFeedback {
326                            authority_name: name,
327                            display_name,
328                            operation: OperationType::Effects,
329                            ping_type,
330                            result: Err(()),
331                        });
332                        (name, Err(SuiErrorKind::TimeoutError.into()))
333                    }
334                }
335            };
336
337            futures.push(future);
338        }
339
340        let mut effects_digest_aggregators: BTreeMap<
341            TransactionEffectsDigest,
342            StatusAggregator<()>,
343        > = BTreeMap::new();
344        // Collect responses from validators which observed the transaction getting rejected,
345        // and rejected the transaction with errors non-retriable with new transaction submissions.
346        let mut non_retriable_errors_aggregator =
347            StatusAggregator::<TransactionRequestError>::new(committee.clone());
348        // Collect responses from validators which observed the transaction getting rejected,
349        // and rejected the transaction with errors retriable with new transaction submissions.
350        let mut retriable_errors_aggregator =
351            StatusAggregator::<TransactionRequestError>::new(committee.clone());
352        // Collect responses from validators which observed the transaction getting rejected,
353        // but do not have a local reason to reject the transaction. The validator could have
354        // accepted the transaction during voting, or the reason has been lost.
355        let mut reason_not_found_aggregator = StatusAggregator::<()>::new(committee.clone());
356        // Collect responses from validators which observed the transaction getting executed using fast path.
357        let mut fast_path_aggregator = StatusAggregator::<()>::new(committee.clone());
358
359        // Every validator returns at most one WaitForEffectsResponse.
360        while let Some((name, response)) = futures.next().await {
361            match response {
362                Ok(WaitForEffectsResponse::Executed {
363                    effects_digest,
364                    details: _,
365                    fast_path,
366                }) => {
367                    if fast_path {
368                        if tx_type != TxType::SingleWriter {
369                            tracing::warn!(
370                                "Fast path is only supported for single writer transactions, name={name}"
371                            );
372                        } else {
373                            fast_path_aggregator.insert(name, ());
374                        }
375                    }
376
377                    let aggregator = effects_digest_aggregators
378                        .entry(effects_digest)
379                        .or_insert_with(|| StatusAggregator::<()>::new(committee.clone()));
380                    aggregator.insert(name, ());
381
382                    if aggregator.reached_quorum_threshold() {
383                        let quorum_weight = aggregator.total_votes();
384                        for (other_digest, other_aggregator) in effects_digest_aggregators {
385                            if other_digest != effects_digest && other_aggregator.total_votes() > 0
386                            {
387                                tracing::warn!(
388                                    ?name,
389                                    "Effects digest inconsistency detected: quorum digest {effects_digest:?} (weight {quorum_weight}), other digest {other_digest:?} (weight {})",
390                                    other_aggregator.total_votes()
391                                );
392                                self.metrics.effects_digest_mismatches.inc();
393                            }
394                        }
395                        // Record success and latency
396                        self.metrics
397                            .certified_effects_ack_successes
398                            .with_label_values(&[tx_type.as_str(), ping_label])
399                            .inc();
400                        self.metrics
401                            .certified_effects_ack_latency
402                            .with_label_values(&[tx_type.as_str(), ping_label])
403                            .observe(timer.elapsed().as_secs_f64());
404
405                        if fast_path_aggregator.reached_quorum_threshold() {
406                            // get the display name of the validator that the transaction has been submitted to
407                            let display_name =
408                                authority_aggregator.get_display_name(&submitted_tx_to_validator);
409
410                            self.metrics
411                                .transaction_fastpath_acked
412                                .with_label_values(&[&display_name, ping_label])
413                                .inc();
414                        }
415
416                        return Ok(effects_digest);
417                    }
418                }
419                Ok(WaitForEffectsResponse::Rejected { error }) => {
420                    if let Some(e) = error {
421                        tracing::trace!(name = ?name.concise(), "Rejected at validator: {:?}", e);
422                        let error = TransactionRequestError::RejectedAtValidator(e);
423                        if error.is_submission_retriable() {
424                            retriable_errors_aggregator.insert(name, error);
425                        } else {
426                            non_retriable_errors_aggregator.insert(name, error);
427                        }
428                    } else {
429                        tracing::trace!(name = ?name.concise(), "Not found at validator");
430                        reason_not_found_aggregator.insert(name, ());
431                    }
432                    self.metrics
433                        .rejection_acks
434                        .with_label_values(&[tx_type.as_str(), ping_label])
435                        .inc();
436                }
437                Ok(WaitForEffectsResponse::Expired { epoch, round }) => {
438                    let error = TransactionRequestError::StatusExpired(epoch, round.unwrap_or(0));
439                    // Expired status is submission retriable.
440                    retriable_errors_aggregator.insert(name, error);
441                    self.metrics
442                        .expiration_acks
443                        .with_label_values(&[tx_type.as_str(), ping_label])
444                        .inc();
445                }
446                Err(error) => {
447                    let error = TransactionRequestError::Aborted(error);
448                    if error.is_submission_retriable() {
449                        retriable_errors_aggregator.insert(name, error);
450                    } else {
451                        non_retriable_errors_aggregator.insert(name, error);
452                    }
453                }
454            };
455
456            // Adding vote up between different StatusAggregators without de-duplication is ok,
457            // because each authority only returns one response.
458            let executed_weight: u64 = effects_digest_aggregators
459                .values()
460                .map(|agg| agg.total_votes())
461                .sum();
462            let total_weight = executed_weight
463                + reason_not_found_aggregator.total_votes()
464                + non_retriable_errors_aggregator.total_votes()
465                + retriable_errors_aggregator.total_votes();
466            let remaining_weight = committee.total_votes() - total_weight;
467
468            // Wait for a quorum of responses, to not summarize the responses too early.
469            // If neither of the branches can exit the loop, the loop will eventually terminate when responses are
470            // gathered from all validators. The time is bounded by WAIT_FOR_EFFECTS_TIMEOUT.
471            //
472            // The most important goal here is to aggregate enough useful errors to be actionable.
473            // Breaking the loop at the earliest possible time is not the goal.
474            if total_weight >= committee.quorum_threshold() {
475                // Try returning non-retriable aggregated error first.
476                if non_retriable_errors_aggregator.total_votes() >= committee.validity_threshold() {
477                    return Err(TransactionDriverError::RejectedByValidators {
478                        submission_non_retriable_errors: aggregate_request_errors(
479                            non_retriable_errors_aggregator.status_by_authority(),
480                        ),
481                        submission_retriable_errors: aggregate_request_errors(
482                            retriable_errors_aggregator.status_by_authority(),
483                        ),
484                    });
485                }
486                // Return a retriable aggregated error only if it becomes impossible to gather enough non-retriable errors.
487                // reason_not_found_aggregator is excluded here intentionally because it does not contain a useful error message.
488                if non_retriable_errors_aggregator.total_votes() + remaining_weight
489                    < committee.validity_threshold()
490                    && retriable_errors_aggregator.total_votes()
491                        + non_retriable_errors_aggregator.total_votes()
492                        >= committee.validity_threshold()
493                {
494                    let mut observed_effects_digests =
495                        Vec::<(TransactionEffectsDigest, Vec<AuthorityName>, StakeUnit)>::new();
496                    for (effects_digest, aggregator) in effects_digest_aggregators {
497                        observed_effects_digests.push((
498                            effects_digest,
499                            aggregator.authorities(),
500                            aggregator.total_votes(),
501                        ));
502                    }
503                    return Err(TransactionDriverError::Aborted {
504                        submission_non_retriable_errors: aggregate_request_errors(
505                            non_retriable_errors_aggregator.status_by_authority(),
506                        ),
507                        submission_retriable_errors: aggregate_request_errors(
508                            retriable_errors_aggregator.status_by_authority(),
509                        ),
510                        observed_effects_digests: AggregatedEffectsDigests {
511                            digests: observed_effects_digests,
512                        },
513                    });
514                }
515            }
516        }
517
518        // At this point, no effects digest has reached quorum. But failed responses do not reach
519        // validity threshold either.
520        let retriable_weight =
521            retriable_errors_aggregator.total_votes() + reason_not_found_aggregator.total_votes();
522        // Whether the transaction is retriable regardless of known effects.
523        let mut submission_retriable = retriable_weight >= committee.quorum_threshold();
524        let mut observed_effects_digests =
525            Vec::<(TransactionEffectsDigest, Vec<AuthorityName>, StakeUnit)>::new();
526        for (effects_digest, aggregator) in effects_digest_aggregators {
527            // This effects digest can still get certified, so the transaction is retriable.
528            if aggregator.total_votes() + retriable_weight >= committee.quorum_threshold() {
529                submission_retriable = true;
530            }
531            observed_effects_digests.push((
532                effects_digest,
533                aggregator.authorities(),
534                aggregator.total_votes(),
535            ));
536        }
537        if submission_retriable {
538            Err(TransactionDriverError::Aborted {
539                submission_non_retriable_errors: aggregate_request_errors(
540                    non_retriable_errors_aggregator.status_by_authority(),
541                ),
542                submission_retriable_errors: aggregate_request_errors(
543                    retriable_errors_aggregator.status_by_authority(),
544                ),
545                observed_effects_digests: AggregatedEffectsDigests {
546                    digests: observed_effects_digests,
547                },
548            })
549        } else {
550            if observed_effects_digests.len() <= 1 {
551                debug_fatal!(
552                    "Expect at least 2 effects digests, but got {:?}",
553                    observed_effects_digests
554                );
555            }
556            Err(TransactionDriverError::ForkedExecution {
557                observed_effects_digests: AggregatedEffectsDigests {
558                    digests: observed_effects_digests,
559                },
560                submission_non_retriable_errors: aggregate_request_errors(
561                    non_retriable_errors_aggregator.status_by_authority(),
562                ),
563                submission_retriable_errors: aggregate_request_errors(
564                    retriable_errors_aggregator.status_by_authority(),
565                ),
566            })
567        }
568    }

570    #[instrument(level = "debug", skip_all, err(level = "debug"), ret, fields(validator_display_name = ?display_name))]
571    async fn wait_for_acknowledgment_rpc<A>(
572        &self,
573        name: AuthorityName,
574        display_name: String,
575        client: &Arc<SafeClient<A>>,
576        client_monitor: &Arc<ValidatorClientMonitor<A>>,
577        raw_request: &RawWaitForEffectsRequest,
578        options: &SubmitTransactionOptions,
579    ) -> Result<WaitForEffectsResponse, SuiError>
580    where
581        A: AuthorityAPI + Send + Sync + 'static + Clone,
582    {
583        let effects_start = Instant::now();
584        let backoff = ExponentialBackoff::new(MAX_WAIT_FOR_EFFECTS_RETRY_DELAY);
585        let ping_type = raw_request.get_ping_type();
586        // This loop should only retry errors that are retriable without new submission.
587        for (attempt, delay) in backoff.enumerate() {
588            let request: WaitForEffectsRequest = raw_request.clone().try_into().unwrap();
589            let result = client
590                .wait_for_effects(request, options.forwarded_client_addr)
591                .await;
592            match result {
593                Ok(response) => {
594                    let latency = effects_start.elapsed();
595                    client_monitor.record_interaction_result(OperationFeedback {
596                        authority_name: name,
597                        display_name: display_name.clone(),
598                        operation: OperationType::Effects,
599                        ping_type,
600                        result: Ok(latency),
601                    });
602                    return Ok(response);
603                }
604                Err(e) => {
605                    client_monitor.record_interaction_result(OperationFeedback {
606                        authority_name: name,
607                        display_name: display_name.clone(),
608                        operation: OperationType::Effects,
609                        ping_type,
610                        result: Err(()),
611                    });
612                    if !matches!(e.as_inner(), SuiErrorKind::RpcError(_, _)) {
613                        return Err(e);
614                    }
615                    tracing::trace!(
616                        ?name,
617                        "Wait for effects acknowledgement (attempt {attempt}): rpc error: {:?}",
618                        e
619                    );
620                }
621            };
622            sleep(delay).await;
623        }
624        Err(SuiErrorKind::TimeoutError.into())
625    }

627    /// Creates the final full response.
628    fn get_quorum_transaction_response(
629        &self,
630        effects_digest: TransactionEffectsDigest,
631        executed_data: Box<ExecutedData>,
632    ) -> QuorumTransactionResponse {
633        self.metrics.executed_transactions.inc();
634
635        tracing::debug!("Transaction executed with effects digest: {effects_digest}",);
636
637        let epoch = executed_data.effects.executed_epoch();
638        let details = FinalizedEffects {
639            effects: executed_data.effects,
640            finality_info: EffectsFinalityInfo::QuorumExecuted(epoch),
641        };
642
643        QuorumTransactionResponse {
644            effects: details,
645            events: executed_data.events,
646            input_objects: if !executed_data.input_objects.is_empty() {
647                Some(executed_data.input_objects)
648            } else {
649                None
650            },
651            output_objects: if !executed_data.output_objects.is_empty() {
652                Some(executed_data.output_objects)
653            } else {
654                None
655            },
656            auxiliary_data: None,
657        }
658    }
659}

661fn get_ping_type(tx_digest: &Option<TransactionDigest>, tx_type: TxType) -> Option<PingType> {
662    if tx_digest.is_none() {
663        Some(match tx_type {
664            TxType::SingleWriter => PingType::FastPath,
665            TxType::SharedObject => PingType::Consensus,
666        })
667    } else {
668        None
669    }
