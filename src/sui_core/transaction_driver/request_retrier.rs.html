19/// Select validators with latencies within 2% of the lowest latency.
20const SELECT_LATENCY_DELTA: f64 = 0.02;
21
22/// Provides the next target validator to retry operations,
23/// and gathers the errors along with the operations.
24///
25/// In TransactionDriver, submitting a transaction and getting full effects follow the same pattern:
26/// 1. Retry against all validators until the operation succeeds.
27/// 2. If nonâ€‘retriable errors from a quorum of validators are returned, the operation should fail permanently.
28///
29/// When an `allowed_validators` is provided, only the validators in the list will be used to submit the transaction to.
30/// When the allowed validator list is empty, any validator can be used an then the validators are selected based on their scores.
31///
32/// When a `blocked_validators` is provided, the validators in the list cannot be used to submit the transaction to.
33/// When the blocked validator list is empty, no restrictions are applied.
34///
35/// This component helps to manager this retry pattern.
36pub(crate) struct RequestRetrier<A: Clone> {
37    ranked_clients: VecDeque<(AuthorityName, Arc<SafeClient<A>>)>,
38    pub(crate) non_retriable_errors_aggregator: StatusAggregator<TransactionRequestError>,
39    pub(crate) retriable_errors_aggregator: StatusAggregator<TransactionRequestError>,
40}
41
42impl<A: Clone> RequestRetrier<A> {
43    pub(crate) fn new(
44        auth_agg: &Arc<AuthorityAggregator<A>>,
45        client_monitor: &Arc<ValidatorClientMonitor<A>>,
46        allowed_validators: Vec<String>,
47        blocked_validators: Vec<String>,
48    ) -> Self {
49        let ranked_validators = client_monitor
50            .select_shuffled_preferred_validators(&auth_agg.committee, SELECT_LATENCY_DELTA);
51        let ranked_clients = ranked_validators
52            .into_iter()
53            .map(|name| (name, auth_agg.get_display_name(&name)))
54            .filter(|(_name, display_name)| {
55                allowed_validators.is_empty() || allowed_validators.contains(display_name)
56            })
57            .filter(|(_name, display_name)| {
58                blocked_validators.is_empty() || !blocked_validators.contains(display_name)
59            })
60            .filter_map(|(name, _display_name)| {
61                // There is not guarantee that the `name` are in the `auth_agg.authority_clients` if those are coming from the list
62                // of `allowed_validators`, as the provided `auth_agg` might have been updated with a new committee that doesn't contain the validator in question.
63                auth_agg
64                    .authority_clients
65                    .get(&name)
66                    .map(|client| (name, client.clone()))
67            })
68            .collect::<VecDeque<_>>();
69        let non_retriable_errors_aggregator = StatusAggregator::new(auth_agg.committee.clone());
70        let retriable_errors_aggregator = StatusAggregator::new(auth_agg.committee.clone());
71        Self {
72            ranked_clients,
73            non_retriable_errors_aggregator,
74            retriable_errors_aggregator,
75        }
76    }
77
78    // Selects the next target validator to attempt an operation.
79    pub(crate) fn next_target(
80        &mut self,
81    ) -> Result<(AuthorityName, Arc<SafeClient<A>>), TransactionDriverError> {
82        if let Some((name, client)) = self.ranked_clients.pop_front() {
83            return Ok((name, client));
84        };
85
86        if self
87            .non_retriable_errors_aggregator
88            .reached_validity_threshold()
89        {
90            Err(TransactionDriverError::RejectedByValidators {
91                submission_non_retriable_errors: aggregate_request_errors(
92                    self.non_retriable_errors_aggregator.status_by_authority(),
93                ),
94                submission_retriable_errors: aggregate_request_errors(
95                    self.retriable_errors_aggregator.status_by_authority(),
96                ),
97            })
98        } else {
99            Err(TransactionDriverError::Aborted {
100                submission_non_retriable_errors: aggregate_request_errors(
101                    self.non_retriable_errors_aggregator.status_by_authority(),
102                ),
103                submission_retriable_errors: aggregate_request_errors(
104                    self.retriable_errors_aggregator.status_by_authority(),
105                ),
106                observed_effects_digests: AggregatedEffectsDigests {
107                    digests: Vec::new(),
108                },
109            })
110        }
111    }
112
113    // Adds an error associated with the operation against the authority.
114    //
115    // Returns an error if it has aggregated >= f+1 submission non-retriable errors.
116    // In this case, the transaction cannot finalize unless there is a software bug
117    // or > f malicious validators.
118    pub(crate) fn add_error(
119        &mut self,
120        name: AuthorityName,
121        error: TransactionRequestError,
122    ) -> Result<(), TransactionDriverError> {
123        if error.is_submission_retriable() {
124            self.retriable_errors_aggregator.insert(name, error);
125        } else {
126            self.non_retriable_errors_aggregator.insert(name, error);
127            if self
128                .non_retriable_errors_aggregator
129                .reached_validity_threshold()
130            {
131                return Err(TransactionDriverError::RejectedByValidators {
132                    submission_non_retriable_errors: aggregate_request_errors(
133                        self.non_retriable_errors_aggregator.status_by_authority(),
134                    ),
135                    submission_retriable_errors: aggregate_request_errors(
136                        self.retriable_errors_aggregator.status_by_authority(),
137                    ),
138                });
139            }
140        }
141
142        Ok(())
143    }
144}

169    #[tokio::test]
170    async fn test_next_target() {
171        let auth_agg = Arc::new(get_authority_aggregator(4));
172        let client_monitor = Arc::new(ValidatorClientMonitor::new_for_test(auth_agg.clone()));
173        let mut retrier = RequestRetrier::new(&auth_agg, &client_monitor, vec![], vec![]);
174
175        for name in auth_agg.committee.names() {
176            retrier.next_target().unwrap();
177            retrier
178                .add_error(
179                    *name,
180                    TransactionRequestError::TimedOutSubmittingTransaction,
181                )
182                .unwrap();
183        }
184
185        let Err(error) = retrier.next_target() else {
186            panic!("Expected an error");
187        };
188        assert!(error.is_submission_retriable(), "{}", error);
189    }

191    #[tokio::test]
192    async fn test_allowed_validators() {
193        use sui_types::crypto::{AuthorityKeyPair, KeypairTraits, get_key_pair};
194
195        let auth_agg = Arc::new(get_authority_aggregator(4));
196        let client_monitor = Arc::new(ValidatorClientMonitor::new_for_test(auth_agg.clone()));
197
198        // Create validators that don't exist in the auth_agg
199        let (_, key_pair1): (_, AuthorityKeyPair) = get_key_pair();
200        let (_, key_pair2): (_, AuthorityKeyPair) = get_key_pair();
201        let unknown_validator1: AuthorityName = key_pair1.public().into();
202        let unknown_validator2: AuthorityName = key_pair2.public().into();
203
204        // Mix of unknown validators and one known validator
205        let authorities: Vec<_> = auth_agg.committee.names().copied().collect();
206
207        println!("Case 1. Mix of unknown validators and one known validator");
208        {
209            let allowed_validators = vec![
210                unknown_validator1.concise().to_string(),
211                unknown_validator2.concise().to_string(),
212                authorities[0].concise().to_string(), // This one exists in auth_agg
213            ];
214
215            let retrier =
216                RequestRetrier::new(&auth_agg, &client_monitor, allowed_validators, vec![]);
217
218            // Should only have 1 remaining client (the known validator)
219            assert_eq!(retrier.ranked_clients.len(), 1);
220            assert_eq!(retrier.ranked_clients[0].0, authorities[0]);
221        }
222
223        println!("Case 2. Only unknown validators are provided");
224        {
225            let allowed_validators = vec![
226                unknown_validator1.concise().to_string(),
227                unknown_validator2.concise().to_string(),
228            ];
229
230            let retrier =
231                RequestRetrier::new(&auth_agg, &client_monitor, allowed_validators, vec![]);
232
233            // Should have no remaining clients since none of the allowed validators exist
234            assert_eq!(retrier.ranked_clients.len(), 0);
235        }
236    }

238    #[tokio::test]
239    async fn test_blocked_validators() {
240        let auth_agg = Arc::new(get_authority_aggregator(4));
241        let client_monitor = Arc::new(ValidatorClientMonitor::new_for_test(auth_agg.clone()));
242
243        // Create a list of validators that should be blocked and never picked up by the retrier.
244        let blocked_validators = auth_agg
245            .committee
246            .names()
247            .take(3)
248            .copied()
249            .collect::<Vec<_>>();
250        let blocked_display_names = blocked_validators
251            .iter()
252            .map(|name| auth_agg.get_display_name(name))
253            .collect::<Vec<_>>();
254
255        // Only the last validator will be picked up.
256        let allowed_validator = auth_agg.committee.names().nth(3).unwrap();
257
258        let mut retrier =
259            RequestRetrier::new(&auth_agg, &client_monitor, vec![], blocked_display_names);
260
261        // The last validator will be picked up.
262        assert_eq!(retrier.next_target().unwrap().0, *allowed_validator);
263        // No more validators will be picked up.
264        assert!(retrier.next_target().is_err());
265    }

297                .unwrap();
298            // 100% stake.
299            retrier
300                .add_error(
301                    authorities[1],
302                    TransactionRequestError::TimedOutSubmittingTransaction,
303                )
304                .unwrap();
305            // Still there is no aggregated error.
306        }
307
308        // Add mix of retriable and non-retriable errors.
309        {
310            let client_monitor = Arc::new(ValidatorClientMonitor::new_for_test(auth_agg.clone()));

312
313            // 25% stake retriable error.

337                        SuiErrorKind::UserInputError {
338                            error: UserInputError::EmptyCommandInput,
339                        }
340                        .into(),
341                    ),
342                )
343                .unwrap_err();
344            // The aggregated error is non-retriable.
345            assert!(!aggregated_error.is_submission_retriable());
