37// Using a long timeout for transaction submission is ok, because good performing validators
38// are chosen first.
39const SUBMIT_TRANSACTION_TIMEOUT: Duration = Duration::from_secs(10);
40
41pub(crate) struct TransactionSubmitter {
42    metrics: Arc<TransactionDriverMetrics>,
43}
44
45impl TransactionSubmitter {
46    pub(crate) fn new(metrics: Arc<TransactionDriverMetrics>) -> Self {
47        Self { metrics }
48    }

50    #[instrument(level = "debug", skip_all, err(level = "debug"))]
51    pub(crate) async fn submit_transaction<A>(
52        &self,
53        authority_aggregator: &Arc<AuthorityAggregator<A>>,
54        client_monitor: &Arc<ValidatorClientMonitor<A>>,
55        tx_type: TxType,
56        amplification_factor: u64,
57        request: SubmitTxRequest,
58        options: &SubmitTransactionOptions,
59    ) -> Result<(AuthorityName, SubmitTxResult), TransactionDriverError>
60    where
61        A: AuthorityAPI + Send + Sync + 'static + Clone,
62    {
63        let start_time = Instant::now();
64
65        self.metrics
66            .submit_amplification_factor
67            .observe(amplification_factor as f64);
68
69        let mut retrier = RequestRetrier::new(
70            authority_aggregator,
71            client_monitor,
72            tx_type,
73            options.allowed_validators.clone(),
74        );
75
76        let ping_label = if request.ping_type.is_some() {
77            "true"
78        } else {
79            "false"
80        };
81        let mut retries = 0;
82        let mut request_rpcs = FuturesUnordered::new();
83
84        // This loop terminates when there are enough (f+1) non-retriable errors when submitting the transaction,
85        // or all feasible targets returned errors or timed out.
86        loop {
87            // Try to fill up to amplification_factor concurrent requests
88            while request_rpcs.len() < amplification_factor as usize {
89                match retrier.next_target() {
90                    Ok((name, client)) => {
91                        let display_name = authority_aggregator.get_display_name(&name);
92                        self.metrics
93                            .validator_selections
94                            .with_label_values(&[&display_name, tx_type.as_str(), ping_label])
95                            .inc();
96
97                        // Create a future that returns the name and display_name along with the result
98                        let submit_fut = self.submit_transaction_once(
99                            client,
100                            &request,
101                            options,
102                            client_monitor,
103                            name,
104                            display_name.clone(),
105                        );
106
107                        let wrapped_fut = async move {
108                            let result = submit_fut.await;
109                            (name, display_name, result)
110                        };
111
112                        request_rpcs.push(wrapped_fut);
113                    }
114                    Err(_) if request_rpcs.is_empty() => {
115                        // No more targets and no requests in flight
116                        return Err(TransactionDriverError::Aborted {
117                            submission_non_retriable_errors: aggregate_request_errors(
118                                retrier
119                                    .non_retriable_errors_aggregator
120                                    .status_by_authority(),
121                            ),
122                            submission_retriable_errors: aggregate_request_errors(
123                                retrier.retriable_errors_aggregator.status_by_authority(),
124                            ),
125                            observed_effects_digests: AggregatedEffectsDigests {
126                                digests: Vec::new(),
127                            },
128                        });
129                    }
130                    Err(_) => {
131                        // No more targets but still have requests in flight
132                        break;
133                    }
134                }
135            }
136
137            match request_rpcs.next().await {
138                Some((name, display_name, Ok(result))) => {
139                    self.metrics
140                        .validator_submit_transaction_successes
141                        .with_label_values(&[&display_name, tx_type.as_str(), ping_label])
142                        .inc();
143                    self.metrics
144                        .submit_transaction_retries
145                        .observe(retries as f64);
146                    let elapsed = start_time.elapsed().as_secs_f64();
147                    self.metrics
148                        .submit_transaction_latency
149                        .with_label_values(&[&display_name, tx_type.as_str(), ping_label])
150                        .observe(elapsed);
151
152                    return Ok((name, result));
153                }
154                Some((name, display_name, Err(e))) => {
155                    let error_type = e.categorize().into();
156                    self.metrics
157                        .validator_submit_transaction_errors
158                        .with_label_values(&[
159                            &display_name,
160                            error_type,
161                            tx_type.as_str(),
162                            ping_label,
163                        ])
164                        .inc();
165
166                    retries += 1;
167                    retrier.add_error(name, e)?;
168                }
169                None => {
170                    // All requests have been processed.
171                    return Err(TransactionDriverError::Aborted {
172                        submission_non_retriable_errors: aggregate_request_errors(
173                            retrier
174                                .non_retriable_errors_aggregator
175                                .status_by_authority(),
176                        ),
177                        submission_retriable_errors: aggregate_request_errors(
178                            retrier.retriable_errors_aggregator.status_by_authority(),
179                        ),
180                        observed_effects_digests: AggregatedEffectsDigests {
181                            digests: Vec::new(),
182                        },
183                    });
184                }
185            };
186
187            // Yield to prevent this retry loop from starving other tasks.
188            tokio::task::yield_now().await;
189        }
190    }

192    #[instrument(level = "debug", skip_all, err(level = "debug"), ret, fields(validator_display_name = ?display_name))]
193    pub(crate) async fn submit_transaction_once<A>(
194        &self,
195        client: Arc<SafeClient<A>>,
196        request: &SubmitTxRequest,
197        options: &SubmitTransactionOptions,
198        client_monitor: &Arc<ValidatorClientMonitor<A>>,
199        validator: AuthorityName,
200        display_name: String,
201    ) -> Result<SubmitTxResult, TransactionRequestError>
202    where
203        A: AuthorityAPI + Send + Sync + 'static + Clone,
204    {
205        let submit_start = Instant::now();
206
207        let resp = timeout(
208            SUBMIT_TRANSACTION_TIMEOUT,
209            client.submit_transaction(request.clone(), options.forwarded_client_addr),
210        )
211        .await
212        .map_err(|_| {
213            client_monitor.record_interaction_result(OperationFeedback {
214                authority_name: validator,
215                display_name: display_name.clone(),
216                operation: OperationType::Submit,
217                ping_type: request.ping_type,
218                result: Err(()),
219            });
220            TransactionRequestError::TimedOutSubmittingTransaction
221        })?
222        .map_err(|error| {
223            if is_validator_error(error.categorize()) {
224                client_monitor.record_interaction_result(OperationFeedback {
225                    authority_name: validator,
226                    display_name: display_name.clone(),
227                    operation: OperationType::Submit,
228                    ping_type: request.ping_type,
229                    result: Err(()),
230                });
231            }
232            TransactionRequestError::RejectedAtValidator(error)
233        })?;
234
235        if resp.results.len() != 1 {
236            return Err(TransactionRequestError::ValidatorInternal(format!(
237                "Expected exactly 1 result, got {}",
238                resp.results.len()
239            )));
240        }
241        let result = resp.results.into_iter().next().unwrap();
242
243        // Since only one transaction is submitted, it is ok to return error when the submission is rejected.
244        if let SubmitTxResult::Rejected { error } = &result {
245            if is_validator_error(error.categorize()) {
246                client_monitor.record_interaction_result(OperationFeedback {
247                    authority_name: validator,
248                    display_name,
249                    operation: OperationType::Submit,
250                    ping_type: request.ping_type,
251                    result: Err(()),
252                });
253            }
254            return Err(TransactionRequestError::RejectedAtValidator(error.clone()));
255        }
256
257        let latency = submit_start.elapsed();
258        client_monitor.record_interaction_result(OperationFeedback {
259            authority_name: validator,
260            display_name,
261            operation: OperationType::Submit,
262            ping_type: request.ping_type,
263            result: Ok(latency),
264        });
265        Ok(result)
266    }
267}

269// Whether the failure is caused by the peer validator, as opposed to the user or this node.
270fn is_validator_error(category: ErrorCategory) -> bool {
271    matches!(
272        category,
273        ErrorCategory::Aborted
274            | ErrorCategory::Internal
275            | ErrorCategory::ValidatorOverloaded
276            | ErrorCategory::Unavailable
277    )
