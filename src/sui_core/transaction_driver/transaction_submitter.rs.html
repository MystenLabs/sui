37// Using a long timeout for transaction submission is ok, because good performing validators
38// are chosen first.
39const SUBMIT_TRANSACTION_TIMEOUT: Duration = Duration::from_secs(10);
40
41// Delay to submit the transaction to an additional backup validator.
42const SUBMIT_TRANSACTION_BACKUP_REQUEST_DELAY: Duration = Duration::from_secs(1);
43
44pub(crate) struct TransactionSubmitter {
45    metrics: Arc<TransactionDriverMetrics>,
46}
47
48impl TransactionSubmitter {
49    pub(crate) fn new(metrics: Arc<TransactionDriverMetrics>) -> Self {
50        Self { metrics }
51    }

53    #[instrument(level = "debug", skip_all, err(level = "debug"))]
54    pub(crate) async fn submit_transaction<A>(
55        &self,
56        authority_aggregator: &Arc<AuthorityAggregator<A>>,
57        client_monitor: &Arc<ValidatorClientMonitor<A>>,
58        tx_type: TxType,
59        amplification_factor: u64,
60        request: SubmitTxRequest,
61        options: &SubmitTransactionOptions,
62    ) -> Result<(AuthorityName, SubmitTxResult), TransactionDriverError>
63    where
64        A: AuthorityAPI + Send + Sync + 'static + Clone,
65    {
66        let start_time = Instant::now();
67
68        // Limit the amplification factor to [1.=committee size].
69        let amplification_factor = amplification_factor
70            .max(1)
71            .min(authority_aggregator.committee.num_members() as u64);
72        self.metrics
73            .submit_amplification_factor
74            .observe(amplification_factor as f64);
75
76        let mut retrier = RequestRetrier::new(
77            authority_aggregator,
78            client_monitor,
79            options.allowed_validators.clone(),
80            options.blocked_validators.clone(),
81        );
82
83        let ping_label = if request.ping_type.is_some() {
84            "true"
85        } else {
86            "false"
87        };
88        let mut initial_requests = true;
89        let mut retries = 0;
90        let mut backups = 0;
91        let mut request_rpcs = FuturesUnordered::new();
92
93        // This loop terminates when there are enough (f+1) non-retriable errors when submitting the transaction,
94        // or all feasible targets returned errors or timed out.
95        loop {
96            let num_additional_requests = if initial_requests {
97                // Initially, try to fill up to amplification_factor concurrent requests
98                initial_requests = false;
99                amplification_factor
100            } else {
101                // Start another request after seeing a failure (retry) or backup delay has elapsed.
102                1
103            };
104            for _ in 0..num_additional_requests {
105                match retrier.next_target() {
106                    Ok((name, client)) => {
107                        let display_name = authority_aggregator.get_display_name(&name);
108                        self.metrics
109                            .validator_selections
110                            .with_label_values(&[
111                                display_name.as_str(),
112                                tx_type.as_str(),
113                                ping_label,
114                            ])
115                            .inc();
116
117                        // Create a future that returns the name and display_name along with the result
118                        let submit_fut = self.submit_transaction_once(
119                            client,
120                            &request,
121                            options,
122                            client_monitor,
123                            name,
124                            display_name.clone(),
125                        );
126
127                        let wrapped_fut = async move {
128                            let result = submit_fut.await;
129                            (name, display_name, result)
130                        };
131
132                        request_rpcs.push(wrapped_fut);
133                    }
134                    Err(_) => {
135                        if !request_rpcs.is_empty() {
136                            // No more targets but still have requests in flight. Continue to wait for them.
137                            break;
138                        }
139                        // No more targets and no requests in flight. Gather the errors and return.
140                        return Err(TransactionDriverError::Aborted {
141                            submission_non_retriable_errors: aggregate_request_errors(
142                                retrier
143                                    .non_retriable_errors_aggregator
144                                    .status_by_authority(),
145                            ),
146                            submission_retriable_errors: aggregate_request_errors(
147                                retrier.retriable_errors_aggregator.status_by_authority(),
148                            ),
149                            observed_effects_digests: AggregatedEffectsDigests {
150                                digests: Vec::new(),
151                            },
152                        });
153                    }
154                }
155            }
156
157            // Wait for the next available result, or backup delay has elapsed.
158            tokio::select! {
159                result = request_rpcs.next() => {
160                    match result {
161                        Some((name, display_name, Ok(result))) => {
162                            self.metrics
163                                .validator_submit_transaction_successes
164                                .with_label_values(&[display_name.as_str(), tx_type.as_str(), ping_label])
165                                .inc();
166                            self.metrics
167                                .submit_transaction_retries
168                                .observe(retries as f64);
169                            self.metrics
170                                .submit_transaction_backups
171                                .observe(backups as f64);
172                            let elapsed = start_time.elapsed().as_secs_f64();
173                            self.metrics
174                                .submit_transaction_latency
175                                .with_label_values(&[tx_type.as_str(), ping_label])
176                                .observe(elapsed);
177
178                            return Ok((name, result));
179                        }
180                        Some((name, display_name, Err(e))) => {
181                            let error_type: &str = e.categorize().into();
182                            self.metrics
183                                .validator_submit_transaction_errors
184                                .with_label_values(&[
185                                    display_name.as_str(),
186                                    error_type,
187                                    tx_type.as_str(),
188                                    ping_label,
189                                ])
190                                .inc();
191
192                            retries += 1;
193                            retrier.add_error(name, e)?;
194                        }
195                        None => {
196                            // All requests have been processed.
197                            return Err(TransactionDriverError::Aborted {
198                                submission_non_retriable_errors: aggregate_request_errors(
199                                    retrier
200                                        .non_retriable_errors_aggregator
201                                        .status_by_authority(),
202                                ),
203                                submission_retriable_errors: aggregate_request_errors(
204                                    retrier.retriable_errors_aggregator.status_by_authority(),
205                                ),
206                                observed_effects_digests: AggregatedEffectsDigests {
207                                    digests: Vec::new(),
208                                },
209                            });
210                        }
211                    }
212                }
213                _ = tokio::time::sleep(SUBMIT_TRANSACTION_BACKUP_REQUEST_DELAY) => {
214                    // Backup delay elapsed without response, continue to start another request
215                    backups += 1;
216                }
217            }
218
219            // Yield to prevent this retry loop from starving other tasks.
220            tokio::task::yield_now().await;
221        }
222    }

224    #[instrument(level = "debug", skip_all, err(level = "debug"), ret, fields(validator_display_name = ?display_name))]
225    pub(crate) async fn submit_transaction_once<A>(
226        &self,
227        client: Arc<SafeClient<A>>,
228        request: &SubmitTxRequest,
229        options: &SubmitTransactionOptions,
230        client_monitor: &Arc<ValidatorClientMonitor<A>>,
231        validator: AuthorityName,
232        display_name: String,
233    ) -> Result<SubmitTxResult, TransactionRequestError>
234    where
235        A: AuthorityAPI + Send + Sync + 'static + Clone,
236    {
237        let submit_start = Instant::now();
238
239        let resp = timeout(
240            SUBMIT_TRANSACTION_TIMEOUT,
241            client.submit_transaction(request.clone(), options.forwarded_client_addr),
242        )
243        .await
244        .map_err(|_| {
245            client_monitor.record_interaction_result(OperationFeedback {
246                authority_name: validator,
247                display_name: display_name.clone(),
248                operation: OperationType::Submit,
249                ping_type: request.ping_type,
250                result: Err(()),
251            });
252            TransactionRequestError::TimedOutSubmittingTransaction
253        })?
254        .map_err(|error| {
255            if is_validator_error(error.categorize()) {
256                client_monitor.record_interaction_result(OperationFeedback {
257                    authority_name: validator,
258                    display_name: display_name.clone(),
259                    operation: OperationType::Submit,
260                    ping_type: request.ping_type,
261                    result: Err(()),
262                });
263            }
264            TransactionRequestError::RejectedAtValidator(error)
265        })?;
266
267        if resp.results.len() != 1 {
268            return Err(TransactionRequestError::ValidatorInternal(format!(
269                "Expected exactly 1 result, got {}",
270                resp.results.len()
271            )));
272        }
273        let result = resp.results.into_iter().next().unwrap();
274
275        // Since only one transaction is submitted, it is ok to return error when the submission is rejected.
276        if let SubmitTxResult::Rejected { error } = &result {
277            if is_validator_error(error.categorize()) {
278                client_monitor.record_interaction_result(OperationFeedback {
279                    authority_name: validator,
280                    display_name,
281                    operation: OperationType::Submit,
282                    ping_type: request.ping_type,
283                    result: Err(()),
284                });
285            }
286            return Err(TransactionRequestError::RejectedAtValidator(error.clone()));
287        }
288
289        let latency = submit_start.elapsed();
290        client_monitor.record_interaction_result(OperationFeedback {
291            authority_name: validator,
292            display_name,
293            operation: OperationType::Submit,
294            ping_type: request.ping_type,
295            result: Ok(latency),
296        });
297        Ok(result)
298    }
299}
300
301// Whether the failure is caused by the peer validator, as opposed to the user or this node.
302fn is_validator_error(category: ErrorCategory) -> bool {
303    matches!(
304        category,
305        ErrorCategory::Aborted
306            | ErrorCategory::Internal
307            | ErrorCategory::ValidatorOverloaded
308            | ErrorCategory::Unavailable
309    )
