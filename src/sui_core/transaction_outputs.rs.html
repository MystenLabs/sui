32impl TransactionOutputs {
33    // Convert InnerTemporaryStore + Effects into the exact set of updates to the store
34    pub fn build_transaction_outputs(
35        transaction: VerifiedTransaction,
36        effects: TransactionEffects,
37        inner_temporary_store: InnerTemporaryStore,
38        unchanged_loaded_runtime_objects: Vec<ObjectKey>,
39    ) -> TransactionOutputs {
40        let InnerTemporaryStore {
41            input_objects,
42            stream_ended_consensus_objects,
43            mutable_inputs,
44            written,
45            events,
46            accumulator_events,
47            loaded_runtime_objects: _,
48            binary_config: _,
49            runtime_packages_loaded_from_db: _,
50            lamport_version,
51            accumulator_running_max_withdraws: _,
52        } = inner_temporary_store;
53
54        let tx_digest = *transaction.digest();
55
56        // Get the actual set of objects that have been received -- any received
57        // object will show up in the modified-at set.
58        let modified_at: HashSet<_> = effects.modified_at_versions().into_iter().collect();
59        let possible_to_receive = transaction.transaction_data().receiving_objects();
60        let received_objects = possible_to_receive
61            .into_iter()
62            .filter(|obj_ref| modified_at.contains(&(obj_ref.0, obj_ref.1)));
63
64        // We record any received or deleted objects since they could be pruned, and smear object
65        // removals from consensus in the marker table. For deleted entries in the marker table we
66        // need to make sure we don't accidentally overwrite entries.
67        let markers: Vec<_> = {
68            let received = received_objects.clone().map(|objref| {
69                (
70                    // TODO: Add support for receiving consensus objects. For now this assumes fastpath.
71                    FullObjectKey::new(FullObjectID::new(objref.0, None), objref.1),
72                    MarkerValue::Received,
73                )
74            });
75
76            let tombstones = effects
77                .all_tombstones()
78                .into_iter()
79                .map(|(object_id, version)| {
80                    let consensus_key = input_objects
81                        .get(&object_id)
82                        .filter(|o| o.is_consensus())
83                        .map(|o| FullObjectKey::new(o.full_id(), version));
84                    if let Some(consensus_key) = consensus_key {
85                        (consensus_key, MarkerValue::ConsensusStreamEnded(tx_digest))
86                    } else {
87                        (
88                            FullObjectKey::new(FullObjectID::new(object_id, None), version),
89                            MarkerValue::FastpathStreamEnded,
90                        )
91                    }
92                });
93
94            let fastpath_stream_ended =
95                effects
96                    .transferred_to_consensus()
97                    .into_iter()
98                    .map(|(object_id, version, _)| {
99                        // Note: it's a bit of a misnomer to mark an object as `FastpathStreamEnded`
100                        // when it could have been transferred to consensus from `ObjectOwner`, as
101                        // its root owner may not have been a fastpath object. However, whether or
102                        // not it was technically in the fastpath at the version the marker is
103                        // written, it certainly is not in the fastpath *anymore*. This is needed
104                        // to produce the required behavior in `ObjectCacheRead::multi_input_objects_available`
105                        // when checking whether receiving objects are available.
106                        (
107                            FullObjectKey::new(FullObjectID::new(object_id, None), version),
108                            MarkerValue::FastpathStreamEnded,
109                        )
110                    });
111
112            let consensus_stream_ended = effects
113                .transferred_from_consensus()
114                .into_iter()
115                .chain(effects.consensus_owner_changed())
116                .map(|(object_id, version, _)| {
117                    let object = input_objects
118                        .get(&object_id)
119                        .expect("stream-ended object must be in input_objects");
120                    (
121                        FullObjectKey::new(object.full_id(), version),
122                        MarkerValue::ConsensusStreamEnded(tx_digest),
123                    )
124                });
125
126            // We "smear" removed consensus objects in the marker table to allow for proper
127            // sequencing of transactions that are submitted after the consensus stream ends.
128            // This means writing duplicate copies of the `ConsensusStreamEnded` marker for
129            // every output version that was scheduled to be created.
130            // NB: that we do _not_ smear objects that were taken immutably in the transaction
131            // (because these are not assigned output versions).
132            let smeared_objects = effects.stream_ended_mutably_accessed_consensus_objects();
133            let consensus_smears = smeared_objects.into_iter().map(|object_id| {
134                let id = input_objects
135                    .get(&object_id)
136                    .map(|obj| obj.full_id())
137                    .unwrap_or_else(|| {
138                        let start_version = stream_ended_consensus_objects.get(&object_id)
139                            .expect("stream-ended object must be in either input_objects or stream_ended_consensus_objects");
140                        FullObjectID::new(object_id, Some(*start_version))
141                    });
142                (
143                    FullObjectKey::new(id, lamport_version),
144                    MarkerValue::ConsensusStreamEnded(tx_digest),
145                )
146            });
147
148            received
149                .chain(tombstones)
150                .chain(fastpath_stream_ended)
151                .chain(consensus_stream_ended)
152                .chain(consensus_smears)
153                .collect()
154        };
155
156        let locks_to_delete: Vec<_> = mutable_inputs
157            .into_iter()
158            .filter_map(|(id, ((version, digest), owner))| {
159                owner.is_address_owned().then_some((id, version, digest))
160            })
161            .chain(received_objects)
162            .collect();
163
164        let new_locks_to_init: Vec<_> = written
165            .values()
166            .filter_map(|new_object| {
167                if new_object.is_address_owned() {
168                    Some(new_object.compute_object_reference())
169                } else {
170                    None
171                }
172            })
173            .collect();
174
175        let deleted = effects
176            .deleted()
177            .into_iter()
178            .chain(effects.unwrapped_then_deleted())
179            .map(ObjectKey::from)
180            .collect();
181
182        let wrapped = effects.wrapped().into_iter().map(ObjectKey::from).collect();
183
184        TransactionOutputs {
185            transaction: Arc::new(transaction),
186            effects,
187            events,
188            unchanged_loaded_runtime_objects,
189            accumulator_events: Mutex::new(Some(accumulator_events)),
190            markers,
191            wrapped,
192            deleted,
193            locks_to_delete,
194            new_locks_to_init,
195            written,
196        }
197    }
198
199    pub fn take_accumulator_events(&self) -> Vec<AccumulatorEvent> {
200        self.accumulator_events
201            .lock()
202            .take()
203            .expect("take_accumulator_events called twice")
204    }

206    #[cfg(test)]

208        Self {
209            transaction: Arc::new(transaction),
210            effects,
211            events: TransactionEvents { data: vec![] },
212            unchanged_loaded_runtime_objects: vec![],
213            accumulator_events: Mutex::new(Some(vec![])),
214            markers: vec![],
215            wrapped: vec![],
216            deleted: vec![],
217            locks_to_delete: vec![],
218            new_locks_to_init: vec![],
219            written: WrittenObjects::new(),
220        }
221    }

224pub fn unchanged_loaded_runtime_objects(
225    _transaction: &TransactionData,
226    effects: &TransactionEffects,
227    loaded_runtime_objects: &ObjectSet,
228) -> Vec<ObjectKey> {
229    let mut unchanged_loaded_runtime_objects: BTreeMap<_, _> = loaded_runtime_objects
230        .iter()
231        // Don't include loaded packages (which are used for doing UID tracking inside the VM)
232        .filter(|o| !o.is_package())
233        .map(|o| (o.id(), o.version()))
234        .collect();
235
236    // Remove any object that is referenced in the changed objects effects set since it would be
237    // redundant to include it again.
238    for change in effects.object_changes() {
239        unchanged_loaded_runtime_objects.remove(&change.id);
240    }
241
242    unchanged_loaded_runtime_objects
243        .into_iter()
244        .map(|(id, v)| ObjectKey(id, v))
245        .collect()
246}
