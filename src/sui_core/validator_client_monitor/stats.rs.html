15// TODO: A few optimization to consider:
16// 1. There may be times when the entire network is unstable, and in that case
17//    we may not want to punish validators when they have errors.
18// 2. Some reports are more critical than others. For example, a health check
19//    report is more critical than a submit report in terms of failures status.
20
21/// Maximum adjusted latency from completely unreachable (reliability = 0.0) or very slow validators.
22const MAX_LATENCY: Duration = Duration::from_secs(10);
23
24/// Complete client-observed statistics for validator interactions.
25///
26/// This struct maintains client-side metrics for all validators in the network,
27/// including latency measurements, reliability measurements, and failure tracking
28/// as observed from the client's perspective. It uses exponential moving averages (EMA)
29/// to smooth out transient spikes while still responding to sustained changes.
30#[derive(Debug, Clone)]
31pub struct ClientObservedStats {
32    /// Per-validator statistics mapping authority names to their client-observed metrics
33    pub validator_stats: HashMap<AuthorityName, ValidatorClientStats>,
34    /// Configuration parameters for scoring and exclusion policies
35    pub config: ValidatorClientMonitorConfig,
36}
37
38/// Client-observed stats for a single validator.
39///
40/// Tracks reliability, latency, and failure patterns for a specific validator
41/// as observed from the client's perspective. Uses exponential moving averages
42/// to smooth measurements while maintaining responsiveness to changes.
43#[derive(Debug, Clone)]
44pub struct ValidatorClientStats {
45    /// Moving window of success rate (0.0 to 1.0)
46    pub reliability: MovingWindow<f64>,
47    /// Moving window of latencies for each operation type (Submit, Effects, HealthCheck)
48    pub average_latencies: BTreeMap<OperationType, MovingWindow<Duration>>,
49    /// Size of the moving window for latency measurements
50    pub latency_moving_window_size: usize,
51}
52
53impl ValidatorClientStats {
54    pub fn new(
55        init_reliability: f64,
56        reliability_moving_window_size: usize,
57        latency_moving_window_size: usize,
58    ) -> Self {
59        Self {
60            reliability: MovingWindow::new(init_reliability, reliability_moving_window_size),
61            average_latencies: BTreeMap::new(),
62            latency_moving_window_size,
63        }
64    }
65
66    pub fn update_average_latency(&mut self, operation: OperationType, new_latency: Duration) {
67        match self.average_latencies.entry(operation) {
68            Entry::Occupied(mut entry) => {
69                entry.get_mut().add_value(new_latency);
70            }
71            Entry::Vacant(entry) => {
72                entry.insert(MovingWindow::new(
73                    new_latency,
74                    self.latency_moving_window_size,
75                ));
76            }
77        }
78    }
79}
80
81impl ClientObservedStats {
82    pub fn new(config: ValidatorClientMonitorConfig) -> Self {
83        Self {
84            validator_stats: HashMap::new(),
85            config,
86        }
87    }
88
89    /// Record client-observed interaction result with a validator.
90    ///
91    /// Updates reliability scores and latency measurements.
92    pub fn record_interaction_result(&mut self, feedback: OperationFeedback) {
93        let validator_stats = self
94            .validator_stats
95            .entry(feedback.authority_name)
96            .or_insert_with(|| {
97                ValidatorClientStats::new(
98                    1.0,
99                    self.config.reliability_moving_window_size,
100                    self.config.latency_moving_window_size,
101                )
102            });
103
104        match feedback.result {
105            Ok(latency) => {
106                validator_stats.reliability.add_value(1.0);
107                validator_stats.update_average_latency(feedback.operation, latency);
108            }
109            Err(()) => {
110                validator_stats.reliability.add_value(0.0);
111            }
112        }
113    }
114
115    /// Get validator latencies for all validators in the committee for the provided tx type.
116    ///
117    /// Returns a map of all tracked validators to their stats. For a validator
118    pub fn get_all_validator_stats(
119        &self,
120        committee: &Committee,
121        tx_type: TxType,
122    ) -> HashMap<AuthorityName, Duration> {
123        committee
124            .names()
125            .map(|validator| {
126                let latency = self.calculate_client_latency(validator, tx_type);
127                (*validator, latency)
128            })
129            .collect()
130    }
131
132    /// Calculate adjusted latency for a single validator for the provided tx type.
133    ///
134    /// Returns the average latency for relevant operations (Consensus and FastPath only)
135    /// with reliability penalty applied. Lower values are better.
136    ///
137    /// Only considers:
138    /// - Consensus operations (for SharedObject transactions)
139    /// - FastPath operations (for SingleWriter transactions)
140    ///
141    /// Returns latency in seconds, with reliability penalty applied as a multiplier.
142    fn calculate_client_latency(&self, validator: &AuthorityName, tx_type: TxType) -> Duration {
143        let Some(stats) = self.validator_stats.get(validator) else {
144            return MAX_LATENCY;
145        };
146
147        let operation = match tx_type {
148            TxType::SharedObject => OperationType::Consensus,
149            TxType::SingleWriter => OperationType::FastPath,
150        };
151        let Some(latency) = stats.average_latencies.get(&operation) else {
152            // For the target validator and operation type, no latency data has been recorded yet.
153            return MAX_LATENCY;
154        };
155
156        // Get the latency for the relevant operation
157        let base_latency = latency.get();
158        let reliability = stats.reliability.get();
159        let reliability_weight = self.config.reliability_weight;
160        let penalty = MAX_LATENCY.mul_f64((1.0 - reliability) * reliability_weight);
161        (base_latency + penalty).min(MAX_LATENCY)
162    }
163
164    /// Retain only the specified validators, removing any others.
165    ///
166    /// Called periodically during health checks to clean up statistics for validators
167    /// that are no longer in the active set. This prevents memory leaks and
168    /// ensures scores are only calculated for current validators.
169    pub fn retain_validators(&mut self, current_validators: &[AuthorityName]) {
170        let cur_len = self.validator_stats.len();
171        let validator_set: HashSet<_> = current_validators.iter().collect();
172        self.validator_stats
173            .retain(|validator, _| validator_set.contains(validator));
174        let removed_count = cur_len - self.validator_stats.len();
175        if removed_count > 0 {
176            debug!("Removed {} stale validator data", removed_count);
177        }
178    }
179}
