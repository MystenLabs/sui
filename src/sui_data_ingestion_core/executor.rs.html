25/// Environment variable to override the default maximum number of checkpoints that can be processed concurrently.
26const MAX_CHECKPOINTS_IN_PROGRESS_VAR_NAME: &str = "MAX_CHECKPOINTS_IN_PROGRESS";
27
28/// Default maximum number of checkpoints in progress.
29const DEFAULT_MAX_CHECKPOINTS_IN_PROGRESS: usize = 10_000;
30
31/// Maximum number of checkpoints that can be processed concurrently.
32///
33/// This value can be overridden by setting the `MAX_CHECKPOINTS_IN_PROGRESS` environment variable
34/// before starting the process. If the environment variable is unset, the default value of
35/// `DEFAULT_MAX_CHECKPOINTS_IN_PROGRESS` will be used.
36///
37/// This is read once at startup and cached. Changing the environment variable at runtime will not
38/// have any effect.
39pub static MAX_CHECKPOINTS_IN_PROGRESS: Lazy<usize> = Lazy::new(|| {
40    let max_checkpoints_opt = std::env::var(MAX_CHECKPOINTS_IN_PROGRESS_VAR_NAME)
41        .ok()
42        .and_then(|s| s.parse().ok());
43    if let Some(max_checkpoints) = max_checkpoints_opt {
44        info!(
45            "Using custom value for '{}' max checkpoints in progress: {}",
46            MAX_CHECKPOINTS_IN_PROGRESS_VAR_NAME, max_checkpoints
47        );
48        max_checkpoints
49    } else {
50        info!(
51            "Using default value for '{}' -- max checkpoints in progress: {}",
52            MAX_CHECKPOINTS_IN_PROGRESS_VAR_NAME, DEFAULT_MAX_CHECKPOINTS_IN_PROGRESS
53        );
54        DEFAULT_MAX_CHECKPOINTS_IN_PROGRESS
55    }
56});

58pub struct IndexerExecutor<P> {
59    pools: Vec<Pin<Box<dyn Future<Output = ()> + Send>>>,
60    pool_senders: Vec<mpsc::Sender<Arc<CheckpointData>>>,
61    progress_store: ProgressStoreWrapper<P>,
62    pool_progress_sender: mpsc::Sender<(String, CheckpointSequenceNumber)>,
63    pool_progress_receiver: mpsc::Receiver<(String, CheckpointSequenceNumber)>,
64    metrics: DataIngestionMetrics,
65}
66
67impl<P: ProgressStore> IndexerExecutor<P> {
68    pub fn new(progress_store: P, number_of_jobs: usize, metrics: DataIngestionMetrics) -> Self {
69        let (pool_progress_sender, pool_progress_receiver) =
70            mpsc::channel(number_of_jobs * *MAX_CHECKPOINTS_IN_PROGRESS);
71        Self {
72            pools: vec![],
73            pool_senders: vec![],
74            progress_store: ProgressStoreWrapper::new(progress_store),
75            pool_progress_sender,
76            pool_progress_receiver,
77            metrics,
78        }
79    }
80
81    /// Registers new worker pool in executor
82    pub async fn register<W: Worker + 'static>(&mut self, pool: WorkerPool<W>) -> Result<()> {
83        let checkpoint_number = self.progress_store.load(pool.task_name.clone()).await?;
84        let (sender, receiver) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
85        self.pools.push(Box::pin(pool.run(
86            checkpoint_number,
87            receiver,
88            self.pool_progress_sender.clone(),
89        )));
90        self.pool_senders.push(sender);
91        Ok(())
92    }
93
94    /// Main executor loop
95    pub async fn run(
96        mut self,
97        path: PathBuf,
98        remote_store_url: Option<String>,
99        remote_store_options: Vec<(String, String)>,
100        reader_options: ReaderOptions,
101        mut exit_receiver: oneshot::Receiver<()>,
102    ) -> Result<ExecutorProgress> {
103        let mut reader_checkpoint_number = self.progress_store.min_watermark()?;
104        let upper_limit = reader_options.upper_limit;
105        let (checkpoint_reader, mut checkpoint_recv, gc_sender, _exit_sender) =
106            CheckpointReader::initialize(
107                path,
108                reader_checkpoint_number,
109                remote_store_url,
110                remote_store_options,
111                reader_options,
112            );
113        spawn_monitored_task!(checkpoint_reader.run());
114
115        for pool in std::mem::take(&mut self.pools) {
116            spawn_monitored_task!(pool);
117        }
118        loop {
119            tokio::select! {
120                _ = &mut exit_receiver => break,
121                Some((task_name, sequence_number)) = self.pool_progress_receiver.recv() => {
122                    self.progress_store.save(task_name.clone(), sequence_number).await?;
123                    let seq_number = self.progress_store.min_watermark()?;
124                    if seq_number > reader_checkpoint_number {
125                        gc_sender.send(seq_number).await?;
126                        reader_checkpoint_number = seq_number;
127                    }
128                    self.metrics.data_ingestion_checkpoint.with_label_values(&[&task_name]).set(sequence_number as i64);
129                    if let Some(limit) = upper_limit
130                        && sequence_number > limit && self.pool_senders.len() == 1 {
131                            break;
132                        }
133                }
134                Some(checkpoint) = checkpoint_recv.recv() => {
135                    for sender in &self.pool_senders {
136                        sender.send(checkpoint.clone()).await?;
137                    }
138                }
139            }
140        }
141        Ok(self.progress_store.stats())
142    }
143
144    pub async fn update_watermark(
145        &mut self,
146        task_name: String,
147        watermark: CheckpointSequenceNumber,
148    ) -> Result<()> {
149        self.progress_store.save(task_name, watermark).await
150    }
151}

153pub async fn setup_single_workflow<W: Worker + 'static>(
154    worker: W,
155    remote_store_url: String,
156    initial_checkpoint_number: CheckpointSequenceNumber,
157    concurrency: usize,
158    reader_options: Option<ReaderOptions>,
159) -> Result<(
160    impl Future<Output = Result<ExecutorProgress>>,
161    oneshot::Sender<()>,
162)> {
163    setup_single_workflow_with_options(
164        worker,
165        remote_store_url,
166        vec![],
167        initial_checkpoint_number,
168        concurrency,
169        reader_options,
170    )
171    .await
172}
173
174pub async fn setup_single_workflow_with_options<W: Worker + 'static>(
175    worker: W,
176    remote_store_url: String,
177    remote_store_options: Vec<(String, String)>,
178    initial_checkpoint_number: CheckpointSequenceNumber,
179    concurrency: usize,
180    reader_options: Option<ReaderOptions>,
181) -> Result<(
182    impl Future<Output = Result<ExecutorProgress>>,
183    oneshot::Sender<()>,
184)> {
185    let (exit_sender, exit_receiver) = oneshot::channel();
186    let metrics = DataIngestionMetrics::new(&Registry::new());
187    let progress_store = ShimProgressStore(initial_checkpoint_number);
188    let mut executor = IndexerExecutor::new(progress_store, 1, metrics);
189    let worker_pool = WorkerPool::new(worker, "workflow".to_string(), concurrency);
190    executor.register(worker_pool).await?;
191    Ok((
192        executor.run(
193            tempfile::tempdir()?.keep(),
194            Some(remote_store_url),
195            remote_store_options,
196            reader_options.unwrap_or_default(),
197            exit_receiver,
198        ),
199        exit_sender,
200    ))
