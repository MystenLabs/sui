30pub struct CheckpointReader {
31    /// Used to read from a local directory when running with a colocated FN.
32    /// When fetching from a remote store, a temp dir can be passed in and it will be an no-op
33    path: PathBuf,
34    remote_store_url: Option<String>,
35    remote_store_options: Vec<(String, String)>,
36    current_checkpoint_number: CheckpointSequenceNumber,
37    last_pruned_watermark: CheckpointSequenceNumber,
38    checkpoint_sender: mpsc::Sender<Arc<CheckpointData>>,
39    processed_receiver: mpsc::Receiver<CheckpointSequenceNumber>,
40    #[allow(clippy::type_complexity)]
41    remote_fetcher_receiver: Option<mpsc::Receiver<Result<(Arc<CheckpointData>, usize)>>>,
42    exit_receiver: oneshot::Receiver<()>,
43    options: ReaderOptions,
44    data_limiter: DataLimiter,
45}

47#[derive(Clone)]
48pub struct ReaderOptions {
49    pub tick_internal_ms: u64,
50    pub timeout_secs: u64,
51    /// number of maximum concurrent requests to the remote store. Increase it for backfills
52    pub batch_size: usize,
53    pub data_limit: usize,
54    pub upper_limit: Option<CheckpointSequenceNumber>,
55    /// Whether to delete processed checkpoint files from the local directory.
56    pub gc_checkpoint_files: bool,
57}

63            timeout_secs: 5,
64            batch_size: 10,
65            data_limit: 0,
66            upper_limit: None,

72enum RemoteStore {

74    Rest(sui_rpc_api::Client),
75    Hybrid(Box<dyn ObjectStore>, sui_rpc_api::Client),

78impl CheckpointReader {
79    /// Represents a single iteration of the reader.
80    /// Reads files in a local directory, validates them, and forwards `CheckpointData` to the executor.
81    async fn read_local_files(&self) -> Result<Vec<Arc<CheckpointData>>> {
82        let mut checkpoints = vec![];
83        for offset in 0..*MAX_CHECKPOINTS_IN_PROGRESS {
84            let sequence_number = self.current_checkpoint_number + offset as u64;
85            if self.exceeds_capacity(sequence_number) {
86                break;
87            }
88            match fs::read(self.path.join(format!("{}.chk", sequence_number))) {
89                Ok(bytes) => checkpoints.push(Blob::from_bytes::<Arc<CheckpointData>>(&bytes)?),
90                Err(err) => match err.kind() {
91                    std::io::ErrorKind::NotFound => break,
92                    _ => Err(err)?,
93                },
94            }
95        }
96        Ok(checkpoints)
97    }
98
99    fn exceeds_capacity(&self, checkpoint_number: CheckpointSequenceNumber) -> bool {
100        ((*MAX_CHECKPOINTS_IN_PROGRESS as u64 + self.last_pruned_watermark) <= checkpoint_number)
101            || self.data_limiter.exceeds()
102    }
103
104    pub async fn fetch_from_object_store(
105        store: &dyn ObjectStore,
106        checkpoint_number: CheckpointSequenceNumber,
107    ) -> Result<(Arc<CheckpointData>, usize)> {
108        let path = Path::from(format!("{}.chk", checkpoint_number));
109        let response = store.get(&path).await?;
110        let bytes = response.bytes().await?;
111        Ok((
112            Blob::from_bytes::<Arc<CheckpointData>>(&bytes)?,
113            bytes.len(),
114        ))
115    }
116
117    async fn fetch_from_full_node(
118        client: &Client,
119        checkpoint_number: CheckpointSequenceNumber,
120    ) -> Result<(Arc<CheckpointData>, usize)> {
121        let checkpoint = client
122            .clone()
123            .get_full_checkpoint(checkpoint_number)
124            .await?;
125        let size = bcs::serialized_size(&checkpoint)?;
126        Ok((Arc::new(checkpoint), size))
127    }
128
129    async fn remote_fetch_checkpoint_internal(
130        store: &RemoteStore,
131        checkpoint_number: CheckpointSequenceNumber,
132    ) -> Result<(Arc<CheckpointData>, usize)> {
133        match store {
134            RemoteStore::ObjectStore(store) => {
135                Self::fetch_from_object_store(store, checkpoint_number).await
136            }
137            RemoteStore::Rest(client) => {
138                Self::fetch_from_full_node(client, checkpoint_number).await
139            }
140            RemoteStore::Hybrid(store, client) => {
141                match Self::fetch_from_full_node(client, checkpoint_number).await {
142                    Ok(result) => Ok(result),
143                    Err(_) => Self::fetch_from_object_store(store, checkpoint_number).await,
144                }
145            }
146        }
147    }
148
149    async fn remote_fetch_checkpoint(
150        store: &RemoteStore,
151        checkpoint_number: CheckpointSequenceNumber,
152    ) -> Result<(Arc<CheckpointData>, usize)> {
153        let mut backoff = backoff::ExponentialBackoff::default();
154        let max_elapsed_time = Duration::from_secs(60);
155        backoff.max_elapsed_time = Some(max_elapsed_time);
156        backoff.initial_interval = Duration::from_millis(100);
157        backoff.current_interval = backoff.initial_interval;
158        backoff.multiplier = 1.0;
159        loop {
160            match tokio::time::timeout(
161                max_elapsed_time,
162                Self::remote_fetch_checkpoint_internal(store, checkpoint_number),
163            )
164            .await
165            {
166                Ok(Ok(data)) => return Ok(data),
167                Ok(Err(err)) => match backoff.next_backoff() {
168                    Some(duration) => {
169                        if !err.to_string().contains("404") {
170                            debug!(
171                                "remote reader retry in {} ms. Error is {:?}",
172                                duration.as_millis(),
173                                err
174                            );
175                        }
176                        tokio::time::sleep(duration).await
177                    }
178                    None => return Err(err),
179                },
180                Err(err) => match backoff.next_backoff() {
181                    Some(duration) => tokio::time::sleep(duration).await,
182                    None => return Err(err.into()),
183                },
184            }
185        }
186    }
187
188    fn start_remote_fetcher(&mut self) -> mpsc::Receiver<Result<(Arc<CheckpointData>, usize)>> {
189        let batch_size = self.options.batch_size;
190        let start_checkpoint = self.current_checkpoint_number;
191        let (sender, receiver) = mpsc::channel(batch_size);
192        let url = self
193            .remote_store_url
194            .clone()
195            .expect("remote store url must be set");
196        let store = if let Some((fn_url, remote_url)) = url.split_once('|') {
197            let object_store = create_remote_store_client(
198                remote_url.to_string(),
199                self.remote_store_options.clone(),
200                self.options.timeout_secs,
201            )
202            .expect("failed to create remote store client");
203            RemoteStore::Hybrid(object_store, sui_rpc_api::Client::new(fn_url).unwrap())
204        } else if url.ends_with("/rest") {
205            RemoteStore::Rest(sui_rpc_api::Client::new(url).unwrap())
206        } else {
207            let object_store = create_remote_store_client(
208                url,
209                self.remote_store_options.clone(),
210                self.options.timeout_secs,
211            )
212            .expect("failed to create remote store client");
213            RemoteStore::ObjectStore(object_store)
214        };
215
216        spawn_monitored_task!(async move {
217            let mut checkpoint_stream = (start_checkpoint..u64::MAX)
218                .map(|checkpoint_number| Self::remote_fetch_checkpoint(&store, checkpoint_number))
219                .pipe(futures::stream::iter)
220                .buffered(batch_size);
221
222            while let Some(checkpoint) = checkpoint_stream.next().await {
223                if sender.send(checkpoint).await.is_err() {
224                    info!("remote reader dropped");
225                    break;
226                }
227            }
228        });
229        receiver
230    }
231
232    fn remote_fetch(&mut self) -> Vec<Arc<CheckpointData>> {
233        let mut checkpoints = vec![];
234        if self.remote_fetcher_receiver.is_none() {
235            self.remote_fetcher_receiver = Some(self.start_remote_fetcher());
236        }
237        while !self.exceeds_capacity(self.current_checkpoint_number + checkpoints.len() as u64) {
238            match self.remote_fetcher_receiver.as_mut().unwrap().try_recv() {
239                Ok(Ok((checkpoint, size))) => {
240                    self.data_limiter.add(&checkpoint, size);
241                    checkpoints.push(checkpoint);
242                }
243                Ok(Err(err)) => {
244                    error!("remote reader transient error {:?}", err);
245                    self.remote_fetcher_receiver = None;
246                    break;
247                }
248                Err(TryRecvError::Disconnected) => {
249                    error!("remote reader channel disconnect error");
250                    self.remote_fetcher_receiver = None;
251                    break;
252                }
253                Err(TryRecvError::Empty) => break,
254            }
255        }
256        checkpoints
257    }

259    async fn sync(&mut self) -> Result<()> {
260        let backoff = backoff::ExponentialBackoff::default();
261        let mut checkpoints = backoff::future::retry(backoff, || async {
262            self.read_local_files().await.map_err(|err| {
263                info!("transient local read error {:?}", err);
264                backoff::Error::transient(err)
265            })
266        })
267        .await?;
268
269        let mut read_source: &str = "local";
270        if self.remote_store_url.is_some()
271            && (checkpoints.is_empty()
272                || checkpoints[0].checkpoint_summary.sequence_number
273                    > self.current_checkpoint_number)
274        {
275            checkpoints = self.remote_fetch();
276            read_source = "remote";
277        } else {
278            // cancel remote fetcher execution because local reader has made progress
279            self.remote_fetcher_receiver = None;
280        }
281
282        info!(
283            "Read from {}. Current checkpoint number: {}, pruning watermark: {}, new updates: {:?}",
284            read_source,
285            self.current_checkpoint_number,
286            self.last_pruned_watermark,
287            checkpoints.len(),
288        );
289        for checkpoint in checkpoints {
290            if read_source == "local"
291                && checkpoint.checkpoint_summary.sequence_number > self.current_checkpoint_number
292            {
293                break;
294            }
295            assert_eq!(
296                checkpoint.checkpoint_summary.sequence_number,
297                self.current_checkpoint_number
298            );
299            self.checkpoint_sender.send(checkpoint).await?;
300            self.current_checkpoint_number += 1;
301        }
302        Ok(())
303    }
304
305    /// Cleans the local directory by removing all processed checkpoint files.
306    fn gc_processed_files(&mut self, watermark: CheckpointSequenceNumber) -> Result<()> {
307        self.data_limiter.gc(watermark);
308        self.last_pruned_watermark = watermark;
309        if !self.options.gc_checkpoint_files {
310            return Ok(());
311        }
312        info!("cleaning processed files, watermark is {}", watermark);
313        for entry in fs::read_dir(self.path.clone())? {
314            let entry = entry?;
315            let filename = entry.file_name();
316            if let Some(sequence_number) = Self::checkpoint_number_from_file_path(&filename)
317                && sequence_number < watermark
318            {
319                fs::remove_file(entry.path())?;
320            }
321        }
322        Ok(())
323    }
324
325    fn checkpoint_number_from_file_path(file_name: &OsString) -> Option<CheckpointSequenceNumber> {
326        file_name
327            .to_str()
328            .and_then(|s| s.rfind('.').map(|pos| &s[..pos]))
329            .and_then(|s| s.parse().ok())
330    }

332    pub fn initialize(
333        path: PathBuf,
334        starting_checkpoint_number: CheckpointSequenceNumber,
335        remote_store_url: Option<String>,
336        remote_store_options: Vec<(String, String)>,
337        options: ReaderOptions,
338    ) -> (
339        Self,
340        mpsc::Receiver<Arc<CheckpointData>>,
341        mpsc::Sender<CheckpointSequenceNumber>,
342        oneshot::Sender<()>,
343    ) {
344        let (checkpoint_sender, checkpoint_recv) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
345        let (processed_sender, processed_receiver) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
346        let (exit_sender, exit_receiver) = oneshot::channel();
347        let reader = Self {
348            path,
349            remote_store_url,
350            remote_store_options,
351            current_checkpoint_number: starting_checkpoint_number,
352            last_pruned_watermark: starting_checkpoint_number,
353            checkpoint_sender,
354            processed_receiver,
355            remote_fetcher_receiver: None,
356            exit_receiver,
357            data_limiter: DataLimiter::new(options.data_limit),
358            options,
359        };
360        (reader, checkpoint_recv, processed_sender, exit_sender)
361    }

363    #[cfg(not(target_os = "macos"))]
364    fn init_watcher(
365        inotify_sender: mpsc::Sender<()>,
366        path: &std::path::Path,
367    ) -> RecommendedWatcher {
368        use notify::Watcher;
369        let mut watcher = notify::recommended_watcher(move |res| {
370            if let Err(err) = res {
371                eprintln!("watch error: {:?}", err);
372            }
373            inotify_sender
374                .blocking_send(())
375                .expect("Failed to send inotify update");
376        })
377        .expect("Failed to init inotify");
378        watcher
379            .watch(path, RecursiveMode::NonRecursive)
380            .expect("Inotify watcher failed");
381        watcher
382    }
383
384    pub async fn run(mut self) -> Result<()> {
385        let (_inotify_sender, mut inotify_recv) = mpsc::channel::<()>(1);
386        std::fs::create_dir_all(self.path.clone()).expect("failed to create a directory");
387
388        #[cfg(not(target_os = "macos"))]
389        let _watcher = Self::init_watcher(_inotify_sender, &self.path);
390
391        self.gc_processed_files(self.last_pruned_watermark)
392            .expect("Failed to clean the directory");
393
394        loop {
395            tokio::select! {
396                _ = &mut self.exit_receiver => break,
397                Some(gc_checkpoint_number) = self.processed_receiver.recv() => {
398                    self.gc_processed_files(gc_checkpoint_number).expect("Failed to clean the directory");
399                }
400                Ok(Some(_)) | Err(_) = timeout(Duration::from_millis(self.options.tick_internal_ms), inotify_recv.recv())  => {
401                    self.sync().await.expect("Failed to read checkpoint files");
402                }
403            }
404        }
405        Ok(())
406    }

409pub struct DataLimiter {
410    limit: usize,
411    queue: BTreeMap<CheckpointSequenceNumber, usize>,
412    in_progress: usize,
413}
414
415impl DataLimiter {
416    fn new(limit: usize) -> Self {
417        Self {
418            limit,
419            queue: BTreeMap::new(),
420            in_progress: 0,
421        }
422    }
423
424    fn exceeds(&self) -> bool {
425        self.limit > 0 && self.in_progress >= self.limit
426    }
427
428    fn add(&mut self, checkpoint: &CheckpointData, size: usize) {
429        if self.limit == 0 {
430            return;
431        }
432        self.in_progress += size;
433        self.queue
434            .insert(checkpoint.checkpoint_summary.sequence_number, size);
435    }
436
437    fn gc(&mut self, watermark: CheckpointSequenceNumber) {
438        if self.limit == 0 {
439            return;
440        }
441        self.queue = self.queue.split_off(&watermark);
442        self.in_progress = self.queue.values().sum();
443    }
