33pub struct CheckpointReader {
34    /// Used to read from a local directory when running with a colocated FN.
35    /// When fetching from a remote store, a temp dir can be passed in and it will be an no-op
36    path: PathBuf,
37    remote_store_url: Option<String>,
38    remote_store_options: Vec<(String, String)>,
39    current_checkpoint_number: CheckpointSequenceNumber,
40    last_pruned_watermark: CheckpointSequenceNumber,
41    checkpoint_sender: mpsc::Sender<Arc<CheckpointData>>,
42    processed_receiver: mpsc::Receiver<CheckpointSequenceNumber>,
43    #[allow(clippy::type_complexity)]
44    remote_fetcher_receiver: Option<mpsc::Receiver<Result<(Arc<CheckpointData>, usize)>>>,
45    exit_receiver: oneshot::Receiver<()>,
46    options: ReaderOptions,
47    data_limiter: DataLimiter,
48}

55    pub batch_size: usize,

81impl CheckpointReader {
82    /// Represents a single iteration of the reader.
83    /// Reads files in a local directory, validates them, and forwards `CheckpointData` to the executor.
84    async fn read_local_files(&self) -> Result<Vec<Arc<CheckpointData>>> {
85        let mut checkpoints = vec![];
86        for offset in 0..*MAX_CHECKPOINTS_IN_PROGRESS {
87            let sequence_number = self.current_checkpoint_number + offset as u64;
88            if self.exceeds_capacity(sequence_number) {
89                break;
90            }
91            match fs::read(self.path.join(format!("{}.binpb.zst", sequence_number))) {
92                Ok(bytes) => {
93                    let decompressed = zstd::decode_all(&bytes[..])?;
94                    let proto_checkpoint = ProtoCheckpoint::decode(&decompressed[..])?;
95                    let checkpoint: Checkpoint = (&proto_checkpoint).try_into()?;
96                    let checkpoint_data: CheckpointData = checkpoint.into();
97                    checkpoints.push(Arc::new(checkpoint_data));
98                }
99                Err(err) => match err.kind() {
100                    std::io::ErrorKind::NotFound => break,
101                    _ => Err(err)?,
102                },
103            }
104        }
105        Ok(checkpoints)
106    }
107
108    fn exceeds_capacity(&self, checkpoint_number: CheckpointSequenceNumber) -> bool {
109        ((*MAX_CHECKPOINTS_IN_PROGRESS as u64 + self.last_pruned_watermark) <= checkpoint_number)
110            || self.data_limiter.exceeds()
111    }

113    pub async fn fetch_from_object_store(
114        store: &dyn ObjectStore,
115        checkpoint_number: CheckpointSequenceNumber,
116    ) -> Result<(Arc<CheckpointData>, usize)> {
117        let path = Path::from(format!("{}.binpb.zst", checkpoint_number));
118        let response = store.get(&path).await?;
119        let bytes = response.bytes().await?;
120        let decompressed = zstd::decode_all(&bytes[..])?;
121        let proto_checkpoint = ProtoCheckpoint::decode(&decompressed[..])?;
122        let checkpoint: Checkpoint = (&proto_checkpoint).try_into()?;
123        let checkpoint_data: CheckpointData = checkpoint.into();
124        Ok((Arc::new(checkpoint_data), bytes.len()))
125    }
126
127    async fn fetch_from_full_node(
128        client: &Client,
129        checkpoint_number: CheckpointSequenceNumber,
130    ) -> Result<(Arc<CheckpointData>, usize)> {
131        let checkpoint = client
132            .clone()
133            .get_full_checkpoint(checkpoint_number)
134            .await?
135            .into();
136        let size = bcs::serialized_size(&checkpoint)?;
137        Ok((Arc::new(checkpoint), size))
138    }
139
140    async fn remote_fetch_checkpoint_internal(
141        store: &RemoteStore,
142        checkpoint_number: CheckpointSequenceNumber,
143    ) -> Result<(Arc<CheckpointData>, usize)> {
144        match store {
145            RemoteStore::ObjectStore(store) => {
146                Self::fetch_from_object_store(store, checkpoint_number).await
147            }
148            RemoteStore::Rest(client) => {
149                Self::fetch_from_full_node(client, checkpoint_number).await
150            }
151            RemoteStore::Hybrid(store, client) => {
152                match Self::fetch_from_full_node(client, checkpoint_number).await {
153                    Ok(result) => Ok(result),
154                    Err(_) => Self::fetch_from_object_store(store, checkpoint_number).await,
155                }
156            }
157        }
158    }
159
160    async fn remote_fetch_checkpoint(
161        store: &RemoteStore,
162        checkpoint_number: CheckpointSequenceNumber,
163    ) -> Result<(Arc<CheckpointData>, usize)> {
164        let mut backoff = backoff::ExponentialBackoff::default();
165        let max_elapsed_time = Duration::from_secs(60);
166        backoff.max_elapsed_time = Some(max_elapsed_time);
167        backoff.initial_interval = Duration::from_millis(100);
168        backoff.current_interval = backoff.initial_interval;
169        backoff.multiplier = 1.0;
170        loop {
171            match tokio::time::timeout(
172                max_elapsed_time,
173                Self::remote_fetch_checkpoint_internal(store, checkpoint_number),
174            )
175            .await
176            {
177                Ok(Ok(data)) => return Ok(data),
178                Ok(Err(err)) => match backoff.next_backoff() {
179                    Some(duration) => {
180                        if !err.to_string().contains("404") {
181                            debug!(
182                                "remote reader retry in {} ms. Error is {:?}",
183                                duration.as_millis(),
184                                err
185                            );
186                        }
187                        tokio::time::sleep(duration).await
188                    }
189                    None => return Err(err),
190                },
191                Err(err) => match backoff.next_backoff() {
192                    Some(duration) => tokio::time::sleep(duration).await,
193                    None => return Err(err.into()),
194                },
195            }
196        }
197    }

199    fn start_remote_fetcher(&mut self) -> mpsc::Receiver<Result<(Arc<CheckpointData>, usize)>> {
200        let batch_size = self.options.batch_size;
201        let start_checkpoint = self.current_checkpoint_number;
202        let (sender, receiver) = mpsc::channel(batch_size);
203        let url = self
204            .remote_store_url
205            .clone()
206            .expect("remote store url must be set");
207        let store = if let Some((fn_url, remote_url)) = url.split_once('|') {
208            let object_store = create_remote_store_client(
209                remote_url.to_string(),
210                self.remote_store_options.clone(),
211                self.options.timeout_secs,
212            )
213            .expect("failed to create remote store client");
214            RemoteStore::Hybrid(object_store, sui_rpc_api::Client::new(fn_url).unwrap())
215        } else if url.ends_with("/rest") {
216            RemoteStore::Rest(sui_rpc_api::Client::new(url).unwrap())
217        } else {
218            let object_store = create_remote_store_client(
219                url,
220                self.remote_store_options.clone(),
221                self.options.timeout_secs,
222            )
223            .expect("failed to create remote store client");
224            RemoteStore::ObjectStore(object_store)
225        };
226
227        spawn_monitored_task!(async move {
228            let mut checkpoint_stream = (start_checkpoint..u64::MAX)
229                .map(|checkpoint_number| Self::remote_fetch_checkpoint(&store, checkpoint_number))
230                .pipe(futures::stream::iter)
231                .buffered(batch_size);
232
233            while let Some(checkpoint) = checkpoint_stream.next().await {
234                if sender.send(checkpoint).await.is_err() {
235                    info!("remote reader dropped");
236                    break;
237                }
238            }
239        });
240        receiver
241    }
242
243    fn remote_fetch(&mut self) -> Vec<Arc<CheckpointData>> {
244        let mut checkpoints = vec![];
245        if self.remote_fetcher_receiver.is_none() {
246            self.remote_fetcher_receiver = Some(self.start_remote_fetcher());
247        }
248        while !self.exceeds_capacity(self.current_checkpoint_number + checkpoints.len() as u64) {
249            match self.remote_fetcher_receiver.as_mut().unwrap().try_recv() {
250                Ok(Ok((checkpoint, size))) => {
251                    self.data_limiter.add(&checkpoint, size);
252                    checkpoints.push(checkpoint);
253                }
254                Ok(Err(err)) => {
255                    error!("remote reader transient error {:?}", err);
256                    self.remote_fetcher_receiver = None;
257                    break;
258                }
259                Err(TryRecvError::Disconnected) => {
260                    error!("remote reader channel disconnect error");
261                    self.remote_fetcher_receiver = None;
262                    break;
263                }
264                Err(TryRecvError::Empty) => break,
265            }
266        }
267        checkpoints
268    }

270    async fn sync(&mut self) -> Result<()> {
271        let backoff = backoff::ExponentialBackoff::default();
272        let mut checkpoints = backoff::future::retry(backoff, || async {
273            self.read_local_files().await.map_err(|err| {
274                info!("transient local read error {:?}", err);
275                backoff::Error::transient(err)
276            })
277        })
278        .await?;
279
280        let mut read_source: &str = "local";
281        if self.remote_store_url.is_some()
282            && (checkpoints.is_empty()
283                || checkpoints[0].checkpoint_summary.sequence_number
284                    > self.current_checkpoint_number)
285        {
286            checkpoints = self.remote_fetch();
287            read_source = "remote";
288        } else {
289            // cancel remote fetcher execution because local reader has made progress
290            self.remote_fetcher_receiver = None;
291        }
292
293        info!(
294            "Read from {}. Current checkpoint number: {}, pruning watermark: {}, new updates: {:?}",
295            read_source,
296            self.current_checkpoint_number,
297            self.last_pruned_watermark,
298            checkpoints.len(),
299        );
300        for checkpoint in checkpoints {
301            if read_source == "local"
302                && checkpoint.checkpoint_summary.sequence_number > self.current_checkpoint_number
303            {
304                break;
305            }
306            assert_eq!(
307                checkpoint.checkpoint_summary.sequence_number,
308                self.current_checkpoint_number
309            );
310            self.checkpoint_sender.send(checkpoint).await?;
311            self.current_checkpoint_number += 1;
312        }
313        Ok(())
314    }

316    /// Cleans the local directory by removing all processed checkpoint files.
317    fn gc_processed_files(&mut self, watermark: CheckpointSequenceNumber) -> Result<()> {
318        self.data_limiter.gc(watermark);
319        self.last_pruned_watermark = watermark;
320        if !self.options.gc_checkpoint_files {
321            return Ok(());
322        }
323        info!("cleaning processed files, watermark is {}", watermark);
324        for entry in fs::read_dir(self.path.clone())? {
325            let entry = entry?;
326            let filename = entry.file_name();
327            if let Some(sequence_number) = Self::checkpoint_number_from_file_path(&filename)
328                && sequence_number < watermark
329            {
330                fs::remove_file(entry.path())?;
331            }
332        }
333        Ok(())
334    }
335
336    fn checkpoint_number_from_file_path(file_name: &OsString) -> Option<CheckpointSequenceNumber> {
337        file_name
338            .to_str()
339            .and_then(|s| s.rfind('.').map(|pos| &s[..pos]))
340            .and_then(|s| s.parse().ok())
341    }

343    pub fn initialize(
344        path: PathBuf,
345        starting_checkpoint_number: CheckpointSequenceNumber,
346        remote_store_url: Option<String>,
347        remote_store_options: Vec<(String, String)>,
348        options: ReaderOptions,
349    ) -> (
350        Self,
351        mpsc::Receiver<Arc<CheckpointData>>,
352        mpsc::Sender<CheckpointSequenceNumber>,
353        oneshot::Sender<()>,
354    ) {
355        let (checkpoint_sender, checkpoint_recv) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
356        let (processed_sender, processed_receiver) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
357        let (exit_sender, exit_receiver) = oneshot::channel();
358        let reader = Self {
359            path,
360            remote_store_url,
361            remote_store_options,
362            current_checkpoint_number: starting_checkpoint_number,
363            last_pruned_watermark: starting_checkpoint_number,
364            checkpoint_sender,
365            processed_receiver,
366            remote_fetcher_receiver: None,
367            exit_receiver,
368            data_limiter: DataLimiter::new(options.data_limit),
369            options,
370        };
371        (reader, checkpoint_recv, processed_sender, exit_sender)
372    }

374    #[cfg(not(target_os = "macos"))]
375    fn init_watcher(
376        inotify_sender: mpsc::Sender<()>,
377        path: &std::path::Path,
378    ) -> RecommendedWatcher {
379        use notify::Watcher;
380        let mut watcher = notify::recommended_watcher(move |res| {
381            if let Err(err) = res {
382                eprintln!("watch error: {:?}", err);
383            }
384            inotify_sender
385                .blocking_send(())
386                .expect("Failed to send inotify update");
387        })
388        .expect("Failed to init inotify");
389        watcher
390            .watch(path, RecursiveMode::NonRecursive)
391            .expect("Inotify watcher failed");
392        watcher
393    }
394
395    pub async fn run(mut self) -> Result<()> {
396        let (_inotify_sender, mut inotify_recv) = mpsc::channel::<()>(1);
397        std::fs::create_dir_all(self.path.clone()).expect("failed to create a directory");
398
399        #[cfg(not(target_os = "macos"))]
400        let _watcher = Self::init_watcher(_inotify_sender, &self.path);
401
402        self.gc_processed_files(self.last_pruned_watermark)
403            .expect("Failed to clean the directory");
404
405        loop {
406            tokio::select! {
407                _ = &mut self.exit_receiver => break,
408                Some(gc_checkpoint_number) = self.processed_receiver.recv() => {
409                    self.gc_processed_files(gc_checkpoint_number).expect("Failed to clean the directory");
410                }
411                Ok(Some(_)) | Err(_) = timeout(Duration::from_millis(self.options.tick_internal_ms), inotify_recv.recv())  => {
412                    self.sync().await.expect("Failed to read checkpoint files");
413                }
414            }
415        }
416        Ok(())
417    }

420pub struct DataLimiter {
421    limit: usize,
422    queue: BTreeMap<CheckpointSequenceNumber, usize>,
423    in_progress: usize,
424}
425
426impl DataLimiter {
427    fn new(limit: usize) -> Self {
428        Self {
429            limit,
430            queue: BTreeMap::new(),
431            in_progress: 0,
432        }
433    }
434
435    fn exceeds(&self) -> bool {
436        self.limit > 0 && self.in_progress >= self.limit
437    }
438
439    fn add(&mut self, checkpoint: &CheckpointData, size: usize) {
440        if self.limit == 0 {
441            return;
442        }
443        self.in_progress += size;
444        self.queue
445            .insert(checkpoint.checkpoint_summary.sequence_number, size);
446    }
447
448    fn gc(&mut self, watermark: CheckpointSequenceNumber) {
449        if self.limit == 0 {
450            return;
451        }
452        self.queue = self.queue.split_off(&watermark);
453        self.in_progress = self.queue.values().sum();
454    }
