4use crate::{MAX_CHECKPOINTS_IN_PROGRESS, Reducer, Worker};
5use anyhow::Result;
6use async_trait::async_trait;
7use futures::StreamExt;
8use std::collections::HashMap;
9use sui_types::messages_checkpoint::CheckpointSequenceNumber;
10use tokio::sync::mpsc;
11use tokio_stream::wrappers::ReceiverStream;
12
13pub(crate) async fn reduce<W: Worker>(
14    task_name: String,
15    mut current_checkpoint_number: CheckpointSequenceNumber,
16    progress_receiver: mpsc::Receiver<(CheckpointSequenceNumber, W::Result)>,
17    executor_progress_sender: mpsc::Sender<(String, CheckpointSequenceNumber)>,
18    reducer: Option<Box<dyn Reducer<W::Result>>>,
19) -> Result<()> {
20    // convert to a stream of MAX size. This way, each iteration of the loop will process all ready messages
21    let mut stream =
22        ReceiverStream::new(progress_receiver).ready_chunks(*MAX_CHECKPOINTS_IN_PROGRESS);
23    let mut unprocessed = HashMap::new();
24    let mut batch = vec![];
25    let mut progress_update = None;
26
27    while let Some(update_batch) = stream.next().await {
28        for (checkpoint_number, message) in update_batch {
29            unprocessed.insert(checkpoint_number, message);
30        }
31        while let Some(message) = unprocessed.remove(&current_checkpoint_number) {
32            if let Some(ref reducer) = reducer {
33                if reducer.should_close_batch(&batch, Some(&message)) {
34                    commit_with_retry(reducer, std::mem::take(&mut batch)).await?;
35                    batch = vec![message];
36                    progress_update = Some(current_checkpoint_number);
37                } else {
38                    batch.push(message);
39                }
40            }
41            current_checkpoint_number += 1;
42        }
43        match reducer {
44            Some(ref reducer) => {
45                if reducer.should_close_batch(&batch, None) {
46                    commit_with_retry(reducer, std::mem::take(&mut batch)).await?;
47                    progress_update = Some(current_checkpoint_number);
48                }
49            }
50            None => progress_update = Some(current_checkpoint_number),
51        }
52        if let Some(watermark) = progress_update {
53            executor_progress_sender
54                .send((task_name.clone(), watermark))
55                .await?;
56            progress_update = None;
57        }
58    }
59    Ok(())
60}

62async fn commit_with_retry<R: Send + Sync + Clone>(
63    reducer: &dyn Reducer<R>,
64    batch: Vec<R>,
65) -> Result<()> {
66    let backoff = backoff::ExponentialBackoff::default();
67    backoff::future::retry(backoff, || async {
68        reducer.commit(batch.clone()).await.map_err(|err| {
69            eprintln!("transient reducer error {:?}", err);
70            backoff::Error::transient(err)
71        })
72    })
73    .await
74}
75
76#[async_trait]
77impl<R> Reducer<R> for Box<dyn Reducer<R>>
78where
79    R: Send + Sync + Clone,
80{
81    async fn commit(&self, batch: Vec<R>) -> Result<()> {
82        self.as_ref().commit(batch).await
83    }
84
85    fn should_close_batch(&self, batch: &[R], next_item: Option<&R>) -> bool {
86        self.as_ref().should_close_batch(batch, next_item)
87    }
88}
