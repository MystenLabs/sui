4use crate::executor::MAX_CHECKPOINTS_IN_PROGRESS;
5use crate::reducer::reduce;
6use crate::{Reducer, Worker};
7use mysten_metrics::spawn_monitored_task;
8use std::collections::{BTreeSet, VecDeque};
9use std::sync::Arc;
10use std::time::Instant;
11use sui_types::full_checkpoint_content::CheckpointData;
12use sui_types::messages_checkpoint::CheckpointSequenceNumber;
13use tokio::sync::mpsc;
14use tokio::sync::oneshot;
15use tracing::info;
16
17pub struct WorkerPool<W: Worker> {
18    pub task_name: String,
19    concurrency: usize,
20    worker: Arc<W>,
21    reducer: Option<Box<dyn Reducer<W::Result>>>,
22}
23
24impl<W: Worker + 'static> WorkerPool<W> {
25    pub fn new(worker: W, task_name: String, concurrency: usize) -> Self {
26        Self {
27            task_name,
28            concurrency,
29            worker: Arc::new(worker),
30            reducer: None,
31        }
32    }
33    pub fn new_with_reducer(
34        worker: W,
35        task_name: String,
36        concurrency: usize,
37        reducer: Box<dyn Reducer<W::Result>>,
38    ) -> Self {
39        Self {
40            task_name,
41            concurrency,
42            worker: Arc::new(worker),
43            reducer: Some(reducer),
44        }
45    }
46
47    pub async fn run(
48        mut self,
49        watermark: CheckpointSequenceNumber,
50        mut checkpoint_receiver: mpsc::Receiver<Arc<CheckpointData>>,
51        executor_progress_sender: mpsc::Sender<(String, CheckpointSequenceNumber)>,
52    ) {
53        info!(
54            "Starting indexing pipeline {} with concurrency {}. Current watermark is {}.",
55            self.task_name, self.concurrency, watermark
56        );
57        let (progress_sender, mut progress_receiver) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
58        let (reducer_sender, reducer_receiver) = mpsc::channel(*MAX_CHECKPOINTS_IN_PROGRESS);
59        let mut workers = vec![];
60        let mut idle: BTreeSet<_> = (0..self.concurrency).collect();
61        let mut checkpoints = VecDeque::new();
62
63        let mut join_handles = vec![];
64
65        // spawn child workers
66        for worker_id in 0..self.concurrency {
67            let (worker_sender, mut worker_recv) =
68                mpsc::channel::<Arc<CheckpointData>>(*MAX_CHECKPOINTS_IN_PROGRESS);
69            let (term_sender, mut term_receiver) = oneshot::channel::<()>();
70            let cloned_progress_sender = progress_sender.clone();
71            let task_name = self.task_name.clone();
72            workers.push((worker_sender, term_sender));
73
74            let worker = self.worker.clone();
75            let join_handle = spawn_monitored_task!(async move {
76                loop {
77                    tokio::select! {
78                        _ = &mut term_receiver => break,
79                        Some(checkpoint) = worker_recv.recv() => {
80                            let sequence_number = checkpoint.checkpoint_summary.sequence_number;
81                            info!("received checkpoint for processing {} for workflow {}", sequence_number, task_name);
82                            let start_time = Instant::now();
83                            let backoff = backoff::ExponentialBackoff::default();
84                            let result = backoff::future::retry(backoff, || async {
85                                worker
86                                    .clone()
87                                    .process_checkpoint_arc(&checkpoint)
88                                    .await
89                                    .map_err(|err| {
90                                        info!("transient worker execution error {:?} for checkpoint {}", err, sequence_number);
91                                        backoff::Error::transient(err)
92                                    })
93                            })
94                            .await
95                            .expect("checkpoint processing failed for checkpoint");
96                            info!("finished checkpoint processing {} for workflow {} in {:?}", sequence_number, task_name, start_time.elapsed());
97                            if cloned_progress_sender.send((worker_id, sequence_number, result)).await.is_err() {
98                                // The progress channel closing is a sign we need to exit this loop.
99                                break;
100                            }
101                        }
102                    }
103                }
104            });
105
106            // Keep all join handles to ensure all workers are terminated before exiting
107            join_handles.push(join_handle);
108        }
109        spawn_monitored_task!(reduce::<W>(
110            self.task_name.clone(),
111            watermark,
112            reducer_receiver,
113            executor_progress_sender,
114            std::mem::take(&mut self.reducer),
115        ));
116        // main worker pool loop
117        loop {
118            tokio::select! {
119                Some((worker_id, checkpoint_number, message)) = progress_receiver.recv() => {
120                    idle.insert(worker_id);
121                    if reducer_sender.send((checkpoint_number, message)).await.is_err() {
122                        break;
123                    }
124                    while !checkpoints.is_empty() && !idle.is_empty() {
125                        let checkpoint = checkpoints.pop_front().unwrap();
126                        let worker_id = idle.pop_first().unwrap();
127                        if workers[worker_id].0.send(checkpoint).await.is_err() {
128                            // The worker channel closing is a sign we need to exit this loop.
129                            break;
130                        }
131                    }
132                }
133                maybe_checkpoint = checkpoint_receiver.recv() => {
134                    if maybe_checkpoint.is_none() {
135                        break;
136                    }
137                    let checkpoint = maybe_checkpoint.expect("invariant's checked");
138                    let sequence_number = checkpoint.checkpoint_summary.sequence_number;
139                    if sequence_number < watermark {
140                        continue;
141                    }
142                    self.worker.preprocess_hook(&checkpoint).expect("failed to preprocess task");
143                    if idle.is_empty() {
144                        checkpoints.push_back(checkpoint);
145                    } else {
146                        let worker_id = idle.pop_first().unwrap();
147                        if workers[worker_id].0.send(checkpoint).await.is_err() {
148                            // The worker channel closing is a sign we need to exit this loop.
149                            break;
150                        };
151                    }
152                }
153            }
154        }
155
156        // Clean up code for graceful termination
157
158        // Notify the exit handles of all workers to terminate
159        drop(workers);
160
161        // Wait for all workers to finish
162        for join_handle in join_handles {
163            join_handle.await.expect("worker thread panicked");
164        }
165    }
166}
