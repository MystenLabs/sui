4//! GQL Queries
5//! Interface to the rpc for the gql schema defined in `crates\sui-indexer-alt-graphql/schema.graphql`.
6//! Built in 3 modules: epoch_query, txn_query, object_query.
7//! No GQL type escapes this module. From here we return structures defined in this crate
8//! or bcs encoded data of runtime structures.
9//!
10//! This module is private to the `DataStore` and packcaged in its own module for convenience.
11
12use crate::{EpochData, stores::DataStore};
13use anyhow::{Context, Error, anyhow};
14use cynic::QueryBuilder;
15use fastcrypto::encoding::{Base64 as CryptoBase64, Encoding};
16
17// Register the schema which was loaded in the build.rs call.
18#[cynic::schema("rpc")]
19mod schema {
20    use chrono::{DateTime as ChronoDateTime, Utc};
21    cynic::impl_scalar!(u64, UInt53);
22    cynic::impl_scalar!(ChronoDateTime<Utc>, DateTime);
23}

25pub(crate) mod epoch_query {
26    use super::*;
27    use chrono::{DateTime as ChronoDateTime, Utc};
28
29    #[derive(cynic::QueryVariables)]
30    pub(crate) struct EpochDataArgs {
31        pub epoch: Option<u64>,
32    }
33
34    #[derive(cynic::QueryFragment)]
35    #[cynic(variables = "EpochDataArgs")]
36    pub(crate) struct Query {
37        #[arguments(epochId: $epoch)]
38        epoch: Option<Epoch>,
39    }
40
41    #[derive(cynic::Scalar, Clone)]
42    #[cynic(graphql_type = "BigInt")]
43    pub(crate) struct BigInt(String);
44
45    #[derive(cynic::QueryFragment)]
46    pub(crate) struct Epoch {
47        epoch_id: u64,
48        protocol_configs: Option<ProtocolConfigs>,
49        reference_gas_price: Option<BigInt>,
50        start_timestamp: Option<ChronoDateTime<Utc>>,
51    }
52
53    #[derive(cynic::QueryFragment)]
54    pub(crate) struct ProtocolConfigs {
55        protocol_version: u64,
56    }
57
58    pub(crate) async fn query(
59        epoch_id: u64,
60        data_store: &DataStore,
61    ) -> Result<Option<EpochData>, Error> {
62        let query = Query::build(EpochDataArgs {
63            epoch: Some(epoch_id),
64        });
65        let response = data_store.run_query(&query).await?;
66
67        let Some(epoch) = response.data.and_then(|epoch| epoch.epoch) else {
68            return Ok(None);
69        };
70        Ok(Some(EpochData {
71            epoch_id: epoch.epoch_id,
72            protocol_version: epoch
73                .protocol_configs
74                .map(|config| config.protocol_version)
75                .ok_or(anyhow!(format!(
76                    "Canot find epoch info for epoch {}",
77                    epoch_id
78                )))?,
79            rgp: epoch
80                .reference_gas_price
81                .map(|rgp| rgp.0)
82                .ok_or(anyhow!(format!(
83                    "Canot find epoch info for epoch {}",
84                    epoch_id
85                )))?
86                .parse()
87                .unwrap(),
88            start_timestamp: epoch
89                .start_timestamp
90                .map(|ts| ts.timestamp_millis() as u64)
91                .ok_or(anyhow!(format!(
92                    "Canot find epoch info for epoch {}",
93                    epoch_id
94                )))?,
95        }))
96    }
97}

99pub(crate) mod txn_query {
100    use super::*;
101    use sui_types::transaction::TransactionData;
102
103    #[derive(cynic::Scalar, Debug, Clone)]
104    #[cynic(graphql_type = "Base64")]
105    pub(crate) struct Base64(pub String);
106
107    #[derive(cynic::QueryVariables)]
108    pub(crate) struct TransactionDataArgs {
109        pub digest: String,
110    }
111
112    #[derive(cynic::QueryFragment)]
113    #[cynic(variables = "TransactionDataArgs")]
114    pub(crate) struct Query {
115        #[arguments(digest: $digest)]
116        transaction: Option<Transaction>,
117    }
118
119    #[derive(cynic::QueryFragment)]
120    pub(crate) struct Transaction {
121        transaction_bcs: Option<Base64>,
122        effects: Option<TransactionEffects>,
123    }
124
125    #[derive(cynic::QueryFragment)]
126    pub(crate) struct TransactionEffects {
127        checkpoint: Option<Checkpoint>,
128        effects_bcs: Option<Base64>,
129    }
130
131    #[derive(cynic::QueryFragment)]
132    pub(crate) struct Checkpoint {
133        sequence_number: u64,
134    }
135
136    pub(crate) async fn query(
137        digest: String,
138        data_store: &DataStore,
139    ) -> Result<Option<(TransactionData, sui_types::effects::TransactionEffects, u64)>, Error> {
140        let query = Query::build(TransactionDataArgs {
141            digest: digest.clone(),
142        });
143        let response = data_store
144            .run_query(&query)
145            .await
146            .context("Failed to run transaction query")?;
147
148        let Some(transaction) = response.data.and_then(|txn| txn.transaction) else {
149            return Ok(None);
150        };
151
152        let txn_data: TransactionData = bcs::from_bytes(
153            &CryptoBase64::decode(
154                &transaction
155                    .transaction_bcs
156                    .ok_or_else(|| {
157                        anyhow!(format!(
158                            "Transaction data not available (None) for digest: {}",
159                            digest
160                        ),)
161                    })?
162                    .0,
163            )
164            .context(format!(
165                "Transaction data does not decode for digest: {}",
166                digest
167            ))?,
168        )
169        .context(format!(
170            "Cannot deserialize transaction data for digest {}",
171            digest
172        ))?;
173
174        let effect_frag = transaction
175            .effects
176            .ok_or_else(|| anyhow!("Missing effects in transaction data response"))?;
177        let effects: sui_types::effects::TransactionEffects = bcs::from_bytes(
178            &CryptoBase64::decode(
179                &effect_frag
180                    .effects_bcs
181                    .ok_or_else(|| anyhow!("Missing effects bcs in transaction data response"))?
182                    .0,
183            )
184            .context(format!(
185                "Transaction effects do not decode for digest: {}",
186                digest
187            ))?,
188        )
189        .context(format!(
190            "Cannot deserialize transaction effects for digest {}",
191            digest
192        ))?;
193
194        let checkpoint = effect_frag
195            .checkpoint
196            .ok_or_else(|| anyhow!("Missing checkpoint in transaction query response"))?
197            .sequence_number;
198
199        Ok(Some((txn_data, effects, checkpoint)))
200    }
201}

203pub(crate) mod object_query {
204    use sui_types::object::Object;
205
206    use super::*;
207    use crate::{ObjectKey as GqlObjectKey, VersionQuery};
208
209    #[derive(cynic::Scalar, Debug, Clone)]
210    #[cynic(graphql_type = "SuiAddress")]
211    pub(crate) struct SuiAddress(pub String);
212
213    #[derive(cynic::Scalar, Debug, Clone)]
214    #[cynic(graphql_type = "Base64")]
215    pub(crate) struct Base64(pub String);
216
217    #[derive(cynic::InputObject, Debug)]
218    #[cynic(graphql_type = "ObjectKey")]
219    pub(crate) struct ObjectKey {
220        pub address: SuiAddress,
221        pub version: Option<u64>,
222        pub root_version: Option<u64>,
223        pub at_checkpoint: Option<u64>,
224    }
225
226    #[derive(cynic::QueryVariables)]
227    pub(crate) struct MultiGetObjectsVars {
228        pub keys: Vec<ObjectKey>,
229    }
230
231    #[derive(cynic::QueryFragment)]
232    #[cynic(variables = "MultiGetObjectsVars", graphql_type = "Query")]
233    pub(crate) struct MultiGetObjectsQuery {
234        #[arguments(keys: $keys)]
235        pub multi_get_objects: Vec<Option<ObjectFragment>>,
236    }
237
238    #[derive(cynic::QueryFragment)]
239    #[cynic(graphql_type = "Object", schema_module = "crate::gql_queries::schema")]
240    pub(crate) struct ObjectFragment {
241        #[allow(dead_code)]
242        pub address: SuiAddress,
243        pub version: Option<u64>,
244        pub object_bcs: Option<Base64>,
245    }
246
247    // Maximum number of keys to query in a single request.
248    // REVIEW: not clear how this translate to the 5000B limit, so
249    // we are picking a "random" and conservative number.
250    const MAX_KEYS_SIZE: usize = 30;
251
252    pub(crate) async fn query(
253        keys: &[GqlObjectKey],
254        data_store: &DataStore,
255    ) -> Result<Vec<Option<(Object, u64)>>, Error> {
256        let mut keys = keys
257            .iter()
258            .cloned()
259            .map(ObjectKey::from)
260            .collect::<Vec<_>>();
261        let mut key_chunks = vec![];
262        while !keys.is_empty() {
263            let chunk: Vec<_> = keys.drain(..MAX_KEYS_SIZE.min(keys.len())).collect();
264            key_chunks.push(chunk);
265        }
266
267        let mut objects = vec![];
268
269        for keys in key_chunks {
270            let query: cynic::Operation<MultiGetObjectsQuery, MultiGetObjectsVars> =
271                MultiGetObjectsQuery::build(MultiGetObjectsVars { keys });
272            let response = data_store.run_query(&query).await?;
273
274            let list = if let Some(data) = response.data {
275                data.multi_get_objects
276            } else {
277                return Err(anyhow!(
278                    "Missing data in transaction query response. Errors: {:?}",
279                    response.errors,
280                ));
281            };
282
283            let chunk = list
284                .into_iter()
285                .map(|frag| match frag {
286                    Some(frag) => {
287                        let b64 = frag
288                            .object_bcs
289                            .ok_or_else(|| anyhow!("Object bcs is None for object"))?
290                            .0;
291                        let bytes = CryptoBase64::decode(&b64)?;
292                        let obj: Object = bcs::from_bytes(&bytes)?;
293                        let version = frag
294                            .version
295                            .ok_or_else(|| anyhow!("Object version is None for object"))?;
296                        Ok::<_, Error>(Some((obj, version)))
297                    }
298                    None => Ok::<_, Error>(None),
299                })
300                .collect::<Result<Vec<Option<(Object, u64)>>, _>>()?;
301            objects.extend(chunk);
302        }
303        Ok(objects)
304    }
305
306    impl From<GqlObjectKey> for ObjectKey {
307        fn from(key: GqlObjectKey) -> Self {
308            ObjectKey {
309                address: SuiAddress(key.object_id.to_string()),
310                version: match key.version_query {
311                    VersionQuery::Version(v) => Some(v),
312                    _ => None,
313                },
314                root_version: match key.version_query {
315                    VersionQuery::RootVersion(v) => Some(v),
316                    _ => None,
317                },
318                at_checkpoint: match key.version_query {
319                    VersionQuery::AtCheckpoint(v) => Some(v),
320                    _ => None,
321                },
322            }
323        }
324    }
325}

327pub(crate) mod chain_id_query {
328    use super::*;
329
330    #[derive(cynic::QueryFragment)]
331    pub(crate) struct Query {
332        chain_identifier: Option<String>,
333    }
334
335    pub(crate) async fn query(data_store: &DataStore) -> Result<String, Error> {
336        let query = Query::build(());
337        let response = data_store.run_query(&query).await?;
338        let Some(chain_id) = response.data.and_then(|data| data.chain_identifier) else {
339            return Err(anyhow!("Missing chain identifier"));
340        };
341        Ok(chain_id)
342    }
