4//! Multi-tier caching data store for Sui blockchain data.
5//!
6//! This crate provides a flexible data store abstraction for retrieving and caching
7//! Sui blockchain data (transactions, epochs, objects). The stores are loosely modeled
8//! after the GQL schema in `crates/sui-indexer-alt-graphql/schema.graphql`.
9//!
10//! ## Core Traits
11//!
12//! - [`TransactionStore`] - Retrieve transaction data and effects by digest
13//! - [`EpochStore`] - Retrieve epoch information and protocol configuration
14//! - [`ObjectStore`] - Retrieve objects by their keys with flexible version queries
15//!
16//! ## Store Implementations
17//!
18//! - [`stores::DataStore`] - Remote GraphQL-backed store (mainnet/testnet)
19//! - [`stores::FileSystemStore`] - Persistent local disk cache
20//! - [`stores::InMemoryStore`] - Unbounded in-memory cache
21//! - [`stores::LruMemoryStore`] - Bounded LRU cache
22//! - [`stores::ReadThroughStore`] - Composable two-tier caching pattern
23//!
24//! ## Composition
25//!
26//! Use `ReadThroughStore<Primary, Secondary>` to compose cache layers:
27//! - `ReadThroughStore<LruMemoryStore, DataStore>` - LRU + remote
28//! - `ReadThroughStore<InMemoryStore, FileSystemStore>` - Memory + disk
29//!   (e.g., for testing in CI with pre-populated disk cache)
30

49/// Transaction data with effects and checkpoint.
50#[derive(Clone, Debug)]
51pub struct TransactionInfo {
52    pub data: TransactionData,
53    pub effects: TransactionEffects,
54    pub checkpoint: u64,
55}
56
57/// A `TransactionStore` has to be able to retrieve transaction data for a given digest.
58/// The data provided to `sui_execution::executor::Executor::execute_transaction_to_effects`
59/// must be available. Some of that data is not provided by the user. It is naturally available
60/// at runtime on a live system and later saved in effects and in the context of a checkpoint.
61pub trait TransactionStore {
62    /// Given a transaction digest, return transaction info including data, effects,
63    /// and the checkpoint that transaction was executed in.
64    /// Returns `None` if the transaction is not found.
65    fn transaction_data_and_effects(
66        &self,
67        tx_digest: &str,
68    ) -> Result<Option<TransactionInfo>, Error>;
69}
70
71/// Epoch data.
72#[derive(Clone, Debug)]
73pub struct EpochData {
74    pub epoch_id: u64,
75    pub protocol_version: u64,
76    pub rgp: u64,
77    pub start_timestamp: u64,
78}
79
80/// An `EpochStore` retrieves the epoch data and protocol configuration
81/// for a given epoch.
82/// Epoch data is collected by an indexer and it is not stored anywhere otherwise.
83/// This is a very small amount of information and could conceivably be saved locally
84/// and never hit a server.
85pub trait EpochStore {
86    /// Return the `EpochData` for a given epoch.
87    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error>;
88    /// Return the `ProtocolConfig` for a given epoch.
89    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error>;
90}
91
92/// Query for an object.
93/// Specifies an `ObjectID` and the "rule" to retrieve it.
94#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
95pub struct ObjectKey {
96    pub object_id: ObjectID,
97    pub version_query: VersionQuery,
98}
99
100/// Query options for an object.
101/// `Version` request an object at a specific version
102/// `RootVersion` request an object at a given version at most (<=)
103/// `AtCheckpoint` request an object at a given checkpoint. Useful for unknown `Version`.
104#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
105pub enum VersionQuery {
106    Version(u64),
107    RootVersion(u64),
108    AtCheckpoint(u64),
109}
110
111/// The `ObjectStore` trait is used to retrieve objects by their keys,
112/// with different query options.
113///
114/// This trait can execute a subset of what is allowed by
115/// `crates/sui-indexer-alt-graphql/schema.graphql::multiGetObjects`.
116/// That query likely allows more than what most clients need, which is fairly limited in
117/// its usage.
118pub trait ObjectStore {
119    /// Retrieve objects by their keys, with different query options.
120    ///
121    /// If the object is not found, the element in the vector is `None`.
122    /// Otherwise each tuple contains:
123    /// - `Object`: The object data
124    /// - `u64`: The actual version of the object
125    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error>;
126}

132// This is a bit of a hack to allow the FileSystemStore to map network to chain id.
133// It is not exposed in a very simple and consistent way quite yet and something
134// we want to revisit in the future.
135
136/// A trait to set up the data store.
137/// This is used to setup internal state of the data store before use.
138/// At the moment is exclusively used by the FileSystemStore to map network to chain id.
139pub trait SetupStore {
140    /// Set up the data store.
141    /// Returns the chain identifier if available, or None if not available.
142    /// When `chain_id` is `None` this is a no-op, unless the given data store
143    /// has a way to fetch the chain id from the network.
144    /// When `chain_id` is `Some(chain_id)` the given data store should override
145    /// the map from network to chain id if it has one.
146    /// That is a meaningful operation only for the FileSystemStore.
147    fn setup(&self, chain_id: Option<String>) -> Result<Option<String>, Error>;
148}

154/// Write-back trait for transaction data.
155/// Allows storing transaction data, effects, and checkpoint information.
156pub trait TransactionStoreWriter: TransactionStore {
157    /// Store transaction data, effects, and the checkpoint it was executed in.
158    fn write_transaction(
159        &self,
160        tx_digest: &str,
161        transaction_info: TransactionInfo,
162    ) -> Result<(), Error>;
163}
164
165/// Write-back trait for epoch data.
166/// Allows storing epoch information.
167pub trait EpochStoreWriter: EpochStore {
168    /// Store epoch data for a given epoch.
169    fn write_epoch_info(&self, epoch: u64, epoch_data: EpochData) -> Result<(), Error>;
170}
171
172/// Write-back trait for object data.
173/// Allows storing objects and their version mappings based on the query type.
174pub trait ObjectStoreWriter: ObjectStore {
175    /// Store object data based on the ObjectKey and actual version.
176    ///
177    /// Behavior depends on the VersionQuery in the key:
178    /// - `Version(v)`: Stores the object at version `v` (actual_version should equal `v`)
179    /// - `RootVersion(max_v)`: Stores a mapping from `max_v` to `actual_version` and
180    ///   the object at `actual_version`
181    /// - `AtCheckpoint(checkpoint)`: Stores a mapping from `checkpoint` to `actual_version` and
182    ///   the object at `actual_version`
183    fn write_object(
184        &self,
185        key: &ObjectKey,
186        object: Object,
187        actual_version: u64,
188    ) -> Result<(), Error>;
189}

195/// A reporting trait for data stores to print a usage/summary report.
196///
197/// Implementors are free to print any relevant statistics or configuration details.
198/// The writer allows callers to decide where summaries go (stdout, file, buffers, etc.).
199pub trait StoreSummary {
200    fn summary<W: Write>(&self, writer: &mut W) -> Result<()>;
201}

214}
