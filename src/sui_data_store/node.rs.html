4//! Network node configuration for Sui data stores.
5//!
6//! Defines the [`Node`] enum for specifying which Sui network to connect to
7//! (mainnet, testnet, or custom) and provides URL resolution for both
8//! GraphQL and JSON-RPC endpoints.
9
10use std::str::FromStr;
11use sui_types::supported_protocol_versions::Chain;
12
13/// GraphQL endpoint for Sui mainnet.
14pub const MAINNET_GQL_URL: &str = "https://graphql.mainnet.sui.io/graphql";
15/// GraphQL endpoint for Sui testnet.
16pub const TESTNET_GQL_URL: &str = "https://graphql.testnet.sui.io/graphql";
17/// JSON-RPC endpoint for Sui mainnet.
18pub const MAINNET_RPC_URL: &str = "https://fullnode.mainnet.sui.io:443";
19/// JSON-RPC endpoint for Sui testnet.
20pub const TESTNET_RPC_URL: &str = "https://fullnode.testnet.sui.io:443";
21
22/// Represents a Sui network node configuration.
23///
24/// Used to specify which network the data store should connect to.
25#[derive(Clone, Debug)]
26pub enum Node {
27    /// Sui mainnet
28    Mainnet,
29    /// Sui testnet
30    Testnet,
31    /// Custom network with a user-provided URL
32    Custom(String),
33}
34
35impl Node {
36    /// Returns the [`Chain`] identifier for this node.
37    pub fn chain(&self) -> Chain {
38        match self {
39            Node::Mainnet => Chain::Mainnet,
40            Node::Testnet => Chain::Testnet,
41            Node::Custom(_) => Chain::Unknown,
42        }
43    }
44
45    /// Returns a human-readable network name.
46    pub fn network_name(&self) -> String {
47        match self {
48            Node::Mainnet => "mainnet".to_string(),
49            Node::Testnet => "testnet".to_string(),
50            Node::Custom(url) => url.clone(),
51        }
52    }
53
54    /// Returns the GraphQL endpoint URL for this node.
55    pub fn gql_url(&self) -> &str {
56        match self {
57            Node::Mainnet => MAINNET_GQL_URL,
58            Node::Testnet => TESTNET_GQL_URL,
59            Node::Custom(_url) => todo!("custom gql url not implemented"),
60        }
61    }
62
63    /// Returns the JSON-RPC endpoint URL for this node.
64    pub fn node_url(&self) -> &str {
65        match self {
66            Node::Mainnet => MAINNET_RPC_URL,
67            Node::Testnet => TESTNET_RPC_URL,
68            // For custom, assume it's already an RPC URL
69            Node::Custom(url) => url.as_str(),
70        }
71    }
72}
73
74impl FromStr for Node {
75    type Err = String;
76
77    fn from_str(s: &str) -> Result<Self, Self::Err> {
78        match s {
79            "mainnet" => Ok(Node::Mainnet),
80            "testnet" => Ok(Node::Testnet),
81            _ => Ok(Node::Custom(s.to_string())),
82        }
83    }
84}
