4//! File system implementation of the data store interfaces: `TransactionStore`, `EpochStore`,
5//! and `ObjectStore`.
6//! Data is persisted on disk under a simple, human-inspectable directory layout.
7//!
8//! # Directory Structure
9//!
10//! ```text
11//! ~/.sui_data_store/
12//!   node_mapping.csv              (CSV: "node,chain_id" mappings)
13//!   <chain_id>/
14//!     transaction/
15//!       <tx_digest>              (BCS: TransactionFileData)
16//!     epoch/
17//!       <epoch_id>               (BCS: EpochFileData)
18//!     objects/
19//!       <object_id>/
20//!         <version>              (BCS: sui_types::object::Object)
21//!         root_versions          (CSV lines: "<max_version>,<actual_version>")
22//!         checkpoint_versions    (CSV lines: "<checkpoint>,<actual_version>")
23//! ```
24//!
25//! - The top-level directory is fixed to `~/.sui_data_store`.
26//! - The `node_mapping.csv` file maps node types to chain identifiers.
27//! - The next level directory is the chain identifier (e.g., "sui_mainnet", "sui_testnet").
28//! - Node types: `mainnet`, `testnet`, `devnet` (not yet available), and custom URLs.
29//!
30//! # File Formats
31//!
32//! - Transaction files in `transaction/<tx_digest>` store `TransactionFileData` as BCS, which
33//!   includes the original `TransactionData`, its `TransactionEffects`, and the execution
34//!   `checkpoint`.
35//! - Epoch files in `epoch/<epoch_id>` store `EpochFileData` as BCS, capturing epoch metadata.
36//! - Object files in `objects/<object_id>/<version>` store the `Object` at the corresponding
37//!   version as BCS.
38//!
39//! # Version Mapping Files
40//!
41//! The `root_versions` and `checkpoint_versions` files provide stable lookups for queries
42//! that are not targeted to a specific version. Each line is a comma-separated pair and the
43//! files can be edited manually or updated by this store:
44//!
45//! - `root_versions`: maps a root version bound to the actual stored version.
46//!   - line format: `<max_version>,<actual_version>`
47//!   - example: `4,3` means the highest version not exceeding 4 is 3
48//!
49//! - `checkpoint_versions`: maps a checkpoint to the actual stored version for the object
50//!   at that point.
51//!   - line format: `<checkpoint>,<actual_version>`
52//!   - example: `100,1` means at checkpoint 100 the object version is 1
53//!
54//! These mapping files allow answering `VersionQuery::RootVersion(_)` and
55//! `VersionQuery::AtCheckpoint(_)` without requiring a full index; the store writes them as
56//! it learns the concrete versions.

166impl FileSystemStore {
167    pub fn new(node: Node) -> Result<Self, Error> {

171            base_path,

178    pub fn base_path() -> Result<PathBuf, Error> {
179        let home_dir = std::env::var("SUI_DATA_STORE")
180            .or_else(|_| std::env::var("SUI_CONFIG_DIR"))
181            .or_else(|_| std::env::var("HOME"))
182            .or_else(|_| std::env::var("USERPROFILE"))
183            .map_err(|_| {
184                anyhow!(
185                    "Cannot determine home directory. Define a SUI_DATA_STORE environment variable"
186                )
187            })?;
188        Ok(PathBuf::from(home_dir).join(DATA_STORE_DIR))
189    }
190
191    fn node_dir(&self) -> Result<PathBuf, Error> {
192        // Read the chain identifier mapping to determine the directory name
193        let chain_id = self.get_chain_id_for_node()?;
194        Ok(self.base_path.join(chain_id))
195    }
196
197    fn get_chain_id_for_node(&self) -> Result<String, Error> {
198        let mapping_file = self.base_path.join(NODE_MAPPING_FILE);
199
200        if !mapping_file.exists() {
201            return Err(anyhow!(
202                "Node mapping file not found at {}. File must exist with format: node,chain_id",
203                mapping_file.display()
204            ));
205        }
206
207        let file = fs::File::open(&mapping_file).with_context(|| {
208            format!(
209                "Failed to open node mapping file: {}",
210                mapping_file.display()
211            )
212        })?;
213
214        let mut rdr = csv::ReaderBuilder::new()
215            .has_headers(false)
216            .trim(csv::Trim::All)
217            .from_reader(file);
218
219        let node_key = match &self.node {
220            Node::Mainnet => "mainnet",
221            Node::Testnet => "testnet",
222            Node::Custom(url) => url,
223        };
224
225        for result in rdr.records() {
226            let record = result.with_context(|| {
227                format!("Failed to read CSV record from: {}", mapping_file.display())
228            })?;
229
230            if record.len() != 2 {
231                return Err(anyhow!(
232                    "Invalid format in node mapping file {}: expected 2 columns, got {}",
233                    mapping_file.display(),
234                    record.len()
235                ));
236            }
237
238            let node = record[0].trim();
239            let chain_id = record[1].trim();
240
241            if node == node_key {
242                return Ok(chain_id.to_string());
243            }
244        }
245
246        Err(anyhow!(
247            "No mapping found for node '{}' in mapping file {}",
248            node_key,
249            mapping_file.display()
250        ))
251    }

279    fn read_bcs_file<T: serde::de::DeserializeOwned>(&self, path: &Path) -> Result<T, Error> {
280        let bytes =
281            fs::read(path).with_context(|| format!("Failed to read file: {}", path.display()))?;
282        bcs::from_bytes(&bytes)
283            .with_context(|| format!("Failed to deserialize BCS data from: {}", path.display()))
284    }
285
286    fn write_bcs_file<T: serde::Serialize>(&self, path: &Path, data: &T) -> Result<(), Error> {
287        if let Some(parent) = path.parent() {
288            fs::create_dir_all(parent)
289                .with_context(|| format!("Failed to create directory: {}", parent.display()))?;
290        }
291        let bytes = bcs::to_bytes(data)
292            .with_context(|| format!("Failed to serialize BCS data for: {}", path.display()))?;
293        fs::write(path, bytes)
294            .with_context(|| format!("Failed to write file: {}", path.display()))?;
295        Ok(())
296    }
297
298    fn read_version_mapping(&self, path: &Path) -> Result<BTreeMap<u64, u64>, Error> {
299        if !path.exists() {
300            return Ok(BTreeMap::new());
301        }
302
303        let file = fs::File::open(path)
304            .with_context(|| format!("Failed to open version mapping file: {}", path.display()))?;
305
306        let mut rdr = csv::ReaderBuilder::new()
307            .has_headers(false)
308            .trim(csv::Trim::All)
309            .from_reader(file);
310
311        let mut mapping = BTreeMap::new();
312        for result in rdr.records() {
313            let record = result
314                .with_context(|| format!("Failed to read CSV record from: {}", path.display()))?;
315
316            if record.len() != 2 {
317                return Err(anyhow!(
318                    "Invalid format in version mapping file {}: expected 2 columns, got {}",
319                    path.display(),
320                    record.len()
321                ));
322            }
323
324            let key: u64 = record[0].parse().with_context(|| {
325                format!("Failed to parse key '{}' in {}", &record[0], path.display())
326            })?;
327            let version: u64 = record[1].parse().with_context(|| {
328                format!(
329                    "Failed to parse version '{}' in {}",
330                    &record[1],
331                    path.display()
332                )
333            })?;
334
335            mapping.insert(key, version);
336        }
337
338        Ok(mapping)
339    }
340
341    fn write_full_version_mapping(
342        &self,
343        path: &Path,
344        mapping: &BTreeMap<u64, u64>,
345    ) -> Result<(), Error> {
346        if let Some(parent) = path.parent() {
347            fs::create_dir_all(parent)
348                .with_context(|| format!("Failed to create directory: {}", parent.display()))?;
349        }
350        let mut content = String::new();
351        for (k, v) in mapping {
352            content.push_str(&format!("{},{}\n", k, v));
353        }
354        fs::write(path, content)
355            .with_context(|| format!("Failed to write version mapping file: {}", path.display()))?;
356        Ok(())
357    }
358
359    fn load_root_mapping(&self, object_id: &ObjectID) -> Result<(), Error> {
360        if self
361            .root_versions_map
362            .read()
363            .unwrap()
364            .contains_key(object_id)
365        {
366            return Ok(());
367        }
368        let path = self.root_versions_path(object_id)?;
369        let mapping = self.read_version_mapping(&path)?;
370        self.root_versions_map
371            .write()
372            .unwrap()
373            .insert(*object_id, mapping);
374        Ok(())
375    }
376
377    fn load_checkpoint_mapping(&self, object_id: &ObjectID) -> Result<(), Error> {
378        if self
379            .checkpoint_versions_map
380            .read()
381            .unwrap()
382            .contains_key(object_id)
383        {
384            return Ok(());
385        }
386        let path = self.checkpoint_versions_path(object_id)?;
387        let mapping = self.read_version_mapping(&path)?;
388        self.checkpoint_versions_map
389            .write()
390            .unwrap()
391            .insert(*object_id, mapping);
392        Ok(())
393    }
394
395    fn update_root_mapping(
396        &self,
397        object_id: &ObjectID,
398        key: u64,
399        version: u64,
400    ) -> Result<(), Error> {
401        self.load_root_mapping(object_id)?;
402        {
403            let mut maps = self.root_versions_map.write().unwrap();
404            let entry = maps.get_mut(object_id).unwrap();
405            entry.insert(key, version);
406            let path = self.root_versions_path(object_id)?;
407            self.write_full_version_mapping(&path, entry)?;
408        }
409        Ok(())
410    }
411
412    fn update_checkpoint_mapping(
413        &self,
414        object_id: &ObjectID,
415        key: u64,
416        version: u64,
417    ) -> Result<(), Error> {
418        self.load_checkpoint_mapping(object_id)?;
419        {
420            let mut maps = self.checkpoint_versions_map.write().unwrap();
421            let entry = maps.get_mut(object_id).unwrap();
422            entry.insert(key, version);
423            let path = self.checkpoint_versions_path(object_id)?;
424            self.write_full_version_mapping(&path, entry)?;
425        }
426        Ok(())
427    }
428
429    fn get_object_by_version(
430        &self,
431        object_id: &ObjectID,
432        version: u64,
433    ) -> Result<Option<(Object, u64)>, Error> {
434        let object_dir = self.objects_dir()?.join(object_id.to_string());
435        let version_file = object_dir.join(version.to_string());
436        if !version_file.exists() {
437            return Ok(None);
438        }
439        self.read_bcs_file(&version_file)
440            .map(|obj| Some((obj, version)))
441    }
442
443    fn get_object_by_root_version(
444        &self,
445        object_id: &ObjectID,
446        max_version: u64,
447    ) -> Result<Option<(Object, u64)>, Error> {
448        self.load_root_mapping(object_id)?;
449        let maps = self.root_versions_map.read().unwrap();
450        if let Some(map) = maps.get(object_id)
451            && let Some(&actual_version) = map.get(&max_version)
452        {
453            return self.get_object_by_version(object_id, actual_version);
454        }
455        Ok(None)
456    }
457
458    fn get_object_by_checkpoint(
459        &self,
460        object_id: &ObjectID,
461        checkpoint: u64,
462    ) -> Result<Option<(Object, u64)>, Error> {
463        self.load_checkpoint_mapping(object_id)?;
464        let maps = self.checkpoint_versions_map.read().unwrap();
465        if let Some(map) = maps.get(object_id)
466            && let Some(&actual_version) = map.get(&checkpoint)
467        {
468            return self.get_object_by_version(object_id, actual_version);
469        }
470        Ok(None)
471    }

481impl TransactionStore for FileSystemStore {
482    fn transaction_data_and_effects(
483        &self,
484        tx_digest: &str,
485    ) -> Result<Option<TransactionInfo>, Error> {
486        let file_path = self.transaction_dir()?.join(tx_digest);
487        if !file_path.exists() {
488            self.metrics.txn_miss.fetch_add(1, Ordering::Relaxed);
489            return Ok(None);
490        }
491        let txn_data: TransactionFileData = match self
492            .read_bcs_file(&file_path)
493            .with_context(|| format!("Failed to load transaction data for digest: {}", tx_digest))
494        {
495            Ok(v) => v,
496            Err(e) => {
497                self.metrics.txn_error.fetch_add(1, Ordering::Relaxed);
498                return Err(e);
499            }
500        };
501        self.metrics.txn_hit.fetch_add(1, Ordering::Relaxed);
502        Ok(Some(TransactionInfo {
503            data: txn_data.data,
504            effects: txn_data.effects,
505            checkpoint: txn_data.checkpoint,
506        }))
507    }
508}

510impl EpochStore for FileSystemStore {
511    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error> {
512        let file_path = self.epoch_dir()?.join(epoch.to_string());
513        if !file_path.exists() {
514            self.metrics.epoch_miss.fetch_add(1, Ordering::Relaxed);
515            return Ok(None);
516        }
517        let epoch_file_data: EpochFileData = match self
518            .read_bcs_file(&file_path)
519            .with_context(|| format!("Failed to load epoch data for epoch: {}", epoch))
520        {
521            Ok(v) => v,
522            Err(e) => {
523                self.metrics.epoch_error.fetch_add(1, Ordering::Relaxed);
524                return Err(e);
525            }
526        };
527        self.metrics.epoch_hit.fetch_add(1, Ordering::Relaxed);
528        Ok(Some(epoch_file_data.into()))
529    }
530
531    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error> {
532        match self.epoch_info(epoch) {
533            Ok(Some(epoch_data)) => {
534                self.metrics.proto_hit.fetch_add(1, Ordering::Relaxed);
535                Ok(Some(ProtocolConfig::get_for_version(
536                    ProtocolVersion::new(epoch_data.protocol_version),
537                    self.chain(),
538                )))
539            }
540            Ok(None) => {
541                self.metrics.proto_miss.fetch_add(1, Ordering::Relaxed);
542                Ok(None)
543            }
544            Err(e) => {
545                self.metrics.proto_error.fetch_add(1, Ordering::Relaxed);
546                Err(e)
547            }
548        }
549    }
550}

552impl ObjectStore for FileSystemStore {
553    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
554        let mut results = Vec::with_capacity(keys.len());
555        for key in keys {
556            let (object_and_version_res, hit_ctr, miss_ctr, err_ctr) = match &key.version_query {
557                VersionQuery::Version(version) => (
558                    self.get_object_by_version(&key.object_id, *version),
559                    &self.metrics.obj_version_hit,
560                    &self.metrics.obj_version_miss,
561                    &self.metrics.obj_version_error,
562                ),
563                VersionQuery::RootVersion(max_version) => (
564                    self.get_object_by_root_version(&key.object_id, *max_version),
565                    &self.metrics.obj_root_hit,
566                    &self.metrics.obj_root_miss,
567                    &self.metrics.obj_root_error,
568                ),
569                VersionQuery::AtCheckpoint(checkpoint) => (
570                    self.get_object_by_checkpoint(&key.object_id, *checkpoint),
571                    &self.metrics.obj_checkpoint_hit,
572                    &self.metrics.obj_checkpoint_miss,
573                    &self.metrics.obj_checkpoint_error,
574                ),
575            };
576
577            match object_and_version_res {
578                Ok(object_and_version) => {
579                    if object_and_version.is_some() {
580                        hit_ctr.fetch_add(1, Ordering::Relaxed);
581                    } else {
582                        miss_ctr.fetch_add(1, Ordering::Relaxed);
583                    }
584                    results.push(object_and_version);
585                }
586                Err(e) => {
587                    err_ctr.fetch_add(1, Ordering::Relaxed);
588                    return Err(e);
589                }
590            }
591        }
592        Ok(results)
593    }
594}

620impl ObjectStoreWriter for FileSystemStore {
621    fn write_object(
622        &self,
623        key: &ObjectKey,
624        object: Object,
625        actual_version: u64,
626    ) -> Result<(), Error> {
627        let object_dir = self.objects_dir()?.join(key.object_id.to_string());
628        let object_file = object_dir.join(actual_version.to_string());
629        self.write_bcs_file(&object_file, &object)?;
630        match &key.version_query {
631            VersionQuery::Version(_) => {}
632            VersionQuery::RootVersion(max_version) => {
633                self.update_root_mapping(&key.object_id, *max_version, actual_version)?;
634            }
635            VersionQuery::AtCheckpoint(checkpoint) => {
636                self.update_checkpoint_mapping(&key.object_id, *checkpoint, actual_version)?;
637            }
638        }
639        Ok(())
640    }

643impl StoreSummary for FileSystemStore {
644    fn summary<W: std::io::Write>(&self, w: &mut W) -> Result<()> {
645        let m = &self.metrics;
646        let txn_hit = m.txn_hit.load(Ordering::Relaxed);
647        let txn_miss = m.txn_miss.load(Ordering::Relaxed);
648        let txn_err = m.txn_error.load(Ordering::Relaxed);
649        let epoch_hit = m.epoch_hit.load(Ordering::Relaxed);
650        let epoch_miss = m.epoch_miss.load(Ordering::Relaxed);
651        let epoch_err = m.epoch_error.load(Ordering::Relaxed);
652        let proto_hit = m.proto_hit.load(Ordering::Relaxed);
653        let proto_miss = m.proto_miss.load(Ordering::Relaxed);
654        let proto_err = m.proto_error.load(Ordering::Relaxed);
655        let obj_v_hit = m.obj_version_hit.load(Ordering::Relaxed);
656        let obj_v_miss = m.obj_version_miss.load(Ordering::Relaxed);
657        let obj_v_err = m.obj_version_error.load(Ordering::Relaxed);
658        let obj_r_hit = m.obj_root_hit.load(Ordering::Relaxed);
659        let obj_r_miss = m.obj_root_miss.load(Ordering::Relaxed);
660        let obj_r_err = m.obj_root_error.load(Ordering::Relaxed);
661        let obj_c_hit = m.obj_checkpoint_hit.load(Ordering::Relaxed);
662        let obj_c_miss = m.obj_checkpoint_miss.load(Ordering::Relaxed);
663        let obj_c_err = m.obj_checkpoint_error.load(Ordering::Relaxed);
664
665        let total_hit = txn_hit + epoch_hit + proto_hit + obj_v_hit + obj_r_hit + obj_c_hit;
666        let total_miss = txn_miss + epoch_miss + proto_miss + obj_v_miss + obj_r_miss + obj_c_miss;
667        let total_err = txn_err + epoch_err + proto_err + obj_v_err + obj_r_err + obj_c_err;
668        let obj_total_hit = obj_v_hit + obj_r_hit + obj_c_hit;
669        let obj_total_miss = obj_v_miss + obj_r_miss + obj_c_miss;
670        let obj_total_err = obj_v_err + obj_r_err + obj_c_err;
671
672        writeln!(w, "FileSystemStore summary")?;
673        writeln!(
674            w,
675            "  Overall: hit={} miss={} error={}",
676            total_hit, total_miss, total_err
677        )?;
678        writeln!(
679            w,
680            "  Transaction: hit={} miss={} error={}",
681            txn_hit, txn_miss, txn_err
682        )?;
683        writeln!(
684            w,
685            "  Epoch:       hit={} miss={} error={}",
686            epoch_hit, epoch_miss, epoch_err
687        )?;
688        writeln!(
689            w,
690            "  Protocol:    hit={} miss={} error={}",
691            proto_hit, proto_miss, proto_err
692        )?;
693        writeln!(
694            w,
695            "  Objects (all): hit={} miss={} error={}",
696            obj_total_hit, obj_total_miss, obj_total_err
697        )?;
698        writeln!(w, "  Objects:")?;
699        writeln!(
700            w,
701            "    Version:     hit={} miss={} error={}",
702            obj_v_hit, obj_v_miss, obj_v_err
703        )?;
704        writeln!(
705            w,
706            "    RootVersion: hit={} miss={} error={}",
707            obj_r_hit, obj_r_miss, obj_r_err
708        )?;
709        writeln!(
710            w,
711            "    Checkpoint:  hit={} miss={} error={}",
712            obj_c_hit, obj_c_miss, obj_c_err
713        )?;
714        Ok(())
715    }

718impl SetupStore for FileSystemStore {
719    fn setup(&self, chain_id: Option<String>) -> Result<Option<String>, Error> {
720        if let Some(chain_id) = chain_id {
721            // Override the mapping of network -> chain_id in node_mapping.csv
722            self.write_chain_identifier(chain_id)?;
723        }
724        // If chain_id is None, do nothing
725        // Return the chain identifier from the mapping file if available
726        Ok(self.get_chain_id_for_node().ok())
727    }
728}
729
730impl FileSystemStore {
731    fn write_chain_identifier(&self, chain_id: String) -> Result<(), Error> {
732        let mapping_file = self.base_path.join(NODE_MAPPING_FILE);
733
734        // Create the base directory if it doesn't exist
735        if let Some(parent) = mapping_file.parent() {
736            fs::create_dir_all(parent)
737                .with_context(|| format!("Failed to create directory: {}", parent.display()))?;
738        }
739
740        // Read existing mappings to preserve them
741        let mut mappings = Vec::new();
742        if mapping_file.exists() {
743            let file = fs::File::open(&mapping_file).with_context(|| {
744                format!(
745                    "Failed to open existing mapping file: {}",
746                    mapping_file.display()
747                )
748            })?;
749
750            let mut rdr = csv::ReaderBuilder::new()
751                .has_headers(false)
752                .trim(csv::Trim::All)
753                .from_reader(file);
754
755            for result in rdr.records() {
756                let record = result.with_context(|| {
757                    format!("Failed to read CSV record from: {}", mapping_file.display())
758                })?;
759
760                if record.len() == 2 {
761                    mappings.push((record[0].trim().to_string(), record[1].trim().to_string()));
762                }
763            }
764        }
765
766        // Determine the node key for this store
767        let node_key = match &self.node {
768            Node::Mainnet => "mainnet",
769            Node::Testnet => "testnet",
770            Node::Custom(url) => url,
771        };
772
773        // Update or add the mapping for this node
774        let mut found = false;
775        for (node, _) in &mut mappings {
776            if node == node_key {
777                *node = node_key.to_string();
778                found = true;
779                break;
780            }
781        }
782
783        if !found {
784            mappings.push((node_key.to_string(), chain_id.clone()));
785        } else {
786            // Update the existing mapping
787            for (node, existing_chain_id) in &mut mappings {
788                if node == node_key {
789                    *existing_chain_id = chain_id.clone();
790                    break;
791                }
792            }
793        }
794
795        // Write the updated mappings back to the file
796        let file = fs::File::create(&mapping_file).with_context(|| {
797            format!("Failed to create mapping file: {}", mapping_file.display())
798        })?;
799
800        let mut wtr = csv::WriterBuilder::new()
801            .has_headers(false)
802            .from_writer(file);
803
804        for (node, chain_id) in mappings {
805            wtr.write_record([&node, &chain_id]).with_context(|| {
806                format!("Failed to write mapping record: {} -> {}", node, chain_id)
807            })?;
808        }
809
810        wtr.flush()
811            .with_context(|| "Failed to flush mapping file")?;
812
813        Ok(())
814    }
