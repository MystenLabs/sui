34/// GraphQL-backed implementation of the data store traits.
35pub struct DataStore {
36    client: reqwest::Client,
37    rpc: reqwest::Url,
38    node: Node,
39    // Keep the epoch data considering its small size and footprint
40    epoch_map: RwLock<BTreeMap<EpochId, EpochData>>,
41    /// The binary's version passed to the User-Agent header in GQL query requests
42    version: String,
43    /// Metrics for hit/miss accounting
44    metrics: DataStoreMetrics,
45}
46
47#[derive(Default)]
48struct DataStoreMetrics {
49    // transactions
50    txn_hit: AtomicU64,
51    txn_miss: AtomicU64,
52    txn_error: AtomicU64,
53    // epochs
54    epoch_hit: AtomicU64,
55    epoch_miss: AtomicU64,
56    epoch_error: AtomicU64,
57    // protocol config
58    proto_hit: AtomicU64,
59    proto_miss: AtomicU64,
60    proto_error: AtomicU64,
61    // objects by query kind
62    obj_version_hit: AtomicU64,
63    obj_version_miss: AtomicU64,
64    obj_version_error: AtomicU64,
65    obj_root_hit: AtomicU64,
66    obj_root_miss: AtomicU64,
67    obj_root_error: AtomicU64,
68    obj_checkpoint_hit: AtomicU64,
69    obj_checkpoint_miss: AtomicU64,
70    obj_checkpoint_error: AtomicU64,
71}
72
73macro_rules! block_on {
74    ($expr:expr) => {{
75        #[allow(clippy::disallowed_methods, clippy::result_large_err)]
76        {
77            if tokio::runtime::Handle::try_current().is_ok() {
78                // When already inside a Tokio runtime, spawn a scoped thread to
79                // run a separate current-thread runtime without requiring
80                // tokio::task::block_in_place (which may be unavailable).
81                std::thread::scope(|scope| {
82                    scope
83                        .spawn(|| {
84                            let rt = tokio::runtime::Builder::new_current_thread()
85                                .enable_all()
86                                .build()
87                                .expect("failed to build Tokio runtime");
88                            rt.block_on($expr)
89                        })
90                        .join()
91                        .expect("failed to join scoped thread running nested runtime")
92                })
93            } else {
94                let rt = tokio::runtime::Builder::new_current_thread()
95                    .enable_all()
96                    .build()
97                    .expect("failed to build Tokio runtime");
98                rt.block_on($expr)
99            }
100        }
101    }};
102}

104impl TransactionStore for DataStore {
105    fn transaction_data_and_effects(&self, digest: &str) -> Result<Option<TransactionInfo>, Error> {
106        match block_on!(self.transaction(digest)) {
107            Ok(Some((data, effects, checkpoint))) => {
108                self.metrics.txn_hit.fetch_add(1, Ordering::Relaxed);
109                Ok(Some(TransactionInfo {
110                    data,
111                    effects,
112                    checkpoint,
113                }))
114            }
115            Ok(None) => {
116                self.metrics.txn_miss.fetch_add(1, Ordering::Relaxed);
117                Ok(None)
118            }
119            Err(e) => {
120                self.metrics.txn_error.fetch_add(1, Ordering::Relaxed);
121                Err(e)
122            }
123        }
124    }
125}

127impl EpochStore for DataStore {
128    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error> {
129        if let Some(epoch_data) = self.epoch_map.read().unwrap().get(&epoch) {
130            self.metrics.epoch_hit.fetch_add(1, Ordering::Relaxed);
131            return Ok(Some(epoch_data.clone()));
132        }
133        match block_on!(self.epoch(epoch)) {
134            Ok(Some(epoch_data)) => {
135                self.epoch_map
136                    .write()
137                    .unwrap()
138                    .insert(epoch, epoch_data.clone());
139                self.metrics.epoch_hit.fetch_add(1, Ordering::Relaxed);
140                Ok(Some(epoch_data))
141            }
142            Ok(None) => {
143                self.metrics.epoch_miss.fetch_add(1, Ordering::Relaxed);
144                Ok(None)
145            }
146            Err(e) => {
147                self.metrics.epoch_error.fetch_add(1, Ordering::Relaxed);
148                Err(e)
149            }
150        }
151    }
152
153    // Get the protocol config for an epoch directly from the binary
154    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error> {
155        match self.epoch_info(epoch) {
156            Ok(Some(epoch)) => {
157                self.metrics.proto_hit.fetch_add(1, Ordering::Relaxed);
158                Ok(Some(ProtocolConfig::get_for_version(
159                    ProtocolVersion::new(epoch.protocol_version),
160                    self.chain(),
161                )))
162            }
163            Ok(None) => {
164                self.metrics.proto_miss.fetch_add(1, Ordering::Relaxed);
165                Ok(None)
166            }
167            Err(e) => {
168                self.metrics.proto_error.fetch_add(1, Ordering::Relaxed);
169                Err(e)
170            }
171        }
172    }
173}

175impl ObjectStore for DataStore {
176    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
177        match block_on!(self.objects(keys)) {
178            Ok(results) => {
179                assert_eq!(results.len(), keys.len());
180                for (key, res) in keys.iter().zip(results.iter()) {
181                    match key.version_query {
182                        VersionQuery::Version(_) => {
183                            if res.is_some() {
184                                self.metrics.obj_version_hit.fetch_add(1, Ordering::Relaxed);
185                            } else {
186                                self.metrics
187                                    .obj_version_miss
188                                    .fetch_add(1, Ordering::Relaxed);
189                            }
190                        }
191                        VersionQuery::RootVersion(_) => {
192                            if res.is_some() {
193                                self.metrics.obj_root_hit.fetch_add(1, Ordering::Relaxed);
194                            } else {
195                                self.metrics.obj_root_miss.fetch_add(1, Ordering::Relaxed);
196                            }
197                        }
198                        VersionQuery::AtCheckpoint(_) => {
199                            if res.is_some() {
200                                self.metrics
201                                    .obj_checkpoint_hit
202                                    .fetch_add(1, Ordering::Relaxed);
203                            } else {
204                                self.metrics
205                                    .obj_checkpoint_miss
206                                    .fetch_add(1, Ordering::Relaxed);
207                            }
208                        }
209                    }
210                }
211                Ok(results)
212            }
213            Err(e) => {
214                // Attribute the error to each requested key type
215                for key in keys.iter() {
216                    match key.version_query {
217                        VersionQuery::Version(_) => {
218                            self.metrics
219                                .obj_version_error
220                                .fetch_add(1, Ordering::Relaxed);
221                        }
222                        VersionQuery::RootVersion(_) => {
223                            self.metrics.obj_root_error.fetch_add(1, Ordering::Relaxed);
224                        }
225                        VersionQuery::AtCheckpoint(_) => {
226                            self.metrics
227                                .obj_checkpoint_error
228                                .fetch_add(1, Ordering::Relaxed);
229                        }
230                    }
231                }
232                Err(e)
233            }
234        }
235    }
236}

238impl SetupStore for DataStore {
239    fn setup(&self, _chain_id: Option<String>) -> Result<Option<String>, Error> {
240        // Return the chain identifier
241        Ok(Some(block_on!(gql_queries::chain_id_query::query(self))?))
242    }

245impl DataStore {
246    pub fn new(node: Node, version: &str) -> Result<Self, Error> {
247        debug!("Start stores creation");
248        let client = reqwest::Client::builder()
249            .connect_timeout(std::time::Duration::from_secs(3))
250            .timeout(std::time::Duration::from_secs(5))
251            .build()?;
252        let url = node.gql_url();
253        let rpc =
254            reqwest::Url::parse(url).context(format!("Failed to parse GQL RPC URL {}", url))?;
255        let epoch_map = RwLock::new(BTreeMap::new());
256        debug!("End stores creation");
257
258        Ok(Self {
259            client,
260            node,
261            epoch_map,
262            rpc,
263            version: version.to_string(),
264            metrics: DataStoreMetrics::default(),
265        })
266    }
267
268    pub fn node(&self) -> &Node {
269        &self.node
270    }
271    pub fn chain(&self) -> Chain {
272        self.node.chain()
273    }

275    pub(crate) async fn run_query<T, V>(
276        &self,
277        operation: &Operation<T, V>,
278    ) -> Result<GraphQlResponse<T>, Error>
279    where
280        T: serde::de::DeserializeOwned,
281        V: serde::Serialize,
282    {
283        Self::run_query_internal(&self.client, &self.rpc, &self.version, operation).await
284    }
285
286    async fn run_query_internal<T, V>(
287        client: &reqwest::Client,
288        rpc: &reqwest::Url,
289        version: &str,
290        operation: &Operation<T, V>,
291    ) -> Result<GraphQlResponse<T>, Error>
292    where
293        T: serde::de::DeserializeOwned,
294        V: serde::Serialize,
295    {
296        client
297            .post(rpc.clone())
298            .header(USER_AGENT, format!("sui-data-store-v{}", version))
299            .json(&operation)
300            .send()
301            .await
302            .context("Failed to send GQL query")?
303            .json::<GraphQlResponse<T>>()
304            .await
305            .context("Failed to read response in GQL query")
306    }

308    async fn transaction(
309        &self,
310        digest: &str,
311    ) -> Result<Option<(TransactionData, TransactionEffects, u64)>, Error> {
312        let _span = debug_span!("gql_txn_query", digest = %digest).entered();
313        debug!(op = "txn_query", phase = "start", "transaction query");
314        let t0 = Instant::now();
315        let data = gql_queries::txn_query::query(digest.to_string(), self).await;
316        let elapsed = t0.elapsed().as_millis();
317        debug!(
318            op = "txn_query",
319            phase = "end",
320            elapsed_ms = elapsed,
321            "transaction query"
322        );
323        data
324    }
325
326    async fn epoch(&self, epoch_id: u64) -> Result<Option<EpochData>, Error> {
327        let _span = debug_span!("gql_epoch_query", epoch = epoch_id).entered();
328        debug!(op = "epoch_query", phase = "start", "epoch query");
329        let t0 = Instant::now();
330        let data = gql_queries::epoch_query::query(epoch_id, self).await;
331        let elapsed = t0.elapsed().as_millis();
332        debug!(
333            op = "epoch_query",
334            phase = "end",
335            elapsed_ms = elapsed,
336            "epoch query"
337        );
338        data
339    }
340
341    async fn objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
342        let _span = debug_span!("gql_objects_query", num_keys = keys.len()).entered();
343        debug!(op = "objects_query", phase = "start", "objects query");
344        let t0 = Instant::now();
345        let data = gql_queries::object_query::query(keys, self).await?;
346        let elapsed = t0.elapsed().as_millis();
347        debug!(
348            op = "objects_query",
349            phase = "end",
350            elapsed_ms = elapsed,
351            "objects query"
352        );
353        Ok(data)
354    }

357impl StoreSummary for DataStore {
358    fn summary<W: std::io::Write>(&self, w: &mut W) -> Result<()> {
359        let m = &self.metrics;
360        let txn_hit = m.txn_hit.load(Ordering::Relaxed);
361        let txn_miss = m.txn_miss.load(Ordering::Relaxed);
362        let txn_err = m.txn_error.load(Ordering::Relaxed);
363        let epoch_hit = m.epoch_hit.load(Ordering::Relaxed);
364        let epoch_miss = m.epoch_miss.load(Ordering::Relaxed);
365        let epoch_err = m.epoch_error.load(Ordering::Relaxed);
366        let proto_hit = m.proto_hit.load(Ordering::Relaxed);
367        let proto_miss = m.proto_miss.load(Ordering::Relaxed);
368        let proto_err = m.proto_error.load(Ordering::Relaxed);
369        let obj_v_hit = m.obj_version_hit.load(Ordering::Relaxed);
370        let obj_v_miss = m.obj_version_miss.load(Ordering::Relaxed);
371        let obj_v_err = m.obj_version_error.load(Ordering::Relaxed);
372        let obj_r_hit = m.obj_root_hit.load(Ordering::Relaxed);
373        let obj_r_miss = m.obj_root_miss.load(Ordering::Relaxed);
374        let obj_r_err = m.obj_root_error.load(Ordering::Relaxed);
375        let obj_c_hit = m.obj_checkpoint_hit.load(Ordering::Relaxed);
376        let obj_c_miss = m.obj_checkpoint_miss.load(Ordering::Relaxed);
377        let obj_c_err = m.obj_checkpoint_error.load(Ordering::Relaxed);
378        let obj_total_hit = obj_v_hit + obj_r_hit + obj_c_hit;
379        let obj_total_miss = obj_v_miss + obj_r_miss + obj_c_miss;
380        let obj_total_err = obj_v_err + obj_r_err + obj_c_err;
381        let total_hit = txn_hit + epoch_hit + proto_hit + obj_total_hit;
382        let total_miss = txn_miss + epoch_miss + proto_miss + obj_total_miss;
383        let total_err = txn_err + epoch_err + proto_err + obj_total_err;
384
385        writeln!(w, "DataStore (remote GQL) summary")?;
386        writeln!(w, "  Node: {:?}", self.node())?;
387        writeln!(w, "  Cache sizes:")?;
388        writeln!(
389            w,
390            "    Epochs: {} entries",
391            self.epoch_map.read().unwrap().len()
392        )?;
393        writeln!(
394            w,
395            "  Overall:    hit={} miss={} error={}",
396            total_hit, total_miss, total_err
397        )?;
398        writeln!(w, "  Hits/Misses:")?;
399        writeln!(
400            w,
401            "    Transaction: hit={} miss={} error={}",
402            txn_hit, txn_miss, txn_err
403        )?;
404        writeln!(
405            w,
406            "    Epoch:       hit={} miss={} error={}",
407            epoch_hit, epoch_miss, epoch_err
408        )?;
409        writeln!(
410            w,
411            "    Protocol:    hit={} miss={} error={}",
412            proto_hit, proto_miss, proto_err
413        )?;
414        writeln!(
415            w,
416            "    Objects(all): hit={} miss={} error={}",
417            obj_total_hit, obj_total_miss, obj_total_err
418        )?;
419        writeln!(
420            w,
421            "      Version:     hit={} miss={}",
422            obj_v_hit, obj_v_miss
423        )?;
424        writeln!(
425            w,
426            "      RootVersion: hit={} miss={}",
427            obj_r_hit, obj_r_miss
428        )?;
429        writeln!(
430            w,
431            "      Checkpoint:  hit={} miss={}",
432            obj_c_hit, obj_c_miss
433        )?;
434        Ok(())
435    }
