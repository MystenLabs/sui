4//! In-memory implementation of the data store interfaces: `TransactionStore`, `EpochStore`, and `ObjectStore`.
5//! The `InMemoryStore` provides fast in-memory lookups.
6//!
7//! This store is purely cache-based - it only returns data that has been explicitly stored in memory.
8//! For `TransactionStore`, `EpochStore`, and `ObjectStore`, missing data results in `Ok(None)`.
9//!
10//! # Usage Examples
11//!
12//! ```ignore
13//! use crate::data_stores::InMemoryStore;
14//! use crate::Node;
15//!
16//! // Create an in-memory store
17//! let store = InMemoryStore::new(Node::Mainnet);
18//!
19//! // Attempting to fetch data not in cache will return None
20//! let result = store.epoch_info(123); // Returns Ok(None) since cache is empty
21//!
22//! // Data must be explicitly added to cache before it can be retrieved
23//! // (This would typically be done by other parts of the system)
24//! ```
25

46/// In-memory store implementing the data store interfaces.
47struct InMemoryStoreInner {
48    node: Node,
49    transaction_cache: BTreeMap<String, TransactionInfo>,
50    epoch_data_cache: BTreeMap<u64, EpochData>,
51    object_cache: BTreeMap<ObjectID, BTreeMap<u64, Object>>,
52    // The next 2 maps can be organized as flat maps or nested maps
53    // and the best solution depends on usage patterns.
54    // For now, we use flat maps but we may want to review in the future
55    /// Cache mapping (ObjectID, root_version) -> actual_version
56    /// Used for VersionQuery::RootVersion lookups
57    root_version_cache: BTreeMap<(ObjectID, u64), u64>,
58    /// Cache mapping (ObjectID, checkpoint) -> actual_version  
59    /// Used for VersionQuery::AtCheckpoint lookups
60    checkpoint_cache: BTreeMap<(ObjectID, u64), u64>,
61    /// Metrics: hit/miss counters for API calls
62    metrics: MemStoreMetrics,
63}
64
65// The RwLock is needed for 2 reasons:
66// 1. to make tokio happy
67// 2. to allow interior mutability for the cache
68pub struct InMemoryStore(RwLock<InMemoryStoreInner>);
69
70#[derive(Default)]
71struct MemStoreMetrics {
72    // transactions
73    txn_hit: AtomicU64,
74    txn_miss: AtomicU64,
75    txn_error: AtomicU64,
76    // epochs
77    epoch_hit: AtomicU64,
78    epoch_miss: AtomicU64,
79    epoch_error: AtomicU64,
80    // protocol config
81    proto_hit: AtomicU64,
82    proto_miss: AtomicU64,
83    proto_error: AtomicU64,
84    // objects by query kind
85    obj_version_hit: AtomicU64,
86    obj_version_miss: AtomicU64,
87    obj_version_error: AtomicU64,
88    obj_root_hit: AtomicU64,
89    obj_root_miss: AtomicU64,
90    obj_root_error: AtomicU64,
91    obj_checkpoint_hit: AtomicU64,
92    obj_checkpoint_miss: AtomicU64,
93    obj_checkpoint_error: AtomicU64,
94}

96impl InMemoryStore {
97    /// Create a new InMemoryStore with the given node
98    pub fn new(node: Node) -> Self {
99        Self(RwLock::new(InMemoryStoreInner {
100            node,
101            transaction_cache: BTreeMap::new(),
102            epoch_data_cache: BTreeMap::new(),
103            object_cache: BTreeMap::new(),
104            root_version_cache: BTreeMap::new(),
105            checkpoint_cache: BTreeMap::new(),
106            metrics: MemStoreMetrics::default(),
107        }))
108    }
109
110    /// Get the chain for this store
111    pub fn chain(&self) -> Chain {
112        self.0.read().unwrap().node.chain()
113    }
114
115    /// Get the node for this store
116    pub fn node(&self) -> Node {
117        self.0.read().unwrap().node.clone()
118    }
119
120    /// Clear all caches
121    pub fn clear_all_caches(&self) {
122        let mut inner = self.0.write().unwrap();
123        inner.transaction_cache.clear();
124        inner.epoch_data_cache.clear();
125        inner.object_cache.clear();
126        inner.root_version_cache.clear();
127        inner.checkpoint_cache.clear();
128    }
129
130    /// Get cache statistics
131    pub fn cache_stats(&self) -> CacheStats {
132        let inner = self.0.read().unwrap();
133        let object_cache_size = inner
134            .object_cache
135            .values()
136            .map(|versions| versions.len())
137            .sum();
138
139        let root_version_cache_size = inner.root_version_cache.len();
140
141        let checkpoint_cache_size = inner.checkpoint_cache.len();
142
143        CacheStats {
144            transaction_cache_size: inner.transaction_cache.len(),
145            epoch_data_cache_size: inner.epoch_data_cache.len(),
146            object_cache_size,
147            root_version_cache_size,
148            checkpoint_cache_size,
149        }
150    }
151
152    /// Add transaction data to the cache
153    pub fn add_transaction_data(&self, tx_digest: String, transaction_info: TransactionInfo) {
154        self.0
155            .write()
156            .unwrap()
157            .transaction_cache
158            .insert(tx_digest, transaction_info);
159    }
160
161    /// Add epoch data to the cache
162    pub fn add_epoch_data(&self, epoch: u64, epoch_data: EpochData) {
163        self.0
164            .write()
165            .unwrap()
166            .epoch_data_cache
167            .insert(epoch, epoch_data);
168    }
169
170    /// Add object data to the cache
171    pub fn add_object_data(&self, object_id: ObjectID, version: u64, object: Object) {
172        self.0
173            .write()
174            .unwrap()
175            .object_cache
176            .entry(object_id)
177            .or_default()
178            .insert(version, object);
179    }
180}

182/// Statistics about cache usage
183#[derive(Debug, Clone)]
184pub struct CacheStats {
185    pub transaction_cache_size: usize,
186    pub epoch_data_cache_size: usize,
187    pub object_cache_size: usize,
188    pub root_version_cache_size: usize,
189    pub checkpoint_cache_size: usize,

192impl TransactionStore for InMemoryStore {
193    fn transaction_data_and_effects(
194        &self,
195        tx_digest: &str,
196    ) -> Result<Option<TransactionInfo>, Error> {
197        let inner = self.0.read().unwrap();
198        if let Some(cached) = inner.transaction_cache.get(tx_digest) {
199            inner.metrics.txn_hit.fetch_add(1, Ordering::Relaxed);
200            return Ok(Some(cached.clone()));
201        }
202        inner.metrics.txn_miss.fetch_add(1, Ordering::Relaxed);
203        Ok(None)
204    }
205}
206
207impl EpochStore for InMemoryStore {
208    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error> {
209        let inner = self.0.read().unwrap();
210        if let Some(cached) = inner.epoch_data_cache.get(&epoch) {
211            inner.metrics.epoch_hit.fetch_add(1, Ordering::Relaxed);
212            return Ok(Some(cached.clone()));
213        }
214        inner.metrics.epoch_miss.fetch_add(1, Ordering::Relaxed);
215        Ok(None)
216    }
217
218    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error> {
219        match self.epoch_info(epoch) {
220            Ok(Some(epoch_data)) => {
221                let inner = self.0.read().unwrap();
222                inner.metrics.proto_hit.fetch_add(1, Ordering::Relaxed);
223                Ok(Some(ProtocolConfig::get_for_version(
224                    ProtocolVersion::new(epoch_data.protocol_version),
225                    self.chain(),
226                )))
227            }
228            Ok(None) => {
229                let inner = self.0.read().unwrap();
230                inner.metrics.proto_miss.fetch_add(1, Ordering::Relaxed);
231                Ok(None)
232            }
233            Err(e) => {
234                let inner = self.0.read().unwrap();
235                inner.metrics.proto_error.fetch_add(1, Ordering::Relaxed);
236                Err(e)
237            }
238        }
239    }
240}

242impl ObjectStore for InMemoryStore {
243    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
244        let mut results = Vec::with_capacity(keys.len());
245        let inner = self.0.read().unwrap();
246
247        // Check caches for each key
248        for key in keys.iter() {
249            let (object_and_version, hit_ctr, miss_ctr) = match &key.version_query {
250                VersionQuery::Version(version) => {
251                    let res = if let Some(versions_map) = inner.object_cache.get(&key.object_id) {
252                        versions_map
253                            .get(version)
254                            .cloned()
255                            .map(|obj| (obj, *version))
256                    } else {
257                        None
258                    };
259                    (
260                        res,
261                        &inner.metrics.obj_version_hit,
262                        &inner.metrics.obj_version_miss,
263                    )
264                }
265                VersionQuery::RootVersion(max_version) => {
266                    let actual_version = inner
267                        .root_version_cache
268                        .get(&(key.object_id, *max_version))
269                        .copied();
270                    let res = actual_version.and_then(|actual_version| {
271                        inner
272                            .object_cache
273                            .get(&key.object_id)
274                            .and_then(|versions_map| versions_map.get(&actual_version))
275                            .cloned()
276                            .map(|obj| (obj, actual_version))
277                    });
278                    (
279                        res,
280                        &inner.metrics.obj_root_hit,
281                        &inner.metrics.obj_root_miss,
282                    )
283                }
284                VersionQuery::AtCheckpoint(checkpoint) => {
285                    let actual_version = inner
286                        .checkpoint_cache
287                        .get(&(key.object_id, *checkpoint))
288                        .copied();
289                    let res = actual_version.and_then(|actual_version| {
290                        inner
291                            .object_cache
292                            .get(&key.object_id)
293                            .and_then(|versions_map| versions_map.get(&actual_version))
294                            .cloned()
295                            .map(|obj| (obj, actual_version))
296                    });
297                    (
298                        res,
299                        &inner.metrics.obj_checkpoint_hit,
300                        &inner.metrics.obj_checkpoint_miss,
301                    )
302                }
303            };
304            if object_and_version.is_some() {
305                hit_ctr.fetch_add(1, Ordering::Relaxed);
306            } else {
307                miss_ctr.fetch_add(1, Ordering::Relaxed);
308            }
309            results.push(object_and_version);
310        }
311
312        Ok(results)
313    }
314}

342impl ObjectStoreWriter for InMemoryStore {
343    fn write_object(
344        &self,
345        key: &ObjectKey,
346        object: Object,
347        actual_version: u64,
348    ) -> Result<(), Error> {
349        let mut inner = self.0.write().unwrap();
350
351        // Always store the object at its actual version
352        inner
353            .object_cache
354            .entry(key.object_id)
355            .or_default()
356            .insert(actual_version, object);
357
358        // Handle version mappings based on query type
359        match &key.version_query {
360            VersionQuery::Version(_) => {
361                // No additional mapping needed for direct version queries
362            }
363            VersionQuery::RootVersion(max_version) => {
364                inner
365                    .root_version_cache
366                    .insert((key.object_id, *max_version), actual_version);
367            }
368            VersionQuery::AtCheckpoint(checkpoint) => {
369                inner
370                    .checkpoint_cache
371                    .insert((key.object_id, *checkpoint), actual_version);
372            }
373        }
374
375        Ok(())
376    }
377}

385impl StoreSummary for InMemoryStore {
386    fn summary<W: std::io::Write>(&self, w: &mut W) -> Result<()> {
387        let stats = self.cache_stats();
388        let inner = self.0.read().unwrap();
389        let m = &inner.metrics;
390        let txn_hit = m.txn_hit.load(Ordering::Relaxed);
391        let txn_miss = m.txn_miss.load(Ordering::Relaxed);
392        let txn_err = m.txn_error.load(Ordering::Relaxed);
393        let epoch_hit = m.epoch_hit.load(Ordering::Relaxed);
394        let epoch_miss = m.epoch_miss.load(Ordering::Relaxed);
395        let epoch_err = m.epoch_error.load(Ordering::Relaxed);
396        let proto_hit = m.proto_hit.load(Ordering::Relaxed);
397        let proto_miss = m.proto_miss.load(Ordering::Relaxed);
398        let proto_err = m.proto_error.load(Ordering::Relaxed);
399        let obj_v_hit = m.obj_version_hit.load(Ordering::Relaxed);
400        let obj_v_miss = m.obj_version_miss.load(Ordering::Relaxed);
401        let obj_v_err = m.obj_version_error.load(Ordering::Relaxed);
402        let obj_r_hit = m.obj_root_hit.load(Ordering::Relaxed);
403        let obj_r_miss = m.obj_root_miss.load(Ordering::Relaxed);
404        let obj_r_err = m.obj_root_error.load(Ordering::Relaxed);
405        let obj_c_hit = m.obj_checkpoint_hit.load(Ordering::Relaxed);
406        let obj_c_miss = m.obj_checkpoint_miss.load(Ordering::Relaxed);
407        let obj_c_err = m.obj_checkpoint_error.load(Ordering::Relaxed);
408        let obj_total_hit = obj_v_hit + obj_r_hit + obj_c_hit;
409        let obj_total_miss = obj_v_miss + obj_r_miss + obj_c_miss;
410        let obj_total_err = obj_v_err + obj_r_err + obj_c_err;
411        let total_hit = txn_hit + epoch_hit + proto_hit + obj_total_hit;
412        let total_miss = txn_miss + epoch_miss + proto_miss + obj_total_miss;
413        let total_err = txn_err + epoch_err + proto_err + obj_total_err;
414
415        writeln!(w, "InMemoryStore summary")?;
416        writeln!(w, "  Node: {:?}", self.node())?;
417        writeln!(w, "  Cache sizes:")?;
418        writeln!(
419            w,
420            "    Transactions: {} entries",
421            stats.transaction_cache_size
422        )?;
423        writeln!(w, "    Epochs: {} entries", stats.epoch_data_cache_size)?;
424        writeln!(w, "    Objects: {} versions", stats.object_cache_size)?;
425        writeln!(w, "    Root map: {} entries", stats.root_version_cache_size)?;
426        writeln!(
427            w,
428            "    Checkpoint map: {} entries",
429            stats.checkpoint_cache_size
430        )?;
431        writeln!(
432            w,
433            "  Overall:    hit={} miss={} error={}",
434            total_hit, total_miss, total_err
435        )?;
436        writeln!(w, "  Hits/Misses:")?;
437        writeln!(
438            w,
439            "    Transaction: hit={} miss={} error={}",
440            txn_hit, txn_miss, txn_err
441        )?;
442        writeln!(
443            w,
444            "    Epoch:       hit={} miss={} error={}",
445            epoch_hit, epoch_miss, epoch_err
446        )?;
447        writeln!(
448            w,
449            "    Protocol:    hit={} miss={} error={}",
450            proto_hit, proto_miss, proto_err
451        )?;
452        writeln!(
453            w,
454            "    Objects(all): hit={} miss={} error={}",
455            obj_total_hit, obj_total_miss, obj_total_err
456        )?;
457        writeln!(
458            w,
459            "      Version:     hit={} miss={} error={}",
460            obj_v_hit, obj_v_miss, obj_v_err
461        )?;
462        writeln!(
463            w,
464            "      RootVersion: hit={} miss={} error={}",
465            obj_r_hit, obj_r_miss, obj_r_err
466        )?;
467        writeln!(
468            w,
469            "      Checkpoint:  hit={} miss={} error={}",
470            obj_c_hit, obj_c_miss, obj_c_err
471        )?;
472        Ok(())
473    }
