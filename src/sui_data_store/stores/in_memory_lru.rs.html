4//! LRU in-memory implementation of the data store interfaces: `TransactionStore`, `EpochStore`,
5//! and `ObjectStore`.
6//!
7//! Very similar to `InMemoryStore`, but entries are kept in bounded `LruCache`s.
8//! This is useful to avoid unbounded memory growth under heavy load.
9
10use crate::{
11    EpochData, EpochStore, EpochStoreWriter, ObjectKey, ObjectStore, ObjectStoreWriter, SetupStore,
12    StoreSummary, TransactionInfo, TransactionStore, TransactionStoreWriter, VersionQuery,
13    node::Node,
14};
15use anyhow::{Error, Result};
16use lru::LruCache;
17use std::{
18    num::NonZeroUsize,
19    sync::{
20        RwLock,
21        atomic::{AtomicU64, Ordering},
22    },
23};
24use sui_types::{
25    base_types::ObjectID,
26    committee::ProtocolVersion,
27    object::Object,

31/// Default cache capacities
32const DEFAULT_TXN_CAP: usize = 10_000;
33const DEFAULT_EPOCH_CAP: usize = 10_000;
34const DEFAULT_OBJECT_CAP: usize = 50_000;
35const DEFAULT_ROOT_MAP_CAP: usize = 50_000;
36const DEFAULT_CHECKPOINT_MAP_CAP: usize = 50_000;
37
38/// LRU-backed store implementing the data store interfaces.
39struct LruMemoryStoreInner {
40    node: Node,
41    transaction_cache: LruCache<String, TransactionInfo>,
42    epoch_data_cache: LruCache<u64, EpochData>,
43    /// Objects are cached by (ObjectID, actual_version)
44    object_cache: LruCache<(ObjectID, u64), Object>,
45    // The next 2 maps can be organized as flat maps or nested maps
46    // and the best solution depends on usage patterns.
47    // For now, we use flat maps but we may want to review in the future
48    /// Cache mapping (ObjectID, root_version) -> actual_version
49    root_version_cache: LruCache<(ObjectID, u64), u64>,
50    /// Cache mapping (ObjectID, checkpoint) -> actual_version
51    checkpoint_cache: LruCache<(ObjectID, u64), u64>,
52    metrics: LruStoreMetrics,
53}
54
55// The RwLock is needed for 2 reasons:
56// 1. to make tokio happy
57// 2. to allow interior mutability for the cache
58pub struct LruMemoryStore(RwLock<LruMemoryStoreInner>);
59
60#[derive(Default)]
61struct LruStoreMetrics {
62    // transactions
63    txn_hit: AtomicU64,
64    txn_miss: AtomicU64,
65    txn_error: AtomicU64,
66    // epochs
67    epoch_hit: AtomicU64,
68    epoch_miss: AtomicU64,
69    epoch_error: AtomicU64,
70    // protocol config
71    proto_hit: AtomicU64,
72    proto_miss: AtomicU64,
73    proto_error: AtomicU64,
74    // objects by query kind
75    obj_version_hit: AtomicU64,
76    obj_version_miss: AtomicU64,
77    obj_version_error: AtomicU64,
78    obj_root_hit: AtomicU64,
79    obj_root_miss: AtomicU64,
80    obj_root_error: AtomicU64,
81    obj_checkpoint_hit: AtomicU64,
82    obj_checkpoint_miss: AtomicU64,
83    obj_checkpoint_error: AtomicU64,
84}
85
86impl LruMemoryStore {
87    /// Create a new LRU store with default capacities
88    pub fn new(node: Node) -> Self {
89        Self::with_capacities(
90            node,
91            DEFAULT_TXN_CAP,
92            DEFAULT_EPOCH_CAP,
93            DEFAULT_OBJECT_CAP,
94            DEFAULT_ROOT_MAP_CAP,
95            DEFAULT_CHECKPOINT_MAP_CAP,
96        )
97    }
98
99    /// Create a new LRU store with custom capacities
100    pub fn with_capacities(
101        node: Node,
102        txn_cap: usize,
103        epoch_cap: usize,
104        object_cap: usize,
105        root_map_cap: usize,
106        checkpoint_map_cap: usize,
107    ) -> Self {
108        let nz = |n: usize| NonZeroUsize::new(n.max(1)).unwrap();
109        Self(RwLock::new(LruMemoryStoreInner {
110            node,
111            transaction_cache: LruCache::new(nz(txn_cap)),
112            epoch_data_cache: LruCache::new(nz(epoch_cap)),
113            object_cache: LruCache::new(nz(object_cap)),
114            root_version_cache: LruCache::new(nz(root_map_cap)),
115            checkpoint_cache: LruCache::new(nz(checkpoint_map_cap)),
116            metrics: LruStoreMetrics::default(),
117        }))
118    }
119
120    pub fn chain(&self) -> Chain {
121        self.0.read().unwrap().node.chain()
122    }
123
124    pub fn node(&self) -> Node {
125        self.0.read().unwrap().node.clone()
126    }
127}

129impl TransactionStore for LruMemoryStore {
130    fn transaction_data_and_effects(
131        &self,
132        tx_digest: &str,
133    ) -> Result<Option<TransactionInfo>, Error> {
134        let mut inner = self.0.write().unwrap();
135        let cached = inner.transaction_cache.get(tx_digest).cloned();
136        if cached.is_some() {
137            inner.metrics.txn_hit.fetch_add(1, Ordering::Relaxed);
138        } else {
139            inner.metrics.txn_miss.fetch_add(1, Ordering::Relaxed);
140        }
141        Ok(cached)
142    }
143}
144
145impl EpochStore for LruMemoryStore {
146    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error> {
147        let mut inner = self.0.write().unwrap();
148        let cached = inner.epoch_data_cache.get(&epoch).cloned();
149        if cached.is_some() {
150            inner.metrics.epoch_hit.fetch_add(1, Ordering::Relaxed);
151        } else {
152            inner.metrics.epoch_miss.fetch_add(1, Ordering::Relaxed);
153        }
154        Ok(cached)
155    }
156
157    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error> {
158        match self.epoch_info(epoch) {
159            Ok(Some(epoch_data)) => {
160                let inner = self.0.read().unwrap();
161                inner.metrics.proto_hit.fetch_add(1, Ordering::Relaxed);
162                Ok(Some(ProtocolConfig::get_for_version(
163                    ProtocolVersion::new(epoch_data.protocol_version),
164                    self.chain(),
165                )))
166            }
167            Ok(None) => {
168                let inner = self.0.read().unwrap();
169                inner.metrics.proto_miss.fetch_add(1, Ordering::Relaxed);
170                Ok(None)
171            }
172            Err(e) => {
173                let inner = self.0.read().unwrap();
174                inner.metrics.proto_error.fetch_add(1, Ordering::Relaxed);
175                Err(e)
176            }
177        }
178    }
179}

181impl ObjectStore for LruMemoryStore {
182    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
183        let mut results = Vec::with_capacity(keys.len());
184        let mut inner = self.0.write().unwrap();
185
186        for key in keys.iter() {
187            let (object_and_version, hit_ctr, miss_ctr) = match &key.version_query {
188                VersionQuery::Version(version) => {
189                    let res = inner
190                        .object_cache
191                        .get(&(key.object_id, *version))
192                        .cloned()
193                        .map(|obj| (obj, *version));
194                    (
195                        res,
196                        &inner.metrics.obj_version_hit,
197                        &inner.metrics.obj_version_miss,
198                    )
199                }
200                VersionQuery::RootVersion(max_version) => {
201                    let actual_version = inner
202                        .root_version_cache
203                        .get(&(key.object_id, *max_version))
204                        .copied();
205                    let res = actual_version.and_then(|v| {
206                        inner
207                            .object_cache
208                            .get(&(key.object_id, v))
209                            .cloned()
210                            .map(|obj| (obj, v))
211                    });
212                    (
213                        res,
214                        &inner.metrics.obj_root_hit,
215                        &inner.metrics.obj_root_miss,
216                    )
217                }
218                VersionQuery::AtCheckpoint(checkpoint) => {
219                    let actual_version = inner
220                        .checkpoint_cache
221                        .get(&(key.object_id, *checkpoint))
222                        .copied();
223                    let res = actual_version.and_then(|v| {
224                        inner
225                            .object_cache
226                            .get(&(key.object_id, v))
227                            .cloned()
228                            .map(|obj| (obj, v))
229                    });
230                    (
231                        res,
232                        &inner.metrics.obj_checkpoint_hit,
233                        &inner.metrics.obj_checkpoint_miss,
234                    )
235                }
236            };
237            if object_and_version.is_some() {
238                hit_ctr.fetch_add(1, Ordering::Relaxed);
239            } else {
240                miss_ctr.fetch_add(1, Ordering::Relaxed);
241            }
242            results.push(object_and_version);
243        }
244        Ok(results)
245    }
246}

248impl TransactionStoreWriter for LruMemoryStore {
249    fn write_transaction(
250        &self,
251        tx_digest: &str,
252        transaction_info: TransactionInfo,
253    ) -> Result<(), Error> {
254        self.0
255            .write()
256            .unwrap()
257            .transaction_cache
258            .put(tx_digest.to_string(), transaction_info);
259        Ok(())
260    }
261}

263impl EpochStoreWriter for LruMemoryStore {
264    fn write_epoch_info(&self, epoch: u64, epoch_data: EpochData) -> Result<(), Error> {
265        self.0
266            .write()
267            .unwrap()
268            .epoch_data_cache
269            .put(epoch, epoch_data);
270        Ok(())
271    }

274impl ObjectStoreWriter for LruMemoryStore {
275    fn write_object(
276        &self,
277        key: &ObjectKey,
278        object: Object,
279        actual_version: u64,
280    ) -> Result<(), Error> {
281        let mut inner = self.0.write().unwrap();
282
283        // Always store the object at its actual version
284        inner
285            .object_cache
286            .put((key.object_id, actual_version), object);
287
288        // Handle version mappings based on query type
289        match &key.version_query {
290            VersionQuery::Version(_) => {}
291            VersionQuery::RootVersion(max_version) => {
292                inner
293                    .root_version_cache
294                    .put((key.object_id, *max_version), actual_version);
295            }
296            VersionQuery::AtCheckpoint(checkpoint) => {
297                inner
298                    .checkpoint_cache
299                    .put((key.object_id, *checkpoint), actual_version);
300            }
301        }
302
303        Ok(())
304    }
305}

313impl StoreSummary for LruMemoryStore {
314    fn summary<W: std::io::Write>(&self, w: &mut W) -> Result<()> {
315        let inner = self.0.read().unwrap();
316        let m = &inner.metrics;
317        let txn_hit = m.txn_hit.load(Ordering::Relaxed);
318        let txn_miss = m.txn_miss.load(Ordering::Relaxed);
319        let txn_err = m.txn_error.load(Ordering::Relaxed);
320        let epoch_hit = m.epoch_hit.load(Ordering::Relaxed);
321        let epoch_miss = m.epoch_miss.load(Ordering::Relaxed);
322        let epoch_err = m.epoch_error.load(Ordering::Relaxed);
323        let proto_hit = m.proto_hit.load(Ordering::Relaxed);
324        let proto_miss = m.proto_miss.load(Ordering::Relaxed);
325        let proto_err = m.proto_error.load(Ordering::Relaxed);
326        let obj_v_hit = m.obj_version_hit.load(Ordering::Relaxed);
327        let obj_v_miss = m.obj_version_miss.load(Ordering::Relaxed);
328        let obj_v_err = m.obj_version_error.load(Ordering::Relaxed);
329        let obj_r_hit = m.obj_root_hit.load(Ordering::Relaxed);
330        let obj_r_miss = m.obj_root_miss.load(Ordering::Relaxed);
331        let obj_r_err = m.obj_root_error.load(Ordering::Relaxed);
332        let obj_c_hit = m.obj_checkpoint_hit.load(Ordering::Relaxed);
333        let obj_c_miss = m.obj_checkpoint_miss.load(Ordering::Relaxed);
334        let obj_c_err = m.obj_checkpoint_error.load(Ordering::Relaxed);
335        let obj_total_hit = obj_v_hit + obj_r_hit + obj_c_hit;
336        let obj_total_miss = obj_v_miss + obj_r_miss + obj_c_miss;
337        let obj_total_err = obj_v_err + obj_r_err + obj_c_err;
338        let total_hit = txn_hit + epoch_hit + proto_hit + obj_total_hit;
339        let total_miss = txn_miss + epoch_miss + proto_miss + obj_total_miss;
340        let total_err = txn_err + epoch_err + proto_err + obj_total_err;
341
342        writeln!(w, "LruMemoryStore summary")?;
343        writeln!(w, "  Node: {:?}", self.node())?;
344        writeln!(
345            w,
346            "  Overall:    hit={} miss={} error={}",
347            total_hit, total_miss, total_err
348        )?;
349        writeln!(
350            w,
351            "    Transaction: hit={} miss={} error={}",
352            txn_hit, txn_miss, txn_err
353        )?;
354        writeln!(
355            w,
356            "    Epoch:       hit={} miss={} error={}",
357            epoch_hit, epoch_miss, epoch_err
358        )?;
359        writeln!(
360            w,
361            "    Protocol:    hit={} miss={} error={}",
362            proto_hit, proto_miss, proto_err
363        )?;
364        writeln!(
365            w,
366            "    Objects(all): hit={} miss={} error={}",
367            obj_total_hit, obj_total_miss, obj_total_err
368        )?;
369        writeln!(
370            w,
371            "      Version:     hit={} miss={} error={}",
372            obj_v_hit, obj_v_miss, obj_v_err
373        )?;
374        writeln!(
375            w,
376            "      RootVersion: hit={} miss={} error={}",
377            obj_r_hit, obj_r_miss, obj_r_err
378        )?;
379        writeln!(
380            w,
381            "      Checkpoint:  hit={} miss={} error={}",
382            obj_c_hit, obj_c_miss, obj_c_err
383        )?;
384        Ok(())
385    }
