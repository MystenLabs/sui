4use std::fmt;

6/// Lexer for Display V1 format strings. Format strings are a mix of text and expressions.
7/// Expressions are enclosed in curly braces and may contain identifiers separated by dots (a path
8/// of field accesses).
9#[derive(Debug)]
10pub(crate) struct Lexer<'s> {
11    /// Remaining input to be tokenized.
12    src: &'s str,
13
14    /// The number of bytes tokenized so far.
15    off: usize,
16
17    /// Whether the lexer is currently inside a text strand or an expression strand.
18    mode: Mode,
19}
20
21#[derive(Debug)]
22enum Mode {
23    Text,
24    Expr,
25}
26
27/// A lexeme is a token along with its offset in the source string, and the slice of source string
28/// that it originated from.
29#[derive(Copy, Clone, Debug, PartialEq, Eq)]
30pub(crate) struct Lexeme<'s>(pub Token, pub usize, pub &'s str);
31
32/// Like [Lexeme] but owns the slice of source string. Useful for capturing context in an error
33/// message.
34#[derive(Debug)]
35pub(crate) struct OwnedLexeme(Token, usize, String);
36
37#[derive(Copy, Clone, Debug, PartialEq, Eq)]
38pub(crate) enum Token {
39    /// '.'
40    Dot,
41    /// '\X' where X is any byte.
42    Escaped,
43    /// A potential field identifier
44    Ident,
45    /// '{'
46    LCurl,
47    /// '}'
48    RCurl,
49    /// A strand of text.
50    Text,
51}
52
53#[derive(Debug)]
54pub(crate) struct TokenSet<'t>(pub &'t [Token]);
55
56impl<'s> Lexer<'s> {
57    pub(crate) fn new(src: &'s str) -> Self {
58        Self {
59            src,
60            off: 0,
61            mode: Mode::Text,
62        }
63    }
64
65    /// Assuming the lexer is in text mode, return the next text token.
66    fn next_text_token(&mut self) -> Option<Lexeme<'s>> {
67        let bytes = self.src.as_bytes();
68
69        use Token as T;
70        Some(match bytes.first()? {
71            b'\\' if bytes.len() > 1 => {
72                self.take(T::Escaped, 1); // discard the backslash
73                self.take(T::Escaped, 1)
74            }
75            b'\\' => self.take(T::Text, 1),
76            b'{' => {
77                self.mode = Mode::Expr;
78                self.take(T::LCurl, 1)
79            }
80            // This is not a valid token within text, but recognise it so that the parser can
81            // produce a better error message.
82            b'}' => self.take(T::RCurl, 1),
83            _ => self.take_until(T::Text, |c| ['\\', '{', '}'].contains(&c)),
84        })
85    }
86
87    /// Assuming the lexer is in expression mode, return the next expression token.
88    fn next_expr_token(&mut self) -> Option<Lexeme<'s>> {
89        self.skip_whitespace();
90
91        use Token as T;
92        Some(match self.src.as_bytes().first()? {
93            // { is not a valid token within an expression, but recognise it so that the parser can
94            // produce a better error message.
95            b'{' => self.take(T::LCurl, 1),
96            b'}' => {
97                self.mode = Mode::Text;
98                self.take(T::RCurl, 1)
99            }
100            b'.' => self.take(T::Dot, 1),
101            // The lexer takes a very liberal definition of "identifier", the parser will check
102            // whether the identifier is actually valid.
103            _ => self.take_until(T::Ident, |c| {
104                c.is_whitespace() || c == '.' || c == '{' || c == '}'
105            }),
106        })
107    }
108
109    fn skip_whitespace(&mut self) {
110        self.take_until(Token::Text, |c: char| !c.is_whitespace());
111    }
112
113    /// Take a prefix of bytes from `self.src` until a byte satisfying pattern `p` is found, and
114    /// return it as a lexeme of type `t`. If no such byte is found, take the entire remainder of
115    /// the source string.
116    fn take_until(&mut self, t: Token, p: impl FnMut(char) -> bool) -> Lexeme<'s> {
117        self.take(t, self.src.find(p).unwrap_or(self.src.len()))
118    }
119
120    /// Take `n` bytes from the beginning of `self.src` and return them as a lexeme of type `t`.
121    ///
122    /// ## Safety
123    ///
124    /// This function assumes that there are at least `n` bytes left in `self.src`, and will panic
125    /// if that is not the case.
126    fn take(&mut self, t: Token, n: usize) -> Lexeme<'s> {
127        let start = self.off;
128        let slice = &self.src[..n];
129        self.src = &self.src[n..];
130        self.off += n;
131
132        Lexeme(t, start, slice)
133    }
134}

136impl Lexeme<'_> {
137    /// Return the lexeme as an owned lexeme, with the slice of source string copied.
138    pub(crate) fn detach(&self) -> OwnedLexeme {
139        OwnedLexeme(self.0, self.1, self.2.to_owned())
140    }
141}
142
143impl<'s> Iterator for Lexer<'s> {
144    type Item = Lexeme<'s>;
145
146    fn next(&mut self) -> Option<Self::Item> {
147        use Mode as M;
148        match self.mode {
149            M::Text => self.next_text_token(),
150            M::Expr => self.next_expr_token(),
151        }
152    }
153}

155impl fmt::Display for OwnedLexeme {
156    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
157        use OwnedLexeme as L;
158        use Token as T;
159        match self {
160            L(T::Text, _, s) => write!(f, "text {s:?}"),
161            L(T::Escaped, _, s) => write!(f, "escaped character '\\{s}'"),
162            L(T::LCurl, _, _) => write!(f, "'{{'"),
163            L(T::RCurl, _, _) => write!(f, "'}}'"),
164            L(T::Ident, _, s) => write!(f, "identifier {s:?}"),
165            L(T::Dot, _, _) => write!(f, "'.'"),
166        }?;
167
168        write!(f, " at offset {}", self.1)
169    }
170}
171
172impl fmt::Display for Token {
173    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
174        use Token as T;
175        match self {
176            T::Text => write!(f, "text"),
177            T::Escaped => write!(f, "an escaped character"),
178            T::LCurl => write!(f, "'{{'"),
179            T::RCurl => write!(f, "'}}'"),
180            T::Ident => write!(f, "an identifier"),
181            T::Dot => write!(f, "'.'"),
182        }
183    }
184}
185
186impl fmt::Display for TokenSet<'_> {
187    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
188        let TokenSet(tokens) = self;
189
190        if tokens.is_empty() {
191            return write!(f, "nothing");
192        }
193
194        let (head, [tail]) = tokens.split_at(tokens.len() - 1) else {
195            unreachable!("tail contains exactly one token");
196        };
197
198        if head.is_empty() {
199            return write!(f, "{tail}");
200        }
201
202        let mut prefix = "one of ";
203        for token in head {
204            write!(f, "{prefix}{token}")?;
205            prefix = ", ";
206        }
207
208        write!(f, ", or {tail}")?;
209        Ok(())
210    }
211}

219    /// Simple test for a raw literal string.
220    #[test]
221    fn test_all_text() {
222        let lexer = Lexer::new("foo bar");
223        let lexemes: Vec<_> = lexer.collect();
224        assert_eq!(lexemes, vec![L(T::Text, 0, "foo bar")]);
225    }
226
227    /// Escape sequences are all text, but they will be split into multiple tokens.
228    #[test]
229    fn test_escapes() {
230        let lexer = Lexer::new(r#"foo \b\{ar\}"#);
231        let lexemes: Vec<_> = lexer.collect();
232        assert_eq!(
233            lexemes,
234            vec![
235                L(T::Text, 0, "foo "),
236                L(T::Escaped, 5, "b"),
237                L(T::Escaped, 7, "{"),
238                L(T::Text, 8, "ar"),
239                L(T::Escaped, 11, "}"),
240            ]
241        );
242    }
243
244    /// If the last character is a backslash, then treat it as just a backslash, not the start of
245    /// an escape sequence.
246    #[test]
247    fn test_trailing_escape() {
248        let lexer = Lexer::new(r#"foo bar\"#);
249        let lexemes: Vec<_> = lexer.collect();
250        assert_eq!(lexemes, vec![L(T::Text, 0, "foo bar"), L(T::Text, 7, "\\")],);
251    }
