10#[derive(thiserror::Error, Debug)]
11pub enum Error {
12    #[error("Hex {0} contains invalid character")]
13    InvalidHexCharacter(OwnedLexeme),
14
15    #[error("Invalid {0}")]
16    InvalidIdentifier(OwnedLexeme),
17
18    #[error("Invalid {what}: {err}")]
19    InvalidNumber { what: &'static str, err: String },
20
21    #[error("Odd number of characters in hex {0}")]
22    OddHexLiteral(OwnedLexeme),
23
24    #[error("Display format contains too many elements")]
25    TooBig,
26
27    #[error("Display format is nested too deeply")]
28    TooDeep,
29
30    #[error("Display format tries to load too many objects")]
31    TooManyLoads,
32
33    #[error("Unexpected end-of-string, expected {expect}")]
34    UnexpectedEos { expect: ExpectedSet },
35
36    #[error("Unexpected {actual}, expected {expect}")]
37    UnexpectedToken {
38        actual: OwnedLexeme,
39        expect: ExpectedSet,
40    },
41
42    #[error("vector at offset {offset} requires 1 type parameter, found {arity}")]
43    VectorArity { offset: usize, arity: usize },
44}
45
46/// The set of patterns that the parser tried to match against the next token, in a given
47/// invocation of `match_token!` or `match_token_opt!`. This is used to provide a clearer error
48/// message.
49#[derive(Debug, Clone)]
50pub(crate) struct ExpectedSet {
51    /// Other sets of patterns that were attempted on the same location.
52    pub prev: Vec<ExpectedSet>,
53
54    /// The set of patterns that were tried in this invocation.
55    pub tried: &'static [Expected],
56}
57
58#[derive(Debug, Eq, PartialEq, Ord, PartialOrd)]
59pub(crate) enum Expected {
60    /// Expected a token spanning a particular literal slice.
61    Literal(&'static str),
62    /// Expected any slice of source string that matches a specific token.
63    Token(Token),
64}
65
66/// The result of a `match_token_opt!` invocation, which can succeed, or return the set of patterns
67/// it tried, and the offset they were tried at.
68#[derive(Clone)]
69pub(crate) enum Match<T> {
70    Found(T),
71    Tried(Option<usize>, ExpectedSet),
72}
73
74impl Error {
75    // Indicate that `tried` was also tried at `offset`, in case the error is related to other
76    // tokens that were tried at the same location.
77    pub(crate) fn also_tried(self, offset: Option<usize>, tried: ExpectedSet) -> Self {
78        match (offset, self) {
79            (Some(offset), Error::UnexpectedToken { actual, expect }) if offset == actual.2 => {
80                Error::UnexpectedToken {
81                    actual,
82                    expect: expect.union(tried),
83                }
84            }
85
86            (None, Error::UnexpectedEos { expect }) => Error::UnexpectedEos {
87                expect: expect.union(tried),
88            },
89
90            (_, error) => error,
91        }
92    }
93}
94
95impl ExpectedSet {
96    pub(crate) fn new(tried: &'static [Expected]) -> Self {
97        Self {
98            prev: vec![],
99            tried,
100        }
101    }
102
103    pub(crate) fn with_prev(mut self, prev: ExpectedSet) -> Self {
104        self.prev.push(prev);
105        self
106    }
107
108    pub(crate) fn union(mut self, mut other: ExpectedSet) -> Self {
109        // Always arrange for `self` to be the larger set, so that we drain the smaller set into
110        // the larger, to always do the minimal work. Ordering does not matter because tokens
111        // across all sets are collected into a set before displaying them.
112        if self.prev.len() < other.prev.len() {
113            mem::swap(&mut self, &mut other);
114        }
115
116        self.prev.append(&mut other.prev);
117        self.prev.push(other);
118        self
119    }
120
121    pub(crate) fn into_error(self, actual: Option<&Lexeme<'_>>) -> Error {
122        if let Some(actual) = actual {
123            Error::UnexpectedToken {
124                actual: actual.detach(),
125                expect: self,
126            }
127        } else {
128            Error::UnexpectedEos { expect: self }
129        }
130    }
131}
132
133impl<T> Match<T> {
134    pub(crate) fn is_not_found(&self) -> bool {
135        matches!(self, Match::Tried(_, _))
136    }
137}

139impl fmt::Display for Expected {
140    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
141        match self {
142            Expected::Token(token) => write!(f, "{token}"),
143            Expected::Literal(s) => write!(f, "'{s}'"),
144        }
145    }
146}
147
148impl fmt::Display for ExpectedSet {
149    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
150        // Gather all the tokens that were tried.
151        let mut expected = BTreeSet::new();
152        let mut stack = vec![self];
153        while let Some(set) = stack.pop() {
154            expected.extend(set.tried);
155            stack.extend(&set.prev);
156        }
157
158        if expected.is_empty() {
159            return write!(f, "nothing");
160        }
161
162        let mut tokens = expected.into_iter().peekable2();
163        let mut prefix = if tokens.peek2().is_some() {
164            "one of "
165        } else {
166            ""
167        };
168
169        while let Some(token) = tokens.next() {
170            write!(f, "{prefix}{token}")?;
171            prefix = if tokens.peek2().is_some() {
172                ", "
173            } else {
174                ", or "
175            };
176        }
177
178        Ok(())
179    }
