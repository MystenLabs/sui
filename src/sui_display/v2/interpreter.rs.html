23/// The interpreter is responsible for evaluating expressions inside format strings into values.
24pub struct Interpreter<S: V::Store> {
25    store: S,
26
27    /// Cache of the objects that have been fetched so far. This cache is never evicted -- it is
28    /// used to keep objects alive for the lifetime of the interpreter.
29    cache: DashMap<AccountAddress, Option<Arc<V::OwnedSlice>>>,
30
31    root: V::OwnedSlice,
32}
33
34impl<S: V::Store> Interpreter<S> {
35    /// Create a new interpreter instance. `root` is the contents (bytes and layout) of an object
36    /// that acts as the root of all field accesses. `store` is used to fetch additional objects as
37    /// needed.
38    pub fn new(root: V::OwnedSlice, store: S) -> Self {
39        Self {
40            store,
41            cache: DashMap::new(),
42            root,
43        }
44    }
45
46    /// Entrypoint to evaluate a single format string, represented as a sequence of its strands.
47    /// Returns evaluated strands that can then be formatted.
48    pub(crate) async fn eval_strands<'s>(
49        &'s self,
50        strands: &'s [P::Strand<'s>],
51    ) -> Result<Option<Vec<V::Strand<'s>>>, FormatError> {
52        join_all(strands.iter().map(|strand| async move {
53            match strand {
54                P::Strand::Text(s) => Ok(Some(V::Strand::Text(s))),
55                P::Strand::Expr(P::Expr {
56                    offset,
57                    alternates,
58                    transform,
59                }) => {
60                    let transform = transform.unwrap_or_default();
61                    Ok(self
62                        .eval_alts(alternates)
63                        .await?
64                        .map(move |value| V::Strand::Value {
65                            value,
66                            transform,
67                            offset: *offset,
68                        }))
69                }
70            }
71        }))
72        .await
73        .into_iter()
74        .collect()
75    }

77    /// Evaluate each `chain` in turn until one succeeds (produces a non-`None` value).
78    ///
79    /// Returns the result from the first chain that produces a value, or `Ok(None)` if none do.
80    /// Propagates any errors encountered during evaluation.
81    async fn eval_alts<'s>(
82        &'s self,
83        alts: &'s [P::Chain<'s>],
84    ) -> Result<Option<V::Value<'s>>, FormatError> {
85        for chain in alts {
86            if let Some(v) = self.eval_chain(chain).await? {
87                return Ok(Some(v));
88            }
89        }
90
91        Ok(None)
92    }

94    /// Evaluate a chain of field accesses against a root expression.
95    ///
96    /// If the chain does not have a root expression, the object being displayed is used as the
97    /// root. The root is evaluated first, and then each successive accessor is applied to it.
98    ///
99    /// This function returns `Ok(Some(value))` if all nested accesses succeed. An access succeeds
100    /// when the accessor evaluates to `Ok(Some(access))` and the part of the value it is
101    /// describing exists.
102    ///
103    /// Any errors encountered during evaluation are propagated.
104    pub(crate) async fn eval_chain<'s>(
105        &'s self,
106        chain: &'s P::Chain<'s>,
107    ) -> Result<Option<V::Value<'s>>, FormatError> {
108        use V::Accessor as A;
109        use V::Value as VV;
110
111        // Evaluate the root (if it is provided) and the accessors, concurrently.
112        let root: OptionFuture<_> = chain
113            .root
114            .as_ref()
115            .map(|literal| self.eval_literal(literal))
116            .into();
117
118        let accessors = join_all(chain.accessors.iter().map(|a| self.eval_accessor(a)));
119        let (root, accessors) = join!(root, accessors,);
120
121        let mut root = match root {
122            Some(Ok(Some(root))) => root,
123            Some(Ok(None)) => return Ok(None),
124            Some(Err(e)) => return Err(e),
125
126            // If a root was not provided, the object being displayed is the root.
127            None => VV::Slice(self.root.as_slice()),
128        };
129
130        let Some(mut accessors) = accessors
131            .into_iter()
132            .collect::<Result<Option<Vec<_>>, _>>()?
133        else {
134            return Ok(None);
135        };
136
137        accessors.reverse();
138        while let Some(accessor) = accessors.last() {
139            match (root, accessor) {
140                (VV::Address(a), A::DFIndex(i)) => {
141                    let df_id = i.derive_dynamic_field_id(a)?.into();
142                    let Some(slice) = self.object(df_id).await? else {
143                        return Ok(None);
144                    };
145
146                    let field = match FieldVisitor::deserialize(slice.bytes, slice.layout) {
147                        Ok(f) => f,
148                        Err(DFV::Error::Visitor(e)) => return Err(FormatError::Visitor(e)),
149                        Err(_) => return Ok(None),
150                    };
151
152                    if field.kind != DynamicFieldType::DynamicField {
153                        return Ok(None);
154                    }
155
156                    accessors.pop();
157                    root = VV::Slice(V::Slice {
158                        bytes: field.value_bytes,
159                        layout: field.value_layout,
160                    });
161                }
162
163                (VV::Address(a), A::DOFIndex(i)) => {
164                    let df_id = i.derive_dynamic_object_field_id(a)?.into();
165                    let Some(slice) = self.object(df_id).await? else {
166                        return Ok(None);
167                    };
168
169                    let field = match FieldVisitor::deserialize(slice.bytes, slice.layout) {
170                        Ok(f) => f,
171                        Err(DFV::Error::Visitor(e)) => return Err(FormatError::Visitor(e)),
172                        Err(_) => return Ok(None),
173                    };
174
175                    if field.kind != DynamicFieldType::DynamicObject {
176                        return Ok(None);
177                    }
178
179                    let Ok(id) = AccountAddress::from_bytes(field.value_bytes) else {
180                        return Ok(None);
181                    };
182
183                    let Some(value_slice) = self.object(id).await? else {
184                        return Ok(None);
185                    };
186
187                    accessors.pop();
188                    root = VV::Slice(value_slice);
189                }
190
191                // Fetch a single byte from a byte array, as long as the accessor evaluates to a
192                // numeric index.
193                (VV::Bytes(bs), accessor) => {
194                    let Some(&b) = accessor.as_numeric_index().and_then(|i| bs.get(i as usize))
195                    else {
196                        return Ok(None);
197                    };
198
199                    accessors.pop();
200                    root = VV::U8(b);
201                }
202
203                // `V::String` corresponds to `std::string::String` in Move, which contains a
204                // single `bytes` field.
205                (VV::String(s), A::Field(f)) if *f == "bytes" => {
206                    accessors.pop();
207                    root = VV::Bytes(s)
208                }
209
210                // Fetch an element from a vector literal, as long as the accessor evaluates to a
211                // numeric index.
212                (VV::Vector(mut xs), accessor) => {
213                    let Some(i) = accessor.as_numeric_index() else {
214                        return Ok(None);
215                    };
216
217                    accessors.pop();
218                    root = if i as usize >= xs.elements.len() {
219                        return Ok(None);
220                    } else {
221                        xs.elements.swap_remove(i as usize)
222                    };
223                }
224
225                // Fetch a field from a struct or enum literal.
226                (VV::Struct(V::Struct { fields, .. }) | VV::Enum(V::Enum { fields, .. }), a) => {
227                    let Some(value) = fields.get(a) else {
228                        return Ok(None);
229                    };
230
231                    accessors.pop();
232                    root = value;
233                }
234
235                // Use the remaining accessors to extract a value from a slice of a serialized
236                // value. This can consume multiple accessors, but will pause if it encounters a
237                // dynamic (object) field access.
238                (VV::Slice(slice), _) => {
239                    let Some(value) = Extractor::deserialize_slice(slice, &mut accessors)? else {
240                        return Ok(None);
241                    };
242
243                    root = value;
244                }
245
246                // Scalar values do not support accessors.
247                (
248                    VV::Address(_)
249                    | VV::Bool(_)
250                    | VV::String(_)
251                    | VV::U8(_)
252                    | VV::U16(_)
253                    | VV::U32(_)
254                    | VV::U64(_)
255                    | VV::U128(_)
256                    | VV::U256(_),
257                    _,
258                ) => return Ok(None),
259            }
260        }
261
262        Ok(Some(root))
263    }

265    /// Evaluates the contents of an accessor to a value.
266    ///
267    /// Returns `Ok(Some(value))` if the accessor evaluates to a value, otherwise it propagates
268    /// errors or `None` values.
269    async fn eval_accessor<'s>(
270        &'s self,
271        acc: &'s P::Accessor<'s>,
272    ) -> Result<Option<V::Accessor<'s>>, FormatError> {
273        use P::Accessor as PA;
274        use V::Accessor as VA;
275
276        Ok(match acc {
277            PA::Field(f) => Some(VA::Field(f.as_str())),
278            PA::Positional(i) => Some(VA::Positional(*i)),
279            PA::Index(chain) => Box::pin(self.eval_chain(chain)).await?.map(VA::Index),
280            PA::DFIndex(chain) => Box::pin(self.eval_chain(chain)).await?.map(VA::DFIndex),
281            PA::DOFIndex(chain) => Box::pin(self.eval_chain(chain)).await?.map(VA::DOFIndex),
282        })
283    }

285    /// Evaluate literals to values.
286    ///
287    /// Returns `Ok(Some(value))` if all parts of the literal evaluate to `Ok(Some(value))`,
288    /// otherwise it propagates errors or `None` values.
289    pub(crate) async fn eval_literal<'s>(
290        &'s self,
291        lit: &'s P::Literal<'s>,
292    ) -> Result<Option<V::Value<'s>>, FormatError> {
293        use P::Literal as L;
294        use V::Value as VV;
295
296        Ok(match lit {
297            L::Address(a) => Some(VV::Address(*a)),
298            L::Bool(b) => Some(VV::Bool(*b)),
299            L::U8(n) => Some(VV::U8(*n)),
300            L::U16(n) => Some(VV::U16(*n)),
301            L::U32(n) => Some(VV::U32(*n)),
302            L::U64(n) => Some(VV::U64(*n)),
303            L::U128(n) => Some(VV::U128(*n)),
304            L::U256(n) => Some(VV::U256(*n)),
305            L::ByteArray(bs) => Some(VV::Bytes(bs.into())),
306
307            L::String(s) => match s.clone() {
308                Cow::Borrowed(s) => Some(VV::String(Cow::Borrowed(s.as_bytes()))),
309                Cow::Owned(s) => Some(VV::String(Cow::Owned(s.into_bytes()))),
310            },
311
312            L::Vector(v) => self.eval_chains(&v.elements).await.and_then(|elements| {
313                let Some(elements) = elements else {
314                    return Ok(None);
315                };
316
317                // Evaluate the vector's element type and check that it is consistent across all
318                // elements.
319                let type_ = if let Some(explicit) = &v.type_ {
320                    Cow::Borrowed(explicit)
321                } else if let Some(first) = elements.first() {
322                    Cow::Owned(first.type_())
323                } else {
324                    return Err(FormatError::VectorNoType);
325                };
326
327                for e in &elements {
328                    let element_type = e.type_();
329                    if element_type != *type_ {
330                        return Err(FormatError::VectorTypeMismatch {
331                            offset: v.offset,
332                            this: type_.into_owned(),
333                            that: element_type,
334                        });
335                    }
336                }
337
338                Ok(Some(VV::Vector(V::Vector { type_, elements })))
339            })?,
340
341            L::Struct(s) => self.eval_fields(&s.fields).await?.map(|fields| {
342                VV::Struct(V::Struct {
343                    type_: &s.type_,
344                    fields,
345                })
346            }),
347
348            L::Enum(e) => self.eval_fields(&e.fields).await?.map(|fields| {
349                VV::Enum(V::Enum {
350                    type_: &e.type_,
351                    variant_name: e.variant_name,
352                    variant_index: e.variant_index,
353                    fields,
354                })
355            }),
356        })
357    }

359    /// Evaluate the fields of a struct or enum literal, concurrently.
360    ///
361    /// Returns `Ok(Some(fields))` if all the field values evaluate to `Ok(Some(value))`, otherwise
362    /// it propagates errors or `None` values.
363    async fn eval_fields<'s>(
364        &'s self,
365        field: &'s P::Fields<'s>,
366    ) -> Result<Option<V::Fields<'s>>, FormatError> {
367        Ok(match field {
368            P::Fields::Positional(fs) => self.eval_chains(fs).await?.map(V::Fields::Positional),
369            P::Fields::Named(fs) => self
370                .eval_chains(fs.iter().map(|(_, f)| f))
371                .await?
372                .map(|vs| V::Fields::Named(fs.iter().map(|(n, _)| *n).zip(vs).collect())),
373        })
374    }
375
376    /// Evaluate multiple chains concurrently.
377    ///
378    /// If all chains evaluate to `Ok(Some(value))`, returns `Some(vec![value, ...])`, otherwise it
379    /// propagates errors or `None` values.
380    async fn eval_chains<'s>(
381        &'s self,
382        chains: impl IntoIterator<Item = &'s P::Chain<'s>>,
383    ) -> Result<Option<Vec<V::Value<'s>>>, FormatError> {
384        let values = chains
385            .into_iter()
386            .map(|chain| Box::pin(self.eval_chain(chain)));
387
388        join_all(values).await.into_iter().collect()
389    }

391    /// Fetch an object from the store, caching the result.
392    ///
393    /// Returns a `Slice<'s>` that borrows from the cached data. The returned slice is guaranteed
394    /// to remain valid for the lifetime 's because the cache (and the Arc it contains) lives for
395    /// the entire lifetime of the Interpreter.
396    async fn object<'s>(&'s self, id: AccountAddress) -> Result<Option<V::Slice<'s>>, FormatError> {
397        let entry = match self.cache.entry(id) {
398            Entry::Occupied(entry) => entry,
399            Entry::Vacant(entry) => entry.insert_entry(
400                self.store
401                    .object(id)
402                    .await
403                    .map_err(|e| FormatError::Store(Arc::new(e)))?
404                    .map(Arc::new),
405            ),
406        };
407
408        let Some(owned) = entry.get().as_ref() else {
409            return Ok(None);
410        };
411
412        // SAFETY: Extending the lifetime of the slice from the reference (local to this
413        // scope), to the lifetime of the interpreter.  This is safe because the reference is
414        // pointing into an Arc that is owned by the interpreter.
415        let slice = owned.as_slice();
416        Ok(Some(unsafe {
417            mem::transmute::<V::Slice<'_>, V::Slice<'s>>(slice)
418        }))
419    }
