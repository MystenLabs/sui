33/// The interpreter is responsible for evaluating expressions inside format strings into values.
34pub(crate) struct Interpreter<'s, S: Store<'s>> {
35    root: Slice<'s>,
36    store: S,
37    max_depth: usize,
38    max_output_size: usize,
39    used_output: AtomicUsize,
40}
41
42impl<'s, S: Store<'s>> Interpreter<'s, S> {
43    pub(crate) fn new(root: Slice<'s>, store: S, max_depth: usize, max_output_size: usize) -> Self {
44        Self {
45            root,
46            store,
47            max_depth,
48            max_output_size,
49            used_output: AtomicUsize::new(0),
50        }
51    }

53    /// Entrypoint to evaluate a single format string, represented as a sequence of its strands.
54    pub(crate) async fn eval(
55        &self,
56        strands: &'s [P::Strand<'s>],
57    ) -> Result<serde_json::Value, FormatError> {
58        // Detect and handle JSON transforms as a special case because they do not always evaluate
59        // to strings.
60        if let [
61            P::Strand::Expr(P::Expr {
62                alternates,
63                transform: Some(P::Transform::Json),
64            }),
65        ] = &strands
66        {
67            let Some(v) = self.eval_alts(alternates).await? else {
68                return Ok(serde_json::Value::Null);
69            };
70
71            let writer = JsonWriter::new(&self.used_output, self.max_output_size, self.max_depth);
72            return v.format_json(writer);
73        }
74
75        let mut writer = StringWriter::new(&self.used_output, self.max_output_size);
76        for strand in strands {
77            match strand {
78                P::Strand::Text(s) => writer
79                    .write_str(s)
80                    .map_err(|_| FormatError::TooMuchOutput)?,
81
82                P::Strand::Expr(P::Expr {
83                    alternates,
84                    transform,
85                }) => {
86                    let Some(v) = self.eval_alts(alternates).await? else {
87                        return Ok(serde_json::Value::Null);
88                    };
89
90                    v.format(transform.unwrap_or_default(), &mut writer)?;
91                }
92            }
93        }
94
95        Ok(serde_json::Value::String(writer.finish()))
96    }

98    /// Evaluate each `chain` in turn until one succeeds (produces a non-`None` value).
99    ///
100    /// Returns the result from the first chain that produces a value, or `Ok(None)` if none do.
101    /// Propagates any errors encountered during evaluation.
102    async fn eval_alts(&self, alts: &'s [P::Chain<'s>]) -> Result<Option<Value<'s>>, FormatError> {
103        for chain in alts {
104            if let Some(v) = self.eval_chain(chain).await? {
105                return Ok(Some(v));
106            }
107        }
108
109        Ok(None)
110    }
111
112    /// Evaluate a chain of field accesses against a root expression.
113    ///
114    /// If the chain does not have a root expression, the object being displayed is used as the
115    /// root. The root is evaluated first, and then each successive accessor is applied to it.
116    ///
117    /// This function returns `Ok(Some(value))` if all nested accesses succeed. An access succeeds
118    /// when the accessor evaluates to `Ok(Some(access))` and the part of the value it is
119    /// describing exists.
120    ///
121    /// Any errors encountered during evaluation are propagated.
122    async fn eval_chain(&self, chain: &'s P::Chain<'s>) -> Result<Option<Value<'s>>, FormatError> {
123        use Accessor as A;
124        use Value as V;
125
126        // Evaluate the root (if it is provided) and the accessors, concurrently.
127        let root: OptionFuture<_> = chain
128            .root
129            .as_ref()
130            .map(|literal| self.eval_literal(literal))
131            .into();
132
133        let accessors = join_all(chain.accessors.iter().map(|a| self.eval_accessor(a)));
134        let (root, accessors) = join!(root, accessors,);
135
136        let mut root = match root {
137            Some(Ok(Some(root))) => root,
138            Some(Ok(None)) => return Ok(None),
139            Some(Err(e)) => return Err(e),
140
141            // If a root was not provided, the object being displayed is the root.
142            None => V::Slice(self.root),
143        };
144
145        let Some(mut accessors) = accessors
146            .into_iter()
147            .collect::<Result<Option<Vec<_>>, _>>()?
148        else {
149            return Ok(None);
150        };
151
152        accessors.reverse();
153        while let Some(accessor) = accessors.last() {
154            match (root, accessor) {
155                (V::Address(a), A::DFIndex(i)) => {
156                    let bytes = bcs::to_bytes(&i)?;
157                    let type_ = i.type_();
158                    let df_id = derive_dynamic_field_id(a, &type_, &bytes)?;
159
160                    let Some(field) = self
161                        .store
162                        .object(df_id.into())
163                        .await
164                        .map_err(|e| FormatError::Store(Arc::new(e)))?
165                    else {
166                        return Ok(None);
167                    };
168
169                    let field = match FieldVisitor::deserialize(field.bytes, field.layout) {
170                        Ok(f) => f,
171                        Err(DFV::Error::Visitor(e)) => return Err(FormatError::Visitor(e)),
172                        Err(_) => return Ok(None),
173                    };
174
175                    if field.kind != DynamicFieldType::DynamicField {
176                        return Ok(None);
177                    }
178
179                    accessors.pop();
180                    root = V::Slice(Slice {
181                        bytes: field.value_bytes,
182                        layout: field.value_layout,
183                    });
184                }
185
186                (V::Address(a), A::DOFIndex(i)) => {
187                    let bytes = bcs::to_bytes(&i)?;
188                    let type_ = DynamicFieldInfo::dynamic_object_field_wrapper(i.type_()).into();
189                    let df_id = derive_dynamic_field_id(a, &type_, &bytes)?;
190
191                    let Some(field) = self
192                        .store
193                        .object(df_id.into())
194                        .await
195                        .map_err(|e| FormatError::Store(Arc::new(e)))?
196                    else {
197                        return Ok(None);
198                    };
199
200                    let field = match FieldVisitor::deserialize(field.bytes, field.layout) {
201                        Ok(f) => f,
202                        Err(DFV::Error::Visitor(e)) => return Err(FormatError::Visitor(e)),
203                        Err(_) => return Ok(None),
204                    };
205
206                    if field.kind != DynamicFieldType::DynamicObject {
207                        return Ok(None);
208                    }
209
210                    let Ok(id) = AccountAddress::from_bytes(field.value_bytes) else {
211                        return Ok(None);
212                    };
213
214                    let Some(slice) = self
215                        .store
216                        .object(id)
217                        .await
218                        .map_err(|e| FormatError::Store(Arc::new(e)))?
219                    else {
220                        return Ok(None);
221                    };
222
223                    accessors.pop();
224                    root = V::Slice(slice);
225                }
226
227                // Fetch a single byte from a byte array, as long as the accessor evaluates to a
228                // numeric index.
229                (V::Bytes(bs), accessor) => {
230                    let Some(&b) = accessor.as_numeric_index().and_then(|i| bs.get(i as usize))
231                    else {
232                        return Ok(None);
233                    };
234
235                    accessors.pop();
236                    root = V::U8(b);
237                }
238
239                // `V::String` corresponds to `std::string::String` in Move, which contains a
240                // single `bytes` field.
241                (V::String(s), A::Field(f)) if *f == "bytes" => {
242                    accessors.pop();
243                    root = V::Bytes(s)
244                }
245
246                // Fetch an element from a vector literal, as long as the accessor evaluates to a
247                // numeric index.
248                (V::Vector(mut xs), accessor) => {
249                    let Some(i) = accessor.as_numeric_index() else {
250                        return Ok(None);
251                    };
252
253                    accessors.pop();
254                    root = if i as usize >= xs.elements.len() {
255                        return Ok(None);
256                    } else {
257                        xs.elements.swap_remove(i as usize)
258                    };
259                }
260
261                // Fetch a field from a struct or enum literal.
262                (V::Struct(Struct { fields, .. }) | V::Enum(Enum { fields, .. }), a) => {
263                    let Some(value) = fields.get(a) else {
264                        return Ok(None);
265                    };
266
267                    accessors.pop();
268                    root = value;
269                }
270
271                // Use the remaining accessors to extract a value from a slice of a serialized
272                // value. This can consume multiple accessors, but will pause if it encounters a
273                // dynamic (object) field access.
274                (V::Slice(slice), _) => {
275                    let Some(value) = Extractor::deserialize_slice(slice, &mut accessors)? else {
276                        return Ok(None);
277                    };
278
279                    root = value;
280                }
281
282                // Scalar values do not support accessors.
283                (
284                    V::Address(_)
285                    | V::Bool(_)
286                    | V::String(_)
287                    | V::U8(_)
288                    | V::U16(_)
289                    | V::U32(_)
290                    | V::U64(_)
291                    | V::U128(_)
292                    | V::U256(_),
293                    _,
294                ) => return Ok(None),
295            }
296        }
297
298        Ok(Some(root))
299    }

301    /// Evaluates the contents of an accessor to a value.
302    ///
303    /// Returns `Ok(Some(value))` if the accessor evaluates to a value, otherwise it propagates
304    /// errors or `None` values.
305    async fn eval_accessor(
306        &self,
307        acc: &'s P::Accessor<'s>,
308    ) -> Result<Option<Accessor<'s>>, FormatError> {
309        use Accessor as VA;
310        use P::Accessor as PA;
311
312        Ok(match acc {
313            PA::Field(f) => Some(VA::Field(f.as_str())),
314            PA::Positional(i) => Some(VA::Positional(*i)),
315            PA::Index(chain) => Box::pin(self.eval_chain(chain)).await?.map(VA::Index),
316            PA::DFIndex(chain) => Box::pin(self.eval_chain(chain)).await?.map(VA::DFIndex),
317            PA::DOFIndex(chain) => Box::pin(self.eval_chain(chain)).await?.map(VA::DOFIndex),
318        })
319    }

321    /// Evaluate literals to values.
322    ///
323    /// Returns `Ok(Some(value))` if all parts of the literal evaluate to `Ok(Some(value))`,
324    /// otherwise it propagates errors or `None` values.
325    async fn eval_literal(
326        &self,
327        lit: &'s P::Literal<'s>,
328    ) -> Result<Option<Value<'s>>, FormatError> {
329        use P::Literal as L;
330        use Value as V;
331
332        Ok(match lit {
333            L::Address(a) => Some(V::Address(*a)),
334            L::Bool(b) => Some(V::Bool(*b)),
335            L::U8(n) => Some(V::U8(*n)),
336            L::U16(n) => Some(V::U16(*n)),
337            L::U32(n) => Some(V::U32(*n)),
338            L::U64(n) => Some(V::U64(*n)),
339            L::U128(n) => Some(V::U128(*n)),
340            L::U256(n) => Some(V::U256(*n)),
341            L::ByteArray(bs) => Some(V::Bytes(bs.into())),
342
343            L::String(s) => match s.clone() {
344                Cow::Borrowed(s) => Some(V::String(Cow::Borrowed(s.as_bytes()))),
345                Cow::Owned(s) => Some(V::String(Cow::Owned(s.into_bytes()))),
346            },
347
348            L::Vector(v) => self.eval_chains(&v.elements).await.and_then(|elements| {
349                let Some(elements) = elements else {
350                    return Ok(None);
351                };
352
353                // Evaluate the vector's element type and check that it is consistent across all
354                // elements.
355                let type_ = if let Some(explicit) = &v.type_ {
356                    Cow::Borrowed(explicit)
357                } else if let Some(first) = elements.first() {
358                    Cow::Owned(first.type_())
359                } else {
360                    return Err(FormatError::VectorNoType);
361                };
362
363                for e in &elements {
364                    let element_type = e.type_();
365                    if element_type != *type_ {
366                        return Err(FormatError::VectorTypeMismatch(
367                            type_.into_owned(),
368                            element_type,
369                        ));
370                    }
371                }
372
373                Ok(Some(V::Vector(Vector { type_, elements })))
374            })?,
375
376            L::Struct(s) => self.eval_fields(&s.fields).await?.map(|fields| {
377                V::Struct(Struct {
378                    type_: &s.type_,
379                    fields,
380                })
381            }),
382
383            L::Enum(e) => self.eval_fields(&e.fields).await?.map(|fields| {
384                V::Enum(Enum {
385                    type_: &e.type_,
386                    variant_name: e.variant_name,
387                    variant_index: e.variant_index,
388                    fields,
389                })
390            }),
391        })
392    }

394    /// Evaluate the fields of a struct or enum literal, concurrently.
395    ///
396    /// Returns `Ok(Some(fields))` if all the field values evaluate to `Ok(Some(value))`, otherwise
397    /// it propagates errors or `None` values.
398    async fn eval_fields(
399        &self,
400        field: &'s P::Fields<'s>,
401    ) -> Result<Option<Fields<'s>>, FormatError> {
402        Ok(match field {
403            P::Fields::Positional(fs) => self.eval_chains(fs).await?.map(Fields::Positional),
404            P::Fields::Named(fs) => self
405                .eval_chains(fs.iter().map(|(_, f)| f))
406                .await?
407                .map(|vs| Fields::Named(fs.iter().map(|(n, _)| *n).zip(vs).collect())),
408        })
409    }
410
411    /// Evaluate multiple chains concurrently.
412    ///
413    /// If all chains evaluate to `Ok(Some(value))`, returns `Some(vec![value, ...])`, otherwise it
414    /// propagates errors or `None` values.
415    async fn eval_chains(
416        &self,
417        chains: impl IntoIterator<Item = &'s P::Chain<'s>>,
418    ) -> Result<Option<Vec<Value<'s>>>, FormatError> {
419        let values = chains
420            .into_iter()
421            .map(|chain| Box::pin(self.eval_chain(chain)));
422
423        join_all(values).await.into_iter().collect()
424    }
