6/// Lexer for Display V2 format strings. Format strings are a mix of text and expressions.
7/// Expressions are enclosed in braces and may contain multiple alternates, separated by pipes, and
8/// each containing nested field, vector, or dynamic field accesses.
9#[derive(Debug)]
10pub(crate) struct Lexer<'s> {
11    /// Remaining input to be tokenized.
12    src: &'s str,
13
14    /// The number of bytes (not characters) tokenized so far.
15    off: usize,
16
17    /// Nesting of curly braces. At level 0, the lexer is in text mode. At all other levels, it is
18    /// in expression mode.
19    level: usize,
20}
21
22/// A lexeme is a slice of the source string marked with a token. The `bool` field indicates
23/// whether the lexeme was preceded by whitespace or not.
24#[derive(Copy, Clone, Debug, PartialEq, Eq)]
25pub(crate) struct Lexeme<'s>(pub bool, pub Token, pub usize, pub &'s str);
26
27/// Like `Lexeme` but owns the slice of source string. Useful for capturing context in an error
28/// message.
29#[derive(Debug, Clone)]
30pub struct OwnedLexeme(
31    pub(crate) bool,
32    pub(crate) Token,
33    pub(crate) usize,
34    pub(crate) String,
35);
36
37#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
38pub(crate) enum Token {
39    /// '->'
40    Arrow,
41    /// '=>'
42    AArrow,
43    /// '@'
44    At,
45    /// ':'
46    Colon,
47    /// '::'
48    CColon,
49    /// ','
50    Comma,
51    /// '.'
52    Dot,
53    /// An identifier
54    Ident,
55    /// '<'
56    LAngle,
57    /// '{'
58    LBrace,
59    /// '['
60    LBracket,
61    /// '{{'
62    LLBrace,
63    /// '('
64    LParen,
65    /// A decimal number, optionally separated by underscores.
66    NumDec,
67    /// A hexadecimal number, prefixed with '0x' (not included in the span), optionally separated
68    /// by underscores.
69    NumHex,
70    /// '|'
71    Pipe,
72    /// '#'
73    Pound,
74    /// '>'
75    RAngle,
76    /// '}'
77    RBrace,
78    /// ']'
79    RBracket,
80    /// ')'
81    RParen,
82    /// '}}'
83    RRBrace,
84    /// Strings are surrounded by single quotes. Quotes and backslashes inside strings are escaped
85    /// with backslashes.
86    String,
87    /// A strand of text.
88    Text,
89
90    /// An unexpected byte in the input string.
91    Unexpected,
92}

94impl<'s> Lexer<'s> {
95    pub(crate) fn new_for_text(src: &'s str) -> Self {
96        Self {
97            src,
98            off: 0,
99            level: 0,
100        }
101    }
102
103    pub(crate) fn new_for_expr(src: &'s str) -> Self {
104        Self {
105            src,
106            off: 0,
107            // Set a very high level so no reasonable number of curly braces could put the lexer
108            // back into text mode, or cause it to overflow.
109            level: usize::MAX / 2,
110        }
111    }
112
113    /// Assuming the lexer is in text mode, return the next text token.
114    fn next_text_token(&mut self) -> Option<Lexeme<'s>> {
115        let bytes = self.src.as_bytes();
116
117        use Token as T;
118        Some(match bytes.first()? {
119            b'{' if bytes.get(1) == Some(&b'{') => {
120                self.advance(1);
121                self.take(false, T::LLBrace, 1)
122            }
123
124            b'{' => {
125                self.level += 1;
126                self.take(false, T::LBrace, 1)
127            }
128
129            b'}' if bytes.get(1) == Some(&b'}') => {
130                self.advance(1);
131                self.take(false, T::RRBrace, 1)
132            }
133
134            // This is not a valid token within text, but is recognised so that the parser can
135            // produce a better error message. `level` is not decremenetd because we should already
136            // been in text mode, meaning the level is already 0, and a decrement would underflow
137            // it.
138            b'}' => self.take(false, T::RBrace, 1),
139
140            _ => self.take_until(false, T::Text, |b| b"{}".contains(&b)),
141        })
142    }
143
144    /// Assuming the lexer is in expression mode, return the next expression token.
145    fn next_expr_token(&mut self) -> Option<Lexeme<'s>> {
146        let ws = self.take_whitespace();
147        let bytes = self.src.as_bytes();
148
149        use Token as T;
150        Some(match bytes.first()? {
151            b'-' if bytes.get(1) == Some(&b'>') => self.take(ws, T::Arrow, 2),
152
153            b'=' if bytes.get(1) == Some(&b'>') => self.take(ws, T::AArrow, 2),
154
155            b'@' => self.take(ws, T::At, 1),
156
157            b':' if bytes.get(1) == Some(&b':') => self.take(ws, T::CColon, 2),
158
159            b':' => self.take(ws, T::Colon, 1),
160
161            b',' => self.take(ws, T::Comma, 1),
162
163            b'.' => self.take(ws, T::Dot, 1),
164
165            b'0' if bytes.get(1) == Some(&b'x')
166                && bytes.get(2).is_some_and(|b| is_valid_hex_byte(*b)) =>
167            {
168                self.advance(2);
169                self.take_until(ws, T::NumHex, |c| !is_valid_hex_byte(c))
170            }
171
172            b'0'..=b'9' => self.take_until(ws, T::NumDec, |c| !is_valid_decimal_byte(c)),
173
174            b'a'..=b'z' | b'A'..=b'Z' | b'_' => {
175                self.take_until(ws, T::Ident, |c| !is_valid_identifier_byte(c))
176            }
177
178            b'<' => self.take(ws, T::LAngle, 1),
179
180            b'{' => {
181                self.level += 1;
182                self.take(ws, T::LBrace, 1)
183            }
184
185            b'[' => self.take(ws, T::LBracket, 1),
186
187            b'(' => self.take(ws, T::LParen, 1),
188
189            b'|' => self.take(ws, T::Pipe, 1),
190
191            b'#' => self.take(ws, T::Pound, 1),
192
193            b'>' => self.take(ws, T::RAngle, 1),
194
195            b'}' => {
196                self.level -= 1;
197                self.take(ws, T::RBrace, 1)
198            }
199
200            b']' => self.take(ws, T::RBracket, 1),
201
202            b')' => self.take(ws, T::RParen, 1),
203
204            b'\'' => {
205                // Set the escaped indicator to true initially so we don't interpret the starting
206                // quote as an ending quote.
207                let mut escaped = true;
208                for (i, b) in self.src.bytes().enumerate() {
209                    if escaped {
210                        escaped = false;
211                    } else if b == b'\\' {
212                        escaped = true;
213                    } else if b == b'\'' {
214                        self.advance(1);
215                        let content = self.take(ws, T::String, i - 1);
216                        self.advance(1);
217                        return Some(content);
218                    }
219                }
220
221                // Reached the end of the byte stream and didn't find a closing quote -- treat the
222                // partial string as an unexpected token.
223                self.take(ws, T::Unexpected, self.src.len())
224            }
225
226            // If the next byte cannot be recognized, extract the next (potentially variable
227            // length) character, and indicate that it is an unexpected token.
228            _ => {
229                let next_boundary = (1..=self.src.len())
230                    .find(|&i| self.src.is_char_boundary(i))
231                    .unwrap_or(self.src.len());
232                self.take(ws, T::Unexpected, next_boundary)
233            }
234        })
235    }
236
237    /// Eat ASCII whitespace from the beginning of `self.src`. Returns `true` if and only if any
238    /// whitespace was consumed.
239    fn take_whitespace(&mut self) -> bool {
240        let Lexeme(_, _, _, slice) =
241            self.take_until(false, Token::Unexpected, |b| !b.is_ascii_whitespace());
242        !slice.is_empty()
243    }
244
245    /// Take a prefix of bytes from `self.src` until a byte satisfying pattern `p` is found, and
246    /// return it as a lexeme of type `t`. If no such byte is found, take the entire remainder of
247    /// the source string.
248    fn take_until(&mut self, ws: bool, t: Token, p: impl FnMut(u8) -> bool) -> Lexeme<'s> {
249        let n = self.src.bytes().position(p).unwrap_or(self.src.len());
250        self.take(ws, t, n)
251    }
252
253    /// Take `n` bytes from the beginning of `self.src` and return them as a lexeme of type `t`.
254    ///
255    /// ## Safety
256    ///
257    /// This function assumes that there are at least `n` bytes left in `self.src`, and will panic
258    /// if that is not the case.
259    fn take(&mut self, ws: bool, t: Token, n: usize) -> Lexeme<'s> {
260        let start = self.off;
261        let slice = &self.src[..n];
262        self.advance(n);
263
264        Lexeme(ws, t, start, slice)
265    }
266
267    /// Move the cursor forward by `n` bytes.
268    ///
269    /// ## Safety
270    ///
271    /// This function assumes that `n` is less than or equal to the length of `self.src`, and will
272    /// panic if that is not the case.
273    fn advance(&mut self, n: usize) {
274        self.src = &self.src[n..];
275        self.off += n;
276    }
277}

279impl Lexeme<'_> {
280    /// Return the lexeme as an owned lexeme, with the slice of source string copied.
281    pub(crate) fn detach(&self) -> OwnedLexeme {
282        OwnedLexeme(self.0, self.1, self.2, self.3.to_owned())
283    }
284}

286impl<'s> Iterator for Lexer<'s> {
287    type Item = Lexeme<'s>;
288
289    fn next(&mut self) -> Option<Self::Item> {
290        if self.level == 0 {
291            self.next_text_token()
292        } else {
293            self.next_expr_token()
294        }
295    }
296}

298impl fmt::Display for OwnedLexeme {
299    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
300        use OwnedLexeme as L;
301        use Token as T;
302
303        if self.0 {
304            write!(f, "whitespace followed by ")?;
305        }
306
307        match self {
308            L(_, T::Arrow, _, _) => write!(f, "'->'"),
309            L(_, T::AArrow, _, _) => write!(f, "'=>'"),
310            L(_, T::At, _, _) => write!(f, "'@'"),
311            L(_, T::Colon, _, _) => write!(f, "':'"),
312            L(_, T::CColon, _, _) => write!(f, "'::'"),
313            L(_, T::Comma, _, _) => write!(f, "','"),
314            L(_, T::Dot, _, _) => write!(f, "'.'"),
315            L(_, T::Ident, _, s) => write!(f, "identifier {s:?}"),
316            L(_, T::LAngle, _, _) => write!(f, "'<'"),
317            L(_, T::LBrace, _, _) => write!(f, "'{{'"),
318            L(_, T::LBracket, _, _) => write!(f, "'['"),
319            L(_, T::LLBrace, _, _) => write!(f, "'{{{{'"),
320            L(_, T::LParen, _, _) => write!(f, "'('"),
321            L(_, T::NumDec, _, s) => write!(f, "decimal number {s:?}"),
322            L(_, T::NumHex, _, s) => write!(f, "hexadecimal number {s:?}"),
323            L(_, T::Pipe, _, _) => write!(f, "'|'"),
324            L(_, T::Pound, _, _) => write!(f, "'#'"),
325            L(_, T::RAngle, _, _) => write!(f, "'>'"),
326            L(_, T::RBrace, _, _) => write!(f, "'}}'"),
327            L(_, T::RBracket, _, _) => write!(f, "']'"),
328            L(_, T::RParen, _, _) => write!(f, "')'"),
329            L(_, T::RRBrace, _, _) => write!(f, "'}}}}'"),
330            L(_, T::String, _, s) => write!(f, "string {s:?}"),
331            L(_, T::Text, _, s) => write!(f, "text {s:?}"),
332            L(_, T::Unexpected, _, s) => {
333                write!(f, "\"")?;
334                for b in s.bytes() {
335                    match b {
336                        b'"' => write!(f, "\\\"")?,
337                        b'\\' => write!(f, "\\\\")?,
338                        b'\n' => write!(f, "\\n")?,
339                        b'\t' => write!(f, "\\t")?,
340                        b'\r' => write!(f, "\\r")?,
341                        b if b.is_ascii_graphic() || b == b' ' => write!(f, "{}", b as char)?,
342                        b => write!(f, "\\x{:02X}", b)?,
343                    }
344                }
345                write!(f, "\"")
346            }
347        }?;
348
349        write!(f, " at byte offset {}", self.2)
350    }
351}

353impl fmt::Display for Token {
354    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
355        use Token as T;
356        match self {
357            T::Arrow => write!(f, "'->'"),
358            T::AArrow => write!(f, "'=>'"),
359            T::At => write!(f, "'@'"),
360            T::Colon => write!(f, "':'"),
361            T::CColon => write!(f, "'::'"),
362            T::Comma => write!(f, "','"),
363            T::Dot => write!(f, "'.'"),
364            T::Ident => write!(f, "an identifier"),
365            T::LAngle => write!(f, "'<'"),
366            T::LBrace => write!(f, "'{{'"),
367            T::LBracket => write!(f, "'['"),
368            T::LLBrace => write!(f, "'{{{{'"),
369            T::LParen => write!(f, "'('"),
370            T::NumDec => write!(f, "a decimal number"),
371            T::NumHex => write!(f, "a hexadecimal number"),
372            T::Pipe => write!(f, "'|'"),
373            T::Pound => write!(f, "'#'"),
374            T::RAngle => write!(f, "'>'"),
375            T::RBrace => write!(f, "'}}'"),
376            T::RBracket => write!(f, "']'"),
377            T::RParen => write!(f, "')'"),
378            T::RRBrace => write!(f, "'}}}}'"),
379            T::String => write!(f, "a string"),
380            T::Text => write!(f, "text"),
381            T::Unexpected => write!(f, "unexpected input"),
382        }
383    }

386fn is_valid_identifier_byte(b: u8) -> bool {
387    matches!(b, b'_' | b'a'..=b'z' | b'A'..=b'Z' | b'0'..=b'9')
388}
389
390fn is_valid_hex_byte(b: u8) -> bool {
391    matches!(b, b'_' | b'a'..=b'f' | b'A'..=b'F' | b'0'..=b'9')
392}
393
394fn is_valid_decimal_byte(b: u8) -> bool {
395    matches!(b, b'_' | b'0'..=b'9')
396}

412    fn clean(lex: Lexer) -> String {
413        lex.map(|L(ws, t, o, s)| {
414            // Handle potentially invalid UTF-8 by working at byte level
415            let safe_s: String = s
416                .bytes()
417                .map(|b| match b {
418                    b'"' => "\\\"".to_string(),
419                    b'\\' => "\\\\".to_string(),
420                    b'\n' => "\\n".to_string(),
421                    b'\t' => "\\t".to_string(),
422                    b'\r' => "\\r".to_string(),
423                    b if b.is_ascii_graphic() || b == b' ' => (b as char).to_string(),
424                    b => format!("\\x{:02X}", b),
425                })
426                .collect();
427            format!("L({ws:?}, {t:?}, {o:?}, \"{}\")", safe_s)
428        })
429        .collect::<Vec<_>>()
430        .join("\n")
431    }
