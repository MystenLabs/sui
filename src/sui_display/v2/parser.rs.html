25/// A single Display string template is a sequence of strands.
26#[derive(PartialEq, Eq)]
27pub enum Strand<'s> {
28    /// Text strands are ported literally to the output.
29    Text(Cow<'s, str>),
30
31    /// Expr strands are interpreted to some value whose string representation is included in the
32    /// output. They are surrounded by curly braces in the syntax, to differentiate them from text.
33    Expr(Expr<'s>),
34}
35
36/// Expressions are composed of a number of alternates and an optional transform. During
37/// evaluation, each alternate is evaluated in turn until the first one succeeds, and if a
38/// transform is provided, it is applied to the result to convert it to a string.
39#[derive(PartialEq, Eq)]
40pub struct Expr<'s> {
41    alternates: Vec<Chain<'s>>,
42    transform: Option<&'s str>,
43}
44
45/// Chains are a sequence of nested field accesses.
46#[derive(PartialEq, Eq)]
47pub struct Chain<'s> {
48    /// An optional root expression. If not provided, the object being displayed is the root.
49    root: Option<Literal<'s>>,
50
51    /// A sequence of field accessors that go successively deeper into the object.
52    accessors: Vec<Accessor<'s>>,
53}
54
55/// Different ways to nest deeply into an object.
56#[derive(PartialEq, Eq)]
57pub enum Accessor<'s> {
58    /// Access a named field.
59    Field(&'s IdentStr),
60
61    /// Access a positional field.
62    Positional(u8),
63
64    /// Index into a vector, VecMap, or dynamic field.
65    Index(Chain<'s>),
66
67    /// Index into a dynamic field.
68    DFIndex(Chain<'s>),
69
70    /// Index into a dynamic object field.
71    DOFIndex(Chain<'s>),
72}

131/// Helper macro for constructing an `Expected` enum variant based on the kind of pattern being
132/// matched on in the parser. The first argument is the kind, which denotes whether the pattern was
133/// looking for a token, regardless of the contents of the underlying slice, or a literal match on
134/// a particular string slice.
135macro_rules! expected {
136    (Lit, $pat:path, $slice:tt) => {
137        Expected::Literal($slice)
138    };
139
140    (Tok, $pat:path, $slice:tt) => {
141        Expected::Token($pat)
142    };
143}
144
145/// Pattern match on the next token in the lexer, without consuming it. Returns an error if there
146/// is no next token, or if the next token doesn't match any of the provided patterns. The error
147/// enumerates all the tokens that were expected, including the tokens that were checked
148/// `$prev`iously, if any were provided.
149macro_rules! match_token {
150    (
151        $lexer:expr $(, $prev:expr)?;
152        $(
153            $kind:ident($ws:pat, $($pat:path)|+, $off:pat, $slice:tt)
154            $(@ $alias:ident)?
155            $(if $cond:expr)? => $expr:expr
156        ),+
157        $(,)?
158    ) => {{
159        match $lexer.peek() {
160            $(Some($($alias @)? &Lex($ws, $($pat)|+, $off, $slice)) $(if $cond)? => $expr,)+
161            lexeme => return Err(ExpectedSet::new(&[$($(expected!($kind, $pat, $slice)),+),+])
162                $(.with_prev($prev))?
163                .into_error(lexeme)),
164        }
165    }};
166}
167
168/// Expression variant of `match_token!`, which evaluates to `Match::Found(...)` if an arm matches,
169/// consuming the next token, or `Match::Tried(...)` without consuming if there is no next token,
170/// or it doesn't match any of the provided patterns.
171///
172/// In the latter case, the set of patterns checked is included in the output, so that the parser
173/// can accumulate all the patterns it has tried to match against at the current position.
174///
175/// The optional `$prev` argument can be used to include the set of patterns that were checked
176/// before this match.
177macro_rules! match_token_opt {
178    (
179        $lexer:expr $(, $prev:expr)?;
180        $(
181            $kind:ident($ws:pat, $($pat:path)|+, $off:pat, $slice:tt)
182            $(@ $alias:ident)?
183            $(if $cond:expr)? => $expr:expr
184        ),+
185        $(,)?
186    ) => {{
187        match $lexer.peek() {
188            $(Some($($alias @)? &Lex($ws, $($pat)|+, $off, $slice)) $(if $cond)? => Match::Found($expr),)+
189            lexeme => Match::Tried(
190                lexeme.map(|l| l.2),
191                ExpectedSet::new(&[$($(expected!($kind, $pat, $slice)),+),+])
192                    $(.with_prev($prev))?
193            ),
194        }
195    }};
196}

198/// Recursive descent parser for Display V2 format strings, parsing the following grammar:
199///
200///   format   ::= strand*
201///
202///   strand   ::= text | expr
203///
204///   text     ::= part+
205///
206///   part     ::= TEXT | '{{' | '}}'
207///
208///   expr     ::= '{' chain ('|' chain)* (':' IDENT)? '}'
209///
210///   chain    ::= (literal | IDENT) accessor*
211///
212///   accessor ::= '.' IDENT
213///              | '.' NUM_DEC
214///              | '->' '[' chain ']'
215///              | '=>' '[' chain ']'
216///
217///   literal  ::= address | bool | number | string | vector | struct | enum
218///
219///   address  ::= '@' NUM_HEX
220///
221///   bool     ::= 'true' | 'false'
222///
223///   number   ::= (NUM_DEC | NUM_HEX) numeric?
224///
225///   string   ::= ('b' | 'x')? STRING
226///
227///   vector   ::= 'vector'  '<' type ','? '>'
228///              | 'vector' ('<' type ','? '>')?  array
229///
230///   array    ::= '[' chain (',' chain)* ','? ']'
231///
232///   struct   ::= datatype fields
233///
234///   enum     ::= datatype '::' (IDENT '#')? NUM_DEC fields
235///
236///   fields   ::= '(' chain (',' chain)* ','? ')'
237///              | '{' named (',' named)* ','? '}'
238///
239///   named    ::= IDENT ':' chain
240///
241///   type     ::= 'address' | 'bool' | | 'vector' '<' type '>' |  numtype | datatype
242///
243///   datatype ::= NUM_HEX '::' IDENT ('<' type (',' type)* ','? '>')?
244///
245///   numeric  ::= 'u8' | 'u16' | 'u32' | 'u64' | 'u128' | 'u256'
246///
247impl<'s> Parser<'s> {
248    /// Construct a new parser, consuming input from the `src` string.
249    pub(crate) fn new(src: &'s str) -> Self {
250        Self {
251            lexer: Lexer::new(src).peekable2(),
252        }
253    }

255    /// Entrypoint into the parser, parsing the root non-terminal -- `format`.
256    pub(crate) fn run(src: &'s str, limits: Limits) -> Result<Vec<Strand<'s>>, Error> {
257        let mut budget = limits.budget();
258        let mut meter = Meter::new(limits.max_depth, &mut budget);
259        Self::new(src).parse_format(&mut meter)
260    }

262    fn parse_format<'b>(mut self, meter: &mut Meter<'b>) -> Result<Vec<Strand<'s>>, Error> {
263        let mut strands = vec![];
264        while self.lexer.peek().is_some() {
265            strands.push(self.parse_strand(meter)?);
266        }
267
268        Ok(strands)
269    }

271    fn parse_strand<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Strand<'s>, Error> {
272        Ok(match_token! { self.lexer;
273            Tok(_, T::Text | T::LLBrace | T::RRBrace, _, _) => Strand::Text(self.parse_text(meter)?),
274            Tok(_, T::LBrace, _, _) => Strand::Expr(self.parse_expr(meter)?),
275        })
276    }

278    fn parse_text<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Cow<'s, str>, Error> {
279        let mut text = self.parse_part()?;
280        while let Some(Lex(_, T::Text | T::LLBrace | T::RRBrace, _, _)) = self.lexer.peek() {
281            text += self.parse_part()?;
282        }
283
284        meter.alloc()?;
285        Ok(text)
286    }
287
288    fn parse_part(&mut self) -> Result<Cow<'s, str>, Error> {
289        Ok(match_token! { self.lexer;
290            Tok(_, T::Text | T::LLBrace | T::RRBrace, _, slice) => {
291                self.lexer.next();
292                Cow::Borrowed(slice)
293            }
294        })
295    }

297    fn parse_expr<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Expr<'s>, Error> {
298        match_token! { self.lexer; Tok(_, T::LBrace, _, _) => self.lexer.next() };
299        let mut alternates = vec![self.parse_chain(meter)?];
300        let mut transform = None;
301
302        loop {
303            match_token! { self.lexer;
304                Tok(_, T::RBrace, _, _) => {
305                    self.lexer.next();
306                    break;
307                },
308
309                Tok(_, T::Colon, _, _) => {
310                    self.lexer.next();
311                    match_token! { self.lexer; Tok(_, T::Ident, _, t) => {
312                        self.lexer.next();
313                        transform = Some(t);
314                    }};
315                    match_token! { self.lexer; Tok(_, T::RBrace, _, _) => {
316                        self.lexer.next()
317                    }};
318                    break;
319                },
320
321                Tok(_, T::Pipe, _, _) => {
322                    self.lexer.next();
323                    alternates.push(self.parse_chain(meter)?);
324                }
325            }
326        }
327
328        meter.alloc()?;
329        Ok(Expr {
330            alternates,
331            transform,
332        })
333    }

335    fn parse_chain<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Chain<'s>, Error> {
336        let meter = &mut meter.nest()?;
337        let mut accessors = vec![];
338
339        // If there is no root literal, the chain must start with an identifier, representing a
340        // field on the object being displayed.
341        let root = match self.try_parse_literal(meter)? {
342            Match::Found(literal) => Some(literal),
343            Match::Tried(_, tried) => {
344                accessors.push(match_token! { self.lexer, tried;
345                    Tok(_, T::Ident, _, ident) @ lex => {
346                        let ident = IdentStr::new(ident).map_err(|_| Error::InvalidIdentifier(lex.detach()))?;
347                        self.lexer.next();
348                        meter.alloc()?;
349                        Accessor::Field(ident)
350                    }},
351                );
352                None
353            }
354        };
355
356        while let Match::Found(accessor) = self.try_parse_accessor(meter)? {
357            accessors.push(accessor);
358        }
359
360        meter.alloc()?;
361        Ok(Chain { root, accessors })
362    }

364    fn try_parse_literal<'b>(
365        &mut self,
366        meter: &mut Meter<'b>,
367    ) -> Result<Match<Literal<'s>>, Error> {
368        Ok(match_token_opt! { self.lexer;
369            Tok(_, T::At, _, _) => {
370                self.lexer.next();
371                meter.alloc()?;
372                Literal::Address(self.parse_address()?)
373            },
374
375            Lit(_, T::Ident, _, "true") => {
376                self.lexer.next();
377                meter.alloc()?;
378                Literal::Bool(true)
379            },
380
381            Lit(_, T::Ident, _, "false") => {
382                self.lexer.next();
383                meter.alloc()?;
384                Literal::Bool(false)
385            },
386
387            Tok(_, T::NumDec | T::NumHex, _, _)
388            if self.lexer.peek2().is_some_and(|Lex(_, t, _, _)| *t == T::CColon) => {
389                self.parse_data(meter)?
390            },
391
392            Tok(_, T::NumDec, _, dec) => {
393                self.lexer.next();
394                self.parse_numeric_suffix(dec, 10, meter)?
395            },
396
397            Tok(_, T::NumHex, _, hex) => {
398                self.lexer.next();
399                self.parse_numeric_suffix(hex, 16, meter)?
400            },
401
402            Tok(_, T::String, _, slice) => {
403                self.lexer.next();
404                meter.alloc()?;
405                Literal::String(read_string_literal(slice))
406            },
407
408            Lit(_, T::Ident, _, "b")
409            if self.lexer.peek2().is_some_and(|Lex(ws, t, _, _)| !ws && *t == T::String) => {
410                self.lexer.next();
411
412                // SAFETY: Match guard peeks ahead to this token.
413                let Lex(_, _, _, slice) = self.lexer.next().unwrap();
414                let output = read_string_literal(slice);
415
416                meter.alloc()?;
417                Literal::ByteArray(output.into_owned().into_bytes())
418            },
419
420            Lit(_, T::Ident, _, "x")
421            if self.lexer.peek2().is_some_and(|Lex(ws, t, _, _)| !ws && *t == T::String) => {
422                self.lexer.next();
423
424                // SAFETY: Match guard peeks ahead to this token.
425                let lex @ Lex(_, _, _, slice) = self.lexer.next().unwrap();
426                let output = read_hex_literal(&lex, slice)?;
427
428                meter.alloc()?;
429                Literal::ByteArray(output)
430            },
431
432            Tok(_, T::Ident, offset, "vector") => {
433                self.lexer.next();
434
435                let mut type_params = self.parse_type_params(meter)?;
436                let type_ = match type_params.len() {
437                    // SAFETY: Bounds check on `type_params.len()` guarantees safety.
438                    1 => Some(type_params.pop().unwrap()),
439                    0 => None,
440                    arity => return Err(Error::VectorArity { offset, arity }),
441                };
442
443                let elements = self.parse_array_elements(meter)?;
444
445                // If the vector is empty, the type parameter becomes mandatory.
446                if elements.is_empty() && type_.is_none() {
447                    return Err(Error::VectorArity {
448                        offset,
449                        arity: 0,
450                    });
451                }
452
453                meter.alloc()?;
454                Literal::Vector(Box::new(Vector { type_, elements }))
455            },
456        })
457    }

459    fn try_parse_accessor<'b>(
460        &mut self,
461        meter: &mut Meter<'b>,
462    ) -> Result<Match<Accessor<'s>>, Error> {
463        Ok(match_token_opt! { self.lexer;
464            Tok(_, T::Dot, _, _) => {
465                self.lexer.next();
466                match_token! { self.lexer;
467                    Tok(_, T::Ident, _, ident) @ lex => {
468                        let ident = IdentStr::new(ident)
469                            .map_err(|_| Error::InvalidIdentifier(lex.detach()))?;
470                        self.lexer.next();
471                        meter.alloc()?;
472                        Accessor::Field(ident)
473                    },
474
475                    Tok(_, T::NumDec, _, n) => {
476                        self.lexer.next();
477                        let index = read_u8(n, 10, "positional field index")?;
478
479                        meter.alloc()?;
480                        Accessor::Positional(index)
481                    }
482                }
483            },
484
485            Tok(_, T::Arrow, _, _) => {
486                self.lexer.next();
487
488                match_token! { self.lexer; Tok(_, T::LBracket, _, _) => self.lexer.next() };
489                let chain = self.parse_chain(meter)?;
490                match_token! { self.lexer; Tok(_, T::RBracket, _, _) => self.lexer.next() };
491
492                meter.load()?;
493                meter.alloc()?;
494                Accessor::DFIndex(chain)
495            },
496
497            Tok(_, T::AArrow, _, _) => {
498                self.lexer.next();
499
500                match_token! { self.lexer; Tok(_, T::LBracket, _, _) => self.lexer.next() };
501                let chain = self.parse_chain(meter)?;
502                match_token! { self.lexer; Tok(_, T::RBracket, _, _) => self.lexer.next() };
503
504                // Dynamic Object Fields require two successive loads.
505                meter.load()?;
506                meter.load()?;
507                meter.alloc()?;
508                Accessor::DOFIndex(chain)
509            },
510
511            Tok(_, T::LBracket, _, _) => {
512                self.lexer.next();
513
514                let chain = self.parse_chain(meter)?;
515                match_token! { self.lexer; Tok(_, T::RBracket, _, _) => self.lexer.next() };
516
517                meter.alloc()?;
518                Accessor::Index(chain)
519            },
520        })
521    }

523    fn parse_array_elements<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Vec<Chain<'s>>, Error> {
524        let mut elements = vec![];
525
526        if match_token_opt! { self.lexer; Tok(_, T::LBracket, _, _) => { self.lexer.next(); } }
527            .is_not_found()
528        {
529            return Ok(elements);
530        }
531
532        let terminated = match_token_opt! { self.lexer;
533            Tok(_, T::RBracket, _, _) => { self.lexer.next(); }
534        };
535
536        let (offset, terminated) = match terminated {
537            Match::Tried(offset, terminated) => (offset, terminated),
538            Match::Found(_) => return Ok(elements),
539        };
540
541        loop {
542            elements.push(
543                self.parse_chain(meter)
544                    .map_err(|e| e.also_tried(offset, terminated.clone()))?,
545            );
546
547            let delimited = match_token_opt! { self.lexer;
548                Tok(_, T::Comma, _, _) => { self.lexer.next(); }
549            };
550
551            let terminated = match_token_opt! { self.lexer;
552                Tok(_, T::RBracket, _, _) => { self.lexer.next(); }
553            };
554
555            match (delimited, terminated) {
556                (_, Match::Found(_)) => break,
557                (Match::Found(_), _) => continue,
558                (Match::Tried(_, delimited), Match::Tried(_, terminated)) => {
559                    return Err(delimited.union(terminated).into_error(self.lexer.peek()));
560                }
561            }
562        }
563
564        Ok(elements)
565    }

567    fn parse_data<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Literal<'s>, Error> {
568        let type_ = self.parse_datatype(meter)?;
569
570        let enum_ = match_token_opt! { self.lexer;
571            Tok(_, T::CColon, _, _) => { self.lexer.next(); }
572        };
573
574        if let Match::Tried(offset, enum_) = enum_ {
575            meter.alloc()?;
576            return Ok(Literal::Struct(Box::new(Struct {
577                type_,
578                fields: self
579                    .parse_fields(meter)
580                    .map_err(|e| e.also_tried(offset, enum_))?,
581            })));
582        }
583
584        Ok(match_token! { self.lexer;
585            Tok(_, T::Ident, _, variant_name) => {
586                self.lexer.next();
587
588                match_token! { self.lexer; Tok(_, T::Pound, _, _) => self.lexer.next() };
589                let variant_index = match_token! { self.lexer; Tok(_, T::NumDec, _, index) => {
590                    self.lexer.next();
591                    read_u16(index, 10, "enum variant index")?
592                }};
593
594                meter.alloc()?;
595                Literal::Enum(Box::new(Enum {
596                    type_,
597                    variant_name: Some(variant_name),
598                    variant_index,
599                    fields: self.parse_fields(meter)?,
600                }))
601            },
602
603            Tok(_, T::NumDec, _, index) => {
604                self.lexer.next();
605
606                let variant_index = read_u16(index, 10, "enum variant index")?;
607
608                meter.alloc()?;
609                Literal::Enum(Box::new(Enum {
610                    type_,
611                    variant_name: None,
612                    variant_index,
613                    fields: self.parse_fields(meter)?,
614                }))
615            }
616        })
617    }

619    fn parse_fields<'b>(&mut self, meter: &mut Meter<'b>) -> Result<Fields<'s>, Error> {
620        let is_named = match_token! { self.lexer;
621            Tok(_, T::LParen, _, _) => { self.lexer.next(); false },
622            Tok(_, T::LBrace, _, _) => { self.lexer.next(); true }
623        };
624
625        Ok(if is_named {
626            let mut fields = vec![];
627
628            let terminated = match_token_opt! { self.lexer;
629                Tok(_, T::RBrace, _, _) => { self.lexer.next(); }
630            };
631
632            let terminated = match terminated {
633                Match::Tried(_, terminated) => terminated,
634                Match::Found(_) => return Ok(Fields::Named(fields)),
635            };
636
637            loop {
638                let name = match_token! { self.lexer, terminated;
639                    Tok(_, T::Ident, _, n) => { self.lexer.next(); n }
640                };
641
642                match_token! { self.lexer; Tok(_, T::Colon, _, _) => self.lexer.next() };
643
644                let value = self.parse_chain(meter)?;
645                fields.push((name, value));
646
647                let delimited = match_token_opt! { self.lexer;
648                    Tok(_, T::Comma, _, _) => { self.lexer.next(); }
649                };
650
651                let terminated = match_token_opt! { self.lexer;
652                    Tok(_, T::RBrace, _, _) => { self.lexer.next(); }
653                };
654
655                match (delimited, terminated) {
656                    (_, Match::Found(_)) => break,
657                    (Match::Found(_), _) => continue,
658                    (Match::Tried(_, delimited), Match::Tried(_, terminated)) => {
659                        return Err(delimited.union(terminated).into_error(self.lexer.peek()));
660                    }
661                }
662            }
663
664            Fields::Named(fields)
665        } else {
666            let mut fields = vec![];
667
668            let terminated = match_token_opt! { self.lexer;
669                Tok(_, T::RParen, _, _) => { self.lexer.next(); }
670            };
671
672            let (offset, terminated) = match terminated {
673                Match::Tried(offset, terminated) => (offset, terminated),
674                Match::Found(_) => return Ok(Fields::Positional(fields)),
675            };
676
677            loop {
678                fields.push(
679                    self.parse_chain(meter)
680                        .map_err(|e| e.also_tried(offset, terminated.clone()))?,
681                );
682
683                let delimited = match_token_opt! { self.lexer;
684                    Tok(_, T::Comma, _, _) => { self.lexer.next(); }
685                };
686
687                let terminated = match_token_opt! { self.lexer;
688                    Tok(_, T::RParen, _, _) => { self.lexer.next(); }
689                };
690
691                match (delimited, terminated) {
692                    (_, Match::Found(_)) => break,
693                    (Match::Found(_), _) => continue,
694                    (Match::Tried(_, delimited), Match::Tried(_, terminated)) => {
695                        return Err(delimited.union(terminated).into_error(self.lexer.peek()));
696                    }
697                }
698            }
699
700            Fields::Positional(fields)
701        })
702    }

704    fn parse_type(&mut self, meter: &mut Meter<'_>) -> Result<TypeTag, Error> {
705        let meter = &mut meter.nest()?;
706        Ok(match_token! { self.lexer;
707            Lit(_, T::Ident, _, "address") => {
708                self.lexer.next();
709                meter.alloc()?;
710                TypeTag::Address
711            },
712
713            Lit(_, T::Ident, _, "bool") => {
714                self.lexer.next();
715                meter.alloc()?;
716                TypeTag::Bool
717            },
718
719            Lit(_, T::Ident, _, "u8") => {
720                self.lexer.next();
721                meter.alloc()?;
722                TypeTag::U8
723            },
724
725            Lit(_, T::Ident, _, "u16") => {
726                self.lexer.next();
727                meter.alloc()?;
728                TypeTag::U16
729            },
730
731            Lit(_, T::Ident, _, "u32") => {
732                self.lexer.next();
733                meter.alloc()?;
734                TypeTag::U32
735            },
736
737            Lit(_, T::Ident, _, "u64") => {
738                self.lexer.next();
739                meter.alloc()?;
740                TypeTag::U64
741            },
742
743            Lit(_, T::Ident, _, "u128") => {
744                self.lexer.next();
745                meter.alloc()?;
746                TypeTag::U128
747            },
748
749            Lit(_, T::Ident, _, "u256") => {
750                self.lexer.next();
751                meter.alloc()?;
752                TypeTag::U256
753            },
754
755            Lit(_, T::Ident, offset, "vector") => {
756                self.lexer.next();
757                let mut type_params = self.parse_type_params(meter)?;
758                match type_params.len() {
759                    1 => {
760                        // SAFETY: Bounds check on `type_params.len()` guarantees safety.
761                        let inner = type_params.pop().unwrap();
762
763                        meter.alloc()?;
764                        TypeTag::Vector(Box::new(inner))
765                    }
766
767                    arity => return Err(Error::VectorArity { offset, arity }),
768                }
769            },
770
771            Tok(_, T::NumDec | T::NumHex, _, _) => {
772                TypeTag::Struct(Box::new(self.parse_datatype(meter)?))
773            }
774        })
775    }

777    fn parse_datatype(&mut self, meter: &mut Meter<'_>) -> Result<StructTag, Error> {
778        let address = self.parse_address()?;
779
780        match_token! { self.lexer; Tok(_, T::CColon, _, _) => self.lexer.next() };
781        let module = self.parse_identifier()?;
782
783        match_token! { self.lexer; Tok(_, T::CColon, _, _) => self.lexer.next() };
784        let name = self.parse_identifier()?;
785
786        let type_params = self.parse_type_params(meter)?;
787
788        meter.alloc()?;
789        Ok(StructTag {
790            address,
791            module,
792            name,
793            type_params,
794        })
795    }

797    fn parse_type_params(&mut self, meter: &mut Meter<'_>) -> Result<Vec<TypeTag>, Error> {
798        let mut type_params = vec![];
799        if match_token_opt! { self.lexer; Tok(_, T::LAngle, _, _) => { self.lexer.next(); } }
800            .is_not_found()
801        {
802            return Ok(type_params);
803        }
804
805        loop {
806            type_params.push(self.parse_type(meter)?);
807
808            let delimited = match_token_opt! { self.lexer;
809                Tok(_, T::Comma, _, _) => { self.lexer.next(); }
810            };
811
812            let terminated = match_token_opt! { self.lexer;
813                Tok(_, T::RAngle, _, _) => { self.lexer.next(); }
814            };
815
816            match (delimited, terminated) {
817                (_, Match::Found(_)) => break,
818                (Match::Found(_), _) => continue,
819                (Match::Tried(_, delimited), Match::Tried(_, terminated)) => {
820                    return Err(delimited.union(terminated).into_error(self.lexer.peek()));
821                }
822            }
823        }
824
825        Ok(type_params)
826    }

910impl fmt::Debug for Expr<'_> {
911    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
912        let mut prefix = "{ ";
913        for alternate in &self.alternates {
914            write!(f, "{prefix}")?;
915            if f.alternate() {
916                let alternate = format!("{alternate:#?}").replace('\n', "\n  ");
917                write!(f, "{alternate}")?;
918                prefix = "\n| ";
919            } else {
920                write!(f, "{alternate:?}")?;
921                prefix = " | ";
922            }
923        }
924
925        if f.alternate() {
926            writeln!(f)?;
927        }
928
929        if let Some(transform) = self.transform {
930            write!(f, ": {transform}")?;
931        }
932
933        write!(f, "}}")
934    }
935}

937impl fmt::Debug for Chain<'_> {
938    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
939        let mut accessors = self.accessors.iter();
940        if let Some(root) = &self.root {
941            root.fmt(f)?;
942        } else if let Some(Accessor::Field(name)) = accessors.next() {
943            // If there is no root, the first accessor is guaranteed to exist, and it must be a
944            // field accessor.
945            write!(f, "{name}")?;
946        }
947
948        for accessor in accessors {
949            use Accessor as A;
950            match accessor {
951                A::Field(name) => write!(f, ".{name}")?,
952                A::Positional(index) => write!(f, ".{index}")?,
953                A::Index(chain) => write!(f, "[{chain:?}]")?,
954                A::DFIndex(chain) => write!(f, "->[{chain:?}]")?,
955                A::DOFIndex(chain) => write!(f, "=>[{chain:?}]")?,
956            }
957        }
958
959        Ok(())
960    }
961}

1101fn read_string_literal(slice: &str) -> Cow<'_, str> {
1102    let mut start = slice.find('\\').unwrap_or(slice.len());
1103    let mut output = Cow::Borrowed(&slice[0..start]);
1104
1105    while start < slice.len() {
1106        // Skip the escape character.
1107        start += 1;
1108
1109        // Slurp up to the next escape character, or the end of the string.
1110        let end = slice[start + 1..]
1111            .find('\\')
1112            .map_or(slice.len(), |i| start + 1 + i);
1113
1114        output += &slice[start..end];
1115        start = end;
1116    }
1117
1118    output
1119}

1121fn read_hex_literal(lexeme: &Lex<'_>, slice: &str) -> Result<Vec<u8>, Error> {
1122    if !slice.len().is_multiple_of(2) {
1123        return Err(Error::OddHexLiteral(lexeme.detach()));
1124    }
1125
1126    let mut output = Vec::with_capacity(slice.len() / 2);
1127    for i in (0..slice.len()).step_by(2) {
1128        let byte = u8::from_str_radix(&slice[i..i + 2], 16)
1129            .map_err(|_| Error::InvalidHexCharacter(lexeme.detach()))?;
1130        output.push(byte);
1131    }
1132
1133    Ok(output)
1134}
