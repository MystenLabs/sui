171    /// Write out a formatted representation of this value, transformed by `transform`, to the
172    /// provided writer.
173    ///
174    /// This operation can fail if the transform is not supported for this value, or if the output
175    /// is too large. If it succeds, `w` will be modified to include the newly written data.
176    pub(crate) fn format(
177        self,
178        transform: Transform,
179        w: &mut StringWriter<'_>,
180    ) -> Result<(), FormatError> {
181        match transform {
182            Transform::Base64(xmod) => Atom::try_from(self)?.format_as_base64(xmod.engine(), w),
183            Transform::Bcs(xmod) => {
184                let bytes = bcs::to_bytes(&self)?;
185                Ok(write!(w, "{}", xmod.engine().encode(bytes))?)
186            }
187
188            Transform::Hex => Atom::try_from(self)?.format_as_hex(w),
189            Transform::Json => Err(FormatError::TransformInvalid("unexpected 'json' in string")),
190            Transform::Str => Atom::try_from(self)?.format_as_str(w),
191            Transform::Timestamp => Atom::try_from(self)?.format_as_timestamp(w),
192            Transform::Url => Atom::try_from(self)?.format_as_url(w),
193        }
194    }
195
196    /// Write out a formatted representation of this value as JSON, using the provided writer.
197    ///
198    /// This operation can fail if the output is too large. If it succeeds, `w` will be modified to
199    /// account for the size of the written data, which will be returned.
200    pub(crate) fn format_json(
201        self,
202        mut w: JsonWriter<'_>,
203    ) -> Result<serde_json::Value, FormatError> {
204        match self {
205            Value::Address(a) => w.write_str(a.to_canonical_string(true)),
206            Value::Bool(b) => w.write_bool(b),
207            Value::U8(n) => w.write_number(n as u32),
208            Value::U16(n) => w.write_number(n as u32),
209            Value::U32(n) => w.write_number(n),
210            Value::U64(n) => w.write_str(n.to_string()),
211            Value::U128(n) => w.write_str(n.to_string()),
212            Value::U256(n) => w.write_str(n.to_string()),
213
214            Value::Bytes(bs) => w.write_str(Base64Modifier::EMPTY.engine().encode(&bs)),
215            Value::String(bs) => w.write_str(
216                str::from_utf8(&bs)
217                    .map_err(|_| FormatError::TransformInvalid("expected utf8 bytes"))?
218                    .to_owned(),
219            ),
220
221            Value::Struct(s) => s.format_json(w),
222            Value::Enum(e) => e.format_json(w),
223            Value::Vector(v) => v.format_json(w),
224            Value::Slice(s) => s.format_json(w),
225        }
226    }

228    /// The Move type of this value.
229    pub(crate) fn type_(&self) -> TypeTag {
230        match self {
231            Value::Address(_) => TypeTag::Address,
232            Value::Bool(_) => TypeTag::Bool,
233            Value::Bytes(_) => TypeTag::Vector(Box::new(TypeTag::U8)),
234            Value::U8(_) => TypeTag::U8,
235            Value::U16(_) => TypeTag::U16,
236            Value::U32(_) => TypeTag::U32,
237            Value::U64(_) => TypeTag::U64,
238            Value::U128(_) => TypeTag::U128,
239            Value::U256(_) => TypeTag::U256,
240
241            Value::Enum(e) => e.type_.clone().into(),
242            Value::Struct(s) => s.type_.clone().into(),
243
244            Value::Slice(s) => s.layout.into(),
245
246            Value::String(_) => {
247                let (&address, module, name) = RESOLVED_UTF8_STR;
248                TypeTag::Struct(Box::new(StructTag {
249                    address,
250                    module: module.to_owned(),
251                    name: name.to_owned(),
252                    type_params: vec![],
253                }))
254            }
255
256            Value::Vector(v) => v.type_(),
257        }
258    }

260    /// Attempt to coerce this value into a `u64` if that's possible. This works for any numeric
261    /// value that can be represented within 64 bits.
262    pub(crate) fn as_u64(&self) -> Option<u64> {
263        use MoveTypeLayout as L;
264        use Value as V;
265
266        match self {
267            // Numeric literals in Display
268            V::U8(n) => Some(*n as u64),
269            V::U16(n) => Some(*n as u64),
270            V::U32(n) => Some(*n as u64),
271            V::U64(n) => Some(*n),
272            V::U128(n) => u64::try_from(*n).ok(),
273            V::U256(n) => u64::try_from(*n).ok(),
274
275            // Numeric values sliced out of Move values
276            V::Slice(Slice {
277                layout,
278                bytes: data,
279            }) => match layout {
280                L::U8 => Some(bcs::from_bytes::<u8>(data).ok()?.into()),
281                L::U16 => Some(bcs::from_bytes::<u16>(data).ok()?.into()),
282                L::U32 => Some(bcs::from_bytes::<u32>(data).ok()?.into()),
283                L::U64 => bcs::from_bytes::<u64>(data).ok(),
284                L::U128 => bcs::from_bytes::<u128>(data).ok()?.try_into().ok(),
285                L::U256 => bcs::from_bytes::<U256>(data).ok()?.try_into().ok(),
286                L::Address | L::Bool | L::Enum(_) | L::Signer | L::Struct(_) | L::Vector(_) => None,
287            },
288
289            // Everything else cannot be coerced to u64
290            V::Address(_)
291            | V::Bool(_)
292            | V::Bytes(_)
293            | V::Enum(_)
294            | V::String(_)
295            | V::Struct(_)
296            | V::Vector(_) => None,
297        }
298    }

301impl Atom<'_> {
302    /// Format the atom as a hexadecimal string.
303    fn format_as_hex(&self, w: &mut StringWriter<'_>) -> Result<(), FormatError> {
304        match self {
305            Atom::Bool(b) => write!(w, "{:02x}", *b as u8)?,
306            Atom::U8(n) => write!(w, "{n:02x}")?,
307            Atom::U16(n) => write!(w, "{n:04x}")?,
308            Atom::U32(n) => write!(w, "{n:08x}")?,
309            Atom::U64(n) => write!(w, "{n:016x}")?,
310            Atom::U128(n) => write!(w, "{n:032x}")?,
311            Atom::U256(n) => write!(w, "{n:064x}")?,
312
313            Atom::Address(a) => {
314                for b in a.into_bytes() {
315                    write!(w, "{b:02x}")?;
316                }
317            }
318
319            Atom::Bytes(bs) => {
320                for b in bs.iter() {
321                    write!(w, "{b:02x}")?;
322                }
323            }
324        }
325
326        Ok(())
327    }
328
329    /// Format the atom as a string.
330    fn format_as_str(&self, w: &mut StringWriter<'_>) -> Result<(), FormatError> {
331        match self {
332            Atom::Address(a) => write!(w, "{}", a.to_canonical_display(true))?,
333            Atom::Bool(b) => write!(w, "{b}")?,
334            Atom::U8(n) => write!(w, "{n}")?,
335            Atom::U16(n) => write!(w, "{n}")?,
336            Atom::U32(n) => write!(w, "{n}")?,
337            Atom::U64(n) => write!(w, "{n}")?,
338            Atom::U128(n) => write!(w, "{n}")?,
339            Atom::U256(n) => write!(w, "{n}")?,
340            Atom::Bytes(bs) => {
341                let s = str::from_utf8(bs)
342                    .map_err(|_| FormatError::TransformInvalid("expected utf8 bytes"))?;
343                write!(w, "{s}")?;
344            }
345        }
346
347        Ok(())
348    }
349
350    /// Coerce the atom into an `i64`, interpreted as an offset in milliseconds since the Unix
351    /// epoch, and format it as an ISO8601 timestamp.
352    fn format_as_timestamp(&self, w: &mut StringWriter<'_>) -> Result<(), FormatError> {
353        let ts = self
354            .as_i64()
355            .and_then(DateTime::from_timestamp_millis)
356            .ok_or_else(|| {
357                FormatError::TransformInvalid("expected unix timestamp in milliseconds")
358            })?;
359
360        write!(w, "{ts:?}")?;
361        Ok(())
362    }
363
364    /// Like string formatting, but percent-encoding reserved URL characters.
365    fn format_as_url(&self, w: &mut StringWriter<'_>) -> Result<(), FormatError> {
366        match self {
367            Atom::Address(a) => write!(w, "{}", a.to_canonical_display(true))?,
368            Atom::Bool(b) => write!(w, "{b}")?,
369            Atom::U8(n) => write!(w, "{n}")?,
370            Atom::U16(n) => write!(w, "{n}")?,
371            Atom::U32(n) => write!(w, "{n}")?,
372            Atom::U64(n) => write!(w, "{n}")?,
373            Atom::U128(n) => write!(w, "{n}")?,
374            Atom::U256(n) => write!(w, "{n}")?,
375            Atom::Bytes(bs) => {
376                for b in bs.iter() {
377                    match *b {
378                        b'A'..=b'Z' | b'a'..=b'z' | b'0'..=b'9' | b'-' | b'.' | b'_' | b'~' => {
379                            write!(w, "{}", *b as char)?
380                        }
381                        b => write!(w, "%{b:02X}")?,
382                    }
383                }
384            }
385        }
386
387        Ok(())
388    }
389
390    /// Base64-encode the byte representation of this atom.
391    fn format_as_base64(
392        &self,
393        e: &impl Engine,
394        w: &mut StringWriter<'_>,
395    ) -> Result<(), FormatError> {
396        let base64 = match self {
397            Atom::Address(a) => e.encode(a.into_bytes()),
398            Atom::Bool(b) => e.encode([*b as u8]),
399            Atom::U8(n) => e.encode([*n]),
400            Atom::U16(n) => e.encode(n.to_le_bytes()),
401            Atom::U32(n) => e.encode(n.to_le_bytes()),
402            Atom::U64(n) => e.encode(n.to_le_bytes()),
403            Atom::U128(n) => e.encode(n.to_le_bytes()),
404            Atom::U256(n) => e.encode(n.to_le_bytes()),
405            Atom::Bytes(bs) => e.encode(bs),
406        };
407
408        write!(w, "{base64}")?;
409        Ok(())
410    }
411
412    /// Attempt to coerce this atom into an `i64`, if possible.
413    fn as_i64(&self) -> Option<i64> {
414        match self {
415            Atom::U8(n) => Some(*n as i64),
416            Atom::U16(n) => Some(*n as i64),
417            Atom::U32(n) => Some(*n as i64),
418            Atom::U64(n) => i64::try_from(*n).ok(),
419            Atom::U128(n) => i64::try_from(*n).ok(),
420            Atom::U256(n) => u64::try_from(*n).ok().and_then(|v| i64::try_from(v).ok()),
421            _ => None,
422        }
423    }

426impl<'s> Accessor<'s> {
427    /// Coerce this accessor into a numeric index, if possible, and returns its value.
428    ///
429    /// Coercion works for all integer literals, as well as `Slice` literals with a numeric layout,
430    /// as long as their numeric values fit into a `u64`.
431    pub(crate) fn as_numeric_index(&self) -> Option<u64> {
432        use Accessor as A;
433
434        match self {
435            A::Index(value) => value.as_u64(),
436            // All other index types don't represent a numeric index.
437            A::DFIndex(_) | A::DOFIndex(_) | A::Field(_) | A::Positional(_) => None,
438        }
439    }
440
441    /// Coerce this accessor into a field name, if possible, and return its name.
442    pub(crate) fn as_field_name(&self) -> Option<Cow<'s, str>> {
443        use Accessor as A;
444        match self {
445            A::Field(f) => Some(Cow::Borrowed(*f)),
446            A::Positional(i) => Some(Cow::Owned(format!("pos{i}"))),
447            A::Index(_) | A::DFIndex(_) | A::DOFIndex(_) => None,
448        }
449    }
450}

470    /// This operation returns `None` if the value contains compound literals (struct, enum, vector

484

486            Value::Slice(s) => Some(s.layout.clone()),

490            Value::U8(_) => Some(L::U8),
491            Value::U16(_) => Some(L::U16),
492            Value::U32(_) => Some(L::U32),
493            Value::U64(_) => Some(L::U64),
494            Value::U128(_) => Some(L::U128),
495            Value::U256(_) => Some(L::U256),

497            Value::Bytes(_) => Some(L::Vector(Box::new(L::U8))),

532impl Enum<'_> {
533    fn format_json(self, mut w: JsonWriter<'_>) -> Result<serde_json::Value, FormatError> {
534        let mut map = serde_json::Map::new();
535        let mut nested = w.nest()?;
536
537        let name = match self.variant_name {
538            Some(name) => nested.write_str(name.to_owned())?,
539            None => nested.write_number(self.variant_index as u32)?,
540        };

543        self.fields.format_json(nested, &mut map)?;
544        w.write_map(map)
545    }

548impl<'s> Fields<'s> {
549    /// Attempt to fetch a particular field  from a struct or enum literal's fields based on the
550    /// given accessor.
551    pub(crate) fn get(self, accessor: &Accessor<'s>) -> Option<Value<'s>> {
552        match (self, accessor) {
553            (Fields::Positional(mut fs), Accessor::Positional(i)) => {
554                let i = *i as usize;
555                if i < fs.len() {
556                    Some(fs.swap_remove(i))
557                } else {
558                    None
559                }
560            }
561
562            (Fields::Named(mut fs), Accessor::Field(f)) => {
563                let i = fs.iter().position(|(n, _)| n == f)?;
564                Some(fs.swap_remove(i).1)
565            }
566
567            _ => None,
568        }
569    }
570
571    fn len(&self) -> usize {
572        match self {
573            Fields::Positional(fs) => fs.len(),
574            Fields::Named(fs) => fs.len(),
575        }
576    }
577
578    fn format_json(
579        self,
580        mut w: JsonWriter<'_>,
581        map: &mut serde_json::Map<String, serde_json::Value>,
582    ) -> Result<(), FormatError> {
583        match self {
584            Fields::Positional(values) => {
585                for (i, value) in values.into_iter().enumerate() {
586                    let json = value.format_json(w)?;
587                    w.map_push_field(map, format!("pos{i}"), json)?;
588                }
589            }
590
591            Fields::Named(items) => {
592                for (field, value) in items {
593                    let json = value.format_json(w)?;
594                    w.map_push_field(map, field.to_owned(), json)?;
595                }
596            }
597        }
598
599        Ok(())
600    }
601}

603/// Serialize implementation for Value to support serializing the Value to BCS bytes.
604impl Serialize for Value<'_> {
605    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
606    where
607        S: serde::Serializer,
608    {
609        match self {
610            Value::Address(a) => a.serialize(serializer),
611            Value::Bool(b) => b.serialize(serializer),
612            Value::Bytes(b) => b.serialize(serializer),
613            Value::Enum(e) => e.serialize(serializer),
614            Value::Slice(s) => s.serialize(serializer),
615            Value::String(s) => s.serialize(serializer),
616            Value::Struct(s) => s.serialize(serializer),
617            Value::U8(n) => n.serialize(serializer),
618            Value::U16(n) => n.serialize(serializer),
619            Value::U32(n) => n.serialize(serializer),
620            Value::U64(n) => n.serialize(serializer),
621            Value::U128(n) => n.serialize(serializer),
622            Value::U256(n) => n.serialize(serializer),
623            Value::Vector(v) => v.serialize(serializer),
624        }
625    }

628/// This implementation makes it so that serializing a `Slice` to BCS bytes produces the bytes
629/// unchanged (but this property is not guaranteed for any other format).
630impl Serialize for Slice<'_> {
631    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
632    where
633        S: serde::Serializer,
634    {
635        let mut s = serializer.serialize_tuple(self.bytes.len())?;
636        for b in self.bytes {
637            s.serialize_element(b)?;
638        }
639
640        s.end()
641    }
642}
643
644impl Serialize for Vector<'_> {
645    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
646    where
647        S: serde::Serializer,
648    {
649        let mut s = serializer.serialize_seq(Some(self.elements.len()))?;
650        for e in &self.elements {
651            s.serialize_element(e)?;
652        }
653
654        s.end()
655    }
656}
657
658impl Serialize for Struct<'_> {
659    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
660    where
661        S: serde::Serializer,
662    {
663        // Serialize the struct as a tuple, regardless of whether it has named or positional
664        // fields, because `serde`'s field names need to be `&'static str`, which we don't have
665        // (and we don't need).
666        let mut s = serializer.serialize_tuple(self.fields.len())?;
667
668        match &self.fields {
669            // Move values cannot serialize to an empty byte stream, so if there are no fields,
670            // `dummy_field: bool = false` is injected.
671            Fields::Positional(fs) if fs.is_empty() => {
672                s.serialize_element(&false)?;
673            }
674
675            Fields::Named(fs) if fs.is_empty() => {
676                s.serialize_element(&false)?;
677            }
678
679            Fields::Positional(fs) => {
680                for f in fs {
681                    s.serialize_element(f)?;
682                }
683            }
684            Fields::Named(fs) => {
685                for (_, f) in fs {
686                    s.serialize_element(f)?;
687                }
688            }
689        }
690
691        s.end()
692    }
693}
694
695impl Serialize for Enum<'_> {
696    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
697    where
698        S: serde::Serializer,
699    {
700        // Serialize the enum as a tuple, with empty names, for similar reasons as `Struct`, above.
701        let mut s = serializer.serialize_tuple_variant(
702            "",
703            self.variant_index as u32,
704            "",
705            self.fields.len(),
706        )?;
707
708        match &self.fields {
709            Fields::Positional(fs) => {
710                for f in fs {
711                    s.serialize_field(f)?;
712                }
713            }
714            Fields::Named(fs) => {
715                for (_, f) in fs {
716                    s.serialize_field(f)?;
717                }
718            }
719        }
720
721        s.end()
722    }
723}

725impl<'s> TryFrom<Value<'s>> for Atom<'s> {
726    type Error = FormatError;
727
728    fn try_from(value: Value<'s>) -> Result<Atom<'s>, FormatError> {
729        use Atom as A;
730        use MoveTypeLayout as L;
731        use TypeTag as T;
732        use Value as V;
733
734        Ok(match value {
735            V::Address(a) => A::Address(a),
736            V::Bool(b) => A::Bool(b),
737            V::U8(n) => A::U8(n),
738            V::U16(n) => A::U16(n),
739            V::U32(n) => A::U32(n),
740            V::U64(n) => A::U64(n),
741            V::U128(n) => A::U128(n),
742            V::U256(n) => A::U256(n),
743
744            // Byte arrays and strings are indistinguishable at the Atom level
745            V::Bytes(bs) | V::String(bs) => A::Bytes(bs),
746
747            V::Enum(_) => return Err(FormatError::TransformInvalid("unexpected enum")),
748            V::Struct(_) => return Err(FormatError::TransformInvalid("unexpected struct")),
749
750            // Vector literals are supported if they are byte vectors.
751            V::Vector(Vector { type_, elements }) => {
752                if *type_ != T::U8 {
753                    return Err(FormatError::TransformInvalid("unexpected vector"));
754                }
755
756                let bytes: Result<Vec<_>, _> = elements
757                    .into_iter()
758                    .map(|e| match e {
759                        V::U8(b) => Ok(b),
760                        V::Slice(Slice { layout, bytes }) if layout == &L::U8 => {
761                            Ok(bcs::from_bytes(bytes)?)
762                        }
763                        _ => Err(FormatError::TransformInvalid("unexpected vector")),
764                    })
765                    .collect();
766
767                A::Bytes(Cow::Owned(bytes?))
768            }
769
770            V::Slice(Slice { layout, bytes }) => match layout {
771                L::Address => A::Address(bcs::from_bytes(bytes)?),
772                L::Bool => A::Bool(bcs::from_bytes(bytes)?),
773                L::U8 => A::U8(bcs::from_bytes(bytes)?),
774                L::U16 => A::U16(bcs::from_bytes(bytes)?),
775                L::U32 => A::U32(bcs::from_bytes(bytes)?),
776                L::U64 => A::U64(bcs::from_bytes(bytes)?),
777                L::U128 => A::U128(bcs::from_bytes(bytes)?),
778                L::U256 => A::U256(bcs::from_bytes(bytes)?),
779
780                L::Vector(layout) if layout.as_ref() == &L::U8 => {
781                    A::Bytes(Cow::Borrowed(bcs::from_bytes(bytes)?))
782                }
783
784                L::Struct(layout)
785                    if [
786                        move_ascii_str_layout(),
787                        move_utf8_str_layout(),
788                        url_layout(),
789                    ]
790                    .contains(layout.as_ref()) =>
791                {
792                    A::Bytes(Cow::Borrowed(bcs::from_bytes(bytes)?))
793                }
794
795                L::Struct(layout) if [UID::layout(), ID::layout()].contains(layout.as_ref()) => {
796                    A::Address(bcs::from_bytes(bytes)?)
797                }
798
799                L::Signer => return Err(FormatError::TransformInvalid("unexpected signer")),
800                L::Enum(_) => return Err(FormatError::TransformInvalid("unexpected enum")),
801                L::Struct(_) => return Err(FormatError::TransformInvalid("unexpected struct")),
802                L::Vector(_) => return Err(FormatError::TransformInvalid("unexpected vector")),
803            },
804        })
805    }
806}

960        type_: &str,
961        variants: Vec<(&str, Vec<(&str, MoveTypeLayout)>)>,
962    ) -> MoveTypeLayout {
963        let type_: StructTag = type_.parse().unwrap();
964        let variants = variants
965            .into_iter()
966            .enumerate()
967            .map(|(tag, (name, fields))| {
968                let fields = fields
969                    .into_iter()

972                    })
973                    .collect();
974
975                ((Identifier::new(name).unwrap(), tag as u16), fields)
976            })
977            .collect();
978
979        MoveTypeLayout::Enum(Box::new(MoveEnumLayout { type_, variants }))
980    }
