4use move_core_types::account_address::AccountAddress;
5use move_core_types::annotated_value as A;
6use move_core_types::annotated_visitor as AV;
7use move_core_types::u256::U256;
8use move_core_types::visitor_default;
9use sui_types::id::ID;
10use sui_types::id::UID;
11
12use crate::v2::error::FormatError;
13
14/// Visitor to extract addresses from Objects, UIDs, IDs, or addresses.
15pub(crate) struct AddressExtractor;
16
17impl AV::Visitor<'_, '_> for AddressExtractor {
18    type Value = Option<AccountAddress>;
19    type Error = FormatError;
20
21    visitor_default! { <'_, '_> u8, u16, u32, u64, u128, u256 = Ok(None) }
22    visitor_default! { <'_, '_> bool, signer, vector, variant = Ok(None) }
23
24    fn visit_address(
25        &mut self,
26        _: &AV::ValueDriver<'_, '_, '_>,
27        a: AccountAddress,
28    ) -> Result<Self::Value, Self::Error> {
29        Ok(Some(a))
30    }
31
32    fn visit_struct(
33        &mut self,
34        driver: &mut AV::StructDriver<'_, '_, '_>,
35    ) -> Result<Self::Value, Self::Error> {
36        let id = ID::layout();
37        let uid = UID::layout();
38        let layout = driver.struct_layout();
39
40        // Detect an inline `ID` or `UID`.
41        if layout == &id || layout == &uid {
42            return extract_address(driver).map(Some);
43        }
44
45        // Otherwise assume the value is an Object and look for `id: UID` as its first field.
46        let Some(field) = driver.peek_field() else {
47            return Ok(None);
48        };
49
50        if field.name.as_str() != "id" {
51            return Ok(None);
52        }
53
54        let A::MoveTypeLayout::Struct(field) = &field.layout else {
55            return Ok(None);
56        };
57
58        if field.as_ref() != &uid {
59            return Ok(None);
60        }
61
62        extract_address(driver).map(Some)
63    }
64}
65
66/// Attempts to extract an address from the prefix of the struct being visited. Assumes the driver
67/// is pointed at the first field of the struct on entry.
68fn extract_address(
69    driver: &mut AV::StructDriver<'_, '_, '_>,
70) -> Result<AccountAddress, FormatError> {
71    driver.skip_field()?;
72    let bytes = &driver.bytes()[driver.start()..driver.position()];
73    Ok(bcs::from_bytes(bytes)?)
74}
