25/// Abstracts over access to the VM across versions of the execution layer.
26pub trait Executor {
27    fn execute_transaction_to_effects(
28        &self,
29        store: &dyn BackingStore,
30        // Configuration
31        protocol_config: &ProtocolConfig,
32        metrics: Arc<LimitsMetrics>,
33        enable_expensive_checks: bool,
34        execution_params: ExecutionOrEarlyError,
35        // Epoch
36        epoch_id: &EpochId,
37        epoch_timestamp_ms: u64,
38        // Transaction Inputs
39        input_objects: CheckedInputObjects,
40        // Gas related
41        gas: GasData,
42        gas_status: SuiGasStatus,
43        // Transaction
44        transaction_kind: TransactionKind,
45        transaction_signer: SuiAddress,
46        transaction_digest: TransactionDigest,
47        trace_builder_opt: &mut Option<MoveTraceBuilder>,
48    ) -> (
49        InnerTemporaryStore,
50        SuiGasStatus,
51        TransactionEffects,
52        Vec<ExecutionTiming>,
53        Result<(), ExecutionError>,
54    );
55
56    fn dev_inspect_transaction(
57        &self,
58        store: &dyn BackingStore,
59        // Configuration
60        protocol_config: &ProtocolConfig,
61        metrics: Arc<LimitsMetrics>,
62        enable_expensive_checks: bool,
63        execution_params: ExecutionOrEarlyError,
64        // Epoch
65        epoch_id: &EpochId,
66        epoch_timestamp_ms: u64,
67        // Transaction Inputs
68        input_objects: CheckedInputObjects,
69        // Gas related
70        gas: GasData,
71        gas_status: SuiGasStatus,
72        // Transaction
73        transaction_kind: TransactionKind,
74        transaction_signer: SuiAddress,
75        transaction_digest: TransactionDigest,
76        skip_all_checks: bool,
77    ) -> (
78        InnerTemporaryStore,
79        SuiGasStatus,
80        TransactionEffects,
81        Result<Vec<ExecutionResult>, ExecutionError>,
82    );
83
84    fn update_genesis_state(
85        &self,
86        store: &dyn BackingStore,
87        // Configuration
88        protocol_config: &ProtocolConfig,
89        metrics: Arc<LimitsMetrics>,
90        // Epoch
91        epoch_id: EpochId,
92        epoch_timestamp_ms: u64,
93        // Genesis Digest
94        transaction_digest: &TransactionDigest,
95        // Transaction
96        input_objects: CheckedInputObjects,
97        pt: ProgrammableTransaction,
98    ) -> Result<InnerTemporaryStore, ExecutionError>;
99
100    fn type_layout_resolver<'r, 'vm: 'r, 'store: 'r>(
101        &'vm self,
102        store: Box<dyn TypeLayoutStore + 'store>,
103    ) -> Box<dyn LayoutResolver + 'r>;
104}
