64impl executor::Executor for Executor {
65    fn execute_transaction_to_effects(
66        &self,
67        store: &dyn BackingStore,
68        protocol_config: &ProtocolConfig,
69        metrics: Arc<LimitsMetrics>,
70        enable_expensive_checks: bool,
71        execution_params: ExecutionOrEarlyError,
72        epoch_id: &EpochId,
73        epoch_timestamp_ms: u64,
74        input_objects: CheckedInputObjects,
75        gas: GasData,
76        gas_status: SuiGasStatus,
77        transaction_kind: TransactionKind,
78        transaction_signer: SuiAddress,
79        transaction_digest: TransactionDigest,
80        _trace_builder_opt: &mut Option<MoveTraceBuilder>,
81    ) -> (
82        InnerTemporaryStore,
83        SuiGasStatus,
84        TransactionEffects,
85        Vec<ExecutionTiming>,
86        Result<(), ExecutionError>,
87    ) {
88        let gas_coins = gas.payment;
89        let (inner_temp_store, gas_status, effects, result) =
90            execute_transaction_to_effects::<execution_mode::Normal>(
91                store,
92                input_objects,
93                gas_coins,
94                gas_status,
95                transaction_kind,
96                transaction_signer,
97                transaction_digest,
98                &self.0,
99                epoch_id,
100                epoch_timestamp_ms,
101                protocol_config,
102                metrics,
103                enable_expensive_checks,
104                execution_params,
105            );
106        // note: old versions do not report timings.
107        (inner_temp_store, gas_status, effects, vec![], result)
108    }
109
110    fn dev_inspect_transaction(
111        &self,
112        store: &dyn BackingStore,
113        protocol_config: &ProtocolConfig,
114        metrics: Arc<LimitsMetrics>,
115        enable_expensive_checks: bool,
116        execution_params: ExecutionOrEarlyError,
117        epoch_id: &EpochId,
118        epoch_timestamp_ms: u64,
119        input_objects: CheckedInputObjects,
120        gas: GasData,
121        gas_status: SuiGasStatus,
122        transaction_kind: TransactionKind,
123        transaction_signer: SuiAddress,
124        transaction_digest: TransactionDigest,
125        skip_all_checks: bool,
126    ) -> (
127        InnerTemporaryStore,
128        SuiGasStatus,
129        TransactionEffects,
130        Result<Vec<ExecutionResult>, ExecutionError>,
131    ) {
132        let gas_coins = gas.payment;
133        if skip_all_checks {
134            execute_transaction_to_effects::<execution_mode::DevInspect<true>>(
135                store,
136                input_objects,
137                gas_coins,
138                gas_status,
139                transaction_kind,
140                transaction_signer,
141                transaction_digest,
142                &self.0,
143                epoch_id,
144                epoch_timestamp_ms,
145                protocol_config,
146                metrics,
147                enable_expensive_checks,
148                execution_params,
149            )
150        } else {
151            execute_transaction_to_effects::<execution_mode::DevInspect<false>>(
152                store,
153                input_objects,
154                gas_coins,
155                gas_status,
156                transaction_kind,
157                transaction_signer,
158                transaction_digest,
159                &self.0,
160                epoch_id,
161                epoch_timestamp_ms,
162                protocol_config,
163                metrics,
164                enable_expensive_checks,
165                execution_params,
166            )
167        }
168    }
169
170    fn update_genesis_state(
171        &self,
172        store: &dyn BackingStore,
173        protocol_config: &ProtocolConfig,
174        metrics: Arc<LimitsMetrics>,
175        epoch_id: EpochId,
176        epoch_timestamp_ms: u64,
177        transaction_digest: &TransactionDigest,
178        input_objects: CheckedInputObjects,
179        pt: ProgrammableTransaction,
180    ) -> Result<InnerTemporaryStore, ExecutionError> {
181        let tx_context = &mut TxContext::new_from_components(
182            &SuiAddress::default(),
183            transaction_digest,
184            &epoch_id,
185            epoch_timestamp_ms,
186            // genesis transaction: RGP: 1, budget: 1M, sponsor: None
187            // Those values are unused anyway in execution versions before 3 (or latest)
188            1,
189            1,
190            1_000_000,
191            None,
192            protocol_config,
193        );
194        execute_genesis_state_update(
195            store,
196            protocol_config,
197            metrics,
198            &self.0,
199            tx_context,
200            input_objects,
201            pt,
202        )
203    }
204
205    fn type_layout_resolver<'r, 'vm: 'r, 'store: 'r>(
206        &'vm self,
207        store: Box<dyn TypeLayoutStore + 'store>,
208    ) -> Box<dyn LayoutResolver + 'r> {
209        Box::new(TypeLayoutResolver::new(&self.0, store))
210    }
211}

213impl verifier::Verifier for Verifier<'_> {
214    fn meter(&self, config: MeterConfig) -> Box<dyn Meter> {
215        Box::new(SuiVerifierMeter::new(config))
216    }
217
218    fn override_deprecate_global_storage_ops_during_deserialization(&self) -> Option<bool> {
219        None
220    }
221
222    fn meter_compiled_modules(
223        &mut self,
224        _protocol_config: &ProtocolConfig,
225        modules: &[CompiledModule],
226        meter: &mut dyn Meter,
227    ) -> SuiResult<()> {
228        run_metered_move_bytecode_verifier(modules, &self.config, meter, self.metrics)
229    }
