4//! The reconstructed execution context: the per-epoch [`EpochCtx`] (version-correct executor,
5//! protocol config, epoch-start timestamp, RGP) and the per-checkpoint [`PreparedCheckpoint`] that
6//! execution runs against, plus [`resolve_epoch_work`] which resolves an epoch range into the
7//! [`EpochCtx`] map and the overall checkpoint range to hand to the framework `Indexer`. The
8//! ingestion and concurrency machinery itself lives in that `Indexer` (see [`crate::handler`]).
9

27/// Per-epoch resolved context, shared (by Arc) across all of the epoch's checkpoint workers.
28pub(crate) struct EpochCtx {
29    pub(crate) epoch: u64,
30    pub(crate) protocol_config: Arc<ProtocolConfig>,
31    /// Version-correct executor for full execution. Owns its own `MoveRuntime`.
32    pub(crate) executor: Arc<dyn Executor + Send + Sync>,
33    /// Execution metrics (shared across all epochs).
34    pub(crate) metrics: Arc<ExecutionMetrics>,
35    /// The epoch's start timestamp (the first checkpoint's `timestamp_ms`), which is the value the
36    /// executor expects for `epoch_timestamp_ms` (constant across the epoch, as `TxContext` sees it
37    /// on chain) — *not* the per-checkpoint timestamp.
38    pub(crate) epoch_start_timestamp_ms: u64,
39    /// The epoch's reference gas price, used to meter execution faithfully (the gas accounting and
40    /// `TxContext` differ from passing the tx's own price as the RGP).
41    pub(crate) reference_gas_price: u64,
42    /// The system (framework) packages live during this epoch, keyed by id. Unlike normal packages,
43    /// these keep a stable id but are upgraded (new bytecode) across protocol versions, so they
44    /// can't be fetched "latest" and replayed faithfully — they are loaded version-correctly from
45    /// the framework snapshot for the epoch's protocol version (see [`load_epoch_system_packages`]).
46    pub(crate) system_packages: Arc<BTreeMap<ObjectID, Object>>,
47}
48
49/// A checkpoint that has been indexed and is ready to execute. Holds only cheap handles (Arcs); the
50/// bulky object index is shared.
51pub(crate) struct PreparedCheckpoint {
52    pub(crate) cp: u64,
53    pub(crate) ctx: Arc<EpochCtx>,
54    pub(crate) chain_id: ChainIdentifier,
55    /// The full checkpoint (shared); we iterate its `transactions` during execution.
56    pub(crate) checkpoint: Arc<Checkpoint>,
57    /// The checkpoint's object index, shared with the execute stage's
58    /// [`ScanStore`](crate::store::ScanStore). Read directly to find candidate coin types and to
59    /// triage divergences against the materialized read set.
60    pub(crate) objects: Arc<BTreeMap<ObjectKey, Object>>,
61}

63/// Resolve every epoch in `epochs` to its checkpoint range + protocol version, building a per-epoch
64/// [`EpochCtx`] (executor, protocol config, epoch-start timestamp, RGP). Returns the epoch→ctx map
65/// together with the inclusive checkpoint range spanning all of the epochs (handed to the indexer's
66/// ingestion service). `first_bounds` is the already-resolved bounds for the first epoch (so it
67/// isn't fetched twice). `max_checkpoints_per_epoch` caps each epoch from its first checkpoint.
68pub(crate) async fn resolve_epoch_work(
69    rpc: &RpcClient,
70    chain: Chain,
71    epochs: RangeInclusive<u64>,
72    max_checkpoints_per_epoch: Option<u64>,
73    first_bounds: EpochBounds,
74    execution_metrics: &Arc<ExecutionMetrics>,
75) -> Result<(BTreeMap<u64, Arc<EpochCtx>>, u64, u64)> {
76    // System packages are reconstructed from the framework snapshot, which only faithfully
77    // represents released networks (mainnet/testnet). A non-prod chain (devnet/localnet/custom, all
78    // `Chain::Unknown`) may run framework bytecode ahead of, or unrelated to, any snapshot, so the
79    // reconstructed system packages — and thus the backtest — can silently diverge from reality (see
80    // [`load_epoch_system_packages`]).
81    if chain == Chain::Unknown {
82        warn!(
83            "backtesting a non-prod network (chain id resolves to Unknown): system packages are \
84             reconstructed from the framework snapshot and may not match the network's actual \
85             framework, so results may be unreliable"
86        );
87    }
88    let start_epoch = *epochs.start();
89    let mut epoch_ctxs: BTreeMap<u64, Arc<EpochCtx>> = BTreeMap::new();
90    let mut first_checkpoint = u64::MAX;
91    let mut last_checkpoint = 0u64;
92    for epoch in epochs {
93        let bounds = if epoch == start_epoch {
94            first_bounds
95        } else {
96            rpc.epoch_bounds(epoch)
97                .await
98                .with_context(|| format!("resolving epoch {epoch}"))?
99        };
100        let protocol_config =
101            ProtocolConfig::get_for_version(ProtocolVersion::new(bounds.protocol_version), chain);
102        let executor = sui_execution::executor(&protocol_config, /* silent */ true)
103            .map_err(|e| anyhow::anyhow!("building executor for epoch {epoch}: {e}"))?;
104        let system_packages = Arc::new(
105            load_epoch_system_packages(bounds.protocol_version)
106                .with_context(|| format!("resolving framework packages for epoch {epoch}"))?,
107        );
108        let ctx = Arc::new(EpochCtx {
109            epoch,
110            protocol_config: Arc::new(protocol_config),
111            executor,
112            metrics: execution_metrics.clone(),
113            epoch_start_timestamp_ms: bounds.epoch_start_timestamp_ms,
114            reference_gas_price: bounds.reference_gas_price,
115            system_packages,
116        });
117
118        let last = match max_checkpoints_per_epoch {
119            Some(cap) => bounds.last_checkpoint.min(
120                bounds
121                    .first_checkpoint
122                    .saturating_add(cap)
123                    .saturating_sub(1),
124            ),
125            None => bounds.last_checkpoint,
126        };
127        let count = last
128            .saturating_sub(bounds.first_checkpoint)
129            .saturating_add(1);
130        info!(
131            epoch,
132            first_checkpoint = bounds.first_checkpoint,
133            last_checkpoint = last,
134            checkpoints = count,
135            protocol_version = bounds.protocol_version,
136            "queued epoch"
137        );
138        first_checkpoint = first_checkpoint.min(bounds.first_checkpoint);
139        last_checkpoint = last_checkpoint.max(last);
140        epoch_ctxs.insert(epoch, ctx);
141    }
142    Ok((epoch_ctxs, first_checkpoint, last_checkpoint))
143}

145/// The system (framework) packages live at `protocol_version`, as package `Object`s keyed by id.
146/// `load_bytecode_snapshot` resolves the sparse protocol-version → snapshot mapping itself (the
147/// greatest snapshot at or below the version). The reconstructed object's version number is
148/// synthetic (`OBJECT_START_VERSION`); only its bytecode and linkage matter for execution, and
149/// those resolve system packages by id, not version.
150///
151/// This is faithful only for released networks (mainnet/testnet), whose framework is exactly what a
152/// published snapshot captured. On devnet the running framework can be *ahead* of any snapshot (so
153/// the greatest-snapshot-≤-version lookup returns stale bytecode), and on localnet it is whatever
154/// was built locally (no snapshot corresponds to it at all) — in both cases the reconstruction is
155/// silently wrong. [`resolve_epoch_work`] warns when the chain is non-prod (`Chain::Unknown`).
156///
157/// TODO: If a GraphQL checkpoint source is added, revisit this — GraphQL can read the actual
158///       framework objects at a checkpoint (as `sui-replay-2` does), which is more principled than
159///       reconstructing them from the snapshot and would also fix the non-prod case above.
160fn load_epoch_system_packages(protocol_version: u64) -> Result<BTreeMap<ObjectID, Object>> {
161    let packages =
162        sui_framework_snapshot::load_bytecode_snapshot(protocol_version).with_context(|| {
163            format!("loading framework snapshot for protocol version {protocol_version}")
164        })?;
165    Ok(packages
166        .into_iter()
167        .map(|package| {
168            let object = package.genesis_object();
169            (object.id(), object)
170        })
171        .collect())
172}
