4//! Thin gRPC client over a Sui fullnode, used for the two things the checkpoint stream does not
5//! already provide: resolving an epoch to its checkpoint range + protocol version, and fetching
6//! package objects that are not present in a checkpoint's object set.

26/// The checkpoint range and protocol parameters for a single epoch.
27#[derive(Clone, Copy, Debug)]
28pub struct EpochBounds {
29    pub first_checkpoint: u64,
30    pub last_checkpoint: u64,
31    pub protocol_version: u64,
32    pub reference_gas_price: u64,
33    /// The epoch's start timestamp (ms), i.e. its first checkpoint's `timestamp_ms` — the value the
34    /// executor expects for `epoch_timestamp_ms`. Sourced directly from `GetEpoch` so we don't have
35    /// to fetch the first checkpoint just to read its timestamp.
36    pub epoch_start_timestamp_ms: u64,
37}
38
39#[derive(Clone)]
40pub struct RpcClient {
41    client: sui_rpc::client::Client,
42}
43
44impl RpcClient {
45    pub fn new(url: Url) -> Result<Self> {
46        Ok(Self {
47            client: sui_rpc::client::Client::new(url.to_string())
48                .context("Failed to construct gRPC client")?,
49        })
50    }
51
52    /// Resolve `epoch` to its first/last checkpoint, protocol version, and reference gas price.
53    pub async fn epoch_bounds(&self, epoch: u64) -> Result<EpochBounds> {
54        let mut ledger = self.client.clone().ledger_client();
55        let request = rpc::GetEpochRequest::new(epoch).with_read_mask(FieldMask::from_paths([
56            "first_checkpoint",
57            "last_checkpoint",
58            "protocol_config.protocol_version",
59            "reference_gas_price",
60            "start",
61        ]));
62        let response = ledger
63            .get_epoch(request)
64            .await
65            .with_context(|| format!("get_epoch({epoch}) failed"))?
66            .into_inner();
67        let epoch_msg = response
68            .epoch
69            .with_context(|| format!("get_epoch({epoch}) returned no epoch"))?;
70        let protocol_version = epoch_msg
71            .protocol_config
72            .as_ref()
73            .and_then(|c| c.protocol_version)
74            .with_context(|| format!("epoch {epoch} missing protocol_version"))?;
75        let start = epoch_msg
76            .start
77            .with_context(|| format!("epoch {epoch} missing start timestamp"))?;
78        let epoch_start_timestamp_ms = proto_to_timestamp_ms(start)
79            .with_context(|| format!("decoding epoch {epoch} start timestamp"))?;
80        Ok(EpochBounds {
81            first_checkpoint: epoch_msg
82                .first_checkpoint
83                .with_context(|| format!("epoch {epoch} missing first_checkpoint"))?,
84            last_checkpoint: epoch_msg
85                .last_checkpoint
86                .with_context(|| format!("epoch {epoch} missing last_checkpoint"))?,
87            protocol_version,
88            reference_gas_price: epoch_msg.reference_gas_price.unwrap_or(0),
89            epoch_start_timestamp_ms,
90        })
91    }
92
93    /// The network's chain identifier, via the fullnode's `GetServiceInfo` (cheap). The remote
94    /// object store would otherwise derive this by fetching + decoding genesis (checkpoint 0), a
95    /// large, slow, one-time startup cost; this avoids it (see [`crate::ingestion`]).
96    pub async fn chain_id(&self) -> Result<ChainIdentifier> {
97        let mut ledger = self.client.clone().ledger_client();
98        let response = ledger
99            .get_service_info(rpc::GetServiceInfoRequest::default())
100            .await
101            .context("get_service_info failed")?
102            .into_inner();
103        Ok(CheckpointDigest::from_str(response.chain_id())?.into())
104    }
105
106    /// Fetch a package object by id at its latest version. Non-system packages are immutable, so the
107    /// latest version is the one that executed against it; system packages (versioned per epoch) are
108    /// NOT fetched here — they are resolved version-correctly from the framework snapshot (see
109    /// [`crate::context`]). Returns `None` if the fullnode does not have it. Retries transient errors (notably HTTP 429
110    /// rate-limiting) with exponential backoff, because a dropped package fetch would otherwise
111    /// surface as a spurious execution failure rather than a real result.
112    pub async fn fetch_object(&self, id: ObjectID) -> Result<Option<Object>> {
113        let response = retry_transient(|| async {
114            let mut ledger = self.client.clone().ledger_client();
115            let request = rpc::GetObjectRequest::new(&id.into_bytes().into())
116                .with_read_mask(FieldMask::from_paths(["bcs"]));
117            ledger.get_object(request).await.map(|r| r.into_inner())
118        })
119        .await
120        .with_context(|| format!("get_object({id}) failed"))?;
121        let Some(bytes) = response.object.and_then(|o| o.bcs).and_then(|b| b.value) else {
122            return Ok(None);
123        };
124        let object: Object =
125            bcs::from_bytes(&bytes).with_context(|| format!("decoding object {id}"))?;
126        Ok(Some(object))
127    }
128
129    /// Batch-fetch objects by id with a chunked multi-get, returning the decoded objects that were
130    /// found (missing ids and per-object errors are silently omitted — used to warm the package
131    /// cache, where a miss just falls through to the lazy single fetch). Chunks are fetched
132    /// concurrently; each chunk retries transient errors like [`Self::fetch_object`].
133    pub async fn fetch_objects(&self, ids: &[ObjectID]) -> Result<Vec<Object>> {
134        // Each result carries the full BCS object, and packages can be ~MBs, so keep chunks well
135        // under the 128 MiB gRPC message cap.
136        const CHUNK: usize = 50;
137        // How many chunk requests to keep in flight at once.
138        const CONCURRENCY: usize = 8;
139        let chunks: Vec<Vec<ObjectID>> = ids.chunks(CHUNK).map(|chunk| chunk.to_vec()).collect();
140        let batches: Vec<Vec<Object>> = futures::stream::iter(chunks)
141            .map(|chunk| self.fetch_object_chunk(chunk))
142            .buffer_unordered(CONCURRENCY)
143            .try_collect()
144            .await?;
145        Ok(batches.into_iter().flatten().collect())
146    }
147
148    /// Fetch and decode one chunk of objects for [`Self::fetch_objects`], retrying transient errors.
149    /// Takes an owned `chunk` so the returned future borrows only `&self` (not the caller's slice),
150    /// which lets the chunk futures be driven concurrently by `buffer_unordered`.
151    async fn fetch_object_chunk(&self, chunk: Vec<ObjectID>) -> Result<Vec<Object>> {
152        let response = retry_transient(|| async {
153            let mut ledger = self.client.clone().ledger_client();
154            let mut request = rpc::BatchGetObjectsRequest::default()
155                .with_read_mask(FieldMask::from_paths(["bcs"]));
156            for id in &chunk {
157                request
158                    .requests_mut()
159                    .push(rpc::GetObjectRequest::new(&id.into_bytes().into()));
160            }
161            ledger
162                .batch_get_objects(request)
163                .await
164                .map(|r| r.into_inner())
165        })
166        .await
167        .with_context(|| format!("batch_get_objects({} ids) failed", chunk.len()))?;
168        let mut out = Vec::new();
169        for result in response.objects {
170            let Ok(object) = result.to_result() else {
171                continue; // per-object NotFound / error: skip, let the lazy path handle it.
172            };
173            let Some(bytes) = object.bcs.and_then(|b| b.value) else {
174                continue;
175            };
176            match bcs::from_bytes::<Object>(&bytes) {
177                Ok(object) => out.push(object),
178                Err(e) => warn!("decoding batched object: {e:#}"),
179            }
180        }
181        Ok(out)
182    }
183}

185/// Exponential backoff (with the crate's built-in jitter) for transient gRPC failures: interval
186/// caps at 5s and total elapsed at 60s, so a persistently-unreachable node surfaces an error rather
187/// than hanging indefinitely.
188fn transient_backoff() -> ExponentialBackoff {
189    ExponentialBackoff {
190        max_interval: Duration::from_secs(5),
191        max_elapsed_time: Some(Duration::from_secs(60)),
192        ..Default::default()
193    }
194}

196/// Run a fallible gRPC call under [`transient_backoff`], retrying only transient errors (see
197/// [`is_transient`]); any other error is returned immediately.
198async fn retry_transient<T, F, Fut>(make_future: F) -> Result<T, tonic::Status>
199where
200    F: Fn() -> Fut,
201    Fut: Future<Output = Result<T, tonic::Status>>,
202{
203    retry(transient_backoff(), || async {
204        make_future().await.map_err(|status| {
205            if is_transient(&status) {
206                backoff::Error::transient(status)
207            } else {
208                backoff::Error::permanent(status)
209            }
210        })
211    })
212    .await
213}

215/// Whether a gRPC error is worth retrying — rate limiting (429) and transient unavailability.
216fn is_transient(status: &tonic::Status) -> bool {
217    use tonic::Code;
218    matches!(
219        status.code(),
220        Code::ResourceExhausted | Code::Unavailable | Code::Aborted | Code::DeadlineExceeded
221    ) || status.message().contains("429")
222}
