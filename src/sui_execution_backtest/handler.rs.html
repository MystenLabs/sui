4//! The backtest as a single `sui-indexer-alt-framework` concurrent pipeline. The framework's
5//! `Indexer` drives ingestion (bounded range, adaptive concurrency) and runs [`Backtest::process`]
6//! per checkpoint with a configurable fanout; the processor re-executes the checkpoint's
7//! transactions serially on a blocking worker and emits typed rows. The sink is abstracted behind
8//! [`CommitRows`] so the same pipeline writes either to postgres (durable, queryable) or to an
9//! ndjson file (zero-setup) — selected by `--store`.
10

35/// Cap on rows per committed batch. Bounds postgres bind parameters (≤ ~13 columns × this < the
36/// 32,767 limit) and keeps batches small; the framework also commits on its own size/time
37/// thresholds.
38const MAX_BATCH_ROWS: usize = 2000;
39
40/// A row the pipeline produces. Both row types ride the same `process` output (the checkpoint is
41/// executed once) and are split apart at commit time.
42pub(crate) enum Row {
43    Divergence(DivergenceRow),
44    Stats(RunStatsRow),
45}
46
47/// Accumulated rows for one committed batch, split by destination table.
48#[derive(Default)]
49pub(crate) struct BacktestBatch {
50    divergences: Vec<DivergenceRow>,
51    stats: Vec<RunStatsRow>,
52}
53
54impl BacktestBatch {
55    fn len(&self) -> usize {
56        self.divergences.len() + self.stats.len()
57    }
58}
59
60/// Store-specific write of a batch. Defined here (locally) so it can be implemented for both the
61/// foreign postgres `Db` and our `NdjsonStore` without an orphan-rule conflict.
62#[async_trait]
63pub(crate) trait CommitRows: ConcurrentStore {
64    async fn commit_rows(conn: &mut Self::Connection<'_>, batch: &BacktestBatch) -> Result<usize>;
65}
66
67#[async_trait]
68impl CommitRows for Db {
69    async fn commit_rows(conn: &mut Connection<'_>, batch: &BacktestBatch) -> Result<usize> {
70        let mut affected = 0;
71        if !batch.divergences.is_empty() {
72            affected += diesel::insert_into(divergence::table)
73                .values(&batch.divergences)
74                .on_conflict_do_nothing()
75                .execute(conn)
76                .await?;
77        }
78        if !batch.stats.is_empty() {
79            affected += diesel::insert_into(run_stats::table)
80                .values(&batch.stats)
81                .on_conflict_do_nothing()
82                .execute(conn)
83                .await?;
84        }
85        Ok(affected)
86    }
87}
88
89#[async_trait]
90impl CommitRows for NdjsonStore {
91    async fn commit_rows(conn: &mut NdjsonConnection<'_>, batch: &BacktestBatch) -> Result<usize> {
92        let store = conn.store();
93        let mut written = store.write_rows(&batch.divergences)?;
94        written += store.write_rows(&batch.stats)?;
95        // Flush per commit so partial results survive an interrupted run.
96        store.flush()?;
97        Ok(written)
98    }
99}

101/// The backtest pipeline, generic over its sink `S`. Holds the per-epoch execution contexts and the
102/// shared package cache; `process` looks up the checkpoint's epoch context and re-executes it.
103pub(crate) struct Backtest<S> {
104    epochs: Arc<BTreeMap<u64, Arc<EpochCtx>>>,
105    packages: Arc<PackageCache>,
106    chain_id: ChainIdentifier,
107    status: StatusFilter,
108    task: String,
109    stats_enabled: bool,
110    _store: PhantomData<fn() -> S>,
111}
112
113impl<S> Backtest<S> {
114    pub(crate) fn new(
115        epochs: Arc<BTreeMap<u64, Arc<EpochCtx>>>,
116        packages: Arc<PackageCache>,
117        chain_id: ChainIdentifier,
118        status: StatusFilter,
119        task: String,
120        stats_enabled: bool,
121    ) -> Self {
122        Self {
123            epochs,
124            packages,
125            chain_id,
126            status,
127            task,
128            stats_enabled,
129            _store: PhantomData,
130        }
131    }
132
133    /// Prefetch the checkpoint's package closure, then re-execute its transactions serially on a
134    /// blocking worker, returning the divergence rows plus (unless disabled) a per-checkpoint stats
135    /// row. Never errors: per-transaction reconstruction/replay failures are counted, and a panic
136    /// in the blocking worker is logged and treated as an empty result, so one bad checkpoint can't
137    /// wedge the (retry-forever) pipeline.
138    async fn run(&self, checkpoint: &Arc<Checkpoint>) -> Vec<Row> {
139        let epoch = checkpoint.summary.epoch;
140        let cp = checkpoint.summary.sequence_number;
141        let Some(ctx) = self.epochs.get(&epoch).cloned() else {
142            error!(
143                epoch,
144                cp, "no epoch context resolved for checkpoint; skipping"
145            );
146            return Vec::new();
147        };
148
149        // Warm the shared package cache with this checkpoint's closure before the synchronous
150        // execute stage (a miss falls through to the lazy fetch).
151        crate::store::prefetch_package_closure(checkpoint, &self.packages).await;
152
153        let (objects, latest) = ScanStore::index_object_set(&checkpoint.object_set);
154        let tombstones = gather_tombstones(checkpoint);
155        let objects = Arc::new(objects);
156        let store = Arc::new(ScanStore::new(
157            objects.clone(),
158            Arc::new(latest),
159            Arc::new(tombstones),
160            self.packages.clone(),
161            ctx.system_packages.clone(),
162        ));
163        let prepared = Arc::new(PreparedCheckpoint {
164            cp,
165            ctx,
166            chain_id: self.chain_id,
167            checkpoint: checkpoint.clone(),
168            objects,
169        });
170
171        let status = self.status;
172        let task = self.task.clone();
173        let count = prepared.checkpoint.transactions.len();
174        let agg = tokio::task::spawn_blocking(move || {
175            let mut agg = CheckpointStats::default();
176            for i in 0..count {
177                agg.merge(execute_one_transaction(&prepared, &store, i, status, &task));
178            }
179            agg
180        })
181        .await
182        .unwrap_or_else(|e| {
183            error!(epoch, cp, "execution task panicked: {e}");
184            CheckpointStats::default()
185        });
186
187        let mut rows: Vec<Row> = agg.records.into_iter().map(Row::Divergence).collect();
188        if self.stats_enabled {
189            rows.push(Row::Stats(RunStatsRow {
190                task: self.task.clone(),
191                epoch: epoch as i64,
192                checkpoint: cp as i64,
193                checked: agg.checked as i64,
194                executed: agg.executed as i64,
195                divergences: agg.divergences as i64,
196                reconstruction_errors: agg.reconstruction_errors as i64,
197                coin_reservation_skipped: agg.coin_reservation_skipped as i64,
198                execute_skipped: agg.execute_skipped as i64,
199                gas_from_balance: agg.gas_from_balance as i64,
200                cancellation_excluded: agg.cancellation_excluded as i64,
201            }));
202        }
203        rows
204    }
205}

207#[async_trait]
208impl<S: CommitRows> Processor for Backtest<S> {
209    const NAME: &'static str = "backtest";
210    type Value = Row;
211
212    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> Result<Vec<Row>> {
213        Ok(self.run(checkpoint).await)
214    }
215}
216
217#[async_trait]
218impl<S: CommitRows> Handler for Backtest<S> {
219    type Store = S;
220    type Batch = BacktestBatch;
221
222    fn batch(
223        &self,
224        batch: &mut BacktestBatch,
225        values: &mut std::vec::IntoIter<Row>,
226    ) -> BatchStatus {
227        for row in values.by_ref() {
228            match row {
229                Row::Divergence(d) => batch.divergences.push(d),
230                Row::Stats(s) => batch.stats.push(s),
231            }
232            if batch.len() >= MAX_BATCH_ROWS {
233                return BatchStatus::Ready;
234            }
235        }
236        BatchStatus::Pending
237    }
238
239    async fn commit<'a>(
240        &self,
241        batch: &BacktestBatch,
242        conn: &mut <S as Store>::Connection<'a>,
243    ) -> Result<usize> {
244        S::commit_rows(conn, batch).await
245    }
246}

248/// Tombstones (deleted/wrapped/unwrapped-then-deleted) across all of the checkpoint's transactions,
249/// so child reads can tell a dynamic field was removed even though the bundled object set still
250/// carries a stale pre-deletion version of it.
251fn gather_tombstones(checkpoint: &Checkpoint) -> BTreeMap<ObjectID, BTreeSet<SequenceNumber>> {
252    let mut tombstones: BTreeMap<ObjectID, BTreeSet<SequenceNumber>> = BTreeMap::new();
253    for executed in &checkpoint.transactions {
254        for (id, ver) in executed.effects.all_tombstones() {
255            tombstones.entry(id).or_default().insert(ver);
256        }
257    }
258    tombstones
259}
