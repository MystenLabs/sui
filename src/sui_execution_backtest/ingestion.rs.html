4//! A remote-object-store ingestion client whose chain identifier is supplied up front instead of
5//! derived by fetching genesis.
6//!
7//! The framework's store client derives the chain id with `checkpoint(0)` — fetching and decoding
8//! mainnet genesis. Because every checkpoint fetch `try_join`s the chain-id `OnceCell` and the
9//! derivation is slow (and repeatedly retried), this throttles the whole ingestion pipeline well
10//! past startup, not just once — measured (mainnet, remote-store source) as a sustained ~4x
11//! throughput drop. We already know the chain id cheaply from the fullnode's
12//! `GetServiceInfo` (see [`crate::grpc::RpcClient::chain_id`]), so we wrap the store client and
13//! return that, never touching genesis. Only the remote-store source has this problem; the gRPC
14//! source already uses `GetServiceInfo`, and a local store is fast.
15
16use std::sync::Arc;
17
18use anyhow::Result;
19use async_trait::async_trait;
20use object_store::ClientOptions;
21use object_store::http::HttpBuilder;
22use sui_indexer_alt_framework::ingestion::ingestion_client::{
23    CheckpointResult, IngestionClient, IngestionClientTrait,
24};
25use sui_indexer_alt_framework::ingestion::store_client::StoreIngestionClient;
26use sui_indexer_alt_framework::metrics::IngestionMetrics;
27use sui_types::digests::ChainIdentifier;
28use url::Url;
29
30/// Wraps an inner ingestion client, returning a fixed `chain_id` rather than letting the inner
31/// client derive it (which, for the remote store, means fetching genesis). Checkpoint fetches pass
32/// straight through.
33struct FixedChainId {
34    inner: Arc<dyn IngestionClientTrait>,
35    chain_id: ChainIdentifier,
36}
37
38#[async_trait]
39impl IngestionClientTrait for FixedChainId {
40    async fn chain_id(&self) -> Result<ChainIdentifier> {
41        Ok(self.chain_id)
42    }
43
44    async fn checkpoint(&self, checkpoint: u64) -> CheckpointResult {
45        self.inner.checkpoint(checkpoint).await
46    }
47
48    async fn latest_checkpoint_number(&self) -> Result<u64> {
49        self.inner.latest_checkpoint_number().await
50    }
51}
52
53/// Build an [`IngestionClient`] over the remote HTTP object store at `url` whose chain id is fixed
54/// to `chain_id` (so it never fetches genesis). Mirrors the framework's own remote-store setup.
55pub(crate) fn remote_store_client(
56    url: &Url,
57    chain_id: ChainIdentifier,
58    metrics: Arc<IngestionMetrics>,
59) -> Result<IngestionClient> {
60    let store = HttpBuilder::new()
61        .with_url(url.to_string())
62        .with_client_options(ClientOptions::default().with_allow_http(true))
63        .build()
64        .map(Arc::new)?;
65    let inner = Arc::new(StoreIngestionClient::new(
66        store,
67        Some(metrics.total_ingested_bytes.clone()),
68    ));
69    let client = Arc::new(FixedChainId { inner, chain_id });
70    Ok(IngestionClient::from_trait(client, metrics))
71}
