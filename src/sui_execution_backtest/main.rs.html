4//! Backtest the execution layer against historical mainnet data: re-execute past transactions under
5//! the current execution rules and report where the recomputed result diverges from what was
6//! recorded on chain. Useful for measuring the behavioral impact of an execution/protocol change
7//! before it ships.
8//!
9//! Runs as a `sui-indexer-alt-framework` concurrent pipeline (see [`handler`]): the framework's
10//! `Indexer` ingests the resolved checkpoint range with adaptive concurrency and runs the
11//! per-checkpoint processor with a configurable fanout. Each transaction matching `--status` is
12//! re-executed against reconstructed checkpoint state; any whose recomputed success/failure status
13//! disagrees with its on-chain status is recorded as a divergence. Output goes to a swappable sink
14//! (`--store`): postgres (durable + queryable) or an ndjson file (zero-setup).

57/// Backtest-specific migrations (the `divergence` and `run_stats` tables). The framework's
58/// watermark tables come from `sui-pg-db`'s own migrations, which `Db::run_migrations` applies
59/// alongside these.
60pub const MIGRATIONS: EmbeddedMigrations = embed_migrations!("migrations");

62/// Which on-chain transaction statuses to re-execute.
63#[derive(Clone, Copy, Debug, PartialEq, Eq, ValueEnum)]
64pub(crate) enum StatusFilter {
65    /// Only transactions that succeeded on-chain (strict differential baseline).
66    Success,
67    /// Only transactions that failed on-chain.
68    Failed,
69    /// Both; divergence records carry their on-chain status/failure kind for downstream filtering.
70    All,
71}
72
73/// Where divergence (and stats) rows are written.
74#[derive(Clone, Copy, Debug, PartialEq, Eq, ValueEnum)]
75enum StoreKind {
76    /// Newline-delimited JSON file (zero-setup; no resumption across restarts).
77    Ndjson,
78    /// Postgres (durable, queryable, resumable).
79    Postgres,
80}

82#[derive(Parser)]
83#[clap(
84    name = "sui-execution-backtest",
85    about = "Re-execute historical transactions and report divergences from their on-chain effects."
86)]
87struct Args {
88    #[clap(flatten)]
89    client: IngestionClientArgs,
90
91    /// Sui fullnode gRPC url used to resolve epochs and fetch packages. Defaults to `--rpc-api-url`
92    /// when that is provided as the checkpoint source.
93    #[clap(long)]
94    fullnode_url: Option<Url>,
95
96    /// First epoch to scan (inclusive).
97    #[clap(long)]
98    start_epoch: u64,
99
100    /// Last epoch to scan (inclusive).
101    #[clap(long)]
102    end_epoch: u64,
103
104    /// Optional cap on the number of checkpoints to process per epoch (for bounded samples).
105    #[clap(long)]
106    max_checkpoints_per_epoch: Option<u64>,
107
108    /// Which on-chain transaction statuses to re-execute. `success` keeps the strict baseline (only
109    /// txns that succeeded on-chain, so any divergence is a clear regression); `failed` runs only
110    /// failed txns; `all` runs both and tags each divergence record with its `original_status`/
111    /// failure kind so the differential can be applied in the analysis layer.
112    #[clap(long, value_enum, default_value_t = StatusFilter::All)]
113    status: StatusFilter,
114
115    /// CPU width: the number of checkpoints processed (and thus re-executed) concurrently by the
116    /// pipeline's processor fanout. Defaults to the framework's adaptive scaling (up to the number
117    /// of CPUs). Each checkpoint's transactions are executed serially on one blocking worker.
118    #[clap(long)]
119    execute_concurrency: Option<usize>,
120
121    /// Optional directory for an on-disk package cache (speeds up re-scans).
122    #[clap(long)]
123    cache: Option<PathBuf>,
124
125    /// Where to write divergence (and stats) rows.
126    #[clap(long, value_enum, default_value_t = StoreKind::Ndjson)]
127    store: StoreKind,
128
129    /// Postgres connection url (required for `--store postgres`).
130    #[clap(long, required_if_eq("store", "postgres"))]
131    database_url: Option<Url>,
132
133    /// Output file for `--store ndjson` (required for that store).
134    #[clap(long, required_if_eq("store", "ndjson"))]
135    output: Option<PathBuf>,
136
137    /// Run identifier. Namespaces both the output rows (the `task` column, part of the primary key)
138    /// and the postgres watermark, so re-running under changed execution rules starts fresh instead
139    /// of resuming the previous run's watermark and skipping already-processed checkpoints. Omit to
140    /// derive it from the git revision the binary was built from — the HEAD commit, plus a sha of
141    /// uncommitted changes when the working tree is dirty — so a commit or local edit gets a fresh
142    /// namespace while re-running unchanged code resumes where it left off.
143    #[clap(long)]
144    task: Option<String>,
145
146    /// Skip emitting per-checkpoint `run_stats` rows (divergences are still recorded).
147    #[clap(long)]
148    no_stats: bool,
149}

151#[tokio::main]
152async fn main() -> Result<()> {
153    tracing_subscriber::fmt::init();
154
155    let args = Args::parse();
156
157    if args.start_epoch > args.end_epoch {
158        bail!(
159            "start_epoch ({}) must be <= end_epoch ({})",
160            args.start_epoch,
161            args.end_epoch
162        );
163    }
164
165    let fullnode_url: Url = args
166        .fullnode_url
167        .clone()
168        .or_else(|| args.client.rpc_api_url.clone())
169        .context("provide --fullnode-url (or --rpc-api-url) for epoch and package resolution")?;
170    let rpc = RpcClient::new(fullnode_url)?;
171
172    let registry = Registry::new();
173    let execution_metrics = Arc::new(ExecutionMetrics::new(&registry));
174    let ingestion_metrics = IngestionMetrics::new(None, &registry);
175    let packages = Arc::new(PackageCache::new(
176        rpc.clone(),
177        tokio::runtime::Handle::current(),
178        args.cache.clone(),
179    )?);
180
181    // The chain identifier comes from the fullnode's GetServiceInfo (cheap), not by fetching
182    // genesis. For a remote object store we wrap the ingestion client with this id (see
183    // [`ingestion`]) so it never derives it from genesis; the gRPC source already uses
184    // GetServiceInfo and a local store is fast. Epoch resolution below uses the same fullnode
185    // (`rpc`), so genesis is never fetched on any path.
186    let chain_id = rpc.chain_id().await.context("fetching chain id")?;
187    let chain: Chain = chain_id.chain();
188    let ingestion_client = match &args.client.remote_store_url {
189        Some(url) => ingestion::remote_store_client(url, chain_id, ingestion_metrics.clone())?,
190        None => IngestionClient::new(args.client.clone(), ingestion_metrics.clone())?,
191    };
192
193    let first_bounds = rpc
194        .epoch_bounds(args.start_epoch)
195        .await
196        .with_context(|| format!("resolving epoch {}", args.start_epoch))?;
197    let (epochs, first_checkpoint, last_checkpoint) = resolve_epoch_work(
198        &rpc,
199        chain,
200        args.start_epoch..=args.end_epoch,
201        args.max_checkpoints_per_epoch,
202        first_bounds,
203        &execution_metrics,
204    )
205    .await?;
206    let epochs = Arc::new(epochs);
207
208    // The framework watermark is keyed by `{pipeline}@{task}`, so namespacing per run-id keeps a
209    // re-run under changed rules from resuming the previous run's watermark. When no `--task` is
210    // given, derive it from the git revision (HEAD commit, plus an uncommitted-changes sha when the
211    // tree is dirty): a commit or local edit to the execution rules gets a fresh namespace, while
212    // re-running unchanged code resumes. The reader interval is irrelevant here (this pipeline does
213    // no pruning and has no main pipeline), so it is set high.
214    const READER_INTERVAL_MS: u64 = 3_600_000;
215    let run_id = match args.task {
216        Some(run_id) => run_id,
217        None => {
218            let derived = derive_task_from_git().context("deriving task id from git revision")?;
219            info!(task = %derived, "no --task given; derived run identifier from git revision");
220            derived
221        }
222    };
223    let indexer_task = TaskArgs::tasked(run_id.clone(), READER_INTERVAL_MS);
224    let row_task = run_id;
225
226    let plan = BacktestPlan {
227        epochs,
228        packages,
229        chain_id,
230        status: args.status,
231        task: row_task,
232        indexer_task,
233        stats_enabled: !args.no_stats,
234        first_checkpoint,
235        last_checkpoint,
236        fanout: args
237            .execute_concurrency
238            .map(|value| ConcurrencyConfig::Fixed { value }),
239    };
240
241    // We build the ingestion client ourselves — wrapping the remote store in `FixedChainId` (see
242    // [`ingestion`]) so it never derives the chain id from a slow genesis fetch — and hand it to
243    // `Indexer::with_ingestion_clients`. The turnkey `Indexer::new` would instead build the client
244    // from `ClientArgs`, which (a) reintroduces the genesis chain-id fetch on the remote-store path
245    // (measured as a sustained ~4x throughput drop, worst on bounded samples) and (b) re-adds the
246    // streaming CLI flags we deliberately dropped (we only flatten `IngestionClientArgs`).
247    match args.store {
248        StoreKind::Ndjson => {
249            // clap's `required_if_eq` guarantees `--output` is present for this store.
250            let output = args.output.expect("--output required for --store ndjson");
251            let store = NdjsonStore::create(&output)?;
252            run_backtest(store, ingestion_client, registry, plan).await
253        }
254        StoreKind::Postgres => {
255            // clap's `required_if_eq` guarantees `--database-url` is present for this store.
256            let database_url = args
257                .database_url
258                .expect("--database-url required for --store postgres");
259            let store = Db::for_write(database_url, DbArgs::default())
260                .await
261                .context("connecting to postgres")?;
262            store
263                .run_migrations(Some(&MIGRATIONS))
264                .await
265                .context("running migrations")?;
266            run_backtest(store, ingestion_client, registry, plan).await
267        }
268    }
269}

271/// The default run identifier: the source revision the binary was built from. This is the HEAD
272/// commit's short sha, plus — when the working tree is dirty — a second sha fingerprinting the
273/// uncommitted (tracked) changes. So committed code re-runs under a stable id (resuming its
274/// watermark), while each distinct set of edits-under-test gets a fresh namespace.
275///
276/// Git runs against the build-time source tree (`CARGO_MANIFEST_DIR`), so the id reflects the code
277/// the binary came from regardless of the process's working directory. The dirty fingerprint is the
278/// diff against HEAD hashed into a git blob sha (`git diff HEAD | git hash-object --stdin`), which —
279/// unlike `git stash create` — carries no commit timestamp, so re-running the same edits yields the
280/// same id. It is empty when the tree is clean; untracked files are not captured. Errors out if git
281/// is unavailable or the source tree is gone — pass `--task` explicitly in that case.
282fn derive_task_from_git() -> Result<String> {
283    let dir = env!("CARGO_MANIFEST_DIR");
284
285    let head = run_git(dir, &["rev-parse", "--short=12", "HEAD"], None)
286        .context("resolving HEAD (pass --task to set a run id explicitly)")?;
287    let head = String::from_utf8(head).context("git HEAD not utf-8")?;
288    let head = head.trim();
289
290    let diff = run_git(dir, &["diff", "HEAD"], None).context("diffing working tree")?;
291    if diff.is_empty() {
292        return Ok(format!("git-{head}"));
293    }
294    let wip = run_git(dir, &["hash-object", "--stdin"], Some(&diff))
295        .context("hashing working-tree diff")?;
296    let wip = String::from_utf8(wip).context("git hash-object output not utf-8")?;
297    let wip = wip.trim();
298    Ok(format!("git-{head}-{}", &wip[..wip.len().min(12)]))
299}

301/// Run `git -C <dir> <args>`, optionally feeding `stdin`, and return raw stdout. Errors on a
302/// non-zero exit (carrying git's stderr). Output is bytes, not text, because `git diff` can contain
303/// non-UTF-8 content from binary files.
304fn run_git(dir: &str, args: &[&str], stdin: Option<&[u8]>) -> Result<Vec<u8>> {
305    use std::io::Write as _;
306    use std::process::{Command, Stdio};
307
308    let mut cmd = Command::new("git");
309    cmd.arg("-C")
310        .arg(dir)
311        .args(args)
312        .stdout(Stdio::piped())
313        .stderr(Stdio::piped());
314    if stdin.is_some() {
315        cmd.stdin(Stdio::piped());
316    }
317    let mut child = cmd.spawn().context("spawning git")?;
318    if let Some(bytes) = stdin {
319        child
320            .stdin
321            .take()
322            .expect("stdin piped")
323            .write_all(bytes)
324            .context("writing git stdin")?;
325    }
326    let out = child.wait_with_output().context("waiting on git")?;
327    if !out.status.success() {
328        anyhow::bail!(
329            "git {} failed: {}",
330            args.join(" "),
331            String::from_utf8_lossy(&out.stderr).trim()
332        );
333    }
334    Ok(out.stdout)
335}

337/// Everything needed to run the pipeline, independent of the sink. Grouped so [`run_backtest`] can
338/// be generic over the store with a small argument list.
339struct BacktestPlan {
340    epochs: Arc<BTreeMap<u64, Arc<EpochCtx>>>,
341    packages: Arc<PackageCache>,
342    chain_id: ChainIdentifier,
343    status: StatusFilter,
344    /// Run identifier written to the `task` row column.
345    task: String,
346    /// Framework task config that namespaces the watermark (matches `task` when set via `--task`).
347    indexer_task: TaskArgs,
348    stats_enabled: bool,
349    first_checkpoint: u64,
350    last_checkpoint: u64,
351    fanout: Option<ConcurrencyConfig>,
352}
353
354/// Build and run the indexer for a given sink. Generic over the store so the postgres and ndjson
355/// paths share all the wiring.
356async fn run_backtest<S: CommitRows>(
357    store: S,
358    ingestion_client: IngestionClient,
359    registry: Registry,
360    plan: BacktestPlan,
361) -> Result<()> {
362    let indexer_args = IndexerArgs {
363        first_checkpoint: Some(plan.first_checkpoint),
364        last_checkpoint: Some(plan.last_checkpoint),
365        pipeline: Vec::new(),
366        task: plan.indexer_task,
367    };
368
369    let mut indexer = Indexer::with_ingestion_clients(
370        store,
371        indexer_args,
372        ingestion_client,
373        None,
374        IngestionConfig::default(),
375        None,
376        &registry,
377    )
378    .await?;
379
380    let handler = Backtest::new(
381        plan.epochs,
382        plan.packages,
383        plan.chain_id,
384        plan.status,
385        plan.task,
386        plan.stats_enabled,
387    );
388    let config = ConcurrentConfig {
389        fanout: plan.fanout,
390        ..Default::default()
391    };
392    indexer.concurrent_pipeline(handler, config).await?;
393
394    let service = indexer.run().await?;
395    service.main().await.context("indexer terminated")?;
396    Ok(())
