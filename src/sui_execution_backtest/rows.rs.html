4//! The typed rows the backtest emits. Both sinks (`postgres` and `ndjson`) write these: postgres
5//! inserts them via diesel, the ndjson sink serializes them with serde. Counters are `i64` because
6//! postgres has no unsigned integer type; the source counts are `u64` and converted at the edge.
7
8use diesel::prelude::*;
9use serde::Serialize;
10
11use crate::schema::divergence;
12use crate::schema::run_stats;
13
14/// A transaction whose recomputed success/failure status disagrees with its on-chain status.
15#[derive(Insertable, Queryable, Selectable, Serialize, Debug, Clone)]
16#[diesel(table_name = divergence)]
17pub struct DivergenceRow {
18    pub task: String,
19    pub epoch: i64,
20    pub checkpoint: i64,
21    pub tx_digest: String,
22    pub original_status: String,
23    pub original_failure_kind: Option<String>,
24    pub recomputed_status: String,
25    pub recomputed_error_kind: Option<String>,
26    pub recomputed_error_detail: Option<String>,
27    pub missing_modified: i64,
28    pub missing_loaded: i64,
29    pub missing_consensus: i64,
30    pub digest_mismatches: i64,
31}
32
33/// Per-checkpoint replay denominators for a run.
34#[derive(Insertable, Queryable, Selectable, Serialize, Debug, Clone)]
35#[diesel(table_name = run_stats)]
36pub struct RunStatsRow {
37    pub task: String,
38    pub epoch: i64,
39    pub checkpoint: i64,
40    pub checked: i64,
41    pub executed: i64,
42    pub divergences: i64,
43    pub reconstruction_errors: i64,
44    pub coin_reservation_skipped: i64,
45    pub execute_skipped: i64,
46    pub gas_from_balance: i64,
47    pub cancellation_excluded: i64,
48}

50#[cfg(test)]
51mod tests {
52    use diesel_async::RunQueryDsl;
53    use sui_indexer_alt_framework::Indexer;
54
55    use super::*;
56    use crate::MIGRATIONS;
57
58    /// The backtest migrations apply on top of the framework's watermark tables, and both row types
59    /// round-trip through diesel against the real schema.
60    #[tokio::test]
61    async fn migrations_apply_and_rows_round_trip() {
62        let (indexer, _temp_db) = Indexer::new_for_testing(&MIGRATIONS).await;
63        let mut conn = indexer.store().connect().await.unwrap();
64
65        let div = DivergenceRow {
66            task: "t".to_owned(),
67            epoch: 1152,
68            checkpoint: 284396922,
69            tx_digest: "digest".to_owned(),
70            original_status: "success".to_owned(),
71            original_failure_kind: None,
72            recomputed_status: "failure".to_owned(),
73            recomputed_error_kind: Some("InvalidLinkage".to_owned()),
74            recomputed_error_detail: Some("conflicting resolutions".to_owned()),
75            missing_modified: 0,
76            missing_loaded: 0,
77            missing_consensus: 0,
78            digest_mismatches: 0,
79        };
80        diesel::insert_into(divergence::table)
81            .values(&div)
82            .execute(&mut conn)
83            .await
84            .unwrap();
85
86        let stats = RunStatsRow {
87            task: "t".to_owned(),
88            epoch: 1152,
89            checkpoint: 284396922,
90            checked: 10,
91            executed: 10,
92            divergences: 1,
93            reconstruction_errors: 0,
94            coin_reservation_skipped: 0,
95            execute_skipped: 0,
96            gas_from_balance: 2,
97            cancellation_excluded: 0,
98        };
99        diesel::insert_into(run_stats::table)
100            .values(&stats)
101            .execute(&mut conn)
102            .await
103            .unwrap();
104
105        let divs: Vec<DivergenceRow> = divergence::table.load(&mut conn).await.unwrap();
106        assert_eq!(divs.len(), 1);
107        assert_eq!(
108            divs[0].recomputed_error_kind.as_deref(),
109            Some("InvalidLinkage")
110        );
111
112        let all_stats: Vec<RunStatsRow> = run_stats::table.load(&mut conn).await.unwrap();
113        assert_eq!(all_stats.len(), 1);
114        assert_eq!(all_stats[0].divergences, 1);
115    }
116}
