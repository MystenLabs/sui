4//! The read side of execution: a read-only `BackingStore` ([`ScanStore`]) for executing a single
5//! transaction, backed by the objects present in a streamed checkpoint's object set, plus the
6//! shared [`PackageCache`] those reads resolve packages against. Dynamic-field child reads and
7//! `Receiving` reads are served from the same object set (a checkpoint carries the objects
8//! execution loaded). [`prefetch_package_closure`] warms the cache with a checkpoint's package
9//! closure up front; a miss during execution falls back to a lazy gRPC + on-disk fetch.
10

36/// Shared, process-wide package cache: in-memory map layered over an optional on-disk cache,
37/// falling back to a gRPC fetch from a fullnode. Packages are immutable, so a single entry per id
38/// is sufficient. `None` is cached as a negative result to avoid refetching known-missing packages.
39/// An `RwLock` (rather than a `Mutex`) lets concurrent execute workers clone their package copies
40/// in parallel on the hit path; the exclusive lock is taken only to insert a freshly-loaded entry.
41pub struct PackageCache {
42    rpc: RpcClient,
43    handle: Handle,
44    disk: Option<PathBuf>,
45    mem: RwLock<HashMap<ObjectID, Option<Object>>>,
46}
47
48impl PackageCache {
49    pub fn new(rpc: RpcClient, handle: Handle, disk: Option<PathBuf>) -> anyhow::Result<Self> {
50        if let Some(dir) = &disk {
51            std::fs::create_dir_all(dir)
52                .with_context(|| format!("creating package cache dir {}", dir.display()))?;
53        }
54        Ok(Self {
55            rpc,
56            handle,
57            disk,
58            mem: RwLock::new(HashMap::new()),
59        })
60    }
61
62    fn disk_path(&self, id: ObjectID) -> Option<PathBuf> {
63        self.disk
64            .as_ref()
65            .map(|dir| dir.join(id.to_canonical_string(/* with_prefix */ true)))
66    }
67
68    /// Fetch a package object, consulting the in-memory cache, then the on-disk cache, then gRPC.
69    /// Must be called from a blocking context (it drives the async fetch via [`Handle::block_on`]),
70    /// never from a runtime worker thread.
71    fn fetch(&self, id: ObjectID) -> Option<Object> {
72        // Hit path under a shared read lock so concurrent workers clone in parallel.
73        if let Some(hit) = self.mem.read().unwrap().get(&id) {
74            return hit.clone();
75        }
76
77        // On-disk cache.
78        if let Some(object) = self.read_disk(id) {
79            self.mem.write().unwrap().insert(id, Some(object.clone()));
80            return Some(object);
81        }
82
83        // Reaching here means neither the in-memory nor on-disk cache had the package, so the
84        // prefetch closure ([`prefetch_package_closure`]) did not warm it — the correctness net
85        // catching a closure gap. This should be rare; a steady stream of these points at a missing
86        // prefetch source worth adding (e.g. a new runtime-only package edge).
87        warn!(%id, "lazy package fetch during execution (prefetch miss)");
88
89        // gRPC fetch. We may be on a `spawn_blocking` thread (the execution pipeline is synchronous),
90        // where `block_on` is disallowed, so drive the async fetch by spawning it onto the runtime
91        // and blocking on a channel for the result.
92        let fetched = {
93            let (tx, rx) = std::sync::mpsc::channel();
94            let rpc = self.rpc.clone();
95            self.handle.spawn(async move {
96                let _ = tx.send(rpc.fetch_object(id).await);
97            });
98            match rx.recv() {
99                Ok(Ok(object)) => object,
100                Ok(Err(e)) => {
101                    warn!(%id, "package fetch failed: {e:#}");
102                    None
103                }
104                Err(_) => {
105                    warn!(%id, "package fetch task dropped");
106                    None
107                }
108            }
109        };
110
111        if let (Some(object), Some(path)) = (&fetched, self.disk_path(id))
112            && let Ok(bytes) = bcs::to_bytes(object)
113        {
114            let _ = std::fs::write(&path, bytes);
115        }
116        self.mem.write().unwrap().insert(id, fetched.clone());
117        fetched
118    }
119
120    /// Read a package object from the on-disk cache, if present and decodable. No network.
121    fn read_disk(&self, id: ObjectID) -> Option<Object> {
122        let path = self.disk_path(id)?;
123        let bytes = std::fs::read(&path).ok()?;
124        bcs::from_bytes::<Object>(&bytes).ok()
125    }
126
127    /// Warm the cache with `ids` using a single batched multi-get for the ones not already cached
128    /// (in memory or on disk), returning every object now available for those ids. Async and called
129    /// from the prefetch stage (no `block_on` bridge); a miss here just isn't inserted and falls
130    /// through to the lazy [`Self::fetch`] during execution. Failures are logged and swallowed so a
131    /// prefetch hiccup degrades to the lazy path rather than aborting.
132    pub async fn prefetch(&self, ids: &[ObjectID]) -> Vec<Object> {
133        let mut have = Vec::new();
134        let mut to_fetch = Vec::new();
135        {
136            let mem = self.mem.read().unwrap();
137            for &id in ids {
138                match mem.get(&id) {
139                    // Present (`Some(obj)`) or a cached negative (`None`); either way, no fetch.
140                    Some(slot) => have.extend(slot.clone()),
141                    None => to_fetch.push(id),
142                }
143            }
144        }
145        // Disk hits next: load them into memory, off the network path.
146        to_fetch.retain(|&id| match self.read_disk(id) {
147            Some(object) => {
148                self.mem.write().unwrap().insert(id, Some(object.clone()));
149                have.push(object);
150                false
151            }
152            None => true,
153        });
154        if to_fetch.is_empty() {
155            return have;
156        }
157        match self.rpc.fetch_objects(&to_fetch).await {
158            Ok(fetched) => {
159                let mut mem = self.mem.write().unwrap();
160                for object in fetched {
161                    let id = object.id();
162                    if let Some(path) = self.disk_path(id)
163                        && let Ok(bytes) = bcs::to_bytes(&object)
164                    {
165                        let _ = std::fs::write(&path, bytes);
166                    }
167                    mem.insert(id, Some(object.clone()));
168                    have.push(object);
169                }
170            }
171            Err(e) => warn!("batch package prefetch failed: {e:#}"),
172        }
173        have
174    }
175}

177/// Warm `cache` with the package closure of `checkpoint`'s transactions, so the synchronous execute
178/// stage reads packages from memory instead of fetching mid-execution.
179///
180/// The closure is computed statically (no execution): collect the *roots* — the packages the
181/// commands reference (via the system's own [`input_objects`](sui_types::transaction::ProgrammableTransaction::input_objects)
182/// derivation) plus the defining packages of the checkpoint's object types. Then add, for each
183/// fetched root:
184///   - its **linkage table** — the (upgraded) storage ids of its transitive declared dependencies,
185///   - its **type-origin table** — the storage id of every package version that *introduced* one of
186///     its own types. A type added in an upgrade (e.g. a Cetus type first defined in v10) records
187///     that version as its origin, and the executor loads the module at that version; since such a
188///     type need never appear as a static type argument, the linkage table alone misses it.
189///
190/// System packages are not prefetched here: they are served version-correctly per epoch from the
191/// framework snapshot (see [`crate::context::EpochCtx`] and [`ScanStore::get_package_object`]).
192///
193/// This remains an over-approximation of what actually executes (and still misses some
194/// runtime-only edges, e.g. versions pinned only by a stored object's provenance); a miss falls
195/// through to the lazy [`PackageCache::fetch`], which is the correctness net.
196pub(crate) async fn prefetch_package_closure(checkpoint: &Checkpoint, cache: &PackageCache) {
197    let roots: Vec<ObjectID> = collect_roots(checkpoint).into_iter().collect();
198    let root_pkgs = cache.prefetch(&roots).await;
199
200    let mut linked = BTreeSet::new();
201    for object in &root_pkgs {
202        if let Some(pkg) = object.data.try_as_package() {
203            for upgrade in pkg.linkage_table().values() {
204                linked.insert(upgrade.upgraded_id);
205            }
206            for origin in pkg.type_origin_table() {
207                linked.insert(origin.package);
208            }
209        }
210    }
211    let linked: Vec<ObjectID> = linked.into_iter().collect();
212    cache.prefetch(&linked).await;
213}

215/// The root package set: the packages the transactions' commands reference, plus the defining
216/// packages of every object type present in the checkpoint's object set.
217fn collect_roots(checkpoint: &Checkpoint) -> BTreeSet<ObjectID> {
218    let mut roots = BTreeSet::new();
219
220    for executed in &checkpoint.transactions {
221        let TransactionKind::ProgrammableTransaction(pt) = executed.transaction.kind() else {
222            continue;
223        };
224        // Reuse the system's own command package-input derivation: it yields a `MovePackage` for
225        // every package the commands reference — MoveCall targets and their type-argument packages,
226        // `MakeMoveVec` element types, and `Publish`/`Upgrade` dependency packages. (A miss here
227        // just falls through to the lazy fetch, so ignore the rare `input_objects` error.)
228        if let Ok(input_objects) = pt.input_objects() {
229            roots.extend(input_objects.into_iter().filter_map(|kind| match kind {
230                InputObjectKind::MovePackage(id) => Some(id),
231                _ => None,
232            }));
233        }
234    }
235
236    // Input object *types* aren't packages in `input_objects`, so pick up their defining packages
237    // (and those of their type parameters) from the materialized object set.
238    for object in checkpoint.object_set.iter() {
239        if let Some(mo) = object.data.try_as_move() {
240            let ty: TypeTag = mo.type_().clone().into();
241            roots.extend(ty.all_addresses().into_iter().map(ObjectID::from));
242        }
243    }
244
245    roots
246}

248/// Per-transaction read-only store over a checkpoint's object set, with package fallback.
249pub struct ScanStore {
250    /// All object versions present in the checkpoint, keyed by (id, version).
251    objects: Arc<BTreeMap<ObjectKey, Object>>,
252    /// Highest version seen per object id (for version-less `get_object`).
253    latest: Arc<BTreeMap<ObjectID, SequenceNumber>>,
254    /// Tombstone (deletion/wrap/unwrap-then-delete) versions per object id, gathered from every
255    /// transaction's effects in the checkpoint. A child object that the bundled object set still
256    /// carries at a stale version may actually have been *removed* at/under a given root version;
257    /// without this, `read_child_object` would resurrect a deleted dynamic field. See
258    /// `child_at_bound`.
259    tombstones: Arc<BTreeMap<ObjectID, BTreeSet<SequenceNumber>>>,
260    packages: Arc<PackageCache>,
261    /// The system (framework) packages live during this epoch, keyed by id. System packages share a
262    /// stable id across upgrades but differ in bytecode per protocol version, so they are served
263    /// from here (resolved version-correctly from the framework snapshot) rather than from the
264    /// fullnode, which only returns their latest version. See [`crate::context::EpochCtx`].
265    system_packages: Arc<BTreeMap<ObjectID, Object>>,
266}
267
268impl ScanStore {
269    /// Build the per-checkpoint object index once; cheap to clone (Arc) per transaction.
270    pub fn index_object_set(
271        object_set: &ObjectSet,
272    ) -> (
273        BTreeMap<ObjectKey, Object>,
274        BTreeMap<ObjectID, SequenceNumber>,
275    ) {
276        let mut objects = BTreeMap::new();
277        let mut latest: BTreeMap<ObjectID, SequenceNumber> = BTreeMap::new();
278        for object in object_set.iter() {
279            let id = object.id();
280            let version = object.version();
281            latest
282                .entry(id)
283                .and_modify(|v| {
284                    if version > *v {
285                        *v = version;
286                    }
287                })
288                .or_insert(version);
289            objects.insert(ObjectKey(id, version), object.clone());
290        }
291        (objects, latest)
292    }
293
294    pub fn new(
295        objects: Arc<BTreeMap<ObjectKey, Object>>,
296        latest: Arc<BTreeMap<ObjectID, SequenceNumber>>,
297        tombstones: Arc<BTreeMap<ObjectID, BTreeSet<SequenceNumber>>>,
298        packages: Arc<PackageCache>,
299        system_packages: Arc<BTreeMap<ObjectID, Object>>,
300    ) -> Self {
301        Self {
302            objects,
303            latest,
304            tombstones,
305            packages,
306            system_packages,
307        }
308    }
309}

311impl ObjectStore for ScanStore {
312    fn get_object(&self, object_id: &ObjectID) -> Option<Object> {
313        let version = *self.latest.get(object_id)?;
314        self.objects.get(&ObjectKey(*object_id, version)).cloned()
315    }
316
317    fn get_object_by_key(&self, object_id: &ObjectID, version: VersionNumber) -> Option<Object> {

319    }
320}

322impl BackingPackageStore for ScanStore {
323    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
324        // System packages are versioned per epoch (stable id, different bytecode per protocol
325        // version), so serve the epoch-correct snapshot copy rather than the fullnode's latest. Fall
326        // through if absent (e.g. a system package not yet published at this protocol version).
327        if is_system_package(*package_id)
328            && let Some(object) = self.system_packages.get(package_id)
329        {
330            return Ok(Some(PackageObject::new(object.clone())));
331        }
332        // Prefer a package already present in the checkpoint's object set.
333        if let Some(object) = self.get_object(package_id)
334            && object.is_package()
335        {
336            return Ok(Some(PackageObject::new(object)));
337        }
338        Ok(self.packages.fetch(*package_id).map(PackageObject::new))
339    }
340}

342impl RuntimeObjectResolver for ScanStore {
343    fn read_child_object(
344        &self,
345        _parent: &ObjectID,
346        child: &ObjectID,
347        child_version_upper_bound: SequenceNumber,
348    ) -> SuiResult<Option<Object>> {
349        // Return the child at its root version, respecting within-checkpoint deletions. Does not
350        // re-verify `parent` ownership (mirrors sui-replay-2: these are reconstructions of
351        // already-validated executions).
352        Ok(child_at_bound(
353            &self.objects,
354            &self.tombstones,
355            *child,
356            child_version_upper_bound,
357        ))
358    }
359
360    fn get_object_received_at_version(
361        &self,
362        owner: &ObjectID,
363        receiving_object_id: &ObjectID,
364        receive_object_at_version: SequenceNumber,
365        _epoch_id: EpochId,
366    ) -> SuiResult<Option<Object>> {
367        // Per the trait contract every failure mode (absent / wrong version / wrong owner) is
368        // `Ok(None)`, never an error.
369        Ok(received_object(
370            &self.objects,
371            *owner,
372            *receiving_object_id,
373            receive_object_at_version,
374        ))
375    }
376}

384/// The child object as of root version `bound`: the greatest live version `<= bound` in the index,
385/// **unless** the child was removed (tombstoned) more recently than that live version. The bundled
386/// object set carries only live object versions (no tombstones), so a child that was deleted within
387/// the checkpoint still has a stale `<= bound` version present; consulting `tombstones` lets us
388/// return `None` for it, matching on-chain liveness (and correctly returning the object again if it
389/// was re-created after deletion at a version `<= bound`).
390fn child_at_bound(
391    objects: &BTreeMap<ObjectKey, Object>,
392    tombstones: &BTreeMap<ObjectID, BTreeSet<SequenceNumber>>,
393    id: ObjectID,
394    bound: SequenceNumber,
395) -> Option<Object> {
396    let (live_key, obj) = objects
397        .range(ObjectKey(id, SequenceNumber::from_u64(0))..=ObjectKey(id, bound))
398        .next_back()?;
399    // A tombstone version is the (new) version assigned when the object was removed, so it is
400    // greater than the object's last live version. If the most recent tombstone `<= bound` is newer
401    // than the most recent live version `<= bound`, the object is dead at `bound`.
402    if let Some(tombs) = tombstones.get(&id)
403        && tombs
404            .range(..=bound)
405            .next_back()
406            .is_some_and(|&t| t > live_key.1)
407    {
408        return None;
409    }
410    Some(obj.clone())
411}

413/// The object at *exactly* `version`, address-owned by `owner` — the `Receiving` contract. Any
414/// failure (absent / wrong version / wrong owner) yields `None`.
415fn received_object(
416    objects: &BTreeMap<ObjectKey, Object>,
417    owner: ObjectID,
418    id: ObjectID,
419    version: SequenceNumber,
420) -> Option<Object> {
421    let obj = objects.get(&ObjectKey(id, version))?.clone();
422    (obj.owner == Owner::AddressOwner(owner.into())).then_some(obj)
423}

425/// Resolve the transaction's declared input object kinds against `store` into [`InputObjects`],
426/// mirroring `sui-replay-2`'s `get_input_objects_for_replay`. Returns an error if a declared input
427/// object is not present in the checkpoint's object set.
428///
429/// `effects` supplies the per-transaction *consensus* (shared) object versions: a shared object's
430/// declared input kind carries only its `initial_shared_version`, but each transaction reads it at
431/// the specific version consensus assigned — and a hot shared object (e.g. a DeFi pool) is mutated
432/// many times within one checkpoint, so the checkpoint-wide latest version is the wrong state for
433/// all but the last reader. Owned/imm inputs already carry their exact version in the input ref.
434pub fn resolve_input_objects(
435    txn_data: &TransactionData,
436    effects: &TransactionEffects,
437    store: &ScanStore,
438) -> anyhow::Result<InputObjects> {
439    let consensus_inputs = effects.input_consensus_objects();
440    // Per-tx version for the inputs we resolve against — only plain mutate/read-only consensus
441    // inputs are materialized at a specific version in the checkpoint. Other kinds (cancelled,
442    // ended consensus streams) carry no live version and fall through to the latest-version lookup,
443    // which is expected to miss; `shared_kind` surfaces which in the error.
444    let shared_versions: BTreeMap<ObjectID, SequenceNumber> = consensus_inputs
445        .iter()
446        .filter_map(|ico| match ico {
447            InputConsensusObject::Mutate((id, version, _))
448            | InputConsensusObject::ReadOnly((id, version, _)) => Some((*id, *version)),
449            _ => None,
450        })
451        .collect();
452    let mut resolved = Vec::new();
453    for kind in txn_data.input_objects()? {
454        match kind {
455            InputObjectKind::MovePackage(id) => {
456                let object = store
457                    .get_package_object(&id)
458                    .map_err(|e| anyhow::anyhow!("fetching package {id}: {e}"))?
459                    .with_context(|| format!("package {id} not found"))?
460                    .object()
461                    .clone();
462                resolved.push(ObjectReadResult {
463                    input_object_kind: kind,
464                    object: ObjectReadResultKind::Object(object),
465                });
466            }
467            InputObjectKind::ImmOrOwnedMoveObject((id, version, _digest)) => {
468                let object = store.get_object_by_key(&id, version).with_context(|| {
469                    format!("imm-or-owned input object {id} v{version} not in object set")
470                })?;
471                resolved.push(ObjectReadResult {
472                    input_object_kind: InputObjectKind::ImmOrOwnedMoveObject(
473                        object.compute_object_reference(),
474                    ),
475                    object: ObjectReadResultKind::Object(object),
476                });
477            }
478            InputObjectKind::SharedMoveObject { id, .. } => {
479                // Prefer the per-tx assigned version from effects; fall back to latest only if the
480                // object isn't in the consensus-input records (e.g. cancelled/stream-ended edges).
481                let object = match shared_versions.get(&id) {
482                    Some(&version) => store.get_object_by_key(&id, version).with_context(|| {
483                        format!(
484                            "{} shared input object {id} v{version} not in object set",
485                            shared_kind(&consensus_inputs, id)
486                        )
487                    })?,
488                    None => store.get_object(&id).with_context(|| {
489                        format!(
490                            "shared input object {id} ({}) not in object set",
491                            shared_kind(&consensus_inputs, id)
492                        )
493                    })?,
494                };
495                resolved.push(ObjectReadResult {
496                    input_object_kind: kind,
497                    object: ObjectReadResultKind::Object(object),
498                });
499            }
500        }
501    }
502    Ok(InputObjects::new(resolved))
503}

505/// The consensus access kind of shared input `id` (for the "not in object set" diagnostic). Scans
506/// the (small) consensus-input list, called only when building an error message.
507fn shared_kind(consensus_inputs: &[InputConsensusObject], id: ObjectID) -> &'static str {
508    consensus_inputs
509        .iter()
510        .find(|ico| ico.id_and_version().0 == id)
511        .map_or("absent", |ico| match ico {
512            InputConsensusObject::Mutate(_) => "mutate",
513            InputConsensusObject::ReadOnly(_) => "read-only",
514            InputConsensusObject::ReadConsensusStreamEnded(..) => "read-stream-ended",
515            InputConsensusObject::MutateConsensusStreamEnded(..) => "mutate-stream-ended",
516            InputConsensusObject::Cancelled(..) => "cancelled",
517        })
518}

563    /// A dynamic-field child read resolves to the greatest version `<= bound` (its "root version"),
564    /// and is absent when the bound is below every known version or the id is unknown.
565    #[test]
566    fn read_child_object_returns_root_version() {
567        let child = ObjectID::random();
568        let objects = index(vec![obj(child, 3), obj(child, 7)]);
569        let no_tombs = BTreeMap::new();
570        let sv = SequenceNumber::from_u64;
571
572        let at = |b| child_at_bound(&objects, &no_tombs, child, sv(b));
573        assert_eq!(at(7).unwrap().version(), sv(7));
574        assert_eq!(at(5).unwrap().version(), sv(3));
575        assert_eq!(at(9).unwrap().version(), sv(7));
576        assert!(at(2).is_none());
577        assert!(child_at_bound(&objects, &no_tombs, ObjectID::random(), sv(100)).is_none());
578    }

580    /// A child removed within the checkpoint must read as absent at/after the tombstone version,
581    /// even though the object set still carries its pre-deletion version — and read as present
582    /// again if it was re-created afterward.
583    #[test]
584    fn read_child_object_respects_tombstones() {
585        let child = ObjectID::random();
586        // live v3, deleted (tombstone v5), re-created v8.
587        let objects = index(vec![obj(child, 3), obj(child, 8)]);
588        let tombs: BTreeMap<ObjectID, BTreeSet<SequenceNumber>> =
589            [(child, [SequenceNumber::from_u64(5)].into_iter().collect())]
590                .into_iter()
591                .collect();
592        let sv = SequenceNumber::from_u64;
593        let at = |b| child_at_bound(&objects, &tombs, child, sv(b));
594
595        assert_eq!(at(3).unwrap().version(), sv(3)); // before deletion: live v3
596        assert!(at(5).is_none()); // at tombstone: deleted
597        assert!(at(7).is_none()); // after deletion, before re-creation: still gone
598        assert_eq!(at(8).unwrap().version(), sv(8)); // re-created: live again
599        assert_eq!(at(9).unwrap().version(), sv(8));
600    }
