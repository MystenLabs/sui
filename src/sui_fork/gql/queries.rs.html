4//! GQL Queries
5//! Interface to the rpc for the gql schema defined in `crates/sui-indexer-alt-graphql/schema.graphql`.
6//! Built in modules for epochs, transactions, objects, and checkpoints.
7//! No GQL type escapes this module. From here we return structures defined in this crate
8//! or bcs encoded data of runtime structures.
9//!
10//! Most of these query types are generated by cynic using the schema through the web UI:
11//! `https://generator.cynic-rs.dev/`
12

82    pub(crate) async fn query(
83        digest: String,
84        client: &GraphQLClient,
85    ) -> Result<Option<TransactionInfo>, Error> {
86        let query = Query::build(TransactionDataArgs {
87            digest: digest.clone(),
88        });
89        let response = client
90            .run_query(&query)
91            .await
92            .context("Failed to run transaction query")?;
93
94        let Some(transaction) = response.data.and_then(|txn| txn.transaction) else {
95            return Ok(None);
96        };
97
98        let txn_data: TransactionData = bcs::from_bytes(
99            &FastCryptoBase64::decode(
100                &transaction
101                    .transaction_bcs
102                    .ok_or_else(|| {
103                        anyhow!(format!(
104                            "Transaction data not available (None) for digest: {}",
105                            digest
106                        ),)
107                    })?
108                    .0,
109            )
110            .context(format!(
111                "Transaction data does not decode for digest: {}",
112                digest
113            ))?,
114        )
115        .context(format!(
116            "Cannot deserialize transaction data for digest {}",
117            digest
118        ))?;
119
120        let signatures: Vec<GenericSignature> = transaction
121            .signatures
122            .into_iter()
123            .map(|sig| {
124                let encoded = sig
125                    .signature_bytes
126                    .ok_or_else(|| anyhow!("Missing signatureBytes for transaction {}", digest))?;
127                let bytes = FastCryptoBase64::decode(&encoded.0).context(format!(
128                    "User signature does not decode for digest: {}",
129                    digest
130                ))?;
131                GenericSignature::from_bytes(&bytes).map_err(|e| {
132                    anyhow!("User signature parse failed for digest {}: {}", digest, e)
133                })
134            })
135            .collect::<Result<_, _>>()?;
136
137        let effect_frag = transaction
138            .effects
139            .ok_or_else(|| anyhow!("Missing effects in transaction data response"))?;
140        let effects: sui_types::effects::TransactionEffects = bcs::from_bytes(
141            &FastCryptoBase64::decode(
142                &effect_frag
143                    .effects_bcs
144                    .ok_or_else(|| anyhow!("Missing effects bcs in transaction data response"))?
145                    .0,
146            )
147            .context(format!(
148                "Transaction effects do not decode for digest: {}",
149                digest
150            ))?,
151        )
152        .context(format!(
153            "Cannot deserialize transaction effects for digest {}",
154            digest
155        ))?;
156
157        let checkpoint = effect_frag
158            .checkpoint
159            .ok_or_else(|| anyhow!("Missing checkpoint in transaction data response"))?
160            .sequence_number;
161
162        let transaction = VerifiedTransaction::new_unchecked(
163            SuiTransaction::from_generic_sig_data(txn_data, signatures),
164        );
165
166        Ok(Some(TransactionInfo {
167            transaction,
168            effects,
169            checkpoint,
170        }))
171    }

219            .service_config
220            .available_range
221            .first
222            .map(|c| c.sequence_number)
223            .unwrap_or(0))
224    }

329    pub(crate) async fn query(
330        address: SuiAddressType,
331        checkpoint: CheckpointSequenceNumber,
332        client: &GraphQLClient,
333    ) -> Result<Vec<AddressOwnedObject>, Error> {
334        let mut entries = Vec::new();
335        let mut cursor = None;
336
337        loop {
338            let operation = Query::build(AddressObjectsArgs {
339                sequence_number: Some(checkpoint),
340                address: SuiAddress(address.to_string()),
341                first: Some(PAGE_SIZE),
342                after: cursor,
343            });
344            let response = client
345                .run_query(&operation)
346                .await
347                .with_context(|| format!("failed to query owned objects for {address}"))?;
348
349            let data = response.data.ok_or_else(|| {
350                anyhow!(
351                    "missing data in address objects query response for {address}: {:?}",
352                    response.errors,
353                )
354            })?;
355            let checkpoint_data = data
356                .checkpoint
357                .ok_or_else(|| anyhow!("checkpoint {checkpoint} not found for address seeding"))?;
358            let scoped_query = checkpoint_data
359                .query
360                .ok_or_else(|| anyhow!("missing checkpoint-scoped query for address seeding"))?;
361            let Some(address_data) = scoped_query.address else {
362                return Ok(entries);
363            };
364            let Some(objects) = address_data.objects else {
365                return Ok(entries);
366            };
367
368            for node in objects.nodes {
369                entries.push(decode_move_object(node)?);
370            }
371
372            if !objects.page_info.has_next_page {
373                break;
374            }
375            cursor = objects.page_info.end_cursor;
376        }
377
378        Ok(entries)
379    }
380
381    fn decode_move_object(node: MoveObject) -> Result<AddressOwnedObject, Error> {
382        let object_id = node
383            .address
384            .0
385            .parse::<ObjectID>()
386            .with_context(|| format!("invalid seed object id {}", node.address.0))?;
387        let digest = node
388            .digest
389            .ok_or_else(|| anyhow!("seed object {object_id} is missing digest"))?
390            .parse::<ObjectDigest>()
391            .with_context(|| format!("invalid seed object digest for {object_id}"))?;
392        let version = node
393            .version
394            .ok_or_else(|| anyhow!("seed object {object_id} is missing version"))?;
395
396        Ok(AddressOwnedObject {
397            object_ref: (object_id, SequenceNumber::from_u64(version), digest),
398        })
399    }

672    const MAX_KEYS_SIZE: usize = 30;
673
674    pub(crate) async fn query(
675        object_ids: &[ObjectID],
676        checkpoint: CheckpointSequenceNumber,
677        client: &GraphQLClient,
678    ) -> Result<Vec<ObjectSeedMetadata>, Error> {
679        let mut results = Vec::with_capacity(object_ids.len());
680
681        for object_ids in object_ids.chunks(MAX_KEYS_SIZE) {
682            let keys = object_ids
683                .iter()
684                .map(|object_id| ObjectKey {
685                    address: SuiAddress(object_id.to_string()),
686                    version: None,
687                    root_version: None,
688                    at_checkpoint: Some(checkpoint),
689                })
690                .collect();
691            let operation = Query::build(ObjectSeedArgs { keys });
692            let response = client.run_query(&operation).await.with_context(|| {
693                format!("failed to query object seeds at checkpoint {checkpoint}")
694            })?;
695            let data = response.data.with_context(|| {
696                format!(
697                    "missing data in object seed query response at checkpoint {checkpoint}: {:?}",
698                    response.errors,
699                )
700            })?;
701
702            if data.multi_get_objects.len() != object_ids.len() {
703                return Err(anyhow!(
704                    "object seed query returned {} results for {} object IDs",
705                    data.multi_get_objects.len(),
706                    object_ids.len(),
707                ));
708            }
709
710            for (object_id, object) in object_ids.iter().copied().zip_eq(data.multi_get_objects) {
711                results.push(decode_object_seed(object_id, object)?);
712            }
713        }
714
715        Ok(results)
716    }

718    fn decode_object_seed(
719        object_id: ObjectID,
720        object: Option<ObjectSeedObject>,
721    ) -> Result<ObjectSeedMetadata, Error> {
722        let Some(object) = object else {
723            return Ok(ObjectSeedMetadata::Missing);
724        };
725        match object.owner {
726            Some(Owner::AddressOwner(owner)) => parse_owner_address(owner.address, &object_id)?,
727            Some(Owner::ConsensusAddressOwner(owner)) => {
728                parse_owner_address(owner.address, &object_id)?
729            }
730            _ => return Ok(ObjectSeedMetadata::NonAddressOwned),
731        };
732        let digest = object
733            .digest
734            .with_context(|| format!("seed object {object_id} is missing digest"))?

1037    /// Fetch all events for a transaction by paginating through the GraphQL
1038    /// events connection. Returns `None` if the transaction doesn't exist.
1039    /// Returns an empty `TransactionEvents` if the transaction has no events.
1040    pub(crate) async fn query(
1041        digest: &str,
1042        client: &GraphQLClient,
1043    ) -> Result<Option<TransactionEvents>, Error> {
1044        let mut all_events: Vec<Event> = Vec::new();
1045        let mut cursor: Option<String> = None;
1046
1047        loop {
1048            let operation = Query::build(EventsArgs {
1049                digest: digest.to_owned(),
1050                first: Some(PAGE_SIZE),
1051                after: cursor,
1052            });
1053            let response = client
1054                .run_query(&operation)
1055                .await
1056                .context("Failed to run events query")?;
1057
1058            let Some(connection) = response
1059                .data
1060                .and_then(|q| q.transaction)
1061                .and_then(|tx| tx.effects)
1062                .and_then(|fx| fx.events)
1063            else {
1064                if all_events.is_empty() {
1065                    return Ok(Some(TransactionEvents { data: vec![] }));
1066                }
1067                break;
1068            };
1069
1070            for node in connection.nodes {
1071                let Some(bcs_str) = node.event_bcs else {
1072                    continue;
1073                };
1074                let bytes = FastCryptoBase64::decode(&bcs_str.0)
1075                    .context(format!("Event BCS does not decode for tx {}", digest))?;
1076                let event: Event = bcs::from_bytes(&bytes)
1077                    .context(format!("Cannot deserialize event for tx {}", digest))?;
1078                all_events.push(event);
1079            }
1080
1081            if !connection.page_info.has_next_page {
1082                break;
1083            }
1084            cursor = connection.page_info.end_cursor;
1085        }
1086
1087        Ok(Some(TransactionEvents { data: all_events }))
1088    }

1173    // Maximum number of keys to query in a single request.
1174    // REVIEW: not clear how this translate to the 5000B limit, so
1175    // we are picking a "random" and conservative number.
1176    const MAX_KEYS_SIZE: usize = 30;
1177
1178    /// Query objects by keys and return a vector of the same length with the results, which contain
1179    /// an Some<(Object, Version)> tuple if the object is found, or None if not found.
1180    pub(crate) async fn query(
1181        keys: &[GqlObjectKey],
1182        data_store: &GraphQLClient,
1183    ) -> Result<Vec<Option<(Object, u64)>>, Error> {
1184        let mut results: Vec<Option<Option<(Object, u64)>>> = vec![None; keys.len()];
1185        let mut standard_indices: Vec<usize> = Vec::with_capacity(keys.len());
1186        let mut standard_keys: Vec<ObjectKey> = Vec::with_capacity(keys.len());
1187
1188        for (idx, key) in keys.iter().cloned().enumerate() {
1189            match key.version_query {
1190                VersionQuery::VersionAtCheckpoint {
1191                    version,
1192                    checkpoint,
1193                } => {
1194                    results[idx] = Some(
1195                        query_version_at_checkpoint(key.object_id, version, checkpoint, data_store)
1196                            .await?,
1197                    );
1198                }
1199                _ => {
1200                    standard_indices.push(idx);
1201                    standard_keys.push(ObjectKey::from(key));
1202                }
1203            }
1204        }
1205
1206        let mut keys = standard_keys;
1207        let mut key_chunks = vec![];
1208        while !keys.is_empty() {
1209            let chunk: Vec<_> = keys.drain(..MAX_KEYS_SIZE.min(keys.len())).collect();
1210            key_chunks.push(chunk);
1211        }
1212
1213        let mut standard_results = vec![];
1214
1215        for keys in key_chunks {
1216            let query: cynic::Operation<MultiGetObjectsQuery, MultiGetObjectsVars> =
1217                MultiGetObjectsQuery::build(MultiGetObjectsVars { keys });
1218            let response = data_store.run_query(&query).await?;
1219
1220            let list = if let Some(data) = response.data {
1221                data.multi_get_objects
1222            } else {
1223                return Err(anyhow!(
1224                    "Missing data in transaction query response. Errors: {:?}",
1225                    response.errors,
1226                ));
1227            };
1228
1229            standard_results.extend(
1230                list.into_iter()
1231                    .map(decode_object_fragment)
1232                    .collect::<Result<Vec<_>, _>>()?,
1233            );
1234        }
1235
1236        for (idx, object) in standard_indices.into_iter().zip_eq(standard_results) {
1237            results[idx] = Some(object);
1238        }
1239
1240        Ok(results
1241            .into_iter()
1242            .map(|result| result.expect("every query key should produce a result"))
1243            .collect())
1244    }

1246    async fn query_version_at_checkpoint(
1247        object_id: sui_types::base_types::ObjectID,
1248        version: u64,
1249        checkpoint: u64,
1250        data_store: &GraphQLClient,
1251    ) -> Result<Option<(Object, u64)>, Error> {
1252        let query = VersionAtCheckpointQuery::build(VersionAtCheckpointVars {
1253            sequence_number: Some(checkpoint),
1254            address: SuiAddress(object_id.to_string()),
1255            version: Some(version),
1256        });
1257        let response = data_store.run_query(&query).await?;
1258        let checkpoint = response
1259            .data
1260            .and_then(|data| data.checkpoint)
1261            .ok_or_else(|| anyhow!("Missing checkpoint in object query response"))?;
1262        let scoped_query = checkpoint
1263            .query
1264            .ok_or_else(|| anyhow!("Missing checkpoint-scoped query in object response"))?;
1265
1266        decode_object_fragment(scoped_query.object)
1267    }

1269    fn decode_object_fragment(
1270        frag: Option<ObjectFragment>,
1271    ) -> Result<Option<(Object, u64)>, Error> {
1272        match frag {
1273            Some(frag) => {
1274                let b64 = frag
1275                    .object_bcs
1276                    .ok_or_else(|| anyhow!("Object bcs is None for object"))?
1277                    .0;
1278                let bytes = FastCryptoBase64::decode(&b64)?;
1279                let obj: Object = bcs::from_bytes(&bytes)?;
1280                let version = frag
1281                    .version
1282                    .ok_or_else(|| anyhow!("Object version is None for object"))?;
1283                Ok(Some((obj, version)))
1284            }
1285            None => Ok(None),
1286        }
1287    }

1442    /// Fetch a checkpoint by sequence number (or the latest if `None`), returning
1443    /// both the verified summary envelope and the decoded checkpoint contents.
1444    pub(crate) async fn query(
1445        sequence_number: Option<u64>,
1446        data_store: &GraphQLClient,
1447    ) -> Result<Option<(VerifiedCheckpoint, CheckpointContents)>, Error> {
1448        let query = Query::build(CheckpointArgs { sequence_number });
1449        let response = data_store.run_query(&query).await?;
1450        let Some(checkpoint) = response.data.and_then(|data| data.checkpoint) else {
1451            return Ok(None);
1452        };
1453        Ok(Some(decode_checkpoint(checkpoint)?))
1454    }
1455
1456    fn decode_checkpoint(
1457        checkpoint: Checkpoint,
1458    ) -> Result<(VerifiedCheckpoint, CheckpointContents), Error> {
1459        let contents: CheckpointContents =
1460            decode_bcs(checkpoint.content_bcs, "checkpoint contents")?;
1461        let summary: CheckpointSummary = decode_bcs(checkpoint.summary_bcs, "checkpoint summary")?;
1462        let Some(validator_signatures) = checkpoint.validator_signatures else {
1463            return Err(anyhow!(
1464                "Missing validator signatures in checkpoint response"
1465            ));
1466        };
1467
1468        let signature_bytes = FastCryptoBase64::decode(
1469            &validator_signatures
1470                .signature
1471                .ok_or_else(|| anyhow!("Missing aggregated checkpoint signature"))?
1472                .0,
1473        )
1474        .context("checkpoint signature does not decode")?;
1475        let signature = AggregateAuthoritySignature::from_bytes(&signature_bytes)
1476            .context("cannot deserialize aggregated checkpoint signature")?;
1477        let signers_map = validator_signatures
1478            .signers_map
1479            .into_iter()
1480            .map(|signer| {
1481                u32::try_from(signer)
1482                    .map_err(|_| anyhow!("negative signer index in checkpoint signature"))
1483            })
1484            .collect::<Result<RoaringBitmap, Error>>()?;
1485
1486        let certified = CertifiedCheckpointSummary::new_from_data_and_sig(
1487            summary.clone(),
1488            AuthorityStrongQuorumSignInfo {
1489                epoch: summary.epoch,
1490                signature,
1491                signers_map,
1492            },
1493        );
1494        if summary.content_digest != *contents.digest() {
1495            return Err(anyhow!(
1496                "checkpoint summary content digest does not match decoded contents: \
1497                 summary={:?} contents={:?}",
1498                summary.content_digest,
1499                contents.digest(),
1500            ));
1501        }
1502        // TODO: should we fetch the committee and pass that into try_into_verified instead of
1503        // constructing this with new_unchecked?
1504        Ok((VerifiedCheckpoint::new_unchecked(certified), contents))
1505    }
1506
1507    fn decode_bcs<T>(field: Option<Base64>, label: &str) -> Result<T, Error>
1508    where
1509        T: serde::de::DeserializeOwned,
1510    {
1511        let bytes = FastCryptoBase64::decode(
1512            &field
1513                .ok_or_else(|| anyhow!("Missing {} in checkpoint response", label))?
1514                .0,
1515        )
1516        .with_context(|| format!("{} does not decode", label))?;
1517        bcs::from_bytes(&bytes).with_context(|| format!("cannot deserialize {}", label))
1518    }
