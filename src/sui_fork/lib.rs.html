44/// Signed transaction envelope paired with its execution effects and the checkpoint
45/// it was finalized in. The checkpoint is used by [`crate::store::DataStore`] as a
46/// pre-fork guard: remote results whose `checkpoint > forked_at_checkpoint` must not
47/// leak into a fork that has already diverged from the upstream chain.
48#[derive(Clone, Debug)]
49pub(crate) struct TransactionInfo {
50    pub(crate) transaction: VerifiedTransaction,
51    pub(crate) effects: TransactionEffects,
52    pub(crate) checkpoint: CheckpointSequenceNumber,
53}
54
55/// `TransactionRead` trait is used to retrieve transaction data for a given digest.
56pub(crate) trait TransactionRead {
57    /// Given a transaction digest, return the signed transaction, its effects, and the
58    /// checkpoint it was finalized in. Returns `None` if the transaction is not found.
59    fn transaction_data_and_effects(
60        &self,
61        tx_digest: &str,
62    ) -> Result<Option<TransactionInfo>, Error>;
63}

65/// Query for an object.
66/// Specifies an `ObjectID` and the "rule" to retrieve it.
67#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
68pub(crate) struct ObjectKey {
69    pub(crate) object_id: ObjectID,
70    pub(crate) version_query: VersionQuery,
71}
72
73/// Query options for an object.
74/// `Version` request an object at a specific version, or latest if no version is provided
75/// `RootVersion` request an object at a given version at most (<=)
76/// `AtCheckpoint` request an object at a given checkpoint. Useful for unknown `Version`.
77/// `VersionAtCheckpoint` requests an exact version, but only if it existed by the given checkpoint.
78#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
79pub(crate) enum VersionQuery {
80    Version(u64),
81    RootVersion(u64),
82    AtCheckpoint(u64),
83    VersionAtCheckpoint { version: u64, checkpoint: u64 },
84}
85
86/// The `ObjectRead` trait is used to retrieve objects by their keys, with different query options.
87pub(crate) trait ObjectRead {
88    /// Retrieve objects by their keys, with different query options.
89    ///
90    /// If the object is not found, the element in the vector is `None`.
91    /// Otherwise each tuple contains:
92    /// - `Object`: The object data
93    /// - `u64`: The actual version of the object
94    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error>;
95}

97/// Checkpoint read data.
98pub(crate) trait CheckpointRead {
99    /// Return the verified checkpoint summary together with its decoded
100    /// contents. If `sequence` is `None`, return the latest checkpoint.
101    fn get_checkpoint(
102        &self,
103        sequence: Option<CheckpointSequenceNumber>,
104    ) -> Result<Option<(VerifiedCheckpoint, CheckpointContents)>, Error>;
105}
