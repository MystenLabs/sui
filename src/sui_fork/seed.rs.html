4//! Fork manifest and seed resolution for lazy owned-object index initialization.
5//!
6//! The manifest is written for every initialized fork directory. Address and explicit object
7//! seeds resolve lightweight object-ref metadata at the fork checkpoint. Full object BCS is fetched
8//! later when an object read or lazy owned-object index initialization needs it.

30/// CLI seed input before it has been resolved against the upstream chain.
31#[derive(Clone, Debug, Default)]
32pub struct SeedInput {
33    /// Addresses whose owned objects should be recorded in the seed manifest.
34    pub addresses: Vec<SuiAddress>,
35    /// Object IDs to fetch and seed when they are owned by an address.
36    pub object_ids: Vec<ObjectID>,
37}
38
39/// Object reference used to seed lazy owned-object index initialization.
40#[derive(Clone, Debug, Eq, PartialEq, Serialize, Deserialize)]
41pub(crate) struct SeedEntry {
42    pub(crate) object_ref: ObjectRef,
43}
44
45/// Durable manifest for fork metadata and optional pre-fork seed metadata.
46#[derive(Clone, Debug, Eq, PartialEq, Serialize, Deserialize)]
47pub(crate) struct SeedManifest {
48    pub(crate) network: String,
49    pub(crate) checkpoint: CheckpointSequenceNumber,
50    pub(crate) entries: Vec<SeedEntry>,
51}
52
53impl SeedManifest {
54    fn empty(network: String, checkpoint: CheckpointSequenceNumber) -> Self {
55        Self {
56            network,
57            checkpoint,
58            entries: Vec::new(),
59        }
60    }
61}
62
63impl SeedInput {
64    /// Return true when no addresses or objects were requested for seeding.
65    pub fn is_empty(&self) -> bool {
66        self.addresses.is_empty() && self.object_ids.is_empty()
67    }
68}
69
70impl From<AddressOwnedObject> for SeedEntry {
71    fn from(object: AddressOwnedObject) -> Self {
72        Self {
73            object_ref: object.object_ref,
74        }
75    }
76}

78/// Reject seed inputs that would overwrite or reinterpret an existing manifest.
79pub(crate) fn ensure_seed_policy(data_store: &DataStore, input: &SeedInput) -> Result<(), Error> {
80    if data_store.local().seed_manifest_exists() && !input.is_empty() {
81        bail!(
82            "A seed manifest already exists at {}. To fork the same checkpoint with different seeds, use a different --data-dir.",
83            data_store.local().seed_manifest_path().display(),
84        );
85    }
86    Ok(())
87}
88
89/// Ensure an existing fork manifest matches the requested network and checkpoint.
90pub(crate) fn ensure_seed_manifest_matches(
91    manifest: &SeedManifest,
92    network: &str,
93    checkpoint: Option<CheckpointSequenceNumber>,
94) -> Result<(), Error> {
95    if manifest.network != network {
96        bail!(
97            "Seed manifest network {} does not match requested network {}. Use a different --data-dir.",
98            manifest.network,
99            network,
100        );
101    }
102
103    if let Some(checkpoint) = checkpoint
104        && manifest.checkpoint != checkpoint
105    {
106        bail!(
107            "Seed manifest checkpoint {} does not match requested checkpoint {}. Use a different --data-dir.",
108            manifest.checkpoint,
109            checkpoint,
110        );
111    }
112
113    Ok(())
114}

116/// Load or create the seed manifest for the current fork directory.
117pub(crate) async fn prepare_seed_manifest(
118    data_store: &DataStore,
119    network: String,
120    input: &SeedInput,
121) -> Result<SeedManifest, Error> {
122    if data_store.local().seed_manifest_exists() {
123        if !input.is_empty() {
124            bail!(
125                "A seed manifest already exists at {}. To fork the same checkpoint with different seeds, use a different --data-dir.",
126                data_store.local().seed_manifest_path().display(),
127            );
128        }
129        let manifest = data_store.local().read_seed_manifest()?;
130        ensure_seed_manifest_matches(&manifest, &network, Some(data_store.forked_at_checkpoint()))?;
131        return Ok(manifest);
132    }
133
134    let manifest = if input.is_empty() {
135        SeedManifest::empty(network, data_store.forked_at_checkpoint())
136    } else {
137        resolve_seeds(input, network, data_store).await?
138    };
139    data_store.local().write_seed_manifest(&manifest)?;
140    Ok(manifest)
141}

143fn dedupe_addresses(addresses: &[SuiAddress]) -> Vec<SuiAddress> {
144    addresses
145        .iter()
146        .copied()
147        .collect::<BTreeSet<_>>()
148        .into_iter()
149        .collect()
150}
151
152fn dedupe_object_ids(object_ids: &[ObjectID]) -> Vec<ObjectID> {
153    object_ids
154        .iter()
155        .copied()
156        .collect::<BTreeSet<_>>()
157        .into_iter()
158        .collect()
159}

161fn ensure_address_seeding_available(
162    gql: &GraphQLClient,
163    checkpoint: CheckpointSequenceNumber,
164) -> Result<(), Error> {
165    let lowest_available = gql.get_lowest_available_checkpoint_objects()?;
166    if checkpoint < lowest_available {
167        bail!(
168            "address seeding is unavailable at checkpoint {checkpoint}; object ownership enumeration is available starting at checkpoint {lowest_available}. Use --object for older checkpoints.",
169        );
170    }
171    Ok(())
172}
173
174async fn resolve_address_seed(
175    gql: &GraphQLClient,
176    address: SuiAddress,
177    checkpoint: CheckpointSequenceNumber,
178) -> Result<Vec<SeedEntry>, Error> {
179    Ok(gql
180        .get_address_owned_objects_at_checkpoint(address, checkpoint)
181        .await?
182        .into_iter()
183        .map(SeedEntry::from)
184        .collect())
185}
186
187async fn resolve_object_seeds(
188    gql: &GraphQLClient,
189    checkpoint: CheckpointSequenceNumber,
190    object_ids: &[ObjectID],
191) -> Result<Vec<SeedEntry>, Error> {
192    if object_ids.is_empty() {
193        return Ok(Vec::new());
194    }
195
196    let objects = gql
197        .get_object_seed_metadata_at_checkpoint(object_ids, checkpoint)
198        .await?;
199    let mut entries = Vec::new();
200
201    for (object_id, object) in object_ids.iter().zip_eq(objects) {
202        match object {
203            ObjectSeedMetadata::Missing => {
204                warn!(%object_id, checkpoint, "object seed not found at fork checkpoint");
205            }
206            ObjectSeedMetadata::NonAddressOwned => {
207                warn!(
208                    %object_id,
209                    checkpoint,
210                    "object seed is not owned by an address and will not be added to the seed manifest",
211                );
212            }
213            ObjectSeedMetadata::AddressOwned(object) => entries.push(SeedEntry::from(object)),
214        }
215    }
216
217    Ok(entries)
218}
219
220async fn resolve_seeds(
221    input: &SeedInput,
222    network: String,
223    data_store: &DataStore,
224) -> Result<SeedManifest, Error> {
225    let checkpoint = data_store.forked_at_checkpoint();
226    let mut entries = BTreeMap::new();
227
228    if !input.addresses.is_empty() {
229        ensure_address_seeding_available(data_store.gql(), checkpoint)?;
230    }
231
232    for address in dedupe_addresses(&input.addresses) {
233        let address_entries = resolve_address_seed(data_store.gql(), address, checkpoint).await?;
234        if address_entries.is_empty() {
235            warn!(%address, checkpoint, "address seed resolved no owned objects");
236        }
237        for entry in address_entries {
238            entries.insert(entry.object_ref.0, entry);
239        }
240    }
241
242    let remaining_object_ids: Vec<_> = dedupe_object_ids(&input.object_ids)
243        .into_iter()
244        .filter(|object_id| !entries.contains_key(object_id))
245        .collect();
246    for entry in resolve_object_seeds(data_store.gql(), checkpoint, &remaining_object_ids).await? {
247        entries.insert(entry.object_ref.0, entry);
248    }
249
250    Ok(SeedManifest {
251        network,
252        checkpoint,
253        entries: entries.into_values().collect(),
254    })
255}

272    fn object_seed_response_body(
273        object: &Object,
274        owner: SuiAddress,
275        owner_type: &str,
276    ) -> serde_json::Value {
277        json!({
278            "data": {
279                "multiGetObjects": [{
280                    "version": object.version().value(),
281                    "digest": object.digest().to_string(),
282                    "owner": {
283                        "__typename": owner_type,
284                        "address": { "address": owner.to_string() },
285                    },
286                }]
287            }
288        })
289    }

291    fn assert_object_seed_query_shape(query: &str) {
292        assert!(query.contains("multiGetObjects"));
293        assert!(query.contains("version"));
294        assert!(query.contains("digest"));
295        assert!(query.contains("... on AddressOwner"));
296        assert!(query.contains("... on ConsensusAddressOwner"));
297        assert!(!query.contains("objectBcs"));
298
299        let object_selection_before_owner = query
300            .split("multiGetObjects")
301            .nth(1)
302            .expect("query should include multiGetObjects")
303            .split("owner")
304            .next()
305            .expect("query should include owner");
306        assert!(
307            !object_selection_before_owner
308                .lines()
309                .any(|line| line.trim() == "address"),
310            "object seed query should not request Object.address",
311        );
312    }
