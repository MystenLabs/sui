40/// Checkpoint selected for startup, plus whether existing local fork state was found.
41#[derive(Clone, Copy, Debug, Eq, PartialEq)]
42pub(crate) struct ResolvedStartCheckpoint {
43    pub(crate) checkpoint: Option<CheckpointSequenceNumber>,
44    pub(crate) resuming: bool,
45}
46
47/// Resolve the fork checkpoint from local state when possible.
48///
49/// Explicit data directories can be inspected directly before the remote
50/// GraphQL client asks for latest. Default roots cannot be inspected without a
51/// checkpoint because their path contains the checkpoint.
52pub(crate) fn resolve_start_checkpoint_from_local(
53    node: &Node,
54    requested_checkpoint: Option<CheckpointSequenceNumber>,
55    data_dir: Option<&Path>,
56) -> Result<ResolvedStartCheckpoint> {
57    let local = match (data_dir, requested_checkpoint) {
58        (Some(data_dir), _) => Some(FilesystemStore::new_with_root(data_dir.to_path_buf())),
59        (None, Some(checkpoint)) => Some(FilesystemStore::new(node, checkpoint, None)?),
60        (None, None) => None,
61    };
62
63    let Some(local) = local else {
64        return Ok(ResolvedStartCheckpoint {
65            checkpoint: requested_checkpoint,
66            resuming: false,
67        });
68    };
69
70    if local.seed_manifest_exists() {
71        let manifest = local.read_seed_manifest()?;
72        ensure_seed_manifest_matches(&manifest, &node.network_name(), requested_checkpoint)?;
73        return Ok(ResolvedStartCheckpoint {
74            checkpoint: Some(manifest.checkpoint),
75            resuming: true,
76        });
77    }
78
79    if requested_checkpoint.is_some() {
80        return Ok(ResolvedStartCheckpoint {
81            checkpoint: requested_checkpoint,
82            resuming: local
83                .get_highest_checkpoint_sequence_number_if_exists()?
84                .is_some(),
85        });
86    }
87
88    let checkpoint = local.get_highest_checkpoint_sequence_number_if_exists()?;
89    Ok(ResolvedStartCheckpoint {
90        checkpoint,
91        resuming: checkpoint.is_some(),
92    })
93}

95/// Initialize a forked network by fetching the fork checkpoint from the remote
96/// endpoint when needed, applying seed metadata, and starting a local Simulacrum
97/// instance from the highest checkpoint already persisted locally. Also builds
98/// the checkpoint subscription broker; the returned handle must be passed to
99/// [`run`] so the gRPC server exposes the streaming RPC.
100///
101/// `data_dir` is the root folder where the fork state is persisted. If `None`, a default path is
102/// used. See the `[FileSystemStore]` docs for details.
103pub async fn initialize(
104    node: Node,
105    forked_at_checkpoint: CheckpointSequenceNumber,
106    version: &str,
107    data_dir: Option<PathBuf>,
108    seed_input: SeedInput,
109) -> Result<(Context, SubscriptionServiceHandle)> {
110    // 1. Create DataStore — empty local cache, GraphQL wired up.
111    let data_store = DataStore::new(node.clone(), forked_at_checkpoint, version, data_dir).await?;
112    let chain_identifier = data_store.chain();
113    crate::seed::ensure_seed_policy(&data_store, &seed_input)?;
114
115    // 2. Download and persist the startup checkpoint (summary + contents),
116    //    then read the summary back through the cache-aware getter.
117    data_store.download_and_persist_startup_checkpoint()?;
118    crate::seed::prepare_seed_manifest(&data_store, node.network_name(), &seed_input).await?;
119    let checkpoint = data_store
120        .get_highest_verified_checkpoint()?
121        .ok_or_else(|| anyhow!("checkpoint {} not found", forked_at_checkpoint))?;
122
123    // 3. Read system state — fetches object 0x5 + dynamic fields from remote via ObjectStore.
124    let system_state =
125        sui_types::sui_system_state::get_sui_system_state(&data_store).map_err(|e| {
126            anyhow!(
127                "failed to read system state at checkpoint {}: {}",
128                forked_at_checkpoint,
129                e
130            )
131        })?;
132
133    let protocol_version = system_state.protocol_version();
134
135    // 4. Build NetworkConfig with local test validators.
136    let mut rng = OsRng;
137    let config = ConfigBuilder::new_with_temp_dir()
138        .rng(&mut rng)
139        .with_chain_start_timestamp_ms(checkpoint.timestamp_ms)
140        .deterministic_committee_size(NonZeroUsize::MIN)
141        .with_protocol_version(ProtocolVersion::new(protocol_version))
142        .with_chain_override(node.chain())
143        .build();
144
145    // 5. Override validators in system state with local keys from config.
146    let system_state = override_validators(system_state, &config)?;
147
148    // 6. Build KeyStore.
149    let keystore = KeyStore::from_network_config(&config);
150
151    // 7. Create Simulacrum from custom state.
152    let simulacrum = Simulacrum::new_from_custom_state(
153        keystore,
154        checkpoint,
155        system_state,
156        &config,
157        data_store,
158        rng,
159    );
160
161    // 8. Build the checkpoint subscription broker. The sender is owned by
162    //    `Context` (producers push here on `advance_checkpoint`); the handle
163    //    is wired into `RpcService` in `run` so subscribers can register.
164    let registry = Registry::new();
165    let (checkpoint_sender, subscription_handle) = SubscriptionService::build(&registry, None);
166
167    Ok((
168        Context::new(simulacrum, chain_identifier, checkpoint_sender),
169        subscription_handle,
170    ))
171}

173/// Run the forked network. Spawns a `sui-rpc-api` gRPC server bound to
174/// `rpc_addr` on top of the context's `DataStore`, then blocks on Ctrl+C.
175pub async fn run(
176    context: Context,
177    subscription_handle: SubscriptionServiceHandle,
178    rpc_addr: SocketAddr,
179    version: &'static str,
180) -> Result<()> {
181    let reader: Arc<dyn RpcStateReader> = {
182        let sim = context.simulacrum().read().await;
183        Arc::new(sim.store().clone())
184    };
185    let mut service = RpcService::new(reader);
186    service.with_server_version(ServerVersion::new("sui-fork", version));
187    service.with_subscription_service(subscription_handle);
188    let context = Arc::new(context);
189    service.with_executor(Arc::new(ForkedTransactionExecutor::new(context.clone())));
190    service.with_custom_service(ForkingServiceServer::new(ForkingServiceImpl::new(
191        context.clone(),
192    )));
193    service.with_file_descriptor_set(crate::proto::FILE_DESCRIPTOR_SET);
194
195    info!("starting sui-rpc-api server on {rpc_addr}");
196    let server_handle = tokio::spawn(async move { service.start_service(rpc_addr).await });
197
198    info!("forked network running, waiting for shutdown signal (Ctrl+C)");
199    tokio::select! {
200        res = tokio::signal::ctrl_c() => {
201            res?;
202            info!("shutdown signal received, stopping forked network");
203        }
204        join = server_handle => {
205            if let Err(e) = join {
206                return Err(anyhow!("rpc server task panicked: {e}"));
207            }
208            return Err(anyhow!("rpc server task exited unexpectedly"));
209        }
210    }
211    Ok(())
212}

214/// Replace the validator set in the system state with local validators from the NetworkConfig
215/// genesis, so the simulacrum can sign checkpoints with locally available keys.
216fn override_validators(
217    system_state: SuiSystemState,
218    config: &NetworkConfig,
219) -> Result<SuiSystemState> {
220    let genesis_validators = match config.genesis.sui_system_object() {
221        SuiSystemState::V1(inner) => inner.validators,
222        SuiSystemState::V2(inner) => inner.validators,
223        #[cfg(msim)]
224        _ => anyhow::bail!("unsupported genesis system state variant"),
225    };
226
227    match system_state {
228        SuiSystemState::V1(mut inner) => {
229            inner.validators = genesis_validators;
230            Ok(SuiSystemState::V1(inner))
231        }
232        SuiSystemState::V2(mut inner) => {
233            inner.validators = genesis_validators;
234            Ok(SuiSystemState::V2(inner))
235        }
236        #[cfg(msim)]
237        _ => anyhow::bail!("unsupported system state variant"),
238    }
239}

250    fn write_manifest(root: &Path, network: &str, checkpoint: CheckpointSequenceNumber) {
251        FilesystemStore::new_with_root(root.to_path_buf())
252            .write_seed_manifest(&SeedManifest {
253                network: network.to_owned(),
254                checkpoint,
255                entries: Vec::new(),
256            })
257            .expect("seed manifest should write");
258    }
259
260    fn write_checkpoint(root: &Path, sequence: CheckpointSequenceNumber) {
261        let checkpoint = VerifiedCheckpoint::new_unchecked(
262            TestCheckpointBuilder::new(sequence)
263                .build_checkpoint()
264                .summary,
265        );
266        FilesystemStore::new_with_root(root.to_path_buf())
267            .write_checkpoint_summary(&checkpoint)
268            .expect("checkpoint summary should write");
269    }

271    #[test]
272    fn resolve_start_checkpoint_uses_manifest_when_checkpoint_is_omitted() {
273        let temp = tempfile::tempdir().expect("tempdir");
274        write_manifest(temp.path(), "mainnet", 42);
275
276        let resolved = resolve_start_checkpoint_from_local(&Node::Mainnet, None, Some(temp.path()))
277            .expect("checkpoint resolution should not fail");
278
279        assert_eq!(
280            resolved,
281            ResolvedStartCheckpoint {
282                checkpoint: Some(42),
283                resuming: true,
284            }
285        );
286    }

288    #[test]
289    fn resolve_start_checkpoint_rejects_requested_checkpoint_mismatch() {
290        let temp = tempfile::tempdir().expect("tempdir");

292
293        let err = resolve_start_checkpoint_from_local(&Node::Mainnet, Some(43), Some(temp.path()))
294            .expect_err("checkpoint mismatch should fail");
