17/// Encapsulation of an entry in the manifest file corresponding to a single version of the system
18/// packages.
19///
20// Note: the [Snapshot] and [SnapshotPackage] types are similar to the
21// [sui_framework::{SystemPackageMetadata, SystemPackage}] types,
22// and also to the [sui::framework_versions::{FrameworkVersion, FrameworkPackage}] types.
23// They are sort of a stepping stone from one to the other - the [sui_framework] types contain
24// additional information about the compiled bytecode of the package, while the
25// [framework_versions] types do not contain information about the object IDs of the packages.
26//
27// These types serve as a kind of stepping stone; they are constructed from the [sui_framework]
28// types and serialized in the manifest, and then the build script for the [sui] crate reads them
29// from the manifest file and encodes them in the `sui` binary. A little information is dropped in
30// each of these steps.
31#[derive(Serialize, Deserialize)]
32pub struct Snapshot {
33    /// Git revision that this snapshot is taken on.
34    pub git_revision: String,
35
36    /// List of system packages in this version
37    pub packages: Vec<SnapshotPackage>,
38}
39
40/// Entry in the manifest file corresponding to a specific version of a specific system package
41#[derive(Serialize, Deserialize)]
42pub struct SnapshotPackage {
43    /// Name of the package (e.g. "MoveStdLib")
44    pub name: String,
45    /// Path to the package in the monorepo (e.g. "crates/sui-framework/packages/move-stdlib")
46    pub path: String,
47    /// Object ID of the published package
48    pub id: ObjectID,
49}
50
51impl Snapshot {
52    pub fn package_ids(&self) -> impl Iterator<Item = ObjectID> + '_ {
53        self.packages.iter().map(|p| p.id)
54    }
55}
56
57impl SnapshotPackage {
58    pub fn from_system_package_metadata(value: &SystemPackageMetadata) -> Self {
59        Self {
60            name: value.name.clone(),
61            path: value.path.clone(),
62            id: value.compiled.id,
63        }
64    }
65}

67const SYSTEM_PACKAGE_PUBLISH_ORDER: &[ObjectID] = &[
68    MOVE_STDLIB_PACKAGE_ID,
69    SUI_FRAMEWORK_PACKAGE_ID,
70    SUI_SYSTEM_PACKAGE_ID,
71    DEEPBOOK_PACKAGE_ID,
72    BRIDGE_PACKAGE_ID,
73];

75pub fn load_bytecode_snapshot_manifest() -> SnapshotManifest {
76    let Ok(bytes) = fs::read(manifest_path()) else {
77        return SnapshotManifest::default();
78    };
79    serde_json::from_slice::<SnapshotManifest>(&bytes)
80        .expect("Could not deserialize SnapshotManifest")
81}
82
83pub fn update_bytecode_snapshot_manifest(
84    git_revision: &str,
85    version: u64,
86    files: Vec<SnapshotPackage>,
87) {
88    let mut snapshot = load_bytecode_snapshot_manifest();
89
90    snapshot.insert(
91        version,
92        Snapshot {
93            git_revision: git_revision.to_string(),
94            packages: files,
95        },
96    );
97
98    let json =
99        serde_json::to_string_pretty(&snapshot).expect("Could not serialize SnapshotManifest");
100    fs::write(manifest_path(), json).expect("Could not update manifest file");
101}

103pub fn load_bytecode_snapshot(protocol_version: u64) -> anyhow::Result<Vec<SystemPackage>> {
104    let snapshot_path = snapshot_path_for_version(protocol_version)?;
105    let mut snapshots: BTreeMap<ObjectID, SystemPackage> = fs::read_dir(&snapshot_path)?
106        .flatten()
107        .map(|entry| {
108            let file_name = entry.file_name().to_str().unwrap().to_string();
109            let mut file = fs::File::open(snapshot_path.clone().join(file_name))?;
110            let mut buffer = Vec::new();
111            file.read_to_end(&mut buffer)?;
112            let package: SystemPackage = bcs::from_bytes(&buffer)?;
113            Ok((package.id, package))
114        })
115        .collect::<anyhow::Result<_>>()?;
116
117    // system packages need to be restored in a specific order
118    assert!(snapshots.len() <= SYSTEM_PACKAGE_PUBLISH_ORDER.len());
119    let mut snapshot_objects = Vec::new();
120    for package_id in SYSTEM_PACKAGE_PUBLISH_ORDER {
121        if let Some(object) = snapshots.remove(package_id) {
122            snapshot_objects.push(object);
123        }
124    }
125    Ok(snapshot_objects)
126}

128pub fn manifest_path() -> PathBuf {
129    PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("manifest.json")
130}
131
132fn snapshot_path_for_version(version: u64) -> anyhow::Result<PathBuf> {
133    let snapshot_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("bytecode_snapshot");
134    let mut snapshots = BTreeSet::new();
135
136    for entry in fs::read_dir(&snapshot_dir)? {
137        let entry = entry?;
138        let path = entry.path();
139        if path.is_dir()
140            && let Some(snapshot_number) = path
141                .file_name()
142                .and_then(|n| n.to_str())
143                .and_then(|n| n.parse::<u64>().ok())
144        {
145            snapshots.insert(snapshot_number);
146        }
147    }
148
149    if version == ProtocolVersion::MAX.as_u64() && !snapshots.contains(&version) {
150        anyhow::bail!("No snapshot found for version {}", version)
151    }
152
153    snapshots
154        .range(..=version)
155        .next_back()
156        .map(|v| snapshot_dir.join(v.to_string()))
157        .ok_or_else(|| anyhow::anyhow!("No snapshot found for version {}", version))
158}
