23/// GraphQL fragment to query the values of the dynamic fields.
24const QUERY_FRAGMENT: &str =
25    "fragment RECORD_VALUES on DynamicField { value { ... on MoveValue { bcs } } }";

28    client: reqwest::Client,
29    config: MoveRegistryConfig,
30    metrics: Metrics,

33/// Helper types for accessing a shared `DataLoader` instance.
34#[derive(Clone)]
35pub(crate) struct MoveRegistryDataLoader(pub Arc<DataLoader<ExternalNamesLoader>>);
36
37impl ExternalNamesLoader {
38    pub(crate) fn new(config: MoveRegistryConfig, metrics: Metrics) -> Self {
39        Self {
40            client: reqwest::Client::new(),
41            config,
42            metrics,
43        }
44    }
45
46    /// Constructs the GraphQL Query to query the names on an external graphql endpoint.
47    fn construct_names_graphql_query(&self, names: &[Name]) -> (String, HashMap<Name, usize>) {
48        let mut mapping: HashMap<Name, usize> = HashMap::new();
49
50        let mut result = format!(r#"{{ owner(address: "{}") {{"#, self.config.registry_id);
51
52        // we create the GraphQL query keys with a `fetch_{id}` prefix, which is accepted on graphql fields.
53        for (index, name) in names.iter().enumerate() {
54            let bcs_base64 = name.to_base64_string();
55
56            // retain the mapping here (id to bcs representation, so we can pick the right response later on)
57            mapping.insert(name.clone(), index);
58
59            // SAFETY: write! to String always succeeds
60            write!(
61                &mut result,
62                r#"{}: dynamicField(name: {{ type: "{}::name::Name", bcs: {} }}) {{ ...RECORD_VALUES }} "#,
63                fetch_key(&index),
64                self.config.package_address,
65                bcs_base64
66            ).unwrap();
67        }
68
69        result.push_str("}} ");
70        result.push_str(QUERY_FRAGMENT);
71
72        (result, mapping)
73    }
74}

76impl MoveRegistryDataLoader {
77    pub(crate) fn new(config: MoveRegistryConfig, metrics: Metrics) -> Self {
78        let batch_size = config.page_limit as usize;
79        let data_loader = DataLoader::new(ExternalNamesLoader::new(config, metrics), tokio::spawn)
80            .max_batch_size(batch_size);
81        Self(Arc::new(data_loader))
82    }
83}

85#[async_trait::async_trait]
86impl Loader<Name> for ExternalNamesLoader {
87    type Value = AppRecord;
88    type Error = Error;
89
90    /// This function queries the external API to fetch the app records for the requested names.
91    /// This is part of the data loader, so all queries are bulked-up to the maximum of {config.page_limit}.
92    /// We handle the cases where individual queries fail, to ensure that a failed query cannot affect
93    /// a successful one.
94    async fn load(&self, keys: &[Name]) -> Result<HashMap<Name, AppRecord>, Error> {
95        let Some(api_url) = self.config.external_api_url.as_ref() else {
96            return Err(Error::MoveNameRegistry(
97                MoveRegistryError::ExternalApiUrlUnavailable,
98            ));
99        };
100
101        let (query, mapping) = self.construct_names_graphql_query(keys);
102
103        let request_body = GraphQLRequest {
104            query,
105            variables: serde_json::Value::Null,
106        };
107
108        let res = {
109            let _timer_guard = self
110                .metrics
111                .app_metrics
112                .external_mvr_resolution_latency
113                .start_timer();
114
115            self.client
116                .post(api_url)
117                .json(&request_body)
118                .send()
119                .await
120                .map_err(|e| {
121                    Error::MoveNameRegistry(MoveRegistryError::FailedToQueryExternalApi(
122                        e.to_string(),
123                    ))
124                })?
125        };
126
127        if !res.status().is_success() {
128            return Err(Error::MoveNameRegistry(
129                MoveRegistryError::FailedToQueryExternalApi(format!(
130                    "Status code: {}",
131                    res.status()
132                )),
133            ));
134        }
135
136        let response_json: GraphQLResponse<Owner> = res.json().await.map_err(|e| {
137            Error::MoveNameRegistry(MoveRegistryError::FailedToParseExternalResponse(
138                e.to_string(),
139            ))
140        })?;
141
142        let names = response_json.data.owner.names;
143
144        let results = HashMap::from_iter(mapping.into_iter().filter_map(|(k, idx)| {
145            let bcs = names.get(&fetch_key(&idx))?.as_ref()?;
146            let Base64(bytes) = Base64::from_str(&bcs.value.bcs).ok()?;
147            let app_record: AppRecord = bcs::from_bytes(&bytes).ok()?;
148            Some((k, app_record))
149        }));
150
151        Ok(results)
152    }
153}
154
155fn fetch_key(idx: &usize) -> String {
156    format!("f_{}", idx)
157}

159// GraphQL Request and Response types to deserialize for the data loader.
160#[derive(Serialize)]
161struct GraphQLRequest {
162    query: String,
163    variables: serde_json::Value,
164}
165
166#[derive(Deserialize, Debug)]
167struct GraphQLResponse<T> {
168    data: T,
169}
170#[derive(Deserialize, Debug)]
171struct Owner {
172    owner: Names,
173}
174
175#[derive(Deserialize, Debug)]
176struct Names {
177    #[serde(flatten)]
178    names: HashMap<String, Option<OwnerValue>>,
179}
180
181#[derive(Deserialize, Debug)]
182struct OwnerValue {
183    value: NameBCS,
184}
185
186#[derive(Deserialize, Debug)]
187struct NameBCS {
188    bcs: String,
189}
