21const STORE: &str = "PostgresDB";
22
23pub(crate) type PackageCache = PackageStoreWithLruCache<DbPackageStore>;
24pub(crate) type PackageResolver = Arc<Resolver<PackageCache>>;
25
26/// Store which fetches package for the given address from the backend db on every call
27/// to `fetch`
28pub struct DbPackageStore(DataLoader);
29
30/// `DataLoader` key for fetching the latest version of a `Package` by its ID.
31#[derive(Copy, Clone, Hash, Eq, PartialEq, Debug)]
32struct PackageKey(AccountAddress);
33
34impl DbPackageStore {
35    pub fn new(loader: DataLoader) -> Self {
36        Self(loader)
37    }
38}
39
40#[async_trait]
41impl PackageStore for DbPackageStore {
42    async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>> {
43        let Self(DataLoader(loader)) = self;
44        let Some(package) = loader.load_one(PackageKey(id)).await? else {
45            return Err(PackageResolverError::PackageNotFound(id));
46        };
47
48        Ok(package)
49    }
50}
51
52#[async_trait::async_trait]
53impl Loader<PackageKey> for Db {
54    type Value = Arc<Package>;
55    type Error = PackageResolverError;
56
57    async fn load(&self, keys: &[PackageKey]) -> Result<HashMap<PackageKey, Arc<Package>>> {
58        use packages::dsl;
59
60        let ids: BTreeSet<_> = keys.iter().map(|PackageKey(id)| id.to_vec()).collect();
61        let stored_packages: Vec<StoredPackage> = self
62            .execute(move |conn| {
63                async move {
64                    conn.results(move || {
65                        dsl::packages.filter(dsl::package_id.eq_any(ids.iter().cloned()))
66                    })
67                    .await
68                }
69                .scope_boxed()
70            })
71            .await
72            .map_err(|e| PackageResolverError::Store {
73                store: STORE,
74                error: e.to_string(),
75            })?;
76
77        let mut id_to_package = HashMap::new();
78        for stored_package in stored_packages {
79            let move_package = bcs::from_bytes(&stored_package.move_package)?;
80            let package = Package::read_from_package(&move_package)?;
81            id_to_package.insert(PackageKey(*move_package.id()), Arc::new(package));
82        }
83
84        Ok(id_to_package)
85    }
86}
