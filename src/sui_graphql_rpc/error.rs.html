10/// Error codes for the `extensions.code` field of a GraphQL error that originates from outside
11/// GraphQL.
12/// `<https://www.apollographql.com/docs/apollo-server/data/errors/#built-in-error-codes>`
13pub(crate) mod code {
14    pub const BAD_USER_INPUT: &str = "BAD_USER_INPUT";
15    pub const INTERNAL_SERVER_ERROR: &str = "INTERNAL_SERVER_ERROR";
16    pub const REQUEST_TIMEOUT: &str = "REQUEST_TIMEOUT";
17    pub const UNKNOWN: &str = "UNKNOWN";
18}
19
20/// Create a generic GraphQL Server Error.
21///
22/// This error has no path, source, or locations, just a message and an error code.
23pub(crate) fn graphql_error(code: &str, message: impl Into<String>) -> ServerError {
24    let mut ext = ErrorExtensionValues::default();
25    ext.set("code", code);
26
27    ServerError {
28        message: message.into(),
29        source: None,
30        locations: vec![],
31        path: vec![],
32        extensions: Some(ext),
33    }
34}
35
36pub(crate) fn graphql_error_at_pos(
37    code: &str,
38    message: impl Into<String>,
39    pos: Pos,
40) -> ServerError {
41    let mut ext = ErrorExtensionValues::default();
42    ext.set("code", code);
43
44    ServerError {
45        message: message.into(),
46        source: None,
47        locations: vec![pos],
48        path: vec![],
49        extensions: Some(ext),
50    }
51}

53#[derive(Clone, Debug, thiserror::Error)]
54pub enum Error {
55    #[error("Unsupported protocol version requested. Min supported: {0}, max supported: {1}")]
56    ProtocolVersionUnsupported(u64, u64),
57    #[error(transparent)]
58    NameService(#[from] NameServiceError),
59    #[error("'first' and 'last' must not be used together")]
60    CursorNoFirstLast,
61    #[error("Connection's page size of {0} exceeds max of {1}")]
62    PageTooLarge(u64, u32),
63    // Catch-all for client-fault errors
64    #[error("{0}")]
65    Client(String),
66    #[error("Internal error occurred while processing request: {0}")]
67    Internal(String),
68    #[error(transparent)]
69    MoveNameRegistry(#[from] MoveRegistryError),
70}
71
72impl ErrorExtensions for Error {
73    fn extend(&self) -> async_graphql::Error {
74        async_graphql::Error::new(format!("{}", self)).extend_with(|_err, e| match self {
75            Error::NameService(_)
76            | Error::MoveNameRegistry(_)
77            | Error::CursorNoFirstLast
78            | Error::PageTooLarge(_, _)
79            | Error::ProtocolVersionUnsupported(_, _)
80            | Error::Client(_) => {
81                e.set("code", code::BAD_USER_INPUT);
82            }
83            Error::Internal(_) => {
84                e.set("code", code::INTERNAL_SERVER_ERROR);
85            }
86        })
87    }
88}
89
90impl From<IndexerError> for Error {
91    fn from(e: IndexerError) -> Self {
92        Error::Internal(e.to_string())
93    }
