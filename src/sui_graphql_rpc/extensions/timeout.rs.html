9use async_graphql_value::Variables;
10use std::sync::{
11    atomic::{AtomicBool, Ordering},
12    Mutex,
13};
14use std::time::Duration;
15use std::{net::SocketAddr, sync::Arc};
16use tokio::time::timeout;
17use tracing::error;
18use uuid::Uuid;
19
20use crate::{config::ServiceConfig, error::code};
21
22/// Extension factory for creating new `Timeout` instances, per query.
23pub(crate) struct Timeout;
24
25#[derive(Debug, Default)]
26struct TimeoutExt {
27    pub query: Mutex<Option<String>>,
28    pub is_mutation: AtomicBool,
29}
30
31impl ExtensionFactory for Timeout {
32    fn create(&self) -> Arc<dyn Extension> {
33        Arc::new(TimeoutExt {
34            query: Mutex::new(None),
35            is_mutation: AtomicBool::new(false),
36        })
37    }
38}
39
40#[async_trait::async_trait]
41impl Extension for TimeoutExt {
42    async fn parse_query(
43        &self,
44        ctx: &ExtensionContext<'_>,
45        query: &str,
46        variables: &Variables,
47        next: NextParseQuery<'_>,
48    ) -> ServerResult<ExecutableDocument> {
49        let document = next.run(ctx, query, variables).await?;
50        *self.query.lock().unwrap() = Some(ctx.stringify_execute_doc(&document, variables));
51
52        let is_mutation = document
53            .operations
54            .iter()
55            .any(|(_, operation)| operation.node.ty == OperationType::Mutation);
56        self.is_mutation.store(is_mutation, Ordering::Relaxed);
57
58        Ok(document)
59    }
60
61    async fn execute(
62        &self,
63        ctx: &ExtensionContext<'_>,
64        operation_name: Option<&str>,
65        next: NextExecute<'_>,
66    ) -> Response {
67        let cfg: &ServiceConfig = ctx
68            .data()
69            .expect("No service config provided in schema data");
70
71        // increase the timeout if the request is a mutation
72        let is_mutation = self.is_mutation.load(Ordering::Relaxed);
73        let request_timeout = if is_mutation {
74            Duration::from_millis(cfg.limits.mutation_timeout_ms.into())
75        } else {
76            Duration::from_millis(cfg.limits.request_timeout_ms.into())
77        };
78
79        timeout(request_timeout, next.run(ctx, operation_name))
80            .await
81            .unwrap_or_else(|_| {
82                let query_id: &Uuid = ctx.data_unchecked();
83                let session_id: &SocketAddr = ctx.data_unchecked();
84                let error_code = code::REQUEST_TIMEOUT;
85                let guard = self.query.lock().unwrap();
86                let query = match guard.as_ref() {
87                    Some(s) => s.as_str(),
88                    None => "",
89                };
90
91                error!(
92                    %query_id,
93                    %session_id,
94                    %error_code,
95                    %query
96                );
97                let error_msg = if is_mutation {
98                    format!(
99                        "Mutation request timed out. Limit: {}s",
100                        request_timeout.as_secs_f32()
101                    )
102                } else {
103                    format!(
104                        "Query request timed out. Limit: {}s",
105                        request_timeout.as_secs_f32()
106                    )
107                };
108                Response::from_errors(vec![ServerError::new(error_msg, None)])
109            })
110    }
111}
