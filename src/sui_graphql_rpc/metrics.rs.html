92impl Metrics {
93    pub(crate) fn new(registry: &Registry) -> Self {
94        let db_metrics = DBMetrics::new(registry);
95        let request_metrics = RequestMetrics::new(registry);
96        let app_metrics = AppMetrics::new(registry);
97
98        Self {
99            db_metrics: Arc::new(db_metrics),
100            request_metrics: Arc::new(request_metrics),
101            app_metrics: Arc::new(app_metrics),
102        }
103    }
104
105    /// Updates the DB related metrics (latency, error, success)
106    pub(crate) fn observe_db_data(&self, time: Duration, succeeded: bool) {
107        let label = if succeeded { "success" } else { "error" };
108        self.db_metrics.db_fetches.with_label_values(&[label]).inc();
109        self.db_metrics
110            .db_fetch_latency
111            .with_label_values(&[label])
112            .observe(time.as_secs_f64());
113    }
114
115    /// The total time needed for handling the query
116    pub(crate) fn query_latency(&self, time: Duration) {
117        self.request_metrics
118            .query_latency
119            .observe(time.as_secs_f64());
120    }
121
122    /// The time needed for validating the query
123    pub(crate) fn query_validation_latency(&self, time: Duration) {
124        self.request_metrics
125            .query_validation_latency
126            .observe(time.as_secs_f64());
127    }
128
129    /// Increment the total number of queries by one
130    pub(crate) fn inc_num_queries(&self) {
131        self.request_metrics.num_queries.inc();
132    }
133
134    /// Use this function to increment the number of errors per path and per error type.
135    /// The error type is detected automatically from the passed errors.
136    pub(crate) fn inc_errors(&self, errors: &[ServerError]) {
137        for err in errors {
138            if let Some(ext) = &err.extensions {
139                if let Some(async_graphql_value::ConstValue::String(val)) = ext.get("code") {
140                    self.request_metrics
141                        .num_errors
142                        .with_label_values(&[query_label_for_error(&err.path).as_str(), val])
143                        .inc();
144                }
145            } else {
146                self.request_metrics
147                    .num_errors
148                    .with_label_values(&[query_label_for_error(&err.path).as_str(), code::UNKNOWN])
149                    .inc();
150            }
151        }
152    }
153}

280/// When an error occurs, GraphQL returns a vector of PathSegments,
281/// that we can use to retrieve the last node which contains the error.
282pub(crate) fn query_label_for_error(query: &[PathSegment]) -> String {
283    let fields: Vec<_> = query
284        .iter()
285        .filter_map(|s| {
286            if let PathSegment::Field(name) = s {
287                Some(name)
288            } else {
289                None
290            }
291        })
292        .collect();
293
294    match &fields[..] {
295        [] => "".to_string(),
296        [seg] => seg.to_string(),
297        [fst, .., lst] => format!("{fst}..{lst}"),
298    }
299}
