4use sui_types::SYSTEM_PACKAGE_ADDRESSES;
5use tokio::sync::watch;
6use tokio_util::sync::CancellationToken;
7use tracing::info;
8
9use crate::data::package_resolver::PackageResolver;
10
11/// Background task responsible for evicting system packages from the package resolver's cache on
12/// epoch boundaries.
13pub(crate) struct SystemPackageTask {
14    resolver: PackageResolver,
15    epoch_rx: watch::Receiver<u64>,
16    cancel: CancellationToken,
17}
18
19impl SystemPackageTask {
20    pub(crate) fn new(
21        resolver: PackageResolver,
22        epoch_rx: watch::Receiver<u64>,
23        cancel: CancellationToken,
24    ) -> Self {
25        Self {
26            resolver,
27            epoch_rx,
28            cancel,
29        }
30    }
31
32    pub(crate) async fn run(&mut self) {
33        loop {
34            tokio::select! {
35                _ = self.cancel.cancelled() => {
36                    info!("Shutdown signal received, terminating system package eviction task");
37                    return;
38                }
39
40                _ = self.epoch_rx.changed() => {
41                    info!("Detected epoch boundary, evicting system packages from cache");
42                    self.resolver
43                        .package_store()
44                        .evict(SYSTEM_PACKAGE_ADDRESSES.iter().copied());
45                }
46            }
47        }
48    }
49}
