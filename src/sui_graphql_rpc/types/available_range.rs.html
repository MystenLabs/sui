13#[derive(Clone, Debug, PartialEq, Eq, Copy)]
14pub(crate) struct AvailableRange {
15    pub first: u64,
16    pub last: u64,
17}
18
19/// Range of checkpoints that the RPC is guaranteed to produce a consistent response for.
20#[Object]
21impl AvailableRange {
22    async fn first(&self, ctx: &Context<'_>) -> Result<Option<Checkpoint>> {
23        Checkpoint::query(ctx, CheckpointId::by_seq_num(self.first), self.last)
24            .await
25            .extend()
26    }
27
28    async fn last(&self, ctx: &Context<'_>) -> Result<Option<Checkpoint>> {
29        Checkpoint::query(ctx, CheckpointId::by_seq_num(self.last), self.last)
30            .await
31            .extend()
32    }
33}
34
35impl AvailableRange {
36    /// Look up the available range when viewing the data consistently at `checkpoint_viewed_at`.
37    pub(crate) async fn query(db: &Db, checkpoint_viewed_at: u64) -> Result<Self, Error> {
38        let Some(range): Option<Self> = db
39            .execute(move |conn| {
40                async move { Self::result(conn, checkpoint_viewed_at).await }.scope_boxed()
41            })
42            .await
43            .map_err(|e| Error::Internal(format!("Failed to fetch available range: {e}")))?
44        else {
45            return Err(Error::Client(format!(
46                "Requesting data at checkpoint {checkpoint_viewed_at}, outside the available \
47                 range.",
48            )));
49        };
50
51        Ok(range)
52    }
53
54    /// Look up the available range when viewing the data consistently at `checkpoint_viewed_at`.
55    /// Made available on the `Conn` type to make it easier to call as part of other queries.
56    ///
57    /// Returns an error if there was an issue querying the database, Ok(None) if the checkpoint
58    /// being viewed is not in the database's available range, or Ok(Some(AvailableRange))
59    /// otherwise.
60    pub(crate) async fn result(
61        conn: &mut Conn<'_>,
62        checkpoint_viewed_at: u64,
63    ) -> QueryResult<Option<Self>> {
64        use checkpoints::dsl as checkpoints;
65        use objects_snapshot::dsl as snapshots;
66
67        let checkpoint_range: Vec<i64> = conn
68            .results(move || {
69                let rhs = checkpoints::checkpoints
70                    .select(checkpoints::sequence_number)
71                    .order(checkpoints::sequence_number.desc())
72                    .limit(1);
73
74                let lhs = snapshots::objects_snapshot
75                    .select(snapshots::checkpoint_sequence_number)
76                    .order(snapshots::checkpoint_sequence_number.desc())
77                    .limit(1);
78
79                // We need to use `union_all` in case `lhs` and `rhs` have the same value.
80                lhs.union_all(rhs)
81            })
82            .await?;
83
84        let (first, mut last) = match checkpoint_range.as_slice() {
85            [] => (0, 0),
86            [single_value] => (0, *single_value as u64),
87            values => {
88                let min_value = *values.iter().min().unwrap();
89                let max_value = *values.iter().max().unwrap();
90                (min_value as u64, max_value as u64)
91            }
92        };
93
94        if checkpoint_viewed_at < first || last < checkpoint_viewed_at {
95            return Ok(None);
96        }
97
98        last = checkpoint_viewed_at;
99        Ok(Some(Self { first, last }))
100    }
101}
