186    ///
187    /// `scanLimit` restricts the number of candidate transactions scanned when gathering a page of
188    /// results. It is required for queries that apply more than two complex filters (on function,
189    /// kind, sender, recipient, input object, changed object, or ids), and can be at most
190    /// `serviceConfig.maxScanLimit`.
191    ///
192    /// When the scan limit is reached the page will be returned even if it has fewer than `first`
193    /// results when paginating forward (`last` when paginating backwards). If there are more
194    /// transactions to scan, `pageInfo.hasNextPage` (or `pageInfo.hasPreviousPage`) will be set to
195    /// `true`, and `PageInfo.endCursor` (or `PageInfo.startCursor`) will be set to the last
196    /// transaction that was scanned as opposed to the last (or first) transaction in the page.
197    ///
198    /// Requesting the next (or previous) page after this cursor will resume the search, scanning
199    /// the next `scanLimit` many transactions in the direction of pagination, and so on until all
200    /// transactions in the scanning range have been visited.

202    /// By default, the scanning range includes all transactions known to GraphQL, but it can be

327    /// The overall quantity of tokens that will be issued.
328    async fn supply(&self, ctx: &Context<'_>) -> Result<Option<BigInt>> {
329        let mut type_params = self.super_.native.type_().type_params();
330        let Some(coin_type) = type_params.pop() else {
331            return Ok(None);
332        };
333
334        let supply = CoinMetadata::query_total_supply(
335            ctx.data_unchecked(),
336            coin_type,
337            self.super_.super_.checkpoint_viewed_at,
338        )
339        .await
340        .extend()?;
341
342        Ok(supply.map(BigInt::from))
343    }

346impl CoinMetadata {
347    /// Read a `CoinMetadata` from the `db` for the coin whose inner type is `coin_type`.
348    pub(crate) async fn query(
349        db: &Db,
350        coin_type: TypeTag,
351        checkpoint_viewed_at: u64,
352    ) -> Result<Option<CoinMetadata>, Error> {
353        let TypeTag::Struct(coin_struct) = coin_type else {
354            // If the type supplied is not metadata, we know it's not a valid coin type, so there
355            // won't be CoinMetadata for it.
356            return Ok(None);
357        };
358
359        let metadata_type = NativeCoinMetadata::type_(*coin_struct);
360        let Some(object) = Object::query_singleton(db, metadata_type, checkpoint_viewed_at).await?
361        else {
362            return Ok(None);
363        };
364
365        let move_object = MoveObject::try_from(&object).map_err(|_| {
366            Error::Internal(format!(
367                "Expected {} to be CoinMetadata, but it is not an object.",
368                object.address,
369            ))
370        })?;
371
372        let coin_metadata = CoinMetadata::try_from(&move_object).map_err(|_| {
373            Error::Internal(format!(
374                "Expected {} to be CoinMetadata, but it is not.",
375                object.address,
376            ))
377        })?;
378
379        Ok(Some(coin_metadata))
380    }
381
382    pub(crate) async fn query_total_supply(
383        db: &Db,
384        coin_type: TypeTag,
385        checkpoint_viewed_at: u64,
386    ) -> Result<Option<u64>, Error> {
387        let TypeTag::Struct(coin_struct) = coin_type else {
388            // If the type supplied is not metadata, we know it's not a valid coin type, so there
389            // won't be CoinMetadata for it.
390            return Ok(None);
391        };
392
393        Ok(Some(if GAS::is_gas(coin_struct.as_ref()) {
394            TOTAL_SUPPLY_SUI
395        } else {
396            let cap_type = TreasuryCap::type_(*coin_struct);
397            let Some(object) = Object::query_singleton(db, cap_type, checkpoint_viewed_at).await?
398            else {
399                return Ok(None);
400            };
401
402            let Some(native) = object.native_impl() else {
403                return Ok(None);
404            };
405
406            let treasury_cap = TreasuryCap::try_from(native.clone()).map_err(|e| {
407                Error::Internal(format!(
408                    "Error while deserializing treasury cap {}: {e}",
409                    object.address,
410                ))
411            })?;
412
413            treasury_cap.total_supply.value
414        }))
415    }
416}

418impl TryFrom<&MoveObject> for CoinMetadata {
419    type Error = CoinMetadataDowncastError;
420
421    fn try_from(move_object: &MoveObject) -> Result<Self, Self::Error> {
422        if !move_object.native.type_().is_coin_metadata() {
423            return Err(CoinMetadataDowncastError::NotCoinMetadata);
424        }
425
426        Ok(Self {
427            super_: move_object.clone(),
428            native: bcs::from_bytes(move_object.native.contents())
429                .map_err(CoinMetadataDowncastError::Bcs)?,
430        })
431    }
