4use crate::types::{
5    digest::Digest,
6    sui_address::SuiAddress,
7    type_filter::{ModuleFilter, TypeFilter},
8};
9use async_graphql::*;
10
11#[derive(InputObject, Clone, Default)]
12pub(crate) struct EventFilter {
13    /// Filter down to events from transactions sent by this address.
14    pub sender: Option<SuiAddress>,
15
16    /// Filter down to the events from this transaction (given by its transaction digest).
17    pub transaction_digest: Option<Digest>,
18
19    // Enhancement (post-MVP)
20    // after_checkpoint
21    // at_checkpoint
22    // before_checkpoint
23    /// Events emitted by a particular module. An event is emitted by a
24    /// particular module if some function in the module is called by a
25    /// PTB and emits an event.
26    ///
27    /// Modules can be filtered by their package, or package::module.
28    /// We currently do not support filtering by emitting module and event type
29    /// at the same time so if both are provided in one filter, the query will error.
30    pub emitting_module: Option<ModuleFilter>,
31
32    /// This field is used to specify the type of event emitted.
33    ///
34    /// Events can be filtered by their type's package, package::module,
35    /// or their fully qualified type name.
36    ///
37    /// Generic types can be queried by either the generic type name, e.g.
38    /// `0x2::coin::Coin`, or by the full type name, such as
39    /// `0x2::coin::Coin<0x2::sui::SUI>`.
40    pub event_type: Option<TypeFilter>,
41    // Enhancement (post-MVP)
42    // pub start_time
43    // pub end_time
44}
