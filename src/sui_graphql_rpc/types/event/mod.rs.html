34/// A Sui node emits one of the following events:
35/// Move event
36/// Publish event
37/// Transfer object event
38/// Delete object event
39/// New object event
40/// Epoch change event
41#[derive(Clone, Debug)]
42pub(crate) struct Event {
43    pub stored: Option<StoredEvent>,
44    pub native: NativeEvent,
45    /// The checkpoint sequence number this was viewed at.
46    pub checkpoint_viewed_at: u64,
47}
48
49type Query<ST, GB> = data::Query<ST, events::table, GB>;
50
51#[Object]
52impl Event {
53    /// The transaction block that emitted this event. This information is only available for
54    /// events from indexed transactions, and not from transactions that have just been executed or
55    /// dry-run.
56    async fn transaction_block(&self, ctx: &Context<'_>) -> Result<Option<TransactionBlock>> {
57        let Some(stored) = &self.stored else {
58            return Ok(None);
59        };
60
61        TransactionBlock::query(
62            ctx,
63            TransactionBlock::by_seq(stored.tx_sequence_number as u64, self.checkpoint_viewed_at),
64        )
65        .await
66        .extend()
67    }
68
69    /// The Move module containing some function that when called by
70    /// a programmable transaction block (PTB) emitted this event.
71    /// For example, if a PTB invokes A::m1::foo, which internally
72    /// calls A::m2::emit_event to emit an event,
73    /// the sending module would be A::m1.
74    async fn sending_module(&self, ctx: &Context<'_>) -> Result<Option<MoveModule>> {
75        MoveModule::query(
76            ctx,
77            self.native.package_id.into(),
78            &self.native.transaction_module.to_string(),
79            self.checkpoint_viewed_at,
80        )
81        .await
82        .extend()
83    }
84
85    /// Address of the sender of the event
86    async fn sender(&self) -> Result<Option<Address>> {
87        if self.native.sender == NativeSuiAddress::ZERO {
88            return Ok(None);
89        }
90
91        Ok(Some(Address {
92            address: self.native.sender.into(),
93            checkpoint_viewed_at: self.checkpoint_viewed_at,
94        }))
95    }
96
97    /// UTC timestamp in milliseconds since epoch (1/1/1970)
98    async fn timestamp(&self) -> Result<Option<DateTime>, Error> {
99        if let Some(stored) = &self.stored {
100            Ok(Some(DateTime::from_ms(stored.timestamp_ms)?))
101        } else {
102            Ok(None)
103        }
104    }
105
106    /// The event's contents as a Move value.
107    async fn contents(&self) -> Result<MoveValue> {
108        Ok(MoveValue::new(
109            self.native.type_.clone().into(),
110            Base64::from(self.native.contents.clone()),
111        ))
112    }
113
114    /// The Base64 encoded BCS serialized bytes of the event.
115    async fn bcs(&self) -> Result<Base64> {
116        Ok(Base64::from(
117            bcs::to_bytes(&self.native).map_err(|e| Error::Internal(e.to_string()))?,
118        ))
119    }
120}

122impl Event {
123    /// Query the database for a `page` of events. The Page uses the transaction, event, and
124    /// checkpoint sequence numbers as the cursor to determine the correct page of results. The
125    /// query can optionally be further `filter`-ed by the `EventFilter`.
126    ///
127    /// The `checkpoint_viewed_at` parameter represents the checkpoint sequence number at which
128    /// this page was queried. Each entity returned in the connection inherits this checkpoint, so
129    /// that when viewing that entity's state, it's as if it's being viewed at this checkpoint.
130    ///
131    /// The cursors in `page` might also include checkpoint viewed at fields. If these are set,
132    /// they take precedence over the checkpoint that pagination is being conducted in.
133    pub(crate) async fn paginate(
134        db: &Db,
135        page: Page<Cursor>,
136        filter: EventFilter,
137        tx_lo: u64,
138        checkpoint_viewed_at: u64,
139    ) -> Result<Connection<String, Event>, Error> {
140        let cursor_viewed_at = page.validate_cursor_consistency()?;
141        let checkpoint_viewed_at = cursor_viewed_at.unwrap_or(checkpoint_viewed_at);
142        let tx_lo = tx_lo as i64;
143
144        // Construct tx and ev sequence number query with table-relevant filters, if they exist. The
145        // resulting query will look something like `SELECT tx_sequence_number,
146        // event_sequence_number FROM lookup_table WHERE ...`. If no filter is provided we don't
147        // need to use any lookup tables and can just query `events` table, as can be seen in the
148        // code below.
149        let query_constraint = match (filter.sender, &filter.emitting_module, &filter.event_type) {
150            (None, None, None) => None,
151            (Some(sender), None, None) => Some(select_sender(sender)),
152            (sender, None, Some(event_type)) => Some(select_event_type(event_type, sender)),
153            (sender, Some(module), None) => Some(select_emit_module(module, sender)),
154            (_, Some(_), Some(_)) => {
155                return Err(Error::Client(
156                    "Filtering by both emitting module and event type is not supported".to_string(),
157                ))
158            }
159        };
160
161        use checkpoints::dsl;
162        let (prev, next, results) = db
163            .execute(move |conn| async move {
164                // Get the tx_hi per checkpoint_viewed_at, which is either from watermarks or from
165                // the cursor if the latter is provided.
166                let tx_hi: i64 = conn.first(move || {
167                    dsl::checkpoints.select(dsl::network_total_transactions)
168                        .filter(dsl::sequence_number.eq(checkpoint_viewed_at as i64))
169                }).await?;
170
171                let (prev, next, mut events): (bool, bool, Vec<StoredEvent>) =
172                    if let Some(filter_query) =  query_constraint {
173                        let query = add_bounds(filter_query, &filter.transaction_digest, &page, tx_lo, tx_hi);
174
175                        let (prev, next, results) =
176                            page.paginate_raw_query::<EvLookup>(conn, checkpoint_viewed_at, query).await?;
177
178                        let ev_lookups = results
179                            .into_iter()
180                            .map(|x| (x.tx, x.ev))
181                            .collect::<Vec<(i64, i64)>>();
182
183                        if ev_lookups.is_empty() {
184                            return Ok::<_, diesel::result::Error>((prev, next, vec![]));
185                        }
186
187                        // Unlike a multi-get on a single column which can be serviced by a query `IN
188                        // (...)`, because events have a composite primary key, the query planner tends
189                        // to perform a sequential scan when given a list of tuples to lookup. A query
190                        // using `UNION ALL` allows us to leverage the index on the composite key.
191                        let events = conn.results(move || {
192                            // Diesel's DSL does not current support chained `UNION ALL`, so we have to turn
193                            // to `RawQuery` here.
194                            let query_string = ev_lookups.iter()
195                                .map(|&(tx, ev)| {
196                                    format!("SELECT * FROM events WHERE tx_sequence_number = {} AND event_sequence_number = {}", tx, ev)
197                                })
198                                .collect::<Vec<String>>()
199                                .join(" UNION ALL ");
200
201                            query!(query_string).into_boxed()
202                        }).await?;
203                        (prev, next, events)
204                    } else {
205                        // No filter is provided so we add bounds to the basic `SELECT * FROM
206                        // events` query and call it a day.
207                        let query = add_bounds(query!("SELECT * FROM events"), &filter.transaction_digest, &page, tx_lo, tx_hi);
208                        let (prev, next, events_iter) = page.paginate_raw_query::<StoredEvent>(conn, checkpoint_viewed_at, query).await?;
209                        let events = events_iter.collect::<Vec<StoredEvent>>();
210                        (prev, next, events)
211                    };
212
213                // UNION ALL does not guarantee order, so we need to sort the results. Whether
214                // `first` or `last, the result set is always sorted in ascending order.
215                events.sort_by(|a, b| {
216                        a.tx_sequence_number.cmp(&b.tx_sequence_number)
217                            .then_with(|| a.event_sequence_number.cmp(&b.event_sequence_number))
218                });
219
220
221                Ok::<_, diesel::result::Error>((prev, next, events))
222            }.scope_boxed())
223            .await?;
224
225        let mut conn = Connection::new(prev, next);
226
227        // The "checkpoint viewed at" sets a consistent upper bound for the nested queries.
228        for stored in results {
229            let cursor = stored.cursor(checkpoint_viewed_at).encode_cursor();
230            conn.edges.push(Edge::new(
231                cursor,
232                Event::try_from_stored_event(stored, checkpoint_viewed_at)?,
233            ));
234        }
235
236        Ok(conn)
237    }

239    pub(crate) fn try_from_stored_transaction(
240        stored_tx: &StoredTransaction,
241        idx: usize,
242        checkpoint_viewed_at: u64,
243    ) -> Result<Self, Error> {
244        let Some(serialized_event) = &stored_tx.get_event_at_idx(idx) else {
245            return Err(Error::Internal(format!(
246                "Could not find event with event_sequence_number {} at transaction {}",
247                idx, stored_tx.tx_sequence_number
248            )));
249        };
250
251        let native_event: NativeEvent = bcs::from_bytes(serialized_event).map_err(|_| {
252            Error::Internal(format!(
253                "Failed to deserialize event with {} at transaction {}",
254                idx, stored_tx.tx_sequence_number
255            ))
256        })?;
257
258        let stored_event = StoredEvent {
259            tx_sequence_number: stored_tx.tx_sequence_number,
260            event_sequence_number: idx as i64,
261            transaction_digest: stored_tx.transaction_digest.clone(),
262            senders: vec![Some(native_event.sender.to_vec())],
263            package: native_event.package_id.to_vec(),
264            module: native_event.transaction_module.to_string(),
265            event_type: native_event
266                .type_
267                .to_canonical_string(/* with_prefix */ true),
268            bcs: native_event.contents.clone(),
269            timestamp_ms: stored_tx.timestamp_ms,
270            sender: Some(native_event.sender.to_vec()),
271        };
272
273        Ok(Self {
274            stored: Some(stored_event),
275            native: native_event,
276            checkpoint_viewed_at,
277        })
278    }

280    fn try_from_stored_event(
281        stored: StoredEvent,
282        checkpoint_viewed_at: u64,
283    ) -> Result<Self, Error> {
284        let Some(Some(sender_bytes)) = ({ stored.senders.first() }) else {
285            return Err(Error::Internal("No senders found for event".to_string()));
286        };
287        let sender = NativeSuiAddress::from_bytes(sender_bytes)
288            .map_err(|e| Error::Internal(e.to_string()))?;
289        let package_id =
290            ObjectID::from_bytes(&stored.package).map_err(|e| Error::Internal(e.to_string()))?;
291        let type_ =
292            parse_sui_struct_tag(&stored.event_type).map_err(|e| Error::Internal(e.to_string()))?;
293        let transaction_module =
294            Identifier::from_str(&stored.module).map_err(|e| Error::Internal(e.to_string()))?;
295        let contents = stored.bcs.clone();
296        Ok(Event {
297            stored: Some(stored),
298            native: NativeEvent {
299                sender,
300                package_id,
301                transaction_module,
302                type_,
303                contents,
304            },
305            checkpoint_viewed_at,
306        })
307    }
