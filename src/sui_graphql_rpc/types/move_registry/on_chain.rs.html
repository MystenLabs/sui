27/// Regex to parse a dot move name. Version is optional (defaults to latest).
28/// For versioned format, the expected format is `@org/app/1` (1 == version).
29/// For an unversioned format, the expected format is `@org/app`.
30///
31/// The unbound regex can be used to search matches in a type tag.
32/// Use `VERSIONED_NAME_REGEX` for parsing a single name from a str.
33const VERSIONED_NAME_UNBOUND_REGEX: &str =
34    concat!(r"([a-z0-9.\-@]*)", r"\/", "([a-z0-9.-]*)", r"(?:\/(\d+))?",);
35
36/// Regex to parse a dot move name. Version is optional (defaults to latest).
37/// For versioned format, the expected format is `@org/app/1`.
38/// For an unversioned format, the expected format is `@org/app`.
39///
40/// This regex is used to parse a single name (does not do type_tag matching).
41/// Use `VERSIONED_NAME_UNBOUND_REGEX` for type tag matching.
42const VERSIONED_NAME_REGEX: &str = concat!(
43    "^",
44    r"([a-z0-9.\-@]*)",
45    r"\/",
46    "([a-z0-9.-]*)",
47    r"(?:\/(\d+))?",
48    "$"
49);
50
51/// A regular expression that detects all possible dot move names in a type tag.
52pub(crate) static VERSIONED_NAME_UNBOUND_REG: Lazy<Regex> =
53    Lazy::new(|| Regex::new(VERSIONED_NAME_UNBOUND_REGEX).unwrap());
54
55/// A regular expression that detects a single name in the format `@org/app/1`.
56pub(crate) static VERSIONED_NAME_REG: Lazy<Regex> =
57    Lazy::new(|| Regex::new(VERSIONED_NAME_REGEX).unwrap());

81#[derive(Debug, Serialize, Deserialize, Clone, Eq, PartialEq)]
82pub(crate) struct VersionedName {
83    /// A version name defaults at None, which means we need the latest version.
84    pub(crate) version: Option<u64>,
85    /// The on-chain `Name` object that represents the move registry name.
86    pub(crate) name: Name,
87}

89/// Attention: The format of this struct should not change unless the on-chain format changes,
90/// as we define it to deserialize on-chain data.
91#[derive(Debug, Serialize, Deserialize, Hash, Clone, Eq, PartialEq)]
92pub(crate) struct Name {
93    pub(crate) org: Domain,
94    pub(crate) app: Vec<String>,
95}

97impl Name {
98    pub(crate) fn new(org: Domain, app: &str) -> Self {
99        Self {
100            org,
101            app: vec![app.to_string()],
102        }
103    }
104
105    pub(crate) fn type_(package_address: SuiAddress) -> StructTag {
106        StructTag {
107            address: package_address.into(),
108            module: MOVE_REGISTRY_MODULE.to_owned(),
109            name: MOVE_REGISTRY_TYPE.to_owned(),
110            type_params: vec![],
111        }
112    }
113
114    pub(crate) fn to_bytes(&self) -> Vec<u8> {
115        bcs::to_bytes(&self).unwrap()
116    }
117
118    pub(crate) fn to_base64_string(&self) -> String {
119        Base64::from(self.to_bytes()).to_value().to_string()
120    }
121
122    /// Generate the ObjectID for a given `Name`
123    pub(crate) fn to_dynamic_field_id(
124        &self,
125        config: &MoveRegistryConfig,
126    ) -> Result<ObjectID, bcs::Error> {
127        let domain_type_tag = Self::type_(config.package_address);
128
129        sui_types::dynamic_field::derive_dynamic_field_id(
130            config.registry_id,
131            &sui_types::TypeTag::Struct(Box::new(domain_type_tag)),
132            &self.to_bytes(),
133        )
134    }
135}

137impl FromStr for VersionedName {
138    type Err = MoveRegistryError;
139
140    fn from_str(s: &str) -> Result<Self, Self::Err> {
141        let Some(caps) = VERSIONED_NAME_REG.captures(s) else {
142            return Err(MoveRegistryError::InvalidName(s.to_string()));
143        };
144
145        let Some(org_name) = caps.get(1).map(|x| x.as_str()) else {
146            return Err(MoveRegistryError::InvalidName(s.to_string()));
147        };
148
149        // validate org_name by trying to cast our input to a Domain.
150        let domain = Domain::from_str(org_name)
151            .map_err(|_| MoveRegistryError::InvalidName(s.to_string()))?;
152
153        let Some(app_name) = caps.get(2).map(|x| x.as_str()) else {
154            return Err(MoveRegistryError::InvalidName(s.to_string()));
155        };
156
157        // Validate our app's label.
158        validate_label(app_name).map_err(|_| MoveRegistryError::InvalidName(s.to_string()))?;
159
160        let version: Option<u64> = caps
161            .get(3)
162            .map(|x| x.as_str().parse())
163            .transpose()
164            .map_err(|_| MoveRegistryError::InvalidVersion)?;
165
166        Ok(Self {
167            version,
168            name: Name::new(domain, app_name),
169        })
170    }
171}

173impl TryFrom<NativeMoveObject> for AppRecord {
174    type Error = MoveRegistryError;
175
176    fn try_from(object: NativeMoveObject) -> Result<Self, MoveRegistryError> {
177        object
178            .to_rust::<Field<Name, Self>>()
179            .map(|record| record.value)
180            .ok_or_else(|| MoveRegistryError::FailedToDeserializeRecord(object.id()))
181    }
182}

184#[cfg(test)]
185mod tests {
186    use super::VersionedName;
187    use std::str::FromStr;
188
189    #[test]
190    fn parse_some_names() {
191        let versioned = VersionedName::from_str("@org/app/1").unwrap();
192        assert!(versioned.version.is_some_and(|x| x == 1));
193
194        assert!(VersionedName::from_str("@org/app/34")
195            .unwrap()
196            .version
197            .is_some_and(|x| x == 34));
198        assert!(VersionedName::from_str("@org/app")
199            .unwrap()
200            .version
201            .is_none());
202
203        let ok_names = vec![
204            "@org/1-app/1",
205            "@org/1-app/34",
206            "@org/1-app",
207            "nested@org/app",
208            "even.more.nested@org/app",
209        ];
210
211        let composite_ok_names = vec![
212            format!("@org/{}/1", generate_fixed_string(63)),
213            format!("@org/{}-app/34", generate_fixed_string(59)),
214            format!(
215                "@{}/{}",
216                generate_fixed_string(63),
217                generate_fixed_string(63)
218            ),
219            format!(
220                "@{}/{}-{}",
221                generate_fixed_string(63),
222                generate_fixed_string(30),
223                generate_fixed_string(30)
224            ),
225        ];
226
227        for name in ok_names {
228            assert!(VersionedName::from_str(name).is_ok());
229        }
230        for name in composite_ok_names {
231            assert!(VersionedName::from_str(&name).is_ok());
232        }
233
234        let not_ok_names = vec![
235            "@org/-app",
236            "@org/1.app",
237            "@org/app-",
238            "@org/app--",
239            "@org/app?",
240            "@org/app/",
241            "@org/app/v",
242            "@org/app/veh",
243            "@org",
244            "@/veh/app",
245            "app",
246            "@",
247            "@org/ap@",
248            "!org/ap",
249            "#org/ap",
250            "@org/ap#org",
251            "org/@app",
252            "",
253            " ",
254        ];
255        let composite_err_names = vec![
256            format!(
257                "@{}{}--/{}",
258                generate_fixed_string(10),
259                generate_fixed_string(10),
260                generate_fixed_string(63)
261            ),
262            format!(
263                "@--{}-{}/{}",
264                generate_fixed_string(10),
265                generate_fixed_string(10),
266                generate_fixed_string(63)
267            ),
268            format!(
269                "@{}/--{}{}",
270                generate_fixed_string(63),
271                generate_fixed_string(30),
272                generate_fixed_string(30)
273            ),
274        ];
275
276        for name in not_ok_names {
277            assert!(VersionedName::from_str(name).is_err());
278        }
279        for name in composite_err_names {
280            assert!(VersionedName::from_str(&name).is_err());
281        }
282    }
283
284    fn generate_fixed_string(len: usize) -> String {
285        // Define the characters to use in the string
286        let chars = "abcdefghijklmnopqrstuvwxyz0123456789";
287        // Repeat the characters to ensure we have at least 63 characters
288        let repeated_chars = chars.repeat(3);
289
290        // Take the first 63 characters to form the string
291        let fixed_string = &repeated_chars[0..len];
292
293        fixed_string.to_string()
294    }
