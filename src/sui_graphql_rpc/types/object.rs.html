52#[derive(Clone, Debug)]
53pub(crate) struct Object {
54    pub address: SuiAddress,
55    pub version: u64,
56    pub kind: ObjectKind,
57    /// The checkpoint sequence number at which this was viewed at.
58    pub checkpoint_viewed_at: u64,
59    /// Root parent object version for dynamic fields.
60    ///
61    /// This enables consistent dynamic field reads in the case of chained dynamic object fields,
62    /// e.g., `Parent -> DOF1 -> DOF2`. In such cases, the object versions may end up like
63    /// `Parent >= DOF1, DOF2` but `DOF1 < DOF2`. Thus, database queries for dynamic fields must
64    /// bound the object versions by the version of the root object of the tree.
65    ///
66    /// Essentially, lamport timestamps of objects are updated for all top-level mutable objects
67    /// provided as inputs to a transaction as well as any mutated dynamic child objects. However,
68    /// any dynamic child objects that were loaded but not actually mutated don't end up having
69    /// their versions updated.
70    root_version: u64,
71}

588    pub(crate) async fn owner(&self) -> Option<ObjectOwner> {
589        use NativeOwner as O;
590
591        let native = self.0.native_impl()?;
592
593        match &native.owner {
594            O::AddressOwner(address) => {
595                let address = SuiAddress::from(*address);
596                Some(ObjectOwner::Address(AddressOwner {
597                    owner: Some(Owner {
598                        address,
599                        checkpoint_viewed_at: self.0.checkpoint_viewed_at,
600                        root_version: None,
601                    }),
602                }))
603            }
604            O::Immutable => Some(ObjectOwner::Immutable(Immutable { dummy: None })),
605            O::ObjectOwner(address) => {
606                let address = SuiAddress::from(*address);
607                Some(ObjectOwner::Parent(Parent {
608                    parent: Some(Owner {
609                        address,
610                        checkpoint_viewed_at: self.0.checkpoint_viewed_at,
611                        root_version: Some(self.0.root_version()),
612                    }),
613                }))
614            }
615            O::Shared {
616                initial_shared_version,
617            } => Some(ObjectOwner::Shared(Shared {
618                initial_shared_version: initial_shared_version.value().into(),
619            })),
620            O::ConsensusAddressOwner {
621                start_version,
622                owner,
623            } => Some(ObjectOwner::ConsensusAddress(ConsensusAddressOwner {
624                start_version: start_version.value().into(),
625                owner: Some(Owner {
626                    address: SuiAddress::from(*owner),
627                    checkpoint_viewed_at: self.0.checkpoint_viewed_at,
628                    root_version: None,
629                }),
630            })),
631        }
632    }

703    /// `display` is part of the `IMoveObject` interface, but is implemented on `ObjectImpl` to
704    /// allow for a convenience function on `Object`.
705    pub(crate) async fn display(&self, ctx: &Context<'_>) -> Result<Option<Vec<DisplayEntry>>> {
706        let resolver: &PackageResolver = ctx.data_unchecked();
707
708        let Some(native) = self.0.native_impl() else {
709            return Ok(None);
710        };
711
712        let move_object = native
713            .data
714            .try_as_move()
715            .ok_or_else(|| Error::Internal("Failed to convert object into MoveObject".to_string()))
716            .extend()?;
717
718        let type_: TypeTag = move_object.type_().clone().into();
719        let (type_layout, display) = join!(
720            resolver.type_layout(type_.clone()),
721            Display::query(ctx.data_unchecked(), type_.clone()),
722        );
723
724        let type_layout = type_layout.map_err(|e| {
725            Error::Internal(format!(
726                "Error fetching layout for type {}: {e}",
727                move_object
728                    .type_()
729                    .to_canonical_string(/* with_prefix */ true)
730            ))
731        })?;
732
733        let Some(display) = display.extend()? else {
734            return Ok(None);
735        };
736
737        Ok(Some(
738            display
739                .render(move_object.contents(), &type_layout)
740                .extend()?,
741        ))

852    /// Query the database for a `page` of some sub-type of Object. The page uses the bytes of an
853    /// Object ID and the checkpoint when the query was made as the cursor, and can optionally be
854    /// further `filter`-ed. The subtype is created using the `downcast` function, which is allowed
855    /// to fail, if the downcast has failed.
856    ///
857    /// `checkpoint_viewed_at` represents the checkpoint sequence number at which this page was
858    /// queried for. Each entity returned in the connection will inherit this checkpoint, so that
859    /// when viewing that entity's state, it will be as if it was read at the same checkpoint.
860    ///
861    /// If a `Page<Cursor>` is also provided, then this function will defer to the
862    /// `checkpoint_viewed_at` in the cursors. Otherwise, use the value from the parameter, or set
863    /// to None. This is so that paginated queries are consistent with the previous query that
864    /// created the cursor.
865    pub(crate) async fn paginate_subtype<T: OutputType>(
866        db: &Db,
867        page: Page<Cursor>,
868        filter: ObjectFilter,
869        checkpoint_viewed_at: u64,
870        downcast: impl Fn(Object) -> Result<T, Error>,
871    ) -> Result<Connection<String, T>, Error> {
872        // If cursors are provided, defer to the `checkpoint_viewed_at` in the cursor if they are
873        // consistent. Otherwise, use the value from the parameter, or set to None. This is so that
874        // paginated queries are consistent with the previous query that created the cursor.
875        let cursor_viewed_at = page.validate_cursor_consistency()?;
876        let checkpoint_viewed_at = cursor_viewed_at.unwrap_or(checkpoint_viewed_at);
877
878        let Some((prev, next, results)) = db
879            .execute_repeatable(move |conn| {
880                async move {
881                    let Some(range) = AvailableRange::result(conn, checkpoint_viewed_at).await?
882                    else {
883                        return Ok::<_, diesel::result::Error>(None);
884                    };
885
886                    Ok(Some(
887                        page.paginate_raw_query::<StoredHistoryObject>(
888                            conn,
889                            checkpoint_viewed_at,
890                            objects_query(&filter, range, &page),
891                        )
892                        .await?,
893                    ))
894                }
895                .scope_boxed()
896            })
897            .await?
898        else {
899            return Err(Error::Client(
900                "Requested data is outside the available range".to_string(),
901            ));
902        };
903
904        let mut conn: Connection<String, T> = Connection::new(prev, next);
905        for stored in results {
906            // To maintain consistency, the returned cursor should have the same upper-bound as the
907            // checkpoint found on the cursor.
908            let cursor = stored.cursor(checkpoint_viewed_at).encode_cursor();
909            let object =
910                Object::try_from_stored_history_object(stored, checkpoint_viewed_at, None)?;
911            conn.edges.push(Edge::new(cursor, downcast(object)?));
912        }
913
914        Ok(conn)
915    }

1006    /// `checkpoint_viewed_at` represents the checkpoint sequence number at which this `Object` was
1007    /// constructed in. This is stored on `Object` so that when viewing that entity's state, it will
1008    /// be as if it was read at the same checkpoint.
1009    ///
1010    /// `root_version` represents the version of the root object in some nested chain of dynamic
1011    /// fields. This should typically be left `None`, unless the object(s) being resolved is a
1012    /// dynamic field, or if `root_version` has been explicitly set for this object. If None, then
1013    /// we use [`version_for_dynamic_fields`] to infer a root version to then propagate from this
1014    /// object down to its dynamic fields.
1015    pub(crate) fn try_from_stored_history_object(
1016        history_object: StoredHistoryObject,
1017        checkpoint_viewed_at: u64,
1018        root_version: Option<u64>,
1019    ) -> Result<Self, Error> {
1020        let address = addr(&history_object.object_id)?;
1021
1022        let object_status =
1023            NativeObjectStatus::try_from(history_object.object_status).map_err(|_| {
1024                Error::Internal(format!(
1025                    "Unknown object status {} for object {} at version {}",
1026                    history_object.object_status, address, history_object.object_version
1027                ))
1028            })?;
1029
1030        match object_status {
1031            NativeObjectStatus::Active => {
1032                let Some(serialized_object) = &history_object.serialized_object else {
1033                    return Err(Error::Internal(format!(
1034                        "Live object {} at version {} cannot have missing serialized_object field",
1035                        address, history_object.object_version
1036                    )));
1037                };
1038
1039                let native_object = bcs::from_bytes(serialized_object).map_err(|_| {
1040                    Error::Internal(format!("Failed to deserialize object {address}"))
1041                })?;
1042
1043                let root_version =
1044                    root_version.unwrap_or_else(|| version_for_dynamic_fields(&native_object));
1045                Ok(Self {
1046                    address,
1047                    version: history_object.object_version as u64,
1048                    kind: ObjectKind::Indexed(native_object, history_object),
1049                    checkpoint_viewed_at,
1050                    root_version,
1051                })
1052            }
1053            NativeObjectStatus::WrappedOrDeleted => Err(Error::Internal(
1054                "Wrapped or deleted objects should not be loaded from DB.".to_string(),
1055            )),
1056        }
1057    }

1061/// caller override it with the actual root object's version if it has access to it.

1071impl ObjectFilter {
1072    /// Try to create a filter whose results are the intersection of objects in `self`'s results and
1073    /// objects in `other`'s results. This may not be possible if the resulting filter is
1074    /// inconsistent in some way (e.g. a filter that requires one field to be two different values
1075    /// simultaneously).
1076    pub(crate) fn intersect(self, other: ObjectFilter) -> Option<Self> {
1077        macro_rules! intersect {
1078            ($field:ident, $body:expr) => {
1079                intersect::field(self.$field, other.$field, $body)
1080            };
1081        }
1082
1083        let object_ids = intersect::field(self.object_ids, other.object_ids, |a, b| {
1084            let a = BTreeSet::from_iter(a);
1085            let b = BTreeSet::from_iter(b);
1086
1087            let intersection: Vec<_> = a.intersection(&b).cloned().collect();
1088            (!intersection.is_empty()).then_some(intersection)
1089        })?;
1090
1091        Some(Self {
1092            type_: intersect!(type_, TypeFilter::intersect)?,
1093            owner: intersect!(owner, intersect::by_eq)?,
1094            object_ids,
1095        })
1096    }
1097
1098    /// Applies ObjectFilter to the input `RawQuery` and returns a new `RawQuery`.
1099    pub(crate) fn apply(&self, mut query: RawQuery) -> RawQuery {
1100        // Start by applying the filters on IDs and/or keys because they are combined as
1101        // a disjunction, while the remaining queries are conjunctions.
1102        if let Some(object_ids) = &self.object_ids {
1103            // Maximally strict - match a vec of 0 elements
1104            if object_ids.is_empty() {
1105                query = or_filter!(query, "1=0");
1106            } else {
1107                let mut inner = String::new();
1108                let mut prefix = "object_id IN (";
1109                for id in object_ids {
1110                    // SAFETY: Writing to a `String` cannot fail.
1111                    write!(
1112                        &mut inner,
1113                        "{prefix}'\\x{}'::bytea",
1114                        hex::encode(id.into_vec())
1115                    )
1116                    .unwrap();
1117                    prefix = ", ";
1118                }
1119                inner.push(')');
1120                query = or_filter!(query, inner);
1121            }
1122        }
1123
1124        if let Some(owner) = self.owner {
1125            query = filter!(
1126                query,
1127                format!(
1128                    "owner_id = '\\x{}'::bytea AND owner_type = {}",
1129                    hex::encode(owner.into_vec()),
1130                    OwnerType::Address as i16
1131                )
1132            );
1133        }
1134
1135        if let Some(type_) = &self.type_ {
1136            return type_.apply_raw(
1137                query,
1138                "object_type",
1139                "object_type_package",
1140                "object_type_module",
1141                "object_type_name",
1142            );
1143        }
1144
1145        query
1146    }
1147
1148    pub(crate) fn has_filters(&self) -> bool {
1149        self != &Default::default()
1150    }
1151}

1209#[async_trait::async_trait]
1210impl Loader<HistoricalKey> for Db {
1211    type Value = Object;
1212    type Error = Error;
1213
1214    async fn load(&self, keys: &[HistoricalKey]) -> Result<HashMap<HistoricalKey, Object>, Error> {
1215        use objects_version::dsl as v;
1216
1217        if keys.is_empty() {
1218            return Ok(HashMap::new());
1219        }
1220
1221        let id_versions: BTreeSet<_> = keys
1222            .iter()
1223            .map(|key| (key.id.into_vec(), key.version as i64))
1224            .collect();
1225
1226        // Maps from (object_id, version) to sequence_number in the object_versions table.
1227        let object_versions: HashMap<_, _> = self
1228            .execute(move |conn| {
1229                async {
1230                    conn.results(move || {
1231                        let mut query = v::objects_version
1232                            .select(StoredObjectVersion::as_select())
1233                            .into_boxed();
1234
1235                        for (id, version) in id_versions.iter().cloned() {
1236                            // TODO: consider using something other than `or_filter` to avoid returning
1237                            // all results when `id_versions` is empty. It is mitigated today by the
1238                            // early return above.
1239                            query = query
1240                                .or_filter(v::object_id.eq(id).and(v::object_version.eq(version)));
1241                        }
1242
1243                        query
1244                    })
1245                    .await
1246                }
1247                .scope_boxed()
1248            })
1249            .await?
1250            .into_iter()
1251            .map(|v| ((v.object_id, v.object_version), v.cp_sequence_number))
1252            .collect();
1253        let filtered_keys: Vec<_> = keys
1254            .iter()
1255            .filter(|key| {
1256                object_versions
1257                    .get(&(key.id.into_vec(), key.version as i64))
1258                    // Filter by key's checkpoint viewed at here. Doing this in memory because it should be
1259                    // quite rare that this query actually filters something, but encoding it in SQL is
1260                    // complicated.
1261                    .is_some_and(|&seq| key.checkpoint_viewed_at >= seq as u64)
1262            })
1263            .collect();
1264        let point_lookup_keys: Vec<_> = filtered_keys
1265            .iter()
1266            .map(|key| PointLookupKey {
1267                id: key.id,
1268                version: key.version,
1269            })
1270            .collect();
1271        let objects = self.load(&point_lookup_keys).await?;
1272        let results = filtered_keys
1273            .into_iter()
1274            .zip(point_lookup_keys)
1275            .filter_map(|(hist_key, lookup_key)| {
1276                let object = objects.get(&lookup_key)?;
1277                let hist_obj = Object::new_serialized(
1278                    lookup_key.id,
1279                    lookup_key.version,
1280                    object.clone(),
1281                    hist_key.checkpoint_viewed_at,
1282                    lookup_key.version,
1283                );
1284                hist_obj.map(|obj| (*hist_key, obj))
1285            })
1286            .collect();
1287        Ok(results)
1288    }
1289}

1292impl Loader<ParentVersionKey> for Db {
1293    type Value = Object;
1294    type Error = Error;
1295
1296    async fn load(
1297        &self,
1298        keys: &[ParentVersionKey],
1299    ) -> Result<HashMap<ParentVersionKey, Object>, Error> {
1300        // Group keys by checkpoint viewed at and parent version -- we'll issue a separate query for
1301        // each group.
1302        #[derive(Eq, PartialEq, Ord, PartialOrd, Clone, Copy)]
1303        struct GroupKey {
1304            checkpoint_viewed_at: u64,
1305            parent_version: u64,
1306        }
1307
1308        let mut keys_by_cursor_and_parent_version: BTreeMap<_, BTreeSet<_>> = BTreeMap::new();
1309        for key in keys {
1310            let group_key = GroupKey {
1311                checkpoint_viewed_at: key.checkpoint_viewed_at,
1312                parent_version: key.parent_version,
1313            };
1314
1315            keys_by_cursor_and_parent_version
1316                .entry(group_key)
1317                .or_default()
1318                .insert(key.id.into_vec());
1319        }
1320
1321        // Issue concurrent reads for each group of keys.
1322        let futures = keys_by_cursor_and_parent_version
1323            .into_iter()
1324            .map(|(group_key, ids)| {
1325                self.execute(move |conn| {
1326                    async move {
1327                        let stored: Vec<StoredObjectVersion> = conn
1328                            .results(move || {
1329                                use objects_version::dsl as v;
1330
1331                                v::objects_version
1332                                    .select(StoredObjectVersion::as_select())
1333                                    .filter(v::object_id.eq_any(ids.iter().cloned()))
1334                                    .filter(v::object_version.le(group_key.parent_version as i64))
1335                                    .distinct_on(v::object_id)
1336                                    .order_by(v::object_id)
1337                                    .then_order_by(v::object_version.desc())
1338                                    .into_boxed()
1339                            })
1340                            .await?;
1341
1342                        Ok::<_, diesel::result::Error>(
1343                            stored
1344                                .into_iter()
1345                                .map(|stored| (group_key, stored))
1346                                .collect::<Vec<_>>(),
1347                        )
1348                    }
1349                    .scope_boxed()
1350                })
1351            });
1352
1353        let groups = futures::future::join_all(futures).await;
1354        let mut group_map = HashMap::new();
1355        for group in groups {
1356            for (group_key, stored) in
1357                group.map_err(|e| Error::Internal(format!("Failed to fetch objects: {e}")))?
1358            {
1359                // This particular object is invalid -- it didn't exist at the checkpoint we are
1360                // viewing at.
1361                if group_key.checkpoint_viewed_at < stored.cp_sequence_number as u64 {
1362                    continue;
1363                }
1364                let key = ParentVersionKey {
1365                    id: addr(&stored.object_id)?,
1366                    checkpoint_viewed_at: group_key.checkpoint_viewed_at,
1367                    parent_version: group_key.parent_version,
1368                };
1369                group_map.insert(key, stored.object_version);
1370            }
1371        }
1372        let point_lookup_keys = group_map
1373            .iter()
1374            .map(|(parent_key, version)| PointLookupKey {
1375                id: parent_key.id,
1376                version: *version as u64,
1377            })
1378            .collect::<Vec<_>>();
1379        let objects = self.load(&point_lookup_keys).await?;
1380        let results = group_map
1381            .into_keys()
1382            .zip(point_lookup_keys)
1383            .filter_map(|(parent_key, lookup_key)| {
1384                let object = objects.get(&lookup_key)?;
1385                let hist_obj = Object::new_serialized(
1386                    parent_key.id,
1387                    lookup_key.version,
1388                    object.clone(),
1389                    parent_key.checkpoint_viewed_at,
1390                    // If `ParentVersionKey::parent_version` is set, it must have been correctly
1391                    // propagated from the `Object::root_version` of some object.
1392                    parent_key.parent_version,
1393                );
1394                hist_obj.map(|obj| (parent_key, obj))
1395            })
1396            .collect();
1397
1398        Ok(results)
1399    }
1400}

1403impl Loader<LatestAtKey> for Db {
1404    type Value = Object;
1405    type Error = Error;
1406
1407    async fn load(&self, keys: &[LatestAtKey]) -> Result<HashMap<LatestAtKey, Object>, Error> {
1408        // Group keys by checkpoint viewed at -- we'll issue a separate query for each group.
1409        let mut keys_by_cursor_and_parent_version: BTreeMap<_, BTreeSet<_>> = BTreeMap::new();
1410
1411        for key in keys {
1412            keys_by_cursor_and_parent_version
1413                .entry(key.checkpoint_viewed_at)
1414                .or_default()
1415                .insert(key.id);
1416        }
1417
1418        // Issue concurrent reads for each group of keys.
1419        let futures =
1420            keys_by_cursor_and_parent_version
1421                .into_iter()
1422                .map(|(checkpoint_viewed_at, ids)| {
1423                    self.execute_repeatable(move |conn| {
1424                        async move {
1425                            let Some(range) =
1426                                AvailableRange::result(conn, checkpoint_viewed_at).await?
1427                            else {
1428                                return Ok::<Vec<(u64, StoredHistoryObject)>, diesel::result::Error>(
1429                                    vec![],
1430                                );
1431                            };
1432
1433                            let filter = ObjectFilter {
1434                                object_ids: Some(ids.iter().cloned().collect()),
1435                                ..Default::default()
1436                            };
1437
1438                            Ok(conn
1439                                .results(move || {
1440                                    build_objects_query(
1441                                        View::Consistent,
1442                                        range,
1443                                        &Page::bounded(ids.len() as u64),
1444                                        |q| filter.apply(q),
1445                                        |q| q,
1446                                    )
1447                                    .into_boxed()
1448                                })
1449                                .await?
1450                                .into_iter()
1451                                .map(|r| (checkpoint_viewed_at, r))
1452                                .collect())
1453                        }
1454                        .scope_boxed()
1455                    })
1456                });
1457
1458        // Wait for the reads to all finish, and gather them into the result map.
1459        let groups = futures::future::join_all(futures).await;
1460
1461        let mut results = HashMap::new();
1462        for group in groups {
1463            for (checkpoint_viewed_at, stored) in
1464                group.map_err(|e| Error::Internal(format!("Failed to fetch objects: {e}")))?
1465            {
1466                let object =
1467                    Object::try_from_stored_history_object(stored, checkpoint_viewed_at, None)?;
1468
1469                let key = LatestAtKey {
1470                    id: object.address,
1471                    checkpoint_viewed_at,
1472                };
1473
1474                results.insert(key, object);
1475            }
1476        }
1477
1478        Ok(results)
1479    }
1480}

1483impl Loader<PointLookupKey> for Db {
1484    type Value = Option<Vec<u8>>;
1485    type Error = Error;
1486
1487    async fn load(
1488        &self,
1489        keys: &[PointLookupKey],
1490    ) -> Result<HashMap<PointLookupKey, Option<Vec<u8>>>, Error> {
1491        use full_objects_history::dsl as f;
1492
1493        if keys.is_empty() {
1494            return Ok(HashMap::new());
1495        }
1496
1497        let id_versions: BTreeSet<_> = keys
1498            .iter()
1499            .map(|key| (key.id.into_vec(), key.version as i64))
1500            .collect();
1501        let objects = self
1502            .execute(move |conn| {
1503                async {
1504                    conn.results(move || {
1505                        let mut query = f::full_objects_history
1506                            .select(StoredFullHistoryObject::as_select())
1507                            .into_boxed();
1508
1509                        for (id, version) in id_versions.iter() {
1510                            // TODO: consider using something other than `or_filter` to avoid returning
1511                            // all results when `id_versions` is empty. It is mitigated today by the
1512                            // early return above.
1513                            query = query.or_filter(
1514                                f::object_id
1515                                    .eq(id.clone())
1516                                    .and(f::object_version.eq(*version)),
1517                            );
1518                        }
1519
1520                        query
1521                    })
1522                    .await
1523                }
1524                .scope_boxed()
1525            })
1526            .await?;
1527        let objects_map: HashMap<_, _> = objects
1528            .into_iter()
1529            .map(|o| {
1530                (
1531                    PointLookupKey {
1532                        id: addr(&o.object_id).unwrap(),
1533                        version: o.object_version as u64,
1534                    },
1535                    o.serialized_object,
1536                )
1537            })
1538            .collect();
1539
1540        let result = keys
1541            .iter()
1542            .filter_map(|key| {
1543                let serialized = objects_map.get(key)?;
1544                Some((*key, serialized.clone()))
1545            })
1546            .collect();
1547        Ok(result)
1548    }
1549}

1569/// Constructs a raw query to fetch objects from the database. Objects are filtered out if they
1570/// satisfy the criteria but have a later version in the same checkpoint. If no filters are
1571/// specified at all, then this final condition is not applied.
1572fn objects_query(filter: &ObjectFilter, range: AvailableRange, page: &Page<Cursor>) -> RawQuery
1573where
1574{
1575    build_objects_query(
1576        if !filter.has_filters() {
1577            View::Historical
1578        } else {
1579            View::Consistent
1580        },
1581        range,
1582        page,
1583        move |query| filter.apply(query),
1584        move |newer| newer,
1585    )
1586}

1613        let i3 = SuiAddress::from_str("0x3").unwrap();

1616        let f0 = ObjectFilter {
1617            object_ids: Some(vec![i1, i3]),
1618            ..Default::default()
1619        };
1620
1621        // Overlaps with f0 on id i1
1622        let f1 = ObjectFilter {
1623            object_ids: Some(vec![i1, i2]),

1625        };
1626
1627        // An empty filter
1628        let f2 = ObjectFilter {
1629            ..Default::default()
1630        };
1631
1632        // Overlaps with f0 on id i3, and does not overlap with f1
1633        let f3 = ObjectFilter {
1634            object_ids: Some(vec![i3]),
1635            ..Default::default()
1636        };
1637
1638        assert_eq!(
1639            f0.clone().intersect(f1.clone()),

1641                object_ids: Some(vec![i1]),

1643            })
1644        );
1645
1646        assert_eq!(
1647            f2.clone().intersect(f2.clone()),
1648            Some(ObjectFilter::default())
1649        );
1650
1651        assert_eq!(f1.clone().intersect(f2.clone()), Some(f1.clone()));
1652        assert_eq!(f1.clone().intersect(f3.clone()), None);
1653
1654        // Overlaps with f1 on i2, but does not overlap with f0 or f3. Note that this also has an
1655        // owner filter
1656        let f4 = ObjectFilter {
1657            owner: Some(i1),

1669        };
1670
1671        // Does not overlap with f5 because module filter is different.
1672        let f6 = ObjectFilter {
1673            owner: None,
1674            object_ids: Some(vec![i1]),

1676                crate::types::type_filter::ModuleFilter::ByPackage(i1),

1678        };
