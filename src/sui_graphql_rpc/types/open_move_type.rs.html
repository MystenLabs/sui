15/// Abilities are keywords in Sui Move that define how types behave at the compiler level.
16#[derive(Enum, Copy, Clone, Eq, PartialEq)]
17pub(crate) enum MoveAbility {
18    /// Enables values to be copied.
19    Copy,
20    /// Enables values to be popped/dropped.
21    Drop,
22    /// Enables values to be held directly in global storage.
23    Key,
24    /// Enables values to be held inside a struct in global storage.
25    Store,
26}

28/// The visibility modifier describes which modules can access this module member.
29/// By default, a module member can be called only within the same module.
30#[derive(Enum, Copy, Clone, Eq, PartialEq)]
31pub(crate) enum MoveVisibility {
32    /// A public member can be accessed by any module.
33    Public,
34    /// A private member can be accessed in the module it is defined in.
35    Private,
36    /// A friend member can be accessed in the module it is defined in and any other module in
37    /// its package that is explicitly specified in its friend list.
38    Friend,
39}

41scalar!(
42    OpenMoveTypeSignature,
43    "OpenMoveTypeSignature",
44    "The shape of an abstract Move Type (a type that can contain free type parameters, and can \
45     optionally be taken by reference), corresponding to the following recursive type:
46
47type OpenMoveTypeSignature = {
48  ref: (\"&\" | \"&mut\")?,
49  body: OpenMoveTypeSignatureBody,
50}
51
52type OpenMoveTypeSignatureBody =
53    \"address\"
54  | \"bool\"
55  | \"u8\" | \"u16\" | ... | \"u256\"
56  | { vector: OpenMoveTypeSignatureBody }
57  | {
58      datatype {
59        package: string,
60        module: string,
61        type: string,
62        typeParameters: [OpenMoveTypeSignatureBody]
63      }
64    }
65  | { typeParameter: number }"
66);

68#[derive(Serialize, Deserialize, Debug)]
69pub(crate) struct OpenMoveTypeSignature {
70    #[serde(rename = "ref")]
71    ref_: Option<OpenMoveTypeReference>,
72    body: OpenMoveTypeSignatureBody,
73}

75#[derive(Serialize, Deserialize, Debug)]
76pub(crate) enum OpenMoveTypeReference {
77    #[serde(rename = "&")]
78    Immutable,
79
80    #[serde(rename = "&mut")]
81    Mutable,
82}

84#[derive(Serialize, Deserialize, Debug)]
85#[serde(rename_all = "camelCase")]
86pub(crate) enum OpenMoveTypeSignatureBody {
87    TypeParameter(u16),
88    Address,
89    Bool,
90    U8,
91    U16,
92    U32,
93    U64,
94    U128,
95    U256,
96    Vector(Box<OpenMoveTypeSignatureBody>),
97    Datatype {
98        package: String,
99        module: String,
100        #[serde(rename = "type")]
101        type_: String,
102        #[serde(rename = "typeParameters")]
103        type_parameters: Vec<OpenMoveTypeSignatureBody>,
104    },
105}

107/// Represents types that could contain references or free type parameters.  Such types can appear
108/// as function parameters, in fields of structs, or as actual type parameter.
109#[Object]
110impl OpenMoveType {
111    /// Structured representation of the type signature.
112    async fn signature(&self) -> &OpenMoveTypeSignature {
113        &self.signature
114    }
115
116    /// Flat representation of the type signature, as a displayable string.
117    async fn repr(&self) -> String {
118        self.signature.to_string()
119    }
120}

156impl From<OpenSignatureBody> for OpenMoveTypeSignatureBody {
157    fn from(signature: OpenSignatureBody) -> Self {
158        use OpenMoveTypeSignatureBody as OMTSB;
159        use OpenSignatureBody as OSB;
160
161        match signature {
162            OSB::Address => OMTSB::Address,
163            OSB::Bool => OMTSB::Bool,
164            OSB::U8 => OMTSB::U8,
165            OSB::U16 => OMTSB::U16,
166            OSB::U32 => OMTSB::U32,
167            OSB::U64 => OMTSB::U64,
168            OSB::U128 => OMTSB::U128,
169            OSB::U256 => OMTSB::U256,
170
171            OSB::Vector(signature) => OMTSB::Vector(Box::new(OMTSB::from(*signature))),
172
173            OSB::Datatype(struct_, type_params) => OMTSB::Datatype {
174                package: struct_.package.to_canonical_string(/* with_prefix */ true),
175                module: struct_.module.to_string(),
176                type_: struct_.name.to_string(),
177                type_parameters: type_params.into_iter().map(OMTSB::from).collect(),
178            },
179
180            OSB::TypeParameter(idx) => OMTSB::TypeParameter(idx),
181        }
182    }
183}

185impl From<Reference> for OpenMoveTypeReference {
186    fn from(ref_: Reference) -> Self {
187        use OpenMoveTypeReference as M;
188        use Reference as R;
189
190        match ref_ {
191            R::Immutable => M::Immutable,
192            R::Mutable => M::Mutable,
193        }
194    }
195}

197impl From<Ability> for MoveAbility {
198    fn from(ability: Ability) -> Self {
199        use Ability as A;
200        use MoveAbility as M;
201
202        match ability {
203            A::Copy => M::Copy,
204            A::Drop => M::Drop,
205            A::Store => M::Store,
206            A::Key => M::Key,
207        }
208    }
209}
210
211impl From<Visibility> for MoveVisibility {
212    fn from(visibility: Visibility) -> Self {
213        use MoveVisibility as M;
214        use Visibility as V;
215
216        match visibility {
217            V::Private => M::Private,
218            V::Public => M::Public,
219            V::Friend => M::Friend,
220        }
221    }
222}

224impl fmt::Display for OpenMoveTypeSignature {
225    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
226        use OpenMoveTypeReference as R;
227        let OpenMoveTypeSignature { ref_, body } = self;
228
229        if let Some(r) = ref_ {
230            match r {
231                R::Immutable => write!(f, "&")?,
232                R::Mutable => write!(f, "&mut ")?,
233            }
234        }
235
236        write!(f, "{body}")
237    }
238}
239
240impl fmt::Display for OpenMoveTypeSignatureBody {
241    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
242        use OpenMoveTypeSignatureBody as B;
243
244        match self {
245            B::TypeParameter(idx) => write!(f, "${idx}"),
246
247            B::Address => write!(f, "address"),
248            B::Bool => write!(f, "bool"),
249            B::U8 => write!(f, "u8"),
250            B::U16 => write!(f, "u16"),
251            B::U32 => write!(f, "u32"),
252            B::U64 => write!(f, "u64"),
253            B::U128 => write!(f, "u128"),
254            B::U256 => write!(f, "u256"),
255            B::Vector(sig) => write!(f, "vector<{sig}>"),
256
257            B::Datatype {
258                package,
259                module,
260                type_,
261                type_parameters,
262            } => {
263                write!(f, "{package}::{module}::{type_}")?;
264
265                let mut params = type_parameters.iter();
266                let Some(param) = params.next() else {
267                    return Ok(());
268                };
269
270                write!(f, "<{param}")?;
271                for param in params {
272                    write!(f, ", {param}")?;
273                }
274                write!(f, ">")
275            }
276        }
277    }
278}

280/// Convert an `AbilitySet` from the binary format into a vector of `MoveAbility` (a GraphQL type).
281pub(crate) fn abilities(set: AbilitySet) -> Vec<MoveAbility> {
282    set.into_iter().map(MoveAbility::from).collect()
283}

309            OpenMoveTypeSignature {
