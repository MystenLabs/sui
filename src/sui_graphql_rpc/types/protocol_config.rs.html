18/// A single protocol configuration value.
19#[derive(Clone, Debug, SimpleObject)]
20pub(crate) struct ProtocolConfigAttr {
21    pub key: String,
22    pub value: Option<String>,
23}
24
25/// Whether or not a single feature is enabled in the protocol config.
26#[derive(Clone, Debug, SimpleObject)]
27pub(crate) struct ProtocolConfigFeatureFlag {
28    pub key: String,
29    pub value: bool,
30}
31
32#[derive(Clone, Debug)]
33pub(crate) struct ProtocolConfigs {
34    version: u64,
35    configs: BTreeMap<String, Option<String>>,
36    feature_flags: BTreeMap<String, bool>,
37}
38
39/// Constants that control how the chain operates.
40///
41/// These can only change during protocol upgrades which happen on epoch boundaries.
42#[Object]
43impl ProtocolConfigs {
44    /// The protocol is not required to change on every epoch boundary, so the protocol version
45    /// tracks which change to the protocol these configs are from.
46    async fn protocol_version(&self) -> UInt53 {
47        self.version.into()
48    }
49
50    /// List all available feature flags and their values.  Feature flags are a form of boolean
51    /// configuration that are usually used to gate features while they are in development.  Once a
52    /// flag has been enabled, it is rare for it to be disabled.
53    async fn feature_flags(&self) -> Vec<ProtocolConfigFeatureFlag> {
54        self.feature_flags
55            .clone()
56            .into_iter()
57            .map(|(key, value)| ProtocolConfigFeatureFlag { key, value })
58            .collect()
59    }
60
61    /// List all available configurations and their values.  These configurations can take any value
62    /// (but they will all be represented in string form), and do not include feature flags.
63    async fn configs(&self) -> Vec<ProtocolConfigAttr> {
64        self.configs
65            .clone()
66            .into_iter()
67            .map(|(key, value)| ProtocolConfigAttr { key, value })
68            .collect()
69    }
70
71    /// Query for the value of the configuration with name `key`.
72    async fn config(&self, key: String) -> Option<ProtocolConfigAttr> {
73        self.configs.get(&key).map(|value| ProtocolConfigAttr {
74            key,
75            value: value.as_ref().map(|v| v.to_string()),
76        })
77    }
78
79    /// Query for the state of the feature flag with name `key`.
80    async fn feature_flag(&self, key: String) -> Option<ProtocolConfigFeatureFlag> {
81        self.feature_flags
82            .get(&key)
83            .map(|value| ProtocolConfigFeatureFlag { key, value: *value })
84    }
85}
86
87impl ProtocolConfigs {
88    pub(crate) async fn query(db: &Db, protocol_version: Option<u64>) -> Result<Self, Error> {
89        use epochs::dsl as e;
90        use feature_flags::dsl as f;
91        use protocol_configs::dsl as p;
92
93        let version = if let Some(version) = protocol_version {
94            version
95        } else {
96            let latest_version: i64 = db
97                .execute(move |conn| {
98                    async move {
99                        conn.first(move || {
100                            e::epochs
101                                .select(e::protocol_version)
102                                .order_by(e::epoch.desc())
103                        })
104                        .await
105                    }
106                    .scope_boxed()
107                })
108                .await
109                .map_err(|e| {
110                    Error::Internal(format!(
111                        "Failed to fetch latest protocol version in db: {e}"
112                    ))
113                })?;
114            latest_version as u64
115        };
116
117        // TODO: This could be optimized by fetching all configs and flags in a single query.
118        let configs: BTreeMap<String, Option<String>> = db
119            .execute(move |conn| {
120                async move {
121                    conn.results(move || {
122                        p::protocol_configs
123                            .select((p::config_name, p::config_value))
124                            .filter(p::protocol_version.eq(version as i64))
125                    })
126                    .await
127                }
128                .scope_boxed()
129            })
130            .await
131            .map_err(|e| Error::Internal(format!("Failed to fetch protocol configs in db: {e}")))?
132            .into_iter()
133            .collect();
134
135        let feature_flags: BTreeMap<String, bool> = db
136            .execute(move |conn| {
137                async move {
138                    conn.results(move || {
139                        f::feature_flags
140                            .select((f::flag_name, f::flag_value))
141                            .filter(f::protocol_version.eq(version as i64))
142                    })
143                    .await
144                }
145                .scope_boxed()
146            })
147            .await
148            .map_err(|e| Error::Internal(format!("Failed to fetch feature flags in db: {e}")))?
149            .into_iter()
150            .collect();
151
152        Ok(ProtocolConfigs {
153            version,
154            configs,
155            feature_flags,
156        })
157    }
158}
