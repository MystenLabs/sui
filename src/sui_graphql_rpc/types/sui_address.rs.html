12const SUI_ADDRESS_LENGTH: usize = 32;
13
14#[derive(Serialize, Deserialize, Clone, Debug, Eq, PartialEq, Ord, PartialOrd, Hash, Copy)]
15pub(crate) struct SuiAddress([u8; SUI_ADDRESS_LENGTH]);
16
17#[derive(thiserror::Error, Debug, Eq, PartialEq)]
18pub(crate) enum FromStrError {
19    #[error("Invalid SuiAddress. Missing 0x prefix.")]
20    NoPrefix,
21
22    #[error(
23        "Expected SuiAddress string with between 1 and {} digits ({} bytes), received {0}",
24        SUI_ADDRESS_LENGTH * 2,
25        SUI_ADDRESS_LENGTH,
26    )]
27    WrongLength(usize),
28
29    #[error("Invalid character {0:?} at position {1}")]
30    BadHex(char, usize),
31}
32
33#[derive(thiserror::Error, Debug, Eq, PartialEq)]
34pub(crate) enum FromVecError {
35    #[error("Expected SuiAddress with {} bytes, received {0}", SUI_ADDRESS_LENGTH)]
36    WrongLength(usize),
37}

39impl SuiAddress {
40    pub fn from_array(arr: [u8; SUI_ADDRESS_LENGTH]) -> Self {
41        SuiAddress(arr)
42    }
43
44    pub fn into_vec(self) -> Vec<u8> {
45        self.0.to_vec()
46    }
47
48    pub fn as_slice(&self) -> &[u8] {
49        &self.0
50    }
51
52    pub fn from_bytes<T: AsRef<[u8]>>(bytes: T) -> Result<Self, FromVecError> {
53        <[u8; SUI_ADDRESS_LENGTH]>::try_from(bytes.as_ref())
54            .map_err(|_| FromVecError::WrongLength(bytes.as_ref().len()))
55            .map(SuiAddress)
56    }
57}

59#[Scalar(use_type_description = true)]
60impl ScalarType for SuiAddress {
61    fn parse(value: Value) -> InputValueResult<Self> {
62        let Value::String(s) = value else {
63            return Err(InputValueError::expected_type(value));
64        };
65
66        Ok(SuiAddress::from_str(&s)?)
67    }
68
69    fn to_value(&self) -> Value {
70        Value::String(format!("0x{}", hex::encode(self.0)))
71    }
72}

126impl FromStr for SuiAddress {
127    type Err = FromStrError;
128
129    fn from_str(s: &str) -> Result<Self, FromStrError> {
130        let Some(s) = s.strip_prefix("0x") else {
131            return Err(FromStrError::NoPrefix);
132        };
133
134        if s.is_empty() || s.len() > SUI_ADDRESS_LENGTH * 2 {
135            return Err(FromStrError::WrongLength(s.len()));
136        }
137
138        let mut arr = [0u8; SUI_ADDRESS_LENGTH];
139        hex::decode_to_slice(
140            // Left pad with `0`-s up to SUI_ADDRESS_LENGTH * 2 characters long.
141            format!("{:0>width$}", s, width = SUI_ADDRESS_LENGTH * 2),
142            &mut arr[..],
143        )
144        .map_err(|e| match e {
145            hex::FromHexError::InvalidHexCharacter { c, index } => {
146                FromStrError::BadHex(c, index + 2)
147            }
148            hex::FromHexError::OddLength => unreachable!("SAFETY: Prevented by padding"),
149            hex::FromHexError::InvalidStringLength => {
150                unreachable!("SAFETY: Prevented by bounds check")
151            }
152        })?;
153
154        Ok(SuiAddress(arr))
155    }
156}

158impl std::fmt::Display for SuiAddress {
159    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
160        f.write_str(&format!("0x{}", hex::encode(self.0)))

164/// Parse a `SuiAddress` from its stored representation.  Failure is an internal error: the
165/// database should never contain a malformed address (containing the wrong number of bytes).
166pub(crate) fn addr(bytes: impl AsRef<[u8]>) -> Result<SuiAddress, Error> {
167    SuiAddress::from_bytes(bytes.as_ref()).map_err(|e| {
168        let bytes = bytes.as_ref().to_vec();
169        Error::Internal(format!("Error deserializing address: {bytes:?}: {e}"))
170    })
171}

173#[cfg(test)]
174mod tests {
175    use super::*;
176    use async_graphql::Value;
177
178    const STR_ADDRESS: &str = "0x0123456789abcdef0123456789abcdef0123456789abcdef0123456789abcdef";
179    const ARR_ADDRESS: [u8; SUI_ADDRESS_LENGTH] = [
180        1, 35, 69, 103, 137, 171, 205, 239, 1, 35, 69, 103, 137, 171, 205, 239, 1, 35, 69, 103,
181        137, 171, 205, 239, 1, 35, 69, 103, 137, 171, 205, 239,
182    ];
183    const SUI_ADDRESS: SuiAddress = SuiAddress(ARR_ADDRESS);
184
185    #[test]
186    fn test_parse_valid_suiaddress() {
187        let parsed = SuiAddress::from_str(STR_ADDRESS).unwrap();
188        assert_eq!(parsed.0, ARR_ADDRESS);
189    }
190
191    #[test]
192    fn test_to_value() {
193        let value = ScalarType::to_value(&SUI_ADDRESS);
194        assert_eq!(value, Value::String(STR_ADDRESS.to_string()));
195    }
196
197    #[test]
198    fn test_from_array() {
199        let addr = SuiAddress::from_array(ARR_ADDRESS);
200        assert_eq!(addr, SUI_ADDRESS);
201    }
202
203    #[test]
204    fn test_as_slice() {
205        assert_eq!(SUI_ADDRESS.as_slice(), &ARR_ADDRESS);
206    }
207
208    #[test]
209    fn test_round_trip() {
210        let value = ScalarType::to_value(&SUI_ADDRESS);
211        let parsed_back = ScalarType::parse(value).unwrap();
212        assert_eq!(SUI_ADDRESS, parsed_back);
213    }
214
215    #[test]
216    fn test_parse_no_prefix() {
217        let err = SuiAddress::from_str(&STR_ADDRESS[2..]).unwrap_err();
218        assert_eq!(FromStrError::NoPrefix, err);
219    }
220
221    #[test]
222    fn test_parse_invalid_prefix() {
223        let input = "1x".to_string() + &STR_ADDRESS[2..];
224        let err = SuiAddress::from_str(&input).unwrap_err();
225        assert_eq!(FromStrError::NoPrefix, err)
226    }
227
228    #[test]
229    fn test_parse_invalid_length() {
230        let input = STR_ADDRESS.to_string() + "0123";
231        let err = SuiAddress::from_str(&input).unwrap_err();
232        assert_eq!(FromStrError::WrongLength(68), err)
233    }
234
235    #[test]
236    fn test_parse_invalid_characters() {
237        let input = "0xg".to_string() + &STR_ADDRESS[3..];
238        let err = SuiAddress::from_str(&input).unwrap_err();
239        assert_eq!(FromStrError::BadHex('g', 2), err);
240    }
241
242    #[test]
243    fn test_unicode_gibberish() {
244        let parsed = SuiAddress::from_str("aAௗ0㌀0");
245        assert!(parsed.is_err());
246    }

248    #[test]
249    fn bad_scalar_type() {
250        let input = Value::Number(0x42.into());
251        let parsed = <SuiAddress as ScalarType>::parse(input);
252        assert!(parsed.is_err());
253    }
