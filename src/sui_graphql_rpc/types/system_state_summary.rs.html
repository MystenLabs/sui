4use super::{
5    big_int::BigInt, gas::GasCostSummary, safe_mode::SafeMode, stake_subsidy::StakeSubsidy,
6    storage_fund::StorageFund, system_parameters::SystemParameters, uint53::UInt53,
7};
8use async_graphql::*;
9use sui_types::sui_system_state::sui_system_state_summary::SuiSystemStateSummary as NativeSystemStateSummary;
10
11#[derive(Clone, Debug)]
12pub(crate) struct SystemStateSummary {
13    pub native: NativeSystemStateSummary,
14}
15
16/// Aspects that affect the running of the system that are managed by the validators either
17/// directly, or through system transactions.
18#[Object]
19impl SystemStateSummary {
20    /// SUI set aside to account for objects stored on-chain, at the start of the epoch.
21    /// This is also used for storage rebates.
22    async fn storage_fund(&self) -> Option<StorageFund> {
23        Some(StorageFund {
24            total_object_storage_rebates: Some(BigInt::from(
25                self.native.storage_fund_total_object_storage_rebates,
26            )),
27            non_refundable_balance: Some(BigInt::from(
28                self.native.storage_fund_non_refundable_balance,
29            )),
30        })
31    }
32
33    /// Information about whether this epoch was started in safe mode, which happens if the full epoch
34    /// change logic fails for some reason.
35    async fn safe_mode(&self) -> Option<SafeMode> {
36        Some(SafeMode {
37            enabled: Some(self.native.safe_mode),
38            gas_summary: Some(GasCostSummary {
39                computation_cost: self.native.safe_mode_computation_rewards,
40                storage_cost: self.native.safe_mode_storage_rewards,
41                storage_rebate: self.native.safe_mode_storage_rebates,
42                non_refundable_storage_fee: self.native.safe_mode_non_refundable_storage_fee,
43            }),
44        })
45    }
46
47    /// The value of the `version` field of `0x5`, the `0x3::sui::SuiSystemState` object.  This
48    /// version changes whenever the fields contained in the system state object (held in a dynamic
49    /// field attached to `0x5`) change.
50    async fn system_state_version(&self) -> Option<UInt53> {
51        Some(self.native.system_state_version.into())
52    }
53
54    /// Details of the system that are decided during genesis.
55    async fn system_parameters(&self) -> Option<SystemParameters> {
56        Some(SystemParameters {
57            duration_ms: Some(BigInt::from(self.native.epoch_duration_ms)),
58            stake_subsidy_start_epoch: Some(self.native.stake_subsidy_start_epoch.into()),
59            // TODO min validator count can be extracted, but it requires some JSON RPC changes,
60            // so we decided to wait on it for now.
61            min_validator_count: None,
62            max_validator_count: Some(self.native.max_validator_count),
63            min_validator_joining_stake: Some(BigInt::from(
64                self.native.min_validator_joining_stake,
65            )),
66            validator_low_stake_threshold: Some(BigInt::from(
67                self.native.validator_low_stake_threshold,
68            )),
69            validator_very_low_stake_threshold: Some(BigInt::from(
70                self.native.validator_very_low_stake_threshold,
71            )),
72            validator_low_stake_grace_period: Some(BigInt::from(
73                self.native.validator_low_stake_grace_period,
74            )),
75        })
76    }
77
78    /// Parameters related to the subsidy that supplements staking rewards
79    async fn system_stake_subsidy(&self) -> Option<StakeSubsidy> {
80        Some(StakeSubsidy {
81            balance: Some(BigInt::from(self.native.stake_subsidy_balance)),
82            distribution_counter: Some(self.native.stake_subsidy_distribution_counter),
83            current_distribution_amount: Some(BigInt::from(
84                self.native.stake_subsidy_current_distribution_amount,
85            )),
86            period_length: Some(self.native.stake_subsidy_period_length),
87            decrease_rate: Some(self.native.stake_subsidy_decrease_rate as u64),
88        })
89    }
90}
