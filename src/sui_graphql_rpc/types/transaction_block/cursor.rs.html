23/// The cursor returned for each `TransactionBlock` in a connection's page of results. The
24/// `checkpoint_viewed_at` will set the consistent upper bound for subsequent queries made on this
25/// cursor.
26#[derive(Serialize, Deserialize, Clone, PartialEq, Eq)]
27pub(crate) struct TransactionBlockCursor {
28    /// The checkpoint sequence number this was viewed at.
29    #[serde(rename = "c")]
30    pub checkpoint_viewed_at: u64,
31    #[serde(rename = "t")]
32    pub tx_sequence_number: u64,
33    /// Whether the cursor was derived from a `scan_limit`. Only applicable to the `startCursor` and
34    /// `endCursor` returned from a Connection's `PageInfo`, and indicates that the cursor may not
35    /// have a corresponding node in the result set.
36    #[serde(rename = "i")]
37    pub is_scan_limited: bool,
38}
39
40/// Results from raw queries in Diesel can only be deserialized into structs that implements
41/// `QueryableByName`. This struct is used to represent a row of `tx_sequence_number` returned from
42/// subqueries against tx lookup tables.
43#[derive(Clone, Debug)]
44pub struct TxLookup {
45    pub tx_sequence_number: i64,
46}
47
48impl Checkpointed for Cursor {
49    fn checkpoint_viewed_at(&self) -> u64 {
50        self.checkpoint_viewed_at
51    }
52}
53
54impl ScanLimited for Cursor {
55    fn is_scan_limited(&self) -> bool {
56        self.is_scan_limited
57    }
58
59    fn unlimited(&self) -> Self {
60        Cursor::new(TransactionBlockCursor {
61            is_scan_limited: false,
62            tx_sequence_number: self.tx_sequence_number,
63            checkpoint_viewed_at: self.checkpoint_viewed_at,
64        })
65    }
66}
67
68impl Paginated<Cursor> for StoredTransaction {
69    type Source = transactions::table;
70
71    fn filter_ge<ST, GB>(cursor: &Cursor, query: Query<ST, GB>) -> Query<ST, GB> {
72        query.filter(transactions::dsl::tx_sequence_number.ge(cursor.tx_sequence_number as i64))
73    }
74
75    fn filter_le<ST, GB>(cursor: &Cursor, query: Query<ST, GB>) -> Query<ST, GB> {
76        query.filter(transactions::dsl::tx_sequence_number.le(cursor.tx_sequence_number as i64))
77    }
78
79    fn order<ST, GB>(asc: bool, query: Query<ST, GB>) -> Query<ST, GB> {
80        use transactions::dsl;
81        if asc {
82            query.order_by(dsl::tx_sequence_number.asc())
83        } else {
84            query.order_by(dsl::tx_sequence_number.desc())
85        }
86    }
87}

157/// `sql_query` raw queries require `QueryableByName`. The default implementation looks for a table
158/// based on the struct name, and it also expects the struct's fields to reflect the table's
159/// columns. We can override this behavior by implementing `QueryableByName` for our struct. For
160/// `TxBounds`, its fields are derived from `checkpoints`, so we can't leverage the default
161/// implementation directly.
162impl<DB> QueryableByName<DB> for TxLookup
163where
164    DB: Backend,
165    i64: FromSql<diesel::sql_types::BigInt, DB>,
166{
167    fn build<'a>(row: &impl NamedRow<'a, DB>) -> deserialize::Result<Self> {
168        let tx_sequence_number =
169            NamedRow::get::<diesel::sql_types::BigInt, _>(row, "tx_sequence_number")?;
170
171        Ok(Self { tx_sequence_number })
172    }
