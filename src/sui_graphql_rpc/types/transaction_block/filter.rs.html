4use super::TransactionBlockKindInput;
5use crate::types::{digest::Digest, sui_address::SuiAddress, type_filter::FqNameFilter};
6use crate::types::{intersect, uint53::UInt53};
7use async_graphql::InputObject;
8use std::collections::BTreeSet;
9use sui_types::base_types::SuiAddress as NativeSuiAddress;
10
11#[derive(InputObject, Debug, Default, Clone)]
12pub(crate) struct TransactionBlockFilter {
13    /// Filter transactions by move function called. Calls can be filtered by the `package`,
14    /// `package::module`, or the `package::module::name` of their function.
15    pub function: Option<FqNameFilter>,
16
17    /// An input filter selecting for either system or programmable transactions.
18    pub kind: Option<TransactionBlockKindInput>,
19
20    /// Limit to transactions that occured strictly after the given checkpoint.
21    pub after_checkpoint: Option<UInt53>,
22
23    /// Limit to transactions in the given checkpoint.
24    pub at_checkpoint: Option<UInt53>,
25
26    /// Limit to transaction that occured strictly before the given checkpoint.
27    pub before_checkpoint: Option<UInt53>,
28
29    /// Limit to transactions that interacted with the given address. The address could be a
30    /// sender, sponsor, or recipient of the transaction.
31    pub affected_address: Option<SuiAddress>,
32
33    /// Limit to transactions that interacted with the given object. The object could have been
34    /// created, read, modified, deleted, wrapped, or unwrapped by the transaction. Objects that
35    /// were passed as a `Receiving` input are not considered to have been affected by a
36    /// transaction unless they were actually received.
37    #[cfg(feature = "staging")]
38    pub affected_object: Option<SuiAddress>,
39
40    /// Limit to transactions that were sent by the given address.
41    pub sent_address: Option<SuiAddress>,
42
43    /// Limit to transactions that accepted the given object as an input. NOTE: this input filter
44    /// has been deprecated in favor of `affectedObject` which offers an easier to under behavior.
45    ///
46    /// This filter will be removed with 1.36.0 (2024-10-14), or at least one release after
47    /// `affectedObject` is introduced, whichever is later.
48    pub input_object: Option<SuiAddress>,
49
50    /// Limit to transactions that output a versioon of this object. NOTE: this input filter has
51    /// been deprecated in favor of `affectedObject` which offers an easier to understand behavor.
52    ///
53    /// This filter will be removed with 1.36.0 (2024-10-14), or at least one release after
54    /// `affectedObject` is introduced, whichever is later.
55    pub changed_object: Option<SuiAddress>,
56
57    /// Select transactions by their digest.
58    pub transaction_ids: Option<Vec<Digest>>,
59}
60
61impl TransactionBlockFilter {
62    /// Try to create a filter whose results are the intersection of transaction blocks in `self`'s
63    /// results and transaction blocks in `other`'s results. This may not be possible if the
64    /// resulting filter is inconsistent in some way (e.g. a filter that requires one field to be
65    /// two different values simultaneously).
66    pub(crate) fn intersect(self, other: Self) -> Option<Self> {
67        macro_rules! intersect {
68            ($field:ident, $body:expr) => {
69                intersect::field(self.$field, other.$field, $body)
70            };
71        }
72
73        Some(Self {
74            function: intersect!(function, FqNameFilter::intersect)?,
75            kind: intersect!(kind, intersect::by_eq)?,
76
77            after_checkpoint: intersect!(after_checkpoint, intersect::by_max)?,
78            at_checkpoint: intersect!(at_checkpoint, intersect::by_eq)?,
79            before_checkpoint: intersect!(before_checkpoint, intersect::by_min)?,
80
81            affected_address: intersect!(affected_address, intersect::by_eq)?,
82            #[cfg(feature = "staging")]
83            affected_object: intersect!(affected_object, intersect::by_eq)?,
84            sent_address: intersect!(sent_address, intersect::by_eq)?,
85            input_object: intersect!(input_object, intersect::by_eq)?,
86            changed_object: intersect!(changed_object, intersect::by_eq)?,
87
88            transaction_ids: intersect!(transaction_ids, |a, b| {
89                let a = BTreeSet::from_iter(a.into_iter());
90                let b = BTreeSet::from_iter(b.into_iter());
91                Some(a.intersection(&b).cloned().collect())
92            })?,
93        })
94    }
95
96    /// Most filter conditions require a scan limit if used in tandem with other filters. The
97    /// exception to this is sender and checkpoint, since sender is denormalized on all tables, and
98    /// the corresponding tx range can be determined for a checkpoint.
99    pub(crate) fn requires_scan_limit(&self) -> bool {
100        [
101            self.function.is_some(),
102            self.kind.is_some(),
103            self.affected_address.is_some(),
104            #[cfg(feature = "staging")]
105            self.affected_object.is_some(),
106            self.input_object.is_some(),
107            self.changed_object.is_some(),
108            self.transaction_ids.is_some(),
109        ]
110        .into_iter()
111        .filter(|is_set| *is_set)
112        .count()
113            > 1
114    }
115
116    /// If we don't query a lookup table that has a denormalized sender column, we need to
117    /// explicitly specify the sender with a query on `tx_sender`. This function returns the sender
118    /// we need to add an explicit query for if one is required, or `None` otherwise.
119    pub(crate) fn explicit_sender(&self) -> Option<SuiAddress> {
120        let missing_implicit_sender = self.function.is_none()
121            && self.kind.is_none()
122            && self.affected_address.is_none()
123            && self.input_object.is_none()
124            && self.changed_object.is_none();
125
126        #[cfg(feature = "staging")]
127        let missing_implicit_sender = missing_implicit_sender && self.affected_object.is_none();
128
129        missing_implicit_sender
130            .then_some(self.sent_address)
131            .flatten()
132    }
133
134    /// A TransactionBlockFilter is considered not to have any filters if no filters are specified,
135    /// or if the only filters are on `checkpoint`.
136    pub(crate) fn has_filters(&self) -> bool {
137        let has_filters = self.function.is_some()
138            || self.kind.is_some()
139            || self.sent_address.is_some()
140            || self.affected_address.is_some()
141            || self.input_object.is_some()
142            || self.changed_object.is_some()
143            || self.transaction_ids.is_some();
144
145        #[cfg(feature = "staging")]
146        let has_filters = has_filters || self.affected_object.is_some();
147
148        has_filters
149    }
150
151    pub(crate) fn is_empty(&self) -> bool {
152        self.before_checkpoint == Some(UInt53::from(0))
153            || matches!(
154                (self.after_checkpoint, self.before_checkpoint),
155                (Some(after), Some(before)) if after >= before
156            )
157            || matches!(
158                (self.after_checkpoint, self.at_checkpoint),
159                (Some(after), Some(at)) if after >= at
160            )
161            || matches!(
162                (self.at_checkpoint, self.before_checkpoint),
163                (Some(at), Some(before)) if at >= before
164            )
165            // If SystemTx, sender if specified must be 0x0. Conversely, if sender is 0x0, kind must be SystemTx.
166            || matches!(
167                (self.kind, self.sent_address),
168                (Some(kind), Some(signer))
169                    if (kind == TransactionBlockKindInput::SystemTx)
170                        != (signer == SuiAddress::from(NativeSuiAddress::ZERO))
171            )
172    }
173}
