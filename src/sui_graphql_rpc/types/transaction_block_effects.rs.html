46/// Wraps the actual transaction block effects data with the checkpoint sequence number at which the
47/// data was viewed, for consistent results on paginating through and resolving nested types.
48#[derive(Clone, Debug)]
49pub(crate) struct TransactionBlockEffects {
50    pub kind: TransactionBlockEffectsKind,
51    /// The checkpoint sequence number this was viewed at.
52    pub checkpoint_viewed_at: u64,
53}
54
55#[derive(Clone, Debug)]
56pub(crate) enum TransactionBlockEffectsKind {
57    /// A transaction that has been indexed and stored in the database,
58    /// containing all information that the other two variants have, and more.
59    Stored {
60        stored_tx: StoredTransaction,
61        native: NativeTransactionEffects,
62    },
63    /// A transaction block that has been executed via executeTransactionBlock
64    /// but not yet indexed. So it does not contain checkpoint, timestamp or balanceChanges.
65    Executed {
66        tx_data: NativeSenderSignedData,
67        native: NativeTransactionEffects,
68        events: Vec<NativeEvent>,
69    },
70    /// A transaction block that has been executed via dryRunTransactionBlock. Similar to
71    /// Executed, it does not contain checkpoint, timestamp or balanceChanges.
72    DryRun {
73        tx_data: NativeTransactionData,
74        native: NativeTransactionEffects,
75        events: Vec<NativeEvent>,
76    },
77}

120    /// The reason for a transaction failure, if it did fail.
121    /// If the error is a Move abort, the error message will be resolved to a human-readable form if
122    /// possible, otherwise it will fall back to displaying the abort code and location.
123    async fn errors(&self, ctx: &Context<'_>) -> Result<Option<String>> {
124        let resolver: &PackageResolver = ctx.data_unchecked();
125        let status = self.resolve_native_status_impl(resolver).await?;
126
127        match status {
128            NativeExecutionStatus::Success => Ok(None),
129
130            NativeExecutionStatus::Failure {
131                error,
132                command: None,
133            } => Ok(Some(error.to_string())),
134
135            NativeExecutionStatus::Failure {
136                error,
137                command: Some(command),
138            } => {
139                let command = command + 1;
140                let suffix = match command % 10 {
141                    1 if command % 100 != 11 => "st",
142                    2 if command % 100 != 12 => "nd",
143                    3 if command % 100 != 13 => "rd",
144                    _ => "th",
145                };
146
147                let mut msg = String::new();
148                write!(msg, "Error in {command}{suffix} command, ")?;
149
150                let ExecutionFailureStatus::MoveAbort(loc, code) = &error else {
151                    write!(msg, "{error}")?;
152                    return Ok(Some(msg));
153                };
154
155                write!(msg, "from '{}", loc.module.to_canonical_display(true))?;
156                if let Some(fname) = &loc.function_name {
157                    write!(msg, "::{}'", fname)?;
158                } else {
159                    write!(msg, "'")?;
160                }
161
162                let Some(CleverError {
163                    source_line_number,
164                    error_info,
165                    error_code,
166                    ..
167                }) = resolver
168                    .resolve_clever_error(loc.module.clone(), *code)
169                    .await
170                else {
171                    write!(
172                        msg,
173                        " (instruction {}), abort code: {code}",
174                        loc.instruction,
175                    )?;
176                    return Ok(Some(msg));
177                };
178
179                let error_code_str = match error_code {
180                    Some(code) => format!("(code = {code})"),
181                    _ => String::new(),
182                };
183
184                match &error_info {
185                    ErrorConstants::Rendered {
186                        identifier,
187                        constant,
188                    } => {
189                        write!(
190                            msg,
191                            " (line {source_line_number}), abort{error_code_str} '{identifier}': {constant}"
192                        )?;
193                    }
194                    ErrorConstants::Raw { identifier, bytes } => {
195                        let const_str = FBase64::encode(bytes);
196                        write!(
197                            msg,
198                            " (line {source_line_number}), abort{error_code_str} '{identifier}': {const_str}"
199                        )?;
200                    }
201                    ErrorConstants::None => {
202                        write!(
203                            msg,
204                            " (line {source_line_number}){}",
205                            match error_code {
206                                Some(code) => format!(" abort(code = {code})"),
207                                _ => String::new(),
208                            }
209                        )?;
210                    }
211                }
212
213                Ok(Some(msg))
214            }
215        }
216    }

218    /// The error code of the Move abort, populated if this transaction failed with a Move abort.
219    async fn abort_code(&self, ctx: &Context<'_>) -> Result<Option<BigInt>> {
220        let resolver: &PackageResolver = ctx.data_unchecked();
221        let status = self.resolve_native_status_impl(resolver).await?;
222        let NativeExecutionStatus::Failure {
223            error: ExecutionFailureStatus::MoveAbort(loc, code),
224            ..
225        } = status
226        else {
227            return Ok(None);
228        };
229
230        let Some(CleverError {
231            error_code: Some(error_code),
232            ..
233        }) = resolver
234            .resolve_clever_error(loc.module.clone(), code)
235            .await
236        else {
237            return Ok(Some(BigInt::from(code)));
238        };
239
240        Ok(Some(BigInt::from(error_code as u64)))
241    }

243    /// Transactions whose outputs this transaction depends upon.
244    async fn dependencies(
245        &self,
246        ctx: &Context<'_>,
247        first: Option<u64>,
248        after: Option<CDependencies>,
249        last: Option<u64>,
250        before: Option<CDependencies>,
251    ) -> Result<
252        Connection<
253            String,
254            Option<TransactionBlock>,
255            EmptyFields,
256            EmptyFields,
257            DependencyConnectionNames,
258            DependencyConnectionNames,
259        >,
260    > {
261        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
262        let mut connection = Connection::new(false, false);
263
264        let dependencies = self.native().dependencies();
265
266        let Some((prev, next, _, cs)) =
267            page.paginate_consistent_indices(dependencies.len(), self.checkpoint_viewed_at)?
268        else {
269            return Ok(connection);
270        };
271
272        let indices: Vec<CDependencies> = cs.collect();
273
274        let (Some(fst), Some(lst)) = (indices.first(), indices.last()) else {
275            return Ok(connection);
276        };
277
278        let transactions = TransactionBlock::multi_query(
279            ctx,
280            dependencies[fst.ix..=lst.ix]
281                .iter()
282                .map(|d| Digest::from(*d))
283                .collect(),
284            fst.c, // Each element's cursor has the same checkpoint sequence number set
285        )
286        .await
287        .extend()?;
288
289        if transactions.is_empty() {
290            return Ok(connection);
291        };
292
293        connection.has_previous_page = prev;
294        connection.has_next_page = next;
295
296        for c in indices {
297            let digest: Digest = dependencies[c.ix].into();
298            connection.edges.push(Edge::new(
299                c.encode_cursor(),
300                transactions.get(&digest).cloned(),
301            ));
302        }
303
304        Ok(connection)
305    }

312    /// Consensus objects that are referenced by but not changed by this transaction.
313    async fn unchanged_consensus_objects(
314        &self,
315        ctx: &Context<'_>,
316        first: Option<u64>,
317        after: Option<CUnchangedConsensusObject>,
318        last: Option<u64>,
319        before: Option<CUnchangedConsensusObject>,
320    ) -> Result<Connection<String, UnchangedConsensusObject>> {
321        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
322        let mut connection = Connection::new(false, false);
323
324        let input_consensus_objects = self.native().input_consensus_objects();
325
326        let Some((prev, next, _, cs)) = page.paginate_consistent_indices(
327            input_consensus_objects.len(),
328            self.checkpoint_viewed_at,
329        )?
330        else {
331            return Ok(connection);
332        };
333
334        connection.has_previous_page = prev;
335        connection.has_next_page = next;
336
337        for c in cs {
338            let result =
339                UnchangedConsensusObject::try_from(input_consensus_objects[c.ix].clone(), c.c);
340            match result {
341                Ok(unchanged_consensus_object) => {
342                    connection
343                        .edges
344                        .push(Edge::new(c.encode_cursor(), unchanged_consensus_object));
345                }
346                Err(_consensus_object_changed) => continue, // Only add unchanged consensus objects to the connection.
347            }
348        }
349
350        Ok(connection)
351    }

390    /// The effect this transaction had on the balances (sum of coin values per coin type) of
391    /// addresses and objects.
392    async fn balance_changes(
393        &self,
394        ctx: &Context<'_>,
395        first: Option<u64>,
396        after: Option<CBalanceChange>,
397        last: Option<u64>,
398        before: Option<CBalanceChange>,
399    ) -> Result<Connection<String, BalanceChange>> {
400        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
401        let mut connection = Connection::new(false, false);
402
403        let TransactionBlockEffectsKind::Stored { stored_tx, .. } = &self.kind else {
404            return Ok(connection);
405        };
406
407        let Some((prev, next, _, cs)) = page
408            .paginate_consistent_indices(stored_tx.get_balance_len(), self.checkpoint_viewed_at)?
409        else {
410            return Ok(connection);
411        };
412
413        connection.has_previous_page = prev;
414        connection.has_next_page = next;
415
416        for c in cs {
417            let Some(serialized) = &stored_tx.get_balance_at_idx(c.ix) else {
418                continue;
419            };
420
421            let balance_change = BalanceChange::read(serialized, c.c).extend()?;
422            connection
423                .edges
424                .push(Edge::new(c.encode_cursor(), balance_change));
425        }
426
427        Ok(connection)
428    }

430    /// Events emitted by this transaction block.
431    async fn events(
432        &self,
433        ctx: &Context<'_>,
434        first: Option<u64>,
435        after: Option<CEvent>,
436        last: Option<u64>,
437        before: Option<CEvent>,
438    ) -> Result<Connection<String, Event>> {
439        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
440        let mut connection = Connection::new(false, false);
441        let len = match &self.kind {
442            TransactionBlockEffectsKind::Stored { stored_tx, .. } => stored_tx.get_event_len(),
443            TransactionBlockEffectsKind::Executed { events, .. }
444            | TransactionBlockEffectsKind::DryRun { events, .. } => events.len(),
445        };
446        let Some((prev, next, _, cs)) =
447            page.paginate_consistent_indices(len, self.checkpoint_viewed_at)?
448        else {
449            return Ok(connection);
450        };
451
452        connection.has_previous_page = prev;
453        connection.has_next_page = next;
454
455        for c in cs {
456            let event = match &self.kind {
457                TransactionBlockEffectsKind::Stored { stored_tx, .. } => {
458                    Event::try_from_stored_transaction(stored_tx, c.ix, c.c).extend()?
459                }
460                TransactionBlockEffectsKind::Executed { events, .. }
461                | TransactionBlockEffectsKind::DryRun { events, .. } => Event {
462                    stored: None,
463                    native: events[c.ix].clone(),
464                    checkpoint_viewed_at: c.c,
465                },
466            };
467            connection.edges.push(Edge::new(c.encode_cursor(), event));
468        }
469
470        Ok(connection)
471    }

492    /// The checkpoint this transaction was finalized in.
493    async fn checkpoint(&self, ctx: &Context<'_>) -> Result<Option<Checkpoint>> {
494        // If the transaction data is not a stored transaction, it's not in the checkpoint yet so we return None.
495        let TransactionBlockEffectsKind::Stored { stored_tx, .. } = &self.kind else {
496            return Ok(None);
497        };
498
499        Checkpoint::query(
500            ctx,
501            CheckpointId::by_seq_num(stored_tx.checkpoint_sequence_number as u64),
502            self.checkpoint_viewed_at,
503        )
504        .await
505        .extend()
506    }

508    /// Base64 encoded bcs serialization of the on-chain transaction effects.
509    async fn bcs(&self) -> Result<Base64> {
510        let bytes = if let TransactionBlockEffectsKind::Stored { stored_tx, .. } = &self.kind {
511            stored_tx.raw_effects.clone()
512        } else {
513            bcs::to_bytes(&self.native())
514                .map_err(|e| Error::Internal(format!("Error serializing transaction effects: {e}")))
515                .extend()?
516        };
517
518        Ok(Base64::from(bytes))
519    }

522impl TransactionBlockEffects {
523    fn native(&self) -> &NativeTransactionEffects {
524        match &self.kind {
525            TransactionBlockEffectsKind::Stored { native, .. } => native,
526            TransactionBlockEffectsKind::Executed { native, .. } => native,
527            TransactionBlockEffectsKind::DryRun { native, .. } => native,
528        }
529    }
530
531    /// Get the transaction data from the transaction block effects.
532    /// Will error if the transaction data is not available/invalid, but this should not occur.
533    fn transaction_data(&self) -> Result<NativeTransactionData> {
534        Ok(match &self.kind {
535            TransactionBlockEffectsKind::Stored { stored_tx, .. } => {
536                let s: NativeSenderSignedData = bcs::from_bytes(&stored_tx.raw_transaction)
537                    .map_err(|e| {
538                        Error::Internal(format!("Error deserializing transaction data: {e}"))
539                    })?;
540                s.transaction_data().clone()
541            }
542            TransactionBlockEffectsKind::Executed { tx_data, .. } => {
543                tx_data.transaction_data().clone()
544            }
545            TransactionBlockEffectsKind::DryRun { tx_data, .. } => tx_data.clone(),
546        })
547    }
548
549    /// Get the programmable transaction from the transaction block effects.
550    /// * If the transaction was unable to be retrieved, this will return an Err.
551    /// * If the transaction was able to be retrieved but was not a programmable transaction, this
552    ///   will return Ok(None).
553    /// * If the transaction was a programmable transaction, this will return Ok(Some(tx)).
554    fn programmable_transaction(&self) -> Result<Option<ProgrammableTransaction>> {
555        let tx_data = self.transaction_data()?;
556        match tx_data.into_kind() {
557            NativeTransactionKind::ProgrammableTransaction(tx) => Ok(Some(tx)),
558            _ => Ok(None),
559        }
560    }
561
562    /// Resolves the module ID within a Move abort to the storage ID of the package that the
563    /// abort occured in.
564    /// * If the error is not a Move abort, or the Move call in the programmable transaction cannot
565    ///   be found, this function will do nothing.
566    /// * If the error is a Move abort and the storage ID is unable to be resolved an error is
567    ///   returned.
568    async fn resolve_native_status_impl(
569        &self,
570        resolver: &PackageResolver,
571    ) -> Result<NativeExecutionStatus> {
572        let mut status = self.native().status().clone();
573        if let NativeExecutionStatus::Failure {
574            error:
575                ExecutionFailureStatus::MoveAbort(MoveLocation { module, .. }, _)
576                | ExecutionFailureStatus::MovePrimitiveRuntimeError(MoveLocationOpt(Some(MoveLocation {
577                    module,
578                    ..
579                }))),
580            command: Some(command_idx),
581        } = &mut status
582        {
583            // Get the Move call that this error is associated with.
584            if let Some(Command::MoveCall(ptb_call)) = self
585                .programmable_transaction()?
586                .and_then(|ptb| ptb.commands.into_iter().nth(*command_idx))
587            {
588                let module_new = module.clone();
589                // Resolve the runtime module ID in the Move abort to the storage ID of the package
590                // that the abort occured in. This is important to make sure that we look at the
591                // correct version of the module when resolving the error.
592                *module = resolver
593                    .resolve_module_id(module_new, ptb_call.package.into())
594                    .await
595                    .map_err(|e| Error::Internal(format!("Error resolving Move location: {e}")))?;
596            }
597        }
598        Ok(status)
599    }
600}

614impl TryFrom<TransactionBlock> for TransactionBlockEffects {
615    type Error = Error;
616
617    fn try_from(block: TransactionBlock) -> Result<Self, Error> {
618        let checkpoint_viewed_at = block.checkpoint_viewed_at;
619        let kind = match block.inner {
620            TransactionBlockInner::Stored { stored_tx, .. } => {
621                bcs::from_bytes(&stored_tx.raw_effects)
622                    .map(|native| TransactionBlockEffectsKind::Stored {
623                        stored_tx: stored_tx.clone(),
624                        native,
625                    })
626                    .map_err(|e| {
627                        Error::Internal(format!("Error deserializing transaction effects: {e}"))
628                    })
629            }
630            TransactionBlockInner::Executed {
631                tx_data,
632                effects,
633                events,
634            } => Ok(TransactionBlockEffectsKind::Executed {
635                tx_data,
636                native: effects,
637                events,
638            }),
639            TransactionBlockInner::DryRun {
640                tx_data,
641                effects,
642                events,
643            } => Ok(TransactionBlockEffectsKind::DryRun {
644                tx_data,
645                native: effects,
646                events,
647            }),
648        }?;
649
650        Ok(Self {
651            kind,
652            checkpoint_viewed_at,
653        })
654    }
