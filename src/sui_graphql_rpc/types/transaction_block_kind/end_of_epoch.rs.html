135/// System transaction that supersedes `ChangeEpochTransaction` as the new way to run transactions
136/// at the end of an epoch. Behaves similarly to `ChangeEpochTransaction` but can accommodate other
137/// optional transactions to run at the end of the epoch.
138#[Object]
139impl EndOfEpochTransaction {
140    /// The list of system transactions that are allowed to run at the end of the epoch.
141    async fn transactions(
142        &self,
143        ctx: &Context<'_>,
144        first: Option<u64>,
145        before: Option<CTxn>,
146        last: Option<u64>,
147        after: Option<CTxn>,
148    ) -> Result<Connection<String, EndOfEpochTransactionKind>> {
149        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
150
151        let mut connection = Connection::new(false, false);
152        let Some((prev, next, _, cs)) =
153            page.paginate_consistent_indices(self.native.len(), self.checkpoint_viewed_at)?
154        else {
155            return Ok(connection);
156        };
157
158        connection.has_previous_page = prev;
159        connection.has_next_page = next;
160
161        for c in cs {
162            let tx = EndOfEpochTransactionKind::from(self.native[c.ix].clone(), c.c);
163            connection.edges.push(Edge::new(c.encode_cursor(), tx));
164        }
165
166        Ok(connection)
167    }
168}

170/// A system transaction that updates epoch information on-chain (increments the current epoch).
171/// Executed by the system once per epoch, without using gas. Epoch change transactions cannot be
172/// submitted by users, because validators will refuse to sign them.
173///
174/// This transaction kind is deprecated in favour of `EndOfEpochTransaction`.
175#[Object]
176impl ChangeEpochTransaction {
177    /// The next (to become) epoch.
178    async fn epoch(&self, ctx: &Context<'_>) -> Result<Option<Epoch>> {
179        Epoch::query(ctx, Some(self.native.epoch), self.checkpoint_viewed_at)
180            .await
181            .extend()
182    }
183
184    /// The protocol version in effect in the new epoch.
185    async fn protocol_version(&self) -> UInt53 {
186        self.native.protocol_version.as_u64().into()
187    }
188
189    /// The total amount of gas charged for storage during the previous epoch (in MIST).
190    async fn storage_charge(&self) -> BigInt {
191        BigInt::from(self.native.storage_charge)
192    }
193
194    /// The total amount of gas charged for computation during the previous epoch (in MIST).
195    async fn computation_charge(&self) -> BigInt {
196        BigInt::from(self.native.computation_charge)
197    }
198
199    /// The SUI returned to transaction senders for cleaning up objects (in MIST).
200    async fn storage_rebate(&self) -> BigInt {
201        BigInt::from(self.native.storage_rebate)
202    }
203
204    /// The total gas retained from storage fees, that will not be returned by storage rebates when
205    /// the relevant objects are cleaned up (in MIST).
206    async fn non_refundable_storage_fee(&self) -> BigInt {
207        BigInt::from(self.native.non_refundable_storage_fee)
208    }
209
210    /// Time at which the next epoch will start.
211    async fn start_timestamp(&self) -> Result<DateTime, Error> {
212        DateTime::from_ms(self.native.epoch_start_timestamp_ms as i64)
213    }
214
215    /// System packages (specifically framework and move stdlib) that are written before the new
216    /// epoch starts, to upgrade them on-chain. Validators write these packages out when running the
217    /// transaction.
218    async fn system_packages(
219        &self,
220        ctx: &Context<'_>,
221        first: Option<u64>,
222        after: Option<CPackage>,
223        last: Option<u64>,
224        before: Option<CPackage>,
225    ) -> Result<Connection<String, MovePackage>> {
226        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
227
228        let mut connection = Connection::new(false, false);
229        let Some((prev, next, _, cs)) = page.paginate_consistent_indices(
230            self.native.system_packages.len(),
231            self.checkpoint_viewed_at,
232        )?
233        else {
234            return Ok(connection);
235        };
236
237        connection.has_previous_page = prev;
238        connection.has_next_page = next;
239
240        for c in cs {
241            let (version, modules, deps) = &self.native.system_packages[c.ix];
242            let compiled_modules = modules
243                .iter()
244                .map(|bytes| CompiledModule::deserialize_with_defaults(bytes))
245                .collect::<PartialVMResult<Vec<_>>>()
246                .map_err(|e| Error::Internal(format!("Failed to deserialize system modules: {e}")))
247                .extend()?;
248
249            let native = NativeObject::new_system_package(
250                &compiled_modules,
251                *version,
252                deps.clone(),
253                TransactionDigest::ZERO,
254            );
255
256            let runtime_id = native.id();
257            let object = Object::from_native(SuiAddress::from(runtime_id), native, c.c, None);
258            let package = MovePackage::try_from(&object)
259                .map_err(|_| Error::Internal("Failed to create system package".to_string()))
260                .extend()?;
261
262            connection.edges.push(Edge::new(c.encode_cursor(), package));
263        }
264
265        Ok(connection)
266    }
267}

301impl EndOfEpochTransactionKind {
302    fn from(kind: NativeEndOfEpochTransactionKind, checkpoint_viewed_at: u64) -> Self {
303        use EndOfEpochTransactionKind as K;
304        use NativeEndOfEpochTransactionKind as N;
305
306        match kind {
307            N::ChangeEpoch(ce) => K::ChangeEpoch(ChangeEpochTransaction {
308                native: ce,
309                checkpoint_viewed_at,
310            }),
311            N::AuthenticatorStateCreate => {
312                K::AuthenticatorStateCreate(AuthenticatorStateCreateTransaction { dummy: None })
313            }
314            N::AuthenticatorStateExpire(ase) => {
315                K::AuthenticatorStateExpire(AuthenticatorStateExpireTransaction {
316                    native: ase,
317                    checkpoint_viewed_at,
318                })
319            }
320            N::RandomnessStateCreate => {
321                K::RandomnessStateCreate(RandomnessStateCreateTransaction { dummy: None })
322            }
323            N::DenyListStateCreate => {
324                K::CoinDenyListStateCreate(CoinDenyListStateCreateTransaction { dummy: None })
325            }
326            N::BridgeStateCreate(chain_id) => K::BridgeStateCreate(BridgeStateCreateTransaction {
327                native: chain_id,
328                checkpoint_viewed_at,
329            }),
330            N::BridgeCommitteeInit(bridge_shared_version) => {
331                K::BridgeCommitteeInit(BridgeCommitteeInitTransaction {
332                    native: bridge_shared_version,
333                    checkpoint_viewed_at,
334                })
335            }
336            N::StoreExecutionTimeObservations(_) => {
337                K::StoreExecutionTimeObservations(StoreExecutionTimeObservationsTransaction {
338                    dummy: None,
339                })
340            }
341            N::AccumulatorRootCreate => {
342                K::AccumulatorRootCreate(AccumulatorRootCreateTransaction { dummy: None })
343            }
344            N::CoinRegistryCreate => {
345                K::CoinRegistryCreate(CoinRegistryCreateTransaction { dummy: None })
346            }
347            N::DisplayRegistryCreate => {
348                K::DisplayRegistryCreate(DisplayRegistryCreateTransaction { dummy: None })
349            }
350        }
351    }
