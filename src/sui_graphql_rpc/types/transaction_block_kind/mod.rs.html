24/// The kind of transaction block, either a programmable transaction or a system transaction.
25#[derive(Union, PartialEq, Clone, Eq)]
26pub(crate) enum TransactionBlockKind {
27    ConsensusCommitPrologue(ConsensusCommitPrologueTransaction),
28    Genesis(GenesisTransaction),
29    ChangeEpoch(ChangeEpochTransaction),
30    Programmable(ProgrammableTransactionBlock),
31    AuthenticatorState(AuthenticatorStateUpdateTransaction),
32    Randomness(RandomnessStateUpdateTransaction),
33    EndOfEpoch(EndOfEpochTransaction),
34    // GraphQL Union does not allow multiple variants with the same type
35    ProgrammableSystem(ProgrammableSystemTransactionBlock),
36}
37
38impl TransactionBlockKind {
39    pub(crate) fn from(kind: NativeTransactionKind, checkpoint_viewed_at: u64) -> Self {
40        use NativeTransactionKind as K;
41        use TransactionBlockKind as T;
42
43        match kind {
44            K::ProgrammableTransaction(pt) => T::Programmable(ProgrammableTransactionBlock {
45                native: pt,
46                checkpoint_viewed_at,
47            }),
48            K::ProgrammableSystemTransaction(pt) => {
49                T::ProgrammableSystem(ProgrammableSystemTransactionBlock {
50                    native: pt,
51                    checkpoint_viewed_at,
52                })
53            }
54            K::ChangeEpoch(ce) => T::ChangeEpoch(ChangeEpochTransaction {
55                native: ce,
56                checkpoint_viewed_at,
57            }),
58            K::Genesis(g) => T::Genesis(GenesisTransaction {
59                native: g,
60                checkpoint_viewed_at,
61            }),
62            K::ConsensusCommitPrologue(ccp) => T::ConsensusCommitPrologue(
63                ConsensusCommitPrologueTransaction::from_v1(ccp, checkpoint_viewed_at),
64            ),
65            K::ConsensusCommitPrologueV2(ccp) => T::ConsensusCommitPrologue(
66                ConsensusCommitPrologueTransaction::from_v2(ccp, checkpoint_viewed_at),
67            ),
68            K::ConsensusCommitPrologueV3(ccp) => T::ConsensusCommitPrologue(
69                ConsensusCommitPrologueTransaction::from_v3(ccp, checkpoint_viewed_at),
70            ),
71            K::ConsensusCommitPrologueV4(ccp) => T::ConsensusCommitPrologue(
72                ConsensusCommitPrologueTransaction::from_v4(ccp, checkpoint_viewed_at),
73            ),
74            K::AuthenticatorStateUpdate(asu) => {
75                T::AuthenticatorState(AuthenticatorStateUpdateTransaction {
76                    native: asu,
77                    checkpoint_viewed_at,
78                })
79            }
80            K::EndOfEpochTransaction(eoe) => T::EndOfEpoch(EndOfEpochTransaction {
81                native: eoe,
82                checkpoint_viewed_at,
83            }),
84            K::RandomnessStateUpdate(rsu) => T::Randomness(RandomnessStateUpdateTransaction {
85                native: rsu,
86                checkpoint_viewed_at,
87            }),
88        }
89    }
90}
