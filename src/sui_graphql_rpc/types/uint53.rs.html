4use std::fmt;
5
6use async_graphql::*;
7use sui_types::{base_types::SequenceNumber, sui_serde::BigInt};
8
9#[derive(Clone, Copy, Debug, Eq, PartialEq, Ord, PartialOrd)]
10pub(crate) struct UInt53(u64);
11
12/// An unsigned integer that can hold values up to 2^53 - 1. This can be treated similarly to `Int`,
13/// but it is guaranteed to be non-negative, and it may be larger than 2^32 - 1.
14#[Scalar(name = "UInt53")]
15impl ScalarType for UInt53 {
16    fn parse(value: Value) -> InputValueResult<Self> {
17        let Value::Number(n) = value else {
18            return Err(InputValueError::expected_type(value));
19        };
20
21        let Some(n) = n.as_u64() else {
22            return Err(InputValueError::custom("Expected an unsigned integer."));
23        };
24
25        Ok(UInt53(n))
26    }
27
28    fn to_value(&self) -> Value {
29        Value::Number(self.0.into())
30    }
31}
32
33impl fmt::Display for UInt53 {
34    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
35        write!(f, "{}", self.0)
36    }
37}
38
39impl From<u64> for UInt53 {
40    fn from(value: u64) -> Self {
41        Self(value)
42    }
43}
44
45impl From<UInt53> for SequenceNumber {
46    fn from(value: UInt53) -> Self {
47        SequenceNumber::from(value.0)
48    }
49}
50
51impl From<UInt53> for BigInt<u64> {
52    fn from(value: UInt53) -> Self {
53        BigInt::from(value.0)
54    }
55}
56
57impl From<UInt53> for u64 {
58    fn from(value: UInt53) -> Self {
59        value.0
60    }
61}
62
63impl From<UInt53> for i64 {
64    fn from(value: UInt53) -> Self {
65        value.0 as i64
66    }
67}
