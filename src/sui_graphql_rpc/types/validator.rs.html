42/// Loads the exchange rates from the cache and return a tuple (epoch stake subsidy started, and
43/// a BTreeMap holiding the exchange rates for each epoch for each validator.
44///
45/// It automatically filters the exchange rate table to only include data for the epochs that are
46/// less than or equal to the requested epoch.
47#[async_trait::async_trait]
48impl Loader<u64> for Db {
49    type Value = (
50        EpochStakeSubsidyStarted,
51        BTreeMap<NativeSuiAddress, Vec<(EpochId, PoolTokenExchangeRate)>>,
52    );
53    type Error = Error;
54
55    async fn load(
56        &self,
57        keys: &[u64],
58    ) -> Result<
59        HashMap<
60            u64,
61            (
62                EpochStakeSubsidyStarted,
63                BTreeMap<NativeSuiAddress, Vec<(EpochId, PoolTokenExchangeRate)>>,
64            ),
65        >,
66        Error,
67    > {
68        let latest_sui_system_state = self
69            .inner
70            .get_latest_sui_system_state()
71            .await
72            .map_err(|_| Error::Internal("Failed to fetch latest Sui system state".to_string()))?;
73        let governance_api = GovernanceReadApi::new(self.inner.clone());
74        let exchange_rates = exchange_rates(&governance_api, &latest_sui_system_state)
75            .await
76            .map_err(|e| Error::Internal(format!("Error fetching exchange rates. {e}")))?;
77        let mut results = BTreeMap::new();
78
79        // The requested epoch is the epoch for which we want to compute the APY. For the current
80        // ongoing epoch we cannot compute an APY, so we compute it for epoch - 1.
81        // First need to check if that requested epoch is not the current running one. If it is,
82        // then subtract one as the APY cannot be computed for a running epoch.
83        // If no epoch is passed in the key, then we default to the latest epoch - 1
84        // for the same reasons as above.
85        let epoch_to_filter_out = if let Some(epoch) = keys.first() {
86            if epoch == &latest_sui_system_state.epoch {
87                *epoch - 1
88            } else {
89                *epoch
90            }
91        } else {
92            latest_sui_system_state.epoch - 1
93        };
94
95        // filter the exchange rates to only include data for the epochs that are less than or
96        // equal to the requested epoch. This enables us to get historical exchange rates
97        // accurately and pass this to the APY calculation function
98        // TODO we might even filter here by the epoch at which the stake subsidy started
99        // to avoid passing that to the `calculate_apy` function and doing another filter there
100        for er in exchange_rates {
101            results.insert(
102                er.address,
103                er.rates
104                    .into_iter()
105                    .filter(|(epoch, _)| epoch <= &epoch_to_filter_out)
106                    .collect(),
107            );
108        }
109
110        let requested_epoch = match keys.first() {
111            Some(x) => *x,
112            None => latest_sui_system_state.epoch,
113        };
114
115        let mut r = HashMap::new();
116        r.insert(
117            requested_epoch,
118            (latest_sui_system_state.stake_subsidy_start_epoch, results),
119        );
120
121        Ok(r)
122    }
123}

141            protocol_pub_key: Some(Base64::from(v.protocol_pubkey_bytes.clone())),
142            network_pub_key: Some(Base64::from(v.network_pubkey_bytes.clone())),
143            worker_pub_key: Some(Base64::from(v.worker_pubkey_bytes.clone())),
144            proof_of_possession: Some(Base64::from(v.proof_of_possession_bytes.clone())),
145            net_address: Some(v.net_address.clone()),
146            p2p_address: Some(v.p2p_address.clone()),
147            primary_address: Some(v.primary_address.clone()),
148            worker_address: Some(v.worker_address.clone()),
149        };
150        Some(credentials)
151    }

328    /// The addresses of other validators this validator has reported.
329    async fn report_records(
330        &self,
331        ctx: &Context<'_>,
332        first: Option<u64>,
333        before: Option<CAddr>,
334        last: Option<u64>,
335        after: Option<CAddr>,
336    ) -> Result<Connection<String, Address>> {
337        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
338
339        let mut connection = Connection::new(false, false);
340        let Some(addresses) = &self.report_records else {
341            return Ok(connection);
342        };
343
344        let Some((prev, next, _, cs)) =
345            page.paginate_consistent_indices(addresses.len(), self.checkpoint_viewed_at)?
346        else {
347            return Ok(connection);
348        };
349
350        connection.has_previous_page = prev;
351        connection.has_next_page = next;
352
353        for c in cs {
354            connection.edges.push(Edge::new(
355                c.encode_cursor(),
356                Address {
357                    address: addresses[c.ix].address,
358                    checkpoint_viewed_at: c.c,
359                },
360            ));
361        }
362
363        Ok(connection)
364    }
365
366    /// The APY of this validator in basis points.
367    /// To get the APY in percentage, divide by 100.
368    async fn apy(&self, ctx: &Context<'_>) -> Result<Option<u64>, Error> {
369        let DataLoader(loader) = ctx.data_unchecked();
370        let (stake_subsidy_start_epoch, exchange_rates) = loader
371            .load_one(self.requested_for_epoch)
372            .await?
373            .ok_or_else(|| Error::Internal("DataLoading exchange rates failed".to_string()))?;
374        let rates = exchange_rates
375            .get(&self.validator_summary.sui_address)
376            .ok_or_else(|| {
377                Error::Internal(format!(
378                    "Failed to get the exchange rate for this validator address {} for requested epoch {}",
379                    self.validator_summary.sui_address, self.requested_for_epoch
380                ))
381            })?;
382
383        let avg_apy = Some(calculate_apy(stake_subsidy_start_epoch, rates));
384
385        Ok(avg_apy.map(|x| (x * 10000.0) as u64))
386    }
