13/// Representation of `0x3::validator_set::ValidatorSet`.
14#[derive(Clone, Debug, SimpleObject, Default)]
15#[graphql(complex)]
16pub(crate) struct ValidatorSet {
17    /// Total amount of stake for all active validators at the beginning of the epoch.
18    pub total_stake: Option<BigInt>,
19
20    #[graphql(skip)]
21    /// The current list of active validators.
22    pub active_validators: Option<Vec<Validator>>,
23
24    /// Validators that are pending removal from the active validator set, expressed as indices in
25    /// to `activeValidators`.
26    pub pending_removals: Option<Vec<u64>>,
27
28    // TODO: instead of returning the id and size of the table, potentially return the table itself, paginated.
29    /// Object ID of the wrapped object `TableVec` storing the pending active validators.
30    pub pending_active_validators_id: Option<SuiAddress>,
31
32    /// Size of the pending active validators table.
33    pub pending_active_validators_size: Option<u64>,
34
35    /// Object ID of the `Table` storing the mapping from staking pool ids to the addresses
36    /// of the corresponding validators. This is needed because a validator's address
37    /// can potentially change but the object ID of its pool will not.
38    pub staking_pool_mappings_id: Option<SuiAddress>,
39
40    /// Size of the stake pool mappings `Table`.
41    pub staking_pool_mappings_size: Option<u64>,
42
43    /// Object ID of the `Table` storing the inactive staking pools.
44    pub inactive_pools_id: Option<SuiAddress>,
45
46    /// Size of the inactive pools `Table`.
47    pub inactive_pools_size: Option<u64>,
48
49    /// Object ID of the `Table` storing the validator candidates.
50    pub validator_candidates_id: Option<SuiAddress>,
51
52    /// Size of the validator candidates `Table`.
53    pub validator_candidates_size: Option<u64>,
54
55    #[graphql(skip)]
56    pub checkpoint_viewed_at: u64,
57}
58
59type CValidator = JsonCursor<ConsistentIndexCursor>;
60
61#[ComplexObject]
62impl ValidatorSet {
63    /// The current set of active validators.
64    async fn active_validators(
65        &self,
66        ctx: &Context<'_>,
67        first: Option<u64>,
68        before: Option<CValidator>,
69        last: Option<u64>,
70        after: Option<CValidator>,
71    ) -> Result<Connection<String, Validator>> {
72        let page = Page::from_params(ctx.data_unchecked(), first, after, last, before)?;
73
74        let mut connection = Connection::new(false, false);
75        let Some(validators) = &self.active_validators else {
76            return Ok(connection);
77        };
78
79        let Some((prev, next, _, cs)) =
80            page.paginate_consistent_indices(validators.len(), self.checkpoint_viewed_at)?
81        else {
82            return Ok(connection);
83        };
84
85        connection.has_previous_page = prev;
86        connection.has_next_page = next;
87
88        for c in cs {
89            let mut validator = validators[c.ix].clone();
90            validator.checkpoint_viewed_at = c.c;
91            connection
92                .edges
93                .push(Edge::new(c.encode_cursor(), validator));
94        }
95
96        Ok(connection)
97    }
98}
