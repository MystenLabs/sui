25pub struct MoveUtilsApi {
26    inner: IndexerReader,
27}

29impl MoveUtilsApi {
30    pub fn new(inner: IndexerReader) -> Self {
31        Self { inner }
32    }
33}
34
35#[async_trait]
36impl MoveUtilsServer for MoveUtilsApi {
37    async fn get_normalized_move_modules_by_package(
38        &self,
39        package_id: ObjectID,
40    ) -> RpcResult<BTreeMap<String, SuiMoveNormalizedModule>> {
41        let resolver_modules = self.inner.get_package(package_id).await?.modules().clone();
42        let pool = &mut normalized::RcPool::new();
43        let sui_normalized_modules = resolver_modules
44            .into_iter()
45            .map(|(k, v)| {
46                let m = &normalized::Module::new(pool, v.bytecode(), /* include code */ false);
47                (k, m.into())
48            })
49            .collect::<BTreeMap<String, SuiMoveNormalizedModule>>();
50        Ok(sui_normalized_modules)
51    }
52
53    async fn get_normalized_move_module(
54        &self,
55        package: ObjectID,
56        module_name: String,
57    ) -> RpcResult<SuiMoveNormalizedModule> {
58        let mut modules = self.get_normalized_move_modules_by_package(package).await?;
59        let module = modules.remove(&module_name).ok_or_else(|| {
60            SuiRpcInputError::GenericNotFound(format!(
61                "No module was found with name {module_name}",
62            ))
63        })?;
64        Ok(module)
65    }
66
67    async fn get_normalized_move_struct(
68        &self,
69        package: ObjectID,
70        module_name: String,
71        struct_name: String,
72    ) -> RpcResult<SuiMoveNormalizedStruct> {
73        let mut module = self
74            .get_normalized_move_module(package, module_name)
75            .await?;
76        module
77            .structs
78            .remove(&struct_name)
79            .ok_or_else(|| {
80                SuiRpcInputError::GenericNotFound(format!(
81                    "No struct was found with struct name {struct_name}"
82                ))
83            })
84            .map_err(Into::into)
85    }
86
87    async fn get_normalized_move_function(
88        &self,
89        package: ObjectID,
90        module_name: String,
91        function_name: String,
92    ) -> RpcResult<SuiMoveNormalizedFunction> {
93        let mut module = self
94            .get_normalized_move_module(package, module_name)
95            .await?;
96        module
97            .exposed_functions
98            .remove(&function_name)
99            .ok_or_else(|| {
100                SuiRpcInputError::GenericNotFound(format!(
101                    "No function was found with function name {function_name}",
102                ))
103            })
104            .map_err(Into::into)
105    }
106
107    async fn get_move_function_arg_types(
108        &self,
109        package: ObjectID,
110        module: String,
111        function: String,
112    ) -> RpcResult<Vec<MoveFunctionArgType>> {
113        let function = self
114            .get_normalized_move_function(package, module, function)
115            .await?;
116        let args = function
117            .parameters
118            .iter()
119            .map(|p| match p {
120                SuiMoveNormalizedType::Struct { .. } => {
121                    MoveFunctionArgType::Object(ObjectValueKind::ByValue)
122                }
123                SuiMoveNormalizedType::Vector(_) => {
124                    MoveFunctionArgType::Object(ObjectValueKind::ByValue)
125                }
126                SuiMoveNormalizedType::Reference(_) => {
127                    MoveFunctionArgType::Object(ObjectValueKind::ByImmutableReference)
128                }
129                SuiMoveNormalizedType::MutableReference(_) => {
130                    MoveFunctionArgType::Object(ObjectValueKind::ByMutableReference)
131                }
132                _ => MoveFunctionArgType::Pure,
133            })
134            .collect::<Vec<MoveFunctionArgType>>();
135        Ok(args)
136    }
137}
138

141        self.into_rpc()
142    }

145        sui_json_rpc_api::MoveUtilsOpenRpc::module_doc()
146    }
