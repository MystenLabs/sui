4use super::governance_api::GovernanceReadApi;
5use crate::indexer_reader::IndexerReader;
6use async_trait::async_trait;
7use move_core_types::language_storage::StructTag;
8use sui_json_rpc::transaction_builder_api::TransactionBuilderApi as SuiTransactionBuilderApi;
9use sui_json_rpc_types::{SuiObjectDataFilter, SuiObjectDataOptions, SuiObjectResponse};
10use sui_transaction_builder::DataReader;
11use sui_types::base_types::{ObjectID, ObjectInfo, SuiAddress};
12use sui_types::object::Object;
13
14pub(crate) struct TransactionBuilderApi {
15    inner: IndexerReader,
16}
17
18impl TransactionBuilderApi {
19    #[allow(clippy::new_ret_no_self)]
20    pub fn new(inner: IndexerReader) -> SuiTransactionBuilderApi {
21        SuiTransactionBuilderApi::new_with_data_reader(std::sync::Arc::new(Self { inner }))
22    }
23}
24
25#[async_trait]
26impl DataReader for TransactionBuilderApi {
27    async fn get_owned_objects(
28        &self,
29        address: SuiAddress,
30        object_type: StructTag,
31    ) -> Result<Vec<ObjectInfo>, anyhow::Error> {
32        let stored_objects = self
33            .inner
34            .get_owned_objects(
35                address,
36                Some(SuiObjectDataFilter::StructType(object_type)),
37                None,
38                50, // Limit the number of objects returned to 50
39            )
40            .await?;
41
42        stored_objects
43            .into_iter()
44            .map(|object| {
45                let object = Object::try_from(object)?;
46                let object_ref = object.compute_object_reference();
47                let info = ObjectInfo::new(&object_ref, &object);
48                Ok(info)
49            })
50            .collect::<Result<Vec<_>, _>>()
51    }
52
53    async fn get_object_with_options(
54        &self,
55        object_id: ObjectID,
56        options: SuiObjectDataOptions,
57    ) -> Result<SuiObjectResponse, anyhow::Error> {
58        let result = self.inner.get_object_read(object_id).await?;
59        Ok((result, options).try_into()?)
60    }
61
62    async fn get_reference_gas_price(&self) -> Result<u64, anyhow::Error> {
63        let epoch_info = GovernanceReadApi::new(self.inner.clone())
64            .get_epoch_info(None)
65            .await?;
66        Ok(epoch_info
67            .reference_gas_price
68            .ok_or_else(|| anyhow::anyhow!("missing latest reference_gas_price"))?)
69    }
70}
