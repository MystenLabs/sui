4use async_trait::async_trait;
5use fastcrypto::encoding::Base64;
6use jsonrpsee::RpcModule;
7use jsonrpsee::core::RpcResult;
8use jsonrpsee::http_client::HttpClient;
9
10use sui_json_rpc::SuiRpcModule;
11use sui_json_rpc_api::{WriteApiClient, WriteApiServer};
12use sui_json_rpc_types::{
13    DevInspectArgs, DevInspectResults, DryRunTransactionBlockResponse, SuiTransactionBlockResponse,
14    SuiTransactionBlockResponseOptions,
15};
16use sui_open_rpc::Module;
17use sui_types::base_types::SuiAddress;
18use sui_types::quorum_driver_types::ExecuteTransactionRequestType;
19use sui_types::sui_serde::BigInt;
20
21use crate::types::SuiTransactionBlockResponseWithOptions;
22
23pub(crate) struct WriteApi {
24    fullnode: HttpClient,
25}
26
27impl WriteApi {
28    pub fn new(fullnode_client: HttpClient) -> Self {
29        Self {
30            fullnode: fullnode_client,
31        }
32    }
33}
34
35#[async_trait]
36impl WriteApiServer for WriteApi {
37    async fn execute_transaction_block(
38        &self,
39        tx_bytes: Base64,
40        signatures: Vec<Base64>,
41        options: Option<SuiTransactionBlockResponseOptions>,
42        request_type: Option<ExecuteTransactionRequestType>,
43    ) -> RpcResult<SuiTransactionBlockResponse> {
44        let sui_transaction_response = self
45            .fullnode
46            .execute_transaction_block(tx_bytes, signatures, options.clone(), request_type)
47            .await
48            .map_err(crate::errors::client_error_to_error_object)?;
49        Ok(SuiTransactionBlockResponseWithOptions {
50            response: sui_transaction_response,
51            options: options.unwrap_or_default(),
52        }
53        .into())
54    }
55
56    async fn dev_inspect_transaction_block(
57        &self,
58        sender_address: SuiAddress,
59        tx_bytes: Base64,
60        gas_price: Option<BigInt<u64>>,
61        epoch: Option<BigInt<u64>>,
62        additional_args: Option<DevInspectArgs>,
63    ) -> RpcResult<DevInspectResults> {
64        self.fullnode
65            .dev_inspect_transaction_block(
66                sender_address,
67                tx_bytes,
68                gas_price,
69                epoch,
70                additional_args,
71            )
72            .await
73            .map_err(crate::errors::client_error_to_error_object)
74    }
75
76    async fn dry_run_transaction_block(
77        &self,
78        tx_bytes: Base64,
79    ) -> RpcResult<DryRunTransactionBlockResponse> {
80        self.fullnode
81            .dry_run_transaction_block(tx_bytes)
82            .await
83            .map_err(crate::errors::client_error_to_error_object)
84    }
85}
86
87impl SuiRpcModule for WriteApi {
88    fn rpc(self) -> RpcModule<Self> {
89        self.into_rpc()
90    }
91
92    fn rpc_doc_module() -> Module {
93        sui_json_rpc_api::WriteApiOpenRpc::module_doc()
94    }
95}
