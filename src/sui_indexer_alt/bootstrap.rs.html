27pub struct BootstrapGenesis {
28    pub stored_genesis: StoredGenesis,
29    pub stored_epoch_start: StoredEpochStart,
30}

32/// Ensures the genesis table has been populated before the rest of the indexer is run, and returns
33/// the information stored there. If the database has been bootstrapped before, this function will
34/// simply read the previously bootstrapped information. Otherwise, it will wait until the first
35/// checkpoint is available and extract the necessary information from there.
36pub async fn bootstrap(
37    indexer: &Indexer<Db>,
38    retry_interval: Duration,
39    bootstrap_genesis: Option<BootstrapGenesis>,
40) -> Result<StoredGenesis> {
41    info!("Bootstrapping indexer with genesis information");
42
43    let Ok(mut conn) = indexer.store().connect().await else {
44        bail!("Bootstrap failed to get connection for DB");
45    };
46
47    // 1. If the row has already been written, return it.
48    if let Some(genesis) = kv_genesis::table
49        .select(StoredGenesis::as_select())
50        .first(&mut conn)
51        .await
52        .optional()?
53    {
54        info!(
55            chain = genesis.chain()?.as_str(),
56            protocol = ?genesis.initial_protocol_version(),
57            "Indexer already bootstrapped",
58        );
59
60        return Ok(genesis);
61    }
62
63    let BootstrapGenesis {
64        stored_genesis,
65        stored_epoch_start,
66    } = match bootstrap_genesis {
67        // 2. If genesis is provided, use it to bootstrap.
68        Some(bootstrap_genesis) => bootstrap_genesis,
69        // 3. Otherwise, extract the necessary information from the genesis checkpoint:
70        //
71        // - Get the Genesis system transaction from the genesis checkpoint.
72        // - Get the system state object that was written out by the system transaction.
73        None => {
74            let genesis_checkpoint = indexer
75                .ingestion_client()
76                .wait_for(0, retry_interval)
77                .await
78                .context("Failed to fetch genesis checkpoint")?;
79
80            let Checkpoint {
81                summary: checkpoint_summary,
82                transactions,
83                ..
84            } = genesis_checkpoint.as_ref();
85
86            let Some(genesis_transaction) = transactions
87                .iter()
88                .find(|tx| matches!(tx.transaction.kind(), TransactionKind::Genesis(_)))
89            else {
90                bail!("Could not find Genesis transaction");
91            };
92
93            let output_objects: Vec<_> = genesis_transaction
94                .output_objects(&genesis_checkpoint.object_set)
95                .cloned()
96                .collect();
97
98            let sui_system_state = get_sui_system_state(&output_objects.as_slice())
99                .context("Failed to get Genesis SystemState")?;
100
101            let stored_genesis = StoredGenesis {
102                genesis_digest: checkpoint_summary.digest().inner().to_vec(),
103                initial_protocol_version: sui_system_state.protocol_version() as i64,
104            };
105            let stored_epoch_start = StoredEpochStart {
106                epoch: 0,
107                protocol_version: sui_system_state.protocol_version() as i64,
108                cp_lo: 0,
109                start_timestamp_ms: sui_system_state.epoch_start_timestamp_ms() as i64,
110                reference_gas_price: sui_system_state.reference_gas_price() as i64,
111                system_state: bcs::to_bytes(&sui_system_state)
112                    .context("Failed to serialize SystemState")?,
113            };
114
115            BootstrapGenesis {
116                stored_genesis,
117                stored_epoch_start,
118            }
119        }
120    };
121
122    info!(
123        chain = stored_genesis.chain()?.as_str(),
124        protocol = ?stored_genesis.initial_protocol_version(),
125        "Bootstrapped indexer",
126    );
127
128    diesel::insert_into(kv_genesis::table)
129        .values(&stored_genesis)
130        .on_conflict_do_nothing()
131        .execute(&mut conn)
132        .await
133        .context("Failed to write genesis record")?;
134
135    diesel::insert_into(kv_epoch_starts::table)
136        .values(&stored_epoch_start)
137        .on_conflict_do_nothing()
138        .execute(&mut conn)
139        .await
140        .context("Failed to write genesis epoch start record")?;
141
142    Ok(stored_genesis)
143}
