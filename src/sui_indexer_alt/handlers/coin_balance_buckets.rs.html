30/// This handler is used to track the balance buckets of address-owned coins.
31/// The balance bucket is calculated using log10 of the coin balance.
32/// Whenever a coin object's presence, owner or balance bucket changes,
33/// we will insert a new row into the `coin_balance_buckets` table.
34/// A Delete record will be inserted when a coin object is no longer present or no longer owned by an address.
35pub(crate) struct CoinBalanceBuckets;
36
37#[derive(Debug, PartialEq, Eq)]
38pub(crate) struct ProcessedCoinBalanceBucket {
39    pub object_id: ObjectID,
40    pub cp_sequence_number: u64,
41    pub change: CoinBalanceBucketChangeKind,
42}
43
44#[derive(Debug, PartialEq, Eq)]
45pub(crate) enum CoinBalanceBucketChangeKind {
46    Upsert {
47        owner_kind: StoredCoinOwnerKind,
48        owner_id: SuiAddress,
49        coin_type: TypeTag,
50        balance_bucket: i16,
51        /// Indicates whether the coin was created/unwrapped in this checkpoint.
52        created: bool,
53    },
54    Delete,
55}
56
57#[async_trait]
58impl Processor for CoinBalanceBuckets {
59    const NAME: &'static str = "coin_balance_buckets";
60    type Value = ProcessedCoinBalanceBucket;
61
62    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> Result<Vec<Self::Value>> {
63        let cp_sequence_number = checkpoint.checkpoint_summary.sequence_number;
64        let checkpoint_input_objects = checkpoint_input_objects(checkpoint)?;
65        let latest_live_output_objects: BTreeMap<_, _> = checkpoint
66            .latest_live_output_objects()
67            .into_iter()
68            .map(|o| (o.id(), o))
69            .collect();
70        let mut values: BTreeMap<ObjectID, Self::Value> = BTreeMap::new();
71        for (object_id, input_object) in checkpoint_input_objects.iter() {
72            // This loop processes all coins that were owned by a single address prior to the checkpoint,
73            // but is now deleted or wrapped after the checkpoint.
74            if !input_object.is_coin() {
75                continue;
76            }
77            if get_coin_owner(input_object).is_none() {
78                continue;
79            }
80            if latest_live_output_objects.contains_key(object_id) {
81                continue;
82            }
83            values.insert(
84                *object_id,
85                ProcessedCoinBalanceBucket {
86                    object_id: *object_id,
87                    cp_sequence_number,
88                    change: CoinBalanceBucketChangeKind::Delete,
89                },
90            );
91        }
92        for (object_id, output_object) in latest_live_output_objects.iter() {
93            let Some(coin_type) = output_object.coin_type_maybe() else {
94                continue;
95            };
96
97            let (input_bucket, input_owner) = match checkpoint_input_objects.get(object_id) {
98                Some(input_object) => {
99                    let bucket = get_coin_balance_bucket(input_object)?;
100                    let owner = get_coin_owner(input_object);
101                    (Some(bucket), owner)
102                }
103                None => (None, None),
104            };
105
106            let output_balance_bucket = get_coin_balance_bucket(output_object)?;
107            let output_owner = get_coin_owner(output_object);
108
109            match (input_owner, output_owner) {
110                (Some(_), None) => {
111                    // In this case, the coin was owned by a single address prior to the checkpoint,
112                    // but is now either shared or immutable after the checkpoint. We treat this the same
113                    // as if the coin was deleted, from the perspective of the balance bucket.
114                    values.insert(
115                        *object_id,
116                        ProcessedCoinBalanceBucket {
117                            object_id: *object_id,
118                            cp_sequence_number,
119                            change: CoinBalanceBucketChangeKind::Delete,
120                        },
121                    );
122                }
123                (_, Some(new_owner))
124                    if input_owner != output_owner
125                        || input_bucket != Some(output_balance_bucket) =>
126                {
127                    // In this case, the coin is still owned by a single address after the checkpoint,
128                    // but either the owner or the balance bucket has changed. This also includes the case
129                    // where the coin did not exist prior to the checkpoint, and is now created/unwrapped.
130                    values.insert(
131                        *object_id,
132                        ProcessedCoinBalanceBucket {
133                            object_id: *object_id,
134                            cp_sequence_number,
135                            change: CoinBalanceBucketChangeKind::Upsert {
136                                owner_kind: new_owner.0,
137                                owner_id: new_owner.1,
138                                coin_type,
139                                balance_bucket: output_balance_bucket,
140                                created: input_owner.is_none(),
141                            },
142                        },
143                    );
144                }
145                _ => {}
146            }
147        }
148
149        Ok(values.into_values().collect())
150    }
151}

153#[async_trait]
154impl Handler for CoinBalanceBuckets {
155    type Store = Db;
156
157    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
158        let stored = values
159            .iter()
160            .map(|v| v.try_into())
161            .collect::<Result<Vec<StoredCoinBalanceBucket>>>()?;
162
163        let mut references = Vec::new();
164        for value in values {
165            match &value.change {
166                CoinBalanceBucketChangeKind::Upsert { created, .. } => {
167                    if !created {
168                        references.push(StoredCoinBalanceBucketDeletionReference {
169                            object_id: value.object_id.to_vec(),
170                            cp_sequence_number: value.cp_sequence_number as i64,
171                        });
172                    }
173                }
174                // Store record of current version to delete previous version, and another to delete
175                // itself. When pruning, the deletion record will not be pruned in the
176                // `value.cp_sequence_number` checkpoint, but the next one.
177                CoinBalanceBucketChangeKind::Delete => {
178                    references.push(StoredCoinBalanceBucketDeletionReference {
179                        object_id: value.object_id.to_vec(),
180                        cp_sequence_number: value.cp_sequence_number as i64,
181                    });
182                    references.push(StoredCoinBalanceBucketDeletionReference {
183                        object_id: value.object_id.to_vec(),
184                        cp_sequence_number: value.cp_sequence_number as i64 + 1,
185                    });
186                }
187            }
188        }
189
190        let count = diesel::insert_into(coin_balance_buckets::table)
191            .values(&stored)
192            .on_conflict_do_nothing()
193            .execute(conn)
194            .await?;
195        let deleted_refs = if !references.is_empty() {
196            diesel::insert_into(coin_balance_buckets_deletion_reference::table)
197                .values(&references)
198                .on_conflict_do_nothing()
199                .execute(conn)
200                .await?
201        } else {
202            0
203        };
204
205        Ok(count + deleted_refs)
206    }

208    // TODO: Add tests for this function.
209    async fn prune<'a>(
210        &self,
211        from: u64,
212        to_exclusive: u64,
213        conn: &mut Connection<'a>,
214    ) -> anyhow::Result<usize> {
215        // This query first deletes from coin_balance_buckets_deletion_reference and computes
216        // predecessors, then deletes from coin_balance_buckets using the precomputed predecessor
217        // information. The inline compute avoids HashAggregate operations and the ensuing
218        // materialization overhead.
219        //
220        // This works best on under 1.5 million object changes, roughly 15k checkpoints. Performance
221        // degrades sharply beyond this, since the planner switches to hash joins and full table
222        // scans. A HashAggregate approach interestingly becomes more performant in this scenario.
223        //
224        // If the first call to prune succeeds, subsequent calls will find no records to delete from
225        // coin_balance_buckets_deletion_reference, and consequently no records to delete from the
226        // main table. Pruning is thus idempotent after the initial run.
227        //
228        // TODO: use sui_sql_macro's query!
229        let query = format!(
230            "
231            -- Delete reference records and return immediate predecessor refs to the main table.
232            WITH deletion_refs AS (
233                DELETE FROM
234                    coin_balance_buckets_deletion_reference dr
235                WHERE
236                    {} <= cp_sequence_number AND cp_sequence_number < {}
237                RETURNING
238                    object_id, (
239                    SELECT
240                        cb.cp_sequence_number
241                    FROM
242                        coin_balance_buckets cb
243                    WHERE
244                        dr.object_id = cb.object_id
245                    AND cb.cp_sequence_number < dr.cp_sequence_number
246                    ORDER BY
247                        cb.cp_sequence_number DESC
248                    LIMIT
249                        1
250                    ) AS cp_sequence_number
251            ),
252            deleted_coins AS (
253                DELETE FROM
254                    coin_balance_buckets cb
255                USING
256                    deletion_refs dr
257                WHERE
258                    cb.object_id = dr.object_id
259                AND cb.cp_sequence_number = dr.cp_sequence_number
260                RETURNING
261                    cb.object_id
262            )
263            SELECT
264                (SELECT COUNT(*) FROM deleted_coins) AS deleted_coins,
265                (SELECT COUNT(*) FROM deletion_refs) AS deleted_refs
266            ",
267            from, to_exclusive
268        );
269
270        #[derive(QueryableByName)]
271        struct CountResult {
272            #[diesel(sql_type = diesel::sql_types::BigInt)]
273            deleted_coins: i64,
274            #[diesel(sql_type = diesel::sql_types::BigInt)]
275            deleted_refs: i64,
276        }
277
278        let CountResult {
279            deleted_coins,
280            deleted_refs,
281        } = diesel::sql_query(query)
282            .get_result::<CountResult>(conn)
283            .await?;
284
285        ensure!(
286            deleted_coins == deleted_refs,
287            "Deleted coins count ({deleted_coins}) does not match deleted refs count ({deleted_refs})",
288        );
289
290        Ok((deleted_coins + deleted_refs) as usize)
291    }
292}

298impl TryInto<StoredCoinBalanceBucket> for &ProcessedCoinBalanceBucket {
299    type Error = anyhow::Error;
300
301    fn try_into(self) -> Result<StoredCoinBalanceBucket> {
302        match &self.change {
303            CoinBalanceBucketChangeKind::Upsert {
304                owner_kind,
305                owner_id,
306                coin_type,
307                balance_bucket,
308                created: _,
309            } => {
310                let serialized_coin_type = bcs::to_bytes(&coin_type)
311                    .map_err(|_| anyhow!("Failed to serialize type for {}", self.object_id))?;
312                Ok(StoredCoinBalanceBucket {
313                    object_id: self.object_id.to_vec(),
314                    cp_sequence_number: self.cp_sequence_number as i64,
315                    owner_kind: Some(*owner_kind),
316                    owner_id: Some(owner_id.to_vec()),
317                    coin_type: Some(serialized_coin_type),
318                    coin_balance_bucket: Some(*balance_bucket),
319                })
320            }
321            CoinBalanceBucketChangeKind::Delete => Ok(StoredCoinBalanceBucket {
322                object_id: self.object_id.to_vec(),
323                cp_sequence_number: self.cp_sequence_number as i64,
324                owner_kind: None,
325                owner_id: None,
326                coin_type: None,
327                coin_balance_bucket: None,
328            }),
329        }
330    }
331}

333/// Get the owner kind and address of a coin, if it is owned by a single address,
334/// either through fast-path ownership or consensus ownership.
335pub(crate) fn get_coin_owner(object: &Object) -> Option<(StoredCoinOwnerKind, SuiAddress)> {
336    match object.owner() {
337        Owner::AddressOwner(owner_id) => Some((StoredCoinOwnerKind::Fastpath, *owner_id)),
338        Owner::ConsensusAddressOwner { owner, .. } => {
339            Some((StoredCoinOwnerKind::Consensus, *owner))
340        }
341        Owner::Immutable | Owner::ObjectOwner(_) | Owner::Shared { .. } => None,
342    }
343}
344
345pub(crate) fn get_coin_balance_bucket(coin: &Object) -> anyhow::Result<i16> {
346    let Some(coin) = coin.as_coin_maybe() else {
347        // TODO: We should make this an invariant violation.
348        bail!("Failed to deserialize Coin for {}", coin.id());
349    };
350    let balance = coin.balance.value();
351    if balance == 0 {
352        return Ok(0);
353    }
354    let bucket = balance.ilog10() as i16;
355    Ok(bucket)
356}

1035    /// Test concurrent pruning operations to ensure thread safety and data consistency.
1036    /// This test creates the same scenario as test_process_coin_balance_buckets_out_of_order_pruning but runs
1037    /// multiple pruning operations concurrently.
1038    #[tokio::test]
1039    async fn test_process_coin_balance_buckets_concurrent_pruning() {
1040        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
1041        let mut conn = indexer.store().connect().await.unwrap();
1042        let mut builder = TestCheckpointDataBuilder::new(0);
1043
1044        // Create the same scenario as the out-of-order test
1045        builder = builder
1046            .start_transaction(0)
1047            .create_sui_object(0, 100)
1048            .create_sui_object(1, 1000)
1049            .create_sui_object(2, 10000)
1050            .finish_transaction();
1051        let checkpoint0 = builder.build_checkpoint();
1052        let result = CoinBalanceBuckets
1053            .process(&Arc::new(checkpoint0))
1054            .await
1055            .unwrap();
1056        CoinBalanceBuckets::commit(&result, &mut conn)
1057            .await
1058            .unwrap();
1059
1060        builder = builder
1061            .start_transaction(0)
1062            .transfer_object(0, 1)
1063            .transfer_object(1, 1)
1064            .transfer_object(2, 1)
1065            .finish_transaction();
1066        let checkpoint1 = builder.build_checkpoint();
1067        let result = CoinBalanceBuckets
1068            .process(&Arc::new(checkpoint1))
1069            .await
1070            .unwrap();
1071        CoinBalanceBuckets::commit(&result, &mut conn)
1072            .await
1073            .unwrap();
1074
1075        builder = builder
1076            .start_transaction(1)
1077            .transfer_object(0, 0)
1078            .transfer_object(1, 0)
1079            .transfer_object(2, 0)
1080            .finish_transaction();
1081        let checkpoint2 = builder.build_checkpoint();
1082        let result = CoinBalanceBuckets
1083            .process(&Arc::new(checkpoint2))
1084            .await
1085            .unwrap();
1086        CoinBalanceBuckets::commit(&result, &mut conn)
1087            .await
1088            .unwrap();
1089
1090        // Verify initial state
1091        let all_balance_buckets = get_all_balance_buckets(&mut conn).await;
1092        assert_eq!(all_balance_buckets.len(), 9); // 3 coins × 3 checkpoints
1093        let all_deletion_references = coin_balance_buckets_deletion_reference::table
1094            .load::<StoredCoinBalanceBucketDeletionReference>(&mut conn)
1095            .await
1096            .unwrap();
1097        assert_eq!(all_deletion_references.len(), 6);
1098
1099        // Run concurrent pruning operations
1100        let mut handles = Vec::new();
1101
1102        // Clone the store so each spawned task can own its own connection
1103        let store = indexer.store().clone();
1104
1105        // Spawn pruning [2, 3)
1106        let store1 = store.clone();
1107        handles.push(tokio::spawn(async move {
1108            let mut conn = store1.connect().await.unwrap();
1109            CoinBalanceBuckets.prune(2, 3, &mut conn).await
1110        }));
1111
1112        // Spawn pruning [1, 2)
1113        let store2 = store.clone();
1114        handles.push(tokio::spawn(async move {
1115            let mut conn = store2.connect().await.unwrap();
1116            CoinBalanceBuckets.prune(1, 2, &mut conn).await
1117        }));
1118
1119        // Spawn pruning [0, 1)
1120        let store3 = store.clone();
1121        handles.push(tokio::spawn(async move {
1122            let mut conn = store3.connect().await.unwrap();
1123            CoinBalanceBuckets.prune(0, 1, &mut conn).await
1124        }));
1125
1126        // Wait for all pruning operations to complete
1127        let results: Vec<Result<usize, anyhow::Error>> = futures::future::join_all(handles)
1128            .await
1129            .into_iter()
1130            .map(|r| r.unwrap())
1131            .collect();
1132
1133        // Verify all pruning operations succeeded
1134        for result in &results {
1135            assert!(result.is_ok(), "Pruning operation failed: {:?}", result);
1136        }
1137
1138        // Verify final state is consistent
1139        let final_balance_buckets = get_all_balance_buckets(&mut conn).await;
1140        let final_deletion_references = coin_balance_buckets_deletion_reference::table
1141            .load::<StoredCoinBalanceBucketDeletionReference>(&mut conn)
1142            .await
1143            .unwrap();
1144
1145        // After all pruning, we should have only the latest versions (cp_sequence_number = 2)
1146        assert_eq!(final_balance_buckets.len(), 3);
1147        for bucket in &final_balance_buckets {
1148            assert_eq!(bucket.cp_sequence_number, 2);
1149        }
1150
1151        // All deletion references should be cleaned up
1152        assert_eq!(final_deletion_references.len(), 0);
1153
1154        // Verify the total number of pruned rows matches expectations
1155        let total_pruned: usize = results.into_iter().map(|r| r.unwrap()).sum();
1156        assert_eq!(total_pruned, 12);
1157        for bucket in &final_balance_buckets {
1158            assert_eq!(bucket.cp_sequence_number, 2);
1159        }
1160    }
