18pub(crate) struct CpDigests;
19
20#[async_trait]
21impl Processor for CpDigests {
22    const NAME: &'static str = "cp_digests";
23
24    type Value = StoredCpDigest;
25
26    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> Result<Vec<Self::Value>> {
27        let cp_sequence_number = checkpoint.summary.sequence_number as i64;
28        let cp_digest = checkpoint.summary.digest().inner().to_vec();
29        Ok(vec![StoredCpDigest {
30            cp_sequence_number,
31            cp_digest,
32        }])
33    }
34}
35
36#[async_trait]
37impl Handler for CpDigests {
38    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
39        Ok(diesel::insert_into(cp_digests::table)
40            .values(values)
41            .on_conflict_do_nothing()
42            .execute(conn)
43            .await?)
44    }
45
46    async fn prune<'a>(
47        &self,
48        from: u64,
49        to_exclusive: u64,
50        conn: &mut Connection<'a>,
51    ) -> Result<usize> {
52        let filter = cp_digests::table
53            .filter(cp_digests::cp_sequence_number.between(from as i64, to_exclusive as i64 - 1));
54
55        Ok(diesel::delete(filter).execute(conn).await?)
56    }
57}

59#[cfg(test)]
60mod tests {
61    use diesel_async::RunQueryDsl;
62    use sui_indexer_alt_framework::Indexer;
63    use sui_indexer_alt_framework::types::test_checkpoint_data_builder::TestCheckpointBuilder;
64    use sui_indexer_alt_schema::MIGRATIONS;
65
66    use super::*;
67
68    async fn get_all_cp_digests(conn: &mut Connection<'_>) -> Result<Vec<StoredCpDigest>> {
69        let query = cp_digests::table.load(conn).await?;
70        Ok(query)
71    }
72
73    /// Verify that the processor extracts the (sequence_number, digest) pair for each checkpoint
74    /// and that the pruner removes entries by sequence-number range.
75    #[tokio::test]
76    async fn test_cp_digests_pruning() {
77        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
78        let mut conn = indexer.store().connect().await.unwrap();
79
80        // Build and commit 3 checkpoints so we have three rows with distinct digests.
81        let mut builder = TestCheckpointBuilder::new(0);
82        for _ in 0..3 {
83            builder = builder.start_transaction(0).finish_transaction();
84            let checkpoint = Arc::new(builder.build_checkpoint());
85            let values = CpDigests.process(&checkpoint).await.unwrap();
86            CpDigests::commit(&values, &mut conn).await.unwrap();
87        }
88
89        // Prune checkpoints from `[0, 2)`.
90        let rows_pruned = CpDigests.prune(0, 2, &mut conn).await.unwrap();
91        assert_eq!(rows_pruned, 2);
92
93        // Only the entry for checkpoint 2 should remain.
94        let remaining = get_all_cp_digests(&mut conn).await.unwrap();
95        assert_eq!(remaining.len(), 1);
96        assert_eq!(remaining[0].cp_sequence_number, 2);
97    }
