26#[async_trait]
27impl Processor for KvEpochEnds {
28    const NAME: &'static str = "kv_epoch_ends";
29
30    type Value = StoredEpochEnd;
31
32    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> Result<Vec<Self::Value>> {
33        let CheckpointData {
34            checkpoint_summary,
35            transactions,
36            ..
37        } = checkpoint.as_ref();
38
39        let Some(end_of_epoch) = checkpoint_summary.end_of_epoch_data.as_ref() else {
40            return Ok(vec![]);
41        };
42
43        let Some(transaction) = transactions.iter().find(|tx| {
44            matches!(
45                tx.transaction.intent_message().value.kind(),
46                TransactionKind::ChangeEpoch(_) | TransactionKind::EndOfEpochTransaction(_)
47            )
48        }) else {
49            bail!(
50                "Failed to get end of epoch transaction in checkpoint {} with EndOfEpochData",
51                checkpoint_summary.sequence_number,
52            );
53        };
54
55        if let Some(SystemEpochInfoEvent {
56            total_stake,
57            storage_fund_reinvestment,
58            storage_charge,
59            storage_rebate,
60            storage_fund_balance,
61            stake_subsidy_amount,
62            total_gas_fees,
63            total_stake_rewards_distributed,
64            leftover_storage_fund_inflow,
65            ..
66        }) = transaction
67            .events
68            .iter()
69            .flat_map(|events| &events.data)
70            .find_map(|event| {
71                event
72                    .is_system_epoch_info_event()
73                    .then(|| bcs::from_bytes(&event.contents))
74            })
75            .transpose()
76            .context("Failed to deserialize SystemEpochInfoEvent")?
77        {
78            Ok(vec![StoredEpochEnd {
79                epoch: checkpoint_summary.epoch as i64,
80                cp_hi: checkpoint_summary.sequence_number as i64 + 1,
81                tx_hi: checkpoint_summary.network_total_transactions as i64,
82                end_timestamp_ms: checkpoint_summary.timestamp_ms as i64,
83
84                safe_mode: false,
85
86                total_stake: Some(total_stake as i64),
87                storage_fund_balance: Some(storage_fund_balance as i64),
88                storage_fund_reinvestment: Some(storage_fund_reinvestment as i64),
89                storage_charge: Some(storage_charge as i64),
90                storage_rebate: Some(storage_rebate as i64),
91                stake_subsidy_amount: Some(stake_subsidy_amount as i64),
92                total_gas_fees: Some(total_gas_fees as i64),
93                total_stake_rewards_distributed: Some(total_stake_rewards_distributed as i64),
94                leftover_storage_fund_inflow: Some(leftover_storage_fund_inflow as i64),
95
96                epoch_commitments: bcs::to_bytes(&end_of_epoch.epoch_commitments)
97                    .context("Failed to serialize EpochCommitment-s")?,
98            }])
99        } else {
100            Ok(vec![StoredEpochEnd {
101                epoch: checkpoint_summary.epoch as i64,
102                cp_hi: checkpoint_summary.sequence_number as i64 + 1,
103                tx_hi: checkpoint_summary.network_total_transactions as i64,
104                end_timestamp_ms: checkpoint_summary.timestamp_ms as i64,
105
106                safe_mode: true,
107
108                total_stake: None,
109                storage_fund_balance: None,
110                storage_fund_reinvestment: None,
111                storage_charge: None,
112                storage_rebate: None,
113                stake_subsidy_amount: None,
114                total_gas_fees: None,
115                total_stake_rewards_distributed: None,
116                leftover_storage_fund_inflow: None,
117
118                epoch_commitments: bcs::to_bytes(&end_of_epoch.epoch_commitments)
119                    .context("Failed to serialize EpochCommitment-s")?,
120            }])
121        }
122    }
123}

125#[async_trait]
126impl Handler for KvEpochEnds {
127    type Store = Db;
128
129    const MIN_EAGER_ROWS: usize = 1;
130
131    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
132        Ok(diesel::insert_into(kv_epoch_ends::table)
133            .values(values)
134            .on_conflict_do_nothing()
135            .execute(conn)
136            .await?)
137    }
138
139    async fn prune<'a>(
140        &self,
141        from: u64,
142        to_exclusive: u64,
143        conn: &mut Connection<'a>,
144    ) -> Result<usize> {
145        let Range {
146            start: from_epoch,
147            end: to_epoch,
148        } = epoch_interval(conn, from..to_exclusive).await?;
149        if from_epoch < to_epoch {
150            let filter = kv_epoch_ends::table
151                .filter(kv_epoch_ends::epoch.between(from_epoch as i64, to_epoch as i64 - 1));
152            Ok(diesel::delete(filter).execute(conn).await?)
153        } else {
154            Ok(0)
155        }
156    }
157}

171    async fn get_all_kv_epoch_ends(conn: &mut Connection<'_>) -> Result<Vec<StoredEpochEnd>> {
172        let result = kv_epoch_ends::table
173            .order_by(kv_epoch_ends::epoch.asc())
174            .load(conn)
175            .await?;
176        Ok(result)
177    }
178
179    async fn get_epoch_num_of_all_kv_epoch_ends(conn: &mut Connection<'_>) -> Result<Vec<i64>> {
180        let epochs = get_all_kv_epoch_ends(conn).await?;
181        Ok(epochs.iter().map(|e| e.epoch).collect())
182    }

184    #[tokio::test]
185    pub async fn test_kv_epoch_ends_safe_mode() {
186        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
187        let mut conn = indexer.store().connect().await.unwrap();
188
189        let mut builder = TestCheckpointDataBuilder::new(0);
190        let checkpoint = Arc::new(builder.advance_epoch(AdvanceEpochConfig {
191            safe_mode: true,
192            ..Default::default()
193        }));
194        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
195        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
196
197        let epochs = get_all_kv_epoch_ends(&mut conn).await.unwrap();
198        assert_eq!(epochs.len(), 1);
199        assert!(epochs[0].safe_mode);
200        assert_eq!(epochs[0].total_gas_fees, None);
201
202        let checkpoint = Arc::new(builder.advance_epoch(Default::default()));
203        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
204        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
205
206        let epochs = get_all_kv_epoch_ends(&mut conn).await.unwrap();
207        assert_eq!(epochs.len(), 2);
208        assert!(!epochs[1].safe_mode);
209        assert_eq!(epochs[1].total_gas_fees, Some(0));
210    }

212    #[tokio::test]
213    pub async fn test_kv_epoch_ends_same_epoch() {
214        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
215        let mut conn = indexer.store().connect().await.unwrap();
216
217        // Test that there is nothing to commit while we haven't reached epoch end.
218        let mut builder = TestCheckpointDataBuilder::new(0);
219        let checkpoint = Arc::new(builder.build_checkpoint());
220        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
221        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
222        assert_eq!(values.len(), 0);
223        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
224        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
225
226        let checkpoint = Arc::new(builder.build_checkpoint());
227        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
228        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
229        assert_eq!(values.len(), 0);
230        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
231        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
232
233        // When the advance epoch tx is detected, there should be an entry to commit.
234        let checkpoint = Arc::new(builder.advance_epoch(Default::default()));
235        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
236        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
237        assert_eq!(values.len(), 1);
238        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
239        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
240
241        // Afterwards, kv_epoch_ends should not have anything to commit until the next advance epoch
242        // tx.
243        let checkpoint = Arc::new(builder.build_checkpoint());
244        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
245        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
246        assert_eq!(values.len(), 0);
247        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
248        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
249
250        let checkpoint = Arc::new(builder.build_checkpoint());
251        let values = KvEpochEnds.process(&checkpoint).await.unwrap();
252        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
253        assert_eq!(values.len(), 0);
254        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
255        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
256
257        let epochs = get_epoch_num_of_all_kv_epoch_ends(&mut conn).await.unwrap();
258        assert_eq!(epochs, vec![0]);
259
260        let rows_pruned = KvEpochEnds.prune(0, 4, &mut conn).await.unwrap();
261        let epochs = get_epoch_num_of_all_kv_epoch_ends(&mut conn).await.unwrap();
262        assert_eq!(epochs.len(), 0);
263        assert_eq!(rows_pruned, 1);
264    }

281        KvEpochEnds::commit(&values, &mut conn).await.unwrap();
