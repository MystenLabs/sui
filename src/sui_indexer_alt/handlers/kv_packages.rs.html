19pub(crate) struct KvPackages;
20
21#[async_trait]
22impl Processor for KvPackages {
23    const NAME: &'static str = "kv_packages";
24
25    type Value = StoredPackage;
26
27    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> Result<Vec<Self::Value>> {
28        let Checkpoint {
29            summary,
30
31            transactions,
32            ..
33        } = checkpoint.as_ref();
34
35        let cp_sequence_number = summary.sequence_number as i64;
36        let mut values = vec![];
37        for tx in transactions {
38            for obj in tx.output_objects(&checkpoint.object_set) {
39                let Some(package) = obj.data.try_as_package() else {
40                    continue;
41                };
42
43                values.push(StoredPackage {
44                    package_id: obj.id().to_vec(),
45                    original_id: package.original_package_id().to_vec(),
46                    package_version: obj.version().value() as i64,
47                    is_system_package: tx.transaction.sender() == SuiAddress::ZERO,
48                    serialized_object: bcs::to_bytes(obj).with_context(|| {
49                        format!("Error serializing package object {}", obj.id())
50                    })?,
51                    cp_sequence_number,
52                });
53            }
54        }
55
56        Ok(values)
57    }
58}
59
60#[async_trait]
61impl Handler for KvPackages {
62    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
63        Ok(diesel::insert_into(kv_packages::table)
64            .values(values)
65            .on_conflict_do_nothing()
66            .execute(conn)
67            .await?)
68    }
69}
