17pub(crate) struct KvTransactions;
18
19#[async_trait]
20impl Processor for KvTransactions {
21    const NAME: &'static str = "kv_transactions";
22
23    type Value = StoredTransaction;
24
25    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> Result<Vec<Self::Value>> {
26        let CheckpointData {
27            transactions,
28            checkpoint_summary,
29            ..
30        } = checkpoint.as_ref();
31
32        let cp_sequence_number = checkpoint_summary.sequence_number as i64;
33
34        let mut values = Vec::with_capacity(transactions.len());
35        for (i, tx) in transactions.iter().enumerate() {
36            let tx_digest = tx.transaction.digest();
37            let transaction = &tx.transaction.data().transaction_data();
38            let signatures = &tx.transaction.data().tx_signatures();
39
40            let effects = &tx.effects;
41            let events: Vec<_> = tx.events.iter().flat_map(|e| e.data.iter()).collect();
42
43            values.push(StoredTransaction {
44                tx_digest: tx_digest.inner().into(),
45                cp_sequence_number,
46                timestamp_ms: checkpoint_summary.timestamp_ms as i64,
47                raw_transaction: bcs::to_bytes(transaction).with_context(|| {
48                    format!("Serializing transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
49                })?,
50                raw_effects: bcs::to_bytes(effects).with_context(|| {
51                    format!("Serializing effects for transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
52                })?,
53                events: bcs::to_bytes(&events).with_context(|| {
54                    format!("Serializing events for transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
55                })?,
56                user_signatures: bcs::to_bytes(signatures).with_context(|| {
57                    format!("Serializing signatures for transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
58                })?,
59            });
60        }
61
62        Ok(values)
63    }
64}
65
66#[async_trait]
67impl Handler for KvTransactions {
68    type Store = Db;
69
70    const MIN_EAGER_ROWS: usize = 100;
71    const MAX_PENDING_ROWS: usize = 10000;
72
73    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
74        Ok(diesel::insert_into(kv_transactions::table)
75            .values(values)
76            .on_conflict_do_nothing()
77            .execute(conn)
78            .await?)
79    }
80
81    async fn prune<'a>(
82        &self,
83        from: u64,
84        to_exclusive: u64,
85        conn: &mut Connection<'a>,
86    ) -> Result<usize> {
87        let filter = kv_transactions::table.filter(
88            kv_transactions::cp_sequence_number.between(from as i64, to_exclusive as i64 - 1),
89        );
90
91        Ok(diesel::delete(filter).execute(conn).await?)
92    }
93}

95#[cfg(test)]
96mod tests {
97    use super::*;
98    use diesel_async::RunQueryDsl;
99    use sui_indexer_alt_framework::{
100        Indexer, types::test_checkpoint_data_builder::TestCheckpointDataBuilder,
101    };
102    use sui_indexer_alt_schema::MIGRATIONS;
103
104    async fn get_all_kv_transactions(conn: &mut Connection<'_>) -> Result<Vec<StoredTransaction>> {
105        Ok(kv_transactions::table.load(conn).await?)
106    }
107
108    /// The kv_checkpoints pruner does not require cp_sequence_numbers, it can prune directly with the
109    /// checkpoint sequence number range.
110    #[tokio::test]
111    async fn test_kv_transactions_pruning() {
112        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
113        let mut conn = indexer.store().connect().await.unwrap();
114
115        let mut builder = TestCheckpointDataBuilder::new(0);
116        builder = builder.start_transaction(0).finish_transaction();
117        let checkpoint = Arc::new(builder.build_checkpoint());
118        let values = KvTransactions.process(&checkpoint).await.unwrap();
119        KvTransactions::commit(&values, &mut conn).await.unwrap();
120
121        builder = builder.start_transaction(0).finish_transaction();
122        builder = builder.start_transaction(1).finish_transaction();
123        let checkpoint = Arc::new(builder.build_checkpoint());
124        let values = KvTransactions.process(&checkpoint).await.unwrap();
125        KvTransactions::commit(&values, &mut conn).await.unwrap();
126
127        builder = builder.start_transaction(0).finish_transaction();
128        builder = builder.start_transaction(1).finish_transaction();
129        builder = builder.start_transaction(2).finish_transaction();
130        builder = builder.start_transaction(3).finish_transaction();
131        let checkpoint = Arc::new(builder.build_checkpoint());
132        let values = KvTransactions.process(&checkpoint).await.unwrap();
133        KvTransactions::commit(&values, &mut conn).await.unwrap();
134
135        let transactions = get_all_kv_transactions(&mut conn).await.unwrap();
136        assert_eq!(transactions.len(), 7);
137
138        // Prune checkpoints from `[0, 2)`
139        let rows_pruned = KvTransactions.prune(0, 2, &mut conn).await.unwrap();
140        assert_eq!(rows_pruned, 3);
141
142        let remaining_transactions = get_all_kv_transactions(&mut conn).await.unwrap();
143        assert_eq!(remaining_transactions.len(), 4);
144    }
