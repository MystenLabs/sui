21#[async_trait]
22impl Processor for KvTransactions {
23    const NAME: &'static str = "kv_transactions";
24
25    type Value = StoredTransaction;
26
27    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> Result<Vec<Self::Value>> {
28        let Checkpoint {
29            transactions,
30            summary,
31            ..
32        } = checkpoint.as_ref();
33
34        let cp_sequence_number = summary.sequence_number as i64;
35
36        let mut values = Vec::with_capacity(transactions.len());
37        for (i, tx) in transactions.iter().enumerate() {
38            let tx_digest = tx.transaction.digest();
39            let transaction = &tx.transaction;
40            let signatures = &tx.signatures;
41
42            let effects = &tx.effects;
43            let events: Vec<_> = tx.events.iter().flat_map(|e| e.data.iter()).collect();
44
45            values.push(StoredTransaction {
46                tx_digest: tx_digest.inner().into(),
47                cp_sequence_number,
48                timestamp_ms: summary.timestamp_ms as i64,
49                raw_transaction: bcs::to_bytes(transaction).with_context(|| {
50                    format!("Serializing transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
51                })?,
52                raw_effects: bcs::to_bytes(effects).with_context(|| {
53                    format!("Serializing effects for transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
54                })?,
55                events: bcs::to_bytes(&events).with_context(|| {
56                    format!("Serializing events for transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
57                })?,
58                user_signatures: bcs::to_bytes(signatures).with_context(|| {
59                    format!("Serializing signatures for transaction {tx_digest} (cp {cp_sequence_number}, tx {i})")
60                })?,
61            });
62        }
63
64        Ok(values)
65    }
66}

68#[async_trait]
69impl Handler for KvTransactions {
70    const MIN_EAGER_ROWS: usize = 100;
71    const MAX_PENDING_ROWS: usize = 10000;
72
73    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
74        Ok(diesel::insert_into(kv_transactions::table)
75            .values(values)
76            .on_conflict_do_nothing()
77            .execute(conn)
78            .await?)
79    }
80
81    async fn prune<'a>(
82        &self,
83        from: u64,
84        to_exclusive: u64,
85        conn: &mut Connection<'a>,
86    ) -> Result<usize> {
87        let filter = kv_transactions::table.filter(
88            kv_transactions::cp_sequence_number.between(from as i64, to_exclusive as i64 - 1),
89        );
90
91        Ok(diesel::delete(filter).execute(conn).await?)
92    }
93}

104    async fn get_all_kv_transactions(conn: &mut Connection<'_>) -> Result<Vec<StoredTransaction>> {
105        Ok(kv_transactions::table.load(conn).await?)
106    }
107
108    /// The kv_checkpoints pruner does not require cp_sequence_numbers, it can prune directly with the
109    /// checkpoint sequence number range.
110    #[tokio::test]
111    async fn test_kv_transactions_pruning() {
112        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
113        let mut conn = indexer.store().connect().await.unwrap();
114
115        let mut builder = TestCheckpointBuilder::new(0);
116        builder = builder.start_transaction(0).finish_transaction();
117        let checkpoint = Arc::new(builder.build_checkpoint());
118        let values = KvTransactions.process(&checkpoint).await.unwrap();
119
120        KvTransactions::commit(&values, &mut conn).await.unwrap();
121
122        builder = builder.start_transaction(0).finish_transaction();
123        builder = builder.start_transaction(1).finish_transaction();
124        let checkpoint = Arc::new(builder.build_checkpoint());
125        let values = KvTransactions.process(&checkpoint).await.unwrap();
126
127        KvTransactions::commit(&values, &mut conn).await.unwrap();
128
129        builder = builder.start_transaction(0).finish_transaction();
130        builder = builder.start_transaction(1).finish_transaction();
131        builder = builder.start_transaction(2).finish_transaction();
132        builder = builder.start_transaction(3).finish_transaction();
133        let checkpoint = Arc::new(builder.build_checkpoint());
134        let values = KvTransactions.process(&checkpoint).await.unwrap();
135
136        KvTransactions::commit(&values, &mut conn).await.unwrap();
137
138        let transactions = get_all_kv_transactions(&mut conn).await.unwrap();
139        assert_eq!(transactions.len(), 7);
140
141        // Prune checkpoints from `[0, 2)`
142        let rows_pruned = KvTransactions.prune(0, 2, &mut conn).await.unwrap();
143        assert_eq!(rows_pruned, 3);
144
145        let remaining_transactions = get_all_kv_transactions(&mut conn).await.unwrap();
146        assert_eq!(remaining_transactions.len(), 4);
147    }
