28/// Enum to encapsulate different types of updates to be written to the main `obj_info` and
29/// reference tables.
30pub(crate) enum ProcessedObjInfoUpdate {
31    Upsert {
32        object: Object,
33        /// Indicates whether the object was created/unwrapped in this checkpoint.
34        created: bool,
35    },
36    /// Represents object wrap or deletion. An entry is created on the main table, and two entries
37    /// are added to the reference table, one to delete the previous version, and one to delete the
38    /// sentinel row itself.
39    Delete(ObjectID),
40}
41
42pub(crate) struct ProcessedObjInfo {
43    pub cp_sequence_number: u64,
44    pub update: ProcessedObjInfoUpdate,
45}
46
47#[async_trait]
48impl Processor for ObjInfo {
49    const NAME: &'static str = "obj_info";
50    type Value = ProcessedObjInfo;
51
52    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> Result<Vec<Self::Value>> {
53        let cp_sequence_number = checkpoint.summary.sequence_number;
54        let checkpoint_input_objects = checkpoint_input_objects(checkpoint)?;
55        let latest_live_output_objects = checkpoint.latest_live_output_objects();
56
57        let mut values: BTreeMap<ObjectID, Self::Value> = BTreeMap::new();
58        for object_id in checkpoint_input_objects.keys() {
59            if !latest_live_output_objects.contains_key(object_id) {
60                // If an input object is not in the latest live output objects, it must have been
61                // deleted or wrapped in this checkpoint. We keep an entry for it in the table. This
62                // is necessary when we query objects and iterating over them, so that we don't
63                // include the object in the result if it was deleted.
64                values.insert(
65                    *object_id,
66                    ProcessedObjInfo {
67                        cp_sequence_number,
68                        update: ProcessedObjInfoUpdate::Delete(*object_id),
69                    },
70                );
71            }
72        }
73        for (object_id, object) in latest_live_output_objects.iter() {
74            // If an object is newly created/unwrapped in this checkpoint, or if the owner changed,
75            // we need to insert an entry for it in the table.
76            let should_insert = match checkpoint_input_objects.get(object_id) {
77                Some(input_object) => input_object.owner() != object.owner(),
78                None => true,
79            };
80            if should_insert {
81                // We make note of whether the obj was mutated or created to help with pruning.
82                let created = !checkpoint_input_objects.contains_key(object_id);
83                values.insert(
84                    *object_id,
85                    ProcessedObjInfo {
86                        cp_sequence_number,
87                        update: ProcessedObjInfoUpdate::Upsert {
88                            object: object.clone(),
89                            created,
90                        },
91                    },
92                );
93            }
94        }
95
96        Ok(values.into_values().collect())
97    }
98}

100#[async_trait]
101impl Handler for ObjInfo {
102    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
103        let stored = values
104            .iter()
105            .map(|v| v.try_into())
106            .collect::<Result<Vec<StoredObjInfo>>>()?;
107
108        // Entries to commit to the reference table for pruning.
109        let mut references = Vec::new();
110        for value in values {
111            match &value.update {
112                ProcessedObjInfoUpdate::Upsert { object, created } => {
113                    // Created objects don't have a previous entry in the main table. Unwrapped
114                    // objects must have been previously wrapped, and the deletion record for that
115                    // wrap will have handled itself. Thus we don't emit an entry to the reference
116                    // table.
117                    if !created {
118                        references.push(StoredObjInfoDeletionReference {
119                            object_id: object.id().to_vec(),
120                            cp_sequence_number: value.cp_sequence_number as i64,
121                        });
122                    }
123                }
124                // Store record of current version to delete previous version, and another to delete
125                // itself. When pruning, the deletion record will not be pruned in the
126                // `value.cp_sequence_number` checkpoint, but the next one.
127                ProcessedObjInfoUpdate::Delete(object_id) => {
128                    references.push(StoredObjInfoDeletionReference {
129                        object_id: object_id.to_vec(),
130                        cp_sequence_number: value.cp_sequence_number as i64,
131                    });
132                    references.push(StoredObjInfoDeletionReference {
133                        object_id: object_id.to_vec(),
134                        cp_sequence_number: value.cp_sequence_number as i64 + 1,
135                    });
136                }
137            }
138        }
139
140        let count = diesel::insert_into(obj_info::table)
141            .values(&stored)
142            .on_conflict_do_nothing()
143            .execute(conn)
144            .await?;
145
146        let deleted_refs = if !references.is_empty() {
147            diesel::insert_into(obj_info_deletion_reference::table)
148                .values(&references)
149                .on_conflict_do_nothing()
150                .execute(conn)
151                .await?
152        } else {
153            0
154        };
155
156        Ok(count + deleted_refs)
157    }
158
159    /// To prune `obj_info`, entries between `[from, to_exclusive)` are read from the reference
160    /// table, and the previous versions of the objects are deleted from the main table. Finally,
161    /// the reference entries themselves are deleted. The framework guarantees that `to_exclusive <=
162    /// checkpoint_hi_inclusive - retention`, so we only prune data that has been committed but is
163    /// beyond the retention window.
164    async fn prune<'a>(
165        &self,
166        from: u64,
167        to_exclusive: u64,
168        conn: &mut Connection<'a>,
169    ) -> Result<usize> {
170        // This query first deletes from obj_info_deletion_reference and computes predecessors, then
171        // deletes from obj_info using the precomputed predecessor information. The inline compute
172        // avoids HashAggregate operations and the ensuing materialization overhead.
173        //
174        // This works best on under 1.5 million object changes, roughly 15k checkpoints. Performance
175        // degrades sharply beyond this, since the planner switches to hash joins and full table
176        // scans. A HashAggregate approach interestingly becomes more performant in this scenario.
177        //
178        // If the first call to prune succeeds, subsequent calls will find no records to delete from
179        // obj_info_deletion_reference, and consequently no records to delete from the main table.
180        // Pruning is thus idempotent after the initial run.
181        //
182        // TODO: use sui_sql_macro's query!
183        let query = format!(
184            "
185            WITH deletion_refs AS (
186                DELETE FROM
187                    obj_info_deletion_reference dr
188                WHERE
189                    {} <= cp_sequence_number AND cp_sequence_number < {}
190                RETURNING
191                    object_id, (
192                    SELECT
193                        oi.cp_sequence_number
194                    FROM
195                        obj_info oi
196                    WHERE
197                        dr.object_id = oi.object_id
198                    AND oi.cp_sequence_number < dr.cp_sequence_number
199                    ORDER BY
200                        oi.cp_sequence_number DESC
201                    LIMIT
202                        1
203                    ) AS cp_sequence_number
204            ),
205            deleted_objects AS (
206                DELETE FROM
207                    obj_info oi
208                USING
209                    deletion_refs dr
210                WHERE
211                    oi.object_id = dr.object_id
212                AND oi.cp_sequence_number = dr.cp_sequence_number
213                RETURNING
214                    oi.object_id
215            )
216            SELECT
217                (SELECT COUNT(*) FROM deleted_objects) AS deleted_objects,
218                (SELECT COUNT(*) FROM deletion_refs) AS deleted_refs
219            ",
220            from, to_exclusive
221        );
222
223        #[derive(QueryableByName)]
224        struct CountResult {
225            #[diesel(sql_type = diesel::sql_types::BigInt)]
226            deleted_objects: i64,
227            #[diesel(sql_type = diesel::sql_types::BigInt)]
228            deleted_refs: i64,
229        }
230
231        let CountResult {
232            deleted_objects,
233            deleted_refs,
234        } = diesel::sql_query(query)
235            .get_result::<CountResult>(conn)
236            .await?;
237
238        ensure!(
239            deleted_objects == deleted_refs,
240            "Deleted objects count ({deleted_objects}) does not match deleted refs count ({deleted_refs})",
241        );
242
243        Ok((deleted_objects + deleted_refs) as usize)
244    }
245}

251impl TryInto<StoredObjInfo> for &ProcessedObjInfo {
252    type Error = anyhow::Error;
253
254    fn try_into(self) -> Result<StoredObjInfo> {
255        match &self.update {
256            ProcessedObjInfoUpdate::Upsert { object, .. } => {
257                StoredObjInfo::from_object(object, self.cp_sequence_number as i64)
258            }
259            ProcessedObjInfoUpdate::Delete(object_id) => Ok(StoredObjInfo {
260                object_id: object_id.to_vec(),
261                cp_sequence_number: self.cp_sequence_number as i64,
262                owner_kind: None,
263                owner_id: None,
264                package: None,
265                module: None,
266                name: None,
267                instantiation: None,
268            }),
269        }
270    }
271}

847    /// In our processing logic, we consider objects that appear as input to the checkpoint but not
848    /// in the output as wrapped or deleted. This emits a tombstone row. Meanwhile, the remote store
849    /// containing `Checkpoint` used to include unchanged consensus objects in the `input_objects`
850    /// of a `CheckpointTransaction`. Because these read-only consensus objects were not modified, they
851    /// were not included in `output_objects`. But that means within our pipeline, these object
852    /// states were incorrectly treated as deleted, and thus every transaction read emitted a
853    /// tombstone row. This test validates that unless an object appears as an input object from
854    /// `tx.effects.object_changes`, we do not consider it within our pipeline.

962    /// Test concurrent pruning operations to ensure thread safety and data consistency.
963    /// This test creates the same scenario as test_process_out_of_order_pruning but runs
964    /// multiple pruning operations concurrently.
965    #[tokio::test]
966    async fn test_process_concurrent_pruning() {
967        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
968        let mut conn = indexer.store().connect().await.unwrap();
969        let mut builder = TestCheckpointBuilder::new(0);
970
971        // Create the same scenario as the out-of-order test
972        builder = builder
973            .start_transaction(0)
974            .create_owned_object(0)
975            .create_owned_object(1)
976            .create_owned_object(2)
977            .finish_transaction();
978        let checkpoint0 = builder.build_checkpoint();
979        let result = ObjInfo.process(&Arc::new(checkpoint0)).await.unwrap();
980        ObjInfo::commit(&result, &mut conn).await.unwrap();
981
982        builder = builder
983            .start_transaction(0)
984            .transfer_object(0, 1)
985            .transfer_object(1, 1)
986            .transfer_object(2, 1)
987            .finish_transaction();
988        let checkpoint1 = builder.build_checkpoint();
989        let result = ObjInfo.process(&Arc::new(checkpoint1)).await.unwrap();
990        ObjInfo::commit(&result, &mut conn).await.unwrap();
991
992        builder = builder
993            .start_transaction(1)
994            .transfer_object(0, 0)
995            .transfer_object(1, 0)
996            .transfer_object(2, 0)
997            .finish_transaction();
998        let checkpoint2 = builder.build_checkpoint();
999        let result = ObjInfo.process(&Arc::new(checkpoint2)).await.unwrap();
1000        ObjInfo::commit(&result, &mut conn).await.unwrap();
1001
1002        // Verify initial state
1003        let all_obj_info = get_all_obj_info(&mut conn).await.unwrap();
1004        assert_eq!(all_obj_info.len(), 9); // 3 objects  3 checkpoints
1005        let all_obj_info_deletion_references = get_all_obj_info_deletion_references(&mut conn)
1006            .await
1007            .unwrap();
1008        assert_eq!(all_obj_info_deletion_references.len(), 6);
1009
1010        // Run concurrent pruning operations
1011        let mut handles = Vec::new();
1012
1013        // Clone the store so each spawned task can own its own connection
1014        let store = indexer.store().clone();
1015
1016        // Spawn pruning [2, 3)
1017        let store1 = store.clone();
1018        handles.push(tokio::spawn(async move {
1019            let mut conn = store1.connect().await.unwrap();
1020            ObjInfo.prune(2, 3, &mut conn).await
1021        }));
1022
1023        // Spawn pruning [1, 2)
1024        let store2 = store.clone();
1025        handles.push(tokio::spawn(async move {
1026            let mut conn = store2.connect().await.unwrap();
1027            ObjInfo.prune(1, 2, &mut conn).await
1028        }));
1029
1030        // Spawn pruning [0, 1)
1031        let store3 = store.clone();
1032        handles.push(tokio::spawn(async move {
1033            let mut conn = store3.connect().await.unwrap();
1034            ObjInfo.prune(0, 1, &mut conn).await
1035        }));
1036
1037        // Wait for all pruning operations to complete
1038        let results: Vec<Result<usize, anyhow::Error>> = futures::future::join_all(handles)
1039            .await
1040            .into_iter()
1041            .map(|r| r.unwrap())
1042            .collect();
1043
1044        // Verify all pruning operations succeeded
1045        for result in &results {
1046            assert!(result.is_ok(), "Pruning operation failed: {:?}", result);
1047        }
1048
1049        // Verify final state is consistent
1050        let final_obj_info = get_all_obj_info(&mut conn).await.unwrap();
1051        let final_deletion_references = get_all_obj_info_deletion_references(&mut conn)
1052            .await
1053            .unwrap();
1054
1055        // After all pruning, we should have only the latest versions (cp_sequence_number = 2)
1056        assert_eq!(final_obj_info.len(), 3);
1057        for object in &final_obj_info {
1058            assert_eq!(object.cp_sequence_number, 2);
1059        }
1060
1061        // All deletion references should be cleaned up
1062        assert_eq!(final_deletion_references.len(), 0);
1063
1064        // Verify the total number of pruned rows matches expectations
1065        let total_pruned: usize = results.into_iter().map(|r| r.unwrap()).sum();
1066        assert_eq!(total_pruned, 12);
1067        for object in &final_obj_info {
1068            assert_eq!(object.cp_sequence_number, 2);
1069        }
1070    }
