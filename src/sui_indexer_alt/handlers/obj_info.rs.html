25/// Enum to encapsulate different types of updates to be written to the main `obj_info` and
26/// reference tables.
27pub(crate) enum ProcessedObjInfoUpdate {
28    Upsert {
29        object: Object,
30        /// Indicates whether the object was created/unwrapped in this checkpoint.
31        created: bool,
32    },
33    /// Represents object wrap or deletion. An entry is created on the main table, and two entries
34    /// are added to the reference table, one to delete the previous version, and one to delete the
35    /// sentinel row itself.
36    Delete(ObjectID),
37}
38
39pub(crate) struct ProcessedObjInfo {
40    pub cp_sequence_number: u64,
41    pub update: ProcessedObjInfoUpdate,
42}
43
44#[async_trait]
45impl Processor for ObjInfo {
46    const NAME: &'static str = "obj_info";
47    type Value = ProcessedObjInfo;
48
49    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> Result<Vec<Self::Value>> {
50        let cp_sequence_number = checkpoint.checkpoint_summary.sequence_number;
51        let checkpoint_input_objects = checkpoint_input_objects(checkpoint)?;
52        let latest_live_output_objects = checkpoint
53            .latest_live_output_objects()
54            .into_iter()
55            .map(|o| (o.id(), o))
56            .collect::<BTreeMap<_, _>>();
57        let mut values: BTreeMap<ObjectID, Self::Value> = BTreeMap::new();
58        for object_id in checkpoint_input_objects.keys() {
59            if !latest_live_output_objects.contains_key(object_id) {
60                // If an input object is not in the latest live output objects, it must have been
61                // deleted or wrapped in this checkpoint. We keep an entry for it in the table. This
62                // is necessary when we query objects and iterating over them, so that we don't
63                // include the object in the result if it was deleted.
64                values.insert(
65                    *object_id,
66                    ProcessedObjInfo {
67                        cp_sequence_number,
68                        update: ProcessedObjInfoUpdate::Delete(*object_id),
69                    },
70                );
71            }
72        }
73        for (object_id, object) in latest_live_output_objects.iter() {
74            // If an object is newly created/unwrapped in this checkpoint, or if the owner changed,
75            // we need to insert an entry for it in the table.
76            let should_insert = match checkpoint_input_objects.get(object_id) {
77                Some(input_object) => input_object.owner() != object.owner(),
78                None => true,
79            };
80            if should_insert {
81                // We make note of whether the obj was mutated or created to help with pruning.
82                let created = !checkpoint_input_objects.contains_key(object_id);
83                values.insert(
84                    *object_id,
85                    ProcessedObjInfo {
86                        cp_sequence_number,
87                        update: ProcessedObjInfoUpdate::Upsert {
88                            object: (*object).clone(),
89                            created,
90                        },
91                    },
92                );
93            }
94        }
95
96        Ok(values.into_values().collect())
97    }
98}

100#[async_trait]
101impl Handler for ObjInfo {
102    type Store = Db;
103
104    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
105        let stored = values
106            .iter()
107            .map(|v| v.try_into())
108            .collect::<Result<Vec<StoredObjInfo>>>()?;
109
110        // Entries to commit to the reference table for pruning.
111        let mut references = Vec::new();
112        for value in values {
113            match &value.update {
114                ProcessedObjInfoUpdate::Upsert { object, created } => {
115                    // Created objects don't have a previous entry in the main table. Unwrapped
116                    // objects must have been previously wrapped, and the deletion record for that
117                    // wrap will have handled itself. Thus we don't emit an entry to the reference
118                    // table.
119                    if !created {
120                        references.push(StoredObjInfoDeletionReference {
121                            object_id: object.id().to_vec(),
122                            cp_sequence_number: value.cp_sequence_number as i64,
123                        });
124                    }
125                }
126                // Store record of current version to delete previous version, and another to delete
127                // itself. When pruning, the deletion record will not be pruned in the
128                // `value.cp_sequence_number` checkpoint, but the next one.
129                ProcessedObjInfoUpdate::Delete(object_id) => {
130                    references.push(StoredObjInfoDeletionReference {
131                        object_id: object_id.to_vec(),
132                        cp_sequence_number: value.cp_sequence_number as i64,
133                    });
134                    references.push(StoredObjInfoDeletionReference {
135                        object_id: object_id.to_vec(),
136                        cp_sequence_number: value.cp_sequence_number as i64 + 1,
137                    });
138                }
139            }
140        }
141
142        let count = diesel::insert_into(obj_info::table)
143            .values(&stored)
144            .on_conflict_do_nothing()
145            .execute(conn)
146            .await?;
147
148        let deleted_refs = if !references.is_empty() {
149            diesel::insert_into(obj_info_deletion_reference::table)
150                .values(&references)
151                .on_conflict_do_nothing()
152                .execute(conn)
153                .await?
154        } else {
155            0
156        };
157
158        Ok(count + deleted_refs)
159    }
160
161    /// To prune `obj_info`, entries between `[from, to_exclusive)` are read from the reference
162    /// table, and the previous versions of the objects are deleted from the main table. Finally,
163    /// the reference entries themselves are deleted. The framework guarantees that `to_exclusive <=
164    /// checkpoint_hi_inclusive - retention`, so we only prune data that has been committed but is
165    /// beyond the retention window.
166    async fn prune<'a>(
167        &self,
168        from: u64,
169        to_exclusive: u64,
170        conn: &mut Connection<'a>,
171    ) -> Result<usize> {
172        // This query first deletes from obj_info_deletion_reference and computes predecessors, then
173        // deletes from obj_info using the precomputed predecessor information. The inline compute
174        // avoids HashAggregate operations and the ensuing materialization overhead.
175        //
176        // This works best on under 1.5 million object changes, roughly 15k checkpoints. Performance
177        // degrades sharply beyond this, since the planner switches to hash joins and full table
178        // scans. A HashAggregate approach interestingly becomes more performant in this scenario.
179        //
180        // If the first call to prune succeeds, subsequent calls will find no records to delete from
181        // obj_info_deletion_reference, and consequently no records to delete from the main table.
182        // Pruning is thus idempotent after the initial run.
183        //
184        // TODO: use sui_sql_macro's query!
185        let query = format!(
186            "
187            WITH deletion_refs AS (
188                DELETE FROM
189                    obj_info_deletion_reference dr
190                WHERE
191                    {} <= cp_sequence_number AND cp_sequence_number < {}
192                RETURNING
193                    object_id, (
194                    SELECT
195                        oi.cp_sequence_number
196                    FROM
197                        obj_info oi
198                    WHERE
199                        dr.object_id = oi.object_id
200                    AND oi.cp_sequence_number < dr.cp_sequence_number
201                    ORDER BY
202                        oi.cp_sequence_number DESC
203                    LIMIT
204                        1
205                    ) AS cp_sequence_number
206            ),
207            deleted_objects AS (
208                DELETE FROM
209                    obj_info oi
210                USING
211                    deletion_refs dr
212                WHERE
213                    oi.object_id = dr.object_id
214                AND oi.cp_sequence_number = dr.cp_sequence_number
215                RETURNING
216                    oi.object_id
217            )
218            SELECT
219                (SELECT COUNT(*) FROM deleted_objects) AS deleted_objects,
220                (SELECT COUNT(*) FROM deletion_refs) AS deleted_refs
221            ",
222            from, to_exclusive
223        );
224
225        #[derive(QueryableByName)]
226        struct CountResult {
227            #[diesel(sql_type = diesel::sql_types::BigInt)]
228            deleted_objects: i64,
229            #[diesel(sql_type = diesel::sql_types::BigInt)]
230            deleted_refs: i64,
231        }
232
233        let CountResult {
234            deleted_objects,
235            deleted_refs,
236        } = diesel::sql_query(query)
237            .get_result::<CountResult>(conn)
238            .await?;
239
240        ensure!(
241            deleted_objects == deleted_refs,
242            "Deleted objects count ({deleted_objects}) does not match deleted refs count ({deleted_refs})",
243        );
244
245        Ok((deleted_objects + deleted_refs) as usize)
246    }
247}

253impl TryInto<StoredObjInfo> for &ProcessedObjInfo {
254    type Error = anyhow::Error;
255
256    fn try_into(self) -> Result<StoredObjInfo> {
257        match &self.update {
258            ProcessedObjInfoUpdate::Upsert { object, .. } => {
259                StoredObjInfo::from_object(object, self.cp_sequence_number as i64)
260            }
261            ProcessedObjInfoUpdate::Delete(object_id) => Ok(StoredObjInfo {
262                object_id: object_id.to_vec(),
263                cp_sequence_number: self.cp_sequence_number as i64,
264                owner_kind: None,
265                owner_id: None,
266                package: None,
267                module: None,
268                name: None,
269                instantiation: None,
270            }),
271        }
272    }
273}

851    /// In our processing logic, we consider objects that appear as input to the checkpoint but not
852    /// in the output as wrapped or deleted. This emits a tombstone row. Meanwhile, the remote store
853    /// containing `CheckpointData` used to include unchanged consensus objects in the `input_objects`
854    /// of a `CheckpointTransaction`. Because these read-only consensus objects were not modified, they
855    /// were not included in `output_objects`. But that means within our pipeline, these object
856    /// states were incorrectly treated as deleted, and thus every transaction read emitted a
857    /// tombstone row. This test validates that unless an object appears as an input object from
858    /// `tx.effects.object_changes`, we do not consider it within our pipeline.
859    ///
860    /// Use the checkpoint builder to create a shared object. Then, remove this from the checkpoint,
861    /// and replace it with a transaction that takes the shared object as read-only.

870

966    /// Test concurrent pruning operations to ensure thread safety and data consistency.
967    /// This test creates the same scenario as test_process_out_of_order_pruning but runs
968    /// multiple pruning operations concurrently.
969    #[tokio::test]
970    async fn test_process_concurrent_pruning() {
971        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
972        let mut conn = indexer.store().connect().await.unwrap();
973        let mut builder = TestCheckpointDataBuilder::new(0);
974
975        // Create the same scenario as the out-of-order test
976        builder = builder
977            .start_transaction(0)
978            .create_owned_object(0)
979            .create_owned_object(1)
980            .create_owned_object(2)
981            .finish_transaction();
982        let checkpoint0 = builder.build_checkpoint();
983        let result = ObjInfo.process(&Arc::new(checkpoint0)).await.unwrap();
984        ObjInfo::commit(&result, &mut conn).await.unwrap();
985
986        builder = builder
987            .start_transaction(0)
988            .transfer_object(0, 1)
989            .transfer_object(1, 1)
990            .transfer_object(2, 1)
991            .finish_transaction();
992        let checkpoint1 = builder.build_checkpoint();
993        let result = ObjInfo.process(&Arc::new(checkpoint1)).await.unwrap();
994        ObjInfo::commit(&result, &mut conn).await.unwrap();
995
996        builder = builder
997            .start_transaction(1)
998            .transfer_object(0, 0)
999            .transfer_object(1, 0)
1000            .transfer_object(2, 0)
1001            .finish_transaction();
1002        let checkpoint2 = builder.build_checkpoint();
1003        let result = ObjInfo.process(&Arc::new(checkpoint2)).await.unwrap();
1004        ObjInfo::commit(&result, &mut conn).await.unwrap();
1005
1006        // Verify initial state
1007        let all_obj_info = get_all_obj_info(&mut conn).await.unwrap();
1008        assert_eq!(all_obj_info.len(), 9); // 3 objects × 3 checkpoints
1009        let all_obj_info_deletion_references = get_all_obj_info_deletion_references(&mut conn)
1010            .await
1011            .unwrap();
1012        assert_eq!(all_obj_info_deletion_references.len(), 6);
1013
1014        // Run concurrent pruning operations
1015        let mut handles = Vec::new();
1016
1017        // Clone the store so each spawned task can own its own connection
1018        let store = indexer.store().clone();
1019
1020        // Spawn pruning [2, 3)
1021        let store1 = store.clone();
1022        handles.push(tokio::spawn(async move {
1023            let mut conn = store1.connect().await.unwrap();
1024            ObjInfo.prune(2, 3, &mut conn).await
1025        }));
1026
1027        // Spawn pruning [1, 2)
1028        let store2 = store.clone();
1029        handles.push(tokio::spawn(async move {
1030            let mut conn = store2.connect().await.unwrap();
1031            ObjInfo.prune(1, 2, &mut conn).await
1032        }));
1033
1034        // Spawn pruning [0, 1)
1035        let store3 = store.clone();
1036        handles.push(tokio::spawn(async move {
1037            let mut conn = store3.connect().await.unwrap();
1038            ObjInfo.prune(0, 1, &mut conn).await
1039        }));
1040
1041        // Wait for all pruning operations to complete
1042        let results: Vec<Result<usize, anyhow::Error>> = futures::future::join_all(handles)
1043            .await
1044            .into_iter()
1045            .map(|r| r.unwrap())
1046            .collect();
1047
1048        // Verify all pruning operations succeeded
1049        for result in &results {
1050            assert!(result.is_ok(), "Pruning operation failed: {:?}", result);
1051        }
1052
1053        // Verify final state is consistent
1054        let final_obj_info = get_all_obj_info(&mut conn).await.unwrap();
1055        let final_deletion_references = get_all_obj_info_deletion_references(&mut conn)
1056            .await
1057            .unwrap();
1058
1059        // After all pruning, we should have only the latest versions (cp_sequence_number = 2)
1060        assert_eq!(final_obj_info.len(), 3);
1061        for object in &final_obj_info {
1062            assert_eq!(object.cp_sequence_number, 2);
1063        }
1064
1065        // All deletion references should be cleaned up
1066        assert_eq!(final_deletion_references.len(), 0);
1067
1068        // Verify the total number of pruned rows matches expectations
1069        let total_pruned: usize = results.into_iter().map(|r| r.unwrap()).sum();
1070        assert_eq!(total_pruned, 12);
1071        for object in &final_obj_info {
1072            assert_eq!(object.cp_sequence_number, 2);
1073        }
1074    }
