21pub(crate) struct TxDigests;
22
23#[async_trait]
24impl Processor for TxDigests {
25    const NAME: &'static str = "tx_digests";
26
27    type Value = StoredTxDigest;
28
29    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> Result<Vec<Self::Value>> {
30        let Checkpoint {
31            transactions,
32            summary,
33            ..
34        } = checkpoint.as_ref();
35
36        let first_tx = summary.network_total_transactions as usize - transactions.len();
37
38        Ok(transactions
39            .iter()
40            .enumerate()
41            .map(|(i, tx)| StoredTxDigest {
42                tx_sequence_number: (first_tx + i) as i64,
43                tx_digest: tx.transaction.digest().inner().to_vec(),
44            })
45            .collect())
46    }
47}
48
49#[async_trait]
50impl Handler for TxDigests {
51    const MIN_EAGER_ROWS: usize = 100;
52    const MAX_PENDING_ROWS: usize = 10000;
53
54    async fn commit<'a>(values: &[Self::Value], conn: &mut Connection<'a>) -> Result<usize> {
55        Ok(diesel::insert_into(tx_digests::table)
56            .values(values)
57            .on_conflict_do_nothing()
58            .execute(conn)
59            .await?)
60    }
61
62    async fn prune<'a>(
63        &self,
64        from: u64,
65        to_exclusive: u64,
66        conn: &mut Connection<'a>,
67    ) -> Result<usize> {
68        let Range {
69            start: from_tx,
70            end: to_tx,
71        } = tx_interval(conn, from..to_exclusive).await?;
72        let filter = tx_digests::table
73            .filter(tx_digests::tx_sequence_number.between(from_tx as i64, to_tx as i64 - 1));
74
75        Ok(diesel::delete(filter).execute(conn).await?)
76    }
77}

90    async fn get_all_tx_digests(conn: &mut Connection<'_>) -> Result<Vec<i64>> {
91        Ok(tx_digests::table
92            .select(tx_digests::tx_sequence_number)
93            .load(conn)
94            .await?)

97    #[tokio::test]
98    async fn test_tx_digests_pruning_complains_if_no_mapping() {
99        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
100        let mut conn = indexer.store().connect().await.unwrap();
101
102        let result = TxDigests.prune(0, 2, &mut conn).await;
103
104        assert!(result.is_err());
105        assert_eq!(
106            result.unwrap_err().to_string(),
107            "No checkpoint mapping found for checkpoint 0"
108        );
109    }

111    /// The kv_checkpoints pruner does not require cp_sequence_numbers, it can prune directly with the
112    /// checkpoint sequence number range.
113    #[tokio::test]
114    async fn test_tx_digests_pruning() {
115        let (indexer, _db) = Indexer::new_for_testing(&MIGRATIONS).await;
116        let mut conn = indexer.store().connect().await.unwrap();
117
118        let mut builder = TestCheckpointBuilder::new(0);
119        builder = builder.start_transaction(0).finish_transaction();
120        let checkpoint = Arc::new(builder.build_checkpoint());
121        let values = TxDigests.process(&checkpoint).await.unwrap();
122        TxDigests::commit(&values, &mut conn).await.unwrap();
123        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
124        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
125
126        builder = builder.start_transaction(0).finish_transaction();
127        builder = builder.start_transaction(1).finish_transaction();
128        let checkpoint = Arc::new(builder.build_checkpoint());
129        let values = TxDigests.process(&checkpoint).await.unwrap();
130        TxDigests::commit(&values, &mut conn).await.unwrap();
131        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
132        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
133
134        builder = builder.start_transaction(0).finish_transaction();
135        builder = builder.start_transaction(1).finish_transaction();
136        builder = builder.start_transaction(2).finish_transaction();
137        builder = builder.start_transaction(3).finish_transaction();
138        let checkpoint = Arc::new(builder.build_checkpoint());
139        let values = TxDigests.process(&checkpoint).await.unwrap();
140        TxDigests::commit(&values, &mut conn).await.unwrap();
141        let values = CpSequenceNumbers.process(&checkpoint).await.unwrap();
142        CpSequenceNumbers::commit(&values, &mut conn).await.unwrap();
143
144        let fetched_results = get_all_tx_digests(&mut conn).await.unwrap();
145        assert_eq!(fetched_results.len(), 7);
146
147        // Prune checkpoints from `[0, 2)`, expect 4 tx_digests remaining
148        let rows_pruned = TxDigests.prune(0, 2, &mut conn).await.unwrap();
149        assert_eq!(rows_pruned, 3);
150        let remaining_tx_digests = get_all_tx_digests(&mut conn).await.unwrap();
151        assert_eq!(remaining_tx_digests.len(), 4);
152    }
