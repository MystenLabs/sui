56pub async fn setup_indexer(
57    database_url: Url,
58    db_args: DbArgs,
59    indexer_args: IndexerArgs,
60    client_args: ClientArgs,
61    indexer_config: IndexerConfig,
62    bootstrap_genesis: Option<BootstrapGenesis>,
63    registry: &Registry,
64) -> anyhow::Result<Indexer<Db>> {
65    let IndexerConfig {
66        ingestion,
67        committer,
68        pruner,
69        pipeline,
70    } = indexer_config;
71
72    let PipelineLayer {
73        sum_displays,
74        coin_balance_buckets,
75        cp_blooms,
76        cp_bloom_blocks,
77        cp_sequence_numbers,
78        ev_emit_mod,
79        ev_struct_inst,
80        kv_checkpoints,
81        kv_epoch_ends,
82        kv_epoch_starts,
83        kv_feature_flags,
84        kv_objects,
85        kv_packages,
86        kv_protocol_configs,
87        kv_transactions,
88        obj_info,
89        obj_versions,
90        tx_affected_addresses,
91        tx_affected_objects,
92        tx_balance_changes,
93        tx_calls,
94        tx_digests,
95        tx_kinds,
96    } = pipeline;
97
98    let ingestion = ingestion.finish(IngestionConfig::default())?;
99    let committer = committer.finish(CommitterConfig::default())?;
100    let pruner = pruner.finish(PrunerConfig::default())?;
101
102    let retry_interval = ingestion.retry_interval();
103
104    // Prepare the store for the indexer
105    let store = Db::for_write(database_url, db_args)
106        .await
107        .context("Failed to connect to database")?;
108
109    // we want to merge &MIGRATIONS with the migrations from the store
110    store
111        .run_migrations(Some(&MIGRATIONS))
112        .await
113        .context("Failed to run pending migrations")?;
114
115    registry.register(Box::new(DbConnectionStatsCollector::new(
116        Some("indexer_db"),
117        store.clone(),
118    )))?;
119
120    let metrics_prefix = None;
121    let mut indexer = Indexer::new(
122        store,
123        indexer_args,
124        client_args,
125        ingestion,
126        metrics_prefix,
127        registry,
128    )
129    .await?;
130
131    // These macros are responsible for registering pipelines with the indexer. It is responsible
132    // for:
133    //
134    //  - Checking whether the pipeline is enabled in the file-based configuration.
135    //  - Checking for unexpected parameters in the config.
136    //  - Combining shared and per-pipeline configurations.
137    //  - Registering the pipeline with the indexer.
138    //
139    // There are two kinds of pipelines, each with their own macro: `add_concurrent` and
140    // `add_sequential`. They map directly to `Indexer::concurrent_pipeline` and
141    // `Indexer::sequential_pipeline` respectively.
142
143    macro_rules! add_concurrent {
144        ($handler:expr, $config:expr) => {
145            if let Some(layer) = $config {
146                indexer
147                    .concurrent_pipeline(
148                        $handler,
149                        layer.finish(ConcurrentConfig {
150                            committer: committer.clone(),
151                            pruner: Some(pruner.clone()),
152                            ..Default::default()
153                        })?,
154                    )
155                    .await?
156            }
157        };
158    }
159
160    macro_rules! add_sequential {
161        ($handler:expr, $config:expr) => {
162            if let Some(layer) = $config {
163                indexer
164                    .sequential_pipeline(
165                        $handler,
166                        layer.finish(SequentialConfig {
167                            committer: committer.clone(),
168                            ..Default::default()
169                        })?,
170                    )
171                    .await?
172            }
173        };
174    }
175
176    let genesis = bootstrap(&indexer, retry_interval, bootstrap_genesis).await?;
177
178    // Pipelines that rely on genesis information
179    add_concurrent!(KvFeatureFlags(genesis.clone()), kv_feature_flags);
180    add_concurrent!(KvProtocolConfigs(genesis.clone()), kv_protocol_configs);
181
182    // Summary tables (without write-ahead log)
183    add_sequential!(SumDisplays, sum_displays);
184
185    // Concurrent pipelines with retention
186    add_concurrent!(CoinBalanceBuckets, coin_balance_buckets);
187    add_concurrent!(ObjInfo, obj_info);
188
189    // Unpruned concurrent pipelines
190    add_concurrent!(CpBlooms, cp_blooms);
191    add_concurrent!(CpBloomBlocks, cp_bloom_blocks);
192    add_concurrent!(CpSequenceNumbers, cp_sequence_numbers);
193    add_concurrent!(EvEmitMod, ev_emit_mod);
194    add_concurrent!(EvStructInst, ev_struct_inst);
195    add_concurrent!(KvCheckpoints, kv_checkpoints);
196    add_concurrent!(KvEpochEnds, kv_epoch_ends);
197    add_concurrent!(KvEpochStarts, kv_epoch_starts);
198    add_concurrent!(KvObjects, kv_objects);
199    add_concurrent!(KvPackages, kv_packages);
200    add_concurrent!(KvTransactions, kv_transactions);
201    add_concurrent!(ObjVersions, obj_versions);
202    add_concurrent!(TxAffectedAddresses, tx_affected_addresses);
203    add_concurrent!(TxAffectedObjects, tx_affected_objects);
204    add_concurrent!(TxBalanceChanges, tx_balance_changes);
205    add_concurrent!(TxCalls, tx_calls);
206    add_concurrent!(TxDigests, tx_digests);
207    add_concurrent!(TxKinds, tx_kinds);
208
209    Ok(indexer)
210}
