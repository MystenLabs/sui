41pub async fn setup_indexer(
42    database_url: Url,
43    db_args: DbArgs,
44    indexer_args: IndexerArgs,
45    client_args: ClientArgs,
46    indexer_config: IndexerConfig,
47    bootstrap_genesis: Option<BootstrapGenesis>,
48    registry: &Registry,
49    cancel: CancellationToken,
50) -> anyhow::Result<Indexer<Db>> {
51    let IndexerConfig {
52        ingestion,
53        committer,
54        pruner,
55        pipeline,
56    } = indexer_config;
57
58    let PipelineLayer {
59        sum_displays,
60        coin_balance_buckets,
61        cp_sequence_numbers,
62        ev_emit_mod,
63        ev_struct_inst,
64        kv_checkpoints,
65        kv_epoch_ends,
66        kv_epoch_starts,
67        kv_feature_flags,
68        kv_objects,
69        kv_packages,
70        kv_protocol_configs,
71        kv_transactions,
72        obj_info,
73        obj_versions,
74        tx_affected_addresses,
75        tx_affected_objects,
76        tx_balance_changes,
77        tx_calls,
78        tx_digests,
79        tx_kinds,
80    } = pipeline;
81
82    let ingestion = ingestion.finish(IngestionConfig::default())?;
83    let committer = committer.finish(CommitterConfig::default())?;
84    let pruner = pruner.finish(PrunerConfig::default())?;
85
86    let retry_interval = ingestion.retry_interval();
87
88    // Prepare the store for the indexer
89    let store = Db::for_write(database_url, db_args)
90        .await
91        .context("Failed to connect to database")?;
92
93    // we want to merge &MIGRATIONS with the migrations from the store
94    store
95        .run_migrations(Some(&MIGRATIONS))
96        .await
97        .context("Failed to run pending migrations")?;
98
99    registry.register(Box::new(DbConnectionStatsCollector::new(
100        Some("indexer_db"),
101        store.clone(),
102    )))?;
103
104    let metrics_prefix = None;
105    let mut indexer = Indexer::new(
106        store,
107        indexer_args,
108        client_args,
109        ingestion,
110        metrics_prefix,
111        registry,
112        cancel.clone(),
113    )
114    .await?;
115
116    // These macros are responsible for registering pipelines with the indexer. It is responsible
117    // for:
118    //
119    //  - Checking whether the pipeline is enabled in the file-based configuration.
120    //  - Checking for unexpected parameters in the config.
121    //  - Combining shared and per-pipeline configurations.
122    //  - Registering the pipeline with the indexer.
123    //
124    // There are two kinds of pipelines, each with their own macro: `add_concurrent` and
125    // `add_sequential`. They map directly to `Indexer::concurrent_pipeline` and
126    // `Indexer::sequential_pipeline` respectively.
127
128    macro_rules! add_concurrent {
129        ($handler:expr, $config:expr) => {
130            if let Some(layer) = $config {
131                indexer
132                    .concurrent_pipeline(
133                        $handler,
134                        layer.finish(ConcurrentConfig {
135                            committer: committer.clone(),
136                            pruner: Some(pruner.clone()),
137                        })?,
138                    )
139                    .await?
140            }
141        };
142    }
143
144    macro_rules! add_sequential {
145        ($handler:expr, $config:expr) => {
146            if let Some(layer) = $config {
147                indexer
148                    .sequential_pipeline(
149                        $handler,
150                        layer.finish(SequentialConfig {
151                            committer: committer.clone(),
152                            ..Default::default()
153                        })?,
154                    )
155                    .await?
156            }
157        };
158    }
159
160    let genesis = bootstrap(&indexer, retry_interval, cancel.clone(), bootstrap_genesis).await?;
161
162    // Pipelines that rely on genesis information
163    add_concurrent!(KvFeatureFlags(genesis.clone()), kv_feature_flags);
164    add_concurrent!(KvProtocolConfigs(genesis.clone()), kv_protocol_configs);
165
166    // Summary tables (without write-ahead log)
167    add_sequential!(SumDisplays, sum_displays);
168
169    // Concurrent pipelines with retention
170    add_concurrent!(CoinBalanceBuckets, coin_balance_buckets);
171    add_concurrent!(ObjInfo, obj_info);
172
173    // Unpruned concurrent pipelines
174    add_concurrent!(CpSequenceNumbers, cp_sequence_numbers);
175    add_concurrent!(EvEmitMod, ev_emit_mod);
176    add_concurrent!(EvStructInst, ev_struct_inst);
177    add_concurrent!(KvCheckpoints, kv_checkpoints);
178    add_concurrent!(KvEpochEnds, kv_epoch_ends);
179    add_concurrent!(KvEpochStarts, kv_epoch_starts);
180    add_concurrent!(KvObjects, kv_objects);
181    add_concurrent!(KvPackages, kv_packages);
182    add_concurrent!(KvTransactions, kv_transactions);
183    add_concurrent!(ObjVersions, obj_versions);
184    add_concurrent!(TxAffectedAddresses, tx_affected_addresses);
185    add_concurrent!(TxAffectedObjects, tx_affected_objects);
186    add_concurrent!(TxBalanceChanges, tx_balance_changes);
187    add_concurrent!(TxCalls, tx_calls);
188    add_concurrent!(TxDigests, tx_digests);
189    add_concurrent!(TxKinds, tx_kinds);
190
191    Ok(indexer)
192}
