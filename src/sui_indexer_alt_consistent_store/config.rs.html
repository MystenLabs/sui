4use sui_default_config::DefaultConfig;

9#[DefaultConfig]
10#[derive(Default)]
11#[serde(deny_unknown_fields)]
12pub struct ServiceConfig {
13    /// How checkpoints are read by the indexer.
14    pub ingestion: IngestionConfig,
15
16    /// The size and density of the consistent range.
17    pub consistency: ConsistencyConfig,
18
19    /// Parameters for the database.
20    pub rocksdb: DbConfig,
21
22    /// Default configuration for committers that is shared by all pipelines. Pipelines can
23    /// override individual settings in their own configuration sections.
24    pub committer: CommitterLayer,
25
26    /// Per-pipeline configuration.
27    pub pipeline: PipelineLayer,
28
29    /// Configuration for the read side of the service.
30    pub rpc: RpcConfig,
31}
32
33/// This type is identical to [`framework::ingestion::IngestionConfig`], but is set-up to be
34/// serialized and deserialized by `serde`.
35#[DefaultConfig]
36#[serde(deny_unknown_fields)]
37pub struct IngestionConfig {
38    pub checkpoint_buffer_size: usize,
39    pub ingest_concurrency: usize,
40    pub retry_interval_ms: u64,
41}
42
43#[DefaultConfig]
44#[derive(Clone)]
45#[serde(deny_unknown_fields)]
46pub struct ConsistencyConfig {
47    /// The number of snapshots to keep in the buffer.
48    pub snapshots: u64,
49
50    /// The stride between checkpoints.
51    pub stride: u64,
52
53    /// The size of the buffer for queueing up writes for checkpoints, before they are committed.
54    pub buffer_size: usize,
55}

57#[DefaultConfig]

65#[DefaultConfig]

82impl ServiceConfig {
83    /// Generate an example configuration, suitable for demonstrating the fields available to
84    /// configure.
85    pub fn example() -> Self {
86        let mut example = Self::default();
87
88        example.committer = CommitterConfig::default().into();
89        example.pipeline = PipelineLayer::example();
90
91        example
92    }
93
94    /// Generate a configuration suitable for testing. This is the same as the example
95    /// configuration, but with reduced concurrency and faster polling intervals so tests spend
96    /// less time waiting.
97    pub fn for_test() -> Self {
98        let mut for_test = Self::example();
99
100        for_test.ingestion.retry_interval_ms = 10;
101        for_test.ingestion.ingest_concurrency = 1;
102
103        for_test.committer.collect_interval_ms = Some(50);
104        for_test.committer.watermark_interval_ms = Some(50);
105        for_test.committer.write_concurrency = Some(1);
106
107        for_test
108    }
109}

111impl PipelineLayer {
112    /// Generate an example configuration, suitable for demonstrating the fields available to
113    /// configure.
114    pub fn example() -> Self {
115        Self {
116            balances: Some(CommitterLayer::default()),
117            object_by_owner: Some(CommitterLayer::default()),
118            object_by_type: Some(CommitterLayer::default()),
119        }
120    }
121}
122
123impl CommitterLayer {
124    pub fn finish(self, base: CommitterConfig) -> CommitterConfig {
125        CommitterConfig {
126            write_concurrency: self.write_concurrency.unwrap_or(base.write_concurrency),
127            collect_interval_ms: self.collect_interval_ms.unwrap_or(base.collect_interval_ms),
128            watermark_interval_ms: self
129                .watermark_interval_ms
130                .unwrap_or(base.watermark_interval_ms),
131        }
132    }
133}

155impl From<CommitterConfig> for CommitterLayer {
