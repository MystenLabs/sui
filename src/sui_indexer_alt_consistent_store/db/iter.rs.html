12/// An iterator that scans through elements in increasing key order.
13pub(crate) struct FwdIter<'d, K, V> {
14    inner: Option<rocksdb::DBRawIterator<'d>>,
15    _data: PhantomData<(K, V)>,
16}
17
18/// An iterator that scans through elements in decreasing key order.
19pub(crate) struct RevIter<'d, K, V> {
20    inner: Option<rocksdb::DBRawIterator<'d>>,
21    _data: PhantomData<(K, V)>,
22}
23
24impl<'d, K, V> FwdIter<'d, K, V> {
25    pub(crate) fn new(inner: Option<rocksdb::DBRawIterator<'d>>) -> Self {
26        Self {
27            inner,
28            _data: PhantomData,
29        }
30    }
31
32    /// Move the iterator's cursor so that it will yield the first key greater than or equal to
33    /// `probe`. The probe is a byte slice which does not have to be generated from the key type,
34    /// `K` (e.g. it could be generated from a prefix).
35    pub(crate) fn seek(&mut self, probe: impl AsRef<[u8]>) {
36        if let Some(inner) = &mut self.inner {
37            inner.seek(probe);
38        }
39    }
40
41    /// Skip past all keys that start with the given `prefix`. The iterator will be positioned at
42    /// the first key that does not start with the `prefix`, or will be invalid if no such key
43    /// exists.
44    pub(crate) fn skip_past(&mut self, prefix: impl AsRef<[u8]>) {
45        if let Some(inner) = &mut self.inner {
46            let mut end = prefix.as_ref().to_vec();
47            if key::next(&mut end) {
48                // With the exclusive key calculated, we can seek to it directly
49                inner.seek(end);
50            } else {
51                // Overflow from computed next key - invalidate the iterator
52                inner.seek_to_last();
53                inner.next();
54            }
55        }
56    }
57
58    /// Returns the raw bytes for the next key the iterator will yield, if any.
59    pub(crate) fn raw_key(&self) -> Option<&[u8]> {
60        self.inner.as_ref().and_then(|iter| iter.key())
61    }
62
63    /// Returns the raw bytes for the next value the iterator will yield, if any.
64    pub(crate) fn raw_value(&self) -> Option<&[u8]> {
65        self.inner.as_ref().and_then(|iter| iter.value())
66    }
67
68    /// Returns whether the underlying raw iterator is valid (i.e. will produce a next value) or
69    /// not.
70    pub(crate) fn valid(&self) -> bool {
71        self.inner.as_ref().is_some_and(|iter| iter.valid())
72    }
73}
74
75impl<'d, K, V> RevIter<'d, K, V> {
76    pub(crate) fn new(inner: Option<rocksdb::DBRawIterator<'d>>) -> Self {
77        Self {
78            inner,
79            _data: PhantomData,
80        }
81    }
82
83    /// Move the iterator's cursor so that it will yield the first key less than or equal to
84    /// `probe`. The probe is a byte slice which does not have to be generated from the key type,
85    /// `K` (e.g. it could be generated from a prefix).
86    pub(crate) fn seek(&mut self, probe: impl AsRef<[u8]>) {
87        if let Some(inner) = &mut self.inner {
88            inner.seek_for_prev(probe);
89        }
90    }
91
92    /// Skip past all keys that start with the given `prefix` in reverse order. The iterator will be
93    /// positioned at the last key that does not start with `prefix`, or will be invalid if no such
94    /// key exists.
95    pub(crate) fn skip_past(&mut self, prefix: impl AsRef<[u8]>) {
96        if let Some(inner) = &mut self.inner {
97            let end = prefix.as_ref();
98            inner.seek_for_prev(end);
99            // The only way to land on something in the prefix range is if the prefix itself is an
100            // exact key, so an exact match check is sufficient.
101            if inner.key() == Some(end) {
102                inner.prev();
103            }
104        }
105    }
106
107    /// Returns the raw bytes for the next key the iterator will yield, if any.
108    pub(crate) fn raw_key(&self) -> Option<&[u8]> {
109        self.inner.as_ref().and_then(|iter| iter.key())
110    }
111
112    /// Returns the raw bytes for the next value the iterator will yield, if any.
113    pub(crate) fn raw_value(&self) -> Option<&[u8]> {
114        self.inner.as_ref().and_then(|iter| iter.value())
115    }
116
117    /// Returns whether the underlying raw iterator is valid (i.e. will produce a next value) or
118    /// not.
119    pub(crate) fn valid(&self) -> bool {
120        self.inner.as_ref().is_some_and(|iter| iter.valid())
121    }
122}

124impl<K, V> Iterator for FwdIter<'_, K, V>
125where
126    K: Decode<()>,
127    V: DeserializeOwned,
128{
129    type Item = Result<(K, V), Error>;
130
131    fn next(&mut self) -> Option<Self::Item> {
132        let Some(inner) = &mut self.inner else {
133            return None;
134        };
135
136        match decode_item(inner)? {
137            Err(e) => Some(Err(e)),
138            Ok((k, v)) => {
139                inner.next();
140                Some(Ok((k, v)))
141            }
142        }
143    }
144}

146impl<K, V> Iterator for RevIter<'_, K, V>
147where
148    K: Decode<()>,
149    V: DeserializeOwned,
150{
151    type Item = Result<(K, V), Error>;
152
153    fn next(&mut self) -> Option<Self::Item> {
154        let Some(inner) = &mut self.inner else {
155            return None;
156        };
157
158        match decode_item(inner)? {
159            Err(e) => Some(Err(e)),
160            Ok((k, v)) => {
161                inner.prev();
162                Some(Ok((k, v)))
163            }
164        }
165    }
166}
167
168fn decode_item<K, V>(iter: &rocksdb::DBRawIterator<'_>) -> Option<Result<(K, V), Error>>
169where
170    K: Decode<()>,
171    V: DeserializeOwned,
172{
173    if !iter.valid() {
174        return if let Err(e) = iter.status() {
175            Some(Err(Error::Storage(e)))
176        } else {
177            None
178        };
179    }
180
181    let k = match key::decode(iter.key()?).map_err(Error::KeyDecode) {
182        Ok(k) => k,
183        Err(e) => return Some(Err(e)),
184    };
185
186    let v = match bcs::from_bytes(iter.value()?) {
187        Ok(v) => v,
188        Err(e) => return Some(Err(Error::Bcs(e))),
189    };
190
191    Some(Ok((k, v)))
192}
