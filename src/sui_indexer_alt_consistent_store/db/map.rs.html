19/// A structured representation of a single RocksDB column family, providing snapshot-based reads
20/// and a transactional write API.
21pub(crate) struct DbMap<K, V> {
22    db: Arc<Db>,
23    cf: String,
24    _data: PhantomData<fn(K) -> V>,
25}
26
27impl<K, V> DbMap<K, V>
28where
29    K: Encode + Decode<()>,
30    V: Serialize + DeserializeOwned,
31{
32    /// Open a new `DbMap` for the column family `cf` in database `db`.
33    pub(crate) fn new(db: Arc<Db>, cf: impl Into<String>) -> Self {
34        Self {
35            db,
36            cf: cf.into(),
37            _data: PhantomData,
38        }
39    }
40
41    /// Point look-up at `checkpoint` for the given `key`.
42    ///
43    /// Fails if the database does not have a snapshot at `checkpoint`.
44    pub(crate) fn get(&self, checkpoint: u64, key: impl Borrow<K>) -> Result<Option<V>, Error> {
45        self.db.get(checkpoint, &self.cf()?, key.borrow())
46    }
47
48    /// Multi-point look-up at `checkpoint` for the given `key`.
49    ///
50    /// Fails if the database does not have a snapshot at `checkpoint`.
51    pub(crate) fn multi_get<'k, J: Borrow<K> + 'k>(
52        &self,
53        checkpoint: u64,
54        keys: impl IntoIterator<Item = &'k J>,
55    ) -> Result<Vec<Result<Option<V>, Error>>, Error> {
56        let keys = keys.into_iter().map(|k| k.borrow());
57        self.db.multi_get(checkpoint, &self.cf()?, keys)
58    }
59
60    /// Create a forward iterator over the values in the map at the given `checkpoint`, optionally
61    /// bounding the keys on either side by the given `range`. A forward iterator yields keys in
62    /// ascending bincoded lexicographic order.
63    ///
64    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
65    pub(crate) fn iter(
66        &self,
67        checkpoint: u64,
68        range: impl RangeBounds<K>,
69    ) -> Result<iter::FwdIter<'_, K, V>, Error> {
70        self.db.iter(checkpoint, &self.cf()?, range)
71    }
72
73    /// Create a reverse iterator over the values in the map at the given `checkpoint`, optionally
74    /// bounding the keys on either side by the given `range`. A reverse iterator yields keys in
75    /// descending bincoded lexicographic order.
76    ///
77    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
78    pub(crate) fn iter_rev(
79        &self,
80        checkpoint: u64,
81        range: impl RangeBounds<K>,
82    ) -> Result<iter::RevIter<'_, K, V>, Error> {
83        self.db.iter_rev(checkpoint, &self.cf()?, range)
84    }
85
86    /// Create a forward iterator over the values in the map at the given `checkpoint`, where all
87    /// the keys start with the given `prefix`. A forward iterator yields keys in ascending
88    /// bincoded lexicographic order, and the predicate is applied on the bincoded key and the
89    /// bincoded prefix.
90    ///
91    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
92    pub(crate) fn prefix(
93        &self,
94        checkpoint: u64,
95        prefix: &impl Encode,
96    ) -> Result<iter::FwdIter<'_, K, V>, Error> {
97        self.db.prefix(checkpoint, &self.cf()?, prefix)
98    }
99
100    /// Create a reverse iterator over the values in the map at the given `checkpoint`, where all
101    /// the keys start with the gven `prefix`. A reverse iterator yields keys in descending
102    /// bincoded lexicographic order, and the predicate is applied on the bincoded key and the
103    /// bincoded prefix.
104    ///
105    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
106    pub(crate) fn prefix_rev(
107        &self,
108        checkpoint: u64,
109        prefix: &impl Encode,
110    ) -> Result<iter::RevIter<'_, K, V>, Error> {
111        self.db.prefix_rev(checkpoint, &self.cf()?, prefix)
112    }
113
114    /// Record the insertion of `k -> v` for the map's column family in the given `batch`. The
115    /// write is not performed until the batch is written to the database, and its effects will not
116    /// be visible until a snapshot is created after the batch is written.
117    pub(crate) fn insert(
118        &self,
119        k: impl Borrow<K>,
120        v: impl Borrow<V>,
121        batch: &mut rocksdb::WriteBatch,
122    ) -> Result<(), Error> {
123        batch.put_cf(
124            &self.cf()?,
125            key::encode(k.borrow()),
126            bcs::to_bytes(v.borrow())?,
127        );
128        Ok(())
129    }
130
131    /// Add `v` as a merge operand to `k` for the map's column family in the given `batch`. The
132    /// write is not performed until the batch is written to the database, and its effects will not
133    /// be visible until a snapshot is created after the batch is written.
134    ///
135    /// ## Safety
136    ///
137    /// It is an error to add a merge operand to a column family that does not have a merge
138    /// operator configured. This error will only be raised when the batch is written, and from
139    /// that point all writes to that key and column family will fail because of the pending merge.
140    pub(crate) fn merge(
141        &self,
142        k: impl Borrow<K>,
143        v: impl Borrow<V>,
144        batch: &mut rocksdb::WriteBatch,
145    ) -> Result<(), Error> {
146        batch.merge_cf(
147            &self.cf()?,
148            key::encode(k.borrow()),
149            bcs::to_bytes(v.borrow())?,
150        );
151        Ok(())
152    }
153
154    /// Record the removal of `k` from the map's column family in the given `batch`. The removal is
155    /// not performed until the batch is written to the database, and its effects will not be
156    /// visible until a snapshot is created after the batch is written.
157    pub(crate) fn remove(
158        &self,
159        k: impl Borrow<K>,
160        batch: &mut rocksdb::WriteBatch,
161    ) -> Result<(), Error> {
162        batch.delete_cf(&self.cf()?, key::encode(k.borrow()));
163        Ok(())
164    }
165
166    fn cf(&self) -> Result<Arc<rocksdb::BoundColumnFamily<'_>>, Error> {
167        self.db
168            .cf(&self.cf)
169            .ok_or_else(|| Error::NoColumnFamily(self.cf.clone()))
170    }
171}

222    #[test]
223    fn test_insert_remove() {
224        let d = tempfile::tempdir().unwrap();
225
226        let mut opts = rocksdb::Options::default();
227        opts.create_if_missing(true);
228
229        let cfs = vec![("test", rocksdb::Options::default())];
230
231        let db = Arc::new(Db::open(d.path().join("db"), opts, 4, cfs).unwrap());
232        let map: DbMap<u64, u64> = DbMap::new(db.clone(), "test");
233
234        let mut batch = rocksdb::WriteBatch::default();
235        map.insert(42, 43, &mut batch).unwrap();
236        map.insert(44, 45, &mut batch).unwrap();
237        db.write("batch", wm(0), batch).unwrap();
238        db.take_snapshot(wm(0));
239
240        let mut batch = rocksdb::WriteBatch::default();
241        map.remove(42, &mut batch).unwrap();
242        map.insert(43, 42, &mut batch).unwrap();
243        db.write("batch", wm(1), batch).unwrap();
244        db.take_snapshot(wm(1));
245
246        // Point look-ups
247        assert_eq!(map.get(0, 42).unwrap(), Some(43));
248        assert_eq!(map.get(0, 43).unwrap(), None);
249        assert_eq!(map.get(0, 44).unwrap(), Some(45));
250
251        // Multi-gets
252        assert_eq!(
253            map.multi_get(0, &[42, 43, 44])
254                .unwrap()
255                .into_iter()
256                .map(|r| r.unwrap())
257                .collect::<Vec<_>>(),
258            vec![Some(43), None, Some(45)],
259        );
260
261        // Iteration
262        assert_eq!(
263            map.iter(0, ..)
264                .unwrap()
265                .collect::<Result<Vec<_>, _>>()
266                .unwrap(),
267            vec![(42, 43), (44, 45)],
268        );
269
270        assert_eq!(
271            map.iter_rev(0, ..)
272                .unwrap()
273                .collect::<Result<Vec<_>, _>>()
274                .unwrap(),
275            vec![(44, 45), (42, 43)],
276        );
277
278        // Point look-ups
279        assert_eq!(map.get(1, 42).unwrap(), None);
280        assert_eq!(map.get(1, 43).unwrap(), Some(42));
281        assert_eq!(map.get(1, 44).unwrap(), Some(45));
282
283        // Multi-gets
284        assert_eq!(
285            map.multi_get(1, &[42, 43, 44])
286                .unwrap()
287                .into_iter()
288                .map(|r| r.unwrap())
289                .collect::<Vec<_>>(),
290            vec![None, Some(42), Some(45)],
291        );
292
293        // Iteration
294        assert_eq!(
295            map.iter(1, ..)
296                .unwrap()
297                .collect::<Result<Vec<_>, _>>()
298                .unwrap(),
299            vec![(43, 42), (44, 45)],
300        );
301
302        assert_eq!(
303            map.iter_rev(1, ..)
304                .unwrap()
305                .collect::<Result<Vec<_>, _>>()
306                .unwrap(),
307            vec![(44, 45), (43, 42)],
308        );
309    }

311    #[test]
312    fn test_merge() {
313        let d = tempfile::tempdir().unwrap();
314
315        let mut db_opts = rocksdb::Options::default();
316        db_opts.create_if_missing(true);
317
318        let mut counter_opts = rocksdb::Options::default();
319        counter_opts.set_merge_operator_associative("add", |_, val, rands| {
320            let mut sum = val.map_or(0u64, |v| bcs::from_bytes(v).unwrap());
321            for rand in rands {
322                sum += bcs::from_bytes::<u64>(rand).unwrap();
323            }
324
325            Some(bcs::to_bytes(&sum).unwrap())
326        });
327
328        let cfs = vec![("counts", counter_opts)];
329
330        let db = Arc::new(Db::open(d.path().join("db"), db_opts, 4, cfs).unwrap());
331        let counts: DbMap<u64, u64> = DbMap::new(db.clone(), "counts");
332
333        // Successfully perform a merge to counts.
334        let mut batch = rocksdb::WriteBatch::default();
335        counts.merge(42, 1, &mut batch).unwrap();
336        db.write("counts", wm(0), batch).unwrap();
337        db.take_snapshot(wm(0));
338
339        // Merges allow for the accumulation of values.
340        let mut batch = rocksdb::WriteBatch::default();
341        counts.merge(42, 2, &mut batch).unwrap();
342        db.write("counts", wm(1), batch).unwrap();
343        db.take_snapshot(wm(1));
344
345        // A single batch can include multiple merges for the same key.
346        let mut batch = rocksdb::WriteBatch::default();
347        counts.merge(42, 3, &mut batch).unwrap();
348        counts.merge(42, 4, &mut batch).unwrap();
349        db.write("counts", wm(2), batch).unwrap();
350        db.take_snapshot(wm(2));
351
352        assert_eq!(counts.get(0, 42).unwrap(), Some(1));
353        assert_eq!(counts.get(1, 42).unwrap(), Some(3));
354        assert_eq!(counts.get(2, 42).unwrap(), Some(10));
355    }

357    #[test]
358    fn test_bad_merge() {
359        let d = tempfile::tempdir().unwrap();
360
361        let mut opts = rocksdb::Options::default();
362        opts.create_if_missing(true);
363
364        let cfs = vec![("test", rocksdb::Options::default())];
365
366        let db = Arc::new(Db::open(d.path().join("db"), opts, 4, cfs).unwrap());
367        let values: DbMap<u64, u64> = DbMap::new(db.clone(), "test");
368
369        let mut batch = rocksdb::WriteBatch::default();
370        values.insert(42, 1, &mut batch).unwrap();
371        db.write("values", wm(0), batch).unwrap();
372
373        // Trying to merge to a map that has not had a merge operator set-up will fail.
374        let mut batch = rocksdb::WriteBatch::default();
375        values.merge(42, 2, &mut batch).unwrap();
376        assert!(db.write("values", wm(0), batch).is_err());
377
378        // Subsequent writes will also fail, as the invalid merge has been written to the database.
379        let mut batch = rocksdb::WriteBatch::default();
380        values.insert(43, 3, &mut batch).unwrap();
381        assert!(db.write("values", wm(1), batch).is_err());
382    }
