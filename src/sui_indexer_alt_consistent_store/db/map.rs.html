11/// A structured representation of a single RocksDB column family, providing snapshot-based reads
12/// and a transactional write API.
13pub(crate) struct DbMap<K, V> {
14    db: Arc<Db>,
15    cf: String,
16    _data: PhantomData<fn(K) -> V>,
17}
18
19impl<K, V> DbMap<K, V>
20where
21    K: Encode + Decode<()>,
22    V: Serialize + DeserializeOwned,
23{
24    /// Open a new `DbMap` for the column family `cf` in database `db`.
25    pub(crate) fn new(db: Arc<Db>, cf: impl Into<String>) -> Self {
26        Self {
27            db,
28            cf: cf.into(),
29            _data: PhantomData,
30        }
31    }
32
33    /// Point look-up at `checkpoint` for the given `key`.
34    ///
35    /// Fails if the database does not have a snapshot at `checkpoint`.
36    pub(crate) fn get(&self, checkpoint: u64, key: impl Borrow<K>) -> Result<Option<V>, Error> {
37        self.db.get(checkpoint, &self.cf()?, key.borrow())
38    }
39
40    /// Multi-point look-up at `checkpoint` for the given `key`.
41    ///
42    /// Fails if the database does not have a snapshot at `checkpoint`.
43    pub(crate) fn multi_get<'k, J: Borrow<K> + 'k>(
44        &self,
45        checkpoint: u64,
46        keys: impl IntoIterator<Item = &'k J>,
47    ) -> Result<Vec<Result<Option<V>, Error>>, Error> {
48        let keys = keys.into_iter().map(|k| k.borrow());
49        self.db.multi_get(checkpoint, &self.cf()?, keys)
50    }
51
52    /// Create a forward iterator over the values in the map at the given `checkpoint`, optionally
53    /// bounding the keys on either side by the given `range`. A forward iterator yields keys in
54    /// ascending bincoded lexicographic order.
55    ///
56    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
57    pub(crate) fn iter(
58        &self,
59        checkpoint: u64,
60        range: impl RangeBounds<K>,
61    ) -> Result<iter::FwdIter<'_, K, V>, Error> {
62        self.db.iter(checkpoint, &self.cf()?, range)
63    }
64
65    /// Create a reverse iterator over the values in the map at the given `checkpoint`, optionally
66    /// bounding the keys on either side by the given `range`. A reverse iterator yields keys in
67    /// descending bincoded lexicographic order.
68    ///
69    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
70    pub(crate) fn iter_rev(
71        &self,
72        checkpoint: u64,
73        range: impl RangeBounds<K>,
74    ) -> Result<iter::RevIter<'_, K, V>, Error> {
75        self.db.iter_rev(checkpoint, &self.cf()?, range)
76    }
77
78    /// Create a forward iterator over the values in the map at the given `checkpoint`, where all
79    /// the keys start with the given `prefix`. A forward iterator yields keys in ascending
80    /// bincoded lexicographic order, and the predicate is applied on the bincoded key and the
81    /// bincoded prefix.
82    ///
83    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
84    pub(crate) fn prefix(
85        &self,
86        checkpoint: u64,
87        prefix: &impl Encode,
88    ) -> Result<iter::FwdIter<'_, K, V>, Error> {
89        self.db.prefix(checkpoint, &self.cf()?, prefix)
90    }
91
92    /// Create a reverse iterator over the values in the map at the given `checkpoint`, where all
93    /// the keys start with the gven `prefix`. A reverse iterator yields keys in descending
94    /// bincoded lexicographic order, and the predicate is applied on the bincoded key and the
95    /// bincoded prefix.
96    ///
97    /// This operation can fail if the database does not have a snapshot at `checkpoint`.
98    pub(crate) fn prefix_rev(
99        &self,
100        checkpoint: u64,
101        prefix: &impl Encode,
102    ) -> Result<iter::RevIter<'_, K, V>, Error> {
103        self.db.prefix_rev(checkpoint, &self.cf()?, prefix)
104    }
105
106    /// Record the insertion of `k -> v` for the map's column family in the given `batch`. The
107    /// write is not performed until the batch is written to the database, and its effects will not
108    /// be visible until a snapshot is created after the batch is written.
109    pub(crate) fn insert(
110        &self,
111        k: impl Borrow<K>,
112        v: impl Borrow<V>,
113        batch: &mut rocksdb::WriteBatch,
114    ) -> Result<(), Error> {
115        batch.put_cf(
116            &self.cf()?,
117            key::encode(k.borrow()),
118            bcs::to_bytes(v.borrow())?,
119        );
120        Ok(())
121    }
122
123    /// Add `v` as a merge operand to `k` for the map's column family in the given `batch`. The
124    /// write is not performed until the batch is written to the database, and its effects will not
125    /// be visible until a snapshot is created after the batch is written.
126    ///
127    /// ## Safety
128    ///
129    /// It is an error to add a merge operand to a column family that does not have a merge
130    /// operator configured. This error will only be raised when the batch is written, and from
131    /// that point all writes to that key and column family will fail because of the pending merge.
132    pub(crate) fn merge(
133        &self,
134        k: impl Borrow<K>,
135        v: impl Borrow<V>,
136        batch: &mut rocksdb::WriteBatch,
137    ) -> Result<(), Error> {
138        batch.merge_cf(
139            &self.cf()?,
140            key::encode(k.borrow()),
141            bcs::to_bytes(v.borrow())?,
142        );
143        Ok(())
144    }
145
146    /// Record the removal of `k` from the map's column family in the given `batch`. The removal is
147    /// not performed until the batch is written to the database, and its effects will not be
148    /// visible until a snapshot is created after the batch is written.
149    pub(crate) fn remove(
150        &self,
151        k: impl Borrow<K>,
152        batch: &mut rocksdb::WriteBatch,
153    ) -> Result<(), Error> {
154        batch.delete_cf(&self.cf()?, key::encode(k.borrow()));
155        Ok(())
156    }
157
158    fn cf(&self) -> Result<Arc<rocksdb::BoundColumnFamily<'_>>, Error> {
159        self.db
160            .cf(&self.cf)
161            .ok_or_else(|| Error::NoColumnFamily(self.cf.clone()))
162    }
163}

213    #[test]
214    fn test_insert_remove() {
215        let d = tempfile::tempdir().unwrap();
216
217        let mut opts = rocksdb::Options::default();
218        opts.create_if_missing(true);
219
220        let cfs = vec![("test", rocksdb::Options::default())];
221
222        let db = Arc::new(Db::open(d.path().join("db"), opts, 4, cfs).unwrap());
223        let map: DbMap<u64, u64> = DbMap::new(db.clone(), "test");
224
225        let mut batch = rocksdb::WriteBatch::default();
226        map.insert(42, 43, &mut batch).unwrap();
227        map.insert(44, 45, &mut batch).unwrap();
228        db.write("batch", wm(0), batch).unwrap();
229        db.take_snapshot(wm(0));
230
231        let mut batch = rocksdb::WriteBatch::default();
232        map.remove(42, &mut batch).unwrap();
233        map.insert(43, 42, &mut batch).unwrap();
234        db.write("batch", wm(1), batch).unwrap();
235        db.take_snapshot(wm(1));
236
237        // Point look-ups
238        assert_eq!(map.get(0, 42).unwrap(), Some(43));
239        assert_eq!(map.get(0, 43).unwrap(), None);
240        assert_eq!(map.get(0, 44).unwrap(), Some(45));
241
242        // Multi-gets
243        assert_eq!(
244            map.multi_get(0, &[42, 43, 44])
245                .unwrap()
246                .into_iter()
247                .map(|r| r.unwrap())
248                .collect::<Vec<_>>(),
249            vec![Some(43), None, Some(45)],
250        );
251
252        // Iteration
253        assert_eq!(
254            map.iter(0, ..)
255                .unwrap()
256                .collect::<Result<Vec<_>, _>>()
257                .unwrap(),
258            vec![(42, 43), (44, 45)],
259        );
260
261        assert_eq!(
262            map.iter_rev(0, ..)
263                .unwrap()
264                .collect::<Result<Vec<_>, _>>()
265                .unwrap(),
266            vec![(44, 45), (42, 43)],
267        );
268
269        // Point look-ups
270        assert_eq!(map.get(1, 42).unwrap(), None);
271        assert_eq!(map.get(1, 43).unwrap(), Some(42));
272        assert_eq!(map.get(1, 44).unwrap(), Some(45));
273
274        // Multi-gets
275        assert_eq!(
276            map.multi_get(1, &[42, 43, 44])
277                .unwrap()
278                .into_iter()
279                .map(|r| r.unwrap())
280                .collect::<Vec<_>>(),
281            vec![None, Some(42), Some(45)],
282        );
283
284        // Iteration
285        assert_eq!(
286            map.iter(1, ..)
287                .unwrap()
288                .collect::<Result<Vec<_>, _>>()
289                .unwrap(),
290            vec![(43, 42), (44, 45)],
291        );
292
293        assert_eq!(
294            map.iter_rev(1, ..)
295                .unwrap()
296                .collect::<Result<Vec<_>, _>>()
297                .unwrap(),
298            vec![(44, 45), (43, 42)],
299        );
300    }

302    #[test]
303    fn test_merge() {
304        let d = tempfile::tempdir().unwrap();
305
306        let mut db_opts = rocksdb::Options::default();
307        db_opts.create_if_missing(true);
308
309        let mut counter_opts = rocksdb::Options::default();
310        counter_opts.set_merge_operator_associative("add", |_, val, rands| {
311            let mut sum = val.map_or(0u64, |v| bcs::from_bytes(v).unwrap());
312            for rand in rands {
313                sum += bcs::from_bytes::<u64>(rand).unwrap();
314            }
315
316            Some(bcs::to_bytes(&sum).unwrap())
317        });
318
319        let cfs = vec![("counts", counter_opts)];
320
321        let db = Arc::new(Db::open(d.path().join("db"), db_opts, 4, cfs).unwrap());
322        let counts: DbMap<u64, u64> = DbMap::new(db.clone(), "counts");
323
324        // Successfully perform a merge to counts.
325        let mut batch = rocksdb::WriteBatch::default();
326        counts.merge(42, 1, &mut batch).unwrap();
327        db.write("counts", wm(0), batch).unwrap();
328        db.take_snapshot(wm(0));
329
330        // Merges allow for the accumulation of values.
331        let mut batch = rocksdb::WriteBatch::default();
332        counts.merge(42, 2, &mut batch).unwrap();
333        db.write("counts", wm(1), batch).unwrap();
334        db.take_snapshot(wm(1));
335
336        // A single batch can include multiple merges for the same key.
337        let mut batch = rocksdb::WriteBatch::default();
338        counts.merge(42, 3, &mut batch).unwrap();
339        counts.merge(42, 4, &mut batch).unwrap();
340        db.write("counts", wm(2), batch).unwrap();
341        db.take_snapshot(wm(2));
342
343        assert_eq!(counts.get(0, 42).unwrap(), Some(1));
344        assert_eq!(counts.get(1, 42).unwrap(), Some(3));
345        assert_eq!(counts.get(2, 42).unwrap(), Some(10));
346    }

348    #[test]
349    fn test_bad_merge() {
350        let d = tempfile::tempdir().unwrap();
351
352        let mut opts = rocksdb::Options::default();
353        opts.create_if_missing(true);
354
355        let cfs = vec![("test", rocksdb::Options::default())];
356
357        let db = Arc::new(Db::open(d.path().join("db"), opts, 4, cfs).unwrap());
358        let values: DbMap<u64, u64> = DbMap::new(db.clone(), "test");
359
360        let mut batch = rocksdb::WriteBatch::default();
361        values.insert(42, 1, &mut batch).unwrap();
362        db.write("values", wm(0), batch).unwrap();
363
364        // Trying to merge to a map that has not had a merge operator set-up will fail.
365        let mut batch = rocksdb::WriteBatch::default();
366        values.merge(42, 2, &mut batch).unwrap();
367        assert!(db.write("values", wm(0), batch).is_err());
368
369        // Subsequent writes will also fail, as the invalid merge has been written to the database.
370        let mut batch = rocksdb::WriteBatch::default();
371        values.insert(43, 3, &mut batch).unwrap();
372        assert!(db.write("values", wm(1), batch).is_err());
373    }
