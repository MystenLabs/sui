28pub(crate) struct Balances;
29
30#[derive(Serialize, Deserialize)]
31pub struct Delta {
32    owner: SuiAddress,
33    type_: TypeTag,
34    delta: i128,
35}
36
37impl Delta {
38    fn negated(self) -> Self {
39        Self {
40            owner: self.owner,
41            type_: self.type_,
42            delta: -self.delta,
43        }
44    }
45}
46
47#[async_trait]
48impl Processor for Balances {
49    const NAME: &'static str = "balances";
50    type Value = Delta;
51
52    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> anyhow::Result<Vec<Delta>> {
53        let mut deltas = vec![];
54
55        for (_, (i, _)) in checkpoint_input_objects(checkpoint)? {
56            if let Some(d) = delta(i)? {
57                deltas.push(d.negated())
58            };
59        }
60
61        for (_, (o, _)) in checkpoint_output_objects(checkpoint)? {
62            if let Some(d) = delta(o)? {
63                deltas.push(d)
64            };
65        }
66
67        Ok(deltas)
68    }
69}
70
71impl Restore<Schema> for Balances {
72    fn restore(
73        schema: &Schema,
74        object: &Object,
75        batch: &mut rocksdb::WriteBatch,
76    ) -> anyhow::Result<()> {
77        if let Some(d) = delta(object)? {
78            schema.balances.merge(
79                &Key {
80                    owner: d.owner,
81                    type_: d.type_,
82                },
83                d.delta,
84                batch,
85            )?;
86        }
87
88        Ok(())
89    }
90}

92#[async_trait]
93impl sequential::Handler for Balances {
94    type Store = Store<Schema>;
95    type Batch = BTreeMap<Key, i128>;
96
97    /// Submit a write for every checkpoint, for snapshotting purposes.
98    const MAX_BATCH_CHECKPOINTS: usize = 1;
99
100    /// Values are not batched between checkpoints, but we can simplify the output for a single
101    /// checkpoint by combining deltas for the same owner and type.
102    fn batch(batch: &mut Self::Batch, values: Vec<Delta>) {
103        for value in values {
104            batch
105                .entry(Key {
106                    owner: value.owner,
107                    type_: value.type_,
108                })
109                .and_modify(|v| *v += value.delta)
110                .or_insert(value.delta);
111        }
112    }
113
114    async fn commit<'a>(
115        batch: &Self::Batch,
116        conn: &mut Connection<'a, Schema>,
117    ) -> anyhow::Result<usize> {
118        let balances = &conn.store.schema().balances;
119        for (key, delta) in batch {
120            balances.merge(key, delta, &mut conn.batch)?;
121        }
122
123        Ok(batch.len())
124    }
125}
126
127fn delta(obj: &Object) -> anyhow::Result<Option<Delta>> {
128    // Balances are only tracked for address owners. Balances are combined for coins that
129    // are address-owned and consensus address-owned for the same address.
130    let &owner = match obj.owner() {
131        Owner::AddressOwner(owner) | Owner::ConsensusAddressOwner { owner, .. } => owner,
132        Owner::ObjectOwner(_) | Owner::Shared { .. } | Owner::Immutable => return Ok(None),
133    };
134
135    // Only track coins.
136    let Some((type_, balance)) = Coin::extract_balance_if_coin(obj)? else {
137        return Ok(None);
138    };
139
140    Ok(Some(Delta {
141        owner,
142        type_,
143        delta: balance as i128,
144    }))
