20pub(crate) struct ObjectByType;
21
22pub enum Value {
23    Put(Key, VersionDigest),
24    Del(Key),
25}
26
27#[async_trait]
28impl Processor for ObjectByType {
29    const NAME: &'static str = "object_by_type";
30    type Value = Value;
31
32    async fn process(&self, checkpoint: &Arc<CheckpointData>) -> anyhow::Result<Vec<Value>> {
33        let input_objects = checkpoint_input_objects(checkpoint)?;
34        let output_objects = checkpoint_output_objects(checkpoint)?;
35        let mut values = vec![];
36
37        // Objects that are in the inputs but not the outputs have been deleted.
38        for (id, &(input, _)) in &input_objects {
39            let Some(key_in) = Key::from_object(input) else {
40                continue;
41            };
42
43            if !output_objects.contains_key(id) {
44                values.push(Value::Del(key_in));
45            }
46        }
47
48        for (id, (output, digest)) in output_objects {
49            let Some(key_out) = Key::from_object(output) else {
50                continue;
51            };
52
53            // If the ID is in the input objects with a different key, it needs to be deleted at
54            // that location.
55            if let Some(key_in) = input_objects
56                .get(&id)
57                .and_then(|(input, _)| Key::from_object(input))
58                && key_in != key_out
59            {
60                values.push(Value::Del(key_in));
61            }
62
63            // The object is always put at its output location.
64            values.push(Value::Put(key_out, (output.version(), digest)));
65        }
66
67        Ok(values)
68    }
69}
70
71impl Restore<Schema> for ObjectByType {
72    fn restore(
73        schema: &Schema,
74        object: &Object,
75        batch: &mut rocksdb::WriteBatch,
76    ) -> anyhow::Result<()> {
77        if let Some(key) = Key::from_object(object) {
78            let val = (object.version(), object.digest());
79            schema.object_by_type.insert(key, val, batch)?;
80        }
81
82        Ok(())
83    }
84}
85
86#[async_trait]
87impl sequential::Handler for ObjectByType {
88    type Store = Store<Schema>;
89    type Batch = Vec<Value>;
90
91    /// Submit a write for every checkpoint, for snapshotting purposes.
92    const MAX_BATCH_CHECKPOINTS: usize = 1;
93
94    /// No batching actually happens, because `MAX_BATCH_CHECKPOINTS` is 1.
95    fn batch(batch: &mut Self::Batch, values: Vec<Value>) {
96        batch.extend(values);
97    }
98
99    async fn commit<'a>(
100        batch: &Self::Batch,
101        conn: &mut Connection<'a, Schema>,
102    ) -> anyhow::Result<usize> {
103        let object_by_type = &conn.store.schema().object_by_type;
104
105        for value in batch {
106            match value {
107                Value::Put(key, val) => {
108                    object_by_type.insert(key, val, &mut conn.batch)?;
109                }
110                Value::Del(key) => {
111                    object_by_type.remove(key, &mut conn.batch)?;
112                }
113            }
114        }
115
116        Ok(batch.len())
117    }
118}
