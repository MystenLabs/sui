4//! # Consistent Store Indexer and RPC
5//!
6//! This crate defines a service that combines a [`sui_indexer_alt_framework`] indexer, with a gRPC
7//! service serving queries about live data, consistent with some recent (measured in minutes or
8//! hours) checkpoint.
9//!
10//! Supported queries include fetching objects by owner or by type, and fetching an address'
11//! balance (across coin-like objects it owns).
12//!
13//! The service's indexer writes to a RocksDB database which it interacts through a `db`
14//! abstraction, which exposes a type-safe abstraction over the underlying bytes-to-bytes ordered
15//! map offered by [`rocksdb`].
16//!
17//! The database abstraction is also responsible for taking and exposing snapshots of the database,
18//! which is what allows the RPC to serve a query at some checkpoint in the recent past. Snapshots
19//! preserve access to the state of the database at a point in time, they are ephemeral (stored in
20//! memory), and database-wide (not per-column-family).
21//!
22//! It is the `Indexer`'s responsibility to coordinate writes across pipelines, to arrange for the
23//! database to contain a consistent view of the data at checkpoints it should take a snapshot of.
24//! To this end, the indexer only supports sequential pipelines (pipelines also update keys
25//! in-place, which precludes out-of-order writes), but writes are buffered, post-commit to allow
26//! pipelines to make progress on later while checkpoints while waiting for lagging pipelines to
27//! reach the snapshot checkpoint.
28//!
29//! The indexer and RPC agree on a `Schema` which describes the key types, value types and options
30//! for all column families to be set-up in the database.

69/// Set-up and run the Indexer and RPC service, using the provided arguments (expected to be
70/// extracted from the command-line).
71///
72/// `path` is the path to the RocksDB database,which will be created if it does not exist.
73/// `indexer_args` and `client_args` control the behavior of the Indexer, while `rpc_args` controls
74/// the behavior of the RPC service.
75///
76/// The service spins up auxiliary services (to expose metrics, run the indexer, and the RPC), and
77/// will clean these up on shutdown as well.
78pub async fn start_service(
79    path: impl AsRef<Path>,
80    indexer_args: IndexerArgs,
81    client_args: ClientArgs,
82    rpc_args: RpcArgs,
83    version: &'static str,
84    config: ServiceConfig,
85    registry: &Registry,
86) -> anyhow::Result<Service> {
87    let ServiceConfig {
88        ingestion,
89        consistency,
90        rocksdb,
91        committer,
92        pipeline:
93            PipelineLayer {
94                balances,
95                object_by_owner,
96                object_by_type,
97                address_balances,
98            },
99        rpc,
100    } = config;
101
102    let committer = committer.finish(CommitterConfig::default());
103
104    let mut indexer: Indexer<Schema> = Indexer::new(
105        path,
106        indexer_args,
107        client_args,
108        consistency.clone(),
109        ingestion.into(),
110        rocksdb,
111        registry,
112    )
113    .await?;
114
115    let state = State {
116        store: indexer.store().clone(),
117        rpc_config: Arc::new(rpc),
118        consistency_config: Arc::new(consistency),
119    };
120
121    let rpc = RpcService::new(rpc_args, version, registry)
122        .await?
123        .register_encoded_file_descriptor_set(proto::rpc::consistent::v1alpha::FILE_DESCRIPTOR_SET)
124        .add_service(ConsistentServiceServer::new(state.clone()));
125
126    macro_rules! add_sequential {
127        ($handler:expr, $config:expr) => {
128            if let Some(layer) = $config {
129                indexer
130                    .sequential_pipeline(
131                        $handler,
132                        SequentialConfig {
133                            committer: layer.finish(committer.clone()),
134                            checkpoint_lag: 0,
135                        },
136                    )
137                    .await?
138            }
139        };
140    }
141
142    add_sequential!(Balances, balances);
143    add_sequential!(ObjectByOwner, object_by_owner);
144    add_sequential!(ObjectByType, object_by_type);
145    add_sequential!(AddressBalances, address_balances);
146
147    let s_rpc = rpc.run().await?;
148    let s_indexer = indexer.run().await?;
149
150    Ok(s_rpc.merge(s_indexer))
151}
