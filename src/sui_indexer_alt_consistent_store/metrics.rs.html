4use std::sync::Arc;
5
6use prometheus::IntGaugeVec;
7use prometheus::Opts;
8use prometheus::core::Collector;
9use prometheus::core::Desc;
10use prometheus::proto::MetricFamily;
11
12pub struct ColumnFamilyStatsCollector {
13    db: Arc<crate::db::Db>,
14    metrics: ColumnFamilyMetrics,
15    cf_names: Vec<String>,
16}
17
18#[derive(Debug)]
19pub struct ColumnFamilyMetrics {
20    /// Size of the active memtable in bytes.
21    pub current_size_active_mem_tables: IntGaugeVec,
22    /// Size of active, unflushed immutable, and pinned memtable in bytes.
23    pub size_all_mem_tables: IntGaugeVec,
24    /// Memory size for the entries residing in the block cache.
25    pub block_cache_usage: IntGaugeVec,
26    /// Memory size of entries pinned in the block cache.
27    pub block_cache_pinned_usage: IntGaugeVec,
28    /// Estimated memory used by SST table readers, not including memory used.
29    pub estimate_table_readers_mem: IntGaugeVec,
30    /// Total number of bytes that need to be compacted to get all levels down to under target size.
31    pub estimate_pending_compaction_bytes: IntGaugeVec,
32    /// Number of L0 files.
33    pub num_level0_files: IntGaugeVec,
34    /// Number of immutable memtables that have not yet been flushed.
35    pub num_immutable_mem_tables: IntGaugeVec,
36    /// Boolean flag (0/1) indicating whether a memtable flush is pending.
37    pub mem_table_flush_pending: IntGaugeVec,
38    /// Boolean flag (0/1) indicating whether a compaction is pending.
39    pub compaction_pending: IntGaugeVec,
40    /// Number of snapshots.
41    pub num_snapshots: IntGaugeVec,
42    /// Number of running compactions.
43    pub num_running_compactions: IntGaugeVec,
44    /// Number of running flushes.
45    pub num_running_flushes: IntGaugeVec,
46    /// The current delayed write rate. 0 means no delay.
47    pub actual_delayed_write_rate: IntGaugeVec,
48    /// Boolean flag (0/1) indicating whether RocksDB has stopped all writes.
49    pub is_write_stopped: IntGaugeVec,
50}
51
52impl ColumnFamilyStatsCollector {
53    pub fn new(prefix: Option<&str>, db: Arc<crate::db::Db>, cf_names: Vec<String>) -> Self {
54        let metrics = ColumnFamilyMetrics::new(prefix);
55
56        Self {
57            db,
58            metrics,
59            cf_names,
60        }
61    }
62}
63
64/// Create metrics without registering them
65impl ColumnFamilyMetrics {
66    pub(crate) fn new(prefix: Option<&str>) -> Self {
67        let prefix = prefix.unwrap_or("rocksdb");
68        let name = |n| format!("{prefix}_{n}");
69
70        Self {
71            num_level0_files: IntGaugeVec::new(
72                Opts::new(name("num_level0_files"), "Number of level 0 files in the column family"),
73                &["cf_name"],
74            ).unwrap(),
75            current_size_active_mem_tables: IntGaugeVec::new(
76                Opts::new(name("current_size_active_mem_tables"), "The current approximate size of active memtable (bytes)."),
77                &["cf_name"],
78            ).unwrap(),
79            size_all_mem_tables: IntGaugeVec::new(
80                Opts::new(name("size_all_mem_tables"), "The memory size occupied by the column family's in-memory buffer"),
81                &["cf_name"],
82            ).unwrap(),
83            num_snapshots: IntGaugeVec::new(
84                Opts::new(name("num_snapshots"), "Number of snapshots held for the column family"),
85                &["cf_name"],
86            ).unwrap(),
87            actual_delayed_write_rate: IntGaugeVec::new(
88                Opts::new(name("actual_delayed_write_rate"), "The current actual delayed write rate. 0 means no delay"),
89                &["cf_name"],
90            ).unwrap(),
91            is_write_stopped: IntGaugeVec::new(
92                Opts::new(name("is_write_stopped"), "A flag indicating whether writes are stopped on this column family. 1 indicates writes have been stopped."),
93                &["cf_name"],
94            ).unwrap(),
95            block_cache_usage: IntGaugeVec::new(
96                Opts::new(name("block_cache_usage"), "The memory size used by the column family in the block cache."),
97                &["cf_name"],
98            ).unwrap(),
99            block_cache_pinned_usage: IntGaugeVec::new(
100                Opts::new(name("block_cache_pinned_usage"), "The memory size used by the column family in the block cache where entries are pinned"),
101                &["cf_name"],
102            ).unwrap(),
103            estimate_table_readers_mem: IntGaugeVec::new(
104                Opts::new(name("estimate_table_readers_mem"), "The estimated memory size used for reading SST tables in this column family such as filters and index blocks. Note that this number does not include the memory used in block cache."),
105                &["cf_name"],
106            ).unwrap(),
107            num_immutable_mem_tables: IntGaugeVec::new(
108                Opts::new(name("num_immutable_mem_tables"), "The number of immutable memtables that have not yet been flushed."),
109                &["cf_name"],
110            ).unwrap(),
111            mem_table_flush_pending: IntGaugeVec::new(
112                Opts::new(name("mem_table_flush_pending"), "A 1 or 0 flag indicating whether a memtable flush is pending. If this number is 1, it means a memtable is waiting for being flushed, but there might be too many L0 files that prevents it from being flushed."),
113                &["cf_name"],
114            ).unwrap(),
115            compaction_pending: IntGaugeVec::new(
116                Opts::new(name("compaction_pending"), "A 1 or 0 flag indicating whether a compaction job is pending. If this number is 1, it means some part of the column family requires compaction in order to maintain shape of LSM tree, but the compaction is pending because the desired compaction job is either waiting for other dependent compactions to be finished or waiting for an available compaction thread."),
117                &["cf_name"],
118            ).unwrap(),
119            estimate_pending_compaction_bytes: IntGaugeVec::new(
120                Opts::new(name("estimate_pending_compaction_bytes"), "Estimated total number of bytes compaction needs to rewrite to get all levels down to under target size. Not valid for other compactions than level-based."),
121                &["cf_name"],
122            ).unwrap(),
123            num_running_compactions: IntGaugeVec::new(
124                Opts::new(name("num_running_compactions"), "The number of compactions that are currently running for the column family."),
125                &["cf_name"],
126            ).unwrap(),
127            num_running_flushes: IntGaugeVec::new(
128                Opts::new(name("num_running_flushes"), "The number of flushes that are currently running for the column family."),
129                &["cf_name"],
130            ).unwrap(),
131        }
132    }
133
134    pub(crate) fn desc(&self) -> Vec<&Desc> {
135        vec![
136            self.num_level0_files.desc(),
137            self.current_size_active_mem_tables.desc(),
138            self.size_all_mem_tables.desc(),
139            self.num_snapshots.desc(),
140            self.actual_delayed_write_rate.desc(),
141            self.is_write_stopped.desc(),
142            self.block_cache_usage.desc(),
143            self.block_cache_pinned_usage.desc(),
144            self.estimate_table_readers_mem.desc(),
145            self.num_immutable_mem_tables.desc(),
146            self.mem_table_flush_pending.desc(),
147            self.compaction_pending.desc(),
148            self.estimate_pending_compaction_bytes.desc(),
149            self.num_running_compactions.desc(),
150            self.num_running_flushes.desc(),
151        ]
152        .into_iter()
153        .flatten()
154        .collect()
155    }
156
157    pub(crate) fn collect_all(&self) -> Vec<MetricFamily> {
158        vec![
159            self.num_level0_files.collect(),
160            self.current_size_active_mem_tables.collect(),
161            self.size_all_mem_tables.collect(),
162            self.num_snapshots.collect(),
163            self.actual_delayed_write_rate.collect(),
164            self.is_write_stopped.collect(),
165            self.block_cache_usage.collect(),
166            self.block_cache_pinned_usage.collect(),
167            self.estimate_table_readers_mem.collect(),
168            self.num_immutable_mem_tables.collect(),
169            self.mem_table_flush_pending.collect(),
170            self.compaction_pending.collect(),
171            self.estimate_pending_compaction_bytes.collect(),
172            self.num_running_compactions.collect(),
173            self.num_running_flushes.collect(),
174        ]
175        .into_iter()
176        .flatten()
177        .collect()
178    }
179}
180
181impl Collector for ColumnFamilyStatsCollector {
182    fn desc(&self) -> Vec<&Desc> {
183        self.metrics.desc()
184    }
185
186    fn collect(&self) -> Vec<MetricFamily> {
187        for cf_name in &self.cf_names {
188            let cf_metrics = self.db.column_family_metrics(cf_name);
189            self.metrics
190                .current_size_active_mem_tables
191                .with_label_values(&[cf_name])
192                .set(cf_metrics.current_size_active_mem_tables);
193
194            self.metrics
195                .size_all_mem_tables
196                .with_label_values(&[cf_name])
197                .set(cf_metrics.size_all_mem_tables);
198
199            self.metrics
200                .block_cache_usage
201                .with_label_values(&[cf_name])
202                .set(cf_metrics.block_cache_usage);
203
204            self.metrics
205                .block_cache_pinned_usage
206                .with_label_values(&[cf_name])
207                .set(cf_metrics.block_cache_pinned_usage);
208
209            self.metrics
210                .estimate_table_readers_mem
211                .with_label_values(&[cf_name])
212                .set(cf_metrics.estimate_table_readers_mem);
213
214            self.metrics
215                .estimate_pending_compaction_bytes
216                .with_label_values(&[cf_name])
217                .set(cf_metrics.estimate_pending_compaction_bytes);
218
219            self.metrics
220                .num_level0_files
221                .with_label_values(&[cf_name])
222                .set(cf_metrics.num_level0_files);
223
224            self.metrics
225                .actual_delayed_write_rate
226                .with_label_values(&[cf_name])
227                .set(cf_metrics.actual_delayed_write_rate);
228
229            self.metrics
230                .is_write_stopped
231                .with_label_values(&[cf_name])
232                .set(cf_metrics.is_write_stopped);
233
234            self.metrics
235                .num_immutable_mem_tables
236                .with_label_values(&[cf_name])
237                .set(cf_metrics.num_immutable_mem_tables);
238
239            self.metrics
240                .mem_table_flush_pending
241                .with_label_values(&[cf_name])
242                .set(cf_metrics.mem_table_flush_pending);
243
244            self.metrics
245                .compaction_pending
246                .with_label_values(&[cf_name])
247                .set(cf_metrics.compaction_pending);
248
249            self.metrics
250                .num_snapshots
251                .with_label_values(&[cf_name])
252                .set(cf_metrics.num_snapshots);
253
254            self.metrics
255                .num_running_compactions
256                .with_label_values(&[cf_name])
257                .set(cf_metrics.num_running_compactions);
258
259            self.metrics
260                .num_running_flushes
261                .with_label_values(&[cf_name])
262                .set(cf_metrics.num_running_flushes);
263        }
264
265        self.metrics.collect_all()
266    }
