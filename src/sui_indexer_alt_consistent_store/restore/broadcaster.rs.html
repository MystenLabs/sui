32/// Wait at most this long between retries while fetching files from the snapshot.
33const MAX_RETRY_INTERVAL: Duration = Duration::from_secs(60);
34
35/// If a fetch takes longer than this, log a warning.
36const SLOW_FETCH_THRESHOLD: Duration = Duration::from_secs(600);
37
38/// The broadcaster task is responsible for consulting the formal snapshot's metadata file,
39/// fetching object files from the formal snapshot and disseminating them to all subscribers in
40/// `subscribers`.
41///
42/// The task will shut down if all object files have been restored across all subscribers. Returns
43/// `Ok(_)` if all object files were successfully fetched and passed to subscribers, or `Err(_)`
44/// otherwise.
45pub(super) fn broadcaster(
46    object_file_concurrency: usize,
47    subscribers: BTreeMap<String, mpsc::Sender<Arc<LiveObjects>>>,
48    db: Arc<Db>,
49    snapshot: FormalSnapshot,
50    metrics: Arc<RestorerMetrics>,
51) -> Service {
52    Service::new().spawn_aborting(async move {
53        info!("Starting broadcaster");
54
55        let manifest = snapshot
56            .manifest()
57            .await
58            .and_then(|bs| EpochManifest::read(&bs))
59            .context("Failed to read snapshot manifest")?;
60
61        let metadata: Vec<_> = manifest
62            .metadata()
63            .iter()
64            .filter(|m| matches!(m.file_type, FileType::Object))
65            .cloned()
66            .collect();
67
68        metrics.total_partitions.set(metadata.len() as i64);
69        info!(partitions = metadata.len(), "Restoring partitions");
70
71        match stream::iter(metadata)
72            .try_for_each_spawned(object_file_concurrency, |metadata| {
73                let subscribers = subscribers.clone();
74                let db = db.clone();
75                let snapshot = snapshot.clone();
76                let metrics = metrics.clone();
77
78                async move {
79                    let restored = db
80                        .is_restored(
81                            metadata.bucket,
82                            metadata.partition,
83                            subscribers.keys().map(|p| p.as_str()),
84                        )
85                        .context("Failed to check restored markers")
86                        .map_err(Break::Err)?;
87
88                    // If all the pipelines have restored this partition, it can be skipped.
89                    if restored.iter().all(|r| *r) {
90                        info!(
91                            bucket = metadata.bucket,
92                            partition = metadata.partition,
93                            "Skipping",
94                        );
95                        metrics.total_partitions_skipped.inc();
96                        return Ok(());
97                    } else {
98                        info!(
99                            bucket = metadata.bucket,
100                            partition = metadata.partition,
101                            "Fetching",
102                        );
103                    }
104
105                    // Download the object file.
106                    let objects = Arc::new(
107                        fetch_objects(snapshot, metadata, metrics.as_ref())
108                            .await
109                            .map_err(Break::Err)?,
110                    );
111
112                    // Send it to all subscribers who are not restored yet.
113                    let futures = subscribers
114                        .iter()
115                        .zip(restored)
116                        .filter(|(_, restored)| !*restored)
117                        .map(|((_, s), _)| s.send(objects.clone()));
118
119                    if try_join_all(futures).await.is_err() {
120                        info!("Subscription dropped, signalling shutdown");
121                        Err(Break::Break)
122                    } else {
123                        metrics.total_partitions_broadcast.inc();
124                        Ok(())
125                    }
126                }
127            })
128            .await
129        {
130            Ok(()) => {
131                info!("Live objects done, stopping broadcaster");
132                Ok(())
133            }
134
135            Err(Break::Break) => {
136                info!("Channel closed, stopping broadcaster");
137                Ok(())
138            }
139
140            Err(Break::Err(e)) => {
141                error!("Error from broadcaster: {e:#}");
142                Err(e)
143            }
144        }
145    })
146}

148/// Fetch the file described by `metadata` from `snapshot` as a live objects file.
149///
150/// This function will repeatedly retry the fetch, with exponential backoff, until it succeeds. It
151/// also monitors for individual fetches that seem slower than expected, logging a warning if one
152/// is found.
153async fn fetch_objects(
154    snapshot: FormalSnapshot,
155    metadata: FileMetadata,
156    metrics: &RestorerMetrics,
157) -> anyhow::Result<LiveObjects> {
158    let _guard = metrics.objects_fetch_latency.start_timer();
159
160    let attempts = AtomicUsize::new(1);
161    let request = || async {
162        let attempt = attempts.fetch_add(1, Ordering::Relaxed);
163
164        let future = async {
165            snapshot
166                .file(&metadata)
167                .await
168                .and_then(|bs| Ok((bs.len(), LiveObjects::read(&bs, &metadata)?)))
169        };
170
171        match with_slow_future_monitor(future, SLOW_FETCH_THRESHOLD, || {
172            warn!(
173                attempt,
174                bucket = metadata.bucket,
175                partition = metadata.partition,
176                "Fetch slow"
177            );
178        })
179        .await
180        {
181            Ok((bytes, objects)) => {
182                metrics.total_bytes_fetched.inc_by(bytes as u64);
183                metrics.total_partitions_fetched.inc();
184                Ok(objects)
185            }
186
187            Err(e) => {
188                warn!(
189                    attempt,
190                    bucket = metadata.bucket,
191                    partition = metadata.partition,
192                    "Fetch error: {e:#}"
193                );
194
195                metrics.total_objects_fetch_retries.inc();
196                Err(BE::transient(e))
197            }
198        }
199    };
200
201    let backoff = ExponentialBackoff {
202        max_interval: MAX_RETRY_INTERVAL,
203        max_elapsed_time: None,
204        ..Default::default()
205    };
206
207    backoff::future::retry(backoff, request).await
208}
