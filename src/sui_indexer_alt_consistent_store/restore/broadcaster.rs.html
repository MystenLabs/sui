28/// Wait at most this long between retries while fetching files from the snapshot.
29const MAX_RETRY_INTERVAL: Duration = Duration::from_secs(60);
30
31/// If a fetch takes longer than this, log a warning.
32const SLOW_FETCH_THRESHOLD: Duration = Duration::from_secs(600);
33
34/// The broadcaster task is responsible for consulting the formal snapshot's metadata file,
35/// fetching object files from the formal snapshot and disseminating them to all subscribers in
36/// `subscribers`.
37///
38/// The task will shut down if the `cancel` token is signalled, or if all object files have been
39/// restored across all subscribers. Returns `Ok(_)` if all object files were successfully fetched
40/// and passed to subscribers, or `Err(_)` otherwise.
41pub(super) fn broadcaster(
42    object_file_concurrency: usize,
43    subscribers: BTreeMap<String, mpsc::Sender<Arc<LiveObjects>>>,
44    db: Arc<Db>,
45    snapshot: FormalSnapshot,
46    metrics: Arc<RestorerMetrics>,
47    cancel: CancellationToken,
48) -> JoinHandle<Result<(), ()>> {
49    tokio::spawn(async move {
50        info!("Starting broadcaster");
51
52        let manifest = match snapshot
53            .manifest()
54            .await
55            .and_then(|bs| EpochManifest::read(&bs))
56        {
57            Ok(manifest) => manifest,
58            Err(e) => {
59                error!("Failed to read snapshot manifest: {e:#}");
60                cancel.cancel();
61                return Err(());
62            }
63        };
64
65        let metadata: Vec<_> = manifest
66            .metadata()
67            .iter()
68            .filter(|m| matches!(m.file_type, FileType::Object))
69            .cloned()
70            .collect();
71
72        metrics.total_partitions.set(metadata.len() as i64);
73        info!(partitions = metadata.len(), "Restoring partitions");
74
75        match stream::iter(metadata)
76            .try_for_each_spawned(object_file_concurrency, |metadata| {
77                let subscribers = subscribers.clone();
78                let db = db.clone();
79                let snapshot = snapshot.clone();
80                let metrics = metrics.clone();
81
82                let supervisor_cancel = cancel.clone();
83                let cancel = cancel.clone();
84
85                async move {
86                    let restored = db
87                        .is_restored(
88                            metadata.bucket,
89                            metadata.partition,
90                            subscribers.keys().map(|p| p.as_str()),
91                        )
92                        .context("Failed to check restored markers")?;
93
94                    // If all the pipelines have restored this partition, it can be skipped.
95                    if restored.iter().all(|r| *r) {
96                        info!(
97                            bucket = metadata.bucket,
98                            partition = metadata.partition,
99                            "Skipping",
100                        );
101                        metrics.total_partitions_skipped.inc();
102                        return Ok(());
103                    } else {
104                        info!(
105                            bucket = metadata.bucket,
106                            partition = metadata.partition,
107                            "Fetching",
108                        );
109                    }
110
111                    // Download the object file.
112                    let objects = tokio::select! {
113                        objects = fetch_objects(snapshot, metadata, metrics.as_ref()) => Arc::new(objects?),
114                        _ = cancel.cancelled() => {
115                            return Err(Break::Cancel);
116                        }
117                    };
118
119                    // Send it to all subscribers who are not restored yet.
120                    let futures = subscribers
121                        .iter()
122                        .zip(restored)
123                        .filter(|(_, restored)| !*restored)
124                        .map(|((_, s), _)| s.send(objects.clone()));
125
126                    if try_join_all(futures).await.is_err() {
127                        info!("Subscription dropped, signalling shutdown");
128                        supervisor_cancel.cancel();
129                        Err(Break::Cancel)
130                    } else {
131                        metrics.total_partitions_broadcast.inc();
132                        Ok(())
133                    }
134                }
135            })
136            .await
137        {
138            Ok(()) => {
139                info!("Live objects done, stopping broadcaster");
140                Ok(())
141            }
142
143            Err(Break::Cancel) => {
144                info!("Shutdown received, stopping broadcaster");
145                Err(())
146            }
147
148            Err(Break::Err(e)) => {
149                error!("Error from broadcaster: {e:#}");
150                Err(())
151            }
152        }
153    })
154}

156/// Fetch the file described by `metadata` from `snapshot` as a live objects file.
157///
158/// This function will repeatedly retry the fetch, with exponential backoff, until it succeeds. It
159/// also monitors for individual fetches that seem slower than expected, logging a warning if one
160/// is found.
161async fn fetch_objects(
162    snapshot: FormalSnapshot,
163    metadata: FileMetadata,
164    metrics: &RestorerMetrics,
165) -> anyhow::Result<LiveObjects> {
166    let _guard = metrics.objects_fetch_latency.start_timer();
167
168    let attempts = AtomicUsize::new(1);
169    let request = || async {
170        let attempt = attempts.fetch_add(1, Ordering::Relaxed);
171
172        let future = async {
173            snapshot
174                .file(&metadata)
175                .await
176                .and_then(|bs| Ok((bs.len(), LiveObjects::read(&bs, &metadata)?)))
177        };
178
179        match with_slow_future_monitor(future, SLOW_FETCH_THRESHOLD, || {
180            warn!(
181                attempt,
182                bucket = metadata.bucket,
183                partition = metadata.partition,
184                "Fetch slow"
185            );
186        })
187        .await
188        {
189            Ok((bytes, objects)) => {
190                metrics.total_bytes_fetched.inc_by(bytes as u64);
191                metrics.total_partitions_fetched.inc();
192                Ok(objects)
193            }
194
195            Err(e) => {
196                warn!(
197                    attempt,
198                    bucket = metadata.bucket,
199                    partition = metadata.partition,
200                    "Fetch error: {e:#}"
201                );
202
203                metrics.total_objects_fetch_retries.inc();
204                Err(BE::transient(e))
205            }
206        }
207    };
208
209    let backoff = ExponentialBackoff {
210        max_interval: MAX_RETRY_INTERVAL,
211        max_elapsed_time: None,
212        ..Default::default()
213    };
214
215    backoff::future::retry(backoff, request).await
216}
