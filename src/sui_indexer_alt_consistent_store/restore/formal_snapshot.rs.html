30#[derive(clap::Args, Clone, Debug)]
31#[group(required = true)]
32pub struct FormalSnapshotArgs {
33    /// Fetch formal snapshot from AWS S3. Provide the bucket name or endpoint-and-bucket.
34    /// (env: AWS_ENDPOINT, AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY, AWS_DEFAULT_REGION)
35    #[arg(long, group = "source")]
36    pub s3: Option<String>,
37
38    /// Fetch formal snapshot from Google Cloud Storage. Provide the bucket name.
39    /// (env: GOOGLE_SERVICE_ACCOUNT_PATH).
40    #[arg(long, group = "source")]
41    pub gcs: Option<String>,
42
43    /// Fetch formal snapshot from Azure Blob Storage. Provide the container name.
44    /// (env: AZURE_STORAGE_ACCOUNT_NAME, AZURE_STORAGE_ACCESS_KEY)
45    #[arg(long, group = "source")]
46    pub azure: Option<String>,
47
48    /// Fetch formal snapshot from a generic HTTP endpoint.
49    #[arg(long, group = "source")]
50    pub http: Option<Url>,
51
52    /// Fetch formal snapshot from local filesystem. Provide the path to the snapshot directory.
53    #[arg(long, group = "source")]
54    pub path: Option<PathBuf>,
55
56    /// URL of the remote checkpoint store. Used to fetch the mapping between epochs and
57    /// checkpoints.
58    #[arg(long)]
59    pub remote_store_url: Url,
60
61    /// The epoch to restore from. Restores from the latest epoch if not specified.
62    #[arg(long)]
63    pub epoch: Option<u64>,
64}
65
66/// Interface over a formal snapshot at a specific epoch.
67#[derive(Clone)]
68pub(super) struct FormalSnapshot {
69    /// The source of files related to the formal snapshot.
70    source: Arc<dyn Storage + Send + Sync + 'static>,
71
72    /// The point in time this snapshot is from.
73    watermark: Watermark,
74}
75
76impl FormalSnapshot {
77    /// Establish a connection to a formal snapshot source, and validate that it can serve data for
78    /// a formal snapshot at the specified epoch (or the latest available epoch if none is
79    /// specified).
80    pub(super) async fn new(
81        snapshot_args: FormalSnapshotArgs,
82        connection_args: StorageConnectionArgs,
83    ) -> anyhow::Result<Self> {
84        // Connect to the formal snapshot source.
85        let store: Arc<dyn Storage + Send + Sync + 'static> = if let Some(bucket) = snapshot_args.s3
86        {
87            info!(bucket, "S3 storage");
88            AmazonS3Builder::from_env()
89                .with_client_options(connection_args.into())
90                .with_imdsv1_fallback()
91                .with_bucket_name(bucket)
92                .build()
93                .map(Arc::new)?
94        } else if let Some(bucket) = snapshot_args.gcs {
95            info!(bucket, "GCS storage");
96            GoogleCloudStorageBuilder::from_env()
97                .with_client_options(connection_args.into())
98                .with_bucket_name(bucket)
99                .build()
100                .map(Arc::new)?
101        } else if let Some(container) = snapshot_args.azure {
102            info!(container, "Azure storage");
103            MicrosoftAzureBuilder::from_env()
104                .with_client_options(connection_args.into())
105                .with_container_name(container)
106                .build()
107                .map(Arc::new)?
108        } else if let Some(endpoint) = snapshot_args.http {
109            info!(endpoint = %endpoint, "HTTP storage");
110            HttpStorage::new(endpoint, connection_args).map(Arc::new)?
111        } else if let Some(path) = snapshot_args.path {
112            info!(path = %path.display(), "Directory storage");
113            LocalFileSystem::new_with_prefix(path).map(Arc::new)?
114        } else {
115            bail!("No formal snapshot source provided");
116        };
117
118        // Fetch details about all the epochs it has available, and use that to confirm and
119        // validate the epoch to restore from (it is available and complete at the source).
120        let root_manifest = RootManifest::read(
121            store
122                .get("MANIFEST".into())
123                .await
124                .context("Failed to fetch root manifest")?
125                .as_ref(),
126        )?;
127
128        // If an epoch has not been specified, restore from the latest available epoch.
129        let epoch = snapshot_args
130            .epoch
131            .or_else(|| root_manifest.latest())
132            .context("No epochs available in the snapshot store")?;
133
134        ensure!(
135            root_manifest.contains(epoch),
136            "Requested epoch {epoch} is not available in the snapshot store",
137        );
138
139        let is_complete = store
140            .get(format!("epoch_{epoch}/_SUCCESS").into())
141            .await
142            .is_ok();
143
144        ensure!(is_complete, "Snapshot for epoch {epoch} is not complete");
145
146        info!(epoch, "Connected to valid formal snapshot");
147
148        // Use the remote store to associate the epoch with a watermark pointing to its last
149        // transaction.
150        let client = StoreIngestionClient::new(
151            HttpBuilder::new()
152                .with_url(snapshot_args.remote_store_url.to_string())
153                .with_client_options(ClientOptions::new().with_allow_http(true))
154                .build()
155                .map(Arc::new)
156                .context("Failed to connect to remote checkpoint store")?,
157        );
158
159        let end_of_epoch_checkpoints: Vec<_> = client
160            .end_of_epoch_checkpoints()
161            .await
162            .context("Failed to fetch end-of-epoch checkpoints")?;
163
164        let checkpoint = end_of_epoch_checkpoints
165            .get(epoch as usize)
166            .cloned()
167            .with_context(|| format!("Cannot find end-of-epoch checkpoint for epoch {epoch}"))?;
168
169        let Checkpoint { summary, .. } = client
170            .checkpoint(checkpoint)
171            .await
172            .context("Failed to fetch end-of-epoch checkpoint")?;
173
174        ensure!(
175            summary.epoch == epoch,
176            "End-of-epoch checkpoint {checkpoint} does not belong to epoch {epoch}",
177        );
178
179        let watermark = Watermark {
180            epoch_hi_inclusive: epoch,
181            checkpoint_hi_inclusive: summary.sequence_number,
182            tx_hi: summary.network_total_transactions,
183            timestamp_ms_hi_inclusive: summary.timestamp_ms,
184        };
185
186        info!(?watermark, "Anchored formal snapshot");
187
188        Ok(Self {
189            source: store,
190            watermark,
191        })
192    }
193
194    /// The watermark this snapshot was taken from.
195    pub(super) fn watermark(&self) -> Watermark {
196        self.watermark
197    }
198
199    /// Load the manifest for this snapshot, detailing all its files.
200    pub(super) async fn manifest(&self) -> anyhow::Result<Bytes> {
201        let path = format!("epoch_{}/MANIFEST", self.watermark.epoch_hi_inclusive);
202
203        self.source
204            .get(path.into())
205            .await
206            .context("Failed to fetch epoch manifest")
207    }
208
209    /// Load a file from this snapshot, given its metadata.
210    pub(super) async fn file(&self, metadata: &FileMetadata) -> anyhow::Result<Bytes> {
211        let ext = match metadata.file_type {
212            FileType::Object => "obj",
213            FileType::Reference => "ref",
214        };
215
216        let path = format!(
217            "epoch_{}/{}_{}.{ext}",
218            self.watermark.epoch_hi_inclusive, metadata.bucket, metadata.partition
219        );
220
221        self.source
222            .get(path.into())
223            .await
224            .context("Failed to fetch object file")
225    }
226}
