25#[derive(clap::Args, Clone, Debug)]
26#[group(required = true)]
27pub struct FormalSnapshotArgs {
28    /// Fetch formal snapshot from AWS S3. Provide the bucket name or endpoint-and-bucket.
29    /// (env: AWS_ENDPOINT, AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY, AWS_DEFAULT_REGION)
30    #[arg(long, group = "source")]
31    pub s3: Option<String>,
32
33    /// Fetch formal snapshot from Google Cloud Storage. Provide the bucket name.
34    /// (env: GOOGLE_SERVICE_ACCOUNT_PATH).
35    #[arg(long, group = "source")]
36    pub gcs: Option<String>,
37
38    /// Fetch formal snapshot from Azure Blob Storage. Provide the container name.
39    /// (env: AZURE_STORAGE_ACCOUNT_NAME, AZURE_STORAGE_ACCESS_KEY)
40    #[arg(long, group = "source")]
41    pub azure: Option<String>,
42
43    /// Fetch formal snapshot from a generic HTTP endpoint.
44    #[arg(long, group = "source")]
45    pub http: Option<Url>,
46
47    /// Fetch formal snapshot from local filesystem. Provide the path to the snapshot directory.
48    #[arg(long, group = "source")]
49    pub path: Option<PathBuf>,
50
51    /// URL of the remote checkpoint store. Used to fetch the mapping between epochs and
52    /// checkpoints.
53    #[arg(long)]
54    pub remote_store_url: Url,
55
56    /// The epoch to restore from. Restores from the latest epoch if not specified.
57    #[arg(long)]
58    pub epoch: Option<u64>,
59}
60
61/// Interface over a formal snapshot at a specific epoch.
62#[derive(Clone)]
63pub(super) struct FormalSnapshot {
64    /// The source of files related to the formal snapshot.
65    source: Arc<dyn Storage + Send + Sync + 'static>,
66
67    /// The point in time this snapshot is from.
68    watermark: Watermark,
69}
70
71impl FormalSnapshot {
72    /// Establish a connection to a formal snapshot source, and validate that it can serve data for
73    /// a formal snapshot at the specified epoch (or the latest available epoch if none is
74    /// specified).
75    pub(super) async fn new(
76        snapshot_args: FormalSnapshotArgs,
77        connection_args: StorageConnectionArgs,
78    ) -> anyhow::Result<Self> {
79        // Connect to the formal snapshot source.
80        let store: Arc<dyn Storage + Send + Sync + 'static> = if let Some(bucket) = snapshot_args.s3
81        {
82            info!(bucket, "S3 storage");
83            AmazonS3Builder::from_env()
84                .with_client_options(connection_args.into())
85                .with_imdsv1_fallback()
86                .with_bucket_name(bucket)
87                .build()
88                .map(Arc::new)?
89        } else if let Some(bucket) = snapshot_args.gcs {
90            info!(bucket, "GCS storage");
91            GoogleCloudStorageBuilder::from_env()
92                .with_client_options(connection_args.into())
93                .with_bucket_name(bucket)
94                .build()
95                .map(Arc::new)?
96        } else if let Some(container) = snapshot_args.azure {
97            info!(container, "Azure storage");
98            MicrosoftAzureBuilder::from_env()
99                .with_client_options(connection_args.into())
100                .with_container_name(container)
101                .build()
102                .map(Arc::new)?
103        } else if let Some(endpoint) = snapshot_args.http {
104            info!(endpoint = %endpoint, "HTTP storage");
105            HttpStorage::new(endpoint, connection_args).map(Arc::new)?
106        } else if let Some(path) = snapshot_args.path {
107            info!(path = %path.display(), "Directory storage");
108            LocalFileSystem::new_with_prefix(path).map(Arc::new)?
109        } else {
110            bail!("No formal snapshot source provided");
111        };
112
113        // Fetch details about all the epochs it has available, and use that to confirm and
114        // validate the epoch to restore from (it is available and complete at the source).
115        let root_manifest = RootManifest::read(
116            store
117                .get("MANIFEST".into())
118                .await
119                .context("Failed to fetch root manifest")?
120                .as_ref(),
121        )?;
122
123        // If an epoch has not been specified, restore from the latest available epoch.
124        let epoch = snapshot_args
125            .epoch
126            .or_else(|| root_manifest.latest())
127            .context("No epochs available in the snapshot store")?;
128
129        ensure!(
130            root_manifest.contains(epoch),
131            "Requested epoch {epoch} is not available in the snapshot store",
132        );
133
134        let is_complete = store
135            .get(format!("epoch_{epoch}/_SUCCESS").into())
136            .await
137            .is_ok();
138
139        ensure!(is_complete, "Snapshot for epoch {epoch} is not complete");
140
141        info!(epoch, "Connected to valid formal snapshot");
142
143        // Use the remote store to associate the epoch with a watermark pointing to its last
144        // transaction.
145        let client = RemoteIngestionClient::new(snapshot_args.remote_store_url)
146            .context("Failed to connect to remote checkpoint store")?;
147
148        let end_of_epoch_checkpoints: Vec<_> = client
149            .end_of_epoch_checkpoints()
150            .await
151            .context("Failed to fetch end-of-epoch checkpoints")?
152            .json()
153            .await
154            .context("Failed to parse end-of-epoch checkpoints")?;
155
156        let checkpoint = end_of_epoch_checkpoints
157            .get(epoch as usize)
158            .cloned()
159            .with_context(|| format!("Cannot find end-of-epoch checkpoint for epoch {epoch}"))?;
160
161        let CheckpointData {
162            checkpoint_summary, ..
163        } = Blob::from_bytes(
164            &client
165                .checkpoint(checkpoint)
166                .await
167                .context("Failed to fetch end-of-epoch checkpoint")?
168                .bytes()
169                .await
170                .context("Failed to read end-of-epoch checkpoint bytes")?,
171        )
172        .context("Failed to deserialize end-of-epoch checkpoint")?;
173
174        ensure!(
175            checkpoint_summary.epoch == epoch,
176            "End-of-epoch checkpoint {checkpoint} does not belong to epoch {epoch}",
177        );
178
179        let watermark = Watermark {
180            epoch_hi_inclusive: epoch,
181            checkpoint_hi_inclusive: checkpoint_summary.sequence_number,
182            tx_hi: checkpoint_summary.network_total_transactions,
183            timestamp_ms_hi_inclusive: checkpoint_summary.timestamp_ms,
184        };
185
186        info!(?watermark, "Anchored formal snapshot");
187
188        Ok(Self {
189            source: store,
190            watermark,
191        })
192    }
193
194    /// The watermark this snapshot was taken from.
195    pub(super) fn watermark(&self) -> Watermark {
196        self.watermark
197    }
198
199    /// Load the manifest for this snapshot, detailing all its files.
200    pub(super) async fn manifest(&self) -> anyhow::Result<Bytes> {
201        let path = format!("epoch_{}/MANIFEST", self.watermark.epoch_hi_inclusive);
202
203        self.source
204            .get(path.into())
205            .await
206            .context("Failed to fetch epoch manifest")
207    }
208
209    /// Load a file from this snapshot, given its metadata.
210    pub(super) async fn file(&self, metadata: &FileMetadata) -> anyhow::Result<Bytes> {
211        let ext = match metadata.file_type {
212            FileType::Object => "obj",
213            FileType::Reference => "ref",
214        };
215
216        let path = format!(
217            "epoch_{}/{}_{}.{ext}",
218            self.watermark.epoch_hi_inclusive, metadata.bucket, metadata.partition
219        );
220
221        self.source
222            .get(path.into())
223            .await
224            .context("Failed to fetch object file")
225    }
226}
