23const EPOCH_MANIFEST_MAGIC: u32 = 0x00C0FFEE;
24const OBJECT_FILE_MAGIC: u32 = 0x00B7EC75;
25const DIGEST_LEN: usize = Sha3_256::OUTPUT_SIZE;

27/// The root of the formal snapshot store contains JSON-encoded manifest that mentions all the
28/// epochs that have a formal snapshot available.
29#[derive(Deserialize)]
30pub(super) struct RootManifest {
31    available_epochs: Vec<u64>,
32}

34/// The Epoch Manifest is stored in a custom binary format, starting with a magic number, ending
35/// with a digest, and the contents in between is BCS-serialized, with the following structure.
36#[derive(Deserialize, Debug)]
37pub(super) enum EpochManifest {
38    V1(EpochManifestV1),
39}

87impl RootManifest {
88    pub(super) fn read(data: &[u8]) -> anyhow::Result<Self> {
89        serde_json::from_slice(data).context("Failed to parse root manifest")
90    }
91
92    /// Returns the latest epoch available, or `None` if there are no epochs.
93    pub(super) fn latest(&self) -> Option<u64> {
94        self.available_epochs.iter().copied().max()
95    }
96
97    /// Decides whether the given epoch is present in the manifest or not.
98    pub(super) fn contains(&self, epoch: u64) -> bool {
99        self.available_epochs.contains(&epoch)
100    }
101}

103impl EpochManifest {
104    pub(super) fn read(data: &[u8]) -> anyhow::Result<Self> {
105        const MAGIC_LEN: usize = size_of_val(&EPOCH_MANIFEST_MAGIC);
106
107        ensure!(
108            data.len() >= MAGIC_LEN + DIGEST_LEN,
109            "Epoch manifest too short",
110        );
111
112        let mut cursor = Cursor::new(data);
113
114        // Check Magic Number
115        let mut magic = [0u8; MAGIC_LEN];
116        cursor.read_exact(&mut magic)?;
117
118        ensure!(
119            u32::from_be_bytes(magic) == EPOCH_MANIFEST_MAGIC,
120            "Not an epoch manifest"
121        );
122
123        // Check Digest
124        cursor.seek(SeekFrom::End(-(DIGEST_LEN as i64)))?;
125        let end = cursor.position() as usize;
126        let mut digest = [0u8; DIGEST_LEN];
127        cursor.read_exact(&mut digest)?;
128
129        let mut hasher = Sha3_256::new();
130        hasher.update(&data[..end]);
131
132        ensure!(
133            hasher.finalize().digest == digest,
134            "Epoch manifest digest mismatch",
135        );
136
137        // Deserialize
138        bcs::from_bytes(&data[MAGIC_LEN..end]).context("Failed to deserialize epoch manifest")
139    }
140
141    pub(super) fn metadata(&self) -> &[FileMetadata] {
142        match self {
143            EpochManifest::V1(m) => &m.metadata,
144        }
145    }
146}

148impl FileCompression {
149    pub(super) fn reader(self, data: &[u8]) -> anyhow::Result<Box<dyn Read + '_>> {
150        Ok(match self {
151            FileCompression::None => Box::new(Cursor::new(data)),
152            FileCompression::Zstd => Box::new(Decoder::new(Cursor::new(data))?),
153        })
154    }
155}

157impl LiveObjects {
158    pub(super) fn read(bytes: &[u8], metadata: &FileMetadata) -> anyhow::Result<Self> {
159        const MAGIC_LEN: usize = size_of_val(&OBJECT_FILE_MAGIC);
160
161        let mut read = metadata.compression.reader(bytes)?;
162
163        let mut magic = [0u8; MAGIC_LEN];
164        read.read_exact(&mut magic)?;
165
166        ensure!(
167            u32::from_be_bytes(magic) == OBJECT_FILE_MAGIC,
168            "Not an object file"
169        );
170
171        let mut objects = vec![];
172        while let Ok(len) = read.read_varint::<u64>() {
173            if len == 0 {
174                break;
175            }
176
177            let mut e = vec![0u8; 1];
178            read.read_exact(&mut e)?;
179            let encoding = BlobEncoding::try_from(e[0])
180                .with_context(|| format!("Invalid encoding in object file: {}", e[0]))?;
181
182            let mut data = vec![0u8; len as usize];
183            read.read_exact(&mut data)?;
184
185            let object = Blob { data, encoding }
186                .decode()
187                .context("Failed to decode object from blob")?;
188
189            if let LiveObject::Normal(object) = object {
190                objects.push(object);
191            }
192        }
193
194        Ok(Self {
195            bucket: metadata.bucket,
196            partition: metadata.partition,
197            objects,
198        })
199    }
