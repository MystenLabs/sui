4use std::sync::Arc;
5
6use prometheus::Histogram;
7use prometheus::HistogramVec;
8use prometheus::IntCounter;
9use prometheus::IntCounterVec;
10use prometheus::IntGauge;
11use prometheus::Registry;
12use prometheus::register_histogram_vec_with_registry;
13use prometheus::register_histogram_with_registry;
14use prometheus::register_int_counter_vec_with_registry;
15use prometheus::register_int_counter_with_registry;
16use prometheus::register_int_gauge_with_registry;
17
18/// Histogram buckets for the distribution of object file fetching latencies.
19const LATENCY_SEC_BUCKETS: &[f64] = &[
20    0.001, 0.002, 0.005, 0.01, 0.02, 0.05, 0.1, 0.2, 0.5, 1.0, 2.0, 5.0, 10.0, 20.0, 50.0, 100.0,
21    200.0, 500.0, 1000.0,
22];
23
24pub(super) struct RestorerMetrics {
25    pub(crate) objects_fetch_latency: Histogram,
26    pub(crate) total_objects_fetch_retries: IntCounter,
27    pub(crate) total_bytes_fetched: IntCounter,
28    pub(crate) total_partitions_fetched: IntCounter,
29    pub(crate) total_partitions_skipped: IntCounter,
30    pub(crate) total_partitions_broadcast: IntCounter,
31    pub(crate) total_partitions: IntGauge,
32    pub(crate) total_partitions_received: IntCounterVec,
33    pub(crate) total_objects_received: IntCounterVec,
34    pub(crate) total_partitions_restored: IntCounterVec,
35    pub(crate) total_objects_restored: IntCounterVec,
36    pub(crate) worker_partition_latency: HistogramVec,
37}
38
39impl RestorerMetrics {
40    pub(super) fn new(prefix: Option<&str>, registry: &Registry) -> Arc<Self> {
41        let prefix = prefix.unwrap_or("restorer");
42        let name = |n| format!("{prefix}_{n}");
43
44        Arc::new(Self {
45            objects_fetch_latency: register_histogram_with_registry!(
46                name("objects_fetch_latency"),
47                "Time taken to fetch an object file from a formal snapshot, in seconds",
48                LATENCY_SEC_BUCKETS.to_vec(),
49                registry,
50            )
51            .unwrap(),
52
53            total_objects_fetch_retries: register_int_counter_with_registry!(
54                name("total_objects_fetch_retries"),
55                "Total number of retries when fetching object files from a formal snapshot",
56                registry,
57            )
58            .unwrap(),
59
60            total_bytes_fetched: register_int_counter_with_registry!(
61                name("total_bytes_fetched"),
62                "Total number of bytes fetched from a formal snapshot",
63                registry,
64            )
65            .unwrap(),
66
67            total_partitions_fetched: register_int_counter_with_registry!(
68                name("total_partitions_fetched"),
69                "Total number of object file partitions fetched from a formal snapshot",
70                registry,
71            )
72            .unwrap(),
73
74            total_partitions_skipped: register_int_counter_with_registry!(
75                name("total_partitions_skipped"),
76                "Total number of object file partitions skipped (because all pipelines had already restored them)",
77                registry,
78            )
79            .unwrap(),
80
81            total_partitions_broadcast: register_int_counter_with_registry!(
82                name("total_partitions_broadcast"),
83                "Total number of object file partitions broadcast to all pipelines",
84                registry,
85            )
86            .unwrap(),
87
88            total_partitions: register_int_gauge_with_registry!(
89                name("total_partitions"),
90                "The number of object file partitions to be fetched from a formal snapshot",
91                registry,
92            )
93            .unwrap(),
94
95            total_partitions_received: register_int_counter_vec_with_registry!(
96                name("total_partitions_received"),
97                "Total number of object file partitions received, per pipeline",
98                &["pipeline"],
99                registry,
100            )
101            .unwrap(),
102
103            total_objects_received: register_int_counter_vec_with_registry!(
104                name("total_objects_received"),
105                "Total number of objects received, per pipeline",
106                &["pipeline"],
107                registry,
108            )
109            .unwrap(),
110
111            total_partitions_restored: register_int_counter_vec_with_registry!(
112                name("total_partitions_restored"),
113                "Total number of partitions restored, per pipeline",
114                &["pipeline"],
115                registry,
116            )
117            .unwrap(),
118
119            total_objects_restored: register_int_counter_vec_with_registry!(
120                name("total_objects_restored"),
121                "Total number of objects restored, per pipeline",
122                &["pipeline"],
123                registry,
124            )
125            .unwrap(),
126
127            worker_partition_latency: register_histogram_vec_with_registry!(
128                name("worker_partition_latency"),
129                "Time taken by a worker to process a partition, in seconds",
130                &["pipeline"],
131                LATENCY_SEC_BUCKETS.to_vec(),
132                registry,
133            )
134            .unwrap(),
135        })
136    }
137}
