4use std::sync::Arc;
5
6use prometheus::{
7    Histogram, HistogramVec, IntCounter, IntCounterVec, IntGauge, Registry,
8    register_histogram_vec_with_registry, register_histogram_with_registry,
9    register_int_counter_vec_with_registry, register_int_counter_with_registry,
10    register_int_gauge_with_registry,
11};
12
13/// Histogram buckets for the distribution of object file fetching latencies.
14const LATENCY_SEC_BUCKETS: &[f64] = &[
15    0.001, 0.002, 0.005, 0.01, 0.02, 0.05, 0.1, 0.2, 0.5, 1.0, 2.0, 5.0, 10.0, 20.0, 50.0, 100.0,
16    200.0, 500.0, 1000.0,
17];
18
19pub(super) struct RestorerMetrics {
20    pub(crate) objects_fetch_latency: Histogram,
21    pub(crate) total_objects_fetch_retries: IntCounter,
22    pub(crate) total_bytes_fetched: IntCounter,
23    pub(crate) total_partitions_fetched: IntCounter,
24    pub(crate) total_partitions_skipped: IntCounter,
25    pub(crate) total_partitions_broadcast: IntCounter,
26    pub(crate) total_partitions: IntGauge,
27    pub(crate) total_partitions_received: IntCounterVec,
28    pub(crate) total_objects_received: IntCounterVec,
29    pub(crate) total_partitions_restored: IntCounterVec,
30    pub(crate) total_objects_restored: IntCounterVec,
31    pub(crate) worker_partition_latency: HistogramVec,
32}
33
34impl RestorerMetrics {
35    pub(super) fn new(prefix: Option<&str>, registry: &Registry) -> Arc<Self> {
36        let prefix = prefix.unwrap_or("restorer");
37        let name = |n| format!("{prefix}_{n}");
38
39        Arc::new(Self {
40            objects_fetch_latency: register_histogram_with_registry!(
41                name("objects_fetch_latency"),
42                "Time taken to fetch an object file from a formal snapshot, in seconds",
43                LATENCY_SEC_BUCKETS.to_vec(),
44                registry,
45            )
46            .unwrap(),
47
48            total_objects_fetch_retries: register_int_counter_with_registry!(
49                name("total_objects_fetch_retries"),
50                "Total number of retries when fetching object files from a formal snapshot",
51                registry,
52            )
53            .unwrap(),
54
55            total_bytes_fetched: register_int_counter_with_registry!(
56                name("total_bytes_fetched"),
57                "Total number of bytes fetched from a formal snapshot",
58                registry,
59            )
60            .unwrap(),
61
62            total_partitions_fetched: register_int_counter_with_registry!(
63                name("total_partitions_fetched"),
64                "Total number of object file partitions fetched from a formal snapshot",
65                registry,
66            )
67            .unwrap(),
68
69            total_partitions_skipped: register_int_counter_with_registry!(
70                name("total_partitions_skipped"),
71                "Total number of object file partitions skipped (because all pipelines had already restored them)",
72                registry,
73            )
74            .unwrap(),
75
76            total_partitions_broadcast: register_int_counter_with_registry!(
77                name("total_partitions_broadcast"),
78                "Total number of object file partitions broadcast to all pipelines",
79                registry,
80            )
81            .unwrap(),
82
83            total_partitions: register_int_gauge_with_registry!(
84                name("total_partitions"),
85                "The number of object file partitions to be fetched from a formal snapshot",
86                registry,
87            )
88            .unwrap(),
89
90            total_partitions_received: register_int_counter_vec_with_registry!(
91                name("total_partitions_received"),
92                "Total number of object file partitions received, per pipeline",
93                &["pipeline"],
94                registry,
95            )
96            .unwrap(),
97
98            total_objects_received: register_int_counter_vec_with_registry!(
99                name("total_objects_received"),
100                "Total number of objects received, per pipeline",
101                &["pipeline"],
102                registry,
103            )
104            .unwrap(),
105
106            total_partitions_restored: register_int_counter_vec_with_registry!(
107                name("total_partitions_restored"),
108                "Total number of partitions restored, per pipeline",
109                &["pipeline"],
110                registry,
111            )
112            .unwrap(),
113
114            total_objects_restored: register_int_counter_vec_with_registry!(
115                name("total_objects_restored"),
116                "Total number of objects restored, per pipeline",
117                &["pipeline"],
118                registry,
119            )
120            .unwrap(),
121
122            worker_partition_latency: register_histogram_vec_with_registry!(
123                name("worker_partition_latency"),
124                "Time taken by a worker to process a partition, in seconds",
125                &["pipeline"],
126                LATENCY_SEC_BUCKETS.to_vec(),
127                registry,
128            )
129            .unwrap(),
130        })
131    }
132}
