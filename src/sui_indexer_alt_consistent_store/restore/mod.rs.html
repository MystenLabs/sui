42/// Trait implemented by processors that support being restored from live objects in a formal
43/// snapshot.
44pub(crate) trait Restore<S: crate::store::Schema>: Processor {
45    /// How much concurrency to use when processing live objects.
46    const FANOUT: usize = 10;
47
48    fn restore(schema: &S, object: &Object, batch: &mut rocksdb::WriteBatch) -> anyhow::Result<()>;
49}
50
51#[derive(clap::Args, Clone, Debug)]
52pub struct RestoreArgs {
53    /// Number of object files to download concurrently
54    #[arg(long, default_value_t = Self::default().object_file_concurrency)]
55    object_file_concurrency: usize,
56
57    /// Maximum size of the backlog of object files waiting to be processed by one worker.
58    #[arg(long, default_value_t = Self::default().object_file_buffer_size)]
59    object_file_buffer_size: usize,
60}

62/// A service for restoring pipelines from live objects in a formal snapshot.
63pub(crate) struct Restorer<S: Schema> {
64    /// The RocksDB database to restore into.
65    db: Arc<Db>,
66
67    /// A schema over the database, to provide structured access to its contents.
68    schema: Arc<S>,
69
70    /// A source of live objects to restore from.
71    snapshot: FormalSnapshot,
72
73    /// Metrics related to the restoration process.
74    metrics: Arc<RestorerMetrics>,
75
76    /// Channels to send live object partitions down, one per pipeline being restored.
77    restore_tx: BTreeMap<String, mpsc::Sender<Arc<LiveObjects>>>,
78
79    /// Handles for tasks spawned by the restorer. They return `Ok(_)` if they complete
80    /// successfully, or `Err(_)` otherwise (they encountered an error, or they were cancelled).
81    handles: Vec<JoinHandle<Result<(), ()>>>,
82
83    /// Number of object files to download concurrenctly
84    object_file_concurrency: usize,
85
86    /// Maximum size of the backlog of object files waiting to be processed by one worker.
87    object_file_buffer_size: usize,
88
89    /// Cancellation token shared among all continuous tasks in the service.
90    cancel: CancellationToken,
91}
92
93/// Internal type used by tasks to propagate errors or shutdown signals.
94#[derive(thiserror::Error, Debug)]
95enum Break {
96    #[error("Shutdown received")]
97    Cancel,
98
99    #[error(transparent)]
100    Err(#[from] anyhow::Error),
101}

103impl<S: Schema + Send + Sync + 'static> Restorer<S> {
104    /// Create a new instance of the `Restorer`, configured to restore into the database at `path`.
105    ///
106    /// `formal_snapshot_args` describes where to load the formal snapshot from, `connection_args`
107    /// controls how to connect to it, `restore_args` controls the restoration process itself, and
108    /// `config` configures the RocksDB database.
109    async fn new(
110        path: impl AsRef<Path>,
111        formal_snapshot_args: FormalSnapshotArgs,
112        connection_args: StorageConnectionArgs,
113        restore_args: RestoreArgs,
114        config: DbConfig,
115        metrics_prefix: Option<&str>,
116        registry: &Registry,
117        cancel: CancellationToken,
118    ) -> anyhow::Result<Self> {
119        let RestoreArgs {
120            object_file_concurrency,
121            object_file_buffer_size,
122        } = restore_args;
123
124        let metrics = RestorerMetrics::new(metrics_prefix, registry);
125
126        let snapshot = FormalSnapshot::new(formal_snapshot_args, connection_args)
127            .await
128            .context("Failed to connect to formal snapshot source")?;
129
130        // This database will not be read from so we don't need to worry about snapshots.
131        let snapshots = 0;
132        let options: rocksdb::Options = config.into();
133        let db = Db::open(path, options.clone(), snapshots, S::cfs(&options))
134            .map(Arc::new)
135            .context("Failed to open database")?;
136
137        let schema = S::open(&db)
138            .map(Arc::new)
139            .context("Failed to open schema")?;
140
141        Ok(Self {
142            db,
143            schema,
144            snapshot,
145            metrics,
146            restore_tx: BTreeMap::new(),
147            handles: vec![],
148            object_file_concurrency,
149            object_file_buffer_size,
150            cancel,
151        })
152    }
153
154    /// Register and start a new restoration pipeline implemented by `R`. Although their tasks have
155    /// started, they will be idle until the restorer as a whole is run, and starts to fetch
156    /// objects from the formal snapshot.
157    fn restorer<R: Restore<S>>(&mut self) -> anyhow::Result<()> {
158        let (tx, rx) = mpsc::channel(self.object_file_buffer_size);
159        ensure!(
160            self.restore_tx.insert(R::NAME.to_string(), tx).is_none(),
161            "Pipeline {} already registered for restoration",
162            R::NAME,
163        );
164
165        let watermark = self.snapshot.watermark();
166        self.db.restore_at(R::NAME, watermark)?;
167
168        self.handles.push(worker::<S, R>(
169            rx,
170            self.db.clone(),
171            self.schema.clone(),
172            self.metrics.clone(),
173            self.cancel.clone(),
174        ));
175
176        Ok(())
177    }
178
179    /// Start restoring live objects across all registered pipelines. The service will run until it
180    /// can confirm that every registered pipeline has been fully restored, at which point, it will
181    /// clean up the restoration state and set the watermark for the restored pipeline.
182    fn run(mut self) -> JoinHandle<()> {
183        // Remember the pipelines being restored for the clean-up process.
184        let pipelines: Vec<_> = self.restore_tx.keys().cloned().collect();
185        info!(?pipelines, "Starting restoration");
186
187        let broadcaster_h = broadcaster(
188            self.object_file_concurrency,
189            self.restore_tx,
190            self.db.clone(),
191            self.snapshot,
192            self.metrics,
193            self.cancel,
194        );
195
196        self.handles.push(broadcaster_h);
197
198        tokio::spawn(async move {
199            // Wait for the broadcaster and all workers to wind down gracefully, and then clean up.
200            if future::join_all(self.handles)
201                .await
202                .into_iter()
203                .any(|r| matches!(r, Err(_) | Ok(Err(_))))
204            {
205                warn!("Restoration did not complete successfully, not updating watermarks");
206                return;
207            };
208
209            // Mark each pipeline as restored, setting its watermark.
210            for pipeline in pipelines {
211                if let Err(e) = self.db.complete_restore(&pipeline) {
212                    warn!(pipeline, "Failed to finalize restoration: {e:#}");
213                } else {
214                    info!(pipeline, "Restoration complete");
215                }
216            }
217        })
218    }
219}
220
221impl Default for RestoreArgs {
222    fn default() -> Self {
223        Self {
224            object_file_concurrency: 25,
225            object_file_buffer_size: 25,
226        }
227    }
228}

230/// Set-up and run the Restorer, using the provided arguments (expected to be extracted from the
231/// command-line). The service will run until restoration complete, or until the cancellation token
232/// is triggered.
233///
234/// `path` is the path to the RocksDB database to restore into. It will be created if it does not
235/// exist. `formal_snapshot_args` describes the formal snapshot source, `connection_args` controls
236/// how to connect to it, `restore_args` controls the restoration process itself, and `config`
237/// configures the RocksDB database.
238pub async fn start_restorer(
239    path: impl AsRef<Path>,
240    formal_snapshot_args: FormalSnapshotArgs,
241    connection_args: StorageConnectionArgs,
242    restore_args: RestoreArgs,
243    mut pipelines: BTreeSet<String>,
244    config: DbConfig,
245    registry: &Registry,
246    cancel: CancellationToken,
247) -> anyhow::Result<JoinHandle<()>> {
248    let mut restorer: Restorer<crate::Schema> = Restorer::new(
249        path,
250        formal_snapshot_args,
251        connection_args,
252        restore_args,
253        config,
254        Some("restorer"),
255        registry,
256        cancel.child_token(),
257    )
258    .await?;
259
260    macro_rules! add_restorer {
261        ($handler:ty) => {
262            if pipelines.remove(<$handler as Processor>::NAME) {
263                restorer.restorer::<$handler>()?;
264            }
265        };
266    }
267
268    add_restorer!(Balances);
269    add_restorer!(ObjectByOwner);
270    add_restorer!(ObjectByType);
271
272    ensure!(
273        pipelines.is_empty(),
274        "Unknown pipelines to restore: {pipelines:?}"
275    );
276
277    Ok(restorer.run())
278}
