41/// Trait implemented by processors that support being restored from live objects in a formal
42/// snapshot.
43pub(crate) trait Restore<S: crate::store::Schema>: Processor {
44    /// How much concurrency to use when processing live objects.
45    const FANOUT: usize = 10;
46
47    fn restore(schema: &S, object: &Object, batch: &mut rocksdb::WriteBatch) -> anyhow::Result<()>;
48}
49
50#[derive(clap::Args, Clone, Debug)]
51pub struct RestoreArgs {
52    /// Number of object files to download concurrently
53    #[arg(long, default_value_t = Self::default().object_file_concurrency)]
54    object_file_concurrency: usize,
55
56    /// Maximum size of the backlog of object files waiting to be processed by one worker.
57    #[arg(long, default_value_t = Self::default().object_file_buffer_size)]
58    object_file_buffer_size: usize,
59}

61/// A service for restoring pipelines from live objects in a formal snapshot.
62pub(crate) struct Restorer<S: Schema> {
63    /// The RocksDB database to restore into.
64    db: Arc<Db>,
65
66    /// A schema over the database, to provide structured access to its contents.
67    schema: Arc<S>,
68
69    /// A source of live objects to restore from.
70    snapshot: FormalSnapshot,
71
72    /// Metrics related to the restoration process.
73    metrics: Arc<RestorerMetrics>,
74
75    /// Channels to send live object partitions down, one per pipeline being restored.
76    restore_tx: BTreeMap<String, mpsc::Sender<Arc<LiveObjects>>>,
77
78    /// Services spawned by the restorer, for individual pipelines.
79    workers: Vec<Service>,
80
81    /// Number of object files to download concurrenctly
82    object_file_concurrency: usize,
83
84    /// Maximum size of the backlog of object files waiting to be processed by one worker.
85    object_file_buffer_size: usize,
86}

88pub struct Finalizer {
89    db: Arc<Db>,
90    pipelines: Vec<String>,

93impl<S: Schema + Send + Sync + 'static> Restorer<S> {
94    /// Create a new instance of the `Restorer`, configured to restore into the database at `path`.
95    ///
96    /// `formal_snapshot_args` describes where to load the formal snapshot from, `connection_args`
97    /// controls how to connect to it, `restore_args` controls the restoration process itself, and
98    /// `config` configures the RocksDB database.
99    async fn new(
100        path: impl AsRef<Path>,
101        formal_snapshot_args: FormalSnapshotArgs,
102        connection_args: StorageConnectionArgs,
103        restore_args: RestoreArgs,
104        config: DbConfig,
105        metrics_prefix: Option<&str>,
106        registry: &Registry,
107    ) -> anyhow::Result<Self> {
108        let RestoreArgs {
109            object_file_concurrency,
110            object_file_buffer_size,
111        } = restore_args;
112
113        let metrics = RestorerMetrics::new(metrics_prefix, registry);
114
115        let snapshot = FormalSnapshot::new(formal_snapshot_args, connection_args)
116            .await
117            .context("Failed to connect to formal snapshot source")?;
118
119        // This database will not be read from so we don't need to worry about snapshots.
120        let snapshots = 0;
121        let options: rocksdb::Options = config.into();
122        let db = Db::open(path, options.clone(), snapshots, S::cfs(&options))
123            .map(Arc::new)
124            .context("Failed to open database")?;
125
126        let schema = S::open(&db)
127            .map(Arc::new)
128            .context("Failed to open schema")?;
129
130        Ok(Self {
131            db,
132            schema,
133            snapshot,
134            metrics,
135            restore_tx: BTreeMap::new(),
136            workers: vec![],
137            object_file_concurrency,
138            object_file_buffer_size,
139        })
140    }
141
142    /// Register and start a new restoration pipeline implemented by `R`. Although their tasks have
143    /// started, they will be idle until the restorer as a whole is run, and starts to fetch
144    /// objects from the formal snapshot.
145    fn restorer<R: Restore<S>>(&mut self) -> anyhow::Result<()> {
146        let (tx, rx) = mpsc::channel(self.object_file_buffer_size);
147        ensure!(
148            self.restore_tx.insert(R::NAME.to_string(), tx).is_none(),
149            "Pipeline {} already registered for restoration",
150            R::NAME,
151        );
152
153        let watermark = self.snapshot.watermark();
154        self.db.restore_at(R::NAME, watermark)?;
155
156        self.workers.push(worker::<S, R>(
157            rx,
158            self.db.clone(),
159            self.schema.clone(),
160            self.metrics.clone(),
161        ));
162
163        Ok(())
164    }
165
166    /// Start restoring live objects across all registered pipelines. The service will run until it
167    /// can confirm that every registered pipeline has been fully restored, at which point, it will
168    /// clean up the restoration state and set the watermark for the restored pipeline.
169    fn run(self) -> (Service, Finalizer) {
170        // Remember the pipelines being restored for the clean-up process.
171        let finalizer = Finalizer {
172            db: self.db.clone(),
173            pipelines: self.restore_tx.keys().cloned().collect(),
174        };
175
176        info!(pipelines = ?finalizer.pipelines, "Starting restoration");
177        let mut service = broadcaster(
178            self.object_file_concurrency,
179            self.restore_tx,
180            self.db,
181            self.snapshot,
182            self.metrics,
183        );
184
185        for worker in self.workers {
186            service = service.merge(worker);
187        }
188
189        (service, finalizer)
190    }
191}
192
193impl Finalizer {
194    pub fn run(self) -> Service {
195        Service::new().spawn_aborting(async move {
196            // Clean up restoration state for each pipeline.
197            for pipeline in self.pipelines {
198                if let Err(e) = self.db.complete_restore(&pipeline) {
199                    warn!(pipeline, "Failed to clear restoration state: {e:#}");
200                } else {
201                    info!(pipeline, "Restoration state cleared");
202                }
203            }
204
205            Ok(())
206        })
207    }
208}

210impl Default for RestoreArgs {
211    fn default() -> Self {
212        Self {
213            object_file_concurrency: 25,
214            object_file_buffer_size: 25,
215        }
216    }
217}

219/// Set-up and run the Restorer, using the provided arguments (expected to be extracted from the
220/// command-line). The service will run until restoration complete.
221///
222/// `path` is the path to the RocksDB database to restore into. It will be created if it does not
223/// exist. `formal_snapshot_args` describes the formal snapshot source, `connection_args` controls
224/// how to connect to it, `restore_args` controls the restoration process itself, and `config`
225/// configures the RocksDB database.
226pub async fn start_restorer(
227    path: impl AsRef<Path>,
228    formal_snapshot_args: FormalSnapshotArgs,
229    connection_args: StorageConnectionArgs,
230    restore_args: RestoreArgs,
231    mut pipelines: BTreeSet<String>,
232    config: DbConfig,
233    registry: &Registry,
234) -> anyhow::Result<(Service, Finalizer)> {
235    let mut restorer: Restorer<crate::Schema> = Restorer::new(
236        path,
237        formal_snapshot_args,
238        connection_args,
239        restore_args,
240        config,
241        Some("restorer"),
242        registry,
243    )
244    .await?;
245
246    macro_rules! add_restorer {
247        ($handler:ty) => {
248            if pipelines.remove(<$handler as Processor>::NAME) {
249                restorer.restorer::<$handler>()?;
250            }
251        };
252    }
253
254    add_restorer!(Balances);
255    add_restorer!(ObjectByOwner);
256    add_restorer!(ObjectByType);
257    add_restorer!(AddressBalances);
258
259    ensure!(
260        pipelines.is_empty(),
261        "Unknown pipelines to restore: {pipelines:?}"
262    );
263
264    Ok(restorer.run())
265}
