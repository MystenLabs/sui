5
6use anyhow::Context as _;
7use sui_futures::service::Service;
8use sui_futures::stream::TrySpawnStreamExt as _;
9use tokio::sync::mpsc;
10use tokio_stream::wrappers::ReceiverStream;
11use tracing::error;
12use tracing::info;
13
14use crate::db::Db;
15use crate::restore::LiveObjects;
16use crate::restore::Restore;
17use crate::restore::RestorerMetrics;
18use crate::store::Schema;

20/// A worker that processes live objects from a single bucket and partition, for a given pipeline.
21pub(super) fn worker<S: Schema + Send + Sync + 'static, R: Restore<S>>(
22    rx: mpsc::Receiver<Arc<LiveObjects>>,
23    db: Arc<Db>,
24    schema: Arc<S>,
25    metrics: Arc<RestorerMetrics>,
26) -> Service {
27    Service::new().spawn_aborting(async move {
28        info!(pipeline = R::NAME, "Starting worker");
29
30        match ReceiverStream::new(rx)
31            .try_for_each_spawned(<R as Restore<S>>::FANOUT, |objects| {
32                let db = db.clone();
33                let schema = schema.clone();
34                let metrics = metrics.clone();
35
36                async move {
37                    info!(
38                        pipeline = R::NAME,
39                        bucket = objects.bucket,
40                        partition = objects.partition,
41                        objects = objects.objects.len(),
42                        "Recv"
43                    );
44
45                    metrics
46                        .total_partitions_received
47                        .with_label_values(&[R::NAME])
48                        .inc();
49
50                    metrics
51                        .total_objects_received
52                        .with_label_values(&[R::NAME])
53                        .inc_by(objects.objects.len() as u64);
54
55                    let _guard = metrics
56                        .worker_partition_latency
57                        .with_label_values(&[R::NAME])
58                        .start_timer();
59
60                    let mut batch = rocksdb::WriteBatch::default();
61                    for object in &objects.objects {
62                        tokio::task::yield_now().await;
63
64                        R::restore(&schema, object, &mut batch).with_context(|| {
65                            format!(
66                                "Error restoring {}, {}, {} in bucket {}, partition {}",
67                                object.id(),
68                                object.version(),
69                                object.digest(),
70                                objects.bucket,
71                                objects.partition
72                            )
73                        })?;
74                    }
75
76                    db.restore(objects.bucket, objects.partition, R::NAME, batch)
77                        .with_context(|| {
78                            format!(
79                                "Failed to write batch for bucket {}, partition {}",
80                                objects.bucket, objects.partition
81                            )
82                        })?;
83
84                    info!(
85                        pipeline = R::NAME,
86                        bucket = objects.bucket,
87                        partition = objects.partition,
88                        "DONE",
89                    );
90
91                    metrics
92                        .total_partitions_restored
93                        .with_label_values(&[R::NAME])
94                        .inc();
95
96                    metrics
97                        .total_objects_restored
98                        .with_label_values(&[R::NAME])
99                        .inc_by(objects.objects.len() as u64);
100
101                    Ok(())
102                }
103            })
104            .await
105        {
106            Ok(()) => {
107                info!(pipeline = R::NAME, "Live objects done, stopping worker");
108                Ok(())
109            }
110
111            Err(e) => {
112                error!(pipeline = R::NAME, "Error from worker: {e:#}");
113                Err(e)
114            }
115        }
116    })
117}
