22#[derive(thiserror::Error, Debug)]
23pub(super) enum Error {
24    #[error("Invalid 'owner': {0:?}")]
25    InvalidOwner(String),
26
27    #[error("Invalid 'coin_type': {0:?}")]
28    InvalidType(String),
29
30    #[error("Missing 'owner'")]
31    MissingOwner,
32
33    #[error("Missing 'coin_type'")]
34    MissingType,
35
36    #[error("Too many requests in batch: {0} (max: {1})")]
37    TooManyRequests(usize, u32),
38}
39
40impl StatusCode for Error {
41    fn code(&self) -> tonic::Code {
42        match self {
43            Error::InvalidOwner(_)
44            | Error::InvalidType(_)
45            | Error::MissingOwner
46            | Error::MissingType
47            | Error::TooManyRequests(_, _) => tonic::Code::InvalidArgument,
48        }
49    }
50}

52pub(super) fn batch_get_balances(
53    state: &State,
54    checkpoint: u64,
55    request: grpc::BatchGetBalancesRequest,
56) -> Result<grpc::BatchGetBalancesResponse, RpcError<Error>> {
57    let config = &state.rpc_config.pagination;
58    let keys = if request.requests.len() > config.max_batch_size as usize {
59        return Err(Error::TooManyRequests(
60            request.requests.len(),
61            state.rpc_config.pagination.max_batch_size,
62        )
63        .into());
64    } else {
65        request
66            .requests
67            .into_iter()
68            .map(key)
69            .collect::<Result<Vec<_>, _>>()?
70    };
71
72    let index = &state.store.schema().balances;
73    let values = index
74        .multi_get(checkpoint, &keys)
75        .map_err(|e| db_error(e, "failed to batch get balances"))?;
76
77    let mut balances = Vec::with_capacity(values.len());
78    for (key, value) in keys.into_iter().zip(values) {
79        let with_prefix = true;
80        let coin_type = key.type_.to_canonical_string(with_prefix);
81        let balance = value.context("Failed to deserialize balance")?.unwrap_or(0);
82        let balance = u64::try_from(balance)
83            .with_context(|| format!("Bad balance for type {coin_type}: {balance}"))?;
84
85        balances.push(grpc::Balance {
86            owner: Some(key.owner.to_string()),
87            coin_type: Some(coin_type),
88            balance: Some(balance),
89            page_token: None,
90        });
91    }
92
93    Ok(grpc::BatchGetBalancesResponse { balances })
94}

96pub(super) fn get_balance(
97    state: &State,
98    checkpoint: u64,
99    request: grpc::GetBalanceRequest,
100) -> Result<grpc::Balance, RpcError<Error>> {
101    let key = key(request)?;
102    let index = &state.store.schema().balances;
103    let balance = index
104        .get(checkpoint, &key)
105        .map_err(|e| db_error(e, "failed to get balance"))?
106        .unwrap_or(0);
107
108    let with_prefix = true;
109    let coin_type = key.type_.to_canonical_string(with_prefix);
110    let balance = u64::try_from(balance)
111        .with_context(|| format!("Bad balance for type {coin_type}: {balance}"))?;
112
113    Ok(grpc::Balance {
114        owner: Some(key.owner.to_string()),
115        coin_type: Some(coin_type),
116        balance: Some(balance),
117        page_token: None,
118    })
119}

121pub(super) fn list_balances(
122    state: &State,
123    checkpoint: u64,
124    request: grpc::ListBalancesRequest,
125) -> Result<grpc::ListBalancesResponse, RpcError<Error>> {
126    let owner = if request.owner().is_empty() {
127        return Err(Error::MissingOwner.into());
128    } else {
129        owner(request.owner())?
130    };
131
132    let page = Page::from_request(
133        &state.rpc_config.pagination,
134        request.after_token(),
135        request.before_token(),
136        request.page_size(),
137        request.end(),
138    );
139
140    let index = &state.store.schema().balances;
141    // Zero balances may end up in the databases through accumulation. They will eventually be
142    // cleaned up by compaction, but until then, they need to be filtered out of results (similar
143    // to how RocksDB filters out tombstones).
144    let resp = page.paginate_filtered(index, checkpoint, &Compat(owner), |_, _, balance| {
145        *balance > 0
146    })?;
147
148    let mut balances = vec![];
149    for (token, key, balance) in resp.results {
150        let coin_type = key.type_.to_canonical_string(/* with_prefix */ true);
151        let balance = u64::try_from(balance)
152            .with_context(|| format!("Bad balance for type {coin_type}: {balance}"))?;
153
154        balances.push(grpc::Balance {
155            owner: Some(key.owner.to_string()),
156            coin_type: Some(coin_type),
157            balance: Some(balance),
158            page_token: Some(token.into()),
159        });
160    }
161
162    Ok(grpc::ListBalancesResponse {
163        has_previous_page: Some(resp.has_prev),
164        has_next_page: Some(resp.has_next),
165        balances,
166    })
167}

169/// Convert a point lookup into a key for the underlying index.
170fn key(request: grpc::GetBalanceRequest) -> Result<Key, Error> {
171    let owner = if request.owner().is_empty() {
172        return Err(Error::MissingOwner);
173    } else {
174        owner(request.owner())?
175    };
176
177    let type_ = if request.coin_type().is_empty() {
178        return Err(Error::MissingType);
179    } else {
180        TypeTag::from_str(request.coin_type())
181            .map_err(|_| Error::InvalidType(request.coin_type().to_owned()))?
182    };
183
184    Ok(Key { owner, type_ })
185}

187/// Parse the owner's `SuiAddress` from a string. Addresses must start with `0x` followed by
188/// between 1 and 64 hexadecimal characters.
189///
190/// TODO: Switch to using `sui_sdk_types::Address`, once the indexing framework is ported to the
191/// new SDK.
192fn owner(input: &str) -> Result<SuiAddress, Error> {
193    let Some(s) = input.strip_prefix("0x") else {
194        return Err(Error::InvalidOwner(input.to_owned()));
195    };
196
197    let s = if s.is_empty() || s.len() > 64 {
198        return Err(Error::InvalidOwner(s.to_owned()));
199    } else if s.len() != 64 {
200        Cow::Owned(format!("0x{s:0>64}"))
201    } else {
202        Cow::Borrowed(input)
203    };
204
205    SuiAddress::from_str(s.as_ref()).map_err(|_| Error::InvalidOwner(s.to_string()))
206}
