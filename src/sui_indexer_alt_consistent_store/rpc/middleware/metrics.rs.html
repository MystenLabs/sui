4use std::borrow::Cow;
5use std::sync::Arc;
6
7use axum::extract::MatchedPath;
8use http::StatusCode;
9use http::header::CONTENT_TYPE;
10use mysten_network::callback::MakeCallbackHandler;
11use mysten_network::callback::ResponseHandler;
12use prometheus::HistogramTimer;
13use tonic::Code;
14use tonic::metadata::GRPC_CONTENT_TYPE;
15
16use crate::rpc::metrics::RpcMetrics;
17
18#[derive(Clone)]
19pub struct MakeMetricsHandler {
20    metrics: Arc<RpcMetrics>,
21}
22
23pub struct Handler {
24    metrics: Arc<RpcMetrics>,
25    path: Cow<'static, str>,
26    timer: Option<HistogramTimer>,
27}
28
29impl MakeMetricsHandler {
30    pub fn new(metrics: Arc<RpcMetrics>) -> Self {
31        Self { metrics }
32    }
33}
34
35impl MakeCallbackHandler for MakeMetricsHandler {
36    type Handler = Handler;
37
38    fn make_handler(&self, request: &http::request::Parts) -> Self::Handler {
39        let metrics = self.metrics.clone();
40
41        let path = if let Some(matched_path) = request.extensions.get::<MatchedPath>() {
42            if request
43                .headers
44                .get(&CONTENT_TYPE)
45                .is_some_and(|header| header == GRPC_CONTENT_TYPE)
46            {
47                Cow::Owned(request.uri.path().to_owned())
48            } else {
49                Cow::Owned(matched_path.as_str().to_owned())
50            }
51        } else {
52            Cow::Borrowed("unknown")
53        };
54
55        let timer = metrics
56            .request_latency
57            .with_label_values(&[path.as_ref()])
58            .start_timer();
59
60        metrics
61            .requests_received
62            .with_label_values(&[path.as_ref()])
63            .inc();
64
65        Handler {
66            metrics,
67            path,
68            timer: Some(timer),
69        }
70    }
71}
72
73impl ResponseHandler for Handler {
74    fn on_response(&mut self, response: &http::response::Parts) {
75        const GRPC_STATUS: http::HeaderName = http::HeaderName::from_static("grpc-status");
76
77        let (success, status) = if response
78            .headers
79            .get(&CONTENT_TYPE)
80            .is_some_and(|content_type| {
81                content_type
82                    .as_bytes()
83                    // check if the content-type starts_with 'application/grpc' in order to
84                    // consider this as a gRPC request. A prefix comparison is done instead of a
85                    // full equality check in order to account for the various types of
86                    // content-types that are considered as gRPC traffic.
87                    .starts_with(GRPC_CONTENT_TYPE.as_bytes())
88            }) {
89            let code = response
90                .headers
91                .get(&GRPC_STATUS)
92                .map(http::HeaderValue::as_bytes)
93                .map(Code::from_bytes)
94                .unwrap_or(Code::Ok);
95
96            (code == Code::Ok, code_as_str(code))
97        } else {
98            (response.status == StatusCode::OK, response.status.as_str())
99        };
100
101        self.timer.take().map(HistogramTimer::stop_and_record);
102
103        if success {
104            self.metrics
105                .requests_succeeded
106                .with_label_values(&[self.path.as_ref(), status])
107                .inc();
108        } else {
109            self.metrics
110                .requests_failed
111                .with_label_values(&[self.path.as_ref(), status])
112                .inc();
113        }
114    }
115
116    fn on_error<E>(&mut self, _error: &E) {
117        unreachable!("all axum services are required to have an error type of Infallible");
118    }
119}
120
121impl Drop for Handler {
122    fn drop(&mut self) {
123        if self.timer.is_some() {
124            self.metrics
125                .requests_cancelled
126                .with_label_values(&[self.path.as_ref()])
127                .inc();
128        }
129    }
130}
131
132fn code_as_str(code: Code) -> &'static str {
133    match code {
134        Code::Ok => "ok",
135        Code::Cancelled => "cancelled",
136        Code::Unknown => "unknown",
137        Code::InvalidArgument => "invalid-argument",
138        Code::DeadlineExceeded => "deadline-exceeded",
139        Code::NotFound => "not-found",
140        Code::AlreadyExists => "already-exists",
141        Code::PermissionDenied => "permission-denied",
142        Code::ResourceExhausted => "resource-exhausted",
143        Code::FailedPrecondition => "failed-precondition",
144        Code::Aborted => "aborted",
145        Code::OutOfRange => "out-of-range",
146        Code::Unimplemented => "unimplemented",
147        Code::Internal => "internal",
148        Code::Unavailable => "unavailable",
149        Code::DataLoss => "data-loss",
150        Code::Unauthenticated => "unauthenticated",
151    }
152}
