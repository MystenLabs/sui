37#[derive(clap::Args, Clone, Debug)]
38pub struct RpcArgs {
39    /// Address to accept incoming RPC connections on.
40    #[clap(long, default_value_t = Self::default().rpc_listen_address)]
41    pub rpc_listen_address: SocketAddr,
42
43    /// TLS configuration
44    #[clap(flatten)]
45    pub tls: TlsArgs,
46}
47
48#[derive(clap::Args, Clone, Debug, Default)]
49pub struct TlsArgs {
50    /// Address to accept incoming TLS/HTTPS connections on
51    #[clap(long, requires_all = &["tls_cert", "tls_key"])]
52    pub rpc_tls_listen_address: Option<SocketAddr>,
53
54    /// Path to TLS certificate file (PEM format)
55    #[clap(long, requires_all = &["rpc_tls_listen_address", "tls_key"])]
56    pub tls_cert: Option<PathBuf>,
57
58    /// Path to TLS private key file (PEM format)
59    #[clap(long, requires_all = &["rpc_tls_listen_address", "tls_cert"])]
60    pub tls_key: Option<PathBuf>,
61}

63/// Responsible for the set-up of a gRPC service -- adding services, configuring reflection,
64/// health-checks, logging and metrics middleware, etc.
65pub(crate) struct RpcService<'d> {
66    /// Address to accept incoming RPC connections on.
67    rpc_listen_address: SocketAddr,
68
69    /// Optional address to accept incoming TLS RPC connections on.
70    rpc_tls_listen_address: Option<SocketAddr>,
71
72    /// TLS configuration
73    tls_config: Option<RustlsConfig>,
74
75    /// The version string to report with each response, as an HTTP header.
76    version: &'static str,
77
78    /// File descriptors are added to these builders to eventually be exposed via the reflection
79    /// service.
80    reflection_v1: tonic_reflection::server::Builder<'d>,
81    reflection_v1alpha: tonic_reflection::server::Builder<'d>,
82
83    /// Names of gRPC services and associated readiness futures registered with this instance.
84    service_futures: Vec<(&'static str, BoxFuture<'static, ()>)>,
85
86    /// The axum router that wil handle incoming requests.
87    router: Router,
88
89    /// Metrics for the RPC service.
90    metrics: Arc<RpcMetrics>,
91}
92
93pub type BoxError = Box<dyn std::error::Error + Send + Sync + 'static>;
94
95impl<'d> RpcService<'d> {
96    pub(crate) async fn new(
97        args: RpcArgs,
98        version: &'static str,
99        registry: &Registry,
100    ) -> anyhow::Result<Self> {
101        let RpcArgs {
102            rpc_listen_address,
103            tls,
104        } = args;
105
106        let TlsArgs {
107            rpc_tls_listen_address,
108            tls_cert,
109            tls_key,
110        } = tls;
111
112        let tls_config = if let (Some(cert), Some(key)) = (tls_cert, tls_key) {
113            Some(
114                RustlsConfig::from_pem_file(cert, key)
115                    .await
116                    .context("Failed to load TLS configuration")?,
117            )
118        } else {
119            None
120        };
121
122        Ok(Self {
123            rpc_listen_address,
124            rpc_tls_listen_address,
125            tls_config,
126            version,
127            reflection_v1: tonic_reflection::server::Builder::configure(),
128            reflection_v1alpha: tonic_reflection::server::Builder::configure(),
129            service_futures: vec![],
130            router: Router::new(),
131            metrics: Arc::new(RpcMetrics::new(registry)),
132        })
133    }
134
135    /// Register a file descriptor set to be exposed via the reflection service.
136    pub(crate) fn register_encoded_file_descriptor_set(mut self, fds: &'d [u8]) -> Self {
137        self.reflection_v1 = self.reflection_v1.register_encoded_file_descriptor_set(fds);
138        self.reflection_v1alpha = self
139            .reflection_v1alpha
140            .register_encoded_file_descriptor_set(fds);
141        self
142    }
143
144    /// Register a new gRPC service.
145    pub(crate) fn add_service<S, F>(mut self, s: S, ready: F) -> Self
146    where
147        S: Clone + Send + Sync + 'static,
148        S: NamedService,
149        S: tower::Service<Request, Response: IntoResponse, Error = Infallible>,
150        S::Future: Send + 'static,
151        S::Error: Send + Into<BoxError>,
152        F: Future<Output = ()> + Send + 'static,
153    {
154        self.service_futures.push((S::NAME, Box::pin(ready)));
155        self.router = add_service(self.router, s);
156        self
157    }
158
159    /// Run the RPC service. This binds the listener and exposes handlers for the RPC service.
160    pub(crate) async fn run(self) -> anyhow::Result<Service> {
161        let Self {
162            rpc_listen_address,
163            rpc_tls_listen_address,
164            tls_config,
165            version,
166            reflection_v1,
167            reflection_v1alpha,
168            service_futures,
169            mut router,
170            metrics,
171        } = self;
172
173        let reflection_v1 = reflection_v1
174            .register_encoded_file_descriptor_set(tonic_health::pb::FILE_DESCRIPTOR_SET)
175            .build_v1()
176            .unwrap();
177
178        let reflection_v1alpha = reflection_v1alpha
179            .register_encoded_file_descriptor_set(tonic_health::pb::FILE_DESCRIPTOR_SET)
180            .build_v1alpha()
181            .unwrap();
182
183        let (health_reporter, health_service) = tonic_health::server::health_reporter();
184
185        let internal_services = vec![
186            service_name(&reflection_v1),
187            service_name(&reflection_v1alpha),
188            service_name(&health_service),
189        ];
190
191        router = add_service(router, reflection_v1);
192        router = add_service(router, reflection_v1alpha);
193        router = add_service(router, health_service);
194        router = router
195            .layer(CallbackLayer::new(MakeMetricsHandler::new(metrics.clone())))
196            .layer(axum::middleware::from_fn_with_state(
197                Version(version),
198                middleware::version::set_version,
199            ))
200            .layer(CatchPanicLayer::new(metrics));
201
202        for service_name in internal_services {
203            health_reporter
204                .set_service_status(service_name, ServingStatus::Serving)
205                .await;
206        }
207
208        // Create a Service to be attached as secondary to the main service
209        let mut readiness_checks = Service::new();
210
211        for (name, ready) in service_futures {
212            health_reporter
213                .set_service_status(name, ServingStatus::NotServing)
214                .await;
215
216            let reporter = health_reporter.clone();
217            readiness_checks = readiness_checks.spawn(async move {
218                ready.await;
219                reporter
220                    .set_service_status(name, ServingStatus::Serving)
221                    .await;
222                info!("gRPC service {name} is now SERVING");
223                Ok(())
224            });
225        }
226
227        let mut service = Service::new();
228        service = service.attach(readiness_checks);
229
230        // Start HTTPS server if TLS is configured
231        if let (Some(listen_address), Some(config)) = (rpc_tls_listen_address, tls_config) {
232            info!("Starting Consistent RPC TLS service on {listen_address}");
233            let handle = Handle::new();
234            let tls_router = router.clone();
235
236            service = service
237                .with_shutdown_signal({
238                    let handle = handle.clone();
239                    async move {
240                        handle.graceful_shutdown(None);
241                    }
242                })
243                .spawn(async move {
244                    axum_server::bind_rustls(listen_address, config)
245                        .handle(handle)
246                        .serve(tls_router.into_make_service())
247                        .await
248                        .context("Consistent RPC TLS service failed")?;
249                    Ok(())
250                });
251        }
252
253        // Start HTTP server
254        info!("Starting Consistent RPC service on {rpc_listen_address}");
255        let listener = TcpListener::bind(rpc_listen_address)
256            .await
257            .context("Failed to bind Consistent RPC to listen address")?;
258
259        let (stx, srx) = oneshot::channel::<()>();
260        service = service
261            .with_shutdown_signal(async move {
262                let _ = stx.send(());
263            })
264            .spawn(async move {
265                axum::serve(listener, router)
266                    .with_graceful_shutdown(async move {
267                        let _ = srx.await;
268                    })
269                    .await
270                    .context("Consistent RPC HTTP service failed")
271            });
272
273        Ok(service)
274    }
275}

277impl Default for RpcArgs {
278    fn default() -> Self {
279        Self {
280            rpc_listen_address: "0.0.0.0:7001".parse().unwrap(),
281            tls: TlsArgs::default(),
282        }
283    }

286fn service_name<S: NamedService>(_: &S) -> &'static str {
287    S::NAME
288}
289
290fn add_service<S>(router: Router, s: S) -> Router
291where
292    S: Clone + Send + Sync + 'static,
293    S: NamedService,
294    S: tower::Service<Request, Response: IntoResponse, Error = Infallible>,
295    S::Future: Send + 'static,
296    S::Error: Send + Into<BoxError>,
297{
298    router.route_service(&format!("/{}/{{*rest}}", S::NAME), s)
299}
