16/// State exposed to RPC service implementations.
17#[derive(Clone)]
18pub(crate) struct State {
19    /// Access to the database.
20    pub store: Store<Schema>,
21
22    /// RPC Configuration
23    pub rpc_config: Arc<RpcConfig>,
24
25    /// Configuration for the consistent range.
26    pub consistency_config: Arc<ConsistencyConfig>,
27}
28
29#[derive(thiserror::Error, Debug)]
30pub(crate) enum Error {
31    #[error("Bad checkpoint sequence number: {0:?}")]
32    BadCheckpoint(AsciiMetadataValue),
33
34    /// The store has no snapshots.
35    #[error("No snapshots available")]
36    NoSnapshots,
37}
38
39impl State {
40    /// Extract the checkpoint to request data from, from the request metadata, or default to the
41    /// latest available snapshot, if no checkpoint is provided.
42    ///
43    /// Fails if a valid checkpoint is not provided (the metadata exists but doesn't contain the
44    /// decimal representation of a `u64`), or if a checkpoint is not provided and there are no
45    /// snapshots available in the store, meaning a default (latest) checkpoint cannot be
46    /// determined.
47    pub(super) fn checkpoint<T>(
48        &self,
49        request: &tonic::Request<T>,
50    ) -> Result<u64, RpcError<Error>> {
51        let Some(checkpoint) = request.metadata().get(CHECKPOINT_METADATA) else {
52            // If a checkpoint hasn't been supplied default to the latest snapshot.
53            return Ok(self
54                .store
55                .db()
56                .snapshot_range(u64::MAX)
57                .ok_or(Error::NoSnapshots)?
58                .end()
59                .checkpoint_hi_inclusive);
60        };
61
62        Ok(checkpoint
63            .to_str()
64            .map_err(|_| Error::BadCheckpoint(checkpoint.clone()))?
65            .parse()
66            .map_err(|_| Error::BadCheckpoint(checkpoint.clone()))?)
67    }
68}
69
70impl StatusCode for Error {
71    fn code(&self) -> tonic::Code {
72        match self {
73            Error::BadCheckpoint(_) => tonic::Code::InvalidArgument,
74            Error::NoSnapshots => tonic::Code::Unavailable,
75        }
76    }
77}

79/// Convert `content` into a `tonic::Response` annotated with the checkpoint the data came from.
80pub(super) fn checkpointed_response<T>(
81    checkpoint: u64,
82    content: T,
83) -> Result<tonic::Response<T>, RpcError<Error>> {
84    let checkpoint = checkpoint
85        .to_string()
86        .parse()
87        .with_context(|| format!("Invalid checkpoint for metadata: {checkpoint}"))?;
88
89    let mut resp = tonic::Response::new(content);
90    let meta = resp.metadata_mut();
91    meta.insert(CHECKPOINT_METADATA, checkpoint);
92
93    Ok(resp)
94}
