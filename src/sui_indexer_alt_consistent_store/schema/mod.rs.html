4use std::sync::Arc;
5
6use sui_indexer_alt_framework::types::base_types::VersionDigest;
7
8use crate::{
9    db::{Db, map::DbMap},
10    store,
11};
12
13pub(crate) mod balances;
14pub(crate) mod object_by_owner;
15pub(crate) mod object_by_type;
16
17/// All tables written to and read from the consistent store.
18pub(crate) struct Schema {
19    /// The balances of all coin-like objects owned by an account, indexed by owner and type.
20    pub(crate) balances: DbMap<balances::Key, i128>,
21
22    /// Fetch objects by their owner, optionally filtered by type. Coin-like objects are returned
23    /// in descending balance order.
24    pub(crate) object_by_owner: DbMap<object_by_owner::Key, VersionDigest>,
25
26    /// Fetch objects by their type.
27    pub(crate) object_by_type: DbMap<object_by_type::Key, VersionDigest>,
28}
29
30impl store::Schema for Schema {
31    fn cfs(base_options: &rocksdb::Options) -> Vec<(&'static str, rocksdb::Options)> {
32        vec![
33            ("balances", balances::options(base_options)),
34            ("object_by_owner", object_by_owner::options(base_options)),
35            ("object_by_type", object_by_type::options(base_options)),
36        ]
37    }
38
39    fn open(db: &Arc<Db>) -> anyhow::Result<Self> {
40        Ok(Self {
41            balances: DbMap::new(db.clone(), "balances"),
42            object_by_owner: DbMap::new(db.clone(), "object_by_owner"),
43            object_by_type: DbMap::new(db.clone(), "object_by_type"),
44        })
45    }
46}
