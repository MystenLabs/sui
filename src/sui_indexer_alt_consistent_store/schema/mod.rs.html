4use std::sync::Arc;
5
6use sui_indexer_alt_framework::types::base_types::VersionDigest;
7
8use crate::db::Db;
9use crate::db::map::DbMap;
10use crate::store;
11
12pub(crate) mod address_balances;
13pub(crate) mod balances;
14pub(crate) mod object_by_owner;
15pub(crate) mod object_by_type;
16
17/// All tables written to and read from the consistent store.
18pub(crate) struct Schema {
19    /// The various balances as tracked by the accumulator object, indexed by owner and type.
20    pub(crate) address_balances: DbMap<address_balances::Key, u128>,
21
22    /// The balances of all coin-like objects owned by an account, indexed by owner and type.
23    pub(crate) balances: DbMap<balances::Key, i128>,
24
25    /// Fetch objects by their owner, optionally filtered by type. Coin-like objects are returned
26    /// in descending balance order.
27    pub(crate) object_by_owner: DbMap<object_by_owner::Key, VersionDigest>,
28
29    /// Fetch objects by their type.
30    pub(crate) object_by_type: DbMap<object_by_type::Key, VersionDigest>,
31}
32
33impl store::Schema for Schema {
34    fn cfs(base_options: &rocksdb::Options) -> Vec<(&'static str, rocksdb::Options)> {
35        vec![
36            ("address_balances", address_balances::options(base_options)),
37            ("balances", balances::options(base_options)),
38            ("object_by_owner", object_by_owner::options(base_options)),
39            ("object_by_type", object_by_type::options(base_options)),
40        ]
41    }
42
43    fn open(db: &Arc<Db>) -> anyhow::Result<Self> {
44        Ok(Self {
45            address_balances: DbMap::new(db.clone(), "address_balances"),
46            balances: DbMap::new(db.clone(), "balances"),
47            object_by_owner: DbMap::new(db.clone(), "object_by_owner"),
48            object_by_type: DbMap::new(db.clone(), "object_by_type"),
49        })
50    }
51}
