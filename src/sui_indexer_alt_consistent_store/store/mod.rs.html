22/// Defines the schema for the database.
23pub(crate) trait Schema: Sized {
24    /// Configuration for this schema's column families (names and options). Takes database-level
25    /// options as the base options to extend per column-family.
26    fn cfs(base_options: &rocksdb::Options) -> Vec<(&'static str, rocksdb::Options)>;
27
28    /// Construct the Rust value that represents the schema's tables, given access to the database.
29    /// It is expected to be a struct containing various `DbMap`s as fields.
30    fn open(db: &Arc<Db>) -> anyhow::Result<Self>;
31}
32
33/// A wrapper around a rocksdb [`Db`] that implements the indexer framework's [`store::Store`]
34/// interface.
35pub(crate) struct Store<S>(Arc<Inner<S>>);
36
37/// A connection to the store that supports reads, writes and watermarking.
38pub(crate) struct Connection<'s, S> {
39    pub store: &'s Store<S>,
40    pub batch: rocksdb::WriteBatch,
41    watermark: Option<(&'static str, Watermark)>,
42}
43
44/// The contents of the store.
45struct Inner<S> {
46    db: Arc<Db>,
47
48    /// A rust representation of the column families in the database that we want to access from
49    /// this store.
50    schema: S,
51
52    /// Access to a synchronizer queue per-pipeline. This is initialized when the store is given a
53    /// [`Synchronizer`] to run, and is used to send writes to the database, associated with a
54    /// pipeline.
55    queue: OnceLock<Queue>,
56}
57
58impl<S: Schema> Store<S> {
59    /// Create a new store with the database at a given `path`, configured by `config`.
60    ///
61    /// `snapshots` is the maximum number of consistent snapshots to keep in the database at one
62    /// time, and `schema` controls which tables are opened on the database.
63    pub(crate) fn open(
64        path: impl AsRef<Path>,
65        config: DbConfig,
66        snapshots: u64,
67        registry: Option<&Registry>,
68    ) -> anyhow::Result<Self> {
69        let db_options: rocksdb::Options = config.into();
70        let cfs = S::cfs(&db_options);
71        let cf_names = cfs.iter().map(|(name, _)| name.to_string()).collect();
72        let db = Arc::new(
73            Db::open(path, db_options.clone(), snapshots as usize, cfs)
74                .context("Failed to open database")?,
75        );
76
77        let schema = S::open(&db).context("Failed to open schema")?;
78
79        if let Some(registry) = registry {
80            registry
81                .register(Box::new(ColumnFamilyStatsCollector::new(
82                    Some("rocksdb"),
83                    db.clone(),
84                    cf_names,
85                )))
86                .context("Failed to register rocksdb column family stats collector")?;
87        }
88
89        Ok(Self(Arc::new(Inner {
90            db,
91            queue: OnceLock::new(),
92            schema,
93        })))
94    }
95
96    /// Access to the store's database.
97    pub(crate) fn db(&self) -> &Arc<Db> {
98        &self.0.db
99    }
100
101    /// Access to the store's schema.
102    pub(crate) fn schema(&self) -> &S {
103        &self.0.schema
104    }
105
106    /// Run the provided synchronizer, and register its queue with the store. This will fail if the
107    /// store already has a synchronizer running.
108    pub(crate) fn sync(&self, s: Synchronizer) -> anyhow::Result<JoinHandle<()>> {
109        let (handle, queue) = s.run()?;
110        self.0
111            .queue
112            .set(queue)
113            .map_err(|_| anyhow!("Store already has synchronizer"))?;
114        Ok(handle)
115    }
116}

131#[async_trait::async_trait]
132impl<S: Send + Sync + 'static> store::TransactionalStore for Store<S> {
133    async fn transaction<'a, R, F>(&self, f: F) -> anyhow::Result<R>
134    where
135        R: Send + 'a,
136        F: Send + 'a,
137        F: for<'r> FnOnce(&'r mut Connection<'_, S>) -> ScopedBoxFuture<'a, 'r, anyhow::Result<R>>,
138    {
139        let mut conn = self.connect().await?;
140        let r = f(&mut conn).await?;
141
142        let Some((pipeline, watermark)) = conn.watermark else {
143            bail!("No watermark set during transaction");
144        };
145
146        self.0
147            .queue
148            .get()
149            .context("Synchronizer not running for store")?
150            .get(pipeline)
151            .with_context(|| format!("No {pipeline:?} synchronizer queue"))?
152            .send((watermark, conn.batch))
153            .await
154            .map_err(|_| anyhow!("{pipeline:?} synchronizer queue closed"))?;
155
156        Ok(r)
157    }
158}

247    async fn wait_until<F, R>(f: F) -> Result<(), Elapsed>
248    where
249        F: Fn() -> R,
250        R: Future<Output = bool>,
251    {
252        time::timeout(Duration::from_millis(500), async move {
253            let mut interval = time::interval(Duration::from_millis(100));
254            loop {
255                interval.tick().await;
256                if f().await {
257                    return;
258                }
259            }
260        })
261        .await
262    }

264    fn has_range(store: &Store<TestSchema>, lo: Option<u64>, hi: Option<u64>) -> bool {
265        store.db().snapshot_range(u64::MAX).is_some_and(|s| {
266            lo.is_none_or(|lo| lo == s.start().checkpoint_hi_inclusive)
267                && hi.is_none_or(|hi| hi == s.end().checkpoint_hi_inclusive)
268        })
269    }

528    #[tokio::test]
529    async fn test_multiple_pipeline_catchup() {
530        let d = tempfile::tempdir().unwrap();
531        let snapshots = 4;
532
533        {
534            // Initialize the database with some data for one of the pipelines.
535            let db_options: rocksdb::Options = DbConfig::default().into();
536            let db = Arc::new(
537                Db::open(
538                    d.path().join("db"),
539                    db_options.clone(),
540                    snapshots,
541                    TestSchema::cfs(&db_options),
542                )
543                .unwrap(),
544            );
545
546            let schema = TestSchema::open(&db).unwrap();
547
548            let mut batch = rocksdb::WriteBatch::default();
549            schema.b.insert(42, "x".to_owned(), &mut batch).unwrap();
550            db.write("b", CommitterWatermark::new_for_testing(0).into(), batch)
551                .unwrap();
552        }
553
554        let store: Store<TestSchema> =
555            Store::open(d.path().join("db"), DbConfig::default(), 4, None).unwrap();
556
557        let stride = 1;
558        let buffer_size = 10;
559        let first_checkpoint = None;
560        let cancel = CancellationToken::new();
561        let mut sync = Synchronizer::new(
562            store.db().clone(),
563            stride,
564            buffer_size,
565            first_checkpoint,
566            cancel.clone(),
567        );
568
569        sync.register_pipeline("a").unwrap();
570        sync.register_pipeline("b").unwrap();
571        let h_sync = store.sync(sync).unwrap();
572
573        // The pipelines are not in sync to begin with, so the synchronizer is waiting for the
574        // writes for the other pipeline in order to take a snapshot.
575        wait_until(|| async { has_range(&store, None, None) })
576            .await
577            .unwrap_err();
578
579        // Further writes to the pipeline that is ahead will be held back.
580        write(&store, "b", 1, |s, b| {
581            s.b.insert(42, "y".to_owned(), b)?;
582            Ok(())
583        })
584        .await
585        .unwrap();
586
587        // Further writes to the pipeline that is ahead will be held back.
588        wait_until(|| async { has_range(&store, None, None) })
589            .await
590            .unwrap_err();
591
592        write(&store, "a", 0, |s, b| {
593            s.a.insert("x".to_owned(), 42, b)?;
594            Ok(())
595        })
596        .await
597        .unwrap();
598
599        // After the other pipeline was caught up, the synchronizer will take the snapshot, but it
600        // will not yet make the subsequent write to the other pipeline available.
601        wait_until(|| async { has_range(&store, None, Some(0)) })
602            .await
603            .unwrap();
604
605        let s = store.schema();
606        assert_eq!(s.a.get(0, "x".to_owned()).unwrap(), Some(42));
607        assert_eq!(s.b.get(0, 42).unwrap(), Some("x".to_owned()));
608
609        // Catch up the first pipeline without writing any further data.
610        write(&store, "a", 1, |_, _| Ok(())).await.unwrap();
611        wait_until(|| async { has_range(&store, None, Some(1)) })
612            .await
613            .unwrap();
614
615        let s = store.schema();
616        assert_eq!(s.a.get(1, "x".to_owned()).unwrap(), Some(42));
617        assert_eq!(s.b.get(1, 42).unwrap(), Some("y".to_owned()));
618
619        cancel.cancel();
620        h_sync.await.unwrap();
621    }

712        })
713        .await
714        .unwrap();
715
716        // With the fix, no snapshot is taken until after the first checkpoint is written.

731    async fn test_stride() {
732        let d = tempfile::tempdir().unwrap();
733
734        let store: Store<TestSchema> =
735            Store::open(d.path().join("db"), DbConfig::default(), 4, None).unwrap();
736
737        let stride = 3;
738        let buffer_size = 10;
739        let first_checkpoint = None;
740        let cancel = CancellationToken::new();
741        let mut sync = Synchronizer::new(
742            store.db().clone(),
743            stride,
744            buffer_size,
745            first_checkpoint,
746            cancel.clone(),
747        );
748
749        sync.register_pipeline("test").unwrap();
750        let h_sync = store.sync(sync).unwrap();
751
752        // Write a run of checkpoints.
753        for cp in 0..=10 {
754            write(&store, "test", cp, move |s, b| {
755                s.a.insert("x".to_owned(), cp * 3, b)?;
756                s.b.insert(cp * 3, "x".to_owned(), b)?;
757                Ok(())
758            })
759            .await
760            .unwrap();
761        }
762
763        // The synchronizer will take a snapshot before every `stride`-th checkpoint.
764        wait_until(|| async { has_range(&store, Some(2), Some(8)) })
765            .await
766            .unwrap();
767
768        let d = store.db();
769        let s = store.schema();
770
771        assert_eq!(d.snapshots(), 3);
772        for cp in (2..10).step_by(stride as usize) {
773            assert_eq!(s.a.get(cp, "x".to_owned()).unwrap(), Some(cp * 3));
774            assert_eq!(s.b.get(cp, cp * 3).unwrap(), Some("x".to_owned()));
775        }
776
777        // Querying the snapshot range at the latest checkpoint does the same thing as an unbounded
778        // range request.
779        assert_eq!(
780            Some(8),
781            d.snapshot_range(8).map(|r| r.end().checkpoint_hi_inclusive)
782        );
783
784        // Going one checkpoint back causes the range to drop back by the stride.
785        assert_eq!(
786            Some(5),
787            d.snapshot_range(7).map(|r| r.end().checkpoint_hi_inclusive)
788        );
789
790        // Going back beyond the first checkpoint results in an empty range.
791        assert_eq!(None, d.snapshot_range(1));
792
793        cancel.cancel();
794        h_sync.await.unwrap();
795    }

844
845        let store: Store<TestSchema> =

855            buffer_size,

857            cancel.clone(),

859

862
863        write(&store, "test", 0, |s, b| {
864            s.a.insert("x".to_owned(), 42, b)?;

874        .await
875        .unwrap();
876
877        // The out of order batch will appear to succeed, but the synchronizer will detect the

884        // The first write made it through, but the second one did not.
885        let s = store.schema();
886        let db = store.db();
887        assert_eq!(db.snapshots(), 1);
888        assert_eq!(
889            db.snapshot_range(u64::MAX)
890                .map(|s| s.end().checkpoint_hi_inclusive),
891            Some(0)
892        );
893        assert_eq!(s.a.get(0, "x".to_owned()).unwrap(), Some(42));
894        assert_eq!(s.a.get(0, "y".to_owned()).unwrap(), None);
895    }
