28/// Defines the schema for the database.
29pub(crate) trait Schema: Sized {
30    /// Configuration for this schema's column families (names and options). Takes database-level
31    /// options as the base options to extend per column-family.
32    fn cfs(base_options: &rocksdb::Options) -> Vec<(&'static str, rocksdb::Options)>;
33
34    /// Construct the Rust value that represents the schema's tables, given access to the database.
35    /// It is expected to be a struct containing various `DbMap`s as fields.
36    fn open(db: &Arc<Db>) -> anyhow::Result<Self>;
37}
38
39/// A wrapper around a rocksdb [`Db`] that implements the indexer framework's [`store::Store`]
40/// interface.
41pub(crate) struct Store<S>(Arc<Inner<S>>);
42
43/// A connection to the store that supports reads, writes and watermarking.
44pub(crate) struct Connection<'s, S> {
45    pub store: &'s Store<S>,
46    pub batch: rocksdb::WriteBatch,
47    watermark: Option<(String, Watermark)>,
48}
49
50/// The contents of the store.
51struct Inner<S> {
52    db: Arc<Db>,
53
54    /// A rust representation of the column families in the database that we want to access from
55    /// this store.
56    schema: S,
57
58    /// Access to a synchronizer queue per-pipeline. This is initialized when the store is given a
59    /// [`Synchronizer`] to run, and is used to send writes to the database, associated with a
60    /// pipeline.
61    queue: OnceLock<Queue>,
62}
63
64impl<S: Schema> Store<S> {
65    /// Create a new store with the database at a given `path`, configured by `config`.
66    ///
67    /// `snapshots` is the maximum number of consistent snapshots to keep in the database at one
68    /// time, and `schema` controls which tables are opened on the database.
69    pub(crate) fn open(
70        path: impl AsRef<Path>,
71        config: DbConfig,
72        snapshots: u64,
73        registry: Option<&Registry>,
74    ) -> anyhow::Result<Self> {
75        let db_options: rocksdb::Options = config.into();
76        let cfs = S::cfs(&db_options);
77        let cf_names = cfs.iter().map(|(name, _)| name.to_string()).collect();
78        let db = Arc::new(
79            Db::open(path, db_options.clone(), snapshots as usize, cfs)
80                .context("Failed to open database")?,
81        );
82
83        let schema = S::open(&db).context("Failed to open schema")?;
84
85        if let Some(registry) = registry {
86            registry
87                .register(Box::new(ColumnFamilyStatsCollector::new(
88                    Some("rocksdb"),
89                    db.clone(),
90                    cf_names,
91                )))
92                .context("Failed to register rocksdb column family stats collector")?;
93        }
94
95        Ok(Self(Arc::new(Inner {
96            db,
97            queue: OnceLock::new(),
98            schema,
99        })))
100    }
101
102    /// Access to the store's database.
103    pub(crate) fn db(&self) -> &Arc<Db> {
104        &self.0.db
105    }
106
107    /// Access to the store's schema.
108    pub(crate) fn schema(&self) -> &S {
109        &self.0.schema
110    }
111
112    /// Run the provided synchronizer, and register its queue with the store. This will fail if the
113    /// store already has a synchronizer running.
114    pub(crate) fn sync(&self, s: Synchronizer) -> anyhow::Result<Service> {
115        let (service, queue) = s.run()?;
116        self.0
117            .queue
118            .set(queue)
119            .map_err(|_| anyhow!("Store already has synchronizer"))?;
120        Ok(service)
121    }
122}

125impl<S: Send + Sync + 'static> store::Store for Store<S> {
126    type Connection<'s> = Connection<'s, S>;
127
128    async fn connect(&self) -> anyhow::Result<Connection<'_, S>> {
129        Ok(Connection {
130            store: self,
131            batch: rocksdb::WriteBatch::default(),
132            watermark: None,
133        })
134    }

137#[async_trait::async_trait]
138impl<S: Send + Sync + 'static> store::TransactionalStore for Store<S> {
139    async fn transaction<'a, R, F>(&self, f: F) -> anyhow::Result<R>
140    where
141        R: Send + 'a,
142        F: Send + 'a,
143        F: for<'r> FnOnce(&'r mut Connection<'_, S>) -> ScopedBoxFuture<'a, 'r, anyhow::Result<R>>,
144    {
145        let mut conn = self.connect().await?;
146        let r = f(&mut conn).await?;
147
148        let Some((pipeline, watermark)) = conn.watermark else {
149            bail!("No watermark set during transaction");
150        };
151
152        self.0
153            .queue
154            .get()
155            .context("Synchronizer not running for store")?
156            .get(pipeline.as_str())
157            .with_context(|| format!("No {pipeline:?} synchronizer queue"))?
158            .send((watermark, conn.batch))
159            .await
160            .map_err(|_| anyhow!("{pipeline:?} synchronizer queue closed"))?;
161
162        Ok(r)
163    }
164}

166#[async_trait::async_trait]
167impl<S: Send + Sync> store::Connection for Connection<'_, S> {
168    async fn init_watermark(&mut self, pipeline_task: &str, _: u64) -> anyhow::Result<Option<u64>> {
169        Ok(self
170            .committer_watermark(pipeline_task)
171            .await?
172            .map(|w| w.checkpoint_hi_inclusive))
173    }
174
175    async fn committer_watermark(
176        &mut self,
177        pipeline_task: &str,
178    ) -> anyhow::Result<Option<CommitterWatermark>> {
179        Ok(self
180            .store
181            .0
182            .db
183            .commit_watermark(pipeline_task)?
184            .map(Into::into))
185    }
186
187    async fn reader_watermark(
188        &mut self,
189        _pipeline: &'static str,
190    ) -> anyhow::Result<Option<store::ReaderWatermark>> {
191        Ok(None)
192    }
193
194    async fn pruner_watermark(
195        &mut self,
196        _pipeline: &'static str,
197        _delay: Duration,
198    ) -> anyhow::Result<Option<store::PrunerWatermark>> {
199        Ok(None)
200    }
201
202    async fn set_committer_watermark(
203        &mut self,
204        pipeline_task: &str,
205        watermark: CommitterWatermark,
206    ) -> anyhow::Result<bool> {
207        self.watermark = Some((pipeline_task.to_string(), watermark.into()));
208        Ok(true)
209    }
210
211    async fn set_reader_watermark(
212        &mut self,
213        _pipeline: &'static str,
214        _reader_lo: u64,
215    ) -> anyhow::Result<bool> {
216        bail!("Pruning not supported by this store");
217    }
218
219    async fn set_pruner_watermark(
220        &mut self,
221        _pipeline: &'static str,
222        _pruner_hi: u64,
223    ) -> anyhow::Result<bool> {
224        bail!("Pruning not supported by this store");
225    }
226}

266    async fn wait_until<F, R>(f: F) -> Result<(), Elapsed>
267    where
268        F: Fn() -> R,
269        R: Future<Output = bool>,
270    {
271        time::timeout(Duration::from_millis(500), async move {
272            let mut interval = time::interval(Duration::from_millis(100));
273            loop {
274                interval.tick().await;
275                if f().await {
276                    return;
277                }
278            }
279        })
280        .await
281    }

283    fn has_range(store: &Store<TestSchema>, lo: Option<u64>, hi: Option<u64>) -> bool {
284        store.db().snapshot_range(u64::MAX).is_some_and(|s| {
285            lo.is_none_or(|lo| lo == s.start().checkpoint_hi_inclusive)
286                && hi.is_none_or(|hi| hi == s.end().checkpoint_hi_inclusive)
287        })
288    }

507    #[tokio::test]
508    async fn test_multiple_pipeline_catchup() {
509        let d = tempfile::tempdir().unwrap();
510        let snapshots = 4;
511
512        {
513            // Initialize the database with some data for one of the pipelines.
514            let db_options: rocksdb::Options = DbConfig::default().into();
515            let db = Arc::new(
516                Db::open(
517                    d.path().join("db"),
518                    db_options.clone(),
519                    snapshots,
520                    TestSchema::cfs(&db_options),
521                )
522                .unwrap(),
523            );
524
525            let schema = TestSchema::open(&db).unwrap();
526
527            let mut batch = rocksdb::WriteBatch::default();
528            schema.b.insert(42, "x".to_owned(), &mut batch).unwrap();
529            db.write("b", CommitterWatermark::new_for_testing(0).into(), batch)
530                .unwrap();
531        }
532
533        let store: Store<TestSchema> =
534            Store::open(d.path().join("db"), DbConfig::default(), 4, None).unwrap();
535
536        let stride = 1;
537        let buffer_size = 10;
538        let first_checkpoint = None;
539        let mut sync = Synchronizer::new(store.db().clone(), stride, buffer_size, first_checkpoint);
540
541        sync.register_pipeline("a").unwrap();
542        sync.register_pipeline("b").unwrap();
543        let _svc = store.sync(sync).unwrap();
544
545        // The pipelines are not in sync to begin with, so the synchronizer is waiting for the
546        // writes for the other pipeline in order to take a snapshot.
547        wait_until(|| async { has_range(&store, None, None) })
548            .await
549            .unwrap_err();
550
551        // Further writes to the pipeline that is ahead will be held back.
552        write(&store, "b", 1, |s, b| {
553            s.b.insert(42, "y".to_owned(), b)?;
554            Ok(())
555        })
556        .await
557        .unwrap();
558
559        // Further writes to the pipeline that is ahead will be held back.
560        wait_until(|| async { has_range(&store, None, None) })
561            .await
562            .unwrap_err();
563
564        write(&store, "a", 0, |s, b| {
565            s.a.insert("x".to_owned(), 42, b)?;
566            Ok(())
567        })
568        .await
569        .unwrap();
570
571        // After the other pipeline was caught up, the synchronizer will take the snapshot, but it
572        // will not yet make the subsequent write to the other pipeline available.
573        wait_until(|| async { has_range(&store, None, Some(0)) })
574            .await
575            .unwrap();
576
577        let s = store.schema();
578        assert_eq!(s.a.get(0, "x".to_owned()).unwrap(), Some(42));
579        assert_eq!(s.b.get(0, 42).unwrap(), Some("x".to_owned()));
580
581        // Catch up the first pipeline without writing any further data.
582        write(&store, "a", 1, |_, _| Ok(())).await.unwrap();
583        wait_until(|| async { has_range(&store, None, Some(1)) })
584            .await
585            .unwrap();
586
587        let s = store.schema();
588        assert_eq!(s.a.get(1, "x".to_owned()).unwrap(), Some(42));
589        assert_eq!(s.b.get(1, 42).unwrap(), Some("y".to_owned()));
590    }

677            Store::open(d.path().join("db"), DbConfig::default(), 4, None).unwrap();
678
679        let stride = 3;
680        let buffer_size = 10;
681        let first_checkpoint = None;
682        let mut sync = Synchronizer::new(store.db().clone(), stride, buffer_size, first_checkpoint);
683
684        sync.register_pipeline("test").unwrap();
685        let _svc = store.sync(sync).unwrap();
686
687        // Write a run of checkpoints.
688        for cp in 0..=10 {
689            write(&store, "test", cp, move |s, b| {
690                s.a.insert("x".to_owned(), cp * 3, b)?;
691                s.b.insert(cp * 3, "x".to_owned(), b)?;
692                Ok(())
693            })
694            .await
695            .unwrap();
696        }
697
698        // The synchronizer will take a snapshot before every `stride`-th checkpoint.
699        wait_until(|| async { has_range(&store, Some(2), Some(8)) })
700            .await
701            .unwrap();
702
703        let d = store.db();
704        let s = store.schema();
705
706        assert_eq!(d.snapshots(), 3);
707        for cp in (2..10).step_by(stride as usize) {
708            assert_eq!(s.a.get(cp, "x".to_owned()).unwrap(), Some(cp * 3));
709            assert_eq!(s.b.get(cp, cp * 3).unwrap(), Some("x".to_owned()));
710        }
711
712        // Querying the snapshot range at the latest checkpoint does the same thing as an unbounded
713        // range request.
714        assert_eq!(
715            Some(8),
716            d.snapshot_range(8).map(|r| r.end().checkpoint_hi_inclusive)
717        );
718
719        // Going one checkpoint back causes the range to drop back by the stride.
720        assert_eq!(
721            Some(5),
722            d.snapshot_range(7).map(|r| r.end().checkpoint_hi_inclusive)
723        );
724
725        // Going back beyond the first checkpoint results in an empty range.
726        assert_eq!(None, d.snapshot_range(1));
727    }

777
778        write(&store, "test", 0, |s, b| {
779            s.a.insert("x".to_owned(), 42, b)?;
780            Ok(())
781        })
782        .await
783        .unwrap();
784

786            s.a.insert("y".to_owned(), 43, b)?;
787            Ok(())
788        })
789        .await
790        .unwrap();
791
792        // The out of order batch will appear to succeed, but the synchronizer will detect the
793        // situation and stop gracefully, with an error.
794        time::timeout(Duration::from_millis(500), svc.join())
795            .await
796            .unwrap()
797            .unwrap_err();
798
799        // The first write made it through, but the second one did not.
800        let s = store.schema();
801        let db = store.db();
802        assert_eq!(db.snapshots(), 1);
803        assert_eq!(
804            db.snapshot_range(u64::MAX)
805                .map(|s| s.end().checkpoint_hi_inclusive),
806            Some(0)
807        );
808        assert_eq!(s.a.get(0, "x".to_owned()).unwrap(), Some(42));
809        assert_eq!(s.a.get(0, "y".to_owned()).unwrap(), None);
810    }
