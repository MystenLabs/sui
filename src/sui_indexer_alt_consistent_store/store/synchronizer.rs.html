18/// The synchronizer will emit a message if it has been waiting to synchronize with other tasks for
19/// this long without making progress.
20const SLOW_SYNC_WARNING_THRESHOLD: Duration = Duration::from_secs(60);
21
22/// A service that coordinates writes to a database from various registered pipelines, with
23/// generating snapshots for that database. The synchronizer ensures that all pipelines have made
24/// the same amount of progress before taking a database-wide snapshot.
25pub(crate) struct Synchronizer {
26    db: Arc<Db>,
27
28    /// The last watermark written to the database for each registered pipeline. The value is
29    /// `None` if the database has not yet seen a write for that pipeline.
30    last_watermarks: HashMap<&'static str, Option<Watermark>>,
31
32    /// The first checkpoint to be fetched across any pipeline.
33    first_checkpoint: u64,
34
35    /// A snapshot is taken every `stride` checkpoints, after pipelines have caught up with each
36    /// other.
37    stride: u64,
38
39    /// The size of queues that feed each synchronizer task.
40    buffer_size: usize,
41
42    // Cancellation token shared among all synchronizer tasks.
43    cancel: CancellationToken,
44}
45
46/// Write access to each pipeline's synchronizer task.
47pub(super) type Queue = HashMap<&'static str, mpsc::Sender<(Watermark, rocksdb::WriteBatch)>>;
48
49impl Synchronizer {
50    /// Create a new synchronizer service that coordinates write and snapshots to `db`.
51    ///
52    /// `stride` controls the number of checkpoints between snapshots, `buffer_size` controls the
53    /// size of the channels that feed each synchronizer task.
54    ///
55    /// `first_checkpoint` is the first checkpoint the service expects to see written to for
56    /// completely new pipelines. If `None`, the first checkpoint is assumed to be `0`.
57    ///
58    /// The service must be started by calling [Self::run], and it will stop if signalled on
59    /// `cancel`, if it has been instructed to write data out-of-order, or if a write fails.
60    pub(crate) fn new(
61        db: Arc<Db>,
62        stride: u64,
63        buffer_size: usize,
64        first_checkpoint: Option<u64>,
65        cancel: CancellationToken,
66    ) -> Self {
67        Self {
68            db,
69            last_watermarks: HashMap::new(),
70            first_checkpoint: first_checkpoint.unwrap_or(0),
71            stride,
72            buffer_size,
73            cancel,
74        }
75    }
76
77    /// Register a new pipeline with the synchronizer service. The synchronizer will spin up a task
78    /// for each pipeline, and make a channel available to send writes to that task, when it is
79    /// started using [`Self::run`].
80    ///
81    /// Fails if the database fails to return the pipeline's watermark -- registering a brand new
82    /// pipeline is not an error.
83    pub(crate) fn register_pipeline(&mut self, pipeline: &'static str) -> anyhow::Result<()> {
84        let watermark = self
85            .db
86            .commit_watermark(pipeline)
87            .with_context(|| format!("Failed to get {pipeline} initial watermark"))?;
88
89        self.last_watermarks.insert(pipeline, watermark);
90        Ok(())
91    }
92
93    /// Start the service, accepting writes for registered pipelines. This consumes the service and
94    /// returns a `JoinHandle<()>` that will complete when all its tasks have completed, and the
95    /// `queue` data structure which gives access to the write side of the channels feeding each
96    /// task.
97    pub(super) fn run(self) -> anyhow::Result<(JoinHandle<()>, Queue)> {
98        let mut queue = Queue::new();
99        let mut tasks = Vec::new();
100
101        let pre_snap = Arc::new(Barrier::new(self.last_watermarks.len()));
102        let post_snap = Arc::new(Barrier::new(self.last_watermarks.len()));
103
104        // Calculate the first checkpoint we will have data for, across all pipelines.
105        let Some(init_checkpoint) = self
106            .last_watermarks
107            .values()
108            .map(|w| w.map_or(self.first_checkpoint, |w| w.checkpoint_hi_inclusive))
109            .max()
110        else {
111            bail!("No pipelines registered with the synchronizer");
112        };
113
114        // The next snapshot should be taken at the next stride boundary after that initial
115        // checkpoint.
116        let next_snapshot_checkpoint = ((init_checkpoint / self.stride) + 1) * self.stride;
117
118        for (pipeline, last_watermark) in self.last_watermarks {
119            let (tx, rx) = mpsc::channel(self.buffer_size);
120
121            queue.insert(pipeline, tx);
122            tasks.push(synchronizer(
123                self.db.clone(),
124                rx,
125                pipeline,
126                self.first_checkpoint,
127                self.stride,
128                next_snapshot_checkpoint,
129                last_watermark,
130                pre_snap.clone(),
131                post_snap.clone(),
132                self.cancel.child_token(),
133            ));
134        }
135
136        let handle = tokio::spawn(async move {
137            future::join_all(tasks).await;
138            info!("Synchronizer gracefully shut down");
139        });
140
141        Ok((handle, queue))
142    }
143}

145/// The synchronizer task is responsible for landing writes to the database for a given `pipeline`.
146/// It also coordinates with other synchronizers to take snapshots of the database every `stride`
147/// checkpoints, starting from before the write of the `next_snapshot_checkpoint`.
148///
149/// Data arrives as a batch-per-checkpoint on `rx`, and must arrive in checkpoint sequence number
150/// order (the synchronizer will report an error and stop if it detects an out-of-order batch).
151///
152/// `pre_snap` and `post_snap` are barriers shared among all synchronizers -- synchronizers wait on
153/// `pre_snap` before a snapshot is to be taken, and on `post_snap` after the snapshot is taken
154/// (and data from future checkpoints can be written).
155///
156/// The task will stop when it receives a shutdown signal on `cancel`, or if it detects an issue
157/// during writes (an out-of-order batch, or an error during writes).
158fn synchronizer(
159    db: Arc<Db>,
160    mut rx: mpsc::Receiver<(Watermark, rocksdb::WriteBatch)>,
161    pipeline: &'static str,
162    first_checkpoint: u64,
163    stride: u64,
164    mut next_snapshot_checkpoint: u64,
165    mut current_watermark: Option<Watermark>,
166    pre_snap: Arc<Barrier>,
167    post_snap: Arc<Barrier>,
168    cancel: CancellationToken,
169) -> JoinHandle<()> {
170    tokio::spawn(async move {
171        loop {
172            let next_checkpoint = current_watermark
173                .as_ref()
174                .map(|w| w.checkpoint_hi_inclusive + 1)
175                .unwrap_or(first_checkpoint);
176
177            match next_snapshot_checkpoint.cmp(&next_checkpoint) {
178                // The next checkpoint should be included in the next snapshot, so allow it to be
179                // written.
180                Ordering::Greater => {}
181
182                // If the next checkpoint is more than one checkpoint ahead of the next snapshot,
183                // something has gone wrong.
184                Ordering::Less => {
185                    error!(
186                        pipeline,
187                        next_snapshot_checkpoint, next_checkpoint, "Missed snapshot"
188                    );
189                    break;
190                }
191
192                // The next checkpoint does not belong in the next snapshot, so wait for other
193                // synchronizers to reach this point, and take the snapshot before proceeding.
194                //
195                // One arbitrary task (the "leader") is responsible for taking the snapshot, the others
196                // just bump their own synchronization point and wait.
197                Ordering::Equal => {
198                    tokio::select! {
199                        w = with_slow_future_monitor(pre_snap.wait(), SLOW_SYNC_WARNING_THRESHOLD, || {
200                            warn!(pipeline, "Synchronizer stuck, pre-snapshot")
201                        }) => if w.is_leader() {
202                            if let Some(watermark) = current_watermark {
203                                db.take_snapshot(watermark);
204                            } else {
205                                error!(pipeline, next_snapshot_checkpoint, "No watermark available for snapshot");
206                                break;
207                            }
208                        },
209
210                        _ = cancel.cancelled() => {
211                            info!(pipeline, "Shutdown received before pre-snapshot barrier");
212                            break;
213                        }
214                    }
215
216                    next_snapshot_checkpoint += stride;
217                    tokio::select! {
218                        _ = with_slow_future_monitor(post_snap.wait(), SLOW_SYNC_WARNING_THRESHOLD, || {
219                            warn!(pipeline, "Synchronizer stuck, post-snapshot")
220                        }) => {}
221                        _ = cancel.cancelled() => {
222                            info!(pipeline, "Shutdown received before post-snapshot barrier");
223                            break;
224                        }
225                    }
226                }
227            }
228
229            tokio::select! {
230                _ = cancel.cancelled() => {
231                    info!(pipeline, "Shutdown received waiting for batch");
232                    break;
233                }
234
235                Some((watermark, batch)) = rx.recv() => {
236                    debug!(pipeline, ?watermark, "Received batch");
237                    if watermark.checkpoint_hi_inclusive != next_checkpoint {
238                        error!(pipeline, ?watermark, next_checkpoint, "Out-of-order batch");
239                        break;
240                    }
241
242                    if let Err(e) = db.write(pipeline, watermark, batch) {
243                        error!(pipeline, ?e, "Failed to write batch");
244                        break;
245                    }
246
247                    current_watermark = Some(watermark);
248                }
249            }
250        }
251
252        info!(
253            pipeline,
254            next_snapshot_checkpoint, watermark = ?current_watermark, "Stopping sync"
255        );
256    })
257}
