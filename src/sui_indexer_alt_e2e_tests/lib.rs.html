64/// A simulation of the network, accompanied by off-chain services (database, indexer, RPC),
65/// connected by local data ingestion.
66pub struct FullCluster {
67    /// A simulation of the network, executing transactions and producing checkpoints.
68    executor: Simulacrum,
69
70    /// The off-chain services (database, indexer, RPC) that are ingesting data from the
71    /// simulation.
72    offchain: OffchainCluster,
73
74    /// Temporary directory to store checkpoint information in, so that the indexer can pick it up.
75    #[allow(unused)]
76    temp_dir: TempDir,
77}
78
79/// A collection of the off-chain services (an indexer, a database, and JSON-RPC/GraphQL servers
80/// that read from that database), grouped together to simplify set-up and tear-down for tests. The
81/// included RPC servers do not support transaction dry run and execution.
82///
83/// The database is temporary, and will be cleaned up when the cluster is dropped, and the RPCs are
84/// set-up to listen on a random, available port, to avoid conflicts when multiple instances are
85/// running concurrently in the same process.
86pub struct OffchainCluster {
87    /// The address the consistent store is listening on.
88    consistent_listen_address: SocketAddr,
89
90    /// The address the JSON-RPC server is listening on.
91    jsonrpc_listen_address: SocketAddr,
92
93    /// The address the GraphQL server is listening on.
94    graphql_listen_address: SocketAddr,
95
96    /// Read access to the temporary database.
97    db: Db,
98
99    /// The pipelines that the indexer is populating.
100    pipelines: Vec<&'static str>,
101
102    /// A handle to the indexer task -- it will stop when the `cancel` token is triggered (or
103    /// earlier of its own accord).
104    indexer: JoinHandle<()>,
105
106    /// A handle to the consistent store task -- it will stop when the `cancel` token is triggered
107    /// (or earlier of its own accord).
108    consistent_store: JoinHandle<()>,
109
110    /// A handle to the JSON-RPC server task -- it will stop when the `cancel` token is triggered
111    /// (or earlier of its own accord).
112    jsonrpc: JoinHandle<()>,
113
114    /// A handle to the GraphQL server task -- it will stop when the `cancel` token is triggered
115    /// (or earlier of its own accord).
116    graphql: JoinHandle<()>,
117
118    /// Hold on to the database so it doesn't get dropped until the cluster is stopped.
119    #[allow(unused)]
120    database: TempDb,
121
122    /// Hold on to the temporary directory where the consistent store writes its data, so it
123    /// doesn't get cleaned up until the cluster is stopped.
124    #[allow(unused)]
125    dir: TempDir,
126
127    /// This token controls the clean up of the cluster.
128    cancel: CancellationToken,
129}

216    /// was created, and wait for the off-chain services to ingest it. Returns the checkpoint

218    pub async fn create_checkpoint(&mut self) -> VerifiedCheckpoint {
219        let checkpoint = self.executor.create_checkpoint();
220        let indexer = self
221            .offchain
222            .wait_for_indexer(checkpoint.sequence_number, Duration::from_secs(10));
223        let consistent_store = self
224            .offchain

226        let graphql = self
227            .offchain
228            .wait_for_graphql(checkpoint.sequence_number, Duration::from_secs(10));
229
230        try_join!(indexer, consistent_store, graphql)
231            .expect("Timed out waiting for indexer and consistent store");

234    }

302impl OffchainCluster {
303    /// Construct a new off-chain cluster and spin up its constituent services.
304    ///
305    /// - `indexer_args`, `client_args`, and `indexer_config` control the indexer. In particular
306    ///   `client_args` is used to configure the client that the indexer uses to fetch checkpoints.
307    /// - `jsonrpc_config` controls the JSON-RPC server.
308    /// - `graphql_config` controls the GraphQL server.
309    /// - `registry` is used to register metrics for the indexer, JSON-RPC, and GraphQL servers.
310    pub async fn new(
311        client_args: ClientArgs,
312        OffchainClusterConfig {
313            indexer_args,
314            consistent_indexer_args,
315            fullnode_args,
316            indexer_config,
317            consistent_config,
318            jsonrpc_config,
319            graphql_config,
320            bootstrap_genesis,
321        }: OffchainClusterConfig,
322        registry: &prometheus::Registry,
323        cancel: CancellationToken,
324    ) -> anyhow::Result<Self> {
325        let consistent_port = get_available_port();
326        let consistent_listen_address =
327            SocketAddr::new(IpAddr::V4(Ipv4Addr::LOCALHOST), consistent_port);
328
329        let jsonrpc_port = get_available_port();
330        let jsonrpc_listen_address = SocketAddr::new(IpAddr::V4(Ipv4Addr::LOCALHOST), jsonrpc_port);
331
332        let graphql_port = get_available_port();
333        let graphql_listen_address = SocketAddr::new(IpAddr::V4(Ipv4Addr::LOCALHOST), graphql_port);
334
335        let database = TempDb::new().context("Failed to create database")?;
336        let database_url = database.database().url();
337
338        let dir = tempfile::tempdir().context("Failed to create temporary directory")?;
339        let rocksdb_path = dir.path().join("rocksdb");
340
341        let consistent_args = ConsistentArgs {
342            rpc_listen_address: consistent_listen_address,
343            tls: ConsistentTlsArgs::default(),
344        };
345
346        let jsonrpc_args = JsonRpcArgs {
347            rpc_listen_address: jsonrpc_listen_address,
348            ..Default::default()
349        };
350
351        let graphql_args = GraphQlArgs {
352            rpc_listen_address: graphql_listen_address,
353            no_ide: true,
354        };
355
356        let db = Db::for_read(database_url.clone(), DbArgs::default())
357            .await
358            .context("Failed to connect to database")?;
359
360        let indexer = setup_indexer(
361            database_url.clone(),
362            DbArgs::default(),
363            indexer_args,
364            client_args.clone(),
365            indexer_config,
366            bootstrap_genesis,
367            registry,
368            cancel.child_token(),
369        )
370        .await
371        .context("Failed to setup indexer")?;
372
373        let pipelines: Vec<_> = indexer.pipelines().collect();
374        let indexer = indexer.run().await.context("Failed to start indexer")?;
375
376        let consistent_store = start_consistent_store(
377            rocksdb_path,
378            consistent_indexer_args,
379            client_args,
380            consistent_args,
381            "0.0.0",
382            consistent_config,
383            registry,
384            cancel.child_token(),
385        )
386        .await
387        .context("Failed to start Consistent Store")?;
388
389        let jsonrpc = start_jsonrpc(
390            Some(database_url.clone()),
391            None,
392            DbArgs::default(),
393            BigtableArgs::default(),
394            jsonrpc_args,
395            JsonRpcNodeArgs::default(),
396            SystemPackageTaskArgs::default(),
397            jsonrpc_config,
398            registry,
399            cancel.child_token(),
400        )
401        .await
402        .context("Failed to start JSON-RPC server")?;
403
404        let consistent_reader_args = ConsistentReaderArgs {
405            consistent_store_url: Some(
406                Url::parse(&format!("http://{consistent_listen_address}")).unwrap(),
407            ),
408            consistent_store_statement_timeout_ms: None,
409        };
410
411        let graphql = start_graphql(
412            Some(database_url.clone()),
413            None,
414            fullnode_args,
415            DbArgs::default(),
416            BigtableArgs::default(),
417            consistent_reader_args,
418            graphql_args,
419            SystemPackageTaskArgs::default(),
420            "0.0.0",
421            graphql_config,
422            pipelines.iter().map(|p| p.to_string()).collect(),
423            registry,
424            cancel.child_token(),
425        )
426        .await
427        .context("Failed to start GraphQL server")?;
428
429        Ok(Self {
430            consistent_listen_address,
431            jsonrpc_listen_address,
432            graphql_listen_address,
433            db,
434            pipelines,
435            indexer,
436            consistent_store,
437            jsonrpc,
438            graphql,
439            database,
440            dir,
441            cancel,
442        })
443    }

468    /// Returns the latest checkpoint that we have all data for in the database, according to the
469    /// watermarks table. Returns `None` if any of the expected pipelines are missing data.
470    pub async fn latest_checkpoint(&self) -> anyhow::Result<Option<u64>> {
471        use watermarks::dsl as w;
472
473        let mut conn = self
474            .db
475            .connect()
476            .await
477            .context("Failed to connect to database")?;
478
479        let latest: HashMap<String, i64> = w::watermarks
480            .select((w::pipeline, w::checkpoint_hi_inclusive))
481            .filter(w::pipeline.eq_any(&self.pipelines))
482            .load(&mut conn)
483            .await?
484            .into_iter()
485            .collect();
486
487        if latest.len() != self.pipelines.len() {
488            return Ok(None);
489        }
490
491        Ok(latest.into_values().min().map(|l| l as u64))
492    }

494    /// Returns the latest checkpoint that the pruner is willing to prune up to for the given
495    /// `pipeline`.
496    pub async fn latest_pruner_checkpoint(&self, pipeline: &str) -> anyhow::Result<Option<u64>> {
497        use watermarks::dsl as w;
498
499        let mut conn = self
500            .db
501            .connect()
502            .await
503            .context("Failed to connect to database")?;
504
505        let latest: Option<i64> = w::watermarks
506            .select(w::reader_lo)
507            .filter(w::pipeline.eq(pipeline))
508            .first(&mut conn)
509            .await
510            .optional()?;
511
512        Ok(latest.map(|l| l as u64))
513    }

515    /// Returns the latest checkpoint that the consistent store is aware of.
516    pub async fn latest_consistent_store_checkpoint(&self) -> anyhow::Result<u64> {
517        ConsistentServiceClient::connect(self.consistent_store_url().to_string())
518            .await
519            .context("Failed to connect to Consistent Store")?

524            .max_checkpoint
525            .context("Consistent Store has not started yet")
526    }

528    /// Returns the latest checkpoint that the GraphQL service is aware of.
529    pub async fn latest_graphql_checkpoint(&self) -> anyhow::Result<u64> {
530        let query = json!({
531            "query": "query { checkpoint { sequenceNumber } }"
532        });
533
534        let client = Client::new();
535        let request = client.post(self.graphql_url()).json(&query);
536        let response = request
537            .send()
538            .await
539            .context("Request to GraphQL server failed")?;
540
541        let body: Value = response
542            .json()
543            .await
544            .context("Failed to parse GraphQL response")?;
545
546        let sequence_number = body
547            .pointer("/data/checkpoint/sequenceNumber")
548            .context("Failed to find checkpoint sequence number in response")?;
549
550        let sequence_number: i64 = serde_json::from_value(sequence_number.clone())
551            .context("Failed to parse sequence number as i64")?;
552
553        ensure!(sequence_number != i64::MAX, "Indexer has not started yet");
554
555        Ok(sequence_number as u64)
556    }

588    /// Waits until the indexer has caught up to the given `checkpoint`, or the `timeout` is
589    /// reached (an error).
590    pub async fn wait_for_indexer(
591        &self,
592        checkpoint: u64,
593        timeout: Duration,
594    ) -> Result<(), Elapsed> {
595        tokio::time::timeout(timeout, async move {
596            let mut interval = interval(Duration::from_millis(200));
597            loop {
598                interval.tick().await;
599                if matches!(self.latest_checkpoint().await, Ok(Some(l)) if l >= checkpoint) {
600                    break;
601                }
602            }
603        })
604        .await
605    }

690/// Returns ClientArgs that use a temporary local ingestion path and the TempDir of that path.
691pub fn local_ingestion_client_args() -> (ClientArgs, TempDir) {

693        .context("Failed to create data ingestion path")
694        .unwrap();
695    let client_args = ClientArgs {
696        local_ingestion_path: Some(temp_dir.path().to_owned()),
697        remote_store_url: None,
698        rpc_api_url: None,
699        rpc_username: None,
700        rpc_password: None,
701    };
702    (client_args, temp_dir)
703}

705/// Writes a checkpoint file to the given path.
706pub async fn write_checkpoint(path: &Path, checkpoint_data: CheckpointData) -> anyhow::Result<()> {
707    let file_name = format!("{}.chk", checkpoint_data.checkpoint_summary.sequence_number);
708    let file_path = path.join(file_name);
709    let blob = Blob::encode(&checkpoint_data, BlobEncoding::Bcs)?;
710    fs::write(file_path, blob.to_bytes())?;
711    Ok(())
712}
