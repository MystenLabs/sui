4//! Grouping of pipelines into ingestion cohorts based on how far behind the network tip each
5//! pipeline's resume checkpoint is. Pipelines in the same cohort share one ingestion service, so
6//! a far-behind pipeline only applies backpressure to similarly-lagging peers, not to pipelines
7//! that are already at the tip.
8
9use crate::PendingPipeline;
10
11/// Default for [`IngestionConfig::min_cohort_boundary`]: a cohort's boundary (its maximum member
12/// distance from the tip) is at least this many checkpoints, so that pipelines that are roughly
13/// caught up are not fragmented into many tiny cohorts, each with its own ingestion service.
14///
15/// [`IngestionConfig::min_cohort_boundary`]: crate::ingestion::IngestionConfig::min_cohort_boundary
16pub(crate) const DEFAULT_MIN_COHORT_BOUNDARY: u64 = 25_000;
17
18/// Group pipelines into cohorts by their distance from the network tip.
19///
20/// Each pipeline's distance is how far behind `latest_checkpoint` it will resume from. `pipelines`
21/// may be in any order; they are sorted (nearest-tip first) internally. The closest unassigned
22/// pipeline (distance `d`) seeds a cohort that absorbs every pipeline whose distance is at most
23/// `max(2 * d, min_cohort_boundary)`; the next unassigned pipeline seeds the next cohort, and so
24/// on.
25///
26/// Returns the cohorts ordered nearest-tip first, each holding its members in ascending distance
27/// order (pipelines at equal distances keep registration order).
28pub(crate) fn cohorts(
29    mut pipelines: Vec<PendingPipeline>,
30    latest_checkpoint: u64,
31    min_cohort_boundary: u64,
32) -> Vec<Vec<PendingPipeline>> {
33    // Stable sort: pipelines at equal distances keep registration order.
34    pipelines.sort_by_key(|p| latest_checkpoint.saturating_sub(p.next_checkpoint));
35
36    // The closest unassigned pipeline seeds a cohort that absorbs every peer within
37    // `max(2 * its distance, min_cohort_boundary)`; the first pipeline beyond that boundary seeds
38    // the next cohort, and so on.
39    let mut cohorts: Vec<Vec<PendingPipeline>> = vec![];
40    let mut boundary = None;
41    for pipeline in pipelines {
42        let dist = latest_checkpoint.saturating_sub(pipeline.next_checkpoint);
43        if boundary.is_none_or(|b| dist > b) {
44            cohorts.push(vec![]);
45            boundary = Some(dist.saturating_mul(2).max(min_cohort_boundary));
46        }
47        cohorts.last_mut().unwrap().push(pipeline);
48    }
49    cohorts
50}

52#[cfg(test)]
53mod tests {
54    use super::*;
55
56    /// A pipeline resuming exactly `distance` checkpoints behind the tip. Tests fix the tip at
57    /// `u64::MAX` so scenarios can be written directly in distances and the overflow case is
58    /// reachable.
59    fn at(distance: u64) -> PendingPipeline {
60        PendingPipeline {
61            name: "test",
62            next_checkpoint: u64::MAX - distance,
63            build: Box::new(|_| unreachable!("cohort grouping never builds pipelines")),
64        }
65    }
66
67    /// Group `pipelines` (tip at `u64::MAX`) and read each cohort back out as member distances.
68    fn grouped(pipelines: Vec<PendingPipeline>, min_cohort_boundary: u64) -> Vec<Vec<u64>> {
69        cohorts(pipelines, u64::MAX, min_cohort_boundary)
70            .iter()
71            .map(|cohort| {
72                cohort
73                    .iter()
74                    .map(|p| u64::MAX - p.next_checkpoint)
75                    .collect()
76            })
77            .collect()
78    }
79
80    #[test]
81    fn test_empty() {
82        assert_eq!(
83            grouped(vec![], DEFAULT_MIN_COHORT_BOUNDARY),
84            Vec::<Vec<u64>>::new()
85        );
86    }
87
88    #[test]
89    fn test_singleton() {
90        assert_eq!(
91            grouped(vec![at(0)], DEFAULT_MIN_COHORT_BOUNDARY),
92            vec![vec![0]]
93        );
94    }
95
96    #[test]
97    fn test_all_equal_distances() {
98        assert_eq!(
99            grouped(vec![at(42), at(42), at(42)], DEFAULT_MIN_COHORT_BOUNDARY),
100            vec![vec![42, 42, 42]]
101        );
102    }
103
104    /// Input may be given in any order; cohorts and their members come back nearest-tip first.
105    #[test]
106    fn test_sorts_unsorted_input() {
107        assert_eq!(
108            grouped(
109                vec![at(200_000), at(10), at(70_000), at(30_000)],
110                DEFAULT_MIN_COHORT_BOUNDARY
111            ),
112            vec![vec![10], vec![30_000], vec![70_000], vec![200_000]],
113        );
114    }
115
116    /// The minimum boundary is inclusive: a seed at the tip absorbs pipelines exactly
117    /// `min_cohort_boundary` away, but not one checkpoint further.
118    #[test]
119    fn test_min_boundary_inclusive() {
120        assert_eq!(
121            grouped(
122                vec![
123                    at(0),
124                    at(DEFAULT_MIN_COHORT_BOUNDARY),
125                    at(DEFAULT_MIN_COHORT_BOUNDARY + 1)
126                ],
127                DEFAULT_MIN_COHORT_BOUNDARY,
128            ),
129            vec![
130                vec![0, DEFAULT_MIN_COHORT_BOUNDARY],
131                vec![DEFAULT_MIN_COHORT_BOUNDARY + 1]
132            ],
133        );
134    }
135
136    /// Once the seed is further out than half the minimum boundary, twice its distance takes
137    /// over as the cohort boundary.
138    #[test]
139    fn test_double_seed_distance_boundary() {
140        assert_eq!(
141            grouped(
142                vec![at(20_000), at(40_000), at(40_001)],
143                DEFAULT_MIN_COHORT_BOUNDARY
144            ),
145            vec![vec![20_000, 40_000], vec![40_001]]
146        );
147    }
148
149    /// Each cohort's boundary is derived from its own seed, not the previous cohort's.
150    #[test]
151    fn test_chained_reseeding() {
152        assert_eq!(
153            grouped(
154                vec![at(10), at(30_000), at(70_000), at(200_000)],
155                DEFAULT_MIN_COHORT_BOUNDARY
156            ),
157            vec![vec![10], vec![30_000], vec![70_000], vec![200_000]],
158        );
159    }
160
161    /// Doubling an enormous seed distance saturates instead of overflowing.
162    #[test]
163    fn test_distance_overflow_saturates() {
164        assert_eq!(
165            grouped(
166                vec![at(25_000), at(u64::MAX - 1), at(u64::MAX)],
167                DEFAULT_MIN_COHORT_BOUNDARY
168            ),
169            vec![vec![25_000], vec![u64::MAX - 1, u64::MAX]],
170        );
171    }
172
173    /// A custom (non-default) boundary is honored: with a boundary of 100 a seed at the tip
174    /// absorbs a peer 100 away but not one at 101 — whereas the default boundary would have put
175    /// all three in one cohort.
176    #[test]
177    fn test_custom_boundary() {
178        assert_eq!(
179            grouped(vec![at(0), at(100), at(101)], 100),
180            vec![vec![0, 100], vec![101]]
181        );
182        assert_eq!(
183            grouped(vec![at(0), at(100), at(101)], DEFAULT_MIN_COHORT_BOUNDARY),
184            vec![vec![0, 100, 101]]
185        );
186    }
187}
