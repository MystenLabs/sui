20/// Broadcaster task that manages checkpoint flow and spawns broadcast tasks for ranges.
21///
22/// This task:
23/// 1. Maintains an ingest_hi based on subscriber feedback
24/// 2. Spawns a broadcaster task for the requested checkpoint range. Right now the entire requested range is given to
25///    each broadcaster, but with streaming support the end of the range could be different.
26/// 3. The broadcast_range task waits on the watch channel when it hits the ingest_hi limit
27///
28/// The task will shut down if the `cancel` token is signalled, or if the `checkpoints` range completes.
29pub(super) fn broadcaster<R>(
30    checkpoints: R,
31    initial_commit_hi: Option<u64>,
32    config: IngestionConfig,
33    client: IngestionClient,
34    mut commit_hi_rx: mpsc::UnboundedReceiver<(&'static str, u64)>,
35    subscribers: Vec<mpsc::Sender<Arc<CheckpointData>>>,
36    cancel: CancellationToken,
37) -> JoinHandle<()>
38where
39    R: std::ops::RangeBounds<u64> + Send + 'static,
40{
41    tokio::spawn(async move {
42        info!("Starting broadcaster");
43
44        // Extract start and end from the range bounds
45        let start_cp = match checkpoints.start_bound() {
46            std::ops::Bound::Included(&n) => n,
47            std::ops::Bound::Excluded(&n) => n + 1,
48            std::ops::Bound::Unbounded => 0,
49        };
50
51        let end_cp = match checkpoints.end_bound() {
52            // If u64::MAX is provided as an inclusive bound, the saturating_add
53            // here will prevent overflow but the broadcaster will actually
54            // only ingest up to u64::MAX - 1, since the range is [start..end).
55            // This isn't an issue in practice since we won't see that many checkpoints
56            // in our lifetime anyway.
57            std::ops::Bound::Included(&n) => n.saturating_add(1),
58            std::ops::Bound::Excluded(&n) => n,
59            std::ops::Bound::Unbounded => u64::MAX,
60        };
61
62        let buffer_size = config.checkpoint_buffer_size as u64;
63
64        let subscribers = Arc::new(subscribers);
65
66        // Track subscriber watermarks
67        let mut subscribers_hi = HashMap::<&'static str, u64>::new();
68
69        // Initialize ingest_hi watch channel.
70        // Start with None (no backpressure) or Some if we have been provided an initial bound.
71        let initial_ingest_hi = initial_commit_hi.map(|min_hi| min_hi + buffer_size);
72        let (ingest_hi_watch_tx, ingest_hi_watch_rx) = watch::channel(initial_ingest_hi);
73
74        // Spawn the broadcaster task.
75        let mut broadcaster_handle = tokio::spawn(ingest_and_broadcast_range(
76            start_cp,
77            end_cp,
78            config.retry_interval(),
79            config.ingest_concurrency,
80            ingest_hi_watch_rx.clone(),
81            client.clone(),
82            subscribers.clone(),
83            cancel.clone(),
84        ));
85
86        loop {
87            tokio::select! {
88                _ = cancel.cancelled() => {
89                    info!("Shutdown received, stopping ingestion");
90                    break;
91                }
92
93                // Subscriber watermark update
94                // docs::#regulator (see docs/content/guides/developer/advanced/custom-indexer.mdx)
95                Some((name, hi)) = commit_hi_rx.recv() => {
96                    subscribers_hi.insert(name, hi);
97
98                    if let Some(min_hi) = subscribers_hi.values().copied().min() {
99                        let new_ingest_hi = min_hi + buffer_size;
100                        // Update the watch channel, which will notify all waiting tasks
101                        let _ = ingest_hi_watch_tx.send(Some(new_ingest_hi));
102                        debug!(ingest_hi = new_ingest_hi);
103                    }
104                }
105                // docs::/#regulator
106
107                // Handle broadcaster task completion
108                result = &mut broadcaster_handle => {
109                    match result {
110                        Ok(Ok(())) => {
111                            info!("Broadcaster completed successfully");
112                            break;
113                        }
114                        Ok(Err(Error::Cancelled)) => {
115                            info!("Broadcaster was cancelled");
116                            break;
117                        }
118                        Ok(Err(e)) => {
119                            error!("Broadcaster failed: {}", e);
120                            cancel.cancel();
121                            break;
122                        }
123                        Err(e) => {
124                            error!("Broadcaster task panicked: {}", e);
125                            cancel.cancel();
126                            break;
127                        }
128                    }
129                }
130            }
131        }
132
133        info!("Broadcaster finished");
134    })
135}

137/// Fetch and broadcasts checkpoints from a range [start..end) to subscribers.
138/// This task is ingest_hi-aware and will wait if it encounters a checkpoint
139/// beyond the current ingest_hi, resuming when ingest_hi advances to currently
140/// ingesting checkpoints.
141async fn ingest_and_broadcast_range(
142    start: u64,
143    end: u64,
144    retry_interval: Duration,
145    ingest_concurrency: usize,
146    ingest_hi_rx: watch::Receiver<Option<u64>>,
147    client: IngestionClient,
148    subscribers: Arc<Vec<mpsc::Sender<Arc<CheckpointData>>>>,
149    cancel: CancellationToken,
150) -> Result<(), Error> {
151    stream::iter(start..end)
152        .try_for_each_spawned(ingest_concurrency, |cp| {
153            let mut ingest_hi_rx = ingest_hi_rx.clone();
154            let client = client.clone();
155            let subscribers = subscribers.clone();
156
157            // One clone is for the supervisor to signal a cancel if it detects a
158            // subscriber that wants to wind down ingestion, and the other is to pass to
159            // each worker to detect cancellation.
160            // let supervisor_cancel = cancel.clone();
161            let cancel = cancel.clone();
162
163            async move {
164                // docs::#bound (see docs/content/guides/developer/advanced/custom-indexer.mdx)
165                // Wait until ingest_hi allows processing this checkpoint.
166                // None means no backpressure limit. If we get Some(hi) we wait until cp < hi.
167                // wait_for only errors if the sender is dropped (main broadcaster shut down) so
168                // we treat an error returned here as cancellation too.
169                if tokio::select! {
170                    result = ingest_hi_rx.wait_for(|hi| hi.is_none_or(|h| cp < h)) => result.is_err(),
171                    _ = cancel.cancelled() => true,
172                } {
173                    return Err(Error::Cancelled);
174                }
175                // docs::/#bound
176
177                // Fetch the checkpoint or stop if cancelled.
178                let checkpoint = tokio::select! {
179                    cp = client.wait_for(cp, retry_interval) => cp?,
180                    _ = cancel.cancelled() => {
181                        return Err(Error::Cancelled);
182                    }
183                };
184
185                // Send checkpoint to all subscribers.
186                tokio::select! {
187                    result = send_checkpoint(checkpoint, &subscribers) => {
188                        if result.is_ok() {
189                            debug!(checkpoint = cp, "Broadcasted checkpoint");
190                            Ok(())
191                        } else {
192                            // An error is returned meaning some subscriber channel has closed, which we consider
193                            // a cancellation signal for the entire ingestion.
194                            cancel.cancel();
195                            Err(Error::Cancelled)
196                        }
197                    },
198                    _ = cancel.cancelled() => Err(Error::Cancelled),
199                }
200            }
201        })
202        .await
203}

205/// Send a checkpoint to all subscribers.
206/// Returns an error if any subscriber's channel is closed.
207async fn send_checkpoint(
208    checkpoint: Arc<CheckpointData>,
209    subscribers: &[mpsc::Sender<Arc<CheckpointData>>],
210) -> Result<Vec<()>, mpsc::error::SendError<Arc<CheckpointData>>> {
211    let futures = subscribers.iter().map(|s| s.send(checkpoint.clone()));
212    try_join_all(futures).await
213}

259    /// Wait up to a second for a response on the channel, and return it, expecting this operation
260    /// to succeed.
261    async fn expect_recv<T>(rx: &mut mpsc::Receiver<T>) -> Option<T> {
262        timeout(Duration::from_secs(1), rx.recv()).await.unwrap()
263    }

265    /// Wait up to a second for a response on the channel, but expecting this operation to timeout.
266    async fn expect_timeout<T: Debug>(rx: &mut mpsc::Receiver<T>) -> Elapsed {
267        timeout(Duration::from_secs(1), rx.recv())
268            .await
269            .unwrap_err()
270    }

272    /// Verify that a channel receives checkpoints in the given range exactly once each, in any order.
273    /// We need this because with ingest concurrency the order is not guaranteed.
274    async fn expect_checkpoints_in_range<R>(rx: &mut mpsc::Receiver<Arc<CheckpointData>>, range: R)
275    where
276        R: Iterator<Item = u64>,
277    {
278        use std::collections::HashSet;
279
280        let expected: HashSet<u64> = range.collect();
281        let mut received: HashSet<u64> = HashSet::new();
282
283        for _ in 0..expected.len() {
284            let checkpoint = expect_recv(rx).await.unwrap();
285            let seq = *checkpoint.checkpoint_summary.sequence_number();
286            assert!(
287                expected.contains(&seq),
288                "Received unexpected checkpoint {seq}",
289            );
290            assert!(received.insert(seq), "Received duplicate checkpoint {seq}",);
291        }
292    }

451    #[tokio::test]
452    async fn multiple_subscribers() {
453        let (hi_tx, hi_rx) = mpsc::unbounded_channel();
454        let (subscriber_tx, mut subscriber_rx) = mpsc::channel(1);
455        let cancel = CancellationToken::new();
456
457        hi_tx.send(("a", 2)).unwrap();
458        hi_tx.send(("b", 3)).unwrap();
459
460        let mut config = test_config();
461        config.checkpoint_buffer_size = 0; // No buffer
462
463        let cps = 0..10;
464        let h_broadcaster = broadcaster(
465            cps,
466            Some(2),
467            config,
468            mock_client(),
469            hi_rx,
470            vec![subscriber_tx],
471            cancel.clone(),
472        );
473
474        expect_checkpoints_in_range(&mut subscriber_rx, 0..2).await;
475
476        // Watermark stopped because of a's watermark.
477        expect_timeout(&mut subscriber_rx).await;
478
479        // Updating b's watermark doesn't make a difference.
480        hi_tx.send(("b", 4)).unwrap();
481        expect_timeout(&mut subscriber_rx).await;
482
483        // But updating a's watermark does.
484        hi_tx.send(("a", 3)).unwrap();
485        let checkpoint = expect_recv(&mut subscriber_rx).await.unwrap();
486        assert_eq!(*checkpoint.checkpoint_summary.sequence_number(), 2);
487
488        // ...by one checkpoint.
489        expect_timeout(&mut subscriber_rx).await;
490
491        // And we can make more progress by updating it again.
492        hi_tx.send(("a", 4)).unwrap();
493        let checkpoint = expect_recv(&mut subscriber_rx).await.unwrap();
494        assert_eq!(*checkpoint.checkpoint_summary.sequence_number(), 3);
495
496        // But another update to "a" will now not make a difference, because "b" is still behind.
497        hi_tx.send(("a", 5)).unwrap();
498        expect_timeout(&mut subscriber_rx).await;
499
500        cancel.cancel();
501        h_broadcaster.await.unwrap();
502    }
