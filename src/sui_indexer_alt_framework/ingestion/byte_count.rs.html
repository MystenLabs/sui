4use bytes::Buf;
5use prometheus::IntCounter;
6use sui_http::middleware::callback::MakeCallbackHandler;
7use sui_http::middleware::callback::ResponseHandler;
8
9/// A [`MakeCallbackHandler`] that increments an [`IntCounter`] by the number of bytes observed
10/// in the response body of each request.
11#[derive(Clone)]
12pub(crate) struct ByteCountMakeCallbackHandler {
13    counter: IntCounter,
14}
15
16impl ByteCountMakeCallbackHandler {
17    pub(crate) fn new(counter: IntCounter) -> Self {
18        Self { counter }
19    }
20}
21
22impl MakeCallbackHandler for ByteCountMakeCallbackHandler {
23    type RequestHandler = ();
24    type ResponseHandler = ByteCountResponseHandler;
25
26    fn make_handler(
27        &self,
28        _request: &http::request::Parts,
29    ) -> (Self::RequestHandler, Self::ResponseHandler) {
30        (
31            (),
32            ByteCountResponseHandler {
33                counter: self.counter.clone(),
34            },
35        )
36    }
37}
38
39pub(crate) struct ByteCountResponseHandler {
40    counter: IntCounter,
41}
42
43impl ResponseHandler for ByteCountResponseHandler {
44    fn on_response(&mut self, _response: &http::response::Parts) {}
45
46    fn on_service_error<E>(&mut self, _error: &E)
47    where
48        E: std::fmt::Display + 'static,
49    {
50    }
51
52    fn on_body_chunk<B>(&mut self, chunk: &B)
53    where
54        B: Buf,
55    {
56        self.counter.inc_by(chunk.remaining() as u64);
57    }
58}
59
60#[cfg(test)]
61mod tests {
62    use super::*;
63
64    use bytes::Bytes;
65
66    fn run_byte_count(frames: &[&'static str]) -> u64 {
67        let counter = IntCounter::new("test_bytes", "test").unwrap();
68        let make = ByteCountMakeCallbackHandler::new(counter.clone());
69
70        let request = http::Request::new(());
71        let (parts, _) = request.into_parts();
72        let ((), mut handler) = make.make_handler(&parts);
73
74        for frame in frames {
75            handler.on_body_chunk(&Bytes::from_static(frame.as_bytes()));
76        }
77
78        counter.get()
79    }
80
81    #[test]
82    fn test_byte_count_0_frames() {
83        assert_eq!(run_byte_count(&[]), 0);
84    }
85
86    #[test]
87    fn test_byte_count_1_frame() {
88        assert_eq!(run_byte_count(&["a"]), 1);
89    }
90
91    #[test]
92    fn test_byte_count_2_frames() {
93        assert_eq!(run_byte_count(&["a", "bb"]), 3);
94    }
95}
