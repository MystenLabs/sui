4use anyhow::anyhow;
5use async_trait::async_trait;
6use prost_types::FieldMask;
7use sui_rpc::Client as RpcClient;
8use sui_rpc::field::FieldMaskUtil;
9use sui_rpc::proto::sui::rpc::v2::GetCheckpointRequest;
10use sui_types::full_checkpoint_content::Checkpoint;
11use tonic::Code;
12
13use crate::ingestion::ingestion_client::FetchData;
14use crate::ingestion::ingestion_client::FetchError;
15use crate::ingestion::ingestion_client::FetchResult;
16use crate::ingestion::ingestion_client::IngestionClientTrait;
17
18#[async_trait]
19impl IngestionClientTrait for RpcClient {
20    async fn fetch(&self, checkpoint: u64) -> FetchResult {
21        let request: GetCheckpointRequest = GetCheckpointRequest::by_sequence_number(checkpoint)
22            .with_read_mask(FieldMask::from_paths([
23                "summary.bcs",
24                "signature",
25                "contents.bcs",
26                "transactions.transaction.bcs",
27                "transactions.effects.bcs",
28                "transactions.effects.unchanged_loaded_runtime_objects",
29                "transactions.events.bcs",
30                "objects.objects.bcs",
31            ]));
32
33        let response = self
34            .clone()
35            .ledger_client()
36            .get_checkpoint(request)
37            .await
38            .map_err(|status| match status.code() {
39                Code::NotFound => FetchError::NotFound,
40                _ => FetchError::Transient {
41                    reason: "get_checkpoint",
42                    error: anyhow!(status),
43                },
44            })?
45            .into_inner();
46
47        let checkpoint =
48            Checkpoint::try_from(response.checkpoint()).map_err(|e| FetchError::Permanent {
49                reason: "proto_conversion",
50                error: e.into(),
51            })?;
52
53        Ok(FetchData::Checkpoint(checkpoint))
54    }
55}
