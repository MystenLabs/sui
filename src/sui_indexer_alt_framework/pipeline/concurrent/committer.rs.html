21/// If the committer needs to retry a commit, it will wait this long initially.
22const INITIAL_RETRY_INTERVAL: Duration = Duration::from_millis(100);
23
24/// If the committer needs to retry a commit, it will wait at most this long between retries.
25const MAX_RETRY_INTERVAL: Duration = Duration::from_secs(1);
26
27/// The committer task is responsible for writing batches of rows to the database. It receives
28/// batches on `rx` and writes them out to the `db` concurrently (`config.write_concurrency`
29/// controls the degree of fan-out).
30///
31/// The writing of each batch will be repeatedly retried on an exponential back-off until it
32/// succeeds. Once the write succeeds, the [WatermarkPart]s for that batch are sent on `tx` to the
33/// watermark task, as long as `skip_watermark` is not true.
34///
35/// This task will shutdown via its `cancel`lation token, or if its receiver or sender channels are
36/// closed.
37pub(super) fn committer<H: Handler + 'static>(
38    config: CommitterConfig,
39    skip_watermark: bool,
40    rx: mpsc::Receiver<BatchedRows<H>>,
41    tx: mpsc::Sender<Vec<WatermarkPart>>,
42    db: H::Store,
43    metrics: Arc<IndexerMetrics>,
44    cancel: CancellationToken,
45) -> JoinHandle<()> {
46    tokio::spawn(async move {
47        info!(pipeline = H::NAME, "Starting committer");
48        let checkpoint_lag_reporter = CheckpointLagMetricReporter::new_for_pipeline::<H>(
49            &metrics.partially_committed_checkpoint_timestamp_lag,
50            &metrics.latest_partially_committed_checkpoint_timestamp_lag_ms,
51            &metrics.latest_partially_committed_checkpoint,
52        );
53
54        match ReceiverStream::new(rx)
55            .try_for_each_spawned(
56                config.write_concurrency,
57                |BatchedRows { values, watermark }| {
58                    let values = Arc::new(values);
59                    let tx = tx.clone();
60                    let db = db.clone();
61                    let metrics = metrics.clone();
62                    let cancel = cancel.clone();
63                    let checkpoint_lag_reporter = checkpoint_lag_reporter.clone();
64
65                    // Repeatedly try to get a connection to the DB and write the batch. Use an
66                    // exponential backoff in case the failure is due to contention over the DB
67                    // connection pool.
68                    let backoff = ExponentialBackoff {
69                        initial_interval: INITIAL_RETRY_INTERVAL,
70                        current_interval: INITIAL_RETRY_INTERVAL,
71                        max_interval: MAX_RETRY_INTERVAL,
72                        max_elapsed_time: None,
73                        ..Default::default()
74                    };
75
76                    let highest_checkpoint = watermark.iter().map(|w| w.checkpoint()).max();
77                    let highest_checkpoint_timestamp =
78                        watermark.iter().map(|w| w.timestamp_ms()).max();
79
80                    use backoff::Error as BE;
81                    let commit = move || {
82                        let values = values.clone();
83                        let db = db.clone();
84                        let metrics = metrics.clone();
85                        let checkpoint_lag_reporter = checkpoint_lag_reporter.clone();
86                        async move {
87                            if values.is_empty() {
88                                return Ok(());
89                            }
90
91                            metrics
92                                .total_committer_batches_attempted
93                                .with_label_values(&[H::NAME])
94                                .inc();
95
96                            let guard = metrics
97                                .committer_commit_latency
98                                .with_label_values(&[H::NAME])
99                                .start_timer();
100
101                            let mut conn = db.connect().await.map_err(|e| {
102                                warn!(
103                                    pipeline = H::NAME,
104                                    "Committed failed to get connection for DB"
105                                );
106
107                                metrics
108                                    .total_committer_batches_failed
109                                    .with_label_values(&[H::NAME])
110                                    .inc();
111
112                                BE::transient(Break::Err(e))
113                            })?;
114
115                            let affected = H::commit(values.as_slice(), &mut conn).await;
116                            let elapsed = guard.stop_and_record();
117
118                            match affected {
119                                Ok(affected) => {
120                                    debug!(
121                                        pipeline = H::NAME,
122                                        elapsed_ms = elapsed * 1000.0,
123                                        affected,
124                                        committed = values.len(),
125                                        "Wrote batch",
126                                    );
127
128                                    checkpoint_lag_reporter.report_lag(
129                                        // unwrap is safe because we would have returned if values is empty.
130                                        highest_checkpoint.unwrap(),
131                                        highest_checkpoint_timestamp.unwrap(),
132                                    );
133
134                                    metrics
135                                        .total_committer_batches_succeeded
136                                        .with_label_values(&[H::NAME])
137                                        .inc();
138
139                                    metrics
140                                        .total_committer_rows_committed
141                                        .with_label_values(&[H::NAME])
142                                        .inc_by(values.len() as u64);
143
144                                    metrics
145                                        .total_committer_rows_affected
146                                        .with_label_values(&[H::NAME])
147                                        .inc_by(affected as u64);
148
149                                    metrics
150                                        .committer_tx_rows
151                                        .with_label_values(&[H::NAME])
152                                        .observe(affected as f64);
153
154                                    Ok(())
155                                }
156
157                                Err(e) => {
158                                    warn!(
159                                        pipeline = H::NAME,
160                                        elapsed_ms = elapsed * 1000.0,
161                                        committed = values.len(),
162                                        "Error writing batch: {e}",
163                                    );
164
165                                    metrics
166                                        .total_committer_batches_failed
167                                        .with_label_values(&[H::NAME])
168                                        .inc();
169
170                                    Err(BE::transient(Break::Err(e)))
171                                }
172                            }
173                        }
174                    };
175
176                    async move {
177                        tokio::select! {
178                            _ = cancel.cancelled() => {
179                                return Err(Break::Cancel);
180                            }
181
182                            // Double check that the commit actually went through, (this backoff should
183                            // not produce any permanent errors, but if it does, we need to shutdown
184                            // the pipeline).
185                            commit = backoff::future::retry(backoff, commit) => {
186                                let () = commit?;
187                            }
188                        };
189
190                        if !skip_watermark && tx.send(watermark).await.is_err() {
191                            info!(pipeline = H::NAME, "Watermark closed channel");
192                            return Err(Break::Cancel);
193                        }
194
195                        Ok(())
196                    }
197                },
198            )
199            .await
200        {
201            Ok(()) => {
202                info!(pipeline = H::NAME, "Batches done, stopping committer");
203            }
204
205            Err(Break::Cancel) => {
206                info!(pipeline = H::NAME, "Shutdown received, stopping committer");
207            }
208
209            Err(Break::Err(e)) => {
210                error!(pipeline = H::NAME, "Error from committer: {e}");
211                cancel.cancel();
212            }
213        }
214    })
215}

270    #[async_trait]
271    impl Handler for DataPipeline {
272        type Store = MockStore;
273
274        async fn commit<'a>(
275            values: &[StoredData],
276            conn: &mut MockConnection<'a>,
277        ) -> anyhow::Result<usize> {
278            for value in values {
279                // If there's a delay, sleep for that duration
280                if value.commit_delay_ms > 0 {
281                    tokio::time::sleep(Duration::from_millis(value.commit_delay_ms)).await;
282                }
283
284                // If there are remaining failures, fail the commit and decrement the counter
285                {
286                    let remaining = value
287                        .commit_failure_remaining
288                        .fetch_sub(1, Ordering::Relaxed);
289                    ensure!(
290                        remaining == 0,
291                        "Commit failed, remaining failures: {}",
292                        remaining - 1
293                    );
294                }
295
296                conn.0
297                    .commit_data(
298                        DataPipeline::NAME,
299                        value.cp_sequence_number,
300                        value.tx_sequence_numbers.clone(),
301                    )
302                    .await?;
303            }
304            Ok(values.len())
305        }
306    }

353    async fn test_concurrent_batch_processing() {
354        let mut setup = setup_test(MockStore::default(), false).await;
355
356        // Send batches
357        let batch1 = BatchedRows {
358            values: vec![
359                StoredData {
360                    cp_sequence_number: 1,
361                    tx_sequence_numbers: vec![1, 2, 3],
362                    ..Default::default()
363                },
364                StoredData {
365                    cp_sequence_number: 2,
366                    tx_sequence_numbers: vec![4, 5, 6],
367                    ..Default::default()
368                },
369            ],
370            watermark: vec![
371                WatermarkPart {
372                    watermark: CommitterWatermark {
373                        epoch_hi_inclusive: 0,
374                        checkpoint_hi_inclusive: 1,
375                        tx_hi: 3,
376                        timestamp_ms_hi_inclusive: 1000,
377                    },
378                    batch_rows: 1,
379                    total_rows: 1, // Total rows from checkpoint 1
380                },
381                WatermarkPart {
382                    watermark: CommitterWatermark {
383                        epoch_hi_inclusive: 0,
384                        checkpoint_hi_inclusive: 2,
385                        tx_hi: 6,
386                        timestamp_ms_hi_inclusive: 2000,
387                    },
388                    batch_rows: 1,
389                    total_rows: 1, // Total rows from checkpoint 2
390                },
391            ],
392        };
393
394        let batch2 = BatchedRows {
395            values: vec![StoredData {
396                cp_sequence_number: 3,
397                tx_sequence_numbers: vec![7, 8, 9],
398                ..Default::default()
399            }],
400            watermark: vec![WatermarkPart {
401                watermark: CommitterWatermark {
402                    epoch_hi_inclusive: 0,
403                    checkpoint_hi_inclusive: 3,
404                    tx_hi: 9,
405                    timestamp_ms_hi_inclusive: 3000,
406                },
407                batch_rows: 1,
408                total_rows: 1, // Total rows from checkpoint 3
409            }],
410        };
411
412        setup.batch_tx.send(batch1).await.unwrap();
413        setup.batch_tx.send(batch2).await.unwrap();
414
415        // Verify watermarks. Blocking until committer has processed.
416        let watermark1 = setup.watermark_rx.recv().await.unwrap();
417        let watermark2 = setup.watermark_rx.recv().await.unwrap();
418        assert_eq!(watermark1.len(), 2);
419        assert_eq!(watermark2.len(), 1);
420
421        // Verify data was committed
422        {
423            let data = setup.store.data.get(DataPipeline::NAME).unwrap();
424            assert_eq!(data.len(), 3);
425            assert_eq!(data.get(&1).unwrap().value(), &vec![1, 2, 3]);
426            assert_eq!(data.get(&2).unwrap().value(), &vec![4, 5, 6]);
427            assert_eq!(data.get(&3).unwrap().value(), &vec![7, 8, 9]);
428        }
429
430        // Clean up
431        drop(setup.batch_tx);
432        let _ = setup.committer_handle.await;
433    }

437        let mut setup = setup_test(MockStore::default(), false).await;
438
439        // Create a batch with a single item that will fail once before succeeding
440        let batch = BatchedRows {
441            values: vec![StoredData {
442                cp_sequence_number: 1,
443                tx_sequence_numbers: vec![1, 2, 3],
444                commit_failure_remaining: Arc::new(AtomicUsize::new(1)),
445                commit_delay_ms: 1_000, // Long commit delay for testing state between retry
446            }],
447            watermark: vec![WatermarkPart {
448                watermark: CommitterWatermark {
449                    epoch_hi_inclusive: 0,
450                    checkpoint_hi_inclusive: 1,
451                    tx_hi: 3,
452                    timestamp_ms_hi_inclusive: 1000,
453                },
454                batch_rows: 1,
455                total_rows: 1,
456            }],
457        };
458
459        // Send the batch
460        setup.batch_tx.send(batch).await.unwrap();
461
462        // Wait for the first attempt to fail and before the retry succeeds
