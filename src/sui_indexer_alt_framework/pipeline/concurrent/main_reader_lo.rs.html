4use std::sync::Arc;
5use std::sync::atomic::AtomicU64;
6use std::sync::atomic::Ordering;
7use std::time::Duration;
8
9use sui_futures::service::Service;
10use tokio::sync::SetOnce;
11use tokio::time::MissedTickBehavior;
12use tokio::time::interval;
13use tracing::info;
14use tracing::warn;
15
16use crate::pipeline::concurrent::Handler;
17use crate::store::Connection;
18use crate::store::Store;

20/// Starts a task for a tasked pipeline to track the main reader lo. The existence of
21/// `reader_interval` indicates whether the indexer was tasked, necessitating this task, or not.
22pub(super) fn track_main_reader_lo<H: Handler + 'static>(
23    reader_lo: Arc<SetOnce<AtomicU64>>,
24    reader_interval: Option<Duration>,
25    store: H::Store,
26) -> Service {
27    Service::new().spawn_aborting(async move {
28        let Some(reader_interval) = reader_interval else {
29            info!(
30                pipeline = H::NAME,
31                "Not a tasked indexer, skipping main reader lo task"
32            );
33            reader_lo.set(AtomicU64::new(0)).ok();
34            return Ok(());
35        };
36
37        // If we miss ticks, skip them to ensure we have the latest watermark.
38        let mut reader_interval = interval(reader_interval);
39        reader_interval.set_missed_tick_behavior(MissedTickBehavior::Skip);
40
41        loop {
42            reader_interval.tick().await;
43
44            let mut conn = match store.connect().await {
45                Ok(conn) => conn,
46                Err(e) => {
47                    warn!(pipeline = H::NAME, "Failed to connect to store: {e}");
48                    continue;
49                }
50            };
51
52            let watermark = match conn.reader_watermark(H::NAME).await {
53                // If the reader watermark is not present (either because the watermark entry does
54                // not exist, or the reaer watermark is not set), we assume that pruning is not
55                // enabled and checkpoints >= 0 are valid.
56                Ok(watermark) => watermark.map_or(0, |wm| wm.reader_lo),
57                Err(e) => {
58                    warn!(pipeline = H::NAME, "Failed to get reader watermark: {e}");
59                    continue;
60                }
61            };
62
63            if let Some(lo) = reader_lo.get() {
64                lo.store(watermark, Ordering::Relaxed);
65            } else {
66                reader_lo.set(AtomicU64::new(watermark)).ok();
67            }
68        }
69    })
70}
