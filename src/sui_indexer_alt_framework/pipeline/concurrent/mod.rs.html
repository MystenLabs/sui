38/// Status returned by `Handler::batch` to indicate whether the batch is ready to be committed.
39#[derive(Debug, Clone, Copy, PartialEq, Eq)]
40pub enum BatchStatus {
41    /// The batch can accept more values.
42    Pending,
43    /// The batch is full and should be committed.
44    Ready,
45}
46
47/// Handlers implement the logic for a given indexing pipeline: How to process checkpoint data (by
48/// implementing [Processor]) into rows for their table, and how to write those rows to the database.
49///
50/// The handler is also responsible for tuning the various parameters of the pipeline (provided as
51/// associated values). Reasonable defaults have been chosen to balance concurrency with memory
52/// usage, but each handle may choose to override these defaults, e.g.
53///
54/// - Handlers that produce many small rows may wish to increase their batch/chunk/max-pending
55///   sizes).
56/// - Handlers that do more work during processing may wish to increase their fanout so more of it
57///   can be done concurrently, to preserve throughput.
58///
59/// Concurrent handlers can only be used in concurrent pipelines, where checkpoint data is
60/// processed and committed out-of-order and a watermark table is kept up-to-date with the latest
61/// checkpoint below which all data has been committed.
62///
63/// Back-pressure is handled through the `MAX_PENDING_SIZE` constant -- if more than this many rows
64/// build up, the collector will stop accepting new checkpoints, which will eventually propagate
65/// back to the ingestion service.
66#[async_trait]
67pub trait Handler: Processor {
68    type Store: Store;
69    type Batch: Default + Send + Sync + 'static;
70
71    /// If at least this many rows are pending, the committer will commit them eagerly.
72    const MIN_EAGER_ROWS: usize = 50;
73
74    /// If there are more than this many rows pending, the committer applies backpressure.
75    const MAX_PENDING_ROWS: usize = 5000;
76
77    /// The maximum number of watermarks that can show up in a single batch.
78    /// This limit exists to deal with pipelines that produce no data for a majority of
79    /// checkpoints -- the size of these pipeline's batches will be dominated by watermark updates.
80    const MAX_WATERMARK_UPDATES: usize = 10_000;
81
82    /// Add values from the iterator to the batch. The implementation may take all, some, or none
83    /// of the values from the iterator by calling `.next()`.
84    ///
85    /// Returns `BatchStatus::Ready` if the batch is full and should be committed,
86    /// or `BatchStatus::Pending` if the batch can accept more values.
87    ///
88    /// Note: The handler can signal batch readiness via `BatchStatus::Ready`, but the framework
89    /// may also decide to commit a batch based on the trait parameters above.
90    fn batch(
91        &self,
92        batch: &mut Self::Batch,
93        values: &mut std::vec::IntoIter<Self::Value>,
94    ) -> BatchStatus;
95
96    /// Commit the batch to the database, returning the number of rows affected.
97    async fn commit<'a>(
98        &self,
99        batch: &Self::Batch,
100        conn: &mut <Self::Store as Store>::Connection<'a>,
101    ) -> anyhow::Result<usize>;
102
103    /// Clean up data between checkpoints `_from` and `_to_exclusive` (exclusive) in the database, returning
104    /// the number of rows affected. This function is optional, and defaults to not pruning at all.
105    async fn prune<'a>(
106        &self,
107        _from: u64,
108        _to_exclusive: u64,
109        _conn: &mut <Self::Store as Store>::Connection<'a>,
110    ) -> anyhow::Result<usize> {
111        Ok(0)
112    }
113}

115/// Configuration for a concurrent pipeline
116#[derive(Serialize, Deserialize, Debug, Clone, Default)]
117pub struct ConcurrentConfig {
118    /// Configuration for the writer, that makes forward progress.
119    pub committer: CommitterConfig,
120
121    /// Configuration for the pruner, that deletes old data.
122    pub pruner: Option<PrunerConfig>,
123
124    /// Override for `Processor::FANOUT` (processor concurrency).
125    pub fanout: Option<usize>,
126
127    /// Override for `Handler::MIN_EAGER_ROWS` (eager batch threshold).
128    pub min_eager_rows: Option<usize>,
129
130    /// Override for `Handler::MAX_PENDING_ROWS` (backpressure threshold).
131    pub max_pending_rows: Option<usize>,
132
133    /// Override for `Handler::MAX_WATERMARK_UPDATES` (watermarks per batch cap).
134    pub max_watermark_updates: Option<usize>,
135}

137#[derive(Serialize, Deserialize, Debug, Clone)]
138pub struct PrunerConfig {
139    /// How often the pruner should check whether there is any data to prune, in milliseconds.
140    pub interval_ms: u64,
141
142    /// How long to wait after the reader low watermark was set, until it is safe to prune up until
143    /// this new watermark, in milliseconds.
144    pub delay_ms: u64,
145
146    /// How much data to keep, this is measured in checkpoints.
147    pub retention: u64,
148
149    /// The maximum range to try and prune in one request, measured in checkpoints.
150    pub max_chunk_size: u64,
151
152    /// The max number of tasks to run in parallel for pruning.
153    pub prune_concurrency: u64,
154}

156/// Values ready to be written to the database. This is an internal type used to communicate
157/// between the collector and the committer parts of the pipeline.
158///
159/// Values inside each batch may or may not be from the same checkpoint. Values in the same
160/// checkpoint can also be split across multiple batches.
161struct BatchedRows<H: Handler> {
162    /// The batch to write
163    batch: H::Batch,
164    /// Number of rows in the batch
165    batch_len: usize,
166    /// Proportions of all the watermarks that are represented in this chunk
167    watermark: Vec<WatermarkPart>,
168}
169
170impl<H, V> BatchedRows<H>
171where
172    H: Handler<Batch = Vec<V>, Value = V>,
173{
174    #[cfg(test)]
175    pub fn from_vec(batch: Vec<V>, watermark: Vec<WatermarkPart>) -> Self {
176        let batch_len = batch.len();
177        Self {
178            batch,
179            batch_len,
180            watermark,
181        }
182    }
183}

207/// Start a new concurrent (out-of-order) indexing pipeline served by the handler, `H`. Starting
208/// strictly after the `watermark` (or from the beginning if no watermark was provided).
209///
210/// Each pipeline consists of a processor task which takes checkpoint data and breaks it down into
211/// rows, ready for insertion, a collector which batches those rows into an appropriate size for
212/// the database, a committer which writes the rows out concurrently, and a watermark task to
213/// update the high watermark.
214///
215/// Committing is performed out-of-order: the pipeline may write out checkpoints out-of-order,
216/// either because it received the checkpoints out-of-order or because of variance in processing
217/// time.
218///
219/// The pipeline also maintains a row in the `watermarks` table for the pipeline which tracks the
220/// watermark below which all data has been committed (modulo pruning).
221///
222/// Checkpoint data is fed into the pipeline through the `checkpoint_rx` channel, and internal
223/// channels are created to communicate between its various components. The pipeline will shutdown
224/// if any of its input or output channels close, any of its independent tasks fail, or if it is
225/// signalled to shutdown through the returned service handle.
226pub(crate) fn pipeline<H: Handler + Send + Sync + 'static>(
227    handler: H,
228    next_checkpoint: u64,
229    config: ConcurrentConfig,
230    store: H::Store,
231    task: Option<Task>,
232    checkpoint_rx: mpsc::Receiver<Arc<Checkpoint>>,
233    metrics: Arc<IndexerMetrics>,
234) -> Service {
235    info!(
236        pipeline = H::NAME,
237        "Starting pipeline with config: {config:#?}",
238    );
239
240    let ConcurrentConfig {
241        committer: committer_config,
242        pruner: pruner_config,
243        fanout,
244        min_eager_rows,
245        max_pending_rows,
246        max_watermark_updates,
247    } = config;
248
249    let fanout = fanout.unwrap_or(H::FANOUT);
250    let min_eager_rows = min_eager_rows.unwrap_or(H::MIN_EAGER_ROWS);
251    let max_pending_rows = max_pending_rows.unwrap_or(H::MAX_PENDING_ROWS);
252    let max_watermark_updates = max_watermark_updates.unwrap_or(H::MAX_WATERMARK_UPDATES);
253
254    let (processor_tx, collector_rx) = mpsc::channel(fanout + PIPELINE_BUFFER);
255
256    //docs::#buff (see docs/content/guides/developer/advanced/custom-indexer.mdx)
257    let (collector_tx, committer_rx) =
258        mpsc::channel(committer_config.write_concurrency + PIPELINE_BUFFER);
259    //docs::/#buff
260    let (committer_tx, watermark_rx) =
261        mpsc::channel(committer_config.write_concurrency + PIPELINE_BUFFER);
262    let main_reader_lo = Arc::new(SetOnce::new());
263
264    let handler = Arc::new(handler);
265
266    let s_processor = processor(
267        handler.clone(),
268        checkpoint_rx,
269        processor_tx,
270        metrics.clone(),
271        fanout,
272    );
273
274    let s_collector = collector::<H>(
275        handler.clone(),
276        committer_config.clone(),
277        collector_rx,
278        collector_tx,
279        main_reader_lo.clone(),
280        metrics.clone(),
281        min_eager_rows,
282        max_pending_rows,
283        max_watermark_updates,
284    );
285
286    let s_committer = committer::<H>(
287        handler.clone(),
288        committer_config.clone(),
289        committer_rx,
290        committer_tx,
291        store.clone(),
292        metrics.clone(),
293    );
294
295    let s_commit_watermark = commit_watermark::<H>(
296        next_checkpoint,
297        committer_config,
298        watermark_rx,
299        store.clone(),
300        task.as_ref().map(|t| t.task.clone()),
301        metrics.clone(),
302    );
303
304    let s_track_reader_lo = track_main_reader_lo::<H>(
305        main_reader_lo.clone(),
306        task.as_ref().map(|t| t.reader_interval),
307        store.clone(),
308    );
309
310    let s_reader_watermark =
311        reader_watermark::<H>(pruner_config.clone(), store.clone(), metrics.clone());
312
313    let s_pruner = pruner(handler, pruner_config, store, metrics);
314
315    s_processor
316        .merge(s_collector)
317        .merge(s_committer)
318        .merge(s_commit_watermark)
319        .attach(s_track_reader_lo)
320        .attach(s_reader_watermark)
321        .attach(s_pruner)
322}

401            // Group values by checkpoint
402            let mut grouped: std::collections::HashMap<u64, Vec<u64>> =

405                grouped
406                    .entry(value.checkpoint)

420        ) -> anyhow::Result<usize> {
421            conn.0.prune_data(DataPipeline::NAME, from, to_exclusive)

669        //

679

681            committer: CommitterConfig {
682                collect_interval_ms: 5_000, // Long interval to prevent timer-driven collection
683                write_concurrency: 1,
684                ..Default::default()
685            },
686            ..Default::default()
687        };
688        let store = MockStore::default();
689        let setup = TestSetup::new(config, store, 0).await;
690
691        // Wait for initial setup
