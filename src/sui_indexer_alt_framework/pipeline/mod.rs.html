18/// Extra buffer added to channels between tasks in a pipeline. There does not need to be a huge
19/// capacity here because tasks already buffer rows to insert internally.
20const PIPELINE_BUFFER: usize = 5;
21
22/// Issue a warning every time the number of pending watermarks exceeds this number. This can
23/// happen if the pipeline was started with its initial checkpoint overridden to be strictly
24/// greater than its current watermark -- in that case, the pipeline will never be able to update
25/// its watermarks.
26const WARN_PENDING_WATERMARKS: usize = 10000;
27
28#[derive(Serialize, Deserialize, Debug, Clone)]
29pub struct CommitterConfig {
30    /// Number of concurrent writers per pipeline.
31    pub write_concurrency: usize,
32
33    /// The collector will check for pending data at least this often, in milliseconds.
34    pub collect_interval_ms: u64,
35
36    /// Watermark task will check for pending watermarks this often, in milliseconds.
37    pub watermark_interval_ms: u64,
38
39    /// Maximum random jitter to add to the watermark interval, in milliseconds.
40    pub watermark_interval_jitter_ms: u64,
41}
42
43/// Processed values associated with a single checkpoint. This is an internal type used to
44/// communicate between the processor and the collector parts of the pipeline.
45struct IndexedCheckpoint<P: Processor> {
46    /// Values to be inserted into the database from this checkpoint
47    values: Vec<P::Value>,
48    /// The watermark associated with this checkpoint
49    watermark: CommitterWatermark,
50}
51
52/// A representation of the proportion of a watermark.
53#[derive(Debug, Clone)]
54struct WatermarkPart {
55    /// The watermark itself
56    watermark: CommitterWatermark,
57    /// The number of rows from this watermark that are in this part
58    batch_rows: usize,
59    /// The total number of rows from this watermark
60    total_rows: usize,
61}
62
63impl CommitterConfig {
64    pub fn collect_interval(&self) -> Duration {
65        Duration::from_millis(self.collect_interval_ms)
66    }
67
68    pub fn watermark_interval(&self) -> Duration {
69        Duration::from_millis(self.watermark_interval_ms)
70    }
71
72    /// Returns the next watermark update instant with a random jitter added. The jitter is a
73    /// random value between 0 and `watermark_interval_jitter_ms`.
74    pub fn watermark_interval_with_jitter(&self) -> tokio::time::Instant {
75        let jitter = if self.watermark_interval_jitter_ms == 0 {
76            0
77        } else {
78            rand::thread_rng().gen_range(0..=self.watermark_interval_jitter_ms)
79        };
80        tokio::time::Instant::now() + Duration::from_millis(self.watermark_interval_ms + jitter)
81    }
82}
83
84impl<P: Processor> IndexedCheckpoint<P> {
85    fn new(
86        epoch: u64,
87        cp_sequence_number: u64,
88        tx_hi: u64,
89        timestamp_ms: u64,
90        values: Vec<P::Value>,
91    ) -> Self {
92        Self {
93            watermark: CommitterWatermark {
94                epoch_hi_inclusive: epoch,
95                checkpoint_hi_inclusive: cp_sequence_number,
96                tx_hi,
97                timestamp_ms_hi_inclusive: timestamp_ms,
98            },
99            values,
100        }
101    }
102
103    /// Number of rows from this checkpoint
104    fn len(&self) -> usize {
105        self.values.len()
106    }
107
108    /// The checkpoint sequence number that this data is from
109    fn checkpoint(&self) -> u64 {
110        self.watermark.checkpoint_hi_inclusive
111    }
112}
113
114impl WatermarkPart {
115    fn checkpoint(&self) -> u64 {
116        self.watermark.checkpoint_hi_inclusive
117    }
118
119    fn timestamp_ms(&self) -> u64 {
120        self.watermark.timestamp_ms_hi_inclusive
121    }
122
123    /// Check if all the rows from this watermark are represented in this part.
124    fn is_complete(&self) -> bool {
125        self.batch_rows == self.total_rows
126    }
127
128    /// Add the rows from `other` to this part.
129    fn add(&mut self, other: WatermarkPart) {
130        debug_assert_eq!(self.checkpoint(), other.checkpoint());
131        self.batch_rows += other.batch_rows;
132    }
133
134    /// Record that `rows` have been taken from this part.
135    fn take(&mut self, rows: usize) -> WatermarkPart {
136        debug_assert!(
137            self.batch_rows >= rows,
138            "Can't take more rows than are available"
139        );
140
141        self.batch_rows -= rows;
142        WatermarkPart {
143            watermark: self.watermark,
144            batch_rows: rows,
145            total_rows: self.total_rows,
146        }
147    }
148}
149
150impl Default for CommitterConfig {
151    fn default() -> Self {
152        Self {
153            write_concurrency: 5,
154            collect_interval_ms: 500,
155            watermark_interval_ms: 500,
156            watermark_interval_jitter_ms: 0,
157        }
158    }
159}

180    #[test]
181    fn test_watermark_part_getters() {
182        let watermark = CommitterWatermark {
183            epoch_hi_inclusive: 1,
184            checkpoint_hi_inclusive: 100,
185            tx_hi: 1000,
186            timestamp_ms_hi_inclusive: 1234567890,
187        };
188
189        let part = WatermarkPart {
190            watermark,
191            batch_rows: 50,
192            total_rows: 200,

196        assert_eq!(part.timestamp_ms(), 1234567890);

221    #[test]
222    fn test_watermark_part_becomes_complete_after_adding_new_batch() {
223        let mut part = WatermarkPart {
224            watermark: CommitterWatermark::default(),
225            batch_rows: 199,
226            total_rows: 200,
227        };
228
229        // Add a batch that makes it complete
230        part.add(WatermarkPart {
231            watermark: CommitterWatermark::default(),
232            batch_rows: 1,
233            total_rows: 200,
234        });
235
236        assert!(part.is_complete());
237        assert_eq!(part.batch_rows, 200);
238    }

240    #[test]
241    fn test_watermark_part_becomes_incomplete_after_taking_away_batch() {
242        let mut part = WatermarkPart {
243            watermark: CommitterWatermark::default(),
244            batch_rows: 200,
245            total_rows: 200,
246        };
247        assert!(part.is_complete(), "Initial part should be complete");
248
249        // Take away a portion of the batch
250        let extracted_part = part.take(10);
251
252        // Verify state of extracted part
253        assert!(!extracted_part.is_complete());
254        assert_eq!(extracted_part.batch_rows, 10);
255        assert_eq!(extracted_part.total_rows, 200);
256    }
