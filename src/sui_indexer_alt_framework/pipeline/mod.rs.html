16/// Extra buffer added to channels between tasks in a pipeline. There does not need to be a huge
17/// capacity here because tasks already buffer rows to insert internally.
18const PIPELINE_BUFFER: usize = 5;
19
20/// Issue a warning every time the number of pending watermarks exceeds this number. This can
21/// happen if the pipeline was started with its initial checkpoint overridden to be strictly
22/// greater than its current watermark -- in that case, the pipeline will never be able to update
23/// its watermarks.
24///
25/// This may be a legitimate thing to do when backfilling a table, but in that case
26/// `--skip-watermarks` should be used.
27const WARN_PENDING_WATERMARKS: usize = 10000;
28
29#[derive(Serialize, Deserialize, Debug, Clone)]
30pub struct CommitterConfig {
31    /// Number of concurrent writers per pipeline.
32    pub write_concurrency: usize,
33
34    /// The collector will check for pending data at least this often, in milliseconds.
35    pub collect_interval_ms: u64,
36
37    /// Watermark task will check for pending watermarks this often, in milliseconds.
38    pub watermark_interval_ms: u64,
39}
40
41/// Processed values associated with a single checkpoint. This is an internal type used to
42/// communicate between the processor and the collector parts of the pipeline.
43struct IndexedCheckpoint<P: Processor> {
44    /// Values to be inserted into the database from this checkpoint
45    values: Vec<P::Value>,
46    /// The watermark associated with this checkpoint
47    watermark: CommitterWatermark,
48}
49
50/// A representation of the proportion of a watermark.
51#[derive(Debug, Clone)]
52struct WatermarkPart {
53    /// The watermark itself
54    watermark: CommitterWatermark,
55    /// The number of rows from this watermark that are in this part
56    batch_rows: usize,
57    /// The total number of rows from this watermark
58    total_rows: usize,
59}
60
61/// Internal type used by workers to propagate errors or shutdown signals up to their
62/// supervisor.
63#[derive(thiserror::Error, Debug)]
64enum Break {
65    #[error("Shutdown received")]
66    Cancel,
67
68    #[error(transparent)]
69    Err(#[from] anyhow::Error),
70}
71
72impl CommitterConfig {
73    pub fn collect_interval(&self) -> Duration {
74        Duration::from_millis(self.collect_interval_ms)
75    }
76
77    pub fn watermark_interval(&self) -> Duration {
78        Duration::from_millis(self.watermark_interval_ms)
79    }
80}
81
82impl<P: Processor> IndexedCheckpoint<P> {
83    fn new(
84        epoch: u64,
85        cp_sequence_number: u64,
86        tx_hi: u64,
87        timestamp_ms: u64,
88        values: Vec<P::Value>,
89    ) -> Self {
90        Self {
91            watermark: CommitterWatermark {
92                epoch_hi_inclusive: epoch,
93                checkpoint_hi_inclusive: cp_sequence_number,
94                tx_hi,
95                timestamp_ms_hi_inclusive: timestamp_ms,
96            },
97            values,
98        }
99    }
100
101    /// Number of rows from this checkpoint
102    fn len(&self) -> usize {
103        self.values.len()
104    }
105
106    /// The checkpoint sequence number that this data is from
107    fn checkpoint(&self) -> u64 {
108        self.watermark.checkpoint_hi_inclusive
109    }
110}
111
112impl WatermarkPart {
113    fn checkpoint(&self) -> u64 {
114        self.watermark.checkpoint_hi_inclusive
115    }
116
117    fn timestamp_ms(&self) -> u64 {
118        self.watermark.timestamp_ms_hi_inclusive
119    }
120
121    /// Check if all the rows from this watermark are represented in this part.
122    fn is_complete(&self) -> bool {
123        self.batch_rows == self.total_rows
124    }
125
126    /// Add the rows from `other` to this part.
127    fn add(&mut self, other: WatermarkPart) {
128        debug_assert_eq!(self.checkpoint(), other.checkpoint());
129        self.batch_rows += other.batch_rows;
130    }
131
132    /// Record that `rows` have been taken from this part.
133    fn take(&mut self, rows: usize) -> WatermarkPart {
134        debug_assert!(
135            self.batch_rows >= rows,
136            "Can't take more rows than are available"
137        );
138
139        self.batch_rows -= rows;
140        WatermarkPart {
141            watermark: self.watermark,
142            batch_rows: rows,
143            total_rows: self.total_rows,
144        }
145    }
146}
147
148impl Default for CommitterConfig {
149    fn default() -> Self {
150        Self {
151            write_concurrency: 5,
152            collect_interval_ms: 500,
153            watermark_interval_ms: 500,
154        }
155    }
156}

180    #[test]
181    fn test_watermark_part_getters() {
182        let watermark = CommitterWatermark {
183            epoch_hi_inclusive: 1,
184            checkpoint_hi_inclusive: 100,
185            tx_hi: 1000,
186            timestamp_ms_hi_inclusive: 1234567890,
187        };
188
189        let part = WatermarkPart {
190            watermark,
191            batch_rows: 50,
192            total_rows: 200,
193        };
194
195        assert_eq!(part.checkpoint(), 100);
196        assert_eq!(part.timestamp_ms(), 1234567890);
197    }

201        let part = WatermarkPart {
202            watermark: CommitterWatermark::default(),
203            batch_rows: 200,
204            total_rows: 200,
205        };

221    #[test]
222    fn test_watermark_part_becomes_complete_after_adding_new_batch() {
223        let mut part = WatermarkPart {
224            watermark: CommitterWatermark::default(),
225            batch_rows: 199,
226            total_rows: 200,
227        };
228
229        // Add a batch that makes it complete
230        part.add(WatermarkPart {
231            watermark: CommitterWatermark::default(),
232            batch_rows: 1,
233            total_rows: 200,
234        });
235
236        assert!(part.is_complete());
237        assert_eq!(part.batch_rows, 200);
238    }
239
240    #[test]
241    fn test_watermark_part_becomes_incomplete_after_taking_away_batch() {
242        let mut part = WatermarkPart {
243            watermark: CommitterWatermark::default(),
244            batch_rows: 200,
245            total_rows: 200,
246        };
247        assert!(part.is_complete(), "Initial part should be complete");
248
249        // Take away a portion of the batch
250        let extracted_part = part.take(10);
251
252        // Verify state of extracted part
253        assert!(!extracted_part.is_complete());
254        assert_eq!(extracted_part.batch_rows, 10);
255        assert_eq!(extracted_part.total_rows, 200);
256    }
