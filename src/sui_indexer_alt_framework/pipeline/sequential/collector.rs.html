23/// A fully-assembled batch handed off from the collector to the committer task.
24///
25/// Batches are produced strictly in checkpoint order by the collector and consumed in the
26/// same order by the committer, so watermarks advance monotonically even with pipelining.
27pub(super) struct BatchedRows<H: Handler> {
28    pub batch: H::Batch,
29    pub watermark: CommitterWatermark,
30    pub batch_rows: usize,
31}
32
33/// Collector task — drains `rx` into a reorder buffer, assembles batches in checkpoint order,
34/// and hands each ready batch off to the committer via `committer_tx`.
35///
36/// Data arrives out of order, grouped by checkpoint, on `rx`. The collector orders them and
37/// waits to dispatch them until either a configurable polling interval has passed (controlled
38/// by `config.collect_interval()`), or `H::BATCH_SIZE` rows have been accumulated and the next
39/// expected checkpoint has arrived.
40///
41/// Backpressure: when `committer_tx` is full (bounded by `pipeline_depth`), the collector
42/// blocks on send and stops reading `rx`, so the processor→collector channel fills and the
43/// adaptive fanout / ingest controllers cut upstream concurrency.
44///
45/// `max_pending_rows` is a soft cap: when exceeded, the inner drain loop yields to the flush
46/// phase so we don't accumulate work faster than we try to flush it. `rx.recv()` itself is
47/// never gated, because the missing predecessor for a stuck `pending` may still be sitting in
48/// `rx`; hard-gating receive would risk deadlock.
49pub(super) fn collector<H: Handler>(
50    handler: Arc<H>,
51    config: SequentialConfig,
52    mut next_checkpoint: u64,
53    mut rx: mpsc::Receiver<IndexedCheckpoint<H>>,
54    metrics: Arc<IndexerMetrics>,
55    min_eager_rows: usize,
56    max_pending_rows: usize,
57    max_batch_checkpoints: usize,
58    committer_tx: mpsc::Sender<BatchedRows<H>>,
59) -> Service {
60    Service::new().spawn_aborting(async move {
61        // The `poll` interval controls the maximum time to wait between commits, regardless of the
62        // amount of data available.
63        let mut poll = interval(config.committer.collect_interval());
64        poll.set_missed_tick_behavior(MissedTickBehavior::Delay);
65
66        // The task keeps track of the highest (inclusive) checkpoint it has added to a batch
67        // through `next_checkpoint`. By extension it also knows the next checkpoint to expect.
68
69        // Data for checkpoints that haven't been written yet.
70        let mut pending: BTreeMap<u64, IndexedCheckpoint<H>> = BTreeMap::new();
71        let mut pending_rows: usize = 0;
72
73        info!(pipeline = H::NAME, "Starting collector");
74
75        loop {
76            // IDLE: block until timer fires or enough data accumulates
77            tokio::select! {
78                biased;
79
80                Some(mut indexed) = rx.recv() => {
81                    // Eagerly drain the channel to avoid scheduler ping-pong. `rx.recv`
82                    // itself is never gated — the reorder buffer must stay unbounded,
83                    // because a missing predecessor may still be sitting in `rx` and
84                    // capping receive could deadlock. The `max_pending_rows` check inside
85                    // the drain loop is only a yield to the flush phase, not a hard-cap on
86                    // the amount of rows buffered.
87                    let mut recv_rows = 0usize;
88                    loop {
89                        recv_rows += indexed.len();
90                        pending_rows += indexed.len();
91                        pending.insert(indexed.checkpoint(), indexed);
92
93                        if pending_rows >= max_pending_rows {
94                            break;
95                        }
96
97                        match rx.try_recv() {
98                            Ok(next) => indexed = next,
99                            Err(_) => break,
100                        }
101                    }
102
103                    // Although there isn't an explicit collector in the sequential pipeline,
104                    // keeping this metric consistent with the concurrent pipeline is useful
105                    // to monitor the backpressure from committer to processor.
106                    metrics
107                        .total_collector_rows_received
108                        .with_label_values(&[H::NAME])
109                        .inc_by(recv_rows as u64);
110
111                    if pending_rows < min_eager_rows {
112                        continue;
113                    }
114                }
115
116                // Timer: always fall through to flush (an empty tick still advances
117                // watermarks for any future-checkpoint progress that may have become
118                // possible).
119                _ = poll.tick() => {}
120            }
121
122            if pending.len() > WARN_PENDING_WATERMARKS {
123                warn!(
124                    pipeline = H::NAME,
125                    pending = pending.len(),
126                    "Pipeline has a large number of pending watermarks",
127                );
128            }
129
130            // FLUSHING: emit batches until no further contiguous commit is possible
131            loop {
132                let guard = metrics
133                    .collector_gather_latency
134                    .with_label_values(&[H::NAME])
135                    .start_timer();
136
137                let mut batch = H::Batch::default();
138                let mut batch_rows = 0;
139                let mut batch_checkpoints = 0;
140                let mut watermark: Option<CommitterWatermark> = None;
141
142                // Push data into the next batch as long as it's from contiguous checkpoints
143                // and we haven't gathered information from too many checkpoints already.
144                //
145                // We don't worry about overall size because the handler may have optimized
146                // writes by combining rows, but we will limit the number of checkpoints we try
147                // and batch together as a way to impose some limit on the size of the batch
148                // (and therefore the length of the write transaction).
149                // docs::#batch  (see docs/content/guides/developer/advanced/custom-indexer.mdx)
150                while batch_checkpoints < max_batch_checkpoints {
151                    let Some(entry) = pending.first_entry() else {
152                        break;
153                    };
154
155                    match next_checkpoint.cmp(entry.key()) {
156                        // Next pending checkpoint is from the future.
157                        Ordering::Less => break,
158
159                        // This is the next checkpoint -- include it.
160                        Ordering::Equal => {
161                            let indexed = entry.remove();
162                            batch_rows += indexed.len();
163                            batch_checkpoints += 1;
164                            handler.batch(&mut batch, indexed.values.into_iter());
165                            watermark = Some(indexed.watermark);
166                            next_checkpoint += 1;
167                        }
168
169                        // Next pending checkpoint is in the past, ignore it to avoid double
170                        // writes.
171                        Ordering::Greater => {
172                            metrics
173                                .total_watermarks_out_of_order
174                                .with_label_values(&[H::NAME])
175                                .inc();
176
177                            let indexed = entry.remove();
178                            pending_rows -= indexed.len();
179                        }
180                    }
181                }
182                // docs::/#batch
183
184                let elapsed = guard.stop_and_record();
185                debug!(
186                    pipeline = H::NAME,
187                    elapsed_ms = elapsed * 1000.0,
188                    rows = batch_rows,
189                    pending = pending_rows,
190                    "Gathered batch",
191                );
192
193                // Nothing assembled this iteration — either pending is empty or the next
194                // expected checkpoint hasn't arrived. Stop flushing and return to idle.
195                if batch_checkpoints == 0 {
196                    assert_eq!(batch_rows, 0);
197                    break;
198                }
199
200                let Some(watermark) = watermark else {
201                    break;
202                };
203
204                metrics
205                    .collector_batch_size
206                    .with_label_values(&[H::NAME])
207                    .observe(batch_rows as f64);
208
209                metrics
210                    .watermark_epoch
211                    .with_label_values(&[H::NAME])
212                    .set(watermark.epoch_hi_inclusive as i64);
213
214                metrics
215                    .watermark_checkpoint
216                    .with_label_values(&[H::NAME])
217                    .set(watermark.checkpoint_hi_inclusive as i64);
218
219                metrics
220                    .watermark_transaction
221                    .with_label_values(&[H::NAME])
222                    .set(watermark.tx_hi as i64);
223
224                metrics
225                    .watermark_timestamp_ms
226                    .with_label_values(&[H::NAME])
227                    .set(watermark.timestamp_ms_hi_inclusive as i64);
228
229                // Hand the assembled batch off to the committer. When `committer_tx` is full,
230                // this blocks — natural backpressure: the collector stops flushing and stops
231                // reading `rx`, so the processor→collector channel fills and the adaptive
232                // fanout / ingest controllers cut upstream concurrency.
233                let batched = BatchedRows {
234                    batch,
235                    watermark,
236                    batch_rows,
237                };
238                if committer_tx.send(batched).await.is_err() {
239                    info!(
240                        pipeline = H::NAME,
241                        "Committer task closed; stopping collector"
242                    );
243                    return Ok(());
244                }
245
246                pending_rows -= batch_rows;
247            }
248
249            if rx.is_closed() && rx.is_empty() && !has_ready_checkpoint(next_checkpoint, &pending) {
250                info!(
251                    pipeline = H::NAME,
252                    "Processor closed channel and no more data to commit"
253                );
254                break;
255            }
256        }
257
258        info!(pipeline = H::NAME, "Stopping collector");
259        Ok(())
260    })
261}

263// Whether the first entry in `pending` is ready to be consumed by the collector — either to be
264// included in the next batch (if it matches `next_checkpoint`) or discarded as a stale duplicate
265// (if it predates `next_checkpoint`).
266fn has_ready_checkpoint<T>(next_checkpoint: u64, pending: &BTreeMap<u64, T>) -> bool {
267    pending
268        .first_key_value()
269        .is_some_and(|(&first, _)| first <= next_checkpoint)
270}
