83    /// Fetch a `Node` by its globally unique `ID`. Returns `null` if the node cannot be found (e.g., the underlying data was pruned or never existed).
84    async fn node(&self, ctx: &Context<'_>, id: Id) -> Option<Result<Node, RpcError>> {
85        async {
86            let scope = self.scope(ctx)?;
87            Ok(match id {
88                Id::Address(a) => Some(Node::Address(Box::new(Address::with_address(scope, a)))),
89
90                Id::Checkpoint(s) => Checkpoint::with_sequence_number(scope, Some(s))
91                    .map(Box::new)
92                    .map(Node::Checkpoint),
93
94                Id::DynamicFieldByAddress(a) => {
95                    let object = Object::with_address(scope, a);
96                    DynamicField::from_object(&object, ctx)
97                        .await?
98                        .map(Box::new)
99                        .map(Node::DynamicField)
100                }
101
102                Id::DynamicFieldByRef(a, v, d) => {
103                    let object = Object::with_ref(&scope, a, v, d);
104                    DynamicField::from_object(&object, ctx)
105                        .await?
106                        .map(Box::new)
107                        .map(Node::DynamicField)
108                }
109
110                Id::Epoch(e) => Some(Node::Epoch(Box::new(Epoch::with_id(scope, e)))),
111
112                Id::MoveObjectByAddress(a) => {
113                    let object = Object::with_address(scope, a);
114                    MoveObject::from_object(&object, ctx)
115                        .await?
116                        .map(Box::new)
117                        .map(Node::MoveObject)
118                }
119
120                Id::MoveObjectByRef(a, v, d) => {
121                    let object = Object::with_ref(&scope, a, v, d);
122                    MoveObject::from_object(&object, ctx)
123                        .await?
124                        .map(Box::new)
125                        .map(Node::MoveObject)
126                }
127
128                Id::MovePackage(a) => Some(Node::MovePackage(Box::new(MovePackage::with_address(
129                    scope, a,
130                )))),
131
132                Id::ObjectByAddress(a) => {
133                    Some(Node::Object(Box::new(Object::with_address(scope, a))))
134                }
135
136                Id::ObjectByRef(a, v, d) => {
137                    Some(Node::Object(Box::new(Object::with_ref(&scope, a, v, d))))
138                }
139
140                Id::Transaction(d) => Some(Node::Transaction(Box::new(Transaction::with_digest(
141                    scope, d,
142                )))),
143            })
144        }
145        .await
146        .transpose()
147    }

149    /// Look-up an account by its SuiAddress.
150    ///
151    /// If `rootVersion` is specified, nested dynamic field accesses will be fetched at or before this version. This can be used to fetch a child or descendant object bounded by its root object's version, when its immediate parent is wrapped, or a value in a dynamic object field. For any wrapped or child (object-owned) object, its root object can be defined recursively as:
152    ///
153    /// - The root object of the object it is wrapped in, if it is wrapped.
154    /// - The root object of its owner, if it is owned by another object.
155    /// - The object itself, if it is not object-owned or wrapped.
156    ///
157    /// Specifying a `rootVersion` disables nested queries for paginating owned objects or dynamic fields (these queries are only supported at checkpoint boundaries).
158    ///
159    /// If `atCheckpoint` is specified, the address will be fetched at the latest version as of this checkpoint. This will fail if the provided checkpoint is after the RPC's latest checkpoint.

161    /// If none of the above are specified, the address is fetched at the checkpoint being viewed.
162    ///
163    /// If the address is fetched by name and the name does not resolve to an address (e.g. the name does not exist or has expired), `null` is returned.

169        root_version: Option<UInt53>,
170        at_checkpoint: Option<UInt53>,
171    ) -> Option<Result<Address, RpcError<address::Error>>> {
172        async {
173            Address::by_key(
174                ctx,
175                self.scope(ctx)?,
176                AddressKey {
177                    address,
178                    name,
179                    root_version,
180                    at_checkpoint,
181                },
182            )
183            .await
184        }
185        .await
186        .transpose()
187    }

736    /// Simulate a transaction to preview its effects without executing it on chain.
737    ///
738    /// Accepts a JSON transaction matching the [Sui gRPC API schema](https://docs.sui.io/references/fullnode-protocol#sui-rpc-v2-Transaction).
739    /// The JSON format allows for partial transaction specification where certain fields can be automatically resolved by the server.
740    ///
741    /// Alternatively, for already serialized transactions, you can pass BCS-encoded data:
742    /// `{"bcs": {"value": "<base64>"}}`
743    ///
744    /// Unlike `executeTransaction`, this does not require signatures since the transaction is not committed to the blockchain. This allows for previewing transaction effects, estimating gas costs, and testing transaction logic without spending gas or requiring valid signatures.
745    ///
746    /// - `checksEnabled`: If true, enables transaction validation checks during simulation. Defaults to true.
747    /// - `doGasSelection`: If true, enables automatic gas coin selection and budget estimation. Defaults to false.
748    async fn simulate_transaction(
749        &self,
750        ctx: &Context<'_>,
751        transaction: Json,
752        checks_enabled: Option<bool>,
753        do_gas_selection: Option<bool>,
754    ) -> Result<SimulationResult, RpcError<TransactionInputError>> {
755        let fullnode_client: &FullnodeClient = ctx.data()?;
756
757        // Convert Json to serde_json::Value and parse as proto::Transaction
758        let json_value: serde_json::Value = transaction
759            .try_into()
760            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
761        let proto_tx: proto::Transaction = serde_json::from_value(json_value)
762            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
763
764        // Simulate transaction via gRPC
765        match fullnode_client
766            .simulate_transaction(
767                proto_tx,
768                checks_enabled.unwrap_or(true),
769                do_gas_selection.unwrap_or(false),
770            )
771            .await
772        {
773            Ok(response) => {
774                let scope = self.scope(ctx)?;
775                let tx_data = response
776                    .transaction
777                    .as_ref()
778                    .and_then(|executed_tx| executed_tx.transaction.as_ref())
779                    .and_then(|tx| tx.bcs.as_ref())
780                    .ok_or_else(|| anyhow!("Missing transaction or BCS in simulation response"))?
781                    .deserialize()
782                    .context("Failed to deserialize transaction from response")?;
783
784                SimulationResult::from_simulation_response(scope, response, tx_data).map_err(upcast)
785            }
786            Err(GrpcExecutionError(status))
787                if matches!(status.code(), Code::InvalidArgument | Code::NotFound) =>
788            {
789                Err(bad_user_input(TransactionInputError::InvalidArgument(
790                    status.message().to_string(),
791                )))
792            }
793            Err(other_error) => Err(anyhow!(other_error)
794                .context("Failed to simulate transaction")
795                .into()),
796        }
797    }
