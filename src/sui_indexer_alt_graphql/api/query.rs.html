82    /// Fetch a `Node` by its globally unique `ID`. Returns `null` if the node cannot be found (e.g., the underlying data was pruned or never existed).
83    async fn node(&self, ctx: &Context<'_>, id: Id) -> Option<Result<Node, RpcError>> {
84        async {
85            let scope = self.scope(ctx)?;
86            Ok(match id {
87                Id::Address(a) => Some(Node::Address(Box::new(Address::with_address(scope, a)))),
88
89                Id::Checkpoint(s) => Checkpoint::with_sequence_number(scope, Some(s))
90                    .map(Box::new)
91                    .map(Node::Checkpoint),
92
93                Id::DynamicFieldByAddress(a) => {
94                    let object = Object::with_address(scope, a);
95                    DynamicField::from_object(&object, ctx)
96                        .await?
97                        .map(Box::new)
98                        .map(Node::DynamicField)
99                }
100
101                Id::DynamicFieldByRef(a, v, d) => {
102                    let object = Object::with_ref(&scope, a, v, d);
103                    DynamicField::from_object(&object, ctx)
104                        .await?
105                        .map(Box::new)
106                        .map(Node::DynamicField)
107                }
108
109                Id::Epoch(e) => Some(Node::Epoch(Box::new(Epoch::with_id(scope, e)))),
110
111                Id::MoveObjectByAddress(a) => {
112                    let object = Object::with_address(scope, a);
113                    MoveObject::from_object(&object, ctx)
114                        .await?
115                        .map(Box::new)
116                        .map(Node::MoveObject)
117                }
118
119                Id::MoveObjectByRef(a, v, d) => {
120                    let object = Object::with_ref(&scope, a, v, d);
121                    MoveObject::from_object(&object, ctx)
122                        .await?
123                        .map(Box::new)
124                        .map(Node::MoveObject)
125                }
126
127                Id::MovePackage(a) => Some(Node::MovePackage(Box::new(MovePackage::with_address(
128                    scope, a,
129                )))),
130
131                Id::ObjectByAddress(a) => {
132                    Some(Node::Object(Box::new(Object::with_address(scope, a))))
133                }
134
135                Id::ObjectByRef(a, v, d) => {
136                    Some(Node::Object(Box::new(Object::with_ref(&scope, a, v, d))))
137                }
138
139                Id::Transaction(d) => Some(Node::Transaction(Box::new(Transaction::with_digest(
140                    scope, d,
141                )))),
142            })
143        }
144        .await
145        .transpose()
146    }

148    /// Look-up an account by its SuiAddress.
149    ///
150    /// If `rootVersion` is specified, nested dynamic field accesses will be fetched at or before this version. This can be used to fetch a child or descendant object bounded by its root object's version, when its immediate parent is wrapped, or a value in a dynamic object field. For any wrapped or child (object-owned) object, its root object can be defined recursively as:
151    ///
152    /// - The root object of the object it is wrapped in, if it is wrapped.
153    /// - The root object of its owner, if it is owned by another object.
154    /// - The object itself, if it is not object-owned or wrapped.
155    ///
156    /// Specifying a `rootVersion` disables nested queries for paginating owned objects or dynamic fields (these queries are only supported at checkpoint boundaries).
157    ///
158    /// If `atCheckpoint` is specified, the address will be fetched at the latest version as of this checkpoint. This will fail if the provided checkpoint is after the RPC's latest checkpoint.
159    ///
160    /// If none of the above are specified, the address is fetched at the checkpoint being viewed.
161    ///
162    /// If the address is fetched by name and the name does not resolve to an address (e.g. the name does not exist or has expired), `null` is returned.
163    async fn address(

167        name: Option<Domain>,
168        root_version: Option<UInt53>,
169        at_checkpoint: Option<UInt53>,
170    ) -> Option<Result<Address, RpcError<address::Error>>> {
171        async {
172            Address::by_key(
173                ctx,
174                self.scope(ctx)?,
175                AddressKey {
176                    address,
177                    name,
178                    root_version,
179                    at_checkpoint,
180                },
181            )
182            .await
183        }
184        .await
185        .transpose()
186    }

735    /// Simulate a transaction to preview its effects without executing it on chain.
736    ///
737    /// Accepts a JSON transaction matching the [Sui gRPC API schema](https://docs.sui.io/references/fullnode-protocol#sui-rpc-v2-Transaction).
738    /// The JSON format allows for partial transaction specification where certain fields can be automatically resolved by the server.
739    ///
740    /// Alternatively, for already serialized transactions, you can pass BCS-encoded data:
741    /// `{"bcs": {"value": "<base64>"}}`
742    ///
743    /// Unlike `executeTransaction`, this does not require signatures since the transaction is not committed to the blockchain. This allows for previewing transaction effects, estimating gas costs, and testing transaction logic without spending gas or requiring valid signatures.
744    ///
745    /// - `checksEnabled`: If true, enables transaction validation checks during simulation. Defaults to true.
746    /// - `doGasSelection`: If true, enables automatic gas coin selection and budget estimation. Defaults to false.
747    async fn simulate_transaction(
748        &self,
749        ctx: &Context<'_>,
750        transaction: Json,
751        checks_enabled: Option<bool>,
752        do_gas_selection: Option<bool>,
753    ) -> Result<SimulationResult, RpcError<TransactionInputError>> {
754        let fullnode_client: &FullnodeClient = ctx.data()?;
755
756        // Convert Json to serde_json::Value and parse as proto::Transaction
757        let json_value: serde_json::Value = transaction
758            .try_into()
759            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
760        let proto_tx: proto::Transaction = serde_json::from_value(json_value)
761            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
762
763        // Simulate transaction using proto
764        match fullnode_client
765            .simulate_transaction(
766                proto_tx,
767                checks_enabled.unwrap_or(true),
768                do_gas_selection.unwrap_or(false),
769            )
770            .await
771        {
772            Ok(response) => {
773                let scope = self.scope(ctx)?;
774                let tx_data = response
775                    .transaction
776                    .as_ref()
777                    .and_then(|executed_tx| executed_tx.transaction.as_ref())
778                    .and_then(|tx| tx.bcs.as_ref())
779                    .ok_or_else(|| anyhow!("Missing transaction or BCS in simulation response"))?
780                    .deserialize()
781                    .context("Failed to deserialize transaction from response")?;
782
783                SimulationResult::from_simulation_response(scope, response, tx_data).map_err(upcast)
784            }
785            Err(GrpcExecutionError(status)) => Ok(SimulationResult {
786                effects: None,
787                outputs: None,
788                error: Some(status.to_string()),
789            }),
790            Err(other_error) => Err(anyhow!(other_error)
791                .context("Failed to simulate transaction")
792                .into()),
793        }
794    }
