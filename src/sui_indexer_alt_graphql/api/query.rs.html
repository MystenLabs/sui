91    /// Fetch a `Node` by its globally unique `ID`. Returns `null` if the node cannot be found (e.g., the underlying data was pruned or never existed).
92    async fn node(&self, ctx: &Context<'_>, id: Id) -> Option<Result<Node, RpcError>> {
93        async {
94            let scope = self.scope(ctx)?;
95            Ok(match id {
96                Id::Address(a) => Some(Node::Address(Box::new(Address::with_address(scope, a)))),
97
98                Id::Checkpoint(s) => Checkpoint::with_sequence_number(scope, Some(s))
99                    .map(Box::new)
100                    .map(Node::Checkpoint),
101
102                Id::DynamicFieldByAddress(a) => {
103                    let object = Object::with_address(scope, a);
104                    DynamicField::from_object(&object, ctx)
105                        .await?
106                        .map(Box::new)
107                        .map(Node::DynamicField)
108                }
109
110                Id::DynamicFieldByRef(a, v, d) => {
111                    let object = Object::with_ref(&scope, a, v, d);
112                    DynamicField::from_object(&object, ctx)
113                        .await?
114                        .map(Box::new)
115                        .map(Node::DynamicField)
116                }
117
118                Id::Epoch(e) => Some(Node::Epoch(Box::new(Epoch::with_id(scope, e)))),
119
120                Id::MoveObjectByAddress(a) => {
121                    let object = Object::with_address(scope, a);
122                    MoveObject::from_object(&object, ctx)
123                        .await?
124                        .map(Box::new)
125                        .map(Node::MoveObject)
126                }
127
128                Id::MoveObjectByRef(a, v, d) => {
129                    let object = Object::with_ref(&scope, a, v, d);
130                    MoveObject::from_object(&object, ctx)
131                        .await?
132                        .map(Box::new)
133                        .map(Node::MoveObject)
134                }
135
136                Id::MovePackage(a) => Some(Node::MovePackage(Box::new(MovePackage::with_address(
137                    scope, a,
138                )))),
139
140                Id::ObjectByAddress(a) => {
141                    Some(Node::Object(Box::new(Object::with_address(scope, a))))
142                }
143
144                Id::ObjectByRef(a, v, d) => {
145                    Some(Node::Object(Box::new(Object::with_ref(&scope, a, v, d))))
146                }
147
148                Id::Transaction(d) => Some(Node::Transaction(Box::new(Transaction::with_digest(
149                    scope, d,
150                )))),
151            })
152        }
153        .await
154        .transpose()
155    }

159    /// If `rootVersion` is specified, nested dynamic field accesses will be fetched at or before this version. This can be used to fetch a child or descendant object bounded by its root object's version, when its immediate parent is wrapped, or a value in a dynamic object field. For any wrapped or child (object-owned) object, its root object can be defined recursively as:
160    ///

203    /// Fetch a checkpoint by its sequence number or digest, or the latest checkpoint if neither is provided.
204    ///
205    /// It is an error to specify both `sequenceNumber` and `digest`.
206    ///
207    /// Returns `null` if the checkpoint does not exist in the store, either because it never existed or because it was pruned.
208    async fn checkpoint(
209        &self,
210        ctx: &Context<'_>,
211        sequence_number: Option<UInt53>,
212        digest: Option<Digest>,
213    ) -> Option<Result<Checkpoint, RpcError<checkpoint::Error>>> {
214        async {
215            let scope = self.scope(ctx)?;
216            match (sequence_number, digest) {
217                (Some(_), Some(_)) => Err(bad_user_input(checkpoint::Error::BothBoundsSet)),
218                (None, Some(digest)) => Checkpoint::by_digest(ctx, scope, digest.into())
219                    .await
220                    .map_err(upcast),
221                (sequence_number, None) => Ok(Checkpoint::with_sequence_number(
222                    scope,
223                    sequence_number.map(|s| s.into()),
224                )),
225            }
226        }
227        .await
228        .transpose()
229    }

250            }
251            .await,
252        )
253    }

753    /// Simulate a transaction to preview its effects without executing it on chain.
754    ///
755    /// Accepts a JSON transaction matching the [Sui gRPC API schema](https://docs.sui.io/references/fullnode-protocol#sui-rpc-v2-Transaction).
756    /// The JSON format allows for partial transaction specification where certain fields can be automatically resolved by the server.
757    ///
758    /// Alternatively, for already serialized transactions, you can pass BCS-encoded data:
759    /// `{"bcs": {"value": "<base64>"}}`
760    ///
761    /// Unlike `executeTransaction`, this does not require signatures since the transaction is not committed to the blockchain. This allows for previewing transaction effects, estimating gas costs, and testing transaction logic without spending gas or requiring valid signatures.
762    ///
763    /// - `checksEnabled`: If true, enables transaction validation checks during simulation. Defaults to true.
764    /// - `doGasSelection`: If true, enables automatic gas coin selection and budget estimation. Defaults to false.
765    async fn simulate_transaction(
766        &self,
767        ctx: &Context<'_>,
768        transaction: Json,
769        checks_enabled: Option<bool>,
770        do_gas_selection: Option<bool>,
771    ) -> Result<SimulationResult, RpcError<TransactionInputError>> {
772        let Some(fullnode_client) = ctx.data_opt::<FullnodeClient>() else {
773            return Err(feature_unavailable("simulating transactions"));
774        };
775
776        // Convert Json to serde_json::Value and parse as proto::Transaction
777        let json_value: serde_json::Value = transaction
778            .try_into()
779            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
780        let proto_tx: proto::Transaction = serde_json::from_value(json_value)
781            .map_err(|err| bad_user_input(TransactionInputError::InvalidTransactionJson(err)))?;
782
783        let read_mask = FieldMask::from_paths([
784            "transaction.effects",
785            "transaction.transaction",
786            "transaction.events.bcs",
787            "transaction.balance_changes",
788            "transaction.objects.objects.bcs",
789            "transaction.transaction.bcs",
790            "command_outputs",
791        ]);
792
793        // Simulate transaction via gRPC
794        match fullnode_client
795            .simulate_transaction(
796                proto_tx,
797                checks_enabled.unwrap_or(true),
798                do_gas_selection.unwrap_or(false),
799                read_mask,
800            )
801            .await
802        {
803            Ok(response) => {
804                let scope = self.scope(ctx)?;
805                let tx_data = response
806                    .transaction
807                    .as_ref()
808                    .and_then(|executed_tx| executed_tx.transaction.as_ref())
809                    .and_then(|tx| tx.bcs.as_ref())
810                    .ok_or_else(|| anyhow!("Missing transaction or BCS in simulation response"))?
811                    .deserialize()
812                    .context("Failed to deserialize transaction from response")?;
813
814                SimulationResult::from_simulation_response(scope, response, tx_data).map_err(upcast)
815            }
816            Err(GrpcExecutionError(status))
817                if matches!(status.code(), Code::InvalidArgument | Code::NotFound) =>
818            {
819                Err(bad_user_input(TransactionInputError::InvalidArgument(
820                    status.message().to_string(),
821                )))
822            }
823            Err(other_error) => Err(anyhow!(other_error)
824                .context("Failed to simulate transaction")
825                .into()),
826        }
827    }
