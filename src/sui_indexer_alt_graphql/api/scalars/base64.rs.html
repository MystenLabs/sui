4use std::str::FromStr;
5
6use async_graphql::*;
7use fastcrypto::encoding::Base64 as NativeBase64;
8use fastcrypto::encoding::Encoding as _;
9
10#[derive(Clone, Debug, PartialEq, Eq)]
11pub(crate) struct Base64(pub(crate) Vec<u8>);
12
13/// String containing Base64-encoded binary data.
14#[Scalar]
15impl ScalarType for Base64 {
16    fn parse(value: Value) -> InputValueResult<Self> {
17        let Value::String(s) = value else {
18            return Err(InputValueError::expected_type(value));
19        };
20
21        Base64::from_str(&s)
22    }
23
24    fn to_value(&self) -> Value {
25        Value::String(NativeBase64::encode(self.0.clone()))
26    }
27}
28
29impl FromStr for Base64 {
30    type Err = InputValueError<Self>;
31
32    fn from_str(s: &str) -> Result<Self, Self::Err> {
33        Ok(Base64(
34            NativeBase64::decode(s).map_err(|_| InputValueError::custom("Invalid Base64"))?,
35        ))
36    }
37}
38
39impl From<&Vec<u8>> for Base64 {
40    fn from(bytes: &Vec<u8>) -> Self {
41        Base64(bytes.clone())
42    }
43}
44
45impl From<&[u8]> for Base64 {
46    fn from(bytes: &[u8]) -> Self {
47        Base64(bytes.to_vec())
48    }
49}
50
51impl From<Vec<u8>> for Base64 {
52    fn from(bytes: Vec<u8>) -> Self {
53        Base64(bytes)
54    }
55}

61    #[test]
62    fn test_parse_valid_base64() {
63        let input = Value::String("SGVsbG8gd29ybGQ=".to_string());
64        let parsed = <Base64 as ScalarType>::parse(input).unwrap();
65        assert_eq!(parsed.0, b"Hello world");
66    }
