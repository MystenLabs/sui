15#[derive(Serialize, Deserialize, Clone, Debug, PartialEq, Eq)]
16#[serde(transparent)]
17pub(crate) struct BigInt(String);
18
19/// String representation of an arbitrary width, possibly signed integer
20#[Scalar]
21impl ScalarType for BigInt {
22    fn parse(value: Value) -> InputValueResult<Self> {
23        let Value::String(s) = value else {
24            return Err(InputValueError::expected_type(value));
25        };
26
27        BigInt::from_str(&s)
28    }
29
30    fn to_value(&self) -> Value {
31        Value::String(self.0.clone())
32    }
33}
34
35impl FromStr for BigInt {
36    type Err = InputValueError<Self>;
37
38    fn from_str(s: &str) -> Result<Self, Self::Err> {
39        let mut r = s;
40        let mut signed = false;
41        // check that all are digits and first can start with -
42        if let Some(suffix) = s.strip_prefix('-') {
43            r = suffix;
44            signed = true;
45        }
46        r = r.trim_start_matches('0');
47
48        if r.is_empty() {
49            Ok(BigInt("0".to_string()))
50        } else if r.chars().all(|c| c.is_ascii_digit()) {
51            Ok(BigInt(format!("{}{}", if signed { "-" } else { "" }, r)))
52        } else {
53            Err(InputValueError::custom("Not a number".to_string()))
54        }
55    }
56}
57
58macro_rules! impl_From {
59    ($($t:ident),*) => {
60        $(impl From<$t> for BigInt {
61            fn from(value: $t) -> Self {
62                BigInt(value.to_string())
63            }
64        })*
65    }
66}

70#[cfg(test)]
71mod tests {
72    use super::*;
73
74    #[test]
75    fn from_value() {
76        assert_eq!(BigInt::from_str("123").unwrap(), BigInt("123".to_string()));
77        assert_eq!(
78            BigInt::from_str("-123").unwrap(),
79            BigInt("-123".to_string())
80        );
81        assert_eq!(
82            BigInt::from_str("00233").unwrap(),
83            BigInt("233".to_string())
84        );
85        assert_eq!(BigInt::from_str("0").unwrap(), BigInt("0".to_string()));
86        assert_eq!(BigInt::from_str("-0").unwrap(), BigInt("0".to_string()));
87        assert_eq!(BigInt::from_str("000").unwrap(), BigInt("0".to_string()));
88        assert_eq!(BigInt::from_str("-000").unwrap(), BigInt("0".to_string()));
89
90        assert!(BigInt::from_str("123a").is_err());
91        assert!(BigInt::from_str("a123").is_err());
92        assert!(BigInt::from_str("123-").is_err());
93        assert!(BigInt::from_str(" 123").is_err());
94    }
95
96    #[test]
97    fn from_primitives() {
98        assert_eq!(BigInt::from(123u8), BigInt("123".to_string()));
99
100        assert_eq!(BigInt::from(12_345u16), BigInt("12345".to_string()));
101
102        assert_eq!(BigInt::from(123_456u32), BigInt("123456".to_string()));
103
104        assert_eq!(
105            BigInt::from(-12_345_678_901i64),
106            BigInt("-12345678901".to_string()),
107        );
108
109        assert_eq!(
110            BigInt::from(12_345_678_901u64),
111            BigInt("12345678901".to_string()),
112        );
113
114        assert_eq!(
115            BigInt::from(-123_456_789_012_345_678_901i128),
116            BigInt("-123456789012345678901".to_string()),
117        );
118
119        assert_eq!(
120            BigInt::from(123_456_789_012_345_678_901u128),
121            BigInt("123456789012345678901".to_string()),
122        );
123
124        assert_eq!(
125            BigInt::from(U256::from_str("12345678901234567890123456789012345678901").unwrap()),
126            BigInt("12345678901234567890123456789012345678901".to_string())
127        );
128
129        assert_eq!(BigInt::from(1000i64 - 1200i64), BigInt("-200".to_string()));
130        assert_eq!(BigInt::from(-1200i64), BigInt("-1200".to_string()));
131    }
