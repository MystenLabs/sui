4use std::str::FromStr;
5
6use fastcrypto::encoding::Base58;
7use fastcrypto::encoding::Encoding;
8use sui_types::digests::TransactionDigest;
9
10use crate::api::scalars::impl_string_input;
11
12const DIGEST_LENGTH: usize = 32;
13
14#[derive(Copy, Clone, Debug, Hash, Eq, PartialEq, Ord, PartialOrd)]
15pub(crate) struct Digest([u8; DIGEST_LENGTH]);
16
17#[derive(thiserror::Error, Debug)]
18pub(crate) enum Error {
19    #[error("Invalid Base58: {0}")]
20    InvalidBase58(String),
21
22    #[error("Expected digest to be {}B, but got {0}B", DIGEST_LENGTH)]
23    BadDigestLength(usize),
24}
25
26impl_string_input!(Digest);
27
28impl From<Digest> for TransactionDigest {
29    fn from(digest: Digest) -> Self {
30        digest.0.into()
31    }
32}
33
34impl FromStr for Digest {
35    type Err = Error;
36
37    fn from_str(s: &str) -> Result<Self, Self::Err> {
38        let buffer = Base58::decode(s).map_err(|_| Error::InvalidBase58(s.to_string()))?;
39        Digest::try_from(&buffer[..])
40    }
41}
42
43impl TryFrom<&[u8]> for Digest {
44    type Error = Error;
45
46    fn try_from(value: &[u8]) -> Result<Self, Error> {
47        let mut result = [0u8; DIGEST_LENGTH];
48
49        if value.len() != DIGEST_LENGTH {
50            return Err(Error::BadDigestLength(value.len()));
51        }
52
53        result.copy_from_slice(value);
54        Ok(Digest(result))
55    }
56}

58#[cfg(test)]
59mod tests {
60    use super::*;
61
62    #[test]
63    fn test_parse_valid_digest() {
64        assert_eq!(
65            Digest::from_str("DMBdBZnpYR4EeTXzXL8A6BtVafqGjAWGsFZhP2zJYmXU").unwrap(),
66            Digest([
67                183, 119, 223, 39, 204, 68, 220, 4, 126, 234, 232, 146, 106, 249, 98, 12, 170, 209,
68                98, 203, 243, 77, 154, 225, 177, 216, 169, 101, 51, 116, 79, 223,
69            ])
70        );
71    }
72
73    #[test]
74    fn test_parse_invalid_digest() {
75        assert!(matches!(
76            Digest::from_str("ILoveBase58").unwrap_err(),
77            Error::InvalidBase58(_),
78        ));
79    }
80
81    #[test]
82    fn test_parse_long_digest() {
83        assert!(matches!(
84            Digest::from_str(&Base58::encode(vec![0u8; DIGEST_LENGTH + 1])).unwrap_err(),
85            Error::BadDigestLength(_),
86        ))
87    }
88
89    #[test]
90    fn test_parse_short_digest() {
91        assert!(matches!(
92            Digest::from_str(&Base58::encode(vec![0u8; DIGEST_LENGTH - 1])).unwrap_err(),
93            Error::BadDigestLength(_),
94        ))
95    }
