4use std::{ops::Deref, str::FromStr};
5
6use serde::{Deserialize, Serialize};
7use sui_name_service::Domain as NativeDomain;
8
9use super::impl_string_input;
10
11/// Wrap SuiNS domain type to expose as a string scalar in GraphQL.
12#[derive(Serialize, Deserialize)]
13pub(crate) struct Domain(NativeDomain);
14
15impl_string_input!(Domain);
16
17impl Deref for Domain {
18    type Target = NativeDomain;
19
20    fn deref(&self) -> &Self::Target {
21        &self.0
22    }
23}
24
25impl FromStr for Domain {
26    type Err = <NativeDomain as FromStr>::Err;
27
28    fn from_str(s: &str) -> Result<Self, Self::Err> {
29        Ok(Domain(NativeDomain::from_str(s)?))
30    }
31}
32
33impl From<NativeDomain> for Domain {
34    fn from(value: NativeDomain) -> Self {
35        Domain(value)
36    }
37}
