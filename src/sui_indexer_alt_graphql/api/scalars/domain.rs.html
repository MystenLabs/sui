4use std::ops::Deref;
5use std::str::FromStr;
6
7use serde::Deserialize;
8use serde::Serialize;
9use sui_name_service::Domain as NativeDomain;
10
11use crate::api::scalars::impl_string_input;
12
13/// Wrap SuiNS domain type to expose as a string scalar in GraphQL.
14#[derive(Serialize, Deserialize)]
15pub(crate) struct Domain(NativeDomain);
16
17impl_string_input!(Domain);
18
19impl Deref for Domain {
20    type Target = NativeDomain;
21
22    fn deref(&self) -> &Self::Target {
23        &self.0
24    }
25}
26
27impl FromStr for Domain {
28    type Err = <NativeDomain as FromStr>::Err;
29
30    fn from_str(s: &str) -> Result<Self, Self::Err> {
31        Ok(Domain(NativeDomain::from_str(s)?))
32    }
33}
34
35impl From<NativeDomain> for Domain {
36    fn from(value: NativeDomain) -> Self {
37        Domain(value)
38    }
39}
40
41impl From<Domain> for NativeDomain {
42    fn from(value: Domain) -> Self {
43        value.0
44    }
45}
