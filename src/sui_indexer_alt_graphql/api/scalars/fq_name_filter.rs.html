4use std::fmt;
5use std::str::FromStr;
6
7use sui_types::parse_sui_fq_name;
8
9use crate::api::scalars::impl_string_input;
10use crate::api::scalars::module_filter::ModuleFilter;
11use crate::api::scalars::sui_address::SuiAddress;
12use crate::api::types::transaction::filter::Error;
13
14/// GraphQL scalar containing a filter on fully-qualified names.
15#[derive(Clone, Debug, Eq, PartialEq)]
16pub(crate) enum FqNameFilter {
17    /// Filter the module member by the package or module it's from.
18    Module(ModuleFilter),
19
20    /// Exact match on the module member.
21    FqName(SuiAddress, String, String),
22}
23
24impl FqNameFilter {
25    pub(crate) fn package(&self) -> SuiAddress {
26        match self {
27            FqNameFilter::Module(m) => m.package(),
28            FqNameFilter::FqName(p, _, _) => *p,
29        }
30    }
31
32    pub(crate) fn module(&self) -> Option<&str> {
33        match self {
34            FqNameFilter::Module(m) => m.module(),
35            FqNameFilter::FqName(_, m, _) => Some(m.as_str()),
36        }
37    }
38
39    pub(crate) fn name(&self) -> Option<&str> {
40        match self {
41            FqNameFilter::Module(_) => None,
42            FqNameFilter::FqName(_, _, n) => Some(n.as_str()),
43        }
44    }
45}
46
47impl_string_input!(FqNameFilter);
48
49impl FromStr for FqNameFilter {
50    type Err = Error;
51    fn from_str(s: &str) -> Result<Self, Error> {
52        if let Ok((module, name)) = parse_sui_fq_name(s) {
53            Ok(FqNameFilter::FqName(
54                SuiAddress::from(*module.address()),
55                module.name().to_string(),
56                name,
57            ))
58        } else if let Ok(filter) = ModuleFilter::from_str(s) {
59            Ok(FqNameFilter::Module(filter))
60        } else {
61            Err(Error::InvalidFormat("package[::module[::function]]"))
62        }
63    }
64}
65
66impl fmt::Display for FqNameFilter {
67    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
68        match self {
69            FqNameFilter::Module(m) => write!(f, "{m}"),
70            FqNameFilter::FqName(p, m, n) => write!(f, "{p}::{m}::{n}"),
71        }
72    }
73}
