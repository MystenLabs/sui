4use anyhow::Context as _;
5use async_graphql::Description;
6use async_graphql::InputValueResult;
7use async_graphql::Scalar;
8use async_graphql::ScalarType;
9use async_graphql::Value;
10
11use crate::error::RpcError;
12
13/// Arbitrary JSON data.
14#[derive(Debug)]
15pub(crate) struct Json(Value);
16
17#[Scalar(name = "JSON", use_type_description = true)]
18impl ScalarType for Json {
19    fn parse(value: Value) -> InputValueResult<Self> {
20        Ok(Self(value))
21    }
22
23    fn to_value(&self) -> Value {
24        self.0.clone()
25    }
26}
27
28impl Description for Json {
29    fn description() -> &'static str {
30        "Arbitrary JSON data."
31    }
32}
33
34impl From<Value> for Json {
35    fn from(value: Value) -> Self {
36        Self(value)
37    }
38}
39
40impl TryFrom<serde_json::Value> for Json {
41    type Error = RpcError;
42
43    fn try_from(value: serde_json::Value) -> Result<Self, RpcError> {
44        Ok(Self(value.try_into().context("Bad JSON")?))
45    }
46}
47
48impl TryInto<serde_json::Value> for Json {
49    type Error = serde_json::Error;
50
51    fn try_into(self) -> Result<serde_json::Value, Self::Error> {
52        self.0.into_json()
53    }
54}
