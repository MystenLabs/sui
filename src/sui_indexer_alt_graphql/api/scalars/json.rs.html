4use async_graphql::{Description, InputValueResult, Scalar, ScalarType, Value};
5
6/// Arbitrary JSON data.
7#[derive(Debug)]
8pub(crate) struct Json(Value);
9
10#[Scalar(name = "JSON", use_type_description = true)]
11impl ScalarType for Json {
12    fn parse(value: Value) -> InputValueResult<Self> {
13        Ok(Self(value))
14    }
15
16    fn to_value(&self) -> Value {
17        self.0.clone()
18    }
19}
20
21impl Description for Json {
22    fn description() -> &'static str {
23        "Arbitrary JSON data."
24    }
25}
26
27impl From<Value> for Json {
28    fn from(value: Value) -> Self {
29        Self(value)
30    }
31}
32
33impl TryInto<serde_json::Value> for Json {
34    type Error = serde_json::Error;
35
36    fn try_into(self) -> Result<serde_json::Value, Self::Error> {
37        self.0.into_json()
38    }
39}
