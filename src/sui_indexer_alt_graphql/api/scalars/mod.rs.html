18/// Opt-in to an implementation of `ScalarType` for a `$Type` that implements `FromStr`, solely for
19/// use as an input (not an output). The type masquarades as a `String` in the GraphQL schema, to
20/// avoid adding a new scalar.
21macro_rules! impl_string_input {
22    ($Type:ident) => {
23        #[async_graphql::Scalar(name = "String", visible = false)]
24        impl async_graphql::ScalarType for $Type {
25            fn parse(value: async_graphql::Value) -> async_graphql::InputValueResult<Self> {
26                if let async_graphql::Value::String(s) = value {
27                    Ok(Self::from_str(&s)?)
28                } else {
29                    Err(async_graphql::InputValueError::expected_type(value))
30                }
31            }
32
33            fn to_value(&self) -> async_graphql::Value {
34                unimplemented!("String inputs should not be output");
35            }
36        }
37    };
38}
39
40pub(crate) use impl_string_input;
