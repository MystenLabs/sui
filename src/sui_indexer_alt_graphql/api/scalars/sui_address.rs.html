18const SUI_ADDRESS_LENGTH: usize = 32;
19
20#[derive(Copy, Clone, Debug, Hash, Eq, PartialEq, Ord, PartialOrd, Serialize, Deserialize)]
21pub(crate) struct SuiAddress([u8; SUI_ADDRESS_LENGTH]);
22
23#[derive(thiserror::Error, Debug)]
24pub(crate) enum Error {
25    #[error("Invalid hexadecimal character at offset {0}")]
26    BadHex(usize),
27
28    #[error("Missing '0x' prefix while parsing SuiAddress: {0:?}")]
29    NoPrefix(String),
30
31    #[error(
32        "Expected between 1 and {} hexadecimal digits ({} bytes), received {0}",
33        SUI_ADDRESS_LENGTH * 2,
34        SUI_ADDRESS_LENGTH,
35    )]
36    WrongLength(usize),
37}
38
39/// String containing 32 byte hex-encoded address, with a leading '0x'. Leading zeroes can be omitted on input but will always appear in outputs (SuiAddress in output is guaranteed to be 66 characters long).
40#[Scalar]
41impl ScalarType for SuiAddress {
42    fn parse(value: Value) -> InputValueResult<Self> {
43        let Value::String(s) = value else {
44            return Err(InputValueError::expected_type(value));
45        };
46
47        Ok(SuiAddress::from_str(&s)?)
48    }
49
50    fn to_value(&self) -> Value {
51        Value::String(self.to_string())
52    }
53}
54
55impl SuiAddress {
56    pub fn into_vec(self) -> Vec<u8> {
57        self.0.to_vec()
58    }
59}

61impl FromStr for SuiAddress {
62    type Err = Error;
63
64    fn from_str(s: &str) -> Result<Self, Error> {
65        let Some(s) = s.strip_prefix("0x") else {
66            return Err(Error::NoPrefix(s.to_owned()));
67        };
68
69        if s.is_empty() || s.len() > SUI_ADDRESS_LENGTH * 2 {
70            return Err(Error::WrongLength(s.len()));
71        }
72
73        // Parse a single hexadecimal character from the string, or return an error pointing to the
74        // bad character in the source string.
75        let hex = |i: usize| -> Result<u8, Error> {
76            u8::from_str_radix(&s[i..=i], 16).map_err(|_| Error::BadHex(i + 2))
77        };
78
79        let mut arr = [0u8; SUI_ADDRESS_LENGTH];
80
81        let mut i = arr.len() - 1;
82        let mut j = s.len();
83
84        // Keep filling the array from the back until we have one hex digit left to process.
85        while j > 2 {
86            arr[i] = (hex(j - 2)? << 4) | hex(j - 1)?;
87            i -= 1;
88            j -= 2;
89        }
90
91        // Process the last hex digit, which will be compromised of either two or one character.
92        if j == 2 {
93            arr[i] = (hex(0)? << 4) | hex(1)?;
94        } else {
95            arr[i] = hex(0)?;
96        }
97
98        Ok(SuiAddress(arr))
99    }
100}

102impl fmt::Display for SuiAddress {
103    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
104        write!(f, "0x")?;
105        for byte in &self.0 {
106            write!(f, "{:02x}", byte)?;
107        }
108        Ok(())
109    }
110}
