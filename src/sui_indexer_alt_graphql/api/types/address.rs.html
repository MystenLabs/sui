36/// The possible relationship types for a transaction: sent or affected.
37#[derive(Enum, Copy, Clone, Eq, PartialEq)]
38pub(crate) enum AddressTransactionRelationship {
39    /// Transactions this address has sent.
40    Sent,
41    /// Transactions that this address was involved in, either as the sender, sponsor, or as the owner of some object that was created, modified or transferred.
42    Affected,
43}
44
45/// Interface implemented by GraphQL types representing entities that are identified by an address.
46///
47/// An address uniquely represents either the public key of an account, or an object's ID, but never both. It is not possible to determine which type an address represents up-front. If an object is wrapped, its contents will not be accessible via its address, but it will still be possible to access other objects it owns.
48#[allow(clippy::duplicated_attributes)]
49#[derive(Interface)]
50#[graphql(
51    name = "IAddressable",
52    field(name = "address", ty = "SuiAddress"),
53    field(
54        name = "balance",
55        arg(name = "coin_type", ty = "TypeInput"),
56        ty = "Result<Option<Balance>, RpcError<balance::Error>>",
57        desc = "Fetch the total balance for coins with marker type `coinType` (e.g. `0x2::sui::SUI`), owned by this address.\n\nIf the address does not own any coins of that type, a balance of zero is returned.",
58    ),
59    field(
60        name = "balances",
61        arg(name = "first", ty = "Option<u64>"),
62        arg(name = "after", ty = "Option<balance::Cursor>"),
63        arg(name = "last", ty = "Option<u64>"),
64        arg(name = "before", ty = "Option<balance::Cursor>"),
65        ty = "Result<Option<Connection<String, Balance>>, RpcError<balance::Error>>",
66        desc = "Total balance across coins owned by this address, grouped by coin type.",
67    ),
68    field(
69        name = "default_suins_name",
70        ty = "Result<Option<String>, RpcError<object::Error>>",
71        desc = "The domain explicitly configured as the default SuiNS name for this address."
72    ),
73    field(
74        name = "multi_get_balances",
75        arg(name = "keys", ty = "Vec<TypeInput>"),
76        ty = "Result<Option<Vec<Balance>>, RpcError<balance::Error>>",
77        desc = "Fetch the total balances keyed by coin types (e.g. `0x2::sui::SUI`) owned by this address.\n\nReturns `null` when no checkpoint is set in scope (e.g. execution scope). If the address does not own any coins of a given type, a balance of zero is returned for that type.",
78    ),
79    field(
80        name = "objects",
81        arg(name = "first", ty = "Option<u64>"),
82        arg(name = "after", ty = "Option<object::CLive>"),
83        arg(name = "last", ty = "Option<u64>"),
84        arg(name = "before", ty = "Option<object::CLive>"),
85        arg(name = "filter", ty = "Option<ObjectFilter>"),
86        ty = "Result<Option<Connection<String, MoveObject>>, RpcError<object::Error>>",
87        desc = "Objects owned by this address, optionally filtered by type."
88    )
89)]
90pub(crate) enum IAddressable {
91    Address(Address),
92    CoinMetadata(CoinMetadata),
93    DynamicField(DynamicField),
94    MoveObject(MoveObject),
95    MovePackage(MovePackage),
96    Object(Object),
97    Validator(Validator),
98}

108    /// The Address' identifier, a 32-byte number represented as a 64-character hex string, with a lead "0x".
109    pub(crate) async fn address(&self) -> Result<SuiAddress, RpcError> {
110        Ok(self.address.into())
111    }

113    /// Attempts to fetch the object at this address.
114    pub(crate) async fn as_object(
115        &self,
116        ctx: &Context<'_>,
117    ) -> Result<Option<Object>, RpcError<object::Error>> {
118        Object::by_key(
119            ctx,
120            self.scope.clone(),
121            ObjectKey {
122                address: self.address.into(),
123                version: None,
124                root_version: self.scope.root_version().map(Into::into),
125                at_checkpoint: None,
126            },
127        )
128        .await
129    }

131    /// Fetch the total balance for coins with marker type `coinType` (e.g. `0x2::sui::SUI`), owned by this address.
132    ///
133    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
134    /// If the address does not own any coins of that type, a balance of zero is returned.
135    pub(crate) async fn balance(
136        &self,
137        ctx: &Context<'_>,
138        coin_type: TypeInput,
139    ) -> Result<Option<Balance>, RpcError<balance::Error>> {
140        Balance::fetch_one(ctx, &self.scope, self.address, coin_type.into()).await
141    }

143    /// Total balance across coins owned by this address, grouped by coin type.
144    pub(crate) async fn balances(
145        &self,
146        ctx: &Context<'_>,
147        first: Option<u64>,
148        after: Option<balance::Cursor>,
149        last: Option<u64>,
150        before: Option<balance::Cursor>,
151    ) -> Result<Option<Connection<String, Balance>>, RpcError<balance::Error>> {
152        let pagination: &PaginationConfig = ctx.data()?;
153        let limits = pagination.limits("IAddressable", "balances");
154        let page = Page::from_params(limits, first, after, last, before)?;
155        Balance::paginate(ctx, self.scope.clone(), self.address, page)
156            .await
157            .map(Some)
158    }

278    /// Objects owned by this address, optionally filtered by type.
279    pub(crate) async fn objects(
280        &self,
281        ctx: &Context<'_>,
282        first: Option<u64>,
283        after: Option<object::CLive>,
284        last: Option<u64>,
285        before: Option<object::CLive>,
286        #[graphql(validator(custom = "OFValidator::allows_empty()"))] filter: Option<ObjectFilter>,
287    ) -> Result<Option<Connection<String, MoveObject>>, RpcError<object::Error>> {
288        let pagination: &PaginationConfig = ctx.data()?;
289        let limits = pagination.limits("IAddressable", "objects");
290        let page = Page::from_params(limits, first, after, last, before)?;
291
292        // Create a filter that constrains to ADDRESS kind and this owner
293        let Some(filter) = filter.unwrap_or_default().intersect(ObjectFilter {
294            owner_kind: Some(OwnerKind::Address),
295            owner: Some(self.address.into()),
296            ..Default::default()
297        }) else {
298            return Ok(Some(Connection::new(false, false)));
299        };
300
301        let objects = Object::paginate_live(ctx, self.scope.clone(), page, filter).await?;
302        let mut move_objects = Connection::new(objects.has_previous_page, objects.has_next_page);
303
304        for edge in objects.edges {
305            let move_obj = MoveObject::from_super(edge.node);
306            move_objects.edges.push(Edge::new(edge.cursor, move_obj));
307        }
308
309        Ok(Some(move_objects))
310    }

312    /// Transactions associated with this address.
313    ///
314    /// Similar behavior to the `transactions` in Query but supporting the additional `AddressTransactionRelationship` filter, which defaults to `SENT`.
315    pub(crate) async fn transactions(
316        &self,
317        ctx: &Context<'_>,
318        first: Option<u64>,
319        after: Option<CTransaction>,
320        last: Option<u64>,
321        before: Option<CTransaction>,
322        relation: Option<AddressTransactionRelationship>,
323        #[graphql(validator(custom = "TFValidator"))] filter: Option<TransactionFilter>,
324    ) -> Result<Option<Connection<String, Transaction>>, RpcError> {
325        let pagination: &PaginationConfig = ctx.data()?;
326        let limits = pagination.limits("Address", "transactions");
327        let page = Page::from_params(limits, first, after, last, before)?;
328
329        // Default relation to SENT if not provided
330        let relation = relation.unwrap_or(AddressTransactionRelationship::Sent);
331
332        // Create address-specific filter based on relationship
333        let address_filter = match relation {
334            AddressTransactionRelationship::Sent => TransactionFilter {
335                sent_address: Some(self.address.into()),
336                ..Default::default()
337            },
338            AddressTransactionRelationship::Affected => TransactionFilter {
339                affected_address: Some(self.address.into()),
340                ..Default::default()
341            },
342        };
343
344        // Intersect with user-provided filter
345        let Some(filter) = filter.unwrap_or_default().intersect(address_filter) else {
346            return Ok(Some(Connection::new(false, false)));
347        };
348
349        Transaction::paginate(ctx, self.scope.clone(), page, filter)
350            .await
351            .map(Some)
352    }
