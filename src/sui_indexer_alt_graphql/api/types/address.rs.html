49/// The possible relationship types for a transaction: sent or affected.
50#[derive(Enum, Copy, Clone, Eq, PartialEq)]
51pub(crate) enum AddressTransactionRelationship {
52    /// Transactions this address has sent.
53    Sent,
54    /// Transactions that this address was involved in, either as the sender, sponsor, or as the owner of some object that was created, modified or transferred.
55    Affected,
56}

125/// Identifies a specific version of an address.
126///
127/// Exactly one of `address` or `name` must be specified. Additionally, at most one of `rootVersion` or `atCheckpoint` can be specified. If neither bound is provided, the address is fetched at the checkpoint being viewed.
128///
129/// See `Query.address` for more details.
130#[derive(InputObject)]
131pub(crate) struct AddressKey {
132    /// The address.
133    pub(crate) address: Option<SuiAddress>,
134
135    /// A SuiNS name to resolve to an address.
136    pub(crate) name: Option<Domain>,
137
138    /// If specified, sets a root version bound for this address.
139    pub(crate) root_version: Option<UInt53>,
140
141    /// If specified, sets a checkpoint bound for this address.
142    pub(crate) at_checkpoint: Option<UInt53>,
143}
144
145#[derive(thiserror::Error, Debug, Clone)]
146pub(crate) enum Error {
147    #[error("Checkpoint {0} in the future")]
148    Future(u64),
149
150    #[error(transparent)]
151    Object(#[from] Arc<object::Error>),
152
153    #[error(
154        "At most one of a root version, or a checkpoint bound can be specified when fetching an address"
155    )]
156    OneBound,
157
158    #[error("Exactly one of `address` or `name` must be specified")]
159    OneId,
160}

376    /// Objects owned by this address, optionally filtered by type.
377    pub(crate) async fn objects(
378        &self,
379        ctx: &Context<'_>,
380        first: Option<u64>,
381        after: Option<object::CLive>,
382        last: Option<u64>,
383        before: Option<object::CLive>,
384        #[graphql(validator(custom = "OFValidator::allows_empty()"))] filter: Option<ObjectFilter>,
385    ) -> Option<Result<Connection<String, MoveObject>, RpcError<object::Error>>> {
386        Some(
387            async {
388                let pagination: &PaginationConfig = ctx.data()?;
389                let limits = pagination.limits("IAddressable", "objects");
390                let page = Page::from_params(limits, first, after, last, before)?;
391
392                // Create a filter that constrains to ADDRESS kind and this owner
393                let Some(filter) = filter.unwrap_or_default().intersect(ObjectFilter {
394                    owner_kind: Some(OwnerKind::Address),
395                    owner: Some(self.address.into()),
396                    ..Default::default()
397                }) else {
398                    return Ok(Connection::new(false, false));
399                };
400
401                let objects = Object::paginate_live(ctx, self.scope.clone(), page, filter).await?;
402                let mut move_objects =
403                    Connection::new(objects.has_previous_page, objects.has_next_page);
404
405                for edge in objects.edges {
406                    let move_obj = MoveObject::from_super(edge.node);
407                    move_objects.edges.push(Edge::new(edge.cursor, move_obj));
408                }
409
410                Ok(move_objects)
411            }
412            .await,
413        )
414    }

416    /// Transactions associated with this address.
417    ///
418    /// Similar behavior to the `transactions` in Query but supporting the additional `AddressTransactionRelationship` filter, which defaults to `SENT`.
419    pub(crate) async fn transactions(
420        &self,
421        ctx: &Context<'_>,
422        first: Option<u64>,
423        after: Option<CTransaction>,
424        last: Option<u64>,
425        before: Option<CTransaction>,
426        relation: Option<AddressTransactionRelationship>,
427        #[graphql(validator(custom = "TFValidator"))] filter: Option<TransactionFilter>,
428    ) -> Option<Result<Connection<String, Transaction>, RpcError>> {
429        Some(
430            async {
431                let pagination: &PaginationConfig = ctx.data()?;
432                let limits = pagination.limits("Address", "transactions");
433                let page = Page::from_params(limits, first, after, last, before)?;
434
435                // Default relation to SENT if not provided
436                let relation = relation.unwrap_or(AddressTransactionRelationship::Sent);
437
438                // Create address-specific filter based on relationship
439                let address_filter = match relation {
440                    AddressTransactionRelationship::Sent => TransactionFilter {
441                        sent_address: Some(self.address.into()),
442                        ..Default::default()
443                    },
444                    AddressTransactionRelationship::Affected => TransactionFilter {
445                        affected_address: Some(self.address.into()),
446                        ..Default::default()
447                    },
448                };
449
450                // Intersect with user-provided filter
451                let Some(filter) = filter.unwrap_or_default().intersect(address_filter) else {
452                    return Ok(Connection::new(false, false));
453                };
454
455                Transaction::paginate(ctx, self.scope.clone(), page, filter).await
456            }
457            .await,
458        )
459    }

462impl Address {
463    /// Fetch an address by its key. Exactly one of an address or a name must be provided.
464    /// Additionally, the key can specify a root version bound, or a checkpoint bound, or neither.
465    ///
466    /// If a name is provided, it is resolved to an address using SuiNS. If the name does not
467    /// resolve to an address (e.g. the name does not exist or has expired), `Ok(None)` is
468    /// returned.
469    pub(crate) async fn by_key(
470        ctx: &Context<'_>,
471        scope: Scope,
472        key: AddressKey,
473    ) -> Result<Option<Self>, RpcError<Error>> {
474        let bounds = key.root_version.is_some() as u8 + key.at_checkpoint.is_some() as u8;
475
476        if bounds > 1 {
477            return Err(bad_user_input(Error::OneBound));
478        }
479
480        // Determine the scope based on bounds
481        let scope = if let Some(v) = key.root_version {
482            scope.with_root_version(v.into())
483        } else if let Some(cp) = key.at_checkpoint {
484            let watermark: &Arc<Watermarks> = ctx.data()?;
485            if u64::from(cp) > watermark.high_watermark().checkpoint() {
486                return Err(bad_user_input(Error::Future(cp.into())));
487            }
488            scope.with_root_checkpoint(cp.into())
489        } else {
490            scope
491        };
492
493        // Resolve the address either directly or via name lookup.
494        let address = match (key.address, key.name) {
495            (Some(addr), None) => addr.into(),
496            (None, Some(name)) => {
497                let Some(target) = NameRecord::by_domain(ctx, scope.clone(), name.into())
498                    .await
499                    .map_err(convert)?
500                    .and_then(|r| r.record.target_address)
501                else {
502                    return Ok(None);
503                };
504
505                target
506            }
507
508            (None, None) | (Some(_), Some(_)) => {
509                return Err(bad_user_input(Error::OneId));
510            }
511        };
512
513        Ok(Some(Self::with_address(scope, address)))
514    }
515
516    /// Construct an address that is represented by just its identifier (`SuiAddress`).
517    /// This does not check whether the address is valid or exists in the system.
518    pub(crate) fn with_address(scope: Scope, address: NativeSuiAddress) -> Self {
519        Self { scope, address }
520    }
