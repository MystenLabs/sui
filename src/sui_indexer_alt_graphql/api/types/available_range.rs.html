33/// Identifies a GraphQL query component that is used to determine the range of checkpoints for which data is available (for data that can be tied to a particular checkpoint).
34///
35/// Provides retention information for the type and optional field and filters. If field or filters are not provided we fall back to the available range for the type.
36#[derive(Debug, Clone, Eq, PartialEq)]
37pub(crate) struct AvailableRangeKey {
38    /// The GraphQL type to check retention for
39    pub(crate) type_: String,
40
41    /// The specific field within the type to check retention for
42    pub(crate) field: Option<String>,
43
44    /// Optional filter to check retention for filtered queries
45    pub(crate) filters: Option<Vec<String>>,
46}
47
48#[derive(Clone)]
49pub(crate) struct AvailableRange {
50    pub(crate) scope: Scope,
51    pub(crate) first: u64,
52}
53
54/// Checkpoint range for which data is available.
55#[Object]
56impl AvailableRange {
57    /// Inclusive lower checkpoint for which data is available.
58    async fn first(&self) -> Option<Result<Checkpoint, RpcError>> {
59        Some(Ok(Checkpoint::with_sequence_number(
60            self.scope.clone(),
61            Some(self.first),
62        )?))
63    }
64
65    /// Inclusive upper checkpoint for which data is available.
66    async fn last(&self) -> Option<Result<Checkpoint, RpcError>> {
67        Some(Ok(Checkpoint::with_sequence_number(
68            self.scope.clone(),
69            None,
70        )?))
71    }
72}

74impl AvailableRange {
75    /// Computes the available checkpoint range for a GraphQL query.
76    ///
77    /// The first checkpoint is the max reader_lo of the pipelines match by the AvailableRangeKey.
78    /// The last checkpoint is scope.checkpoint_viewed_at.
79    pub(crate) fn new(
80        ctx: &Context<'_>,
81        scope: &Scope,
82        available_range_key: AvailableRangeKey,
83    ) -> Result<Self, RpcError<Error>> {
84        available_range_key.validate(&ctx.schema_env.registry)?;
85        let watermarks: &Arc<Watermarks> = ctx.data()?;
86        let first = available_range_key.reader_lo(watermarks).map_err(upcast)?;
87
88        Ok(Self {
89            scope: scope.clone(),
90            first,
91        })
92    }
93}
94
95impl AvailableRangeKey {
96    /// The max reader_lo for the pipelines that match the available range key.
97    pub(crate) fn reader_lo(self, watermarks: &Watermarks) -> Result<u64, RpcError> {
98        let mut pipelines = BTreeSet::new();
99        let filters = BTreeSet::from_iter(self.filters.unwrap_or_default());
100        collect_pipelines(&self.type_, self.field.as_deref(), filters, &mut pipelines);
101
102        pipelines.iter().try_fold(0, |acc, pipeline| {
103            let watermark = watermarks
104                .per_pipeline()
105                .get(pipeline)
106                .ok_or_else(|| pipeline_unavailable(pipeline))?;
107            Ok(acc.max(watermark.lo().checkpoint()))
108        })
109    }
110
111    /// Validates that this key references valid types and fields in the GraphQL schema.
112    fn validate(&self, registry: &Registry) -> Result<(), RpcError<Error>> {
113        let fields = match registry.types.get(&self.type_) {
114            Some(MetaType::Object { fields, .. } | MetaType::Interface { fields, .. }) => fields,
115            Some(_) => {
116                return Err(bad_user_input(Error::NotAnObjectOrInterface(
117                    self.type_.to_string(),
118                )));
119            }
120            None => {
121                return Err(bad_user_input(Error::TypeNotFound(self.type_.to_string())));
122            }
123        };
124
125        if let Some(field_name) = &self.field {
126            fields.get(field_name).ok_or_else(|| {
127                bad_user_input(Error::FieldNotFound(
128                    self.type_.to_string(),
129                    field_name.to_string(),
130                ))
131            })?;
132        }
133
134        Ok(())
135    }
136}

138/// Return the appropriate error for the query or filter if the pipeline is not available.
139/// Certain queryies or filters are not available if the pipeline supporting them is not configured.
140pub(crate) fn pipeline_unavailable(pipeline: &str) -> RpcError {
141    match pipeline {
142        "consistent" => feature_unavailable("consistent queries across objects and balances"),
143        "cp_sequence_numbers" => feature_unavailable("querying checkpoints"),
144        "ev_emit_mod" => feature_unavailable("querying events by emitting module"),
145        "ev_struct_inst" => feature_unavailable("querying events by type"),
146        "obj_versions" => feature_unavailable("querying object versions"),
147        "tx_affected_addresses" => {
148            feature_unavailable("filtering transactions by affected address")
149        }
150        "tx_affected_objects" => feature_unavailable("filtering transactions by affected object"),
151        "tx_balance_changes" => feature_unavailable("querying transaction balance changes"),
152        "tx_calls" => feature_unavailable("filtering transactions by function calls"),
153        "tx_digests" => feature_unavailable("querying transactions"),
154        "tx_kinds" => feature_unavailable("filtering transactions by kind"),
155        _ => anyhow!("unrecognized pipeline name: {}", pipeline).into(),
156    }
157}

159/// Expands a series of type/field/filter patterns to generate the collect_pipelines function and collects macro invocation data into
160/// a HashMap of (Type, Field) to (DelegateType, DelegateField) for tests.
161macro_rules! collect_pipelines {
162    (
163        $($type:ident . [$($field:ident),*]
164        $(=> $delegate_type:ident . $delegate_field:tt $( ( $(.., $f_to_add:literal)? ) )? )?
165        $(|$pipes:ident, $filt:ident| $block:block)?
166        ;
167    )*) => {
168        /// Populates `pipelines` with pipeline names by matching the type, field, and filters to their dependent pipelines where data is available.
169        /// The mapping is defined in the collect_pipelines! macro innvocation.
170        fn collect_pipelines(
171            type_: &str,
172            field: Option<&str>,
173            mut filters: BTreeSet<String>,
174            pipelines: &mut BTreeSet<String>,
175        ) {
176            match (type_, field) {
177                $(
178                    (stringify!($type), _field @ Some($(stringify!($field)) |*)) => {
179                        $(
180                            $($(
181                                filters.insert($f_to_add.to_string());
182                            )?)?
183                            let delegate_field = delegate!(_field, $delegate_field);
184                            collect_pipelines(stringify!($delegate_type), delegate_field, filters, pipelines);
185                        )?
186                        $(
187                            let $filt = filters;
188                            let $pipes = pipelines;
189                            $block
190                        )?
191                    }
192                )*
193                _ => {}
194            }
195        }
196
197        /// Map from (Type, Field) to (DelegateType, DelegateField) for testing.
198        /// Maps to ("", "") if the type/field is not delegated. The "*" wildcard is expanded.
199        #[cfg(test)]
200        static TYPE_FIELD_DELEGATIONS: std::sync::LazyLock<std::collections::HashMap<(&'static str, &'static str), (&'static str, &'static str)>> =
201            std::sync::LazyLock::new(|| {
202                let mut map = std::collections::HashMap::new();
203                $(
204                    let type_name = stringify!($type);
205                    let fields = [$(stringify!($field),)*];
206                    let delegate_type = stringify!($($delegate_type)?);
207                    for field in fields {
208                        let delegate_field = match stringify!($($delegate_field)?) {
209                            "*" => field,
210                            _ => stringify!($($delegate_field)?),
211                        };
212                        map.insert((type_name, field), (delegate_type, delegate_field));
213                    }
214                )*
215                map
216            });
217        };
218}

220macro_rules! delegate {
221    ($field:ident, *) => {
222        $field
223    };
224    ($field:ident, $delegate:tt) => {
225        Some(stringify!($delegate))
226    };
227}

490    /// If a type implements an interface, every field in the interface is delegated to that interface in the macro invocation.
491    /// ex. CoinMetadata.[balance] => IMoveObject.balance();  in the macro invocation should throw an error because
492    ///     Type 'CoinMetadata' field 'balance' should delegate to interface 'IAddressable' but delegates to 'IMoveObject'
493    fn type_delegation_matches_interface(registry: &Registry) {
494        let type_delegations = &TYPE_FIELD_DELEGATIONS;
495
496        for (interface_name, meta_type) in registry.types.iter() {
497            let MetaType::Interface {
498                name,
499                possible_types,
500                fields,
501                ..
502            } = meta_type
503            else {
504                continue;
505            };
506
507            // Node is part of the GraphQL Global Identification specification, it does not have
508            // any retention requirements, so can be safely skipped.
509            if name == "Node" {
510                continue;
511            }
512
513            for type_name in possible_types {
514                for (interface_field_name, _) in fields {
515                    let Some((delegate_type, delegate_field)) =
516                        type_delegations.get(&(type_name.as_str(), interface_field_name.as_str()))
517                    else {
518                        panic!(
519                            "Type '{}' field '{}' should delegate to interface '{}' but does not",
520                            type_name, interface_field_name, interface_name
521                        );
522                    };
523
524                    assert_eq!(
525                        delegate_type, interface_name,
526                        "Type '{}' field '{}' should delegate to interface '{}' but delegates to '{}'",
527                        type_name, interface_field_name, interface_name, delegate_type
528                    );
529                    assert_eq!(
530                        delegate_field, interface_field_name,
531                        "Type '{}' field '{}' should delegate to interface field '{}' but delegates to '{}'",
532                        type_name, interface_field_name, interface_field_name, delegate_field
533                    );
534                }
535            }
536        }
537    }

539    /// Validates that the macro invocation matches the schema. This will error if there are any
540    /// types or fields in the macro invocation are not found in the schema registry.
541    fn macro_invocation_matches_schema(registry: &Registry) {
542        let type_fields = &TYPE_FIELD_DELEGATIONS;
543
544        for ((type_name, field_name), _) in type_fields.iter() {
545            let fields = match registry.types.get(*type_name) {
546                Some(MetaType::Object { fields, .. } | MetaType::Interface { fields, .. }) => {
547                    fields
548                }
549                Some(_) => panic!("Type '{}' is not an Object or Interface type", type_name),
550                None => panic!("Type '{}' not found in schema registry", type_name),
551            };
552            fields.get(*field_name).unwrap_or_else(|| {
553                panic!("Field '{}' not found in type '{}'", field_name, type_name)
554            });
555        }
556    }

558    /// Calls collect_pipeline on types, fields, and filters in the schema registry to
559    /// 1. store a snapshot of the input and output pipelines for regression testing and auditing.
560    /// 2. validate pipelines in the collect_pipelines macro innovation are mapped to a feature in pipeline_unavailable.
561    ///
562    /// If a filter does not result in a different set of pipelines from the unfiltered case,
563    /// it is not included in the snapshot.
564    fn registry_collect_pipelines_snapshot(registry: &Registry) {
565        const PAGINATION_ARGS: &[&str] = &["first", "after", "last", "before"];
566
567        let mut output = String::new();
568        let mut all_pipelines = BTreeSet::new();
569
570        for (type_name, meta_type) in registry.types.iter() {
571            let (MetaType::Object { fields, .. } | MetaType::Interface { fields, .. }) = meta_type
572            else {
573                continue;
574            };
575
576            for (field_name, meta_field) in fields.iter() {
577                if should_ignore_in_snapshot(type_name, field_name) {
578                    continue;
579                }
580                let filter_fields: Vec<String> = meta_field
581                    .args
582                    .iter()
583                    .filter(|(name, _)| !PAGINATION_ARGS.contains(&name.as_str()))
584                    .flat_map(|(param_name, meta_input_value)| {
585                        let concrete_type = MetaTypeName::concrete_typename(&meta_input_value.ty);
586                        match registry.types.get(concrete_type) {
587                            Some(MetaType::InputObject { input_fields, .. }) => {
588                                input_fields.keys().cloned().collect()
589                            }
590                            Some(MetaType::Scalar { .. }) => {
591                                vec![param_name.clone()]
592                            }
593                            _ => vec![],
594                        }
595                    })
596                    .collect();
597
598                // Call collect_pipelines with no filters to get the unfiltered pipelines.
599                let mut unfiltered_pipelines = BTreeSet::new();
600                super::collect_pipelines(
601                    type_name,
602                    Some(field_name),
603                    BTreeSet::new(),
604                    &mut unfiltered_pipelines,
605                );
606                let unfiltered_output_str =
607                    formatted_output_str(type_name, field_name, &unfiltered_pipelines, None);
608                output.push_str(&unfiltered_output_str);
609                all_pipelines.extend(unfiltered_pipelines.iter().cloned());
610
611                // Call collect_pipelines with each filter to get the filtered pipelines.
612                for filter_field in filter_fields.iter().map(Some) {
613                    let filters = filter_field.iter().copied().map(String::from).collect();
614                    let mut pipelines = BTreeSet::new();
615                    super::collect_pipelines(type_name, Some(field_name), filters, &mut pipelines);
616                    let output_str =
617                        formatted_output_str(type_name, field_name, &pipelines, filter_field);
618                    if unfiltered_pipelines != pipelines {
619                        output.push_str(&output_str);
620                        all_pipelines.extend(pipelines.iter().cloned());
621                    }
622                }
623            }
624        }
625        // Validate all pipelines have feature mappings.
626        for pipeline in all_pipelines {
627            let result = super::pipeline_unavailable(pipeline.as_str());
628            if let RpcError::InternalError(err) = result {
629                panic!(
630                    "Pipeline '{}' does not have a feature mapping in pipeline_unavailable: {}",
631                    pipeline, err
632                );
633            }
634        }
635        insta::assert_snapshot!(output);
636    }
637
638    fn formatted_output_str(
639        type_name: &str,
640        field_name: &str,
641        pipelines: &BTreeSet<String>,
642        filter_field: Option<&String>,
643    ) -> String {
644        let filter_suffix = filter_field.map_or(String::new(), |f| format!(" (filter: {f})"));
645        let pipeline_list = pipelines
646            .iter()
647            .map(|s| format!("\"{s}\""))
648            .collect::<Vec<_>>()
649            .join(", ");
650
651        let output_str =
652            format!("{type_name}.{field_name}{filter_suffix}\n  => {{{pipeline_list}}}\n\n");
653        output_str
654    }

656    /// If the type or fields is a generated type or field, or an introspection type, it is skipped in the snapshot.
657    fn should_ignore_in_snapshot(type_name: &str, field_name: &str) -> bool {
658        const GENERATED_TYPES: &[&str] = &["Connection", "Edge"];
659        const GENERATED_FIELDS: &[&str] = &["node", "nodes", "edges", "cursor", "pageInfo"];
660        const INTROSPECTION_TYPE: &str = "__";
661        if type_name.starts_with(INTROSPECTION_TYPE)
662            || GENERATED_TYPES
663                .iter()
664                .any(|suffix| type_name.ends_with(suffix))
665                && GENERATED_FIELDS
666                    .iter()
667                    .any(|suffix| field_name.ends_with(suffix))
668        {
669            return true;
670        }
671        false
672    }
