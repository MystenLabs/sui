26/// The total balance for a particular coin type.
27#[derive(SimpleObject)]
28pub(crate) struct Balance {
29    /// Coin type for the balance, such as `0x2::sui::SUI`.
30    pub(crate) coin_type: Option<MoveType>,
31
32    /// The sum total of the accumulator balance and individual coin balances owned by the address.
33    pub(crate) total_balance: Option<BigInt>,
34
35    /// Total balance across all owned coin objects of the coin type.
36    pub(crate) coin_balance: Option<BigInt>,
37
38    /// The balance as tracked by the accumulator object for the address.
39    pub(crate) address_balance: Option<BigInt>,
40}
41
42#[derive(thiserror::Error, Debug, Clone)]
43pub(crate) enum Error {
44    #[error("Cursors are pinned to different checkpoints: {0} vs {1}")]
45    CursorInconsistency(u64, u64),
46
47    #[error("Request is outside consistent range")]
48    OutOfRange(u64),
49
50    #[error("Checkpoint {0} in the future")]
51    Future(u64),
52
53    #[error(
54        "Cannot query balances for a parent object's address if its version is bounded. Fetch the parent at a checkpoint in the consistent range to query its balances."
55    )]
56    RootVersionOwnership,
57}
58
59pub(crate) type Cursor = cursor::BcsCursor<(u64, Vec<u8>)>;
60
61impl Balance {
62    fn try_from_proto(proto: ProtoBalance, scope: Scope) -> Result<Self, RpcError<Error>> {
63        let coin_type: TypeTag = proto
64            .coin_type
65            .context("coin type missing")?
66            .parse()
67            .context("invalid coin type")?;
68
69        Ok(Balance {
70            coin_type: Some(MoveType::from_native(coin_type, scope)),
71            total_balance: Some(BigInt::from(proto.total_balance.unwrap_or(0))),
72            coin_balance: Some(BigInt::from(proto.coin_balance.unwrap_or(0))),
73            address_balance: Some(BigInt::from(proto.address_balance.unwrap_or(0))),
74        })
75    }
76
77    /// Fetch the balance for a single coin type owned by the given address, live at the current
78    /// checkpoint.
79    ///
80    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
81    pub(crate) async fn fetch_one(
82        ctx: &Context<'_>,
83        scope: &Scope,
84        address: SuiAddress,
85        coin_type: TypeTag,
86    ) -> Result<Option<Balance>, RpcError<Error>> {
87        if scope.root_version().is_some() {
88            return Err(bad_user_input(Error::RootVersionOwnership));
89        }
90
91        let Some(checkpoint) = scope.root_checkpoint() else {
92            return Ok(None);
93        };
94
95        query_limits::rich::debit(ctx)?;
96        let consistent_reader: &ConsistentReader = ctx.data()?;
97        let balance = consistent_reader
98            .get_balance(
99                Some(checkpoint),
100                address.to_string(),
101                coin_type.to_canonical_string(true),
102            )
103            .await
104            .map_err(|e| consistent_error(checkpoint, e))?;
105
106        Ok(Some(Balance::try_from_proto(balance, scope.clone())?))
107    }
108
109    /// Fetch balances for multiple coin types owned by the given address, live at the current
110    /// checkpoint. Returns `None` when no checkpoint is set in scope (e.g. execution scope).
111    pub(crate) async fn fetch_many(
112        ctx: &Context<'_>,
113        scope: &Scope,
114        address: SuiAddress,
115        coin_types: Vec<TypeTag>,
116    ) -> Result<Option<Vec<Balance>>, RpcError<Error>> {
117        if scope.root_version().is_some() {
118            return Err(bad_user_input(Error::RootVersionOwnership));
119        }
120
121        let Some(checkpoint) = scope.root_checkpoint() else {
122            return Ok(None);
123        };
124
125        query_limits::rich::debit(ctx)?;
126        let consistent_reader: &ConsistentReader = ctx.data()?;
127        let balances = consistent_reader
128            .batch_get_balances(
129                checkpoint,
130                address.to_string(),
131                coin_types
132                    .into_iter()
133                    .map(|t| t.to_canonical_string(true))
134                    .collect(),
135            )
136            .await
137            .map_err(|e| consistent_error(checkpoint, e))?;
138
139        Ok(Some(
140            balances
141                .into_iter()
142                .map(|balance| Balance::try_from_proto(balance, scope.clone()))
143                .collect::<Result<_, _>>()?,
144        ))
145    }
146
147    /// Paginate through balances for coins owned by the given address, live at the current
148    /// checkpoint.
149    pub(crate) async fn paginate(
150        ctx: &Context<'_>,
151        scope: Scope,
152        address: SuiAddress,
153        page: Page<Cursor>,
154    ) -> Result<Connection<String, Balance>, RpcError<Error>> {
155        if scope.root_version().is_some() {
156            return Err(bad_user_input(Error::RootVersionOwnership));
157        }
158
159        let Some(root_checkpoint) = scope.root_checkpoint() else {
160            return Ok(Connection::new(false, false));
161        };
162
163        query_limits::rich::debit(ctx)?;
164        let consistent_reader: &ConsistentReader = ctx.data()?;
165
166        // Figure out which checkpoint to pin results to, based on the pagination cursors and
167        // defaulting to the current scope. If both cursors are provided, they must agree on the
168        // checkpoint they are pinning, and this checkpoint must be at or below the scope's latest
169        // checkpoint.
170        let checkpoint = match (page.after(), page.before()) {
171            (Some(a), Some(b)) if a.0 != b.0 => {
172                return Err(bad_user_input(Error::CursorInconsistency(a.0, b.0)));
173            }
174            (None, None) => root_checkpoint,
175            (Some(c), _) | (_, Some(c)) => c.0,
176        };
177
178        let Some(scope) = scope.with_checkpoint_viewed_at(ctx, checkpoint) else {
179            return Err(bad_user_input(Error::Future(checkpoint)));
180        };
181
182        let balances = consistent_reader
183            .list_balances(
184                Some(checkpoint),
185                address.to_string(),
186                Some(page.limit() as u32),
187                page.after().map(|c| c.1.clone()),
188                page.before().map(|c| c.1.clone()),
189                page.is_from_front(),
190            )
191            .await
192            .map_err(|e| consistent_error(checkpoint, e))?;
193
194        let mut conn = Connection::new(false, false);
195        if balances.results.is_empty() {
196            return Ok(conn);
197        }
198
199        conn.has_previous_page = balances.has_previous_page;
200        conn.has_next_page = balances.has_next_page;
201
202        for edge in balances.results {
203            let balance = edge.value;
204
205            let cursor = Cursor::new((checkpoint, edge.token));
206            let balance = Balance::try_from_proto(balance, scope.clone())?;
207
208            conn.edges.push(Edge::new(cursor.encode_cursor(), balance));
209        }
210
211        Ok(conn)
212    }
213}

215/// Convert an error from the consistent reader into an RpcError, assuming the request was made at
216/// the given `checkpoint`.
217fn consistent_error(checkpoint: u64, error: consistent_reader::Error) -> RpcError<Error> {
218    match error {
219        consistent_reader::Error::NotConfigured => {
220            feature_unavailable("fetching balances for addresses")
221        }
222
223        consistent_reader::Error::OutOfRange(_) => bad_user_input(Error::OutOfRange(checkpoint)),
224
225        consistent_reader::Error::Internal(error) => {
226            error.context("Failed to fetch balance").into()
227        }
228    }
