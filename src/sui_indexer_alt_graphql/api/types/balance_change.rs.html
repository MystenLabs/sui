19/// Effects to the balance (sum of coin values per coin type) of addresses and objects.
20#[derive(Clone, SimpleObject)]
21pub(crate) struct BalanceChange {
22    /// The address or object whose balance has changed.
23    pub(crate) owner: Option<Address>,
24    /// The inner type of the coin whose balance has changed (e.g. `0x2::sui::SUI`).
25    pub(crate) coin_type: Option<MoveType>,
26    /// The signed balance change.
27    pub(crate) amount: Option<BigInt>,
28}
29
30impl BalanceChange {
31    /// Create a BalanceChange from a gRPC BalanceChange.
32    pub(crate) fn from_grpc(scope: Scope, grpc: &GrpcBalanceChange) -> Result<Self, RpcError> {
33        let address = grpc.address().parse().context("Failed to parse address")?;
34        let coin_type: TypeTag = grpc
35            .coin_type()
36            .parse()
37            .context("Failed to parse coin type")?;
38        let amount: i128 = grpc.amount().parse().context("Failed to parse amount")?;
39
40        Ok(Self {
41            owner: Some(Address::with_address(scope.clone(), address)),
42            coin_type: Some(MoveType::from_native(coin_type, scope)),
43            amount: Some(BigInt::from(amount)),
44        })
45    }
46
47    /// Create a BalanceChange from a stored BalanceChange (database).
48    pub(crate) fn from_stored(scope: Scope, stored: StoredBalanceChange) -> Result<Self, RpcError> {
49        let StoredBalanceChange::V1 {
50            owner,
51            coin_type,
52            amount,
53        } = stored;
54
55        // Extract address from owner
56        let address = match owner {
57            Owner::AddressOwner(addr)
58            | Owner::ObjectOwner(addr)
59            | Owner::ConsensusAddressOwner { owner: addr, .. } => Some(addr),
60            Owner::Shared { .. } | Owner::Immutable => None,
61        };
62
63        let coin_type = TypeTag::from_str(&coin_type).context("Failed to parse coin type")?;
64
65        Ok(Self {
66            owner: address.map(|addr| Address::with_address(scope.clone(), addr)),
67            coin_type: Some(MoveType::from_native(coin_type, scope)),
68            amount: Some(BigInt::from(amount)),
69        })
70    }
71
72    /// Convert a stored BalanceChange to a gRPC BalanceChange.
73    pub(crate) fn stored_to_grpc(stored: StoredBalanceChange) -> GrpcBalanceChange {
74        let StoredBalanceChange::V1 {
75            owner,
76            coin_type,
77            amount,
78        } = stored;
79
80        // Extract address from owner
81        let address = match owner {
82            Owner::AddressOwner(addr)
83            | Owner::ObjectOwner(addr)
84            | Owner::ConsensusAddressOwner { owner: addr, .. } => Some(addr),
85            Owner::Shared { .. } | Owner::Immutable => None,
86        };
87
88        let mut grpc = GrpcBalanceChange::default();
89        if let Some(addr) = address {
90            grpc.set_address(addr.to_string());
91        }
92        grpc.set_coin_type(coin_type);
93        grpc.set_amount(amount.to_string());
94        grpc
95    }
96}
