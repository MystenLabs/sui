22// Filter for checkpoint-based queries across checkpoints, packages, and epochs.
23#[derive(InputObject, Debug, Default, Clone)]
24pub(crate) struct CheckpointFilter {
25    /// Limit query results to checkpoints at this epoch.
26    pub at_epoch: Option<UInt53>,
27
28    /// Limit query results to checkpoints that occured strictly after the given checkpoint.
29    pub after_checkpoint: Option<UInt53>,
30
31    /// Limit query results to checkpoints that occured at the given checkpoint.
32    pub at_checkpoint: Option<UInt53>,
33
34    /// Limit query results to checkpoints that occured strictly before the given checkpoint.
35    pub before_checkpoint: Option<UInt53>,
36}
37
38#[derive(QueryableByName, Debug)]
39pub(crate) struct CpBounds {
40    #[diesel(sql_type = BigInt, column_name = "cp_lo")]
41    cp_lo: i64,
42    #[diesel(sql_type = BigInt, column_name = "cp_hi_inclusive")]
43    cp_hi_inclusive: i64,
44}
45
46impl CheckpointFilter {
47    pub(crate) fn intersect(self, other: Self) -> Option<Self> {
48        macro_rules! intersect {
49            ($field:ident, $body:expr) => {
50                intersect::field(self.$field, other.$field, $body)
51            };
52        }
53
54        Some(Self {
55            at_epoch: intersect!(at_epoch, intersect::by_eq)?,
56            after_checkpoint: intersect!(after_checkpoint, intersect::by_max)?,
57            at_checkpoint: intersect!(at_checkpoint, intersect::by_eq)?,
58            before_checkpoint: intersect!(before_checkpoint, intersect::by_min)?,
59        })
60    }
61
62    /// The active filters in CheckpointFilter. Used to find the pipelines that are available to serve queries with these filters applied.
63    pub(crate) fn active_filters(&self) -> Vec<String> {
64        let mut filters = vec![];
65        if self.at_epoch.is_some() {
66            filters.push("atEpoch".to_string());
67        }
68        if self.at_checkpoint.is_some() {
69            filters.push("atCheckpoint".to_string());
70        }
71        if self.after_checkpoint.is_some() {
72            filters.push("afterCheckpoint".to_string());
73        }
74        if self.before_checkpoint.is_some() {
75            filters.push("beforeCheckpoint".to_string());
76        }
77        filters
78    }
79}

81/// The bounds on checkpoint sequence number, imposed by filters. The outermost bounds are
82/// determined by the lowest checkpoint that is safe to read from (reader_lo) and the highest
83/// checkpoint that has been processed based on the context's watermark(checkpoint_viewed_at).
84///
85/// ```ignore
86///     reader_lo                                                     checkpoint_viewed_at
87///     [-----------------------------------------------------------------]
88/// ```
89///
90/// The bounds are further tightened by the filters if they are present.
91///
92/// ```ignore
93///    filter.after_checkpoint                         filter.before_checkpoint
94///         [------------[--------------------------]------------]
95///                         filter.at_checkpoint
96/// ```
97pub(crate) fn checkpoint_bounds(
98    cp_after: Option<u64>,
99    cp_at: Option<u64>,
100    cp_before: Option<u64>,
101    reader_lo: u64,
102    checkpoint_viewed_at: u64,
103) -> Option<RangeInclusive<u64>> {
104    let cp_after_inclusive = match cp_after.map(|x| x.checked_add(1)) {
105        Some(Some(after)) => Some(after),
106        Some(None) => return None,
107        None => None,
108    };
109    // Inclusive checkpoint sequence number lower bound. If there are no lower bound filters,
110    // we will use the smallest checkpoint available from the database, retrieved from
111    // the watermark.
112    //
113    // SAFETY: we can unwrap because of `Some(reader_lo)`
114    let cp_lo = max_option([cp_after_inclusive, cp_at, Some(reader_lo)]).unwrap();
115
116    let cp_before_inclusive = match cp_before.map(|x| x.checked_sub(1)) {
117        Some(Some(before)) => Some(before),
118        Some(None) => return None,
119        None => None,
120    };
121
122    // Inclusive checkpoint sequence upper bound. If there are no upper bound filters,
123    // we will use `checkpoint_viewed_at`.
124    //
125    // SAFETY: we can unwrap because of the `Some(checkpoint_viewed_at)`
126    let cp_hi = min_option([cp_before_inclusive, cp_at, Some(checkpoint_viewed_at)]).unwrap();
127
128    cp_hi
129        .checked_sub(cp_lo)
130        .map(|_| RangeInclusive::new(cp_lo, cp_hi))
131}

133/// The cp_sequence_numbers within checkpoint bounds with cursors applied inclusively.
134///
135/// pg_lo: The maximum of the cursor and the start of the checkpoint bound.
136/// pg_hi_inclusive: The minimum of the cursor and the end of the checkpoint bound.
137pub(super) fn cp_unfiltered(cp_bounds: &RangeInclusive<u64>, page: &Page<CCheckpoint>) -> Vec<u64> {
138    let cp_lo = *cp_bounds.start();
139    let cp_hi = *cp_bounds.end();
140
141    // Inclusive cursor bounds
142    let pg_lo = page.after().map_or(cp_lo, |cursor| cursor.max(cp_lo));
143    let pg_hi_inclusive = page.before().map_or(cp_hi, |cursor| cursor.min(cp_hi));
144
145    if page.is_from_front() {
146        (pg_lo..=pg_hi_inclusive)
147            .take(page.limit_with_overhead())
148            .collect()
149    } else {
150        // Graphql last syntax expects results to be in ascending order. If we are paginating backwards,
151        // we reverse the results after applying limits.
152        let mut results: Vec<_> = (pg_lo..=pg_hi_inclusive)
153            .rev()
154            .take(page.limit_with_overhead())
155            .collect();
156        results.reverse();
157        results
158    }
159}

161/// The checkpoint sequence numbers in a range bounded by checkpoints in an epoch.
162/// The range is further tightened by bounds derived from filters if they are present.
163pub(super) async fn cp_by_epoch(
164    ctx: &Context<'_>,
165    page: &Page<CCheckpoint>,
166    cp_bounds: &RangeInclusive<u64>,
167    epoch: u64,
168) -> Result<Vec<u64>, RpcError> {
169    query_limits::rich::debit::<Infallible>(ctx)?;
170    let pg_reader: &PgReader = ctx.data()?;
171
172    let query = query!(
173        r#"
174        SELECT
175            GREATEST(es.cp_lo, {BigInt}) AS "cp_lo",
176            LEAST(COALESCE(ee.cp_hi - 1, {BigInt}), {BigInt}) AS "cp_hi_inclusive"
177        FROM
178            kv_epoch_starts es
179        LEFT JOIN kv_epoch_ends ee ON es.epoch = ee.epoch
180        WHERE
181            es.epoch = {BigInt}
182        "#,
183        *cp_bounds.start() as i64,
184        *cp_bounds.end() as i64,
185        *cp_bounds.end() as i64,
186        epoch as i64,
187    );
188
189    let mut conn = pg_reader
190        .connect()
191        .await
192        .context("Failed to connect to database")?;
193
194    let results: Vec<CpBounds> = conn
195        .results(query)
196        .await
197        .context("Failed to execute epoch checkpoint query")?;
198
199    let (cp_lo, cp_hi_inclusive) = match results.first() {
200        Some(bounds) => (bounds.cp_lo as u64, bounds.cp_hi_inclusive as u64),
201        None => return Ok(vec![]),
202    };
203
204    Ok(cp_unfiltered(&(cp_lo..=cp_hi_inclusive), page))
205}

207/// Determines the maximum value in an arbitrary number of Option<impl Ord>.
208fn max_option<T: Ord>(xs: impl IntoIterator<Item = Option<T>>) -> Option<T> {
209    xs.into_iter().flatten().max()
210}
211
212/// Determines the minimum value in an arbitrary number of Option<impl Ord>.
213fn min_option<T: Ord>(xs: impl IntoIterator<Item = Option<T>>) -> Option<T> {
214    xs.into_iter().flatten().min()
215}
