18// Filter for checkpoint-based queries across checkpoints, packages, and epochs.
19#[derive(InputObject, Debug, Default, Clone)]
20pub(crate) struct CheckpointFilter {
21    /// Limit query results to checkpoints at this epoch.
22    pub at_epoch: Option<UInt53>,
23
24    /// Limit query results to checkpoints that occured strictly after the given checkpoint.
25    pub after_checkpoint: Option<UInt53>,
26
27    /// Limit query results to checkpoints that occured at the given checkpoint.
28    pub at_checkpoint: Option<UInt53>,
29
30    /// Limit query results to checkpoints that occured strictly before the given checkpoint.
31    pub before_checkpoint: Option<UInt53>,
32}
33
34#[derive(QueryableByName, Debug)]
35pub(crate) struct CpBounds {
36    #[diesel(sql_type = BigInt, column_name = "cp_lo")]
37    cp_lo: i64,
38    #[diesel(sql_type = BigInt, column_name = "cp_hi_inclusive")]
39    cp_hi_inclusive: i64,
40}
41
42impl CheckpointFilter {
43    pub(crate) fn intersect(self, other: Self) -> Option<Self> {
44        macro_rules! intersect {
45            ($field:ident, $body:expr) => {
46                intersect::field(self.$field, other.$field, $body)
47            };
48        }
49
50        Some(Self {
51            at_epoch: intersect!(at_epoch, intersect::by_eq)?,
52            after_checkpoint: intersect!(after_checkpoint, intersect::by_max)?,
53            at_checkpoint: intersect!(at_checkpoint, intersect::by_eq)?,
54            before_checkpoint: intersect!(before_checkpoint, intersect::by_min)?,
55        })
56    }
57}

59/// The bounds on checkpoint sequence number, imposed by filters. The outermost bounds are
60/// determined by the lowest checkpoint that is safe to read from (reader_lo) and the highest
61/// checkpoint that has been processed based on the context's watermark(checkpoint_viewed_at).
62///
63/// ```ignore
64///     reader_lo                                                     checkpoint_viewed_at
65///     [-----------------------------------------------------------------]
66/// ```
67///
68/// The bounds are further tightened by the filters if they are present.
69///
70/// ```ignore
71///    filter.after_checkpoint                         filter.before_checkpoint
72///         [------------[--------------------------]------------]
73///                         filter.at_checkpoint
74/// ```
75///
76pub(crate) fn checkpoint_bounds(
77    cp_after: Option<u64>,
78    cp_at: Option<u64>,
79    cp_before: Option<u64>,
80    reader_lo: u64,
81    checkpoint_viewed_at: u64,
82) -> Option<RangeInclusive<u64>> {
83    let cp_after_inclusive = match cp_after.map(|x| x.checked_add(1)) {
84        Some(Some(after)) => Some(after),
85        Some(None) => return None,
86        None => None,
87    };
88    // Inclusive checkpoint sequence number lower bound. If there are no lower bound filters,
89    // we will use the smallest checkpoint available from the database, retrieved from
90    // the watermark.
91    //
92    // SAFETY: we can unwrap because of `Some(reader_lo)`
93    let cp_lo = max_option([cp_after_inclusive, cp_at, Some(reader_lo)]).unwrap();
94
95    let cp_before_inclusive = match cp_before.map(|x| x.checked_sub(1)) {
96        Some(Some(before)) => Some(before),
97        Some(None) => return None,
98        None => None,
99    };
100
101    // Inclusive checkpoint sequence upper bound. If there are no upper bound filters,
102    // we will use `checkpoint_viewed_at`.
103    //
104    // SAFETY: we can unwrap because of the `Some(checkpoint_viewed_at)`
105    let cp_hi = min_option([cp_before_inclusive, cp_at, Some(checkpoint_viewed_at)]).unwrap();
106
107    cp_hi
108        .checked_sub(cp_lo)
109        .map(|_| RangeInclusive::new(cp_lo, cp_hi))
110}

112/// The cp_sequence_numbers within checkpoint bounds with cursors applied inclusively.
113///
114/// pg_lo: The maximum of the cursor and the start of the checkpoint bound.
115/// pg_hi_inclusive: The minimum of the cursor and the end of the checkpoint bound.
116///
117pub(super) fn cp_unfiltered(cp_bounds: &RangeInclusive<u64>, page: &Page<CCheckpoint>) -> Vec<u64> {
118    let cp_lo = *cp_bounds.start();
119    let cp_hi = *cp_bounds.end();
120
121    // Inclusive cursor bounds
122    let pg_lo = page.after().map_or(cp_lo, |cursor| cursor.max(cp_lo));
123    let pg_hi_inclusive = page.before().map_or(cp_hi, |cursor| cursor.min(cp_hi));
124
125    if page.is_from_front() {
126        (pg_lo..=pg_hi_inclusive)
127            .take(page.limit_with_overhead())
128            .collect()
129    } else {
130        // Graphql last syntax expects results to be in ascending order. If we are paginating backwards,
131        // we reverse the results after applying limits.
132        let mut results: Vec<_> = (pg_lo..=pg_hi_inclusive)
133            .rev()
134            .take(page.limit_with_overhead())
135            .collect();
136        results.reverse();
137        results
138    }
139}

141/// The checkpoint sequence numbers in a range bounded by checkpoints in an epoch.
142/// The range is further tightened by bounds derived from filters if they are present.
143pub(super) async fn cp_by_epoch(
144    ctx: &Context<'_>,
145    page: &Page<CCheckpoint>,
146    cp_bounds: &RangeInclusive<u64>,
147    epoch: u64,
148) -> Result<Vec<u64>, RpcError> {
149    let pg_reader: &PgReader = ctx.data()?;
150
151    let query = query!(
152        r#"
153        SELECT
154            GREATEST(es.cp_lo, {BigInt}) AS "cp_lo",
155            LEAST(COALESCE(ee.cp_hi - 1, {BigInt}), {BigInt}) AS "cp_hi_inclusive"
156        FROM
157            kv_epoch_starts es
158        LEFT JOIN kv_epoch_ends ee ON es.epoch = ee.epoch
159        WHERE
160            es.epoch = {BigInt}
161        "#,
162        *cp_bounds.start() as i64,
163        *cp_bounds.end() as i64,
164        *cp_bounds.end() as i64,
165        epoch as i64,
166    );
167
168    let mut conn = pg_reader
169        .connect()
170        .await
171        .context("Failed to connect to database")?;
172
173    let results: Vec<CpBounds> = conn
174        .results(query)
175        .await
176        .context("Failed to execute epoch checkpoint query")?;
177
178    let (cp_lo, cp_hi_inclusive) = match results.first() {
179        Some(bounds) => (bounds.cp_lo as u64, bounds.cp_hi_inclusive as u64),
180        None => return Ok(vec![]),
181    };
182
183    Ok(cp_unfiltered(&(cp_lo..=cp_hi_inclusive), page))
184}

186/// Determines the maximum value in an arbitrary number of Option<impl Ord>.
187fn max_option<T: Ord>(xs: impl IntoIterator<Item = Option<T>>) -> Option<T> {
188    xs.into_iter().flatten().max()
189}
190
191/// Determines the minimum value in an arbitrary number of Option<impl Ord>.
192fn min_option<T: Ord>(xs: impl IntoIterator<Item = Option<T>>) -> Option<T> {
193    xs.into_iter().flatten().min()
194}
