62/// Checkpoints contain finalized transactions and are used for node synchronization and global transaction ordering.
63#[Object]
64impl Checkpoint {
65    /// The checkpoint's globally unique identifier, which can be passed to `Query.node` to refetch it.
66    pub(crate) async fn id(&self) -> Id {
67        Id::Checkpoint(self.sequence_number)
68    }
69
70    /// The checkpoint's position in the total order of finalized checkpoints, agreed upon by consensus.
71    async fn sequence_number(&self) -> UInt53 {
72        self.sequence_number.into()
73    }
74
75    /// Query the RPC as if this checkpoint were the latest checkpoint.
76    async fn query(&self, ctx: &Context<'_>) -> Option<Result<Query, RpcError>> {
77        async {
78            let scope = Some(
79                self.scope
80                    .with_checkpoint_viewed_at(ctx, self.sequence_number)
81                    .context("Checkpoint in the future")?,
82            );
83
84            Ok(Some(Query { scope }))
85        }
86        .await
87        .transpose()
88    }
89
90    #[graphql(flatten)]
91    async fn contents(&self, ctx: &Context<'_>) -> Result<CheckpointContents, RpcError> {
92        CheckpointContents::fetch(ctx, self.scope.clone(), self.sequence_number).await
93    }
94}

96#[Object]
97impl CheckpointContents {
98    /// A commitment by the committee at each checkpoint on the artifacts of the checkpoint.
99    /// e.g., object checkpoint states
100    async fn artifacts_digest(&self) -> Option<Result<String, RpcError>> {
101        let (summary, _, _) = self.contents.as_ref()?;
102
103        for commitment in &summary.checkpoint_commitments {
104            if let CheckpointCommitment::CheckpointArtifactsDigest(digest) = commitment {
105                return Some(Ok(digest.base58_encode()));
106            }
107        }
108
109        None
110    }
111
112    /// A 32-byte hash that uniquely identifies the checkpoint, encoded in Base58. This is a hash of the checkpoint's summary.
113    async fn digest(&self) -> Option<Result<String, RpcError>> {
114        let (summary, _, _) = self.contents.as_ref()?;
115        Some(Ok(summary.digest().base58_encode()))
116    }
117
118    /// A 32-byte hash that uniquely identifies the checkpoint's content, encoded in Base58.
119    async fn content_digest(&self) -> Option<Result<String, RpcError>> {
120        let (summary, _, _) = self.contents.as_ref()?;
121        Some(Ok(summary.content_digest.base58_encode()))
122    }
123
124    /// The epoch that this checkpoint is part of.
125    async fn epoch(&self) -> Option<Epoch> {
126        let (summary, _, _) = self.contents.as_ref()?;
127        Some(Epoch::with_id(self.scope.clone(), summary.epoch))
128    }
129
130    /// The total number of transactions in the network by the end of this checkpoint.
131    async fn network_total_transactions(&self) -> Option<UInt53> {
132        let (summary, _, _) = self.contents.as_ref()?;
133        Some(summary.network_total_transactions.into())
134    }
135
136    /// The digest of the previous checkpoint's summary.
137    async fn previous_checkpoint_digest(&self) -> Option<Result<String, RpcError>> {
138        let (summary, _, _) = self.contents.as_ref()?;
139        Some(Ok(summary.previous_digest.as_ref()?.base58_encode()))
140    }
141
142    /// The computation cost, storage cost, storage rebate, and non-refundable storage fee accumulated during this epoch, up to and including this checkpoint. These values increase monotonically across checkpoints in the same epoch, and reset on epoch boundaries.
143    async fn rolling_gas_summary(&self) -> Option<GasCostSummary> {
144        let (summary, _, _) = self.contents.as_ref()?;
145        Some(GasCostSummary::from(
146            summary.epoch_rolling_gas_cost_summary.clone(),
147        ))
148    }
149
150    /// The Base64 serialized BCS bytes of this checkpoint's summary.
151    async fn summary_bcs(&self) -> Option<Result<Base64, RpcError>> {
152        async {
153            let Some((summary, _, _)) = &self.contents else {
154                return Ok(None);
155            };
156            Ok(Some(Base64::from(
157                bcs::to_bytes(summary).context("Failed to serialize checkpoint summary")?,
158            )))
159        }
160        .await
161        .transpose()
162    }
163
164    /// The Base64 serialized BCS bytes of this checkpoint's contents.
165    async fn content_bcs(&self) -> Option<Result<Base64, RpcError>> {
166        async {
167            let Some((_, content, _)) = &self.contents else {
168                return Ok(None);
169            };
170            Ok(Some(Base64::from(
171                bcs::to_bytes(content).context("Failed to serialize checkpoint content")?,
172            )))
173        }
174        .await
175        .transpose()
176    }
177
178    /// The timestamp at which the checkpoint is agreed to have happened according to consensus. Transactions that access time in this checkpoint will observe this timestamp.
179    async fn timestamp(&self) -> Option<Result<DateTime, RpcError>> {
180        async {
181            let Some((summary, _, _)) = &self.contents else {
182                return Ok(None);
183            };
184
185            Ok(Some(DateTime::from_ms(summary.timestamp_ms as i64)?))
186        }
187        .await
188        .transpose()
189    }
190
191    /// The aggregation of signatures from a quorum of validators for the checkpoint proposal.
192    async fn validator_signatures(&self) -> Option<Result<ValidatorAggregatedSignature, RpcError>> {
193        let (_, _, authority_info) = self.contents.as_ref()?;
194        Some(Ok(ValidatorAggregatedSignature::with_authority_info(
195            self.scope.clone(),
196            authority_info.clone(),
197        )))
198    }
199
200    // The transactions in this checkpoint.
201    async fn transactions(
202        &self,
203        ctx: &Context<'_>,
204        first: Option<u64>,
205        after: Option<CTransaction>,
206        last: Option<u64>,
207        before: Option<CTransaction>,
208        #[graphql(validator(custom = "TFValidator"))] filter: Option<TransactionFilter>,
209    ) -> Option<Result<Connection<String, Transaction>, RpcError>> {
210        async {
211            let Some((summary, _, _)) = &self.contents else {
212                return Ok(None);
213            };
214            let pagination: &PaginationConfig = ctx.data()?;
215            let limits = pagination.limits("Checkpoint", "transactions");
216            let page = Page::from_params(limits, first, after, last, before)?;
217
218            let Some(filter) = filter.unwrap_or_default().intersect(TransactionFilter {
219                at_checkpoint: Some(UInt53::from(summary.sequence_number)),
220                ..Default::default()
221            }) else {
222                return Ok(Some(Connection::new(false, false)));
223            };
224
225            Ok(Some(
226                Transaction::paginate(ctx, self.scope.clone(), page, filter).await?,
227            ))
228        }
229        .await
230        .transpose()
231    }
232}

234impl Checkpoint {
235    /// Construct a checkpoint that is represented by just its identifier (its sequence number).
236    ///
237    /// If no sequence_number is provided, defaults to the scope's checkpoint.
238    /// Returns `None` if the checkpoint is set in the future relative to the current scope's
239    /// checkpoint, or when no checkpoint is set in scope (e.g. execution scope, where checkpoint
240    /// queries return None to prevent temporal inconsistency).
241    pub(crate) fn with_sequence_number(scope: Scope, sequence_number: Option<u64>) -> Option<Self> {
242        let scope_checkpoint = scope.checkpoint_viewed_at()?;
243        let sequence_number = sequence_number.unwrap_or(scope_checkpoint);
244
245        (sequence_number <= scope_checkpoint).then_some(Self {
246            scope,
247            sequence_number,
248        })
249    }
250
251    /// Paginate through checkpoints with filters applied.
252    ///
253    /// Returns empty results when no checkpoint is set in scope (e.g. execution scope).
254    pub(crate) async fn paginate(
255        ctx: &Context<'_>,
256        scope: Scope,
257        page: Page<CCheckpoint>,
258        filter: CheckpointFilter,
259    ) -> Result<Connection<String, Checkpoint>, RpcError> {
260        let watermarks: &Arc<Watermarks> = ctx.data()?;
261        let available_range_key = AvailableRangeKey {
262            type_: "Query".to_string(),
263            field: Some("checkpoints".to_string()),
264            filters: Some(filter.active_filters()),
265        };
266        let reader_lo = available_range_key.reader_lo(watermarks)?;
267
268        let Some(cp_hi_inclusive) = scope.checkpoint_viewed_at() else {
269            // In execution scope, checkpoint pagination returns empty results
270            return Ok(Connection::new(false, false));
271        };
272
273        let Some(cp_bounds) = checkpoint_bounds(
274            filter.after_checkpoint.map(u64::from),
275            filter.at_checkpoint.map(u64::from),
276            filter.before_checkpoint.map(u64::from),
277            reader_lo,
278            cp_hi_inclusive,
279        ) else {
280            return Ok(Connection::new(false, false));
281        };
282
283        let results = if let Some(epoch) = filter.at_epoch {
284            cp_by_epoch(ctx, &page, &cp_bounds, epoch.into()).await?
285        } else {
286            cp_unfiltered(&cp_bounds, &page)
287        };
288
289        page.paginate_results(
290            results,
291            |c| JsonCursor::new(*c),
292            |c| Ok(Self::with_sequence_number(scope.clone(), Some(c)).unwrap()),
293        )
294    }
295}

297impl CheckpointContents {
298    /// Attempt to fill the contents. If the contents are already filled, returns a clone,
299    /// otherwise attempts to fetch from the store. The resulting value may still have an empty
300    /// contents field, because it could not be found in the store.
301    pub(crate) async fn fetch(
302        ctx: &Context<'_>,
303        scope: Scope,
304        sequence_number: u64,
305    ) -> Result<Self, RpcError> {
306        let kv_loader: &KvLoader = ctx.data()?;
307        let contents = kv_loader
308            .load_one_checkpoint(sequence_number)
309            .await
310            .context("Failed to fetch checkpoint contents")?;
311
312        Ok(Self { scope, contents })
313    }
