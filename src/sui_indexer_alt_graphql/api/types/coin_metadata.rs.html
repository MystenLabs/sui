64/// Whether the currency is regulated or not.
65#[derive(Enum, Copy, Clone, Eq, PartialEq)]
66pub(crate) enum RegulatedState {
67    /// A `DenyCap` or a `RegulatedCoinMetadata` exists for this currency.
68    Regulated,
69
70    /// The currency was created without a deny list.
71    Unregulated,
72}
73
74#[derive(SimpleObject, Default)]
75pub(crate) struct RegulatedFields {
76    /// Whether the currency is regulated or not. `null` indicates that the regulatory status is unknown.
77    regulated_state: Option<RegulatedState>,
78
79    /// Whether the `DenyCap` can be used to enable a global pause that behaves as if all addresses were added to the deny list. `null` indicates that it is not known whether the currency can be paused or not. This field is only populated on currencies held in the Coin Registry. To determine whether a legacy currency can be paused, check the contents of its `DenyCap`, if it can be found.
80    allow_global_pause: Option<bool>,
81
82    /// If the currency is regulated, this object represents the capability to modify the deny list. If a capability is known but wrapped, its address can be fetched but other fields will not be accessible.
83    deny_cap: Option<MoveObject>,
84}

428    #[graphql(flatten)]
429    pub(crate) async fn regulated_fields(
430        &self,
431        ctx: &Context<'_>,
432    ) -> Result<RegulatedFields, RpcError<object::Error>> {
433        let Some(contents) = self.metadata_contents(ctx).await.map_err(upcast)? else {
434            return Ok(RegulatedFields::default());
435        };
436
437        let scope = &self.super_.super_.super_.scope;
438
439        // If the currency's metadata is stored in the Coin Registry, and its regulation state is
440        // known, return it.
441        match &contents.native {
442            NativeContents::Registry(NativeCurrency {
443                regulated: NativeRegulated::Unregulated,
444                ..
445            }) => {
446                return Ok(RegulatedFields {
447                    regulated_state: Some(RegulatedState::Unregulated),
448                    allow_global_pause: Some(false),
449                    deny_cap: None,
450                });
451            }
452
453            NativeContents::Registry(NativeCurrency {
454                regulated:
455                    NativeRegulated::Regulated {
456                        cap,
457                        allow_global_pause,
458                        variant: _,
459                    },
460                ..
461            }) => {
462                return Ok(RegulatedFields {
463                    regulated_state: Some(RegulatedState::Regulated),
464                    allow_global_pause: *allow_global_pause,
465                    deny_cap: Some(MoveObject::from_super(Object::with_address(
466                        scope.without_root_bound(),
467                        (*cap).into(),
468                    ))),
469                });
470            }
471
472            _ => {}
473        }
474
475        // ...otherwise fall back to looking up the RegulatedCoinMetadata singleton object (for
476        // coin registry currencies where the regulated state is not known and for legacy currencies).
477        let TypeTag::Struct(coin_type) = &contents.coin_type else {
478            return Ok(RegulatedFields::default());
479        };
480
481        if GAS::is_gas(coin_type.as_ref()) {
482            return Ok(RegulatedFields {
483                regulated_state: Some(RegulatedState::Unregulated),
484                allow_global_pause: Some(false),
485                deny_cap: None,
486            });
487        }
488
489        let type_ = RegulatedCoinMetadata::type_(*coin_type.clone());
490        let Some(object) = Object::singleton(ctx, scope.without_root_bound(), type_).await? else {
491            // If there is no RegulatedCoinMetadata object, the coin is unregulated.
492            return Ok(RegulatedFields {
493                regulated_state: Some(RegulatedState::Unregulated),
494                allow_global_pause: Some(false),
495                deny_cap: None,
496            });
497        };
498
499        let Some(contents) = object.contents(ctx).await.map_err(upcast)? else {
500            // We were able to find a regulated coin metadata object but couldn't load its
501            // contents -- it's definitely regulated, but we don't have more details.
502            return Ok(RegulatedFields {
503                regulated_state: Some(RegulatedState::Regulated),
504                allow_global_pause: None,
505                deny_cap: None,
506            });
507        };
508
509        let move_object = contents
510            .data
511            .try_as_move()
512            .context("Query by type returned a package")?;
513
514        let metadata: RegulatedCoinMetadata = bcs::from_bytes(move_object.contents())
515            .context("Failed to deserialize RegulatedCoinMetadata")?;
516
517        Ok(RegulatedFields {
518            regulated_state: Some(RegulatedState::Regulated),
519            allow_global_pause: None,
520            deny_cap: Some(MoveObject::from_super(Object::with_address(
521                scope.without_root_bound(),
522                metadata.deny_cap_object.bytes.into(),
523            ))),
524        })
525    }

527    #[graphql(flatten)]
528    pub(crate) async fn supply_fields(
529        &self,
530        ctx: &Context<'_>,
531    ) -> Result<SupplyFields, RpcError<object::Error>> {
532        let Some(contents) = self.metadata_contents(ctx).await.map_err(upcast)? else {
533            return Ok(SupplyFields::default());
534        };
535
536        // If the currency's metadata is stored in the Coin Registry, and its supply is known,
537        // return it.
538        match &contents.native {
539            NativeContents::Registry(NativeCurrency {
540                supply: Some(NativeSupply::Fixed(s)),
541                ..
542            }) => {
543                return Ok(SupplyFields {
544                    supply_state: Some(SupplyState::Fixed),
545                    supply: Some(BigInt::from(*s)),
546                });
547            }
548
549            NativeContents::Registry(NativeCurrency {
550                supply: Some(NativeSupply::BurnOnly(s)),
551                ..
552            }) => {
553                return Ok(SupplyFields {
554                    supply_state: Some(SupplyState::BurnOnly),
555                    supply: Some(BigInt::from(*s)),
556                });
557            }
558
559            _ => {}
560        }
561
562        // ...otherwise fall back to looking up the TreasuryCap singleton object (for coin registry
563        // currencies where the supply state is not known and for legacy currencies).
564        let TypeTag::Struct(coin_type) = &contents.coin_type else {
565            return Ok(SupplyFields::default());
566        };
567
568        if GAS::is_gas(coin_type.as_ref()) {
569            return Ok(SupplyFields {
570                supply_state: Some(SupplyState::Fixed),
571                supply: Some(BigInt::from(TOTAL_SUPPLY_MIST)),
572            });
573        }
574
575        let type_ = TreasuryCap::type_(*coin_type.clone());
576        let scope = self.super_.super_.super_.scope.without_root_bound();
577        let Some(object) = Object::singleton(ctx, scope, type_).await? else {
578            return Ok(SupplyFields::default());
579        };
580
581        let Some(contents) = object.contents(ctx).await.map_err(upcast)? else {
582            return Ok(SupplyFields::default());
583        };
584
585        let move_object = contents
586            .data
587            .try_as_move()
588            .context("Query by type returned a package")?;
589
590        let treasury_cap: TreasuryCap =
591            bcs::from_bytes(move_object.contents()).context("Failed to deserialize TreasuryCap")?;
592
593        // Treat the supply as fixed if the TreasuryCap is immutable, or owned by the zero address.
594        let supply_state = if matches!(
595            contents.owner(),
596            NativeOwner::Immutable
597                | NativeOwner::AddressOwner(NativeAddress::ZERO)
598                | NativeOwner::ConsensusAddressOwner {
599                    owner: NativeAddress::ZERO,
600                    ..
601                }
602        ) {
603            Some(SupplyState::Fixed)
604        } else {
605            None
606        };
607
608        Ok(SupplyFields {
609            supply_state,
610            supply: Some(BigInt::from(treasury_cap.total_supply.value)),
611        })
612    }

637    /// Create a CoinMetadata from a `MoveObject`, after checking whether it is a CoinMetadata.
638    pub(crate) async fn from_move_object(
639        move_object: &MoveObject,
640        ctx: &Context<'_>,
641    ) -> Result<Option<Self>, RpcError> {
642        let Some(native) = move_object.native(ctx).await?.as_ref() else {
643            return Ok(None);
644        };
645
646        if !native.type_().is_coin_metadata() && !native.type_().is_currency() {
647            return Ok(None);
648        }
649
650        Ok(Some(Self::from_super(move_object.clone())))
651    }

653    /// Find a CoinMetadata object by the coin type it represents.
654    ///
655    /// First checks if the currency is represented in the Coin Registry, and otherwise looks for a
656    /// live singleton `CoinMetadata` object.
657    ///
658    /// Returns `None` if the currency is not in the Coin Registry and there is no live
659    /// `CoinMetadata` object for the given coin type (it may have been deleted, wrapped, or never
660    /// created).
661    pub(crate) async fn by_coin_type(
662        ctx: &Context<'_>,
663        scope: Scope,
664        coin_type: TypeTag,
665    ) -> Result<Option<Self>, RpcError<object::Error>> {
666        let currency_id: SuiAddress = NativeCurrency::derive_object_id(coin_type.clone())
667            .context("Failed to derive Currency ID")?
668            .into();
669
670        if let Some(object) = Object::latest(ctx, scope.clone(), currency_id)
671            .await
672            .map_err(upcast)?
673        {
674            return Ok(Some(Self::from_super(MoveObject::from_super(object))));
675        }
676
677        let type_ = StructTag {
678            address: SUI_FRAMEWORK_ADDRESS,
679            module: COIN_MODULE_NAME.to_owned(),
680            name: COIN_METADATA_STRUCT_NAME.to_owned(),
681            type_params: vec![coin_type],
682        };
683
684        Ok(Object::singleton(ctx, scope, type_)
685            .await?
686            .map(|obj| Self::from_super(MoveObject::from_super(obj))))
687    }

689    /// Get the native CoinMetadata data, loading it lazily if needed.
690    async fn metadata_contents(
691        &self,
692        ctx: &Context<'_>,
693    ) -> Result<&Option<MetadataContents>, RpcError> {
694        self.contents
695            .get_or_try_init(async || {
696                let Some(native_move) = self.super_.native(ctx).await?.as_ref() else {
697                    return Ok(None);
698                };
699
700                let coin_type = || {
701                    StructTag::from(native_move.type_().clone())
702                        .type_params
703                        .pop()
704                        .context("No coin type parameter")
705                };
706
707                Ok(Some(if native_move.type_().is_currency() {
708                    MetadataContents {
709                        coin_type: coin_type()?,
710                        native: NativeContents::Registry(
711                            bcs::from_bytes(native_move.contents())
712                                .context("Failed to deserialize Currency")?,
713                        ),
714                    }
715                } else if native_move.type_().is_coin_metadata() {
716                    MetadataContents {
717                        coin_type: coin_type()?,
718                        native: NativeContents::Metadata(
719                            bcs::from_bytes(native_move.contents())
720                                .context("Failed to deserialize CoinMetadata")?,
721                        ),
722                    }
723                } else {
724                    return Err(anyhow!("Not a CoinMetadata").into());
725                }))
726            })
727            .await
728    }
