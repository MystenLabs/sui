48/// Whether the currency is regulated or not.
49#[derive(Enum, Copy, Clone, Eq, PartialEq)]
50pub(crate) enum RegulatedState {
51    /// A `DenyCap` or a `RegulatedCoinMetadata` exists for this currency.
52    Regulated,
53
54    /// The currency was created without a deny list.
55    Unregulated,
56}
57
58#[derive(SimpleObject, Default)]
59pub(crate) struct RegulatedFields {
60    /// Whether the currency is regulated or not. `null` indicates that the regulatory status is unknown.
61    regulated_state: Option<RegulatedState>,
62
63    /// Whether the `DenyCap` can be used to enable a global pause that behaves as if all addresses were added to the deny list. `null` indicates that it is not known whether the currency can be paused or not. This field is only populated on currencies held in the Coin Registry. To determine whether a legacy currency can be paused, check the contents of its `DenyCap`, if it can be found.
64    allow_global_pause: Option<bool>,
65
66    /// If the currency is regulated, this object represents the capability to modify the deny list. If a capability is known but wrapped, its address can be fetched but other fields will not be accessible.
67    deny_cap: Option<MoveObject>,
68}

389    #[graphql(flatten)]
390    pub(crate) async fn regulated_fields(
391        &self,
392        ctx: &Context<'_>,
393    ) -> Result<RegulatedFields, RpcError<object::Error>> {
394        let Some(contents) = self.metadata_contents(ctx).await.map_err(upcast)? else {
395            return Ok(RegulatedFields::default());
396        };
397
398        let scope = &self.super_.super_.super_.scope;
399
400        // If the currency's metadata is stored in the Coin Registry, and its regulation state is
401        // known, return it.
402        match &contents.native {
403            NativeContents::Registry(NativeCurrency {
404                regulated: NativeRegulated::Unregulated,
405                ..
406            }) => {
407                return Ok(RegulatedFields {
408                    regulated_state: Some(RegulatedState::Unregulated),
409                    allow_global_pause: Some(false),
410                    deny_cap: None,
411                });
412            }
413
414            NativeContents::Registry(NativeCurrency {
415                regulated:
416                    NativeRegulated::Regulated {
417                        cap,
418                        allow_global_pause,
419                        variant: _,
420                    },
421                ..
422            }) => {
423                return Ok(RegulatedFields {
424                    regulated_state: Some(RegulatedState::Regulated),
425                    allow_global_pause: *allow_global_pause,
426                    deny_cap: Some(MoveObject::from_super(Object::with_address(
427                        scope.without_root_version(),
428                        (*cap).into(),
429                    ))),
430                });
431            }
432
433            _ => {}
434        }
435
436        // ...otherwise fall back to looking up the RegulatedCoinMetadata singleton object (for
437        // coin registry currencies where the regulated state is not known and for legacy currencies).
438        let TypeTag::Struct(coin_type) = &contents.coin_type else {
439            return Ok(RegulatedFields::default());
440        };
441
442        if GAS::is_gas(coin_type.as_ref()) {
443            return Ok(RegulatedFields {
444                regulated_state: Some(RegulatedState::Unregulated),
445                allow_global_pause: Some(false),
446                deny_cap: None,
447            });
448        }
449
450        let type_ = RegulatedCoinMetadata::type_(*coin_type.clone());
451        let Some(object) = Object::singleton(ctx, scope.without_root_version(), type_).await?
452        else {
453            // If there is no RegulatedCoinMetadata object, the coin is unregulated.
454            return Ok(RegulatedFields {
455                regulated_state: Some(RegulatedState::Unregulated),
456                allow_global_pause: Some(false),
457                deny_cap: None,
458            });
459        };
460
461        let Some(contents) = object.contents(ctx).await.map_err(upcast)? else {
462            // We were able to find a regulated coin metadata object but couldn't load its
463            // contents -- it's definitely regulated, but we don't have more details.
464            return Ok(RegulatedFields {
465                regulated_state: Some(RegulatedState::Regulated),
466                allow_global_pause: None,
467                deny_cap: None,
468            });
469        };
470
471        let move_object = contents
472            .data
473            .try_as_move()
474            .context("Query by type returned a package")?;
475
476        let metadata: RegulatedCoinMetadata = bcs::from_bytes(move_object.contents())
477            .context("Failed to deserialize RegulatedCoinMetadata")?;
478
479        Ok(RegulatedFields {
480            regulated_state: Some(RegulatedState::Regulated),
481            allow_global_pause: None,
482            deny_cap: Some(MoveObject::from_super(Object::with_address(
483                scope.without_root_version(),
484                metadata.deny_cap_object.bytes.into(),
485            ))),
486        })
487    }

489    #[graphql(flatten)]
490    pub(crate) async fn supply_fields(
491        &self,
492        ctx: &Context<'_>,
493    ) -> Result<SupplyFields, RpcError<object::Error>> {
494        let Some(contents) = self.metadata_contents(ctx).await.map_err(upcast)? else {
495            return Ok(SupplyFields::default());
496        };
497
498        // If the currency's metadata is stored in the Coin Registry, and its supply is known,
499        // return it.
500        match &contents.native {
501            NativeContents::Registry(NativeCurrency {
502                supply: Some(NativeSupply::Fixed(s)),
503                ..
504            }) => {
505                return Ok(SupplyFields {
506                    supply_state: Some(SupplyState::Fixed),
507                    supply: Some(BigInt::from(*s)),
508                });
509            }
510
511            NativeContents::Registry(NativeCurrency {
512                supply: Some(NativeSupply::BurnOnly(s)),
513                ..
514            }) => {
515                return Ok(SupplyFields {
516                    supply_state: Some(SupplyState::BurnOnly),
517                    supply: Some(BigInt::from(*s)),
518                });
519            }
520
521            _ => {}
522        }
523
524        // ...otherwise fall back to looking up the TreasuryCap singleton object (for coin registry
525        // currencies where the supply state is not known and for legacy currencies).
526        let TypeTag::Struct(coin_type) = &contents.coin_type else {
527            return Ok(SupplyFields::default());
528        };
529
530        if GAS::is_gas(coin_type.as_ref()) {
531            return Ok(SupplyFields {
532                supply_state: Some(SupplyState::Fixed),
533                supply: Some(BigInt::from(TOTAL_SUPPLY_MIST)),
534            });
535        }
536
537        let type_ = TreasuryCap::type_(*coin_type.clone());
538        let scope = self.super_.super_.super_.scope.without_root_version();
539        let Some(object) = Object::singleton(ctx, scope, type_).await? else {
540            return Ok(SupplyFields::default());
541        };
542
543        let Some(contents) = object.contents(ctx).await.map_err(upcast)? else {
544            return Ok(SupplyFields::default());
545        };
546
547        let move_object = contents
548            .data
549            .try_as_move()
550            .context("Query by type returned a package")?;
551
552        let treasury_cap: TreasuryCap =
553            bcs::from_bytes(move_object.contents()).context("Failed to deserialize TreasuryCap")?;
554
555        // Treat the supply as fixed if the TreasuryCap is immutable, or owned by the zero address.
556        let supply_state = if matches!(
557            contents.owner(),
558            NativeOwner::Immutable
559                | NativeOwner::AddressOwner(NativeAddress::ZERO)
560                | NativeOwner::ConsensusAddressOwner {
561                    owner: NativeAddress::ZERO,
562                    ..
563                }
564        ) {
565            Some(SupplyState::Fixed)
566        } else {
567            None
568        };
569
570        Ok(SupplyFields {
571            supply_state,
572            supply: Some(BigInt::from(treasury_cap.total_supply.value)),
573        })
574    }

599    /// Create a CoinMetadata from a `MoveObject`, after checking whether it is a CoinMetadata.
600    pub(crate) async fn from_move_object(
601        move_object: &MoveObject,
602        ctx: &Context<'_>,
603    ) -> Result<Option<Self>, RpcError> {
604        let Some(native) = move_object.native(ctx).await?.as_ref() else {
605            return Ok(None);
606        };
607
608        if !native.type_().is_coin_metadata() && !native.type_().is_currency() {
609            return Ok(None);
610        }
611
612        Ok(Some(Self::from_super(move_object.clone())))
613    }

615    /// Find a CoinMetadata object by the coin type it represents.
616    ///
617    /// First checks if the currency is represented in the Coin Registry, and otherwise looks for a
618    /// live singleton `CoinMetadata` object.
619    ///
620    /// Returns `None` if the currency is not in the Coin Registry and there is no live
621    /// `CoinMetadata` object for the given coin type (it may have been deleted, wrapped, or never
622    /// created).
623    pub(crate) async fn by_coin_type(
624        ctx: &Context<'_>,
625        scope: Scope,
626        coin_type: TypeTag,
627    ) -> Result<Option<Self>, RpcError<object::Error>> {
628        let currency_id: SuiAddress = NativeCurrency::derive_object_id(coin_type.clone())
629            .context("Failed to derive Currency ID")?
630            .into();
631
632        if let Some(object) = Object::latest(ctx, scope.clone(), currency_id)
633            .await
634            .map_err(upcast)?
635        {
636            return Ok(Some(Self::from_super(MoveObject::from_super(object))));
637        }
638
639        let type_ = StructTag {
640            address: SUI_FRAMEWORK_ADDRESS,
641            module: COIN_MODULE_NAME.to_owned(),
642            name: COIN_METADATA_STRUCT_NAME.to_owned(),
643            type_params: vec![coin_type],
644        };
645
646        Ok(Object::singleton(ctx, scope, type_)
647            .await?
648            .map(|obj| Self::from_super(MoveObject::from_super(obj))))
649    }

651    /// Get the native CoinMetadata data, loading it lazily if needed.
652    async fn metadata_contents(
653        &self,
654        ctx: &Context<'_>,
655    ) -> Result<&Option<MetadataContents>, RpcError> {
656        self.contents
657            .get_or_try_init(async || {
658                let Some(native_move) = self.super_.native(ctx).await?.as_ref() else {
659                    return Ok(None);
660                };
661
662                let coin_type = || {
663                    StructTag::from(native_move.type_().clone())
664                        .type_params
665                        .pop()
666                        .context("No coin type parameter")
667                };
668
669                Ok(Some(if native_move.type_().is_currency() {
670                    MetadataContents {
671                        coin_type: coin_type()?,
672                        native: NativeContents::Registry(
673                            bcs::from_bytes(native_move.contents())
674                                .context("Failed to deserialize Currency")?,
675                        ),
676                    }
677                } else if native_move.type_().is_coin_metadata() {
678                    MetadataContents {
679                        coin_type: coin_type()?,
680                        native: NativeContents::Metadata(
681                            bcs::from_bytes(native_move.contents())
682                                .context("Failed to deserialize CoinMetadata")?,
683                        ),
684                    }
685                } else {
686                    return Err(anyhow!("Not a CoinMetadata").into());
687                }))
688            })
689            .await
690    }
691}
