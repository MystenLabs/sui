4use async_graphql::SimpleObject;
5use sui_rpc::proto::sui::rpc::v2 as proto;
6
7use crate::{
8    api::types::{
9        move_type::MoveType, move_value::MoveValue,
10        transaction_kind::programmable::commands::TransactionArgument,
11    },
12    scope::Scope,
13};
14
15/// The intermediate results for each command of a transaction simulation.
16#[derive(Clone, SimpleObject)]
17pub struct CommandResult {
18    /// Return results of each command in this transaction.
19    pub return_values: Option<Vec<CommandOutput>>,
20    /// Changes made to arguments that were mutably borrowed by each command in this transaction.
21    pub mutated_references: Option<Vec<CommandOutput>>,
22}
23
24/// A value produced or modified during command execution.
25///
26/// This can represent either a return value from a command or an argument that was mutated by reference.
27#[derive(Clone, SimpleObject)]
28pub struct CommandOutput {
29    /// The transaction argument that this value corresponds to (if any).
30    pub argument: Option<TransactionArgument>,
31    /// The structured Move value, if available.
32    pub value: Option<MoveValue>,
33}
34
35impl CommandResult {
36    pub(crate) fn from_proto(result: proto::CommandResult, scope: Scope) -> Self {
37        Self {
38            return_values: Some(
39                result
40                    .return_values
41                    .into_iter()
42                    .map(|output| CommandOutput::from_proto(output, scope.clone()))
43                    .collect(),
44            ),
45            mutated_references: Some(
46                result
47                    .mutated_by_ref
48                    .into_iter()
49                    .map(|output| CommandOutput::from_proto(output, scope.clone()))
50                    .collect(),
51            ),
52        }
53    }
54}
55
56impl CommandOutput {
57    pub(crate) fn from_proto(output: proto::CommandOutput, scope: Scope) -> Self {
58        Self {
59            argument: output
60                .argument
61                .and_then(|arg| TransactionArgument::try_from(arg).ok()),
62            value: output.value.and_then(|bcs| {
63                let type_name = bcs.name?;
64                let value_bytes = bcs.value?.to_vec();
65
66                // Parse the type name string into a MoveType
67                let type_tag = type_name.parse().ok()?;
68                let move_type = MoveType::from_native(type_tag, scope.clone());
69
70                Some(MoveValue::new(move_type, value_bytes))
71            }),
72        }
73    }
74}
