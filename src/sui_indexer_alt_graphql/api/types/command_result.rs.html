4use async_graphql::SimpleObject;
5use sui_rpc::proto::sui::rpc::v2 as proto;
6
7use crate::api::types::move_type::MoveType;
8use crate::api::types::move_value::MoveValue;
9use crate::api::types::transaction_kind::programmable::commands::TransactionArgument;
10use crate::scope::Scope;
11
12/// The intermediate results for each command of a transaction simulation.
13#[derive(Clone, SimpleObject)]
14pub struct CommandResult {
15    /// Return results of each command in this transaction.
16    pub return_values: Option<Vec<CommandOutput>>,
17    /// Changes made to arguments that were mutably borrowed by each command in this transaction.
18    pub mutated_references: Option<Vec<CommandOutput>>,
19}
20
21/// A value produced or modified during command execution.
22///
23/// This can represent either a return value from a command or an argument that was mutated by reference.
24#[derive(Clone, SimpleObject)]
25pub struct CommandOutput {
26    /// The transaction argument that this value corresponds to (if any).
27    pub argument: Option<TransactionArgument>,
28    /// The structured Move value, if available.
29    pub value: Option<MoveValue>,
30}
31
32impl CommandResult {
33    pub(crate) fn from_proto(result: proto::CommandResult, scope: Scope) -> Self {
34        Self {
35            return_values: Some(
36                result
37                    .return_values
38                    .into_iter()
39                    .map(|output| CommandOutput::from_proto(output, scope.clone()))
40                    .collect(),
41            ),
42            mutated_references: Some(
43                result
44                    .mutated_by_ref
45                    .into_iter()
46                    .map(|output| CommandOutput::from_proto(output, scope.clone()))
47                    .collect(),
48            ),
49        }
50    }
51}
52
53impl CommandOutput {
54    pub(crate) fn from_proto(output: proto::CommandOutput, scope: Scope) -> Self {
55        Self {
56            argument: output
57                .argument
58                .and_then(|arg| TransactionArgument::try_from(arg).ok()),
59            value: output.value.and_then(|bcs| {
60                let type_name = bcs.name?;
61                let value_bytes = bcs.value?.to_vec();
62
63                // Parse the type name string into a MoveType
64                let type_tag = type_name.parse().ok()?;
65                let move_type = MoveType::from_native(type_tag, scope.clone());
66
67                Some(MoveValue::new(move_type, value_bytes))
68            }),
69        }
70    }
71}
