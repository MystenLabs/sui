128/// Dynamic fields are heterogenous fields that can be added or removed from an object at runtime. Their names are arbitrary Move values that have `copy`, `drop`, and `store`.
129///
130/// There are two sub-types of dynamic fields:
131///
132/// - Dynamic fields can store any value that has `store`. Objects stored in this kind of field will be considered wrapped (not accessible via its ID by external tools like explorers, wallets, etc. accessing storage).
133/// - Dynamic object fields can only store objects (values that have the `key` ability, and an `id: UID` as its first field) that have `store`, but they will still be directly accessible off-chain via their ID after being attached as a field.
134#[Object]

143        }
144    }

425    /// The dynamic field's value, as a Move value for dynamic fields and as a MoveObject for dynamic object fields.
426    async fn value(&self, ctx: &Context<'_>) -> Option<Result<DynamicFieldValue, RpcError>> {
427        async {
428            let Some(native) = self.native(ctx).await? else {
429                return Ok(None);
430            };
431
432            if native.kind == DynamicFieldType::DynamicField {
433                return Ok(Some(DynamicFieldValue::MoveValue(MoveValue::new(
434                    MoveType::from_native(native.value_type.clone(), native.scope.clone()),
435                    native.value_bytes.clone(),
436                ))));
437            }
438
439            let address: SuiAddress = bcs::from_bytes(&native.value_bytes)
440                .context("Failed to deserialize dynamic object field ID")?;
441
442            let object = Object::latest(ctx, native.scope.clone(), address).await?;
443
444            let Some(object) = object else {
445                return Ok(None);
446            };
447
448            Ok(Some(DynamicFieldValue::MoveObject(MoveObject::from_super(
449                object,
450            ))))
451        }
452        .await
453        .transpose()
454    }

479    /// Create a dynamic field from a `MoveObject`, after checking whether it is a dynamic field.
480    pub(crate) async fn from_move_object(
481        move_object: &MoveObject,
482        ctx: &Context<'_>,
483    ) -> Result<Option<Self>, RpcError> {
484        let Some(native) = move_object.native(ctx).await?.as_ref() else {
485            return Ok(None);
486        };

489            return Ok(None);
490        }

492        Ok(Some(Self::from_super(move_object.clone())))

495    /// Look up a dynamic field owned by `parent`, with the given `kind` and `name`. `scope`
496    /// includes checkpoint and version bounds that should be applied to the lookup.
497    pub(crate) async fn by_name(
498        ctx: &Context<'_>,
499        scope: Scope,
500        parent: SuiAddress,
501        kind: DynamicFieldType,
502        name: DynamicFieldName,
503    ) -> Result<Option<Self>, RpcError<Error>> {
504        match name {
505            DynamicFieldName {
506                type_: Some(type_),
507                bcs: Some(bcs),
508                literal: None,
509            } => Self::by_serialized_name(ctx, scope, parent, kind, type_, bcs)
510                .await
511                .map_err(upcast),
512
513            DynamicFieldName {
514                literal: Some(literal),
515                type_: None,
516                bcs: None,
517            } => Self::by_literal_name(ctx, scope, parent, kind, literal).await,
518
519            _ => Err(bad_user_input(Error::NameInput)),
520        }
521    }

523    /// Look up a dynamic field by its serialized name (type and BCS bytes).
524    pub(crate) async fn by_serialized_name(
525        ctx: &Context<'_>,
526        scope: Scope,
527        parent: SuiAddress,
528        kind: DynamicFieldType,
529        type_: TypeInput,
530        bcs: Base64,
531    ) -> Result<Option<Self>, RpcError> {
532        use DynamicFieldType as DFT;
533
534        let type_ = match kind {
535            DFT::DynamicField => type_.0,
536            DFT::DynamicObject => DynamicFieldInfo::dynamic_object_field_wrapper(type_.0).into(),
537        };
538
539        let field_id: SuiAddress = derive_dynamic_field_id(parent, &type_, &bcs.0)?.into();
540
541        let object = Object::latest(ctx, scope.clone(), field_id).await?;
542
543        let Some(object) = object else {
544            return Ok(None);
545        };
546
547        let move_object = MoveObject::from_super(object);
548        Ok(Some(DynamicField::from_super(move_object)))
549    }

551    /// Look up a dynamic field by its literal name (Display v2 expression).
552    ///
553    /// Literals don't support field accesses or dynamic loads, so the interpreter is supplied
554    /// with a dummy store and root object.
555    pub(crate) async fn by_literal_name(
556        ctx: &Context<'_>,
557        scope: Scope,
558        parent: SuiAddress,
559        kind: DynamicFieldType,
560        literal: String,
561    ) -> Result<Option<Self>, RpcError<Error>> {
562        use DynamicFieldType as DFT;
563
564        struct NopStore;
565
566        #[async_trait]
567        impl sui_display::v2::Store for NopStore {
568            async fn object(
569                &self,
570                _: AccountAddress,
571            ) -> anyhow::Result<Option<sui_display::v2::OwnedSlice>> {
572                bail!("Dynamic loads not supported")
573            }
574        }
575
576        let limits: &Limits = ctx.data()?;
577        let limits = limits.display();
578
579        let root = sui_display::v2::OwnedSlice {
580            layout: MoveTypeLayout::Bool,
581            bytes: bcs::to_bytes(&false).unwrap(),
582        };
583
584        let parsed =
585            sui_display::v2::Name::parse(limits, &literal).map_err(|e| bad_user_input(e.into()))?;
586
587        let interpreter = sui_display::v2::Interpreter::new(root, NopStore);
588
589        let value = match parsed.eval(&interpreter).await {
590            Ok(Some(value)) => value,
591            Ok(None) => return Err(bad_user_input(Error::FieldAccess)),
592            Err(sui_display::v2::FormatError::Store(_)) => {
593                return Err(bad_user_input(Error::StoreAccess));
594            }
595            Err(e) => return Err(bad_user_input(e.into())),
596        };
597
598        let field_id: SuiAddress = match kind {
599            DFT::DynamicField => value
600                .derive_dynamic_field_id(parent)
601                .context("Failed to derive dynamic field ID")?,
602
603            DFT::DynamicObject => value
604                .derive_dynamic_object_field_id(parent)
605                .context("Failed to derive dynamic object field ID")?,
606        }
607        .into();
608
609        let object = Object::latest(ctx, scope.clone(), field_id)
610            .await
611            .map_err(upcast)?;
612
613        let Some(object) = object else {
614            return Ok(None);
615        };
616
617        let move_object = MoveObject::from_super(object);
618        Ok(Some(DynamicField::from_super(move_object)))
619    }

621    /// Paginate dynamic fields owned by a parent object
622    pub(crate) async fn paginate(
623        ctx: &Context<'_>,
624        scope: Scope,
625        parent: SuiAddress,
626        page: Page<CLive>,
627    ) -> Result<Connection<String, DynamicField>, RpcError<object::Error>> {
628        // Create a filter for dynamic fields: they are objects owned by the parent
629        // with type 0x2::dynamic_field::Field
630        let filter = ObjectFilter {
631            owner_kind: Some(OwnerKind::Object),
632            owner: Some(parent),
633            type_: Some(TypeFilter::Type(StructTag {
634                address: SUI_FRAMEWORK_ADDRESS,
635                module: DYNAMIC_FIELD_MODULE_NAME.to_owned(),
636                name: DYNAMIC_FIELD_FIELD_STRUCT_NAME.to_owned(),
637                type_params: vec![],
638            })),
639        };
640
641        let objects = Object::paginate_live(ctx, scope, page, filter).await?;
642        let mut dynamic_fields = Connection::new(objects.has_previous_page, objects.has_next_page);
643
644        for edge in objects.edges {
645            let move_obj = MoveObject::from_super(edge.node);
646            let dynamic_field = DynamicField::from_super(move_obj);
647            dynamic_fields
648                .edges
649                .push(Edge::new(edge.cursor, dynamic_field));
650        }
651
652        Ok(dynamic_fields)
653    }
654
655    /// Get the native dynamic field data, loading it lazily if needed.
656    pub(crate) async fn native(&self, ctx: &Context<'_>) -> Result<&Option<NativeField>, RpcError> {
657        self.native
658            .get_or_try_init(async || {
659                let Some(value) = self.super_.contents(ctx).await.ok().flatten().transpose()?
660                else {
661                    return Ok(None);
662                };
663
664                let Some(layout) = value.type_.layout_impl().await? else {
665                    return Ok(None);
666                };
667
668                let DFV::Field {
669                    kind,
670                    name_layout,
671                    name_bytes,
672                    value_layout,
673                    value_bytes,
674                    ..
675                } = DFV::FieldVisitor::deserialize(&value.native, &layout)
676                    .context("Failed to deserialize dynamic field")?;
677
678                Ok(Some(NativeField {
679                    kind,
680                    name_bytes: name_bytes.to_owned(),
681                    name_type: name_layout.into(),
682                    value_bytes: value_bytes.to_owned(),
683                    value_type: value_layout.into(),
684                    scope: self.super_.super_.super_.scope.clone(),
685                }))
686            })
687            .await
688    }
