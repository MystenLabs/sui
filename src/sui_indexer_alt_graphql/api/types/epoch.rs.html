79/// Activity on Sui is partitioned in time, into epochs.
80///
81/// Epoch changes are opportunities for the network to reconfigure itself (perform protocol or system package upgrades, or change the committee) and distribute staking rewards. The network aims to keep epochs roughly the same duration as each other.
82///
83/// During a particular epoch the following data is fixed:
84///
85/// - protocol version,
86/// - reference gas price,
87/// - system package versions,
88/// - validators in the committee.
89#[Object]
90impl Epoch {

130    /// State of the Coin DenyList object (0x403) at the start of this epoch.
131    ///
132    /// The DenyList controls access to Regulated Coins. Writes to the DenyList are accumulated and only take effect on the next epoch boundary. Consequently, it's possible to determine the state of the DenyList for a transaction by reading it at the start of the epoch the transaction is in.
133    async fn coin_deny_list(&self, ctx: &Context<'_>) -> Option<Result<Object, RpcError>> {
134        async {

204    /// A commitment by the committee at the end of epoch on the contents of the live object set at that time.
205    /// This can be used to verify state snapshots.

207        async {

215            for commitment in commitments {
216                if let CheckpointCommitment::ECMHLiveObjectSetDigest(digest) = commitment {

219            }

311    /// The total number of checkpoints in this epoch.
312    ///
313    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
314    async fn total_checkpoints(&self, ctx: &Context<'_>) -> Option<Result<UInt53, RpcError>> {
315        async {
316            let (Some(start), end) = try_join!(self.start(ctx), self.end(ctx))? else {
317                return Ok(None);
318            };
319
320            let lo = start.cp_lo as u64;
321            let hi = match end.as_ref() {
322                Some(end) => end.cp_hi as u64,
323                None => {
324                    let Some(bound) = self.scope.checkpoint_viewed_at_exclusive_bound() else {
325                        return Ok(None);
326                    };
327                    bound
328                }
329            };
330
331            Ok(Some(UInt53::from(hi - lo)))
332        }
333        .await
334        .transpose()
335    }

384    /// The total number of transaction blocks in this epoch.
385    ///
386    /// If the epoch has not finished yet, this number is computed based on the number of transactions at the latest known checkpoint.
387    async fn total_transactions(&self, ctx: &Context<'_>) -> Option<Result<UInt53, RpcError>> {
388        async {
389            let watermarks: &Arc<Watermarks> = ctx.data()?;
390            let (sequence_numbers, end) = try_join!(self.sequence_numbers(ctx), self.end(ctx))?;
391
392            let Some(start) = &sequence_numbers.start else {
393                return Ok(None);
394            };
395
396            let lo = start.tx_lo as u64;
397            let hi = if let Some(end) = end {
398                // If the epoch has already ended as of the latest checkpoint, its end record
399                // stores its transaction high watermark.
400                end.tx_hi as u64
401            } else if let Some(next) = &sequence_numbers.next {
402                // Otherwise, we have attempted to fetch the transaction low watermark of the
403                // checkpoint *after* the one being viewed at.
404                next.tx_lo as u64
405            } else {
406                // If all else fails, assume that the checkpoint being viewed at is the latest one
407                // known to the service, and use its global transaction high watermark.
408                watermarks.high_watermark().transaction()
409            };
410
411            Ok(Some(UInt53::from(hi.saturating_sub(lo))))
412        }
413        .await
414        .transpose()
415    }

417    /// The transactions in this epoch, optionally filtered by transaction filters.
418    ///
419    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
420    async fn transactions(
421        &self,
422        ctx: &Context<'_>,
423        first: Option<u64>,
424        after: Option<CTransaction>,
425        last: Option<u64>,
426        before: Option<CTransaction>,
427        #[graphql(validator(custom = "TFValidator"))] filter: Option<TransactionFilter>,
428    ) -> Option<Result<Connection<String, Transaction>, RpcError>> {
429        async {
430            let (Some(start), end) = try_join!(self.start(ctx), self.end(ctx))? else {
431                return Ok(None);
432            };
433            let Some(checkpoint_viewed_at_exclusive_bound) =
434                self.scope.checkpoint_viewed_at_exclusive_bound()
435            else {
436                return Ok(None);
437            };
438
439            let pagination: &PaginationConfig = ctx.data()?;
440            let limits = pagination.limits("Epoch", "transactions");
441            let page = Page::from_params(limits, first, after, last, before)?;
442
443            let cp_lo_exclusive = (start.cp_lo as u64).checked_sub(1);
444            let cp_hi = end.as_ref().map_or_else(
445                || checkpoint_viewed_at_exclusive_bound,
446                |end| end.cp_hi as u64,
447            );
448
449            let Some(filter) = filter.unwrap_or_default().intersect(TransactionFilter {
450                after_checkpoint: cp_lo_exclusive.map(UInt53::from),
451                before_checkpoint: Some(UInt53::from(cp_hi)),
452                ..Default::default()
453            }) else {
454                return Ok(Some(Connection::new(false, false)));
455            };
456
457            Ok(Some(
458                Transaction::paginate(ctx, self.scope.clone(), page, filter).await?,
459            ))
460        }
461        .await
462        .transpose()
463    }

490    /// Construct an epoch that is represented by just its identifier (its sequence number). This
491    /// does not check whether the epoch exists, so should not be used to "fetch" an epoch based on
492    /// an ID provided as user input.
493    pub(crate) fn with_id(scope: Scope, epoch_id: u64) -> Self {
494        Self {
495            epoch_id,
496            scope,
497            start: OnceCell::new(),
498            end: OnceCell::new(),
499            sequence_numbers: OnceCell::new(),
500        }
501    }

503    /// Load the epoch from the store, and return it fully inflated (with contents already
504    /// fetched). If `epoch_id` is provided, the epoch with that ID is loaded. Otherwise, the
505    /// latest epoch for the current checkpoint is loaded.
506    ///
507    /// Returns `None` if the epoch does not exist, started after the checkpoint being viewed,
508    /// or when no checkpoint is set in scope (e.g. execution scope).
509    pub(crate) async fn fetch(
510        ctx: &Context<'_>,
511        scope: Scope,
512        epoch_id: Option<UInt53>,
513    ) -> Result<Option<Self>, RpcError> {
514        // In execution scope, epoch queries return None
515        let Some(checkpoint_viewed_at) = scope.checkpoint_viewed_at() else {
516            return Ok(None);
517        };
518
519        let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
520
521        let stored = match epoch_id {
522            Some(id) => pg_loader
523                .load_one(EpochStartKey(id.into()))
524                .await
525                .context("Failed to fetch epoch start information by start key")?,
526            None => pg_loader
527                .load_one(CheckpointBoundedEpochStartKey(checkpoint_viewed_at))
528                .await
529                .context(
530                    "Failed to fetch epoch start information by checkpoint bounded start key",
531                )?,
532        }
533        .filter(|start| start.cp_lo as u64 <= checkpoint_viewed_at);
534
535        Ok(stored.map(|start| Self {
536            epoch_id: start.epoch as u64,
537            scope: scope.clone(),
538            start: OnceCell::from(Some(start)),
539            end: OnceCell::new(),
540            sequence_numbers: OnceCell::new(),
541        }))
542    }

544    /// Paginate through epochs.
545    pub(crate) async fn paginate(
546        ctx: &Context<'_>,
547        scope: &Scope,
548        page: Page<CEpoch>,
549    ) -> Result<Option<Connection<String, Epoch>>, RpcError> {
550        let Some(latest_epoch) = Epoch::fetch(ctx, scope.clone(), None).await? else {
551            return Ok(Some(Connection::new(false, false)));
552        };
553
554        page.paginate_indices(1 + latest_epoch.epoch_id as usize, |id| {
555            Ok(Epoch::with_id(scope.clone(), id as u64))
556        })
557        .map(Some)
558    }

560    /// Get the epoch start information.
561    ///
562    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
563    async fn start(&self, ctx: &Context<'_>) -> Result<&Option<StoredEpochStart>, RpcError> {
564        let Some(checkpoint_viewed_at) = self.scope.checkpoint_viewed_at() else {
565            return Ok(&None);
566        };
567
568        self.start
569            .get_or_try_init(async || {
570                let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
571
572                let stored = pg_loader
573                    .load_one(EpochStartKey(self.epoch_id))
574                    .await
575                    .context("Failed to fetch epoch start information")?
576                    .filter(|start| start.cp_lo as u64 <= checkpoint_viewed_at);
577
578                Ok(stored)
579            })
580            .await
581    }
582
583    /// Attempt to fetch information about the end of an epoch from the store. May return an empty
584    /// response if the epoch has not ended yet, as of the checkpoint being viewed, or when
585    /// no checkpoint is set in scope (e.g. execution scope).
586    async fn end(&self, ctx: &Context<'_>) -> Result<&Option<StoredEpochEnd>, RpcError> {
587        let Some(checkpoint_bound) = self.scope.checkpoint_viewed_at_exclusive_bound() else {
588            return Ok(&None);
589        };
590
591        self.end
592            .get_or_try_init(async || {
593                let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
594                let stored = pg_loader
595                    .load_one(EpochEndKey(self.epoch_id))
596                    .await
597                    .context("Failed to fetch epoch end information")?
598                    .filter(|end| end.cp_hi as u64 <= checkpoint_bound);
599
600                Ok(stored)
601            })
602            .await
603    }
604
605    async fn sequence_numbers(&self, ctx: &Context<'_>) -> Result<&SequenceNumbers, RpcError> {
606        self.sequence_numbers
607            .get_or_try_init(async || {
608                let Some(start) = self.start(ctx).await? else {
609                    return Ok(SequenceNumbers::default());
610                };
611
612                let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
613
614                let start = pg_loader.load_one(CpSequenceNumberKey(start.cp_lo as u64));
615                let next: OptionFuture<_> = self
616                    .scope
617                    .checkpoint_viewed_at()
618                    .map(|cp| pg_loader.load_one(CpSequenceNumberKey(cp + 1)))
619                    .into();
620
621                let (start, next) = join!(start, next);
622                let start = start.context("Failed to fetch epoch start sequence numbers")?;
623                let next = next
624                    .transpose()
625                    .context("Failed to fetch latest checkpoint sequence numbers")?
626                    .flatten();
627
628                Ok(SequenceNumbers { start, next })
629            })
630            .await
631    }

633    /// The contents of the system state inner object at the start of this epoch.
634    async fn system_state_impl(&self, ctx: &Context<'_>) -> Result<Option<MoveValue>, RpcError> {
635        let Some(start) = self.start(ctx).await? else {
636            return Ok(None);
637        };
638
639        let scope = self.scope.with_root_checkpoint(start.cp_lo as u64);
640        let struct_name = match start.system_state.first() {
641            Some(0) => SUI_SYSTEM_STATE_INNER_V1_STRUCT_NAME,
642            Some(1) => SUI_SYSTEM_STATE_INNER_V2_STRUCT_NAME,
643            _ => {
644                return Ok(None);
645            }
646        };
647
648        let tag = TypeTag::Struct(Box::new(StructTag {
649            address: SUI_SYSTEM_ADDRESS,
650            module: SUI_SYSTEM_STATE_INNER_MODULE_NAME.to_owned(),
651            name: struct_name.to_owned(),
652            type_params: vec![],
653        }));
654
655        let type_ = MoveType::from_native(tag, scope);
656        let native = start.system_state[1..].to_owned();
657
658        Ok(Some(MoveValue { type_, native }))
659    }
660}
