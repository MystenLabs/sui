26#[derive(InputObject, Debug, Default, Clone)]
27pub(crate) struct EventFilter {
28    /// Limit to events that occured strictly after the given checkpoint.
29    pub after_checkpoint: Option<UInt53>,
30
31    /// Limit to events in the given checkpoint.
32    pub at_checkpoint: Option<UInt53>,
33
34    /// Limit to event that occured strictly before the given checkpoint.
35    pub before_checkpoint: Option<UInt53>,
36
37    /// Filter on events by transaction sender address.
38    pub sender: Option<SuiAddress>,
39
40    /// Events emitted by a particular module. An event is emitted by a particular module if some function in the module is called by a PTB and emits an event.
41    ///
42    /// Modules can be filtered by their package, or package::module. We currently do not support filtering by emitting module and event type at the same time so if both are provided in one filter, the query will error.
43    pub module: Option<ModuleFilter>,
44
45    /// This field is used to specify the type of event emitted.
46    ///
47    /// Events can be filtered by their type's package, package::module, or their fully qualified type name.
48    ///
49    /// Generic types can be queried by either the generic type name, e.g. `0x2::coin::Coin`, or by the full type name, such as `0x2::coin::Coin<0x2::sui::SUI>`.
50    pub type_: Option<TypeFilter>,
51}
52
53impl EventFilter {
54    /// Builds a SQL query to select and filter events based on sender, module, and type filters.
55    /// Uses the provided transaction bounds subquery to limit results to a specific transaction range
56    pub(crate) fn query<'q>(&self) -> Result<Query<'q>, RpcError> {
57        let table = match (&self.module, &self.type_) {
58            (Some(_), Some(_)) => {
59                return Err(feature_unavailable(
60                    "Filtering by both emitting module and event type is not supported",
61                ));
62            }
63            (Some(_), None) => query!("ev_emit_mod"),
64            (None, _) => query!("ev_struct_inst"),
65        };
66
67        let mut query = query!(
68            r#"
69            SELECT
70                tx_sequence_number
71            FROM
72                {}
73            WHERE
74                tx_sequence_number >= (SELECT tx_lo FROM tx_lo)
75                AND tx_sequence_number < (SELECT tx_hi FROM tx_hi)
76            "#,
77            table,
78        );
79
80        if let Some(sender) = self.sender {
81            query += query!(" AND sender = {Bytea}", sender.into_vec());
82        }
83
84        if let Some(package) = self.module.as_ref().map(|m| m.package()) {
85            query += query!(" AND package = {Bytea}", package.into_vec());
86        }
87
88        if let Some(module) = self.module.as_ref().and_then(|m| m.module()) {
89            query += query!(" AND module = {Text}", module.to_string());
90        }
91
92        if let Some(package) = self.type_.as_ref().map(|t| t.package()) {
93            query += query!(" AND package = {Bytea}", package.into_vec());
94        }
95
96        if let Some(module) = self.type_.as_ref().and_then(|t| t.module()) {
97            query += query!(" AND module = {Text}", module.to_string());
98        }
99
100        if let Some(name) = self.type_.as_ref().and_then(|t| t.type_name()) {
101            query += query!(" AND name = {Text}", name.to_string());
102        }
103
104        if let Some(type_params) = self.type_.as_ref().and_then(|t| t.type_params()) {
105            query += query!(
106                " AND instantiation = {Bytea}",
107                bcs::to_bytes(type_params).context("Failed to serialize type parameters")?
108            );
109        }
110
111        Ok(query)
112    }
113
114    // Check if the Event matches sender, module, or type filters in EventFilter if they are provided.
115    pub(crate) fn matches(&self, event: &NativeEvent) -> bool {
116        if let Some(sender) = &self.sender
117            && sender != &SuiAddress::from(event.sender)
118        {
119            return false;
120        }
121
122        if let Some(module) = &self.module {
123            if module.package() != SuiAddress::from(event.package_id) {
124                return false;
125            }
126
127            if let Some(module) = module.module()
128                && module != event.transaction_module.as_str()
129            {
130                return false;
131            }
132        }
133
134        if let Some(type_) = &self.type_ {
135            if type_.package() != SuiAddress::from(event.type_.address) {
136                return false;
137            }
138
139            if let Some(module) = type_.module()
140                && module != event.type_.module.as_str()
141            {
142                return false;
143            }
144
145            if let Some(type_name) = type_.type_name()
146                && type_name != event.type_.name.as_str()
147            {
148                return false;
149            }
150
151            if let Some(type_params) = type_.type_params()
152                && type_params != event.type_.type_params.as_slice()
153            {
154                return false;
155            }
156        }
157
158        true
159    }
160}

176/// The event indices (sequence_number) in a transaction's events array that are within the cursor bounds, inclusively.
177/// Event transaction numbers are always returned in ascending order.
178pub(super) fn tx_ev_bounds(
179    page: &Page<CEvent>,
180    tx_sequence_number: u64,
181    event_count: usize,
182) -> Range<usize> {
183    // Find start index from 'after' cursor, defaults to 0
184    let ev_lo = page
185        .after()
186        .filter(|c| c.tx_sequence_number == tx_sequence_number)
187        .map(|c| c.ev_sequence_number as usize)
188        .unwrap_or(0)
189        .min(event_count);
190
191    // Find exclusive end index from 'before' cursor, default to event_count
192    let ev_hi = page
193        .before()
194        .filter(|c| c.tx_sequence_number == tx_sequence_number)
195        .map(|c| (c.ev_sequence_number as usize).saturating_add(1))
196        .unwrap_or(event_count)
197        .max(ev_lo)
198        .min(event_count);
199
200    ev_lo..ev_hi
201}
