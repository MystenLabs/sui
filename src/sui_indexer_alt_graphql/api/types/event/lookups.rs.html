19/// The page of Event cursors and Events emitted from transactions with cursors and limits with
20/// overhead applied.
21///
22/// Note: tx_sequence_numbers are ordered ASC or DESC depending on the page direction, while events
23/// in each transactions are returned in ASC sequence number from the store.
24pub(crate) async fn events_from_sequence_numbers(
25    scope: &Scope,
26    ctx: &Context<'_>,
27    page: &Page<CEvent>,
28    tx_sequence_numbers: &[u64],
29    filter: &EventFilter,
30) -> Result<Vec<(EventCursor, Event)>, RpcError> {
31    let kv_loader: &KvLoader = ctx.data()?;
32
33    let digests = tx_digests(ctx, tx_sequence_numbers).await?;
34
35    let digest_to_events = kv_loader
36        .load_many_transaction_events(digests.iter().map(|d| d.1).collect())
37        .await
38        .context("Failed to load transaction events")?;
39
40    let events = digests.into_iter().map(|(tx_sequence_number, digest)| {
41        let events = digest_to_events
42            .get(&digest)
43            .context("Failed to get events")?;
44
45        Ok((tx_sequence_number, digest, events))
46    });
47
48    let results = tx_events_paginated(scope, page, events, filter)?;
49
50    Ok(results)
51}
52
53/// Helper function to map sequence numbers to a page of Event and Event Cursors.
54fn tx_events_paginated<'e>(
55    scope: &Scope,
56    page: &Page<CEvent>,
57    contents: impl Iterator<
58        Item = anyhow::Result<(u64, TransactionDigest, &'e TransactionEventsContents)>,
59    >,
60    filter: &EventFilter,
61) -> Result<Vec<(EventCursor, Event)>, RpcError> {
62    let mut results = Vec::new();
63    let limit = page.limit_with_overhead();
64
65    'outer: for events in contents {
66        let (tx_sequence_number, transaction_digest, contents) = events?;
67        let events = contents.events()?;
68
69        let bounds: Either<Range<usize>, Rev<Range<usize>>> = if page.is_from_front() {
70            Either::Left(tx_ev_bounds(page, tx_sequence_number, events.len()))
71        } else {
72            Either::Right(tx_ev_bounds(page, tx_sequence_number, events.len()).rev())
73        };
74
75        for ev_sequence_number in bounds {
76            let event_cursor = EventCursor {
77                tx_sequence_number,
78                ev_sequence_number: ev_sequence_number as u64,
79            };
80
81            let native = &events[ev_sequence_number];
82            if !filter.matches(native) {
83                continue;
84            }
85
86            let event = Event {
87                scope: scope.clone(),
88                native: native.clone(),
89                transaction_digest,
90                sequence_number: ev_sequence_number as u64,
91                timestamp_ms: contents.timestamp_ms(),
92            };
93
94            results.push((event_cursor, event));
95
96            if results.len() >= limit {
97                break 'outer;
98            }
99        }
100    }
101
102    if !page.is_from_front() {
103        results.reverse();
104    }
105
106    Ok(results)
107}
