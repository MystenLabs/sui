24/// The page of Event cursors and Events emitted from transactions with cursors and limits with
25/// overhead applied.
26///
27/// Note: tx_sequence_numbers are ordered ASC or DESC depending on the page direction, while events
28/// in each transactions are returned in ASC sequence number from the store.
29pub(crate) async fn events_from_sequence_numbers(
30    scope: &Scope,
31    ctx: &Context<'_>,
32    page: &Page<CEvent>,
33    tx_sequence_numbers: &[u64],
34    filter: &EventFilter,
35) -> Result<Vec<(EventCursor, Event)>, RpcError> {
36    let kv_loader: &KvLoader = ctx.data()?;
37
38    let digests = tx_digests(ctx, tx_sequence_numbers).await?;
39
40    let digest_to_events = kv_loader
41        .load_many_transaction_events(digests.iter().map(|d| d.1).collect())
42        .await
43        .context("Failed to load transaction events")?;
44
45    let events = digests.into_iter().map(|(tx_sequence_number, digest)| {
46        let events = digest_to_events
47            .get(&digest)
48            .context("Failed to get events")?;
49
50        Ok((tx_sequence_number, digest, events))
51    });
52
53    let results = tx_events_paginated(scope, page, events, filter)?;
54
55    Ok(results)
56}
57
58/// Helper function to map sequence numbers to a page of Event and Event Cursors.
59fn tx_events_paginated<'e>(
60    scope: &Scope,
61    page: &Page<CEvent>,
62    contents: impl Iterator<
63        Item = anyhow::Result<(u64, TransactionDigest, &'e TransactionEventsContents)>,
64    >,
65    filter: &EventFilter,
66) -> Result<Vec<(EventCursor, Event)>, RpcError> {
67    let mut results = Vec::new();
68    let limit = page.limit_with_overhead();
69
70    'outer: for events in contents {
71        let (tx_sequence_number, transaction_digest, contents) = events?;
72        let events = contents.events()?;
73
74        let bounds: Either<Range<usize>, Rev<Range<usize>>> = if page.is_from_front() {
75            Either::Left(tx_ev_bounds(page, tx_sequence_number, events.len()))
76        } else {
77            Either::Right(tx_ev_bounds(page, tx_sequence_number, events.len()).rev())
78        };
79
80        for ev_sequence_number in bounds {
81            let event_cursor = EventCursor {
82                tx_sequence_number,
83                ev_sequence_number: ev_sequence_number as u64,
84            };
85
86            let native = &events[ev_sequence_number];
87            if !filter.matches(native) {
88                continue;
89            }
90
91            let event = Event {
92                scope: scope.clone(),
93                native: native.clone(),
94                transaction_digest,
95                sequence_number: ev_sequence_number as u64,
96                timestamp_ms: contents.timestamp_ms(),
97            };
98
99            results.push((event_cursor, event));
100
101            if results.len() >= limit {
102                break 'outer;
103            }
104        }
105    }
106
107    if !page.is_from_front() {
108        results.reverse();
109    }
110
111    Ok(results)
112}
