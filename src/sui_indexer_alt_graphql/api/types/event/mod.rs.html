37#[derive(Serialize, Deserialize, PartialEq, Eq, Clone, Debug, PartialOrd, Ord, Copy)]
38pub(crate) struct EventCursor {
39    #[serde(rename = "t")]
40    pub tx_sequence_number: u64,
41    #[serde(rename = "e")]
42    pub ev_sequence_number: u64,
43}
44
45pub(crate) type CEvent = JsonCursor<EventCursor>;
46
47#[derive(Clone)]
48pub(crate) struct Event {
49    pub(crate) scope: Scope,
50    pub(crate) native: NativeEvent,
51    /// Digest of the transaction that emitted this event
52    pub(crate) transaction_digest: TransactionDigest,
53    /// Position of this event within the transaction's events list (0-indexed)
54    pub(crate) sequence_number: u64,
55    /// Timestamp when the transaction containing this event was finalized (checkpoint time)
56    pub(crate) timestamp_ms: u64,
57}
58
59#[Object]
60impl Event {
61    /// The Move value emitted for this event.
62    async fn contents(&self) -> Option<MoveValue> {
63        let type_ = MoveType::from_native(self.native.type_.clone().into(), self.scope.clone());
64        Some(MoveValue::new(type_, self.native.contents.clone()))
65    }
66
67    /// The Base64 encoded BCS serialized bytes of the entire Event structure from sui-types.
68    /// This includes: package_id, transaction_module, sender, type, and contents (which itself contains the BCS-serialized Move struct data).
69    async fn event_bcs(&self) -> Result<Option<Base64>, RpcError> {
70        let bcs_bytes = bcs::to_bytes(&self.native).context("Failed to serialize event")?;
71        Ok(Some(Base64(bcs_bytes)))
72    }
73
74    /// Address of the sender of the transaction that emitted this event.
75    async fn sender(&self) -> Option<Address> {
76        if self.native.sender == NativeSuiAddress::ZERO {
77            return None;
78        }
79
80        Some(Address::with_address(
81            self.scope.clone(),
82            self.native.sender,
83        ))
84    }
85
86    /// The position of the event among the events from the same transaction.
87    async fn sequence_number(&self) -> UInt53 {
88        UInt53::from(self.sequence_number)
89    }
90
91    /// Timestamp corresponding to the checkpoint this event's transaction was finalized in.
92    /// All events from the same transaction share the same timestamp.
93    async fn timestamp(&self) -> Result<Option<DateTime>, RpcError> {
94        Ok(Some(DateTime::from_ms(self.timestamp_ms as i64)?))
95    }
96
97    /// The transaction that emitted this event. This information is only available for events from indexed transactions, and not from transactions that have just been executed or dry-run.
98    async fn transaction(&self) -> Option<Transaction> {
99        Some(Transaction::with_id(
100            self.scope.clone(),
101            self.transaction_digest,
102        ))
103    }
104
105    /// The module containing the function that was called in the programmable transaction, that resulted in this event being emitted.
106    async fn transaction_module(&self) -> Option<MoveModule> {
107        let package = MovePackage::with_address(self.scope.clone(), self.native.package_id.into());
108        Some(MoveModule::with_fq_name(
109            package,
110            self.native.transaction_module.to_string(),
111        ))
112    }
113}

115impl Event {
116    /// Paginates events based on the provided filters and page parameters.
117    ///
118    /// Returns empty results when no checkpoint is set in scope (e.g. execution scope).
119    pub(crate) async fn paginate(
120        ctx: &Context<'_>,
121        scope: Scope,
122        page: Page<CEvent>,
123        filter: EventFilter,
124    ) -> Result<Connection<String, Event>, RpcError> {
125        let pg_reader: &PgReader = ctx.data()?;
126
127        // TODO: (henry) Use watermarks once we have a strategy for kv pruning.
128        let reader_lo = 0;
129
130        let Some(mut query) = filter.tx_bounds(ctx, &scope, reader_lo, &page).await? else {
131            return Ok(Connection::new(false, false));
132        };
133
134        #[derive(QueryableByName)]
135        struct TxSequenceNumber(
136            #[diesel(sql_type = BigInt, column_name = "tx_sequence_number")] i64,
137        );
138
139        query += filter.query()?;
140        query += query!(
141            r#" ORDER BY tx_sequence_number {} LIMIT {BigInt}"#,
142            page.order_by_direction(),
143            page.limit_with_overhead() as i64,
144        );
145
146        let mut conn = pg_reader
147            .connect()
148            .await
149            .context("Failed to connect to database")?;
150
151        let tx_sequence_numbers: Vec<u64> = conn
152            .results(query)
153            .await
154            .context("Failed to execute query")?
155            .into_iter()
156            .map(|tx_seq: TxSequenceNumber| tx_seq.0 as u64)
157            .unique()
158            .collect();
159
160        let events = lookups::events_from_sequence_numbers(
161            &scope,
162            ctx,
163            &page,
164            &tx_sequence_numbers,
165            &filter,
166        )
167        .await?;
168
169        page.paginate_results(events, |(c, _)| JsonCursor::new(*c), |(_, e)| Ok(e))
170    }
171}
