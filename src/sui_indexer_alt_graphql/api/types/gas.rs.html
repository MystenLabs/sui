4use async_graphql::Object;
5use sui_types::gas::GasCostSummary as NativeGasCostSummary;
6
7use crate::api::scalars::uint53::UInt53;
8
9pub(crate) struct GasCostSummary {
10    native: NativeGasCostSummary,
11}
12
13/// Summary of charges from transactions.
14///
15/// Storage is charged in three parts -- `storage_cost`, `-storage_rebate`, and `non_refundable_storage_fee` -- independently of `computation_cost`.
16///
17/// The overall cost of a transaction, deducted from its gas coins, is its `computation_cost + storage_cost - storage_rebate`. `non_refundable_storage_fee` is collected from objects being mutated or deleted and accumulated by the system in storage funds, the remaining storage costs of previous object versions are what become the `storage_rebate`. The ratio between `non_refundable_storage_fee` and `storage_rebate` is set by the protocol.
18#[Object]
19impl GasCostSummary {
20    /// The sum cost of computation/execution
21    async fn computation_cost(&self) -> Option<UInt53> {
22        Some(self.native.computation_cost.into())
23    }
24    /// Cost for storage at the time the transaction is executed, calculated as the size of the objects being mutated in bytes multiplied by a storage cost per byte (part of the protocol).
25    async fn storage_cost(&self) -> Option<UInt53> {
26        Some(self.native.storage_cost.into())
27    }
28    /// Amount the user gets back from the storage cost of the previous versions of objects being mutated or deleted.
29    async fn storage_rebate(&self) -> Option<UInt53> {
30        Some(self.native.storage_rebate.into())
31    }
32    /// Amount that is retained by the system in the storage fund from the cost of the previous versions of objects being mutated or deleted.
33    async fn non_refundable_storage_fee(&self) -> Option<UInt53> {
34        Some(self.native.non_refundable_storage_fee.into())
35    }
36}
37
38impl From<NativeGasCostSummary> for GasCostSummary {
39    fn from(native: NativeGasCostSummary) -> Self {
40        Self { native }
41    }
42}
