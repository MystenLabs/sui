4use std::sync::Arc;
5
6use anyhow::Context as _;
7use async_graphql::Context;
8use async_graphql::Object;
9use async_graphql::SimpleObject;
10use sui_package_resolver::FunctionDef;
11use tokio::sync::OnceCell;
12
13use crate::api::types::move_module::MoveModule;
14use crate::api::types::move_module::MoveVisibility;
15use crate::api::types::move_type::MoveAbility;
16use crate::api::types::move_type::abilities;
17use crate::api::types::open_move_type::OpenMoveType;
18use crate::error::RpcError;
19
20pub(crate) struct MoveFunction {
21    /// The module that this function is defined in.
22    module: MoveModule,
23
24    /// The function's unqualified name.
25    name: String,
26
27    /// The lazily loaded definition of the function.
28    contents: Arc<OnceCell<Option<FunctionDef>>>,
29}
30
31/// Declaration of a type parameter on a Move function.
32#[derive(SimpleObject)]
33pub(crate) struct MoveFunctionTypeParameter {
34    /// Ability constraints on this type parameter.
35    constraints: Vec<MoveAbility>,
36}
37
38/// A function defined in a Move module.
39#[Object]
40impl MoveFunction {
41    /// The module that this function is defined in.
42    async fn module(&self) -> &MoveModule {
43        &self.module
44    }
45
46    /// The function's unqualified name.
47    async fn name(&self) -> &str {
48        &self.name
49    }
50
51    /// The function's fully-qualified name, including package address, module name, and function name.
52    async fn fully_qualified_name(&self) -> String {
53        format!(
54            "{}::{}::{}",
55            self.module.package.address_impl(),
56            self.module.name,
57            self.name
58        )
59    }
60
61    /// Whether the function is marked `entry` or not.
62    async fn is_entry(&self, ctx: &Context<'_>) -> Option<Result<bool, RpcError>> {
63        let contents = self.contents(ctx).await.ok()?.as_ref()?;
64        Some(Ok(contents.is_entry))
65    }
66
67    /// The function's parameter types. These types can reference type parameters introduced by this function (see `typeParameters`).
68    async fn parameters(&self, ctx: &Context<'_>) -> Option<Result<Vec<OpenMoveType>, RpcError>> {
69        let contents = self.contents(ctx).await.ok()?.as_ref()?;
70        Some(Ok(contents
71            .parameters
72            .iter()
73            .cloned()
74            .map(OpenMoveType::from)
75            .collect()))
76    }
77
78    /// The function's return types. There can be multiple because functions in Move can return multiple values. These types can reference type parameters introduced by this function (see `typeParameters`).
79    async fn return_(&self, ctx: &Context<'_>) -> Option<Result<Vec<OpenMoveType>, RpcError>> {
80        let contents = self.contents(ctx).await.ok()?.as_ref()?;
81        Some(Ok(contents
82            .return_
83            .iter()
84            .cloned()
85            .map(OpenMoveType::from)
86            .collect()))
87    }
88
89    /// Constraints on the function's formal type parameters.
90    ///
91    /// Move bytecode does not name type parameters, so when they are referenced (e.g. in parameter and return types), they are identified by their index in this list.
92    async fn type_parameters(
93        &self,
94        ctx: &Context<'_>,
95    ) -> Option<Result<Vec<MoveFunctionTypeParameter>, RpcError>> {
96        let contents = self.contents(ctx).await.ok()?.as_ref()?;
97        Some(Ok(contents
98            .type_params
99            .iter()
100            .map(|c| MoveFunctionTypeParameter {
101                constraints: abilities(*c),
102            })
103            .collect()))
104    }
105
106    /// The function's visibility: `public`, `public(friend)`, or `private`.
107    async fn visibility(&self, ctx: &Context<'_>) -> Option<Result<MoveVisibility, RpcError>> {
108        let contents = self.contents(ctx).await.ok()?.as_ref()?;
109        Some(Ok(contents.visibility.into()))
110    }
111}
112
113impl MoveFunction {
114    /// Construct a function that is represented by its fully-qualified name (package, module and
115    /// name). This does not check that the function actually exists, so should not be used to
116    /// "fetch" a function based on user input.
117    pub(crate) fn with_fq_name(module: MoveModule, name: String) -> Self {
118        Self {
119            module,
120            name,
121            contents: Arc::new(OnceCell::new()),
122        }
123    }
124
125    /// Construct a function with a pre-loaded function definition.
126    pub(crate) fn from_def(module: MoveModule, name: String, def: FunctionDef) -> Self {
127        Self {
128            module,
129            name,
130            contents: Arc::new(OnceCell::from(Some(def))),
131        }
132    }
133
134    /// Get the function definition, loading it lazily if needed.
135    async fn contents(&self, ctx: &Context<'_>) -> Result<&Option<FunctionDef>, RpcError> {
136        self.contents
137            .get_or_try_init(|| async {
138                let Some(contents) = self.module.contents(ctx).await? else {
139                    return Ok(None);
140                };
141
142                let def = contents
143                    .parsed
144                    .function_def(&self.name)
145                    .context("Failed to get function definition")?;
146
147                Ok(def)
148            })
149            .await
150    }
151}
