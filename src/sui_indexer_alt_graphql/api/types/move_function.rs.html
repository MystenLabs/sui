4use std::sync::Arc;
5
6use anyhow::Context as _;
7use async_graphql::{Context, Object, SimpleObject};
8use sui_package_resolver::FunctionDef;
9use tokio::sync::OnceCell;
10
11use crate::error::RpcError;
12
13use super::{
14    move_module::{MoveModule, MoveVisibility},
15    move_type::{MoveAbility, abilities},
16    open_move_type::OpenMoveType,
17};

19pub(crate) struct MoveFunction {
20    /// The module that this function is defined in.
21    module: MoveModule,
22
23    /// The function's unqualified name.
24    name: String,
25
26    /// The lazily loaded definition of the function.
27    contents: Arc<OnceCell<Option<FunctionDef>>>,
28}
29
30/// Declaration of a type parameter on a Move function.
31#[derive(SimpleObject)]
32pub(crate) struct MoveFunctionTypeParameter {
33    /// Ability constraints on this type parameter.
34    constraints: Vec<MoveAbility>,
35}
36
37/// A function defined in a Move module.
38#[Object]
39impl MoveFunction {
40    /// The module that this function is defined in.
41    async fn module(&self) -> &MoveModule {
42        &self.module
43    }
44
45    /// The function's unqualified name.
46    async fn name(&self) -> &str {
47        &self.name
48    }
49
50    /// Whether the function is marked `entry` or not.
51    async fn is_entry(&self, ctx: &Context<'_>) -> Result<Option<bool>, RpcError> {
52        let Some(contents) = self.contents(ctx).await?.as_ref() else {
53            return Ok(None);
54        };
55
56        Ok(Some(contents.is_entry))
57    }
58
59    /// The function's parameter types. These types can reference type parameters introduced by this function (see `typeParameters`).
60    async fn parameters(&self, ctx: &Context<'_>) -> Result<Option<Vec<OpenMoveType>>, RpcError> {
61        let Some(contents) = self.contents(ctx).await?.as_ref() else {
62            return Ok(None);
63        };
64
65        Ok(Some(
66            contents
67                .parameters
68                .iter()
69                .cloned()
70                .map(OpenMoveType::from)
71                .collect(),
72        ))
73    }
74
75    /// The function's return types. There can be multiple because functions in Move can return multiple values. These types can reference type parameters introduced by this function (see `typeParameters`).
76    async fn return_(&self, ctx: &Context<'_>) -> Result<Option<Vec<OpenMoveType>>, RpcError> {
77        let Some(contents) = self.contents(ctx).await?.as_ref() else {
78            return Ok(None);
79        };
80
81        Ok(Some(
82            contents
83                .return_
84                .iter()
85                .cloned()
86                .map(OpenMoveType::from)
87                .collect(),
88        ))
89    }
90
91    /// Constraints on the function's formal type parameters.
92    ///
93    /// Move bytecode does not name type parameters, so when they are referenced (e.g. in parameter and return types), they are identified by their index in this list.
94    async fn type_parameters(
95        &self,
96        ctx: &Context<'_>,
97    ) -> Result<Option<Vec<MoveFunctionTypeParameter>>, RpcError> {
98        let Some(contents) = self.contents(ctx).await?.as_ref() else {
99            return Ok(None);
100        };
101
102        Ok(Some(
103            contents
104                .type_params
105                .iter()
106                .map(|c| MoveFunctionTypeParameter {
107                    constraints: abilities(*c),
108                })
109                .collect(),
110        ))
111    }
112
113    /// The function's visibility: `public`, `public(friend)`, or `private`.
114    async fn visibility(&self, ctx: &Context<'_>) -> Result<Option<MoveVisibility>, RpcError> {
115        let Some(contents) = self.contents(ctx).await?.as_ref() else {
116            return Ok(None);
117        };
118
119        Ok(Some(contents.visibility.into()))
120    }
121}
122
123impl MoveFunction {
124    /// Construct a function that is represented by its fully-qualified name (package, module and
125    /// name). This does not check that the function actually exists, so should not be used to
126    /// "fetch" a function based on user input.
127    pub(crate) fn with_fq_name(module: MoveModule, name: String) -> Self {
128        Self {
129            module,
130            name,
131            contents: Arc::new(OnceCell::new()),
132        }
133    }
134
135    /// Construct a function with a pre-loaded function definition.
136    pub(crate) fn from_def(module: MoveModule, name: String, def: FunctionDef) -> Self {
137        Self {
138            module,
139            name,
140            contents: Arc::new(OnceCell::from(Some(def))),
141        }
142    }
143
144    /// Get the function definition, loading it lazily if needed.
145    async fn contents(&self, ctx: &Context<'_>) -> Result<&Option<FunctionDef>, RpcError> {
146        self.contents
147            .get_or_try_init(|| async {
148                let Some(contents) = self.module.contents(ctx).await? else {
149                    return Ok(None);
150                };
151
152                let def = contents
153                    .parsed
154                    .function_def(&self.name)
155                    .context("Failed to get function definition")?;
156
157                Ok(def)
158            })
159            .await
160    }
161}
