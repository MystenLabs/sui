42/// The visibility modifier describes which modules can access this module member.
43///
44/// By default, a module member can be called only within the same module.
45#[derive(Enum, Copy, Clone, Eq, PartialEq)]
46pub(crate) enum MoveVisibility {
47    /// A public member can be accessed by any module.
48    Public,
49    /// A private member can be accessed in the module it is defined in.
50    Private,
51    /// A friend member can be accessed in the module it is defined in and any other module in its package that is explicitly specified in its friend list.
52    Friend,
53}

57    pub(crate) parsed: ParsedModule,
58}

60/// Cursor for iterating over datatypes in a module. Points to the datatype by its name.

63/// Cursor for iterating over enums in a module. Points to the enum by its name.
64type CEnum = JsonCursor<String>;
65
66/// Cursor for iterating over friend modules. Points to the friend by its index in the friend list.
67type CFriend = JsonCursor<usize>;

69/// Cursor for iterating over functioons in a module. Points to the function by its name.
70type CFunction = JsonCursor<String>;

72/// Cursor for iterating over structs in a module. Points to the struct by its name.

101        &self,

104    ) -> Result<Option<MoveDatatype>, RpcError> {
105        let Some(contents) = self.contents(ctx).await?.as_ref() else {

107        };

110            .parsed

120    /// Paginate through this module's datatype definitions.
121    async fn datatypes(
122        &self,
123        ctx: &Context<'_>,
124        first: Option<u64>,
125        after: Option<CDatatype>,
126        last: Option<u64>,
127        before: Option<CDatatype>,
128    ) -> Result<Option<Connection<String, MoveDatatype>>, RpcError> {
129        let pagination: &PaginationConfig = ctx.data()?;
130        let limits = pagination.limits("MoveModule", "datatypes");
131        let page = Page::from_params(limits, first, after, last, before)?;
132
133        let Some(contents) = self.contents(ctx).await?.as_ref() else {
134            return Ok(None);
135        };
136
137        let datatype_range = contents.parsed.datatypes(
138            page.after().map(|c| c.as_ref()),
139            page.before().map(|c| c.as_ref()),
140        );
141
142        let mut conn = Connection::new(false, false);
143        let datatypes = if page.is_from_front() {
144            datatype_range.take(page.limit()).collect()
145        } else {
146            let mut datatypes: Vec<_> = datatype_range.rev().take(page.limit()).collect();
147            datatypes.reverse();
148            datatypes
149        };
150
151        conn.has_previous_page = datatypes
152            .first()
153            .is_some_and(|fst| contents.parsed.datatypes(None, Some(fst)).next().is_some());
154
155        conn.has_next_page = datatypes
156            .last()
157            .is_some_and(|lst| contents.parsed.datatypes(Some(lst), None).next().is_some());
158
159        for datatype_name in datatypes {
160            conn.edges.push(Edge::new(
161                JsonCursor::new(datatype_name.to_owned()).encode_cursor(),
162                MoveDatatype::with_fq_name(self.clone(), datatype_name.to_owned()),
163            ));
164        }
165
166        Ok(Some(conn))
167    }

169    /// Textual representation of the module's bytecode.
170    async fn disassembly(&self, ctx: &Context<'_>) -> Result<Option<String>, RpcError> {
171        let limits: &Limits = ctx.data()?;
172
173        let Some(contents) = self.contents(ctx).await?.as_ref() else {
174            return Ok(None);
175        };
176
177        Ok(Some(
178            Disassembler::from_module_with_max_size(
179                contents.parsed.bytecode(),
180                Loc::invalid(),
181                Some(limits.max_disassembled_module_size),
182            )
183            .context("Failed to initialize disassembler")?
184            .disassemble()
185            .map_err(resource_exhausted)?,
186        ))
187    }

264    /// Modules that this module considers friends. These modules can call `public(package)` functions in this module.
265    async fn friends(
266        &self,
267        ctx: &Context<'_>,
268        first: Option<u64>,
269        after: Option<CFriend>,
270        last: Option<u64>,
271        before: Option<CFriend>,
272    ) -> Result<Option<Connection<String, MoveModule>>, RpcError> {
273        let pagination: &PaginationConfig = ctx.data()?;
274        let limits = pagination.limits("MoveModule", "friends");
275        let page = Page::from_params(limits, first, after, last, before)?;
276
277        let Some(contents) = self.contents(ctx).await?.as_ref() else {
278            return Ok(None);
279        };
280
281        let bytecode = contents.parsed.bytecode();
282        let runtime_id = *bytecode.self_id().address();
283
284        let friends = bytecode.friend_decls();
285        page.paginate_indices(friends.len(), |i| {
286            let decl = &friends[i];
287            let friend_pkg = bytecode.address_identifier_at(decl.address);
288            let friend_mod = bytecode.identifier_at(decl.name);
289
290            if *friend_pkg != runtime_id {
291                return Err(anyhow!("Cross-package friend modules").into());
292            }
293
294            Ok(MoveModule::with_fq_name(
295                self.package.clone(),
296                friend_mod.to_string(),
297            ))
298        })
299        .map(Some)
300    }

443impl MoveModule {
444    /// Construct a module that is represented by just its fully-qualified name. This does not
445    /// check that the module actually exists, so should not be used to "fetch" a module based on
446    /// user input.
447    pub(crate) fn with_fq_name(package: MovePackage, name: String) -> Self {
448        Self {
449            package,
450            name,
451            contents: Arc::new(OnceCell::new()),
452        }
453    }
454
455    /// Get the native CompiledModule, loading it lazily if needed.
456    pub(crate) async fn contents(
457        &self,
458        ctx: &Context<'_>,
459    ) -> Result<&Option<ModuleContents>, RpcError> {
460        self.contents
461            .get_or_try_init(|| async {
462                let (native, parsed) = join!(self.package.native(ctx), self.package.parsed(ctx));
463                let (Some(native), Some(parsed)) = (native?.as_ref(), parsed?.as_ref()) else {
464                    return Ok(None);
465                };
466
467                let Some(native) = native.serialized_module_map().get(&self.name).cloned() else {
468                    return Ok(None);
469                };
470
471                let parsed = parsed
472                    .module(&self.name)
473                    .context("Couldn't find parsed module, for existing native module")?
474                    .clone();
475
476                Ok(Some(ModuleContents { native, parsed }))
477            })
478            .await
479    }
480}

482impl From<Visibility> for MoveVisibility {
483    fn from(visibility: Visibility) -> Self {
484        use MoveVisibility as M;
485        use Visibility as V;
486
487        match visibility {
488            V::Private => M::Private,
489            V::Public => M::Public,
490            V::Friend => M::Friend,
491        }
492    }
