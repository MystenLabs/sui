46/// The visibility modifier describes which modules can access this module member.
47///
48/// By default, a module member can be called only within the same module.
49#[derive(Enum, Copy, Clone, Eq, PartialEq)]
50pub(crate) enum MoveVisibility {
51    /// A public member can be accessed by any module.
52    Public,
53    /// A private member can be accessed in the module it is defined in.
54    Private,
55    /// A friend member can be accessed in the module it is defined in and any other module in its package that is explicitly specified in its friend list.
56    Friend,
57}

100    /// The datatype (struct or enum) named `name` in this module.
101    async fn datatype(
102        &self,
103        ctx: &Context<'_>,
104        name: String,
105    ) -> Option<Result<MoveDatatype, RpcError>> {
106        async {
107            let Some(contents) = self.contents(ctx).await?.as_ref() else {
108                return Ok(None);
109            };
110
111            let Some(def) = contents
112                .parsed
113                .data_def(&name)
114                .context("Failed to get datatype definition")?
115            else {
116                return Ok(None);
117            };
118
119            Ok(Some(MoveDatatype::from_def(self.clone(), name, def)))
120        }
121        .await
122        .transpose()
123    }

125    /// Paginate through this module's datatype definitions.
126    async fn datatypes(
127        &self,
128        ctx: &Context<'_>,
129        first: Option<u64>,
130        after: Option<CDatatype>,
131        last: Option<u64>,
132        before: Option<CDatatype>,
133    ) -> Option<Result<Connection<String, MoveDatatype>, RpcError>> {
134        async {
135            let pagination: &PaginationConfig = ctx.data()?;
136            let limits = pagination.limits("MoveModule", "datatypes");
137            let page = Page::from_params(limits, first, after, last, before)?;
138
139            let Some(contents) = self.contents(ctx).await?.as_ref() else {
140                return Ok(None);
141            };
142
143            let datatype_range = contents.parsed.datatypes(
144                page.after().map(|c| c.as_ref()),
145                page.before().map(|c| c.as_ref()),
146            );
147
148            let mut conn = Connection::new(false, false);
149            let datatypes = if page.is_from_front() {
150                datatype_range.take(page.limit()).collect()
151            } else {
152                let mut datatypes: Vec<_> = datatype_range.rev().take(page.limit()).collect();
153                datatypes.reverse();
154                datatypes
155            };
156
157            conn.has_previous_page = datatypes
158                .first()
159                .is_some_and(|fst| contents.parsed.datatypes(None, Some(fst)).next().is_some());
160
161            conn.has_next_page = datatypes
162                .last()
163                .is_some_and(|lst| contents.parsed.datatypes(Some(lst), None).next().is_some());
164
165            for datatype_name in datatypes {
166                conn.edges.push(Edge::new(
167                    JsonCursor::new(datatype_name.to_owned()).encode_cursor(),
168                    MoveDatatype::with_fq_name(self.clone(), datatype_name.to_owned()),
169                ));
170            }
171
172            Ok(Some(conn))
173        }
174        .await
175        .transpose()
176    }

178    /// Textual representation of the module's bytecode.
179    async fn disassembly(&self, ctx: &Context<'_>) -> Option<Result<String, RpcError>> {
180        async {
181            let limits: &Limits = ctx.data()?;
182
183            let Some(contents) = self.contents(ctx).await?.as_ref() else {
184                return Ok(None);
185            };
186
187            Ok(Some(
188                Disassembler::from_module_with_max_size(
189                    contents.parsed.bytecode(),
190                    Loc::invalid(),
191                    Some(limits.max_disassembled_module_size),
192                )
193                .context("Failed to initialize disassembler")?
194                .disassemble()
195                .map_err(resource_exhausted)?,
196            ))
197        }
198        .await
199        .transpose()
200    }

282    /// Modules that this module considers friends. These modules can call `public(package)` functions in this module.
283    async fn friends(
284        &self,
285        ctx: &Context<'_>,
286        first: Option<u64>,
287        after: Option<CFriend>,
288        last: Option<u64>,
289        before: Option<CFriend>,
290    ) -> Option<Result<Connection<String, MoveModule>, RpcError>> {
291        async {
292            let pagination: &PaginationConfig = ctx.data()?;
293            let limits = pagination.limits("MoveModule", "friends");
294            let page = Page::from_params(limits, first, after, last, before)?;
295
296            let Some(contents) = self.contents(ctx).await?.as_ref() else {
297                return Ok(None);
298            };
299
300            let bytecode = contents.parsed.bytecode();
301            let runtime_id = *bytecode.self_id().address();
302
303            let friends = bytecode.friend_decls();
304            let conn: Connection<String, MoveModule> =
305                page.paginate_indices(friends.len(), |i| -> Result<_, RpcError> {
306                    let decl = &friends[i];
307                    let friend_pkg = bytecode.address_identifier_at(decl.address);
308                    let friend_mod = bytecode.identifier_at(decl.name);
309
310                    if *friend_pkg != runtime_id {
311                        return Err(anyhow!("Cross-package friend modules").into());
312                    }
313
314                    Ok(MoveModule::with_fq_name(
315                        self.package.clone(),
316                        friend_mod.to_string(),
317                    ))
318                })?;
319            Ok(Some(conn))
320        }
321        .await
322        .transpose()
323    }

487impl MoveModule {
488    /// Construct a module that is represented by just its fully-qualified name. This does not
489    /// check that the module actually exists, so should not be used to "fetch" a module based on
490    /// user input.
491    pub(crate) fn with_fq_name(package: MovePackage, name: String) -> Self {
492        Self {
493            package,
494            name,
495            contents: Arc::new(OnceCell::new()),
496        }
497    }
498
499    /// Get the native CompiledModule, loading it lazily if needed.
500    pub(crate) async fn contents(
501        &self,
502        ctx: &Context<'_>,
503    ) -> Result<&Option<ModuleContents>, RpcError> {
504        self.contents
505            .get_or_try_init(|| async {
506                let (native, parsed) = join!(self.package.native(ctx), self.package.parsed(ctx));
507                let (Some(native), Some(parsed)) = (native?.as_ref(), parsed?.as_ref()) else {
508                    return Ok(None);
509                };
510
511                let Some(native) = native.serialized_module_map().get(&self.name).cloned() else {
512                    return Ok(None);
513                };
514
515                let parsed = parsed
516                    .module(&self.name)
517                    .context("Couldn't find parsed module, for existing native module")?
518                    .clone();
519
520                Ok(Some(ModuleContents { native, parsed }))
521            })
522            .await
523    }
524}

526impl From<Visibility> for MoveVisibility {
527    fn from(visibility: Visibility) -> Self {
528        use MoveVisibility as M;
529        use Visibility as V;
530
531        match visibility {
532            V::Private => M::Private,
533            V::Public => M::Public,
534            V::Friend => M::Friend,
535        }
536    }
