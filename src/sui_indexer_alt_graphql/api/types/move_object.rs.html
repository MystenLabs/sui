121    pub(crate) async fn id(&self) -> Id {
122        let a = self.super_.super_.address;
123        if let Some((v, d)) = self.super_.version_digest {
124            Id::MoveObjectByRef(a, v, d)
125        } else {
126            Id::MoveObjectByAddress(a)
127        }
128    }

202    /// The structured representation of the object's contents.
203    pub(crate) async fn contents(&self, ctx: &Context<'_>) -> Option<Result<MoveValue, RpcError>> {
204        let native = self.native(ctx).await.map(Option::as_ref).transpose()?;
205        Some(native.map(|n| {
206            let scope = self
207                .super_
208                .super_
209                .scope
210                .with_root_version(n.version().value());
211            let type_ = MoveType::from_native(n.type_().clone().into(), scope);
212            MoveValue::new(type_, n.contents().to_owned())
213        }))
214    }

244    /// Dynamic fields owned by this object.
245    ///
246    /// Dynamic fields on wrapped objects can be accessed using `Address.dynamicFields`.
247    pub(crate) async fn dynamic_fields(
248        &self,
249        ctx: &Context<'_>,
250        first: Option<u64>,
251        after: Option<CLive>,
252        last: Option<u64>,
253        before: Option<CLive>,
254    ) -> Option<Result<Connection<String, DynamicField>, RpcError<object::Error>>> {
255        Some(
256            async {
257                let pagination: &PaginationConfig = ctx.data()?;
258                let limits = pagination.limits("IMoveObject", "dynamicFields");
259                let page = Page::from_params(limits, first, after, last, before)?;
260
261                let dynamic_fields = DynamicField::paginate(
262                    ctx,
263                    self.super_.super_.scope.clone(),
264                    self.super_.super_.address.into(),
265                    page,
266                )
267                .await?;
268
269                Ok(dynamic_fields)
270            }
271            .await,
272        )
273    }

295    /// Whether this object can be transfered using the `TransferObjects` Programmable Transaction Command or `sui::transfer::public_transfer`.
296    ///
297    /// Both these operations require the object to have both the `key` and `store` abilities.
298    pub(crate) async fn has_public_transfer(
299        &self,
300        ctx: &Context<'_>,
301    ) -> Option<Result<bool, RpcError>> {
302        async {
303            let Some(native) = self.native(ctx).await?.as_ref() else {
304                return Ok(None);
305            };
306
307            let type_ = MoveType::from_native(
308                native.type_().clone().into(),
309                self.super_.super_.scope.clone(),
310            );
311
312            Ok(type_
313                .abilities_impl()
314                .await?
315                .map(|s| s.has_key() && s.has_store()))
316        }
317        .await
318        .transpose()
319    }

498impl MoveObject {
499    /// Create a `MoveObject` from an `Object` that is assumed to be a `MoveObject`. Its contents
500    /// will be lazily loaded when needed, erroring if the `Object` is not a `MoveObject`.
501    pub(crate) fn from_super(super_: Object) -> Self {
502        Self {
503            super_,
504            native: Arc::new(OnceCell::new()),
505        }
506    }
507
508    /// Try to upcast an `Object` to a `MoveObject`. This function returns `None` if `object`'s
509    /// contents cannot be fetched, or it is not a move object.
510    pub(crate) async fn from_object(
511        object: &Object,
512        ctx: &Context<'_>,
513    ) -> Result<Option<Self>, RpcError> {
514        let Some(super_contents) = object.contents(ctx).await? else {
515            return Ok(None);
516        };
517
518        let Some(native) = super_contents.data.try_as_move() else {
519            return Ok(None);
520        };
521
522        Ok(Some(Self {
523            super_: object.clone(),
524            native: Arc::new(OnceCell::from(Some(native.clone()))),
525        }))
526    }
527
528    /// Get the native MoveObject, loading it lazily if needed.
529    pub(crate) async fn native(
530        &self,
531        ctx: &Context<'_>,
532    ) -> Result<&Option<NativeMoveObject>, RpcError> {
533        self.native
534            .get_or_try_init(async || {
535                let Some(contents) = self.super_.contents(ctx).await? else {
536                    return Ok(None);
537                };
538
539                let native = contents
540                    .data
541                    .try_as_move()
542                    .context("Object is not a MoveObject")?;
543
544                Ok(Some(native.clone()))
545            })
546            .await
547    }
