19/// Abilities are keywords in Sui Move that define how types behave at the compiler level.
20#[derive(Enum, Copy, Clone, Eq, PartialEq)]

109/// Represents instances of concrete types (no type parameters, no references).
110#[Object]
111impl MoveType {
112    /// Flat representation of the type signature, as a displayable string.
113    async fn repr(&self) -> String {
114        self.native.to_canonical_string(/* with_prefix */ true)
115    }
116
117    /// Structured representation of the type signature.
118    async fn signature(&self) -> Result<MoveTypeSignature, RpcError> {
119        MoveTypeSignature::try_from(self.native.clone())
120    }
121
122    /// Structured representation of the "shape" of values that match this type. May return no
123    /// layout if the type is invalid.
124    async fn layout(&self) -> Result<Option<MoveTypeLayout>, RpcError> {
125        let Some(layout) = self.layout_impl().await? else {
126            return Ok(None);
127        };
128
129        Ok(Some(MoveTypeLayout::try_from(layout)?))
130    }
131
132    /// The abilities this concrete type has. Returns no abilities if the type is invalid.
133    async fn abilities(&self) -> Result<Option<Vec<MoveAbility>>, RpcError> {
134        Ok(self.abilities_impl().await?.map(abilities))
135    }
136}

186impl MoveType {
187    /// Read a type from user input. Canonicalizing involves replacing package addresses, to the
188    /// ones that originally defined the type, while in the input a type can be referred to at any
189    /// package at or after the one that first defined it.
190    pub(crate) async fn canonicalize(
191        input: TypeTag,
192        scope: Scope,
193    ) -> Result<Option<Self>, RpcError<Error>> {
194        use ResolverError as RE;
195
196        let canonical = match scope.package_resolver().canonical_type(input.clone()).await {
197            Ok(canonical) => canonical,
198
199            Err(
200                RE::FunctionNotFound(_, _, _)
201                | RE::ModuleNotFound(_, _)
202                | RE::NotAPackage(_)
203                | RE::NotAnIdentifier(_)
204                | RE::PackageNotFound(_)
205                | RE::DatatypeNotFound(_, _, _),
206            ) => return Ok(None),
207
208            Err(
209                err @ (RE::TooManyTypeNodes(_, _)
210                | RE::TooManyTypeParams(_, _)
211                | RE::TypeParamNesting(_, _)
212                | RE::ValueNesting(_)),
213            ) => return Err(resource_exhausted(err)),
214
215            Err(err @ (RE::TypeArityMismatch(_, _) | RE::TypeParamOOB(_, _))) => {
216                return Err(bad_user_input(Error::Resolve(err)));
217            }
218
219            Err(
220                err @ (RE::Bcs(_)
221                | RE::Store { .. }
222                | RE::Deserialize(_)
223                | RE::EmptyPackage(_)
224                | RE::LinkageNotFound(_)
225                | RE::NoTypeOrigin(_, _, _)
226                | RE::UnexpectedReference
227                | RE::UnexpectedSigner
228                | RE::UnexpectedError(_)),
229            ) => {
230                return Err(anyhow!(err)
231                    .context(format!(
232                        "Error canonicalizing type {}",
233                        input.to_canonical_display(/* with_prefix */ true)
234                    ))
235                    .into());
236            }
237        };
238
239        Ok(Some(Self {
240            native: canonical.into(),
241            scope,
242        }))
243    }
244
245    /// Construct a `MoveType` from a native `TypeTag`. Use this when surfacing a stored type i.e.
246    /// not user input.
247    pub(crate) fn from_native(tag: TypeTag, scope: Scope) -> Self {
248        Self {
249            native: tag.into(),
250            scope,
251        }
252    }
253
254    /// Construct a `MoveType` directly from a `TypeInput`. Use this when you already have a
255    /// `TypeInput` (which is MoveType's internal representation) and don't want conversion to fail.
256    pub(crate) fn from_input(input: TypeInput, scope: Scope) -> Self {
257        Self {
258            native: input,
259            scope,
260        }
261    }
262
263    /// Get the native `TypeTag` for this type, if it is valid.
264    pub(crate) fn to_type_tag(&self) -> Option<TypeTag> {
265        self.native.to_type_tag().ok()
266    }
267
268    /// Get the annotated type layout for this type, if it is valid.
269    pub(crate) async fn layout_impl(&self) -> Result<Option<A::MoveTypeLayout>, RpcError> {
270        let Some(tag) = self.to_type_tag() else {
271            return Ok(None);
272        };
273
274        let layout = self
275            .scope
276            .package_resolver()
277            .type_layout(tag)
278            .await
279            .map_err(|err| {
280                internal_resolution_error(err, || {
281                    format!(
282                        "Error calculating layout for {}",
283                        self.native.to_canonical_display(/* with_prefix */ true)
284                    )
285                })
286            })?;
287
288        Ok(Some(layout))
289    }
290
291    /// Get the abilities for this type, if it is valid.
292    pub(crate) async fn abilities_impl(&self) -> Result<Option<AbilitySet>, RpcError> {
293        let Some(tag) = self.to_type_tag() else {
294            return Ok(None);
295        };
296
297        let set = self
298            .scope
299            .package_resolver()
300            .abilities(tag)
301            .await
302            .map_err(|err| {
303                internal_resolution_error(err, || {
304                    format!(
305                        "Error calculating abilities for {}",
306                        self.native.to_canonical_display(/* with_prefix */ true)
307                    )
308                })
309            })?;
310
311        Ok(Some(set))
312    }
313}

315impl TryFrom<TypeInput> for MoveTypeSignature {
316    type Error = RpcError;
317
318    fn try_from(tag: TypeInput) -> Result<Self, RpcError> {
319        use TypeInput as T;
320
321        Ok(match tag {
322            T::Signer => return Err(anyhow!("Unexpected 'signer' type").into()),
323
324            T::U8 => Self::U8,
325            T::U16 => Self::U16,
326            T::U32 => Self::U32,
327            T::U64 => Self::U64,
328            T::U128 => Self::U128,
329            T::U256 => Self::U256,
330
331            T::Bool => Self::Bool,
332            T::Address => Self::Address,
333
334            T::Vector(v) => Self::Vector(Box::new(Self::try_from(*v)?)),
335
336            T::Struct(s) => Self::Datatype {
337                package: s.address.to_canonical_string(/* with_prefix */ true),
338                module: s.module,
339                type_: s.name,
340                type_parameters: s
341                    .type_params
342                    .into_iter()
343                    .map(Self::try_from)
344                    .collect::<Result<Vec<_>, _>>()?,
345            },
346        })
347    }
348}

350impl TryFrom<A::MoveTypeLayout> for MoveTypeLayout {
351    type Error = RpcError;
352
353    fn try_from(layout: A::MoveTypeLayout) -> Result<Self, RpcError> {
354        use A::MoveTypeLayout as TL;
355
356        Ok(match layout {
357            TL::Signer => return Err(anyhow!("Unexpected 'signer' type").into()),
358
359            TL::U8 => Self::U8,
360            TL::U16 => Self::U16,
361            TL::U32 => Self::U32,
362            TL::U64 => Self::U64,
363            TL::U128 => Self::U128,
364            TL::U256 => Self::U256,
365
366            TL::Bool => Self::Bool,
367            TL::Address => Self::Address,
368
369            TL::Vector(v) => Self::Vector(Box::new(Self::try_from(*v)?)),
370            TL::Struct(s) => Self::Struct((*s).try_into()?),
371            TL::Enum(e) => Self::Enum((*e).try_into()?),
372        })
373    }
374}

376impl TryFrom<A::MoveEnumLayout> for MoveEnumLayout {
377    type Error = RpcError;
378
379    fn try_from(layout: A::MoveEnumLayout) -> Result<Self, RpcError> {
380        let A::MoveEnumLayout { variants, .. } = layout;
381        let mut variant_layouts = Vec::new();
382        for ((name, _), variant_fields) in variants {
383            let mut field_layouts = Vec::new();
384            for field in variant_fields {
385                field_layouts.push(MoveFieldLayout::try_from(field)?);
386            }
387            variant_layouts.push(MoveVariantLayout {
388                name: name.to_string(),
389                layout: field_layouts,
390            });
391        }
392
393        Ok(MoveEnumLayout {
394            variants: variant_layouts,
395        })
396    }
397}

425impl From<Ability> for MoveAbility {
426    fn from(ability: Ability) -> Self {
427        use Ability as A;
428        use MoveAbility as M;
429
430        match ability {
431            A::Copy => M::Copy,
432            A::Drop => M::Drop,
433            A::Store => M::Store,
434            A::Key => M::Key,
435        }
436    }

439/// Convert an `AbilitySet` from the binary format into a vector of `MoveAbility` (a GraphQL type).
440pub(crate) fn abilities(set: AbilitySet) -> Vec<MoveAbility> {
441    set.into_iter().map(MoveAbility::from).collect()

444/// Convert a package resolver error into an `RpcError` where the type involved has already been
445/// vetted (it is a checked external input, or it is a type that comes from some stored value).
446fn internal_resolution_error<F, C>(err: ResolverError, context: F) -> RpcError
447where
448    C: fmt::Display + Send + Sync + 'static,
449    F: FnOnce() -> C,
450{
451    use ResolverError as RE;
452
453    match &err {
454        RE::TooManyTypeNodes(_, _)
455        | RE::TooManyTypeParams(_, _)
456        | RE::TypeParamNesting(_, _)
457        | RE::ValueNesting(_) => resource_exhausted(err),
458
459        RE::Bcs(_)
460        | RE::Store { .. }
461        | RE::Deserialize(_)
462        | RE::EmptyPackage(_)
463        | RE::FunctionNotFound(_, _, _)
464        | RE::LinkageNotFound(_)
465        | RE::ModuleNotFound(_, _)
466        | RE::NoTypeOrigin(_, _, _)
467        | RE::NotAPackage(_)
468        | RE::NotAnIdentifier(_)
469        | RE::PackageNotFound(_)
470        | RE::DatatypeNotFound(_, _, _)
471        | RE::TypeArityMismatch(_, _)
472        | RE::TypeParamOOB(_, _)
473        | RE::UnexpectedReference
474        | RE::UnexpectedSigner
475        | RE::UnexpectedError(_) => anyhow!(err).context(context()).into(),
476    }
