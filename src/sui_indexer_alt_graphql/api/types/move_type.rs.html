123/// Represents instances of concrete types (no type parameters, no references).
124#[Object]
125impl MoveType {
126    /// Flat representation of the type signature, as a displayable string.
127    async fn repr(&self) -> String {
128        self.to_type_input().to_canonical_string(true)
129    }
130
131    /// Structured representation of the type signature.
132    async fn signature(&self) -> Result<MoveTypeSignature, RpcError> {
133        MoveTypeSignature::try_from(self.to_type_input())
134    }
135
136    /// Structured representation of the "shape" of values that match this type. May return no
137    /// layout if the type is invalid.
138    async fn layout(&self) -> Option<Result<MoveTypeLayout, RpcError>> {
139        async {
140            let Some(layout) = self.layout_impl().await? else {
141                return Ok(None);
142            };
143
144            Ok(Some(MoveTypeLayout::try_from(layout)?))
145        }
146        .await
147        .transpose()
148    }
149
150    /// The abilities this concrete type has. Returns no abilities if the type is invalid.
151    async fn abilities(&self) -> Option<Result<Vec<MoveAbility>, RpcError>> {
152        Some(Ok(abilities(self.abilities_impl().await.ok()??)))
153    }
154}

204impl MoveType {
205    /// Read a type from user input. Canonicalizing involves replacing package addresses, to the
206    /// ones that originally defined the type, while in the input a type can be referred to at any
207    /// package at or after the one that first defined it.
208    pub(crate) async fn canonicalize(
209        input: TypeTag,
210        scope: Scope,
211    ) -> Result<Option<Self>, RpcError<Error>> {
212        use ResolverError as RE;
213
214        let canonical = match scope.package_resolver().canonical_type(input.clone()).await {
215            Ok(canonical) => canonical,
216
217            Err(
218                RE::FunctionNotFound(_, _, _)
219                | RE::ModuleNotFound(_, _)
220                | RE::NotAPackage(_)
221                | RE::NotAnIdentifier(_)
222                | RE::PackageNotFound(_)
223                | RE::DatatypeNotFound(_, _, _),
224            ) => return Ok(None),
225
226            Err(
227                err @ (RE::TooManyTypeNodes(_, _)
228                | RE::TooManyTypeParams(_, _)
229                | RE::TypeParamNesting(_, _)
230                | RE::ValueNesting(_)),
231            ) => return Err(resource_exhausted(err)),
232
233            Err(err @ (RE::TypeArityMismatch(_, _) | RE::TypeParamOOB(_, _))) => {
234                return Err(bad_user_input(Error::Resolve(err)));
235            }
236
237            Err(
238                err @ (RE::Bcs(_)
239                | RE::Store { .. }
240                | RE::Deserialize(_)
241                | RE::EmptyPackage(_)
242                | RE::LinkageNotFound(_)
243                | RE::NoTypeOrigin(_, _, _)
244                | RE::UnexpectedReference
245                | RE::UnexpectedSigner
246                | RE::UnexpectedError(_)),
247            ) => {
248                return Err(anyhow!(err)
249                    .context(format!(
250                        "Error canonicalizing type {}",
251                        input.to_canonical_display(/* with_prefix */ true)
252                    ))
253                    .into());
254            }
255        };
256
257        Ok(Some(Self {
258            native: Native::Input(canonical.into()),
259            scope,
260        }))
261    }
262
263    /// Construct a `MoveType` from a native `TypeTag`. Use this when surfacing a stored type i.e.
264    /// not user input.
265    pub(crate) fn from_native(tag: TypeTag, scope: Scope) -> Self {
266        Self::from_input(tag.into(), scope)
267    }
268
269    /// Construct a `MoveType` directly from a `TypeInput`. Use this when you already have a
270    /// `TypeInput` (which is one of MoveType's internal representation) and don't want conversion
271    /// to fail.
272    pub(crate) fn from_input(input: TypeInput, scope: Scope) -> Self {
273        Self {
274            native: Native::Input(input),
275            scope,
276        }
277    }
278
279    /// Construct a `MoveType` directly from a `A::MoveTypeLayout`. Use this when you already have
280    /// a type layout, to avoid further type resolution requests from having to call out to the
281    /// package resolver.
282    pub(crate) fn from_layout(layout: A::MoveTypeLayout, scope: Scope) -> Self {
283        Self {
284            native: Native::Layout(layout),
285            scope,
286        }
287    }
288
289    /// Get the native `TypeTag` for this type, if it is valid.
290    pub(crate) fn to_type_tag(&self) -> Option<TypeTag> {
291        match &self.native {
292            Native::Input(input) => input.to_type_tag().ok(),
293            Native::Layout(layout) => Some(layout.into()),
294        }
295    }
296
297    /// Convert the native representation into a `TypeInput`.
298    pub(crate) fn to_type_input(&self) -> TypeInput {
299        match &self.native {
300            Native::Input(input) => input.clone(),
301            Native::Layout(layout) => TypeInput::from(TypeTag::from(layout)),
302        }
303    }
304
305    /// Get the annotated type layout for this type, if it is valid.
306    pub(crate) async fn layout_impl(&self) -> Result<Option<A::MoveTypeLayout>, RpcError> {
307        let input = match &self.native {
308            Native::Layout(layout) => return Ok(Some(layout.clone())),
309            Native::Input(input) => input,
310        };
311
312        let Some(tag) = input.to_type_tag().ok() else {
313            return Ok(None);
314        };
315
316        let layout = self
317            .scope
318            .package_resolver()
319            .type_layout(tag)
320            .await
321            .map_err(|err| {
322                internal_resolution_error(err, || {
323                    format!(
324                        "Error calculating layout for {}",
325                        input.to_canonical_display(/* with_prefix */ true)
326                    )
327                })
328            })?;
329
330        Ok(Some(layout))
331    }
332
333    /// Get the abilities for this type, if it is valid.
334    pub(crate) async fn abilities_impl(&self) -> Result<Option<AbilitySet>, RpcError> {
335        let Some(tag) = self.to_type_tag() else {
336            return Ok(None);
337        };
338
339        let set = self
340            .scope
341            .package_resolver()
342            .abilities(tag.clone())
343            .await
344            .map_err(|err| {
345                internal_resolution_error(err, || {
346                    format!(
347                        "Error calculating abilities for {}",
348                        tag.to_canonical_display(/* with_prefix */ true)
349                    )
350                })
351            })?;
352
353        Ok(Some(set))
354    }
355}

357impl TryFrom<TypeInput> for MoveTypeSignature {
358    type Error = RpcError;
359
360    fn try_from(tag: TypeInput) -> Result<Self, RpcError> {
361        use TypeInput as T;
362
363        Ok(match tag {
364            T::Signer => return Err(anyhow!("Unexpected 'signer' type").into()),
365
366            T::U8 => Self::U8,
367            T::U16 => Self::U16,
368            T::U32 => Self::U32,
369            T::U64 => Self::U64,
370            T::U128 => Self::U128,
371            T::U256 => Self::U256,
372
373            T::Bool => Self::Bool,
374            T::Address => Self::Address,
375
376            T::Vector(v) => Self::Vector(Box::new(Self::try_from(*v)?)),
377
378            T::Struct(s) => Self::Datatype {
379                package: s.address.to_canonical_string(/* with_prefix */ true),
380                module: s.module,
381                type_: s.name,
382                type_parameters: s
383                    .type_params
384                    .into_iter()
385                    .map(Self::try_from)
386                    .collect::<Result<Vec<_>, _>>()?,
387            },
388        })
389    }
390}

392impl TryFrom<A::MoveTypeLayout> for MoveTypeLayout {
393    type Error = RpcError;
394
395    fn try_from(layout: A::MoveTypeLayout) -> Result<Self, RpcError> {
396        use A::MoveTypeLayout as TL;
397
398        Ok(match layout {
399            TL::Signer => return Err(anyhow!("Unexpected 'signer' type").into()),
400
401            TL::U8 => Self::U8,
402            TL::U16 => Self::U16,
403            TL::U32 => Self::U32,
404            TL::U64 => Self::U64,
405            TL::U128 => Self::U128,
406            TL::U256 => Self::U256,
407
408            TL::Bool => Self::Bool,
409            TL::Address => Self::Address,
410
411            TL::Vector(v) => Self::Vector(Box::new(Self::try_from(*v)?)),
412            TL::Struct(s) => Self::Struct((*s).try_into()?),
413            TL::Enum(e) => Self::Enum((*e).try_into()?),
414        })
415    }
416}

418impl TryFrom<A::MoveEnumLayout> for MoveEnumLayout {
419    type Error = RpcError;
420
421    fn try_from(layout: A::MoveEnumLayout) -> Result<Self, RpcError> {
422        let A::MoveEnumLayout { variants, .. } = layout;
423        let mut variant_layouts = Vec::new();
424        for ((name, _), variant_fields) in variants {
425            let mut field_layouts = Vec::new();
426            for field in variant_fields {
427                field_layouts.push(MoveFieldLayout::try_from(field)?);
428            }
429            variant_layouts.push(MoveVariantLayout {
430                name: name.to_string(),
431                layout: field_layouts,
432            });
433        }
434
435        Ok(MoveEnumLayout {
436            variants: variant_layouts,
437        })
438    }
439}

467impl From<Ability> for MoveAbility {
468    fn from(ability: Ability) -> Self {
469        use Ability as A;
470        use MoveAbility as M;
471
472        match ability {
473            A::Copy => M::Copy,
474            A::Drop => M::Drop,
475            A::Store => M::Store,
476            A::Key => M::Key,
477        }
478    }
479}

481/// Convert an `AbilitySet` from the binary format into a vector of `MoveAbility` (a GraphQL type).
482pub(crate) fn abilities(set: AbilitySet) -> Vec<MoveAbility> {
483    set.into_iter().map(MoveAbility::from).collect()

486/// Convert a package resolver error into an `RpcError` where the type involved has already been
487/// vetted (it is a checked external input, or it is a type that comes from some stored value).
488fn internal_resolution_error<F, C>(err: ResolverError, context: F) -> RpcError
489where
490    C: fmt::Display + Send + Sync + 'static,
491    F: FnOnce() -> C,
492{
493    use ResolverError as RE;
494
495    match &err {
496        RE::TooManyTypeNodes(_, _)
497        | RE::TooManyTypeParams(_, _)
498        | RE::TypeParamNesting(_, _)
499        | RE::ValueNesting(_) => resource_exhausted(err),
500
501        RE::Bcs(_)
502        | RE::Store { .. }
503        | RE::Deserialize(_)
504        | RE::EmptyPackage(_)
505        | RE::FunctionNotFound(_, _, _)
506        | RE::LinkageNotFound(_)
507        | RE::ModuleNotFound(_, _)
508        | RE::NoTypeOrigin(_, _, _)
509        | RE::NotAPackage(_)
510        | RE::NotAnIdentifier(_)
511        | RE::PackageNotFound(_)
512        | RE::DatatypeNotFound(_, _, _)
513        | RE::TypeArityMismatch(_, _)
514        | RE::TypeParamOOB(_, _)
515        | RE::UnexpectedReference
516        | RE::UnexpectedSigner
517        | RE::UnexpectedError(_) => anyhow!(err).context(context()).into(),
518    }
