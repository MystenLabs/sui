22#[derive(Clone)]
23pub(crate) struct MoveValue {
24    pub(crate) type_: MoveType,
25    pub(crate) native: Vec<u8>,
26}
27
28#[derive(thiserror::Error, Debug)]
29#[error("Move value is too big")]
30pub(crate) struct MoveValueTooBigError;
31
32#[Object]
33impl MoveValue {
34    /// The BCS representation of this value, Base64-encoded.
35    async fn bcs(&self) -> Option<Base64> {
36        Some(Base64::from(self.native.clone()))
37    }
38
39    /// A rendered JSON blob based on an on-chain template, substituted with data from this value.
40    ///
41    /// Returns `null` if the value's type does not have an associated `Display` template.
42    async fn display(&self, ctx: &Context<'_>) -> Result<Option<Display>, RpcError> {
43        let limits: &Limits = ctx.data()?;
44        let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
45
46        let Some(TypeTag::Struct(type_)) = self.type_.to_type_tag() else {
47            return Ok(None);
48        };
49
50        let (layout, display) = join!(
51            self.type_.layout_impl(),
52            pg_loader.load_one(DisplayKey(*type_)),
53        );
54
55        let (Some(layout), Some(display)) = (layout?, display.context("Failed to fetch Display")?)
56        else {
57            return Ok(None);
58        };
59
60        let event: DisplayVersionUpdatedEvent = bcs::from_bytes(&display.display)
61            .context("Failed to deserialize DisplayVersionUpdatedEvent")?;
62
63        let mut output = IndexMap::new();
64        let mut errors = IndexMap::new();
65
66        for (field, value) in
67            sui_display::v1::Format::parse(limits.max_display_field_depth, &event.fields)
68                .map_err(resource_exhausted)?
69                .display(limits.max_display_output_size, &self.native, &layout)
70                .map_err(resource_exhausted)?
71        {
72            match value {
73                Ok(v) => {
74                    output.insert(Name::new(&field), Value::String(v));
75                }
76
77                Err(e) => {
78                    output.insert(Name::new(&field), Value::Null);
79                    errors.insert(Name::new(&field), Value::String(e.to_string()));
80                }
81            };
82        }
83
84        Ok(Some(Display {
85            output: (!output.is_empty()).then(|| Json::from(Value::from(output))),
86            errors: (!errors.is_empty()).then(|| Json::from(Value::from(errors))),
87        }))
88    }
89
90    /// Representation of a Move value in JSON, where:
91    ///
92    /// - Addresses, IDs, and UIDs are represented in canonical form, as JSON strings.
93    /// - Bools are represented by JSON boolean literals.
94    /// - u8, u16, and u32 are represented as JSON numbers.
95    /// - u64, u128, and u256 are represented as JSON strings.
96    /// - Balances, Strings, and Urls are represented as JSON strings.
97    /// - Vectors of bytes are represented as Base64 blobs, and other vectors are represented by JSON arrays.
98    /// - Structs are represented by JSON objects.
99    /// - Enums are represented by JSON objects, with a field named `@variant` containing the variant name.
100    /// - Empty optional values are represented by `null`.
101    async fn json(&self, ctx: &Context<'_>) -> Result<Option<Json>, RpcError> {
102        let limits: &Limits = ctx.data()?;
103
104        let Some(layout) = self.type_.layout_impl().await? else {
105            return Ok(None);
106        };
107
108        let value = ProtoVisitorBuilder::new(limits.max_move_value_bound)
109            .deserialize_value(&self.native, &layout)
110            .map_err(|_| resource_exhausted(MoveValueTooBigError))?;
111
112        Ok(Some(Json::from(proto_to_json(value))))
113    }
114
115    /// The value's type.
116    async fn type_(&self) -> Option<MoveType> {
117        Some(self.type_.clone())
118    }
119}
120
121impl MoveValue {
122    pub(crate) fn new(type_: MoveType, native: Vec<u8>) -> Self {
123        Self { type_, native }
124    }
125}
126
127/// Convert a Protobuf value into a GraphQL JSON value.
128fn proto_to_json(proto: proto::Value) -> async_graphql::Value {
129    match proto.kind {
130        Some(Kind::NullValue(_)) | None => async_graphql::Value::Null,
131        Some(Kind::BoolValue(b)) => async_graphql::Value::Boolean(b),
132        Some(Kind::StringValue(s)) => async_graphql::Value::String(s),
133
134        // The [`ProtoVisitor`] only produces numbers for `u8`, `u16`, and `u32` values, so they
135        // can be encoded as a whole number in JSON without loss of precision by conversion to
136        // `u32`.
137        Some(Kind::NumberValue(n)) => async_graphql::Value::Number(Number::from(n as u32)),
138
139        Some(Kind::StructValue(map)) => async_graphql::Value::Object(
140            map.fields
141                .into_iter()
142                .map(|(k, v)| (Name::new(k), proto_to_json(v)))
143                .collect(),
144        ),
145
146        Some(Kind::ListValue(list)) => {
147            async_graphql::Value::List(list.values.into_iter().map(proto_to_json).collect())
148        }
149    }
150}
