103    /// Attempts to treat this value as an `Address`.
104    ///
105    /// If the value is of type `address` or `0x2::object::ID`, it is interpreted as an address pointer, and it is scoped to the current checkpoint.
106    ///
107    /// If the value is of type `0x2::object::UID`, it is interpreted as a wrapped object whose version is bounded by the root version of the current value. Such values do not support nested owned object queries, but `Address.addressAt` can be used to re-scope it to a checkpoint (defaults to the current checkpoint), instead of a root version, allowing owned object queries.
108    ///
109    /// Values of other types cannot be interpreted as addresses, and `null` is returned.
110    async fn as_address(&self) -> Option<Result<Address, RpcError>> {
111        use TypeTag as T;
112        let tag = self.type_.to_type_tag()?;
113
114        async {
115            let address = match tag {
116                T::Address => Address::with_address(
117                    self.type_.scope.without_root_bound(),
118                    bcs::from_bytes(&self.native)?,
119                ),
120
121                T::Struct(s) if *s == ID::type_() => Address::with_address(
122                    self.type_.scope.without_root_bound(),
123                    bcs::from_bytes(&self.native)?,
124                ),
125
126                T::Struct(s) if *s == UID::type_() => {
127                    Address::with_address(self.type_.scope.clone(), bcs::from_bytes(&self.native)?)
128                }
129
130                _ => return Ok(None),
131            };
132
133            Ok(Some(address))
134        }
135        .await
136        .transpose()
137    }

139    /// Attempts to treat this value as a `vector<T>` and paginate over its elements.
140    ///
141    /// Values of other types cannot be interpreted as vectors, and `null` is returned.
142    async fn as_vector(
143        &self,
144        ctx: &Context<'_>,
145        first: Option<u64>,
146        after: Option<CVector>,
147        last: Option<u64>,
148        before: Option<CVector>,
149    ) -> Option<Result<Connection<String, MoveValue>, RpcError>> {
150        struct Visitor<'p, 's> {
151            page: &'p Page<CVector>,
152            scope: &'s Scope,
153        }
154
155        impl AV::Visitor<'_, '_> for Visitor<'_, '_> {
156            type Value = Option<Connection<String, MoveValue>>;
157            type Error = AV::Error;
158
159            visitor_default! { <'_, '_> u8, u16, u32, u64, u128, u256 = Ok(None) }
160            visitor_default! { <'_, '_> bool, address, signer, struct, variant = Ok(None) }
161
162            fn visit_vector(
163                &mut self,
164                driver: &mut AV::VecDriver<'_, '_, '_>,
165            ) -> Result<Self::Value, Self::Error> {
166                let mut conn = Connection::new(false, false);
167
168                let total = driver.len() as usize;
169                let Some(range) = self.page.range(total) else {
170                    return Ok(Some(conn));
171                };
172
173                conn.has_previous_page = 0 < range.start;
174                conn.has_next_page = range.end < total;
175
176                let layout = driver.element_layout().clone();
177                let type_ = MoveType::from_layout(layout, self.scope.clone());
178
179                for i in 0..total {
180                    let start = driver.position();
181                    driver.skip_element()?;
182                    let end = driver.position();
183
184                    if range.contains(&i) {
185                        conn.edges.push(Edge::new(
186                            JsonCursor::new(i).encode_cursor(),
187                            MoveValue::new(type_.clone(), driver.bytes()[start..end].to_vec()),
188                        ));
189                    }
190                }
191
192                Ok(Some(conn))
193            }
194        }
195
196        async {
197            let pagination: &PaginationConfig = ctx.data()?;
198            let limits = pagination.limits("MoveValue", "asVector");
199            let page = Page::from_params(limits, first, after, last, before)?;
200
201            if !matches!(self.type_.to_type_tag(), Some(TypeTag::Vector(_))) {
202                return Ok(None);
203            }
204
205            let Some(layout) = self.type_.layout_impl().await? else {
206                return Ok(None);
207            };
208
209            let mut visitor = Visitor {
210                page: &page,
211                scope: &self.type_.scope,
212            };
213
214            Ok(
215                A::MoveValue::visit_deserialize(&self.native, &layout, &mut visitor)
216                    .context("Failed to deserialize vector")?,
217            )
218        }
219        .await
220        .transpose()
221    }

224    async fn bcs(&self) -> Option<Base64> {
225        Some(Base64::from(self.native.clone()))

228    /// A rendered JSON blob based on an on-chain template, substituted with data from this value.
229    ///
230    /// Returns `null` if the value's type does not have an associated `Display` template.
231    async fn display(&self, ctx: &Context<'_>) -> Option<Result<Display, RpcError>> {
232        async {
233            let limits: &Limits = ctx.data()?;
234            let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
235
236            let Some(TypeTag::Struct(type_)) = self.type_.to_type_tag() else {
237                return Ok(None);
238            };
239
240            let (layout, display) = join!(
241                self.type_.layout_impl(),
242                pg_loader.load_one(DisplayKey(*type_)),
243            );
244
245            let (Some(layout), Some(display)) =
246                (layout?, display.context("Failed to fetch Display")?)
247            else {
248                return Ok(None);
249            };
250
251            let event: DisplayVersionUpdatedEvent = bcs::from_bytes(&display.display)
252                .context("Failed to deserialize DisplayVersionUpdatedEvent")?;
253
254            let mut output = IndexMap::new();
255            let mut errors = IndexMap::new();
256
257            for (field, value) in
258                sui_display::v1::Format::parse(limits.max_display_field_depth, &event.fields)
259                    .map_err(resource_exhausted)?
260                    .display(limits.max_display_output_size, &self.native, &layout)
261                    .map_err(resource_exhausted)?
262            {
263                match value {
264                    Ok(v) => {
265                        output.insert(Name::new(&field), Value::String(v));
266                    }
267
268                    Err(e) => {
269                        output.insert(Name::new(&field), Value::Null);
270                        errors.insert(Name::new(&field), Value::String(e.to_string()));
271                    }
272                };
273            }
274
275            Ok(Some(Display {
276                output: (!output.is_empty()).then(|| Json::from(Value::from(output))),
277                errors: (!errors.is_empty()).then(|| Json::from(Value::from(errors))),
278            }))
279        }
280        .await
281        .transpose()
282    }

284    /// Extract a nested value at the given path.
285    ///
286    /// `path` is a Display v2 'chain' expression, allowing access to nested, named and positional fields, vector indices, VecMap keys, and dynamic (object) field accesses.
287    async fn extract(
288        &self,
289        ctx: &Context<'_>,
290        path: String,
291    ) -> Option<Result<MoveValue, RpcError<Error>>> {
292        async {
293            let limits: &Limits = ctx.data()?;
294            let extract = sui_display::v2::Extract::parse(limits.display(), &path)
295                .map_err(|e| format_error(Error::Path, e))?;
296
297            let Some(layout) = self.type_.layout_impl().await.map_err(upcast)? else {
298                return Ok(None);
299            };
300
301            // Create a store for dynamic field resolution
302            let store = DisplayStore::new(ctx, &self.type_.scope);
303
304            // Create an interpreter that combines the root value with the store
305            let root = sui_display::v2::OwnedSlice {
306                bytes: self.native.clone(),
307                layout,
308            };
309
310            // Evaluate the extraction and convert to an owned slice
311            let interpreter = sui_display::v2::Interpreter::new(root, store);
312            let Some(value) = extract
313                .extract(&interpreter)
314                .await
315                .map_err(|e| format_error(Error::Path, e))?
316            else {
317                return Ok(None);
318            };
319
320            let Some(sui_display::v2::OwnedSlice {
321                layout,
322                bytes: native,
323            }) = value.into_owned_slice()
324            else {
325                return Err(bad_user_input(Error::NotASlice));
326            };
327
328            let type_ = MoveType::from_layout(layout, self.type_.scope.clone());
329            Ok(Some(MoveValue { type_, native }))
330        }
331        .await
332        .transpose()
333    }

335    /// Render a single Display v2 format string against this value.
336    ///
337    /// Returns `null` if the value does not have a valid type, or if any of the expressions in the format string fail to evaluate (e.g. field does not exist).
338    async fn format(
339        &self,
340        ctx: &Context<'_>,
341        format: String,
342    ) -> Option<Result<Json, RpcError<Error>>> {
343        async {
344            let limits: &Limits = ctx.data()?;
345            let parsed = sui_display::v2::Format::parse(limits.display(), &format)
346                .map_err(|e| format_error(Error::Format, e))?;
347
348            let Some(layout) = self.type_.layout_impl().await.map_err(upcast)? else {
349                return Ok(None);
350            };
351
352            let store = DisplayStore::new(ctx, &self.type_.scope);
353            let root = sui_display::v2::OwnedSlice {
354                bytes: self.native.clone(),
355                layout,
356            };
357
358            let interpreter = sui_display::v2::Interpreter::new(root, store);
359            let value = parsed
360                .format(
361                    &interpreter,
362                    limits.max_move_value_depth,
363                    limits.max_display_output_size,
364                )
365                .await
366                .map_err(|e| format_error(Error::Format, e))?;
367
368            Ok(Some(Json::try_from(value).map_err(upcast)?))
369        }
370        .await
371        .transpose()
372    }

374    /// Representation of a Move value in JSON, where:
375    ///
376    /// - Addresses, IDs, and UIDs are represented in canonical form, as JSON strings.
377    /// - Bools are represented by JSON boolean literals.
378    /// - u8, u16, and u32 are represented as JSON numbers.
379    /// - u64, u128, and u256 are represented as JSON strings.
380    /// - Balances, Strings, and Urls are represented as JSON strings.
381    /// - Vectors of bytes are represented as Base64 blobs, and other vectors are represented by JSON arrays.
382    /// - Structs are represented by JSON objects.
383    /// - Enums are represented by JSON objects, with a field named `@variant` containing the variant name.
384    /// - Empty optional values are represented by `null`.
385    async fn json(&self, ctx: &Context<'_>) -> Option<Result<Json, RpcError>> {
386        async {
387            let limits: &Limits = ctx.data()?;
388
389            let Some(layout) = self.type_.layout_impl().await? else {
390                return Ok(None);
391            };
392
393            let value = JsonVisitor::new(limits)
394                .deserialize_value(&self.native, &layout)
395                .map_err(|e| match &e {
396                    VisitorError::Visitor(_) | VisitorError::UnexpectedType => anyhow!(e).into(),
397                    VisitorError::TooBig | VisitorError::TooDeep => resource_exhausted(e),
398                })?;
399
400            Ok(Some(Json::try_from(value)?))
401        }
402        .await
403        .transpose()
404    }

406    /// The value's type.

408        Some(self.type_.clone())
409    }

448impl JsonWriter<'_> {
449    fn debit(&mut self, size: usize) -> Result<(), VisitorError> {
450        if *self.size_budget < size {
451            return Err(VisitorError::TooBig);
452        }
453
454        *self.size_budget -= size;
455        Ok(())
456    }

459#[async_trait]
460impl<'f, 'r> sui_display::v2::Store for DisplayStore<'f, 'r> {
461    async fn object(
462        &self,
463        id: AccountAddress,
464    ) -> anyhow::Result<Option<sui_display::v2::OwnedSlice>> {
465        // NOTE: We can't use `anyhow::Context` here because `RpcError` doesn't implement
466        // `std::error::Error`.
467        let object = Object::latest(self.ctx, self.scope.clone(), id.into())
468            .await
469            .map_err(|e| anyhow!("Failed to fetch object: {e:?}"))?;
470
471        let Some(object) = object else {
472            return Ok(None);
473        };
474
475        let Some(native) = object
476            .contents(self.ctx)
477            .await
478            .map_err(|e| anyhow!("Failed to get object contents: {e:?}"))?
479        else {
480            return Ok(None);
481        };
482
483        let Some(move_object) = native.data.try_as_move() else {
484            return Ok(None);
485        };
486
487        let type_ = MoveType::from_native(
488            move_object.type_().clone().into(),
489            object.super_.scope.clone(),
490        );
491
492        let Some(layout) = type_
493            .layout_impl()
494            .await
495            .map_err(|e| anyhow!("Failed to get layout: {e:?}"))?
496        else {
497            return Ok(None);
498        };
499
500        let bytes = move_object.contents().to_owned();
501        Ok(Some(sui_display::v2::OwnedSlice { layout, bytes }))
502    }
503}

505impl RV::Writer for JsonWriter<'_> {
506    type Value = serde_json::Value;
507    type Error = VisitorError;
508
509    type Vec = Vec<serde_json::Value>;
510    type Map = serde_json::Map<String, serde_json::Value>;
511
512    type Nested<'b>
513        = JsonWriter<'b>
514    where
515        Self: 'b;
516
517    fn nest(&mut self) -> Result<Self::Nested<'_>, Self::Error> {
518        if self.depth_budget == 0 {
519            return Err(VisitorError::TooDeep);
520        }
521
522        Ok(JsonWriter {
523            size_budget: self.size_budget,
524            depth_budget: self.depth_budget - 1,
525        })
526    }
527
528    fn write_null(&mut self) -> Result<Self::Value, Self::Error> {
529        self.debit("null".len())?;
530        Ok(serde_json::Value::Null)
531    }
532
533    fn write_bool(&mut self, value: bool) -> Result<Self::Value, Self::Error> {
534        self.debit(if value { "true".len() } else { "false".len() })?;
535        Ok(serde_json::Value::Bool(value))
536    }
537
538    fn write_number(&mut self, value: u32) -> Result<Self::Value, Self::Error> {
539        self.debit(if value == 0 { 1 } else { value.ilog10() } as usize)?;
540        Ok(serde_json::Value::Number(value.into()))
541    }
542
543    fn write_str(&mut self, value: String) -> Result<Self::Value, Self::Error> {
544        // Account for the quotes around the string.
545        self.debit(2 + value.len())?;
546        Ok(serde_json::Value::String(value))
547    }
548
549    fn write_vec(&mut self, value: Self::Vec) -> Result<Self::Value, Self::Error> {
550        // Account for the opening bracket.
551        self.debit(1)?;
552        Ok(serde_json::Value::Array(value))
553    }
554
555    fn write_map(&mut self, value: Self::Map) -> Result<Self::Value, Self::Error> {
556        // Account for the opening brace.
557        self.debit(1)?;
558        Ok(serde_json::Value::Object(value))
559    }
560
561    fn vec_push_element(
562        &mut self,
563        vec: &mut Self::Vec,
564        val: Self::Value,
565    ) -> Result<(), Self::Error> {
566        // Account for comma (or closing bracket).
567        self.debit(1)?;
568        vec.push(val);
569        Ok(())
570    }
571
572    fn map_push_field(
573        &mut self,
574        map: &mut Self::Map,
575        key: String,
576        val: Self::Value,
577    ) -> Result<(), Self::Error> {
578        // Account for quotes, colon, and comma (or closing brace).
579        self.debit(4 + key.len())?;
580        map.insert(key, val);
581        Ok(())
582    }
583}

597fn format_error(
598    wrap: impl FnOnce(sui_display::v2::FormatError) -> Error,
599    e: sui_display::v2::FormatError,
600) -> RpcError<Error> {
601    use sui_display::v2::FormatError as FE;
602    match &e {
603        FE::InvalidHexCharacter(_)
604        | FE::InvalidIdentifier(_)
605        | FE::InvalidNumber { .. }
606        | FE::OddHexLiteral(_)
607        | FE::TransformInvalid(_)
608        | FE::TransformInvalid_ { .. }
609        | FE::UnexpectedEos { .. }
610        | FE::UnexpectedRemaining(_)
611        | FE::UnexpectedToken { .. }
612        | FE::VectorArity { .. }
613        | FE::VectorNoType
614        | FE::VectorTypeMismatch { .. } => bad_user_input(wrap(e)),
615
616        FE::TooBig | FE::TooDeep | FE::TooManyLoads | FE::TooMuchOutput => resource_exhausted(e),
617        FE::Bcs(_) | FE::Visitor(_) | FE::Store(_) => anyhow!(e).into(),
618    }
