16/// An interface implemented by types that can be uniquely identified by a globally unique `ID`, following the GraphQL Global Object Identification specification.
17#[derive(Interface)]
18#[graphql(
19    name = "Node",
20    field(
21        name = "id",
22        ty = "Id",
23        desc = "The node's globally unique identifier, which can be passed to `Query.node` to refetch it."
24    )
25)]
26pub(crate) enum Node {
27    Address(Box<Address>),
28    Checkpoint(Box<Checkpoint>),
29    DynamicField(Box<DynamicField>),
30    Epoch(Box<Epoch>),
31    MoveObject(Box<MoveObject>),
32    MovePackage(Box<MovePackage>),
33    Object(Box<Object>),
34    Transaction(Box<Transaction>),
35}
