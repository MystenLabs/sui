14/// A filter over the live object set, the filter can be one of:
15///
16/// - A filter on type (all live objects whose type matches that filter).
17/// - Fetching all objects owned by an address or object, optionally filtered by type.
18/// - Fetching all shared or immutable objects, filtered by type.
19#[derive(InputObject, Default, Debug, Clone, Eq, PartialEq)]
20pub(crate) struct ObjectFilter {
21    /// Filter on whether the object is address-owned, object-owned, shared, or immutable.
22    ///
23    /// - If this field is set to "ADDRESS" or "OBJECT", then an owner filter must also be provided.
24    /// - If this field is set to "SHARED" or "IMMUTABLE", then a type filter must also be provided.
25    pub owner_kind: Option<OwnerKind>,
26
27    /// Specifies the address of the owning address or object.
28    ///
29    /// This field is required if `ownerKind` is "ADDRESS" or "OBJECT". If provided without `ownerKind`, `ownerKind` defaults to "ADDRESS".
30    pub owner: Option<SuiAddress>,
31
32    /// Filter on the object's type.
33    ///
34    /// The filter can be one of:
35    ///
36    /// - A package address: `0x2`,
37    /// - A module: `0x2::coin`,
38    /// - A fully-qualified name: `0x2::coin::Coin`,
39    /// - A type instantiation: `0x2::coin::Coin<0x2::sui::SUI>`.
40    pub type_: Option<TypeFilter>,
41}
42
43#[derive(Default)]
44pub(crate) struct ObjectFilterValidator {
45    /// Whether to allow an empty filter on input.
46    allow_empty: bool,
47}
48
49impl ObjectFilterValidator {
50    /// Create a validator that allows empty filters.
51    pub(crate) fn allows_empty() -> Self {
52        Self { allow_empty: true }
53    }
54}
55
56impl ObjectFilter {
57    /// Try to create a filter whose results are the intersection of `self`'s results and `other`'s
58    /// results. May return `None` if the filters are incompatible (would result in no matches)
59    pub(crate) fn intersect(self, other: Self) -> Option<Self> {
60        macro_rules! intersect {
61            ($field:ident, $body:expr) => {
62                intersect::field(self.$field, other.$field, $body)
63            };
64        }
65
66        Some(Self {
67            owner_kind: intersect!(owner_kind, intersect::by_eq)?,
68            owner: intersect!(owner, intersect::by_eq)?,
69            type_: intersect!(type_, TypeFilter::intersect)?,
70        })
71    }
72}
73
74impl CustomValidator<ObjectFilter> for ObjectFilterValidator {
75    fn check(&self, filter: &ObjectFilter) -> Result<(), InputValueError<ObjectFilter>> {
76        match filter {
77            ObjectFilter {
78                owner_kind: Some(kind @ (OwnerKind::Address | OwnerKind::Object)),
79                owner: None,
80                type_: _,
81            } => Err(InputValueError::custom(format!(
82                "{kind} owner kind requires an `owner` to be specified"
83            ))),
84
85            ObjectFilter {
86                owner_kind: Some(kind @ (OwnerKind::Shared | OwnerKind::Immutable)),
87                owner: Some(_),
88                type_: _,
89            } => Err(InputValueError::custom(format!(
90                "Unexpected `owner` for {kind} owner kind",
91            ))),
92
93            ObjectFilter {
94                owner_kind: Some(kind @ (OwnerKind::Shared | OwnerKind::Immutable)),
95                owner: None,
96                type_: None,
97            } => Err(InputValueError::custom(format!(
98                "{kind} owner kind requires a `type` to be specified"
99            ))),
100
101            ObjectFilter {
102                owner_kind: None,
103                owner: None,
104                type_: None,
105            } => {
106                if self.allow_empty {
107                    Ok(())
108                } else {
109                    Err(InputValueError::custom("No `ObjectFilter` specified"))
110                }
111            }
112
113            // Valid address/object owner filter
114            ObjectFilter {
115                owner_kind: None | Some(OwnerKind::Address | OwnerKind::Object),
116                owner: Some(_),
117                type_: _,
118            } => Ok(()),
119
120            // Valid shared/immutable owner filter
121            ObjectFilter {
122                owner_kind: Some(OwnerKind::Shared | OwnerKind::Immutable),
123                owner: None,
124                type_: Some(_),
125            } => Ok(()),
126
127            // Valid type-only filter
128            ObjectFilter {
129                owner_kind: None,
130                owner: None,
131                type_: Some(_),
132            } => Ok(()),
133        }
134    }
135}

137#[cfg(test)]
138mod tests {
139    use super::*;
140
141    #[test]
142    fn test_default_intersection() {
143        let f = ObjectFilter {
144            owner_kind: Some(OwnerKind::Object),
145            owner: Some("0x1".parse().unwrap()),
146            type_: Some("0x2::coin".parse().unwrap()),
147        };
148
149        assert_eq!(ObjectFilter::default().intersect(f.clone()).unwrap(), f);
150        assert_eq!(f.clone().intersect(ObjectFilter::default()).unwrap(), f);
151    }
152
153    #[test]
154    fn test_owner_kind_intersection() {
155        let a = ObjectFilter {
156            owner_kind: Some(OwnerKind::Address),
157            owner: None,
158            type_: None,
159        };
160
161        let o = ObjectFilter {
162            owner_kind: Some(OwnerKind::Object),
163            owner: None,
164            type_: None,
165        };
166
167        // Same owner_kind should intersect successfully
168        assert_eq!(a.clone().intersect(a.clone()).unwrap(), a);
169        assert_eq!(o.clone().intersect(o.clone()).unwrap(), o);
170
171        // Different owner_kind should return None
172        assert!(a.clone().intersect(o.clone()).is_none());
173        assert!(o.intersect(a).is_none());
174    }
175
176    #[test]
177    fn test_owner_intersection() {
178        let a1 = ObjectFilter {
179            owner_kind: None,
180            owner: Some("0x1".parse().unwrap()),
181            type_: None,
182        };
183
184        let a2 = ObjectFilter {
185            owner_kind: None,
186            owner: Some("0x2".parse().unwrap()),
187            type_: None,
188        };
189
190        // Same owner should intersect successfully
191        assert_eq!(a1.clone().intersect(a1.clone()).unwrap(), a1);
192        assert_eq!(a2.clone().intersect(a2.clone()).unwrap(), a2);
193
194        // Different owner_kind should return None
195        assert!(a1.clone().intersect(a2.clone()).is_none());
196        assert!(a2.intersect(a1).is_none());
197    }
198
199    #[test]
200    fn test_type_filter_intersection() {
201        let coin = ObjectFilter {
202            owner_kind: None,
203            owner: None,
204            type_: Some("0x2::coin::Coin".parse().unwrap()),
205        };
206
207        let sui = ObjectFilter {
208            owner_kind: None,
209            owner: None,
210            type_: Some("0x2::coin::Coin<0x2::sui::SUI>".parse().unwrap()),
211        };
212
213        let token = ObjectFilter {
214            owner_kind: None,
215            owner: None,
216            type_: Some("0x2::token::Token".parse().unwrap()),
217        };
218
219        // Compatible type filters intersect to become the more specific filter, regardless of
220        // intersection order.
221        assert_eq!(coin.clone().intersect(sui.clone()).unwrap(), sui);
222        assert_eq!(sui.clone().intersect(coin.clone()).unwrap(), sui);
223
224        // Incompatible filters don't intersect
225        assert!(coin.clone().intersect(token.clone()).is_none());
226        assert!(token.intersect(coin).is_none())
227    }

229    #[test]
230    fn test_combined_intersection() {
231        let a_coin = ObjectFilter {
232            owner_kind: Some(OwnerKind::Address),
233            owner: Some("0x1".parse().unwrap()),
234            type_: Some("0x2::coin::Coin".parse().unwrap()),
235        };
236
237        let a_sui = ObjectFilter {
238            owner_kind: Some(OwnerKind::Address),
239            owner: Some("0x1".parse().unwrap()),
240            type_: Some("0x2::coin::Coin<0x2::sui::SUI>".parse().unwrap()),
241        };
242
243        let o_coin = ObjectFilter {
244            owner_kind: Some(OwnerKind::Object),
245            owner: Some("0x1".parse().unwrap()),
246            type_: Some("0x2::coin::Coin".parse().unwrap()),
247        };
248
249        assert_eq!(a_coin.clone().intersect(a_sui.clone()).unwrap(), a_sui);
250        assert!(a_coin.clone().intersect(o_coin.clone()).is_none());
251    }
