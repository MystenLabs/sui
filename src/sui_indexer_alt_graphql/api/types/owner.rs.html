4use async_graphql::SimpleObject;
5use async_graphql::Union;
6use sui_types::object::Owner as NativeOwner;
7
8use crate::api::scalars::uint53::UInt53;
9use crate::api::types::address::Address;
10use crate::scope::Scope;
11
12/// The object's owner kind.
13#[derive(Union, Clone)]
14pub(crate) enum Owner {
15    Address(AddressOwner),
16    Object(ObjectOwner),
17    Shared(Shared),
18    Immutable(Immutable),
19    ConsensusAddress(ConsensusAddressOwner),
20}
21
22/// Object is exclusively owned by a single address, and is mutable.
23#[derive(SimpleObject, Clone)]
24pub(crate) struct AddressOwner {
25    /// The owner's address.
26    address: Option<Address>,
27}
28
29/// Object is exclusively owned by a single object, and is mutable. Note that the owning object may be inaccessible because it is wrapped.
30#[derive(SimpleObject, Clone)]
31pub(crate) struct ObjectOwner {
32    /// The owner's address.
33    address: Option<Address>,
34}
35
36/// Object is shared, can be used by any address, and is mutable.
37#[derive(SimpleObject, Clone)]
38pub(crate) struct Shared {
39    /// The version at which the object became shared.
40    initial_shared_version: Option<UInt53>,
41}
42
43/// Object is accessible to all addresses, and is immutable.
44#[derive(SimpleObject, Clone)]
45pub(crate) struct Immutable {
46    #[graphql(name = "_")]
47    dummy: Option<bool>,
48}
49
50/// Object is exclusively owned by a single adderss and sequenced via consensus.
51#[derive(SimpleObject, Clone)]
52pub(crate) struct ConsensusAddressOwner {
53    /// The version at which the object most recently bcame a consensus object. This serves the same function as `Shared.initialSharedVersion`, except it may change if the object's `owner` type changes.
54    start_version: Option<UInt53>,
55
56    /// The owner's address.
57    address: Option<Address>,
58}
59
60impl Owner {
61    pub(crate) fn from_native(scope: Scope, native: NativeOwner) -> Self {
62        use NativeOwner as NO;
63        use Owner as O;
64
65        match native {
66            NO::AddressOwner(a) => O::Address(AddressOwner {
67                address: Some(Address::with_address(scope.without_root_bound(), a)),
68            }),
69
70            NO::ObjectOwner(a) => O::Object(ObjectOwner {
71                address: Some(Address::with_address(scope, a)),
72            }),
73
74            NO::Shared {
75                initial_shared_version,
76            } => O::Shared(Shared {
77                initial_shared_version: Some(initial_shared_version.into()),
78            }),
79
80            NO::Immutable => O::Immutable(Immutable { dummy: None }),
81
82            NO::ConsensusAddressOwner {
83                start_version,
84                owner,
85            } => O::ConsensusAddress(ConsensusAddressOwner {
86                start_version: Some(start_version.into()),
87                address: Some(Address::with_address(scope.without_root_bound(), owner)),
88            }),
89        }
90    }
91}
