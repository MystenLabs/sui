29/// A protocol configuration that can hold an arbitrary value (or no value at all).
30#[derive(Clone, SimpleObject)]
31pub(crate) struct ProtocolConfig {
32    /// Configuration name.
33    pub key: String,
34
35    /// Configuration value.
36    pub value: Option<String>,
37}
38
39/// A boolean protocol configuration.
40#[derive(Clone, SimpleObject)]
41pub(crate) struct FeatureFlag {
42    /// Feature flag name.
43    pub key: String,
44
45    /// Feature flag value.
46    pub value: bool,
47}
48
49#[derive(Clone)]
50struct ConfigContent(BTreeMap<String, Option<String>>);
51
52#[derive(Clone)]
53struct FlagContent(BTreeMap<String, bool>);
54
55/// Constants that control how the chain operates.
56///
57/// These can only change during protocol upgrades which happen on epoch boundaries. Configuration is split into feature flags (which are just booleans), and configs which can take any value (including no value at all), and will be represented by a string.
58#[Object]
59impl ProtocolConfigs {
60    async fn protocol_version(&self) -> UInt53 {
61        self.protocol_version.into()
62    }
63
64    #[graphql(flatten)]
65    async fn configs(&self, ctx: &Context<'_>) -> Result<ConfigContent, RpcError> {
66        ConfigContent::fetch(ctx, self.protocol_version).await
67    }
68
69    #[graphql(flatten)]
70    async fn feature_flags(&self, ctx: &Context<'_>) -> Result<FlagContent, RpcError> {
71        FlagContent::fetch(ctx, self.protocol_version).await
72    }
73}
74
75#[Object]
76impl ConfigContent {
77    /// Query for the value of the configuration with name `key`.
78    async fn config(&self, key: String) -> Option<ProtocolConfig> {
79        self.0.get(&key).map(|value| ProtocolConfig {
80            key: key.clone(),
81            value: value.clone(),
82        })
83    }
84
85    /// List all available configurations and their values.
86    async fn configs(&self) -> Vec<ProtocolConfig> {
87        self.0
88            .clone()
89            .into_iter()
90            .map(|(key, value)| ProtocolConfig { key, value })
91            .collect()
92    }
93}
94
95#[Object]
96impl FlagContent {
97    /// Query for the state of the feature flag with name `key`.
98    async fn feature_flag(&self, key: String) -> Option<FeatureFlag> {
99        self.0.get(&key).map(|value| FeatureFlag {
100            key: key.clone(),
101            value: *value,
102        })
103    }
104
105    /// List all available feature flags and their values.
106    async fn feature_flags(&self) -> Vec<FeatureFlag> {
107        self.0
108            .clone()
109            .into_iter()
110            .map(|(key, value)| FeatureFlag { key, value })
111            .collect()
112    }
113}

115impl ProtocolConfigs {
116    /// Construct a protocol config object that is represented by just its identifier (its protocol version).
117    pub(crate) fn with_protocol_version(protocol_version: u64) -> Self {
118        Self { protocol_version }
119    }
120
121    /// Fetch the protocol config for the latest epoch in scope.
122    ///
123    /// Returns `None` when no checkpoint is set in scope (e.g. execution scope).
124    pub(crate) async fn latest(ctx: &Context<'_>, scope: &Scope) -> Result<Option<Self>, RpcError> {
125        let Some(cp) = scope.checkpoint_viewed_at() else {
126            return Ok(None);
127        };
128
129        let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
130        let Some(stored) = pg_loader
131            .load_one(CheckpointBoundedEpochStartKey(cp))
132            .await
133            .context("Failed to fetch latest epoch")?
134        else {
135            return Ok(None);
136        };
137
138        Ok(Some(Self {
139            protocol_version: stored.protocol_version as u64,
140        }))
141    }
142}

144impl ConfigContent {
145    async fn fetch(ctx: &Context<'_>, protocol_version: u64) -> Result<Self, RpcError> {
146        use kv_protocol_configs::dsl as p;
147
148        let pg_reader: &PgReader = ctx.data()?;
149        let mut conn = pg_reader
150            .connect()
151            .await
152            .context("Failed to connect to database")?;
153
154        let configs: Vec<StoredProtocolConfig> = conn
155            .results(p::kv_protocol_configs.filter(p::protocol_version.eq(protocol_version as i64)))
156            .await
157            .context("Fail to fetch protocol configs")?;
158
159        Ok(Self(
160            configs
161                .into_iter()
162                .map(|c| (c.config_name, c.config_value))
163                .collect(),
164        ))
165    }
166}
