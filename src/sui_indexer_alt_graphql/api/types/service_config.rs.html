48    /// The estimate is an upperbound of how many nodes there would be in the output assuming every requested field is present, paginated requests return full page sizes, and multi-get queries find all requested keys. Below is a worked example query:
49    ///
50    /// ```graphql
51    /// |  0: query {                            # 514 = total
52    /// |  1:   checkpoint {                     # 1
53    /// |  2:     sequenceNumber                 # 1
54    /// |  3:   }
55    /// |  4:
56    /// |  5:   multiGetObjects([$a, $b, $c]) {  # 1 (* 3)
57    /// |  6:     address                        # 3
58    /// |  7:     digest                         # 3
59    /// |  8:   }
60    /// |  9:
61    /// | 10:   # default page size is 20
62    /// | 11:   transactions {                   # 1 (* 20)
63    /// | 12:     pageInfo {                     # 1
64    /// | 13:       hasNextPage                  # 1
65    /// | 14:       endCursor                    # 1
66    /// | 15:     }
67    /// | 16:
68    /// | 17:     nodes                          # 1
69    /// | 18:     {                              # 20
70    /// | 19:       digest                       # 20
71    /// | 20:       effects {                    # 20

105    /// Number of elements a paginated connection will return if a page size is not supplied.
106    ///
107    /// Accepts `type` and `field` arguments which identify the connection that is being queried. If the field in question is paginated, its default page size is returned. If it does not exist or is not paginated, `null` is returned.
108    async fn default_page_size(
109        &self,
110        ctx: &Context<'_>,
111        type_: String,
112        field: String,
113    ) -> Result<Option<u32>, RpcError> {
114        let registry = &ctx.schema_env.registry;
115
116        if !registry
117            .concrete_type_by_name(&type_)
118            .and_then(|t| t.field_by_name(&field))
119            .is_some_and(is_connection)
120        {
121            return Ok(None);
122        }
123
124        let config: &PaginationConfig = ctx.data()?;
125        Ok(Some(config.limits(&type_, &field).default))
126    }
127
128    /// Maximum number of elements that can be requested from a paginated connection. A request to fetch more elements will result in an error.
129    ///
130    /// Accepts `type` and `field` arguments which identify the connection that is being queried. If the field in question is paginated, its max page size is returned. If it does not exist or is not paginated, `null` is returned.
131    async fn max_page_size(
132        &self,
133        ctx: &Context<'_>,
134        type_: String,
135        field: String,
136    ) -> Result<Option<u32>, RpcError> {
137        let registry = &ctx.schema_env.registry;
138
139        if !registry
140            .concrete_type_by_name(&type_)
141            .and_then(|t| t.field_by_name(&field))
142            .is_some_and(is_connection)
143        {
144            return Ok(None);
145        }
146
147        let config: &PaginationConfig = ctx.data()?;
148        Ok(Some(config.limits(&type_, &field).max))
149    }
