4use anyhow::Context;
5use async_graphql::SimpleObject;
6use sui_rpc::proto::sui::rpc::v2 as proto;
7use sui_types::transaction::TransactionData;
8
9use crate::api::types::command_result::CommandResult;
10use crate::api::types::transaction_effects::TransactionEffects;
11use crate::error::RpcError;
12use crate::scope::Scope;
13
14/// The result of simulating a transaction, including the predicted effects.
15#[derive(Clone, SimpleObject)]
16pub struct SimulationResult {
17    /// The predicted effects of the transaction if it were executed.
18    pub effects: Option<TransactionEffects>,
19
20    /// The intermediate outputs for each command of the transaction simulation, including contents of mutated references and return values.
21    pub outputs: Option<Vec<CommandResult>>,
22}
23
24impl SimulationResult {
25    /// Create a SimulationResult from a gRPC SimulateTransactionResponse.
26    pub(crate) fn from_simulation_response(
27        scope: Scope,
28        response: proto::SimulateTransactionResponse,
29        transaction_data: TransactionData,
30    ) -> Result<Self, RpcError> {
31        let executed_transaction = response
32            .transaction
33            .as_ref()
34            .context("SimulateTransactionResponse should have transaction")?;
35
36        // Create scope with execution objects
37        let scope = scope.with_executed_transaction(executed_transaction)?;
38
39        let effects = TransactionEffects::from_executed_transaction(
40            scope.clone(),
41            executed_transaction,
42            transaction_data.clone(),
43            vec![], // No signatures for simulated transactions
44        )?;
45
46        // Extract command results from the response
47        let outputs = Some(
48            response
49                .command_outputs
50                .into_iter()
51                .map(|output| CommandResult::from_proto(output, scope.clone()))
52                .collect(),
53        );
54
55        Ok(Self {
56            effects: Some(effects),
57            outputs,
58        })
59    }
60}
