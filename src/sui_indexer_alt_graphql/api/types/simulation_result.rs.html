4use anyhow::Context;
5use async_graphql::SimpleObject;
6use sui_rpc::proto::sui::rpc::v2 as proto;
7use sui_types::transaction::TransactionData;
8
9use crate::{error::RpcError, scope::Scope};
10
11use super::{command_result::CommandResult, event::Event, transaction_effects::TransactionEffects};
12
13/// The result of simulating a transaction, including the predicted effects, events, and any errors.
14#[derive(Clone, SimpleObject)]
15pub struct SimulationResult {
16    /// The predicted effects of the transaction if it were executed.
17    ///
18    /// `None` if the simulation failed due to an error.
19    pub effects: Option<TransactionEffects>,
20
21    /// The events that would be emitted if the transaction were executed.
22    ///
23    /// `None` if the simulation failed or no events would be emitted.
24    pub events: Option<Vec<Event>>,
25
26    /// The intermediate outputs for each command of the transaction simulation, including contents of mutated references and return values.
27    pub outputs: Option<Vec<CommandResult>>,
28
29    /// Error message if the simulation failed.
30    ///
31    /// `None` if the simulation was successful.
32    pub error: Option<String>,
33}
34
35impl SimulationResult {
36    /// Create a SimulationResult from a gRPC SimulateTransactionResponse.
37    pub(crate) fn from_simulation_response(
38        scope: Scope,
39        response: proto::SimulateTransactionResponse,
40        transaction_data: TransactionData,
41    ) -> Result<Self, RpcError> {
42        let effects = Some(TransactionEffects::from_simulation_response(
43            scope.clone(),
44            response.clone(),
45            transaction_data.clone(),
46        )?);
47
48        // Parse events - break into clear steps
49        let executed_transaction = response
50            .transaction
51            .as_ref()
52            .context("No transaction in simulation response")?;
53
54        let events_bcs = executed_transaction
55            .events
56            .as_ref()
57            .and_then(|events| events.bcs.as_ref());
58
59        let transaction_events = events_bcs
60            .map(|bcs| bcs.deserialize())
61            .transpose()
62            .context("Failed to deserialize events BCS")?;
63
64        let events = transaction_events.map(|events: sui_types::effects::TransactionEvents| {
65            events
66                .data
67                .into_iter()
68                .enumerate()
69                .map(|(sequence, native_event)| Event {
70                    scope: scope.clone(),
71                    native: native_event,
72                    transaction_digest: transaction_data.digest(),
73                    sequence_number: sequence as u64,
74                    timestamp_ms: 0, // No timestamp for simulation
75                })
76                .collect()
77        });
78
79        // Extract command results from the response
80        let outputs = Some(
81            response
82                .command_outputs
83                .into_iter()
84                .map(|output| CommandResult::from_proto(output, scope.clone()))
85                .collect(),
86        );
87
88        Ok(Self {
89            effects,
90            events,
91            outputs,
92            error: None,
93        })
94    }
95}
