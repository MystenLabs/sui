14#[derive(InputObject, Debug, Default, Clone)]
15pub(crate) struct TransactionFilter {
16    /// Filter to transactions that occurred strictly after the given checkpoint.
17    pub after_checkpoint: Option<UInt53>,
18
19    /// Filter to transactions in the given checkpoint.
20    pub at_checkpoint: Option<UInt53>,
21
22    /// Filter to transaction that occurred strictly before the given checkpoint.
23    pub before_checkpoint: Option<UInt53>,
24
25    /// Filter transactions by move function called. Calls can be filtered by the `package`, `package::module`, or the `package::module::name` of their function.
26    pub function: Option<FqNameFilter>,
27
28    /// An input filter selecting for either system or programmable transactions.
29    pub kind: Option<TransactionKindInput>,
30
31    /// Limit to transactions that interacted with the given address.
32    /// The address could be a sender, sponsor, or recipient of the transaction.
33    pub affected_address: Option<SuiAddress>,
34
35    /// Limit to transactions that interacted with the given object.
36    /// The object could have been created, read, modified, deleted, wrapped, or unwrapped by the transaction.
37    /// Objects that were passed as a `Receiving` input are not considered to have been affected by a transaction unless they were actually received.
38    pub affected_object: Option<SuiAddress>,
39
40    /// Limit to transactions that were sent by the given address.
41    pub sent_address: Option<SuiAddress>,
42}
43
44/// An input filter selecting for either system or programmable transactions.
45#[derive(Enum, Copy, Clone, Eq, PartialEq, Debug)]
46pub(crate) enum TransactionKindInput {
47    /// A system transaction can be one of several types of transactions.
48    /// See [unions/transaction-block-kind] for more details.
49    SystemTx = 0,
50    /// A user submitted transaction block.
51    ProgrammableTx = 1,
52}

54pub(crate) struct TransactionFilterValidator;
55
56impl CustomValidator<TransactionFilter> for TransactionFilterValidator {
57    fn check(&self, filter: &TransactionFilter) -> Result<(), InputValueError<TransactionFilter>> {
58        let filters = filter.affected_address.is_some() as u8
59            + filter.affected_object.is_some() as u8
60            + filter.function.is_some() as u8
61            + filter.kind.is_some() as u8;
62        if filters > 1 {
63            return Err(InputValueError::custom(
64                "At most one of [affectedAddress, affectedObject, function, kind] can be specified",
65            ));
66        }
67
68        Ok(())
69    }
70}

78impl TransactionFilter {
79    /// Try to create a filter whose results are the intersection of transaction blocks in `self`'s
80    /// results and transaction blocks in `other`'s results. This may not be possible if the
81    /// resulting filter is inconsistent in some way (e.g. a filter that requires one field to be
82    /// two different values simultaneously).
83    pub(crate) fn intersect(self, other: Self) -> Option<Self> {
84        macro_rules! intersect {
85            ($field:ident, $body:expr) => {
86                intersect::field(self.$field, other.$field, $body)
87            };
88        }
89
90        Some(Self {
91            after_checkpoint: intersect!(after_checkpoint, intersect::by_max)?,
92            at_checkpoint: intersect!(at_checkpoint, intersect::by_eq)?,
93            before_checkpoint: intersect!(before_checkpoint, intersect::by_min)?,
94            function: intersect!(function, intersect::by_eq)?,
95            kind: intersect!(kind, intersect::by_eq)?,
96            affected_address: intersect!(affected_address, intersect::by_eq)?,
97            affected_object: intersect!(affected_object, intersect::by_eq)?,
98            sent_address: intersect!(sent_address, intersect::by_eq)?,
99        })
100    }
101}
