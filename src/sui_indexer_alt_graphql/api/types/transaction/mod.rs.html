69/// Description of a transaction, the unit of activity on Sui.
70#[Object]
71impl Transaction {
72    /// The transaction's globally unique identifier, which can be passed to `Query.node` to refetch it.
73    pub(crate) async fn id(&self) -> Id {
74        Id::Transaction(self.digest)
75    }
76
77    /// A 32-byte hash that uniquely identifies the transaction contents, encoded in Base58.
78    async fn digest(&self) -> String {
79        Base58::encode(self.digest)
80    }
81
82    /// The results to the chain of executing this transaction.
83    async fn effects(&self) -> Option<TransactionEffects> {
84        Some(TransactionEffects::from(self.clone()))
85    }
86
87    /// The type of this transaction as well as the commands and/or parameters comprising the transaction of this kind.
88    async fn kind(&self, ctx: &Context<'_>) -> Option<Result<TransactionKind, RpcError>> {
89        async {
90            let contents = self.contents.fetch(ctx, self.digest).await?;
91            let Some(content) = &contents.contents else {
92                return Ok(None);
93            };
94
95            let transaction_data = content.data()?;
96            Ok(TransactionKind::from(
97                transaction_data.kind().clone(),
98                contents.scope.clone(),
99            ))
100        }
101        .await
102        .transpose()
103    }
104
105    #[graphql(flatten)]
106    async fn contents(&self, ctx: &Context<'_>) -> Result<TransactionContents, RpcError> {
107        self.contents.fetch(ctx, self.digest).await
108    }
109}

111#[Object]
112impl TransactionContents {
113    /// This field is set by senders of a transaction block. It is an epoch reference that sets a deadline after which validators will no longer consider the transaction valid. By default, there is no deadline for when a transaction must execute.
114    async fn expiration(&self) -> Option<Result<Epoch, RpcError>> {
115        async {
116            let Some(content) = &self.contents else {
117                return Ok(None);
118            };
119
120            let transaction_data = content.data()?;
121            match transaction_data.expiration() {
122                TransactionExpiration::None => Ok(None),
123                TransactionExpiration::Epoch(epoch_id) => {
124                    Ok(Some(Epoch::with_id(self.scope.clone(), *epoch_id)))
125                }
126                TransactionExpiration::ValidDuring { max_epoch, .. } => {
127                    if let Some(epoch_id) = max_epoch {
128                        Ok(Some(Epoch::with_id(self.scope.clone(), *epoch_id)))
129                    } else {
130                        Ok(None)
131                    }
132                }
133            }
134        }
135        .await
136        .transpose()
137    }
138
139    /// The gas input field provides information on what objects were used as gas as well as the owner of the gas object(s) and information on the gas price and budget.
140    async fn gas_input(&self) -> Option<Result<GasInput, RpcError>> {
141        async {
142            let Some(content) = &self.contents else {
143                return Ok(None);
144            };
145
146            let transaction_data = content.data()?;
147            Ok(Some(GasInput::from_gas_data(
148                self.scope.clone(),
149                transaction_data.gas_data().clone(),
150            )))
151        }
152        .await
153        .transpose()
154    }
155
156    /// The address corresponding to the public key that signed this transaction. System transactions do not have senders.
157    async fn sender(&self) -> Option<Result<Address, RpcError>> {
158        async {
159            let Some(content) = &self.contents else {
160                return Ok(None);
161            };
162
163            let sender = content.data()?.sender();
164            Ok((sender != NativeSuiAddress::ZERO)
165                .then(|| Address::with_address(self.scope.clone(), sender)))
166        }
167        .await
168        .transpose()
169    }
170
171    /// The Base64-encoded BCS serialization of this transaction, as a `TransactionData`.
172    async fn transaction_bcs(&self) -> Option<Result<Base64, RpcError>> {
173        async {
174            let Some(content) = &self.contents else {
175                return Ok(None);
176            };
177
178            Ok(Some(Base64(content.raw_transaction()?)))
179        }
180        .await
181        .transpose()
182    }
183
184    /// The transaction as a JSON blob, matching the gRPC proto format (excluding BCS).
185    async fn transaction_json(&self) -> Option<Result<Json, RpcError>> {
186        async {
187            let Some(content) = &self.contents else {
188                return Ok(None);
189            };
190
191            let mut proto_transaction = content.proto_transaction()?;
192            // Clear the bcs field as transactionJson is intended to provide a full structured output
193            proto_transaction.bcs = None;
194            let json_value = serde_json::to_value(&proto_transaction)
195                .context("Failed to serialize transaction to JSON")?;
196            Ok(Some(json_value.try_into()?))
197        }
198        .await
199        .transpose()
200    }
201
202    /// User signatures for this transaction.
203    async fn signatures(&self) -> Result<Vec<UserSignature>, RpcError> {
204        let Some(content) = &self.contents else {
205            return Ok(vec![]);
206        };
207
208        let signatures = content.signatures()?;
209        Ok(signatures
210            .into_iter()
211            .map(UserSignature::from_generic_signature)
212            .collect())
213    }
214}

216impl Transaction {
217    /// Construct a transaction that is represented by just its identifier (its transaction
218    /// digest). This does not check whether the transaction exists, so should not be used to
219    /// "fetch" a transaction based on a digest provided as user input.
220    pub(crate) fn with_digest(scope: Scope, digest: TransactionDigest) -> Self {
221        Self {
222            digest,
223            contents: TransactionContents::empty(scope),
224        }
225    }
226
227    /// Load the transaction from the store, and return it fully inflated (with contents already
228    /// fetched). Returns `None` if the transaction does not exist (either never existed or was
229    /// pruned from the store).
230    pub(crate) async fn fetch(
231        ctx: &Context<'_>,
232        scope: Scope,
233        digest: Digest,
234    ) -> Result<Option<Self>, RpcError> {
235        let contents = TransactionContents::empty(scope)
236            .fetch(ctx, digest.into())
237            .await?;
238
239        let Some(tx) = &contents.contents else {
240            return Ok(None);
241        };
242
243        Ok(Some(Self {
244            digest: tx.digest()?,
245            contents,
246        }))
247    }
248
249    /// Cursor based pagination through transactions with filters applied.
250    ///
251    /// Returns empty results when no checkpoint is set in scope (e.g. execution scope).
252    pub(crate) async fn paginate(
253        ctx: &Context<'_>,
254        scope: Scope,
255        page: Page<CTransaction>,
256        filter: TransactionFilter,
257    ) -> Result<Connection<String, Transaction>, RpcError> {
258        let watermarks: &Arc<Watermarks> = ctx.data()?;
259        let available_range_key = AvailableRangeKey {
260            type_: "Query".to_string(),
261            field: Some("transactions".to_string()),
262            filters: Some(filter.active_filters()),
263        };
264        let reader_lo = available_range_key.reader_lo(watermarks)?;
265
266        let Some(query) = filter.tx_bounds(ctx, &scope, reader_lo, &page).await? else {
267            return Ok(Connection::new(false, false));
268        };
269
270        let TransactionFilter {
271            after_checkpoint: _,
272            at_checkpoint: _,
273            before_checkpoint: _,
274            function,
275            kind,
276            affected_address,
277            affected_object,
278            sent_address,
279        } = filter;
280
281        let tx_sequence_numbers = if let Some(function) = function {
282            tx_call(ctx, query, &page, function, sent_address).await?
283        } else if let Some(kind) = kind {
284            tx_kind(ctx, query, &page, kind, sent_address).await?
285        } else if let Some(affected_object) = affected_object {
286            tx_affected_object(ctx, query, &page, affected_object, sent_address).await?
287        } else if let Some(address) = affected_address {
288            tx_affected_address(ctx, query, &page, address, sent_address).await?
289        } else if let Some(address) = sent_address {
290            tx_affected_address(ctx, query, &page, address, sent_address).await?
291        } else {
292            tx_unfiltered(ctx, query, &page).await?
293        };
294
295        page.paginate_results(
296            tx_digests(ctx, &tx_sequence_numbers).await?,
297            |(s, _)| JsonCursor::new(*s),
298            |(_, d)| Ok(Self::with_digest(scope.clone(), d)),
299        )
300    }
301}

303impl TransactionContents {
304    fn empty(scope: Scope) -> Self {
305        Self {
306            scope,
307            contents: None,
308        }
309    }
310
311    /// Attempt to fill the contents. If the contents are already filled, returns a clone,
312    /// otherwise attempts to fetch from the store. The resulting value may still have an empty
313    /// contents field, because it could not be found in the store.
314    pub(crate) async fn fetch(
315        &self,
316        ctx: &Context<'_>,
317        digest: TransactionDigest,
318    ) -> Result<Self, RpcError> {
319        if self.contents.is_some() {
320            return Ok(self.clone());
321        }
322        let Some(checkpoint_viewed_at) = self.scope.checkpoint_viewed_at() else {
323            return Ok(self.clone());
324        };
325
326        let kv_loader: &KvLoader = ctx.data()?;
327        let Some(transaction) = kv_loader
328            .load_one_transaction(digest)
329            .await
330            .context("Failed to fetch transaction contents")?
331        else {
332            return Ok(self.clone());
333        };
334
335        // Discard the loaded result if we are viewing it at a checkpoint before it existed.
336        let cp_num = transaction
337            .cp_sequence_number()
338            .context("Any transaction fetched from the DB should have a checkpoint set")?;
339        if cp_num > checkpoint_viewed_at {
340            return Ok(self.clone());
341        }
342
343        Ok(Self {
344            scope: self.scope.clone(),
345            contents: Some(Arc::new(transaction)),
346        })
347    }
348}

367pub(crate) async fn tx_digests(
368    ctx: &Context<'_>,
369    tx_sequence_numbers: &[u64],
370) -> Result<Vec<(u64, TransactionDigest)>, RpcError> {
371    let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
372
373    try_join_all(tx_sequence_numbers.iter().map(|&tx| async move {
374        let stored = pg_loader
375            .load_one(TxDigestKey(tx))
376            .await
377            .context("Failed to load transaction digest")?
378            .context("Failed to find transaction digest")?;
379
380        let digest = TransactionDigest::try_from(stored.tx_digest)
381            .context("Failed to deserialize transaction digest")?;
382
383        Ok((tx, digest))
384    }))
385    .await
386}

388async fn tx_affected_address(
389    ctx: &Context<'_>,
390    mut query: Query<'_>,
391    page: &Page<CTransaction>,
392    affected_address: SuiAddress,
393    sent_address: Option<SuiAddress>,
394) -> Result<Vec<u64>, RpcError> {
395    query += query!(
396        r#"
397        SELECT
398            tx_sequence_number
399        FROM
400            tx_affected_addresses
401        WHERE
402            affected = {Bytea}
403        "#,
404        affected_address.into_vec(),
405    );
406
407    if let Some(address) = sent_address {
408        query += query!(" AND sender = {Bytea}", address.into_vec());
409    }
410
411    tx_sequence_numbers(ctx, query, page).await
412}

440async fn tx_call(
441    ctx: &Context<'_>,
442    mut query: Query<'_>,
443    page: &Page<CTransaction>,
444    function: FqNameFilter,
445    sent_address: Option<SuiAddress>,
446) -> Result<Vec<u64>, RpcError> {
447    query += query!(
448        r#"
449        SELECT
450            tx_sequence_number
451        FROM
452            tx_calls
453        WHERE
454            package = {Bytea}
455        "#,
456        function.package().into_vec(),
457    );
458
459    if let Some(module) = function.module() {
460        query += query!(" AND module = {Text}", module.to_string());
461    }
462
463    if let Some(name) = function.name() {
464        query += query!(" AND function = {Text}", name.to_string());
465    }
466
467    if let Some(address) = sent_address {
468        query += query!(" AND sender = {Bytea}", address.into_vec());
469    }
470
471    tx_sequence_numbers(ctx, query, page).await
472}

474async fn tx_kind(
475    ctx: &Context<'_>,
476    mut query: Query<'_>,
477    page: &Page<CTransaction>,
478    kind: TransactionKindInput,
479    sent_address: Option<SuiAddress>,
480) -> Result<Vec<u64>, RpcError> {
481    match (kind, sent_address) {
482        // We can simplify the query to just the `tx_affected_addresses` table if ProgrammableTX
483        // and sender are specified.
484        (TransactionKindInput::ProgrammableTx, Some(address)) => {
485            tx_affected_address(ctx, query, page, address, Some(address)).await
486        }
487
488        (TransactionKindInput::SystemTx, Some(_)) => Ok(vec![]),
489
490        // Otherwise, we can ignore the sender always, and just query the `tx_kinds` table.
491        (_, None) => {
492            query += query!(
493                r#"
494                SELECT
495                    tx_sequence_number
496                FROM
497                    tx_kinds
498                WHERE
499                    tx_kind = {BigInt}
500                "#,
501                kind as i64,
502            );
503
504            tx_sequence_numbers(ctx, query, page).await
505        }
506    }
507}

509async fn tx_sequence_numbers(
510    ctx: &Context<'_>,
511    mut query: Query<'_>,
512    page: &Page<CTransaction>,
513) -> Result<Vec<u64>, RpcError> {
514    query_limits::rich::debit(ctx)?;
515    let pg_reader: &PgReader = ctx.data()?;
516
517    query += query!(
518        r#"
519            AND (SELECT tx_lo FROM tx_lo) <= tx_sequence_number
520            AND tx_sequence_number < (SELECT tx_hi FROM tx_hi)
521        ORDER BY
522            tx_sequence_number {} /* order_by_direction */
523        LIMIT
524            {BigInt} /* limit_with_overhead */
525        "#,
526        page.order_by_direction(),
527        page.limit_with_overhead() as i64,
528    );
529
530    #[derive(QueryableByName)]
531    struct TxSequenceNumber {
532        #[diesel(sql_type = BigInt)]
533        tx_sequence_number: i64,
534    }
535
536    let mut conn = pg_reader
537        .connect()
538        .await
539        .context("Failed to connect to database")?;
540
541    let wrapped_tx_sequence_numbers: Vec<TxSequenceNumber> = conn
542        .results(query)
543        .await
544        .context("Failed to execute query")?;
545
546    let tx_sequence_numbers = if page.is_from_front() {
547        wrapped_tx_sequence_numbers
548            .iter()
549            .map(|t| t.tx_sequence_number as u64)
550            .collect()
551    } else {
552        wrapped_tx_sequence_numbers
553            .iter()
554            .rev()
555            .map(|t| t.tx_sequence_number as u64)
556            .collect()
557    };
558
559    Ok(tx_sequence_numbers)
560}

562/// The tx_sequence_numbers with cursors applied inclusively.
563/// Results are limited to `page.limit() + 2` to allow has_previous_page and has_next_page calculations.
564async fn tx_unfiltered(
565    ctx: &Context<'_>,
566    mut query: Query<'_>,
567    page: &Page<CTransaction>,
568) -> Result<Vec<u64>, RpcError> {
569    query_limits::rich::debit(ctx)?;
570    let pg_reader: &PgReader = ctx.data()?;
571
572    query += query!(
573        r#"
574        SELECT
575            (SELECT tx_lo FROM tx_lo),
576            (SELECT tx_hi FROM tx_hi)
577        "#
578    );
579
580    #[derive(QueryableByName)]
581    struct TxBounds {
582        #[diesel(sql_type = BigInt)]
583        tx_hi: i64,
584
585        #[diesel(sql_type = BigInt)]
586        tx_lo: i64,
587    }
588
589    let mut conn = pg_reader
590        .connect()
591        .await
592        .context("Failed to connect to database")?;
593
594    let results: Vec<TxBounds> = conn
595        .results(query)
596        .await
597        .context("Failed to execute query")?;
598
599    let (mut tx_lo, mut tx_hi) = results
600        .first()
601        .context("No valid checkpoints found")
602        .map(|bounds| (bounds.tx_lo as u64, bounds.tx_hi as u64))?;
603
604    let limit = page.limit_with_overhead() as u64;
605    if page.is_from_front() {
606        tx_hi = tx_hi.min(tx_lo.saturating_add(limit));
607    } else {
608        tx_lo = tx_lo.max(tx_hi.saturating_sub(limit));
609    }
610
611    let tx_sequence_numbers = (tx_lo..tx_hi).collect();
612    Ok(tx_sequence_numbers)
