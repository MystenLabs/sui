45/// The execution status of this transaction: success or failure.
46#[derive(Enum, Copy, Clone, Eq, PartialEq)]
47pub(crate) enum ExecutionStatus {
48    /// The transaction was successfully executed.
49    Success,
50    /// The transaction could not be executed.
51    Failure,
52}

54#[derive(Clone)]
55pub(crate) struct TransactionEffects {
56    pub(crate) digest: TransactionDigest,
57    pub(crate) contents: EffectsContents,
58}

64}

72/// The results of executing a transaction.
73#[Object]
74impl TransactionEffects {
75    /// A 32-byte hash that uniquely identifies the transaction contents, encoded in Base58.
76    ///
77    /// Note that this is different from the execution digest, which is the unique hash of the transaction effects.
78    async fn digest(&self) -> String {
79        Base58::encode(self.digest)
80    }
81
82    /// The transaction that ran to produce these effects.
83    async fn transaction(&self) -> Option<Transaction> {
84        Some(Transaction::from(self.clone()))
85    }
86
87    #[graphql(flatten)]
88    async fn contents(&self, ctx: &Context<'_>) -> Result<EffectsContents, RpcError> {
89        self.contents.fetch(ctx, self.digest).await
90    }
91}

93#[Object]
94impl EffectsContents {
95    /// The checkpoint this transaction was finalized in.
96    async fn checkpoint(&self) -> Option<Checkpoint> {
97        let Some(content) = &self.contents else {
98            return None;
99        };
100
101        content
102            .cp_sequence_number()
103            .and_then(|cp| Checkpoint::with_sequence_number(self.scope.clone(), Some(cp)))
104    }
105
106    /// Whether the transaction executed successfully or not.
107    async fn status(&self) -> Result<Option<ExecutionStatus>, RpcError> {
108        let Some(content) = &self.contents else {
109            return Ok(None);
110        };
111
112        let effects = content.effects()?;
113        let status = match effects.status() {
114            NativeExecutionStatus::Success => ExecutionStatus::Success,
115            NativeExecutionStatus::Failure { .. } => ExecutionStatus::Failure,
116        };
117
118        Ok(Some(status))
119    }
120
121    /// The latest version of all objects (apart from packages) that have been created or modified by this transaction, immediately following this transaction.
122    async fn lamport_version(&self) -> Result<Option<UInt53>, RpcError> {
123        let Some(content) = &self.contents else {
124            return Ok(None);
125        };
126
127        let effects = content.effects()?;
128        Ok(Some(UInt53::from(effects.lamport_version().value())))
129    }
130
131    /// Rich execution error information for failed transactions.
132    async fn execution_error(&self) -> Result<Option<ExecutionError>, RpcError> {
133        let Some(content) = &self.contents else {
134            return Ok(None);
135        };
136
137        let effects = content.effects()?;
138        let status = effects.status();
139
140        // Extract programmable transaction if available
141        let programmable_tx = content
142            .data()
143            .ok()
144            .and_then(|tx_data| match tx_data.into_kind() {
145                sui_types::transaction::TransactionKind::ProgrammableTransaction(tx) => Some(tx),
146                _ => None,
147            });
148
149        ExecutionError::from_execution_status(&self.scope, status, programmable_tx.as_ref()).await
150    }
151
152    /// Timestamp corresponding to the checkpoint this transaction was finalized in.
153    ///
154    /// `null` for executed/simulated transactions that have not been included in a checkpoint.
155    async fn timestamp(&self) -> Result<Option<DateTime>, RpcError> {
156        let Some(content) = &self.contents else {
157            return Ok(None);
158        };
159
160        content
161            .timestamp_ms()
162            .map(|ms| DateTime::from_ms(ms as i64))
163            .transpose()
164    }
165
166    /// The epoch this transaction was finalized in.
167    async fn epoch(&self) -> Result<Option<Epoch>, RpcError> {
168        let Some(content) = &self.contents else {
169            return Ok(None);
170        };
171
172        let effects = content.effects()?;
173        Ok(Some(Epoch::with_id(
174            self.scope.clone(),
175            effects.executed_epoch(),
176        )))
177    }

179    /// Events emitted by this transaction.
180    async fn events(
181        &self,
182        ctx: &Context<'_>,
183        first: Option<u64>,
184        after: Option<CEvent>,
185        last: Option<u64>,
186        before: Option<CEvent>,
187    ) -> Result<Option<Connection<String, Event>>, RpcError> {
188        let Some(content) = &self.contents else {
189            return Ok(Some(Connection::new(false, false)));
190        };
191
192        let pagination: &PaginationConfig = ctx.data()?;
193        let limits = pagination.limits("TransactionEffects", "events");
194        let page = Page::from_params(limits, first, after, last, before)?;
195
196        let events = content.events()?;
197        page.paginate_indices(events.len(), |i| {
198            let transaction_digest = content.digest()?;
199            let timestamp_ms = content.timestamp_ms();
200
201            Ok(Event {
202                scope: self.scope.clone(),
203                native: events[i].clone(),
204                transaction_digest,
205                sequence_number: i as u64,
206                timestamp_ms,
207            })
208        })
209        .map(Some)
210    }

212    /// The effect this transaction had on the balances (sum of coin values per coin type) of addresses and objects.
213    async fn balance_changes(
214        &self,
215        ctx: &Context<'_>,
216        first: Option<u64>,
217        after: Option<CBalanceChange>,
218        last: Option<u64>,
219        before: Option<CBalanceChange>,
220    ) -> Result<Option<Connection<String, BalanceChange>>, RpcError> {
221        let Some(content) = &self.contents else {
222            return Ok(Some(Connection::new(false, false)));
223        };
224
225        let pagination: &PaginationConfig = ctx.data()?;
226        let limits = pagination.limits("TransactionEffects", "balanceChanges");
227        let page = Page::from_params(limits, first, after, last, before)?;
228
229        // First try to get balance changes from execution context (content)
230        if let Some(grpc_balance_changes) = content.balance_changes() {
231            return page
232                .paginate_indices(grpc_balance_changes.len(), |i| {
233                    BalanceChange::from_grpc(self.scope.clone(), &grpc_balance_changes[i])
234                })
235                .map(Some);
236        }
237
238        // Fall back to loading from database
239        let transaction_digest = content.digest()?;
240        let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
241        let key = TxBalanceChangeKey(transaction_digest);
242
243        let Some(stored_balance_changes) = pg_loader
244            .load_one(key)
245            .await
246            .context("Failed to load balance changes")?
247        else {
248            return Ok(Some(Connection::new(false, false)));
249        };
250
251        // Deserialize balance changes from BCS bytes
252        let balance_changes: Vec<StoredBalanceChange> =
253            bcs::from_bytes(&stored_balance_changes.balance_changes)
254                .context("Failed to deserialize balance changes")?;
255
256        page.paginate_indices(balance_changes.len(), |i| {
257            BalanceChange::from_stored(self.scope.clone(), balance_changes[i].clone())
258        })
259        .map(Some)
260    }

372impl TransactionEffects {
373    /// Create a new TransactionEffects from an ExecutedTransaction.
374    pub(crate) fn from_executed_transaction(
375        scope: Scope,
376        executed_transaction: &ExecutedTransaction,
377        transaction_data: TransactionData,
378        signatures: Vec<GenericSignature>,
379    ) -> Result<Self, RpcError> {
380        let contents = NativeTransactionContents::from_executed_transaction(
381            executed_transaction,
382            transaction_data,
383            signatures,
384        )
385        .context("Failed to create TransactionContents from ExecutedTransaction")?;
386
387        let digest = contents
388            .digest()
389            .context("Failed to get digest from transaction contents")?;
390
391        Ok(Self {
392            digest,
393            contents: EffectsContents {
394                scope,
395                contents: Some(Arc::new(contents)),
396            },
397        })
398    }
399
400    /// Load the effects from the store, and return it fully inflated (with contents already
401    /// fetched). Returns `None` if the effects do not exist (either never existed or were pruned
402    /// from the store).
403    pub(crate) async fn fetch(
404        ctx: &Context<'_>,
405        scope: Scope,
406        digest: Digest,
407    ) -> Result<Option<Self>, RpcError> {
408        let contents = EffectsContents::empty(scope)
409            .fetch(ctx, digest.into())
410            .await?;
411
412        let Some(tx) = &contents.contents else {
413            return Ok(None);
414        };
415
416        Ok(Some(Self {
417            digest: tx.digest()?,
418            contents,
419        }))
420    }
421}
422
423impl EffectsContents {
424    fn empty(scope: Scope) -> Self {
425        Self {
426            scope,
427            contents: None,
428        }
429    }
430
431    /// Attempt to fill the contents. If the contents are already filled, returns a clone,
432    /// otherwise attempts to fetch from the store. The resulting value may still have an empty
433    /// contents field, because it could not be found in the store.
434    pub(crate) async fn fetch(
435        &self,
436        ctx: &Context<'_>,
437        digest: TransactionDigest,
438    ) -> Result<Self, RpcError> {
439        if self.contents.is_some() {
440            return Ok(self.clone());
441        }
442        let Some(checkpoint_viewed_at) = self.scope.checkpoint_viewed_at() else {
443            return Ok(self.clone());
444        };
445
446        let kv_loader: &KvLoader = ctx.data()?;
447        let Some(transaction) = kv_loader
448            .load_one_transaction(digest)
449            .await
450            .context("Failed to fetch transaction contents")?
451        else {
452            return Ok(self.clone());
453        };
454
455        let cp_num = transaction
456            .cp_sequence_number()
457            .context("Fetched transaction should have checkpoint sequence number")?;
458
459        if cp_num > checkpoint_viewed_at {
460            return Ok(self.clone());
461        }
462
463        Ok(Self {
464            scope: self.scope.clone(),
465            contents: Some(Arc::new(transaction)),
466        })
467    }
468}

470impl From<Transaction> for TransactionEffects {
471    fn from(tx: Transaction) -> Self {
472        let TransactionContents { scope, contents } = tx.contents;
473
474        Self {
475            digest: tx.digest,
476            contents: EffectsContents { scope, contents },
477        }
478    }
