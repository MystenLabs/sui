48/// The execution status of this transaction: success or failure.
49#[derive(Enum, Copy, Clone, Eq, PartialEq)]
50pub(crate) enum ExecutionStatus {
51    /// The transaction was successfully executed.
52    Success,
53    /// The transaction could not be executed.
54    Failure,
55}
56
57#[derive(Clone)]
58pub(crate) struct TransactionEffects {
59    pub(crate) digest: TransactionDigest,
60    pub(crate) contents: EffectsContents,
61}

63#[derive(Clone)]
64pub(crate) struct EffectsContents {
65    pub(crate) scope: Scope,
66    pub(crate) contents: Option<Arc<NativeTransactionContents>>,
67}

75/// The results of executing a transaction.
76#[Object]
77impl TransactionEffects {
78    /// A 32-byte hash that uniquely identifies the transaction contents, encoded in Base58.
79    ///
80    /// Note that this is different from the execution digest, which is the unique hash of the transaction effects.
81    async fn digest(&self) -> String {
82        Base58::encode(self.digest)
83    }
84
85    /// The transaction that ran to produce these effects.
86    async fn transaction(&self) -> Option<Transaction> {
87        Some(Transaction::from(self.clone()))
88    }
89
90    #[graphql(flatten)]
91    async fn contents(&self, ctx: &Context<'_>) -> Result<EffectsContents, RpcError> {
92        self.contents.fetch(ctx, self.digest).await
93    }
94}

97impl EffectsContents {
98    /// The checkpoint this transaction was finalized in.
99    async fn checkpoint(&self) -> Option<Checkpoint> {
100        let Some(content) = &self.contents else {
101            return None;
102        };
103
104        content
105            .cp_sequence_number()
106            .and_then(|cp| Checkpoint::with_sequence_number(self.scope.clone(), Some(cp)))
107    }
108
109    /// Whether the transaction executed successfully or not.
110    async fn status(&self) -> Option<Result<ExecutionStatus, RpcError>> {
111        let content = self.contents.as_ref()?;
112
113        Some(
114            content
115                .effects()
116                .map(|effects| match effects.status() {
117                    NativeExecutionStatus::Success => ExecutionStatus::Success,
118                    NativeExecutionStatus::Failure { .. } => ExecutionStatus::Failure,
119                })
120                .map_err(RpcError::from),
121        )
122    }

124    /// The latest version of all objects (apart from packages) that have been created or modified by this transaction, immediately following this transaction.
125    async fn lamport_version(&self) -> Option<Result<UInt53, RpcError>> {
126        let content = self.contents.as_ref()?;
127
128        Some(
129            content
130                .effects()
131                .map(|effects| UInt53::from(effects.lamport_version().value()))
132                .map_err(RpcError::from),
133        )
134    }

136    /// Rich execution error information for failed transactions.
137    async fn execution_error(&self) -> Option<Result<ExecutionError, RpcError>> {
138        let content = self.contents.as_ref()?;
139
140        let result = async {
141            let effects = content.effects()?;
142            let status = effects.status();
143
144            // Extract programmable transaction if available
145            let programmable_tx =
146                content
147                    .data()
148                    .ok()
149                    .and_then(|tx_data| match tx_data.into_kind() {
150                        sui_types::transaction::TransactionKind::ProgrammableTransaction(tx) => {
151                            Some(tx)
152                        }
153                        _ => None,
154                    });
155
156            ExecutionError::from_execution_status(&self.scope, status, programmable_tx.as_ref())
157                .await
158        }
159        .await;
160
161        result.transpose()
162    }

164    /// Timestamp corresponding to the checkpoint this transaction was finalized in.
165    ///
166    /// `null` for executed/simulated transactions that have not been included in a checkpoint.

168        let content = self.contents.as_ref()?;
169        let timestamp_ms = content.timestamp_ms()?;
170        Some(DateTime::from_ms(timestamp_ms as i64))

185    /// Events emitted by this transaction.
186    async fn events(
187        &self,
188        ctx: &Context<'_>,
189        first: Option<u64>,
190        after: Option<CEvent>,
191        last: Option<u64>,
192        before: Option<CEvent>,
193    ) -> Option<Result<Connection<String, Event>, RpcError>> {
194        let content = self.contents.as_ref()?;
195
196        Some(
197            async {
198                let pagination: &PaginationConfig = ctx.data()?;
199                let limits = pagination.limits("TransactionEffects", "events");
200                let page = Page::from_params(limits, first, after, last, before)?;
201
202                let events = content.events()?;
203                page.paginate_indices(events.len(), |i| {
204                    let transaction_digest = content.digest()?;
205                    let timestamp_ms = content.timestamp_ms();
206
207                    Ok(Event {
208                        scope: self.scope.clone(),
209                        native: events[i].clone(),
210                        transaction_digest,
211                        sequence_number: i as u64,
212                        timestamp_ms,
213                    })
214                })
215            }
216            .await,
217        )
218    }

220    /// The effect this transaction had on the balances (sum of coin values per coin type) of addresses and objects.
221    async fn balance_changes(
222        &self,
223        ctx: &Context<'_>,
224        first: Option<u64>,
225        after: Option<CBalanceChange>,
226        last: Option<u64>,
227        before: Option<CBalanceChange>,
228    ) -> Option<Result<Connection<String, BalanceChange>, RpcError>> {
229        let content = self.contents.as_ref()?;
230
231        Some(
232            async {
233                let pagination: &PaginationConfig = ctx.data()?;
234                let limits = pagination.limits("TransactionEffects", "balanceChanges");
235                let page = Page::from_params(limits, first, after, last, before)?;
236
237                // First try to get balance changes from execution context (content)
238                if let Some(grpc_balance_changes) = content.balance_changes() {
239                    return page.paginate_indices(grpc_balance_changes.len(), |i| {
240                        BalanceChange::from_grpc(self.scope.clone(), &grpc_balance_changes[i])
241                    });
242                }
243
244                // Fall back to loading from database
245                let transaction_digest = content.digest()?;
246                let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
247                let key = TxBalanceChangeKey(transaction_digest);
248
249                let Some(stored_balance_changes) = pg_loader
250                    .load_one(key)
251                    .await
252                    .context("Failed to load balance changes")?
253                else {
254                    return Ok(Connection::new(false, false));
255                };
256
257                // Deserialize balance changes from BCS bytes
258                let balance_changes: Vec<StoredBalanceChange> =
259                    bcs::from_bytes(&stored_balance_changes.balance_changes)
260                        .context("Failed to deserialize balance changes")?;
261
262                page.paginate_indices(balance_changes.len(), |i| {
263                    BalanceChange::from_stored(self.scope.clone(), balance_changes[i].clone())
264                })
265            }
266            .await,
267        )
268    }

270    /// The balance changes as a JSON array, matching the gRPC proto format.
271    async fn balance_changes_json(&self, ctx: &Context<'_>) -> Option<Result<Json, RpcError>> {
272        let content = self.contents.as_ref()?;
273
274        Some(
275            async {
276                // First try to get balance changes from execution context
277                if let Some(grpc_balance_changes) = content.balance_changes() {
278                    let json_value = serde_json::to_value(grpc_balance_changes)
279                        .context("Failed to serialize balance changes to JSON")?;
280                    return json_value.try_into();
281                }
282
283                // Fall back to loading from database
284                let transaction_digest = content.digest()?;
285                let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
286                let key = TxBalanceChangeKey(transaction_digest);
287
288                let Some(stored_balance_changes) = pg_loader
289                    .load_one(key)
290                    .await
291                    .context("Failed to load balance changes")?
292                else {
293                    // No balance changes found, return empty array
294                    return serde_json::json!([]).try_into();
295                };
296
297                // Deserialize and convert to gRPC format
298                let balance_changes: Vec<StoredBalanceChange> =
299                    bcs::from_bytes(&stored_balance_changes.balance_changes)
300                        .context("Failed to deserialize balance changes")?;
301
302                let grpc_balance_changes: Vec<_> = balance_changes
303                    .into_iter()
304                    .map(BalanceChange::stored_to_grpc)
305                    .collect();
306
307                let json_value = serde_json::to_value(&grpc_balance_changes)
308                    .context("Failed to serialize balance changes to JSON")?;
309                json_value.try_into()
310            }
311            .await,
312        )
313    }

452impl TransactionEffects {
453    /// Create a new TransactionEffects from an ExecutedTransaction.
454    pub(crate) fn from_executed_transaction(
455        scope: Scope,
456        executed_transaction: &ExecutedTransaction,
457        transaction_data: TransactionData,
458        signatures: Vec<GenericSignature>,
459    ) -> Result<Self, RpcError> {
460        let contents = NativeTransactionContents::from_executed_transaction(
461            executed_transaction,
462            transaction_data,
463            signatures,
464        )
465        .context("Failed to create TransactionContents from ExecutedTransaction")?;
466
467        let digest = contents
468            .digest()
469            .context("Failed to get digest from transaction contents")?;
470
471        Ok(Self {
472            digest,
473            contents: EffectsContents {
474                scope,
475                contents: Some(Arc::new(contents)),
476            },
477        })
478    }
479
480    /// Load the effects from the store, and return it fully inflated (with contents already
481    /// fetched). Returns `None` if the effects do not exist (either never existed or were pruned
482    /// from the store).
483    pub(crate) async fn fetch(
484        ctx: &Context<'_>,
485        scope: Scope,
486        digest: Digest,
487    ) -> Result<Option<Self>, RpcError> {
488        let contents = EffectsContents::empty(scope)
489            .fetch(ctx, digest.into())
490            .await?;
491
492        let Some(tx) = &contents.contents else {
493            return Ok(None);
494        };
495
496        Ok(Some(Self {
497            digest: tx.digest()?,
498            contents,
499        }))
500    }
501}

503impl EffectsContents {
504    fn empty(scope: Scope) -> Self {
505        Self {
506            scope,
507            contents: None,
508        }
509    }
510
511    /// Attempt to fill the contents. If the contents are already filled, returns a clone,
512    /// otherwise attempts to fetch from the store. The resulting value may still have an empty
513    /// contents field, because it could not be found in the store.
514    pub(crate) async fn fetch(
515        &self,
516        ctx: &Context<'_>,
517        digest: TransactionDigest,
518    ) -> Result<Self, RpcError> {
519        if self.contents.is_some() {
520            return Ok(self.clone());
521        }
522        let Some(checkpoint_viewed_at) = self.scope.checkpoint_viewed_at() else {
523            return Ok(self.clone());
524        };
525
526        let kv_loader: &KvLoader = ctx.data()?;
527        let Some(transaction) = kv_loader
528            .load_one_transaction(digest)
529            .await
530            .context("Failed to fetch transaction contents")?
531        else {
532            return Ok(self.clone());
533        };
534
535        let cp_num = transaction
536            .cp_sequence_number()
537            .context("Fetched transaction should have checkpoint sequence number")?;
538
539        if cp_num > checkpoint_viewed_at {
540            return Ok(self.clone());
541        }
542
543        Ok(Self {
544            scope: self.scope.clone(),
545            contents: Some(Arc::new(transaction)),
546        })
547    }
548}

550impl From<Transaction> for TransactionEffects {
551    fn from(tx: Transaction) -> Self {
552        let TransactionContents { scope, contents } = tx.contents;
553
554        Self {
555            digest: tx.digest,
556            contents: EffectsContents { scope, contents },
557        }
