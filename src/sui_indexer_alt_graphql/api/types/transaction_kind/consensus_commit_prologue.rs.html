20#[derive(Clone)]
21pub(crate) struct ConsensusCommitPrologueTransaction {
22    pub(crate) scope: Scope,
23    pub(crate) epoch: u64,
24    pub(crate) round: u64,
25    pub(crate) sub_dag_index: Option<u64>,
26    pub(crate) commit_timestamp_ms: u64,
27    pub(crate) consensus_commit_digest: Option<ConsensusCommitDigest>,
28    pub(crate) additional_state_digest: Option<AdditionalConsensusStateDigest>,
29}

31/// System transaction that runs at the beginning of a checkpoint, and is responsible for setting the current value of the clock, based on the timestamp from consensus.
32#[Object]
33impl ConsensusCommitPrologueTransaction {
34    /// Epoch of the commit prologue transaction.
35    ///
36    /// Present in V1, V2, V3, V4.
37    async fn epoch(&self) -> Option<Epoch> {
38        Some(Epoch::with_id(self.scope.clone(), self.epoch))
39    }
40
41    /// Consensus round of the commit.
42    ///
43    /// Present in V1, V2, V3, V4.
44    async fn round(&self) -> Option<UInt53> {
45        Some(self.round.into())
46    }
47
48    /// Unix timestamp from consensus.
49    ///
50    /// Present in V1, V2, V3, V4.
51    async fn commit_timestamp(&self) -> Option<Result<DateTime, RpcError>> {
52        Some(DateTime::from_ms(self.commit_timestamp_ms as i64))
53    }
54
55    /// Digest of consensus output, encoded as a Base58 string.
56    ///
57    /// Present in V2, V3, V4.
58    async fn consensus_commit_digest(&self) -> Option<String> {
59        self.consensus_commit_digest
60            .as_ref()
61            .map(|digest| Base58::encode(digest.inner()))
62    }
63
64    /// The sub DAG index of the consensus commit. This field is populated if there
65    /// are multiple consensus commits per round.
66    ///
67    /// Present in V3, V4.
68    async fn sub_dag_index(&self) -> Option<UInt53> {
69        self.sub_dag_index.map(|idx| idx.into())
70    }
71
72    /// Digest of any additional state computed by the consensus handler.
73    /// Used to detect forking bugs as early as possible.
74    ///
75    /// Present in V4.
76    async fn additional_state_digest(&self) -> Option<String> {
77        self.additional_state_digest
78            .as_ref()
79            .map(|digest| digest.to_string())
80    }
81}
