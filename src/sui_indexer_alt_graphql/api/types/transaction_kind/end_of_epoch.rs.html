61/// System transaction that is executed at the end of an epoch to expire JSON Web Keys (JWKs) that are no longer valid, based on their associated epoch. This is part of the on-chain state management for zkLogin and authentication.
62#[Object]
63impl AuthenticatorStateExpireTransaction {
64    /// Expire JWKs that have a lower epoch than this.
65    async fn min_epoch(&self) -> Option<Epoch> {
66        Some(Epoch::with_id(self.scope.clone(), self.native.min_epoch))
67    }
68
69    /// The initial version that the AuthenticatorStateUpdate was shared at.
70    async fn authenticator_obj_initial_shared_version(&self) -> Option<UInt53> {
71        Some(
72            self.native
73                .authenticator_obj_initial_shared_version
74                .value()
75                .into(),
76        )
77    }
78}

150/// System transaction that supersedes `ChangeEpochTransaction` as the new way to run transactions at the end of an epoch. Behaves similarly to `ChangeEpochTransaction` but can accommodate other optional transactions to run at the end of the epoch.
151#[Object]
152impl EndOfEpochTransaction {
153    /// The list of system transactions that are allowed to run at the end of the epoch.
154    async fn transactions(
155        &self,
156        ctx: &Context<'_>,
157        first: Option<u64>,
158        after: Option<CTransaction>,
159        last: Option<u64>,
160        before: Option<CTransaction>,
161    ) -> Result<Option<Connection<String, EndOfEpochTransactionKind>>, RpcError> {
162        let pagination: &PaginationConfig = ctx.data()?;
163        let limits = pagination.limits("EndOfEpochTransaction", "transactions");
164        let page = Page::from_params(limits, first, after, last, before)?;
165
166        page.paginate_indices(self.native.len(), |i| {
167            Ok(EndOfEpochTransactionKind::from(
168                self.native[i].clone(),
169                self.scope.clone(),
170            ))
171        })
172        .map(Some)
173    }
174}
175
176impl EndOfEpochTransactionKind {
177    pub fn from(kind: NativeEndOfEpochTransactionKind, scope: Scope) -> Self {
178        use EndOfEpochTransactionKind as K;
179        use NativeEndOfEpochTransactionKind as N;
180
181        match kind {
182            N::ChangeEpoch(ce) => K::ChangeEpoch(ChangeEpochTransaction { native: ce, scope }),
183            N::AuthenticatorStateCreate => {
184                K::AuthenticatorStateCreate(AuthenticatorStateCreateTransaction { dummy: None })
185            }
186            N::AuthenticatorStateExpire(expire_data) => {
187                K::AuthenticatorStateExpire(AuthenticatorStateExpireTransaction {
188                    native: expire_data,
189                    scope,
190                })
191            }
192            N::RandomnessStateCreate => {
193                K::RandomnessStateCreate(RandomnessStateCreateTransaction { dummy: None })
194            }
195            N::DenyListStateCreate => {
196                K::CoinDenyListStateCreate(CoinDenyListStateCreateTransaction { dummy: None })
197            }
198            N::StoreExecutionTimeObservations(_) => {
199                K::StoreExecutionTimeObservations(StoreExecutionTimeObservationsTransaction {
200                    dummy: None,
201                })
202            }
203            N::BridgeStateCreate(chain_id) => {
204                K::BridgeStateCreate(BridgeStateCreateTransaction { native: chain_id })
205            }
206            N::BridgeCommitteeInit(bridge_version) => {
207                K::BridgeCommitteeInit(BridgeCommitteeInitTransaction {
208                    bridge_object_version: Some(bridge_version.value().into()),
209                })
210            }
211            N::AccumulatorRootCreate => {
212                K::AccumulatorRootCreate(AccumulatorRootCreateTransaction { dummy: None })
213            }
214            N::CoinRegistryCreate => {
215                K::CoinRegistryCreate(CoinRegistryCreateTransaction { dummy: None })
216            }
217            N::DisplayRegistryCreate => {
218                K::DisplayRegistryCreate(DisplayRegistryCreateTransaction { dummy: None })
219            }
220        }
221    }
