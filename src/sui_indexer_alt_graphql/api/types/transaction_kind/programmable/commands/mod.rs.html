31/// A single command in the programmable transaction.
32#[derive(Union, Clone)]
33pub enum Command {
34    MakeMoveVec(MakeMoveVecCommand),
35    MergeCoins(MergeCoinsCommand),
36    MoveCall(MoveCallCommand),
37    Publish(PublishCommand),
38    SplitCoins(SplitCoinsCommand),
39    TransferObjects(TransferObjectsCommand),
40    Upgrade(UpgradeCommand),
41    Other(OtherCommand),
42}
43
44/// Placeholder for unimplemented command types
45#[derive(SimpleObject, Clone)]
46pub struct OtherCommand {
47    /// Placeholder field for unimplemented commands
48    #[graphql(name = "_")]
49    pub dummy: Option<bool>,
50}
51
52impl Command {
53    pub fn from(scope: Scope, command: NativeCommand) -> Self {
54        match command {
55            NativeCommand::MakeMoveVec(type_opt, elements) => {
56                Command::MakeMoveVec(MakeMoveVecCommand {
57                    type_: type_opt
58                        .map(|type_input| MoveType::from_input(type_input, scope.clone())),
59                    elements: Some(
60                        elements
61                            .into_iter()
62                            .map(TransactionArgument::from)
63                            .collect(),
64                    ),
65                })
66            }
67            NativeCommand::MoveCall(call) => Command::MoveCall(MoveCallCommand {
68                native: *call,
69                scope,
70            }),
71            NativeCommand::SplitCoins(coin, amounts) => Command::SplitCoins(SplitCoinsCommand {
72                coin: Some(TransactionArgument::from(coin)),
73                amounts: amounts.into_iter().map(TransactionArgument::from).collect(),
74            }),
75            NativeCommand::MergeCoins(coin, coins) => Command::MergeCoins(MergeCoinsCommand {
76                coin: Some(TransactionArgument::from(coin)),
77                coins: coins.into_iter().map(TransactionArgument::from).collect(),
78            }),
79            NativeCommand::TransferObjects(objects, address) => {
80                Command::TransferObjects(TransferObjectsCommand {
81                    inputs: objects.into_iter().map(TransactionArgument::from).collect(),
82                    address: Some(TransactionArgument::from(address)),
83                })
84            }
85            NativeCommand::Publish(modules, dependencies) => Command::Publish(PublishCommand {
86                modules: Some(modules.into_iter().map(Base64::from).collect()),
87                dependencies: Some(dependencies.into_iter().map(SuiAddress::from).collect()),
88            }),
89            NativeCommand::Upgrade(modules, dependencies, current_package, upgrade_ticket) => {
90                Command::Upgrade(UpgradeCommand {
91                    modules: Some(modules.into_iter().map(Base64::from).collect()),
92                    dependencies: Some(dependencies.into_iter().map(SuiAddress::from).collect()),
93                    current_package: Some(SuiAddress::from(current_package)),
94                    upgrade_ticket: Some(TransactionArgument::from(upgrade_ticket)),
95                })
96            }
97        }
98    }
99}
