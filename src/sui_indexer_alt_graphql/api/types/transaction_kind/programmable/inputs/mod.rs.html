23/// Input argument to a Programmable Transaction Block (PTB) command.
24#[derive(Union)]
25pub enum TransactionInput {
26    Pure(Pure),
27    Value(MoveValue),
28    OwnedOrImmutable(OwnedOrImmutable),
29    SharedInput(SharedInput),
30    Receiving(Receiving),
31    BalanceWithdraw(BalanceWithdraw),
32}
33
34impl TransactionInput {
35    pub fn from(
36        input: sui_types::transaction::CallArg,
37        layout: Option<MoveTypeLayout>,
38        scope: Scope,
39    ) -> Self {
40        use sui_types::transaction::CallArg as CA;
41        use sui_types::transaction::ObjectArg as OA;
42
43        match (input, layout) {
44            // If the layout for the pure arg can be inferred, then represent it as a MoveValue.
45            (CA::Pure(native), Some(layout)) => Self::Value(MoveValue {
46                type_: MoveType::from_layout(layout, scope),
47                native,
48            }),
49
50            (CA::Pure(bytes), None) => Self::Pure(Pure {
51                bytes: Some(Base64::from(bytes)),
52            }),
53
54            (CA::Object(OA::ImmOrOwnedObject((id, version, digest))), _) => Self::OwnedOrImmutable(
55                OwnedOrImmutable::from_object_ref(id, version, digest, scope),
56            ),
57
58            (
59                CA::Object(OA::SharedObject {
60                    id,
61                    initial_shared_version,
62                    mutability,
63                }),
64                _,
65            ) => Self::SharedInput(SharedInput::from_shared_object(
66                id,
67                initial_shared_version,
68                mutability.is_exclusive(),
69            )),
70
71            (CA::Object(OA::Receiving((id, version, digest))), _) => {
72                Self::Receiving(Receiving::from_object_ref(id, version, digest, scope))
73            }
74
75            (CA::FundsWithdrawal(withdrawal), _) => {
76                Self::BalanceWithdraw(BalanceWithdraw::from_native(withdrawal, scope))
77            }
78        }
79    }
80}
