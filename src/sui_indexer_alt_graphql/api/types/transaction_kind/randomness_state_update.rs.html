4use async_graphql::Object;
5use sui_types::transaction::RandomnessStateUpdate as NativeRandomnessStateUpdate;
6
7use crate::api::scalars::base64::Base64;
8
9#[derive(Clone)]
10pub(crate) struct RandomnessStateUpdateTransaction {
11    pub(crate) native: NativeRandomnessStateUpdate,
12}
13
14/// System transaction to update the source of on-chain randomness.
15#[Object]
16impl RandomnessStateUpdateTransaction {
17    /// Epoch of the randomness state update transaction.
18    async fn epoch(&self) -> Option<u64> {
19        Some(self.native.epoch)
20    }
21
22    /// Randomness round of the update.
23    async fn randomness_round(&self) -> Option<u64> {
24        Some(self.native.randomness_round.0)
25    }
26
27    /// Updated random bytes, Base64 encoded.
28    async fn random_bytes(&self) -> Option<Base64> {
29        Some(Base64::from(self.native.random_bytes.clone()))
30    }
31
32    /// The initial version of the randomness object that it was shared at.
33    async fn randomness_obj_initial_shared_version(&self) -> Option<u64> {
34        Some(self.native.randomness_obj_initial_shared_version.value())
35    }
36}
