4use crate::api::scalars::sui_address::SuiAddress;
5use async_graphql::Object;
6use sui_types::move_package::TypeOrigin as NativeTypeOrigin;
7
8pub struct TypeOrigin {
9    native: NativeTypeOrigin,
10}
11
12/// Information about which previous versions of a package introduced its types.
13#[Object]
14impl TypeOrigin {
15    /// Module defining the type.
16    pub(crate) async fn module(&self) -> Option<String> {
17        Some(self.native.module_name.clone())
18    }
19
20    /// Name of the struct.
21    #[graphql(name = "struct")]
22    pub(crate) async fn struct_(&self) -> Option<String> {
23        Some(self.native.datatype_name.clone())
24    }
25
26    /// The storage ID of the package that first defined this type.
27    pub(crate) async fn defining_id(&self) -> Option<SuiAddress> {
28        Some(self.native.package.into())
29    }
30}
31
32impl From<NativeTypeOrigin> for TypeOrigin {
33    fn from(native: NativeTypeOrigin) -> Self {
34        Self { native }
35    }
36}
