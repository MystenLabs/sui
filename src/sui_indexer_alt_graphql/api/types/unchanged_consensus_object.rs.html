26/// Reason why a transaction that attempted to access a consensus-managed object was cancelled.
27#[derive(Enum, Copy, Clone, Eq, PartialEq)]
28pub(crate) enum ConsensusObjectCancellationReason {
29    /// Read operation was cancelled.
30    CancelledRead,
31    /// Object congestion prevented execution.
32    Congested,
33    /// Randomness service was unavailable.
34    RandomnessUnavailable,
35    /// Internal use only.
36    Unknown,
37}
38
39/// Details pertaining to consensus-managed objects that are referenced by but not changed by a transaction.
40#[derive(Union)]
41pub(crate) enum UnchangedConsensusObject {
42    Read(ConsensusObjectRead),
43    MutateConsensusStreamEnded(MutateConsensusStreamEnded),
44    ReadConsensusStreamEnded(ReadConsensusStreamEnded),
45    Cancelled(ConsensusObjectCancelled),
46    PerEpochConfig(PerEpochConfig),
47}

49/// A consensus-managed object that was read by this transaction but not modified.
50pub(crate) struct ConsensusObjectRead {
51    scope: Scope,
52    object_id: ObjectID,
53    version: SequenceNumber,
54    digest: ObjectDigest,
55}

57/// A transaction that wanted to mutate a consensus-managed object but couldn't because it became not-consensus-managed before the transaction executed (for example, it was deleted, turned into an owned object, or wrapped).
58#[derive(SimpleObject)]
59pub(crate) struct MutateConsensusStreamEnded {

62
63    /// The sequence number associated with the consensus stream ending.

67/// A transaction that wanted to read a consensus-managed object but couldn't because it became not-consensus-managed before the transaction executed (for example, it was deleted, turned into an owned object, or wrapped).
68#[derive(SimpleObject)]
69pub(crate) struct ReadConsensusStreamEnded {
70    /// The ID of the consensus-managed object.

72
73    /// The sequence number associated with the consensus stream ending.
74    sequence_number: Option<UInt53>,

77/// A transaction that was cancelled before it could access the consensus-managed object, so the object was an input but remained unchanged.
78#[derive(SimpleObject)]
79pub(crate) struct ConsensusObjectCancelled {
80    /// The ID of the consensus-managed object that the transaction intended to access.
81    address: Option<SuiAddress>,
82    /// Reason why the transaction was cancelled.
83    cancellation_reason: Option<ConsensusObjectCancellationReason>,
84}

86/// A per-epoch configuration object that was accessed by this transaction and remains constant during the epoch.
87pub(crate) struct PerEpochConfig {
88    scope: Scope,
89    object_id: ObjectID,
90    /// The epoch when the transaction was executed
91    epoch: u64,
92}
93
94#[Object]
95impl PerEpochConfig {
96    /// The per-epoch configuration object as of when the transaction was executed.
97    async fn object(&self, ctx: &Context<'_>) -> Option<Result<Object, RpcError>> {
98        async {
99            let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
100            let Some(epoch_start) = pg_loader
101                .load_one(EpochStartKey(self.epoch))
102                .await
103                .context("Failed to fetch epoch start information")?
104            else {
105                return Ok(None);
106            };
107
108            let cp = epoch_start.cp_lo as u64;
109            let scope = self.scope.with_root_checkpoint(cp);
110            Object::latest(ctx, scope, self.object_id.into()).await
111        }
112        .await
113        .transpose()
114    }
115}
116
117#[Object]
118impl ConsensusObjectRead {
119    /// The version of the consensus-managed object that was read by this transaction.
120    async fn object(&self) -> Option<Object> {
121        Some(Object::with_ref(
122            &self.scope,
123            self.object_id.into(),
124            self.version,
125            self.digest,
126        ))
127    }
128}
129
130impl UnchangedConsensusObject {
131    pub(crate) fn from_native(
132        scope: Scope,
133        native: (ObjectID, NativeUnchangedConsensusKind),
134        epoch: u64,
135    ) -> Self {
136        let (object_id, kind) = native;
137
138        match kind {
139            NativeUnchangedConsensusKind::ReadOnlyRoot((version, digest)) => {
140                Self::Read(ConsensusObjectRead {
141                    scope,
142                    object_id,
143                    version,
144                    digest,
145                })
146            }
147            NativeUnchangedConsensusKind::MutateConsensusStreamEnded(sequence_number) => {
148                Self::MutateConsensusStreamEnded(MutateConsensusStreamEnded {
149                    address: Some(object_id.into()),
150                    sequence_number: Some(sequence_number.into()),
151                })
152            }
153            NativeUnchangedConsensusKind::ReadConsensusStreamEnded(sequence_number) => {
154                Self::ReadConsensusStreamEnded(ReadConsensusStreamEnded {
155                    address: Some(object_id.into()),
156                    sequence_number: Some(sequence_number.into()),
157                })
158            }
159            NativeUnchangedConsensusKind::Cancelled(sequence_number) => {
160                let cancellation_reason = match sequence_number {
161                    SequenceNumber::CANCELLED_READ => {
162                        ConsensusObjectCancellationReason::CancelledRead
163                    }
164                    SequenceNumber::CONGESTED => ConsensusObjectCancellationReason::Congested,
165                    SequenceNumber::RANDOMNESS_UNAVAILABLE => {
166                        ConsensusObjectCancellationReason::RandomnessUnavailable
167                    }
168                    _ => ConsensusObjectCancellationReason::Unknown,
169                };
170                Self::Cancelled(ConsensusObjectCancelled {
171                    address: Some(object_id.into()),
172                    cancellation_reason: Some(cancellation_reason),
173                })
174            }
175            NativeUnchangedConsensusKind::PerEpochConfig => Self::PerEpochConfig(PerEpochConfig {
176                scope,
177                object_id,
178                epoch,
179            }),
180        }
181    }
