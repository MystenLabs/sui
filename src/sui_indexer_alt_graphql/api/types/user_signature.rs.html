4use async_graphql::Object;
5use sui_types::signature::GenericSignature;
6
7use crate::api::scalars::base64::Base64;
8
9/// A user signature for a transaction.
10#[derive(Clone)]
11pub(crate) struct UserSignature {
12    pub(crate) native: GenericSignature,
13}
14
15#[Object]
16impl UserSignature {
17    /// The signature bytes, Base64-encoded.
18    /// For simple signatures: flag || signature || pubkey
19    /// For complex signatures: flag || bcs_serialized_struct
20    async fn signature_bytes(&self) -> Option<Base64> {
21        Some(Base64(self.native.as_ref().to_vec()))
22    }
23}
24
25// TODO(DVX-786): Support signature scheme details.
26impl UserSignature {
27    pub(crate) fn from_generic_signature(signature: GenericSignature) -> Self {
28        Self { native: signature }
29    }
30}
