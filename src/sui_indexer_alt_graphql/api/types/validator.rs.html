4use std::sync::Arc;
5
6use async_graphql::Context;
7use async_graphql::Object;
8use async_graphql::connection::Connection;
9
10use crate::api::scalars::cursor::JsonCursor;
11use crate::api::scalars::uint53::UInt53;
12use crate::api::types::move_type::MoveType;
13use crate::api::types::move_value::MoveValue;
14use crate::api::types::validator_set::ValidatorContents;
15use crate::api::types::validator_set::ValidatorSetContents;
16use crate::error::RpcError;
17use crate::pagination::Page;
18use crate::pagination::PaginationConfig;
19
20#[derive(Clone)]
21pub(crate) struct Validator {
22    pub(crate) contents: Arc<ValidatorSetContents>,
23    pub(crate) idx: usize,
24}
25
26type CAddr = JsonCursor<usize>;
27
28#[Object]
29impl Validator {
30    /// The number of epochs for which this validator has been below the low stake threshold.
31    async fn at_risk(&self) -> Option<UInt53> {
32        let validator = self.validator()?;
33        Some(UInt53::from(validator.at_risk))
34    }
35
36    /// On-chain representation of the underlying `0x3::validator::Validator` value.
37    async fn contents(&self) -> Option<MoveValue> {
38        let validator = self.validator()?;
39        let native = validator.bytes.clone();
40
41        let type_ = MoveType::from_native(
42            ValidatorContents::tag().into(),
43            self.contents.scope().clone(),
44        );
45
46        Some(MoveValue { type_, native })
47    }
48
49    /// Other validators this validator has reported.
50    async fn report_records(
51        &self,
52        ctx: &Context<'_>,
53        first: Option<u64>,
54        before: Option<CAddr>,
55        last: Option<u64>,
56        after: Option<CAddr>,
57    ) -> Option<Result<Connection<String, Validator>, RpcError>> {
58        let validator = self.validator()?;
59
60        Some(
61            async {
62                let pagination: &PaginationConfig = ctx.data()?;
63                let limits = pagination.limits("Validator", "reportRecords");
64                let page = Page::from_params(limits, first, after, last, before)?;
65
66                page.paginate_indices(validator.reports.len(), |i| {
67                    let idx = validator.reports[i];
68                    Ok(Validator {
69                        contents: Arc::clone(&self.contents),
70                        idx,
71                    })
72                })
73            }
74            .await,
75        )
76    }
77}
78
79impl Validator {
80    fn validator(&self) -> Option<&ValidatorContents> {
81        self.contents.active_validators.get(self.idx)
82    }
83}
