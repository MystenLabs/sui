25/// Representation of `0x3::validator_set::ValidatorSet`.
26#[derive(Clone, Debug)]
27pub(crate) struct ValidatorSet {
28    pub(crate) contents: Arc<ValidatorSetContents>,
29}
30
31#[derive(Clone, Debug)]
32pub(crate) struct ValidatorSetContents {
33    pub(crate) scope: Scope,
34    pub(crate) native: ValidatorSetV1,
35    pub(crate) report_records: IndexMap<NativeSuiAddress, Vec<usize>>,
36}
37
38/// Representation of `0x3::validator_set::ValidatorSet`.
39#[Object]
40impl ValidatorSet {
41    /// The current list of active validators.
42    async fn active_validators(
43        &self,
44        ctx: &Context<'_>,
45        first: Option<u64>,
46        after: Option<CValidator>,
47        last: Option<u64>,
48        before: Option<CValidator>,
49    ) -> Result<Option<Connection<String, Validator>>, RpcError> {
50        let pagination: &PaginationConfig = ctx.data()?;
51        let limits = pagination.limits("ValidatorSet", "activeValidators");
52        let page = Page::from_params(limits, first, after, last, before)?;
53
54        page.paginate_indices(self.contents.native.active_validators.len(), |idx| {
55            Ok(Validator {
56                contents: Arc::clone(&self.contents),
57                idx,
58            })
59        })
60        .map(Some)
61    }
62
63    /// Object ID of the `Table` storing the inactive staking pools.
64    async fn inactive_pools_id(&self) -> Option<SuiAddress> {
65        Some(self.contents.native.inactive_validators.id.into())
66    }
67
68    /// Size of the inactive pools `Table`.
69    async fn inactive_pools_size(&self) -> Option<u64> {
70        Some(self.contents.native.inactive_validators.size)
71    }
72
73    // TODO: instead of returning the id and size of the table, potentially return the table itself, paginated.
74    /// Object ID of the wrapped object `TableVec` storing the pending active validators.
75    async fn pending_active_validators_id(&self) -> Option<SuiAddress> {
76        Some(
77            self.contents
78                .native
79                .pending_active_validators
80                .contents
81                .id
82                .into(),
83        )
84    }
85
86    /// Size of the pending active validators table.
87    async fn pending_active_validators_size(&self) -> Option<u64> {
88        Some(self.contents.native.pending_active_validators.contents.size)
89    }
90
91    /// Validators that are pending removal from the active validator set, expressed as indices in to `activeValidators`.
92    async fn pending_removals(&self) -> Option<Vec<u64>> {
93        Some(self.contents.native.pending_removals.clone())
94    }
95
96    /// Object ID of the `Table` storing the mapping from staking pool ids to the addresses of the corresponding validators.
97    /// This is needed because a validator's address can potentially change but the object ID of its pool will not.
98    async fn staking_pool_mappings_id(&self) -> Option<SuiAddress> {
99        Some(self.contents.native.staking_pool_mappings.id.into())
100    }
101
102    /// Size of the stake pool mappings `Table`.
103    async fn staking_pool_mappings_size(&self) -> Option<u64> {
104        Some(self.contents.native.staking_pool_mappings.size)
105    }
106
107    /// Total amount of stake for all active validators at the beginning of the epoch.
108    async fn total_stake(&self) -> Option<BigInt> {
109        Some(self.contents.native.total_stake.into())
110    }
111
112    /// Object ID of the `Table` storing the validator candidates.
113    async fn validator_candidates_id(&self) -> Option<SuiAddress> {
114        Some(self.contents.native.validator_candidates.id.into())
115    }
116
117    /// Size of the validator candidates `Table`.
118    async fn validator_candidates_size(&self) -> Option<u64> {
119        Some(self.contents.native.validator_candidates.size)
120    }
121}
122
123impl ValidatorSet {
124    pub(crate) fn from_validator_set_v1(
125        scope: Scope,
126        native: ValidatorSetV1,
127        report_records: VecMap<NativeSuiAddress, VecSet<NativeSuiAddress>>,
128    ) -> Self {
129        let address_to_index: BTreeMap<_, _> = native
130            .active_validators
131            .iter()
132            .enumerate()
133            .map(|(i, v)| (v.metadata.sui_address, i))
134            .collect();
135        let report_records = report_records
136            .contents
137            .into_iter()
138            .map(|Entry { key, value }| {
139                (
140                    key,
141                    value
142                        .contents
143                        .into_iter()
144                        .map(|v| address_to_index[&v])
145                        .collect(),
146                )
147            })
148            .collect();
149        Self {
150            contents: Arc::new(ValidatorSetContents {
151                scope,
152                native,
153                report_records,
154            }),
155        }
156    }
