39/// An enum that specifies the intent scope to be used to parse the bytes for signature verification.
40#[derive(Enum, Copy, Clone, Eq, PartialEq)]
41pub(crate) enum ZkLoginIntentScope {
42    /// Indicates that the bytes are to be parsed as transaction data bytes.
43    TransactionData,
44    /// Indicates that the bytes are to be parsed as a personal message.
45    PersonalMessage,
46}
47
48/// The result of the zkLogin signature verification.
49#[derive(SimpleObject, Clone, Debug)]
50pub(crate) struct ZkLoginVerifyResult {
51    /// The boolean result of the verification. If true, errors should be empty.
52    pub success: Option<bool>,
53    /// The error field capture reasons why the signature could not be verified, assuming the inputs are valid and there are no internal errors.
54    pub error: Option<String>,
55}
56
57#[derive(thiserror::Error, Debug)]
58pub(crate) enum Error {
59    #[error("Cannot parse signature")]
60    BadSignature,
61
62    #[error("Not a zkLogin signature")]
63    NotZkLogin,
64
65    #[error("Failed to deserialize TransactionData from bytes")]
66    NotTransactionData,
67}

69/// Verified a zkLogin signature is from the given `author`.
70///
71/// `bytes` are either a serialized `TransactionData` or a personal message, depending on
72/// `intent_scope`.
73pub(crate) async fn verify_signature(
74    ctx: &Context<'_>,
75    scope: Scope,
76    Base64(bytes): Base64,
77    Base64(signature): Base64,
78    intent_scope: ZkLoginIntentScope,
79    author: SuiAddress,
80) -> Result<ZkLoginVerifyResult, RpcError<Error>> {
81    let config: &ZkLoginConfig = ctx.data()?;
82
83    let epoch = Epoch::fetch(ctx, scope.clone(), None)
84        .await
85        .map_err(upcast)?
86        .context("Failed to fetch current Epoch")?;
87
88    let sig @ GenericSignature::ZkLoginAuthenticator(_) = GenericSignature::from_bytes(&signature)
89        .map_err(|_| bad_user_input(Error::BadSignature))?
90    else {
91        return Err(bad_user_input(Error::NotZkLogin));
92    };
93
94    let jwk_object = DynamicField::by_serialized_name(
95        ctx,
96        scope,
97        SUI_AUTHENTICATOR_STATE_ADDRESS.into(),
98        DynamicFieldType::DynamicField,
99        TypeInput(TypeTag::U64),
100        Base64(bcs::to_bytes(&1u64).unwrap()),
101    )
102    .await
103    .map_err(upcast)?
104    .context("JWK dynamic field not found")?;
105
106    let authenticator_field = jwk_object
107        .native(ctx)
108        .await
109        .map_err(upcast)?
110        .as_ref()
111        .context("Couldn't fetch JWK dynamic field contents")?;
112
113    let authenticator_jwks: AuthenticatorStateInner =
114        bcs::from_bytes(&authenticator_field.value_bytes)
115            .context("Failed to deserialize JWK dynamic field contents")?;
116
117    let mut jwks = HashMap::new();
118    for ActiveJwk { jwk_id, jwk, .. } in authenticator_jwks.active_jwks {
119        match jwks.entry(jwk_id.clone()) {
120            Entry::Occupied(_) => {
121                warn!("JWK with kid {jwk_id:?} already exists, skipping");
122            }
123            Entry::Vacant(entry) => {
124                entry.insert(jwk.clone());
125            }
126        }
127    }
128
129    let params = VerifyParams::new(
130        jwks,
131        vec![],
132        config.env,
133        true,
134        true,
135        true,
136        config.max_epoch_upper_bound_delta,
137        true,
138        true,
139    );
140
141    Ok(match intent_scope {
142        ZkLoginIntentScope::TransactionData => verify(
143            sig,
144            &IntentMessage::new(
145                Intent::sui_transaction(),
146                bcs::from_bytes::<TransactionData>(&bytes)
147                    .map_err(|_| bad_user_input(Error::NotTransactionData))?,
148            ),
149            author,
150            epoch.epoch_id,
151            &params,
152        ),
153
154        ZkLoginIntentScope::PersonalMessage => verify(
155            sig,
156            &IntentMessage::new(
157                Intent::personal_message(),
158                PersonalMessage { message: bytes },
159            ),
160            author,
161            epoch.epoch_id,
162            &params,
163        ),
164    })
165}
166
167fn verify<T: Serialize>(
168    sig: GenericSignature,
169    message: &IntentMessage<T>,
170    author: SuiAddress,
171    epoch_id: u64,
172    params: &VerifyParams,
173) -> ZkLoginVerifyResult {
174    match sig.verify_authenticator(
175        message,
176        author.into(),
177        epoch_id,
178        params,
179        Arc::new(VerifiedDigestCache::new_empty()),
180    ) {
181        Ok(()) => ZkLoginVerifyResult {
182            success: Some(true),
183            error: None,
184        },
185
186        Err(e) => ZkLoginVerifyResult {
187            success: Some(false),
188            error: Some(e.to_string()),
189        },
190    }
191}
