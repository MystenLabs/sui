34/// An enum that specifies the intent scope to be used to parse the bytes for signature verification.
35#[derive(Enum, Copy, Clone, Eq, PartialEq)]
36pub(crate) enum ZkLoginIntentScope {
37    /// Indicates that the bytes are to be parsed as transaction data bytes.
38    TransactionData,
39    /// Indicates that the bytes are to be parsed as a personal message.
40    PersonalMessage,
41}
42
43/// The result of the zkLogin signature verification.
44#[derive(SimpleObject, Clone, Debug)]
45pub(crate) struct ZkLoginVerifyResult {
46    /// The boolean result of the verification. If true, errors should be empty.
47    pub success: Option<bool>,
48    /// The error field capture reasons why the signature could not be verified, assuming the inputs are valid and there are no internal errors.
49    pub error: Option<String>,
50}
51
52#[derive(thiserror::Error, Debug)]
53pub(crate) enum Error {
54    #[error("Cannot parse signature")]
55    BadSignature,
56
57    #[error("Not a zkLogin signature")]
58    NotZkLogin,
59
60    #[error("Failed to deserialize TransactionData from bytes")]
61    NotTransactionData,
62}
63
64/// Verified a zkLogin signature is from the given `author`.
65///
66/// `bytes` are either a serialized `TransactionData` or a personal message, depending on
67/// `intent_scope`.
68pub(crate) async fn verify_signature(
69    ctx: &Context<'_>,
70    scope: Scope,
71    Base64(bytes): Base64,
72    Base64(signature): Base64,
73    intent_scope: ZkLoginIntentScope,
74    author: SuiAddress,
75) -> Result<ZkLoginVerifyResult, RpcError<Error>> {
76    let config: &ZkLoginConfig = ctx.data()?;
77
78    let epoch = Epoch::fetch(ctx, scope.clone(), None)
79        .await
80        .map_err(upcast)?
81        .context("Failed to fetch current Epoch")?;
82
83    let sig @ GenericSignature::ZkLoginAuthenticator(_) = GenericSignature::from_bytes(&signature)
84        .map_err(|_| bad_user_input(Error::BadSignature))?
85    else {
86        return Err(bad_user_input(Error::NotZkLogin));
87    };
88
89    let jwk_object = DynamicField::by_name(
90        ctx,
91        scope,
92        SUI_AUTHENTICATOR_STATE_ADDRESS.into(),
93        DynamicFieldType::DynamicField,
94        DynamicFieldName {
95            type_: TypeInput(TypeTag::U64),
96            bcs: Base64(bcs::to_bytes(&1u64).unwrap()),
97        },
98    )
99    .await
100    .map_err(upcast)?
101    .context("JWK dynamic field not found")?;
102
103    let authenticator_field = jwk_object
104        .native(ctx)
105        .await
106        .map_err(upcast)?
107        .as_ref()
108        .context("Couldn't fetch JWK dynamic field contents")?;
109
110    let authenticator_jwks: AuthenticatorStateInner =
111        bcs::from_bytes(&authenticator_field.value_bytes)
112            .context("Failed to deserialize JWK dynamic field contents")?;
113
114    let mut jwks = HashMap::new();
115    for ActiveJwk { jwk_id, jwk, .. } in authenticator_jwks.active_jwks {
116        match jwks.entry(jwk_id.clone()) {
117            Entry::Occupied(_) => {
118                warn!("JWK with kid {jwk_id:?} already exists, skipping");
119            }
120            Entry::Vacant(entry) => {
121                entry.insert(jwk.clone());
122            }
123        }
124    }
125
126    let params = VerifyParams::new(
127        jwks,
128        vec![],
129        config.env,
130        true,
131        true,
132        true,
133        config.max_epoch_upper_bound_delta,
134        true,
135    );
136
137    Ok(match intent_scope {
138        ZkLoginIntentScope::TransactionData => verify(
139            sig,
140            &IntentMessage::new(
141                Intent::sui_transaction(),
142                bcs::from_bytes::<TransactionData>(&bytes)
143                    .map_err(|_| bad_user_input(Error::NotTransactionData))?,
144            ),
145            author,
146            epoch.epoch_id,
147            &params,
148        ),
149
150        ZkLoginIntentScope::PersonalMessage => verify(
151            sig,
152            &IntentMessage::new(
153                Intent::personal_message(),
154                PersonalMessage { message: bytes },
155            ),
156            author,
157            epoch.epoch_id,
158            &params,
159        ),
160    })
161}
162
163fn verify<T: Serialize>(
164    sig: GenericSignature,
165    message: &IntentMessage<T>,
166    author: SuiAddress,
167    epoch_id: u64,
168    params: &VerifyParams,
169) -> ZkLoginVerifyResult {
170    match sig.verify_authenticator(
171        message,
172        author.into(),
173        epoch_id,
174        params,
175        Arc::new(VerifiedDigestCache::new_empty()),
176    ) {
177        Ok(()) => ZkLoginVerifyResult {
178            success: Some(true),
179            error: None,
180        },
181
182        Err(e) => ZkLoginVerifyResult {
183            success: Some(false),
184            error: Some(e.to_string()),
185        },
186    }
187}
