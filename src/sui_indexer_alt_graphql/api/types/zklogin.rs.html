31/// An enum that specifies the intent scope to be used to parse the bytes for signature verification.
32#[derive(Enum, Copy, Clone, Eq, PartialEq)]
33pub(crate) enum ZkLoginIntentScope {
34    /// Indicates that the bytes are to be parsed as transaction data bytes.
35    TransactionData,
36    /// Indicates that the bytes are to be parsed as a personal message.
37    PersonalMessage,
38}
39
40/// The result of the zkLogin signature verification.
41#[derive(SimpleObject, Clone, Debug)]
42pub(crate) struct ZkLoginVerifyResult {
43    /// Whether the signature was verified successfully.
44    pub success: Option<bool>,
45}
46
47#[derive(thiserror::Error, Debug)]
48pub(crate) enum Error {
49    #[error("Cannot parse signature")]
50    BadSignature,
51
52    #[error("Not a zkLogin signature")]
53    NotZkLogin,
54
55    #[error("Failed to deserialize TransactionData from bytes")]
56    NotTransactionData,
57
58    #[error("Verification failed: {0}")]
59    VerificationFailed(String),
60}

62/// Verified a zkLogin signature is from the given `author`.
63///
64/// `bytes` are either a serialized `TransactionData` or a personal message, depending on
65/// `intent_scope`.
66pub(crate) async fn verify_signature(
67    ctx: &Context<'_>,
68    scope: Scope,
69    Base64(bytes): Base64,
70    Base64(signature): Base64,
71    intent_scope: ZkLoginIntentScope,
72    author: SuiAddress,
73) -> Result<ZkLoginVerifyResult, RpcError<Error>> {
74    let config: &ZkLoginConfig = ctx.data()?;
75
76    let epoch = Epoch::fetch(ctx, scope.clone(), None)
77        .await
78        .map_err(upcast)?
79        .context("Failed to fetch current Epoch")?;
80
81    let sig @ GenericSignature::ZkLoginAuthenticator(_) = GenericSignature::from_bytes(&signature)
82        .map_err(|_| bad_user_input(Error::BadSignature))?
83    else {
84        return Err(bad_user_input(Error::NotZkLogin));
85    };
86
87    let jwks = fetch_jwks(ctx, scope).await.map_err(upcast)?;
88
89    let zklogin_circuit_mode = chain_zklogin_circuit_mode(ctx, &epoch)
90        .await
91        .map_err(upcast)?;
92
93    let params = VerifyParams::new(
94        jwks,
95        vec![],
96        config.env,
97        zklogin_circuit_mode,
98        true,
99        true,
100        true,
101        config.max_epoch_upper_bound_delta,
102        true,
103        true,
104    );
105
106    match intent_scope {
107        ZkLoginIntentScope::TransactionData => verify(
108            sig,
109            &IntentMessage::new(
110                Intent::sui_transaction(),
111                bcs::from_bytes::<TransactionData>(&bytes)
112                    .map_err(|_| bad_user_input(Error::NotTransactionData))?,
113            ),
114            author,
115            epoch.epoch_id,
116            &params,
117        ),
118
119        ZkLoginIntentScope::PersonalMessage => verify(
120            sig,
121            &IntentMessage::new(
122                Intent::personal_message(),
123                PersonalMessage { message: bytes },
124            ),
125            author,
126            epoch.epoch_id,
127            &params,
128        ),
129    }
130    .map_err(|e| bad_user_input(Error::VerificationFailed(e.to_string())))?;
131
132    Ok(ZkLoginVerifyResult {
133        success: Some(true),
134    })
135}

137fn verify<T: Serialize>(
138    sig: GenericSignature,
139    message: &IntentMessage<T>,
140    author: SuiAddress,
141    epoch_id: u64,
142    params: &VerifyParams,
143) -> Result<(), sui_types::error::SuiError> {
144    sig.verify_authenticator(
145        message,
146        author.into(),
147        epoch_id,
148        params,
149        Arc::new(VerifiedDigestCache::new_empty()),
150    )
151}
