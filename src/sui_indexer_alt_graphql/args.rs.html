18/// Arguments for configuring KV store access (either Bigtable or Ledger gRPC).
19///
20/// These options are mutually exclusive - only one KV store source can be configured at a time.
21#[derive(clap::Args, Debug, Clone, Default)]
22#[group(required = false)]
23pub struct KvArgs {
24    /// Bigtable instance ID to make KV store requests to.
25    #[arg(long, group = "kv_source")]
26    pub bigtable_instance: Option<String>,
27
28    /// GCP project ID for the BigTable instance (defaults to the token provider's project).
29    #[arg(long)]
30    pub bigtable_project: Option<String>,
31
32    /// App profile ID to use for Bigtable client. If not provided, the default profile will be used.
33    #[arg(long)]
34    pub bigtable_app_profile_id: Option<String>,
35
36    /// Maximum gRPC decoding message size for KV responses, in bytes.
37    ///
38    /// Applies to both Bigtable and Ledger gRPC readers.
39    #[arg(long)]
40    pub kv_max_decoding_message_size: Option<usize>,
41
42    /// gRPC endpoint URL for the ledger service (e.g., archive.mainnet.sui.io)
43    #[arg(long, group = "kv_source")]
44    pub ledger_grpc_url: Option<Uri>,
45
46    /// Time spent waiting for a request to the kv store to complete, in milliseconds.
47    #[arg(long)]
48    pub kv_statement_timeout_ms: Option<u64>,
49}
50
51impl KvArgs {
52    /// Extract BigtableArgs from KvArgs
53    pub fn bigtable_args(&self) -> BigtableArgs {
54        BigtableArgs {
55            bigtable_statement_timeout_ms: self.kv_statement_timeout_ms,
56            bigtable_project: self.bigtable_project.clone(),
57            bigtable_app_profile_id: self.bigtable_app_profile_id.clone(),
58            bigtable_max_decoding_message_size: self.kv_max_decoding_message_size,
59        }
60    }
61
62    /// Extract LedgerGrpcArgs from KvArgs
63    pub fn ledger_grpc_args(&self) -> LedgerGrpcArgs {
64        LedgerGrpcArgs {
65            ledger_grpc_statement_timeout_ms: self.kv_statement_timeout_ms,
66            ledger_grpc_max_decoding_message_size: self.kv_max_decoding_message_size,
67        }
68    }
69}

72pub struct Args {

77#[allow(clippy::large_enum_variant)]
78#[derive(clap::Subcommand, Debug, Clone)]
79pub enum Command {
80    /// Run the RPC service.
81    Rpc {
82        /// The URL of the database to connect to.
83        #[clap(
84            long,
85            default_value = "postgres://postgres:postgrespw@localhost:5432/sui_indexer_alt"
86        )]
87        database_url: Url,
88
89        #[command(flatten)]
90        fullnode_args: FullnodeArgs,
91
92        #[command(flatten)]
93        db_args: DbArgs,
94
95        #[command(flatten)]
96        kv_args: KvArgs,
97
98        #[command(flatten)]
99        consistent_reader_args: ConsistentReaderArgs,
100
101        #[command(flatten)]
102        rpc_args: RpcArgs,
103
104        #[command(flatten)]
105        system_package_task_args: SystemPackageTaskArgs,
106
107        #[command(flatten)]
108        metrics_args: MetricsArgs,
109
110        /// Path to the RPC's configuration TOML file. If one is not provided, the default values for
111        /// the configuration will be set.
112        #[arg(long)]
113        config: Option<PathBuf>,
114
115        /// Path to indexer configuration TOML files (multiple can be supplied). These are used to
116        /// identify the pipelines that the RPC will monitor for watermark purposes.
117        #[arg(long, action = clap::ArgAction::Append)]
118        indexer_config: Vec<PathBuf>,
119    },
120
121    /// Output the contents of the default configuration to STDOUT.
122    GenerateConfig,
123}
