20/// Error codes for the `extensions.code` field of a GraphQL error that originates from outside
21/// GraphQL.
22///
23/// <https://www.apollographql.com/docs/apollo-server/data/errors/#built-in-error-codes>
24pub(crate) mod code {
25    pub const BAD_USER_INPUT: &str = "BAD_USER_INPUT";
26    pub const FEATURE_UNAVAILABLE: &str = "FEATURE_UNAVAILABLE";
27    pub const GRAPHQL_PARSE_FAILED: &str = "GRAPHQL_PARSE_FAILED";
28    pub const GRAPHQL_VALIDATION_FAILED: &str = "GRAPHQL_VALIDATION_FAILED";
29    pub const INTERNAL_SERVER_ERROR: &str = "INTERNAL_SERVER_ERROR";
30    pub const REQUEST_TIMEOUT: &str = "REQUEST_TIMEOUT";
31    pub const RESOURCE_EXHAUSTED: &str = "RESOURCE_EXHAUSTED";
32}
33
34#[derive(thiserror::Error, Debug)]
35pub(crate) enum RpcError<E: std::error::Error = Infallible> {
36    /// An error that is the user's fault.
37    BadUserInput(Arc<E>),
38
39    /// This feature is not available, because GraphQL does not have access to the underlying
40    /// store.
41    FeatureUnavailable { what: &'static str },
42
43    /// An error that is produced by the framework, it gets wrapped so that we can add an error
44    /// extension to it.
45    GraphQlError(async_graphql::Error),
46
47    /// An error produced by the internal workings of the service (our fault).
48    InternalError(Arc<anyhow::Error>),
49
50    /// A user error related to pagination and cursors.
51    Pagination(#[from] pagination::Error),
52
53    /// The request took too long to process.
54    RequestTimeout { kind: &'static str, limit: Duration },
55
56    /// Expended some limit, such as node count, depth, etc, during query execution.
57    ResourceExhausted(Arc<anyhow::Error>),
58}
59
60impl<E: std::error::Error> From<RpcError<E>> for async_graphql::Error {
61    fn from(err: RpcError<E>) -> Self {
62        match err {
63            RpcError::BadUserInput(err) => err.to_string().extend_with(|_, ext| {
64                ext.set("code", code::BAD_USER_INPUT);
65            }),
66
67            RpcError::FeatureUnavailable { what } => {
68                format!("{what} not available").extend_with(|_, ext| {
69                    ext.set("code", code::FEATURE_UNAVAILABLE);
70                })
71            }
72
73            RpcError::GraphQlError(mut err) => {
74                fill_error_code(&mut err.extensions, code::INTERNAL_SERVER_ERROR);
75                err
76            }
77
78            RpcError::InternalError(err) => {
79                // Discard the root cause (which will be the main error message), and then capture
80                // the rest as a context chain.
81                let mut chain = err.chain();
82                let Some(top) = chain.next() else {
83                    return "Unknown error".extend_with(|_, ext| {
84                        ext.set("code", code::INTERNAL_SERVER_ERROR);
85                    });
86                };
87
88                let chain: Vec<_> = chain.map(|e| e.to_string()).collect();
89                top.to_string().extend_with(|_, ext| {
90                    ext.set("code", code::INTERNAL_SERVER_ERROR);
91                    if !chain.is_empty() {
92                        ext.set("chain", chain);
93                    }
94                })
95            }
96
97            RpcError::Pagination(err) => err.to_string().extend_with(|_, ext| {
98                ext.set("code", code::BAD_USER_INPUT);
99            }),
100
101            RpcError::RequestTimeout { kind, limit } => {
102                format!("{kind} timed out after {:.2}s", limit.as_secs_f64()).extend_with(
103                    |_, ext| {
104                        ext.set("code", code::REQUEST_TIMEOUT);
105                    },
106                )
107            }
108
109            RpcError::ResourceExhausted(err) => {
110                // Discard the root cause (which will be the main error message), and then capture
111                // the rest as a context chain.
112                let mut chain = err.chain();
113                let Some(top) = chain.next() else {
114                    return "Unknown error".extend_with(|_, ext| {
115                        ext.set("code", code::RESOURCE_EXHAUSTED);
116                    });
117                };
118
119                let chain: Vec<_> = chain.map(|e| e.to_string()).collect();
120                top.to_string().extend_with(|_, ext| {
121                    ext.set("code", code::RESOURCE_EXHAUSTED);
122                    if !chain.is_empty() {
123                        ext.set("chain", chain);
124                    }
125                })
126            }
127        }
128    }
129}

131impl<E: std::error::Error> Clone for RpcError<E> {
132    fn clone(&self) -> Self {
133        match self {
134            Self::BadUserInput(e) => Self::BadUserInput(e.clone()),
135            &Self::FeatureUnavailable { what } => Self::FeatureUnavailable { what },
136            Self::GraphQlError(e) => Self::GraphQlError(e.clone()),
137            Self::InternalError(e) => Self::InternalError(e.clone()),
138            Self::Pagination(e) => Self::Pagination(e.clone()),
139            &Self::RequestTimeout { kind, limit } => Self::RequestTimeout { kind, limit },
140            Self::ResourceExhausted(e) => Self::ResourceExhausted(e.clone()),
141        }
142    }
143}

145// Cannot use `#[from]` for this conversion because [`async_graphql::Error`] does not implement
146// `std::error::Error`, so it cannot participate in the source/chaining APIs.
147impl<E: std::error::Error> From<async_graphql::Error> for RpcError<E> {
148    fn from(err: async_graphql::Error) -> Self {
149        RpcError::GraphQlError(err)
150    }
151}
152
153// Cannot use `#[from]` for this conversion because [`anyhow::Error`] does not implement `Clone`,
154// so it needs to be wrapped in an [`Arc`].
155impl<E: std::error::Error> From<anyhow::Error> for RpcError<E> {
156    fn from(err: anyhow::Error) -> Self {
157        RpcError::InternalError(Arc::new(err))
158    }
159}

161impl<E: std::error::Error> From<RpcError<E>> for async_graphql::ServerError {
162    fn from(err: RpcError<E>) -> Self {
163        let async_graphql::Error {
164            message,
165            source,
166            extensions,
167        } = async_graphql::Error::from(err);
168
169        async_graphql::ServerError {
170            message,
171            source,
172            locations: vec![],
173            path: vec![],
174            extensions,
175        }
176    }
177}

185/// Signal an error that is the user's fault.
186pub(crate) fn bad_user_input<E: std::error::Error>(err: E) -> RpcError<E> {
187    RpcError::BadUserInput(Arc::new(err))
188}

190/// Signal that feature `what` is not available.
191pub(crate) fn feature_unavailable<E: std::error::Error>(what: &'static str) -> RpcError<E> {
192    RpcError::FeatureUnavailable { what }
193}

195/// Signal a timeout. `kind` specifies what operation timed out and is included in the error
196/// message.
197pub(crate) fn request_timeout(kind: &'static str, limit: Duration) -> RpcError {
198    RpcError::RequestTimeout { kind, limit }
199}

201/// Signal some resource has been consumed during query execution.
202pub(crate) fn resource_exhausted<E>(err: impl Into<anyhow::Error>) -> RpcError<E>
203where
204    E: std::error::Error + Send + Sync + 'static,
205{
206    RpcError::ResourceExhausted(Arc::new(err.into()))
207}

209/// Upcast an `RpcError` with no user error type into an `RpcError<E>`.
210pub(crate) fn upcast<E: std::error::Error>(err: RpcError) -> RpcError<E> {
211    match err {
212        RpcError::BadUserInput(e) => match *e.as_ref() {},
213        RpcError::Pagination(e) => RpcError::Pagination(e),
214        RpcError::GraphQlError(e) => RpcError::GraphQlError(e),
215        RpcError::InternalError(e) => RpcError::InternalError(e),
216        RpcError::FeatureUnavailable { what } => RpcError::FeatureUnavailable { what },
217        RpcError::RequestTimeout { kind, limit } => RpcError::RequestTimeout { kind, limit },
218        RpcError::ResourceExhausted(e) => RpcError::ResourceExhausted(e),
219    }
220}

222/// Convert an `RpcError<A>` into an `RpcError<B>`, as long as `B` can wrap an `Arc<A>`.
223pub(crate) fn convert<A, B>(err: RpcError<A>) -> RpcError<B>
224where
225    A: std::error::Error,
226    B: std::error::Error + From<Arc<A>>,
227{
228    match err {
229        RpcError::BadUserInput(e) => RpcError::BadUserInput(Arc::new(e.into())),
230        RpcError::Pagination(e) => RpcError::Pagination(e),
231        RpcError::GraphQlError(e) => RpcError::GraphQlError(e),
232        RpcError::InternalError(e) => RpcError::InternalError(e),
233        RpcError::FeatureUnavailable { what } => RpcError::FeatureUnavailable { what },
234        RpcError::RequestTimeout { kind, limit } => RpcError::RequestTimeout { kind, limit },
235        RpcError::ResourceExhausted(e) => RpcError::ResourceExhausted(e),
236    }
237}

239/// Add a code to the error, if one does not exist already in the error extensions.
240pub(crate) fn fill_error_code(ext: &mut Option<ErrorExtensionValues>, code: &str) {
241    match ext {
242        Some(ext) if ext.get("code").is_some() => {}
243        Some(ext) => ext.set("code", code),
244        None => {
245            let mut singleton = ErrorExtensionValues::default();
246            singleton.set("code", code);
247            *ext = Some(singleton);
248        }
249    }
250}

252/// Get a list of error codes from a GraphQL response. We use these to figure out whether we should
253/// log the query at the `debug` or `info` level.
254pub(crate) fn error_codes(response: &Response) -> Vec<&str> {
255    response
256        .errors
257        .iter()
258        .flat_map(|err| &err.extensions)
259        .flat_map(|ext| ext.get("code"))
260        .filter_map(|code| {
261            if let Value::String(code) = code {
262                Some(code.as_str())
263            } else {
264                None
265            }
266        })
267        .collect()
268}

270/// Handler for panics that occur during request processing. Converts panics into GraphQL error
271/// responses with a 500 status code.
272#[derive(Clone)]
273pub(crate) struct PanicHandler {
274    metrics: Arc<RpcMetrics>,
275}
276
277impl PanicHandler {
278    pub fn new(metrics: Arc<RpcMetrics>) -> Self {
279        Self { metrics }
280    }
281}
282
283impl ResponseForPanic for PanicHandler {
284    type ResponseBody = axum::body::Body;
285
286    fn response_for_panic(
287        &mut self,
288        err: Box<dyn std::any::Any + Send + 'static>,
289    ) -> axum::http::Response<Self::ResponseBody> {
290        self.metrics.queries_panicked.inc();
291
292        let err: RpcError = if let Some(s) = err.downcast_ref::<String>() {
293            anyhow!(s.clone()).context("Request panicked")
294        } else if let Some(s) = err.downcast_ref::<&str>() {
295            anyhow!(s.to_string()).context("Request panicked")
296        } else {
297            anyhow!("Request panicked")
298        }
299        .into();
300
301        let mut res = Json(Response::from_errors(vec![err.into()])).into_response();
302        *res.status_mut() = axum::http::StatusCode::INTERNAL_SERVER_ERROR;
303        res
304    }
305}

311    #[derive(thiserror::Error, Debug)]
312    #[error("Boom!")]

315    #[test]
316    fn test_bad_user_input() {
317        let err: async_graphql::Error = bad_user_input(Error).into();
318
319        assert_eq!(err.message, "Boom!");
320
321        let ext = err.extensions.as_ref().expect("No extensions");
322        assert_eq!(ext.get("code"), Some(&code::BAD_USER_INPUT.into()));
323    }
324
325    /// If the GraphQL error does not have a code, it should be set to `INTERNAL_SERVER_ERROR`.
326    #[test]
327    fn test_graphql_error() {
328        let err: async_graphql::Error =
329            RpcError::<Infallible>::from(async_graphql::Error::new("Boom!")).into();
330
331        assert_eq!(err.message, "Boom!");
332
333        let ext = err.extensions.as_ref().expect("No extensions");
334        assert_eq!(ext.get("code"), Some(&code::INTERNAL_SERVER_ERROR.into()));
335    }
336
337    /// If the GraphQL error does already have a code, it should be left as is.
338    #[test]
339    fn test_graphql_error_existing_code() {
340        let err: async_graphql::Error = RpcError::<Infallible>::from(
341            async_graphql::Error::new("Boom!")
342                .extend_with(|_, ext| ext.set("code", code::BAD_USER_INPUT)),
343        )
344        .into();
345
346        assert_eq!(err.message, "Boom!");
347
348        let ext = err.extensions.as_ref().expect("No extensions");
349        assert_eq!(ext.get("code"), Some(&code::BAD_USER_INPUT.into()));
350    }
351
352    #[test]
353    fn test_internal_error() {
354        let err: async_graphql::Error = RpcError::<Infallible>::from(
355            anyhow!("Root cause")
356                .context("Immediate predecessor")
357                .context("Main message"),
358        )
359        .into();
360
361        assert_eq!(err.message, "Main message");
362
363        let ext = err.extensions.as_ref().expect("No extensions");
364        assert_eq!(ext.get("code"), Some(&code::INTERNAL_SERVER_ERROR.into()));
365        assert_eq!(
366            ext.get("chain"),
367            Some(&vec!["Immediate predecessor", "Root cause"].into())
368        );
369    }
370
371    #[test]
372    fn test_request_timeout() {
373        let err: async_graphql::Error = request_timeout("Kind", Duration::from_secs(5)).into();
374
375        assert_eq!(err.message, "Kind timed out after 5.00s");
376
377        let ext = err.extensions.as_ref().expect("No extensions");
378        assert_eq!(ext.get("code"), Some(&code::REQUEST_TIMEOUT.into()));
379    }
