9/// Error codes for the `extensions.code` field of a GraphQL error that originates from outside
10/// GraphQL.
11///
12/// <https://www.apollographql.com/docs/apollo-server/data/errors/#built-in-error-codes>
13pub(crate) mod code {
14    pub const BAD_USER_INPUT: &str = "BAD_USER_INPUT";
15    pub const FEATURE_UNAVAILABLE: &str = "FEATURE_UNAVAILABLE";
16    pub const GRAPHQL_PARSE_FAILED: &str = "GRAPHQL_PARSE_FAILED";
17    pub const GRAPHQL_VALIDATION_FAILED: &str = "GRAPHQL_VALIDATION_FAILED";
18    pub const INTERNAL_SERVER_ERROR: &str = "INTERNAL_SERVER_ERROR";
19    pub const REQUEST_TIMEOUT: &str = "REQUEST_TIMEOUT";
20    pub const RESOURCE_EXHAUSTED: &str = "RESOURCE_EXHAUSTED";
21}
22
23#[derive(thiserror::Error, Debug, Clone)]
24pub(crate) enum RpcError<E: std::error::Error = Infallible> {
25    /// An error that is the user's fault.
26    BadUserInput(Arc<E>),
27
28    /// This feature is not available, because GraphQL does not have access to the underlying
29    /// store.
30    FeatureUnavailable { what: &'static str },
31
32    /// An error that is produced by the framework, it gets wrapped so that we can add an error
33    /// extension to it.
34    GraphQlError(async_graphql::Error),
35
36    /// An error produced by the internal workings of the service (our fault).
37    InternalError(Arc<anyhow::Error>),
38
39    /// A user error related to pagination and cursors.
40    Pagination(#[from] pagination::Error),
41
42    /// The request took too long to process.
43    RequestTimeout { kind: &'static str, limit: Duration },
44
45    /// Expended some limit, such as node count, depth, etc, during query execution.
46    ResourceExhausted(Arc<anyhow::Error>),
47}
48
49impl<E: std::error::Error> From<RpcError<E>> for async_graphql::Error {
50    fn from(err: RpcError<E>) -> Self {
51        match err {
52            RpcError::BadUserInput(err) => err.to_string().extend_with(|_, ext| {
53                ext.set("code", code::BAD_USER_INPUT);
54            }),
55
56            RpcError::FeatureUnavailable { what } => {
57                format!("{what} not available").extend_with(|_, ext| {
58                    ext.set("code", code::FEATURE_UNAVAILABLE);
59                })
60            }
61
62            RpcError::GraphQlError(mut err) => {
63                fill_error_code(&mut err.extensions, code::INTERNAL_SERVER_ERROR);
64                err
65            }
66
67            RpcError::InternalError(err) => {
68                // Discard the root cause (which will be the main error message), and then capture
69                // the rest as a context chain.
70                let mut chain = err.chain();
71                let Some(top) = chain.next() else {
72                    return "Unknown error".extend_with(|_, ext| {
73                        ext.set("code", code::INTERNAL_SERVER_ERROR);
74                    });
75                };
76
77                let chain: Vec<_> = chain.map(|e| e.to_string()).collect();
78                top.to_string().extend_with(|_, ext| {
79                    ext.set("code", code::INTERNAL_SERVER_ERROR);
80                    if !chain.is_empty() {
81                        ext.set("chain", chain);
82                    }
83                })
84            }
85
86            RpcError::Pagination(err) => err.to_string().extend_with(|_, ext| {
87                ext.set("code", code::BAD_USER_INPUT);
88            }),
89
90            RpcError::RequestTimeout { kind, limit } => {
91                format!("{kind} timed out after {:.2}s", limit.as_secs_f64()).extend_with(
92                    |_, ext| {
93                        ext.set("code", code::REQUEST_TIMEOUT);
94                    },
95                )
96            }
97
98            RpcError::ResourceExhausted(err) => {
99                // Discard the root cause (which will be the main error message), and then capture
100                // the rest as a context chain.
101                let mut chain = err.chain();
102                let Some(top) = chain.next() else {
103                    return "Unknown error".extend_with(|_, ext| {
104                        ext.set("code", code::RESOURCE_EXHAUSTED);
105                    });
106                };
107
108                let chain: Vec<_> = chain.map(|e| e.to_string()).collect();
109                top.to_string().extend_with(|_, ext| {
110                    ext.set("code", code::RESOURCE_EXHAUSTED);
111                    if !chain.is_empty() {
112                        ext.set("chain", chain);
113                    }
114                })
115            }
116        }
117    }
118}

120// Cannot use `#[from]` for this conversion because [`async_graphql::Error`] does not implement
121// `std::error::Error`, so it cannot participate in the source/chaining APIs.
122impl<E: std::error::Error> From<async_graphql::Error> for RpcError<E> {
123    fn from(err: async_graphql::Error) -> Self {
124        RpcError::GraphQlError(err)
125    }
126}
127
128// Cannot use `#[from]` for this conversion because [`anyhow::Error`] does not implement `Clone`,
129// so it needs to be wrapped in an [`Arc`].
130impl<E: std::error::Error> From<anyhow::Error> for RpcError<E> {
131    fn from(err: anyhow::Error) -> Self {
132        RpcError::InternalError(Arc::new(err))
133    }
134}

136impl<E: std::error::Error> From<RpcError<E>> for async_graphql::ServerError {
137    fn from(err: RpcError<E>) -> Self {
138        let async_graphql::Error {
139            message,
140            source,
141            extensions,
142        } = async_graphql::Error::from(err);
143
144        async_graphql::ServerError {
145            message,
146            source,
147            locations: vec![],
148            path: vec![],
149            extensions,
150        }
151    }
152}

154impl<E: std::error::Error> From<bcs::Error> for RpcError<E> {
155    fn from(err: bcs::Error) -> Self {
156        RpcError::InternalError(Arc::new(err.into()))
157    }

160/// Signal an error that is the user's fault.
161pub(crate) fn bad_user_input<E: std::error::Error>(err: E) -> RpcError<E> {
162    RpcError::BadUserInput(Arc::new(err))
163}
164
165/// Signal that feature `what` is not available.
166pub(crate) fn feature_unavailable<E: std::error::Error>(what: &'static str) -> RpcError<E> {
167    RpcError::FeatureUnavailable { what }
168}
169
170/// Signal a timeout. `kind` specifies what operation timed out and is included in the error
171/// message.
172pub(crate) fn request_timeout(kind: &'static str, limit: Duration) -> RpcError {
173    RpcError::RequestTimeout { kind, limit }
174}
175
176/// Signal some resource has been consumed during query execution.
177pub(crate) fn resource_exhausted<E>(err: impl Into<anyhow::Error>) -> RpcError<E>
178where
179    E: std::error::Error + Send + Sync + 'static,
180{
181    RpcError::ResourceExhausted(Arc::new(err.into()))
182}

184/// Upcast an `RpcError` with no user error type into an `RpcError<E>`.
185pub(crate) fn upcast<E: std::error::Error>(err: RpcError) -> RpcError<E> {
186    match err {
187        RpcError::BadUserInput(e) => match *e.as_ref() {},
188        RpcError::Pagination(e) => RpcError::Pagination(e),
189        RpcError::GraphQlError(e) => RpcError::GraphQlError(e),
190        RpcError::InternalError(e) => RpcError::InternalError(e),
191        RpcError::FeatureUnavailable { what } => RpcError::FeatureUnavailable { what },
192        RpcError::RequestTimeout { kind, limit } => RpcError::RequestTimeout { kind, limit },
193        RpcError::ResourceExhausted(e) => RpcError::ResourceExhausted(e),
194    }
195}

197/// Add a code to the error, if one does not exist already in the error extensions.
198pub(crate) fn fill_error_code(ext: &mut Option<ErrorExtensionValues>, code: &str) {
199    match ext {
200        Some(ext) if ext.get("code").is_some() => {}
201        Some(ext) => ext.set("code", code),
202        None => {
203            let mut singleton = ErrorExtensionValues::default();
204            singleton.set("code", code);
205            *ext = Some(singleton);
206        }
207    }
208}

210/// Get a list of error codes from a GraphQL response. We use these to figure out whether we should
211/// log the query at the `debug` or `info` level.
212pub(crate) fn error_codes(response: &Response) -> Vec<&str> {
213    response
214        .errors
215        .iter()
216        .flat_map(|err| &err.extensions)
217        .flat_map(|ext| ext.get("code"))
218        .filter_map(|code| {
219            if let Value::String(code) = code {
220                Some(code.as_str())
221            } else {
222                None
223            }
224        })
225        .collect()
226}

228#[cfg(test)]
229mod tests {

232    use super::*;
233
234    #[derive(thiserror::Error, Debug)]
235    #[error("Boom!")]
236    struct Error;

238    #[test]
239    fn test_bad_user_input() {
240        let err: async_graphql::Error = bad_user_input(Error).into();
241
242        assert_eq!(err.message, "Boom!");
243
244        let ext = err.extensions.as_ref().expect("No extensions");
245        assert_eq!(ext.get("code"), Some(&code::BAD_USER_INPUT.into()));
246    }
247
248    /// If the GraphQL error does not have a code, it should be set to `INTERNAL_SERVER_ERROR`.
249    #[test]
250    fn test_graphql_error() {
251        let err: async_graphql::Error =
252            RpcError::<Infallible>::from(async_graphql::Error::new("Boom!")).into();
253
254        assert_eq!(err.message, "Boom!");
255
256        let ext = err.extensions.as_ref().expect("No extensions");
257        assert_eq!(ext.get("code"), Some(&code::INTERNAL_SERVER_ERROR.into()));
258    }
259
260    /// If the GraphQL error does already have a code, it should be left as is.
261    #[test]
262    fn test_graphql_error_existing_code() {
263        let err: async_graphql::Error = RpcError::<Infallible>::from(
264            async_graphql::Error::new("Boom!")
265                .extend_with(|_, ext| ext.set("code", code::BAD_USER_INPUT)),
266        )
267        .into();
268
269        assert_eq!(err.message, "Boom!");
270
271        let ext = err.extensions.as_ref().expect("No extensions");
272        assert_eq!(ext.get("code"), Some(&code::BAD_USER_INPUT.into()));
273    }
274
275    #[test]
276    fn test_internal_error() {
277        let err: async_graphql::Error = RpcError::<Infallible>::from(
278            anyhow!("Root cause")
279                .context("Immediate predecessor")
280                .context("Main message"),
281        )
282        .into();
283
284        assert_eq!(err.message, "Main message");
285
286        let ext = err.extensions.as_ref().expect("No extensions");
287        assert_eq!(ext.get("code"), Some(&code::INTERNAL_SERVER_ERROR.into()));
288        assert_eq!(
289            ext.get("chain"),
290            Some(&vec!["Immediate predecessor", "Root cause"].into())
291        );
292    }
293
294    #[test]
295    fn test_request_timeout() {
296        let err: async_graphql::Error = request_timeout("Kind", Duration::from_secs(5)).into();
297
298        assert_eq!(err.message, "Kind timed out after 5.00s");
299
300        let ext = err.extensions.as_ref().expect("No extensions");
301        assert_eq!(ext.get("code"), Some(&code::REQUEST_TIMEOUT.into()));
302    }
