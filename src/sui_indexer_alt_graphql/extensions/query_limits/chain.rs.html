4use std::rc::Rc;
5
6use async_graphql::Name;
7use async_graphql::PathSegment;
8
9/// Chains represent a tree of [PathSegment]s, where each link in the chain knows its parent. They
10/// are used to recover the path of (nested) fields at which an error occurred.
11pub(super) struct Chain {
12    seg: PathSegment,
13    pred: Option<Rc<Chain>>,
14}
15
16impl Chain {
17    /// Create a new chain with `name` appended to `pred`.
18    pub(super) fn new(pred: Option<Rc<Chain>>, name: Name) -> Rc<Self> {
19        Rc::new(Self {
20            seg: PathSegment::Field(name.as_str().to_owned()),
21            pred,
22        })
23    }
24
25    /// Recover the path ending at this chain node.
26    pub(super) fn to_path(&self) -> Vec<PathSegment> {
27        let mut path = vec![];
28        let mut curr = self;
29        loop {
30            path.push(curr.seg.clone());
31            if let Some(pred) = &curr.pred {
32                curr = pred;
33            } else {
34                break;
35            }
36        }
37
38        path.reverse();
39        path
40    }
41
42    /// Convenience function for taking a predecessor edge and turning it into a path.
43    pub(super) fn path(pred: &Option<Rc<Chain>>) -> Vec<PathSegment> {
44        pred.as_ref().map_or(vec![], |p| p.to_path())
45    }
46}
