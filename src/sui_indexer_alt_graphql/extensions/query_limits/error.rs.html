4use async_graphql::ErrorExtensions;
5use async_graphql::PathSegment;
6use async_graphql::Pos;
7use async_graphql::ServerError;
8use async_graphql::parser::types::OperationType;
9
10use crate::error::code;
11
12pub(super) struct Error {
13    kind: ErrorKind,
14    path: Vec<PathSegment>,
15    pos: Option<Pos>,
16}
17
18#[derive(thiserror::Error, Debug)]
19pub(super) enum ErrorKind {
20    #[error("Query nesting is over {0}")]
21    InputNesting(u32),
22
23    #[error("Query has over {0} nodes")]
24    InputNodes(u32),
25
26    #[error(transparent)]
27    InternalError(#[from] anyhow::Error),
28
29    #[error("Too many keys supplied to multi-get: {actual} > {limit}")]
30    MultiGetTooLarge { limit: u32, actual: usize },
31
32    #[error(
33        "Query is estimated to produce over {0} output nodes. Try fetching fewer fields or \
34         fetching fewer items per page in paginated or multi-get fields."
35    )]
36    OutputNodes(u32),
37
38    #[error("Page size is too large: {actual} > {limit}")]
39    PageSizeTooLarge { limit: u32, actual: u64 },
40
41    #[error("Request too large {actual}B > {limit}B")]
42    PayloadSizeOverall { limit: u32, actual: u64 },
43
44    #[error("Query payload too large: {actual}B > {limit}B")]
45    PayloadSizeQuery { limit: u32, actual: u32 },
46
47    #[error("Transaction payload exceeded limit of {limit}B")]
48    PayloadSizeTx { limit: u32 },
49
50    #[error("{0} not supported")]
51    SchemaNotSupported(OperationType),
52
53    #[error("Fragment {0} referred to but not found in document")]
54    UnknownFragment(String),
55}
56
57impl Error {
58    /// An error that occurred at a specific position in the query.
59    pub(super) fn new(kind: ErrorKind, path: Vec<PathSegment>, pos: Pos) -> Self {
60        Self {
61            kind,
62            path,
63            pos: Some(pos),
64        }
65    }
66
67    /// An error that applies to the query as a whole.
68    pub(super) fn new_global(kind: ErrorKind) -> Self {
69        Self {
70            kind,
71            path: vec![],
72            pos: None,
73        }
74    }
75}
76
77impl From<Error> for ServerError {
78    fn from(err: Error) -> Self {
79        let Error { kind, pos, path } = err;
80
81        let code = if matches!(kind, ErrorKind::InternalError(_)) {
82            code::INTERNAL_SERVER_ERROR
83        } else {
84            code::GRAPHQL_VALIDATION_FAILED
85        };
86
87        let async_graphql::Error {
88            message,
89            source,
90            extensions,
91        } = kind.extend_with(|_, ext| ext.set("code", code));
92
93        ServerError {
94            message,
95            source,
96            locations: pos.into_iter().collect(),
97            path,
98            extensions,
99        }
100    }
101}
