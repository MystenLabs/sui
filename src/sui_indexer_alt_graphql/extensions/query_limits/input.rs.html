4use std::mem;

6use async_graphql::{
7    ServerResult,
8    parser::types::{ExecutableDocument, Selection},
9};
10use serde::{Deserialize, Serialize};
11
12use super::{
13    QueryLimitsConfig,
14    chain::Chain,
15    error::{Error, ErrorKind},
16};

18/// How many input nodes the query used, and how deep the deepest part of the query was.
19#[derive(Serialize, Deserialize)]
20pub(super) struct Usage {
21    pub(super) nodes: u32,
22    pub(super) depth: u32,
23}
24
25/// Check input node limits for the query in `doc` regarding depth and number of nodes. These
26/// limits are over the abstract syntax tree of the query, so an input node could be a field
27/// selection, or a fragment spread.
28///
29/// For the purposes of this check, fragments are treated as if they are inlined into the query
30/// (the depth of the fragment definition is added to the depth at which the fragment is spread,
31/// and if a fragment is spread multiple times, it will contribute to the number of nodes used that
32/// many times).
33///
34/// The check returns the number of nodes and max depth found if they are at or below the limits,
35/// or an error specifying which limit was hit, and at what point in the query it was hit
36/// otherwise.
37pub(super) fn check(limits: &QueryLimitsConfig, doc: &ExecutableDocument) -> ServerResult<Usage> {
38    let mut node_budget = limits.max_query_nodes;
39    let mut depth_budget = limits.max_query_depth;
40
41    let mut next_level = vec![];
42    let mut curr_level = vec![];
43
44    for (_, op) in doc.operations.iter() {
45        let sels = &op.node.selection_set.node.items;
46        next_level.extend(sels.iter().map(|sel| (None, sel)));
47    }
48
49    while let Some((chain, next)) = next_level.first() {
50        if depth_budget == 0 {
51            Err(Error::new(
52                ErrorKind::InputNesting(limits.max_query_depth),
53                Chain::path(chain),
54                next.pos,
55            ))?
56        } else {
57            depth_budget -= 1;
58        }
59
60        mem::swap(&mut next_level, &mut curr_level);
61
62        for (pred, selection) in curr_level.drain(..) {
63            if node_budget == 0 {
64                Err(Error::new(
65                    ErrorKind::InputNodes(limits.max_query_nodes),
66                    Chain::path(&pred),
67                    selection.pos,
68                ))?
69            } else {
70                node_budget -= 1;
71            }
72
73            match &selection.node {
74                Selection::Field(f) => {
75                    let chain = Some(Chain::new(pred, f.node.name.node.clone()));
76                    let items = &f.node.selection_set.node.items;
77                    next_level.extend(items.iter().map(|sel| (chain.clone(), sel)))
78                }
79
80                Selection::InlineFragment(f) => {
81                    let items = &f.node.selection_set.node.items;
82                    next_level.extend(items.iter().map(|sel| (pred.clone(), sel)))
83                }
84
85                Selection::FragmentSpread(fs) => {
86                    let name = &fs.node.fragment_name.node;
87                    let def = doc.fragments.get(name).ok_or_else(|| {
88                        Error::new(
89                            ErrorKind::UnknownFragment(name.as_str().to_owned()),
90                            Chain::path(&pred),
91                            fs.pos,
92                        )
93                    })?;
94
95                    let items = &def.node.selection_set.node.items;
96                    next_level.extend(items.iter().map(|sel| (pred.clone(), sel)))
97                }
98            }
99        }
100    }
101
102    Ok(Usage {
103        nodes: limits.max_query_nodes - node_budget,
104        depth: limits.max_query_depth - depth_budget,
105    })
106}
