32pub(crate) struct QueryLimitsConfig {
33    pub(crate) max_output_nodes: u32,
34    pub(crate) max_query_nodes: u32,
35    pub(crate) max_query_depth: u32,
36    pub(crate) max_query_payload_size: u32,
37    pub(crate) max_tx_payload_size: u32,
38
39    pub(crate) tx_payload_args: BTreeSet<(&'static str, &'static str, &'static str)>,
40}
41
42/// Extension factory for adding checks that the query is within configurable limits.
43pub(crate) struct QueryLimitsChecker {
44    limits: Arc<QueryLimitsConfig>,
45    metrics: Arc<RpcMetrics>,
46}
47
48struct QueryLimitsCheckerExt {
49    limits: Arc<QueryLimitsConfig>,
50    metrics: Arc<RpcMetrics>,
51    doc: Mutex<Option<ParsedDocument>>,
52    usage: Mutex<Option<Usage>>,
53}
54
55struct ParsedDocument {
56    var: Variables,
57    doc: ExecutableDocument,
58}
59
60struct Usage {
61    input: input::Usage,
62    payload: payload::Usage,
63    output: output::Usage,
64}
65
66impl QueryLimitsConfig {
67    /// Requests to this service can definitely not exceed this size, in bytes.
68    pub(crate) fn max_payload_size(&self) -> u32 {
69        self.max_query_payload_size + self.max_tx_payload_size
70    }
71}
72
73impl QueryLimitsChecker {
74    pub(crate) fn new(limits: QueryLimitsConfig, metrics: Arc<RpcMetrics>) -> Self {
75        Self {
76            limits: Arc::new(limits),
77            metrics,
78        }
79    }
80}
81
82impl ExtensionFactory for QueryLimitsChecker {
83    fn create(&self) -> Arc<dyn Extension> {
84        Arc::new(QueryLimitsCheckerExt {
85            limits: self.limits.clone(),
86            metrics: self.metrics.clone(),
87            doc: Mutex::new(None),
88            usage: Mutex::new(None),
89        })
90    }
91}

93#[async_trait::async_trait]
94impl Extension for QueryLimitsCheckerExt {
95    /// Performs initial checks about content length, and then stashes the parsed document so that
96    /// we can run our validation checks on it.
97    async fn parse_query(
98        &self,
99        ctx: &ExtensionContext<'_>,
100        query: &str,
101        variables: &Variables,
102        next: NextParseQuery<'_>,
103    ) -> ServerResult<ExecutableDocument> {
104        let &ContentLength(length) = ctx.data_unchecked();
105        if length > self.limits.max_payload_size() as u64 {
106            Err(Error::new_global(ErrorKind::PayloadSizeOverall {
107                limit: self.limits.max_payload_size(),
108                actual: length,
109            }))?;
110        }
111
112        let doc = next.run(ctx, query, variables).await?;
113
114        // Stash the parsed document so that we can run our validations on it, after the
115        // framework's validations have run.
116        *self.doc.lock().unwrap() = Some(ParsedDocument {
117            var: variables.clone(),
118            doc: doc.clone(),
119        });
120
121        Ok(doc)
122    }
123
124    /// Run the framework's validation rules, and then run our own validation rules to validate
125    /// that the query is within configurable limits before it is run, to protect the rest of the
126    /// system from doing too much work. Tests ensure that:
127    ///
128    /// - The query is not too large or too deep, as an AST.
129    /// - If the query is large, that does not translate into a lot of query work (it's okay to
130    ///   have large binary payloads to handle execution, but we don't want a query with a big
131    ///   footprint to translate into a query that requires a lot of work to execute).
132    /// - The query will not produce too large a response (estimated based on the upperbound number
133    ///   of output nodes that input query could produce).
134    async fn validation(
135        &self,
136        ctx: &ExtensionContext<'_>,
137        next: NextValidation<'_>,
138    ) -> Result<ValidationResult, Vec<ServerError>> {
139        let res = next.run(ctx).await?;
140
141        let Some(ParsedDocument { doc, var }) = self.doc.lock().unwrap().take() else {
142            return Ok(res);
143        };
144
145        let &ContentLength(length) = ctx.data_unchecked();
146        let pagination_config: &PaginationConfig = ctx.data_unchecked();
147
148        let _guard = self.metrics.limits_validation_latency.start_timer();
149
150        let input = input::check(self.limits.as_ref(), &doc)?;
151
152        let payload = payload::check(
153            self.limits.as_ref(),
154            length,
155            &ctx.schema_env.registry,
156            &doc,
157            &var,
158        )?;
159
160        let output = output::check(
161            self.limits.as_ref(),
162            pagination_config,
163            &ctx.schema_env.registry,
164            &doc,
165            &var,
166        )?;
167
168        self.metrics.input_depth.observe(input.depth as f64);
169        self.metrics.input_nodes.observe(input.nodes as f64);
170        self.metrics.total_payload_size.observe(length as f64);
171        self.metrics
172            .query_payload_size
173            .observe(payload.query_payload_size as f64);
174        self.metrics
175            .tx_payload_size
176            .observe(payload.tx_payload_size as f64);
177        self.metrics.output_nodes.observe(output.nodes as f64);
178
179        if let Some(ShowUsage(_)) = ctx.data_opt() {
180            *self.usage.lock().unwrap() = Some(Usage {
181                input,
182                payload,
183                output,
184            });
185        }
186
187        Ok(res)
188    }
189
190    async fn request(&self, ctx: &ExtensionContext<'_>, next: NextRequest<'_>) -> Response {
191        let mut response = next.run(ctx).await;
192
193        if let Some(Usage {
194            input,
195            payload,
196            output,
197        }) = self.usage.lock().unwrap().take()
198        {
199            response = response.extension(
200                "usage",
201                value!({
202                    "input": input,
203                    "payload": payload,
204                    "output": output
205                }),
206            );
207        }
208
209        response
210    }
211}

354    fn schema(
355        limits: QueryLimitsConfig,
356        pagination_config: PaginationConfig,
357    ) -> Schema<Query, Mutation, EmptySubscription> {
358        // Create a throwaway registry and metrics so we can set-up the extension.
359        let registry = prometheus::Registry::new();
360        let metrics = RpcMetrics::new(&registry);
361
362        Schema::build(Query, Mutation, EmptySubscription)
363            .extension(QueryLimitsChecker::new(limits, metrics))
364            .data(pagination_config)
365            .finish()
366    }

369        Request::from(query)

371            .data(ContentLength(query.len() as u64))
