17/// How many output nodes are estimated to be output from this query.
18#[derive(Serialize, Deserialize)]
19pub(super) struct Usage {
20    pub(super) nodes: u32,
21}
22
23struct OutputNodeBudget<'c> {
24    /// The maximum number of output nodes that can be output from a query.
25    max_output_nodes: u32,
26
27    /// The number of additional output nodes this query is allowed to output.
28    output_node_budget: u32,
29
30    /// Configuration for default page sizes.
31    pagination_config: &'c PaginationConfig,
32}
33
34/// Validation rule that estimates the maximum number of output nodes the query will produce. It
35/// accounts for the fact that paginated and multi-get fields will produce multiple output nodes
36/// for each of their recursive input nodes, and assumes that they will always produce the maximal
37/// number of output nodes given their arguments (page size or number of keys).
38///
39/// Note that input nodes that result in lists being output will treat the list as an output node,
40/// and then every element as its own output node.
41struct OutputNodeRule<'r, 'c> {
42    budget: &'r mut OutputNodeBudget<'c>,
43
44    /// An estimate of how many times the query rooted at a given field will be evaluated (any
45    /// estimate of how many fields this query will produce needs to be multiplied by this factor).
46    multiplicity: u32,
47
48    /// Set when recursing into a paginated field to use as the multiplicative factor when
49    /// recursing into its `nodes` or `edges` field.
50    page_size: Option<u32>,
51}
52
53impl<'r> OutputNodeRule<'r, '_> {
54    /// Try to deduct `amount` from the output budget.
55    fn deduct(&mut self, driver: &FieldDriver<'_, 'r>, amount: u32) -> Result<(), Error> {
56        if amount > self.budget.output_node_budget {
57            return Err(driver.err(ErrorKind::OutputNodes(self.budget.max_output_nodes)));
58        } else {
59            self.budget.output_node_budget -= amount;
60        }
61        Ok(())
62    }
63
64    /// Returns the page size implied by the current field's arguments, assuming it is a paginated
65    /// field.
66    ///
67    /// - If the field is not paginated, returns `None`,
68    /// - If the page size exceeds the configured max page size, returns an error.
69    /// - If the page size would cause the estimated output node size to overflow, returns an
70    ///   error.
71    ///
72    /// If the field does not specify a page size, a default page size is fetched from the config,
73    /// based on the parent type and field name.
74    fn page_size(&self, driver: &FieldDriver<'_, 'r>) -> Result<Option<u32>, Error> {
75        if !is_connection(driver.meta_field()) {
76            return Ok(None);
77        }
78
79        let first = self.size_arg(driver, "first");
80        let last = self.size_arg(driver, "last");
81
82        let type_ = driver.parent_type().name();
83        let name = driver.meta_field().name.as_str();
84        let limits = self.budget.pagination_config.limits(type_, name);
85
86        let size = match (first, last) {
87            (Some(f), Some(l)) => f.max(l),
88            (Some(p), _) | (_, Some(p)) => p,
89            (None, None) => limits.default as u64,
90        };
91
92        if size > limits.max as u64 {
93            return Err(driver.err(ErrorKind::PageSizeTooLarge {
94                limit: limits.max,
95                actual: size,
96            }));
97        }
98
99        // SAFETY: `size <= limits.max <= u32::MAX`.
100        Ok(Some(size as u32))
101    }
102
103    /// Look for an argument on the current field with the name `name`, and assume that it is a
104    /// numeric argument. If the argument is not present, or is not a number, returns `None`.
105    fn size_arg(&self, driver: &FieldDriver<'_, 'r>, name: &str) -> Option<u64> {
106        let Value::Number(num) = driver.resolve_arg(name)? else {
107            return None;
108        };
109
110        num.as_u64()
111    }
112
113    /// Returns the number of keys that will be fetched by the current field, assuming it is a
114    /// multi-get field (the name starts with `multiGet`, and it contains a `keys` argument that
115    /// is a list).
116    fn multi_get_size(&self, driver: &FieldDriver<'_, 'r>) -> Result<Option<u32>, Error> {
117        if !driver.meta_field().name.starts_with("multiGet") {
118            return Ok(None);
119        }
120
121        if let Some(Value::List(vs)) = driver.resolve_arg("keys") {
122            let keys = vs.len();
123            let limit = self.budget.pagination_config.max_multi_get_size();
124            if keys > limit as usize {
125                return Err(driver.err(ErrorKind::MultiGetTooLarge {
126                    limit,
127                    actual: keys,
128                }));
129            }
130
131            // SAFETY: `keys < limit <= u32::MAX`.
132            Ok(Some(keys as u32))
133        } else {
134            Ok(None)
135        }
136    }
137}

139impl<'r> Visitor<'r> for OutputNodeRule<'r, '_> {
140    fn visit_field(&mut self, driver: &FieldDriver<'_, 'r>) -> Result<(), Error> {
141        let mut multiplicity = self.multiplicity;
142        let mut page_size = self.page_size;
143
144        // Start by deducting the cost of the current field.
145        self.deduct(driver, multiplicity)?;
146
147        if ["nodes", "edges"].contains(&driver.meta_field().name.as_str()) {
148            // If the current field looks like a page content field for a Connection, and there is
149            // a page size set (meaning this field is nested within a Connection field), then
150            // nested fields will be multiplied by the connection's page size.
151            if let Some(size) = page_size {
152                multiplicity = self.multiplicity.saturating_mul(size);
153                page_size = None;
154
155                // Deduct the cost of an output node per element of the page.
156                self.deduct(driver, multiplicity)?;
157            }
158        } else if let Some(size) = self.multi_get_size(driver)? {
159            // If the current field is a multi-get, then the multiplicity increases and we know any
160            // page size that was set previously is no longer relevant.
161            multiplicity = self.multiplicity.saturating_mul(size);
162            page_size = None;
163
164            // Deduct the cost of an output node per element of the page.
165            self.deduct(driver, multiplicity)?;
166        } else if let Some(size) = self.page_size(driver)? {
167            // If the current field is a Connection, its arguments determine the page size that
168            // impacts nested fields.
169            page_size = Some(size);
170        }
171
172        driver.visit_selection_set(&mut OutputNodeRule {
173            budget: self.budget,
174            multiplicity,
175            page_size,
176        })
177    }
178}

180/// Test that the query does not produce an excessively large output by estimating the number
181/// of output nodes it will produce before executing it.
182///
183/// This check must be done after the input limit check, because it relies on the query depth being
184/// bounded to protect it from recursing too deeply.
185pub(super) fn check(
186    limits: &QueryLimitsConfig,
187    pagination_config: &PaginationConfig,
188    registry: &Registry,
189    doc: &ExecutableDocument,
190    variables: &Variables,
191) -> ServerResult<Usage> {
192    let mut budget = OutputNodeBudget {
193        max_output_nodes: limits.max_output_nodes,
194        output_node_budget: limits.max_output_nodes,
195        pagination_config,
196    };
197
198    Driver::visit_document(
199        registry,
200        doc,
201        variables,
202        &mut OutputNodeRule {
203            budget: &mut budget,
204            multiplicity: 1,
205            page_size: None,
206        },
207    )?;
208
209    Ok(Usage {
210        nodes: limits.max_output_nodes - budget.output_node_budget,
211    })
