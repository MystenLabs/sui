21/// The proportion of the query content that was transaction payload versus query payload.
22#[derive(Serialize, Deserialize)]
23pub(super) struct Usage {
24    pub(super) query_payload_size: u32,
25    pub(super) tx_payload_size: u32,
26}
27
28/// Validation rule that finds parts of the request that correspond to transaction payloads
29/// (transaction bytes, signatures, etc), and deducts them from an overall budget.
30///
31/// NOTE: This rule only checks top-level fields (it does not look for transaction payloads in
32/// nested structures) as currently transaction payloads can only be found there.
33struct TxPayloadRule<'r> {
34    /// The maximum number of bytes in a request that can come from transaction payloads.
35    max_tx_payload_size: u32,
36
37    /// The budget for transaction payloads remaining.
38    tx_payload_budget: u32,
39
40    /// Names of variables that have been used in transaction payloads -- variables that show up in
41    /// transaction payloads multiple times are only debited from the budget once.
42    used_vars: BTreeSet<&'r Name>,
43
44    /// Set of type, field, and argument triples that are known to contain transaction payloads.
45    tx_payload_args: &'r BTreeSet<(&'static str, &'static str, &'static str)>,
46}
47
48impl<'r> TxPayloadRule<'r> {
49    /// Whether the type and field pointed at currently by `driver` contains any transaction
50    /// payload arguments or not.
51    fn has_tx_payload(&self, driver: &FieldDriver<'_, 'r>) -> bool {
52        let type_ = driver.parent_type().name();
53        let field = driver.meta_field().name.as_str();
54
55        self.tx_payload_args
56            .range(&(type_, field, "")..)
57            .next()
58            .is_some_and(|&(t, f, _)| t == type_ && f == field)
59    }
60
61    /// Debit the size of `value` assuming `name` is a transaction payload argument of the type and
62    /// field pointed at by `driver`.
63    ///
64    /// Fails if there is insufficient remaining budget.
65    fn check_tx_arg(
66        &mut self,
67        driver: &FieldDriver<'_, 'r>,
68        name: &'r Positioned<Name>,
69        value: &'r Positioned<Value>,
70    ) -> Result<(), Error> {
71        use Value as V;
72
73        let type_ = driver.parent_type().name();
74        let field = driver.meta_field().name.as_str();
75        let arg = name.node.as_str();
76
77        if !self.tx_payload_args.contains(&(type_, field, arg)) {
78            return Ok(());
79        }
80
81        let mut stack = vec![&value.node];
82        while let Some(v) = stack.pop() {
83            match v {
84                V::Variable(name) => self.check_tx_var(driver, value.pos, name)?,
85
86                V::String(s) => {
87                    // Pay for the string, plus the quotes around it.
88                    let debit = s.len() + 2;
89                    if debit > self.tx_payload_budget as usize {
90                        return Err(driver.err_at(
91                            value.pos,
92                            ErrorKind::PayloadSizeTx {
93                                limit: self.max_tx_payload_size,
94                            },
95                        ));
96                    } else {
97                        // SAFETY: We know that debit <= self.tx_payload_budget, which is a u32, so
98                        // the cast and subtraction are both safe.
99                        self.tx_payload_budget -= debit as u32;
100                    }
101                }
102
103                V::List(vs) => {
104                    // Pay for the opening and closing brackets and every comma up-front so that
105                    // deeply nested lists are not free.
106                    let debit = vs.len().saturating_sub(1) + 2;
107                    if debit > self.tx_payload_budget as usize {
108                        return Err(driver.err_at(
109                            value.pos,
110                            ErrorKind::PayloadSizeTx {
111                                limit: self.max_tx_payload_size,
112                            },
113                        ));
114                    } else {
115                        // SAFETY: We know that debit <= self.tx_payload_budget, which is a u32, so
116                        // the cast and subtraction are both safe.
117                        self.tx_payload_budget -= debit as u32;
118                        stack.extend(vs)
119                    }
120                }
121
122                V::Null
123                | V::Number(_)
124                | V::Boolean(_)
125                | V::Binary(_)
126                | V::Enum(_)
127                | V::Object(_) => {
128                    // Transaction payloads cannot be any of these types, so this request is
129                    // destined to fail. Ignore these values for now, so that it can fail later on
130                    // with a more legible error message.
131                    //
132                    // From a limits perspective, it is safe to ignore these values here, because
133                    // they will still be counted as part of the query payload (and so are still
134                    // subject to a limit).
135                }
136            }
137        }
138
139        Ok(())
140    }
141
142    /// Debit the size of the value for the variable `name` (assuming it is used in a transaction
143    /// payload argument), if it has not already been used in a transaction payload (in which case
144    /// it is already accounted for).
145    ///
146    /// Fails if there is insufficient remaining budget.
147    fn check_tx_var(
148        &mut self,
149        driver: &FieldDriver<'_, 'r>,
150        pos: Pos,
151        name: &'r Name,
152    ) -> Result<(), Error> {
153        use ConstValue as CV;
154
155        // Already used in a transaction payload, don't double count.
156        if !self.used_vars.insert(name) {
157            return Ok(());
158        }
159
160        // Can't find the variable, so it can't count towards the transaction payload.
161        let Some(value) = driver.resolve_var(name) else {
162            return Ok(());
163        };
164
165        let mut stack = vec![value];
166        while let Some(value) = stack.pop() {
167            match &value {
168                CV::String(s) => {
169                    // Pay for the string, plus the quotes around it.
170                    let debit = s.len() + 2;
171                    if debit > self.tx_payload_budget as usize {
172                        return Err(driver.err_at(
173                            pos,
174                            ErrorKind::PayloadSizeTx {
175                                limit: self.max_tx_payload_size,
176                            },
177                        ));
178                    } else {
179                        // SAFETY: We know that debit <= self.tx_payload_budget, which is a u32, so
180                        // the cast and subtraction are both safe.
181                        self.tx_payload_budget -= debit as u32;
182                    }
183                }
184
185                CV::List(vs) => {
186                    // Pay for the opening and closing brackets and every comma up-front so that
187                    // deeply nested lists are not free.
188                    let debit = vs.len().saturating_sub(1) + 2;
189                    if debit > self.tx_payload_budget as usize {
190                        return Err(driver.err_at(
191                            pos,
192                            ErrorKind::PayloadSizeTx {
193                                limit: self.max_tx_payload_size,
194                            },
195                        ));
196                    } else {
197                        // SAFETY: We know that debit <= self.tx_payload_budget, which is a u32, so
198                        // the cast and subtraction are both safe.
199                        self.tx_payload_budget -= debit as u32;
200                        stack.extend(vs)
201                    }
202                }
203
204                CV::Null
205                | CV::Number(_)
206                | CV::Boolean(_)
207                | CV::Binary(_)
208                | CV::Enum(_)
209                | CV::Object(_) => {
210                    // As in `check_tx_arg`, these are safe to ignore.
211                }
212            }
213        }
214
215        Ok(())
216    }
217}
218
219impl<'r> Visitor<'r> for TxPayloadRule<'r> {
220    fn visit_field(&mut self, driver: &FieldDriver<'_, 'r>) -> Result<(), Error> {
221        if !self.has_tx_payload(driver) {
222            return Ok(());
223        }
224
225        for (name, value) in &driver.field().node.arguments {
226            self.check_tx_arg(driver, name, value)?;
227        }
228
229        Ok(())
230    }
231}

233/// Test that arguments holding transaction payloads (serialized transactions, signatures, etc) are
234/// within the service's transaction payload limit, cumulatively.
235///
236/// Similarly, test that the remaining transaction payload is within the query payload limit (which
237/// is usually smaller).
238///
239/// This check must be done after the input limit check, because it relies on the query depth being
240/// bounded to protect it from recursing too deeply.
241pub(super) fn check(
242    limits: &QueryLimitsConfig,
243    total_payload_size: u64,
244    registry: &Registry,
245    doc: &ExecutableDocument,
246    variables: &Variables,
247) -> ServerResult<Usage> {
248    let mut rule = TxPayloadRule {
249        max_tx_payload_size: limits.max_tx_payload_size,
250        tx_payload_budget: limits.max_tx_payload_size,
251        used_vars: BTreeSet::new(),
252        tx_payload_args: &limits.tx_payload_args,
253    };
254
255    Driver::visit_document(registry, doc, variables, &mut rule)?;
256
257    let tx_payload_size = limits.max_tx_payload_size - rule.tx_payload_budget;
258    let query_payload_size = total_payload_size as u32 - tx_payload_size;
259
260    if query_payload_size > limits.max_query_payload_size {
261        Err(Error::new_global(ErrorKind::PayloadSizeQuery {
262            limit: limits.max_query_payload_size,
263            actual: query_payload_size,
264        }))?;
265    }
266
267    Ok(Usage {
268        query_payload_size,
269        tx_payload_size,
270    })
