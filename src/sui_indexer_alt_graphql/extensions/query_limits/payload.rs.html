25/// The proportion of the query content that was transaction payload versus query payload.
26#[derive(Serialize, Deserialize)]
27pub(super) struct Usage {
28    pub(super) query_payload_size: u32,
29    pub(super) tx_payload_size: u32,
30}
31
32/// Validation rule that finds parts of the request that correspond to transaction payloads
33/// (transaction bytes, signatures, etc), and deducts them from an overall budget.
34///
35/// NOTE: This rule only checks top-level fields (it does not look for transaction payloads in
36/// nested structures) as currently transaction payloads can only be found there.
37struct TxPayloadRule<'r> {
38    /// The maximum number of bytes in a request that can come from transaction payloads.
39    max_tx_payload_size: u32,
40
41    /// The budget for transaction payloads remaining.
42    tx_payload_budget: u32,
43
44    /// Names of variables that have been used in transaction payloads -- variables that show up in
45    /// transaction payloads multiple times are only debited from the budget once.
46    used_vars: BTreeSet<&'r Name>,
47
48    /// Set of type, field, and argument triples that are known to contain transaction payloads.
49    tx_payload_args: &'r BTreeSet<(&'static str, &'static str, &'static str)>,
50}
51
52impl<'r> TxPayloadRule<'r> {
53    /// Whether the type and field pointed at currently by `driver` contains any transaction
54    /// payload arguments or not.
55    fn has_tx_payload(&self, driver: &FieldDriver<'_, 'r>) -> bool {
56        let type_ = driver.parent_type().name();
57        let field = driver.meta_field().name.as_str();
58
59        self.tx_payload_args
60            .range(&(type_, field, "")..)
61            .next()
62            .is_some_and(|&(t, f, _)| t == type_ && f == field)
63    }
64
65    /// Debit the size of `value` assuming `name` is a transaction payload argument of the type and
66    /// field pointed at by `driver`.
67    ///
68    /// Fails if there is insufficient remaining budget.
69    fn check_tx_arg(
70        &mut self,
71        driver: &FieldDriver<'_, 'r>,
72        name: &'r Positioned<Name>,
73        value: &'r Positioned<Value>,
74    ) -> Result<(), Error> {
75        use Value as V;
76
77        let type_ = driver.parent_type().name();
78        let field = driver.meta_field().name.as_str();
79        let arg = name.node.as_str();
80
81        if !self.tx_payload_args.contains(&(type_, field, arg)) {
82            return Ok(());
83        }
84
85        macro_rules! debit {
86            ($debit: expr) => {{
87                let debit: usize = $debit;
88                if debit > self.tx_payload_budget as usize {
89                    return Err(driver.err_at(
90                        value.pos,
91                        ErrorKind::PayloadSizeTx {
92                            limit: self.max_tx_payload_size,
93                        },
94                    ));
95                } else {
96                    // SAFETY: We know that debit <= self.tx_payload_budget, which is a u32, so
97                    // the cast and subtraction are both safe.
98                    self.tx_payload_budget -= debit as u32;
99                }
100            }};
101        }
102
103        let mut stack = vec![&value.node];
104        while let Some(v) = stack.pop() {
105            match v {
106                V::Variable(name) => self.check_tx_var(driver, value.pos, name)?,
107
108                V::String(s) => {
109                    // Pay for the string, plus the quotes around it.
110                    debit!(2 + s.len());
111                }
112
113                V::List(vs) => {
114                    // Pay for the opening and closing brackets and every comma up-front so that
115                    // deeply nested lists are not free.
116                    debit!(2 + vs.len().saturating_sub(1));
117                    stack.extend(vs);
118                }
119
120                V::Object(fs) => {
121                    // Pay for the opening and closing braces, colons, commas, and field names
122                    // up-front so that deeply nested objects are not free.
123                    debit!(
124                        2 // { and }
125                        +   fs.len().saturating_sub(1) // commas
126                        +   fs.keys().map(|k| k.as_str().len() + 1).sum::<usize>() // keys, colons
127                    );
128
129                    stack.extend(fs.values());
130                }
131
132                V::Number(n) => {
133                    // Estimate the string representation of the number.
134                    debit!(n.to_string().len());
135                }
136
137                V::Boolean(b) => {
138                    debit!(if *b { "true".len() } else { "false".len() });
139                }
140
141                V::Enum(name) => {
142                    // Pay for the enum name.
143                    debit!(name.len());
144                }
145
146                V::Binary(bs) => {
147                    // Pay for the Base64-encoded representation with quotes. Base64 encoding:
148                    // every 3 bytes becomes 4 characters.
149                    debit!(2 + bs.len().div_ceil(3) * 4);
150                }
151
152                V::Null => {
153                    debit!("null".len());
154                }
155            }
156        }
157
158        Ok(())
159    }
160
161    /// Debit the size of the value for the variable `name` (assuming it is used in a transaction
162    /// payload argument), if it has not already been used in a transaction payload (in which case
163    /// it is already accounted for).
164    ///
165    /// Fails if there is insufficient remaining budget.
166    fn check_tx_var(
167        &mut self,
168        driver: &FieldDriver<'_, 'r>,
169        pos: Pos,
170        name: &'r Name,
171    ) -> Result<(), Error> {
172        use ConstValue as CV;
173
174        // Already used in a transaction payload, don't double count.
175        if !self.used_vars.insert(name) {
176            return Ok(());
177        }
178
179        // Can't find the variable, so it can't count towards the transaction payload.
180        let Some(value) = driver.resolve_var(name) else {
181            return Ok(());
182        };
183
184        macro_rules! debit {
185            ($debit: expr) => {{
186                let debit: usize = $debit;
187                if debit > self.tx_payload_budget as usize {
188                    return Err(driver.err_at(
189                        pos,
190                        ErrorKind::PayloadSizeTx {
191                            limit: self.max_tx_payload_size,
192                        },
193                    ));
194                } else {
195                    // SAFETY: We know that debit <= self.tx_payload_budget, which is a u32, so
196                    // the cast and subtraction are both safe.
197                    self.tx_payload_budget -= debit as u32;
198                }
199            }};
200        }
201
202        let mut stack = vec![value];
203        while let Some(value) = stack.pop() {
204            // The cases below are the same as those in `check_tx_arg`, but without having to
205            // handle variables and with no bare identifiers (all identifiers -- field names and
206            // enums -- are quoted).
207            match &value {
208                // Pay for the string, plus the quotes around it.
209                CV::String(s) => {
210                    debit!(2 + s.len());
211                }
212
213                // Pay for the opening and closing brackets and every comma up-front so that deeply
214                // nested lists are not free.
215                CV::List(vs) => {
216                    debit!(2 + vs.len().saturating_sub(1));
217                    stack.extend(vs);
218                }
219
220                // Pay for the opening and closing braces, colons, commas, and field names (with
221                // quotes) up-front so that deeply nested objects are not free.
222                CV::Object(fs) => {
223                    debit!(
224                        2 // { and }
225                        +   fs.len().saturating_sub(1) // commas
226                        +   fs.keys().map(|k| k.as_str().len() + 3).sum::<usize>() // keys, quotes, colons
227                    );
228
229                    stack.extend(fs.values());
230                }
231
232                CV::Number(n) => {
233                    debit!(n.to_string().len());
234                }
235
236                CV::Boolean(b) => {
237                    debit!(if *b { "true".len() } else { "false".len() });
238                }
239
240                // Pay for the enum name, with quotes.
241                CV::Enum(name) => {
242                    debit!(2 + name.as_str().len());
243                }
244
245                CV::Binary(bs) => {
246                    debit!(2 + bs.len().div_ceil(3) * 4);
247                }
248
249                CV::Null => {
250                    debit!("null".len());
251                }
252            }
253        }
254
255        Ok(())
256    }
257}

259impl<'r> Visitor<'r> for TxPayloadRule<'r> {
260    fn visit_field(&mut self, driver: &FieldDriver<'_, 'r>) -> Result<(), Error> {
261        if !self.has_tx_payload(driver) {
262            return Ok(());
263        }
264
265        for (name, value) in &driver.field().node.arguments {
266            self.check_tx_arg(driver, name, value)?;
267        }
268
269        Ok(())
270    }
271}

273/// Test that arguments holding transaction payloads (serialized transactions, signatures, etc) are
274/// within the service's transaction payload limit, cumulatively.
275///
276/// Similarly, test that the remaining transaction payload is within the query payload limit (which
277/// is usually smaller).
278///
279/// This check must be done after the input limit check, because it relies on the query depth being
280/// bounded to protect it from recursing too deeply.
281pub(super) fn check(
282    limits: &QueryLimitsConfig,
283    total_payload_size: u64,
284    registry: &Registry,
285    doc: &ExecutableDocument,
286    variables: &Variables,
287) -> ServerResult<Usage> {
288    let mut rule = TxPayloadRule {
289        max_tx_payload_size: limits.max_tx_payload_size,
290        tx_payload_budget: limits.max_tx_payload_size,
291        used_vars: BTreeSet::new(),
292        tx_payload_args: &limits.tx_payload_args,
293    };
294
295    Driver::visit_document(registry, doc, variables, &mut rule)?;
296
297    let tx_payload_size = limits.max_tx_payload_size - rule.tx_payload_budget;
298    let query_payload_size = total_payload_size as u32 - tx_payload_size;
299
300    if query_payload_size > limits.max_query_payload_size {
301        Err(Error::new_global(ErrorKind::PayloadSizeQuery {
302            limit: limits.max_query_payload_size,
303            actual: query_payload_size,
304        }))?;
305    }
306
307    Ok(Usage {
308        query_payload_size,
309        tx_payload_size,
310    })
