4use std::sync::atomic::{AtomicUsize, Ordering};
5
6use anyhow::anyhow;
7use async_graphql::Context;
8
9use crate::config::Limits;
10use crate::error::{RpcError, resource_exhausted};
11
12/// Meters the number of "rich" queries performed in a single request. Rich queries are queries that
13/// require dedicated requests to the backing store (i.e. they cannot be batched using a
14/// data-loader).
15#[derive(Default)]
16pub(crate) struct Meter(AtomicUsize);
17
18/// Increment the rich query meter by one. If the meter exceeds the configured limit, a
19/// `RESOURCE_EXHAUSTED` error is returned.
20pub(crate) fn debit<E>(ctx: &Context<'_>) -> Result<(), RpcError<E>>
21where
22    E: std::error::Error + Send + Sync + 'static,
23{
24    let limits: &Limits = ctx.data_unchecked();
25    let meter: &Meter = ctx.data_unchecked();
26
27    // Use fetch_add with Relaxed ordering since we only need atomicity, not synchronization.
28    // The comparison against the limit will catch any overflow.
29    let prev = meter.0.fetch_add(1, Ordering::Relaxed);
30    if prev >= limits.max_rich_queries {
31        return Err(resource_exhausted(anyhow!(
32            "Exceeded the maximum number ({}) of queries that require dedicated access to a \
33             backing store in a single request.",
34            limits.max_rich_queries
35        )));
36    }
37    Ok(())
38}
