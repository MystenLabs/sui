4use axum::http::HeaderName;
5use axum::http::HeaderValue;
6use headers::Error;
7use headers::Header;
8
9static SHOW_USAGE: HeaderName = HeaderName::from_static("x-sui-rpc-show-usage");
10
11/// Header indicating that the client would like their GraphQL response to be extended with how
12/// much of each limit the query is using. The value of this header doesn't matter, it just has to
13/// be present.
14pub(crate) struct ShowUsage(pub HeaderValue);
15
16impl Header for ShowUsage {
17    fn name() -> &'static HeaderName {
18        &SHOW_USAGE
19    }
20
21    fn decode<'i, I: Iterator<Item = &'i HeaderValue>>(values: &mut I) -> Result<Self, Error> {
22        Ok(ShowUsage(values.next().ok_or_else(Error::invalid)?.clone()))
23    }
24
25    fn encode<E: Extend<HeaderValue>>(&self, values: &mut E) {
26        values.extend([self.0.clone()]);
27    }
28}
