22/// Type representing the `__typename` meta field in GraphQL. This doesn't appear in the schema, but
23/// the visitor may still need to visit it.
24static TYPENAME: LazyLock<MetaField> = LazyLock::new(|| MetaField {
25    name: "__typename".to_owned(),
26    description: Some("Meta field that returns the name of the Object's type.".to_owned()),
27    args: Default::default(),
28    ty: "String".to_owned(),
29    deprecation: Default::default(),
30    cache_control: Default::default(),
31    external: false,
32    requires: None,
33    provides: None,
34    shareable: false,
35    inaccessible: false,
36    tags: Default::default(),
37    override_from: None,
38    visible: None,
39    compute_complexity: None,
40    directive_invocations: Default::default(),
41});

43/// Visitors traverse the AST of a query, performing an action on each field.
44pub(super) trait Visitor<'r> {
45    /// This function is called on each field in the query.
46    ///
47    /// The `driver` parameter gives access to the field's value and type information, and controls
48    /// when and how to recurse into nested field selections.
49    fn visit_field<'t>(&mut self, driver: &FieldDriver<'t, 'r>) -> Result<(), Error>;
50}
51
52/// Like [Driver] but specialized to a specific field.
53///
54/// ## Lifetimes
55/// - `'t`: The lifetime of the traversal.
56/// - `'r`: The lifetime of the request.
57pub(super) struct FieldDriver<'t, 'r> {
58    driver: &'t Driver<'r>,
59    chain: Option<Rc<Chain>>,
60    meta_type: &'r MetaType,
61    meta_field: &'r MetaField,
62    field: &'r Positioned<Field>,
63}
64
65/// The driver abstracts over recursively visiting a GraphQL document. It keeps track of type
66/// information, path and source position for the current field.
67pub(super) struct Driver<'r> {
68    registry: &'r Registry,
69    fragments: &'r HashMap<Name, Positioned<FragmentDefinition>>,
70    variables: &'r Variables,
71}
72
73impl<'r> FieldDriver<'_, 'r> {
74    /// Metadata about the field's parent type.
75    pub(super) fn parent_type(&self) -> &'r MetaType {
76        self.meta_type
77    }
78
79    /// Metadata about the field.
80    pub(super) fn meta_field(&self) -> &'r MetaField {
81        self.meta_field
82    }
83
84    /// The field's AST node.
85    pub(super) fn field(&self) -> &'r Positioned<Field> {
86        self.field
87    }
88
89    /// Find an argument on the current field by its name, and return its fully resolved value if
90    /// it exists, or `None` if it does not. Missing variables resolve to `null`.
91    pub(super) fn resolve_arg(&self, name: &str) -> Option<ConstValue> {
92        self.field.node.arguments.iter().find_map(|(n, v)| {
93            (n.node.as_str() == name).then_some(self.resolve_val(v.node.clone()))
94        })
95    }
96
97    /// Return `val` with variables all resolved. Missing variables resolve to `null`.
98    pub(super) fn resolve_val(&self, val: Value) -> ConstValue {
99        let val: Result<_, Infallible> = val.into_const_with(|name| {
100            Ok(self.resolve_var(&name).cloned().unwrap_or(ConstValue::Null))
101        });
102
103        match val {
104            Ok(v) => v,
105            Err(e) => match e {},
106        }
107    }
108
109    /// Find the value bound to a given GraphQL variable name.
110    pub(super) fn resolve_var(&self, name: &Name) -> Option<&'r ConstValue> {
111        self.driver.variables.get(name)
112    }
113
114    /// Helper to contextualize an error with the field's position and path.
115    pub(super) fn err(&self, kind: ErrorKind) -> Error {
116        self.err_at(self.field.pos, kind)
117    }
118
119    /// Helper to contextualize an error with the field's path and a custom position.
120    pub(super) fn err_at(&self, pos: Pos, kind: ErrorKind) -> Error {
121        Error::new(kind, Chain::path(&self.chain), pos)
122    }
123
124    /// Continue traversing the query, visiting the selection of fields nested within these field,
125    /// if there are any.
126    pub(super) fn visit_selection_set<V: Visitor<'r> + ?Sized>(
127        &self,
128        visitor: &mut V,
129    ) -> Result<(), Error> {
130        self.driver.visit_selection_set(
131            &self.meta_field.ty,
132            self.chain.clone(),
133            &self.field.node.selection_set,
134            visitor,
135        )
136    }
137}

139impl<'r> Driver<'r> {
140    /// Entry point for visiting a document with a particular visitor.
141    pub(super) fn visit_document<V: Visitor<'r> + ?Sized>(
142        registry: &'r Registry,
143        doc: &'r ExecutableDocument,
144        variables: &'r Variables,
145        visitor: &mut V,
146    ) -> Result<(), Error> {
147        let driver = Self {
148            registry,
149            fragments: &doc.fragments,
150            variables,
151        };
152
153        for (name, op) in doc.operations.iter() {
154            driver.visit_operation(name, op, visitor)?;
155        }
156
157        Ok(())
158    }
159
160    fn visit_operation<V: Visitor<'r> + ?Sized>(
161        &self,
162        _name: Option<&Name>,
163        op: &'r Positioned<OperationDefinition>,
164        visitor: &mut V,
165    ) -> Result<(), Error> {
166        let Some(root) = (match op.node.ty {
167            OperationType::Query => Some(&self.registry.query_type),
168            OperationType::Mutation => self.registry.mutation_type.as_ref(),
169            OperationType::Subscription => self.registry.subscription_type.as_ref(),
170        }) else {
171            return Err(Error::new_global(ErrorKind::SchemaNotSupported(op.node.ty)));
172        };
173
174        self.visit_selection_set(root, None, &op.node.selection_set, visitor)
175    }
176
177    fn visit_selection_set<V: Visitor<'r> + ?Sized>(
178        &self,
179        type_: &'r str,
180        chain: Option<Rc<Chain>>,
181        selection_set: &'r Positioned<SelectionSet>,
182        visitor: &mut V,
183    ) -> Result<(), Error> {
184        let Some(meta_type) = self.registry.concrete_type_by_name(type_) else {
185            return Err(Error::new(
186                ErrorKind::InternalError(anyhow!("Type '{type_}' not found in schema")),
187                Chain::path(&chain),
188                selection_set.pos,
189            ));
190        };
191
192        for sel in &selection_set.node.items {
193            match &sel.node {
194                Selection::Field(field) => {
195                    let name = &field.node.name.node;
196                    let meta_field = if name == "__typename" {
197                        &*TYPENAME
198                    } else {
199                        meta_type.field_by_name(name.as_str()).ok_or_else(|| {
200                            Error::new(
201                                ErrorKind::InternalError(anyhow!(
202                                    "Field '{type_}.{name}' not found in schema"
203                                )),
204                                Chain::path(&chain),
205                                field.pos,
206                            )
207                        })?
208                    };
209
210                    visitor.visit_field(&FieldDriver {
211                        driver: self,
212                        chain: Some(Chain::new(chain.clone(), name.clone())),
213                        meta_type,
214                        meta_field,
215                        field,
216                    })?;
217                }
218
219                Selection::FragmentSpread(fragment_spread) => {
220                    let name = &fragment_spread.node.fragment_name.node;
221                    let fragment_def = self.fragments.get(name).ok_or_else(|| {
222                        Error::new(
223                            ErrorKind::UnknownFragment(name.as_str().to_owned()),
224                            Chain::path(&chain),
225                            fragment_spread.pos,
226                        )
227                    })?;
228
229                    self.visit_selection_set(
230                        fragment_def.node.type_condition.node.on.node.as_str(),
231                        chain.clone(),
232                        &fragment_def.node.selection_set,
233                        visitor,
234                    )?;
235                }
236
237                Selection::InlineFragment(inline_fragment) => {
238                    self.visit_selection_set(
239                        inline_fragment
240                            .node
241                            .type_condition
242                            .as_ref()
243                            .map_or(type_, |cond| cond.node.on.node.as_str()),
244                        chain.clone(),
245                        &inline_fragment.node.selection_set,
246                        visitor,
247                    )?;
248                }
249            }
250        }
251
252        Ok(())
253    }
