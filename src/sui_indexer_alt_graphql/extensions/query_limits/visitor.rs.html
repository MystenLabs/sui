31/// Type representing the `__typename` meta field in GraphQL. This doesn't appear in the schema, but
32/// the visitor may still need to visit it.
33static TYPENAME: LazyLock<MetaField> = LazyLock::new(|| MetaField {
34    name: "__typename".to_owned(),
35    description: Some("Meta field that returns the name of the Object's type.".to_owned()),
36    args: Default::default(),
37    ty: "String".to_owned(),
38    deprecation: Default::default(),
39    cache_control: Default::default(),
40    external: false,
41    requires: None,
42    provides: None,
43    shareable: false,
44    inaccessible: false,
45    tags: Default::default(),
46    override_from: None,
47    visible: None,
48    compute_complexity: None,
49    directive_invocations: Default::default(),
50});

52/// Visitors traverse the AST of a query, performing an action on each field.
53pub(super) trait Visitor<'r> {
54    /// This function is called on each field in the query.
55    ///
56    /// The `driver` parameter gives access to the field's value and type information, and controls
57    /// when and how to recurse into nested field selections.
58    fn visit_field<'t>(&mut self, driver: &FieldDriver<'t, 'r>) -> Result<(), Error>;
59}
60
61/// Like [Driver] but specialized to a specific field.
62///
63/// ## Lifetimes
64/// - `'t`: The lifetime of the traversal.
65/// - `'r`: The lifetime of the request.
66pub(super) struct FieldDriver<'t, 'r> {
67    driver: &'t Driver<'r>,
68    chain: Option<Rc<Chain>>,
69    meta_type: &'r MetaType,
70    meta_field: &'r MetaField,
71    field: &'r Positioned<Field>,
72}
73
74/// The driver abstracts over recursively visiting a GraphQL document. It keeps track of type
75/// information, path and source position for the current field.
76pub(super) struct Driver<'r> {
77    registry: &'r Registry,
78    fragments: &'r HashMap<Name, Positioned<FragmentDefinition>>,
79    variables: &'r Variables,
80}
81
82impl<'r> FieldDriver<'_, 'r> {
83    /// Metadata about the field's parent type.
84    pub(super) fn parent_type(&self) -> &'r MetaType {
85        self.meta_type
86    }
87
88    /// Metadata about the field.
89    pub(super) fn meta_field(&self) -> &'r MetaField {
90        self.meta_field
91    }
92
93    /// The field's AST node.
94    pub(super) fn field(&self) -> &'r Positioned<Field> {
95        self.field
96    }
97
98    /// Find an argument on the current field by its name, and return its fully resolved value if
99    /// it exists, or `None` if it does not. Missing variables resolve to `null`.
100    pub(super) fn resolve_arg(&self, name: &str) -> Option<ConstValue> {
101        self.field.node.arguments.iter().find_map(|(n, v)| {
102            (n.node.as_str() == name).then_some(self.resolve_val(v.node.clone()))
103        })
104    }
105
106    /// Return `val` with variables all resolved. Missing variables resolve to `null`.
107    pub(super) fn resolve_val(&self, val: Value) -> ConstValue {
108        let val: Result<_, Infallible> = val.into_const_with(|name| {
109            Ok(self.resolve_var(&name).cloned().unwrap_or(ConstValue::Null))
110        });
111
112        match val {
113            Ok(v) => v,
114            Err(e) => match e {},
115        }
116    }
117
118    /// Find the value bound to a given GraphQL variable name.
119    pub(super) fn resolve_var(&self, name: &Name) -> Option<&'r ConstValue> {
120        self.driver.variables.get(name)
121    }
122
123    /// Helper to contextualize an error with the field's position and path.
124    pub(super) fn err(&self, kind: ErrorKind) -> Error {
125        self.err_at(self.field.pos, kind)
126    }
127
128    /// Helper to contextualize an error with the field's path and a custom position.
129    pub(super) fn err_at(&self, pos: Pos, kind: ErrorKind) -> Error {
130        Error::new(kind, Chain::path(&self.chain), pos)
131    }
132
133    /// Continue traversing the query, visiting the selection of fields nested within these field,
134    /// if there are any.
135    pub(super) fn visit_selection_set<V: Visitor<'r> + ?Sized>(
136        &self,
137        visitor: &mut V,
138    ) -> Result<(), Error> {
139        self.driver.visit_selection_set(
140            &self.meta_field.ty,
141            self.chain.clone(),
142            &self.field.node.selection_set,
143            visitor,
144        )
145    }
146}

148impl<'r> Driver<'r> {
149    /// Entry point for visiting a document with a particular visitor.
150    pub(super) fn visit_document<V: Visitor<'r> + ?Sized>(
151        registry: &'r Registry,
152        doc: &'r ExecutableDocument,
153        variables: &'r Variables,
154        visitor: &mut V,
155    ) -> Result<(), Error> {
156        let driver = Self {
157            registry,
158            fragments: &doc.fragments,
159            variables,
160        };
161
162        for (name, op) in doc.operations.iter() {
163            driver.visit_operation(name, op, visitor)?;
164        }
165
166        Ok(())
167    }
168
169    fn visit_operation<V: Visitor<'r> + ?Sized>(
170        &self,
171        _name: Option<&Name>,
172        op: &'r Positioned<OperationDefinition>,
173        visitor: &mut V,
174    ) -> Result<(), Error> {
175        let Some(root) = (match op.node.ty {
176            OperationType::Query => Some(&self.registry.query_type),
177            OperationType::Mutation => self.registry.mutation_type.as_ref(),
178            OperationType::Subscription => self.registry.subscription_type.as_ref(),
179        }) else {
180            return Err(Error::new_global(ErrorKind::SchemaNotSupported(op.node.ty)));
181        };
182
183        self.visit_selection_set(root, None, &op.node.selection_set, visitor)
184    }
185
186    fn visit_selection_set<V: Visitor<'r> + ?Sized>(
187        &self,
188        type_: &'r str,
189        chain: Option<Rc<Chain>>,
190        selection_set: &'r Positioned<SelectionSet>,
191        visitor: &mut V,
192    ) -> Result<(), Error> {
193        let Some(meta_type) = self.registry.concrete_type_by_name(type_) else {
194            return Err(Error::new(
195                ErrorKind::InternalError(anyhow!("Type '{type_}' not found in schema")),
196                Chain::path(&chain),
197                selection_set.pos,
198            ));
199        };
200
201        for sel in &selection_set.node.items {
202            match &sel.node {
203                Selection::Field(field) => {
204                    let name = &field.node.name.node;
205                    let meta_field = if name == "__typename" {
206                        &*TYPENAME
207                    } else {
208                        meta_type.field_by_name(name.as_str()).ok_or_else(|| {
209                            Error::new(
210                                ErrorKind::InternalError(anyhow!(
211                                    "Field '{type_}.{name}' not found in schema"
212                                )),
213                                Chain::path(&chain),
214                                field.pos,
215                            )
216                        })?
217                    };
218
219                    visitor.visit_field(&FieldDriver {
220                        driver: self,
221                        chain: Some(Chain::new(chain.clone(), name.clone())),
222                        meta_type,
223                        meta_field,
224                        field,
225                    })?;
226                }
227
228                Selection::FragmentSpread(fragment_spread) => {
229                    let name = &fragment_spread.node.fragment_name.node;
230                    let fragment_def = self.fragments.get(name).ok_or_else(|| {
231                        Error::new(
232                            ErrorKind::UnknownFragment(name.as_str().to_owned()),
233                            Chain::path(&chain),
234                            fragment_spread.pos,
235                        )
236                    })?;
237
238                    self.visit_selection_set(
239                        fragment_def.node.type_condition.node.on.node.as_str(),
240                        chain.clone(),
241                        &fragment_def.node.selection_set,
242                        visitor,
243                    )?;
244                }
245
246                Selection::InlineFragment(inline_fragment) => {
247                    self.visit_selection_set(
248                        inline_fragment
249                            .node
250                            .type_condition
251                            .as_ref()
252                            .map_or(type_, |cond| cond.node.on.node.as_str()),
253                        chain.clone(),
254                        &inline_fragment.node.selection_set,
255                        visitor,
256                    )?;
257                }
258            }
259        }
260
261        Ok(())
262    }
