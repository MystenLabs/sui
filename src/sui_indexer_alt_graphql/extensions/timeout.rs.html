27/// How long to wait for each kind of operation before timing out.
28pub(crate) struct TimeoutConfig {
29    pub(crate) query: Duration,
30    pub(crate) mutation: Duration,
31}
32
33/// The timeout extension is responsible for limiting the amount of time spent serving any single
34/// request. It is configured by [RpcConfig] which it expects to find in its context. Timeout
35/// durations are configured separately for mutations and for queries.
36pub(crate) struct Timeout(Arc<TimeoutConfig>);
37
38struct TimeoutExt {
39    config: Arc<TimeoutConfig>,
40    /// Map from operation name to its type, populated once during parsing.
41    /// `None` key represents an anonymous (single) operation.
42    operation_types: OnceLock<HashMap<Option<String>, OperationType>>,
43}
44
45impl Timeout {
46    pub(crate) fn new(config: TimeoutConfig) -> Self {
47        Self(Arc::new(config))
48    }
49}
50
51impl ExtensionFactory for Timeout {
52    fn create(&self) -> Arc<dyn Extension> {
53        Arc::new(TimeoutExt {
54            config: self.0.clone(),
55            operation_types: OnceLock::new(),
56        })
57    }
58}
59
60#[async_trait::async_trait]
61impl Extension for TimeoutExt {
62    async fn parse_query(
63        &self,
64        ctx: &ExtensionContext<'_>,
65        query: &str,
66        variables: &Variables,
67        next: NextParseQuery<'_>,
68    ) -> ServerResult<ExecutableDocument> {
69        let document = next.run(ctx, query, variables).await?;
70
71        let types: HashMap<_, _> = document
72            .operations
73            .iter()
74            .map(|(name, op)| (name.map(|n| n.to_string()), op.node.ty))
75            .collect();
76        let _ = self.operation_types.set(types);
77
78        Ok(document)
79    }
80
81    async fn execute(
82        &self,
83        ctx: &ExtensionContext<'_>,
84        operation_name: Option<&str>,
85        next: NextExecute<'_>,
86    ) -> Response {
87        let key = operation_name.map(|s| s.to_string());
88        let operation_type = self
89            .operation_types
90            .get()
91            .and_then(|types| types.get(&key).copied())
92            .unwrap_or(OperationType::Query);
93
94        // Subscriptions are long-lived streams — a per-request timeout does not apply.
95        if operation_type == OperationType::Subscription {
96            return next.run(ctx, operation_name).await;
97        }
98
99        let is_mutation = operation_type == OperationType::Mutation;
100        let limit = if is_mutation {
101            self.config.mutation
102        } else {
103            self.config.query
104        };
105
106        timeout(limit, CaptureSpanTrace, next.run(ctx, operation_name))
107            .await
108            .unwrap_or_else(|e| {
109                let kind = if is_mutation { "Mutation" } else { "Query" };
110                let Session { uuid, .. } = ctx.data_unchecked();
111                warn!(request_id = %uuid, %kind, "Request timed out: {e}");
112                Response::from_errors(vec![ServerError::from(request_timeout(kind, limit))])
113            })
114    }
115}

151    /// The request takes less than the timeout to handle, so it should pass.
152    #[tokio::test]
153    async fn test_query_timeout_pass() {
154        let zero = Duration::from_millis(0);
155        let delay = Duration::from_millis(200);
156        let response = Schema::build(Root(delay / 2), EmptyMutation, EmptySubscription)
157            .extension(Timeout::new(TimeoutConfig {
158                query: delay,
159                mutation: zero,
160            }))
161            .finish()
162            .execute("query { op }")
163            .await;
164
165        assert!(response.is_ok());
166    }

255    async fn test_timeout_fail(
256        query_timeout: Duration,
257        mutation_timeout: Duration,
258        delay: Duration,
259        request: &str,
260        expected_error: &str,
261    ) -> String {
262        // Enable tracing, configured by environment variables.
263        let (_guard, handle) = TelemetryConfig::new()
264            .with_set_global_default(false)
265            // Enable to match default in main.rs
266            .with_enable_error_layer(true)
267            // Required for handle.get_test_layer_events()
268            .with_enable_test_layer(true)
269            .init();
270
271        let root = Root(delay);
272
273        let response = Schema::build(root.clone(), root, EmptySubscription)
274            // Timeout reads session data. This either needs to be set by the GraphQL framework or
275            // a test like this if bypassing the GraphQL framework.
276            .data(Session {
277                // ffffffff-ffff-ffff-ffff-ffffffffffff
278                uuid: Uuid::from_bytes([255; 16]),
279                addr: SocketAddr::new(Ipv4Addr::UNSPECIFIED.into(), 0),
280                client: ClientInfo::default(),
281            })
282            .extension(Timeout::new(TimeoutConfig {
283                query: query_timeout,
284                mutation: mutation_timeout,
285            }))
286            .extension(Tracing)
287            .finish()
288            .execute(request)
289            .await;
290
291        assert!(response.is_err());
292
293        let error = &response.errors[0];
294        assert!(error.message.contains(expected_error));
295        assert_eq!(
296            error.extensions.as_ref().unwrap().get("code"),
297            Some(&Value::String(code::REQUEST_TIMEOUT.into()))
298        );
299
300        let events = handle.get_test_layer_events();
301        assert_eq!(events.len(), 1);
302        // Remove line number strings to avoid test churn
303        // example: "             at /Users/evanwall/.cargo/git/checkouts/async-graphql-7336e61dcafca7ed/7be9351/src/extensions/tracing.rs:135"
304        let re = Regex::new(r"\s+at\s.+").unwrap();
305        events[0].split("\n").filter(|s| !re.is_match(s)).join("\n")
306    }
