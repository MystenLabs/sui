27/// How long to wait for each kind of operation before timing out.
28pub(crate) struct TimeoutConfig {
29    pub(crate) query: Duration,
30    pub(crate) mutation: Duration,
31}
32
33/// The timeout extension is responsible for limiting the amount of time spent serving any single
34/// request. It is configured by [RpcConfig] which it expects to find in its context. Timeout
35/// durations are configured separately for mutations and for queries.
36pub(crate) struct Timeout(Arc<TimeoutConfig>);
37
38struct TimeoutExt {
39    config: Arc<TimeoutConfig>,
40    is_mutation: AtomicBool,
41}
42
43impl Timeout {
44    pub(crate) fn new(config: TimeoutConfig) -> Self {
45        Self(Arc::new(config))
46    }
47}
48
49impl ExtensionFactory for Timeout {
50    fn create(&self) -> Arc<dyn Extension> {
51        Arc::new(TimeoutExt {
52            config: self.0.clone(),
53            is_mutation: AtomicBool::new(false),
54        })
55    }
56}
57
58#[async_trait::async_trait]
59impl Extension for TimeoutExt {
60    async fn parse_query(
61        &self,
62        ctx: &ExtensionContext<'_>,
63        query: &str,
64        variables: &Variables,
65        next: NextParseQuery<'_>,
66    ) -> ServerResult<ExecutableDocument> {
67        let document = next.run(ctx, query, variables).await?;
68
69        self.is_mutation.store(
70            document
71                .operations
72                .iter()
73                .any(|(_, op)| op.node.ty == OperationType::Mutation),
74            Ordering::Relaxed,
75        );
76
77        Ok(document)
78    }
79
80    async fn execute(
81        &self,
82        ctx: &ExtensionContext<'_>,
83        operation_name: Option<&str>,
84        next: NextExecute<'_>,
85    ) -> Response {
86        let is_mutation = self.is_mutation.load(Ordering::Relaxed);
87        let limit = if is_mutation {
88            self.config.mutation
89        } else {
90            self.config.query
91        };
92
93        timeout(limit, CaptureSpanTrace, next.run(ctx, operation_name))
94            .await
95            .unwrap_or_else(|e| {
96                let kind = if is_mutation { "Mutation" } else { "Query" };
97                let Session { uuid, .. } = ctx.data_unchecked();
98                warn!(request_id = %uuid, %kind, "Request timed out: {e}");
99                Response::from_errors(vec![ServerError::from(request_timeout(kind, limit))])
100            })
101    }
102}

136    /// The request takes less than the timeout to handle, so it should pass.
137    #[tokio::test]
138    async fn test_query_timeout_pass() {
139        let zero = Duration::from_millis(0);
140        let delay = Duration::from_millis(200);
141        let response = Schema::build(Root(delay / 2), EmptyMutation, EmptySubscription)
142            .extension(Timeout::new(TimeoutConfig {
143                query: delay,
144                mutation: zero,
145            }))
146            .finish()
147            .execute("query { op }")
148            .await;
149
150        assert!(response.is_ok());
151    }

240    async fn test_timeout_fail(
241        query_timeout: Duration,
242        mutation_timeout: Duration,
243        delay: Duration,
244        request: &str,
245        expected_error: &str,
246    ) -> String {
247        // Enable tracing, configured by environment variables.
248        let (_guard, handle) = TelemetryConfig::new()
249            .with_set_global_default(false)
250            // Enable to match default in main.rs
251            .with_enable_error_layer(true)
252            // Required for handle.get_test_layer_events()
253            .with_enable_test_layer(true)
254            .init();
255
256        let root = Root(delay);
257
258        let response = Schema::build(root.clone(), root, EmptySubscription)
259            // Timeout reads session data. This either needs to be set by the GraphQL framework or
260            // a test like this if bypassing the GraphQL framework.
261            .data(Session {
262                // ffffffff-ffff-ffff-ffff-ffffffffffff
263                uuid: Uuid::from_bytes([255; 16]),
264                addr: SocketAddr::new(Ipv4Addr::UNSPECIFIED.into(), 0),
265            })
266            .extension(Timeout::new(TimeoutConfig {
267                query: query_timeout,
268                mutation: mutation_timeout,
269            }))
270            .extension(Tracing)
271            .finish()
272            .execute(request)
273            .await;
274
275        assert!(response.is_err());
276
277        let error = &response.errors[0];
278        assert!(error.message.contains(expected_error));
279        assert_eq!(
280            error.extensions.as_ref().unwrap().get("code"),
281            Some(&Value::String(code::REQUEST_TIMEOUT.into()))
282        );
283
284        let events = handle.get_test_layer_events();
285        assert_eq!(events.len(), 1);
286        // Remove line number strings to avoid test churn
287        // example: "             at /Users/evanwall/.cargo/git/checkouts/async-graphql-7336e61dcafca7ed/7be9351/src/extensions/tracing.rs:135"
288        let re = Regex::new(r"\s+at\s.+").unwrap();
289        events[0].split("\n").filter(|s| !re.is_match(s)).join("\n")
290    }
