21/// Extension that holds a DB URL to probe as part of health checks.
22#[derive(Clone)]
23pub(crate) struct DbProbe(pub(crate) Option<Url>);
24
25/// Query params for the health check endpoint.
26#[derive(Deserialize)]
27pub(crate) struct Params {
28    /// customise the max allowed checkpoint lag. If it is omitted, the default lag configured for
29    /// the service is used.
30    max_checkpoint_lag_ms: Option<u64>,
31}
32
33/// Response body for the health check endpoint. This is output as JSON.
34#[derive(Serialize)]
35pub(crate) struct Response {
36    #[serde(skip_serializing_if = "Option::is_none")]
37    checkpoint_lag_ms: Option<u64>,
38
39    #[serde(skip_serializing_if = "IndexMap::is_empty")]
40    pipelines: IndexMap<String, u64>,
41
42    #[serde(skip_serializing_if = "Vec::is_empty")]
43    errors: Vec<String>,
44}
45
46/// Health check endpoint outputs information about the services health -- how recent its
47/// information is, and any health check related errors. The response status code is
48/// INTERNAL_SERVER_ERROR if there are any errors, and OK otherwise (if the service is healthy).
49pub(crate) async fn check(
50    Extension(watermarks): Extension<WatermarksLock>,
51    Extension(config): Extension<HealthConfig>,
52    Extension(DbProbe(db_url)): Extension<DbProbe>,
53    Query(params): Query<Params>,
54) -> Response {
55    let mut errors = vec![];
56
57    let (lag, pipelines) = check_watermarks(&watermarks).await;
58
59    if let Err(e) = check_connection(&db_url).await {
60        errors.push(format!("DB probe failed: {e}"));
61    }
62
63    let max_lag = params
64        .max_checkpoint_lag_ms
65        .unwrap_or(config.max_checkpoint_lag.as_millis() as u64);
66
67    if lag > max_lag {
68        errors.push("Watermark lag is too high".to_owned());
69    }
70
71    Response {
72        checkpoint_lag_ms: Some(lag),
73        pipelines,
74        errors,
75    }
76}
77
78/// Returns the lag between the latest checkpoint the indexer is aware of and the current time.
79async fn check_watermarks(watermarks: &WatermarksLock) -> (u64, IndexMap<String, u64>) {
80    let now = Utc::now();
81    let watermarks = watermarks.read().await;
82
83    let mut pipeline_lags: Vec<(String, u64)> = watermarks
84        .per_pipeline()
85        .iter()
86        .map(|(pipeline, watermark)| (pipeline.clone(), watermark.lag_ms(now)))
87        .collect();
88    pipeline_lags.sort_by(|(name_a, lag_a), (name_b, lag_b)| {
89        lag_b.cmp(lag_a).then_with(|| name_a.cmp(name_b))
90    });
91
92    let pipelines = pipeline_lags.into_iter().collect();
93    (watermarks.lag_ms(now), pipelines)
94}
95
96/// Checks that the service can talk to the database.
97async fn check_connection(url: &Option<Url>) -> anyhow::Result<()> {
98    // No URL configured to probe.
99    let Some(url) = url else {
100        return Ok(());
101    };
102
103    let addrs = url
104        .socket_addrs(|| None)
105        .map_err(|_| anyhow!("Could not resolve URL"))?;
106
107    TcpStream::connect(addrs.as_slice())
108        .await
109        .map_err(|_| anyhow!("Failed to establish TCP connection"))?;
110
111    Ok(())
112}

114impl IntoResponse for Response {
115    fn into_response(self) -> AxumResponse {
116        let status = if self.errors.is_empty() {
117            StatusCode::OK
118        } else {
119            StatusCode::INTERNAL_SERVER_ERROR
120        };
121
122        (status, Json(self)).into_response()
123    }
