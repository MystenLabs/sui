69#[derive(clap::Args, Clone, Debug)]
70pub struct RpcArgs {
71    /// Address to accept incoming RPC connections on.
72    #[clap(long, default_value_t = Self::default().rpc_listen_address)]
73    pub rpc_listen_address: SocketAddr,
74
75    /// Do not expose the GraphiQL IDE.
76    #[clap(long, default_value_t = Self::default().no_ide)]
77    pub no_ide: bool,
78}

80/// This type is responsible for the set-up and lifecycle of all services related to a GraphQL RPC
81/// service (the RPC service, online IDE, health checks etc). It is agnostic to the schema
82/// being served (which must be provided to `run`).
83pub struct RpcService<Q, M, S> {
84    /// Address to accept incoming RPC connections on.
85    rpc_listen_address: SocketAddr,
86
87    /// Whether to expose GraphiQL IDE on GET requests
88    with_ide: bool,
89
90    /// The version string to report with each response, as an HTTP header.
91    version: &'static str,
92
93    /// The axum router that will handle incoming requests.
94    router: Router,
95
96    /// The GraphQL schema this service will serve.
97    schema: SchemaBuilder<Q, M, S>,
98
99    /// Metrics for the RPC service.
100    metrics: Arc<RpcMetrics>,
101
102    /// Cancellation token controls lifecycle of all RPC-related services.
103    cancel: CancellationToken,
104}
105
106impl<Q, M, S> RpcService<Q, M, S>
107where
108    Q: ObjectType + 'static,
109    M: ObjectType + 'static,
110    S: SubscriptionType + 'static,
111{
112    pub fn new(
113        args: RpcArgs,
114        version: &'static str,
115        schema: SchemaBuilder<Q, M, S>,
116        registry: &Registry,
117        cancel: CancellationToken,
118    ) -> Self {
119        let RpcArgs {
120            rpc_listen_address,
121            no_ide,
122        } = args;
123
124        let metrics = RpcMetrics::new(registry);
125        let router = Router::new();
126
127        // The logging extension should be outermost so that it can surround all other extensions.
128        let schema = schema.extension(Logging(metrics.clone()));
129
130        Self {
131            rpc_listen_address,
132            with_ide: !no_ide,
133            version,
134            router,
135            schema,
136            metrics,
137            cancel,
138        }
139    }
140
141    /// Return a copy of the metrics.
142    pub fn metrics(&self) -> Arc<RpcMetrics> {
143        self.metrics.clone()
144    }
145
146    /// Add a handler to the axum router.
147    pub fn route(self, path: &str, router: MethodRouter) -> Self {
148        let router = self.router.route(path, router);
149        Self { router, ..self }
150    }
151
152    /// Add an extension as a layer to the axum router.
153    pub fn layer<T>(self, layer: T) -> Self
154    where
155        T: Clone + Send + Sync + 'static,
156    {
157        let router = self.router.layer(Extension(layer));
158        Self { router, ..self }
159    }
160
161    /// Add an extension to the GraphQL schema.
162    pub fn extension(self, ext: impl ExtensionFactory) -> Self {
163        let schema = self.schema.extension(ext);
164        Self { schema, ..self }
165    }
166
167    /// Add data to the GraphQL schema that can be accessed from the Context from any request.
168    pub fn data<D: Any + Send + Sync>(self, data: D) -> Self {
169        let schema = self.schema.data(data);
170        Self { schema, ..self }
171    }
172
173    /// Run the RPC service. This binds the listener and exposes handlers for the RPC service and IDE
174    /// (if enabled).
175    pub async fn run(self) -> anyhow::Result<JoinHandle<()>>
176    where
177        Q: ObjectType + 'static,
178        M: ObjectType + 'static,
179        S: SubscriptionType + 'static,
180    {
181        let Self {
182            rpc_listen_address,
183            with_ide,
184            version,
185            mut router,
186            schema,
187            metrics: _,
188            cancel,
189        } = self;
190
191        if with_ide {
192            info!("Starting GraphiQL IDE at 'http://{rpc_listen_address}/graphql'");
193            router = router.route("/graphql", get(graphiql));
194        } else {
195            info!("Skipping GraphiQL IDE setup");
196        }
197
198        router = router
199            .layer(Extension(schema.finish()))
200            .layer(axum::middleware::from_fn_with_state(
201                Version(version),
202                middleware::version::set_version,
203            ))
204            .layer(
205                cors::CorsLayer::new()
206                    .allow_methods([Method::POST])
207                    .allow_origin(cors::Any)
208                    .allow_headers(cors::Any),
209            );
210
211        info!("Starting GraphQL service on {rpc_listen_address}");
212        let listener = TcpListener::bind(rpc_listen_address)
213            .await
214            .context("Failed to bind GraphQL to listen address")?;
215
216        let service = axum::serve(
217            listener,
218            router.into_make_service_with_connect_info::<SocketAddr>(),
219        )
220        .with_graceful_shutdown({
221            let cancel = cancel.clone();
222            async move {
223                cancel.cancelled().await;
224                info!("Shutdown received, shutting down GraphQL service");
225            }
226        });
227
228        Ok(tokio::spawn(async move {
229            if let Err(e) = service.await.context("Failed to start GraphQL service") {
230                error!("Failed to start GraphQL service: {e:?}");
231                cancel.cancel();
232            }
233        }))
234    }
235}

241            no_ide: false,
242        }
243    }

246/// The GraphQL schema this service will serve, without any extensions or context added.
247pub fn schema() -> SchemaBuilder<Query, Mutation, EmptySubscription> {
248    Schema::build(Query::default(), Mutation, EmptySubscription)
249        .register_output_type::<IAddressable>()
250        .register_output_type::<IMoveDatatype>()
251        .register_output_type::<IMoveObject>()
252        .register_output_type::<IObject>()
253}

255/// Set-up and run the RPC service, using the provided arguments (expected to be extracted from the
256/// command-line). The service will continue to run until the cancellation token is triggered, and
257/// will signal cancellation on the token when it is shutting down.
258///
259/// Access to most reads is controlled by the `database_url` -- if it is `None`, those reads will
260/// not work. KV queries can optionally be served by a Bigtable instance, if `bigtable_instance` is
261/// provided, otherwise these requests are served by the database. If a `bigtable_instance` is
262/// provided, the `GOOGLE_APPLICATION_CREDENTIALS` environment variable must point to the
263/// credentials JSON file.
264///
265/// `version` is the version string reported in response headers by the service as part of every
266/// request.
267///
268/// The service may spin up auxiliary services (such as the system package task) to support itself,
269/// and will clean these up on shutdown as well.
270pub async fn start_rpc(
271    database_url: Option<Url>,
272    bigtable_instance: Option<String>,
273    fullnode_args: FullnodeArgs,
274    db_args: DbArgs,
275    bigtable_args: BigtableArgs,
276    consistent_reader_args: ConsistentReaderArgs,
277    args: RpcArgs,
278    system_package_task_args: SystemPackageTaskArgs,
279    version: &'static str,
280    config: RpcConfig,
281    pg_pipelines: Vec<String>,
282    registry: &Registry,
283    cancel: CancellationToken,
284) -> anyhow::Result<JoinHandle<()>> {
285    let rpc = RpcService::new(args, version, schema(), registry, cancel.child_token());
286    let metrics = rpc.metrics();
287
288    // Create gRPC full node client wrapper
289    let fullnode_client = FullnodeClient::new(
290        Some("graphql_fullnode"),
291        fullnode_args,
292        registry,
293        cancel.child_token(),
294    )
295    .await?;
296
297    let pg_reader = PgReader::new(
298        Some("graphql_db"),
299        database_url.clone(),
300        db_args,
301        registry,
302        cancel.child_token(),
303    )
304    .await?;
305
306    let bigtable_reader = if let Some(instance_id) = bigtable_instance {
307        let reader = BigtableReader::new(
308            instance_id,
309            "indexer-alt-graphql".to_owned(),
310            bigtable_args,
311            registry,
312        )
313        .await?;
314
315        Some(reader)
316    } else {
317        None
318    };
319
320    let consistent_reader = ConsistentReader::new(
321        Some("graphql_consistent"),
322        consistent_reader_args,
323        registry,
324        cancel.child_token(),
325    )
326    .await?;
327
328    let pg_loader = Arc::new(pg_reader.as_data_loader());
329    let kv_loader = if let Some(reader) = bigtable_reader.as_ref() {
330        KvLoader::new_with_bigtable(Arc::new(reader.as_data_loader()))
331    } else {
332        KvLoader::new_with_pg(pg_loader.clone())
333    };
334
335    let package_store = Arc::new(PackageCache::new(DbPackageStore::new(pg_loader.clone())));
336
337    let system_package_task = SystemPackageTask::new(
338        system_package_task_args,
339        pg_reader.clone(),
340        package_store.clone(),
341        cancel.child_token(),
342    );
343
344    // Fetch and cache the chain identifier from the database.
345    let chain_identifier = chain_identifier::task(
346        &pg_reader,
347        config.watermark.watermark_polling_interval,
348        cancel.child_token(),
349    )
350    .await?;
351
352    let watermark_task = WatermarkTask::new(
353        config.watermark,
354        pg_pipelines,
355        pg_reader.clone(),
356        bigtable_reader,
357        consistent_reader.clone(),
358        metrics.clone(),
359        cancel.child_token(),
360    );
361
362    let rpc = rpc
363        .route("/graphql", post(graphql))
364        .route("/graphql/health", get(health::check))
365        .layer(watermark_task.watermarks())
366        .layer(config.health)
367        .layer(DbProbe(database_url))
368        .extension(Timeout::new(config.limits.timeouts()))
369        .extension(QueryLimitsChecker::new(
370            config.limits.query_limits(),
371            metrics,
372        ))
373        .data(config.limits.pagination())
374        .data(config.limits)
375        .data(config.name_service)
376        .data(config.zklogin)
377        .data(chain_identifier)
378        .data(pg_reader)
379        .data(consistent_reader)
380        .data(pg_loader)
381        .data(kv_loader)
382        .data(package_store)
383        .data(fullnode_client);
384
385    let h_rpc = rpc.run().await?;
386    let h_system_package_task = system_package_task.run();
387    let h_watermark = watermark_task.run();
388
389    Ok(tokio::spawn(async move {
390        let _ = h_rpc.await;
391        cancel.cancel();
392        let _ = h_system_package_task.await;
393        let _ = h_watermark.await;
394    }))
395}

397/// Handler for RPC requests (POST requests making GraphQL queries).
398async fn graphql(
399    ConnectInfo(addr): ConnectInfo<SocketAddr>,
400    Extension(schema): Extension<Schema<Query, Mutation, EmptySubscription>>,
401    Extension(watermark): Extension<WatermarksLock>,
402    TypedHeader(content_length): TypedHeader<ContentLength>,
403    show_usage: Option<TypedHeader<ShowUsage>>,
404    request: GraphQLRequest,
405) -> GraphQLResponse {
406    let mut request = request
407        .into_inner()
408        .data(content_length)
409        .data(Session::new(addr))
410        .data(watermark.read().await.clone());
411
412    if let Some(TypedHeader(show_usage)) = show_usage {
413        request = request.data(show_usage);
414    }
415
416    schema.execute(request).await.into()
417}

419/// Handler for GET requests for the online IDE. GraphQL requests are forwarded to the POST handler
420/// at the same path.
421async fn graphiql(path: MatchedPath) -> Html<String> {
422    Html(GraphiQLSource::build().endpoint(path.as_str()).finish())
423}
