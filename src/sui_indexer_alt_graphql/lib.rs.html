89    #[clap(long, default_value_t = Self::default().rpc_listen_address)]

97/// This type is responsible for the set-up and lifecycle of all services related to a GraphQL RPC
98/// service (the RPC service, online IDE, health checks etc). It is agnostic to the schema
99/// being served (which must be provided to `run`).
100pub struct RpcService<Q, M, S> {
101    /// Address to accept incoming RPC connections on.
102    rpc_listen_address: SocketAddr,
103
104    /// Whether to expose GraphiQL IDE on GET requests
105    with_ide: bool,
106
107    /// The version string to report with each response, as an HTTP header.
108    version: &'static str,
109
110    /// The axum router that will handle incoming requests.
111    router: Router,
112
113    /// The GraphQL schema this service will serve.
114    schema: SchemaBuilder<Q, M, S>,
115
116    /// Metrics for the RPC service.
117    metrics: Arc<RpcMetrics>,
118}
119
120impl<Q, M, S> RpcService<Q, M, S>
121where
122    Q: ObjectType + 'static,
123    M: ObjectType + 'static,
124    S: SubscriptionType + 'static,
125{
126    pub fn new(
127        args: RpcArgs,
128        version: &'static str,
129        schema: SchemaBuilder<Q, M, S>,
130        registry: &Registry,
131    ) -> Self {
132        let RpcArgs {
133            rpc_listen_address,
134            no_ide,
135        } = args;
136
137        let metrics = RpcMetrics::new(registry);
138        let router = Router::new();
139
140        // The logging extension should be outermost so that it can surround all other extensions.
141        let schema = schema
142            .extension(Logging(metrics.clone()))
143            .extension(Tracing);
144
145        Self {
146            rpc_listen_address,
147            with_ide: !no_ide,
148            version,
149            router,
150            schema,
151            metrics,
152        }
153    }
154
155    /// Return a copy of the metrics.
156    pub fn metrics(&self) -> Arc<RpcMetrics> {
157        self.metrics.clone()
158    }
159
160    /// Add a handler to the axum router.
161    pub fn route(self, path: &str, router: MethodRouter) -> Self {
162        let router = self.router.route(path, router);
163        Self { router, ..self }
164    }
165
166    /// Add an extension as a layer to the axum router.
167    pub fn layer<T>(self, layer: T) -> Self
168    where
169        T: Clone + Send + Sync + 'static,
170    {
171        let router = self.router.layer(Extension(layer));
172        Self { router, ..self }
173    }
174
175    /// Add an extension to the GraphQL schema.
176    pub fn extension(self, ext: impl ExtensionFactory) -> Self {
177        let schema = self.schema.extension(ext);
178        Self { schema, ..self }
179    }
180
181    /// Add data to the GraphQL schema that can be accessed from the Context from any request.
182    pub fn data<D: Any + Send + Sync>(self, data: D) -> Self {
183        let schema = self.schema.data(data);
184        Self { schema, ..self }
185    }
186
187    /// Run the RPC service. This binds the listener and exposes handlers for the RPC service and IDE
188    /// (if enabled).
189    pub async fn run(self) -> anyhow::Result<Service>
190    where
191        Q: ObjectType + 'static,
192        M: ObjectType + 'static,
193        S: SubscriptionType + 'static,
194    {
195        let Self {
196            rpc_listen_address,
197            with_ide,
198            version,
199            mut router,
200            schema,
201            metrics,
202        } = self;
203
204        if with_ide {
205            info!("Starting GraphiQL IDE at 'http://{rpc_listen_address}/graphql'");
206            router = router.route("/graphql", get(graphiql));
207        } else {
208            info!("Skipping GraphiQL IDE setup");
209        }
210
211        router = router
212            .layer(Extension(schema.finish()))
213            .layer(axum::middleware::from_fn_with_state(
214                Version(version),
215                middleware::version::set_version,
216            ))
217            .layer(
218                cors::CorsLayer::new()
219                    .allow_methods([Method::POST])
220                    .allow_origin(cors::Any)
221                    .allow_headers(cors::Any),
222            )
223            .layer(catch_panic::CatchPanicLayer::custom(PanicHandler::new(
224                metrics,
225            )));
226
227        info!("Starting GraphQL service on {rpc_listen_address}");
228        let listener = TcpListener::bind(rpc_listen_address)
229            .await
230            .context("Failed to bind GraphQL to listen address")?;
231
232        let (stx, srx) = oneshot::channel::<()>();
233        Ok(Service::new()
234            .with_shutdown_signal(async move {
235                let _ = stx.send(());
236            })
237            .spawn(async move {
238                axum::serve(
239                    listener,
240                    router.into_make_service_with_connect_info::<SocketAddr>(),
241                )
242                .with_graceful_shutdown(async move {
243                    let _ = srx.await;
244                    info!("Shutdown received, shutting down GraphQL service");
245                })
246                .await
247                .context("Failed to start GraphQL service")
248            }))
249    }
250}

261/// The GraphQL schema this service will serve, without any extensions or context added.
262pub fn schema() -> SchemaBuilder<Query, Mutation, EmptySubscription> {
263    Schema::build(Query::default(), Mutation, EmptySubscription)
264        .register_output_type::<IAddressable>()
265        .register_output_type::<IMoveDatatype>()
266        .register_output_type::<IMoveObject>()
267        .register_output_type::<IObject>()
268}

270/// Set-up and run the RPC service, using the provided arguments (expected to be extracted from the
271/// command-line).
272///
273/// Access to most reads is controlled by the `database_url` -- if it is `None`, those reads will
274/// not work. KV queries can optionally be served by a Bigtable instance or Ledger gRPC service
275/// via `kv_args`. If a Bigtable instance is configured, the `GOOGLE_APPLICATION_CREDENTIALS`
276/// environment variable must point to the credentials JSON file.
277///
278/// `version` is the version string reported in response headers by the service as part of every
279/// request.
280///
281/// The service may spin up auxiliary services (such as the system package task) to support itself,
282/// and will clean these up on shutdown as well.
283pub async fn start_rpc(
284    database_url: Option<Url>,
285    fullnode_args: FullnodeArgs,
286    db_args: DbArgs,
287    kv_args: args::KvArgs,
288    consistent_reader_args: ConsistentReaderArgs,
289    args: RpcArgs,
290    system_package_task_args: SystemPackageTaskArgs,
291    version: &'static str,
292    config: RpcConfig,
293    pg_pipelines: Vec<String>,
294    registry: &Registry,
295) -> anyhow::Result<Service> {
296    let rpc = RpcService::new(args, version, schema(), registry);
297    let metrics = rpc.metrics();
298
299    // Create gRPC full node client wrapper
300    let fullnode_client =
301        FullnodeClient::new(Some("graphql_fullnode"), fullnode_args, registry).await?;
302
303    let pg_reader =
304        PgReader::new(Some("graphql_db"), database_url.clone(), db_args, registry).await?;
305
306    let bigtable_reader = if let Some(instance_id) = kv_args.bigtable_instance.as_ref() {
307        let reader = BigtableReader::new(
308            instance_id.clone(),
309            "indexer-alt-graphql".to_owned(),
310            kv_args.bigtable_args(),
311            registry,
312        )
313        .await?;
314
315        Some(reader)
316    } else {
317        None
318    };
319
320    let ledger_grpc_reader = if let Some(ledger_grpc_url) = kv_args.ledger_grpc_url.as_ref() {
321        let reader = LedgerGrpcReader::new(
322            ledger_grpc_url.clone(),
323            kv_args.ledger_grpc_args(),
324            Some("graphql_ledger_grpc"),
325            registry,
326        )
327        .await
328        .context("Failed to create Ledger gRPC reader")?;
329        Some(reader)
330    } else {
331        None
332    };
333
334    let consistent_reader =
335        ConsistentReader::new(Some("graphql_consistent"), consistent_reader_args, registry).await?;
336
337    let pg_loader = Arc::new(pg_reader.as_data_loader());
338    let kv_loader = if let Some(reader) = bigtable_reader.as_ref() {
339        KvLoader::new_with_bigtable(Arc::new(reader.as_data_loader()))
340    } else if let Some(reader) = ledger_grpc_reader.as_ref() {
341        KvLoader::new_with_ledger_grpc(Arc::new(reader.as_data_loader()))
342    } else {
343        KvLoader::new_with_pg(pg_loader.clone())
344    };
345
346    let package_store = Arc::new(PackageCache::new(DbPackageStore::new(pg_loader.clone())));
347
348    let system_package_task = SystemPackageTask::new(
349        system_package_task_args,
350        pg_reader.clone(),
351        package_store.clone(),
352    );
353
354    // Fetch and cache the chain identifier from the database.
355    let (chain_identifier, s_chain_id) = chain_identifier::task(
356        pg_reader.clone(),
357        config.watermark.watermark_polling_interval,
358    );
359
360    let watermark_task = WatermarkTask::new(
361        config.watermark,
362        pg_pipelines,
363        pg_reader.clone(),
364        bigtable_reader,
365        ledger_grpc_reader,
366        consistent_reader.clone(),
367        metrics.clone(),
368    );
369
370    let rpc = rpc
371        .route("/graphql", post(graphql))
372        .route("/graphql/health", get(health::check))
373        .layer(watermark_task.watermarks())
374        .layer(config.health)
375        .layer(DbProbe(database_url))
376        .extension(Timeout::new(config.limits.timeouts()))
377        .extension(QueryLimitsChecker::new(
378            config.limits.query_limits(),
379            metrics,
380        ))
381        .data(config.limits.pagination())
382        .data(config.limits)
383        .data(config.name_service)
384        .data(config.zklogin)
385        .data(chain_identifier)
386        .data(pg_reader)
387        .data(consistent_reader)
388        .data(pg_loader)
389        .data(kv_loader)
390        .data(package_store)
391        .data(fullnode_client);
392
393    let s_rpc = rpc.run().await?;
394    let s_system_package_task = system_package_task.run();
395    let s_watermark = watermark_task.run();
396
397    Ok(s_rpc
398        .attach(s_chain_id)
399        .attach(s_system_package_task)
400        .attach(s_watermark))
401}

403/// Handler for RPC requests (POST requests making GraphQL queries).
404async fn graphql(
405    ConnectInfo(addr): ConnectInfo<SocketAddr>,
406    Extension(schema): Extension<Schema<Query, Mutation, EmptySubscription>>,
407    Extension(watermark): Extension<WatermarksLock>,
408    TypedHeader(content_length): TypedHeader<ContentLength>,
409    show_usage: Option<TypedHeader<ShowUsage>>,
410    request: GraphQLRequest,
411) -> GraphQLResponse {
412    let mut request = request
413        .into_inner()
414        .data(content_length)
415        .data(Session::new(addr))
416        .data(watermark.read().await.clone())
417        .data(rich::Meter::default());
418
419    if let Some(TypedHeader(show_usage)) = show_usage {
420        request = request.data(show_usage);
421    }
422
423    schema.execute(request).await.into()
424}

426/// Handler for GET requests for the online IDE. GraphQL requests are forwarded to the POST handler
427/// at the same path.
428async fn graphiql(path: MatchedPath) -> Html<String> {
429    Html(GraphiQLSource::build().endpoint(path.as_str()).finish())
430}

477    #[tokio::test]
478    async fn test_panic_handling() {
479        struct Query;
480
481        #[Object]
482        impl Query {
483            async fn panic(&self) -> bool {
484                assert_eq!(1, 2, "Boom!");
485                true
486            }
487        }
488
489        async fn graphql(
490            Extension(schema): Extension<Schema<Query, EmptyMutation, EmptySubscription>>,
491            request: GraphQLRequest,
492        ) -> GraphQLResponse {
493            let request = request
494                .into_inner()
495                .data(Session::new("0.0.0.0:0".parse().unwrap()));
496            schema.execute(request).await.into()
497        }
498
499        let registry = Registry::new();
500        let rpc_listen_address =
501            SocketAddr::new(IpAddr::V4(Ipv4Addr::LOCALHOST), get_available_port());
502
503        let rpc = RpcService::new(
504            RpcArgs {
505                rpc_listen_address,
506                no_ide: true,
507            },
508            "test",
509            Schema::build(Query, EmptyMutation, EmptySubscription),
510            &registry,
511        )
512        .route("/graphql", post(graphql));
513
514        let metrics = rpc.metrics();
515        let _svc = rpc.run().await.unwrap();
516
517        let url = format!("http://{rpc_listen_address}/graphql");
518        let client = Client::new();
519
520        let resp = client
521            .post(&url)
522            .json(&json!({
523                "query": "{ panic }"
524            }))
525            .send()
526            .await
527            .expect("Request should succeed");
528
529        assert_eq!(resp.status(), 500);
530
531        let body: Value = resp.json().await.expect("Response should be JSON");
532
533        // Verify the response is a GraphQL error
534        let error = &body["errors"].as_array().unwrap()[0];
535
536        assert!(
537            error["message"]
538                .as_str()
539                .unwrap()
540                .contains("Request panicked")
541        );
542
543        assert_eq!(
544            error["extensions"]["code"].as_str(),
545            Some(code::INTERNAL_SERVER_ERROR)
546        );
547
548        // The panic message is in the chain
549        let chain = error["extensions"]["chain"].as_array().unwrap();
550        assert!(chain.iter().any(|c| c.as_str().unwrap().contains("Boom!")));
551
552        // Verify the panic is recorded in metrics
553        assert_eq!(metrics.queries_panicked.get(), 1);
554    }
