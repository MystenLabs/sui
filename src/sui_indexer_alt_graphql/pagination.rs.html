16/// Configuration for page size limits, specifying a max multi-get size, as well as a default and
17/// max page size for each paginated fields. Page limits can be customized for specific fields,
18/// otherwise falling back to a blanket default.
19pub(crate) struct PaginationConfig {
20    /// Maximum number of keys that can be fetched in a single multi-get.
21    max_multi_get_size: u32,
22
23    /// Fallback page limit configuration.
24    fallback: PageLimits,
25
26    /// Type and field name-specific overrides for page limits.
27    overrides: BTreeMap<(&'static str, &'static str), PageLimits>,
28}
29
30/// The configuration for a single paginated field.
31pub(crate) struct PageLimits {
32    pub(crate) default: u32,
33    pub(crate) max: u32,
34}
35
36#[derive(Debug, Eq, PartialEq)]
37pub(crate) struct Page<C> {
38    /// The exclusive lower bound of the page (no bound means start from the beginning of the
39    /// data-set).
40    after: Option<C>,
41
42    /// The exclusive upper bound of the page (no bound means continue to the end of the data-set).
43    before: Option<C>,
44
45    /// Maximum number of entries in the page.
46    limit: u64,
47
48    /// In case there are more than `limit` entries in the range described by `(after, before)`,
49    /// this field states whether the entries up to limit are taken from the `Front` or `Back` of
50    /// that range.
51    end: End,
52}
53
54/// Whether the page is extracted from the beginning or the end of the range bounded by the cursors.
55#[derive(PartialEq, Eq, Debug, Clone, Copy)]
56pub(crate) enum End {
57    Front,
58    Back,
59}
60
61#[derive(thiserror::Error, Debug, Clone)]
62pub(crate) enum Error {
63    #[error("Cannot provide both 'first' and 'last' parameters for connection")]
64    FirstAndLast,
65
66    #[error("Page size of {limit} exceeds max of {max} for connection")]
67    TooLarge { limit: u64, max: u32 },
68}
69
70impl PaginationConfig {
71    pub(crate) fn new(
72        max_multi_get_size: u32,
73        fallback: PageLimits,
74        overrides: BTreeMap<(&'static str, &'static str), PageLimits>,
75    ) -> Self {
76        Self {
77            max_multi_get_size,
78            fallback,
79            overrides,
80        }
81    }
82
83    /// Maximum number of keys that can be fetched in a single multi-get.
84    pub(crate) fn max_multi_get_size(&self) -> u32 {
85        self.max_multi_get_size
86    }
87
88    /// Fetch the default and max page size for this type and field.
89    pub(crate) fn limits<'l, 't: 'l>(&'l self, type_: &'t str, name: &'t str) -> &'l PageLimits {
90        self.overrides.get(&(type_, name)).unwrap_or(&self.fallback)
91    }
92}

94impl PageLimits {
95    /// Limits for fetching a single item.
96    pub(crate) fn singleton() -> Self {
97        Self { default: 1, max: 1 }
98    }

101impl<C> Page<C> {
102    /// Convert connection parameters into a page. Entries for the page are drawn from the range
103    /// `(after, before)` (Both bounds are optional). The number of entries in the page is
104    /// controlled by `first` and `last`.
105    ///
106    /// - Setting both is in an error.
107    /// - Setting `first` indicates that the entries are taken from the front of the range.
108    /// - Setting `last` indicates that the entries are taken from the end of the range.
109    /// - Setting neither defaults the limit to the default page size in `limits`, taken from the
110    ///   front of the range.
111    ///
112    /// It is an error to set a limit on page size that is greater than the `limit`'s max page
113    /// size.
114    pub(crate) fn from_params(
115        limits: &PageLimits,
116        first: Option<u64>,
117        after: Option<C>,
118        last: Option<u64>,
119        before: Option<C>,
120    ) -> Result<Self, Error> {
121        let page = match (first, after, last, before) {
122            (Some(_), _, Some(_), _) => return Err(Error::FirstAndLast),
123
124            (limit, after, None, before) => Page {
125                after,
126                before,
127                limit: limit.unwrap_or(limits.default as u64),
128                end: End::Front,
129            },
130
131            (None, after, Some(limit), before) => Page {
132                after,
133                before,
134                limit,
135                end: End::Back,
136            },
137        };
138
139        if page.limit > limits.max as u64 {
140            return Err(Error::TooLarge {
141                limit: page.limit,
142                max: limits.max,
143            });
144        }
145
146        Ok(page)
147    }
148
149    pub(crate) fn after(&self) -> Option<&C> {
150        self.after.as_ref()
151    }
152
153    pub(crate) fn before(&self) -> Option<&C> {
154        self.before.as_ref()
155    }
156
157    pub(crate) fn limit(&self) -> usize {
158        self.limit as usize
159    }
160
161    /// Returns the limit + 2 for has_previous_page and has_next_page pagination calculations
162    pub(crate) fn limit_with_overhead(&self) -> usize {
163        self.limit() + 2
164    }
165
166    pub(crate) fn is_from_front(&self) -> bool {
167        matches!(self.end, End::Front)
168    }
169
170    /// Direction for sorting SQL queries.
171    pub(crate) fn order_by_direction(&self) -> Query<'_> {
172        if self.is_from_front() {
173            query!("ASC")
174        } else {
175            query!("DESC")
176        }
177    }
178}

180impl Page<JsonCursor<usize>> {
181    /// Treat the cursors of this Page as indices into a range [0, total).
182    ///
183    /// Returns a connection where the cursors correspond to a sub-range of indices and the
184    /// nodes are selected by calling `node` with each index in that sub-range.
185    pub(crate) fn paginate_indices<N: OutputType, E>(
186        &self,
187        total: usize,
188        node: impl Fn(usize) -> Result<N, E>,
189    ) -> Result<Connection<String, N>, E> {
190        let mut lo = self.after().map_or(0, |a| a.saturating_add(1)).min(total);
191        let mut hi = self.before().map_or(total, |b| **b).min(total);
192        let mut conn = Connection::new(false, false);
193
194        if hi <= lo {
195            return Ok(conn);
196        } else if (hi - lo) > self.limit() {
197            if self.is_from_front() {
198                hi = lo + self.limit();
199            } else {
200                lo = hi - self.limit();
201            }
202        }
203
204        conn.has_previous_page = 0 < lo;
205        conn.has_next_page = hi < total;
206        for i in lo..hi {
207            conn.edges
208                .push(Edge::new(JsonCursor::new(i).encode_cursor(), node(i)?));
209        }
210
211        Ok(conn)
212    }
213}

215impl<C: CursorType + Eq + PartialEq + Clone> Page<C> {
216    /// Process the results of a paginated query.
217    ///
218    /// `results` is expected to be the result of a query modified to fetch the page indicated by
219    /// `self`. This function determines whether those results are consistent with the cursors for
220    /// this page, and whether there are more results in the set being paginated, before or after
221    /// the current page.
222    ///
223    /// `results` should contain the elements of the page, in order, as well as at least one record
224    /// either side of the page, if there is one.
225    ///
226    /// `cursor` is a function that extracts the cursor from an element of `results`, and `node` is
227    /// a function that extracts the node. Returns a GraphQL `Connection` populated with edges
228    /// derived from `results`.
229    pub(crate) fn paginate_results<T, N: OutputType, E>(
230        &self,
231        results: Vec<T>,
232        cursor: impl Fn(&T) -> C,
233        node: impl Fn(T) -> Result<N, E>,
234    ) -> Result<Connection<String, N>, E> {
235        let edges: Vec<_> = results.into_iter().map(|r| (cursor(&r), r)).collect();
236        let first = edges.first().map(|(c, _)| c.clone());
237        let last = edges.last().map(|(c, _)| c.clone());
238
239        let (prev, next, prefix, suffix) =
240            match (self.after(), first, last, self.before(), self.end) {
241                // Results came back empty, despite supposedly including the `after` and `before`
242                // cursors, so the bounds must have been invalid, no matter which end the page was
243                // drawn from.
244                (_, None, _, _, _) | (_, _, None, _, _) => {
245                    return Ok(Connection::new(false, false));
246                }
247
248                // Page drawn from the front, and the cursor for the first element does not match
249                // `after`. This absence implies the bound was invalid, so we return an empty
250                // result.
251                (Some(a), Some(f), _, _, End::Front) if f != *a => {
252                    return Ok(Connection::new(false, false));
253                }
254
255                // Similar to above case, but for back of results.
256                (_, _, Some(l), Some(b), End::Back) if l != *b => {
257                    return Ok(Connection::new(false, false));
258                }
259
260                // From here onwards, we know that the results are non-empty. In the forward
261                // pagination scenario, the presence of a previous page is determined by whether a
262                // cursor supplied on the end the page is being drawn from is found in the first
263                // position. The presence of a next page is determined by whether we have more
264                // results than the provided limit, and/ or if the end cursor element appears in
265                // the result set.
266                (after, Some(f), Some(l), before, End::Front) => {
267                    let has_previous_page = after.is_some_and(|a| *a == f);
268                    let prefix = has_previous_page as usize;
269
270                    // If results end with the before cursor, we will at least need to trim one
271                    // element from the suffix and we trim more off the end if there is more after
272                    // applying the limit.
273                    let mut suffix = before.is_some_and(|b| *b == l) as usize;
274                    suffix += edges.len().saturating_sub(self.limit() + prefix + suffix);
275                    let has_next_page = suffix > 0;
276
277                    (has_previous_page, has_next_page, prefix, suffix)
278                }
279
280                // Symmetric to the previous case, but drawing from the back.
281                (after, Some(f), Some(l), before, End::Back) => {
282                    let has_next_page = before.is_some_and(|b| *b == l);
283                    let suffix = has_next_page as usize;
284
285                    let mut prefix = after.is_some_and(|a| *a == f) as usize;
286                    prefix += edges.len().saturating_sub(self.limit() + prefix + suffix);
287                    let has_previous_page = prefix > 0;
288
289                    (has_previous_page, has_next_page, prefix, suffix)
290                }
291            };
292
293        // If there are no elements left after trimming, then forget whether there's a previous or
294        // next page, because there will be no start or end cursor for this page to anchor on.
295        if edges.len() == prefix + suffix {
296            return Ok(Connection::new(false, false));
297        }
298
299        // We finally made it -- trim the prefix and suffix edges from the result and send it!
300        let mut edges = edges.into_iter();
301        if prefix > 0 {
302            edges.nth(prefix - 1);
303        }
304        if suffix > 0 {
305            edges.nth_back(suffix - 1);
306        }
307
308        let mut conn = Connection::new(prev, next);
309        for (c, n) in edges {
310            conn.edges.push(Edge::new(c.encode_cursor(), node(n)?));
311        }
312
313        Ok(conn)
314    }
315}

317/// Decides whether the field's return type is paginated.
318pub(crate) fn is_connection(field: &MetaField) -> bool {
319    let type_ = field.ty.as_str();
320    type_.ends_with("Connection") || type_.ends_with("Connection!")
321}
