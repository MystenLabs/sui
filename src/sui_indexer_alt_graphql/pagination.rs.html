17/// Configuration for page size limits, specifying a max multi-get size, as well as a default and
18/// max page size for each paginated fields. Page limits can be customized for specific fields,
19/// otherwise falling back to a blanket default.
20pub(crate) struct PaginationConfig {
21    /// Maximum number of keys that can be fetched in a single multi-get.
22    max_multi_get_size: u32,
23
24    /// Fallback page limit configuration.
25    fallback: PageLimits,
26
27    /// Type and field name-specific overrides for page limits.
28    overrides: BTreeMap<(&'static str, &'static str), PageLimits>,
29}
30
31/// The configuration for a single paginated field.
32pub(crate) struct PageLimits {
33    pub(crate) default: u32,
34    pub(crate) max: u32,
35}
36
37#[derive(Debug, Eq, PartialEq)]
38pub(crate) struct Page<C> {
39    /// The exclusive lower bound of the page (no bound means start from the beginning of the
40    /// data-set).
41    after: Option<C>,
42
43    /// The exclusive upper bound of the page (no bound means continue to the end of the data-set).
44    before: Option<C>,
45
46    /// Maximum number of entries in the page.
47    limit: u64,
48
49    /// In case there are more than `limit` entries in the range described by `(after, before)`,
50    /// this field states whether the entries up to limit are taken from the `Front` or `Back` of
51    /// that range.
52    end: End,
53}
54
55/// Whether the page is extracted from the beginning or the end of the range bounded by the cursors.
56#[derive(PartialEq, Eq, Debug, Clone, Copy)]
57pub(crate) enum End {
58    Front,
59    Back,
60}
61
62#[derive(thiserror::Error, Debug, Clone)]
63pub(crate) enum Error {
64    #[error("Cannot provide both 'first' and 'last' parameters for connection")]
65    FirstAndLast,
66
67    #[error("Page size of {limit} exceeds max of {max} for connection")]
68    TooLarge { limit: u64, max: u32 },
69}
70
71impl PaginationConfig {
72    pub(crate) fn new(
73        max_multi_get_size: u32,
74        fallback: PageLimits,
75        overrides: BTreeMap<(&'static str, &'static str), PageLimits>,
76    ) -> Self {
77        Self {
78            max_multi_get_size,
79            fallback,
80            overrides,
81        }
82    }
83
84    /// Maximum number of keys that can be fetched in a single multi-get.
85    pub(crate) fn max_multi_get_size(&self) -> u32 {
86        self.max_multi_get_size
87    }
88
89    /// Fetch the default and max page size for this type and field.
90    pub(crate) fn limits<'l, 't: 'l>(&'l self, type_: &'t str, name: &'t str) -> &'l PageLimits {
91        self.overrides.get(&(type_, name)).unwrap_or(&self.fallback)
92    }
93}

95impl PageLimits {
96    /// Limits for fetching a single item.
97    pub(crate) fn singleton() -> Self {
98        Self { default: 1, max: 1 }
99    }
100}

102impl<C> Page<C> {
103    /// Convert connection parameters into a page. Entries for the page are drawn from the range
104    /// `(after, before)` (Both bounds are optional). The number of entries in the page is
105    /// controlled by `first` and `last`.
106    ///
107    /// - Setting both is in an error.
108    /// - Setting `first` indicates that the entries are taken from the front of the range.
109    /// - Setting `last` indicates that the entries are taken from the end of the range.
110    /// - Setting neither defaults the limit to the default page size in `limits`, taken from the
111    ///   front of the range.
112    ///
113    /// It is an error to set a limit on page size that is greater than the `limit`'s max page
114    /// size.
115    pub(crate) fn from_params(
116        limits: &PageLimits,
117        first: Option<u64>,
118        after: Option<C>,
119        last: Option<u64>,
120        before: Option<C>,
121    ) -> Result<Self, Error> {
122        let page = match (first, after, last, before) {
123            (Some(_), _, Some(_), _) => return Err(Error::FirstAndLast),
124
125            (limit, after, None, before) => Page {
126                after,
127                before,
128                limit: limit.unwrap_or(limits.default as u64),
129                end: End::Front,
130            },
131
132            (None, after, Some(limit), before) => Page {
133                after,
134                before,
135                limit,
136                end: End::Back,
137            },
138        };
139
140        if page.limit > limits.max as u64 {
141            return Err(Error::TooLarge {
142                limit: page.limit,
143                max: limits.max,
144            });
145        }
146
147        Ok(page)
148    }
149
150    pub(crate) fn after(&self) -> Option<&C> {
151        self.after.as_ref()
152    }
153
154    pub(crate) fn before(&self) -> Option<&C> {
155        self.before.as_ref()
156    }
157
158    pub(crate) fn limit(&self) -> usize {
159        self.limit as usize
160    }
161
162    /// Returns the limit + 2 for has_previous_page and has_next_page pagination calculations
163    pub(crate) fn limit_with_overhead(&self) -> usize {
164        self.limit() + 2
165    }
166
167    pub(crate) fn is_from_front(&self) -> bool {
168        matches!(self.end, End::Front)
169    }
170
171    /// Direction for sorting SQL queries.
172    pub(crate) fn order_by_direction(&self) -> Query<'_> {
173        if self.is_from_front() {
174            query!("ASC")
175        } else {
176            query!("DESC")
177        }
178    }
179}

181impl Page<JsonCursor<usize>> {
182    /// Translate the cursors of this Page into a range of indices, in [0, total).
183    ///
184    /// Returns `None` if the cursors are inconsistent, otherwise returns Some(lo..hi) where `lo`
185    /// and `hi` are the inclusive lower and exclusive upper bounds of the range of indices
186    /// corresponding to the cursors for this page.
187    pub(crate) fn range(&self, total: usize) -> Option<Range<usize>> {
188        let mut lo = self.after().map_or(0, |a| a.saturating_add(1)).min(total);
189        let mut hi = self.before().map_or(total, |b| **b).min(total);
190
191        if hi <= lo {
192            return None;
193        }
194
195        if (hi - lo) > self.limit() {
196            if self.is_from_front() {
197                hi = lo + self.limit();
198            } else {
199                lo = hi - self.limit();
200            }
201        }
202
203        Some(lo..hi)
204    }
205
206    /// Treat the cursors of this Page as indices into a range [0, total).
207    ///
208    /// Returns a connection where the cursors correspond to a sub-range of indices and the
209    /// nodes are selected by calling `node` with each index in that sub-range.
210    pub(crate) fn paginate_indices<N: OutputType, E>(
211        &self,
212        total: usize,
213        node: impl Fn(usize) -> Result<N, E>,
214    ) -> Result<Connection<String, N>, E> {
215        let mut conn = Connection::new(false, false);
216        let Some(range) = self.range(total) else {
217            return Ok(conn);
218        };
219
220        conn.has_previous_page = 0 < range.start;
221        conn.has_next_page = range.end < total;
222        for i in range {
223            conn.edges
224                .push(Edge::new(JsonCursor::new(i).encode_cursor(), node(i)?));
225        }
226
227        Ok(conn)
228    }
229}

231impl<C: CursorType + Eq + PartialEq + Clone> Page<C> {
232    /// Process the results of a paginated query.
233    ///
234    /// `results` is expected to be the result of a query modified to fetch the page indicated by
235    /// `self`. This function determines whether those results are consistent with the cursors for
236    /// this page, and whether there are more results in the set being paginated, before or after
237    /// the current page.
238    ///
239    /// `results` should contain the elements of the page, in order, as well as at least one record
240    /// either side of the page, if there is one.
241    ///
242    /// `cursor` is a function that extracts the cursor from an element of `results`, and `node` is
243    /// a function that extracts the node. Returns a GraphQL `Connection` populated with edges
244    /// derived from `results`.
245    pub(crate) fn paginate_results<T, N: OutputType, E>(
246        &self,
247        results: Vec<T>,
248        cursor: impl Fn(&T) -> C,
249        node: impl Fn(T) -> Result<N, E>,
250    ) -> Result<Connection<String, N>, E> {
251        let edges: Vec<_> = results.into_iter().map(|r| (cursor(&r), r)).collect();
252        let first = edges.first().map(|(c, _)| c.clone());
253        let last = edges.last().map(|(c, _)| c.clone());
254
255        let (prev, next, prefix, suffix) =
256            match (self.after(), first, last, self.before(), self.end) {
257                // Results came back empty, despite supposedly including the `after` and `before`
258                // cursors, so the bounds must have been invalid, no matter which end the page was
259                // drawn from.
260                (_, None, _, _, _) | (_, _, None, _, _) => {
261                    return Ok(Connection::new(false, false));
262                }
263
264                // Page drawn from the front, and the cursor for the first element does not match
265                // `after`. This absence implies the bound was invalid, so we return an empty
266                // result.
267                (Some(a), Some(f), _, _, End::Front) if f != *a => {
268                    return Ok(Connection::new(false, false));
269                }
270
271                // Similar to above case, but for back of results.
272                (_, _, Some(l), Some(b), End::Back) if l != *b => {
273                    return Ok(Connection::new(false, false));
274                }
275
276                // From here onwards, we know that the results are non-empty. In the forward
277                // pagination scenario, the presence of a previous page is determined by whether a
278                // cursor supplied on the end the page is being drawn from is found in the first
279                // position. The presence of a next page is determined by whether we have more
280                // results than the provided limit, and/ or if the end cursor element appears in
281                // the result set.
282                (after, Some(f), Some(l), before, End::Front) => {
283                    let has_previous_page = after.is_some_and(|a| *a == f);
284                    let prefix = has_previous_page as usize;
285
286                    // If results end with the before cursor, we will at least need to trim one
287                    // element from the suffix and we trim more off the end if there is more after
288                    // applying the limit.
289                    let mut suffix = before.is_some_and(|b| *b == l) as usize;
290                    suffix += edges.len().saturating_sub(self.limit() + prefix + suffix);
291                    let has_next_page = suffix > 0;
292
293                    (has_previous_page, has_next_page, prefix, suffix)
294                }
295
296                // Symmetric to the previous case, but drawing from the back.
297                (after, Some(f), Some(l), before, End::Back) => {
298                    let has_next_page = before.is_some_and(|b| *b == l);
299                    let suffix = has_next_page as usize;
300
301                    let mut prefix = after.is_some_and(|a| *a == f) as usize;
302                    prefix += edges.len().saturating_sub(self.limit() + prefix + suffix);
303                    let has_previous_page = prefix > 0;
304
305                    (has_previous_page, has_next_page, prefix, suffix)
306                }
307            };
308
309        // If there are no elements left after trimming, then forget whether there's a previous or
310        // next page, because there will be no start or end cursor for this page to anchor on.
311        if edges.len() == prefix + suffix {
312            return Ok(Connection::new(false, false));
313        }
314
315        // We finally made it -- trim the prefix and suffix edges from the result and send it!
316        let mut edges = edges.into_iter();
317        if prefix > 0 {
318            edges.nth(prefix - 1);
319        }
320        if suffix > 0 {
321            edges.nth_back(suffix - 1);
322        }
323
324        let mut conn = Connection::new(prev, next);
325        for (c, n) in edges {
326            conn.edges.push(Edge::new(c.encode_cursor(), node(n)?));
327        }
328
329        Ok(conn)
330    }
331}

333/// Decides whether the field's return type is paginated.
334pub(crate) fn is_connection(field: &MetaField) -> bool {
335    let type_ = field.ty.as_str();
336    type_.ends_with("Connection") || type_.ends_with("Connection!")
337}
