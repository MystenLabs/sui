39/// A map of objects from an executed transaction, keyed by (ObjectID, SequenceNumber).
40/// None values indicate tombstones for deleted/wrapped objects.
41pub(crate) type ExecutionObjectMap =
42    Arc<BTreeMap<(ObjectID, SequenceNumber), Option<NativeObject>>>;

44/// Where in-memory lookups (objects, transaction contents, etc.) in this scope draw their data
45/// from. Encodes the mutually exclusive modes a [`Scope`] can be in. Indexed mode hits the
46/// database/kv_loader (no in-memory payload). Executed mode is the mutation/simulate path, with a
47/// freshly executed transaction's outputs. Streamed mode is the subscription path, with an
48/// in-memory checkpoint payload.
49#[derive(Clone)]
50pub(crate) enum DataSource {
51    /// Reads go through the indexed-checkpoint path (kv_loader / DB). No in-memory payload.
52    Indexed,
53    /// A freshly executed transaction's input/output objects.
54    Executed {
55        execution_objects: ExecutionObjectMap,
56    },
57    /// A streamed checkpoint with all per-tx contents and a checkpoint-wide execution-objects
58    /// map. If the scope is anchored to a particular transaction in the checkpoint, its digest
59    /// is in [`Scope::effects_viewed_at_digest`].
60    #[cfg(feature = "staging")]
61    Streamed {
62        checkpoint: Arc<ProcessedCheckpoint>,
63    },
64}
65
66/// Identifies the transaction whose effects are currently in view. Descendant resolvers default
67/// fields like `IAddressable.asTransactionObject` to this transaction, and `*Contents::fetch`
68/// consults `contents` as a hit before falling back to streaming/kv_loader.
69///
70/// `contents` is `Some` when the anchoring caller already had them in hand (e.g. when navigating
71/// `Transaction.effects` on a hydrated `Transaction`). `None` for digest-only anchors (e.g.
72/// subscription resolvers, where contents live in the [`DataSource::Streamed`] checkpoint and are
73/// fetched on demand).
74#[derive(Clone)]
75struct ActiveTransaction {
76    digest: TransactionDigest,
77    contents: Option<Arc<NativeTransactionContents>>,
78}

80/// Root object bound for consistent dynamic field reads.
81///
82/// This enables consistent dynamic field reads in the case of chained dynamic object fields,
83/// e.g., `Parent -> DOF1 -> DOF2`. In such cases, the object versions may end up like
84/// `Parent >= DOF1, DOF2` but `DOF1 < DOF2`.
85///
86/// Lamport timestamps of objects are updated for all top-level mutable objects provided as
87/// inputs to a transaction as well as any mutated dynamic child objects. However, any dynamic
88/// child objects that were loaded but not actually mutated don't end up having their versions
89/// updated. So, database queries for nested dynamic fields must be bounded by the version of
90/// the root object, and not the immediate parent.
91///
92/// The bound can be expressed either in terms of a specific object version or a checkpoint.
93#[derive(Clone, Copy, Debug)]
94pub(crate) enum RootBound {
95    Version(u64),
96    Checkpoint(u64),
97}

99/// A way to share information between fields in a request, similar to [Context].
100///
101/// Unlike [Context], [Scope] is not referenced by every field resolver. Instead, fields must
102/// explicitly thread [Scope]-s to where they are needed, and are able to override them when
103/// necessary, allowing a nested scope to shadow values in its parent scope.
104#[derive(Clone)]
105pub(crate) struct Scope {
106    /// The checkpoint we are viewing this data at. Queries for the latest versions of an entity
107    /// are relative to this checkpoint.
108    ///
109    /// None indicates execution context where we're viewing fresh transaction effects not yet indexed.
110    checkpoint_viewed_at: Option<u64>,
111
112    /// The transaction whose effects descendant resolvers default to (e.g.
113    /// `IAddressable.asTransactionObject` without an explicit `transactionDigest`). Set when a
114    /// resolver brings a single transaction into view: indexed paths that already hold the
115    /// `TransactionContents` anchor with contents (so `*Contents::fetch` can short-circuit via
116    /// `active_transaction_contents_for`); subscription resolvers anchor digest-only and rely on
117    /// the [`DataSource::Streamed`] checkpoint for content lookups.
118    ///
119    /// This is *not* a "view bound" up to and including a transaction; it identifies one
120    /// transaction. Object visibility is end-of-checkpoint regardless of this field.
121    active_transaction: Option<ActiveTransaction>,
122
123    /// Root object bound for dynamic fields.
124    ///
125    /// This can be expressed either in terms of a specific object version or a checkpoint.
126    root_bound: Option<RootBound>,
127
128    /// Where in-memory lookups in this scope draw their data from. See [`DataSource`].
129    data_source: DataSource,
130
131    /// Access to packages for type resolution.
132    package_store: Arc<dyn PackageStore>,
133
134    /// Limits for package/type resolution.
135    resolver_limits: sui_package_resolver::Limits,
136}

141    pub(crate) fn new<E: std::error::Error>(ctx: &Context<'_>) -> Result<Self, RpcError<E>> {
142        let watermark: &Arc<Watermarks> = ctx.data()?;
143        let package_store: &Arc<PackageCache> = ctx.data()?;

148            active_transaction: None,

151            package_store: package_store.clone(),
152            resolver_limits: limits.package_resolver(),
153        })
154    }

176    /// Anchor a nested scope to a transaction by digest only. Used when the caller does not yet
177    /// hold the transaction's contents. If the scope is already anchored to the same digest
178    /// (with or without hydrated contents), the existing anchor is preserved so descendants
179    /// keep access to any cached contents. Does not change object visibility.
180    pub(crate) fn with_active_transaction_digest(&self, digest: TransactionDigest) -> Self {
181        if self.active_transaction_digest() == Some(digest) {
182            return self.clone();
183        }
184        self.with_active_transaction(ActiveTransaction {
185            digest,
186            contents: None,
187        })
188    }

190    /// Anchor a nested scope to a transaction whose contents the caller already holds. Lets

193    /// object visibility.
194    pub(crate) fn with_active_transaction_contents(
195        &self,
196        digest: TransactionDigest,
197        contents: Arc<NativeTransactionContents>,

200            digest,
201            contents: Some(contents),

242    /// Create a nested scope pinned to a checkpoint. Returns `None` if the checkpoint is in
243    /// the future, or if the current scope is in execution context (no checkpoint is set).
244    pub(crate) fn with_checkpoint_viewed_at(
245        &self,
246        ctx: &Context<'_>,
247        checkpoint_viewed_at: u64,
248    ) -> Option<Self> {
249        let watermark: &Arc<Watermarks> = ctx.data().ok()?;
250        let cp_hi_inclusive = watermark.high_watermark().checkpoint();
251        (checkpoint_viewed_at <= cp_hi_inclusive).then(|| Self {
252            checkpoint_viewed_at: Some(checkpoint_viewed_at),
253            active_transaction: self.active_transaction.clone(),
254            root_bound: self.root_bound,
255            data_source: self.data_source.clone(),
256            package_store: self.package_store.clone(),
257            resolver_limits: self.resolver_limits.clone(),
258        })
259    }

319        if let Some(RootBound::Checkpoint(cp)) = self.root_bound {
320            Some(cp)

333    /// The digest of the transaction this scope is anchored to, if any.
334    pub(crate) fn active_transaction_digest(&self) -> Option<TransactionDigest> {
335        self.active_transaction.as_ref().map(|a| a.digest)

338    /// Already-fetched contents for `digest`, if the scope was anchored to that same transaction
339    /// with hydrated contents. Returns `None` for any other digest, or when the anchor is
340    /// digest-only; callers should fall through to the normal fetch path.
341    pub(crate) fn active_transaction_contents_for(
342        &self,
343        digest: TransactionDigest,
344    ) -> Option<&Arc<NativeTransactionContents>> {
345        self.active_transaction
346            .as_ref()
347            .filter(|a| a.digest == digest)
348            .and_then(|a| a.contents.as_ref())
349    }

351    /// The execution objects map active for object lookups in this scope. Resolves through the
352    /// scope's [`DataSource`]: in `Streamed` mode, returns the checkpoint-wide map (object
353    /// visibility is end-of-checkpoint, matching the indexed Query path); in `Executed` mode,
354    /// returns the freshly extracted map; in `Indexed` mode, returns `None` so callers fall
355    /// through to the DB.
356    fn execution_objects_in_view(&self) -> Option<&ExecutionObjectMap> {
357        match &self.data_source {
358            DataSource::Indexed => None,
359            DataSource::Executed { execution_objects } => Some(execution_objects),
360            #[cfg(feature = "staging")]
361            DataSource::Streamed { checkpoint } => Some(&checkpoint.execution_objects),
362        }
363    }

365    /// Get an object from the execution context cache, if available.
366    ///
367    /// Returns None if the object doesn't exist in the cache or if it was deleted/wrapped.
368    pub(crate) fn execution_output_object(
369        &self,
370        object_id: ObjectID,
371        version: SequenceNumber,
372    ) -> Option<&NativeObject> {
373        self.execution_objects_in_view()?
374            .get(&(object_id, version))
375            .and_then(|opt| opt.as_ref())
376    }

378    /// Get the latest version of an object from the execution context cache, if available.
379    /// Returns None if the object doesn't exist in the cache or if it was deleted/wrapped.
380    pub(crate) fn execution_output_object_latest(
381        &self,
382        object_id: ObjectID,
383    ) -> Option<&NativeObject> {
384        self.execution_objects_in_view()?
385            .range(..=(object_id, SequenceNumber::MAX))
386            .last()
387            .and_then(|(_, opt)| opt.as_ref())
388    }

413/// Extract object contents from an ExecutedTransaction, including tombstones for deleted/wrapped objects.
414///
415/// Returns a BTreeMap mapping (ObjectID, SequenceNumber) to Option<NativeObject>,
416/// where None indicates the object was deleted or wrapped at that version.
417fn extract_objects_from_executed_transaction(
418    executed_transaction: &grpc::ExecutedTransaction,
419) -> Result<ExecutionObjectMap, RpcError> {
420    use anyhow::Context;
421
422    let mut map = BTreeMap::new();
423
424    // Extract all objects from the ObjectSet.
425    if let Some(object_set) = &executed_transaction.objects {
426        for obj in &object_set.objects {
427            if let Some(bcs) = &obj.bcs {
428                let native_obj: NativeObject = bcs
429                    .deserialize()
430                    .context("Failed to deserialize object BCS")?;
431                map.insert((native_obj.id(), native_obj.version()), Some(native_obj));
432            }
433        }
434    }
435
436    // Add tombstones for objects that no longer exist from effects
437    if let Some(effects) = &executed_transaction.effects {
438        // Get lamport version directly from gRPC effects
439        let lamport_version = SequenceNumber::from_u64(
440            effects
441                .lamport_version
442                .context("Effects should have lamport_version")?,
443        );
444
445        for changed_obj in &effects.changed_objects {
446            if changed_obj.output_state() == OutputObjectState::DoesNotExist {
447                let object_id = changed_obj
448                    .object_id
449                    .as_ref()
450                    .and_then(|id| id.parse().ok())
451                    .context("ChangedObject should have valid object_id")?;
452
453                // Deleted/wrapped objects don't have an output_version, so we use the transaction's
454                // lamport version as the tombstone version. This ensures execution_output_object_latest
455                // returns None for these objects.
456                map.insert((object_id, lamport_version), None);
457            }
458        }
459    }
460
461    Ok(Arc::new(map))
462}

484#[async_trait]
485impl PackageStore for Scope {
486    /// Fetches a package, first checking execution context objects if available,
487    /// then falling back to the underlying package store.
488    async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>, PackageResolverError> {
489        let object_id = ObjectID::from(id);
490
491        // First check execution context objects if we have any
492        if let Some(execution_objects) = self.execution_objects_in_view() {
493            let latest_package = execution_objects
494                .range((object_id, SequenceNumber::MIN)..=(object_id, SequenceNumber::MAX))
495                .last()
496                .and_then(|(_, opt_object)| opt_object.as_ref())
497                .and_then(|object| {
498                    // Check if this object is actually a package
499                    object.data.try_as_package()
500                });
501
502            if let Some(package) = latest_package {
503                return Package::read_from_package(package).map(Arc::new);
504            }
505        }
506
507        // Package not found in execution context, fall back to the underlying store
508        self.package_store.fetch(id).await
509    }
