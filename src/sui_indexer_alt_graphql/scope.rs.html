23/// A way to share information between fields in a request, similar to [Context].
24///
25/// Unlike [Context], [Scope] is not referenced by every field resolver. Instead, fields must
26/// explicitly thread [Scope]-s to where they are needed, and are able to override them when
27/// necessary, allowing a nested scope to shadow values in its parent scope.
28#[derive(Clone)]
29pub(crate) struct Scope {
30    /// The checkpoint we are viewing this data at. Queries for the latest versions of an entity
31    /// are relative to this checkpoint.
32    ///
33    /// None indicates execution context where we're viewing fresh transaction effects not yet indexed.
34    checkpoint_viewed_at: Option<u64>,
35
36    /// Root parent object version for dynamic fields.
37    ///
38    /// This enables consistent dynamic field reads in the case of chained dynamic object fields,
39    /// e.g., `Parent -> DOF1 -> DOF2`. In such cases, the object versions may end up like
40    /// `Parent >= DOF1, DOF2` but `DOF1 < DOF2`.
41    ///
42    /// Lamport timestamps of objects are updated for all top-level mutable objects provided as
43    /// inputs to a transaction as well as any mutated dynamic child objects. However, any dynamic
44    /// child objects that were loaded but not actually mutated don't end up having their versions
45    /// updated. So, database queries for nested dynamic fields must be bounded by the version of
46    /// the root object, and not the immediate parent.
47    root_version: Option<u64>,
48
49    /// Cache of objects available in execution context (freshly executed transaction).
50    /// Maps (ObjectID, SequenceNumber) to the actual object data.
51    /// This enables any Object GraphQL type to access fresh data without database queries.
52    execution_objects: Arc<BTreeMap<(ObjectID, SequenceNumber), NativeObject>>,
53
54    /// Access to packages for type resolution.
55    package_store: Arc<dyn PackageStore>,
56
57    /// Limits for package/type resolution.
58    resolver_limits: sui_package_resolver::Limits,
59}
60
61impl Scope {
62    /// Create a new scope at the top-level (initialized by information we have at the root of a
63    /// request).
64    pub(crate) fn new<E: std::error::Error>(ctx: &Context<'_>) -> Result<Self, RpcError<E>> {
65        let watermark: &Arc<Watermarks> = ctx.data()?;
66        let package_store: &Arc<PackageCache> = ctx.data()?;
67        let limits: &Limits = ctx.data()?;
68
69        Ok(Self {
70            checkpoint_viewed_at: Some(watermark.high_watermark().checkpoint()),
71            root_version: None,
72            execution_objects: Arc::new(BTreeMap::new()),
73            package_store: package_store.clone(),
74            resolver_limits: limits.package_resolver(),
75        })
76    }
77
78    /// Create a nested scope pinned to a past checkpoint. Returns `None` if the checkpoint is in
79    /// the future, or if the current scope is in execution context (no checkpoint is set).
80    pub(crate) fn with_checkpoint_viewed_at(&self, checkpoint_viewed_at: u64) -> Option<Self> {
81        let current_cp = self.checkpoint_viewed_at?;
82        (checkpoint_viewed_at <= current_cp).then(|| Self {
83            checkpoint_viewed_at: Some(checkpoint_viewed_at),
84            root_version: self.root_version,
85            execution_objects: Arc::clone(&self.execution_objects),
86            package_store: self.package_store.clone(),
87            resolver_limits: self.resolver_limits.clone(),
88        })
89    }
90
91    /// Create a nested scope for execution context (freshly executed transaction).
92    /// This clears the checkpoint context to indicate fresh execution data and
93    /// sets execution objects from the transaction output.
94    pub(crate) fn with_execution_output<I>(&self, objects: I) -> Self
95    where
96        I: IntoIterator<Item = NativeObject>,
97    {
98        let execution_objects = Arc::new(
99            objects
100                .into_iter()
101                .map(|obj| ((obj.id(), obj.version()), obj))
102                .collect::<BTreeMap<_, _>>(),
103        );
104
105        Self {
106            checkpoint_viewed_at: None,
107            root_version: self.root_version,
108            execution_objects,
109            package_store: self.package_store.clone(),
110            resolver_limits: self.resolver_limits.clone(),
111        }
112    }
113
114    /// Create a nested scope with a root version set.
115    pub(crate) fn with_root_version(&self, root_version: u64) -> Self {
116        Self {
117            checkpoint_viewed_at: self.checkpoint_viewed_at,
118            root_version: Some(root_version),
119            execution_objects: Arc::clone(&self.execution_objects),
120            package_store: self.package_store.clone(),
121            resolver_limits: self.resolver_limits.clone(),
122        }
123    }
124
125    /// Reset the root version.
126    pub(crate) fn without_root_version(&self) -> Self {
127        Self {
128            checkpoint_viewed_at: self.checkpoint_viewed_at,
129            root_version: None,
130            execution_objects: Arc::clone(&self.execution_objects),
131            package_store: self.package_store.clone(),
132            resolver_limits: self.resolver_limits.clone(),
133        }
134    }
135
136    /// Get the checkpoint being viewed, if any.
137    /// Returns `None` in execution context (freshly executed transaction).
138    ///
139    /// Call sites must explicitly handle the execution context case and decide whether
140    /// their operation makes sense without checkpoint context.
141    pub(crate) fn checkpoint_viewed_at(&self) -> Option<u64> {
142        self.checkpoint_viewed_at
143    }
144
145    /// Root parent object version for dynamic fields.
146    pub(crate) fn root_version(&self) -> Option<u64> {
147        self.root_version
148    }
149
150    /// Get the exclusive checkpoint bound, if any.
151    ///
152    /// Returns `None` in execution context (freshly executed transaction).
153    pub(crate) fn checkpoint_viewed_at_exclusive_bound(&self) -> Option<u64> {
154        self.checkpoint_viewed_at.map(|cp| cp + 1)
155    }
156
157    /// Get an object from the execution context cache, if available.
158    pub(crate) fn execution_output_object(
159        &self,
160        object_id: ObjectID,
161        version: SequenceNumber,
162    ) -> Option<&NativeObject> {
163        self.execution_objects.get(&(object_id, version))
164    }
165
166    /// A package resolver with access to the packages known at this scope.
167    pub(crate) fn package_resolver(&self) -> Resolver<Self> {
168        Resolver::new_with_limits(self.clone(), self.resolver_limits.clone())
169    }
170}
171
172impl Debug for Scope {
173    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
174        f.debug_struct("Scope")
175            .field("checkpoint_viewed_at", &self.checkpoint_viewed_at)
176            .field("root_version", &self.root_version)
177            .field("resolver_limits", &self.resolver_limits)
178            .finish()
179    }
180}
181
182#[async_trait]
183impl PackageStore for Scope {
184    /// Fetches a package, first checking execution context objects if available,
185    /// then falling back to the underlying package store.
186    async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>, PackageResolverError> {
187        let object_id = ObjectID::from(id);
188
189        // First check execution context objects if we have any
190        if !self.execution_objects.is_empty() {
191            let latest_package = self
192                .execution_objects
193                .range((object_id, SequenceNumber::MIN)..=(object_id, SequenceNumber::MAX))
194                .last()
195                .and_then(|(_, object)| {
196                    // Check if this object is actually a package
197                    object.data.try_as_package()
198                });
199
200            if let Some(package) = latest_package {
201                return Package::read_from_package(package).map(Arc::new);
202            }
203        }
204
205        // Package not found in execution context, fall back to the underlying store
206        self.package_store.fetch(id).await
207    }
