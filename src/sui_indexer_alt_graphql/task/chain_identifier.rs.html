19#[derive(Clone, Default)]
20pub(crate) struct ChainIdentifier(Arc<SetOnce<NativeChainIdentifier>>);
21
22impl ChainIdentifier {
23    /// Wait for the chain identifier to be set, and return it.
24    pub(crate) async fn wait(&self) -> NativeChainIdentifier {
25        *self.0.wait().await
26    }
27
28    // Set the chain identifier.
29    fn set(&self, chain: NativeChainIdentifier) -> anyhow::Result<()> {
30        self.0.set(chain).context("Chain Identifier already set")
31    }
32}
33
34/// Repeatedly try to fetch the chain identifier from the database at a given interval, eventually
35/// setting it in a `SetOnce` construct. If no database is available, returns a default
36/// ChainIdentifier immediately.
37///
38/// Returns the container that is populated with the chain identifier once it has been fetched, and
39/// a handle to the service performing the fetching.
40pub(crate) fn task(pg_reader: PgReader, interval: Duration) -> (ChainIdentifier, Service) {
41    let crx = ChainIdentifier::default();
42    let ctx = crx.clone();
43
44    let svc = Service::new().spawn_aborting(async move {
45        if !pg_reader.has_database() {
46            ctx.set(NativeChainIdentifier::default())
47                .context("Failed to set ")
48                .expect("SAFETY: Only this task sets the chain identifier, immediately before returning.");
49            return Ok(());
50        }
51
52        let mut interval = time::interval(interval);
53
54        loop {
55            interval.tick().await;
56
57            let mut conn = match pg_reader.connect().await {
58                Ok(conn) => conn,
59                Err(e) => {
60                    warn!("Failed to connect to database: {e}");
61                    continue;
62                }
63            };
64
65            match fetch(&mut conn).await {
66                Ok(chain) => {
67                    ctx.set(chain)
68                        .expect("SAFETY: Only this task sets the chain identifier, immediately before returning.");
69                    return Ok(());
70                }
71
72                Err(e) => {
73                    warn!("Failed to fetch chain identifier: {e}");
74                    continue;
75                }
76            }
77        }
78    });
79
80    (crx, svc)
81}
82
83/// Try to fetch the chain identifier from the database. Fails if the genesis checkpoint digest has
84/// not been written to the database yet, or if it cannot be deserialized as a checkpoint digest.
85async fn fetch(conn: &mut Connection<'_>) -> anyhow::Result<NativeChainIdentifier> {
86    use kv_genesis::dsl as g;
87
88    let digest_bytes: Vec<_> = conn
89        .first(g::kv_genesis.select(g::genesis_digest))
90        .await
91        .context("Failed to fetch digest information")?;
92
93    let digest =
94        CheckpointDigest::try_from(digest_bytes).context("Failed to deserialize genesis digest")?;
95
96    Ok(NativeChainIdentifier::from(digest))
97}
