15/// Repeatedly try to fetch the chain identifier from the database at a given interval, eventually
16/// returning it, or an error if the task was cancelled before it had a chance to complete.
17/// If no database is available, returns a default ChainIdentifier immediately.
18pub(crate) async fn task(
19    pg_reader: &PgReader,
20    interval: Duration,
21    cancel: CancellationToken,
22) -> anyhow::Result<ChainIdentifier> {
23    // If no database is available, return default chain identifier
24    if !pg_reader.has_database() {
25        return Ok(ChainIdentifier::default());
26    }
27
28    let mut interval = time::interval(interval);
29
30    loop {
31        tokio::select! {
32            _ = cancel.cancelled() => {
33                bail!("Shutdown signal received, terminating chain identifier task");
34            }
35
36            _ = interval.tick() => {
37                let mut conn = match pg_reader.connect().await {
38                    Ok(conn) => conn,
39                    Err(e) => {
40                        warn!("Failed to connect to database: {e}");
41                        continue;
42                    }
43                };
44
45                match fetch(&mut conn).await {
46                    Ok(chain_identifier) => return Ok(chain_identifier),
47                    Err(e) => {
48                        warn!("Failed to fetch chain identifier: {e}");
49                        continue;
50                    }
51                }
52            }
53        }
54    }
55}

57/// Try to fetch the chain identifier from the database. Fails if the genesis checkpoint digest has
58/// not been written to the database yet, or if it cannot be deserialized as a checkpoint digest.
59async fn fetch(conn: &mut Connection<'_>) -> anyhow::Result<ChainIdentifier> {
60    use kv_genesis::dsl as g;
61
62    let digest_bytes: Vec<_> = conn
63        .first(g::kv_genesis.select(g::genesis_digest))
64        .await
65        .context("Failed to fetch digest information")?;
66
67    let digest =
68        CheckpointDigest::try_from(digest_bytes).context("Failed to deserialize genesis digest")?;
69
70    Ok(ChainIdentifier::from(digest))
71}
