4//! Checkpoint stream task.
5//!
6//! Maintains the connection to the fullnode's `SubscribeCheckpoints` gRPC
7//! stream and broadcasts every checkpoint, in order, to GraphQL subscribers.
8//!
9//! ```text
10//!   loop forever:
11//!     connect (retry on transient errors, fail on auth/config)
12//!     for each incoming checkpoint:
13//!       if seq > last_broadcast + 1:
14//!         recover_gap(last_broadcast + 1 ..= seq - 1)
15//!       broadcast(checkpoint)
16//!     on stream end: reconnect
17//! ```
18//!
19//! # Reconnect
20//!
21//! gRPC connections drop for many reasons: rolling deploys, network blips,
22//! server restarts. When the stream errors or closes, the task reconnects
23//! with bounded exponential backoff. Transient errors retry indefinitely;
24//! permanent gRPC errors (auth, config) terminate the task.
25//!
26//! # Gap recovery
27//!
28//! `SubscribeCheckpoints` resumes at the current tip on reconnect, not where
29//! we left off. So after a drop, the first new message typically has a
30//! sequence number well past the last we broadcast — checkpoints produced
31//! during the drop are now invisible on the live stream.
32//!
33//! Detection happens on every incoming message: if `seq > last_broadcast + 1`,
34//! the task fetches the missing checkpoints from kv-rpc (a `LedgerService`
35//! endpoint serving historical checkpoints) and broadcasts them in order
36//! before broadcasting the live message.
37//!
38//! ```text
39//!   stream:    ... 10, 11   ╳ drop ╳   16, 17, 18 ...
40//!                                       ↑ first after reconnect
41//!
42//!   on receive 16 (last_broadcast = 11):
43//!     fetch 12..=15 from kv-rpc, broadcast 12..=15
44//!     then broadcast 16
45//! ```
46//!
47//! Recovery is chunked, with each chunk waiting for the kv-rpc indexer to
48//! catch up. See `gap_recovery` for chunk size and retry semantics.
49//!
50//! # Subscriber view
51//!
52//! From a subscriber's perspective, the broadcast is strictly contiguous,
53//! in-order, and contains no duplicates — there's no visible boundary
54//! between recovered and live messages. Long disconnects converge naturally:
55//! if recovery itself takes long enough that the validator advances past
56//! the recovery target, the next live message is itself a gap, and detection
57//! fires again.

176/// Handle on the checkpoint broadcast that subscription resolvers consume from. Registered in
177/// the GraphQL context as a nominal type so its members do not clash with other context entries
178/// that may be added later.
179#[cfg(any(feature = "staging", test))]
180pub(crate) struct SubscriptionBroadcast {
181    /// Receiver created with the channel and never `.recv()`'d. Subscribers call `resubscribe()`
182    /// on it to get their own receivers; we also read `broadcaster.len()` to count how many
183    /// checkpoints the live stream has broadcast since channel creation, which combined with
184    /// `first_live_checkpoint` gives the current network tip without a separate atomic.
185    broadcaster: CheckpointBroadcaster,
186    /// Sequence number of the first checkpoint the live upstream stream broadcast through this
187    /// channel. Distinct from any `resume_from` arg passed by subscribers (those refer to the
188    /// kv-rpc resume fallback, not the live broadcast).
189    first_live_checkpoint: u64,
190}
191
192#[cfg(any(feature = "staging", test))]
193impl SubscriptionBroadcast {
194    pub(crate) fn new(broadcaster: CheckpointBroadcaster, first_live_checkpoint: u64) -> Self {
195        Self {
196            broadcaster,
197            first_live_checkpoint,
198        }
199    }
200
201    /// Direct access to the broadcast receiver template. Subscribers should call
202    /// `.resubscribe()` to get their own receiver.
203    #[cfg(feature = "staging")]
204    pub(crate) fn broadcaster(&self) -> &CheckpointBroadcaster {
205        &self.broadcaster
206    }
207
208    /// Sequence number of the most recently broadcast checkpoint. Derived from the tokio-managed
209    /// `broadcaster.len()` plus the immutable `first_live_checkpoint`, so there is no separate
210    /// shared counter that could race with the broadcast `send`.
211    pub(crate) fn network_tip(&self) -> u64 {
212        self.first_live_checkpoint
213            .saturating_add(self.broadcaster.len() as u64)
214            .saturating_sub(1)
215    }
216
217    /// Subscribe to broadcasted checkpoints, optionally resuming from `resume_from + 1`.
218    ///
219    /// Linear "catch up, then follow": Phase 1 scans toward the tip and pins a `handoff` near it,
220    /// Phase 2 follows the live broadcast from `handoff + 1`, so the phases meet with no gap. Any
221    /// anomaly disconnects (the client reconnects from its last cursor), logged by reason.
222    pub(crate) fn subscribe<F: CheckpointFetcher + Clone + Send + 'static>(
223        self: Arc<Self>,
224        resume_from: Option<u64>,
225        fetcher: F,
226        config: &SubscriptionConfig,
227    ) -> impl Stream<Item = Result<Arc<ProcessedCheckpoint>, RpcError>> + 'static {
228        // Resubscribe and pin the handoff once the scan is within this many checkpoints of the tip.
229        // Half the buffer leaves room for checkpoints arriving during the handoff, so it won't lag.
230        let handoff_threshold = config.broadcast_buffer as u64 / 2;
231
232        let config = config.clone();
233
234        stream! {
235            let mut last_yielded: Option<u64> = resume_from;
236            let mut receiver = self.broadcaster.resubscribe();
237            // `resubscribe` is future-only (delivers `handoff + 1`), so Phase 1 stops exactly at
238            // `handoff` rather than chasing the tip.
239            let mut handoff: Option<u64> = None;
240
241            // Phase 1: scan toward the tip; within `handoff_threshold`, resubscribe + pin, stop at it.
242            if let Some(start_after) = last_yielded {
243                for await item in scan_checkpoints(fetcher, self.clone(), start_after, &config) {
244                    let processed = item?;
245                    let seq = processed.summary.sequence_number;
246                    last_yielded = Some(seq);
247                    yield Ok(processed);
248
249                    if handoff.is_none()
250                        && self.network_tip().saturating_sub(seq) <= handoff_threshold
251                    {
252                        receiver = self.broadcaster.resubscribe();
253                        handoff = Some(self.network_tip());
254                    }
255
256                    if handoff.is_some_and(|h| seq >= h) {
257                        break;
258                    }
259                }
260            }
261
262            // Phase 2: follow live from the pinned receiver, or a fresh one if Phase 1 was skipped.
263            // Tags the disconnect log: a `Lagged` before the first live item is catch-up overflow.
264            let mut delivered_live = false;
265            loop {
266                match receiver.recv().await {
267                    Ok(processed) => match last_yielded {
268                        // Already covered by the scan (resubscribe/tip race): skip.
269                        Some(ly) if processed.summary.sequence_number <= ly => continue,
270                        // Unreachable under pin-and-meet (live is contiguous with the scan; a real
271                        // lag surfaces as `Lagged`). Disconnect defensively rather than emit a hole.
272                        Some(ly) if processed.summary.sequence_number > ly + 1 => {
273                            warn!(
274                                last_yielded = ly,
275                                received = processed.summary.sequence_number,
276                                "Unexpected gap between scan and live; disconnecting"
277                            );
278                            yield Err(reconnect_error());
279                            return;
280                        }
281                        _ => {
282                            last_yielded = Some(processed.summary.sequence_number);
283                            delivered_live = true;
284                            yield Ok(processed);
285                        }
286                    },
287                    Err(broadcast::error::RecvError::Lagged(missed)) if !delivered_live => {
288                        warn!(missed, "Subscriber fell behind during catch-up (likely kv-rpc lag)");
289                        yield Err(reconnect_error());
290                        return;
291                    }
292                    // Slow subscriber (Lagged after going live) or closed channel: disconnect.
293                    Err(e) => {
294                        yield Err(broadcast_error(e));
295                        return;
296                    }
297                }
298            }
299        }
300    }
301}

303/// Convert a broadcast `RecvError` into an `RpcError` to be yielded to the subscriber.
304#[cfg(any(feature = "staging", test))]
305pub(crate) fn broadcast_error(e: broadcast::error::RecvError) -> RpcError {
306    match e {
307        broadcast::error::RecvError::Lagged(missed_count) => {
308            warn!(missed_count, "Subscription lagged, disconnecting");
309            anyhow::anyhow!(
310                "Subscription too slow: missed {missed_count} checkpoints. \
311                 Please reconnect and resume from your last seen checkpoint."
312            )
313            .into()
314        }
315        broadcast::error::RecvError::Closed => {
316            warn!("Checkpoint broadcast channel closed");
317            anyhow::anyhow!("Checkpoint stream has been shut down. Please reconnect.").into()
318        }
319    }
320}

322/// General error for a subscription interrupted by a gap or catch-up overflow. The specific reason
323/// is logged at the call site; clients only see that they should reconnect and resume.
324#[cfg(any(feature = "staging", test))]
325fn reconnect_error() -> RpcError {
326    anyhow::anyhow!(
327        "Subscription interrupted. Please reconnect and resume from your last seen checkpoint."
328    )
329    .into()
330}

332/// Background service that connects to a fullnode's gRPC SubscribeCheckpoints endpoint,
333/// processes incoming checkpoints, and broadcasts them to subscription resolvers.
334pub(crate) struct CheckpointStreamTask {
335    uri: Uri,
336    sender: broadcast::Sender<Arc<ProcessedCheckpoint>>,
337    streaming_packages: Arc<StreamingPackageStore>,
338    package_eviction_tx: UnboundedSender<(u64, Vec<AccountAddress>)>,
339    readiness: Arc<SubscriptionReadiness>,
340    /// kv-rpc reader used to fill upstream gaps. Required: streaming subscriptions need a
341    /// fallback source to recover from disconnects. lib.rs ensures this is configured when
342    /// streaming is enabled.
343    ledger_grpc_reader: LedgerGrpcReader,
344    /// Watermarks receiver used by gap recovery to wait for kv-rpc to be at or past a target
345    /// before fetching.
346    watermarks_rx: watch::Receiver<Arc<Watermarks>>,
347    gap_recovery_chunk_size: usize,
348}
349
350impl CheckpointStreamTask {
351    /// Returns the task and the anchor receiver created with the broadcast channel. The
352    /// receiver must outlive `run()` (which consumes the task), so the caller holds onto it
353    /// and pairs it with the first streamed checkpoint (from `SubscriptionReadiness`) to
354    /// build a `SubscriptionBroadcast` once readiness fires.
355    pub(crate) fn new(
356        uri: Uri,
357        config: &SubscriptionConfig,
358        streaming_packages: Arc<StreamingPackageStore>,
359        package_eviction_tx: UnboundedSender<(u64, Vec<AccountAddress>)>,
360        readiness: Arc<SubscriptionReadiness>,
361        ledger_grpc_reader: LedgerGrpcReader,
362        watermarks_rx: watch::Receiver<Arc<Watermarks>>,
363    ) -> (Self, CheckpointBroadcaster) {
364        let (sender, broadcaster) = broadcast::channel(config.broadcast_buffer);
365        let task = Self {
366            uri,
367            sender,
368            streaming_packages,
369            package_eviction_tx,
370            readiness,
371            ledger_grpc_reader,
372            watermarks_rx,
373            gap_recovery_chunk_size: config.gap_recovery_chunk_size,
374        };
375        (task, broadcaster)
376    }
377
378    /// Connect to the fullnode's gRPC SubscribeCheckpoints endpoint.
379    async fn connect(&self) -> anyhow::Result<Streaming<SubscribeCheckpointsResponse>> {
380        let endpoint = Endpoint::from(self.uri.clone()).connect_timeout(CONNECTION_TIMEOUT);
381
382        let mut client = SubscriptionServiceClient::connect(endpoint)
383            .await
384            .context("Failed to connect to checkpoint stream")?
385            .max_decoding_message_size(MAX_GRPC_MESSAGE_SIZE_BYTES);
386
387        let mut request = SubscribeCheckpointsRequest::default();
388        request.read_mask = Some(checkpoint_field_mask());
389
390        let stream = client
391            .subscribe_checkpoints(request)
392            .await
393            .context("Failed to subscribe to checkpoint stream")?
394            .into_inner();
395
396        Ok(stream)
397    }
398
399    /// Drive the upstream checkpoint stream. Reconnects on any failure with bounded backoff,
400    /// detects gaps on each incoming checkpoint, and fills them via kv-rpc before broadcasting
401    /// the live message. Permanent connect errors (auth / config) terminate the task; transient
402    /// errors retry indefinitely.
403    pub(crate) fn run(self) -> Service {
404        Service::new().spawn_aborting(async move {
405            let mut last_broadcast: Option<u64> = None;
406            let mut first_live_recorded = false;
407
408            loop {
409                info!("Connecting to checkpoint stream at {}...", self.uri);
410                let stream = backoff::future::retry(reconnect_backoff(), || async {
411                    self.connect().await.map_err(classify_connect_error)
412                })
413                .await?;
414                info!("Connected to checkpoint stream at {}", self.uri);
415
416                self.consume_stream(stream, &mut last_broadcast, &mut first_live_recorded)
417                    .await?;
418                warn!("Checkpoint stream ended, reconnecting");
419            }
420        })
421    }
422
423    /// Consume one connected stream until it ends or errors. On stream-level errors, logs
424    /// and returns `Ok(())` so the caller reconnects. Per-message errors (malformed proto,
425    /// gap recovery failure) propagate as `Err` and terminate the task.
426    ///
427    /// Generic over the stream type so tests can feed synthetic messages without standing
428    /// up a real gRPC server. Tonic's `Streaming<T>` already implements `Stream`, so the
429    /// production call site is unchanged.
430    async fn consume_stream<S>(
431        &self,
432        mut stream: S,
433        last_broadcast: &mut Option<u64>,
434        first_live_recorded: &mut bool,
435    ) -> anyhow::Result<()>
436    where
437        S: futures::Stream<Item = Result<SubscribeCheckpointsResponse, tonic::Status>> + Unpin,
438    {
439        while let Some(result) = stream.next().await {
440            let response = match result {
441                Ok(r) => r,
442                Err(e) => {
443                    warn!("Stream error: {e:#}");
444                    return Ok(());
445                }
446            };
447
448            let Some(checkpoint) = response.checkpoint else {
449                continue;
450            };
451            let seq = checkpoint
452                .sequence_number
453                .context("Checkpoint without sequence_number")?;
454
455            if !*first_live_recorded {
456                self.readiness.record_first_live_checkpoint(seq);
457                *first_live_recorded = true;
458            }
459
460            // Gap detection: synchronously fill any hole between last_broadcast and this
461            // message before broadcasting the live cp.
462            if let Some(last) = *last_broadcast
463                && seq > last + 1
464            {
465                info!(from = last + 1, to = seq - 1, "Recovering gap");
466                recover_gap(
467                    &self.ledger_grpc_reader,
468                    &self.watermarks_rx,
469                    &self.sender,
470                    last + 1,
471                    seq - 1,
472                    self.gap_recovery_chunk_size,
473                )
474                .await?;
475            }
476
477            self.index_and_broadcast(checkpoint, seq)?;
478            *last_broadcast = Some(seq);
479        }
480
481        Ok(())
482    }
483
484    /// Index packages from the checkpoint into the streaming store, signal eviction,
485    /// then process and broadcast. Indexing exposes the checkpoint's packages to
486    /// subscribers immediately, ahead of `kv_packages`. Recovered checkpoints from
487    /// `recover_gap` skip this step because the gate already waits for both
488    /// `ledger_grpc` and `kv_packages` to catch up.
489    fn index_and_broadcast(&self, checkpoint: ProtoCheckpoint, seq: u64) -> anyhow::Result<()> {
490        let packages = extract_packages(&checkpoint);
491        if !packages.is_empty() {
492            self.streaming_packages.index_packages(seq, &packages);
493            let ids = packages.iter().map(|p| p.storage_id()).collect();
494            // Send errors only if the eviction task has exited; nothing will drain the
495            // store, but we keep serving.
496            let _ = self.package_eviction_tx.send((seq, ids));
497        }
498        let processed = process_checkpoint(checkpoint)?;
499        // Ignore send errors: no active subscribers is a normal state.
500        let _ = self.sender.send(Arc::new(processed));
501        Ok(())
502    }
503}

505/// Backoff between reconnect attempts: 1s growing to 5s max, no overall deadline.
506/// Tight cap so we recover quickly from rolling deploys; infinite retry so the streaming
507/// server stays alive through extended outages.
508fn reconnect_backoff() -> ExponentialBackoff {
509    ExponentialBackoff {
510        initial_interval: Duration::from_secs(1),
511        max_interval: Duration::from_secs(5),
512        max_elapsed_time: None,
513        ..Default::default()
514    }
515}
516
517/// Classify connect errors as transient or permanent. gRPC codes that signal config problems
518/// (auth, missing endpoint, malformed request) terminate the task; everything else retries.
519fn classify_connect_error(e: anyhow::Error) -> backoff::Error<anyhow::Error> {
520    use tonic::Code;
521    let permanent = e.downcast_ref::<tonic::Status>().is_some_and(|s| {
522        matches!(
523            s.code(),
524            Code::Unauthenticated
525                | Code::PermissionDenied
526                | Code::InvalidArgument
527                | Code::Unimplemented
528        )
529    });
530
531    if permanent {
532        warn!("Permanent connect failure (config / auth issue): {e:#}");
533        backoff::Error::permanent(e)
534    } else {
535        warn!("Connect failed, will retry: {e:#}");
536        backoff::Error::transient(e)
537    }
538}

540pub(super) fn process_checkpoint(
541    checkpoint: ProtoCheckpoint,
542) -> anyhow::Result<ProcessedCheckpoint> {
543    let sequence_number = checkpoint
544        .sequence_number
545        .context("Checkpoint without sequence_number")?;
546
547    let summary: CheckpointSummary = checkpoint
548        .summary
549        .as_ref()
550        .and_then(|s| s.bcs.as_ref())
551        .context("Missing summary.bcs")?
552        .deserialize()
553        .context("Failed to deserialize checkpoint summary")?;
554
555    let contents: CheckpointContents = checkpoint
556        .contents
557        .as_ref()
558        .and_then(|c| c.bcs.as_ref())
559        .context("Missing contents.bcs")?
560        .deserialize()
561        .context("Failed to deserialize checkpoint contents")?;
562
563    let signature: AuthorityStrongQuorumSignInfo = {
564        let sdk_sig = ValidatorAggregatedSignature::try_from(
565            checkpoint.signature.as_ref().context("Missing signature")?,
566        )
567        .context("Failed to parse checkpoint signature")?;
568        AuthorityStrongQuorumSignInfo::from(sdk_sig)
569    };
570
571    let timestamp_ms = summary.timestamp_ms;
572    let cp_sequence_number = sequence_number;
573    let tx_lo = summary.network_total_transactions - checkpoint.transactions.len() as u64;
574    #[cfg(feature = "staging")]
575    let checkpoint_objects = deserialize_checkpoint_objects(&checkpoint)?;
576
577    // Seed the checkpoint-wide execution-objects map with every existing object version
578    // carried by the proto. Tombstones for deleted/wrapped outputs are added below from each
579    // transaction's effects because the proto doesn't carry deleted-object payloads.
580    #[cfg(feature = "staging")]
581    let mut execution_objects: BTreeMap<(ObjectID, SequenceNumber), Option<NativeObject>> =
582        checkpoint_objects
583            .iter()
584            .map(|(k, v)| (*k, Some(v.clone())))
585            .collect();
586
587    let mut transactions = Vec::with_capacity(checkpoint.transactions.len());
588    for (i, proto) in checkpoint.transactions.iter().enumerate() {
589        #[cfg(feature = "staging")]
590        add_tombstones(&mut execution_objects, proto)?;
591        transactions.push(process_transaction(
592            proto,
593            timestamp_ms,
594            cp_sequence_number,
595            tx_lo + i as u64,
596        )?);
597    }
598
599    Ok(ProcessedCheckpoint {
600        summary,
601        contents,
602        signature,
603        transactions,
604        #[cfg(feature = "staging")]
605        execution_objects: Arc::new(execution_objects),
606    })
607}

609fn process_transaction(
610    proto: &ProtoExecutedTransaction,
611    timestamp_ms: u64,
612    cp_sequence_number: u64,
613    tx_sequence_number: u64,
614) -> anyhow::Result<ProcessedTransaction> {
615    let transaction_data: TransactionData = proto
616        .transaction
617        .as_ref()
618        .and_then(|t| t.bcs.as_ref())
619        .context("Missing transaction.bcs")?
620        .deserialize()
621        .context("Failed to deserialize transaction data")?;
622
623    let effects: TransactionEffects = proto
624        .effects
625        .as_ref()
626        .and_then(|e| e.bcs.as_ref())
627        .context("Missing effects.bcs")?
628        .deserialize()
629        .context("Failed to deserialize transaction effects")?;
630
631    let events: Vec<Arc<_>> = proto
632        .events
633        .as_ref()
634        .and_then(|e| e.bcs.as_ref())
635        .map(|bcs| {
636            bcs.deserialize()
637                .context("Failed to deserialize transaction events")
638        })
639        .transpose()?
640        .map(|e: TransactionEvents| e.data.into_iter().map(Arc::new).collect())
641        .unwrap_or_default();
642
643    let signatures: Vec<GenericSignature> = proto
644        .signatures
645        .iter()
646        .map(|sig| {
647            let bytes = sig
648                .bcs
649                .as_ref()
650                .context("Missing signature bcs")?
651                .value
652                .as_deref()
653                .unwrap_or(&[]);
654            GenericSignature::from_bytes(bytes)
655                .map_err(|e| anyhow::anyhow!("Failed to parse user signature: {e}"))
656        })
657        .collect::<anyhow::Result<_>>()?;
658
659    let contents = NativeTransactionContents::ExecutedTransaction(
660        sui_indexer_alt_reader::kv_loader::ExecutedTransactionData {
661            effects: Box::new(effects),
662            events,
663            transaction_data: Box::new(transaction_data),
664            signatures,
665            balance_changes: proto.balance_changes.clone(),
666            proto_effects: proto.effects.clone(),
667            proto_transaction: proto.transaction.clone(),
668            timestamp_ms: Some(timestamp_ms),
669            cp_sequence_number: Some(cp_sequence_number),
670        },
671    );
672
673    Ok(ProcessedTransaction {
674        tx_sequence_number,
675        contents: Arc::new(contents),
676    })
677}

679/// Extract package objects from a streamed checkpoint. Returned packages will be
680/// inserted into the streaming index and queued for eventual eviction.
681fn extract_packages(checkpoint: &ProtoCheckpoint) -> Vec<Arc<Package>> {
682    let Some(object_set) = &checkpoint.objects else {
683        return Vec::new();
684    };
685    let mut packages = Vec::new();
686    for obj in &object_set.objects {
687        let Some(bcs) = &obj.bcs else { continue };
688        let Ok(native_obj) = bcs.deserialize::<NativeObject>() else {
689            continue;
690        };
691        let Some(move_package) = native_obj.data.try_as_package() else {
692            continue;
693        };
694        let Ok(package) = Package::read_from_package(move_package) else {
695            continue;
696        };
697        packages.push(Arc::new(package));
698    }
699    packages
700}

702/// Deserialize all objects from the checkpoint-level ObjectSet.
703#[cfg(feature = "staging")]
704fn deserialize_checkpoint_objects(
705    checkpoint: &ProtoCheckpoint,
706) -> anyhow::Result<BTreeMap<(ObjectID, SequenceNumber), NativeObject>> {
707    let mut map = BTreeMap::new();
708    if let Some(object_set) = &checkpoint.objects {
709        for obj in &object_set.objects {
710            if let Some(bcs) = &obj.bcs {
711                let native_obj: NativeObject = bcs
712                    .deserialize()
713                    .context("Failed to deserialize checkpoint object BCS")?;
714                map.insert((native_obj.id(), native_obj.version()), native_obj);
715            }
716        }
717    }
718    Ok(map)
719}

721/// Add tombstones (`None` entries) to a checkpoint-wide execution-objects map for any
722/// `DoesNotExist` outputs in this transaction's effects. The proto's `objects` field doesn't
723/// carry payloads for deleted/wrapped objects, so tombstones must come from effects; without
724/// them, `execution_output_object_latest` would return the pre-deletion version of an object
725/// that no longer exists at end-of-checkpoint.
726#[cfg(feature = "staging")]
727fn add_tombstones(
728    map: &mut BTreeMap<(ObjectID, SequenceNumber), Option<NativeObject>>,
729    proto_tx: &ProtoExecutedTransaction,
730) -> anyhow::Result<()> {
731    let Some(effects) = &proto_tx.effects else {
732        return Ok(());
733    };
734
735    let lamport_version = SequenceNumber::from_u64(
736        effects
737            .lamport_version
738            .context("Effects should have lamport_version")?,
739    );
740
741    for changed_obj in &effects.changed_objects {
742        if changed_obj.output_state() != OutputObjectState::DoesNotExist {
743            continue;
744        }
745        let object_id: ObjectID = changed_obj
746            .object_id
747            .as_ref()
748            .and_then(|id| id.parse().ok())
749            .context("ChangedObject should have valid object_id")?;
750        map.insert((object_id, lamport_version), None);
751    }
752
753    Ok(())
754}

763    /// Drain `n` items from the (pinned) stream, returning their sequence numbers. The stream
764    /// is borrowed via `Pin<&mut _>` so multiple `drain_n` calls can interleave with other test
765    /// operations (e.g., bumping the broadcast tip between phases).
766    async fn drain_n<S>(mut stream: std::pin::Pin<&mut S>, n: usize) -> Vec<u64>
767    where
768        S: Stream<Item = Result<Arc<ProcessedCheckpoint>, RpcError>>,
769    {
770        use futures::StreamExt;
771        let mut out = Vec::with_capacity(n);
772        for _ in 0..n {
773            let item = stream.as_mut().next().await.unwrap().unwrap();
774            out.push(item.summary.sequence_number);
775        }
776        out
777    }
