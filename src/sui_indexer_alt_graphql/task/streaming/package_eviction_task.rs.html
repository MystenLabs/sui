21/// Background task that evicts packages from the `StreamedPackageStore` once the
22/// `kv_packages` pipeline has indexed the checkpoint that introduced them.
23///
24/// The stream task sends `(checkpoint_seq, package_ids)` entries to this task over
25/// an unbounded mpsc channel. On each timer tick, the task drains all entries whose
26/// checkpoint is at or below the current `kv_packages` watermark.
27pub(crate) struct PackageEvictionTask<S> {
28    streaming_packages: Arc<StreamedPackageStore<S>>,
29    receiver: UnboundedReceiver<(u64, Vec<AccountAddress>)>,
30    watermarks: WatermarksLock,
31    eviction_interval: Duration,
32}
33
34impl<S: Send + Sync + 'static> PackageEvictionTask<S> {
35    pub(crate) fn new(
36        streaming_packages: Arc<StreamedPackageStore<S>>,
37        receiver: UnboundedReceiver<(u64, Vec<AccountAddress>)>,
38        watermarks: WatermarksLock,
39        eviction_interval: Duration,
40    ) -> Self {
41        Self {
42            streaming_packages,
43            receiver,
44            watermarks,
45            eviction_interval,
46        }
47    }
48
49    pub(crate) fn run(self) -> Service {
50        let Self {
51            streaming_packages,
52            receiver,
53            watermarks,
54            eviction_interval,
55        } = self;
56
57        Service::new().spawn_aborting(async move {
58            let mut stream = Box::pin(UnboundedReceiverStream::new(receiver).peekable());
59            let mut interval = tokio::time::interval(eviction_interval);
60
61            loop {
62                interval.tick().await;
63
64                let Some(kv_packages_hi) = kv_packages_watermark(&watermarks).await else {
65                    continue;
66                };
67
68                // Drain entries at or below the watermark. `peek().now_or_never()`
69                // gives us a non-blocking peek, so we can stop immediately when the
70                // channel is empty or the next entry is above the watermark.
71                loop {
72                    let peeked_cp = stream
73                        .as_mut()
74                        .peek()
75                        .now_or_never()
76                        .map(|opt| opt.map(|(cp, _)| *cp));
77
78                    match peeked_cp {
79                        None => break,
80                        Some(None) => return Ok(()),
81                        Some(Some(cp)) if cp > kv_packages_hi => break,
82                        Some(Some(_)) => {
83                            let (cp_seq, package_ids) = stream
84                                .next()
85                                .await
86                                .expect("peek returned Some; next must yield same item");
87                            streaming_packages.evict_checkpoint(cp_seq, &package_ids);
88                            debug!(
89                                checkpoint = cp_seq,
90                                "Evicted {} packages",
91                                package_ids.len()
92                            );
93                        }
94                    }
95                }
96            }
97        })
98    }
99}
100
101/// Read the current `kv_packages` high watermark from the shared watermarks.
102/// Returns `None` if the pipeline is not being tracked.
103async fn kv_packages_watermark(watermarks: &WatermarksLock) -> Option<u64> {
104    let watermarks = watermarks.read().await;
105    watermarks
106        .per_pipeline()
107        .get(KV_PACKAGES_PIPELINE)
108        .map(|p: &Pipeline| p.hi().checkpoint())
109}

113mod tests {

125    #[async_trait::async_trait]
126    impl sui_package_resolver::PackageStore for MockStore {
127        async fn fetch(
128            &self,
129            id: AccountAddress,

131            Err(PackageResolverError::PackageNotFound(id))

134
135    fn addr(n: u8) -> AccountAddress {
136        let mut bytes = [0u8; AccountAddress::LENGTH];
137        bytes[AccountAddress::LENGTH - 1] = n;
138        AccountAddress::new(bytes)
139    }
140
141    fn pkg(id: AccountAddress, version: u64) -> Arc<sui_package_resolver::Package> {
142        Arc::new(sui_package_resolver::Package::for_test(
143            id,
144            SequenceNumber::from_u64(version),
145        ))
146    }
147
148    async fn set_kv_packages_hi(watermarks: &WatermarksLock, hi: u64) {
149        *watermarks.write().await = Arc::new(Watermarks::for_test(&[("kv_packages", hi)]));
150    }
151
152    #[tokio::test(start_paused = true)]
153    async fn evicts_entries_at_or_below_watermark() {
154        let store = Arc::new(StreamedPackageStore::new(MockStore));
155        let (tx, rx) = unbounded_channel();
156        let watermarks: WatermarksLock = Default::default();
157
158        store.index_packages(5, &[pkg(addr(1), 1)]);
159        tx.send((5, vec![addr(1)])).unwrap();
160        set_kv_packages_hi(&watermarks, 10).await;
161
162        let _service = PackageEvictionTask::new(
163            store.clone(),
164            rx,
165            watermarks.clone(),
166            Duration::from_secs(5),
167        )
168        .run();
169
170        tokio::time::advance(Duration::from_secs(6)).await;
171        tokio::task::yield_now().await;
172
173        assert!(store.fetch(addr(1)).await.is_err());
174    }
