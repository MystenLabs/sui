12/// Package store for streaming subscriptions that holds packages not yet indexed by the DB.
13///
14/// Packages from streamed checkpoints are indexed here. Once the `kv_packages` pipeline
15/// catches up, a separate eviction task removes them — at that point the inner store
16/// (LRU → PackageCache → DB) can serve them instead.
17///
18/// Each package entry stores the checkpoint that introduced it, so that eviction of an
19/// older checkpoint does not accidentally remove a system package that was upgraded at a
20/// later checkpoint.
21pub(crate) struct StreamedPackageStore<S> {
22    /// Primary index: packages from streamed checkpoints not yet in the DB.
23    packages: DashMap<AccountAddress, IndexedPackage>,
24
25    /// Fallback store (typically the shared PackageCache → DB).
26    inner: S,
27}
28
29struct IndexedPackage {
30    checkpoint: u64,
31    package: Arc<Package>,
32}
33
34impl<S> StreamedPackageStore<S> {
35    pub(crate) fn new(inner: S) -> Self {
36        Self {
37            packages: DashMap::new(),
38            inner,
39        }
40    }
41
42    /// Index packages from a streamed checkpoint. Called by the checkpoint stream task.
43    ///
44    /// Checkpoints are processed sequentially, so the latest insert for a given package
45    /// ID is always the newest version.
46    pub(crate) fn index_packages(&self, checkpoint_seq: u64, packages: &[Arc<Package>]) {
47        for package in packages {
48            self.packages.insert(
49                package.storage_id(),
50                IndexedPackage {
51                    checkpoint: checkpoint_seq,
52                    package: package.clone(),
53                },
54            );
55        }
56    }
57
58    /// Remove packages that were introduced at `checkpoint_seq` from the primary index.
59    ///
60    /// Uses `DashMap::remove_if` for atomic checked removal: a package is only removed
61    /// if its stored checkpoint still matches. This handles system package upgrades where
62    /// the same ID is re-inserted at a later checkpoint.
63    pub(crate) fn evict_checkpoint(&self, checkpoint_seq: u64, package_ids: &[AccountAddress]) {
64        for id in package_ids {
65            self.packages
66                .remove_if(id, |_, entry| entry.checkpoint == checkpoint_seq);
67        }
68    }
69}
70
71#[async_trait::async_trait]
72impl<S: PackageStore> PackageStore for StreamedPackageStore<S> {
73    async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>> {
74        if let Some(entry) = self.packages.get(&id) {
75            return Ok(entry.package.clone());
76        }
77
78        self.inner.fetch(id).await
79    }
80}

82#[cfg(test)]
83mod tests {
84    use std::collections::HashMap;
85    use std::sync::Mutex;
86
87    use sui_package_resolver::error::Error as PackageResolverError;
88    use sui_types::base_types::SequenceNumber;
89
90    use super::*;
91
92    struct MockStore {
93        packages: Mutex<HashMap<AccountAddress, Arc<Package>>>,
94    }
95
96    impl MockStore {
97        fn new() -> Self {
98            Self {
99                packages: Mutex::new(HashMap::new()),
100            }
101        }
102
103        fn with(id: AccountAddress, package: Arc<Package>) -> Self {
104            let store = Self::new();
105            store.packages.lock().unwrap().insert(id, package);
106            store
107        }
108    }
109
110    #[async_trait::async_trait]
111    impl PackageStore for MockStore {
112        async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>> {
113            self.packages
114                .lock()
115                .unwrap()
116                .get(&id)
117                .cloned()
118                .ok_or(PackageResolverError::PackageNotFound(id))
119        }
120    }
121
122    fn addr(n: u8) -> AccountAddress {
123        let mut bytes = [0u8; AccountAddress::LENGTH];
124        bytes[AccountAddress::LENGTH - 1] = n;
125        AccountAddress::new(bytes)
126    }
127
128    fn pkg(id: AccountAddress, version: u64) -> Arc<Package> {
129        Arc::new(Package::for_test(id, SequenceNumber::from_u64(version)))
130    }
131
132    #[tokio::test]
133    async fn fetch_hits_primary_index() {
134        let store = StreamedPackageStore::new(MockStore::new());
135        let p = pkg(addr(1), 1);
136        store.index_packages(5, std::slice::from_ref(&p));
137
138        assert!(Arc::ptr_eq(&store.fetch(addr(1)).await.unwrap(), &p));
139    }
140
141    #[tokio::test]
142    async fn fetch_falls_through_to_inner() {
143        let p = pkg(addr(1), 1);
144        let store = StreamedPackageStore::new(MockStore::with(addr(1), p.clone()));
145
146        assert!(Arc::ptr_eq(&store.fetch(addr(1)).await.unwrap(), &p));
147    }
148
149    #[tokio::test]
150    async fn fetch_primary_takes_precedence_over_inner() {
151        let primary = pkg(addr(1), 2);
152        let store = StreamedPackageStore::new(MockStore::with(addr(1), pkg(addr(1), 1)));
153        store.index_packages(5, std::slice::from_ref(&primary));
154
155        assert!(Arc::ptr_eq(&store.fetch(addr(1)).await.unwrap(), &primary));
156    }
157
158    #[tokio::test]
159    async fn evict_removes_matching_checkpoint() {
160        let store = StreamedPackageStore::new(MockStore::new());
161        store.index_packages(5, &[pkg(addr(1), 1)]);
162
163        store.evict_checkpoint(5, &[addr(1)]);
164
165        assert!(store.fetch(addr(1)).await.is_err());
166    }
167
168    #[tokio::test]
169    async fn evict_skips_mismatched_checkpoint() {
170        // Simulates a system package upgrade: same ID indexed twice at different
171        // checkpoints. Evicting the older checkpoint should NOT remove the newer entry.
172        let store = StreamedPackageStore::new(MockStore::new());
173        store.index_packages(5, &[pkg(addr(1), 1)]);
174        let upgraded = pkg(addr(1), 2);
175        store.index_packages(10, std::slice::from_ref(&upgraded));
176
177        store.evict_checkpoint(5, &[addr(1)]);
178
179        assert!(Arc::ptr_eq(&store.fetch(addr(1)).await.unwrap(), &upgraded));
180    }
181
182    #[tokio::test]
183    async fn evict_handles_multiple_packages() {
184        let store = StreamedPackageStore::new(MockStore::new());
185        store.index_packages(5, &[pkg(addr(1), 1), pkg(addr(2), 1), pkg(addr(3), 1)]);
186
187        store.evict_checkpoint(5, &[addr(1), addr(2), addr(3)]);
188
189        assert!(store.fetch(addr(1)).await.is_err());
190        assert!(store.fetch(addr(2)).await.is_err());
191        assert!(store.fetch(addr(3)).await.is_err());
192    }
