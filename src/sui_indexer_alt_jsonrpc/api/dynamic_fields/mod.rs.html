23#[open_rpc(namespace = "suix", tag = "Dynamic Fields API")]
24#[rpc(server, namespace = "suix")]
25trait DynamicFieldsApi {
26    /// Return the information from a dynamic field based on its parent ID and name.
27    #[method(name = "getDynamicFieldObject")]
28    async fn get_dynamic_field_object(
29        &self,
30        /// The ID of the parent object
31        parent_object_id: ObjectID,
32        /// The Name of the dynamic field
33        name: DynamicFieldName,
34    ) -> RpcResult<SuiObjectResponse>;
35
36    /// Query dynamic fields by their parent object. Returns a paginated list of object info.
37    ///
38    /// If a cursor is provided, the query will start from the dynamic field after the one pointed
39    /// to by this cursor, otherwise pagination starts from the first page of dynamic fields
40    /// owned by the object.
41    ///
42    /// The size of each page is controlled by the `limit` parameter.
43    #[method(name = "getDynamicFields")]
44    async fn get_dynamic_fields(
45        &self,
46        /// The ID of the parent object
47        parent_object_id: ObjectID,
48        /// Cursor to start paginating from.
49        cursor: Option<String>,
50        /// Maximum number of objects to return per page.
51        limit: Option<usize>,
52    ) -> RpcResult<Page<DynamicFieldInfoResponse, String>>;
53}
54
55pub struct DynamicFields(pub Context);
56
57#[async_trait::async_trait]
58impl DynamicFieldsApiServer for DynamicFields {
59    async fn get_dynamic_field_object(
60        &self,
61        parent_object_id: ObjectID,
62        name: DynamicFieldName,
63    ) -> RpcResult<SuiObjectResponse> {
64        let Self(ctx) = self;
65        Ok(response::dynamic_field_object(ctx, parent_object_id, name).await?)
66    }
67
68    async fn get_dynamic_fields(
69        &self,
70        parent_object_id: ObjectID,
71        cursor: Option<String>,
72        limit: Option<usize>,
73    ) -> RpcResult<Page<DynamicFieldInfoResponse, String>> {
74        let Self(ctx) = self;
75
76        let Page {
77            data: object_ids,
78            next_cursor,
79            has_next_page,
80        } = objects::filter::dynamic_fields(ctx, parent_object_id, cursor, limit).await?;
81
82        let df_futures = object_ids
83            .iter()
84            .map(|id| response::dynamic_field_info(ctx, *id));
85
86        let data = future::join_all(df_futures)
87            .await
88            .into_iter()
89            .zip(object_ids)
90            .map(|(r, id)| {
91                r.with_internal_context(|| format!("Failed to get object {id} at latest version"))
92            })
93            .collect::<Result<Vec<_>, _>>()?;
94
95        Ok(Page {
96            data,
97            next_cursor: next_cursor.map(|id| id.to_string()),
98            has_next_page,
99        })
100    }
101}
102
103impl RpcModule for DynamicFields {
104    fn schema(&self) -> Module {
105        DynamicFieldsApiOpenRpc::module_doc()
106    }
107
108    fn into_impl(self) -> jsonrpsee::RpcModule<Self> {
109        self.into_rpc()
110    }
