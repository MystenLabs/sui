28/// Fetch the latest version of a dynamic field object, identified by its parent ID and name.
29pub(super) async fn dynamic_field_object(
30    ctx: &Context,
31    parent_object_id: ObjectID,
32    name: DynamicFieldName,
33) -> Result<SuiObjectResponse, RpcError<Error>> {
34    let layout = resolve_type(ctx, &name.type_).await?;
35    let bytes = SuiJsonValue::new(name.value)
36        .map_err(|e| invalid_params(Error::BadName(e)))?
37        .to_bcs_bytes(&layout)
38        .map_err(|e| invalid_params(Error::TypeMismatch(name.type_.clone(), e)))?;
39
40    let df = load_df(ctx, parent_object_id, &name.type_, &bytes);
41    let dof = load_dof(ctx, parent_object_id, &name.type_, &bytes);
42    let (df, dof) = try_join!(df, dof)
43        .with_context(|| format!("Failed to fetch dynamic field on {parent_object_id}"))?;
44
45    let Some(object) = df.or(dof) else {
46        return Ok(SuiObjectResponse::new_with_error(
47            SuiObjectResponseError::DynamicFieldNotFound { parent_object_id },
48        ));
49    };
50
51    let options = SuiObjectDataOptions::full_content();
52
53    use RpcError as E;
54    Ok(SuiObjectResponse::new_with_data(
55        objects::response::object_data_with_options(ctx, object, &options)
56            .await
57            .map_err(|e| match e {
58                E::InvalidParams(e) => match e {},
59                E::Timeout(e) => E::Timeout(e),
60                E::InternalError(e) => E::InternalError(e),
61            })?,
62    ))
63}

65/// Fetch the latest version of the object identified by `object_id`, treat it as if it is a
66/// `sui::dynamic_field::Field<K, V>`, and extract the name and value from it.
67pub(super) async fn dynamic_field_info(
68    ctx: &Context,
69    object_id: ObjectID,
70) -> Result<DynamicFieldInfoResponse, RpcError<Error>> {
71    let object = load_live(ctx, object_id)
72        .await
73        .context("Failed to load dynamic field")?
74        .context("Could not find latest content for dynamic field")?;
75
76    let Some(move_object) = object.data.try_as_move() else {
77        rpc_bail!("Dynamic field at {object_id} is not a Move Object");
78    };
79
80    let type_ = move_object.type_().clone().into();
81    let layout = resolve_type(ctx, &type_).await?;
82
83    let field = DFV::FieldVisitor::deserialize(move_object.contents(), &layout)
84        .context("Failed to deserialize dynamic field info")?;
85
86    let type_ = field.kind;
87    let name_type: TypeTag = field.name_layout.into();
88    let bcs_name = BcsName::Base64 {
89        bcs_name: field.name_bytes.to_owned(),
90    };
91
92    let name_value = BoundedVisitor::deserialize_value(field.name_bytes, field.name_layout)
93        .context("Failed to deserialize dynamic field name")?;
94
95    let name = DynamicFieldName {
96        type_: name_type,
97        value: SuiMoveValue::from(name_value).to_json_value(),
98    };
99
100    let value_metadata = field
101        .value_metadata()
102        .context("Failed to extract dynamic field value metadata")?;
103
104    Ok(match value_metadata {
105        DFV::ValueMetadata::DynamicField(object_type) => DynamicFieldInfoResponse {
106            name,
107            bcs_name,
108            type_,
109            object_type: object_type.to_canonical_string(/* with_prefix */ true),
110            object_id: object.id(),
111            version: object.version(),
112            digest: object.digest(),
113        },
114
115        DFV::ValueMetadata::DynamicObjectField(object_id) => {
116            let object = load_live(ctx, object_id)
117                .await
118                .context("Failed to load dynamic object field value")?
119                .context("Could not find latest content for dynamic object field value")?;
120
121            let Some(object_type) = object.data.type_().cloned() else {
122                rpc_bail!("Dynamic object field value at {object_id} is not a Move Object");
123            };
124
125            DynamicFieldInfoResponse {
126                name,
127                bcs_name,
128                type_,
129                object_type: object_type.to_canonical_string(/* with_prefix */ true),
130                object_id: object.id(),
131                version: object.version(),
132                digest: object.digest(),
133            }
134        }
135    })
136}

138/// Resolve the layout for a given type tag, using the package resolver in the context.
139/// Re-interprets errors from the package resolver, categorizing them as RPC user errors or
140/// internal errors.
141async fn resolve_type(ctx: &Context, type_: &TypeTag) -> Result<MoveTypeLayout, RpcError<Error>> {
142    ctx.package_resolver()
143        .type_layout(type_.clone())
144        .await
145        .map_err(|e| {
146            use sui_package_resolver::error::Error as PRE;
147            match &e {
148                // These errors can be triggered by passing a type that doesn't exist for the
149                // dynamic field name.
150                PRE::NotAPackage(_)
151                | PRE::PackageNotFound(_)
152                | PRE::ModuleNotFound(_, _)
153                | PRE::DatatypeNotFound(_, _, _)
154                | PRE::TypeArityMismatch(_, _) => invalid_params(Error::BadType(type_.clone(), e)),
155
156                // These errors can be triggered by requesting a type whose layout is too large
157                // (requires too may resources to resolve)
158                PRE::TooManyTypeNodes(_, _)
159                | PRE::TooManyTypeParams(_, _)
160                | PRE::TypeParamNesting(_, _) => invalid_params(Error::BadType(type_.clone(), e)),
161
162                // The other errors are a form of internal error.
163                PRE::Bcs(_)
164                | PRE::Store { .. }
165                | PRE::Deserialize(_)
166                | PRE::EmptyPackage(_)
167                | PRE::FunctionNotFound(_, _, _)
168                | PRE::LinkageNotFound(_)
169                | PRE::NoTypeOrigin(_, _, _)
170                | PRE::NotAnIdentifier(_)
171                | PRE::TypeParamOOB(_, _)
172                | PRE::UnexpectedReference
173                | PRE::UnexpectedSigner
174                | PRE::UnexpectedError(_)
175                | PRE::ValueNesting(_) => {
176                    RpcError::from(anyhow!(e).context("Failed to resolve type layout"))
177                }
178            }
179        })
180}

182/// Try to load a dynamic field from `parent_id`, whose name has type `type_` and value `name` (as
183/// BCS bytes). Fetches the `Field<K, V>` object from store.
184async fn load_df(
185    ctx: &Context,
186    parent_id: ObjectID,
187    type_: &TypeTag,
188    value: &[u8],
189) -> Result<Option<Object>, RpcError<Error>> {
190    let id = derive_dynamic_field_id(parent_id, type_, value)
191        .context("Failed to derive dynamic field ID")?;
192
193    Ok(load_live(ctx, id)
194        .await
195        .context("Failed to load dynamic field")?)
196}

198/// Try to load a dynamic object field from `parent_id`, whose name has type `type_` and value
199/// `name` (as BCS bytes). Fetches the object pointed to by the `Field<Wrapper<K>, ID>` object.
200///
201/// This function returns `None`, if the Field object does not exist in the store or does not have
202/// contents.
203async fn load_dof(
204    ctx: &Context,
205    parent_id: ObjectID,
206    type_: &TypeTag,
207    name: &[u8],
208) -> Result<Option<Object>, RpcError<Error>> {
209    let wrapper: TypeTag = DynamicFieldInfo::dynamic_object_field_wrapper(type_.clone()).into();
210    let id = derive_dynamic_field_id(parent_id, &wrapper, name)
211        .context("Failed to derive dynamic object field ID")?;
212
213    let Some(object) = load_live(ctx, id)
214        .await
215        .context("Failed to load dynamic object field")?
216    else {
217        return Ok(None);
218    };
219
220    let Some(move_object) = object.data.try_as_move() else {
221        rpc_bail!("Dynamic field at {id} is not a Move Object");
222    };
223
224    // Peel off the UID and the name from the serialized object. An ObjectID should be left.
225    let value = ObjectID::from_bytes(&move_object.contents()[ObjectID::LENGTH + name.len()..])
226        .context("Failed to extract object ID from dynamic object field")?;
227
228    Ok(load_live(ctx, value)
229        .await
230        .context("Failed to load dynamic field object")?)
231}
