32/// Fetch the latest version of a dynamic field object, identified by its parent ID and name.
33pub(super) async fn dynamic_field_object(
34    ctx: &Context,
35    parent_object_id: ObjectID,
36    name: DynamicFieldName,
37) -> Result<SuiObjectResponse, RpcError<Error>> {
38    let layout = resolve_type(ctx, &name.type_).await?;
39    let bytes = SuiJsonValue::new(name.value)
40        .map_err(|e| invalid_params(Error::BadName(e)))?
41        .to_bcs_bytes(&layout)
42        .map_err(|e| invalid_params(Error::TypeMismatch(name.type_.clone(), e)))?;
43
44    let df = load_df(ctx, parent_object_id, &name.type_, &bytes);
45    let dof = load_dof(ctx, parent_object_id, &name.type_, &bytes);
46    let (df, dof) = try_join!(df, dof)
47        .with_context(|| format!("Failed to fetch dynamic field on {parent_object_id}"))?;
48
49    let Some(object) = df.or(dof) else {
50        return Ok(SuiObjectResponse::new_with_error(
51            SuiObjectResponseError::DynamicFieldNotFound { parent_object_id },
52        ));
53    };
54
55    let options = SuiObjectDataOptions::full_content();
56
57    use RpcError as E;
58    Ok(SuiObjectResponse::new_with_data(
59        objects::response::object_data_with_options(ctx, object, &options)
60            .await
61            .map_err(|e| match e {
62                E::InvalidParams(e) => match e {},
63                E::Timeout(e) => E::Timeout(e),
64                E::InternalError(e) => E::InternalError(e),
65            })?,
66    ))
67}

69/// Fetch the latest version of the object identified by `object_id`, treat it as if it is a
70/// `sui::dynamic_field::Field<K, V>`, and extract the name and value from it.
71pub(super) async fn dynamic_field_info(
72    ctx: &Context,
73    object_id: ObjectID,
74) -> Result<DynamicFieldInfoResponse, RpcError<Error>> {
75    let object = load_live(ctx, object_id)
76        .await
77        .context("Failed to load dynamic field")?
78        .context("Could not find latest content for dynamic field")?;
79
80    let Some(move_object) = object.data.try_as_move() else {
81        rpc_bail!("Dynamic field at {object_id} is not a Move Object");
82    };
83
84    let type_ = move_object.type_().clone().into();
85    let layout = resolve_type(ctx, &type_).await?;
86
87    let field = DFV::FieldVisitor::deserialize(move_object.contents(), &layout)
88        .context("Failed to deserialize dynamic field info")?;
89
90    let type_ = field.kind;
91    let name_type: TypeTag = field.name_layout.into();
92    let bcs_name = BcsName::Base64 {
93        bcs_name: field.name_bytes.to_owned(),
94    };
95
96    let name_value = BoundedVisitor::deserialize_value(field.name_bytes, field.name_layout)
97        .context("Failed to deserialize dynamic field name")?;
98
99    let name = DynamicFieldName {
100        type_: name_type,
101        value: SuiMoveValue::from(name_value).to_json_value(),
102    };
103
104    let value_metadata = field
105        .value_metadata()
106        .context("Failed to extract dynamic field value metadata")?;
107
108    Ok(match value_metadata {
109        DFV::ValueMetadata::DynamicField(object_type) => DynamicFieldInfoResponse {
110            name,
111            bcs_name,
112            type_,
113            object_type: object_type.to_canonical_string(/* with_prefix */ true),
114            object_id: object.id(),
115            version: object.version(),
116            digest: object.digest(),
117        },
118
119        DFV::ValueMetadata::DynamicObjectField(object_id) => {
120            let object = load_live(ctx, object_id)
121                .await
122                .context("Failed to load dynamic object field value")?
123                .context("Could not find latest content for dynamic object field value")?;
124
125            let Some(object_type) = object.data.type_().cloned() else {
126                rpc_bail!("Dynamic object field value at {object_id} is not a Move Object");
127            };
128
129            DynamicFieldInfoResponse {
130                name,
131                bcs_name,
132                type_,
133                object_type: object_type.to_canonical_string(/* with_prefix */ true),
134                object_id: object.id(),
135                version: object.version(),
136                digest: object.digest(),
137            }
138        }
139    })
140}

142/// Resolve the layout for a given type tag, using the package resolver in the context.
143/// Re-interprets errors from the package resolver, categorizing them as RPC user errors or
144/// internal errors.
145async fn resolve_type(ctx: &Context, type_: &TypeTag) -> Result<MoveTypeLayout, RpcError<Error>> {
146    ctx.package_resolver()
147        .type_layout(type_.clone())
148        .await
149        .map_err(|e| {
150            use sui_package_resolver::error::Error as PRE;
151            match &e {
152                // These errors can be triggered by passing a type that doesn't exist for the
153                // dynamic field name.
154                PRE::NotAPackage(_)
155                | PRE::PackageNotFound(_)
156                | PRE::ModuleNotFound(_, _)
157                | PRE::DatatypeNotFound(_, _, _)
158                | PRE::TypeArityMismatch(_, _) => invalid_params(Error::BadType(type_.clone(), e)),
159
160                // These errors can be triggered by requesting a type whose layout is too large
161                // (requires too may resources to resolve)
162                PRE::TooManyTypeNodes(_, _)
163                | PRE::TooManyTypeParams(_, _)
164                | PRE::TypeParamNesting(_, _) => invalid_params(Error::BadType(type_.clone(), e)),
165
166                // The other errors are a form of internal error.
167                PRE::Bcs(_)
168                | PRE::Store { .. }
169                | PRE::Deserialize(_)
170                | PRE::EmptyPackage(_)
171                | PRE::FunctionNotFound(_, _, _)
172                | PRE::LinkageNotFound(_)
173                | PRE::NoTypeOrigin(_, _, _)
174                | PRE::NotAnIdentifier(_)
175                | PRE::TypeParamOOB(_, _)
176                | PRE::UnexpectedReference
177                | PRE::UnexpectedSigner
178                | PRE::UnexpectedError(_)
179                | PRE::ValueNesting(_) => {
180                    RpcError::from(anyhow!(e).context("Failed to resolve type layout"))
181                }
182            }
183        })
184}

186/// Try to load a dynamic field from `parent_id`, whose name has type `type_` and value `name` (as
187/// BCS bytes). Fetches the `Field<K, V>` object from store.
188async fn load_df(
189    ctx: &Context,
190    parent_id: ObjectID,
191    type_: &TypeTag,
192    value: &[u8],
193) -> Result<Option<Object>, RpcError<Error>> {
194    let id = derive_dynamic_field_id(parent_id, type_, value)
195        .context("Failed to derive dynamic field ID")?;
196
197    Ok(load_live(ctx, id)
198        .await
199        .context("Failed to load dynamic field")?)
200}

202/// Try to load a dynamic object field from `parent_id`, whose name has type `type_` and value
203/// `name` (as BCS bytes). Fetches the object pointed to by the `Field<Wrapper<K>, ID>` object.
204///
205/// This function returns `None`, if the Field object does not exist in the store or does not have
206/// contents.
207async fn load_dof(
208    ctx: &Context,
209    parent_id: ObjectID,
210    type_: &TypeTag,
211    name: &[u8],
212) -> Result<Option<Object>, RpcError<Error>> {
213    let wrapper: TypeTag = DynamicFieldInfo::dynamic_object_field_wrapper(type_.clone()).into();
214    let id = derive_dynamic_field_id(parent_id, &wrapper, name)
215        .context("Failed to derive dynamic object field ID")?;
216
217    let Some(object) = load_live(ctx, id)
218        .await
219        .context("Failed to load dynamic object field")?
220    else {
221        return Ok(None);
222    };
223
224    let Some(move_object) = object.data.try_as_move() else {
225        rpc_bail!("Dynamic field at {id} is not a Move Object");
226    };
227
228    // Peel off the UID and the name from the serialized object. An ObjectID should be left.
229    let value = ObjectID::from_bytes(&move_object.contents()[ObjectID::LENGTH + name.len()..])
230        .context("Failed to extract object ID from dynamic object field")?;
231
232    Ok(load_live(ctx, value)
233        .await
234        .context("Failed to load dynamic field object")?)
235}
