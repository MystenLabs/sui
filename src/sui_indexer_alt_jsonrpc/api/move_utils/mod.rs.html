4use jsonrpsee::core::RpcResult;
5use jsonrpsee::proc_macros::rpc;

8use sui_open_rpc_macros::open_rpc;

11use crate::api::rpc_module::RpcModule;
12use crate::context::Context;
13
14mod error;
15mod response;

17#[open_rpc(namespace = "sui", tag = "Move APIs")]
18#[rpc(server, namespace = "sui")]
19trait MoveApi {
20    #[method(name = "getNormalizedMoveFunction")]
21    async fn get_normalized_move_function(
22        &self,
23        package: ObjectID,
24        module_name: String,
25        function_name: String,
26    ) -> RpcResult<SuiMoveNormalizedFunction>;
27}
28
29pub(crate) struct MoveUtils(pub Context);
30
31#[async_trait::async_trait]
32impl MoveApiServer for MoveUtils {
33    async fn get_normalized_move_function(
34        &self,
35        package: ObjectID,
36        module_name: String,
37        function_name: String,
38    ) -> RpcResult<SuiMoveNormalizedFunction> {
39        let Self(ctx) = self;
40        Ok(response::function(ctx, package, &module_name, &function_name).await?)
41    }
42}
43
44impl RpcModule for MoveUtils {
45    fn schema(&self) -> Module {
46        MoveApiOpenRpc::module_doc()
47    }
48
49    fn into_impl(self) -> jsonrpsee::RpcModule<Self> {
50        self.into_rpc()
51    }
52}
