23/// Attempt to translate the given SuiNS `name` to its address, as long as the mapping exists,
24/// and it hasn't expired.
25pub(super) async fn resolved_address(
26    ctx: &Context,
27    name: &str,
28) -> Result<Option<SuiAddress>, RpcError<Error>> {
29    use Error as E;
30
31    let domain: Domain = name
32        .parse()
33        .map_err(|e| invalid_params(E::NameService(e)))?;
34
35    let config = &ctx.config().name_service;
36    let domain_record_id = config.record_field_id(&domain);
37    let parent_record_id = config.record_field_id(&domain.parent());
38
39    let domain_object = load_live(ctx, domain_record_id);
40    let parent_object: OptionFuture<_> = domain
41        .is_subdomain()
42        .then(|| load_live(ctx, parent_record_id))
43        .into();
44
45    // Fetch the current timestamp, the domain record. If the domain being resolved is a
46    // sub-domain, then also fetch the parent record, because its expiry is controlled by its
47    // parent's.
48    let (timestamp_ms, domain_object, parent_object) =
49        join!(latest_timestamp_ms(ctx), domain_object, parent_object);
50
51    let timestamp_ms = timestamp_ms.context("Failed to fetch latest timestamp")?;
52
53    let Some(domain_object) = domain_object.context("Failed to fetch domain record")? else {
54        return Err(invalid_params(E::NotFound(domain.to_string())));
55    };
56
57    let domain_record =
58        NameRecord::try_from(domain_object).context("Failed to deserialize domain record")?;
59
60    // If the domain being fetched is not a leaf node, then check its expiry directly.
61    if !domain_record.is_leaf_record() {
62        return if !domain_record.is_node_expired(timestamp_ms) {
63            Ok(domain_record.target_address)
64        } else {
65            return Err(invalid_params(E::NameService(
66                NameServiceError::NameExpired,
67            )));
68        };
69    }
70
71    // Otherwise its expiry depends on its parent's record: It must exist, and the domain's record
72    // must recognise the fetched parent as its parent, and the parent must not be expired.
73    let Some(parent_object) = parent_object
74        .transpose()
75        .context("Failed to fetch parent record")?
76        .flatten()
77    else {
78        // If the domain object exists but the parent object does not, it could indicate the
79        // sub-domain has expired because the parent has been re-registered.
80        return Err(invalid_params(E::NotFound(domain.parent().to_string())));
81    };
82
83    let parent_record =
84        NameRecord::try_from(parent_object).context("Failed to deserialize parent record")?;
85
86    if parent_record.is_valid_leaf_parent(&domain_record)
87        && !parent_record.is_node_expired(timestamp_ms)
88    {
89        Ok(domain_record.target_address)
90    } else {
91        Err(invalid_params(E::NameService(
92            NameServiceError::NameExpired,
93        )))
94    }
95}

97/// Attempt to translate the given `address` to its SuiNS name, as long as the reverse mapping
98/// exists, and the forward mapping points to the address.
99pub(super) async fn resolved_name(
100    ctx: &Context,
101    address: SuiAddress,
102) -> Result<Option<String>, RpcError<Error>> {
103    let config = &ctx.config().name_service;
104
105    let reverse_record_id = config.reverse_record_field_id(address.as_ref());
106    let Some(reverse_record_object) = load_live(ctx, reverse_record_id)
107        .await
108        .context("Failed to fetch reverse record")?
109    else {
110        return Ok(None);
111    };
112
113    let reverse_record: Field<SuiAddress, Domain> = bcs::from_bytes(
114        reverse_record_object
115            .data
116            .try_as_move()
117            .context("Reverse record not a Move object")?
118            .contents(),
119    )
120    .context("Failed to deserialize reverse record")?;
121
122    // Before returning the domain, check that it is still valid. If forward resolution fails with
123    // a user error, it means the reverse record is no longer valid. Internal errors are unexpected
124    // and should be propagated.
125    //
126    // There is strong on-chain enforcement that the forward and reverse mappings are consistent
127    // with each other, so we don't need to check the forward mapping, if we find one.
128    let domain = reverse_record.value.to_string();
129    match resolved_address(ctx, &domain).await {
130        Ok(Some(_)) => Ok(Some(domain)),
131        Ok(None) | Err(RpcError::InvalidParams(_)) => Ok(None),
132        Err(e) => Err(e).internal_context("Failed to resolve address"),
133    }
134}

136/// Fetch the latest timestamp from the database, based on the watermark for the `obj_info`
137/// pipeline, because we know that the `obj_info` pipeline is being queried as part of address
138/// resolution.
139async fn latest_timestamp_ms(ctx: &Context) -> Result<u64, RpcError<Error>> {
140    use watermarks::dsl as w;
141
142    let mut conn = ctx
143        .pg_reader()
144        .connect()
145        .await
146        .context("Failed to connect to database")?;
147
148    let query = w::watermarks
149        .select(w::timestamp_ms_hi_inclusive)
150        .filter(w::pipeline.eq("obj_info"));
151
152    let timestamp_ms: i64 = conn
153        .first(query)
154        .await
155        .context("Failed to fetch latest timestamp")?;
156
157    Ok(timestamp_ms as u64)
158}
