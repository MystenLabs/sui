30/// Fetch the necessary data from the stores in `ctx` and transform it to build a response for a
31/// the latest version of an object, identified by its ID, according to the response `options`.
32pub(super) async fn live_object(
33    ctx: &Context,
34    object_id: ObjectID,
35    options: &SuiObjectDataOptions,
36) -> Result<SuiObjectResponse, RpcError> {
37    let Some(object) = load_live(ctx, object_id)
38        .await
39        .context("Failed to load latest object")?
40    else {
41        return Ok(SuiObjectResponse::new_with_error(
42            SuiObjectResponseError::NotExists { object_id },
43        ));
44    };
45
46    Ok(SuiObjectResponse::new_with_data(
47        object_data_with_options(ctx, object, options).await?,
48    ))
49}

51/// Fetch the necessary data from the stores in `ctx` and transform it to build a response for a
52/// past object identified by its ID and version, according to the response `options`.
53pub(super) async fn past_object(
54    ctx: &Context,
55    object_id: ObjectID,
56    version: SequenceNumber,
57    options: &SuiObjectDataOptions,
58) -> Result<SuiPastObjectResponse, RpcError> {
59    let Some(object) = ctx
60        .kv_loader()
61        .load_one_object(object_id, version.value())
62        .await
63        .context("Failed to load object from store")?
64    else {
65        return Ok(SuiPastObjectResponse::VersionNotFound(object_id, version));
66    };
67
68    Ok(SuiPastObjectResponse::VersionFound(
69        object_data_with_options(ctx, object, options).await?,
70    ))
71}

73/// Extract a representation of the object according to its response options.
74pub(crate) async fn object_data_with_options(
75    ctx: &Context,
76    object: Object,
77    options: &SuiObjectDataOptions,
78) -> Result<SuiObjectData, RpcError> {
79    let type_ = options.show_type.then(|| ObjectType::from(&object));
80
81    let owner = options.show_owner.then(|| object.owner().clone());
82
83    let previous_transaction = options
84        .show_previous_transaction
85        .then(|| object.previous_transaction);
86
87    let storage_rebate = options.show_storage_rebate.then(|| object.storage_rebate);
88
89    let content: OptionFuture<_> = options
90        .show_content
91        .then(|| object_data::<SuiParsedData>(ctx, &object))
92        .into();
93
94    let bcs: OptionFuture<_> = options
95        .show_bcs
96        .then(|| object_data::<SuiRawData>(ctx, &object))
97        .into();
98
99    let display: OptionFuture<_> = options.show_display.then(|| display(ctx, &object)).into();
100
101    let (content, bcs, display) = join!(content, bcs, display);
102
103    let content = content
104        .transpose()
105        .internal_context("Failed to deserialize object content")?;
106
107    let bcs = bcs
108        .transpose()
109        .internal_context("Failed to deserialize object to BCS")?;
110
111    Ok(SuiObjectData {
112        object_id: object.id(),
113        version: object.version(),
114        digest: object.digest(),
115        type_,
116        owner,
117        previous_transaction,
118        storage_rebate,
119        display,
120        content,
121        bcs,
122    })
123}

125/// Extract the contents of an object, in a format chosen by the `D` type parameter.
126/// This operaton can fail if it's not possible to get the type layout for the object's type.
127async fn object_data<D: SuiData>(ctx: &Context, object: &Object) -> Result<D, RpcError> {
128    Ok(match object.data.clone() {
129        Data::Package(move_package) => D::try_from_package(move_package)?,
130
131        Data::Move(move_object) => {
132            let type_: TypeTag = move_object.type_().clone().into();
133            let MoveTypeLayout::Struct(layout) = ctx
134                .package_resolver()
135                .type_layout(type_.clone())
136                .await
137                .with_context(|| {
138                    format!(
139                        "Failed to resolve type layout for {}",
140                        type_.to_canonical_display(/*with_prefix */ true)
141                    )
142                })?
143            else {
144                rpc_bail!(
145                    "Type {} is not a struct",
146                    type_.to_canonical_display(/*with_prefix */ true)
147                );
148            };
149
150            D::try_from_object(move_object, *layout)?
151        }
152    })
153}

155/// Creates a response containing an object's Display fields. If this operation fails for any
156/// reason, the value is captured in the response's error field, rather than using a `Result`, so
157/// that the failure to generate a Display does not prevent the rest of the object's data from
158/// being returned.
159async fn display(ctx: &Context, object: &Object) -> DisplayFieldsResponse {
160    let fields = match display_fields(ctx, object).await {
161        Ok(fields) => fields,
162        Err(e) => {
163            return DisplayFieldsResponse {
164                data: None,
165                error: Some(SuiObjectResponseError::DisplayError {
166                    error: format!("{e:#}"),
167                }),
168            };
169        }
170    };
171
172    let mut field_values = BTreeMap::new();
173    let mut field_errors = String::new();
174    let mut prefix = "";
175
176    for (name, value) in fields {
177        match value {
178            Ok(value) => {
179                field_values.insert(name, value);
180            }
181            Err(e) => {
182                write!(field_errors, "{prefix}Error for field {name:?}: {e:#}").unwrap();
183                prefix = "; ";
184            }
185        }
186    }
187
188    DisplayFieldsResponse {
189        data: Some(field_values),
190        error: (!field_errors.is_empty()).then_some(SuiObjectResponseError::DisplayError {
191            error: field_errors,
192        }),
193    }
194}

196/// Generate the Display fields for an object by fetching its latest Display format, parsing it,
197/// and extracting values from the object's contents according to expressions in each field's
198/// format string.
199///
200/// This operation can fail if the object is not a Move object, the Display format is not found, or
201/// one of its fields fails to parse as a valid format string. Generating each field can also fail
202/// if a field is nested too deeply, is not present, or has an invalid type for a format string.
203async fn display_fields(
204    ctx: &Context,
205    object: &Object,
206) -> anyhow::Result<BTreeMap<String, anyhow::Result<String>>> {
207    let Some(object) = object.data.try_as_move() else {
208        bail!("Display is only supported for Move objects");
209    };
210
211    let config = &ctx.config().objects;
212    let type_: StructTag = object.type_().clone().into();
213
214    let layout = ctx.package_resolver().type_layout(type_.clone().into());
215    let display = ctx.pg_loader().load_one(DisplayKey(type_.clone()));
216
217    let (layout, display) = join!(layout, display);
218
219    let layout = layout.context("Failed to resolve type layout")?;
220    let Some(stored) = display.context("Failed to load Display format")? else {
221        bail!(
222            "Display format not found for {}",
223            type_.to_canonical_display(/*with_prefix */ true)
224        );
225    };
226
227    let event: DisplayVersionUpdatedEvent =
228        bcs::from_bytes(&stored.display).context("Failed to deserialize Display format")?;
229
230    let format = Format::parse(config.max_display_field_depth, &event.fields)?;
231    format.display(config.max_display_output_size, object.contents(), &layout)
232}
