46/// Fetch the necessary data from the stores in `ctx` and transform it to build a response for a
47/// the latest version of an object, identified by its ID, according to the response `options`.
48pub(super) async fn live_object(
49    ctx: &Context,
50    object_id: ObjectID,
51    options: &SuiObjectDataOptions,
52) -> Result<SuiObjectResponse, RpcError> {
53    let object = if let Some(object) = load_live(ctx, object_id)
54        .await
55        .context("Failed to load latest object")?
56    {
57        object
58    } else if let Some(object) = try_resolve_address_balance_object(ctx, object_id)
59        .await
60        .context("Failed to resolve address balance object")?
61    {
62        object
63    } else {
64        return Ok(SuiObjectResponse::new_with_error(
65            SuiObjectResponseError::NotExists { object_id },
66        ));
67    };
68
69    Ok(SuiObjectResponse::new_with_data(
70        object_data_with_options(ctx, object, options).await?,
71    ))
72}

74/// Fetch the necessary data from the stores in `ctx` and transform it to build a response for a
75/// past object identified by its ID and version, according to the response `options`.
76pub(super) async fn past_object(
77    ctx: &Context,
78    object_id: ObjectID,
79    version: SequenceNumber,
80    options: &SuiObjectDataOptions,
81) -> Result<SuiPastObjectResponse, RpcError> {
82    let Some(object) = ctx
83        .kv_loader()
84        .load_one_object(object_id, version.value())
85        .await
86        .context("Failed to load object from store")?
87    else {
88        return Ok(SuiPastObjectResponse::VersionNotFound(object_id, version));
89    };
90
91    Ok(SuiPastObjectResponse::VersionFound(
92        object_data_with_options(ctx, object, options).await?,
93    ))
94}

96/// Extract a representation of the object according to its response options.
97pub(crate) async fn object_data_with_options(
98    ctx: &Context,
99    object: Object,
100    options: &SuiObjectDataOptions,
101) -> Result<SuiObjectData, RpcError> {
102    let type_ = options.show_type.then(|| ObjectType::from(&object));
103
104    let owner = options.show_owner.then(|| object.owner().clone());
105
106    let previous_transaction = options
107        .show_previous_transaction
108        .then(|| object.previous_transaction);
109
110    let storage_rebate = options.show_storage_rebate.then(|| object.storage_rebate);
111
112    let content: OptionFuture<_> = options
113        .show_content
114        .then(|| object_data::<SuiParsedData>(ctx, &object))
115        .into();
116
117    let bcs: OptionFuture<_> = options
118        .show_bcs
119        .then(|| object_data::<SuiRawData>(ctx, &object))
120        .into();
121
122    let display: OptionFuture<_> = options.show_display.then(|| display(ctx, &object)).into();
123
124    let (content, bcs, display) = join!(content, bcs, display);
125
126    let content = content
127        .transpose()
128        .internal_context("Failed to deserialize object content")?;
129
130    let bcs = bcs
131        .transpose()
132        .internal_context("Failed to deserialize object to BCS")?;
133
134    Ok(SuiObjectData {
135        object_id: object.id(),
136        version: object.version(),
137        digest: object.digest(),
138        type_,
139        owner,
140        previous_transaction,
141        storage_rebate,
142        display,
143        content,
144        bcs,
145    })
146}

148/// Extract the contents of an object, in a format chosen by the `D` type parameter.
149/// This operaton can fail if it's not possible to get the type layout for the object's type.
150async fn object_data<D: SuiData>(ctx: &Context, object: &Object) -> Result<D, RpcError> {
151    Ok(match object.data.clone() {
152        Data::Package(move_package) => D::try_from_package(move_package)?,
153
154        Data::Move(move_object) => {
155            let type_: TypeTag = move_object.type_().clone().into();
156            let MoveTypeLayout::Struct(layout) = ctx
157                .package_resolver()
158                .type_layout(type_.clone())
159                .await
160                .with_context(|| {
161                    format!(
162                        "Failed to resolve type layout for {}",
163                        type_.to_canonical_display(/*with_prefix */ true)
164                    )
165                })?
166            else {
167                rpc_bail!(
168                    "Type {} is not a struct",
169                    type_.to_canonical_display(/*with_prefix */ true)
170                );
171            };
172
173            D::try_from_object(move_object, *layout)?
174        }
175    })
176}

178/// Creates a response containing an object's Display fields. If this operation fails for any
179/// reason, the value is captured in the response's error field, rather than using a `Result`, so
180/// that the failure to generate a Display does not prevent the rest of the object's data from
181/// being returned.
182async fn display(ctx: &Context, object: &Object) -> DisplayFieldsResponse {
183    let fields = match display_fields(ctx, object).await {
184        Ok(fields) => fields,
185        Err(e) => {
186            return DisplayFieldsResponse {
187                data: None,
188                error: Some(SuiObjectResponseError::DisplayError {
189                    error: format!("{e:#}"),
190                }),
191            };
192        }
193    };
194
195    let mut field_values = BTreeMap::new();
196    let mut field_errors = String::new();
197    let mut prefix = "";
198
199    for (name, value) in fields {
200        match value {
201            Ok(value) => {
202                field_values.insert(name, value);
203            }
204            Err(e) => {
205                write!(field_errors, "{prefix}Error for field {name:?}: {e:#}").unwrap();
206                prefix = "; ";
207            }
208        }
209    }
210
211    DisplayFieldsResponse {
212        data: Some(field_values),
213        error: (!field_errors.is_empty()).then_some(SuiObjectResponseError::DisplayError {
214            error: field_errors,
215        }),
216    }
217}

219/// Generate the Display fields for an object by fetching its latest Display format, parsing it,
220/// and extracting values from the object's contents according to expressions in each field's
221/// format string.
222///
223/// This operation can fail if the object is not a Move object, the Display format is not found, or
224/// one of its fields fails to parse as a valid format string. Generating each field can also fail
225/// if a field is nested too deeply, is not present, or has an invalid type for a format string.
226async fn display_fields(
227    ctx: &Context,
228    object: &Object,
229) -> anyhow::Result<BTreeMap<String, anyhow::Result<Json>>> {
230    let Some(object) = object.data.try_as_move() else {
231        bail!("Display is only supported for Move objects");
232    };
233
234    let config = &ctx.config().objects;
235    let type_: StructTag = object.type_().clone().into();
236
237    let layout = ctx.package_resolver().type_layout(type_.clone().into());
238    let display_v1 = display_v1(ctx, &type_);
239    let display_v2 = display_v2(ctx, &type_);
240
241    let (layout, display_v1, display_v2) = join!(layout, display_v1, display_v2);
242
243    let layout = layout.context("Failed to resolve type layout")?;
244
245    if let Some(display_v2) = display_v2? {
246        let store = DisplayStore::new(ctx);
247        let root = sui_display::v2::OwnedSlice::new(layout, object.contents().to_owned());
248        let interpreter = sui_display::v2::Interpreter::new(root, store);
249        let fields = sui_display::v2::Display::parse(config.display(), display_v2.fields())?
250            .display(
251                ctx.config().package_resolver.max_move_value_depth,
252                config.max_display_output_size,
253                &interpreter,
254            )
255            .await?;
256
257        Ok(fields
258            .into_iter()
259            .map(|(field, value)| (field, value.map_err(Into::into)))
260            .collect())
261    } else if let Some(display_v1) = display_v1? {
262        let format = Format::parse(config.max_display_field_depth, &display_v1.fields)?;
263        Ok(format
264            .display(config.max_display_output_size, object.contents(), &layout)?
265            .into_iter()
266            .map(|(field, value)| (field, value.map(Json::String)))
267            .collect())
268    } else {
269        bail!(
270            "Display format not found for {}",
271            type_.to_canonical_display(/*with_prefix */ true)
272        );
273    }
274}

276/// Try to load the V1 Display format for this type.
277async fn display_v1(
278    ctx: &Context,
279    type_: &StructTag,
280) -> anyhow::Result<Option<DisplayVersionUpdatedEvent>> {
281    let Some(stored) = ctx
282        .pg_loader()
283        .load_one(DisplayKey(type_.clone()))
284        .await
285        .context("Failed to load Display v1")?
286    else {
287        return Ok(None);
288    };
289
290    let event: DisplayVersionUpdatedEvent =
291        bcs::from_bytes(&stored.display).context("Failed to deserialize Display v1")?;
292
293    Ok(Some(event))
294}

296/// Try to load the V2 Display format for this type.
297async fn display_v2(
298    ctx: &Context,
299    type_: &StructTag,
300) -> anyhow::Result<Option<display_registry::Display>> {
301    let object_id = display_registry::display_object_id(type_.clone().into())
302        .context("Failed to derive Display v2 object ID")?;
303
304    let Some(object) = load_live(ctx, object_id)
305        .await
306        .context("Failed to fetch Display v2 object")?
307    else {
308        return Ok(None);
309    };
310
311    let Some(move_object) = object.data.try_as_move() else {
312        return Ok(None);
313    };
314
315    let display = bcs::from_bytes(move_object.contents())
316        .context("Failed to deserialize Display v2 object")?;
317    Ok(Some(display))
318}

320impl<'c> DisplayStore<'c> {
321    fn new(ctx: &'c Context) -> Self {
322        Self { ctx }
323    }
324}

326#[async_trait]
327impl sui_display::v2::Store for DisplayStore<'_> {
328    async fn latest(
329        &self,
330        id: move_core_types::account_address::AccountAddress,
331    ) -> anyhow::Result<Option<(move_core_types::annotated_value::MoveTypeLayout, Vec<u8>)>> {
332        let Some(object) = load_live(self.ctx, id.into())
333            .await
334            .context("Failed to fetch object")?
335        else {
336            return Ok(None);
337        };
338
339        let Some(move_object) = object.data.try_as_move() else {
340            return Ok(None);
341        };
342
343        let type_: TypeTag = move_object.type_().clone().into();
344        let layout = self
345            .ctx
346            .package_resolver()
347            .type_layout(type_)
348            .await
349            .context("Failed to resolve type layout")?;
350
351        Ok(Some((layout, move_object.contents().to_owned())))
352    }
