38/// Fetch the necessary data from the stores in `ctx` and transform it to build a response for a
39/// the latest version of an object, identified by its ID, according to the response `options`.
40pub(super) async fn live_object(
41    ctx: &Context,
42    object_id: ObjectID,
43    options: &SuiObjectDataOptions,
44) -> Result<SuiObjectResponse, RpcError> {
45    let Some(object) = load_live(ctx, object_id)
46        .await
47        .context("Failed to load latest object")?
48    else {
49        return Ok(SuiObjectResponse::new_with_error(
50            SuiObjectResponseError::NotExists { object_id },
51        ));
52    };
53
54    Ok(SuiObjectResponse::new_with_data(
55        object_data_with_options(ctx, object, options).await?,
56    ))
57}

59/// Fetch the necessary data from the stores in `ctx` and transform it to build a response for a
60/// past object identified by its ID and version, according to the response `options`.
61pub(super) async fn past_object(
62    ctx: &Context,
63    object_id: ObjectID,
64    version: SequenceNumber,
65    options: &SuiObjectDataOptions,
66) -> Result<SuiPastObjectResponse, RpcError> {
67    let Some(object) = ctx
68        .kv_loader()
69        .load_one_object(object_id, version.value())
70        .await
71        .context("Failed to load object from store")?
72    else {
73        return Ok(SuiPastObjectResponse::VersionNotFound(object_id, version));
74    };
75
76    Ok(SuiPastObjectResponse::VersionFound(
77        object_data_with_options(ctx, object, options).await?,
78    ))
79}

81/// Extract a representation of the object according to its response options.
82pub(crate) async fn object_data_with_options(
83    ctx: &Context,
84    object: Object,
85    options: &SuiObjectDataOptions,
86) -> Result<SuiObjectData, RpcError> {
87    let type_ = options.show_type.then(|| ObjectType::from(&object));
88
89    let owner = options.show_owner.then(|| object.owner().clone());
90
91    let previous_transaction = options
92        .show_previous_transaction
93        .then(|| object.previous_transaction);
94
95    let storage_rebate = options.show_storage_rebate.then(|| object.storage_rebate);
96
97    let content: OptionFuture<_> = options
98        .show_content
99        .then(|| object_data::<SuiParsedData>(ctx, &object))
100        .into();
101
102    let bcs: OptionFuture<_> = options
103        .show_bcs
104        .then(|| object_data::<SuiRawData>(ctx, &object))
105        .into();
106
107    let display: OptionFuture<_> = options.show_display.then(|| display(ctx, &object)).into();
108
109    let (content, bcs, display) = join!(content, bcs, display);
110
111    let content = content
112        .transpose()
113        .internal_context("Failed to deserialize object content")?;
114
115    let bcs = bcs
116        .transpose()
117        .internal_context("Failed to deserialize object to BCS")?;
118
119    Ok(SuiObjectData {
120        object_id: object.id(),
121        version: object.version(),
122        digest: object.digest(),
123        type_,
124        owner,
125        previous_transaction,
126        storage_rebate,
127        display,
128        content,
129        bcs,
130    })
131}

133/// Extract the contents of an object, in a format chosen by the `D` type parameter.
134/// This operaton can fail if it's not possible to get the type layout for the object's type.
135async fn object_data<D: SuiData>(ctx: &Context, object: &Object) -> Result<D, RpcError> {
136    Ok(match object.data.clone() {
137        Data::Package(move_package) => D::try_from_package(move_package)?,
138
139        Data::Move(move_object) => {
140            let type_: TypeTag = move_object.type_().clone().into();
141            let MoveTypeLayout::Struct(layout) = ctx
142                .package_resolver()
143                .type_layout(type_.clone())
144                .await
145                .with_context(|| {
146                    format!(
147                        "Failed to resolve type layout for {}",
148                        type_.to_canonical_display(/*with_prefix */ true)
149                    )
150                })?
151            else {
152                rpc_bail!(
153                    "Type {} is not a struct",
154                    type_.to_canonical_display(/*with_prefix */ true)
155                );
156            };
157
158            D::try_from_object(move_object, *layout)?
159        }
160    })
161}

163/// Creates a response containing an object's Display fields. If this operation fails for any
164/// reason, the value is captured in the response's error field, rather than using a `Result`, so
165/// that the failure to generate a Display does not prevent the rest of the object's data from
166/// being returned.
167async fn display(ctx: &Context, object: &Object) -> DisplayFieldsResponse {
168    let fields = match display_fields(ctx, object).await {
169        Ok(fields) => fields,
170        Err(e) => {
171            return DisplayFieldsResponse {
172                data: None,
173                error: Some(SuiObjectResponseError::DisplayError {
174                    error: format!("{e:#}"),
175                }),
176            };
177        }
178    };
179
180    let mut field_values = BTreeMap::new();
181    let mut field_errors = String::new();
182    let mut prefix = "";
183
184    for (name, value) in fields {
185        match value {
186            Ok(value) => {
187                field_values.insert(name, value);
188            }
189            Err(e) => {
190                write!(field_errors, "{prefix}Error for field {name:?}: {e:#}").unwrap();
191                prefix = "; ";
192            }
193        }
194    }
195
196    DisplayFieldsResponse {
197        data: Some(field_values),
198        error: (!field_errors.is_empty()).then_some(SuiObjectResponseError::DisplayError {
199            error: field_errors,
200        }),
201    }
202}

204/// Generate the Display fields for an object by fetching its latest Display format, parsing it,
205/// and extracting values from the object's contents according to expressions in each field's
206/// format string.
207///
208/// This operation can fail if the object is not a Move object, the Display format is not found, or
209/// one of its fields fails to parse as a valid format string. Generating each field can also fail
210/// if a field is nested too deeply, is not present, or has an invalid type for a format string.
211async fn display_fields(
212    ctx: &Context,
213    object: &Object,
214) -> anyhow::Result<BTreeMap<String, anyhow::Result<String>>> {
215    let Some(object) = object.data.try_as_move() else {
216        bail!("Display is only supported for Move objects");
217    };
218
219    let config = &ctx.config().objects;
220    let type_: StructTag = object.type_().clone().into();
221
222    let layout = ctx.package_resolver().type_layout(type_.clone().into());
223    let display = ctx.pg_loader().load_one(DisplayKey(type_.clone()));
224
225    let (layout, display) = join!(layout, display);
226
227    let layout = layout.context("Failed to resolve type layout")?;
228    let Some(stored) = display.context("Failed to load Display format")? else {
229        bail!(
230            "Display format not found for {}",
231            type_.to_canonical_display(/*with_prefix */ true)
232        );
233    };
234
235    let event: DisplayVersionUpdatedEvent =
236        bcs::from_bytes(&stored.display).context("Failed to deserialize Display format")?;
237
238    let format = Format::parse(config.max_display_field_depth, &event.fields)?;
239    format.display(config.max_display_output_size, object.contents(), &layout)
240}
