29#[open_rpc(namespace = "sui", tag = "Protocol API")]
30#[rpc(server, namespace = "sui")]
31trait ProtocolApi {
32    /// Return the protocol config table for the given version number. If the version number is not
33    /// specified, the protocol config for the latest indexed epoch is returned.
34    #[method(name = "getProtocolConfig")]
35    async fn get_protocol_config(
36        &self,
37        /// An optional protocol version specifier. If omitted, the protocol config for the latest indexed epoch will be returned.
38        version: Option<BigInt<u64>>,
39    ) -> RpcResult<ProtocolConfigResponse>;
40}
41
42pub(crate) struct Protocol(pub Context);
43
44#[derive(Debug, thiserror::Error)]
45pub enum Error {
46    #[error("Protocol version {0} not found")]
47    ProtocolVersionNotFound(u64),
48}
49
50#[async_trait::async_trait]
51impl ProtocolApiServer for Protocol {
52    async fn get_protocol_config(
53        &self,
54        version: Option<BigInt<u64>>,
55    ) -> RpcResult<ProtocolConfigResponse> {
56        Ok(protocol_config_response(&self.0, version).await?)
57    }
58}

67    }

70/// Load data and generate response for `getProtocolConfig`.
71///
72/// The response is built from the `kv_protocol_configs` and `kv_feature_flags` tables (which the
73/// indexer materializes at each epoch boundary), rather than from this binary's own
74/// `ProtocolConfig`, so that the RPC keeps serving correct values when the chain's protocol version
75/// is newer than the binary.
76///
77/// The tables store values stringified, which costs some fidelity compared to the legacy response:
78/// only protocol versions the indexer has seen can be served; `attributes` reports every integer as
79/// `u64` (declared widths are not preserved); and in `configs`, all numbers are JSON strings
80/// (legacy rendered u16/u32 fields as JSON numbers) and non-scalar values are absent.
81async fn protocol_config_response(
82    ctx: &Context,
83    version: Option<BigInt<u64>>,
84) -> Result<ProtocolConfigResponse, RpcError<Error>> {
85    use kv_feature_flags::dsl as f;
86    use kv_protocol_configs::dsl as p;
87
88    let version = match version {
89        Some(version) => *version,
90        None => latest_protocol_version(ctx).await?,
91    };
92
93    let mut conn = ctx
94        .pg_reader()
95        .connect()
96        .await
97        .context("Failed to connect to the database")?;
98
99    let stored_configs: Vec<StoredProtocolConfig> = conn
100        .results(p::kv_protocol_configs.filter(p::protocol_version.eq(version as i64)))
101        .await
102        .context("Failed to fetch protocol configs")?;
103
104    let stored_flags: Vec<StoredFeatureFlag> = conn
105        .results(f::kv_feature_flags.filter(f::protocol_version.eq(version as i64)))
106        .await
107        .context("Failed to fetch feature flags")?;
108
109    if stored_configs.is_empty() && stored_flags.is_empty() {
110        return Err(invalid_params(Error::ProtocolVersionNotFound(version)));
111    }
112
113    let feature_flags: BTreeMap<String, bool> = stored_flags
114        .into_iter()
115        .map(|flag| (flag.flag_name, flag.flag_value))
116        .collect();
117
118    // `configs` is the complete view of the protocol config: attributes that are set at this
119    // version, merged with the feature flags.
120    let mut attributes = BTreeMap::new();
121    let mut configs = BTreeMap::new();
122    for config in stored_configs {
123        if let Some(value) = &config.config_value {
124            configs.insert(config.config_name.clone(), raw_config_to_json(value));
125        }
126        let attribute = config.config_value.as_deref().and_then(parse_attribute);
127        attributes.insert(config.config_name, attribute);
128    }
129
130    for (flag, value) in &feature_flags {
131        configs.insert(flag.clone(), Value::Bool(*value));
132    }
133
134    Ok(ProtocolConfigResponse {
135        // Min and max supported versions are properties of a binary -- these are the versions this
136        // RPC's own software knows about, which may trail the version being served.
137        min_supported_protocol_version: ProtocolVersion::MIN,
138        max_supported_protocol_version: ProtocolVersion::MAX,
139        protocol_version: version.into(),
140        feature_flags,
141        attributes,
142        configs,
143    })
144}

146/// Fetch the protocol version of the latest epoch from the database.
147async fn latest_protocol_version(ctx: &Context) -> Result<u64, RpcError<Error>> {
148    use kv_epoch_starts::dsl as e;
149
150    let mut conn = ctx
151        .pg_reader()
152        .connect()
153        .await
154        .context("Failed to connect to the database")?;
155
156    let version: i64 = conn
157        .first(
158            e::kv_epoch_starts
159                .select(e::protocol_version)
160                .order(e::epoch.desc()),
161        )
162        .await
163        .context("Failed to fetch the latest protocol version")?;
164
165    Ok(version as u64)
166}

168/// Recover a typed attribute value from its stored string representation. The database does not
169/// preserve integer widths, so all integer values parse as `u64`.
170fn parse_attribute(value: &str) -> Option<SuiProtocolConfigValue> {
171    match value {
172        "true" => Some(SuiProtocolConfigValue::Bool(true)),
173        "false" => Some(SuiProtocolConfigValue::Bool(false)),
174        _ => value.parse().ok().map(SuiProtocolConfigValue::U64),
175    }
176}
177
178/// Convert a stored config value to its JSON representation: booleans as booleans, everything
179/// else as a string. Numbers are uniformly strings (rather than following the legacy convention
180/// of rendering u16/u32 as JSON numbers) because the stored value does not carry its declared
181/// width, and inferring it from the magnitude would make a config's JSON type change when its
182/// value grows.
183fn raw_config_to_json(value: &str) -> Value {
184    match value {
185        "true" => Value::Bool(true),
186        "false" => Value::Bool(false),
187        _ => Value::String(value.to_owned()),
188    }
