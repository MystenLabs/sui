57pub(crate) struct SuiTransactionBlockResponseQuery {
58    /// If None, no filter will be applied.
59    pub filter: Option<TransactionFilter>,
60    /// Configures which fields to include in the response, by default only digest is included.
61    pub options: Option<SuiTransactionBlockResponseOptions>,

64#[serde_as]
65#[derive(Clone, Debug, JsonSchema, Serialize, Deserialize)]
66pub(crate) enum TransactionFilter {
67    /// Query by checkpoint.
68    Checkpoint(
69        #[schemars(with = "BigInt<u64>")]
70        #[serde_as(as = "Readable<BigInt<u64>, _>")]
71        CheckpointSequenceNumber,
72    ),
73    /// Query by move function.
74    MoveFunction {
75        package: ObjectID,
76        module: Option<String>,
77        function: Option<String>,
78    },
79    /// Query for transactions that touch this object.
80    AffectedObject(ObjectID),
81    /// Query by sender address.
82    FromAddress(SuiAddress),
83    /// Query by sender and recipient address.
84    FromAndToAddress { from: SuiAddress, to: SuiAddress },
85    /// Query transactions that have a given address as sender or recipient.
86    FromOrToAddress { addr: SuiAddress },
87    /// Query by recipient address. On this RPC, this is an alias for `FromOrToAddress`.
88    ToAddress(SuiAddress),
89}

94/// Fetch the digests for a page of transactions that satisfy the given `filter` and pagination
95/// parameters. Returns the digests and a cursor pointing to the last result (if there are any
96/// results).
97pub(super) async fn transactions(
98    ctx: &Context,
99    filter: &Option<TransactionFilter>,
100    cursor: Option<String>,
101    limit: Option<usize>,
102    descending_order: Option<bool>,
103) -> Result<Digests, RpcError<Error>> {
104    let config = &ctx.config().transactions;
105    let page: Page<Cursor> = Page::from_params(
106        config.default_page_size,
107        config.max_page_size,
108        cursor,
109        limit,
110        descending_order,
111    )?;
112
113    use TransactionFilter as F;
114    match filter {
115        None => all_transactions(ctx, &page).await,
116
117        Some(F::Checkpoint(seq)) => by_checkpoint(ctx, &page, *seq).await,
118
119        Some(F::MoveFunction {
120            package,
121            module,
122            function,
123        }) => tx_calls(ctx, &page, package, module.as_ref(), function.as_ref()).await,
124
125        Some(F::AffectedObject(object)) => tx_affected_objects(ctx, &page, *object).await,
126
127        Some(F::FromAddress(from)) => tx_affected_addresses(ctx, &page, Some(*from), *from).await,
128
129        Some(F::FromAndToAddress { from, to }) => {
130            tx_affected_addresses(ctx, &page, Some(*from), *to).await
131        }
132
133        Some(F::FromOrToAddress { addr }) => tx_affected_addresses(ctx, &page, None, *addr).await,
134
135        Some(F::ToAddress(addr)) => tx_affected_addresses(ctx, &page, None, *addr).await,
136    }
137}

139/// Fetch a page of transaction digests without filtering them.
140async fn all_transactions(ctx: &Context, page: &Page<Cursor>) -> Result<Digests, RpcError<Error>> {
141    use tx_digests::dsl as d;
142
143    let query = d::tx_digests
144        .select((d::tx_sequence_number, d::tx_digest))
145        .into_boxed();
146
147    let results: Vec<(i64, Vec<u8>)> = ctx
148        .pg_reader()
149        .connect()
150        .await
151        .context("Failed to connect to the database")?
152        .results(paginate(page, "tx_digests", d::tx_sequence_number, query))
153        .await
154        .context("Failed to fetch transaction sequence numbers")?;
155
156    from_digests(page.limit, results)
157}

159/// Fetch a page of transaction digests from the given `checkpoint` (by sequence number).
160async fn by_checkpoint(
161    ctx: &Context,
162    page: &Page<Cursor>,
163    checkpoint: u64,
164) -> Result<Digests, RpcError<Error>> {
165    let Some((summary, contents, _)) = ctx
166        .kv_loader()
167        .load_one_checkpoint(checkpoint)
168        .await
169        .context("Failed to load checkpoint")?
170    else {
171        return Ok(PageResponse::empty());
172    };
173
174    // Transaction sequence number bounds from the checkpoint
175    let cp_hi = summary.network_total_transactions;
176    let cp_lo = summary.network_total_transactions - contents.inner().len() as u64;
177
178    let Page {
179        cursor,
180        limit,
181        descending,
182    } = page;
183
184    // Transaction sequence number bounds from the page
185    let pg_lo: u64;
186    let pg_hi: u64;
187    if *descending {
188        pg_hi = cursor.as_ref().map(|c| c.0).map_or(cp_hi, |c| c.min(cp_hi));
189        pg_lo = pg_hi.saturating_sub(1 + *limit as u64).max(cp_lo);
190    } else {
191        pg_lo = cursor
192            .as_ref()
193            .map(|c| c.0.saturating_add(1))
194            .map_or(cp_lo, |c| c.max(cp_lo));
195
196        pg_hi = pg_lo.saturating_add(1 + *limit as u64).min(cp_hi);
197    }
198
199    let digests = contents.inner();
200    let mut results = Vec::with_capacity(pg_hi.saturating_sub(pg_lo) as usize);
201    for tx in pg_lo..pg_hi {
202        let ix = (tx - cp_lo) as usize;
203        let digest = digests
204            .get_digests(ix)
205            .context("Transaction out of bounds in checkpoint")?
206            .transaction
207            .inner()
208            .to_vec();
209
210        results.push((tx as i64, digest));
211    }
212
213    if *descending {
214        results.reverse();
215    }
216
217    from_digests(*limit, results)
218}

220/// Fetch a page of transaction digests that called the described function(s). Functions can be
221/// selected by just their package, their module, or their fully-qualified name. It is an error to
222/// supply a package and function, but no module.
223async fn tx_calls(
224    ctx: &Context,
225    page: &Page<Cursor>,
226    package: &ObjectID,
227    module: Option<&String>,
228    function: Option<&String>,
229) -> Result<Digests, RpcError<Error>> {
230    use tx_calls::dsl as c;
231
232    if let (None, Some(function)) = (module, function) {
233        return Err(invalid_params(Error::MissingModule {
234            function: function.clone(),
235        }));
236    }
237
238    let mut query = c::tx_calls
239        .select(c::tx_sequence_number)
240        .filter(c::package.eq(package.as_slice()))
241        .into_boxed();
242
243    if let Some(module) = module {
244        query = query.filter(c::module.eq(module.as_str()));
245    }
246
247    if let Some(function) = function {
248        query = query.filter(c::function.eq(function.as_str()));
249    }
250
251    let results: Vec<i64> = ctx
252        .pg_reader()
253        .connect()
254        .await
255        .context("Failed to connect to the database")?
256        .results(paginate(page, "tx_calls", c::tx_sequence_number, query))
257        .await
258        .context("Failed to fetch transaction sequence numbers")?;
259
260    from_sequence_numbers(ctx, page.limit, results).await
261}

263/// Fetch a page of transaction digests that touched `object` (created it, modified it, deleted it
264/// or wrapped it).
265async fn tx_affected_objects(
266    ctx: &Context,
267    page: &Page<Cursor>,
268    object: ObjectID,
269) -> Result<Digests, RpcError<Error>> {
270    use tx_affected_objects::dsl as o;
271
272    let query = o::tx_affected_objects
273        .select(o::tx_sequence_number)
274        .filter(o::affected.eq(object.as_slice()))
275        .into_boxed();
276
277    let results: Vec<i64> = ctx
278        .pg_reader()
279        .connect()
280        .await
281        .context("Failed to connect to the database")?
282        .results(paginate(
283            page,
284            "tx_affected_objects",
285            o::tx_sequence_number,
286            query,
287        ))
288        .await
289        .context("Failed to fetch transaction sequence numbers")?;
290
291    from_sequence_numbers(ctx, page.limit, results).await
292}

294/// Fetch a page of transaction digests that touched the provided addresses (`from` and `to`).
295///
296/// - If both are supplied, then returns transactions that were sent by `from` and affected `to`.
297/// - If `from == to` this is equivalent to filtering down to the transactions that were sent by `from`.
298/// - If `from` is not supplied, this finds the transactions that `to` was affected by in some way
299///   (either it is the sender, or it is the recipient of one of the output objects).
300async fn tx_affected_addresses(
301    ctx: &Context,
302    page: &Page<Cursor>,
303    from: Option<SuiAddress>,
304    to: SuiAddress,
305) -> Result<Digests, RpcError<Error>> {
306    use tx_affected_addresses::dsl as a;
307
308    let mut query = a::tx_affected_addresses
309        .select(a::tx_sequence_number)
310        .filter(a::affected.eq(to.to_inner()))
311        .into_boxed();
312
313    if let Some(from) = from {
314        query = query.filter(a::sender.eq(from.to_inner()));
315    }
316
317    let results: Vec<i64> = ctx
318        .pg_reader()
319        .connect()
320        .await
321        .context("Failed to connect to the database")?
322        .results(paginate(
323            page,
324            "tx_affected_addresses",
325            a::tx_sequence_number,
326            query,
327        ))
328        .await
329        .context("Failed to fetch transaction sequence numbers")?;
330
331    from_sequence_numbers(ctx, page.limit, results).await
332}

334/// Modify `query` to be paginated according to `page`, using `tx_sequence_number` as the column
335/// containing the sequence number. The query is also modified to limit results returned by the
336/// reader low-watermark from the `watermarks` table (using the `cp_sequence_numbers` table to
337/// translate a checkpoint bound into a transaction sequence number bound). This helps the database
338/// avoid scanning dead tuples due to pruning.
339///
340/// The query fetches one more element than the limit, to determine if there is a next page.
341fn paginate<'q, TX, ST, QS>(
342    page: &Page<Cursor>,
343    pipeline: &'static str,
344    tx_sequence_number: TX,
345    mut query: BoxedSelectStatement<'q, ST, FromClause<QS>, Pg>,
346) -> BoxedSelectStatement<'q, ST, FromClause<QS>, Pg>
347where
348    QS: QuerySource,
349    TX: Copy + Send + Sync + 'q,
350    TX: ValidGrouping<()> + QueryFragment<Pg>,
351    TX: Column<Table = QS> + AppearsOnTable<QS>,
352    TX: ExpressionMethods + Expression<SqlType = SqlBigInt>,
353    TX::IsAggregate: MixedAggregates<Never, Output = No>,
354{
355    query = query.filter(tx_sequence_number.ge(sql!(as SqlBigInt,
356        "COALESCE(
357            (
358                SELECT
359                    MAX(tx_lo)
360                FROM
361                    watermarks w
362                INNER JOIN
363                    cp_sequence_numbers c
364                ON
365                    w.reader_lo = c.cp_sequence_number
366                WHERE
367                    w.pipeline IN ({Text}, 'tx_digests')
368            ),
369            0
370        )",
371        pipeline,
372    )));
373
374    if let Some(JsonCursor(tx)) = page.cursor {
375        if page.descending {
376            query = query.filter(tx_sequence_number.lt(tx as i64));
377        } else {
378            query = query.filter(tx_sequence_number.gt(tx as i64));
379        }
380    }
381
382    if page.descending {
383        query = query.order(tx_sequence_number.desc());
384    } else {
385        query = query.order(tx_sequence_number.asc());
386    }
387
388    query.limit(page.limit + 1)
389}

391/// Convert a list of raw transaction sequence numbers from the database into a page of parsed
392/// transaction digests, ready to be loaded. This requires loading digests from the data loader.
393async fn from_sequence_numbers(
394    ctx: &Context,
395    limit: i64,
396    mut rows: Vec<i64>,
397) -> Result<Digests, RpcError<Error>> {
398    let has_next_page = rows.len() > limit as usize;
399    if has_next_page {
400        rows.truncate(limit as usize);
401    }
402
403    let next_cursor = rows
404        .last()
405        .map(|last| JsonCursor(*last).encode())
406        .transpose()
407        .context("Failed to encode next cursor")?;
408
409    // Get digests from the table and retry any of the digests that are not found.
410    let config = &ctx.config().transactions;
411    let max_retries = config.tx_retry_count;
412    let mut retry_interval =
413        tokio::time::interval(Duration::from_millis(config.tx_retry_interval_ms));
414    let mut keys: Vec<_> = rows.iter().map(|&seq| TxDigestKey(seq as u64)).collect();
415    let mut digests: HashMap<TxDigestKey, StoredTxDigest> = HashMap::new();
416    let mut retries = 0;
417    for _ in 0..=max_retries {
418        retry_interval.tick().await;
419
420        digests.extend(
421            ctx.pg_loader()
422                .load_many(keys.clone())
423                .await
424                .context("Failed to load transaction digests")?,
425        );
426
427        // Only retry the keys that are not found.
428        keys.retain(|key| !digests.contains_key(key));
429        if keys.is_empty() {
430            break;
431        }
432        retries += 1;
433        ctx.metrics()
434            .read_retries
435            .with_label_values(&["tx_digest"])
436            .inc();
437    }
438
439    ctx.metrics()
440        .read_retries_per_request
441        .with_label_values(&["tx_digest"])
442        .observe(retries as f64);
443
444    let mut data = Vec::with_capacity(rows.len());
445    for seq in rows {
446        let bytes = digests
447            .get(&TxDigestKey(seq as u64))
448            .with_context(|| format!("Missing transaction digest for transaction {seq}"))?
449            .tx_digest
450            .as_slice();
451
452        let digest = TransactionDigest::try_from(bytes)
453            .context("Failed to deserialize transaction digests")?;
454
455        data.push(digest);
456    }
457
458    Ok(PageResponse {
459        data,
460        next_cursor,
461        has_next_page,
462    })
463}

465/// Convert a list of raw transaction sequence numbers and digests from the database into a page of
466/// parsed transaction digests, ready to be loaded.
467fn from_digests(limit: i64, mut rows: Vec<(i64, Vec<u8>)>) -> Result<Digests, RpcError<Error>> {
468    let has_next_page = rows.len() > limit as usize;
469    if has_next_page {
470        rows.truncate(limit as usize);
471    }
472
473    let data = rows
474        .iter()
475        .map(|(_, digest)| TransactionDigest::try_from(digest.as_slice()))
476        .collect::<Result<Vec<TransactionDigest>, _>>()
477        .context("Failed to deserialize transaction digests")?;
478
479    let next_cursor = rows
480        .last()
481        .map(|(last, _)| JsonCursor(*last).encode())
482        .transpose()
483        .context("Failed to encode next cursor")?;
484
485    Ok(PageResponse {
486        data,
487        next_cursor,
488        has_next_page,
489    })
