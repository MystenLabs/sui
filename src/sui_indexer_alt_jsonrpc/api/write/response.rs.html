43pub(super) async fn transaction(
44    ctx: &Context,
45    tx_data: TransactionData,
46    tx_signatures: Vec<GenericSignature>,
47    executed_tx: &proto::ExecutedTransaction,
48    options: &SuiTransactionBlockResponseOptions,
49) -> Result<SuiTransactionBlockResponse, RpcError<Error>> {
50    let tx_digest = tx_data.digest();
51    let mut result = SuiTransactionBlockResponse::new(tx_digest);
52    result.checkpoint = executed_tx.checkpoint;
53    result.timestamp_ms = executed_tx
54        .timestamp
55        .and_then(|ts| sui_rpc::proto::proto_to_timestamp_ms(ts).ok());
56
57    if options.show_input {
58        result.transaction = Some(input(ctx, tx_data.clone(), tx_signatures).await?);
59    }
60
61    if options.show_raw_input {
62        result.raw_transaction = raw_input(&tx_data)?;
63    }
64
65    if options.show_raw_effects {
66        result.raw_effects = raw_effects(executed_tx)?;
67    }
68
69    if options.show_effects || options.show_object_changes {
70        let effects = deserialize_effects(executed_tx)?;
71
72        if options.show_effects {
73            result.effects = Some(effects_response(&effects)?);
74        }
75
76        if options.show_object_changes {
77            result.object_changes = Some(object_changes(tx_data.sender(), &effects, executed_tx)?);
78        }
79    }
80
81    if options.show_events {
82        result.events = Some(events(ctx, tx_digest, executed_tx).await?);
83    }
84
85    if options.show_balance_changes {
86        result.balance_changes = Some(balance_changes(executed_tx)?);
87    }
88
89    Ok(result)
90}

92pub(super) async fn dry_run(
93    ctx: &Context,
94    tx_data: TransactionData,
95    executed_tx: &proto::ExecutedTransaction,
96    suggested_gas_price: Option<u64>,
97) -> Result<DryRunTransactionBlockResponse, RpcError<Error>> {
98    let effects = deserialize_effects(executed_tx)?;
99    let tx_digest = tx_data.digest();
100
101    Ok(DryRunTransactionBlockResponse {
102        effects: effects_response(&effects)?,
103        events: events(ctx, tx_digest, executed_tx).await?,
104        object_changes: object_changes(tx_data.sender(), &effects, executed_tx)?,
105        balance_changes: balance_changes(executed_tx)?,
106        input: input(ctx, tx_data, vec![]).await?.data,
107        execution_error_source: None,
108        suggested_gas_price,
109    })
110}

112/// Build a representation of the transaction's input data for the response.
113async fn input(
114    ctx: &Context,
115    tx_data: TransactionData,
116    tx_signatures: Vec<GenericSignature>,
117) -> Result<SuiTransactionBlock, RpcError<Error>> {
118    let data =
119        SuiTransactionBlockData::try_from_with_package_resolver(tx_data, ctx.package_resolver())
120            .await
121            .context("Failed to convert transaction data")?;
122    Ok(SuiTransactionBlock {
123        data,
124        tx_signatures,
125    })
126}

128/// Serialize transaction data to raw BCS bytes.
129fn raw_input(tx_data: &TransactionData) -> Result<Vec<u8>, RpcError<Error>> {
130    Ok(bcs::to_bytes(tx_data).context("Failed to serialize transaction")?)
131}

133/// Extract the raw effects BCS bytes from the gRPC response.
134fn raw_effects(executed_tx: &proto::ExecutedTransaction) -> Result<Vec<u8>, RpcError<Error>> {
135    let effects_bcs = executed_tx
136        .effects
137        .as_ref()
138        .and_then(|e| e.bcs.as_ref())
139        .context("Missing effects.bcs in gRPC response")?;
140    Ok(effects_bcs.value().to_vec())
141}

143/// Deserialize events from the gRPC response and resolve their layouts.
144async fn events(
145    ctx: &Context,
146    tx_digest: TransactionDigest,
147    executed_tx: &proto::ExecutedTransaction,
148) -> Result<SuiTransactionBlockEvents, RpcError<Error>> {
149    let events_bcs = executed_tx.events.as_ref().and_then(|e| e.bcs.as_ref());
150
151    let events: Vec<Event> = match events_bcs {
152        Some(bcs) => bcs
153            .deserialize()
154            .context("Failed to deserialize event BCS from gRPC response")?,
155        None => vec![],
156    };
157
158    let mut sui_events = Vec::with_capacity(events.len());
159    for (ix, event) in events.into_iter().enumerate() {
160        let layout = match ctx
161            .package_resolver()
162            .type_layout(event.type_.clone().into())
163            .await
164            .context("Failed to resolve event type layout")?
165        {
166            MoveTypeLayout::Struct(s) => MoveDatatypeLayout::Struct(s),
167            MoveTypeLayout::Enum(e) => MoveDatatypeLayout::Enum(e),
168            _ => {
169                return Err(anyhow::anyhow!(
170                    "Event {ix} is not a struct or enum: {}",
171                    event.type_.to_canonical_string(true)
172                )
173                .into());
174            }
175        };
176        sui_events.push(
177            SuiEvent::try_from(event, tx_digest, ix as u64, None, layout)
178                .context("Failed to convert event into JSON-RPC response type")?,
179        );
180    }
181
182    Ok(SuiTransactionBlockEvents { data: sui_events })
183}

185/// Convert balance changes from the gRPC response.
186fn balance_changes(
187    executed_tx: &proto::ExecutedTransaction,
188) -> Result<Vec<SuiBalanceChange>, RpcError<Error>> {
189    let mut changes = Vec::with_capacity(executed_tx.balance_changes.len());
190    for bc in &executed_tx.balance_changes {
191        let addr: SuiAddress = bc
192            .address
193            .as_ref()
194            .context("Missing address in balance change")?
195            .parse()
196            .context("Invalid owner address in balance change")?;
197        let owner = Owner::AddressOwner(addr);
198
199        let coin_type = TypeTag::from_str(
200            bc.coin_type
201                .as_ref()
202                .context("Missing coin_type in balance change")?,
203        )
204        .context("Invalid coin type in balance change")?;
205
206        let amount: i128 = bc
207            .amount
208            .as_ref()
209            .context("Missing amount in balance change")?
210            .parse()
211            .context("Invalid balance change amount")?;
212
213        changes.push(SuiBalanceChange {
214            owner,
215            coin_type,
216            amount,
217        });
218    }
219
220    Ok(changes)
221}

223/// Build object changes by correlating effects with the output objects from the gRPC response.
224fn object_changes(
225    sender: SuiAddress,
226    effects: &TransactionEffects,
227    executed_tx: &proto::ExecutedTransaction,
228) -> Result<Vec<SuiObjectChange>, RpcError<Error>> {
229    let native_changes = effects.object_changes();
230
231    // Build a map of (ObjectID, version) -> Object from the proto objects. Objects that are
232    // Wrapped or Deleted will not have BCS content and are skipped here.
233    let mut objects: HashMap<(ObjectID, u64), Object> = HashMap::new();
234    if let Some(object_set) = &executed_tx.objects {
235        for proto_obj in &object_set.objects {
236            if let Some(bcs) = &proto_obj.bcs {
237                let obj: Object = bcs
238                    .deserialize()
239                    .context("Failed to deserialize changed object from gRPC response")?;
240                objects.insert((obj.id(), obj.version().value()), obj);
241            }
242        }
243    }
244
245    let fetch_object = |id: ObjectID,
246                        v: Option<SequenceNumber>,
247                        d: Option<ObjectDigest>|
248     -> Result<Option<(Object, ObjectDigest)>, RpcError<Error>> {
249        let Some(v) = v else { return Ok(None) };
250        let Some(d) = d else { return Ok(None) };
251        let key = (id, v.value());
252        match objects.get(&key) {
253            Some(o) => Ok(Some((o.clone(), d))),
254            None => Err(crate::error::internal_error!(
255                "Object {id} at version {} referenced in effects but missing BCS in gRPC response",
256                v.value(),
257            )),
258        }
259    };
260
261    let mut changes = Vec::with_capacity(native_changes.len());
262    for change in &native_changes {
263        let &ObjectChange {
264            id: object_id,
265            id_operation,
266            input_version,
267            input_digest,
268            output_version,
269            output_digest,
270            ..
271        } = change;
272
273        let input = fetch_object(object_id, input_version, input_digest)?;
274        let output = fetch_object(object_id, output_version, output_digest)?;
275
276        changes.extend(to_sui_object_change(
277            sender,
278            object_id,
279            id_operation,
280            input,
281            output,
282            effects.lamport_version(),
283        )?);
284    }
285
286    Ok(changes)
287}

289/// Deserialize `TransactionEffects` from the BCS field in the gRPC response.
290fn deserialize_effects(
291    executed_tx: &proto::ExecutedTransaction,
292) -> Result<TransactionEffects, RpcError<Error>> {
293    let effects_bcs = executed_tx
294        .effects
295        .as_ref()
296        .and_then(|e| e.bcs.as_ref())
297        .context("Missing effects.bcs in gRPC response")?;
298    Ok(effects_bcs
299        .deserialize()
300        .context("Failed to deserialize effects from gRPC response")?)
301}
302
303fn effects_response(
304    effects: &TransactionEffects,
305) -> Result<SuiTransactionBlockEffects, RpcError<Error>> {
306    Ok(effects
307        .clone()
308        .try_into()
309        .context("Failed to convert effects into JSON-RPC response type")?)
