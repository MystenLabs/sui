4use std::sync::Arc;

7use prometheus::Registry;
8use sui_indexer_alt_reader::bigtable_reader::BigtableArgs;
9use sui_indexer_alt_reader::bigtable_reader::BigtableReader;
10use sui_indexer_alt_reader::kv_loader::KvLoader;
11use sui_indexer_alt_reader::package_resolver::DbPackageStore;
12use sui_indexer_alt_reader::package_resolver::PackageCache;
13use sui_indexer_alt_reader::pg_reader::PgReader;
14use sui_indexer_alt_reader::pg_reader::db::DbArgs;
15use sui_package_resolver::Resolver;
16use url::Url;
17
18use crate::config::RpcConfig;
19use crate::metrics::RpcMetrics;

21/// A bundle of different interfaces to data, for use by JSON-RPC method implementations.
22#[derive(Clone)]
23pub(crate) struct Context {
24    /// Direct access to the database, for running SQL queries.
25    pg_reader: PgReader,
26
27    /// Access to the database for performing point look-ups. Access is through the same connection
28    /// pool as `reader`, but this interface groups multiple point lookups into a single database
29    /// query.
30    pg_loader: Arc<DataLoader<PgReader>>,
31
32    /// Access to the kv store for performing point look-ups. This may either be backed by Bigtable
33    /// or Postgres db, depending on the configuration.
34    kv_loader: KvLoader,
35
36    /// Access to the database for accessing information about types from their packages (again
37    /// through the same connection pool as `reader`).
38    package_resolver: Arc<Resolver<Arc<PackageCache>>>,
39
40    /// Access to the RPC's metrics.
41    metrics: Arc<RpcMetrics>,
42
43    /// Access to the RPC's configuration.
44    config: Arc<RpcConfig>,
45}
46
47impl Context {
48    /// Set-up access to the stores through all the interfaces available in the context. If
49    /// `bigtable_instance` is set, KV lookups will be sent to it, otherwise they will be sent to
50    /// the `database. If `database_url` is `None`, the interfaces will be set-up but will fail to
51    /// accept any connections.
52    pub(crate) async fn new(
53        database_url: Option<Url>,
54        bigtable_instance: Option<String>,
55        db_args: DbArgs,
56        bigtable_args: BigtableArgs,
57        config: RpcConfig,
58        metrics: Arc<RpcMetrics>,
59        registry: &Registry,
60    ) -> Result<Self, anyhow::Error> {
61        let pg_reader = PgReader::new(None, database_url, db_args, registry).await?;
62        let pg_loader = Arc::new(pg_reader.as_data_loader());
63
64        let kv_loader = if let Some(instance_id) = bigtable_instance {
65            let bigtable_reader = BigtableReader::new(
66                instance_id,
67                "indexer-alt-jsonrpc".to_owned(),
68                bigtable_args,
69                registry,
70            )
71            .await?;
72
73            KvLoader::new_with_bigtable(Arc::new(bigtable_reader.as_data_loader()))
74        } else {
75            KvLoader::new_with_pg(pg_loader.clone())
76        };
77
78        let store = Arc::new(PackageCache::new(DbPackageStore::new(pg_loader.clone())));
79        let package_resolver = Arc::new(Resolver::new_with_limits(
80            store,
81            config.package_resolver.clone(),
82        ));
83
84        Ok(Self {
85            pg_reader,
86            pg_loader,
87            kv_loader,
88            package_resolver,
89            metrics,
90            config: Arc::new(config),
91        })
92    }
93
94    /// For performing arbitrary SQL queries on the Postgres db.
95    pub(crate) fn pg_reader(&self) -> &PgReader {
96        &self.pg_reader
97    }
98
99    /// For performing point look-ups on the Postgres db only.
100    pub(crate) fn pg_loader(&self) -> &Arc<DataLoader<PgReader>> {
101        &self.pg_loader
102    }
103
104    /// For performing point look-ups on the kv store.
105    /// Depends on the configuration of the indexer, the kv store may be backed by
106    /// eitherBigtable or Postgres.
107    pub(crate) fn kv_loader(&self) -> &KvLoader {
108        &self.kv_loader
109    }
110
111    /// For querying type and function signature information.
112    pub(crate) fn package_resolver(&self) -> &Resolver<Arc<PackageCache>> {
113        self.package_resolver.as_ref()
114    }
115
116    /// Access to the RPC metrics.
117    pub(crate) fn metrics(&self) -> &RpcMetrics {
118        self.metrics.as_ref()
119    }
120
121    /// Access to the RPC configuration.
122    pub(crate) fn config(&self) -> &RpcConfig {
123        self.config.as_ref()
124    }
125}
