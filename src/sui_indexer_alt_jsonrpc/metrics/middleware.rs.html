30/// Tower Layer that adds middleware to record statistics about RPC requests (how long they took to
31/// serve, how many we have served, how many succeeded or failed, etc).
32#[derive(Clone)]
33pub(crate) struct MetricsLayer {
34    metrics: Arc<RpcMetrics>,
35    methods: Arc<HashSet<String>>,
36    slow_request_threshold: Duration,
37}
38
39/// The Tower Service responsible for wrapping the JSON-RPC request handler with metrics handling.
40pub(crate) struct MetricsService<S> {
41    layer: MetricsLayer,
42    inner: S,
43}
44
45struct RequestMetrics {
46    timer: HistogramTimer,
47    succeeded: IntCounterVec,
48    failed: IntCounterVec,
49    cancelled: IntCounterVec,
50}
51
52#[pin_project(PinnedDrop)]
53pub(crate) struct MetricsFuture<'a, F> {
54    metrics: Option<RequestMetrics>,
55    method: Cow<'a, str>,
56    // RPC request params for logging
57    params: Option<Cow<'a, RawValue>>,
58    slow_request_threshold: Duration,
59    #[pin]
60    inner: F,
61}
62
63impl MetricsLayer {
64    /// Create a new metrics layer that only records statistics for the given methods (any other
65    /// methods will be replaced with "<UNKNOWN>").
66    pub fn new(
67        metrics: Arc<RpcMetrics>,
68        methods: HashSet<String>,
69        slow_request_threshold: Duration,
70    ) -> Self {
71        Self {
72            metrics,
73            methods: Arc::new(methods),
74            slow_request_threshold,
75        }
76    }
77}

79impl<S> Layer<S> for MetricsLayer {
80    type Service = MetricsService<S>;
81
82    fn layer(&self, inner: S) -> Self::Service {
83        MetricsService {
84            layer: self.clone(),
85            inner,
86        }
87    }
88}
89
90impl<'a, S> RpcServiceT<'a> for MetricsService<S>
91where
92    S: RpcServiceT<'a>,
93{
94    type Future = MetricsFuture<'a, S::Future>;
95
96    fn call(&self, request: Request<'a>) -> Self::Future {
97        let method = if self.layer.methods.contains(request.method_name()) {
98            request.method.clone()
99        } else {
100            "<UNKNOWN>".into()
101        };
102
103        self.layer
104            .metrics
105            .requests_received
106            .with_label_values(&[method.as_ref()])
107            .inc();
108
109        let timer = self
110            .layer
111            .metrics
112            .request_latency
113            .with_label_values(&[method.as_ref()])
114            .start_timer();
115
116        MetricsFuture {
117            metrics: Some(RequestMetrics {
118                timer,
119                succeeded: self.layer.metrics.requests_succeeded.clone(),
120                failed: self.layer.metrics.requests_failed.clone(),
121                cancelled: self.layer.metrics.requests_cancelled.clone(),
122            }),
123            method,
124            params: request.params.clone(),
125            slow_request_threshold: self.layer.slow_request_threshold,
126            inner: self.inner.call(request),
127        }
128    }
129}

131impl<F> Future for MetricsFuture<'_, F>
132where
133    F: Future<Output = MethodResponse>,
134{
135    type Output = F::Output;
136
137    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
138        let this = self.project();
139
140        let Poll::Ready(resp) = this.inner.poll(cx) else {
141            return Poll::Pending;
142        };
143
144        let Some(metrics) = this.metrics.take() else {
145            return Poll::Ready(resp);
146        };
147
148        let method = this.method.as_ref();
149        let elapsed_ms = metrics.timer.stop_and_record() * 1000.0;
150        let slow_threshold_ms = this.slow_request_threshold.as_millis() as f64;
151
152        // Determine if we need detailed logging that includes params and response.
153        let needs_detailed_log = if let Some(code) = resp.as_error_code() {
154            code == INTERNAL_ERROR_CODE || tracing::enabled!(tracing::Level::DEBUG)
155        } else {
156            elapsed_ms > slow_threshold_ms
157        };
158
159        // Only compute params and response when needed for detailed logging
160        let (params, response) = if needs_detailed_log {
161            let params = this.params.as_ref().map(|p| p.get()).unwrap_or("[]");
162            let result = resp.as_result();
163            let response = if result.len() > 1000 {
164                format!("{}...", &result[..997])
165            } else {
166                result.to_string()
167            };
168            (params, response)
169        } else {
170            ("", String::new())
171        };
172
173        if let Some(code) = resp.as_error_code() {
174            metrics
175                .failed
176                .with_label_values(&[method, &format!("{code}")])
177                .inc();
178
179            if code == INTERNAL_ERROR_CODE {
180                error!(
181                    method,
182                    params, code, response, elapsed_ms, "Request failed with internal error"
183                );
184            } else if tracing::enabled!(tracing::Level::DEBUG) {
185                debug!(
186                    method,
187                    params, code, response, elapsed_ms, "Request failed with non-internal error"
188                );
189            } else {
190                info!(method, code, elapsed_ms, "Request failed");
191            }
192        } else {
193            metrics.succeeded.with_label_values(&[method]).inc();
194            if elapsed_ms > slow_threshold_ms {
195                warn!(
196                    method,
197                    params,
198                    response,
199                    elapsed_ms,
200                    threshold_ms = slow_threshold_ms,
201                    "Slow request - exceeded threshold but succeeded"
202                );
203            } else {
204                info!(method, elapsed_ms, "Request succeeded");
205            }
206        }
207
208        Poll::Ready(resp)
209    }
210}

212#[pinned_drop]
213impl<F> PinnedDrop for MetricsFuture<'_, F> {
214    fn drop(self: Pin<&mut Self>) {
215        let this = self.project();
216
217        if let Some(metrics) = this.metrics.take() {
218            let method = this.method.as_ref();
219            let elapsed_ms = metrics.timer.stop_and_record() * 1000.0;
220
221            metrics.cancelled.with_label_values(&[method]).inc();
222
223            info!(method, elapsed_ms, "Request cancelled");
224
225            if elapsed_ms > this.slow_request_threshold.as_millis() as f64 {
226                let params = this.params.as_ref().map(|p| p.get()).unwrap_or("[]");
227                warn!(
228                    method,
229                    params, elapsed_ms, "Cancelled request took a long time"
230                );
231            }
232        }
233    }
234}
