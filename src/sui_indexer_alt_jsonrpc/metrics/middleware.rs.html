27/// Tower Layer that adds middleware to record statistics about RPC requests (how long they took to
28/// serve, how many we have served, how many succeeded or failed, etc).
29#[derive(Clone)]
30pub(crate) struct MetricsLayer {
31    metrics: Arc<RpcMetrics>,
32    methods: Arc<HashSet<String>>,
33    slow_request_threshold: Duration,
34}
35
36/// The Tower Service responsible for wrapping the JSON-RPC request handler with metrics handling.
37pub(crate) struct MetricsService<S> {
38    layer: MetricsLayer,
39    inner: S,
40}
41
42struct RequestMetrics {
43    timer: HistogramTimer,
44    succeeded: IntCounterVec,
45    failed: IntCounterVec,
46    cancelled: IntCounterVec,
47}
48
49#[pin_project(PinnedDrop)]
50pub(crate) struct MetricsFuture<'a, F> {
51    metrics: Option<RequestMetrics>,
52    method: Cow<'a, str>,
53    // RPC request params for logging
54    params: Option<Cow<'a, RawValue>>,
55    slow_request_threshold: Duration,
56    #[pin]
57    inner: F,
58}
59
60impl MetricsLayer {
61    /// Create a new metrics layer that only records statistics for the given methods (any other
62    /// methods will be replaced with "<UNKNOWN>").
63    pub fn new(
64        metrics: Arc<RpcMetrics>,
65        methods: HashSet<String>,
66        slow_request_threshold: Duration,
67    ) -> Self {
68        Self {
69            metrics,
70            methods: Arc::new(methods),
71            slow_request_threshold,
72        }
73    }
74}
75
76impl<S> Layer<S> for MetricsLayer {
77    type Service = MetricsService<S>;
78
79    fn layer(&self, inner: S) -> Self::Service {
80        MetricsService {
81            layer: self.clone(),
82            inner,
83        }
84    }
85}
86
87impl<'a, S> RpcServiceT<'a> for MetricsService<S>
88where
89    S: RpcServiceT<'a>,
90{
91    type Future = MetricsFuture<'a, S::Future>;
92
93    fn call(&self, request: Request<'a>) -> Self::Future {
94        let method = if self.layer.methods.contains(request.method_name()) {
95            request.method.clone()
96        } else {
97            "<UNKNOWN>".into()
98        };
99
100        self.layer
101            .metrics
102            .requests_received
103            .with_label_values(&[method.as_ref()])
104            .inc();
105
106        let timer = self
107            .layer
108            .metrics
109            .request_latency
110            .with_label_values(&[method.as_ref()])
111            .start_timer();
112
113        MetricsFuture {
114            metrics: Some(RequestMetrics {
115                timer,
116                succeeded: self.layer.metrics.requests_succeeded.clone(),
117                failed: self.layer.metrics.requests_failed.clone(),
118                cancelled: self.layer.metrics.requests_cancelled.clone(),
119            }),
120            method,
121            params: request.params.clone(),
122            slow_request_threshold: self.layer.slow_request_threshold,
123            inner: self.inner.call(request),
124        }
125    }
126}

128impl<F> Future for MetricsFuture<'_, F>
129where
130    F: Future<Output = MethodResponse>,
131{
132    type Output = F::Output;
133
134    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
135        let this = self.project();
136
137        let Poll::Ready(resp) = this.inner.poll(cx) else {
138            return Poll::Pending;
139        };
140
141        let Some(metrics) = this.metrics.take() else {
142            return Poll::Ready(resp);
143        };
144
145        let method = this.method.as_ref();
146        let elapsed_ms = metrics.timer.stop_and_record() * 1000.0;
147        let slow_threshold_ms = this.slow_request_threshold.as_millis() as f64;
148
149        // Determine if we need detailed logging that includes params and response.
150        let needs_detailed_log = if let Some(code) = resp.as_error_code() {
151            code == INTERNAL_ERROR_CODE || tracing::enabled!(tracing::Level::DEBUG)
152        } else {
153            elapsed_ms > slow_threshold_ms
154        };
155
156        // Only compute params and response when needed for detailed logging
157        let (params, response) = if needs_detailed_log {
158            let params = this.params.as_ref().map(|p| p.get()).unwrap_or("[]");
159            let result = resp.as_result();
160            let response = if result.len() > 1000 {
161                format!("{}...", &result[..997])
162            } else {
163                result.to_string()
164            };
165            (params, response)
166        } else {
167            ("", String::new())
168        };
169
170        if let Some(code) = resp.as_error_code() {
171            metrics
172                .failed
173                .with_label_values(&[method, &format!("{code}")])
174                .inc();
175
176            if code == INTERNAL_ERROR_CODE {
177                error!(
178                    method,
179                    params, code, response, elapsed_ms, "Request failed with internal error"
180                );
181            } else if tracing::enabled!(tracing::Level::DEBUG) {
182                debug!(
183                    method,
184                    params, code, response, elapsed_ms, "Request failed with non-internal error"
185                );
186            } else {
187                info!(method, code, elapsed_ms, "Request failed");
188            }
189        } else {
190            metrics.succeeded.with_label_values(&[method]).inc();
191            if elapsed_ms > slow_threshold_ms {
192                warn!(
193                    method,
194                    params,
195                    response,
196                    elapsed_ms,
197                    threshold_ms = slow_threshold_ms,
198                    "Slow request - exceeded threshold but succeeded"
199                );
200            } else {
201                info!(method, elapsed_ms, "Request succeeded");
202            }
203        }
204
205        Poll::Ready(resp)
206    }
207}

209#[pinned_drop]
210impl<F> PinnedDrop for MetricsFuture<'_, F> {
211    fn drop(self: Pin<&mut Self>) {
212        let this = self.project();
213
214        if let Some(metrics) = this.metrics.take() {
215            let method = this.method.as_ref();
216            let elapsed_ms = metrics.timer.stop_and_record() * 1000.0;
217
218            metrics.cancelled.with_label_values(&[method]).inc();
219
220            info!(method, elapsed_ms, "Request cancelled");
221
222            if elapsed_ms > this.slow_request_threshold.as_millis() as f64 {
223                let params = this.params.as_ref().map(|p| p.get()).unwrap_or("[]");
224                warn!(
225                    method,
226                    params, elapsed_ms, "Cancelled request took a long time"
227                );
228            }
229        }
230    }
231}
