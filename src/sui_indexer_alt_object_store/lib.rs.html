37/// Used to potentially migrate from the legacy watermark format that does not include `reader_lo`,
38/// `pruner_hi`, and `pruner_timestamp_ms`.
39#[derive(Serialize, Deserialize, Clone, Debug)]
40struct LegacyObjectStoreWatermark {
41    epoch_hi_inclusive: u64,
42    checkpoint_hi_inclusive: Option<u64>,
43    tx_hi: u64,
44    timestamp_ms_hi_inclusive: u64,
45    #[serde(default)]
46    reader_lo: Option<u64>,
47    #[serde(default)]
48    pruner_hi: Option<u64>,
49    #[serde(default)]
50    pruner_timestamp_ms: Option<u64>,
51}

75    async fn get_watermark_for_read(
76        &self,
77        pipeline: &str,
78    ) -> anyhow::Result<Option<(ObjectStoreWatermark, u64)>> {
79        let object_path = watermark_path(pipeline);
80        let result = match self.object_store.get(&object_path).await {
81            Ok(result) => result,
82            Err(ObjectStoreError::NotFound { .. }) => return Ok(None),
83            Err(e) => return Err(e.into()),
84        };
85
86        let bytes = result.bytes().await?;
87        let watermark =
88            serde_json::from_slice::<ObjectStoreWatermark>(&bytes).with_context(|| {
89                format!("Failed to parse watermark from object store pipeline={pipeline}")
90            })?;
91        // Hide watermarks where `checkpoint_hi_inclusive < reader_lo`.
92        let Some(checkpoint_hi_inclusive) = watermark
93            .checkpoint_hi_inclusive
94            .filter(|&cp| watermark.reader_lo <= cp)
95        else {
96            return Ok(None);
97        };
98
99        Ok(Some((watermark, checkpoint_hi_inclusive)))
100    }
101
102    async fn get_watermark_for_write(
103        &self,
104        pipeline: &str,
105    ) -> anyhow::Result<(ObjectStoreWatermark, Option<String>, Option<String>)> {
106        let object_path = watermark_path(pipeline);
107        let result = match self.object_store.get(&object_path).await {
108            Ok(result) => result,
109            Err(e) => return Err(e.into()),
110        };
111
112        let e_tag = result.meta.e_tag.clone();
113        let version = result.meta.version.clone();
114        let bytes = result.bytes().await?;
115        let watermark = serde_json::from_slice::<ObjectStoreWatermark>(&bytes)
116            .context("Failed to parse watermark from object store")?;
117
118        Ok((watermark, e_tag, version))
119    }
120
121    async fn set_watermark(
122        &self,
123        pipeline: &str,
124        watermark: ObjectStoreWatermark,
125        e_tag: Option<String>,
126        version: Option<String>,
127    ) -> anyhow::Result<()> {
128        let object_path = watermark_path(pipeline);
129        let json_bytes = serde_json::to_vec(&watermark)?;
130        let payload: PutPayload = Bytes::from(json_bytes).into();
131        self.object_store
132            .put_opts(
133                &object_path,
134                payload,
135                PutMode::Update(object_store::UpdateVersion { e_tag, version }).into(),
136            )
137            .await?;
138        Ok(())
139    }
140}

159impl Connection for ObjectStoreConnection {
160    async fn init_watermark(
161        &mut self,
162        pipeline_task: &str,
163        checkpoint_hi_inclusive: Option<u64>,
164    ) -> anyhow::Result<Option<InitWatermark>> {
165        let object_path = watermark_path(pipeline_task);
166        let reader_lo = checkpoint_hi_inclusive.map_or(0, |cp| cp + 1);
167        let watermark = ObjectStoreWatermark {
168            epoch_hi_inclusive: 0,
169            checkpoint_hi_inclusive,
170            tx_hi: 0,
171            timestamp_ms_hi_inclusive: 0,
172            reader_lo,
173            pruner_hi: reader_lo,
174            pruner_timestamp_ms: 0,
175        };
176        let json_bytes = serde_json::to_vec(&watermark)?;
177        let payload: PutPayload = Bytes::from(json_bytes).into();
178        // Try create-if-not-exists write first.
179        let (checkpoint_hi_inclusive, reader_lo) = match self
180            .object_store
181            .put_opts(&object_path, payload, PutMode::Create.into())
182            .await
183        {
184            Ok(_) => (checkpoint_hi_inclusive, Some(reader_lo)),
185            Err(object_store::Error::AlreadyExists { .. }) => {
186                // Fall back to reading existing watermark.
187                let result = match self.object_store.get(&object_path).await {
188                    Ok(result) => result,
189                    Err(e) => return Err(e.into()),
190                };
191                let e_tag = result.meta.e_tag.clone();
192                let version = result.meta.version.clone();
193                let bytes = result.bytes().await?;
194                let legacy_watermark: LegacyObjectStoreWatermark = serde_json::from_slice(&bytes)
195                    .with_context(|| {
196                        format!(
197                            "Failed to parse legacy watermark from object store pipeline={pipeline_task}"
198                        )
199                    })?;
200
201                // Write data from the legacy watermark using the new format if it is missing newly added fields.
202                if legacy_watermark.reader_lo.is_none()
203                    || legacy_watermark.pruner_hi.is_none()
204                    || legacy_watermark.pruner_timestamp_ms.is_none()
205                {
206                    let watermark = ObjectStoreWatermark {
207                        epoch_hi_inclusive: legacy_watermark.epoch_hi_inclusive,
208                        checkpoint_hi_inclusive: legacy_watermark.checkpoint_hi_inclusive,
209                        tx_hi: legacy_watermark.tx_hi,
210                        timestamp_ms_hi_inclusive: legacy_watermark.timestamp_ms_hi_inclusive,
211                        ..watermark
212                    };
213                    self.set_watermark(pipeline_task, watermark, e_tag, version)
214                        .await?;
215                }
216
217                (
218                    legacy_watermark.checkpoint_hi_inclusive,
219                    legacy_watermark.reader_lo,
220                )
221            }
222            Err(e) => return Err(e.into()),
223        };
224        Ok(Some(InitWatermark {
225            checkpoint_hi_inclusive,
226            reader_lo,
227        }))
228    }

238    async fn committer_watermark(
239        &mut self,
240        pipeline_task: &str,
241    ) -> anyhow::Result<Option<CommitterWatermark>> {
242        Ok(self
243            .get_watermark_for_read(pipeline_task)
244            .await?
245            .map(|(w, checkpoint_hi_inclusive)| CommitterWatermark {
246                epoch_hi_inclusive: w.epoch_hi_inclusive,
247                checkpoint_hi_inclusive,
248                tx_hi: w.tx_hi,
249                timestamp_ms_hi_inclusive: w.timestamp_ms_hi_inclusive,
250            }))
251    }
252
253    async fn set_committer_watermark(
254        &mut self,
255        pipeline_task: &str,
256        watermark: CommitterWatermark,
257    ) -> anyhow::Result<bool> {
258        let (current_watermark, e_tag, version) =
259            self.get_watermark_for_write(pipeline_task).await?;
260
261        if current_watermark
262            .checkpoint_hi_inclusive
263            .is_some_and(|cp| cp >= watermark.checkpoint_hi_inclusive)
264        {
265            return Ok(false);
266        }
267
268        let new_watermark = ObjectStoreWatermark {
269            epoch_hi_inclusive: watermark.epoch_hi_inclusive,
270            checkpoint_hi_inclusive: Some(watermark.checkpoint_hi_inclusive),
271            tx_hi: watermark.tx_hi,
272            timestamp_ms_hi_inclusive: watermark.timestamp_ms_hi_inclusive,
273            ..current_watermark
274        };
275        self.set_watermark(pipeline_task, new_watermark, e_tag, version)
276            .await?;
277        Ok(true)
278    }
279}

282impl ConcurrentConnection for ObjectStoreConnection {
283    async fn reader_watermark(
284        &mut self,
285        pipeline: &str,
286    ) -> anyhow::Result<Option<ReaderWatermark>> {
287        Ok(self
288            .get_watermark_for_read(pipeline)
289            .await?
290            .map(|(w, checkpoint_hi_inclusive)| ReaderWatermark {
291                checkpoint_hi_inclusive,
292                reader_lo: w.reader_lo,
293            }))
294    }
295
296    async fn pruner_watermark(
297        &mut self,
298        pipeline: &'static str,
299        delay: Duration,
300    ) -> anyhow::Result<Option<PrunerWatermark>> {
301        let Some((watermark, _)) = self.get_watermark_for_read(pipeline).await? else {
302            return Ok(None);
303        };
304        // Compute max(0, (pruner_timestamp + delay) - now). Use u128 to avoid overflow
305        // when summing the two operands, and saturating_sub so we never underflow when
306        // the wait period has already elapsed. saturating_sub is safe because the caller treats
307        // anything < 1 the same.
308        let pruner_ready_ms = (watermark.pruner_timestamp_ms as u128) + delay.as_millis();
309        let now_ms = SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis();
310        let wait_for_ms = i64::try_from(pruner_ready_ms.saturating_sub(now_ms))?;
311        Ok(Some(PrunerWatermark {
312            wait_for_ms,
313            reader_lo: watermark.reader_lo,
314            pruner_hi: watermark.pruner_hi,
315        }))
316    }
317
318    async fn set_reader_watermark(
319        &mut self,
320        pipeline: &'static str,
321        reader_lo: u64,
322    ) -> anyhow::Result<bool> {
323        let (current_watermark, e_tag, version) = self.get_watermark_for_write(pipeline).await?;
324
325        if reader_lo <= current_watermark.reader_lo {
326            return Ok(false);
327        }
328
329        let new_watermark = ObjectStoreWatermark {
330            reader_lo,
331            ..current_watermark
332        };
333        self.set_watermark(pipeline, new_watermark, e_tag, version)
334            .await?;
335        Ok(true)
336    }
337
338    async fn set_pruner_watermark(
339        &mut self,
340        pipeline: &'static str,
341        pruner_hi: u64,
342    ) -> anyhow::Result<bool> {
343        let (current_watermark, e_tag, version) = self.get_watermark_for_write(pipeline).await?;
344
345        if pruner_hi <= current_watermark.pruner_hi {
346            return Ok(false);
347        }
348
349        let new_watermark = ObjectStoreWatermark {
350            pruner_hi,
351            ..current_watermark
352        };
353        self.set_watermark(pipeline, new_watermark, e_tag, version)
354            .await?;
355        Ok(true)
356    }
357}

367/// Reusable implementation of [`Connection::accepts_chain_id`] for object-store-backed
368/// connections. Stores `chain_id` at `_metadata/chain_id/{pipeline_task}` on first call
369/// via a conditional create; on subsequent calls reads and compares.
370pub async fn accepts_chain_id(
371    object_store: &dyn object_store::ObjectStore,
372    pipeline_task: &str,
373    chain_id: [u8; 32],
374) -> anyhow::Result<bool> {
375    let path = chain_id_path(pipeline_task);
376    match object_store
377        .put_opts(
378            &path,
379            chain_id.to_vec().into(),
380            object_store::PutOptions {
381                mode: PutMode::Create,
382                ..Default::default()
383            },
384        )
385        .await
386    {
387        Ok(_) => Ok(true),
388        Err(ObjectStoreError::AlreadyExists { .. }) => {
389            let bytes = object_store.get(&path).await?.bytes().await?;
390            let stored: [u8; 32] = bytes.as_ref().try_into().ok().with_context(|| {
391                format!(
392                    "stored chain_id at {} has wrong length: {}",
393                    path,
394                    bytes.len()
395                )
396            })?;
397            Ok(stored == chain_id)
398        }
399        Err(e) => Err(e.into()),
400    }

467    /// Build a delay such that `pruner_watermark` returns `wait_for_ms ~= 1h` for a
468    /// watermark with `pruner_timestamp_ms = 0` (which is what `bootstrap` produces).
469    /// `pruner_watermark` computes `(pruner_timestamp + delay) - now` saturating to 0,
470    /// so the delay must put `pruner_timestamp + delay` past `now` for the wait to be
471    /// non-zero; i.e. delay must be at least `now`.
472    fn delay_for_one_hour_wait() -> Duration {
473        let now_ms = SystemTime::now()
474            .duration_since(UNIX_EPOCH)
475            .unwrap()
476            .as_millis() as u64;
477        Duration::from_millis(now_ms + ONE_HOUR_MS)
478    }
