22/// A reader backed by the gRPC LedgerService's streaming list APIs.
23#[derive(Clone)]
24pub struct AlphaLedgerGrpcReader {
25    client: Client,
26    timeout: Option<Duration>,
27}
28
29/// A single item from a list stream and the resume cursor the server emitted alongside it.
30#[derive(Debug, Clone)]
31pub struct PageItem<T> {
32    pub payload: T,
33    pub cursor: Bytes,
34}
35
36/// A page drained from a single gRPC list stream.
37#[derive(Debug, Clone)]
38pub struct StreamPage<T> {
39    /// Items that matched the filters, in stream order.
40    pub items: Vec<PageItem<T>>,
41    first_wm_cursor: Option<Bytes>,
42    last_wm_cursor: Option<Bytes>,
43    pub end_reason: Option<proto::QueryEndReason>,
44}
45
46#[derive(Debug)]
47enum FrameKind<T> {
48    Frame {
49        payload: Option<T>,
50        cursor: Option<Bytes>,
51        end_reason: Option<proto::QueryEndReason>,
52    },
53    /// A frame with none of the known fields set (unknown/future frame kind).
54    Unknown,
55}
56
57impl AlphaLedgerGrpcReader {
58    pub async fn new(
59        uri: Uri,
60        args: LedgerGrpcArgs,
61        prefix: Option<&str>,
62        registry: &Registry,
63    ) -> anyhow::Result<Self> {
64        let timeout = args.statement_timeout();
65        let mut client = Client::new(uri)?
66            .with_max_decoding_message_size(args.ledger_grpc_max_decoding_message_size)
67            .request_layer(GrpcMetricsLayer::new(
68                prefix.unwrap_or("ledger_grpc"),
69                registry,
70            ));
71
72        if let Some(timeout) = timeout {
73            client = client.with_response_headers_timeout(timeout);
74        }
75
76        Ok(Self { client, timeout })
77    }
78
79    pub async fn list_transactions(
80        &self,
81        request: proto::ListTransactionsRequest,
82    ) -> anyhow::Result<StreamPage<ExecutedTransaction>> {
83        let stream = self
84            .client
85            .clone()
86            .ledger_client()
87            .list_transactions(self.request(request))
88            .await
89            .context("ListTransactions stream open failed")?
90            .into_inner();
91
92        drain_list_stream("ListTransactions", stream).await
93    }
94
95    /// Create a gRPC request, optionally with the grpc-timeout header if configured.
96    fn request<T>(&self, input: T) -> tonic::Request<T> {
97        let mut request = tonic::Request::new(input);
98        if let Some(timeout) = self.timeout {
99            request.set_timeout(timeout);
100        }
101        request
102    }
103}

105impl<T> StreamPage<T> {
106    /// Whether further data may exist in the direction of pagination.
107    ///
108    /// `false` iff one of:
109    /// - `reason ∈ {LedgerTip, CheckpointBound}` — authoritative range terminals.
110    /// - `reason = CursorBound` AND no cursor was emitted - the server did not do any scanning and
111    ///   short-circuited. Typically implies that the cursors for the request fell outside the
112    ///   available range.
113    pub fn has_more(&self) -> bool {
114        use proto::QueryEndReason as R;
115        match self.end_reason {
116            None => true,
117            Some(R::Unknown | R::ItemLimit | R::ScanLimit) => true,
118            Some(R::LedgerTip | R::CheckpointBound) => false,
119            Some(R::CursorBound) => self.last_cursor().is_some(),
120            // `QueryEndReason` is non exhaustive — conservatively `true` if a
121            // future variant slips past `apply()`'s `unwrap_or(Unknown)`.
122            Some(_) => true,
123        }
124    }
125
126    /// The page's starting cursor: the standalone-watermark cursor if one preceded any items,
127    /// otherwise the first item's own cursor.
128    pub fn first_cursor(&self) -> Option<&Bytes> {
129        self.first_wm_cursor
130            .as_ref()
131            .or_else(|| self.items.first().map(|item| &item.cursor))
132    }
133
134    /// The page's resume cursor: a standalone-watermark cursor emitted after the last item if one
135    /// exists, otherwise the last item's own cursor.
136    pub fn last_cursor(&self) -> Option<&Bytes> {
137        self.last_wm_cursor
138            .as_ref()
139            .or_else(|| self.items.last().map(|item| &item.cursor))
140    }
141
142    /// Construct a page directly for cross-crate tests, bypassing the drain loop. The watermark
143    /// fields are private (their invariant is maintained by [`Self::apply`]); this is the only
144    /// sanctioned way to set them from outside the crate.
145    #[cfg(feature = "testing")]
146    pub fn for_test(
147        items: Vec<PageItem<T>>,
148        first_wm_cursor: Option<Bytes>,
149        last_wm_cursor: Option<Bytes>,
150        end_reason: Option<proto::QueryEndReason>,
151    ) -> Self {
152        Self {
153            items,
154            first_wm_cursor,
155            last_wm_cursor,
156            end_reason,
157        }
158    }
159
160    /// Fold one frame into the page.
161    ///
162    /// Returns `true` when the frame is `QueryEnd`.
163    fn apply(&mut self, frame: FrameKind<T>) -> bool {
164        let FrameKind::Frame {
165            payload,
166            cursor,
167            end_reason,
168        } = frame
169        else {
170            warn!("ignoring unrecognized frame");
171            return false;
172        };
173        match payload {
174            Some(payload) => {
175                let cursor = cursor.expect("TryFrom validated item cursor");
176                self.last_wm_cursor = None;
177                self.items.push(PageItem { payload, cursor });
178            }
179            None => {
180                if let Some(cursor) = cursor {
181                    self.last_wm_cursor = Some(cursor.clone());
182                    if self.items.is_empty() && self.first_wm_cursor.is_none() {
183                        self.first_wm_cursor = Some(cursor);
184                    }
185                }
186            }
187        }
188        if let Some(reason) = end_reason {
189            // `QueryEnd::reason()` folds an absent or unknown reason into
190            // `Unknown`, so `None` here remains unambiguous shorthand for
191            // "no End frame received" (i.e. the deadline cut the stream short).
192            self.end_reason = Some(reason);
193            return true;
194        }
195        false
196    }
197}
198
199impl<T> Default for StreamPage<T> {
200    fn default() -> Self {
201        Self {
202            items: Vec::new(),
203            first_wm_cursor: None,
204            last_wm_cursor: None,
205            end_reason: None,
206        }
207    }
208}

210impl TryFrom<proto::ListTransactionsResponse> for FrameKind<ExecutedTransaction> {
211    type Error = anyhow::Error;
212
213    fn try_from(response: proto::ListTransactionsResponse) -> anyhow::Result<Self> {
214        let payload = response.transaction;
215        let cursor = response.watermark.and_then(|w| w.cursor);
216        let end_reason = response.end.map(|e| e.reason());
217
218        if payload.is_none() && cursor.is_none() && end_reason.is_none() {
219            return Ok(FrameKind::Unknown);
220        }
221        if payload.is_some() && cursor.is_none() {
222            bail!("Item frame missing watermark.cursor");
223        }
224
225        Ok(FrameKind::Frame {
226            payload,
227            cursor,
228            end_reason,
229        })
230    }
231}

233async fn drain_list_stream<R, T, S>(
234    rpc_name: &'static str,
235    stream: S,
236) -> anyhow::Result<StreamPage<T>>
237where
238    R: TryInto<FrameKind<T>, Error = anyhow::Error>,
239    S: Stream<Item = Result<R, tonic::Status>>,
240{
241    futures::pin_mut!(stream);
242    let mut page = StreamPage::default();
243    while let Some(result) = stream.next().await {
244        match result {
245            Ok(response) => {
246                let frame = response
247                    .try_into()
248                    .with_context(|| format!("{rpc_name}: malformed frame"))?;
249                // Process and break on receiving `QueryEnd`.
250                if page.apply(frame) {
251                    break;
252                }
253            }
254            // `DeadlineExceeded`: server-side `grpc-timeout` header fired. `Cancelled`: client-side
255            // channel timeout fired (or upstream cancel). In either case, preserve partial work if
256            // any progress was made.
257            Err(status)
258                if matches!(
259                    status.code(),
260                    tonic::Code::DeadlineExceeded | tonic::Code::Cancelled
261                ) =>
262            {
263                break;
264            }
265            // Consider other errors as the request failed, safest to discard partial work.
266            Err(status) => {
267                bail!(
268                    "{rpc_name}: stream error {:?}: {}",
269                    status.code(),
270                    status.message()
271                );
272            }
273        }
274    }
275
276    // Exited via `break` or via `None`. If `has_more()` promises further data, the resume cursor
277    // comes from the latest watermark received — on the fused last item, a ScanLimit end frame, or
278    // a prior beacon; a bare end frame is only sent when no progress claim exists.
279    ensure!(
280        !page.has_more() || page.last_cursor().is_some(),
281        "{rpc_name}: server reported more results but did not provide resume cursor — cannot continue",
282    );
283
284    Ok(page)
285}

344        let mut page: StreamPage<ExecutedTransaction> = StreamPage::default();
345        page.apply(frame(item_response(b"c1")));
346        page.apply(frame(watermark_response(b"w2")));

357            Some(b"c1".as_ref())

359        assert_eq!(page.last_cursor().map(|c| c.as_ref()), Some(b"c3".as_ref()));
