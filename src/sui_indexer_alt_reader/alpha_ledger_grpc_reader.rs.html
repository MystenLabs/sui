26/// A reader backed by the gRPC LedgerService's v2alpha experimental query APIs.
27#[derive(Clone)]
28pub struct AlphaLedgerGrpcReader {
29    client: LedgerServiceClient<GrpcMetricsService<Channel>>,
30    timeout: Option<Duration>,
31}
32
33/// A single item from a list stream and the resume cursor the server emitted alongside it.
34#[derive(Debug, Clone)]
35pub struct PageItem<T> {
36    pub payload: T,
37    pub cursor: Bytes,
38}
39
40/// A page drained from a single gRPC list stream.
41#[derive(Debug, Clone)]
42pub struct StreamPage<T> {
43    /// Items that matched the filters, in stream order.
44    pub items: Vec<PageItem<T>>,
45    first_wm_cursor: Option<Bytes>,
46    last_wm_cursor: Option<Bytes>,
47    pub end_reason: Option<proto::QueryEndReason>,
48}
49
50#[derive(Debug)]
51enum FrameKind<T> {
52    Item { payload: T, cursor: Bytes },
53    Watermark { cursor: Option<Bytes> },
54    End { reason: i32 },
55    Unknown,
56}
57
58impl AlphaLedgerGrpcReader {
59    pub async fn new(
60        uri: Uri,
61        args: LedgerGrpcArgs,
62        prefix: Option<&str>,
63        registry: &Registry,
64    ) -> anyhow::Result<Self> {
65        let mut endpoint = Channel::builder(uri.clone());
66        if let Some(timeout) = args.statement_timeout() {
67            endpoint = endpoint.timeout(timeout);
68        }
69
70        if uri.scheme_str() == Some("https") {
71            let tls_config = ClientTlsConfig::new().with_native_roots();
72            endpoint = endpoint.tls_config(tls_config)?;
73        }
74
75        let channel = endpoint.connect_lazy();
76        let layered =
77            GrpcMetricsLayer::new(prefix.unwrap_or("ledger_grpc"), registry).layer(channel);
78
79        let timeout = args.statement_timeout();
80        let client = LedgerServiceClient::new(layered.clone())
81            .max_decoding_message_size(args.ledger_grpc_max_decoding_message_size);
82
83        Ok(Self { client, timeout })
84    }
85
86    pub async fn list_transactions(
87        &self,
88        request: proto::ListTransactionsRequest,
89    ) -> anyhow::Result<StreamPage<ExecutedTransaction>> {
90        let mut client = self.client.clone();
91        let stream = client
92            .list_transactions(self.request(request))
93            .await
94            .context("ListTransactions stream open failed")?
95            .into_inner();
96
97        drain_list_stream("ListTransactions", stream).await
98    }
99
100    /// Create a gRPC request, optionally with the grpc-timeout header if configured.
101    fn request<T>(&self, input: T) -> tonic::Request<T> {
102        let mut request = tonic::Request::new(input);
103        if let Some(timeout) = self.timeout {
104            request.set_timeout(timeout);
105        }
106        request
107    }
108}

110impl<T> StreamPage<T> {
111    /// Whether further data may exist in the direction of pagination.
112    ///
113    /// `false` iff one of:
114    /// - `reason ∈ {LedgerTip, CheckpointBound}` — authoritative range terminals.
115    /// - `reason = CursorBound` AND no cursor was emitted - the server did not do any scanning and
116    ///   short-circuited. Typically implies that the cursors for the request fell outside the
117    ///   available range.
118    pub fn has_more(&self) -> bool {
119        use proto::QueryEndReason as R;
120        match self.end_reason {
121            None => true,
122            Some(R::Unspecified | R::ItemLimit | R::ScanLimit) => true,
123            Some(R::LedgerTip | R::CheckpointBound) => false,
124            Some(R::CursorBound) => self.last_cursor().is_some(),
125            // `QueryEndReason` is non exhaustive — conservatively `true` if a
126            // future variant slips past `apply()`'s `unwrap_or(Unspecified)`.
127            Some(_) => true,
128        }
129    }
130
131    /// The page's starting cursor: the standalone-watermark cursor if one preceded any items,
132    /// otherwise the first item's own cursor.
133    pub fn first_cursor(&self) -> Option<&Bytes> {
134        self.first_wm_cursor
135            .as_ref()
136            .or_else(|| self.items.first().map(|item| &item.cursor))
137    }
138
139    /// The page's resume cursor: a standalone-watermark cursor emitted after the last item if one
140    /// exists, otherwise the last item's own cursor.
141    pub fn last_cursor(&self) -> Option<&Bytes> {
142        self.last_wm_cursor
143            .as_ref()
144            .or_else(|| self.items.last().map(|item| &item.cursor))
145    }
146
147    /// Fold one frame into the page.
148    ///
149    /// Returns `true` when the frame is `QueryEnd`.
150    fn apply(&mut self, frame: FrameKind<T>) -> bool {
151        match frame {
152            FrameKind::Item { payload, cursor } => {
153                self.last_wm_cursor = None;
154                self.items.push(PageItem { payload, cursor });
155            }
156            FrameKind::Watermark { cursor } => {
157                self.last_wm_cursor = cursor.clone();
158                if self.items.is_empty() && self.first_wm_cursor.is_none() {
159                    self.first_wm_cursor = cursor.clone();
160                }
161            }
162            FrameKind::End { reason } => {
163                // Fold an unknown reason into `Unspecified` so `None` remains unambiguous shorthand
164                // for "no End frame received" (i.e. the deadline cut the stream short).
165                self.end_reason =
166                    Some(proto::QueryEndReason::try_from(reason).unwrap_or_else(|_| {
167                        warn!(reason, "unknown QueryEndReason");
168                        proto::QueryEndReason::Unspecified
169                    }));
170                return true;
171            }
172            FrameKind::Unknown => {
173                warn!("unknown Frame");
174            }
175        }
176        false
177    }
178}
179
180impl<T> Default for StreamPage<T> {
181    fn default() -> Self {
182        Self {
183            items: Vec::new(),
184            first_wm_cursor: None,
185            last_wm_cursor: None,
186            end_reason: None,
187        }
188    }
189}

191impl TryFrom<proto::ListTransactionsResponse> for FrameKind<ExecutedTransaction> {
192    type Error = anyhow::Error;
193
194    fn try_from(response: proto::ListTransactionsResponse) -> anyhow::Result<Self> {
195        use proto::list_transactions_response::Response;
196
197        let Some(response) = response.response else {
198            return Ok(FrameKind::Unknown);
199        };
200        Ok(match response {
201            Response::Item(item) => {
202                let cursor = item
203                    .watermark
204                    .and_then(|w| w.cursor)
205                    .context("Item frame missing watermark.cursor")?;
206                let payload = item
207                    .transaction
208                    .context("Item frame missing transaction payload")?;
209                FrameKind::Item { payload, cursor }
210            }
211            Response::Watermark(watermark) => FrameKind::Watermark {
212                cursor: watermark.cursor,
213            },
214            Response::End(end) => FrameKind::End { reason: end.reason },
215            _ => FrameKind::Unknown,
216        })
217    }
218}

220async fn drain_list_stream<R, T, S>(
221    rpc_name: &'static str,
222    stream: S,
223) -> anyhow::Result<StreamPage<T>>
224where
225    R: TryInto<FrameKind<T>, Error = anyhow::Error>,
226    S: Stream<Item = Result<R, tonic::Status>>,
227{
228    futures::pin_mut!(stream);
229    let mut page = StreamPage::default();
230    while let Some(result) = stream.next().await {
231        match result {
232            Ok(response) => {
233                let frame = response
234                    .try_into()
235                    .with_context(|| format!("{rpc_name}: malformed frame"))?;
236                // Process and break on receiving `QueryEnd`.
237                if page.apply(frame) {
238                    break;
239                }
240            }
241            // `DeadlineExceeded`: server-side `grpc-timeout` header fired. `Cancelled`: client-side
242            // channel timeout fired (or upstream cancel). In either case, preserve partial work if
243            // any progress was made.
244            Err(status)
245                if matches!(
246                    status.code(),
247                    tonic::Code::DeadlineExceeded | tonic::Code::Cancelled
248                ) =>
249            {
250                break;
251            }
252            // Consider other errors as the request failed, safest to discard partial work.
253            Err(status) => {
254                bail!(
255                    "{rpc_name}: stream error {:?}: {}",
256                    status.code(),
257                    status.message()
258                );
259            }
260        }
261    }
262
263    // Exited via `break` or via `None`. If `has_more()` promises further data, the cursor to resume
264    // from must be present — either a standalone watermark advanced past the last item, or the last
265    // item's own cursor.
266    ensure!(
267        !page.has_more() || page.last_cursor().is_some(),
268        "{rpc_name}: server reported more results but did not provide resume cursor — cannot continue",
269    );
270
271    Ok(page)
272}

278    /// Well-formed `Item` frame with both `transaction` payload and a cursor-bearing watermark.
279    fn item_response(cursor: &[u8]) -> proto::ListTransactionsResponse {
280        let mut watermark = proto::Watermark::default();
281        watermark.cursor = Some(Bytes::copy_from_slice(cursor));
282        let mut item = proto::TransactionItem::default();
283        item.watermark = Some(watermark);
284        item.transaction = Some(ExecutedTransaction::default());
285        let mut response = proto::ListTransactionsResponse::default();
286        response.response = Some(proto::list_transactions_response::Response::Item(item));
287        response
288    }

290    fn watermark_response(cursor: &[u8]) -> proto::ListTransactionsResponse {
291        let mut watermark = proto::Watermark::default();
292        watermark.cursor = Some(Bytes::copy_from_slice(cursor));
293        let mut response = proto::ListTransactionsResponse::default();
294        response.response = Some(proto::list_transactions_response::Response::Watermark(
295            watermark,

322    #[test]
323    fn drains_items_tracking_latest_cursor_and_end_reason() {
324        let mut page: StreamPage<ExecutedTransaction> = StreamPage::default();
325        page.apply(frame(item_response(b"c1")));
326        page.apply(frame(watermark_response(b"w2")));
327        page.apply(frame(item_response(b"c3")));
328        page.apply(frame(end_response(proto::QueryEndReason::ItemLimit)));
329
330        assert_eq!(page.items.len(), 2);
331        // Per-item cursors are preserved on `PageItem` — that's the whole point of the
332        // payload/cursor split. The standalone watermark at `c2` does not produce a `PageItem`.
333        assert_eq!(page.items[0].cursor.as_ref(), b"c1".as_ref());
334        assert_eq!(page.items[1].cursor.as_ref(), b"c3".as_ref());
335        assert_eq!(
336            page.first_cursor().map(|c| c.as_ref()),
337            Some(b"c1".as_ref())
338        );
339        assert_eq!(page.last_cursor().map(|c| c.as_ref()), Some(b"c3".as_ref()));
340        assert_eq!(page.end_reason, Some(proto::QueryEndReason::ItemLimit));
341        // w2 watermark was briefly set on last_wm_cursor, but got wiped from subsequent item.
342        assert!(page.last_wm_cursor.is_none());
343    }
