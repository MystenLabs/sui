25/// Key for fetching a checkpoint's content by its sequence number.
26#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
27pub struct CheckpointKey(pub u64);
28
29#[async_trait::async_trait]
30impl Loader<CheckpointKey> for PgReader {
31    type Value = StoredCheckpoint;
32    type Error = Error;
33
34    async fn load(
35        &self,
36        keys: &[CheckpointKey],
37    ) -> Result<HashMap<CheckpointKey, Self::Value>, Error> {
38        use kv_checkpoints::dsl as c;
39
40        if keys.is_empty() {
41            return Ok(HashMap::new());
42        }
43
44        let mut conn = self.connect().await?;
45
46        let seqs: BTreeSet<_> = keys.iter().map(|d| d.0 as i64).collect();
47        let checkpoints: Vec<StoredCheckpoint> = conn
48            .results(c::kv_checkpoints.filter(c::sequence_number.eq_any(seqs)))
49            .await?;
50
51        Ok(checkpoints
52            .into_iter()
53            .map(|c| (CheckpointKey(c.sequence_number as u64), c))
54            .collect())
55    }
56}

58#[async_trait::async_trait]
59impl Loader<CheckpointKey> for BigtableReader {
60    type Value = (
61        CheckpointSummary,
62        CheckpointContents,
63        AuthorityQuorumSignInfo<true>,
64    );
65    type Error = Error;
66
67    async fn load(
68        &self,
69        keys: &[CheckpointKey],
70    ) -> Result<HashMap<CheckpointKey, Self::Value>, Error> {
71        if keys.is_empty() {
72            return Ok(HashMap::new());
73        }
74
75        let checkpoint_keys: Vec<_> = keys.iter().map(|k| k.0).collect();
76
77        Ok(self
78            .checkpoints(&checkpoint_keys)
79            .await?
80            .into_iter()
81            .map(|c| {
82                (
83                    CheckpointKey(c.summary.sequence_number),
84                    (c.summary, c.contents, c.signatures),
85                )
86            })
87            .collect())
88    }
89}

91#[async_trait::async_trait]
92impl Loader<CheckpointKey> for LedgerGrpcReader {
93    type Value = (
94        CheckpointSummary,
95        CheckpointContents,
96        AuthorityQuorumSignInfo<true>,
97    );
98    type Error = Error;
99
100    async fn load(
101        &self,
102        keys: &[CheckpointKey],
103    ) -> Result<HashMap<CheckpointKey, Self::Value>, Error> {
104        if keys.is_empty() {
105            return Ok(HashMap::new());
106        }
107
108        let mut results = HashMap::new();
109        for key in keys {
110            let request = proto::GetCheckpointRequest::by_sequence_number(key.0).with_read_mask(
111                FieldMask::from_paths(["summary.bcs", "signature", "contents.bcs"]),
112            );
113
114            match self.get_checkpoint(request).await {
115                Ok(response) => {
116                    let checkpoint = response.checkpoint.context("No checkpoint returned")?;
117
118                    let summary: CheckpointSummary = checkpoint
119                        .summary
120                        .as_ref()
121                        .and_then(|s| s.bcs.as_ref())
122                        .context("Missing summary.bcs")?
123                        .deserialize()
124                        .context("Failed to deserialize checkpoint summary")?;
125
126                    let contents: CheckpointContents = checkpoint
127                        .contents
128                        .as_ref()
129                        .and_then(|c| c.bcs.as_ref())
130                        .context("Missing contents.bcs")?
131                        .deserialize()
132                        .context("Failed to deserialize checkpoint contents")?;
133
134                    let signature: AuthorityQuorumSignInfo<true> = {
135                        let sdk_sig = sui_sdk_types::ValidatorAggregatedSignature::try_from(
136                            checkpoint.signature.as_ref().context("Missing signature")?,
137                        )
138                        .context("Failed to parse signature")?;
139                        AuthorityQuorumSignInfo::from(sdk_sig)
140                    };
141
142                    results.insert(*key, (summary, contents, signature));
143                }
144                Err(status) if status.code() == tonic::Code::NotFound => continue,
145                Err(e) => return Err(e.into()),
146            }
147        }
148        Ok(results)
149    }
150}
