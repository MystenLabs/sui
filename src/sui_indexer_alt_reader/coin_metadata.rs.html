23/// Key for fetching the  of a CoinMetadata object, based on its coin marker type, e.g.
24/// `0x2::sui::SUI`.
25#[derive(Debug, Clone, PartialEq, Eq, Hash)]
26pub struct CoinMetadataKey(pub StructTag);
27
28#[async_trait::async_trait]
29impl Loader<CoinMetadataKey> for PgReader {
30    type Value = StoredObjInfo;
31    type Error = Error;
32
33    #[allow(clippy::double_parens)]
34    async fn load(
35        &self,
36        keys: &[CoinMetadataKey],
37    ) -> Result<HashMap<CoinMetadataKey, StoredObjInfo>, Error> {
38        use obj_info::dsl as o;
39
40        let (candidates, newer) = diesel::alias!(obj_info as candidates, obj_info as newer);
41
42        macro_rules! candidates {
43            ($($field:ident),* $(,)?) => {
44                candidates.fields(($(o::$field),*))
45            };
46        }
47
48        macro_rules! newer {
49            ($($field:ident),* $(,)?) => {
50                newer.fields(($(o::$field),*))
51            };
52        }
53
54        if keys.is_empty() {
55            return Ok(HashMap::new());
56        }
57
58        let mut conn = self.connect().await?;
59
60        let instantiations = keys
61            .iter()
62            .map(|CoinMetadataKey(tag)| {
63                let params: Vec<TypeTag> = vec![tag.clone().into()];
64                bcs::to_bytes(&params)
65            })
66            .collect::<Result<Vec<_>, _>>()
67            .context("Failed to serialize coin metadata type parameters")?;
68
69        let query = candidates
70            .distinct_on(candidates!(package, module, name, instantiation))
71            .left_join(
72                newer.on(candidates!(object_id)
73                    .eq(newer!(object_id))
74                    .and(candidates!(cp_sequence_number).lt(newer!(cp_sequence_number)))),
75            )
76            .select(candidates!(
77                object_id,
78                cp_sequence_number,
79                owner_kind,
80                owner_id,
81                package,
82                module,
83                name,
84                instantiation,
85            ))
86            .filter(newer!(object_id).is_null())
87            .filter(candidates!(package).eq(SUI_FRAMEWORK_ADDRESS.into_bytes()))
88            .filter(candidates!(module).eq(COIN_MODULE_NAME.as_str()))
89            .filter(candidates!(name).eq(COIN_METADATA_STRUCT_NAME.as_str()))
90            .filter(candidates!(instantiation).eq_any(&instantiations));
91
92        let obj_info: Vec<StoredObjInfo> = conn.results(query).await?;
93        let instantiations_to_stored: HashMap<_, _> = obj_info
94            .iter()
95            .map(|stored| (&stored.instantiation, stored))
96            .collect();
97
98        Ok(keys
99            .iter()
100            .zip(instantiations)
101            .filter_map(|(key, inst)| {
102                let stored = *instantiations_to_stored.get(&Some(inst))?;
103                Some((key.clone(), stored.clone()))
104            })
105            .collect())
106    }
107}
