28/// Like `anyhow::bail!`, but returns this module's `Error` type, not `anyhow::Error`.
29macro_rules! bail {
30    ($e:expr) => {
31        return Err(Error::Internal(anyhow!($e)));
32    };
33}

35#[derive(clap::Args, Debug, Clone, Default)]
36pub struct ConsistentReaderArgs {

38    #[arg(long)]
39    pub consistent_store_url: Option<Url>,

46/// A reader backed by the consistent store gRPC service.
47#[derive(Clone)]
48pub struct ConsistentReader {
49    client: Option<Client>,
50    timeout: Option<Duration>,
51    metrics: Arc<ConsistentReaderMetrics>,
52    cancel: CancellationToken,
53}
54
55/// Response from a paginated query.
56pub struct Page<T> {
57    pub results: Vec<Edge<T>>,
58    pub has_previous_page: bool,
59    pub has_next_page: bool,
60}
61
62/// An individual element from a paginated query, includes the value and its token (pagination
63/// cursor).
64pub struct Edge<T> {
65    pub token: Vec<u8>,
66    pub value: T,
67}

69type Client = ConsistentServiceClient<Channel>;

71#[derive(thiserror::Error, Debug)]
72pub enum Error {
73    #[error(transparent)]
74    Internal(#[from] anyhow::Error),
75
76    #[error("{}", .0.message())]
77    OutOfRange(#[source] tonic::Status),
78
79    #[error("Consistent store client not configured")]
80    NotConfigured,
81}

83impl ConsistentReaderArgs {
84    pub fn statement_timeout(&self) -> Option<Duration> {
85        self.consistent_store_statement_timeout_ms
86            .map(Duration::from_millis)

90impl ConsistentReader {
91    pub async fn new(
92        prefix: Option<&str>,
93        args: ConsistentReaderArgs,
94        registry: &Registry,
95        cancel: CancellationToken,
96    ) -> Result<Self, Error> {
97        let client = if let Some(url) = &args.consistent_store_url {
98            let mut endpoint = Channel::from_shared(url.to_string())
99                .context("Failed to create channel for gRPC endpoint")?;
100
101            if let Some(timeout) = args.statement_timeout() {
102                endpoint = endpoint.timeout(timeout);
103            }
104
105            let channel = endpoint
106                .connect()
107                .await
108                .context("Failed to connect to gRPC endpoint")?;
109
110            Some(ConsistentServiceClient::new(channel))
111        } else {
112            None
113        };
114
115        let timeout = args.statement_timeout();
116        let metrics = ConsistentReaderMetrics::new(prefix, registry);
117
118        Ok(Self {
119            client,
120            timeout,
121            metrics,
122            cancel,
123        })
124    }

138    /// Batch get balances for multiple (owner, coin_type) pairs at a given checkpoint.
139    #[instrument(skip(self), level = "debug")]
140    pub async fn batch_get_balances(
141        &self,
142        checkpoint: u64,
143        address: String,
144        coin_types: Vec<String>,
145    ) -> Result<Vec<(TypeTag, u64)>, Error> {
146        let response = self
147            .request(
148                "batch_get_balances",
149                Some(checkpoint),
150                |mut client, request| async move { client.batch_get_balances(request).await },
151                BatchGetBalancesRequest {
152                    requests: coin_types
153                        .into_iter()
154                        .map(|coin_type| proto::GetBalanceRequest {
155                            owner: Some(address.clone()),
156                            coin_type: Some(coin_type),
157                        })
158                        .collect(),
159                },
160            )
161            .await?;
162
163        let mut results = vec![];
164        for balance in response.balances {
165            let edge: Edge<(TypeTag, u64)> = balance.try_into()?;
166            results.push(edge.value);
167        }
168
169        Ok(results)
170    }

196    /// Paginate coin balances for `address`, at checkpoint `checkpoint`.
197    #[instrument(skip(self), level = "debug")]
198    pub async fn list_balances(
199        &self,
200        checkpoint: u64,
201        address: String,
202        page_size: Option<u32>,
203        after_token: Option<Vec<u8>>,
204        before_token: Option<Vec<u8>>,
205        is_from_front: bool,
206    ) -> Result<Page<(TypeTag, u64)>, Error> {
207        let response = self
208            .request(
209                "list_balances",
210                Some(checkpoint),
211                |mut client, request| async move { client.list_balances(request).await },
212                proto::ListBalancesRequest {
213                    owner: Some(address),
214                    page_size,
215                    after_token: after_token.map(Into::into),
216                    before_token: before_token.map(Into::into),
217                    end: if is_from_front {
218                        Some(End::Front.into())
219                    } else {
220                        Some(End::Back.into())
221                    },
222                },
223            )
224            .await?;
225
226        let has_next_page = response.has_next_page();
227        let has_previous_page = response.has_previous_page();
228
229        let results = response
230            .balances
231            .into_iter()
232            .map(TryFrom::try_from)
233            .collect::<Result<Vec<_>, _>>()?;
234
235        Ok(Page {
236            results,
237            has_next_page,
238            has_previous_page,
239        })
240    }

341    async fn request<I, O, Fut, F>(
342        &self,
343        method: &str,
344        checkpoint: Option<u64>,
345        response: F,
346        input: I,
347    ) -> Result<O, Error>
348    where
349        F: FnOnce(Client, tonic::Request<I>) -> Fut,
350        Fut: Future<Output = Result<tonic::Response<O>, tonic::Status>>,
351    {
352        let Some(client) = self.client.clone() else {
353            return Err(Error::NotConfigured);
354        };
355
356        self.metrics
357            .requests_received
358            .with_label_values(&[method])
359            .inc();
360
361        let _timer = self
362            .metrics
363            .latency
364            .with_label_values(&[method])
365            .start_timer();
366
367        let mut request = tonic::Request::new(input);
368
369        if let Some(timeout) = self.timeout {
370            request.set_timeout(timeout);
371        }
372
373        if let Some(checkpoint) = checkpoint {
374            request.metadata_mut().insert(
375                CHECKPOINT_METADATA,
376                checkpoint
377                    .to_string()
378                    .parse()
379                    .with_context(|| format!("Invalid checkpoint {checkpoint}"))?,
380            );
381        }
382
383        let response = tokio::select! {
384            _ = self.cancel.cancelled() => {
385                bail!("Request cancelled");
386            }
387
388            r = response(client, request) => {
389                r.map(|r| r.into_inner()).map_err(Into::into)
390            }
391        };
392
393        if response.is_ok() {
394            self.metrics
395                .requests_succeeded
396                .with_label_values(&[method])
397                .inc();
398        } else {
399            self.metrics
400                .requests_failed
401                .with_label_values(&[method])
402                .inc()
403        }
404
405        response
406    }

409impl TryFrom<Balance> for Edge<(TypeTag, u64)> {
410    type Error = Error;
411
412    fn try_from(proto: Balance) -> Result<Self, Error> {
413        let coin_type: TypeTag = proto
414            .coin_type
415            .context("coin type missing")?
416            .parse()
417            .context("invalid coin type")?;
418
419        let balance: u64 = proto.balance.unwrap_or(0);
420
421        let token: Vec<u8> = proto.page_token.unwrap_or_default().into();
422
423        Ok(Edge {
424            token,
425            value: (coin_type, balance),
426        })
427    }
428}

430impl TryFrom<Object> for Edge<ObjectRef> {
431    type Error = Error;
432
433    fn try_from(proto: Object) -> Result<Self, Error> {
434        let object_id: ObjectID = proto
435            .object_id
436            .context("object ID missing")?
437            .parse()
438            .context("invalid object ID")?;
439
440        let digest: ObjectDigest = proto
441            .digest
442            .context("digest missing")?
443            .parse()
444            .context("invalid digest")?;
445
446        let version: SequenceNumber = proto.version.context("version missing")?.into();
447        let token: Vec<u8> = proto.page_token.unwrap_or_default().into();
448
449        Ok(Edge {
450            token,
451            value: (object_id, version, digest),
452        })
453    }
454}

456impl From<tonic::Status> for Error {
457    fn from(status: tonic::Status) -> Self {
458        match status.code() {
459            tonic::Code::OutOfRange => Error::OutOfRange(status),
460            _ => Error::Internal(anyhow!(status.code()).context(status.message().to_string())),
461        }
462    }
