4use std::collections::HashMap;
5
6use anyhow::Context;
7use async_graphql::dataloader::Loader;
8use diesel::{ExpressionMethods, QueryDsl};
9use move_core_types::language_storage::StructTag;
10use sui_indexer_alt_schema::{displays::StoredDisplay, schema::sum_displays};
11
12use crate::{error::Error, pg_reader::PgReader};
13
14/// Key for fetching a Display object by the type it corresponds to.
15#[derive(Debug, Clone, PartialEq, Eq, Hash)]
16pub struct DisplayKey(pub StructTag);
17
18#[async_trait::async_trait]
19impl Loader<DisplayKey> for PgReader {
20    type Value = StoredDisplay;
21    type Error = Error;
22
23    async fn load(&self, keys: &[DisplayKey]) -> Result<HashMap<DisplayKey, Self::Value>, Error> {
24        use sum_displays::dsl as d;
25
26        if keys.is_empty() {
27            return Ok(HashMap::new());
28        }
29
30        let mut conn = self.connect().await?;
31
32        let types = keys
33            .iter()
34            .map(|d| bcs::to_bytes(&d.0))
35            .collect::<Result<Vec<_>, _>>()
36            .context("Failed to serialize display types")?;
37
38        let displays: Vec<StoredDisplay> = conn
39            .results(d::sum_displays.filter(d::object_type.eq_any(types.clone())))
40            .await?;
41
42        let raw_type_to_stored: HashMap<_, _> = displays
43            .into_iter()
44            .map(|d| (d.object_type.clone(), d))
45            .collect();
46
47        Ok(keys
48            .iter()
49            .zip(types)
50            .filter_map(|(k, t)| Some((k.clone(), raw_type_to_stored.get(&t).cloned()?)))
51            .collect())
52    }
53}
