16/// Key for fetching transaction events contents (Events, TimestampMs) by digest.
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
18pub struct TransactionEventsKey(pub TransactionDigest);
19
20/// Partial transaction and events for when you only need transaction content for events
21#[derive(Debug, Clone, Queryable, Selectable)]
22#[diesel(table_name = kv_transactions)]
23pub struct StoredTransactionEvents {
24    pub events: Vec<u8>,
25    pub timestamp_ms: i64,
26}
27
28#[async_trait::async_trait]
29impl Loader<TransactionEventsKey> for PgReader {
30    type Value = StoredTransactionEvents;
31    type Error = Error;
32
33    async fn load(
34        &self,
35        keys: &[TransactionEventsKey],
36    ) -> Result<HashMap<TransactionEventsKey, Self::Value>, Error> {
37        use kv_transactions::dsl as t;
38
39        if keys.is_empty() {
40            return Ok(HashMap::new());
41        }
42
43        let mut conn = self.connect().await.map_err(Arc::new)?;
44
45        let digests: BTreeSet<_> = keys.iter().map(|d| d.0.into_inner()).collect();
46        let transactions: Vec<(Vec<u8>, StoredTransactionEvents)> = conn
47            .results(
48                t::kv_transactions
49                    .select((t::tx_digest, StoredTransactionEvents::as_select()))
50                    .filter(t::tx_digest.eq_any(digests)),
51            )
52            .await?;
53        let digest_to_stored: HashMap<_, _> = transactions
54            .into_iter()
55            .map(|(tx_digest, stored)| (tx_digest.clone(), stored))
56            .collect();
57
58        Ok(keys
59            .iter()
60            .filter_map(|key| {
61                let slice: &[u8] = key.0.as_ref();
62                Some((*key, digest_to_stored.get(slice).cloned()?))
63            })
64            .collect())
65    }
66}

68#[async_trait::async_trait]
69impl Loader<TransactionEventsKey> for BigtableReader {
70    type Value = TransactionEventsData;
71    type Error = Error;
72
73    async fn load(
74        &self,
75        keys: &[TransactionEventsKey],
76    ) -> Result<HashMap<TransactionEventsKey, Self::Value>, Self::Error> {
77        if keys.is_empty() {
78            return Ok(HashMap::new());
79        }
80
81        let digests: Vec<_> = keys.iter().map(|k| k.0).collect();
82        Ok(self
83            .transactions_events(&digests)
84            .await?
85            .into_iter()
86            .map(|(digest, events)| (TransactionEventsKey(digest), events))
87            .collect())
88    }
89}
