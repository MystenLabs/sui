16/// Histogram buckets for the distribution of latency (time between sending a DB request and
17/// receiving a response).
18const LATENCY_SEC_BUCKETS: &[f64] = &[
19    0.001, 0.002, 0.005, 0.01, 0.02, 0.05, 0.1, 0.2, 0.5, 1.0, 2.0, 5.0, 10.0, 20.0, 50.0, 100.0,
20    200.0, 500.0, 1000.0,
21];
22
23#[derive(Clone)]
24pub(crate) struct DbReaderMetrics {
25    pub latency: Histogram,
26    pub requests_received: IntCounter,
27    pub requests_succeeded: IntCounter,
28    pub requests_failed: IntCounter,
29}

55impl DbReaderMetrics {
56    pub(crate) fn new(prefix: Option<&str>, registry: &Registry) -> Arc<Self> {
57        let prefix = prefix.unwrap_or("db");
58        let name = |n| format!("{prefix}_{n}");
59
60        Arc::new(Self {
61            latency: register_histogram_with_registry!(
62                name("latency"),
63                "Time taken by the database to respond to queries",
64                LATENCY_SEC_BUCKETS.to_vec(),
65                registry,
66            )
67            .unwrap(),
68
69            requests_received: register_int_counter_with_registry!(
70                name("requests_received"),
71                "Number of database requests sent by the service",
72                registry,
73            )
74            .unwrap(),
75
76            requests_succeeded: register_int_counter_with_registry!(
77                name("requests_succeeded"),
78                "Number of database requests that completed successfully",
79                registry,
80            )
81            .unwrap(),
82
83            requests_failed: register_int_counter_with_registry!(
84                name("requests_failed"),
85                "Number of database requests that completed with an error",
86                registry,
87            )
88            .unwrap(),
89        })
90    }
91}
