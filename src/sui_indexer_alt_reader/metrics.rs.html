4use std::sync::Arc;
5
6use prometheus::{
7    Histogram, HistogramVec, IntCounter, IntCounterVec, Registry,
8    register_histogram_vec_with_registry, register_histogram_with_registry,
9    register_int_counter_vec_with_registry, register_int_counter_with_registry,
10};
11
12/// Histogram buckets for the distribution of latency (time between sending a DB request and
13/// receiving a response).
14const LATENCY_SEC_BUCKETS: &[f64] = &[
15    0.001, 0.002, 0.005, 0.01, 0.02, 0.05, 0.1, 0.2, 0.5, 1.0, 2.0, 5.0, 10.0, 20.0, 50.0, 100.0,
16    200.0, 500.0, 1000.0,
17];
18
19#[derive(Clone)]
20pub(crate) struct DbReaderMetrics {
21    pub latency: Histogram,
22    pub requests_received: IntCounter,
23    pub requests_succeeded: IntCounter,
24    pub requests_failed: IntCounter,
25}
26
27#[derive(Clone)]
28pub(crate) struct ConsistentReaderMetrics {
29    pub latency: HistogramVec,
30    pub requests_received: IntCounterVec,
31    pub requests_succeeded: IntCounterVec,
32    pub requests_failed: IntCounterVec,
33}
34
35#[derive(Clone)]
36pub(crate) struct FullnodeClientMetrics {
37    pub latency: HistogramVec,
38    pub requests_received: IntCounterVec,
39    pub requests_succeeded: IntCounterVec,
40    pub requests_failed: IntCounterVec,
41}
42
43impl DbReaderMetrics {
44    pub(crate) fn new(prefix: Option<&str>, registry: &Registry) -> Arc<Self> {
45        let prefix = prefix.unwrap_or("db");
46        let name = |n| format!("{prefix}_{n}");
47
48        Arc::new(Self {
49            latency: register_histogram_with_registry!(
50                name("latency"),
51                "Time taken by the database to respond to queries",
52                LATENCY_SEC_BUCKETS.to_vec(),
53                registry,
54            )
55            .unwrap(),
56
57            requests_received: register_int_counter_with_registry!(
58                name("requests_received"),
59                "Number of database requests sent by the service",
60                registry,
61            )
62            .unwrap(),
63
64            requests_succeeded: register_int_counter_with_registry!(
65                name("requests_succeeded"),
66                "Number of database requests that completed successfully",
67                registry,
68            )
69            .unwrap(),
70
71            requests_failed: register_int_counter_with_registry!(
72                name("requests_failed"),
73                "Number of database requests that completed with an error",
74                registry,
75            )
76            .unwrap(),
77        })
78    }
79}
