25/// Key for fetching the contents a particular version of an object.
26#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
27pub struct VersionedObjectKey(pub ObjectID, pub u64);
28
29#[async_trait::async_trait]
30impl Loader<VersionedObjectKey> for PgReader {
31    type Value = StoredObject;
32    type Error = Error;
33
34    async fn load(
35        &self,
36        keys: &[VersionedObjectKey],
37    ) -> Result<HashMap<VersionedObjectKey, StoredObject>, Error> {
38        use kv_objects::dsl as o;
39
40        if keys.is_empty() {
41            return Ok(HashMap::new());
42        }
43
44        let mut conn = self.connect().await?;
45
46        let mut query = o::kv_objects.into_boxed();
47
48        for VersionedObjectKey(id, version) in keys {
49            query = query.or_filter(
50                o::object_id
51                    .eq(id.into_bytes())
52                    .and(o::object_version.eq(*version as i64)),
53            );
54        }
55
56        let objects: Vec<StoredObject> = conn.results(query).await?;
57
58        let key_to_stored: HashMap<_, _> = objects
59            .iter()
60            .map(|stored| {
61                let id = &stored.object_id[..];
62                let version = stored.object_version as u64;
63                ((id, version), stored)
64            })
65            .collect();
66
67        Ok(keys
68            .iter()
69            .filter_map(|key| {
70                let slice: &[u8] = key.0.as_ref();
71                let stored = *key_to_stored.get(&(slice, key.1))?;
72                Some((*key, stored.clone()))
73            })
74            .collect())
75    }
76}

78#[async_trait::async_trait]
79impl Loader<VersionedObjectKey> for BigtableReader {
80    type Value = Object;
81    type Error = Error;
82
83    async fn load(
84        &self,
85        keys: &[VersionedObjectKey],
86    ) -> Result<HashMap<VersionedObjectKey, Object>, Error> {
87        if keys.is_empty() {
88            return Ok(HashMap::new());
89        }
90
91        let object_keys: Vec<ObjectKey> = keys
92            .iter()
93            .map(|key| ObjectKey(key.0, key.1.into()))
94            .collect();
95
96        Ok(self
97            .objects(&object_keys)
98            .await?
99            .into_iter()
100            .map(|o| (VersionedObjectKey(o.id(), o.version().into()), o))
101            .collect())
102    }
103}

105#[async_trait::async_trait]
106impl Loader<VersionedObjectKey> for LedgerGrpcReader {
107    type Value = Object;
108    type Error = Error;
109
110    async fn load(
111        &self,
112        keys: &[VersionedObjectKey],
113    ) -> Result<HashMap<VersionedObjectKey, Object>, Error> {
114        if keys.is_empty() {
115            return Ok(HashMap::new());
116        }
117
118        let requests = keys
119            .iter()
120            .map(|key| {
121                let mut req = proto::GetObjectRequest::new(&key.0.into());
122                req.version = Some(key.1);
123                req
124            })
125            .collect();
126
127        let mut request = proto::BatchGetObjectsRequest::default();
128        request.requests = requests;
129        request.read_mask = Some(FieldMask::from_paths(["bcs"]));
130
131        let batch_response = self.batch_get_objects(request).await?;
132
133        let mut results = HashMap::new();
134        for obj_result in batch_response.objects {
135            if let Some(proto::get_object_result::Result::Object(object)) = obj_result.result {
136                let obj: Object = object
137                    .bcs
138                    .as_ref()
139                    .context("Missing bcs in object")?
140                    .deserialize()
141                    .context("Failed to deserialize object")?;
142                results.insert(VersionedObjectKey(obj.id(), obj.version().into()), obj);
143            }
144        }
145        Ok(results)
146    }
147}
