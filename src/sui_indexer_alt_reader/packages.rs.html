14/// Key for fetching the original ID of a package
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
16pub struct PackageOriginalIdKey(pub ObjectID);
17
18/// Key for fetching the latest version of a package, based on its *original ID* and a checkpoint
19/// bound.
20#[derive(Debug, Clone, Copy, PartialEq, Eq, Ord, PartialOrd, Hash)]
21pub struct CheckpointBoundedOriginalPackageKey(pub ObjectID, pub u64);
22
23/// Key for fetching a package by its original ID and version.
24#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
25pub struct VersionedOriginalPackageKey(pub ObjectID, pub u64);
26
27#[async_trait::async_trait]
28impl Loader<PackageOriginalIdKey> for PgReader {
29    type Value = StoredPackageOriginalId;
30    type Error = Error;
31
32    async fn load(
33        &self,
34        keys: &[PackageOriginalIdKey],
35    ) -> Result<HashMap<PackageOriginalIdKey, StoredPackageOriginalId>, Error> {
36        if keys.is_empty() {
37            return Ok(HashMap::new());
38        }
39
40        let mut conn = self.connect().await?;
41
42        let ids: Vec<_> = keys.iter().map(|k| k.0.into_bytes()).collect();
43        let query = diesel::sql_query(
44            r#"
45                SELECT
46                    k.package_id,
47                    v.original_id,
48                    v.cp_sequence_number
49                FROM (
50                    SELECT UNNEST($1) package_id
51                ) k
52                CROSS JOIN LATERAL (
53                    SELECT
54                        original_id,
55                        cp_sequence_number
56                    FROM
57                        kv_packages
58                    WHERE
59                        kv_packages.package_id = k.package_id
60                    LIMIT
61                        1
62                ) v
63            "#,
64        )
65        .bind::<Array<Bytea>, _>(ids);
66
67        let stored: Vec<StoredPackageOriginalId> = conn.results(query).await?;
68        let id_to_stored: HashMap<_, _> = stored
69            .iter()
70            .map(|package| (&package.package_id[..], package))
71            .collect();
72
73        Ok(keys
74            .iter()
75            .filter_map(|key| {
76                let stored = *id_to_stored.get(key.0.into_bytes().as_ref())?;
77                Some((*key, stored.clone()))
78            })
79            .collect())
80    }
81}

83#[async_trait::async_trait]
84impl Loader<CheckpointBoundedOriginalPackageKey> for PgReader {
85    type Value = StoredPackage;
86    type Error = Error;
87
88    async fn load(
89        &self,
90        keys: &[CheckpointBoundedOriginalPackageKey],
91    ) -> Result<HashMap<CheckpointBoundedOriginalPackageKey, StoredPackage>, Error> {
92        if keys.is_empty() {
93            return Ok(HashMap::new());
94        }
95
96        let mut conn = self.connect().await?;
97
98        let ids: Vec<_> = keys.iter().map(|k| k.0.into_bytes()).collect();
99        let cps: Vec<_> = keys.iter().map(|k| k.1 as i64).collect();
100        let query = diesel::sql_query(
101            r#"
102                SELECT
103                    v.*
104                FROM (
105                    SELECT
106                        UNNEST($1) original_id,
107                        UNNEST($2) cp_sequence_number
108                ) k
109                CROSS JOIN LATERAL (
110                    SELECT
111                        package_id,
112                        package_version,
113                        original_id,
114                        is_system_package,
115                        serialized_object,
116                        cp_sequence_number
117                    FROM
118                        kv_packages
119                    WHERE
120                        kv_packages.original_id = k.original_id
121                    AND kv_packages.cp_sequence_number <= k.cp_sequence_number
122                    ORDER BY
123                        cp_sequence_number DESC,
124                        package_version DESC
125                    LIMIT
126                        1
127                ) v
128            "#,
129        )
130        .bind::<Array<Bytea>, _>(ids)
131        .bind::<Array<BigInt>, _>(cps);
132
133        let stored_packages: Vec<StoredPackage> = conn.results(query).await?;
134
135        // A single data loader request may contain multiple keys for the same package ID. Store
136        // them in an ordered map, so that we can find the latest version for each key.
137        let mut key_to_stored = BTreeMap::new();
138        for package in stored_packages {
139            let id = ObjectID::from_bytes(&package.original_id)
140                .context("Failed to deserialize ObjectID")?;
141
142            let cp_sequence_number = package.cp_sequence_number as u64;
143            key_to_stored.insert(
144                CheckpointBoundedOriginalPackageKey(id, cp_sequence_number),
145                package,
146            );
147        }
148
149        Ok(keys
150            .iter()
151            .filter_map(|key| {
152                let (bound, stored) = key_to_stored.range(..=key).last()?;
153                (key.0 == bound.0).then(|| (*key, stored.clone()))
154            })
155            .collect())
156    }
157}

159#[async_trait::async_trait]
160impl Loader<VersionedOriginalPackageKey> for PgReader {
161    type Value = StoredPackage;
162    type Error = Error;
163
164    async fn load(
165        &self,
166        keys: &[VersionedOriginalPackageKey],
167    ) -> Result<HashMap<VersionedOriginalPackageKey, StoredPackage>, Error> {
168        if keys.is_empty() {
169            return Ok(HashMap::new());
170        }
171
172        let mut conn = self.connect().await?;
173
174        let ids: Vec<_> = keys.iter().map(|k| k.0.into_bytes()).collect();
175        let versions: Vec<_> = keys.iter().map(|k| k.1 as i64).collect();
176        let query = diesel::sql_query(
177            r#"
178                SELECT
179                    v.*
180                FROM (
181                    SELECT
182                        UNNEST($1) original_id,
183                        UNNEST($2) package_version
184                ) k
185                CROSS JOIN LATERAL (
186                    SELECT
187                        package_id,
188                        package_version,
189                        original_id,
190                        is_system_package,
191                        serialized_object,
192                        cp_sequence_number
193                    FROM
194                        kv_packages
195                    WHERE
196                        kv_packages.original_id = k.original_id
197                    AND kv_packages.package_version = k.package_version
198                    LIMIT
199                        1
200                ) v
201            "#,
202        )
203        .bind::<Array<Bytea>, _>(ids)
204        .bind::<Array<BigInt>, _>(versions);
205
206        let stored_packages: Vec<StoredPackage> = conn.results(query).await?;
207        let key_to_stored: HashMap<_, _> = stored_packages
208            .iter()
209            .map(|stored| {
210                let id = &stored.original_id[..];
211                let version = stored.package_version as u64;
212                ((id, version), stored)
213            })
214            .collect();
215
216        Ok(keys
217            .iter()
218            .filter_map(|key| {
219                let slice: &[u8] = key.0.as_ref();
220                let stored = *key_to_stored.get(&(slice, key.1))?;
221                Some((*key, stored.clone()))
222            })
223            .collect())
224    }
225}
