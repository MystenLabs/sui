25/// This wrapper type exists to perform error conversion between the data fetching layer and the
26/// RPC layer, metrics collection, and debug logging of database queries.
27#[derive(Clone)]
28pub struct PgReader {
29    db: Option<db::Db>,
30    metrics: Arc<DbReaderMetrics>,
31    cancel: CancellationToken,
32}
33
34pub struct Connection<'p> {
35    conn: db::Connection<'p>,
36    metrics: Arc<DbReaderMetrics>,
37}
38
39impl PgReader {
40    /// Create a new database reader. If `database_url` is `None`, the reader will not accept any
41    /// connection requests (they will all fail).
42    ///
43    /// `prefix` is used to prefix the metrics collected by this reader.
44    pub async fn new(
45        prefix: Option<&str>,
46        database_url: Option<Url>,
47        db_args: db::DbArgs,
48        registry: &Registry,
49        cancel: CancellationToken,
50    ) -> anyhow::Result<Self> {
51        let db = if let Some(database_url) = database_url {
52            let db = db::Db::for_read(database_url, db_args)
53                .await
54                .context("Failed to create database for reading")?;
55
56            registry
57                .register(Box::new(DbConnectionStatsCollector::new(
58                    prefix,
59                    db.clone(),
60                )))
61                .context("Failed to register database connection stats collector")?;
62
63            Some(db)
64        } else {
65            None
66        };
67
68        let metrics = DbReaderMetrics::new(prefix, registry);
69
70        Ok(Self {
71            db,
72            metrics,
73            cancel,
74        })
75    }
76
77    /// Create a data loader backed by this reader.
78    pub fn as_data_loader(&self) -> DataLoader<Self> {
79        DataLoader::new(self.clone(), tokio::spawn)
80    }
81
82    /// Check if this reader has a database available.
83    pub fn has_database(&self) -> bool {
84        self.db.is_some()
85    }
86
87    /// Acquire a connection to the database. This can potentially fail if the service is cancelled
88    /// while the connection is being acquired.
89    pub async fn connect(&self) -> anyhow::Result<Connection<'_>> {
90        let Some(db) = &self.db else {
91            bail!("No database to connect to");
92        };
93
94        tokio::select! {
95            _ = self.cancel.cancelled() => {
96                bail!("Cancelled while connecting to the database");
97            }
98
99            conn = db.connect() => {
100                Ok(Connection {
101                    conn: conn.context("Failed to connect to database")?,
102                    metrics: self.metrics.clone(),
103                })
104            }
105        }
106    }
107}
108
109impl Connection<'_> {
110    pub async fn first<'q, Q, ST, U>(&mut self, query: Q) -> anyhow::Result<U>
111    where
112        Q: LimitDsl,
113        Q::Output: Query + QueryFragment<Pg> + QueryId + Send + 'q,
114        <Q::Output as Query>::SqlType: CompatibleType<U, Pg, SqlType = ST>,
115        U: Send + FromSqlRow<ST, Pg> + 'static,
116        Pg: QueryMetadata<<Q::Output as Query>::SqlType>,
117        ST: 'static,
118    {
119        let query = query.limit(1);
120        let query_debug = diesel::debug_query(&query).to_string();
121        debug!("{query_debug}");
122
123        self.metrics.requests_received.inc();
124        let _guard = self.metrics.latency.start_timer();
125
126        let res = query.get_result(&mut self.conn).await;
127        if res.as_ref().is_err_and(is_timeout) {
128            warn!(query = query_debug, "Query timed out");
129        }
130
131        if res.is_ok() {
132            self.metrics.requests_succeeded.inc();
133        } else {
134            self.metrics.requests_failed.inc();
135        }
136
137        Ok(res?)
138    }
139
140    pub async fn results<'q, Q, ST, U>(&mut self, query: Q) -> anyhow::Result<Vec<U>>
141    where
142        Q: Query + QueryFragment<Pg> + QueryId + Send + 'q,
143        Q::SqlType: CompatibleType<U, Pg, SqlType = ST>,
144        U: Send + FromSqlRow<ST, Pg> + 'static,
145        Pg: QueryMetadata<Q::SqlType>,
146        ST: 'static,
147    {
148        let query_debug = diesel::debug_query(&query).to_string();
149        debug!("{query_debug}");
150
151        self.metrics.requests_received.inc();
152        let _guard = self.metrics.latency.start_timer();
153
154        let res = query.get_results(&mut self.conn).await;
155        if res.as_ref().is_err_and(is_timeout) {
156            warn!(query = query_debug, "Query timed out");
157        }
158
159        if res.is_ok() {
160            self.metrics.requests_succeeded.inc();
161        } else {
162            self.metrics.requests_failed.inc();
163        }
164
165        Ok(res?)
166    }
167}

169/// Detect whether the error is due to a timeout.
170fn is_timeout(err: &diesel::result::Error) -> bool {
171    let diesel::result::Error::DatabaseError(_, info) = err else {
172        return false;
173    };
174
175    info.message() == "canceling statement due to statement timeout"
176}
