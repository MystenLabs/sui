18#[derive(clap::Args, Debug, Clone)]
19pub struct SystemPackageTaskArgs {
20    /// How long to wait between checking for epoch changes.
21    #[clap(long, default_value_t = Self::default().epoch_polling_interval_ms)]
22    epoch_polling_interval_ms: u64,
23}
24
25/// Background task responsible for evicting system package from the package resolver's cache after
26/// detecting an epoch boundary.
27pub struct SystemPackageTask {
28    /// Access to the database
29    pg_reader: PgReader,
30
31    /// The cached store underlying the package resolver.
32    package_cache: Arc<PackageCache>,
33
34    /// How long to wait between checks.
35    interval: Duration,
36
37    /// Signal to cancel the task.
38    cancel: CancellationToken,
39}
40
41impl SystemPackageTaskArgs {
42    pub fn epoch_polling_interval(&self) -> Duration {
43        Duration::from_millis(self.epoch_polling_interval_ms)
44    }
45}
46
47impl SystemPackageTask {
48    pub fn new(
49        args: SystemPackageTaskArgs,
50        pg_reader: PgReader,
51        package_cache: Arc<PackageCache>,
52        cancel: CancellationToken,
53    ) -> Self {
54        Self {
55            pg_reader,
56            package_cache,
57            interval: args.epoch_polling_interval(),
58            cancel,
59        }
60    }
61
62    /// Start a new task that regularly polls the database for the latest epoch and evicts system
63    /// packages if it detects that the epoch has changed (which means that a framework upgrade
64    /// could have happened).
65    ///
66    /// This operation consumes the `self` and returns a handle to the spawned tokio task. The task
67    /// will continue to run until its cancellation token is triggered.
68    pub fn run(self) -> JoinHandle<()> {
69        tokio::spawn(async move {
70            let Self {
71                pg_reader,
72                package_cache,
73                interval,
74                cancel,
75            } = self;
76
77            let mut last_epoch: i64 = 0;
78            let mut interval = time::interval(interval);
79
80            loop {
81                tokio::select! {
82                    _ = cancel.cancelled() => {
83                        info!("Shutdown signal received, terminating system package eviction task");
84                        break;
85                    }
86
87                    _ = interval.tick() => {
88                        let mut conn = match pg_reader.connect().await {
89                            Ok(conn) => conn,
90                            Err(e) => {
91                                error!("Failed to connect to database: {:?}", e);
92                                continue;
93                            }
94                        };
95
96                        #[derive(QueryableByName, Copy, Clone)]
97                        struct Watermark {
98                            #[diesel(sql_type = BigInt)]
99                            epoch_hi_inclusive: i64,
100
101                            #[diesel(sql_type = BigInt)]
102                            checkpoint_hi_inclusive: i64,
103                        }
104
105                        let query = query!(
106                            r#"
107                            SELECT
108                                epoch_hi_inclusive,
109                                checkpoint_hi_inclusive
110                            FROM
111                                watermarks
112                            WHERE
113                                pipeline = 'kv_packages'
114                            "#
115                        );
116
117                        let Watermark {
118                            epoch_hi_inclusive: next_epoch,
119                            checkpoint_hi_inclusive,
120                        } = match conn
121                            .results(query)
122                            .await
123                            .as_deref()
124                        {
125                            Ok([watermark]) => *watermark,
126
127                            Ok([]) => {
128                                info!("Package index isn't populated yet, no epoch information");
129                                continue;
130                            }
131
132                            Ok(_) => {
133                                error!("Expected exactly one row from the watermarks table");
134                                continue;
135                            },
136
137                            Err(e) => {
138                                error!("Failed to fetch latest epoch: {e}");
139                                continue;
140                            }
141                        };
142
143                        if next_epoch <= last_epoch {
144                            continue;
145                        }
146
147                        info!(last_epoch, next_epoch, "Detected epoch boundary");
148                        last_epoch = next_epoch;
149
150                        #[derive(QueryableByName, Clone)]
151                        struct SystemPackage {
152                            #[diesel(sql_type = Bytea)]
153                            original_id: Vec<u8>,
154                        }
155
156                        let query = query!(
157                            r#"
158                            SELECT DISTINCT
159                                original_id
160                            FROM
161                                kv_packages
162                            WHERE
163                                is_system_package
164                            AND cp_sequence_number <= {BigInt}
165                            "#,
166                            checkpoint_hi_inclusive
167                        );
168
169                        let system_packages: Vec<SystemPackage> = match conn.results(query).await {
170                            Ok(system_packages) => system_packages,
171
172                            Err(e) => {
173                                error!("Failed to fetch system packages: {e}");
174                                continue;
175                            }
176                        };
177
178                        let Ok(system_packages) = system_packages
179                            .into_iter()
180                            .map(|pkg| AccountAddress::from_bytes(pkg.original_id))
181                            .collect::<Result<Vec<_>, _>>()
182                        else {
183                            error!("Failed to deserialize system package addresses");
184                            continue;
185                        };
186
187                        info!(system_packages = ?system_packages, "Evicting...");
188                        package_cache.evict(system_packages)
189                    }
190                }
191            }
192        })
193    }
194}
195
196impl Default for SystemPackageTaskArgs {
197    fn default() -> Self {
198        Self {
199            epoch_polling_interval_ms: 10_000,
200        }
201    }
