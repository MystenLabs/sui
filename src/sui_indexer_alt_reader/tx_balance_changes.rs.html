4use std::collections::{BTreeSet, HashMap};
5
6use async_graphql::dataloader::Loader;
7use diesel::{ExpressionMethods, JoinOnDsl, QueryDsl, SelectableHelper};
8use sui_indexer_alt_schema::{
9    schema::{tx_balance_changes, tx_digests},
10    transactions::StoredTxBalanceChange,
11};
12use sui_types::digests::TransactionDigest;
13
14use crate::{error::Error, pg_reader::PgReader};
15
16/// Key for fetching a transaction's balance changes by digest.
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
18pub struct TxBalanceChangeKey(pub TransactionDigest);
19
20#[async_trait::async_trait]
21impl Loader<TxBalanceChangeKey> for PgReader {
22    type Value = StoredTxBalanceChange;
23    type Error = Error;
24
25    async fn load(
26        &self,
27        keys: &[TxBalanceChangeKey],
28    ) -> Result<HashMap<TxBalanceChangeKey, Self::Value>, Error> {
29        use tx_balance_changes::dsl as b;
30        use tx_digests::dsl as t;
31
32        if keys.is_empty() {
33            return Ok(HashMap::new());
34        }
35
36        let mut conn = self.connect().await?;
37
38        let digests: BTreeSet<_> = keys.iter().map(|d| d.0.into_inner()).collect();
39        let balance_changes: Vec<(Vec<u8>, StoredTxBalanceChange)> = conn
40            .results(
41                b::tx_balance_changes
42                    .inner_join(t::tx_digests.on(b::tx_sequence_number.eq(t::tx_sequence_number)))
43                    .select((t::tx_digest, StoredTxBalanceChange::as_select()))
44                    .filter(t::tx_digest.eq_any(digests)),
45            )
46            .await?;
47
48        let digest_to_balance_changes: HashMap<_, _> = balance_changes.into_iter().collect();
49
50        Ok(keys
51            .iter()
52            .filter_map(|key| {
53                let slice: &[u8] = key.0.as_ref();
54                Some((*key, digest_to_balance_changes.get(slice).cloned()?))
55            })
56            .collect())
57    }
58}
