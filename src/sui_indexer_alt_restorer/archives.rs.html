4use crate::Args;
5use sui_data_ingestion_core::end_of_epoch_data;
6use tracing::info;
7
8#[derive(Clone, Debug)]
9pub(crate) struct ArchivalCheckpointInfo {
10    pub next_checkpoint_after_epoch: u64,
11}
12
13impl ArchivalCheckpointInfo {
14    /// Reads checkpoint information from archival storage to determine,
15    /// specifically the next checkpoint number after `start_epoch` for watermarking.
16    pub async fn read_archival_checkpoint_info(args: &Args) -> anyhow::Result<Self> {
17        let checkpoints = end_of_epoch_data(args.archive_url.clone(), vec![], 5).await?;
18        let next_checkpoint_after_epoch = checkpoints[args.start_epoch as usize] + 1;
19        info!(
20            epoch = args.start_epoch,
21            checkpoint = next_checkpoint_after_epoch,
22            "Next checkpoint after epoch",
23        );
24        Ok(Self {
25            next_checkpoint_after_epoch,
26        })
27    }
28}
