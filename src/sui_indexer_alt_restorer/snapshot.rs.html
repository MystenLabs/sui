31pub type DigestByBucketAndPartition = BTreeMap<u32, BTreeMap<u32, [u8; 32]>>;

33pub struct SnapshotRestorer {
34    pub restore_args: Args,
35    pub next_checkpoint_after_epoch: u64,
36    pub snapshot_reader: StateSnapshotReaderV1,
37    pub db: Db,
38}

40impl SnapshotRestorer {
41    pub async fn new(args: &Args, next_checkpoint_after_epoch: u64) -> Result<Self, Error> {
42        let remote_store_config = ObjectStoreConfig {
43            object_store: Some(ObjectStoreType::S3),
44            aws_endpoint: Some(args.endpoint.clone()),
45            aws_virtual_hosted_style_request: true,
46            object_store_connection_limit: args.concurrency,
47            no_sign_request: true,
48            ..Default::default()
49        };
50
51        let local_path = PathBuf::from(&args.snapshot_local_dir);
52        let snapshot_dir = local_path.join("snapshot");
53        let local_store_config = ObjectStoreConfig {
54            object_store: Some(ObjectStoreType::File),
55            directory: Some(snapshot_dir.clone().to_path_buf()),
56            ..Default::default()
57        };
58
59        let m = MultiProgress::new();
60        let snapshot_reader = StateSnapshotReaderV1::new(
61            args.start_epoch,
62            &remote_store_config,
63            &local_store_config,
64            NonZeroUsize::new(args.concurrency).unwrap(),
65            m,
66            true, // skip_reset_local_store
67            3,    // max_retries
68        )
69        .await?;
70        let db = Db::for_write(args.database_url.clone(), args.db_args.clone()).await?;
71
72        Ok(Self {
73            restore_args: args.clone(),
74            snapshot_reader,
75            db,
76            next_checkpoint_after_epoch,
77        })
78    }
79
80    pub async fn restore(&mut self) -> Result<(), Error> {
81        info!(
82            epoch = self.restore_args.start_epoch,
83            "Starting snapshot restore"
84        );
85        let (sha3_digests, num_part_files) = self.snapshot_reader.compute_checksum().await?;
86        let (input_files, epoch_dir, remote_object_store, _concurrency) =
87            self.snapshot_reader.export_metadata().await?;
88        let owned_input_files: Vec<(u32, (u32, FileMetadata))> = input_files
89            .into_iter()
90            .map(|(bucket, (part_num, metadata))| (*bucket, (part_num, metadata.clone())))
91            .collect();
92        info!("Start snapshot restore.");
93        self.restore_object_infos(
94            owned_input_files,
95            epoch_dir,
96            remote_object_store,
97            sha3_digests,
98            num_part_files,
99        )
100        .await?;
101        info!(
102            epoch = self.restore_args.start_epoch,
103            "Finished snapshot restore"
104        );
105        Ok(())
106    }
107
108    async fn restore_object_infos(
109        &self,
110        input_files: Vec<(u32, (u32, FileMetadata))>,
111        epoch_dir: Path,
112        remote_object_store: Arc<dyn ObjectStoreGetExt>,
113        sha3_digests: Arc<Mutex<DigestByBucketAndPartition>>,
114        num_part_files: usize,
115    ) -> anyhow::Result<()> {
116        let move_object_progress_bar = Arc::new(self.snapshot_reader.get_multi_progress().add(
117            ProgressBar::new(num_part_files as u64).with_style(
118                ProgressStyle::with_template(
119                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} move object files restored ({msg})",
120                )
121                .unwrap(),
122            ),
123        ));
124
125        futures::stream::iter(input_files)
126            .try_for_each_spawned(
127                self.restore_args.concurrency,
128                |(bucket, (part_num, file_metadata))| {
129                    let epoch_dir = epoch_dir.clone();
130                    let remote_object_store = remote_object_store.clone();
131                    let sha3_digests = sha3_digests.clone();
132                    let bar = move_object_progress_bar.clone();
133                    let db = self.db.clone();
134                    let next_cp = self.next_checkpoint_after_epoch;
135
136                    async move {
137                        debug!(
138                            bucket = bucket,
139                            part_num = part_num,
140                            "Start downloading move object file"
141                        );
142                        let mut conn = db.connect().await?;
143                        let (bytes, _) = download_bytes(
144                            remote_object_store,
145                            &file_metadata,
146                            epoch_dir,
147                            sha3_digests,
148                            &&bucket,
149                            &part_num,
150                            Some(512), // max_timeout_secs
151                        )
152                        .await;
153                        debug!(
154                            bucket = bucket,
155                            part_num = part_num,
156                            "Finished downloading move object file"
157                        );
158                        let object_infos = LiveObjectIter::new(&file_metadata, bytes.clone())?
159                            .filter_map(|object| match object {
160                                LiveObject::Normal(obj) => {
161                                    Some(StoredObjInfo::from_object(&obj, next_cp as i64))
162                                }
163                                LiveObject::Wrapped(_) => None,
164                            })
165                            .collect::<Result<Vec<_>, _>>()?;
166                        let num_object_infos = object_infos.len();
167                        // NOTE: chunk to avoid hitting the PG limit
168                        let chunk_size: usize = i16::MAX as usize / StoredObjInfo::FIELD_COUNT;
169                        debug!(
170                            bucket = bucket,
171                            part_num = part_num,
172                            num_object_infos = num_object_infos,
173                            "Start inserting object infos"
174                        );
175                        for chunk in object_infos.chunks(chunk_size) {
176                            diesel::insert_into(obj_info::table)
177                                .values(chunk)
178                                .on_conflict_do_nothing()
179                                .execute(&mut conn)
180                                .await?;
181                        }
182                        debug!(
183                            bucket = bucket,
184                            part_num = part_num,
185                            num_object_infos = num_object_infos,
186                            "Finished inserting object infos"
187                        );
188                        bar.inc(1);
189                        bar.set_message(format!("Bucket: {}, Part: {}", bucket, part_num));
190                        Ok::<(), anyhow::Error>(())
191                    }
192                },
193            )
194            .await?;
195        Ok(())
196    }
