12#[derive(Insertable, Queryable, Debug, Clone, FieldCount)]

19    pub end_timestamp_ms: i64,

27    pub total_gas_fees: Option<i64>,
28    pub total_stake_rewards_distributed: Option<i64>,
29    pub leftover_storage_fund_inflow: Option<i64>,
30    pub epoch_commitments: Vec<u8>,
31}
