50    async fn get_staked_sui(&self, owner: SuiAddress) -> Result<Vec<StakedSui>, Error> {
51        let state = self.state.clone();
52        let result =
53            spawn_monitored_task!(async move { state.get_staked_sui(owner).await }).await??;
54
55        self.metrics
56            .get_stake_sui_result_size
57            .observe(result.len() as f64);
58        self.metrics
59            .get_stake_sui_result_size_total
60            .inc_by(result.len() as u64);
61        Ok(result)
62    }

64    async fn get_stakes_by_ids(
65        &self,
66        staked_sui_ids: Vec<ObjectID>,
67    ) -> Result<Vec<DelegatedStake>, Error> {
68        let state = self.state.clone();
69        let stakes_read = spawn_monitored_task!(async move {
70            staked_sui_ids
71                .iter()
72                .map(|id| state.get_object_read(id))
73                .collect::<Result<Vec<_>, _>>()
74        })
75        .await??;
76
77        if stakes_read.is_empty() {
78            return Ok(vec![]);
79        }
80
81        let mut stakes: Vec<(StakedSui, bool)> = vec![];
82        for stake in stakes_read.into_iter() {
83            match stake {
84                ObjectRead::Exists(_, o, _) => stakes.push((StakedSui::try_from(&o)?, true)),
85                ObjectRead::Deleted(oref) => {
86                    match self
87                        .state
88                        .find_object_lt_or_eq_version(&oref.0, &oref.1.one_before().unwrap())
89                        .await?
90                    {
91                        Some(o) => stakes.push((StakedSui::try_from(&o)?, false)),
92                        None => Err(SuiRpcInputError::UserInputError(
93                            UserInputError::ObjectNotFound {
94                                object_id: oref.0,
95                                version: None,
96                            },
97                        ))?,
98                    }
99                }
100                ObjectRead::NotExists(id) => Err(SuiRpcInputError::UserInputError(
101                    UserInputError::ObjectNotFound {
102                        object_id: id,
103                        version: None,
104                    },
105                ))?,
106            }
107        }
108
109        self.get_delegated_stakes(stakes).await
110    }

112    async fn get_stakes(&self, owner: SuiAddress) -> Result<Vec<DelegatedStake>, Error> {
113        let timer = self.metrics.get_stake_sui_latency.start_timer();
114        let stakes = self.get_staked_sui(owner).await?;
115        if stakes.is_empty() {
116            return Ok(vec![]);
117        }
118        drop(timer);
119
120        let _timer = self.metrics.get_delegated_sui_latency.start_timer();
121
122        let self_clone = self.clone();
123        spawn_monitored_task!(
124            self_clone.get_delegated_stakes(stakes.into_iter().map(|s| (s, true)).collect())
125        )
126        .await?
127    }

129    async fn get_delegated_stakes(
130        &self,
131        stakes: Vec<(StakedSui, bool)>,
132    ) -> Result<Vec<DelegatedStake>, Error> {
133        let pools = stakes.into_iter().fold(
134            BTreeMap::<_, Vec<_>>::new(),
135            |mut pools, (stake, exists)| {
136                pools
137                    .entry(stake.pool_id())
138                    .or_default()
139                    .push((stake, exists));
140                pools
141            },
142        );
143
144        let system_state = self.get_system_state()?;
145        let system_state_summary: SuiSystemStateSummary =
146            system_state.clone().into_sui_system_state_summary();
147
148        let rates = exchange_rates(&self.state, system_state_summary.epoch)
149            .await?
150            .into_iter()
151            .map(|rates| (rates.pool_id, rates))
152            .collect::<BTreeMap<_, _>>();
153
154        let mut delegated_stakes = vec![];
155        for (pool_id, stakes) in pools {
156            // Rate table and rate can be null when the pool is not active
157            let rate_table = rates.get(&pool_id).ok_or_else(|| {
158                SuiRpcInputError::GenericNotFound(
159                    "Cannot find rates for staking pool {pool_id}".to_string(),
160                )
161            })?;
162            let current_rate = rate_table.rates.first().map(|(_, rate)| rate);
163
164            let mut delegations = vec![];
165            for (stake, exists) in stakes {
166                let status = if !exists {
167                    StakeStatus::Unstaked
168                } else if system_state_summary.epoch >= stake.activation_epoch() {
169                    // TODO: use dev_inspect to call a move function to get the estimated reward
170                    let estimated_reward = if let Some(current_rate) = current_rate {
171                        let stake_rate = rate_table
172                            .rates
173                            .iter()
174                            .find_map(|(epoch, rate)| {
175                                if *epoch == stake.activation_epoch() {
176                                    Some(rate.clone())
177                                } else {
178                                    None
179                                }
180                            })
181                            .unwrap_or_default();
182                        let estimated_reward = ((stake_rate.rate() / current_rate.rate()) - 1.0)
183                            * stake.principal() as f64;
184                        max(0, estimated_reward.round() as u64)
185                    } else {
186                        0
187                    };
188                    StakeStatus::Active { estimated_reward }
189                } else {
190                    StakeStatus::Pending
191                };
192                delegations.push(Stake {
193                    staked_sui_id: stake.id(),
194                    // TODO: this might change when we implement warm up period.
195                    stake_request_epoch: stake.activation_epoch() - 1,
196                    stake_active_epoch: stake.activation_epoch(),
197                    principal: stake.principal(),
198                    status,
199                })
200            }
201            delegated_stakes.push(DelegatedStake {
202                validator_address: rate_table.address,
203                staking_pool: pool_id,
204                stakes: delegations,
205            })
206        }
207        Ok(delegated_stakes)
208    }

211        Ok(self.state.get_system_state()?)
212    }

279pub fn calculate_apys(
280    stake_subsidy_start_epoch: u64,
281    exchange_rate_table: Vec<ValidatorExchangeRates>,
282) -> Vec<ValidatorApy> {
283    let mut apys = vec![];
284
285    for rates in exchange_rate_table.into_iter().filter(|r| r.active) {
286        // we start the apy calculation from the epoch when the stake subsidy starts
287        let exchange_rates = rates.rates.into_iter().filter_map(|(epoch, rate)| {
288            if epoch >= stake_subsidy_start_epoch {
289                Some(rate)
290            } else {
291                None
292            }
293        });
294
295        // we need at least 2 data points to calculate apy
296        let average_apy = if exchange_rates.clone().count() >= 2 {
297            // rates are sorted by epoch in descending order.
298            let er_e = exchange_rates.clone().dropping(1);
299            // rate e+1
300            let er_e_1 = exchange_rates.dropping_back(1);
301            let apys = er_e
302                .zip(er_e_1)
303                .map(calculate_apy)
304                .filter(|apy| *apy > 0.0 && *apy < 0.1)
305                .take(30)
306                .collect::<Vec<_>>();
307
308            let apy_counts = apys.len() as f64;
309            apys.iter().sum::<f64>() / apy_counts
310        } else {
311            0.0
312        };
313        apys.push(ValidatorApy {
314            address: rates.address,
315            apy: average_apy,
316        });
317    }
318    apys
319}

353// APY_e = (ER_e+1 / ER_e) ^ 365
354fn calculate_apy((rate_e, rate_e_1): (PoolTokenExchangeRate, PoolTokenExchangeRate)) -> f64 {
355    (rate_e.rate() / rate_e_1.rate()).powf(365.0) - 1.0
356}

358/// Cached exchange rates for validators for the given epoch, the cache size is 1, it will be cleared when the epoch changes.
359/// rates are in descending order by epoch.
360#[cached(
361    type = "SizedCache<EpochId, Vec<ValidatorExchangeRates>>",
362    create = "{ SizedCache::with_size(1) }",
363    convert = "{ _current_epoch }",
364    result = true
365)]
366async fn exchange_rates(
367    state: &Arc<dyn StateRead>,
368    _current_epoch: EpochId,
369) -> RpcInterimResult<Vec<ValidatorExchangeRates>> {
370    let system_state = state.get_system_state()?;
371    let system_state_summary: SuiSystemStateSummary = system_state.into_sui_system_state_summary();
372
373    // Get validator rate tables
374    let mut tables = vec![];
375
376    for validator in system_state_summary.active_validators {
377        tables.push((
378            validator.sui_address,
379            validator.staking_pool_id,
380            validator.exchange_rates_id,
381            validator.exchange_rates_size,
382            true,
383        ));
384    }
385
386    // Get inactive validator rate tables
387    for df in state.get_dynamic_fields(
388        system_state_summary.inactive_pools_id,
389        None,
390        system_state_summary.inactive_pools_size as usize,
391    )? {
392        let pool_id: ID = bcs::from_bytes(&df.1.bcs_name).map_err(|e| {
393            SuiErrorKind::ObjectDeserializationError {
394                error: e.to_string(),
395            }
396        })?;
397        let validator = get_validator_from_table(
398            state.get_object_store().as_ref(),
399            system_state_summary.inactive_pools_id,
400            &pool_id,
401        )?; // TODO(wlmyng): roll this into StateReadError
402        tables.push((
403            validator.sui_address,
404            validator.staking_pool_id,
405            validator.exchange_rates_id,
406            validator.exchange_rates_size,
407            false,
408        ));
409    }
410
411    let mut exchange_rates = vec![];
412    // Get exchange rates for each validator
413    for (address, pool_id, exchange_rates_id, exchange_rates_size, active) in tables {
414        let mut rates = state
415            .get_dynamic_fields(exchange_rates_id, None, exchange_rates_size as usize)?
416            .into_iter()
417            .map(|df| {
418                let epoch: EpochId = bcs::from_bytes(&df.1.bcs_name).map_err(|e| {
419                    SuiErrorKind::ObjectDeserializationError {
420                        error: e.to_string(),
421                    }
422                })?;
423
424                let exchange_rate: PoolTokenExchangeRate = get_dynamic_field_from_store(
425                    &state.get_object_store().as_ref(),
426                    exchange_rates_id,
427                    &epoch,
428                )?;
429
430                Ok::<_, SuiError>((epoch, exchange_rate))
431            })
432            .collect::<Result<Vec<_>, _>>()?;
433
434        // Rates for some epochs might be missing due to safe mode, we need to backfill them.
435        rates = backfill_rates(rates);
436
437        exchange_rates.push(ValidatorExchangeRates {
438            address,
439            pool_id,
440            active,
441            rates,
442        });
443    }
444    Ok(exchange_rates)
445}

447#[derive(Clone, Debug)]
448pub struct ValidatorExchangeRates {
449    pub address: SuiAddress,
450    pub pool_id: ObjectID,
451    pub active: bool,
452    pub rates: Vec<(EpochId, PoolTokenExchangeRate)>,
453}
454
455/// Backfill missing rates for some epochs due to safe mode. If a rate is missing for epoch e,
456/// we will use the rate for epoch e-1 to fill it.
457/// Rates returned are in descending order by epoch.
458fn backfill_rates(
459    rates: Vec<(EpochId, PoolTokenExchangeRate)>,
460) -> Vec<(EpochId, PoolTokenExchangeRate)> {
461    if rates.is_empty() {
462        return rates;
463    }
464
465    let min_epoch = *rates.iter().map(|(e, _)| e).min().unwrap();
466    let max_epoch = *rates.iter().map(|(e, _)| e).max().unwrap();
467    let mut filled_rates = Vec::new();
468    let mut prev_rate = None;
469
470    for epoch in min_epoch..=max_epoch {
471        match rates.iter().find(|(e, _)| *e == epoch) {
472            Some((e, rate)) => {
473                prev_rate = Some(rate.clone());
474                filled_rates.push((*e, rate.clone()));
475            }
476            None => {
477                if let Some(rate) = prev_rate.clone() {
478                    filled_rates.push((epoch, rate));
479                }
480            }
481        }
482    }
483    filled_rates.reverse();
484    filled_rates
485}
