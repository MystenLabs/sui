18const SPAM_LABEL: &str = "SPAM";
19const LATENCY_SEC_BUCKETS: &[f64] = &[
20    0.001, 0.005, 0.01, 0.05, 0.1, 0.25, 0.5, 1., 2.5, 5., 10., 20., 30., 60., 90.,
21];
22
23#[derive(Debug, Clone)]
24pub struct Metrics {
25    /// Counter of requests, route is a label (ie separate timeseries per route)
26    requests_by_route: IntCounterVec,
27    /// Gauge of inflight requests, route is a label (ie separate timeseries per route)
28    inflight_requests_by_route: IntGaugeVec,
29    /// Request latency, route is a label
30    req_latency_by_route: HistogramVec,
31    /// Failed requests by route
32    errors_by_route: IntCounterVec,
33    server_errors_by_route: IntCounterVec,
34    client_errors_by_route: IntCounterVec,
35    transient_errors_by_route: IntCounterVec,
36    /// Client info
37    client: IntCounterVec,
38    /// Request size
39    rpc_request_size: HistogramVec,
40    /// Response size
41    rpc_response_size: HistogramVec,
42
43    method_whitelist: HashSet<String>,
44}
45
46impl Metrics {
47    pub fn new(registry: &prometheus::Registry, method_whitelist: &[&str]) -> Self {
48        Self {
49            requests_by_route: register_int_counter_vec_with_registry!(
50                "rpc_requests_by_route",
51                "Number of requests by route",
52                &["route"],
53                registry,
54            )
55            .unwrap(),
56            inflight_requests_by_route: register_int_gauge_vec_with_registry!(
57                "inflight_rpc_requests_by_route",
58                "Number of inflight requests by route",
59                &["route"],
60                registry,
61            )
62            .unwrap(),
63            req_latency_by_route: register_histogram_vec_with_registry!(
64                "req_latency_by_route",
65                "Latency of a request by route",
66                &["route"],
67                LATENCY_SEC_BUCKETS.to_vec(),
68                registry,
69            )
70            .unwrap(),
71            client_errors_by_route: register_int_counter_vec_with_registry!(
72                "client_errors_by_route",
73                "Number of client errors by route",
74                &["route"],
75                registry,
76            )
77            .unwrap(),
78            server_errors_by_route: register_int_counter_vec_with_registry!(
79                "server_errors_by_route",
80                "Number of server errors by route",
81                &["route"],
82                registry,
83            )
84            .unwrap(),
85            transient_errors_by_route: register_int_counter_vec_with_registry!(
86                "transient_errors_by_route",
87                "Number of transient errors by route",
88                &["route"],
89                registry,
90            )
91            .unwrap(),
92            errors_by_route: register_int_counter_vec_with_registry!(
93                "errors_by_route",
94                "Number of client and server errors by route",
95                &["route"],
96                registry
97            )
98            .unwrap(),
99            client: register_int_counter_vec_with_registry!(
100                "rpc_client",
101                "Connected RPC client's info",
102                &["client_type", "api_version"],
103                registry,
104            )
105            .unwrap(),
106            rpc_request_size: register_histogram_vec_with_registry!(
107                "rpc_request_size",
108                "Request size of rpc requests",
109                &["route"],
110                prometheus::exponential_buckets(32.0, 2.0, 19)
111                    .unwrap()
112                    .to_vec(),
113                registry,
114            )
115            .unwrap(),
116            rpc_response_size: register_histogram_vec_with_registry!(
117                "rpc_response_size",
118                "Response size of rpc requests",
119                &["route"],
120                prometheus::exponential_buckets(1024.0, 2.0, 20)
121                    .unwrap()
122                    .to_vec(),
123                registry,
124            )
125            .unwrap(),
126            method_whitelist: method_whitelist.iter().map(|s| (*s).into()).collect(),
127        }
128    }
129
130    fn check_spam<'a>(&'a self, method_name: &'a str) -> &'a str {
131        if self.method_whitelist.contains(method_name) {
132            method_name
133        } else {
134            SPAM_LABEL
135        }
136    }
137
138    fn on_request(&self, request: &jsonrpsee::types::Request<'_>) {
139        let method_name = request.method_name();
140        let method_name = self.check_spam(method_name);
141        self.inflight_requests_by_route
142            .with_label_values(&[method_name])
143            .inc();
144        self.requests_by_route
145            .with_label_values(&[method_name])
146            .inc();
147
148        self.rpc_request_size
149            .with_label_values(&[method_name])
150            .observe(request.params().len_bytes() as f64);
151    }
152
153    fn on_response(
154        &self,
155        method_name: &str,
156        started_at: Instant,
157        response: &jsonrpsee::MethodResponse,
158    ) {
159        let method_name = self.check_spam(method_name);
160        self.inflight_requests_by_route
161            .with_label_values(&[method_name])
162            .dec();
163        let req_latency_secs = (Instant::now() - started_at).as_secs_f64();
164        self.req_latency_by_route
165            .with_label_values(&[method_name])
166            .observe(req_latency_secs);
167
168        if let Some(code) = response.as_error_code() {
169            if code == jsonrpsee::types::error::CALL_EXECUTION_FAILED_CODE
170                || code == jsonrpsee::types::error::INTERNAL_ERROR_CODE
171            {
172                self.server_errors_by_route
173                    .with_label_values(&[method_name])
174                    .inc();
175            } else if code == TRANSIENT_ERROR_CODE {
176                self.transient_errors_by_route
177                    .with_label_values(&[method_name])
178                    .inc();
179            } else {
180                self.client_errors_by_route
181                    .with_label_values(&[method_name])
182                    .inc();
183            }
184            self.errors_by_route.with_label_values(&[method_name]).inc();
185        }
186
187        self.rpc_response_size
188            .with_label_values(&[method_name])
189            .observe(response.as_result().len() as f64)
190    }
191
192    pub fn on_http_request(&self, headers: &axum::http::HeaderMap<axum::http::HeaderValue>) {
193        let client_type = headers
194            .get(CLIENT_SDK_TYPE_HEADER)
195            .and_then(|v| v.to_str().ok())
196            .unwrap_or("Unknown");
197
198        let api_version = headers
199            .get(CLIENT_TARGET_API_VERSION_HEADER)
200            .and_then(|v| v.to_str().ok())
201            .unwrap_or("Unknown");
202        self.client
203            .with_label_values(&[client_type, api_version])
204            .inc();
205    }
206}

208#[derive(Clone)]
209pub struct MetricsLayer<S> {
210    inner: S,
211    metrics: Arc<Metrics>,
212}
213
214impl<S> MetricsLayer<S> {
215    pub fn new(service: S, metrics: Arc<Metrics>) -> Self {
216        Self {
217            inner: service,
218            metrics,
219        }
220    }
221}
222
223impl<'a, S> RpcServiceT<'a> for MetricsLayer<S>
224where
225    S: RpcServiceT<'a> + Send + Sync,
226    S::Future: 'a,
227{
228    type Future = futures::future::BoxFuture<'a, jsonrpsee::MethodResponse>;
229
230    fn call(&self, req: jsonrpsee::types::Request<'a>) -> Self::Future {
231        let metrics = self.metrics.clone();
232        metrics.on_request(&req);
233        let method_name = req.method_name().to_owned();
234        let started_at = Instant::now();
235        let fut = self.inner.call(req);
236
237        async move {
238            let response = fut.await;
239            metrics.on_response(&method_name, started_at, &response);
240            response
241        }
242        .boxed()
243    }
244}

246#[derive(Clone)]
247pub struct TimeoutLayer<S>(S, Duration);
248
249impl<S> TimeoutLayer<S> {
250    pub fn new(service: S, duration: Duration) -> Self {
251        Self(service, duration)
252    }
253}
254
255impl<'a, S> RpcServiceT<'a> for TimeoutLayer<S>
256where
257    S: RpcServiceT<'a> + Send + Sync,
258    S::Future: 'a,
259{
260    type Future = futures::future::BoxFuture<'a, jsonrpsee::MethodResponse>;
261
262    fn call(&self, req: jsonrpsee::types::Request<'a>) -> Self::Future {
263        let future = tokio::time::timeout(self.1, self.0.call(req));
264        async move {
265            match future.await {
266                Ok(res) => res,
267                Err(_) => jsonrpsee::MethodResponse::error(
268                    Id::Null,
269                    ErrorObject::from(ErrorCode::InternalError),
270                ),
271            }
272        }
273        .boxed()
274    }
