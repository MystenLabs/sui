18const TOO_MANY_REQUESTS_MSG: &str = "Too many requests";
19
20#[derive(Clone)]
21pub struct TrafficControllerService<S> {
22    inner: S,
23    traffic_controller: Option<Arc<TrafficController>>,
24}
25
26impl<S> TrafficControllerService<S> {
27    pub fn new(service: S, traffic_controller: Option<Arc<TrafficController>>) -> Self {
28        Self {
29            inner: service,
30            traffic_controller,
31        }
32    }
33}
34
35impl<'a, S> RpcServiceT<'a> for TrafficControllerService<S>
36where
37    S: RpcServiceT<'a> + Send + Sync + Clone + 'static,
38    S::Future: 'a,
39{
40    type Future = futures::future::BoxFuture<'a, jsonrpsee::MethodResponse>;
41
42    fn call(&self, req: jsonrpsee::types::Request<'a>) -> Self::Future {
43        let service = self.inner.clone();
44        let traffic_controller = self.traffic_controller.clone();
45
46        async move {
47            if let Some(traffic_controller) = traffic_controller {
48                let client = req.extensions().get::<IpAddr>().cloned();
49                if let Err(response) = handle_traffic_req(&traffic_controller, &client).await {
50                    response
51                } else {
52                    let response = service.call(req).await;
53                    handle_traffic_resp(&traffic_controller, client, &response).await;
54                    response
55                }
56            } else {
57                service.call(req).await
58            }
59        }
60        .boxed()
61    }
62}
63
64async fn handle_traffic_req(
65    traffic_controller: &Arc<TrafficController>,
66    client: &Option<IpAddr>,
67) -> Result<(), MethodResponse> {
68    if !traffic_controller.check(client, &None).await {
69        // Entity in blocklist
70        let err_obj =
71            ErrorObject::borrowed(ErrorCode::ServerIsBusy.code(), TOO_MANY_REQUESTS_MSG, None);
72        Err(MethodResponse::error(Id::Null, err_obj))
73    } else {
74        Ok(())
75    }
76}
77
78async fn handle_traffic_resp(
79    traffic_controller: &Arc<TrafficController>,
80    client: Option<IpAddr>,
81    response: &MethodResponse,
82) {
83    let error = response.as_error_code().map(ErrorCode::from);
84    traffic_controller.tally(TrafficTally {
85        direct: client,
86        through_fullnode: None,
87        error_info: error.map(|e| {
88            let error_type = e.to_string();
89            let error_weight = normalize(e);
90            (error_weight, error_type)
91        }),
92        // For now, count everything as spam with equal weight
93        // on the rpc node side, including gas-charging endpoints
94        // such as `sui_executeTransactionBlock`, as this can enable
95        // node operators who wish to rate limit their transcation
96        // traffic and incentivize high volume clients to choose a
97        // suitable rpc provider (or run their own). Later we may want
98        // to provide a weight distribution based on the method being called.
99        spam_weight: Weight::one(),
100        timestamp: SystemTime::now(),
101    });
102}
103
104// TODO: refine error matching here
105fn normalize(err: ErrorCode) -> Weight {
106    match err {
107        ErrorCode::InvalidRequest | ErrorCode::InvalidParams => Weight::one(),
108        // e.g. invalid client signature
109        ErrorCode::ServerError(i) if i == TRANSACTION_EXECUTION_CLIENT_ERROR_CODE => Weight::one(),
110        _ => Weight::zero(),
111    }
112}

114pub fn determine_client_ip<T>(
115    client_id_source: ClientIdSource,
116    request: &mut axum::http::Request<T>,
117) {
118    let headers = request.headers();
119    let client = match client_id_source {
120        ClientIdSource::SocketAddr => request
121            .extensions()
122            .get::<ConnectInfo<SocketAddr>>()
123            .map(|info| info.0.ip()),
124        ClientIdSource::XForwardedFor(num_hops) => {
125            let do_header_parse = |header: &axum::http::HeaderValue| match header.to_str() {
126                Ok(header_val) => {
127                    let header_contents = header_val.split(',').map(str::trim).collect::<Vec<_>>();
128                    if num_hops == 0 {
129                        error!(
130                            "x-forwarded-for: 0 specified. x-forwarded-for contents: {:?}. Please assign nonzero value for \
131                                number of hops here, or use `socket-addr` client-id-source type if requests are not being proxied \
132                                to this node. Skipping traffic controller request handling.",
133                            header_contents,
134                        );
135                        return None;
136                    }
137                    let contents_len = header_contents.len();
138                    let Some(client_ip) = header_contents.get(contents_len - num_hops) else {
139                        error!(
140                            "x-forwarded-for header value of {:?} contains {} values, but {} hops were specificed. \
141                                Expected {} values. Skipping traffic controller request handling.",
142                            header_contents,
143                            contents_len,
144                            num_hops,
145                            num_hops + 1,
146                        );
147                        return None;
148                    };
149                    parse_ip(client_ip)
150                }
151                Err(e) => {
152                    error!("Invalid UTF-8 in x-forwarded-for header: {:?}", e);
153                    None
154                }
155            };
156            if let Some(header) = headers.get("x-forwarded-for") {
157                do_header_parse(header)
158            } else if let Some(header) = headers.get("X-Forwarded-For") {
159                do_header_parse(header)
160            } else {
161                error!(
162                    "x-forwarded-for header not present for request despite node configuring x-forwarded-for tracking type"
163                );
164                None
165            }
166        }
167    };
168
169    if let Some(ip) = client {
170        request.extensions_mut().insert(ip);
171    }
172}
