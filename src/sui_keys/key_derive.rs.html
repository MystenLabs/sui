22pub const DERIVATION_PATH_COIN_TYPE: u32 = 784;
23pub const DERVIATION_PATH_PURPOSE_ED25519: u32 = 44;
24pub const DERVIATION_PATH_PURPOSE_SECP256K1: u32 = 54;
25pub const DERVIATION_PATH_PURPOSE_SECP256R1: u32 = 74;
26
27/// Ed25519 follows SLIP-0010 using hardened path: m/44'/784'/0'/0'/{index}'
28/// Secp256k1 follows BIP-32/44 using path where the first 3 levels are hardened: m/54'/784'/0'/0/{index}
29/// Secp256r1 follows BIP-32/44 using path where the first 3 levels are hardened: m/74'/784'/0'/0/{index}
30/// Note that the purpose node is used to distinguish signature schemes.
31pub fn derive_key_pair_from_path(
32    seed: &[u8],
33    derivation_path: Option<DerivationPath>,
34    key_scheme: &SignatureScheme,
35) -> Result<(SuiAddress, SuiKeyPair), SuiError> {
36    let path = validate_path(key_scheme, derivation_path)?;
37    match key_scheme {
38        SignatureScheme::ED25519 => {
39            let indexes = path.into_iter().map(|i| i.into()).collect::<Vec<_>>();
40            let derived = derive_ed25519_private_key(seed, &indexes);
41            let sk = Ed25519PrivateKey::from_bytes(&derived)
42                .map_err(|e| SuiErrorKind::SignatureKeyGenError(e.to_string()))?;
43            let kp: Ed25519KeyPair = sk.into();
44            Ok((kp.public().into(), SuiKeyPair::Ed25519(kp)))
45        }
46        SignatureScheme::Secp256k1 => {
47            let child_xprv = XPrv::derive_from_path(seed, &path)
48                .map_err(|e| SuiErrorKind::SignatureKeyGenError(e.to_string()))?;
49            let kp = Secp256k1KeyPair::from(
50                Secp256k1PrivateKey::from_bytes(child_xprv.private_key().to_bytes().as_slice())
51                    .map_err(|e| SuiErrorKind::SignatureKeyGenError(e.to_string()))?,
52            );
53            Ok((kp.public().into(), SuiKeyPair::Secp256k1(kp)))
54        }
55        SignatureScheme::Secp256r1 => {
56            let child_xprv = XPrv::derive_from_path(seed, &path)
57                .map_err(|e| SuiErrorKind::SignatureKeyGenError(e.to_string()))?;
58            let kp = Secp256r1KeyPair::from(
59                Secp256r1PrivateKey::from_bytes(child_xprv.private_key().to_bytes().as_slice())
60                    .map_err(|e| SuiErrorKind::SignatureKeyGenError(e.to_string()))?,
61            );
62            Ok((kp.public().into(), SuiKeyPair::Secp256r1(kp)))
63        }
64        SignatureScheme::BLS12381
65        | SignatureScheme::MultiSig
66        | SignatureScheme::ZkLoginAuthenticator
67        | SignatureScheme::PasskeyAuthenticator => Err(SuiErrorKind::UnsupportedFeatureError {
68            error: format!("key derivation not supported {:?}", key_scheme),
69        }
70        .into()),
71    }
72}

74pub fn validate_path(
75    key_scheme: &SignatureScheme,
76    path: Option<DerivationPath>,
77) -> Result<DerivationPath, SuiError> {
78    match key_scheme {
79        SignatureScheme::ED25519 => {
80            match path {
81                Some(p) => {
82                    // The derivation path must be hardened at all levels with purpose = 44, coin_type = 784
83                    if let &[purpose, coin_type, account, change, address] = p.as_ref() {
84                        if Some(purpose)
85                            == ChildNumber::new(DERVIATION_PATH_PURPOSE_ED25519, true).ok()
86                            && Some(coin_type)
87                                == ChildNumber::new(DERIVATION_PATH_COIN_TYPE, true).ok()
88                            && account.is_hardened()
89                            && change.is_hardened()
90                            && address.is_hardened()
91                        {
92                            Ok(p)
93                        } else {
94                            Err(
95                                SuiErrorKind::SignatureKeyGenError("Invalid path".to_string())
96                                    .into(),
97                            )
98                        }
99                    } else {
100                        Err(SuiErrorKind::SignatureKeyGenError("Invalid path".to_string()).into())
101                    }
102                }
103                None => Ok(format!(
104                    "m/{DERVIATION_PATH_PURPOSE_ED25519}'/{DERIVATION_PATH_COIN_TYPE}'/0'/0'/0'"
105                )
106                .parse()
107                .map_err(|_| {
108                    SuiErrorKind::SignatureKeyGenError("Cannot parse path".to_string())
109                })?),
110            }
111        }
112        SignatureScheme::Secp256k1 => {
113            match path {
114                Some(p) => {
115                    // The derivation path must be hardened at first 3 levels with purpose = 54, coin_type = 784
116                    if let &[purpose, coin_type, account, change, address] = p.as_ref() {
117                        if Some(purpose)
118                            == ChildNumber::new(DERVIATION_PATH_PURPOSE_SECP256K1, true).ok()
119                            && Some(coin_type)
120                                == ChildNumber::new(DERIVATION_PATH_COIN_TYPE, true).ok()
121                            && account.is_hardened()
122                            && !change.is_hardened()
123                            && !address.is_hardened()
124                        {
125                            Ok(p)
126                        } else {
127                            Err(
128                                SuiErrorKind::SignatureKeyGenError("Invalid path".to_string())
129                                    .into(),
130                            )
131                        }
132                    } else {
133                        Err(SuiErrorKind::SignatureKeyGenError("Invalid path".to_string()).into())
134                    }
135                }
136                None => Ok(format!(
137                    "m/{DERVIATION_PATH_PURPOSE_SECP256K1}'/{DERIVATION_PATH_COIN_TYPE}'/0'/0/0"
138                )
139                .parse()
140                .map_err(|_| {
141                    SuiErrorKind::SignatureKeyGenError("Cannot parse path".to_string())
142                })?),
143            }
144        }
145        SignatureScheme::Secp256r1 => {
146            match path {
147                Some(p) => {
148                    // The derivation path must be hardened at first 3 levels with purpose = 74, coin_type = 784
149                    if let &[purpose, coin_type, account, change, address] = p.as_ref() {
150                        if Some(purpose)
151                            == ChildNumber::new(DERVIATION_PATH_PURPOSE_SECP256R1, true).ok()
152                            && Some(coin_type)
153                                == ChildNumber::new(DERIVATION_PATH_COIN_TYPE, true).ok()
154                            && account.is_hardened()
155                            && !change.is_hardened()
156                            && !address.is_hardened()
157                        {
158                            Ok(p)
159                        } else {
160                            Err(
161                                SuiErrorKind::SignatureKeyGenError("Invalid path".to_string())
162                                    .into(),
163                            )
164                        }
165                    } else {
166                        Err(SuiErrorKind::SignatureKeyGenError("Invalid path".to_string()).into())
167                    }
168                }
169                None => Ok(format!(
170                    "m/{DERVIATION_PATH_PURPOSE_SECP256R1}'/{DERIVATION_PATH_COIN_TYPE}'/0'/0/0"
171                )
172                .parse()
173                .map_err(|_| {
174                    SuiErrorKind::SignatureKeyGenError("Cannot parse path".to_string())
175                })?),
176            }
177        }
178        SignatureScheme::BLS12381
179        | SignatureScheme::MultiSig
180        | SignatureScheme::ZkLoginAuthenticator
181        | SignatureScheme::PasskeyAuthenticator => Err(SuiErrorKind::UnsupportedFeatureError {
182            error: format!("key derivation not supported {:?}", key_scheme),
183        }
184        .into()),
185    }
186}

188pub fn generate_new_key(
189    key_scheme: SignatureScheme,
190    derivation_path: Option<DerivationPath>,
191    word_length: Option<String>,
192) -> Result<(SuiAddress, SuiKeyPair, SignatureScheme, String), anyhow::Error> {
193    let mnemonic = Mnemonic::new(parse_word_length(word_length)?, Language::English);
194    let seed = Seed::new(&mnemonic, "");
195    match derive_key_pair_from_path(seed.as_bytes(), derivation_path, &key_scheme) {
196        Ok((address, kp)) => Ok((address, kp, key_scheme, mnemonic.phrase().to_string())),
197        Err(e) => Err(anyhow!("Failed to generate keypair: {:?}", e)),
198    }
199}
200
201fn parse_word_length(s: Option<String>) -> Result<MnemonicType, anyhow::Error> {
202    match s {
203        None => Ok(MnemonicType::Words12),
204        Some(s) => match s.as_str() {
205            "word12" => Ok(MnemonicType::Words12),
206            "word15" => Ok(MnemonicType::Words15),
207            "word18" => Ok(MnemonicType::Words18),
208            "word21" => Ok(MnemonicType::Words21),
209            "word24" => Ok(MnemonicType::Words24),
210            _ => anyhow::bail!("Invalid word length"),
211        },
212    }
