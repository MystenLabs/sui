4use std::{fmt::Display, str::FromStr};
5
6use serde::Serialize;
7use sui_types::base_types::SuiAddress;
8
9/// An address or an alias associated with a key in the wallet
10/// This is used to distinguish between an address or an alias,
11/// enabling a user to use an alias for any command that requires an address.
12#[derive(Debug, Serialize, Clone)]
13pub enum KeyIdentity {
14    Address(SuiAddress),
15    Alias(String),
16}
17
18impl FromStr for KeyIdentity {
19    type Err = anyhow::Error;
20    fn from_str(s: &str) -> Result<Self, Self::Err> {
21        if s.starts_with("0x") {
22            Ok(KeyIdentity::Address(SuiAddress::from_str(s)?))
23        } else {
24            Ok(KeyIdentity::Alias(s.to_string()))
25        }
26    }
27}
28
29impl From<SuiAddress> for KeyIdentity {
30    fn from(addr: SuiAddress) -> Self {
31        KeyIdentity::Address(addr)
32    }
33}
34
35impl Display for KeyIdentity {
36    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
37        let v = match self {
38            KeyIdentity::Address(x) => x.to_string(),
39            KeyIdentity::Alias(x) => x.to_string(),
40        };
41        write!(f, "{}", v)
42    }
43}
