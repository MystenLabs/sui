66#[async_trait]
67#[enum_dispatch]
68pub trait AccountKeystore: Send + Sync {
69    /// Generate a new keypair and add it into the keystore.
70    async fn generate(
71        &mut self,
72        alias: Option<String>,
73        generate_options: GenerateOptions,
74    ) -> Result<GeneratedKey, anyhow::Error> {
75        let (key_scheme, derivation_path, word_length) = match generate_options {
76            GenerateOptions::Default => (SignatureScheme::ED25519, None, None),
77            GenerateOptions::Local(opts) => {
78                (opts.key_scheme, opts.derivation_path, opts.word_length)
79            }
80            GenerateOptions::ExternalSigner(_) => {
81                return Err(anyhow!(
82                    "Generating new keypair is not supported for this keystore type"
83                ));
84            }
85        };
86
87        let (address, kp, scheme, _phrase) =
88            generate_new_key(key_scheme, derivation_path, word_length)?;
89        let public_key = kp.public();
90        self.import(alias, kp).await?;
91        Ok(GeneratedKey {
92            address,
93            public_key,
94            scheme,
95        })
96    }
97
98    /// Import a keypair into the keystore from a `SuiKeyPair` and optional alias.
99    async fn import(
100        &mut self,
101        alias: Option<String>,
102        keypair: SuiKeyPair,
103    ) -> Result<(), anyhow::Error>;
104    /// Remove a keypair from the keystore by its address.
105    async fn remove(&mut self, address: SuiAddress) -> Result<(), anyhow::Error>;
106    /// Return an array of `PublicKey`, consisting of every public key in the keystore.
107    fn entries(&self) -> Vec<PublicKey>;
108    /// Return `SuiKeyPair` for the given address.
109    fn export(&self, address: &SuiAddress) -> Result<&SuiKeyPair, anyhow::Error>;
110
111    /// Sign a hash with the keypair corresponding to the given address.
112    async fn sign_hashed(
113        &self,
114        address: &SuiAddress,
115        msg: &[u8],
116    ) -> Result<Signature, signature::Error>;
117
118    /// Sign a message with the keypair corresponding to the given address with the given intent.
119    async fn sign_secure<T>(
120        &self,
121        address: &SuiAddress,
122        msg: &T,
123        intent: Intent,
124    ) -> Result<Signature, signature::Error>
125    where
126        T: Serialize + Sync;
127
128    /// Return an array of `SuiAddress`, consisting of every address in the keystore.
129    fn addresses(&self) -> Vec<SuiAddress> {
130        self.entries().iter().map(|k| k.into()).collect()
131    }
132    /// Return an array of (&SuiAddress, &Alias), consisting of every address and its corresponding alias in the keystore.
133    fn addresses_with_alias(&self) -> Vec<(&SuiAddress, &Alias)>;
134    /// Return an array of `Alias`, consisting of every alias stored. Each alias corresponds to a key.
135    fn aliases(&self) -> Vec<&Alias>;
136    /// Mutable references to each alias in the keystore.
137    fn aliases_mut(&mut self) -> Vec<&mut Alias>;
138    /// Get the alias of an address.
139    fn get_alias(&self, address: &SuiAddress) -> Result<String, anyhow::Error>;
140    /// Check if an alias exists by its name
141    fn alias_exists(&self, alias: &str) -> bool {
142        self.aliases().iter().any(|a| a.alias == alias)
143    }
144
145    /// Get address by its identity: a type which is either an address or an alias.
146    fn get_by_identity(&self, key_identity: &KeyIdentity) -> Result<SuiAddress, anyhow::Error> {
147        match key_identity {
148            KeyIdentity::Address(addr) => Ok(*addr),
149            KeyIdentity::Alias(alias) => Ok(*self
150                .addresses_with_alias()
151                .iter()
152                .find(|(_, a)| a.alias == *alias)
153                .ok_or_else(|| anyhow!("Cannot resolve alias {alias} to an address"))?
154                .0),
155        }
156    }
157
158    /// Returns an alias string. Optional string can be passed, checks for duplicates
159    fn create_alias(&self, alias: Option<String>) -> Result<String, anyhow::Error>;
160
161    /// Updates the alias of an existing keypair.
162    async fn update_alias(
163        &mut self,
164        old_alias: &str,
165        new_alias: Option<&str>,
166    ) -> Result<String, anyhow::Error>;
167
168    // Internal function. Use update_alias instead
169    fn update_alias_value(
170        &mut self,
171        old_alias: &str,
172        new_alias: Option<&str>,
173    ) -> Result<String, anyhow::Error> {
174        if !self.aliases().iter().any(|a| a.alias == old_alias) {
175            bail!("The provided alias {old_alias} does not exist");
176        }
177        let new_alias_name = self.create_alias(new_alias.map(str::to_string))?;
178        for a in self.aliases_mut() {
179            if a.alias == old_alias {
180                let pk = &a.public_key_base64;
181                *a = Alias {
182                    alias: new_alias_name.clone(),
183                    public_key_base64: pk.clone(),
184                };
185            }
186        }
187        Ok(new_alias_name)
188    }
189
190    /// Import from a mnemonic phrase.
191    async fn import_from_mnemonic(
192        &mut self,
193        phrase: &str,
194        key_scheme: SignatureScheme,
195        derivation_path: Option<DerivationPath>,
196        alias: Option<String>,
197    ) -> Result<SuiAddress, anyhow::Error> {
198        let mnemonic = Mnemonic::from_phrase(phrase, Language::English)
199            .map_err(|e| anyhow::anyhow!("Invalid mnemonic phrase: {:?}", e))?;
200        let seed = Seed::new(&mnemonic, "");
201        match derive_key_pair_from_path(seed.as_bytes(), derivation_path, &key_scheme) {
202            Ok((address, kp)) => {
203                self.import(alias, kp).await?;
204                Ok(address)
205            }
206            Err(e) => Err(anyhow!("error getting keypair {:?}", e)),
207        }
208    }
209}

270#[async_trait]
271impl AccountKeystore for FileBasedKeystore {
272    async fn sign_hashed(
273        &self,
274        address: &SuiAddress,
275        msg: &[u8],
276    ) -> Result<Signature, signature::Error> {
277        Ok(Signature::new_hashed(
278            msg,
279            self.keys.get(address).ok_or_else(|| {
280                signature::Error::from_source(format!("Cannot find key for address: [{address}]"))
281            })?,
282        ))
283    }
284    async fn sign_secure<T>(
285        &self,
286        address: &SuiAddress,
287        msg: &T,
288        intent: Intent,
289    ) -> Result<Signature, signature::Error>
290    where
291        T: Serialize + Sync,
292    {
293        Ok(Signature::new_secure(
294            &IntentMessage::new(intent, msg),
295            self.keys.get(address).ok_or_else(|| {
296                signature::Error::from_source(format!("Cannot find key for address: [{address}]"))
297            })?,
298        ))
299    }
300
301    async fn import(
302        &mut self,
303        alias: Option<String>,
304        keypair: SuiKeyPair,
305    ) -> Result<(), anyhow::Error> {
306        let address: SuiAddress = (&keypair.public()).into();
307        let alias = self.create_alias(alias)?;
308        self.aliases.insert(
309            address,
310            Alias {
311                alias,
312                public_key_base64: keypair.public().encode_base64(),
313            },
314        );
315        self.keys.insert(address, keypair);
316        self.save().await?;
317        Ok(())
318    }
319
320    async fn remove(&mut self, address: SuiAddress) -> Result<(), anyhow::Error> {
321        self.aliases.remove(&address);
322        self.keys.remove(&address);
323        self.save().await?;
324        Ok(())
325    }
326
327    /// Return an array of `Alias`, consisting of every alias and its corresponding public key.
328    fn aliases(&self) -> Vec<&Alias> {
329        self.aliases.values().collect()
330    }
331
332    fn addresses_with_alias(&self) -> Vec<(&SuiAddress, &Alias)> {
333        self.aliases.iter().collect::<Vec<_>>()
334    }
335
336    /// Return an array of `Alias`, consisting of every alias and its corresponding public key.
337    fn aliases_mut(&mut self) -> Vec<&mut Alias> {
338        self.aliases.values_mut().collect()
339    }
340
341    fn entries(&self) -> Vec<PublicKey> {
342        self.keys.values().map(|key| key.public()).collect()
343    }
344
345    /// This function returns an error if the provided alias already exists. If the alias
346    /// has not already been used, then it returns the alias.
347    /// If no alias has been passed, it will generate a new alias.
348    fn create_alias(&self, alias: Option<String>) -> Result<String, anyhow::Error> {
349        match alias {
350            Some(a) if self.alias_exists(&a) => {
351                bail!("Alias {a} already exists. Please choose another alias.")
352            }
353            Some(a) => validate_alias(&a),
354            None => Ok(random_name(
355                &self
356                    .aliases()
357                    .into_iter()
358                    .map(|x| x.alias.to_string())
359                    .collect::<HashSet<_>>(),
360            )),
361        }
362    }
363
364    /// Get the alias if it exists, or return an error if it does not exist.
365    fn get_alias(&self, address: &SuiAddress) -> Result<String, anyhow::Error> {
366        match self.aliases.get(address) {
367            Some(alias) => Ok(alias.alias.clone()),
368            None => bail!("Cannot find alias for address {address}"),
369        }
370    }
371
372    fn export(&self, address: &SuiAddress) -> Result<&SuiKeyPair, anyhow::Error> {
373        match self.keys.get(address) {
374            Some(key) => Ok(key),
375            None => Err(anyhow!("Cannot find key for address: [{address}]")),
376        }
377    }
378
379    /// Updates an old alias to the new alias and saves it to the alias file.
380    /// If the new_alias is None, it will generate a new random alias.
381    async fn update_alias(
382        &mut self,
383        old_alias: &str,
384        new_alias: Option<&str>,
385    ) -> Result<String, anyhow::Error> {
386        let new_alias_name = self.update_alias_value(old_alias, new_alias)?;
387        self.save_aliases().await?;
388        Ok(new_alias_name)
389    }
390}

392impl FileBasedKeystore {
393    pub fn load_or_create(path: &PathBuf) -> Result<Self, anyhow::Error> {
394        let keys = if path.exists() {
395            #[cfg(unix)]
396            let _ = set_reduced_file_permissions(path).inspect_err(|error| {
397                eprintln!(
398                    "[{}]: while attempting to ensure reduced file permissions on '{}'. Cannot set \
399                    permissions for keystore file. Error: {error}",
400                    "warning".bold().yellow(),
401                    path.display(),
402                );
403            });
404
405            let reader =
406                BufReader::new(fs::File::open(path).with_context(|| {
407                    format!("Cannot open the keystore file: {}", path.display())
408                })?);
409            let kp_strings: Vec<String> = serde_json::from_reader(reader).with_context(|| {
410                format!("Cannot deserialize the keystore file: {}", path.display(),)
411            })?;
412            kp_strings
413                .iter()
414                .map(|kpstr| {
415                    let key = SuiKeyPair::decode_base64(kpstr);
416                    key.map(|k| (SuiAddress::from(&k.public()), k))
417                })
418                .collect::<Result<BTreeMap<_, _>, _>>()
419                .map_err(|e| anyhow!("Invalid keystore file: {}. {}", path.display(), e))?
420        } else {
421            BTreeMap::new()
422        };
423
424        // check aliases
425        let mut aliases_path = path.clone();
426        aliases_path.set_extension(ALIASES_FILE_EXTENSION);
427
428        let aliases = if aliases_path.exists() {
429            let reader = BufReader::new(fs::File::open(&aliases_path).with_context(|| {
430                format!(
431                    "Cannot open aliases file in keystore: {}",
432                    aliases_path.display()
433                )
434            })?);
435
436            let aliases: Vec<Alias> = serde_json::from_reader(reader).with_context(|| {
437                format!(
438                    "Cannot deserialize aliases file in keystore: {}",
439                    aliases_path.display(),
440                )
441            })?;
442
443            aliases
444                .into_iter()
445                .map(|alias| {
446                    let key = PublicKey::decode_base64(&alias.public_key_base64);
447                    key.map(|k| (Into::<SuiAddress>::into(&k), alias))
448                })
449                .collect::<Result<BTreeMap<_, _>, _>>()
450                .map_err(|e| {
451                    anyhow!(
452                        "Invalid aliases file in keystore: {}. {}",
453                        aliases_path.display(),
454                        e
455                    )
456                })?
457        } else if keys.is_empty() {
458            BTreeMap::new()
459        } else {
460            let names: Vec<String> = random_names(HashSet::new(), keys.len());
461            let aliases = keys
462                .iter()
463                .zip(names)
464                .map(|((sui_address, skp), alias)| {
465                    let public_key_base64 = skp.public().encode_base64();
466                    (
467                        *sui_address,
468                        Alias {
469                            alias,
470                            public_key_base64,
471                        },
472                    )
473                })
474                .collect::<BTreeMap<_, _>>();
475            let aliases_store = serde_json::to_string_pretty(&aliases.values().collect::<Vec<_>>())
476                .with_context(|| {
477                    format!(
478                        "Cannot serialize aliases to file in keystore: {}",
479                        aliases_path.display()
480                    )
481                })?;
482
483            fs::write(aliases_path, aliases_store)?;
484            aliases
485        };
486
487        Ok(Self {
488            keys,
489            aliases,
490            path: Some(path.to_path_buf()),
491        })
492    }
493
494    pub fn set_path(&mut self, path: &Path) {
495        self.path = Some(path.to_path_buf());
496    }
497
498    pub async fn save_aliases(&self) -> Result<(), anyhow::Error> {
499        if let Some(path) = &self.path {
500            let aliases_store =
501                serde_json::to_string_pretty(&self.aliases.values().collect::<Vec<_>>())
502                    .with_context(|| {
503                        format!(
504                            "Cannot serialize aliases to file in keystore: {}",
505                            path.display()
506                        )
507                    })?;
508
509            let mut aliases_path = path.clone();
510            aliases_path.set_extension(ALIASES_FILE_EXTENSION);
511            // no reactor for tokio::fs::write in simtest, so we use spawn_blocking
512            tokio::task::spawn_blocking(move || fs::write(aliases_path, aliases_store))
513                .await?
514                .with_context(|| format!("Cannot write aliases to file: {}", path.display()))?;
515        }
516        Ok(())
517    }
518
519    /// Keys saved as Base64 with 33 bytes `flag || privkey` ($BASE64_STR).
520    /// To see Bech32 format encoding, use `sui keytool export $SUI_ADDRESS` where
521    /// $SUI_ADDRESS can be found with `sui keytool list`. Or use `sui keytool convert $BASE64_STR`
522    pub async fn save_keystore(&self) -> Result<(), anyhow::Error> {
523        if let Some(path) = &self.path {
524            let store = serde_json::to_string_pretty(
525                &self
526                    .keys
527                    .values()
528                    .map(|k| k.encode_base64())
529                    .collect::<Vec<_>>(),
530            )
531            .with_context(|| format!("Cannot serialize keystore to file: {}", path.display()))?;
532            let keystore_path = path.clone();
533            // no reactor for tokio::fs::write in simtest, so we use spawn_blocking
534            tokio::task::spawn_blocking(move || {
535                let ret = fs::write(&keystore_path, store);
536                #[cfg(unix)]
537                if ret.is_ok() {
538                    let _ = set_reduced_file_permissions(&keystore_path).inspect_err(|error| {
539                        eprintln!(
540                            "While attempting to set reduced file permissions on '{}'. Cannot set \
541                                permissions for keystore file. Error: {error}",
542                            keystore_path.display()
543                        );
544                    });
545                }
546                ret
547            })
548            .await?
549            .with_context(|| format!("Cannot write keystore to file: {}", path.display()))?;
550        }
551        Ok(())
552    }
553
554    pub async fn save(&self) -> Result<(), anyhow::Error> {
555        self.save_aliases().await?;
556        self.save_keystore().await?;
557        Ok(())
558    }
559
560    pub fn key_pairs(&self) -> Vec<&SuiKeyPair> {
561        self.keys.values().collect()
562    }
563}

565#[cfg(unix)]
566fn set_reduced_file_permissions(path: impl AsRef<Path>) -> Result<(), anyhow::Error> {
567    let path = path.as_ref();
568    let metadata = fs::metadata(path)?;
569    let mode = metadata.permissions().mode();
570    if mode & 0o177 != 0 {
571        fs::set_permissions(path, fs::Permissions::from_mode(0o600)).with_context(|| {
572            format!(
573                "Cannot set permissions for keystore file: {}.",
574                path.display(),
575            )
576        })?;
577    }
578    Ok(())
579}

679    /// has not already been used, then it returns the alias.
680    /// If no alias has been passed, it will generate a new alias.
681    fn create_alias(&self, alias: Option<String>) -> Result<String, anyhow::Error> {
682        match alias {
683            Some(a) if self.alias_exists(&a) => {
684                bail!("Alias {a} already exists. Please choose another alias.")
685            }
686            Some(a) => validate_alias(&a),
687            None => Ok(random_name(
688                &self
689                    .aliases()
690                    .iter()
691                    .map(|x| x.alias.to_string())
692                    .collect::<HashSet<_>>(),
693            )),
694        }
695    }
696
697    fn aliases_mut(&mut self) -> Vec<&mut Alias> {
698        self.aliases.values_mut().collect()
699    }
700
701    /// Updates an old alias to the new alias. If the new_alias is None,
702    /// it will generate a new random alias.
703    async fn update_alias(
704        &mut self,
705        old_alias: &str,
706        new_alias: Option<&str>,
707    ) -> Result<String, anyhow::Error> {
708        self.update_alias_value(old_alias, new_alias)
709    }
710}

712impl InMemKeystore {
713    pub fn new_insecure_for_tests(initial_key_number: usize) -> Self {
714        let mut rng = StdRng::from_seed([0; 32]);
715        let keys = (0..initial_key_number)
716            .map(|_| get_key_pair_from_rng(&mut rng))
717            .map(|(ad, k)| (ad, SuiKeyPair::Ed25519(k)))
718            .collect::<BTreeMap<SuiAddress, SuiKeyPair>>();
719
720        let aliases = keys
721            .iter()
722            .zip(random_names(HashSet::new(), keys.len()))
723            .map(|((sui_address, skp), alias)| {
724                let public_key_base64 = skp.public().encode_base64();
725                (
726                    *sui_address,
727                    Alias {
728                        alias,
729                        public_key_base64,
730                    },
731                )
732            })
733            .collect::<BTreeMap<_, _>>();
734
735        Self { aliases, keys }
736    }
737}

739pub(crate) fn validate_alias(alias: &str) -> Result<String, anyhow::Error> {
740    let re = Regex::new(r"^[A-Za-z][A-Za-z0-9-_\.]*$")
741        .map_err(|_| anyhow!("Cannot build the regex needed to validate the alias naming"))?;
742    let alias = alias.trim();
743    ensure!(
744        re.is_match(alias),
745        "Invalid alias. A valid alias must start with a letter and can contain only letters, digits, hyphens (-), dots (.), or underscores (_)."
746    );
747    Ok(alias.to_string())
748}
