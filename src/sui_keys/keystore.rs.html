70#[async_trait]
71#[enum_dispatch]
72pub trait AccountKeystore: Send + Sync {
73    /// Generate a new keypair and add it into the keystore.
74    async fn generate(
75        &mut self,
76        alias: Option<String>,
77        generate_options: GenerateOptions,
78    ) -> Result<GeneratedKey, anyhow::Error> {
79        let (key_scheme, derivation_path, word_length) = match generate_options {
80            GenerateOptions::Default => (SignatureScheme::ED25519, None, None),
81            GenerateOptions::Local(opts) => {
82                (opts.key_scheme, opts.derivation_path, opts.word_length)
83            }
84            GenerateOptions::ExternalSigner(_) => {
85                return Err(anyhow!(
86                    "Generating new keypair is not supported for this keystore type"
87                ));
88            }
89        };
90
91        let (address, kp, scheme, _phrase) =
92            generate_new_key(key_scheme, derivation_path, word_length)?;
93        let public_key = kp.public();
94        self.import(alias, kp).await?;
95        Ok(GeneratedKey {
96            address,
97            public_key,
98            scheme,
99        })
100    }
101
102    /// Import a keypair into the keystore from a `SuiKeyPair` and optional alias.
103    async fn import(
104        &mut self,
105        alias: Option<String>,
106        keypair: SuiKeyPair,
107    ) -> Result<(), anyhow::Error>;
108    /// Remove a keypair from the keystore by its address.
109    async fn remove(&mut self, address: SuiAddress) -> Result<(), anyhow::Error>;
110    /// Return an array of `PublicKey`, consisting of every public key in the keystore.
111    fn entries(&self) -> Vec<PublicKey>;
112    /// Return `SuiKeyPair` for the given address.
113    fn export(&self, address: &SuiAddress) -> Result<&SuiKeyPair, anyhow::Error>;
114
115    /// Sign a hash with the keypair corresponding to the given address.
116    async fn sign_hashed(
117        &self,
118        address: &SuiAddress,
119        msg: &[u8],
120    ) -> Result<Signature, signature::Error>;
121
122    /// Sign a message with the keypair corresponding to the given address with the given intent.
123    async fn sign_secure<T>(
124        &self,
125        address: &SuiAddress,
126        msg: &T,
127        intent: Intent,
128    ) -> Result<Signature, signature::Error>
129    where
130        T: Serialize + Sync;
131
132    /// Return an array of `SuiAddress`, consisting of every address in the keystore.
133    fn addresses(&self) -> Vec<SuiAddress> {
134        self.entries().iter().map(|k| k.into()).collect()
135    }
136    /// Return an array of (&SuiAddress, &Alias), consisting of every address and its corresponding alias in the keystore.
137    fn addresses_with_alias(&self) -> Vec<(&SuiAddress, &Alias)>;
138    /// Return an array of `Alias`, consisting of every alias stored. Each alias corresponds to a key.
139    fn aliases(&self) -> Vec<&Alias>;
140    /// Mutable references to each alias in the keystore.
141    fn aliases_mut(&mut self) -> Vec<&mut Alias>;
142    /// Get the alias of an address.
143    fn get_alias(&self, address: &SuiAddress) -> Result<String, anyhow::Error>;
144    /// Check if an alias exists by its name
145    fn alias_exists(&self, alias: &str) -> bool {
146        self.aliases().iter().any(|a| a.alias == alias)
147    }
148
149    /// Get address by its identity: a type which is either an address or an alias.
150    fn get_by_identity(&self, key_identity: &KeyIdentity) -> Result<SuiAddress, anyhow::Error> {
151        match key_identity {
152            KeyIdentity::Address(addr) => Ok(*addr),
153            KeyIdentity::Alias(alias) => Ok(*self
154                .addresses_with_alias()
155                .iter()
156                .find(|(_, a)| a.alias == *alias)
157                .ok_or_else(|| anyhow!("Cannot resolve alias {alias} to an address"))?
158                .0),
159        }
160    }
161
162    /// Returns an alias string. Optional string can be passed, checks for duplicates
163    fn create_alias(&self, alias: Option<String>) -> Result<String, anyhow::Error>;
164
165    /// Updates the alias of an existing keypair.
166    async fn update_alias(
167        &mut self,
168        old_alias: &str,
169        new_alias: Option<&str>,
170    ) -> Result<String, anyhow::Error>;
171
172    // Internal function. Use update_alias instead
173    fn update_alias_value(
174        &mut self,
175        old_alias: &str,
176        new_alias: Option<&str>,
177    ) -> Result<String, anyhow::Error> {
178        if !self.aliases().iter().any(|a| a.alias == old_alias) {
179            bail!("The provided alias {old_alias} does not exist");
180        }
181        let new_alias_name = self.create_alias(new_alias.map(str::to_string))?;
182        for a in self.aliases_mut() {
183            if a.alias == old_alias {
184                let pk = &a.public_key_base64;
185                *a = Alias {
186                    alias: new_alias_name.clone(),
187                    public_key_base64: pk.clone(),
188                };
189            }
190        }
191        Ok(new_alias_name)
192    }
193
194    /// Import from a mnemonic phrase.
195    async fn import_from_mnemonic(
196        &mut self,
197        phrase: &str,
198        key_scheme: SignatureScheme,
199        derivation_path: Option<DerivationPath>,
200        alias: Option<String>,
201    ) -> Result<SuiAddress, anyhow::Error> {
202        let mnemonic = Mnemonic::from_phrase(phrase, Language::English)
203            .map_err(|e| anyhow::anyhow!("Invalid mnemonic phrase: {:?}", e))?;
204        let seed = Seed::new(&mnemonic, "");
205        match derive_key_pair_from_path(seed.as_bytes(), derivation_path, &key_scheme) {
206            Ok((address, kp)) => {
207                self.import(alias, kp).await?;
208                Ok(address)
209            }
210            Err(e) => Err(anyhow!("error getting keypair {:?}", e)),
211        }
212    }
213}

230            }
231        }
232    }

274#[async_trait]
275impl AccountKeystore for FileBasedKeystore {
276    async fn sign_hashed(
277        &self,
278        address: &SuiAddress,
279        msg: &[u8],
280    ) -> Result<Signature, signature::Error> {
281        Ok(Signature::new_hashed(
282            msg,
283            self.keys.get(address).ok_or_else(|| {
284                signature::Error::from_source(format!("Cannot find key for address: [{address}]"))
285            })?,
286        ))
287    }
288    async fn sign_secure<T>(
289        &self,
290        address: &SuiAddress,
291        msg: &T,
292        intent: Intent,
293    ) -> Result<Signature, signature::Error>
294    where
295        T: Serialize + Sync,
296    {
297        Ok(Signature::new_secure(
298            &IntentMessage::new(intent, msg),
299            self.keys.get(address).ok_or_else(|| {
300                signature::Error::from_source(format!("Cannot find key for address: [{address}]"))
301            })?,
302        ))
303    }
304
305    async fn import(
306        &mut self,
307        alias: Option<String>,
308        keypair: SuiKeyPair,
309    ) -> Result<(), anyhow::Error> {
310        let address: SuiAddress = (&keypair.public()).into();
311        let alias = self.create_alias(alias)?;
312        self.aliases.insert(
313            address,
314            Alias {
315                alias,
316                public_key_base64: keypair.public().encode_base64(),
317            },
318        );
319        self.keys.insert(address, keypair);
320        self.save().await?;
321        Ok(())
322    }
323
324    async fn remove(&mut self, address: SuiAddress) -> Result<(), anyhow::Error> {
325        self.aliases.remove(&address);
326        self.keys.remove(&address);
327        self.save().await?;
328        Ok(())
329    }
330
331    /// Return an array of `Alias`, consisting of every alias and its corresponding public key.
332    fn aliases(&self) -> Vec<&Alias> {
333        self.aliases.values().collect()
334    }
335
336    fn addresses_with_alias(&self) -> Vec<(&SuiAddress, &Alias)> {
337        self.aliases.iter().collect::<Vec<_>>()
338    }
339
340    /// Return an array of `Alias`, consisting of every alias and its corresponding public key.
341    fn aliases_mut(&mut self) -> Vec<&mut Alias> {
342        self.aliases.values_mut().collect()
343    }
344
345    fn entries(&self) -> Vec<PublicKey> {
346        self.keys.values().map(|key| key.public()).collect()
347    }
348
349    /// This function returns an error if the provided alias already exists. If the alias
350    /// has not already been used, then it returns the alias.
351    /// If no alias has been passed, it will generate a new alias.
352    fn create_alias(&self, alias: Option<String>) -> Result<String, anyhow::Error> {
353        match alias {
354            Some(a) if self.alias_exists(&a) => {
355                bail!("Alias {a} already exists. Please choose another alias.")
356            }
357            Some(a) => validate_alias(&a),
358            None => Ok(random_name(
359                &self
360                    .aliases()
361                    .into_iter()
362                    .map(|x| x.alias.to_string())
363                    .collect::<HashSet<_>>(),
364            )),
365        }
366    }
367
368    /// Get the alias if it exists, or return an error if it does not exist.
369    fn get_alias(&self, address: &SuiAddress) -> Result<String, anyhow::Error> {
370        match self.aliases.get(address) {
371            Some(alias) => Ok(alias.alias.clone()),
372            None => bail!("Cannot find alias for address {address}"),
373        }
374    }
375
376    fn export(&self, address: &SuiAddress) -> Result<&SuiKeyPair, anyhow::Error> {
377        match self.keys.get(address) {
378            Some(key) => Ok(key),
379            None => Err(anyhow!("Cannot find key for address: [{address}]")),
380        }
381    }
382
383    /// Updates an old alias to the new alias and saves it to the alias file.
384    /// If the new_alias is None, it will generate a new random alias.
385    async fn update_alias(
386        &mut self,
387        old_alias: &str,
388        new_alias: Option<&str>,
389    ) -> Result<String, anyhow::Error> {
390        let new_alias_name = self.update_alias_value(old_alias, new_alias)?;
391        self.save_aliases().await?;
392        Ok(new_alias_name)
393    }
394}

396impl FileBasedKeystore {
397    pub fn load_or_create(path: &PathBuf) -> Result<Self, anyhow::Error> {
398        let keys = if path.exists() {
399            #[cfg(unix)]
400            let _ = set_reduced_file_permissions(path).inspect_err(|error| {
401                eprintln!(
402                    "[{}]: while attempting to ensure reduced file permissions on '{}'. Cannot set \
403                    permissions for keystore file. Error: {error}",
404                    "warning".bold().yellow(),
405                    path.display(),
406                );
407            });
408
409            let reader =
410                BufReader::new(std::fs::File::open(path).with_context(|| {
411                    format!("Cannot open the keystore file: {}", path.display())
412                })?);
413            let kp_strings: Vec<String> = serde_json::from_reader(reader).with_context(|| {
414                format!("Cannot deserialize the keystore file: {}", path.display(),)
415            })?;
416            kp_strings
417                .iter()
418                .map(|kpstr| {
419                    let key = SuiKeyPair::decode_base64(kpstr);
420                    key.map(|k| (SuiAddress::from(&k.public()), k))
421                })
422                .collect::<Result<BTreeMap<_, _>, _>>()
423                .map_err(|e| anyhow!("Invalid keystore file: {}. {}", path.display(), e))?
424        } else {
425            BTreeMap::new()
426        };
427
428        // check aliases
429        let mut aliases_path = path.clone();
430        aliases_path.set_extension(ALIASES_FILE_EXTENSION);
431
432        let aliases = if aliases_path.exists() {
433            let reader = BufReader::new(std::fs::File::open(&aliases_path).with_context(|| {
434                format!(
435                    "Cannot open aliases file in keystore: {}",
436                    aliases_path.display()
437                )
438            })?);
439
440            let aliases: Vec<Alias> = serde_json::from_reader(reader).with_context(|| {
441                format!(
442                    "Cannot deserialize aliases file in keystore: {}",
443                    aliases_path.display(),
444                )
445            })?;
446
447            aliases
448                .into_iter()
449                .map(|alias| {
450                    let key = PublicKey::decode_base64(&alias.public_key_base64);
451                    key.map(|k| (Into::<SuiAddress>::into(&k), alias))
452                })
453                .collect::<Result<BTreeMap<_, _>, _>>()
454                .map_err(|e| {
455                    anyhow!(
456                        "Invalid aliases file in keystore: {}. {}",
457                        aliases_path.display(),
458                        e
459                    )
460                })?
461        } else if keys.is_empty() {
462            BTreeMap::new()
463        } else {
464            let names: Vec<String> = random_names(HashSet::new(), keys.len());
465            let aliases = keys
466                .iter()
467                .zip(names)
468                .map(|((sui_address, skp), alias)| {
469                    let public_key_base64 = skp.public().encode_base64();
470                    (
471                        *sui_address,
472                        Alias {
473                            alias,
474                            public_key_base64,
475                        },
476                    )
477                })
478                .collect::<BTreeMap<_, _>>();
479            let aliases_store = serde_json::to_string_pretty(&aliases.values().collect::<Vec<_>>())
480                .with_context(|| {
481                    format!(
482                        "Cannot serialize aliases to file in keystore: {}",
483                        aliases_path.display()
484                    )
485                })?;
486
487            std::fs::write(aliases_path, aliases_store)?;
488            aliases
489        };
490
491        Ok(Self {
492            keys,
493            aliases,
494            path: Some(path.to_path_buf()),
495        })
496    }
497
498    pub fn set_path(&mut self, path: &Path) {
499        self.path = Some(path.to_path_buf());
500    }
501
502    pub async fn save_aliases(&self) -> Result<(), anyhow::Error> {
503        if let Some(path) = &self.path {
504            let aliases_store =
505                serde_json::to_string_pretty(&self.aliases.values().collect::<Vec<_>>())
506                    .with_context(|| {
507                        format!(
508                            "Cannot serialize aliases to file in keystore: {}",
509                            path.display()
510                        )
511                    })?;
512
513            let mut aliases_path = path.clone();
514            aliases_path.set_extension(ALIASES_FILE_EXTENSION);
515            // no reactor for tokio::fs::write in simtest, so we use spawn_blocking
516            tokio::task::spawn_blocking(move || std::fs::write(aliases_path, aliases_store))
517                .await?
518                .with_context(|| format!("Cannot write aliases to file: {}", path.display()))?;
519        }
520        Ok(())
521    }
522
523    /// Keys saved as Base64 with 33 bytes `flag || privkey` ($BASE64_STR).
524    /// To see Bech32 format encoding, use `sui keytool export $SUI_ADDRESS` where
525    /// $SUI_ADDRESS can be found with `sui keytool list`. Or use `sui keytool convert $BASE64_STR`
526    pub async fn save_keystore(&self) -> Result<(), anyhow::Error> {
527        if let Some(path) = &self.path {
528            let store = serde_json::to_string_pretty(
529                &self
530                    .keys
531                    .values()
532                    .map(|k| k.encode_base64())
533                    .collect::<Vec<_>>(),
534            )
535            .with_context(|| format!("Cannot serialize keystore to file: {}", path.display()))?;
536            let keystore_path = path.clone();
537            // no reactor for tokio::fs::write in simtest, so we use spawn_blocking
538            tokio::task::spawn_blocking(move || {
539                let ret = std::fs::write(&keystore_path, store);
540                #[cfg(unix)]
541                if ret.is_ok() {
542                    let _ = set_reduced_file_permissions(&keystore_path).inspect_err(|error| {
543                        eprintln!(
544                            "While attempting to set reduced file permissions on '{}'. Cannot set \
545                                permissions for keystore file. Error: {error}",
546                            keystore_path.display()
547                        );
548                    });
549                }
550                ret
551            })
552            .await?
553            .with_context(|| format!("Cannot write keystore to file: {}", path.display()))?;
554        }
555        Ok(())
556    }
557
558    pub async fn save(&self) -> Result<(), anyhow::Error> {
559        self.save_aliases().await?;
560        self.save_keystore().await?;
561        Ok(())
562    }
563
564    pub fn key_pairs(&self) -> Vec<&SuiKeyPair> {
565        self.keys.values().collect()
566    }
567}

569#[cfg(unix)]
570fn set_reduced_file_permissions(path: impl AsRef<Path>) -> Result<(), anyhow::Error> {
571    let path = path.as_ref();
572    let metadata = fs::metadata(path)?;
573    let mode = metadata.permissions().mode();
574    if mode & 0o177 != 0 {
575        fs::set_permissions(path, fs::Permissions::from_mode(0o600)).with_context(|| {
576            format!(
577                "Cannot set permissions for keystore file: {}.",
578                path.display(),
579            )
580        })?;
581    }
582    Ok(())
583}

682    /// This function returns an error if the provided alias already exists. If the alias
683    /// has not already been used, then it returns the alias.
684    /// If no alias has been passed, it will generate a new alias.
685    fn create_alias(&self, alias: Option<String>) -> Result<String, anyhow::Error> {
686        match alias {
687            Some(a) if self.alias_exists(&a) => {
688                bail!("Alias {a} already exists. Please choose another alias.")
689            }
690            Some(a) => validate_alias(&a),
691            None => Ok(random_name(
692                &self
693                    .aliases()
694                    .iter()
695                    .map(|x| x.alias.to_string())
696                    .collect::<HashSet<_>>(),
697            )),
698        }
699    }
700
701    fn aliases_mut(&mut self) -> Vec<&mut Alias> {
702        self.aliases.values_mut().collect()
703    }
704
705    /// Updates an old alias to the new alias. If the new_alias is None,
706    /// it will generate a new random alias.
707    async fn update_alias(
708        &mut self,
709        old_alias: &str,
710        new_alias: Option<&str>,
711    ) -> Result<String, anyhow::Error> {
712        self.update_alias_value(old_alias, new_alias)
713    }
714}

716impl InMemKeystore {
717    pub fn new_insecure_for_tests(initial_key_number: usize) -> Self {
718        let mut rng = StdRng::from_seed([0; 32]);
719        let keys = (0..initial_key_number)
720            .map(|_| get_key_pair_from_rng(&mut rng))
721            .map(|(ad, k)| (ad, SuiKeyPair::Ed25519(k)))
722            .collect::<BTreeMap<SuiAddress, SuiKeyPair>>();
723
724        let aliases = keys
725            .iter()
726            .zip(random_names(HashSet::new(), keys.len()))
727            .map(|((sui_address, skp), alias)| {
728                let public_key_base64 = skp.public().encode_base64();
729                (
730                    *sui_address,
731                    Alias {
732                        alias,
733                        public_key_base64,
734                    },
735                )
736            })
737            .collect::<BTreeMap<_, _>>();
738
739        Self { aliases, keys }
740    }
741}

743pub(crate) fn validate_alias(alias: &str) -> Result<String, anyhow::Error> {
744    let re = Regex::new(r"^[A-Za-z][A-Za-z0-9-_\.]*$")
745        .map_err(|_| anyhow!("Cannot build the regex needed to validate the alias naming"))?;
746    let alias = alias.trim();
747    ensure!(
748        re.is_match(alias),
749        "Invalid alias. A valid alias must start with a letter and can contain only letters, digits, hyphens (-), dots (.), or underscores (_)."
750    );
751    Ok(alias.to_string())
752}
