184/// Return a random name formatted as first-second from a list of strings.
185///
186/// The main purpose of this function is to generate random aliases for addresses.
187pub fn random_name(conflicts: &HashSet<String>) -> String {
188    let mut rng = thread_rng();
189    // as long as the generated name is in the list of conflicts,
190    // we try to find a different name that is not in the list yet
191    loop {
192        let output = generate(&mut rng);
193        if !conflicts.contains(&output) {
194            return output;
195        }
196    }
197}
198
199/// Return a unique collection of names.
200pub fn random_names(mut conflicts: HashSet<String>, output_size: usize) -> Vec<String> {
201    let mut names = Vec::with_capacity(output_size);
202    names.resize_with(output_size, || {
203        let name = random_name(&conflicts);
204        conflicts.insert(name.clone());
205        name
206    });
207    names
208}
209
210// Generate a random name as a pair from left and right string arrays
211fn generate(rng: &mut ThreadRng) -> String {
212    let left_idx = rng.gen_range(0..LEFT_LENGTH);
213    let right_idx = rng.gen_range(0..RIGHT_LENGTH);
214    format!(
215        "{}-{}",
216        LEFT_NAMES.get(left_idx).unwrap(),
217        RIGHT_NAMES.get(right_idx).unwrap()
218    )
219}
