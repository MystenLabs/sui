4//! Request-scoped BigTable facade.
5//!
6//! Downstream read APIs on this wrapper acquire a permit from a single
7//! `Arc<Semaphore>` before dispatching to the underlying
8//! `sui_kvstore::BigTableClient`.
9//! For streaming methods, the permit is bundled into the returned stream's
10//! state and released only when the stream is dropped. Pipeline helpers drain
11//! those streams inside one chunk future so live permit-holding streams do not
12//! cross stage boundaries. For non-streaming methods, the permit is held for
13//! the duration of the call.
14//!
15//! Bitmap scans are intentionally not gated by this downstream semaphore:
16//! their fanout is bounded by `max_bitmap_filter_literals`, and filtered list
17//! requests must still be able to run bitmap discovery with a low downstream
18//! request budget.
19//!
20//! The wrapper is constructed per request (`BigTableClient::new(client, capacity, ...)`)
21//! and threaded down through the pipeline.
22//!
23//! Each successful acquire wait is labeled with `stage` so the wait histogram
24//! can be sliced by call-site type. Per-request peak permits and successful
25//! acquisition count are recorded once on `BigTableClientContext` drop.
26

71/// Histograms for tuning the request-scoped `BigTableClient` semaphore.
72/// Cardinality is bounded: 3 methods × 5 stages = 15 series each for
73/// `permit_wait_ms` and `permit_hold_ms`, and 3 series each for the request-end
74/// `permits_peak` and `ops_total` histograms.
75#[derive(Clone)]
76pub(crate) struct Metrics {
77    /// Time spent waiting for one limiter permit, recorded once per successful
78    /// `acquire_owned()` call.
79    pub permit_wait_ms: HistogramVec,
80    /// Time one limiter permit is held from acquisition through release.
81    pub permit_hold_ms: HistogramVec,
82    /// Peak in-use permit count observed during a single request, recorded
83    /// once when the request's `BigTableClient` is dropped.
84    pub permits_peak: HistogramVec,
85    /// Total limiter acquisitions issued by a single request, recorded once
86    /// when the request's `BigTableClient` is dropped.
87    pub ops_total: HistogramVec,
88}

146/// Per-request shared state. Cloned along with `BigTableClient` (each clone is
147/// just an `Arc` bump). When the last clone is dropped, `Drop` records the
148/// per-request peak and ops-total histograms.
149pub(crate) struct BigTableClientContext {
150    metrics: Arc<Metrics>,
151    method: &'static str,
152    active_permits: AtomicUsize,
153    peak_used: AtomicUsize,
154    ops_count: AtomicUsize,
155}
156
157impl Drop for BigTableClientContext {
158    fn drop(&mut self) {
159        let peak = *self.peak_used.get_mut() as f64;
160        let ops = *self.ops_count.get_mut() as f64;
161        self.metrics
162            .permits_peak
163            .with_label_values(&[self.method])
164            .observe(peak);
165        self.metrics
166            .ops_total
167            .with_label_values(&[self.method])
168            .observe(ops);
169    }
170}

361    async fn acquire(&self, stage: &'static str) -> Result<LimitedPermit, RpcError> {
362        acquire_limited(self.limiter.clone(), self.context.clone(), stage)
363            .await
364            .map_err(|_| RpcError::new(tonic::Code::Internal, "request concurrency limiter closed"))

367    async fn tx_seq_checkpoint(&self, tx_seq: u64) -> Result<u64, RpcError> {
368        self.resolve_tx_checkpoints(&[tx_seq])
369            .await?
370            .into_iter()
371            .next()
372            .map(|(_, cp)| cp)
373            .ok_or_else(|| {
374                RpcError::new(
375                    tonic::Code::Internal,
376                    format!("missing checkpoint mapping for transaction sequence number {tx_seq}"),
377                )
378            })

381    /// Build a resolver closure for tx-bitmap watermarks (identity
382    /// decode). Hand directly to [`crate::pipeline::resolve_scan_watermarks`].
383    pub(crate) fn tx_wm_resolver(
384        &self,
385        direction: ScanDirection,
386    ) -> impl Fn(u64) -> WmResolverFut + Send + 'static {
387        let client = self.clone();
388        move |position| {
389            let client = client.clone();
390            Box::pin(async move {
391                let lookup_tx_seq = if direction.is_ascending() {
392                    match position.checked_sub(1) {
393                        Some(tx_seq) => tx_seq,
394                        None => return Ok(None),
395                    }
396                } else {
397                    position
398                };
399                client
400                    .tx_seq_checkpoint(lookup_tx_seq)
401                    .await
402                    .map(Some)
403                    .map_err(|e| ScanStop::Fault(anyhow::Error::new(e)))
404            })
405        }
406    }

408    /// Build a resolver closure for event-bitmap watermarks. Hand directly to
409    /// [`crate::pipeline::resolve_scan_watermarks`].
410    pub(crate) fn event_wm_resolver(
411        &self,
412        direction: ScanDirection,
413    ) -> impl Fn(EventPosition) -> WmResolverFut + Send + 'static {
414        let client = self.clone();
415        move |position| {
416            let client = client.clone();
417            Box::pin(async move {
418                let lookup_tx_seq = if direction.is_ascending() {
419                    if position.event_index > 0 {
420                        position.tx_seq
421                    } else {
422                        match position.tx_seq.checked_sub(1) {
423                            Some(tx_seq) => tx_seq,
424                            None => return Ok(None),
425                        }
426                    }
427                } else {
428                    position.tx_seq
429                };
430                client
431                    .tx_seq_checkpoint(lookup_tx_seq)
432                    .await
433                    .map(Some)
434                    .map_err(|e| ScanStop::Fault(anyhow::Error::new(e)))
435            })
436        }
437    }

445struct LimitedPermit {
446    _permit: OwnedSemaphorePermit,
447    context: Arc<BigTableClientContext>,
448    stage: &'static str,
449    acquired_at: Instant,
450}
451
452impl Drop for LimitedPermit {
453    fn drop(&mut self) {
454        let hold_ms = self.acquired_at.elapsed().as_secs_f64() * 1000.0;
455        self.context
456            .metrics
457            .permit_hold_ms
458            .with_label_values(&[self.context.method, self.stage])
459            .observe(hold_ms);
460        record_release(&self.context);
461    }
462}
463
464async fn acquire_limited(
465    limiter: Arc<Semaphore>,
466    context: Arc<BigTableClientContext>,
467    stage: &'static str,
468) -> Result<LimitedPermit, tokio::sync::AcquireError> {
469    let start = Instant::now();
470    let permit = limiter.acquire_owned().await?;
471    let acquired_at = Instant::now();
472    record_acquired(&context, stage, acquired_at.duration_since(start));
473    Ok(LimitedPermit {
474        _permit: permit,
475        context,
476        stage,
477        acquired_at,
478    })
479}
480
481fn record_acquired(context: &BigTableClientContext, stage: &'static str, wait: Duration) {
482    let wait_ms = wait.as_secs_f64() * 1000.0;
483    context
484        .metrics
485        .permit_wait_ms
486        .with_label_values(&[context.method, stage])
487        .observe(wait_ms);
488    context.ops_count.fetch_add(1, Ordering::Relaxed);
489    let active = context.active_permits.fetch_add(1, Ordering::AcqRel) + 1;
490    context.peak_used.fetch_max(active, Ordering::Relaxed);
491}
492
493fn record_release(context: &BigTableClientContext) {
494    let previous = context
495        .active_permits
496        .fetch_update(Ordering::AcqRel, Ordering::Acquire, |active| {
497            active.checked_sub(1)
498        })
499        .unwrap_or(0);
500    debug_assert!(previous > 0, "limited client permit release underflow");
501}

503/// Wrap a stream so that `permit` is held until the stream is dropped (either
504/// by full drain, early termination by the consumer, or cancellation
505/// propagating through `Drop`). The returned stream is a thin pass-through —
506/// items and errors flow through unchanged.
507fn gate_stream<T>(
508    permit: impl Send + 'static,
509    inner: BoxStream<'static, Result<T, anyhow::Error>>,
510) -> BoxStream<'static, Result<T, anyhow::Error>>
511where
512    T: Send + 'static,
513{
514    async_stream::try_stream! {
515        let _permit = permit;
516        futures::pin_mut!(inner);
517        while let Some(row) = inner.next().await {
518            yield row?;
519        }
520    }
521    .boxed()
522}

537    /// `gate_stream` should hold the permit until the wrapped stream is
538    /// dropped. With a capacity-1 semaphore, the wrapped stream's existence
539    /// blocks any other acquire, even between yields.
540    #[tokio::test(start_paused = true)]
541    async fn gate_stream_holds_permit_for_full_drain() {
542        let limiter = Arc::new(Semaphore::new(1));
543        let permit = limiter.clone().acquire_owned().await.unwrap();
544        assert_eq!(limiter.available_permits(), 0);
545
546        let inner: BoxStream<'static, Result<u64, anyhow::Error>> =
547            stream::iter(vec![Ok(1u64), Ok(2), Ok(3)]).boxed();
548        let mut wrapped = gate_stream(permit, inner);
549
550        for expected in 1u64..=3 {
551            let item = wrapped.try_next().await.unwrap().unwrap();
552            assert_eq!(item, expected);
553            assert_eq!(
554                limiter.available_permits(),
555                0,
556                "permit should still be held mid-drain"
557            );
558        }
559        assert!(wrapped.try_next().await.unwrap().is_none());
560        // Dropping the stream releases the permit.
561        drop(wrapped);
562        assert_eq!(limiter.available_permits(), 1);
563    }
564
565    /// Dropping the wrapped stream early — before the inner stream is
566    /// drained — should release the permit immediately.
567    #[tokio::test(start_paused = true)]
568    async fn gate_stream_releases_permit_on_early_drop() {
569        let limiter = Arc::new(Semaphore::new(1));
570        let permit = limiter.clone().acquire_owned().await.unwrap();
571        assert_eq!(limiter.available_permits(), 0);
572
573        // Wrap an infinite stream so we can drop in the middle.
574        let counter = Arc::new(AtomicUsize::new(0));
575        let counter_clone = counter.clone();
576        let inner: BoxStream<'static, Result<u64, anyhow::Error>> =
577            stream::repeat_with(move || {
578                let n = counter_clone.fetch_add(1, Ordering::SeqCst);
579                Ok(n as u64)
580            })
581            .boxed();
582        let mut wrapped = gate_stream(permit, inner);
583
584        let _ = wrapped.try_next().await.unwrap().unwrap();
585        let _ = wrapped.try_next().await.unwrap().unwrap();
586        assert_eq!(limiter.available_permits(), 0);
587
588        drop(wrapped);
589        // The drop is synchronous; the permit is released immediately.
590        // (`tokio::sync::Semaphore` doesn't need yield-to-runtime to observe
591        // the release.)
592        tokio::time::sleep(Duration::from_millis(0)).await;
593        assert_eq!(limiter.available_permits(), 1);
594    }

596    /// Inner errors should propagate through unchanged; the permit should

638    /// passthrough to `sui_kvstore::BigTableClient`.
639    fn test_context(metrics: Arc<Metrics>, method: &'static str) -> Arc<BigTableClientContext> {
640        Arc::new(BigTableClientContext {
641            metrics,
642            method,
643            active_permits: AtomicUsize::new(0),
644            peak_used: AtomicUsize::new(0),
645            ops_count: AtomicUsize::new(0),
646        })
647    }

649    #[test]
650    fn record_acquired_tracks_active_peak_and_release() {
651        let (metrics, registry) = Metrics::for_testing();
652        let context = test_context(metrics, "list_transactions");
653
654        record_acquired(&context, stage::TRANSACTIONS, Duration::from_millis(7));
655        assert_eq!(context.ops_count.load(Ordering::Relaxed), 1);
656        assert_eq!(context.active_permits.load(Ordering::Relaxed), 1);
657        assert_eq!(context.peak_used.load(Ordering::Relaxed), 1);
658
659        record_acquired(&context, stage::OBJECTS, Duration::from_millis(3));
660        assert_eq!(context.ops_count.load(Ordering::Relaxed), 2);
661        assert_eq!(context.active_permits.load(Ordering::Relaxed), 2);
662        assert_eq!(context.peak_used.load(Ordering::Relaxed), 2);
663
664        record_release(&context);
665        assert_eq!(context.active_permits.load(Ordering::Relaxed), 1);
666        assert_eq!(context.peak_used.load(Ordering::Relaxed), 2);
667        record_release(&context);
668        assert_eq!(context.active_permits.load(Ordering::Relaxed), 0);
669
670        drop(context);
671        let families = registry.gather();
672        assert_eq!(
673            histogram_sample_count(family(&families, "kv_rpc_bigtable_permit_wait_ms")),
674            2
675        );
676        assert_eq!(
677            histogram_sample_count(family(&families, "kv_rpc_bigtable_permits_peak")),
678            1,
679            "permits_peak observed once on context drop"
680        );
681        assert_eq!(
682            histogram_sample_count(family(&families, "kv_rpc_bigtable_ops_total")),
683            1,
684            "ops_total observed once on context drop"
685        );
686    }

689    async fn limited_permit_records_hold_once_on_drop() {
690        let (metrics, registry) = Metrics::for_testing();
691        let context = test_context(metrics, "list_transactions");

713            labels

719    #[test]
720    fn stage_labels_are_bounded_and_counted() {
721        let (metrics, registry) = Metrics::for_testing();
722        let context = test_context(metrics, "list_events");
723
724        record_acquired(&context, stage::TRANSACTIONS, Duration::from_millis(1));
725        record_acquired(&context, stage::OBJECTS, Duration::from_millis(2));
726        record_release(&context);
727        record_release(&context);
728        drop(context);
729
730        let families = registry.gather();
731        assert_eq!(
732            histogram_sample_count(family(&families, "kv_rpc_bigtable_permits_peak")),
733            1,
734            "permits_peak observed once on context drop"
735        );
736        assert_eq!(
737            histogram_sample_count(family(&families, "kv_rpc_bigtable_ops_total")),
738            1,
739            "ops_total observed once on context drop"
740        );
741        assert_eq!(
742            histogram_sample_count(family(&families, "kv_rpc_bigtable_permit_wait_ms")),
743            2,
744            "one permit_wait observation per successful acquire"
745        );
746
747        let permit_wait = family(&families, "kv_rpc_bigtable_permit_wait_ms");
748        let by_stage: std::collections::BTreeMap<String, u64> = permit_wait
749            .get_metric()
750            .iter()
751            .map(|m| {
752                let stage = m
753                    .get_label()
754                    .iter()
755                    .find(|l| l.name() == "stage")
756                    .unwrap()
757                    .value()
758                    .to_string();
759                (stage, m.get_histogram().get_sample_count())
760            })
761            .collect();
762        assert_eq!(by_stage.get("transactions"), Some(&1));
763        assert_eq!(by_stage.get("objects"), Some(&1));
764    }
