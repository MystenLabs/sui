74#[derive(Clone)]
75pub(crate) struct KvRpcMetrics {
76    bigtable_limiter: Arc<BigTableLimiterMetrics>,
77    response_render_latency_ms: HistogramVec,
78    stream_item_yield_wait_ms: HistogramVec,
79    bitmap_buckets_evaluated: HistogramVec,
80}
81
82impl KvRpcMetrics {
83    fn new(registry: &Registry) -> Arc<Self> {
84        Arc::new(Self {
85            bigtable_limiter: BigTableLimiterMetrics::new(registry),
86            response_render_latency_ms: register_histogram_vec_with_registry!(
87                "kv_rpc_response_render_latency_ms",
88                "Wall time spent rendering one v2alpha response item.",
89                &["method"],
90                prometheus::exponential_buckets(0.01, 2.0, 18).unwrap(),
91                registry,
92            )
93            .unwrap(),
94            stream_item_yield_wait_ms: register_histogram_vec_with_registry!(
95                "kv_rpc_stream_item_yield_wait_ms",
96                "Wall time from yielding one v2alpha response item until the stream is polled again.",
97                &["method"],
98                prometheus::exponential_buckets(0.01, 2.0, 18).unwrap(),
99                registry,
100            )
101            .unwrap(),
102            bitmap_buckets_evaluated: register_histogram_vec_with_registry!(
103                "kv_rpc_bitmap_buckets_evaluated",
104                "Buckets evaluated against the per-request bitmap scan budget. Caps at the configured budget; actual backend bucket reads may exceed it by up to max_bitmap_filter_literals (one extra fetched-and-discarded bucket per leaf stream at exhaustion). Tail near the per-request cap means clients are hitting SCAN_LIMIT and paging.",
105                &["method"],
106                // Exponential 1..2048: many requests evaluate few buckets
107                // (small or empty scans), a tail saturates at the default 1k cap.
108                prometheus::exponential_buckets(1.0, 2.0, 12).unwrap(),
109                registry,
110            )
111            .unwrap(),
112        })
113    }
114
115    fn observe_response_render(&self, method: &'static str, elapsed: std::time::Duration) {
116        self.response_render_latency_ms
117            .with_label_values(&[method])
118            .observe(elapsed.as_secs_f64() * 1000.0);
119    }
120
121    fn observe_stream_item_yield_wait(&self, method: &'static str, elapsed: std::time::Duration) {
122        self.stream_item_yield_wait_ms
123            .with_label_values(&[method])
124            .observe(elapsed.as_secs_f64() * 1000.0);
125    }
126
127    fn observe_bitmap_buckets_evaluated(&self, method: &'static str, buckets: u64) {
128        self.bitmap_buckets_evaluated
129            .with_label_values(&[method])
130            .observe(buckets as f64);
131    }
132}

157impl KvRpcServer {
158    pub async fn new(
159        instance_id: String,
160        project_id: Option<String>,
161        app_profile_id: Option<String>,
162        channel_timeout: Option<Duration>,
163        server_version: Option<ServerVersion>,
164        registry: &Registry,
165        credentials_path: Option<String>,
166        pool_config: PoolConfig,
167        service_info_watermark_pipelines: Vec<&'static str>,
168        ledger_history: LedgerHistoryConfig,
169        request_bigtable_concurrency: usize,
170        stages: StagesConfig,
171    ) -> anyhow::Result<Self> {
172        ledger_history.validate()?;
173        let mut client = BigTableClient::new_remote_with_credentials(
174            instance_id,
175            project_id,
176            false,
177            channel_timeout,
178            None,
179            "sui-kv-rpc".to_string(),
180            Some(registry),
181            app_profile_id,
182            pool_config,
183            credentials_path,
184        )
185        .await?;
186        let genesis = client
187            .get_checkpoints(&[0])
188            .await?
189            .pop()
190            .expect("failed to fetch genesis checkpoint from the KV store");
191        let summary = genesis.summary.expect("genesis checkpoint missing summary");
192        let chain_id = ChainIdentifier::from(summary.digest());
193        let metrics = KvRpcMetrics::new(registry);
194        Self::init(
195            client,
196            chain_id,
197            server_version,
198            service_info_watermark_pipelines,
199            metrics,
200            ledger_history,
201            request_bigtable_concurrency,
202            stages,
203        )
204    }

223    /// Construct a KvRpcServer backed by a local BigTable emulator with explicit
224    /// ledger-history and per-stage configuration. Tests use this to exercise
225    /// scan-budget / chunk-size behaviour (e.g. a small `tx_seq_digest` chunk
226    /// size) against a modest synthetic dataset.
227    pub async fn new_local_with_config(
228        host: String,
229        instance_id: String,
230        server_version: Option<ServerVersion>,
231        ledger_history: LedgerHistoryConfig,
232        request_bigtable_concurrency: usize,
233        stages: StagesConfig,
234    ) -> anyhow::Result<Self> {
235        let client = BigTableClient::new_local(host, instance_id).await?;
236        // Emulator/test path: metrics are inert (no scrape endpoint), but the
237        // request-scoped BigTable wrapper still expects a handle.
238        let metrics = KvRpcMetrics::new(&Registry::default());
239        Self::init(
240            client,
241            ChainIdentifier::from(sui_types::digests::CheckpointDigest::default()),
242            server_version,
243            default_service_info_watermark_pipelines(false),
244            metrics,
245            ledger_history,
246            request_bigtable_concurrency,
247            stages,
248        )
249    }

251    fn init(
252        client: BigTableClient,
253        chain_id: ChainIdentifier,
254        server_version: Option<ServerVersion>,
255        service_info_watermark_pipelines: Vec<&'static str>,
256        metrics: Arc<KvRpcMetrics>,
257        ledger_history: LedgerHistoryConfig,
258        request_bigtable_concurrency: usize,
259        stages: StagesConfig,
260    ) -> anyhow::Result<Self> {
261        ensure!(
262            !service_info_watermark_pipelines.is_empty(),
263            "at least one service info watermark pipeline must be configured"
264        );
265        ledger_history.validate()?;
266
267        let cache = Arc::new(RwLock::new(None));
268
269        let package_store: Arc<dyn PackageStore> = Arc::new(PackageStoreWithLruCache::new(
270            BigTablePackageStore::new(client.clone()),
271        ));
272        let package_resolver = Arc::new(Resolver::new(package_store));
273
274        let server = Self {
275            chain_id,
276            client,
277            server_version,
278            service_info_watermark_pipelines,
279            cache,
280            package_resolver,
281            metrics,
282            ledger_history,
283            request_bigtable_concurrency,
284            stages,
285        };
286
287        let server_clone = server.clone();
288        tokio::spawn(async move {
289            loop {
290                match v2::get_service_info(
291                    server_clone.client.clone(),
292                    server_clone.chain_id,
293                    server_clone.server_version.clone(),
294                    &server_clone.service_info_watermark_pipelines,
295                )
296                .await
297                {
298                    Ok(info) => {
299                        let mut cache = server_clone.cache.write().await;
300                        *cache = Some(info);
301                    }
302                    Err(e) => error!("Failed to update service info cache: {:?}", e),
303                }
304                sleep(Duration::from_millis(10)).await;
305            }
306        });
307
308        Ok(server)
309    }

311    /// Start this server as a tonic gRPC service on the given address.
312    /// Returns a `Service` handle for lifecycle management.
313    pub async fn start_service(
314        self,
315        listen_address: SocketAddr,
316        config: ServerConfig,
317    ) -> anyhow::Result<sui_futures::service::Service> {
318        use mysten_network::callback::CallbackLayer;
319        use sui_rpc_api::{
320            RpcMetrics, RpcMetricsMakeCallbackHandler, grpc_method_paths_from_file_descriptor_sets,
321        };
322
323        let mut builder = Server::builder();
324
325        if let Some(identity) = config.tls_identity {
326            builder = builder.tls_config(ServerTlsConfig::new().identity(identity))?;
327        }
328
329        // Single source of truth for every encoded FileDescriptorSet that
330        // backs a gRPC service mounted below. Consumed by both the
331        // reflection services and the metrics allowlist so they cannot drift
332        // out of sync.
333        let enable_experimental_query_apis = config.enable_experimental_query_apis;
334        let mut file_descriptor_sets: Vec<&'static [u8]> = vec![
335            sui_rpc_api::proto::google::protobuf::FILE_DESCRIPTOR_SET,
336            sui_rpc_api::proto::google::rpc::FILE_DESCRIPTOR_SET,
337            sui_rpc::proto::sui::rpc::v2::FILE_DESCRIPTOR_SET,
338        ];
339        if enable_experimental_query_apis {
340            file_descriptor_sets.push(sui_rpc::proto::sui::rpc::v2alpha::FILE_DESCRIPTOR_SET);
341        }
342
343        let registry = config.metrics_registry.unwrap_or_default();
344        let grpc_method_allowlist = Arc::new(grpc_method_paths_from_file_descriptor_sets(
345            &file_descriptor_sets,
346        )?);
347        let mut router = builder
348            .layer(CallbackLayer::new(
349                RpcMetricsMakeCallbackHandler::with_grpc_method_allowlist(
350                    Arc::new(RpcMetrics::new(&registry)),
351                    grpc_method_allowlist,
352                ),
353            ))
354            .add_service(LedgerServiceServer::new(self.clone()));
355
356        if enable_experimental_query_apis {
357            router = router.add_service(KvLedgerServiceServer::new(self));
358        }
359
360        if config.enable_reflection {
361            let mut reflection_v1_builder = tonic_reflection::server::Builder::configure();
362            let mut reflection_v1alpha_builder = tonic_reflection::server::Builder::configure();
363            for &fds in &file_descriptor_sets {
364                reflection_v1_builder =
365                    reflection_v1_builder.register_encoded_file_descriptor_set(fds);
366                reflection_v1alpha_builder =
367                    reflection_v1alpha_builder.register_encoded_file_descriptor_set(fds);
368            }
369            router = router
370                .add_service(reflection_v1_builder.build_v1()?)
371                .add_service(reflection_v1alpha_builder.build_v1alpha()?);
372        }
373
374        let (tx, rx) = tokio::sync::oneshot::channel::<()>();
375        let server_future = router.serve_with_shutdown(listen_address, async {
376            let _ = rx.await;
377        });
378
379        let service = sui_futures::service::Service::new()
380            .spawn(async move {
381                server_future.await?;
382                Ok(())
383            })
384            .with_shutdown_signal(async move {
385                let _ = tx.send(());
386            });
387
388        Ok(service)
389    }
390}

392pub fn default_service_info_watermark_pipelines(
393    enable_experimental_query_apis: bool,
394) -> Vec<&'static str> {
395    let mut pipelines = DEFAULT_SERVICE_INFO_WATERMARK_PIPELINES.to_vec();
396    if enable_experimental_query_apis {
397        pipelines.extend_from_slice(&EXPERIMENTAL_QUERY_SERVICE_INFO_WATERMARK_PIPELINES);
398    }
399    pipelines
