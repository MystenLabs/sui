283    pub async fn new(
284        instance_id: String,
285        project_id: Option<String>,
286        app_profile_id: Option<String>,
287        channel_timeout: Option<Duration>,
288        server_version: Option<ServerVersion>,
289        registry: &Registry,
290        credentials_path: Option<String>,
291        pool_config: PoolConfig,
292        service_info_watermark_pipelines: Vec<&'static str>,
293        ledger_history: LedgerHistoryConfig,
294        request_bigtable_concurrency: usize,
295        stages: StagesConfig,
296    ) -> anyhow::Result<Self> {
297        ledger_history.validate()?;
298        let mut client = BigTableClient::new_remote_with_credentials(
299            instance_id,
300            project_id,
301            false,
302            channel_timeout,
303            None,
304            "sui-kv-rpc".to_string(),
305            Some(registry),
306            app_profile_id,
307            pool_config,
308            credentials_path,
309        )
310        .await?;
311        let genesis = client
312            .get_checkpoints(&[0])
313            .await?
314            .pop()
315            .expect("failed to fetch genesis checkpoint from the KV store");
316        let summary = genesis.summary.expect("genesis checkpoint missing summary");
317        let chain_id = ChainIdentifier::from(summary.digest());
318        let metrics = KvRpcMetrics::new(registry);
319        Self::init(
320            client,
321            chain_id,
322            server_version,
323            service_info_watermark_pipelines,
324            metrics,
325            ledger_history,
326            request_bigtable_concurrency,
327            stages,
328        )
329    }

348    /// Construct a KvRpcServer backed by a local BigTable emulator with explicit
349    /// ledger-history and per-stage configuration. Tests use this to exercise
350    /// scan-budget / chunk-size behaviour (e.g. a small `tx_seq_digest` chunk
351    /// size) against a modest synthetic dataset.
352    pub async fn new_local_with_config(
353        host: String,
354        instance_id: String,
355        server_version: Option<ServerVersion>,
356        ledger_history: LedgerHistoryConfig,
357        request_bigtable_concurrency: usize,
358        stages: StagesConfig,
359    ) -> anyhow::Result<Self> {
360        let client = BigTableClient::new_local(host, instance_id).await?;
361        // Emulator/test path: metrics are inert (no scrape endpoint), but the
362        // request-scoped BigTable wrapper still expects a handle.
363        let metrics = KvRpcMetrics::new(&Registry::default());
364        Self::init(
365            client,
366            ChainIdentifier::from(sui_types::digests::CheckpointDigest::default()),
367            server_version,
368            default_service_info_watermark_pipelines(false),
369            metrics,
370            ledger_history,
371            request_bigtable_concurrency,
372            stages,
373        )
374    }

376    fn init(
377        client: BigTableClient,
378        chain_id: ChainIdentifier,
379        server_version: Option<ServerVersion>,
380        service_info_watermark_pipelines: Vec<&'static str>,
381        metrics: Arc<KvRpcMetrics>,
382        ledger_history: LedgerHistoryConfig,
383        request_bigtable_concurrency: usize,
384        stages: StagesConfig,
385    ) -> anyhow::Result<Self> {
386        ensure!(
387            !service_info_watermark_pipelines.is_empty(),
388            "at least one service info watermark pipeline must be configured"
389        );
390        ledger_history.validate()?;
391
392        let cache = Arc::new(RwLock::new(None));
393
394        let package_store: Arc<dyn PackageStore> = Arc::new(PackageStoreWithLruCache::new(
395            BigTablePackageStore::new(client.clone()),
396        ));
397        let package_resolver = Arc::new(Resolver::new(package_store));
398
399        let server = Self {
400            chain_id,
401            client,
402            server_version,
403            service_info_watermark_pipelines,
404            cache,
405            package_resolver,
406            metrics,
407            ledger_history,
408            request_bigtable_concurrency,
409            stages,
410            query_apis_enabled: false,
411        };
412
413        let server_clone = server.clone();
414        tokio::spawn(async move {
415            loop {
416                match v2::get_service_info(
417                    server_clone.client.clone(),
418                    server_clone.chain_id,
419                    server_clone.server_version.clone(),
420                    &server_clone.service_info_watermark_pipelines,
421                )
422                .await
423                {
424                    Ok(info) => {
425                        let mut cache = server_clone.cache.write().await;
426                        *cache = Some(info);
427                    }
428                    Err(e) => error!("Failed to update service info cache: {:?}", e),
429                }
430                sleep(Duration::from_millis(10)).await;
431            }
432        });
433
434        Ok(server)
435    }

447    /// Start this server as a tonic gRPC service on the given address.
448    /// Returns a `Service` handle for lifecycle management.
449    pub async fn start_service(
450        mut self,
451        listen_address: SocketAddr,
452        config: ServerConfig,
453    ) -> anyhow::Result<sui_futures::service::Service> {
454        use sui_http::middleware::callback::CallbackLayer;
455        use sui_rpc_api::{
456            RpcMetrics, RpcMetricsMakeCallbackHandler, grpc_method_paths_from_file_descriptor_sets,
457        };
458
459        let mut builder = Server::builder();
460
461        if let Some(identity) = config.tls_identity {
462            builder = builder.tls_config(ServerTlsConfig::new().identity(identity))?;
463        }
464
465        // Single source of truth for every encoded FileDescriptorSet that
466        // backs a gRPC service mounted below. Consumed by both the
467        // reflection services and the metrics allowlist so they cannot drift
468        // out of sync.
469        self.query_apis_enabled = config.enable_experimental_query_apis;
470        let file_descriptor_sets: Vec<&'static [u8]> = vec![
471            sui_rpc_api::proto::google::protobuf::FILE_DESCRIPTOR_SET,
472            sui_rpc_api::proto::google::rpc::FILE_DESCRIPTOR_SET,
473            sui_rpc::proto::sui::rpc::v2::FILE_DESCRIPTOR_SET,
474        ];
475
476        let registry = config.metrics_registry.unwrap_or_default();
477        let grpc_method_allowlist = Arc::new(grpc_method_paths_from_file_descriptor_sets(
478            &file_descriptor_sets,
479        )?);
480        let mut router = builder
481            .layer(CallbackLayer::new(
482                RpcMetricsMakeCallbackHandler::with_grpc_method_allowlist(
483                    Arc::new(RpcMetrics::new(&registry)),
484                    grpc_method_allowlist,
485                ),
486            ))
487            .layer(
488                mysten_network::request_log::GrpcRequestLogLayer::from_encoded_file_descriptor_sets(
489                    file_descriptor_sets.iter().copied(),
490                )?,
491            )
492            .add_service(ledger_service_with_response_compression(self));
493
494        if config.enable_reflection {
495            let mut reflection_v1_builder = tonic_reflection::server::Builder::configure();
496            let mut reflection_v1alpha_builder = tonic_reflection::server::Builder::configure();
497            for &fds in &file_descriptor_sets {
498                reflection_v1_builder =
499                    reflection_v1_builder.register_encoded_file_descriptor_set(fds);
500                reflection_v1alpha_builder =
501                    reflection_v1alpha_builder.register_encoded_file_descriptor_set(fds);
502            }
503            router = router
504                .add_service(reflection_v1_builder.build_v1()?)
505                .add_service(reflection_v1alpha_builder.build_v1alpha()?);
506        }
507
508        let (tx, rx) = tokio::sync::oneshot::channel::<()>();
509        let server_future = router.serve_with_shutdown(listen_address, async {
510            let _ = rx.await;
511        });
512
513        let service = sui_futures::service::Service::new()
514            .spawn(async move {
515                server_future.await?;
516                Ok(())
517            })
518            .with_shutdown_signal(async move {
519                let _ = tx.send(());
520            });
521
522        Ok(service)
523    }
