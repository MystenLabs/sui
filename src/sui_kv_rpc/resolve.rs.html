4//! Shared BigTable resolution layer: turns streams (or bounded sets) of
5//! checkpoint/transaction identifiers into fully-resolved entities — their
6//! transactions and objects — using the chunked, concurrency-limited pipeline
7//! primitives. Both the v2 point-get handlers and the list handlers
8//! build on these so request-size chunking, stage overlap, and object
9//! deduplication are identical across them.

33/// Stage-C keyed-batch output: a checkpoint paired with a map of just its own
34/// transaction rows (the `pipelined_keyed_batches` per-item map).

36    (u64, CheckpointData),
37    Arc<HashMap<TransactionDigest, TransactionData>>,
38);

40/// Attach a per-item `ObjectMap` to each item in `upstream`. When
41/// `needs_objects` is false every item is paired with an empty map (no
42/// BigTable read); otherwise object keys are derived per item via `keys_of`
43/// and fetched through `pipelined_keyed_batches` — chunk-bounded, concurrent,
44/// and deduplicated by a request-scoped `ObjectCache`, overlapping with the
45/// still-arriving upstream. The cache is owned by the returned stream, so its
46/// in-flight dispatches abort if the consumer drops the stream.
47pub(crate) fn with_object_maps<I, E>(
48    upstream: BoxStream<'static, Result<Watermarked<I>, E>>,
49    client: BigTableClient,
50    objects_stage: ResolvedStageConfig,
51    needs_objects: bool,
52    keys_of: impl Fn(&I) -> Vec<ObjectKey> + Send + 'static,
53) -> BoxStream<'static, Result<Watermarked<(I, ObjectMap)>, E>>
54where
55    I: Send + 'static,
56    E: From<anyhow::Error> + Send + 'static,
57{
58    if needs_objects {
59        let object_cache = ObjectCache::new(Arc::new(BigTableObjectFetcher::new(client)));
60        let with_keys = upstream
61            .map_ok(move |m| {
62                m.map_item(|item| {
63                    let keys = keys_of(&item);
64                    (item, keys)
65                })
66            })
67            .boxed();
68        pipelined_keyed_batches(
69            with_keys,
70            objects_stage.chunk_size,
71            objects_stage.chunk_size,
72            objects_stage.concurrency,
73            move |keys| {
74                let object_cache = object_cache.clone();
75                async move {
76                    object_cache
77                        .get_many(keys)
78                        .await
79                        .map_err(|e| E::from(anyhow::Error::new(e)))
80                }
81            },
82        )
83        .boxed()
84    } else {
85        let empty: ObjectMap = Arc::new(HashMap::new());
86        upstream
87            .map_ok(move |m| {
88                let empty = empty.clone();
89                m.map_item(move |item| (item, empty.clone()))
90            })
91            .boxed()
92    }
93}

95/// Resolve a stream of `(cp_seq, CheckpointData)` into
96/// `(cp_seq, cp_data, txs, objects)`. Stage C fetches each chunk's
97/// transactions; stage D attaches their objects. Shared by `get_checkpoint`
98pub(crate) fn resolve_checkpoints<E>(
99    client: BigTableClient,
100    read_mask: &FieldMaskTree,
101    transactions_stage: ResolvedStageConfig,
102    objects_stage: ResolvedStageConfig,
103    cp_data_stream: BoxStream<'static, Result<Watermarked<(u64, CheckpointData)>, E>>,
104) -> BoxStream<'static, Result<Watermarked<ResolvedCp>, E>>
105where
106    E: From<anyhow::Error> + Send + 'static,
107{
108    let tx_columns: Arc<[&'static str]> = list_transactions_columns(read_mask).into();
109    let needs_objects = read_mask.contains(Checkpoint::OBJECTS_FIELD);
110
111    // Stage C: (cp_seq, CheckpointData) -> + Vec<TransactionData>. Derive each
112    // checkpoint's transaction digests and fetch them through
113    // `pipelined_keyed_batches` — chunk-bounded and concurrent, batching at
114    // `chunk_size` keys per BigTable request.
115    let with_keys = cp_data_stream
116        .map(|res: Result<Watermarked<(u64, CheckpointData)>, E>| {
117            let m = res?;
118            Ok::<_, E>(match m {
119                Watermarked::Item((cp_seq, cp_data)) => {
120                    let keys = cp_data
121                        .contents
122                        .as_ref()
123                        .ok_or_else(|| {
124                            E::from(anyhow::anyhow!(
125                                "checkpoint {cp_seq} contents column missing"
126                            ))
127                        })?
128                        .iter()
129                        .map(|d| d.transaction)
130                        .collect::<Vec<TransactionDigest>>();
131                    Watermarked::Item(((cp_seq, cp_data), keys))
132                }
133                Watermarked::Watermark(p) => Watermarked::Watermark(p),
134            })
135        })
136        .boxed();
137
138    let cp_with_map_stream = pipelined_keyed_batches(
139        with_keys,
140        transactions_stage.chunk_size,
141        transactions_stage.chunk_size,
142        transactions_stage.concurrency,
143        {
144            let client = client.clone();
145            let columns = tx_columns.clone();
146            move |keys| {
147                let client = client.clone();
148                let columns = columns.clone();
149                async move {
150                    let requested = keys.len();
151                    let stream = fetch_transaction_rows(client, columns, keys)
152                        .await
153                        .map_err(E::from)?;
154                    futures::pin_mut!(stream);
155                    let mut map: HashMap<TransactionDigest, TransactionData> = HashMap::new();
156                    while let Some(row) = stream.next().await {
157                        let (digest, tx) = row.map_err(E::from)?;
158                        map.insert(digest, tx);
159                    }
160                    if map.len() != requested {
161                        return Err(E::from(anyhow::anyhow!(
162                            "resolve_checkpoints: BigTable returned fewer transactions than \
163                             requested ({} of {} rows)",
164                            map.len(),
165                            requested
166                        )));
167                    }
168                    Ok(map)
169                }
170            }
171        },
172    );
173
174    let cp_with_txs_stream = cp_with_map_stream
175        .map(|res: Result<Watermarked<CpTxMap>, E>| {
176            let m = res?;
177            Ok::<_, E>(match m {
178                Watermarked::Item(((cp_seq, cp_data), tx_map)) => {
179                    let contents = cp_data.contents.as_ref().ok_or_else(|| {
180                        E::from(anyhow::anyhow!(
181                            "checkpoint {cp_seq} contents column missing"
182                        ))
183                    })?;
184                    // The per-item map is uniquely owned here — the reassembler
185                    // just built it and handed it over — so move each body out
186                    // in checkpoint-contents order rather than deep-cloning it.
187                    // The `unwrap_or_else` clone is a correctness fallback for
188                    // the (currently impossible) case of a shared map.
189                    let mut tx_map = Arc::try_unwrap(tx_map).unwrap_or_else(|arc| (*arc).clone());
190                    let mut txs: Vec<TransactionData> = Vec::with_capacity(contents.size());
191                    for d in contents.iter() {
192                        let digest = d.transaction;
193                        let tx = tx_map.remove(&digest).ok_or_else(|| {
194                            E::from(anyhow::anyhow!(
195                                "resolve_checkpoints: BigTable returned fewer transactions \
196                                 than requested (checkpoint {cp_seq} missing transaction {digest})"
197                            ))
198                        })?;
199                        txs.push(tx);
200                    }
201                    Watermarked::Item((cp_seq, cp_data, txs))
202                }
203                Watermarked::Watermark(p) => Watermarked::Watermark(p),
204            })
205        })
206        .boxed();
207
208    // Stage D: each cp must see only its own object keys, since
209    // `render_full_checkpoint` folds the whole map into an `ObjectSet`.
210    with_object_maps(
211        cp_with_txs_stream,
212        client,
213        objects_stage,
214        needs_objects,
215        |(_, _, txs): &CpWithTxs| {
216            txs.iter()
217                .flat_map(compute_object_keys)
218                .collect::<BTreeSet<_>>()
219                .into_iter()
220                .collect()
221        },
222    )
223    .map_ok(|m| m.map_item(|((cp_seq, cp_data, txs), objects)| (cp_seq, cp_data, txs, objects)))
224    .boxed()
225}

227/// Resolve a single already-fetched checkpoint (the point-get path): drive
228/// `resolve_checkpoints` over a one-item stream and return the sole result.
229/// `resolve_checkpoints` emits exactly one item per input checkpoint (empty
230/// checkpoints are pre-emitted) or propagates a fetch error, so the stream is
231/// never empty — hence no "missing result" failure mode.
232pub(crate) async fn resolve_checkpoint(
233    client: BigTableClient,
234    read_mask: &FieldMaskTree,
235    transactions_stage: ResolvedStageConfig,
236    objects_stage: ResolvedStageConfig,
237    cp_seq: u64,
238    cp_data: CheckpointData,
239) -> Result<ResolvedCp, RpcError> {
240    let cp_stream =
241        stream::once(async move { Ok::<_, anyhow::Error>(Watermarked::Item((cp_seq, cp_data))) })
242            .boxed();
243    let stream = resolve_checkpoints(
244        client,
245        read_mask,
246        transactions_stage,
247        objects_stage,
248        cp_stream,
249    );
250    futures::pin_mut!(stream);
251    while let Some(item) = stream.next().await {
252        if let Watermarked::Item(resolved) = item.map_err(RpcError::from)? {
253            return Ok(resolved);
254        }
255    }
256    unreachable!("resolve_checkpoints emits exactly one item per input checkpoint")
257}

259/// Resolve `digests` into transactions and the object data needed to render
260/// complete `ExecutedTransaction.objects` sets or annotate effects object
261/// types, keyed by digest for request-order reconstruction. Transaction rows
262/// stream in by chunk and their object reads overlap the transaction fetch. An
263/// empty `digests` issues no BigTable read.
264pub(crate) async fn resolve_transactions(
265    client: BigTableClient,
266    digests: Vec<TransactionDigest>,
267    read_mask: &FieldMaskTree,
268    transactions_stage: ResolvedStageConfig,
269    objects_stage: ResolvedStageConfig,
270) -> Result<HashMap<TransactionDigest, (TransactionData, ObjectMap)>, RpcError> {
271    if digests.is_empty() {
272        return Ok(HashMap::new());
273    }
274    let columns: Arc<[&'static str]> = transaction_columns(read_mask).into();
275    let needs_objects = needs_transaction_objects(read_mask);
276
277    let digest_stream = stream::iter(
278        digests
279            .into_iter()
280            .map(|d| Ok::<_, anyhow::Error>(Watermarked::Item(d))),
281    )
282    .boxed();
283
284    // Stage A: digests -> (digest, TransactionData), chunked & concurrent.
285    let tx_stream = pipelined_chunks(
286        digest_stream,
287        transactions_stage.chunk_size,
288        transactions_stage.concurrency,
289        {
290            let client = client.clone();
291            let columns = columns.clone();
292            move |chunk| fetch_transaction_rows(client.clone(), columns.clone(), chunk)
293        },
294    );
295
296    // Stage B: attach each tx's own objects.
297    let with_objects = with_object_maps(
298        tx_stream,
299        client,
300        objects_stage,
301        needs_objects,
302        |(_, tx): &(TransactionDigest, TransactionData)| {
303            compute_object_keys(tx).into_iter().collect()
304        },
305    );
306    futures::pin_mut!(with_objects);
307
308    let mut out = HashMap::new();
309    while let Some(item) = with_objects.next().await {
310        if let Watermarked::Item(((digest, tx), objects)) = item.map_err(RpcError::from)? {
311            out.insert(digest, (tx, objects));
312        }
313    }
314    Ok(out)
315}

317/// Stage-A/C fetch: fetch one chunk of transaction rows as a stream. An empty
318/// chunk issues no read (see the empty-RowSet full-scan trap guarded in
319/// `sui-kvstore`).
320async fn fetch_transaction_rows(
321    client: BigTableClient,
322    columns: Arc<[&'static str]>,
323    digests: Vec<TransactionDigest>,
324) -> Result<
325    BoxStream<'static, Result<(TransactionDigest, TransactionData), anyhow::Error>>,
326    anyhow::Error,
327> {
328    if digests.is_empty() {
329        return Ok(stream::empty().boxed());
330    }
331    let column_filter = BigTableClient::column_filter(&columns);
332    Ok(client
333        .get_transactions_stream(digests, Some(column_filter))
334        .await?
335        .boxed())
336}

340/// Whether the checkpoint read mask requests transactions or objects (the heavy
341/// path that needs `resolve_checkpoints`).
342pub(crate) fn needs_transactions_or_objects(mask: &FieldMaskTree) -> bool {
343    mask.contains(Checkpoint::TRANSACTIONS_FIELD) || mask.contains(Checkpoint::OBJECTS_FIELD)
344}

346/// Whether a transaction read mask needs object types resolved (i.e. requests
347/// `effects.changed_objects` or `effects.unchanged_consensus_objects`).
348pub(crate) fn needs_object_types(mask: &FieldMaskTree) -> bool {
349    mask.subtree(ExecutedTransaction::EFFECTS_FIELD.name)
350        .is_some_and(|submask| {
351            submask.contains(TransactionEffects::CHANGED_OBJECTS_FIELD.name)
352                || submask.contains(TransactionEffects::UNCHANGED_CONSENSUS_OBJECTS_FIELD.name)
353        })
354}
355
356/// Whether a transaction read mask needs its canonical object set or object
357/// types embedded in effects.
358pub(crate) fn needs_transaction_objects(mask: &FieldMaskTree) -> bool {
359    mask.contains(ExecutedTransaction::OBJECTS_FIELD.name) || needs_object_types(mask)
360}

362/// Canonical object keys referenced by a transaction's inputs and effects.
363/// Callers use these objects for complete transaction object sets and effects
364/// type annotation.
365pub(crate) fn compute_object_keys(source: &TransactionData) -> BTreeSet<ObjectKey> {
366    match (&source.transaction_data, &source.effects) {
367        (Some(tx_data), Some(effects)) => sui_types::storage::get_transaction_object_set(
368            tx_data,
369            effects,
370            &source.unchanged_loaded_runtime_objects,
371        ),
372        _ => BTreeSet::new(),
373    }
374}

376/// Checkpoint-table columns needed for `mask`. Always includes `s` (summary);
377/// adds `sg` (signatures) and `c` (contents) when requested.
378pub(crate) fn checkpoint_columns(mask: &FieldMaskTree) -> Vec<&'static str> {
379    use tables::checkpoints::col;
380    let mut columns = vec![col::SUMMARY];
381    if mask.contains(Checkpoint::SIGNATURE_FIELD) {
382        columns.push(col::SIGNATURES);
383    }
384    if mask.contains(Checkpoint::CONTENTS_FIELD) {
385        columns.push(col::CONTENTS);
386    }
387    columns
388}

390/// Checkpoint-table columns for the list/get checkpoint handlers. The heavy
391/// (transactions/objects) path additionally needs `signatures` (to reconstruct
392/// `CertifiedCheckpointSummary`) and `contents` (to enumerate tx digests).
393pub(crate) fn list_checkpoint_columns(mask: &FieldMaskTree, needs_full: bool) -> Vec<&'static str> {
394    use tables::checkpoints::col;
395    let mut columns = checkpoint_columns(mask);
396    if needs_full {
397        if !columns.contains(&col::CONTENTS) {
398            columns.push(col::CONTENTS);
399        }
400        if !columns.contains(&col::SIGNATURES) {
401            columns.push(col::SIGNATURES);
402        }
403    }
404    columns
405}

407/// Transaction-table columns needed for `mask`. Always includes `cn`/`ts`
408/// (small metadata); adds `td`, `sg`, `ef`, `ev`, `bc`, `ul` as the mask
409/// requires, including all object-data columns when complete transaction
410/// objects or effects object types must be resolved.
411pub(crate) fn transaction_columns(mask: &FieldMaskTree) -> Vec<&'static str> {
412    transaction_columns_for_mask(mask, needs_transaction_objects(mask))
413}
414
415fn transaction_columns_for_mask(mask: &FieldMaskTree, needs_objects: bool) -> Vec<&'static str> {
416    use tables::transactions::col;
417    let mut columns = vec![col::CHECKPOINT_NUMBER, col::TIMESTAMP];
418
419    if mask
420        .subtree(ExecutedTransaction::TRANSACTION_FIELD.name)
421        .is_some()
422    {
423        columns.push(col::DATA);
424    }
425    if mask
426        .subtree(ExecutedTransaction::SIGNATURES_FIELD.name)
427        .is_some()
428    {
429        columns.push(col::SIGNATURES);
430    }
431    if let Some(effects_submask) = mask.subtree(ExecutedTransaction::EFFECTS_FIELD.name) {
432        columns.push(col::EFFECTS);
433        if effects_submask.contains(TransactionEffects::UNCHANGED_LOADED_RUNTIME_OBJECTS_FIELD.name)
434        {
435            columns.push(col::UNCHANGED_LOADED);
436        }
437    }
438    if mask
439        .subtree(ExecutedTransaction::EVENTS_FIELD.name)
440        .is_some()
441    {
442        columns.push(col::EVENTS);
443    }
444    if mask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD.name) {
445        columns.push(col::BALANCE_CHANGES);
446    }
447    if needs_objects {
448        for column in [col::DATA, col::EFFECTS, col::UNCHANGED_LOADED] {
449            if !columns.contains(&column) {
450                columns.push(column);
451            }
452        }
453    }
454
455    columns
456}

458/// Transaction-table columns needed to resolve a checkpoint's transactions. The
459/// merge target (`full_checkpoint_content::ExecutedTransaction`) has non-Option
460/// `transaction`/`effects`, so `td`+`ef` are always fetched; object resolution
461/// also needs `ul`.
462pub(crate) fn list_transactions_columns(mask: &FieldMaskTree) -> Vec<&'static str> {
463    use tables::transactions::col;
464    let mut columns = if let Some(submask) = mask.subtree(Checkpoint::TRANSACTIONS_FIELD.name) {
465        transaction_columns_for_mask(&submask, needs_object_types(&submask))
466    } else {
467        // Baseline metadata columns even if the proto `transactions` field
468        // isn't requested; we still need the rows to compute object keys.
469        vec![col::CHECKPOINT_NUMBER, col::TIMESTAMP]
470    };
471    // Required to construct the merge target faithfully.
472    for c in [col::DATA, col::EFFECTS] {
473        if !columns.contains(&c) {
474            columns.push(c);
475        }
476    }
477    if mask.contains(Checkpoint::OBJECTS_FIELD) && !columns.contains(&col::UNCHANGED_LOADED) {
478        columns.push(col::UNCHANGED_LOADED);
479    }
480    columns
481}

566    /// Drive the real `resolve_checkpoints` stream over `cps` and collect the
567    /// emitted checkpoint items (dropping watermarks).
568    async fn run_resolver(
569        client: BigTableClient,
570        read_mask: &FieldMaskTree,
571        cps: Vec<(u64, CheckpointData)>,
572    ) -> Result<Vec<ResolvedCp>, RpcError> {
573        let tx_stage = ResolvedStageConfig {
574            chunk_size: TX_CHUNK_SIZE,
575            concurrency: 4,
576        };
577        let obj_stage = ResolvedStageConfig {
578            chunk_size: 100,
579            concurrency: 1,
580        };
581        let input = stream::iter(
582            cps.into_iter()
583                .map(|cp| Ok::<_, RpcError>(Watermarked::Item(cp))),
584        )
585        .boxed();
586        let stream = resolve_checkpoints(client, read_mask, tx_stage, obj_stage, input);
587        futures::pin_mut!(stream);
588        let mut out = Vec::new();
589        while let Some(item) = stream.next().await {
590            if let Watermarked::Item(resolved) = item? {
591                out.push(resolved);
592            }
593        }
594        Ok(out)
595    }

606    #[tokio::test]
607    async fn resolve_checkpoints_splits_fat_checkpoint_transaction_reads() {
608        let (mock, client, _handle) = setup().await;
609        // One checkpoint with more transactions than the stage `chunk_size`, so
610        // the pipeline must split its digests across multiple capped BigTable
611        // requests (the objects stage batches the same way). The backend
612        // `MAX_TX_DIGESTS_PER_REQUEST` clamp is a separate safety net covered by
613        // the `sui-kvstore` `get_transactions_stream_*` tests.
614        let n = TX_CHUNK_SIZE * 2 + 1;
615        let digests: Vec<_> = (0..n as u64).map(digest).collect();
616        for d in &digests {

618        }

621            .await
622            .unwrap();
623

625        let (_, _, txs, _) = &out[0];

629        assert!(
630            calls.len() >= 2,
631            "fat checkpoint should split into >=2 ReadRows calls, got {}",

636                c.row_keys.len() <= TX_CHUNK_SIZE,

638                c.row_keys.len()
639            );
640        }
641        let total: usize = calls.iter().map(|c| c.row_keys.len()).sum();
642        assert_eq!(total, n, "every digest fetched exactly once");
