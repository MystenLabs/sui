18pub const READ_MASK_DEFAULT: &str = sui_rpc_api::read_mask_defaults::CHECKPOINT;
19
20pub async fn get_checkpoint(
21    mut client: BigTableClient,
22    limited_client: LimitedBigTableClient,
23    stages: &StagesConfig,
24    request: GetCheckpointRequest,
25) -> Result<GetCheckpointResponse, RpcError> {
26    let read_mask = {
27        let read_mask = request
28            .read_mask
29            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
30        read_mask.validate::<Checkpoint>().map_err(|path| {
31            FieldViolation::new("read_mask")
32                .with_description(format!("invalid read_mask path: {path}"))
33                .with_reason(ErrorReason::FieldInvalid)
34        })?;
35        FieldMaskTree::from(read_mask)
36    };
37    let needs_full = resolve::needs_transactions_or_objects(&read_mask);
38    let columns = resolve::list_checkpoint_columns(&read_mask, needs_full);
39    let checkpoint = match request.checkpoint_id {
40        Some(CheckpointId::Digest(digest)) => {
41            let digest = digest.parse::<CheckpointDigest>().map_err(|e| {
42                FieldViolation::new("digest")
43                    .with_description(format!("invalid digest: {e}"))
44                    .with_reason(ErrorReason::FieldInvalid)
45            })?;
46            client
47                .get_checkpoint_by_digest_filtered(digest, Some(&columns))
48                .await?
49                .ok_or(CheckpointNotFoundError::digest(digest.into()))?
50        }
51        Some(CheckpointId::SequenceNumber(sequence_number)) => client
52            .get_checkpoints_filtered(&[sequence_number], Some(&columns))
53            .await?
54            .pop()
55            .ok_or(CheckpointNotFoundError::sequence_number(sequence_number))?,
56        _ => {
57            let sequence_number = client
58                .get_watermark_for_pipelines(&[CHECKPOINTS_PIPELINE])
59                .await?
60                .and_then(|wm| wm.checkpoint_hi_inclusive)
61                .ok_or(CheckpointNotFoundError::sequence_number(0))?;
62            client
63                .get_checkpoints_filtered(&[sequence_number], Some(&columns))
64                .await?
65                .pop()
66                .ok_or(CheckpointNotFoundError::sequence_number(sequence_number))?
67        }
68    };
69
70    let message = if needs_full {
71        let transactions_stage = stages.stage(PipelineStage::Transactions);
72        let objects_stage = stages.stage(PipelineStage::Objects);
73        resolve_full_checkpoint(
74            limited_client,
75            checkpoint,
76            &read_mask,
77            transactions_stage,
78            objects_stage,
79        )
80        .await?
81    } else {
82        render::checkpoint_to_response(checkpoint, &read_mask)?
83    };
84    Ok(GetCheckpointResponse::new(message))
85}

87/// Heavy path: resolve the checkpoint's transactions and (when requested)
88/// objects from BigTable, then render the full proto `Checkpoint`.
89async fn resolve_full_checkpoint(
90    limited_client: LimitedBigTableClient,
91    checkpoint: CheckpointData,
92    read_mask: &FieldMaskTree,
93    transactions_stage: ResolvedStageConfig,
94    objects_stage: ResolvedStageConfig,
95) -> Result<Checkpoint, RpcError> {
96    let cp_seq = checkpoint
97        .summary
98        .as_ref()
99        .map(|s| s.sequence_number)
100        .ok_or_else(|| RpcError::new(tonic::Code::Internal, "checkpoint summary column missing"))?;
101
102    let (_, cp_data, txs, objects) = resolve::resolve_checkpoint(
103        limited_client,
104        read_mask,
105        transactions_stage,
106        objects_stage,
107        cp_seq,
108        checkpoint,
109    )
110    .await?;
111
112    render::render_full_checkpoint(cp_data, txs, objects, read_mask)
113}
