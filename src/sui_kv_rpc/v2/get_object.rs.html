18pub const MAX_BATCH_REQUESTS: usize = 1000;
19
20pub(crate) async fn get_object(
21    mut client: BigTableClient,
22    GetObjectRequest {
23        object_id,
24        version,
25        read_mask,
26        ..
27    }: GetObjectRequest,
28) -> Result<GetObjectResponse, RpcError> {
29    let (requests, read_mask) =
30        validate_get_object_requests(vec![(object_id, version)], read_mask)?;
31    let (object_id, version) = requests[0];
32    let object = match version {
33        Some(version) => client
34            .get_objects(&[ObjectKey(object_id.into(), version.into())])
35            .await?
36            .pop()
37            .ok_or_else(|| ObjectNotFoundError::new_with_version(object_id, version))?,
38        None => client
39            .get_latest_object(&object_id.into())
40            .await?
41            .ok_or_else(|| ObjectNotFoundError::new(object_id))?,
42    };
43    let mut message = Object::default();
44    // TODO: support json read mask
45    message.merge(&object, &read_mask);
46    Ok(GetObjectResponse::new(message))
47}

49pub(crate) async fn batch_get_objects(
50    mut client: BigTableClient,
51    BatchGetObjectsRequest {
52        requests,
53        read_mask,
54        ..
55    }: BatchGetObjectsRequest,
56) -> Result<BatchGetObjectsResponse, RpcError> {
57    if requests.len() > MAX_BATCH_REQUESTS {
58        return Err(RpcError::new(
59            tonic::Code::InvalidArgument,
60            format!("number of batch requests exceed limit of {MAX_BATCH_REQUESTS}"),
61        ));
62    }
63
64    // only batch requests with `object_id` and `exact_version` are supported by the KV store
65    if requests.iter().any(|r| r.version.is_none()) {
66        return Err(FieldViolation::new("version")
67            .with_reason(ErrorReason::FieldInvalid)
68            .with_description("KV store supports batch requests with exact object versioning")
69            .into());
70    }
71    let requests = requests
72        .into_iter()
73        .map(|req| (req.object_id, req.version))
74        .collect();
75    let (requests, read_mask) = validate_get_object_requests(requests, read_mask)?;
76    let object_keys: Vec<_> = requests
77        .into_iter()
78        .map(|(object_id, version)| {
79            ObjectKey(
80                object_id.into(),
81                version.expect("invariant's already checked").into(),
82            )
83        })
84        .collect();
85    let response: HashMap<_, _> = client
86        .get_objects(&object_keys)
87        .await?
88        .into_iter()
89        .map(|obj| ((obj.id(), obj.version()), obj))
90        .collect();
91
92    let objects = object_keys
93        .into_iter()
94        .map(|object_key| {
95            if let Some(object) = response.get(&(object_key.0, object_key.1)) {
96                let mut message = Object::default();
97                message.merge(object, &read_mask);
98                return GetObjectResult::new_object(message);
99            }
100            let err: RpcError =
101                ObjectNotFoundError::new_with_version(object_key.0.into(), object_key.1.into())
102                    .into();
103            GetObjectResult::new_error(err.into_status_proto())
104        })
105        .collect();
106
107    Ok(BatchGetObjectsResponse::new(objects))
108}
