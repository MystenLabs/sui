18pub(crate) async fn get_object(
19    mut client: BigTableClient,
20    GetObjectRequest {
21        object_id,
22        version,
23        read_mask,
24        ..
25    }: GetObjectRequest,
26) -> Result<GetObjectResponse, RpcError> {
27    let (requests, read_mask) =
28        validate_get_object_requests(vec![(object_id, version)], read_mask)?;
29    let (object_id, version) = requests[0];
30    let object = match version {
31        Some(version) => client
32            .get_objects(&[ObjectKey(object_id.into(), version.into())])
33            .await?
34            .pop()
35            .ok_or_else(|| ObjectNotFoundError::new_with_version(object_id, version))?,
36        None => client
37            .get_latest_object(&object_id.into())
38            .await?
39            .ok_or_else(|| ObjectNotFoundError::new(object_id))?,
40    };
41    let mut message = Object::default();
42    // TODO: support json read mask
43    message.merge(&object, &read_mask);
44    Ok(GetObjectResponse::new(message))
45}

47pub(crate) async fn batch_get_objects(
48    mut client: BigTableClient,
49    BatchGetObjectsRequest {
50        requests,
51        read_mask,
52        ..
53    }: BatchGetObjectsRequest,
54) -> Result<BatchGetObjectsResponse, RpcError> {
55    // only batch requests with `object_id` and `exact_version` are supported by the KV store
56    if requests.iter().any(|r| r.version.is_none()) {
57        return Err(FieldViolation::new("version")
58            .with_reason(ErrorReason::FieldInvalid)
59            .with_description("KV store supports batch requests with exact object versioning")
60            .into());
61    }
62    let requests = requests
63        .into_iter()
64        .map(|req| (req.object_id, req.version))
65        .collect();
66    let (requests, read_mask) = validate_get_object_requests(requests, read_mask)?;
67    let object_keys: Vec<_> = requests
68        .into_iter()
69        .map(|(object_id, version)| {
70            ObjectKey(
71                object_id.into(),
72                version.expect("invariant's already checked").into(),
73            )
74        })
75        .collect();
76    let response: HashMap<_, _> = client
77        .get_objects(&object_keys)
78        .await?
79        .into_iter()
80        .map(|obj| ((obj.id(), obj.version()), obj))
81        .collect();
82
83    let objects = object_keys
84        .into_iter()
85        .map(|object_key| {
86            if let Some(object) = response.get(&(object_key.0, object_key.1)) {
87                let mut message = Object::default();
88                message.merge(object, &read_mask);
89                return GetObjectResult::new_object(message);
90            }
91            let err: RpcError =
92                ObjectNotFoundError::new_with_version(object_key.0.into(), object_key.1.into())
93                    .into();
94            GetObjectResult::new_error(err.into_status_proto())
95        })
96        .collect();
97
98    Ok(BatchGetObjectsResponse::new(objects))
99}
