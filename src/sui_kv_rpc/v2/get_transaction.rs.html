21pub const MAX_BATCH_REQUESTS: usize = 200;
22pub const READ_MASK_DEFAULT: &str = sui_rpc_api::read_mask_defaults::TRANSACTION;

24pub(crate) fn validate_read_mask(read_mask: Option<FieldMask>) -> Result<FieldMaskTree, RpcError> {
25    let read_mask = read_mask.unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
26    read_mask
27        .validate::<ExecutedTransaction>()
28        .map_err(|path| {
29            FieldViolation::new("read_mask")
30                .with_description(format!("invalid read_mask path: {path}"))
31                .with_reason(ErrorReason::FieldInvalid)
32        })?;
33    Ok(FieldMaskTree::from(read_mask))
34}

36pub async fn get_transaction(
37    client: BigTableClient,
38    stages: &StagesConfig,
39    request: GetTransactionRequest,
40    resolver: &PackageResolver,
41) -> Result<GetTransactionResponse, RpcError> {
42    let transaction_digest = request
43        .digest
44        .ok_or_else(|| {
45            FieldViolation::new("digest")
46                .with_description("missing digest")
47                .with_reason(ErrorReason::FieldMissing)
48        })?
49        .parse::<TransactionDigest>()
50        .map_err(|e| {
51            FieldViolation::new("digest")
52                .with_description(format!("invalid digest: {e}"))
53                .with_reason(ErrorReason::FieldInvalid)
54        })?;
55
56    let read_mask = validate_read_mask(request.read_mask)?;
57
58    let transactions_stage = stages.stage(PipelineStage::Transactions);
59    let objects_stage = stages.stage(PipelineStage::Objects);
60    let mut resolved = resolve::resolve_transactions(
61        client,
62        vec![transaction_digest],
63        &read_mask,
64        transactions_stage,
65        objects_stage,
66    )
67    .await?;
68    let (transaction, objects) = resolved
69        .remove(&transaction_digest)
70        .ok_or(TransactionNotFoundError(transaction_digest.into()))?;
71    Ok(GetTransactionResponse::new(
72        transaction_to_response(transaction, &read_mask, objects.as_ref(), resolver).await?,
73    ))
74}

76pub async fn batch_get_transactions(
77    client: BigTableClient,
78    stages: &StagesConfig,
79    BatchGetTransactionsRequest {
80        digests, read_mask, ..
81    }: BatchGetTransactionsRequest,
82    resolver: &PackageResolver,
83) -> Result<BatchGetTransactionsResponse, RpcError> {
84    let read_mask = validate_read_mask(read_mask)?;
85
86    if digests.len() > MAX_BATCH_REQUESTS {
87        return Err(RpcError::new(
88            tonic::Code::InvalidArgument,
89            format!("number of batch requests exceed limit of {MAX_BATCH_REQUESTS}"),
90        ));
91    }
92
93    let digests = digests
94        .iter()
95        .map(|digest| TransactionDigest::from_str(digest))
96        .collect::<Result<Vec<_>, _>>()?;
97    let transactions_stage = stages.stage(PipelineStage::Transactions);
98    let objects_stage = stages.stage(PipelineStage::Objects);
99    let resolved = resolve::resolve_transactions(
100        client,
101        digests.clone(),
102        &read_mask,
103        transactions_stage,
104        objects_stage,
105    )
106    .await?;
107
108    let mut transactions = Vec::with_capacity(digests.len());
109    for digest in digests {
110        if let Some((tx, objects)) = resolved.get(&digest) {
111            match transaction_to_response(tx.clone(), &read_mask, objects.as_ref(), resolver).await
112            {
113                Ok(tx) => transactions.push(GetTransactionResult::new_transaction(tx)),
114                Err(err) => {
115                    transactions.push(GetTransactionResult::new_error(err.into_status_proto()))
116                }
117            }
118        } else {
119            let err: RpcError = TransactionNotFoundError(digest.into()).into();
120            transactions.push(GetTransactionResult::new_error(err.into_status_proto()));
121        }
122    }
123    Ok(BatchGetTransactionsResponse::new(transactions))
124}

141    fn request(
142        digest: TransactionDigest,
143        paths: impl IntoIterator<Item = &'static str>,
144    ) -> GetTransactionRequest {
145        let mut request = GetTransactionRequest::default();
146        request.digest = Some(digest.to_string());
147        request.read_mask = Some(FieldMask::from_paths(paths));

151    fn assert_batch_objects(
152        response: BatchGetTransactionsResponse,
153        checkpoint: &sui_types::full_checkpoint_content::Checkpoint,
154        identity_only: bool,
155    ) {
156        assert_eq!(response.transactions.len(), 3);
157        for (result, expected_index) in
158            response
159                .transactions
160                .into_iter()
161                .zip_debug_eq([Some(1), None, Some(0)])
162        {
163            let Some(index) = expected_index else {
164                let status = result
165                    .to_result()
166                    .expect_err("missing digest should return an error");
167                assert_eq!(status.code, tonic::Code::NotFound as i32);
168                continue;
169            };
170            let transaction = result
171                .to_result()
172                .expect("known digest should return a transaction");
173            assert_eq!(
174                response_object_keys(&transaction),
175                canonical_transaction_object_keys(checkpoint, index)
176            );
177            let sibling_created_id =
178                TestCheckpointBuilder::derive_object_id(if index == 0 { 11 } else { 10 });
179            assert!(
180                response_object_keys(&transaction)
181                    .iter()
182                    .all(|key| key.0 != sibling_created_id),
183                "transaction object set must exclude its sibling's created object"
184            );
185            if identity_only {
186                assert_identity_only_object_mask(&transaction);
187            }
188        }
189    }

260    #[tokio::test]
261    async fn batch_get_transaction_objects_preserve_order_and_deduplicate_reads() {
262        let (ctx, _registry, mock, server) =
263            query_context_with_mock_and_registry("batch_get_transactions", 1).await;
264        let checkpoint = two_transaction_object_checkpoint();
265        insert_checkpoint_rows(&mock, &checkpoint).await;
266        let stages = StagesConfig::default();
267        let first_digest = checkpoint.transactions[0].transaction.digest();
268        let second_digest = checkpoint.transactions[1].transaction.digest();
269        let missing_digest = TransactionDigest::random();
270        let digests = vec![
271            second_digest.to_string(),
272            missing_digest.to_string(),
273            first_digest.to_string(),
274        ];
275
276        let mut parent_request = BatchGetTransactionsRequest::default();
277        parent_request.digests = digests.clone();
278        parent_request.read_mask = Some(FieldMask::from_paths(["objects"]));
279        let parent_response = batch_get_transactions(
280            ctx.client().clone(),
281            &stages,
282            parent_request,
283            ctx.package_resolver(),
284        )
285        .await
286        .expect("parent objects BatchGetTransactions should succeed");
287        assert_batch_objects(parent_response, &checkpoint, false);
288
289        mock.clear_read_rows_calls().await;
290        let mut nested_request = BatchGetTransactionsRequest::default();
291        nested_request.digests = digests;
292        nested_request.read_mask = Some(FieldMask::from_paths([
293            "objects.objects.object_id",
294            "objects.objects.version",
295        ]));
296        let nested_response = batch_get_transactions(
297            ctx.client().clone(),
298            &stages,
299            nested_request,
300            ctx.package_resolver(),
301        )
302        .await
303        .expect("nested objects BatchGetTransactions should succeed");
304        assert_batch_objects(nested_response, &checkpoint, true);
305
306        let mut actual_object_rows = mock
307            .read_rows_calls()
308            .await
309            .into_iter()
310            .filter(|call| call.table == sui_kvstore::tables::objects::NAME)
311            .flat_map(|call| call.row_keys)
312            .map(|key| key.to_vec())
313            .collect::<Vec<_>>();
314        actual_object_rows.sort();
315        let mut expected_object_rows = (0..checkpoint.transactions.len())
316            .flat_map(|index| canonical_transaction_object_keys(&checkpoint, index))
317            .collect::<BTreeSet<_>>()
318            .into_iter()
319            .map(|key| sui_kvstore::tables::objects::encode_key(&key))
320            .collect::<Vec<_>>();
321        expected_object_rows.sort();
322        assert_eq!(actual_object_rows, expected_object_rows);
323
324        server.abort();
325    }
