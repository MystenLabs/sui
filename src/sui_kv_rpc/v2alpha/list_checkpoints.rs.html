66pub(crate) async fn list_checkpoints(
67    ctx: QueryContext,
68    request: ListCheckpointsRequest,
69) -> Result<ListCheckpointsStream, RpcError> {
70    let started = Instant::now();
71    let filtered = request.filter.is_some();
72    let client: BigTableClient = ctx.client().clone();
73    let checkpoint_hi_exclusive = ctx.checkpoint_hi_exclusive();
74    let lh = ctx.ledger_history();
75    let endpoint = lh.list_checkpoints();
76    let checkpoints_stage = ctx.stage(PipelineStage::Checkpoints);
77    let transactions_stage = ctx.stage(PipelineStage::Transactions);
78    let objects_stage = ctx.stage(PipelineStage::Objects);
79    let tx_seq_digest_stage = ctx.stage(PipelineStage::TxSeqDigest);
80
81    let checkpoint_range = CheckpointRange::from_request(
82        request.start_checkpoint,
83        request.end_checkpoint,
84        checkpoint_hi_exclusive,
85    )?;
86    let read_mask = {
87        let read_mask = request
88            .read_mask
89            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
90        read_mask.validate::<Checkpoint>().map_err(|path| {
91            FieldViolation::new("read_mask")
92                .with_description(format!("invalid read_mask path: {path}"))
93                .with_reason(ErrorReason::FieldInvalid)
94        })?;
95        FieldMaskTree::from(read_mask)
96    };
97    let options = QueryOptions::from_proto(
98        request.options.as_ref(),
99        endpoint.default_limit_items,
100        endpoint.max_limit_items,
101        QueryType::Checkpoints,
102    )?;
103    let limit_items = options.limit_items;
104    let ordering = options.ordering;
105    let direction = options.scan_direction();
106
107    let cp_range = async { Ok::<_, RpcError>(resolve_cp_range(checkpoint_range, &options)) }
108        .instrument(debug_span!("resolve_cp_range"))
109        .await?;
110    let end_reason = cp_range.end_reason;
111    let end_checkpoint = cp_range.end_checkpoint;
112    let end_position = cp_range.end_position;
113    let cp_range = cp_range.range;
114
115    if cp_range.is_empty() {
116        info!(
117            filtered,
118            limit_items,
119            ?ordering,
120            emitted = 0usize,
121            elapsed_ms = started.elapsed().as_millis(),
122            "list_checkpoints: empty range"
123        );
124        // A caught-up tail (e.g. polling at the ledger tip) resolves to an empty
125        // range; still surface the terminal boundary so the client learns the
126        // final checkpoint is complete without waiting for the next item.
127        let terminal = reached_range_end(end_reason).then(|| {
128            watermark_response(terminal_boundary_watermark(
129                &options,
130                end_checkpoint,
131                end_position,
132            ))
133        });
134        return Ok(stream::iter(
135            terminal
136                .into_iter()
137                .chain([end_response(end_reason)])
138                .map(Ok),
139        )
140        .boxed());
141    }
142
143    let needs_full = needs_transactions_or_objects(&read_mask);
144    let cp_columns: Arc<[&'static str]> = list_checkpoint_columns(&read_mask, needs_full).into();
145
146    // Stage A: discover checkpoint rows for the requested response. Filtered
147    // requests use sparse bitmap-eval over transactions and then fetch the
148    // deduped checkpoint rows. Unfiltered requests scan the dense checkpoint
149    // keyspace directly, bounded by limit_items.
150    let cp_data_stream: BoxStream<'static, BitmapScanResult<Watermarked<(u64, CheckpointData)>>> =
151        if let Some(filter) = &request.filter {
152            let scan_budget = ctx.scan_budget(BitmapIndexSpec::tx());
153            let tx_range = client.checkpoint_to_tx_range(cp_range.clone()).await?;
154            let query = ctx.transaction_filter_query(filter)?;
155            let tx_seq_stream = client.eval_bitmap_query_stream(
156                query,
157                tx_range,
158                BitmapIndexSpec::tx(),
159                direction,
160                scan_budget,
161                ctx.bitmap_scan_observer(),
162            );
163            // Stage A2: resolve tx_seq -> cp_seq, streaming each cp as soon as the
164            // scan-order prefix is contiguous, then collapse each checkpoint's
165            // (contiguous) transactions to a single cp_seq. Dedup is its own stage
166            // rather than a per-chunk mapper because it carries state across chunk
167            // boundaries.
168            let cp_seq_stream = pipelined_chunks(
169                tx_seq_stream,
170                tx_seq_digest_stage.chunk_size,
171                tx_seq_digest_stage.concurrency,
172                {
173                    let client = client.clone();
174                    move |tx_seqs| {
175                        let client = client.clone();
176                        async move {
177                            resolve_checkpoint_seqs(client, tx_seqs)
178                                .await
179                                .map(|s| s.map_err(BitmapScanError::Source).boxed())
180                                .map_err(BitmapScanError::Source)
181                        }
182                    }
183                },
184            );
185            let cp_seq_stream = take_items(dedup_consecutive(cp_seq_stream), limit_items);
186            // Stage A3: fetch checkpoint rows for the deduped cp_seqs.
187            pipelined_chunks(
188                cp_seq_stream,
189                checkpoints_stage.chunk_size,
190                checkpoints_stage.concurrency,
191                {
192                    let client = client.clone();
193                    let columns = cp_columns.clone();
194                    move |seqs| {
195                        let client = client.clone();
196                        let columns = columns.clone();
197                        async move {
198                            fetch_checkpoint_data(client, columns, seqs)
199                                .await
200                                .map(|s| s.map_err(BitmapScanError::Source).boxed())
201                                .map_err(BitmapScanError::Source)
202                        }
203                    }
204                },
205            )
206        } else {
207            scan_checkpoint_data(
208                client.clone(),
209                cp_columns.clone(),
210                cp_range.clone(),
211                limit_items,
212                &options,
213            )
214            .await?
215        };
216
217    // Fast path: read_mask doesn't request transactions or objects → render
218    // directly from CheckpointData via the existing `checkpoint_to_response`.
219    if !needs_full {
220        let cp_data_stream = resolve_watermarks(cp_data_stream, client.tx_wm_resolver(direction));
221        return Ok(async_stream::try_stream! {
222            futures::pin_mut!(cp_data_stream);
223            let mut emitted = 0usize;
224            let mut checkpoint_boundary: Option<u64> = None;
225            let mut scan_limit_hit = false;
226            while let Some(item) = cp_data_stream.next().await {
227                match item {
228                    Ok(ResolvedWatermarked::Item((cp_seq, cp_data))) => {
229                        checkpoint_boundary = advance_checkpoint_boundary(checkpoint_boundary, cp_seq, &options);
230                        let wm = item_watermark(&options, cp_seq, cp_seq, checkpoint_boundary);
231                        emitted += 1;
232                        let message =
233                            crate::render::checkpoint_to_response(cp_data, &read_mask)?;
234                        yield response_for(wm, message);
235                    }
236                    Ok(ResolvedWatermarked::Watermark { position: _, cp: raw_cp }) => {
237                        // Tx-space → cp-space translation done in the
238                        // combinator; here we just clamp past anything
239                        // we've already emitted and convert to a
240                        // boundary cursor.
241                        let cp_frontier = if direction.is_ascending() {
242                            Some(raw_cp)
243                        } else {
244                            raw_cp.checked_add(1)
245                        };
246                        let Some(cp_frontier) = clamp_cp_frontier_past_last(cp_frontier, checkpoint_boundary, direction) else {
247                            continue;
248                        };
249                        if let Some(c) = frontier_to_boundary_candidate(cp_frontier, &options) {
250                            checkpoint_boundary = advance_checkpoint_boundary(checkpoint_boundary, c, &options);
251                        }
252                        let wm = boundary_watermark(&options, cp_frontier, cp_frontier, checkpoint_boundary);
253                        yield watermark_response(wm);
254                    }
255                    Err(BitmapScanError::ScanLimit) => {
256                        scan_limit_hit = true;
257                        break;
258                    }
259                    Err(BitmapScanError::Cancelled) => {
260                        Err(RpcError::new(
261                            tonic::Code::Cancelled,
262                            BitmapScanError::Cancelled.to_string(),
263                        ))?;
264                    }
265                    Err(BitmapScanError::Source(inner)) => {
266                        Err(RpcError::from(inner))?;
267                    }
268                }
269            }
270            let reason = if scan_limit_hit {
271                QueryEndReason::ScanLimit
272            } else if emitted == limit_items {
273                QueryEndReason::ItemLimit
274            } else {
275                end_reason
276            };
277            if reached_range_end(reason) {
278                yield watermark_response(terminal_boundary_watermark(&options, end_checkpoint, end_position));
279            }
280            yield end_response(reason);
281            info!(
282                filtered,
283                limit_items,
284                ?ordering,
285                emitted,
286                ?reason,
287                elapsed_ms = started.elapsed().as_millis(),
288                "list_checkpoints: done (summary only)"
289            );
290        }
291        .boxed());
292    }
293
294    // Heavy path: needs transactions and/or objects.
295    let cp_full_stream = resolve::resolve_checkpoints(
296        client.clone(),
297        &read_mask,
298        transactions_stage,
299        objects_stage,
300        cp_data_stream,
301    );
302    let cp_full_stream = resolve_watermarks(cp_full_stream, client.tx_wm_resolver(direction));
303
304    // Stage E: sync render — build full_checkpoint_content::Checkpoint and
305    // merge into the proto Checkpoint (CPU-only, no further IO).
306    Ok(async_stream::try_stream! {
307        futures::pin_mut!(cp_full_stream);
308        let mut emitted = 0usize;
309        let mut checkpoint_boundary: Option<u64> = None;
310        let mut scan_limit_hit = false;
311        while let Some(item) = cp_full_stream.next().await {
312            match item {
313                Ok(ResolvedWatermarked::Item((cp_seq, cp_data, txs, objects))) => {
314                    checkpoint_boundary = advance_checkpoint_boundary(checkpoint_boundary, cp_seq, &options);
315                    let wm = item_watermark(&options, cp_seq, cp_seq, checkpoint_boundary);
316                    let message =
317                        render_full_checkpoint(cp_data, txs, objects.as_ref(), &read_mask)?;
318                    emitted += 1;
319                    yield response_for(wm, message);
320                }
321                Ok(ResolvedWatermarked::Watermark { position: _, cp: raw_cp }) => {
322                    // See light-path arm — same clamp + boundary logic.
323                    let cp_frontier = if direction.is_ascending() {
324                        Some(raw_cp)
325                    } else {
326                        raw_cp.checked_add(1)
327                    };
328                    let Some(cp_frontier) = clamp_cp_frontier_past_last(cp_frontier, checkpoint_boundary, direction) else {
329                        continue;
330                    };
331                    if let Some(c) = frontier_to_boundary_candidate(cp_frontier, &options) {
332                        checkpoint_boundary = advance_checkpoint_boundary(checkpoint_boundary, c, &options);
333                    }
334                    let wm = boundary_watermark(&options, cp_frontier, cp_frontier, checkpoint_boundary);
335                    yield watermark_response(wm);
336                }
337                Err(BitmapScanError::ScanLimit) => {
338                    scan_limit_hit = true;
339                    break;
340                }
341                Err(BitmapScanError::Cancelled) => {
342                    Err(RpcError::new(
343                        tonic::Code::Cancelled,
344                        BitmapScanError::Cancelled.to_string(),
345                    ))?;
346                }
347                Err(BitmapScanError::Source(inner)) => {
348                    Err(RpcError::from(inner))?;
349                }
350            }
351        }
352        let reason = if scan_limit_hit {
353            QueryEndReason::ScanLimit
354        } else if emitted == limit_items {
355            QueryEndReason::ItemLimit
356        } else {
357            end_reason
358        };
359        if reached_range_end(reason) {
360            yield watermark_response(terminal_boundary_watermark(&options, end_checkpoint, end_position));
361        }
362        yield end_response(reason);
363        info!(
364            filtered,
365            limit_items,
366            ?ordering,
367            emitted,
368            ?reason,
369            elapsed_ms = started.elapsed().as_millis(),
370            "list_checkpoints: done"
371        );
372    }
373    .boxed())
374}

376/// Convert a cp-space scan frontier from `filtered_checkpoint_seq_stream`
377/// into a checkpoint-boundary candidate for `advance_checkpoint_boundary`.
378///
379/// ListCheckpoints dedupes cp_seq, so "cp X emitted" ≡ "cp X complete" and
380/// the item path feeds the item's cp straight into `advance_checkpoint_boundary`.
381/// The frontier path needs this adjustment instead:
382///
383/// - Ascending: frontier means "all matching cps strictly less than `p`
384///   have been emitted." The last fully-scanned cp is `p - 1`.
385/// - Descending: frontier means "all matching cps at least `p` have
386///   been emitted." The candidate is `p` directly.
387///
388/// Returns `None` only when `p == 0` ascending (no preceding cp), in
389/// which case the boundary stays at whatever the items have built up.
390fn frontier_to_boundary_candidate(frontier: u64, options: &QueryOptions) -> Option<u64> {
391    if options.is_ascending() {
392        frontier.checked_sub(1)
393    } else {
394        Some(frontier)
395    }
396}

398fn watermark_response(watermark: Watermark) -> ListCheckpointsResponse {
399    let mut response = ListCheckpointsResponse::default();
400    response.response = Some(list_checkpoints_response::Response::Watermark(watermark));
401    response
402}

404async fn scan_checkpoint_data(
405    client: BigTableClient,
406    columns: Arc<[&'static str]>,
407    range: std::ops::Range<u64>,
408    limit: usize,
409    options: &QueryOptions,
410) -> Result<BoxStream<'static, BitmapScanResult<Watermarked<(u64, CheckpointData)>>>, RpcError> {
411    let column_filter = BigTableClient::column_filter(&columns);
412    let rows = client
413        .scan_checkpoints_stream(range, options.scan_direction(), limit, Some(column_filter))
414        .await?;
415    Ok(rows
416        .map_ok(Watermarked::Item)
417        .map_err(BitmapScanError::Source)
418        .boxed())
419}

421async fn fetch_checkpoint_data(
422    client: BigTableClient,
423    columns: Arc<[&'static str]>,
424    seqs: Vec<u64>,
425) -> Result<BoxStream<'static, Result<(u64, CheckpointData), anyhow::Error>>, anyhow::Error> {
426    if seqs.is_empty() {
427        return Ok(stream::empty().boxed());
428    }
429    let column_filter = BigTableClient::column_filter(&columns);
430    let keys: Vec<Vec<u8>> = seqs
431        .iter()
432        .copied()
433        .map(tables::checkpoints::encode_key)
434        .collect();
435    let rows = client
436        .multi_get_stream(
437            tables::checkpoints::NAME,
438            keys,
439            Some(column_filter),
440            stage::CHECKPOINTS,
441        )
442        .await?;
443
444    Ok(async_stream::try_stream! {
445        let mut emitter: InputOrderEmitter<u64, (u64, CheckpointData)> =
446            InputOrderEmitter::new(seqs);
447        futures::pin_mut!(rows);
448        while let Some(row) = rows.next().await {
449            let (key, cells) = row?;
450            let seq = decode_checkpoint_row_key(&key)?;
451            let checkpoint = tables::checkpoints::decode(&cells)?;
452            for v in emitter.push(
453                seq,
454                (seq, checkpoint),
455                "list_checkpoints: checkpoint lookup",
456            )? {
457                yield v;
458            }
459        }
460        for v in emitter.finish("list_checkpoints: missing selected checkpoint row")? {
461            yield v;
462        }
463    }
464    .boxed())
465}

467/// Resolve a chunk of `tx_sequence_number`s (already in scan order) to their
468/// checkpoint sequence numbers, streaming each cp_seq as soon as the scan-order
469/// prefix is contiguously available rather than buffering the whole batch.
470/// BigTable multi-get rows arrive unordered, so `InputOrderEmitter` (keyed by
471/// the input scan order) releases the contiguous front as rows fill in — the
472/// same ordering trick as `fetch_checkpoint_data`. Consecutive duplicates (one
473/// checkpoint's multiple transactions) are collapsed downstream by
474/// `dedup_consecutive`.
475async fn resolve_checkpoint_seqs(
476    client: BigTableClient,
477    tx_seqs: Vec<u64>,
478) -> Result<BoxStream<'static, Result<u64, anyhow::Error>>, anyhow::Error> {
479    if tx_seqs.is_empty() {
480        return Ok(stream::empty().boxed());
481    }
482    let rows = client
483        .resolve_tx_checkpoints_stream(tx_seqs.clone())
484        .await?;
485    Ok(async_stream::try_stream! {
486        let mut emitter: InputOrderEmitter<u64, u64> = InputOrderEmitter::new(tx_seqs);
487        futures::pin_mut!(rows);
488        while let Some(row) = rows.next().await {
489            let (tx_seq, cp_seq) = row?;
490            for cp in emitter.push(tx_seq, cp_seq, "list_checkpoints: tx -> checkpoint resolution")? {
491                yield cp;
492            }
493        }
494        for cp in emitter.finish("list_checkpoints: missing tx -> checkpoint row")? {
495            yield cp;
496        }
497    }
498    .boxed())
499}

501fn response_for(watermark: Watermark, message: Checkpoint) -> ListCheckpointsResponse {

506    let mut response = ListCheckpointsResponse::default();
507    response.response = Some(list_checkpoints_response::Response::Item(item));

520/// Determine the checkpoint_sequence_number scan window from the logical
521/// checkpoint bounds, indexed history, and cursor bounds. Filtered scans
522/// are additionally bounded at runtime by the per-request bitmap bucket
523/// budget; that limit surfaces as SCAN_LIMIT, not as an up-front cp-range
524/// clamp.
525fn resolve_cp_range(checkpoint_range: CheckpointRange, options: &QueryOptions) -> ResolvedRange {
526    let cp_range = checkpoint_range.resolve(options);
527    let range = cp_range.range.clone();
528    options.apply_cursor_bounds(cp_range.with_range(range, options.ordering))
529}

531fn decode_checkpoint_row_key(key: &Bytes) -> Result<u64, RpcError> {
532    let bytes: [u8; 8] = key
533        .as_ref()
534        .try_into()
535        .map_err(|_| RpcError::new(tonic::Code::Internal, "invalid checkpoint row key length"))?;
536    Ok(u64::from_be_bytes(bytes))
537}

539/// Clamp a translated cp frontier so the emitted Boundary cursor never
540/// causes the client to re-request a checkpoint already delivered as an
541/// Item. `ListCheckpoints` dedupes items, so additional matching txs
542/// inside an already-delivered cp cannot produce a second item — it is
543/// safe (and required) to advance past the last delivered cp even when
544/// the tx→cp translation lands on it.
545///
546///   Ascending:  Item(C) resumes at cp ≥ C+1, so the boundary must be ≥ C+1.
547///   Descending: Item(C) resumes at cp < C+1 (i.e. ≤ C), so the boundary
548///               must be ≤ C; emitting at C is equivalent to the item's
549///               own resume (harmless but redundant).
550fn clamp_cp_frontier_past_last(
551    cp_frontier: Option<u64>,
552    last_cp_seq: Option<u64>,
553    direction: ScanDirection,
554) -> Option<u64> {
555    match (cp_frontier, last_cp_seq) {
556        (Some(frontier), Some(last)) if direction.is_ascending() => {
557            Some(frontier.max(last.saturating_add(1)))
558        }
559        (Some(frontier), Some(last)) => Some(frontier.min(last)),
560        (Some(frontier), None) => Some(frontier),
561        (None, _) => None,
562    }
563}

569    #[test]
570    fn clamp_passes_through_when_no_item_emitted() {
571        assert_eq!(
572            clamp_cp_frontier_past_last(Some(10), None, ScanDirection::Ascending),
573            Some(10),

577            Some(10),

579    }

581    #[test]
582    fn clamp_returns_none_when_translation_failed() {
583        assert_eq!(
584            clamp_cp_frontier_past_last(None, None, ScanDirection::Ascending),
585            None,
586        );
587        assert_eq!(
588            clamp_cp_frontier_past_last(None, Some(10), ScanDirection::Ascending),
589            None,
590        );
591        assert_eq!(
592            clamp_cp_frontier_past_last(None, Some(10), ScanDirection::Descending),
593            None,
594        );
595    }
596
597    #[test]
598    fn clamp_ascending_advances_past_last_emitted_cp() {
599        // Frontier behind the last emitted cp — clamp up to last+1 so we
600        // don't re-request a cp already delivered.
601        assert_eq!(
602            clamp_cp_frontier_past_last(Some(5), Some(10), ScanDirection::Ascending),
603            Some(11),
604        );
605        // Frontier exactly at the last emitted cp — same regression
606        // shape: clamp to last+1.
607        assert_eq!(
608            clamp_cp_frontier_past_last(Some(10), Some(10), ScanDirection::Ascending),
609            Some(11),
610        );
611        // Frontier already past last — pass through unchanged.
612        assert_eq!(
613            clamp_cp_frontier_past_last(Some(20), Some(10), ScanDirection::Ascending),
614            Some(20),
615        );
616    }

618    #[test]
619    fn clamp_descending_advances_past_last_emitted_cp() {
620        // Descending: "advanced" means smaller cp. Frontier > last is
621        // behind — clamp down to last (equivalent to the item's own
622        // resume in descending).
623        assert_eq!(
624            clamp_cp_frontier_past_last(Some(20), Some(10), ScanDirection::Descending),
625            Some(10),
626        );
627        // Frontier at last is the boundary case.
628        assert_eq!(
629            clamp_cp_frontier_past_last(Some(10), Some(10), ScanDirection::Descending),
630            Some(10),
631        );
632        // Frontier already past last (smaller) — pass through.
633        assert_eq!(
634            clamp_cp_frontier_past_last(Some(5), Some(10), ScanDirection::Descending),
635            Some(5),
636        );
637    }
