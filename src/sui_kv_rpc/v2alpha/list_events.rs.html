64pub(crate) async fn list_events(
65    ctx: QueryContext,
66    request: ListEventsRequest,
67) -> Result<ListEventsStream, RpcError> {
68    let started = Instant::now();
69    let filtered = request.filter.is_some();
70    let client: BigTableClient = ctx.client().clone();
71    let resolver: crate::PackageResolver = ctx.package_resolver().clone();
72    let checkpoint_hi_exclusive = ctx.checkpoint_hi_exclusive();
73    let lh = ctx.ledger_history();
74    let endpoint = lh.list_events();
75    let tx_seq_digest_stage = ctx.stage(PipelineStage::TxSeqDigest);
76    let transactions_stage = ctx.stage(PipelineStage::Transactions);
77
78    let checkpoint_range = CheckpointRange::from_request(
79        request.start_checkpoint,
80        request.end_checkpoint,
81        checkpoint_hi_exclusive,
82    )?;
83    let read_mask = Arc::new(validate_event_read_mask(request.read_mask)?);
84    let options = QueryOptions::from_proto(
85        request.options.as_ref(),
86        endpoint.default_limit_items,
87        endpoint.max_limit_items,
88        QueryType::Events,
89    )?;
90    let limit_items = options.limit_items;
91    let ordering = options.ordering;
92    let direction = options.scan_direction();
93    let wants_json = read_mask.contains(ProtoEvent::JSON_FIELD.name);
94
95    let event_range = resolve_event_range(&client, checkpoint_range, &options)
96        .instrument(debug_span!("resolve_event_range"))
97        .await?;
98    let end_reason = event_range.end_reason;
99    let end_checkpoint = event_range.end_checkpoint;
100    let end_position = event_range.end_position;
101    let event_range = event_range.range;
102
103    if event_range.is_empty() {
104        info!(
105            filtered,
106            wants_json,
107            limit_items,
108            ?ordering,
109            emitted = 0,
110            elapsed_ms = started.elapsed().as_millis(),
111            "list_events: empty range"
112        );
113        // A caught-up tail (e.g. polling at the ledger tip) resolves to an empty
114        // range; still surface the terminal boundary so the client learns the
115        // final checkpoint is complete without waiting for the next item.
116        let terminal = reached_range_end(end_reason).then(|| {
117            watermark_response(terminal_boundary_watermark(
118                &options,
119                end_checkpoint,
120                end_position,
121            ))
122        });
123        return Ok(futures::stream::iter(
124            terminal
125                .into_iter()
126                .chain([end_response(end_reason)])
127                .map(Ok),
128        )
129        .boxed());
130    }
131
132    let scan_budget = ctx.scan_budget(BitmapIndexSpec::event());
133
134    // Stage A: stream of EventRefs. Filtered requests discover event_seq
135    // values through the event bitmap (per-bucket and periodic-frontier
136    // watermarks). Unfiltered requests scan tx_seq_digest rows and expand
137    // each row's event_count into concrete EventRefs.
138    let request_bigtable_concurrency = ctx.request_bigtable_concurrency();
139    let event_ref_stream: BoxStream<'static, BitmapScanResult<Watermarked<EventRef>>> =
140        if let Some(filter) = &request.filter {
141            let query = ctx.event_filter_query(filter)?;
142            client
143                .eval_bitmap_query_stream(
144                    query,
145                    event_range.clone(),
146                    BitmapIndexSpec::event(),
147                    options.scan_direction(),
148                    scan_budget,
149                    ctx.bitmap_scan_observer(),
150                )
151                .map_ok(|m| {
152                    m.map_item(|event_seq| {
153                        let (tx_seq, event_idx) = event_bitmap_index::decode_event_seq(event_seq);
154                        EventRef {
155                            event_seq,
156                            tx_seq,
157                            event_idx,
158                            tx_seq_digest: None,
159                        }
160                    })
161                })
162                .boxed()
163        } else {
164            unfiltered_event_refs(
165                client.clone(),
166                event_range.clone(),
167                options.clone(),
168                endpoint.max_limit_items as usize,
169            )
170        };
171    let ref_stream = take_items(event_ref_stream, limit_items);
172
173    // Stage B (filtered path only): enrich refs with tx_seq_digest. The
174    // unfiltered path already populates `tx_seq_digest` from point lookups,
175    // so we skip this stage there.
176    let ref_with_digest_stream: BoxStream<'static, BitmapScanResult<Watermarked<EventRef>>> =
177        if filtered {
178            pipelined_chunks(
179                ref_stream,
180                tx_seq_digest_stage.chunk_size,
181                tx_seq_digest_stage.concurrency,
182                {
183                    let client = client.clone();
184                    move |refs| {
185                        let client = client.clone();
186                        async move {
187                            attach_tx_seq_digests(client, refs)
188                                .await
189                                .map(|s| s.map_err(BitmapScanError::Source).boxed())
190                                .map_err(BitmapScanError::Source)
191                        }
192                    }
193                },
194            )
195        } else {
196            ref_stream
197        };
198
199    let columns: Arc<[&'static str]> = Arc::from([col::EVENTS, col::CHECKPOINT_NUMBER]);
200
201    // Stage C: Watermarked<EventRef> -> Watermarked<(EventRef, TransactionData)>.
202    let tx_ref_stream = pipelined_chunks(
203        ref_with_digest_stream,
204        transactions_stage.chunk_size,
205        transactions_stage.concurrency,
206        {
207            let client = client.clone();
208            let columns = columns.clone();
209            move |refs| {
210                let client = client.clone();
211                let columns = columns.clone();
212                async move {
213                    fetch_txs_for_refs(client, columns, refs)
214                        .await
215                        .map(|s| s.map_err(BitmapScanError::Source).boxed())
216                        .map_err(BitmapScanError::Source)
217                }
218            }
219        },
220    );
221
222    // Stage D: render. `buffered` (ordered) lets per-event JSON rendering
223    // overlap while preserving input ref order in the output. Frontier
224    // watermarks pass through unchanged.
225    //
226    // Package resolution uses the server-global `PackageResolver`, backed by
227    // a cross-request LRU over the raw BigTable client. It is intentionally
228    // outside this request's downstream BigTable semaphore: tying a global
229    // cache miss to one request's budget would let that request's budget stall
230    // unrelated requests waiting on the same package. The local buffer below
231    // only bounds how many render/package-resolution attempts this request
232    // runs concurrently.
233    //
234    // TODO: add global single-flight dedupe around package cache misses so
235    // concurrent requests for the same uncached package share one BigTable
236    // fetch.
237    let event_stream = tx_ref_stream
238        .map(move |item| {
239            let resolver = resolver.clone();
240            let read_mask = read_mask.clone();
241            async move {
242                match item? {
243                    Watermarked::Item((event_ref, tx)) => {
244                        let rendered =
245                            render_event(event_ref, tx, &read_mask, &resolver, wants_json)
246                                .await
247                                .map_err(|e| BitmapScanError::Source(anyhow::Error::new(e)))?;
248                        Ok::<Watermarked<RenderedEvent>, BitmapScanError>(Watermarked::Item(
249                            rendered,
250                        ))
251                    }
252                    Watermarked::Watermark(p) => Ok(Watermarked::Watermark(p)),
253                }
254            }
255        })
256        .buffered(request_bigtable_concurrency)
257        .boxed();
258
259    let event_stream = resolve_watermarks(event_stream, client.event_wm_resolver(direction));
260
261    Ok(async_stream::try_stream! {
262        futures::pin_mut!(event_stream);
263        let mut emitted = 0usize;
264        let mut checkpoint_boundary: Option<u64> = None;
265        let mut scan_limit_hit = false;
266        while let Some(item) = event_stream.next().await {
267            match item {
268                Ok(ResolvedWatermarked::Item(rendered)) => {
269                    checkpoint_boundary = advance_boundary_excluding_cp(checkpoint_boundary, rendered.checkpoint_number, &options);
270                    let wm = item_watermark(
271                        &options,
272                        rendered.checkpoint_number,
273                        rendered.event_seq,
274                        checkpoint_boundary,
275                    );
276                    emitted += 1;
277                    yield event_item_response(rendered.item, wm);
278                }
279                Ok(ResolvedWatermarked::Watermark { position, cp }) => {
280                    checkpoint_boundary = advance_boundary_excluding_cp(checkpoint_boundary, cp, &options);
281                    let wm = boundary_watermark(&options, boundary_cursor_cp(cp, direction), position, checkpoint_boundary);
282                    yield watermark_response(wm);
283                }
284                Err(BitmapScanError::ScanLimit) => {
285                    scan_limit_hit = true;
286                    break;
287                }
288                Err(BitmapScanError::Cancelled) => {
289                    Err(RpcError::new(
290                        tonic::Code::Cancelled,
291                        BitmapScanError::Cancelled.to_string(),
292                    ))?;
293                }
294                Err(BitmapScanError::Source(inner)) => {
295                    Err(RpcError::from(inner))?;
296                }
297            }
298        }
299        let reason = if scan_limit_hit {
300            QueryEndReason::ScanLimit
301        } else if emitted == limit_items {
302            QueryEndReason::ItemLimit
303        } else {
304            end_reason
305        };
306        if reached_range_end(reason) {
307            yield watermark_response(terminal_boundary_watermark(&options, end_checkpoint, end_position));
308        }
309        yield end_response(reason);
310        info!(
311            filtered,
312            wants_json,
313            limit_items,
314            ?ordering,
315            emitted,
316            ?reason,
317            elapsed_ms = started.elapsed().as_millis(),
318            "list_events: done"
319        );
320    }
321    .boxed())
322}

337/// Stage B: for filtered refs (no `tx_seq_digest`), dedupe tx_seqs in the
338/// chunk, fetch their digests via one multi-get, and emit enriched refs in
339/// input ref-index order. A single arriving digest can release multiple
340/// refs (multiple events from one tx).
341async fn attach_tx_seq_digests(
342    client: BigTableClient,
343    refs: Vec<EventRef>,
344) -> Result<BoxStream<'static, Result<EventRef, anyhow::Error>>, anyhow::Error> {
345    if refs.is_empty() {
346        return Ok(futures::stream::empty().boxed());
347    }
348
349    let mut unique_seqs: Vec<u64> = refs.iter().map(|p| p.tx_seq).collect();
350    unique_seqs.sort_unstable();
351    unique_seqs.dedup();
352    // tx_seq -> indices of refs sharing that tx_seq, so a single arriving
353    // digest can release all dependent refs at once.
354    let mut indices_by_seq: HashMap<u64, Vec<usize>> = HashMap::new();
355    for (i, p) in refs.iter().enumerate() {
356        indices_by_seq.entry(p.tx_seq).or_default().push(i);
357    }
358
359    let digest_stream = client.resolve_tx_digests_stream(unique_seqs).await?;
360    Ok(async_stream::try_stream! {
361        // Keyed by ref index so the emitter releases in input-ref order.
362        let input_indices: Vec<usize> = (0..refs.len()).collect();
363        let mut emitter: InputOrderEmitter<usize, EventRef> =
364            InputOrderEmitter::new(input_indices);
365        futures::pin_mut!(digest_stream);
366        while let Some(row) = digest_stream.next().await {
367            let row = row?;
368            let idxs = indices_by_seq
369                .remove(&row.tx_sequence_number)
370                .ok_or_else(|| {
371                    anyhow::anyhow!(
372                        "list_events: unexpected transaction digest row {}",
373                        row.tx_sequence_number
374                    )
375                })?;
376            for idx in idxs {
377                let mut p = refs[idx];
378                p.tx_seq_digest = Some(row);
379                for v in emitter.push(
380                    idx,
381                    p,
382                    "list_events: transaction digest lookup for selected event",
383                )? {
384                    yield v;
385                }
386            }
387        }
388        for v in emitter.finish("list_events: missing selected event transaction digest")? {
389            yield v;
390        }
391    }
392    .boxed())
393}

395/// Stage C: fetch transactions for a chunk of refs, emitting
396/// `(EventRef, TransactionData)` in input ref-index order.
397async fn fetch_txs_for_refs(
398    client: BigTableClient,
399    columns: Arc<[&'static str]>,
400    refs: Vec<EventRef>,
401) -> Result<BoxStream<'static, Result<(EventRef, TransactionData), anyhow::Error>>, anyhow::Error> {
402    if refs.is_empty() {
403        return Ok(futures::stream::empty().boxed());
404    }
405
406    let mut unique_digests: Vec<TransactionDigest> = Vec::new();
407    let mut seen_digests: std::collections::HashSet<TransactionDigest> =
408        std::collections::HashSet::new();
409    let mut indices_by_digest: HashMap<TransactionDigest, Vec<usize>> = HashMap::new();
410    for (i, p) in refs.iter().enumerate() {
411        let row = p.tx_seq_digest.ok_or_else(|| {
412            anyhow::anyhow!(
413                "list_events: selected event {} is missing transaction digest",
414                p.event_seq
415            )
416        })?;
417        if seen_digests.insert(row.digest) {
418            unique_digests.push(row.digest);
419        }
420        indices_by_digest.entry(row.digest).or_default().push(i);
421    }
422
423    let column_filter = BigTableClient::column_filter(&columns);
424    let tx_stream = client
425        .get_transactions_stream(unique_digests, Some(column_filter))
426        .await?;
427    Ok(async_stream::try_stream! {
428        let input_indices: Vec<usize> = (0..refs.len()).collect();
429        let mut emitter: InputOrderEmitter<usize, (EventRef, TransactionData)> =
430            InputOrderEmitter::new(input_indices);
431        futures::pin_mut!(tx_stream);
432        while let Some(row) = tx_stream.next().await {
433            let (digest, tx) = row?;
434            // All refs sharing this digest become ready at once. We clone
435            // `tx` per ref so each event from the same tx has its own
436            // `TransactionData`; downstream consumes by-value when reading
437            // `tx.events.data[event_idx]`.
438            let idxs = indices_by_digest.remove(&digest).ok_or_else(|| {
439                anyhow::anyhow!("list_events: unexpected transaction body row {digest}")
440            })?;
441            for idx in idxs {
442                for v in emitter.push(
443                    idx,
444                    (refs[idx], tx.clone()),
445                    "list_events: transaction body lookup for selected event",
446                )? {
447                    yield v;
448                }
449            }
450        }
451        for v in emitter.finish("list_events: missing selected event transaction body")? {
452            yield v;
453        }
454    }
455    .boxed())
456}

458/// Carries the rendered `EventItem` (without its `watermark` field — the
459/// main loop fills that in with the current checkpoint boundary) plus
460/// the cp and packed event_seq the main loop needs to update the
461/// watermark.
462struct RenderedEvent {
463    item: EventItem,
464    checkpoint_number: u64,
465    event_seq: u64,
466}
467
468async fn render_event(
469    event_ref: EventRef,
470    tx: TransactionData,
471    read_mask: &FieldMaskTree,
472    resolver: &PackageResolver,
473    wants_json: bool,
474) -> Result<RenderedEvent, RpcError> {
475    let tx_events = tx.events.as_ref().ok_or_else(|| {
476        RpcError::new(
477            tonic::Code::Internal,
478            format!(
479                "list_events: selected event {} transaction {} has no events column",
480                event_ref.event_seq, tx.digest
481            ),
482        )
483    })?;
484    let event = tx_events
485        .data
486        .get(event_ref.event_idx as usize)
487        .ok_or_else(|| {
488            RpcError::new(
489                tonic::Code::Internal,
490                format!(
491                    "list_events: selected event {} index {} out of range for transaction {}",
492                    event_ref.event_seq, event_ref.event_idx, tx.digest
493                ),
494            )
495        })?;
496
497    let mut proto_event = ProtoEvent::merge_from(event, read_mask);
498    if wants_json {
499        proto_event.json = render_json(resolver, &event.type_, &event.contents)
500            .await
501            .map(Box::new);
502    }
503
504    let mut item = EventItem::default();
505    item.checkpoint = Some(tx.checkpoint_number);
506    item.event_index = Some(event_ref.event_idx);
507    item.transaction_digest = Some(tx.digest.to_string());
508    item.event = Some(proto_event);
509    item.transaction_offset = event_ref.tx_seq_digest.map(|row| row.tx_offset as u64);
510
511    Ok(RenderedEvent {
512        item,
513        checkpoint_number: tx.checkpoint_number,
514        event_seq: event_ref.event_seq,
515    })
516}

524    response

527/// A reference to a single event from the bitmap scan or tx_seq_digest lookup,
528/// carrying just enough to look up the concrete event after a bulk tx fetch.
529/// Unfiltered discovery already reads tx_seq_digest rows to enumerate real
530/// events, so those rows are carried forward instead of being fetched again.
531#[derive(Clone, Copy, Debug, Eq, PartialEq)]
532struct EventRef {
533    event_seq: u64,
534    tx_seq: u64,
535    event_idx: u32,
536    tx_seq_digest: Option<TxSeqDigestData>,
537}
538
539/// Range-scan `tx_seq_digest` across the tx range covered by `event_range`,
540/// using each row's `event_count` to enumerate real event_seqs per tx without
541/// touching the tx body.
542fn unfiltered_event_refs(
543    client: BigTableClient,
544    event_range: std::ops::Range<u64>,
545    options: QueryOptions,
546    source_limit: usize,
547) -> BoxStream<'static, BitmapScanResult<Watermarked<EventRef>>> {
548    async_stream::try_stream! {
549        let lower_bound = event_range.start;
550        let upper_bound = event_range.end;
551        let Some(tx_range) = tx_range_for_event_range(event_range) else {
552            return;
553        };
554        let (scan_range, scan_limited, frontier_tx) =
555            clamp_tx_scan_range(tx_range, source_limit, &options);
556        let rows = client
557            .scan_tx_seq_digests_stream(scan_range, options.scan_direction(), source_limit)
558            .await
559            .map_err(|e| BitmapScanError::Source(anyhow::Error::new(e)))?;
560
561        futures::pin_mut!(rows);
562        while let Some(row) = rows.next().await {
563            for event_ref in expand_event_refs(row?, lower_bound, upper_bound, &options) {
564                yield Watermarked::Item(event_ref);
565            }
566        }
567
568        if scan_limited {
569            yield Watermarked::Watermark(event_bitmap_index::event_seq_lo(frontier_tx));
570            Err(BitmapScanError::ScanLimit)?;
571        }
572    }
573    .boxed()
574}
575
576fn tx_range_for_event_range(event_range: std::ops::Range<u64>) -> Option<std::ops::Range<u64>> {
577    let last_event_seq = event_range.end.checked_sub(1)?;
578    let start_tx = event_bitmap_index::decode_event_seq(event_range.start).0;
579    let end_tx = event_bitmap_index::decode_event_seq(last_event_seq)
580        .0
581        .checked_add(1)?;
582    (start_tx < end_tx).then_some(start_tx..end_tx)
583}
584
585fn clamp_tx_scan_range(
586    tx_range: std::ops::Range<u64>,
587    source_limit: usize,
588    options: &QueryOptions,
589) -> (std::ops::Range<u64>, bool, u64) {
590    let source_limit = source_limit as u64;
591    if options.is_ascending() {
592        let end = tx_range
593            .start
594            .saturating_add(source_limit)
595            .min(tx_range.end);
596        (tx_range.start..end, end < tx_range.end, end)
597    } else {
598        let start = tx_range
599            .end
600            .saturating_sub(source_limit)
601            .max(tx_range.start);
602        (start..tx_range.end, start > tx_range.start, start)
603    }
604}

606fn expand_event_refs(
607    row: TxSeqDigestData,
608    lower_bound: u64,
609    upper_bound: u64,
610    options: &QueryOptions,
611) -> Vec<EventRef> {
612    if row.event_count == 0 {
613        return Vec::new();
614    }
615
616    let mut refs = Vec::with_capacity(row.event_count as usize);
617    if options.is_ascending() {
618        for event_idx in 0..row.event_count {
619            push_event_ref_if_in_bounds(&mut refs, row, event_idx, lower_bound, upper_bound);
620        }
621    } else {
622        for event_idx in (0..row.event_count).rev() {
623            push_event_ref_if_in_bounds(&mut refs, row, event_idx, lower_bound, upper_bound);
624        }
625    }
626    refs
627}
628
629fn push_event_ref_if_in_bounds(
630    refs: &mut Vec<EventRef>,
631    row: TxSeqDigestData,
632    event_idx: u32,
633    lower_bound: u64,
634    upper_bound: u64,
635) {
636    let event_seq = event_bitmap_index::encode_event_seq(row.tx_sequence_number, event_idx);
637    if event_seq < lower_bound || event_seq >= upper_bound {
638        return;
639    }
640    refs.push(EventRef {
641        event_seq,
642        tx_seq: row.tx_sequence_number,
643        event_idx,
644        tx_seq_digest: Some(row),
645    });
646}

648fn validate_event_read_mask(read_mask: Option<FieldMask>) -> Result<FieldMaskTree, RpcError> {
649    let read_mask = read_mask.unwrap_or_else(|| FieldMask::from_str(EVENT_READ_MASK_DEFAULT));
650    read_mask.validate::<ProtoEvent>().map_err(|path| {
651        FieldViolation::new("read_mask")
652            .with_description(format!("invalid read_mask path: {path}"))
653            .with_reason(ErrorReason::FieldInvalid)
654    })?;
655    Ok(FieldMaskTree::from(read_mask))
656}

658/// Resolve the packed-event_seq scan window from the logical checkpoint bounds.
659/// The checkpoint window is first clamped to indexed history and any cursor
660/// bounds, then shifted into packed event space. Query cursors are applied
661/// in that packed space so they can resume from the middle of a transaction's
662/// events. Filtered scans are additionally bounded at runtime by the
663/// per-request bitmap bucket budget; that limit surfaces as SCAN_LIMIT, not
664/// as an up-front cp-range clamp.
665async fn resolve_event_range(
666    client: &BigTableClient,
667    checkpoint_range: CheckpointRange,
668    options: &QueryOptions,
669) -> Result<ResolvedRange, RpcError> {
670    let cp_range = checkpoint_range.resolve(options);
671    if cp_range.is_empty() {
672        let tx_boundary =
673            checkpoint_to_tx_boundary(client, cp_range.terminal_checkpoint(options.ordering))
674                .await?;
675        let event_boundary = event_bitmap_index::event_seq_lo(tx_boundary);
676        return Ok(cp_range.with_range(event_boundary..event_boundary, options.ordering));
677    }
678
679    let tx_range = client
680        .checkpoint_to_tx_range(cp_range.range.clone())
681        .await?;
682    let start_event_seq = event_bitmap_index::event_seq_lo(tx_range.start);
683    let end_event_seq = event_bitmap_index::event_seq_lo(tx_range.end);
684    Ok(options
685        .apply_cursor_bounds(cp_range.with_range(start_event_seq..end_event_seq, options.ordering)))
686}
687
688async fn checkpoint_to_tx_boundary(
689    client: &BigTableClient,
690    checkpoint: u64,
691) -> Result<u64, RpcError> {
692    if checkpoint == 0 {
693        return Ok(0);
694    }
695    Ok(client.checkpoint_to_tx_range(0..checkpoint).await?.end)
696}
