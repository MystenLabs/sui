88#[async_trait]
89impl KeyValueStoreWriter for BigTableClient {
90    async fn save_objects(&mut self, objects: &[&Object], timestamp_ms: TimestampMs) -> Result<()> {
91        let mut items = Vec::with_capacity(objects.len());
92        for object in objects {
93            let object_key = ObjectKey(object.id(), object.version());
94            items.push((
95                Self::raw_object_key(&object_key)?,
96                vec![(DEFAULT_COLUMN_QUALIFIER, bcs::to_bytes(object)?)],
97            ));
98        }
99        self.multi_set(OBJECTS_TABLE, items, Some(timestamp_ms))
100            .await
101    }
102
103    async fn save_transactions(&mut self, transactions: &[TransactionData]) -> Result<()> {
104        let mut items = Vec::with_capacity(transactions.len());
105        let mut timestamp_ms = None;
106        for transaction in transactions {
107            timestamp_ms = Some(transaction.timestamp);
108            let cells = vec![
109                (
110                    TRANSACTION_COLUMN_QUALIFIER,
111                    bcs::to_bytes(&transaction.transaction)?,
112                ),
113                (
114                    EFFECTS_COLUMN_QUALIFIER,
115                    bcs::to_bytes(&transaction.effects)?,
116                ),
117                (EVENTS_COLUMN_QUALIFIER, bcs::to_bytes(&transaction.events)?),
118                (
119                    TIMESTAMP_COLUMN_QUALIFIER,
120                    bcs::to_bytes(&transaction.timestamp)?,
121                ),
122                (
123                    CHECKPOINT_NUMBER_COLUMN_QUALIFIER,
124                    bcs::to_bytes(&transaction.checkpoint_number)?,
125                ),
126            ];
127            items.push((transaction.transaction.digest().inner().to_vec(), cells));
128        }
129        self.multi_set(TRANSACTIONS_TABLE, items, timestamp_ms)
130            .await
131    }
132
133    async fn save_checkpoint(&mut self, checkpoint: &CheckpointData) -> Result<()> {
134        let summary = &checkpoint.checkpoint_summary.data();
135        let timestamp = summary.timestamp_ms;
136        let contents = &checkpoint.checkpoint_contents;
137        let signatures = &checkpoint.checkpoint_summary.auth_sig();
138        let key = summary.sequence_number.to_be_bytes().to_vec();
139        let cells = vec![
140            (CHECKPOINT_SUMMARY_COLUMN_QUALIFIER, bcs::to_bytes(summary)?),
141            (
142                CHECKPOINT_SIGNATURES_COLUMN_QUALIFIER,
143                bcs::to_bytes(signatures)?,
144            ),
145            (
146                CHECKPOINT_CONTENTS_COLUMN_QUALIFIER,
147                bcs::to_bytes(contents)?,
148            ),
149        ];
150        self.multi_set(CHECKPOINTS_TABLE, [(key.clone(), cells)], Some(timestamp))
151            .await?;
152        self.multi_set(
153            CHECKPOINTS_BY_DIGEST_TABLE,
154            [(
155                checkpoint.checkpoint_summary.digest().inner().to_vec(),
156                vec![(DEFAULT_COLUMN_QUALIFIER, key)],
157            )],
158            Some(timestamp),
159        )
160        .await
161    }
162
163    async fn save_watermark(&mut self, watermark: CheckpointSequenceNumber) -> Result<()> {
164        let watermark_bytes = watermark.to_be_bytes().to_vec();
165        self.multi_set(
166            WATERMARK_ALT_TABLE,
167            [(
168                vec![0],
169                vec![(DEFAULT_COLUMN_QUALIFIER, watermark_bytes.clone())],
170            )],
171            Some(watermark),
172        )
173        .await?;
174        self.multi_set(
175            WATERMARK_TABLE,
176            [(watermark_bytes, vec![(DEFAULT_COLUMN_QUALIFIER, vec![])])],
177            None,
178        )
179        .await
180    }
181
182    async fn save_epoch(&mut self, epoch: EpochInfo) -> Result<()> {
183        let key = epoch.epoch.to_be_bytes().to_vec();
184        self.multi_set(
185            EPOCHS_TABLE,
186            [(
187                key,
188                vec![(DEFAULT_COLUMN_QUALIFIER, bcs::to_bytes(&epoch)?)],
189            )],
190            epoch.end_timestamp_ms.or(epoch.start_timestamp_ms),
191        )
192        .await
193    }
194}

196#[async_trait]
197impl KeyValueStoreReader for BigTableClient {
198    async fn get_objects(&mut self, object_keys: &[ObjectKey]) -> Result<Vec<Object>> {
199        let keys: Result<_, _> = object_keys.iter().map(Self::raw_object_key).collect();
200        let mut objects = vec![];
201        for (_, row) in self.multi_get(OBJECTS_TABLE, keys?, None).await? {
202            for (_, value) in row {
203                objects.push(bcs::from_bytes(&value)?);
204            }
205        }
206        Ok(objects)
207    }
208
209    async fn get_transactions(
210        &mut self,
211        transactions: &[TransactionDigest],
212    ) -> Result<Vec<TransactionData>> {
213        let keys = transactions.iter().map(|tx| tx.inner().to_vec()).collect();
214        let mut result = vec![];
215        for (_, row) in self.multi_get(TRANSACTIONS_TABLE, keys, None).await? {
216            let mut transaction = None;
217            let mut effects = None;
218            let mut events = None;
219            let mut timestamp = 0;
220            let mut checkpoint_number = 0;
221
222            for (column, value) in row {
223                match std::str::from_utf8(&column)? {
224                    TRANSACTION_COLUMN_QUALIFIER => transaction = Some(bcs::from_bytes(&value)?),
225                    EFFECTS_COLUMN_QUALIFIER => effects = Some(bcs::from_bytes(&value)?),
226                    EVENTS_COLUMN_QUALIFIER => events = Some(bcs::from_bytes(&value)?),
227                    TIMESTAMP_COLUMN_QUALIFIER => timestamp = bcs::from_bytes(&value)?,
228                    CHECKPOINT_NUMBER_COLUMN_QUALIFIER => {
229                        checkpoint_number = bcs::from_bytes(&value)?
230                    }
231                    _ => error!("unexpected column {:?} in transactions table", column),
232                }
233            }
234            result.push(TransactionData {
235                transaction: transaction.context("transaction field is missing")?,
236                effects: effects.context("effects field is missing")?,
237                events: events.context("events field is missing")?,
238                timestamp,
239                checkpoint_number,
240            })
241        }
242        Ok(result)
243    }
244
245    async fn get_checkpoints(
246        &mut self,
247        sequence_numbers: &[CheckpointSequenceNumber],
248    ) -> Result<Vec<Checkpoint>> {
249        let keys = sequence_numbers
250            .iter()
251            .map(|sq| sq.to_be_bytes().to_vec())
252            .collect();
253        let mut checkpoints = vec![];
254        for (_, row) in self.multi_get(CHECKPOINTS_TABLE, keys, None).await? {
255            let mut summary = None;
256            let mut contents = None;
257            let mut signatures = None;
258            for (column, value) in row {
259                match std::str::from_utf8(&column)? {
260                    CHECKPOINT_SUMMARY_COLUMN_QUALIFIER => summary = Some(bcs::from_bytes(&value)?),
261                    CHECKPOINT_CONTENTS_COLUMN_QUALIFIER => {
262                        contents = Some(bcs::from_bytes(&value)?)
263                    }
264                    CHECKPOINT_SIGNATURES_COLUMN_QUALIFIER => {
265                        signatures = Some(bcs::from_bytes(&value)?)
266                    }
267                    _ => error!("unexpected column {:?} in checkpoints table", column),
268                }
269            }
270            let checkpoint = Checkpoint {
271                summary: summary.context("summary field is missing")?,
272                contents: contents.context("contents field is missing")?,
273                signatures: signatures.context("signatures field is missing")?,
274            };
275            checkpoints.push(checkpoint);
276        }
277        Ok(checkpoints)
278    }

280    async fn get_checkpoint_by_digest(
281        &mut self,
282        digest: CheckpointDigest,
283    ) -> Result<Option<Checkpoint>> {
284        let key = digest.inner().to_vec();
285        let mut response = self
286            .multi_get(CHECKPOINTS_BY_DIGEST_TABLE, vec![key], None)
287            .await?;
288        if let Some((_, row)) = response.pop()
289            && let Some((_, value)) = row.into_iter().next()
290        {
291            let sequence_number = u64::from_be_bytes(value.as_slice().try_into()?);
292            if let Some(chk) = self.get_checkpoints(&[sequence_number]).await?.pop() {
293                return Ok(Some(chk));
294            }
295        }
296        Ok(None)
297    }
298
299    async fn get_latest_checkpoint(&mut self) -> Result<CheckpointSequenceNumber> {
300        match self
301            .multi_get(WATERMARK_ALT_TABLE, vec![vec![0]], None)
302            .await?
303            .pop()
304            .and_then(|(_, mut row)| row.pop())
305        {
306            Some((_, value_bytes)) => Ok(u64::from_be_bytes(value_bytes.as_slice().try_into()?)),
307            None => Ok(0),
308        }
309    }
310
311    async fn get_latest_checkpoint_summary(&mut self) -> Result<Option<CheckpointSummary>> {
312        let sequence_number = self.get_latest_checkpoint().await?;
313        if sequence_number == 0 {
314            return Ok(None);
315        }
316
317        // Fetch just the summary for the latest checkpoint sequence number.
318        let mut response = self
319            .multi_get(
320                CHECKPOINTS_TABLE,
321                vec![(sequence_number - 1).to_be_bytes().to_vec()],
322                Some(RowFilter {
323                    filter: Some(Filter::ColumnQualifierRegexFilter(
324                        format!("^({CHECKPOINT_SUMMARY_COLUMN_QUALIFIER})$").into(),
325                    )),
326                }),
327            )
328            .await?;
329
330        let Some((_, row)) = response.pop() else {
331            return Ok(None);
332        };
333
334        let mut summary: Option<CheckpointSummary> = None;
335        for (column, value) in row {
336            match std::str::from_utf8(&column)? {
337                CHECKPOINT_SUMMARY_COLUMN_QUALIFIER => summary = Some(bcs::from_bytes(&value)?),
338                _ => error!("unexpected column {:?} in checkpoints table", column),
339            }
340        }
341
342        Ok(summary)
343    }

345    async fn get_latest_object(&mut self, object_id: &ObjectID) -> Result<Option<Object>> {
346        let upper_limit = Self::raw_object_key(&ObjectKey::max_for_id(object_id))?;
347        if let Some((_, row)) = self.reversed_scan(OBJECTS_TABLE, upper_limit).await?.pop()
348            && let Some((_, value)) = row.into_iter().next()
349        {
350            return Ok(Some(bcs::from_bytes(&value)?));
351        }
352        Ok(None)
353    }

355    async fn get_epoch(&mut self, epoch_id: EpochId) -> Result<Option<EpochInfo>> {
356        let key = epoch_id.to_be_bytes().to_vec();
357        Ok(
358            match self.multi_get(EPOCHS_TABLE, vec![key], None).await?.pop() {
359                Some((_, mut row)) => row
360                    .pop()
361                    .map(|value| bcs::from_bytes(&value.1))
362                    .transpose()?,
363                None => None,
364            },
365        )
366    }

370        Ok(
371            match self.reversed_scan(EPOCHS_TABLE, upper_limit).await?.pop() {
372                Some((_, mut row)) => row
373                    .pop()
374                    .map(|value| bcs::from_bytes(&value.1))
375                    .transpose()?,
376                None => None,

381    // Multi-get transactions, selecting columns relevant to events.
382    async fn get_events_for_transactions(
383        &mut self,
384        transaction_digests: &[TransactionDigest],
385    ) -> Result<Vec<(TransactionDigest, TransactionEventsData)>> {
386        let query = self.multi_get(
387            TRANSACTIONS_TABLE,
388            transaction_digests
389                .iter()
390                .map(|tx| tx.inner().to_vec())
391                .collect(),
392            Some(RowFilter {
393                filter: Some(Filter::ColumnQualifierRegexFilter(
394                    format!("^({EVENTS_COLUMN_QUALIFIER}|{TIMESTAMP_COLUMN_QUALIFIER})$").into(),
395                )),
396            }),
397        );
398        let mut results = vec![];
399
400        for (key, row) in query.await? {
401            let mut transaction_events: Option<Option<TransactionEvents>> = None;
402            let mut timestamp_ms = 0;
403            for (column, value) in row {
404                match std::str::from_utf8(&column)? {
405                    EVENTS_COLUMN_QUALIFIER => transaction_events = Some(bcs::from_bytes(&value)?),
406                    TIMESTAMP_COLUMN_QUALIFIER => timestamp_ms = bcs::from_bytes(&value)?,
407                    _ => error!("unexpected column {:?} in transactions table", column),
408                }
409            }
410            let events = transaction_events
411                .context("events field is missing")?
412                .map(|e| e.data)
413                .unwrap_or_default();
414
415            let transaction_digest = TransactionDigest::try_from(key)
416                .context("Failed to deserialize transaction digest")?;
417
418            results.push((
419                transaction_digest,
420                TransactionEventsData {
421                    events,
422                    timestamp_ms,
423                },
424            ));
425        }
426
427        Ok(results)
428    }

449    pub async fn new_remote(
450        instance_id: String,
451        is_read_only: bool,
452        timeout: Option<Duration>,
453        client_name: String,
454        registry: Option<&Registry>,
455        app_profile_id: Option<String>,
456    ) -> Result<Self> {
457        let policy = if is_read_only {
458            "https://www.googleapis.com/auth/bigtable.data.readonly"

460            "https://www.googleapis.com/auth/bigtable.data"

462        let token_provider = gcp_auth::provider().await?;

465            .domain_name("bigtable.googleapis.com");
466        let mut endpoint = Channel::from_static("https://bigtable.googleapis.com")
467            .http2_keep_alive_interval(Duration::from_secs(60))
468            .keep_alive_while_idle(true)
469            .tls_config(tls_config)?;
470        if let Some(timeout) = timeout {
471            endpoint = endpoint.timeout(timeout);
472        }
473        let table_prefix = format!(
474            "projects/{}/instances/{}/tables/",
475            token_provider.project_id().await?,
476            instance_id
477        );
478        let auth_channel = AuthChannel {
479            channel: endpoint.connect_lazy(),
480            policy: policy.to_string(),
481            token_provider: Some(token_provider),
482            token: Arc::new(RwLock::new(None)),
483        };
484        Ok(Self {
485            table_prefix,
486            client: BigtableInternalClient::new(auth_channel),
487            client_name,
488            metrics: registry.map(KvMetrics::new),
489            app_profile_id,
490        })
491    }

500    fn report_bt_stats(&self, request_stats: &Option<RequestStats>, table_name: &str) {
501        let Some(metrics) = &self.metrics else {
502            return;
503        };
504        let labels = [&self.client_name, table_name];
505        if let Some(StatsView::FullReadStatsView(view)) =
506            request_stats.as_ref().and_then(|r| r.stats_view.as_ref())
507        {
508            if let Some(latency) = view
509                .request_latency_stats
510                .as_ref()
511                .and_then(|s| s.frontend_server_latency)
512            {
513                if latency.seconds < 0 || latency.nanos < 0 {
514                    return;
515                }
516                let duration = Duration::new(latency.seconds as u64, latency.nanos as u32);
517                metrics
518                    .kv_bt_chunk_latency_ms
519                    .with_label_values(&labels)
520                    .observe(duration.as_millis() as f64);
521            }
522            if let Some(iteration_stats) = &view.read_iteration_stats {
523                metrics
524                    .kv_bt_chunk_rows_returned_count
525                    .with_label_values(&labels)
526                    .inc_by(iteration_stats.rows_returned_count as u64);
527                metrics
528                    .kv_bt_chunk_rows_seen_count
529                    .with_label_values(&labels)
530                    .inc_by(iteration_stats.rows_seen_count as u64);
531            }
532        }
533    }

535    pub async fn read_rows(
536        &mut self,
537        mut request: ReadRowsRequest,
538        table_name: &str,
539    ) -> Result<Vec<(Vec<u8>, Vec<(Vec<u8>, Vec<u8>)>)>> {
540        if let Some(ref app_profile_id) = self.app_profile_id {
541            request.app_profile_id = app_profile_id.clone();
542        }
543        let mut result = vec![];
544        let mut response = self.client.read_rows(request).await?.into_inner();
545
546        let mut row_key = None;
547        let mut row = vec![];
548        let mut cell_value = vec![];
549        let mut cell_name = None;
550        let mut timestamp = 0;
551
552        while let Some(message) = response.message().await? {
553            self.report_bt_stats(&message.request_stats, table_name);
554            for mut chunk in message.chunks.into_iter() {
555                // new row check
556                if !chunk.row_key.is_empty() {
557                    row_key = Some(chunk.row_key);
558                }
559                match chunk.qualifier {
560                    // new cell started
561                    Some(qualifier) => {
562                        if let Some(cell_name) = cell_name {
563                            row.push((cell_name, cell_value));
564                            cell_value = vec![];
565                        }
566                        cell_name = Some(qualifier);
567                        timestamp = chunk.timestamp_micros;
568                        cell_value.append(&mut chunk.value);
569                    }
570                    None => {
571                        if chunk.timestamp_micros == 0 {
572                            cell_value.append(&mut chunk.value);
573                        } else if chunk.timestamp_micros >= timestamp {
574                            // newer version of cell is available
575                            timestamp = chunk.timestamp_micros;
576                            cell_value = chunk.value;
577                        }
578                    }
579                }
580                if chunk.row_status.is_some() {
581                    if let Some(RowStatus::CommitRow(_)) = chunk.row_status {
582                        if let Some(cell_name) = cell_name {
583                            row.push((cell_name, cell_value));
584                        }
585                        if let Some(row_key) = row_key {
586                            result.push((row_key, row));
587                        }
588                    }
589                    row_key = None;
590                    row = vec![];
591                    cell_value = vec![];
592                    cell_name = None;
593                }
594            }
595        }
596        Ok(result)
597    }

599    async fn multi_set(
600        &mut self,
601        table_name: &str,
602        values: impl IntoIterator<Item = (Bytes, Vec<(&str, Bytes)>)> + std::marker::Send,
603        timestamp_ms: Option<TimestampMs>,
604    ) -> Result<()> {
605        for chunk in values.into_iter().collect::<Vec<_>>().chunks(50_000) {
606            self.multi_set_internal(table_name, chunk.iter().cloned(), timestamp_ms)
607                .await?;
608        }
609        Ok(())
610    }
611
612    async fn multi_set_internal(
613        &mut self,
614        table_name: &str,
615        values: impl IntoIterator<Item = (Bytes, Vec<(&str, Bytes)>)> + std::marker::Send,
616        timestamp_ms: Option<TimestampMs>,
617    ) -> Result<()> {
618        let mut entries = vec![];
619        let timestamp_micros = timestamp_ms
620            .map(|tst| {
621                tst.checked_mul(1000)
622                    .expect("timestamp multiplication overflow") as i64
623            })
624            // default to -1 for current Bigtable server time
625            .unwrap_or(-1);
626        for (row_key, cells) in values {
627            let mutations = cells
628                .into_iter()
629                .map(|(column_name, value)| Mutation {
630                    mutation: Some(mutation::Mutation::SetCell(SetCell {
631                        family_name: COLUMN_FAMILY_NAME.to_string(),
632                        column_qualifier: column_name.to_owned().into_bytes(),
633                        // The timestamp of the cell into which new data should be written.
634                        timestamp_micros,
635                        value,
636                    })),
637                })
638                .collect();
639            entries.push(Entry {
640                row_key,
641                mutations,
642                idempotency: None,
643            });
644        }
645        let request = MutateRowsRequest {
646            table_name: format!("{}{}", self.table_prefix, table_name),
647            entries,
648            ..MutateRowsRequest::default()
649        };
650        let mut response = self.mutate_rows(request).await?;
651        while let Some(part) = response.message().await? {
652            for entry in part.entries {
653                if let Some(status) = entry.status
654                    && status.code != 0
655                {
656                    return Err(anyhow!(
657                        "bigtable write failed {} {}",
658                        status.code,
659                        status.message
660                    ));
661                }
662            }
663        }
664        Ok(())
665    }

667    pub async fn multi_get(
668        &mut self,
669        table_name: &str,
670        keys: Vec<Vec<u8>>,
671        filter: Option<RowFilter>,
672    ) -> Result<Vec<(Vec<u8>, Vec<(Bytes, Bytes)>)>> {
673        let start_time = Instant::now();
674        let num_keys_requested = keys.len();
675        let result = self.multi_get_internal(table_name, keys, filter).await;
676        let elapsed_ms = start_time.elapsed().as_millis() as f64;
677
678        let Some(metrics) = &self.metrics else {
679            return result;
680        };
681
682        let labels = [&self.client_name, table_name];
683        let Ok(rows) = &result else {
684            metrics.kv_get_errors.with_label_values(&labels).inc();
685            return result;
686        };
687
688        metrics
689            .kv_get_batch_size
690            .with_label_values(&labels)
691            .observe(num_keys_requested as f64);
692
693        if num_keys_requested > rows.len() {
694            metrics
695                .kv_get_not_found
696                .with_label_values(&labels)
697                .inc_by((num_keys_requested - rows.len()) as u64);
698        }
699
700        metrics
701            .kv_get_success
702            .with_label_values(&labels)
703            .inc_by(rows.len() as u64);
704
705        metrics
706            .kv_get_latency_ms
707            .with_label_values(&labels)
708            .observe(elapsed_ms);
709
710        if num_keys_requested > 0 {
711            metrics
712                .kv_get_latency_ms_per_key
713                .with_label_values(&labels)
714                .observe(elapsed_ms / num_keys_requested as f64);
715        }
716
717        result
718    }

720    pub async fn multi_get_internal(
721        &mut self,
722        table_name: &str,
723        keys: Vec<Vec<u8>>,
724        filter: Option<RowFilter>,
725    ) -> Result<Vec<(Vec<u8>, Vec<(Bytes, Bytes)>)>> {
726        let version_filter = RowFilter {
727            filter: Some(Filter::CellsPerColumnLimitFilter(1)),
728        };
729        let filter = Some(match filter {
730            Some(filter) => RowFilter {
731                filter: Some(Filter::Chain(Chain {
732                    filters: vec![filter, version_filter],
733                })),
734            },
735            None => version_filter,
736        });
737        let request = ReadRowsRequest {
738            table_name: format!("{}{}", self.table_prefix, table_name),
739            rows_limit: keys.len() as i64,
740            rows: Some(RowSet {
741                row_keys: keys,
742                row_ranges: vec![],
743            }),
744            filter,
745            request_stats_view: 2,
746            ..ReadRowsRequest::default()
747        };
748        let mut result = vec![];
749        for (key, cells) in self.read_rows(request, table_name).await? {
750            result.push((key, cells));
751        }
752        Ok(result)
753    }

755    async fn reversed_scan(
756        &mut self,
757        table_name: &str,
758        upper_limit: Bytes,
759    ) -> Result<Vec<(Bytes, Vec<(Bytes, Bytes)>)>> {
760        let start_time = Instant::now();
761        let result = self.reversed_scan_internal(table_name, upper_limit).await;
762        let elapsed_ms = start_time.elapsed().as_millis() as f64;
763        let labels = [&self.client_name, table_name];
764        match &self.metrics {
765            Some(metrics) => match result {
766                Ok(result) => {
767                    metrics.kv_scan_success.with_label_values(&labels).inc();
768                    if result.is_empty() {
769                        metrics.kv_scan_not_found.with_label_values(&labels).inc();
770                    }
771                    metrics
772                        .kv_scan_latency_ms
773                        .with_label_values(&labels)
774                        .observe(elapsed_ms);
775                    Ok(result)
776                }
777                Err(e) => {
778                    metrics.kv_scan_error.with_label_values(&labels).inc();
779                    Err(e)
780                }
781            },
782            None => result,
783        }
784    }

786    async fn reversed_scan_internal(
787        &mut self,
788        table_name: &str,
789        upper_limit: Bytes,
790    ) -> Result<Vec<(Bytes, Vec<(Bytes, Bytes)>)>> {
791        let range = RowRange {
792            start_key: None,
793            end_key: Some(EndKey::EndKeyClosed(upper_limit)),
794        };
795        let request = ReadRowsRequest {
796            table_name: format!("{}{}", self.table_prefix, table_name),
797            rows_limit: 1,
798            rows: Some(RowSet {
799                row_keys: vec![],
800                row_ranges: vec![range],
801            }),
802            reversed: true,
803            ..ReadRowsRequest::default()
804        };
805        self.read_rows(request, table_name).await
806    }

815impl Service<Request<Body>> for AuthChannel {
816    type Response = Response<Body>;
817    type Error = Box<dyn std::error::Error + Send + Sync>;
818    #[allow(clippy::type_complexity)]
819    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>> + Send>>;
820
821    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
822        self.channel.poll_ready(cx).map_err(Into::into)
823    }
824
825    fn call(&mut self, mut request: Request<Body>) -> Self::Future {
826        let cloned_channel = self.channel.clone();
827        let cloned_token = self.token.clone();
828        let mut inner = std::mem::replace(&mut self.channel, cloned_channel);
829        let policy = self.policy.clone();
830        let token_provider = self.token_provider.clone();
831
832        let mut auth_token = None;
833        if token_provider.is_some() {
834            let guard = self.token.read().expect("failed to acquire a read lock");
835            if let Some(token) = &*guard
836                && !token.has_expired()
837            {
838                auth_token = Some(token.clone());
839            }
840        }
841
842        Box::pin(async move {
843            if let Some(ref provider) = token_provider {
844                let token = match auth_token {
845                    None => {
846                        let new_token = provider.token(&[policy.as_ref()]).await?;
847                        let mut guard = cloned_token.write().unwrap();
848                        *guard = Some(new_token.clone());
849                        new_token
850                    }
851                    Some(token) => token,
852                };
853                let token_string = token.as_str().parse::<String>()?;
854                let header =
855                    HeaderValue::from_str(format!("Bearer {}", token_string.as_str()).as_str())?;
856                request.headers_mut().insert("authorization", header);
857            }
858            // enable reverse scan
859            let header = HeaderValue::from_static("CAE=");
860            request.headers_mut().insert("bigtable-features", header);
861            Ok(inner.call(request).await?)
862        })
863    }
