7use std::sync::RwLock;
8use std::task::Context;
9use std::task::Poll;

11use gcp_auth::Token;
12use gcp_auth::TokenProvider;
13use http::HeaderValue;
14use http::Request;
15use http::Response;
16use tonic::body::Body;

19/// Auth middleware that injects credentials onto any inner `Service`.
20#[derive(Clone)]
21pub(crate) struct AuthChannel<S> {
22    inner: S,
23    policy: String,
24    token_provider: Option<Arc<dyn TokenProvider>>,
25    token: Arc<RwLock<Option<Arc<Token>>>>,
26}
27
28impl<S> AuthChannel<S> {
29    pub(crate) fn new(
30        inner: S,
31        policy: String,
32        token_provider: Option<Arc<dyn TokenProvider>>,
33    ) -> Self {
34        Self {
35            inner,
36            policy,
37            token_provider,
38            token: Arc::new(RwLock::new(None)),
39        }
40    }
41}
42
43impl<S> Service<Request<Body>> for AuthChannel<S>
44where
45    S: Service<Request<Body>, Response = Response<Body>> + Clone + Send + 'static,
46    S::Error: Into<Box<dyn std::error::Error + Send + Sync>>,
47    S::Future: Send,
48{
49    type Response = Response<Body>;
50    type Error = Box<dyn std::error::Error + Send + Sync>;
51    #[allow(clippy::type_complexity)]
52    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>> + Send>>;
53
54    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
55        self.inner.poll_ready(cx).map_err(Into::into)
56    }
57
58    fn call(&mut self, mut request: Request<Body>) -> Self::Future {
59        let cloned_token = self.token.clone();
60        let policy = self.policy.clone();
61        let token_provider = self.token_provider.clone();
62
63        let mut auth_token = None;
64        if token_provider.is_some() {
65            let guard = self.token.read().expect("failed to acquire a read lock");
66            if let Some(token) = &*guard
67                && !token.has_expired()
68            {
69                auth_token = Some(token.clone());
70            }
71        }
72
73        // Take the poll_ready'd inner service, replace with a fresh clone.
74        let cloned = self.inner.clone();
75        let mut ready_inner = std::mem::replace(&mut self.inner, cloned);
76
77        Box::pin(async move {
78            if let Some(ref provider) = token_provider {
79                let token = match auth_token {
80                    None => {
81                        let new_token = provider.token(&[policy.as_ref()]).await?;
82                        let mut guard = cloned_token.write().unwrap();
83                        *guard = Some(new_token.clone());
84                        new_token
85                    }
86                    Some(token) => token,
87                };
88                let token_string = token.as_str().parse::<String>()?;
89                let header =
90                    HeaderValue::from_str(format!("Bearer {}", token_string.as_str()).as_str())?;
91                request.headers_mut().insert("authorization", header);
92            }
93            // enable reverse scan
94            let header = HeaderValue::from_static("CAE=");
95            request.headers_mut().insert("bigtable-features", header);
96
97            ready_inner.call(request).await.map_err(Into::into)
98        })
99    }
