5//!
6//! Generic DNF query planning and ordered bucket-stream merge-joins live in
7//! `sui_inverted_index::bitmap_query`. This module only knows how to scan one
8//! BigTable dimension key into the generic `(bucket_id, RoaringBitmap)` stream
9//! shape.

26/// Identifies which inverted-index table a `BitmapQuery` evaluates against.
27#[derive(Clone, Copy)]
28pub struct BitmapIndexSpec {
29    pub table_name: &'static str,
30    pub schema_version: u32,
31    pub bucket_size: u64,
32    pub bucket_id_width: usize,
33    pub bitmap_column: &'static str,
34}
35
36impl BitmapIndexSpec {
37    pub const fn tx() -> Self {
38        Self {
39            table_name: transaction_bitmap_index::NAME,
40            schema_version: transaction_bitmap_index::SCHEMA_VERSION,
41            bucket_size: transaction_bitmap_index::BUCKET_SIZE,
42            bucket_id_width: 10,
43            bitmap_column: transaction_bitmap_index::col::BITMAP,
44        }
45    }
46
47    pub const fn event() -> Self {
48        Self {
49            table_name: event_bitmap_index::NAME,
50            schema_version: event_bitmap_index::SCHEMA_VERSION,
51            bucket_size: event_bitmap_index::BUCKET_SIZE,
52            bucket_id_width: 12,
53            bitmap_column: event_bitmap_index::col::BITMAP,
54        }
55    }
56
57    fn encode_row_key(&self, dimension_key: &[u8], bucket_id: u64) -> Vec<u8> {
58        // Dispatch on `table_name` rather than `bucket_id_width`: cp- and
59        // tx-bitmap rows share the same width, so width alone isn't a unique
60        // discriminant.
61        match self.table_name {
62            transaction_bitmap_index::NAME => transaction_bitmap_index::encode_row_key(
63                self.schema_version,
64                dimension_key,
65                bucket_id,
66            ),
67            event_bitmap_index::NAME => {
68                event_bitmap_index::encode_row_key(self.schema_version, dimension_key, bucket_id)
69            }
70            name => panic!("unsupported bitmap table {name}"),
71        }
72    }
73}

87impl BitmapBucketSource for BigTableBitmapSource {
88    fn scan_bucket_stream(
89        &self,
90        dimension_key: Vec<u8>,
91        range: Range<u64>,
92        direction: ScanDirection,
93    ) -> BucketStream {
94        // The tx universe is dense (every tx_seq in range is real), so it is
95        // synthesized here rather than stored; see `IndexDimension::TxUniverse`.
96        // Gated on the tx table: the key is only ever produced by the tx filter
97        // layer, and in event-space a dense universe would be semantically wrong.
98        if self.spec.table_name == transaction_bitmap_index::NAME
99            && dimension_key.first() == Some(&IndexDimension::TxUniverse.tag_byte())
100        {
101            return futures::stream::iter(
102                dense_universe_buckets(range, self.spec.bucket_size, direction).map(Ok),
103            )
104            .boxed();
105        }
106        scan_bucket_stream(
107            self.client.clone(),
108            dimension_key,
109            range,
110            self.spec,
111            direction,
112        )
113    }
114}

116/// Stream a single bitmap-index dimension's buckets in order, one
117/// `RoaringBitmap` per bucket with **relative** bit positions.
118fn scan_bucket_stream(
119    mut client: BigTableClient,
120    dimension_key: Vec<u8>,
121    range: Range<u64>,
122    spec: BitmapIndexSpec,
123    direction: ScanDirection,
124) -> BucketStream {
125    async_stream::try_stream! {
126        if range.is_empty() {
127            return;
128        }
129        let start_bucket = range.start / spec.bucket_size;
130        let end_bucket = (range.end - 1) / spec.bucket_size;
131
132        let start_row = spec.encode_row_key(&dimension_key, start_bucket);
133        let end_row = spec.encode_row_key(&dimension_key, end_bucket);
134
135        let stream = client
136            .range_scan_stream(
137                spec.table_name,
138                Some(bytes::Bytes::from(start_row)),
139                Some(bytes::Bytes::from(end_row)),
140                0,
141                !direction.is_ascending(),
142                None,
143            )
144            .await?;
145        futures::pin_mut!(stream);
146
147        while let Some(row) = stream.next().await {
148            let (row_key, cells) = row?;
149            let Some(bitmap_bytes) = cells
150                .iter()
151                .find(|(col, _)| col.as_ref() == spec.bitmap_column.as_bytes())
152                .map(|(_, v)| v)
153            else {
154                continue;
155            };
156
157            let hash_pos = row_key
158                .iter()
159                .rposition(|&b| b == b'#')
160                .context("malformed bitmap index row key: no '#' separator")?;
161            let suffix = &row_key[hash_pos + 1..];
162            let bucket_id: u64 = std::str::from_utf8(suffix)
163                .context("non-ascii bucket_id suffix")?
164                .parse()
165                .context("invalid bucket_id in row key")?;
166
167            let bitmap = RoaringBitmap::deserialize_from(bitmap_bytes.as_ref())
168                .context("deserializing bitmap")?;
169            yield (bucket_id, bitmap);
170        }
171    }
172    .boxed()
173}
