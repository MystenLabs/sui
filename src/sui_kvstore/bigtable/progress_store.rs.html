4use crate::{BigTableClient, KeyValueStoreReader, KeyValueStoreWriter};
5use anyhow::Result;
6use async_trait::async_trait;
7use sui_data_ingestion_core::ProgressStore;
8use sui_types::messages_checkpoint::CheckpointSequenceNumber;
9
10pub struct BigTableProgressStore {
11    client: BigTableClient,
12}
13
14impl BigTableProgressStore {
15    pub fn new(client: BigTableClient) -> Self {
16        Self { client }
17    }
18}
19
20#[async_trait]
21impl ProgressStore for BigTableProgressStore {
22    async fn load(&mut self, _: String) -> Result<CheckpointSequenceNumber> {
23        self.client.get_latest_checkpoint().await
24    }
25
26    async fn save(&mut self, _: String, checkpoint_number: CheckpointSequenceNumber) -> Result<()> {
27        self.client.save_watermark(checkpoint_number).await
28    }
29}
