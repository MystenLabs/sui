43#[derive(Clone, Default, Debug, Deserialize, Serialize)]
44#[serde(default, rename_all = "kebab-case")]
45pub struct BigtablePoolLayer {
46    /// Number of channels to create at startup (default: 10).
47    pub initial_pool_size: Option<usize>,
48    /// Minimum number of channels the pool will maintain (default: 1).
49    pub min_pool_size: Option<usize>,
50    /// Maximum number of channels the pool can scale to (default: 200).
51    pub max_pool_size: Option<usize>,
52    /// Average load per channel below which the pool considers scaling down (default: 5).
53    pub min_rpcs_per_channel: Option<usize>,
54    /// Average load per channel above which the pool scales up (default: 50).
55    pub max_rpcs_per_channel: Option<usize>,
56    /// Maximum channels to remove in a single scale-down operation (default: 2).
57    pub max_resize_delta: Option<usize>,
58    /// Consecutive low-load observations required before scaling down (default: 3).
59    pub downscale_threshold: Option<usize>,
60    /// Milliseconds between maintenance cycles (resize + refresh) (default: 60000).
61    pub maintenance_interval_ms: Option<u64>,
62    /// Channel age in milliseconds before it is eligible for refresh (default: 2700000 = 45 min).
63    pub refresh_age_ms: Option<u64>,
64    /// Random jitter in milliseconds added to refresh age (default: 300000 = 5 min).
65    pub refresh_jitter_ms: Option<u64>,
66}
67
68impl BigtablePoolLayer {
69    pub fn finish(self, deprecated_pool_size: Option<usize>) -> PoolConfig {
70        if deprecated_pool_size.is_some() {
71            warn!(
72                "bigtable-connection-pool-size is deprecated; \
73                 use the [bigtable-pool] section instead"
74            );
75        }
76
77        let base = PoolConfig::default();
78        PoolConfig {
79            initial_pool_size: self
80                .initial_pool_size
81                .or(deprecated_pool_size)
82                .unwrap_or(base.initial_pool_size),
83            min_pool_size: self.min_pool_size.unwrap_or(base.min_pool_size),
84            max_pool_size: self.max_pool_size.unwrap_or(base.max_pool_size),
85            min_rpcs_per_channel: self
86                .min_rpcs_per_channel
87                .unwrap_or(base.min_rpcs_per_channel),
88            max_rpcs_per_channel: self
89                .max_rpcs_per_channel
90                .unwrap_or(base.max_rpcs_per_channel),
91            max_resize_delta: self.max_resize_delta.unwrap_or(base.max_resize_delta),
92            downscale_threshold: self.downscale_threshold.unwrap_or(base.downscale_threshold),
93            maintenance_interval: self
94                .maintenance_interval_ms
95                .map(Duration::from_millis)
96                .unwrap_or(base.maintenance_interval),
97            refresh_age: self
98                .refresh_age_ms
99                .map(Duration::from_millis)
100                .unwrap_or(base.refresh_age),
101            refresh_jitter: self
102                .refresh_jitter_ms
103                .map(Duration::from_millis)
104                .unwrap_or(base.refresh_jitter),
105        }
106    }
107}

158impl ConcurrentLayer {
159    pub(crate) fn max_rows_or_default(&self) -> usize {
160        self.max_rows.unwrap_or(DEFAULT_MAX_ROWS_PER_BIGTABLE_BATCH)
161    }
162
163    pub fn finish(self, base: ConcurrentConfig) -> ConcurrentConfig {
164        ConcurrentConfig {
165            committer: if let Some(c) = self.committer {
166                c.finish(base.committer)
167            } else {
168                base.committer
169            },
170            ingestion: if let Some(i) = self.ingestion {
171                i.finish(base.ingestion)
172            } else {
173                base.ingestion
174            },
175            pruner: None,
176            fanout: self.fanout.or(base.fanout),
177            min_eager_rows: self.min_eager_rows.or(base.min_eager_rows),
178            max_pending_rows: self.max_pending_rows.or(base.max_pending_rows),
179            max_watermark_updates: self.max_watermark_updates.or(base.max_watermark_updates),
180            processor_channel_size: self.processor_channel_size.or(base.processor_channel_size),
181            collector_channel_size: self.collector_channel_size.or(base.collector_channel_size),
182            committer_channel_size: self.committer_channel_size.or(base.committer_channel_size),
183        }
184    }
185}

197#[derive(Clone, Default, Debug, Deserialize, Serialize)]
198#[serde(default, rename_all = "kebab-case")]
199pub struct SequentialLayer {
200    // Framework sequential surface — mirrors the fields actually read by
201    // `sui_indexer_alt_framework::pipeline::sequential`.
202    pub committer: Option<CommitterLayer>,
203    pub ingestion: Option<PipelineIngestionLayer>,
204    pub fanout: Option<ConcurrencyConfig>,
205    pub min_eager_rows: Option<usize>,
206    pub max_pending_rows: Option<usize>,
207    pub max_batch_checkpoints: Option<usize>,
208    pub processor_channel_size: Option<usize>,
209    pub pipeline_depth: Option<usize>,
210
211    // sui-kvstore-specific config extensions
212
213    // Controls the concurrency of the bitmap flushes. The framework doesn't
214    // perform commits for sequential pipelines concurrently, but our store
215    // implementation doesn't actually write to the database on commit for
216    // the bitmap pipelines. The store buffers the bitmaps for the current
217    // working "bucket" ranges internally, merges in rows from each framework
218    // batch on commit (parallelized on background tasks), then finally flushes
219    // the updated bitmaps to bigtable concurrently. Same semantic as
220    // `ConcurrentLayer::write_concurrency`.
221    pub write_concurrency: Option<usize>,
222    /// Maximum rows per in-handler BigTable write RPC. Same semantic as
223    /// `ConcurrentLayer::max_rows`.
224    pub max_rows: Option<usize>,
225    /// Per-pipeline rate limit (rows per second). Overrides the default
226    /// `IndexerConfig::max_rows_per_second` when set.
227    pub max_rows_per_second: Option<u64>,
228}
229
230impl SequentialLayer {
231    pub(crate) fn max_rows_or_default(&self) -> usize {
232        self.max_rows.unwrap_or(DEFAULT_MAX_ROWS_PER_BIGTABLE_BATCH)
233    }
234
235    pub fn finish(self, base: ConcurrentConfig) -> SequentialConfig {
236        let committer = if let Some(c) = self.committer {
237            c.finish(base.committer)
238        } else {
239            base.committer
240        };
241        SequentialConfig {
242            committer,
243            ingestion: if let Some(i) = self.ingestion {
244                i.finish(base.ingestion)
245            } else {
246                base.ingestion
247            },
248            fanout: self.fanout.or(base.fanout),
249            min_eager_rows: self.min_eager_rows.or(base.min_eager_rows),
250            max_pending_rows: self.max_pending_rows.or(base.max_pending_rows),
251            max_batch_checkpoints: self.max_batch_checkpoints,
252            processor_channel_size: self.processor_channel_size.or(base.processor_channel_size),
253            pipeline_depth: self.pipeline_depth,
254        }
255    }
256}

277/// This type is identical to [`framework::ingestion::IngestionConfig`], but is set-up to be
278/// serialized and deserialized by `serde`.
279#[derive(Clone, Debug, Deserialize, Serialize)]
280#[serde(default, rename_all = "kebab-case", deny_unknown_fields)]
281pub struct IngestionConfig {
282    pub ingest_concurrency: framework::config::ConcurrencyConfig,
283    pub retry_interval_ms: u64,
284    pub streaming_backoff_initial_batch_size: NonZeroUsize,
285    pub streaming_backoff_max_batch_size: usize,
286    pub streaming_connection_timeout_ms: u64,
287    pub streaming_statement_timeout_ms: u64,
288
289    /// Deprecated: accepted (and ignored) so old configs don't fail to parse. Replaced by
290    /// per-pipeline `ingestion.subscriber-channel-size`.
291    pub checkpoint_buffer_size: Option<usize>,
292}
293
294impl Default for IngestionConfig {
295    fn default() -> Self {
296        framework::ingestion::IngestionConfig::default().into()
297    }
298}
299
300impl From<framework::ingestion::IngestionConfig> for IngestionConfig {
301    fn from(config: framework::ingestion::IngestionConfig) -> Self {
302        Self {
303            ingest_concurrency: config.ingest_concurrency,
304            retry_interval_ms: config.retry_interval_ms,
305            streaming_backoff_initial_batch_size: config.streaming_backoff_initial_batch_size,
306            streaming_backoff_max_batch_size: config.streaming_backoff_max_batch_size,
307            streaming_connection_timeout_ms: config.streaming_connection_timeout_ms,
308            streaming_statement_timeout_ms: config.streaming_statement_timeout_ms,
309            checkpoint_buffer_size: None,
310        }
311    }
312}

314impl From<IngestionConfig> for framework::ingestion::IngestionConfig {
315    fn from(config: IngestionConfig) -> Self {
316        if config.checkpoint_buffer_size.is_some() {
317            warn!(
318                "Config field `checkpoint-buffer-size` is deprecated and ignored. Remove it from \
319                 your config; set `subscriber-channel-size` under each pipeline's `ingestion` \
320                 section if you need to override the default."
321            );
322        }
323
324        framework::ingestion::IngestionConfig {
325            ingest_concurrency: config.ingest_concurrency,
326            retry_interval_ms: config.retry_interval_ms,
327            streaming_backoff_initial_batch_size: config.streaming_backoff_initial_batch_size,
328            streaming_backoff_max_batch_size: config.streaming_backoff_max_batch_size,
329            streaming_connection_timeout_ms: config.streaming_connection_timeout_ms,
330            streaming_statement_timeout_ms: config.streaming_statement_timeout_ms,
331        }
332    }
