9#[DefaultConfig]
10#[derive(Clone, Default, Debug)]
11pub struct IndexerConfig {
12    pub ingestion: IngestionConfig,
13    pub committer: CommitterLayer,
14    pub pipeline: PipelineLayer,
15    /// Global rate limit (rows per second) shared across all pipelines.
16    pub total_max_rows_per_second: Option<u64>,
17    /// Default per-pipeline rate limit (rows per second).
18    /// Individual pipelines can override via their `ConcurrentLayer`.
19    pub max_rows_per_second: Option<u64>,
20    /// Number of gRPC channels in the Bigtable connection pool (default: 10).
21    /// A good rule of thumb is to target ~25 concurrent RPCs per channel, so
22    /// ceil(sum of write_concurrency across all pipelines / 25).
23    pub bigtable_connection_pool_size: Option<usize>,
24    /// Channel-level timeout in milliseconds for BigTable gRPC calls (default: 60000).
25    pub bigtable_channel_timeout_ms: Option<u64>,
26}

28#[DefaultConfig]
29#[derive(Clone, Default, Debug)]
30pub struct CommitterLayer {
31    pub write_concurrency: Option<usize>,
32    pub collect_interval_ms: Option<u64>,
33    pub watermark_interval_ms: Option<u64>,
34    pub watermark_interval_jitter_ms: Option<u64>,
35}
36
37impl CommitterLayer {
38    pub fn finish(self, base: CommitterConfig) -> CommitterConfig {
39        CommitterConfig {
40            write_concurrency: self.write_concurrency.unwrap_or(base.write_concurrency),
41            collect_interval_ms: self.collect_interval_ms.unwrap_or(base.collect_interval_ms),
42            watermark_interval_ms: self
43                .watermark_interval_ms
44                .unwrap_or(base.watermark_interval_ms),
45            watermark_interval_jitter_ms: self
46                .watermark_interval_jitter_ms
47                .unwrap_or(base.watermark_interval_jitter_ms),
48        }
49    }
50}

52#[DefaultConfig]
53#[derive(Clone, Default, Debug)]
54pub struct ConcurrentLayer {
55    pub committer: Option<CommitterLayer>,
56    /// Maximum rows per BigTable batch for this pipeline.
57    pub max_rows: Option<usize>,
58    /// Per-pipeline rate limit (rows per second). Overrides the default
59    /// `IndexerConfig::max_rows_per_second` when set.
60    pub max_rows_per_second: Option<u64>,
61    pub fanout: Option<usize>,
62    pub min_eager_rows: Option<usize>,
63    pub max_pending_rows: Option<usize>,
64    pub max_watermark_updates: Option<usize>,
65}
66
67impl ConcurrentLayer {
68    pub fn finish(self, base: ConcurrentConfig) -> ConcurrentConfig {
69        ConcurrentConfig {
70            committer: if let Some(c) = self.committer {
71                c.finish(base.committer)
72            } else {
73                base.committer
74            },
75            pruner: None,
76            fanout: self.fanout.or(base.fanout).or(Some(num_cpus::get())),
77            min_eager_rows: self.min_eager_rows.or(base.min_eager_rows),
78            max_pending_rows: self.max_pending_rows.or(base.max_pending_rows),
79            max_watermark_updates: self.max_watermark_updates.or(base.max_watermark_updates),
80        }
81    }
82}

101/// This type is identical to [`framework::ingestion::IngestionConfig`], but is set-up to be
102/// serialized and deserialized by `serde`.
103#[DefaultConfig]
104#[derive(Clone, Debug)]
105#[serde(deny_unknown_fields)]
106pub struct IngestionConfig {
107    pub checkpoint_buffer_size: usize,
108    pub ingest_concurrency: usize,
109    pub retry_interval_ms: u64,
110    pub streaming_backoff_initial_batch_size: usize,
111    pub streaming_backoff_max_batch_size: usize,
112    pub streaming_connection_timeout_ms: u64,
113    pub streaming_statement_timeout_ms: u64,
114}
115
116impl Default for IngestionConfig {
117    fn default() -> Self {
118        framework::ingestion::IngestionConfig::default().into()
119    }
120}
121
122impl From<framework::ingestion::IngestionConfig> for IngestionConfig {
123    fn from(config: framework::ingestion::IngestionConfig) -> Self {
124        Self {
125            checkpoint_buffer_size: config.checkpoint_buffer_size,
126            ingest_concurrency: config.ingest_concurrency,
127            retry_interval_ms: config.retry_interval_ms,
128            streaming_backoff_initial_batch_size: config.streaming_backoff_initial_batch_size,
129            streaming_backoff_max_batch_size: config.streaming_backoff_max_batch_size,
130            streaming_connection_timeout_ms: config.streaming_connection_timeout_ms,
131            streaming_statement_timeout_ms: config.streaming_statement_timeout_ms,
132        }
133    }
134}
