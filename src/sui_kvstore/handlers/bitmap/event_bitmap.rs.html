4//! Event-keyed Roaring bitmap inverted index processor.
5//!
6//! Parallel to [`super::transaction_bitmap`], but bit positions
7//! correspond to packed `event_seq`s (see [`crate::tables::event_bitmap_index`])
8//! rather than `tx_sequence_number`s. Enables `list_events` to resolve matches
9//! directly in event-space with no over-fetch.
10
11use sui_indexer_alt_framework_store_traits::CommitterWatermark;
12use sui_inverted_index::IndexDimension;
13use sui_inverted_index::for_each_event_dimension;
14use sui_types::full_checkpoint_content::Checkpoint;
15use sui_types::transaction::TransactionDataAPI;
16
17use crate::tables::event_bitmap_index;
18
19use crate::handlers::bitmap::BitmapIndexProcessor;

21// Compile-time check that BUCKET_SIZE fits in u32 (required for RoaringBitmap bit positions).
22const _: () = assert!(event_bitmap_index::BUCKET_SIZE <= u32::MAX as u64);
23
24/// Event-keyed bitmap index: one bit per (dimension, packed event_seq).
25pub struct EventBitmapProcessor;
26
27impl BitmapIndexProcessor for EventBitmapProcessor {
28    const NAME: &'static str = "kvstore_event_dimensions";
29    const TABLE: &'static str = "event_bitmap_index";
30    const COLUMN: &'static str = event_bitmap_index::col::BITMAP;
31    const SCHEMA_VERSION: u32 = event_bitmap_index::SCHEMA_VERSION;
32    const BUCKET_ID_WIDTH: usize = event_bitmap_index::BUCKET_ID_WIDTH;
33
34    fn for_each_indexed_bit<F>(&self, checkpoint: &Checkpoint, mut emit: F)
35    where
36        F: FnMut(u64, u32, IndexDimension, &[u8]),
37    {
38        let cp = checkpoint.summary.data();
39        // network_total_transactions is cumulative *including* this checkpoint,
40        // so tx_lo is the first tx_seq in this checkpoint.
41        let tx_lo = cp.network_total_transactions - checkpoint.transactions.len() as u64;
42
43        for (i, tx) in checkpoint.transactions.iter().enumerate() {
44            let tx_seq = tx_lo + i as u64;
45            for_each_event_dimension(
46                tx.transaction.sender(),
47                &tx.effects,
48                tx.events.as_ref(),
49                |event_idx, dim, value| {
50                    let event_seq = event_bitmap_index::encode_event_seq(tx_seq, event_idx);
51                    let bucket_id = event_seq / event_bitmap_index::BUCKET_SIZE;
52                    let bit_position = (event_seq % event_bitmap_index::BUCKET_SIZE) as u32;
53                    emit(bucket_id, bit_position, dim, value);
54                },
55            );
56        }
57    }
58
59    fn is_sealed(bucket_id: u64, watermark: CommitterWatermark) -> bool {
60        // Bucket B is sealed once every future tx's smallest event_seq
61        // (`event_seq_lo(tx) = tx * MAX_EVENTS_PER_TX`) is past bucket B's
62        // upper end. Solve for the smallest tx satisfying that.
63        let seal_tx_hi = ((bucket_id + 1) * event_bitmap_index::BUCKET_SIZE)
64            .div_ceil(event_bitmap_index::MAX_EVENTS_PER_TX as u64);
65        watermark.tx_hi >= seal_tx_hi
66    }
67}

87    fn event(sender_idx: u8) -> Event {
88        Event::new(
89            &ObjectID::ZERO,
90            ident_str!("dup_mod"),
91            TestCheckpointBuilder::derive_address(sender_idx),
92            GAS::type_(),
93            vec![],
94        )
95    }
96
97    /// Row key for the EmitModule dimension at the package level. Every event
98    /// emitted by `event(_)` uses `ObjectID::ZERO` as its package, so this
99    /// helper produces a stable key across the test inputs. (Event extraction
100    /// emits EmitModule and EventType but not Sender — Sender is a tx-level
101    /// dimension, not event-level.)
102    fn emit_module_row_key(bucket_id: u64) -> Vec<u8> {
103        let dimension_key =
104            encode_dimension_key(IndexDimension::EmitModule, ObjectID::ZERO.as_ref());
105        event_bitmap_index::encode_row_key(
106            event_bitmap_index::SCHEMA_VERSION,
107            &dimension_key,
108            bucket_id,
109        )
110    }
111
112    fn row<'a>(values: &'a [BitmapIndexValue], row_key: &[u8]) -> &'a BitmapIndexValue {
113        values
114            .iter()
115            .find(|v| v.row_key.as_ref() == row_key)
116            .expect("row must be emitted")
117    }

119    #[tokio::test]
120    async fn preserves_duplicate_dimensions_across_distinct_events() {
121        let checkpoint = TestCheckpointBuilder::new(0)
122            .start_transaction(1)
123            .with_events(vec![event(1), event(1)])
124            .finish_transaction()
125            .build_checkpoint();
126
127        let values = BitmapIndexHandler::new(EventBitmapProcessor)
128            .process(&Arc::new(checkpoint))
129            .await
130            .unwrap();
131
132        let row_key = emit_module_row_key(0);
133        let value = row(&values, &row_key);
134        assert_eq!(value.bucket_id, 0);
135        assert_eq!(value.bitmap.len(), 2);
136        assert!(value.bitmap.contains(0));
137        assert!(value.bitmap.contains(1));
138    }

173    #[tokio::test]
174    async fn emits_existence_bit_for_every_event() {
175        let checkpoint = TestCheckpointBuilder::new(0)
176            .with_network_total_transactions(100)
177            .start_transaction(1)
178            .with_events(vec![event(1), event(2)])
179            .finish_transaction()
180            .start_transaction(1)
181            .with_events(vec![event(3)])
182            .finish_transaction()
183            .build_checkpoint();
184
185        let values = BitmapIndexHandler::new(EventBitmapProcessor)
186            .process(&Arc::new(checkpoint))
187            .await
188            .unwrap();
189
190        let extant_key = event_bitmap_index::encode_row_key(
191            event_bitmap_index::SCHEMA_VERSION,
192            // EventExtant is keyed by a singleton placeholder value byte.
193            &encode_dimension_key(IndexDimension::EventExtant, &[0x00]),
194            0,
195        );
196        let value = row(&values, &extant_key);
197        assert_eq!(value.bucket_id, 0);
198        // tx 100 emits 2 events (idx 0,1), tx 101 emits 1 (idx 0): 3 bits total,
199        // independent of the events' attributes.
200        assert_eq!(value.bitmap.len(), 3);
201        for (tx_seq, event_idx) in [(100, 0), (100, 1), (101, 0)] {
202            assert!(
203                value
204                    .bitmap
205                    .contains(event_bitmap_index::encode_event_seq(tx_seq, event_idx) as u32)
206            );
207        }
208    }

212        let txs_per_bucket =

214        let checkpoint = TestCheckpointBuilder::new(0)
215            .with_network_total_transactions(txs_per_bucket - 1)
216            .start_transaction(1)
217            .with_events(vec![event(1)])
218            .finish_transaction()
219            .start_transaction(1)
220            .with_events(vec![event(1)])
221            .finish_transaction()
222            .build_checkpoint();
223
224        let values = BitmapIndexHandler::new(EventBitmapProcessor)
225            .process(&Arc::new(checkpoint))
226            .await
227            .unwrap();
228
229        let bucket_0_key = emit_module_row_key(0);
230        let bucket_0 = row(&values, &bucket_0_key);
231        assert_eq!(bucket_0.bucket_id, 0);
232        assert_eq!(bucket_0.bitmap.len(), 1);
233        assert!(
234            bucket_0
235                .bitmap

239        let bucket_1_key = emit_module_row_key(1);

241        assert_eq!(bucket_1.bucket_id, 1);
242        assert_eq!(bucket_1.bitmap.len(), 1);
