19/// Bits contributed to one bitmap-index row for a single framework commit.
20pub struct BitmapIndexValue {
21    pub row_key: Bytes,
22    pub bucket_id: u64,
23    pub bitmap: RoaringBitmap,
24    // Max checkpoint for which this BitmapIndexValue contains bits
25    pub max_cp: u64,
26    // Checkpoint timestamp for max_cp
27    pub max_ts_ms: u64,
28    // Bitmaps for the same row are accumulated across commits. The tasks
29    // that merge the bitmaps together are "sharded." This allows parallelization
30    // of the compute and also reduces the working set size of the task.
31    // The shard is computed by hashing the row key modulo the desired number of shards.
32    // This is just an in-memory value used to route bitmaps to worker tasks, it's not
33    // stored in the database.
34    pub shard_id: u8,
35}
36
37/// A Roaring-bitmap inverted index definition.
38///
39/// The generic handler owns the shared checkpoint-to-row accumulation logic.
40/// Individual processors only describe how a checkpoint contributes bits and
41/// how those bits map into table-specific row keys.
42pub trait BitmapIndexProcessor {
43    const NAME: &'static str;
44    /// The BigTable table that holds this index.
45    const TABLE: &'static str;
46    /// The column qualifier that holds the serialized bitmap.
47    const COLUMN: &'static str;
48    /// Row-key schema version prefix.
49    const SCHEMA_VERSION: u32;
50    /// Decimal width used to zero-pad bucket ids in row keys.
51    const BUCKET_ID_WIDTH: usize;
52
53    fn for_each_indexed_bit<F>(&self, checkpoint: &Checkpoint, emit: F)
54    where
55        F: FnMut(u64, u32, IndexDimension, &[u8]);
56
57    /// Returns true when the persisted committer watermark guarantees no
58    /// future checkpoint will contribute a bit to `bucket_id`. The committer
59    /// uses this to advance the active bucket and evict sealed rows from
60    /// in-memory shard state.
61    ///
62    /// Each processor decides which watermark dimension matters: tx-keyed
63    /// bitmaps gate on `tx_hi`, checkpoint-keyed bitmaps gate on
64    /// `checkpoint_hi_inclusive`, etc.
65    fn is_sealed(bucket_id: u64, watermark: CommitterWatermark) -> bool;
66}
