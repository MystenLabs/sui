20/// Pipeline that writes epoch data in the legacy format to BigTable.
21/// This maintains backwards compatibility during migration.
22pub struct EpochLegacyPipeline;
23
24/// Epoch info with optional previous epoch update info.
25pub struct EpochLegacyBatch {
26    pub epoch_info: EpochInfo,
27    pub prev_epoch_update: Option<PrevEpochUpdate>,
28}
29
30pub struct PrevEpochUpdate {
31    pub epoch_id: u64,
32    pub end_checkpoint: Option<u64>,
33    pub end_timestamp_ms: Option<u64>,
34}
35
36#[async_trait::async_trait]
37impl Processor for EpochLegacyPipeline {
38    const NAME: &'static str = "kvstore_epochs_legacy";
39    type Value = EpochLegacyBatch;
40
41    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
42        let checkpoint_data: CheckpointData = checkpoint.as_ref().clone().into();
43
44        match checkpoint_data.epoch_info()? {
45            Some(epoch_info) => {
46                let prev_epoch_update = if epoch_info.epoch > 0 {
47                    Some(PrevEpochUpdate {
48                        epoch_id: epoch_info.epoch - 1,
49                        end_checkpoint: epoch_info.start_checkpoint.map(|sq| sq - 1),
50                        end_timestamp_ms: epoch_info.start_timestamp_ms,
51                    })
52                } else {
53                    None
54                };
55                Ok(vec![EpochLegacyBatch {
56                    epoch_info,
57                    prev_epoch_update,
58                }])
59            }
60            None => Ok(vec![]),
61        }
62    }
63}
64
65#[async_trait::async_trait]
66impl Handler for EpochLegacyPipeline {
67    type Store = BigTableStore;
68    type Batch = Option<EpochLegacyBatch>;
69
70    const MIN_EAGER_ROWS: usize = 1;
71
72    fn batch(
73        &self,
74        batch: &mut Self::Batch,
75        values: &mut std::vec::IntoIter<Self::Value>,
76    ) -> BatchStatus {
77        if let Some(value) = values.next() {
78            *batch = Some(value);
79            BatchStatus::Ready
80        } else {
81            BatchStatus::Pending
82        }
83    }
84
85    async fn commit<'a>(
86        &self,
87        batch: &Self::Batch,
88        conn: &mut <Self::Store as Store>::Connection<'a>,
89    ) -> anyhow::Result<usize> {
90        let Some(epoch_batch) = batch else {
91            return Ok(0);
92        };
93
94        let mut entries = Vec::with_capacity(2);
95
96        // Update previous epoch with end info if needed (read-modify-write)
97        if let Some(prev_update) = &epoch_batch.prev_epoch_update {
98            let prev_data = conn
99                .client()
100                .get_epoch(prev_update.epoch_id)
101                .await?
102                .with_context(|| {
103                    format!(
104                        "previous epoch {} not found when processing epoch {}",
105                        prev_update.epoch_id, epoch_batch.epoch_info.epoch
106                    )
107                })?;
108            let prev = EpochInfo {
109                epoch: prev_data.epoch.context("missing epoch")?,
110                protocol_version: prev_data.protocol_version,
111                start_timestamp_ms: prev_data.start_timestamp_ms,
112                start_checkpoint: prev_data.start_checkpoint,
113                reference_gas_price: prev_data.reference_gas_price,
114                system_state: prev_data.system_state,
115                end_checkpoint: prev_update.end_checkpoint,
116                end_timestamp_ms: prev_update.end_timestamp_ms,
117            };
118            entries.push(epoch_to_entry(&prev)?);
119        }
120
121        // Write the new epoch info
122        entries.push(epoch_to_entry(&epoch_batch.epoch_info)?);
123
124        conn.client()
125            .write_entries(tables::epochs::NAME, entries)
126            .await?;
127        Ok(1)
128    }
129}
130
131fn epoch_to_entry(epoch: &EpochInfo) -> anyhow::Result<Entry> {
132    Ok(tables::make_entry(
133        tables::epochs::encode_key(epoch.epoch),
134        tables::epochs::encode(epoch)?,
135        epoch.end_timestamp_ms.or(epoch.start_timestamp_ms),
136    ))
137}
