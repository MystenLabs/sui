21/// BigTable's hard limit is 100k mutations per MutateRows request.
22/// We cap at half which is still very large.
23const MAX_MUTATIONS_PER_BATCH: usize = 50_000;
24
25/// This is the batch size the official java client from Google uses.
26pub const DEFAULT_MAX_ROWS: usize = 100;
27
28/// Extension of `Processor` that specifies a BigTable table name.
29pub trait BigTableProcessor: Processor<Value = Entry> {
30    /// The BigTable table to write rows to.
31    const TABLE: &'static str;
32
33    /// How much concurrency to use when processing checkpoint data (default: 10).
34    const FANOUT: usize = 10;
35
36    /// Minimum rows before eager commit (default: 50).
37    const MIN_EAGER_ROWS: usize = 50;
38
39    /// Maximum pending rows before back-pressure kicks in (default: 5000).
40    const MAX_PENDING_ROWS: usize = 5000;
41}
42
43/// Generic wrapper that implements `concurrent::Handler` for any `BigTableProcessor`.
44///
45/// This adapter wraps a `BigTableProcessor` and provides the common batching and commit logic
46/// for writing entries to BigTable. Individual pipelines implement `BigTableProcessor`.
47pub struct BigTableHandler<P> {
48    processor: P,
49    max_rows: usize,
50    rate_limiter: Arc<CompositeRateLimiter>,
51}
52
53/// Batch of BigTable entries.
54/// Uses RwLock for interior mutability so we can remove succeeded entries on partial write failures.
55#[derive(Default)]
56pub struct BigTableBatch {
57    inner: RwLock<BigTableBatchInner>,
58}
59
60#[derive(Default)]
61struct BigTableBatchInner {
62    entries: BTreeMap<Bytes, Entry>,
63    total_mutations: usize,
64}
65
66impl<P> BigTableHandler<P>
67where
68    P: BigTableProcessor,
69{
70    pub(crate) fn new(
71        processor: P,
72        config: &ConcurrentLayer,
73        rate_limiter: Arc<CompositeRateLimiter>,
74    ) -> Self {
75        Self {
76            processor,
77            max_rows: config.max_rows.unwrap_or(DEFAULT_MAX_ROWS),
78            rate_limiter,
79        }
80    }
81}
82
83#[async_trait::async_trait]
84impl<P> Processor for BigTableHandler<P>
85where
86    P: BigTableProcessor + Send + Sync,
87{
88    const NAME: &'static str = P::NAME;
89    const FANOUT: usize = <P as BigTableProcessor>::FANOUT;
90    type Value = Entry;
91
92    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Self::Value>> {
93        self.processor.process(checkpoint).await
94    }
95}
96
97#[async_trait::async_trait]
98impl<P> Handler for BigTableHandler<P>
99where
100    P: BigTableProcessor + Send + Sync,
101{
102    type Store = BigTableStore;
103    type Batch = BigTableBatch;
104
105    const MIN_EAGER_ROWS: usize = P::MIN_EAGER_ROWS;
106    const MAX_PENDING_ROWS: usize = P::MAX_PENDING_ROWS;
107
108    fn batch(
109        &self,
110        batch: &mut Self::Batch,
111        values: &mut std::vec::IntoIter<Self::Value>,
112    ) -> BatchStatus {
113        let mut inner = batch.inner.write().unwrap();
114
115        for entry in values {
116            inner.total_mutations += entry.mutations.len();
117            inner.entries.insert(entry.row_key.clone(), entry);
118
119            if inner.entries.len() >= self.max_rows
120                || inner.total_mutations >= MAX_MUTATIONS_PER_BATCH
121            {
122                return BatchStatus::Ready;
123            }
124        }
125
126        BatchStatus::Pending
127    }
128
129    async fn commit<'a>(
130        &self,
131        batch: &Self::Batch,
132        conn: &mut <Self::Store as Store>::Connection<'a>,
133    ) -> anyhow::Result<usize> {
134        let entries_to_write: Vec<Entry> = {
135            let inner = batch.inner.read().unwrap();
136            if inner.entries.is_empty() {
137                return Ok(0);
138            }
139            inner.entries.values().cloned().collect()
140        };
141        let count = entries_to_write.len();
142
143        self.rate_limiter.acquire(count).await;
144
145        match conn
146            .client()
147            .write_entries(P::TABLE, entries_to_write)
148            .await
149        {
150            Ok(()) => Ok(count),
151            Err(e) => {
152                if let Some(partial) = e.downcast_ref::<PartialWriteError>() {
153                    let mut inner = batch.inner.write().unwrap();
154                    let failed: std::collections::BTreeSet<&Bytes> =
155                        partial.failed_keys.iter().map(|f| &f.key).collect();
156                    inner.entries.retain(|key, _| failed.contains(key));
157                }
158                Err(e)
159            }
160        }
161    }
162}

195    #[tokio::test]
196    async fn test_multi_round_partial_failure() {
197        let mock = MockBigtableServer::new();
198
199        // Round 1: 10 entries, odd indices fail (1,3,5,7,9) → 5 remain.
200        mock.expect(ExpectedCall {
201            row_keys: vec![
202                b"row0", b"row1", b"row2", b"row3", b"row4", b"row5", b"row6", b"row7", b"row8",
203                b"row9",
204            ],
205            failures: HashMap::from([(1, 8), (3, 8), (5, 8), (7, 8), (9, 8)]),
206        })
207        .await;
208
209        // Round 2: 5 entries (row1,row3,row5,row7,row9 in sorted order),
210        // positional indices 0,2,4 fail (row1,row5,row9) → 2 remain.
211        mock.expect(ExpectedCall {
212            row_keys: vec![b"row1", b"row3", b"row5", b"row7", b"row9"],
213            failures: HashMap::from([(0, 8), (2, 8), (4, 8)]),
214        })
215        .await;
216
217        // Round 3: 2 entries (row1,row5,row9), all succeed.
218        mock.expect(ExpectedCall {
219            row_keys: vec![b"row1", b"row5", b"row9"],
220            failures: HashMap::new(),
221        })
222        .await;
223
224        let (addr, _handle) = mock.start().await.unwrap();
225        let client =
226            BigTableClient::new_for_host(addr.to_string(), "test".to_string(), "test").unwrap();
227        let store = BigTableStore::new(client);
228        let mut conn = store.connect().await.unwrap();
229
230        let handler = BigTableHandler::new(
231            TestProcessor,
232            &ConcurrentLayer::default(),
233            Arc::new(CompositeRateLimiter::noop()),
234        );
235        let mut batch = BigTableBatch::default();
236        let entries: Vec<Entry> = (0..10)
237            .map(|i| make_entry(format!("row{i}").as_bytes()))
238            .collect();
239        handler.batch(&mut batch, &mut entries.into_iter());
240
241        // Round 1: partial failure, batch shrinks from 10 to 5.
242        let result = handler.commit(&batch, &mut conn).await;
243        assert!(result.is_err());
244        {
245            let inner = batch.inner.read().unwrap();
246            assert_eq!(inner.entries.len(), 5);
247            for key in [b"row1", b"row3", b"row5", b"row7", b"row9"] {
248                assert!(inner.entries.contains_key(key.as_slice()));
249            }
250        }
251
252        // Round 2: partial failure, batch shrinks from 5 to 3.
253        let result = handler.commit(&batch, &mut conn).await;
254        assert!(result.is_err());
255        {
256            let inner = batch.inner.read().unwrap();
257            assert_eq!(inner.entries.len(), 3);
258            for key in [b"row1", b"row5", b"row9"] {
259                assert!(inner.entries.contains_key(key.as_slice()));
260            }
261        }
262
263        // Round 3: all succeed, batch still has 3 entries (framework clears, not handler).
264        let result = handler.commit(&batch, &mut conn).await;
265        assert_eq!(result.unwrap(), 3);
266        {
267            let inner = batch.inner.read().unwrap();
268            assert_eq!(inner.entries.len(), 3);
269        }
270    }
