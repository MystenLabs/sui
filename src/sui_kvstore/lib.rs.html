124static WRITE_LEGACY_DATA: OnceLock<bool> = OnceLock::new();
125
126/// Set whether to write legacy data (legacy watermark row, epoch DEFAULT_COLUMN, tx column).
127/// Must be called before creating any pipelines. Panics if called more than once.
128pub fn set_write_legacy_data(value: bool) {
129    WRITE_LEGACY_DATA
130        .set(value)
131        .expect("write_legacy_data already set");
132}
133
134pub fn write_legacy_data() -> bool {
135    *WRITE_LEGACY_DATA.get_or_init(|| false)
136}

222#[async_trait]
223pub trait KeyValueStoreReader {
224    async fn get_objects(&mut self, objects: &[ObjectKey]) -> Result<Vec<Object>>;
225    async fn get_transactions(
226        &mut self,
227        transactions: &[TransactionDigest],
228    ) -> Result<Vec<TransactionData>>;
229    async fn get_checkpoints(
230        &mut self,
231        sequence_numbers: &[CheckpointSequenceNumber],
232    ) -> Result<Vec<CheckpointData>>;
233    async fn get_checkpoint_by_digest(
234        &mut self,
235        digest: CheckpointDigest,
236    ) -> Result<Option<CheckpointData>>;
237    /// Return the minimum watermark across the given pipelines, selecting the whole
238    /// watermark with the lowest `checkpoint_hi_inclusive`. Returns `None` if any
239    /// pipeline is missing a watermark.
240    async fn get_watermark_for_pipelines(
241        &mut self,
242        pipelines: &[&str],
243    ) -> Result<Option<Watermark>>;
244    /// Return the minimum watermark across all non-legacy pipelines.
245    async fn get_watermark(&mut self) -> Result<Option<Watermark>> {
246        self.get_watermark_for_pipelines(&WATERMARK_PIPELINES).await
247    }
248    async fn get_latest_object(&mut self, object_id: &ObjectID) -> Result<Option<Object>>;
249    async fn get_epoch(&mut self, epoch_id: EpochId) -> Result<Option<EpochData>>;
250    async fn get_latest_epoch(&mut self) -> Result<Option<EpochData>>;
251    async fn get_protocol_configs(
252        &mut self,
253        protocol_version: u64,
254    ) -> Result<Option<ProtocolConfigData>>;
255    async fn get_events_for_transactions(
256        &mut self,
257        keys: &[TransactionDigest],
258    ) -> Result<Vec<(TransactionDigest, TransactionEventsData)>>;
259
260    /// Resolve package_ids to their original_ids.
261    async fn get_package_original_ids(
262        &mut self,
263        package_ids: &[ObjectID],
264    ) -> Result<Vec<(ObjectID, ObjectID)>>;
265
266    /// Get packages by (original_id, version) pairs.
267    async fn get_packages_by_version(
268        &mut self,
269        keys: &[(ObjectID, u64)],
270    ) -> Result<Vec<PackageData>>;
271
272    /// Get the latest version of a package at or before `cp_bound`.
273    async fn get_package_latest(
274        &mut self,
275        original_id: ObjectID,
276        cp_bound: u64,
277    ) -> Result<Option<PackageData>>;
278
279    /// Paginate package versions for an original_id, filtered by cp_bound.
280    async fn get_package_versions(
281        &mut self,
282        original_id: ObjectID,
283        cp_bound: u64,
284        after_version: Option<u64>,
285        before_version: Option<u64>,
286        limit: usize,
287        descending: bool,
288    ) -> Result<Vec<PackageData>>;
289
290    /// Get packages created in a checkpoint range, ordered by checkpoint.
291    async fn get_packages_by_checkpoint_range(
292        &mut self,
293        cp_after: Option<u64>,
294        cp_before: Option<u64>,
295        limit: usize,
296        descending: bool,
297    ) -> Result<Vec<PackageData>>;
298
299    /// Get all system packages with their latest version at or before `cp_bound`.
300    async fn get_system_packages(
301        &mut self,
302        cp_bound: u64,
303        after_original_id: Option<ObjectID>,
304        limit: usize,
305    ) -> Result<Vec<PackageData>>;
306}

308impl BigTableIndexer {
309    pub async fn new(
310        store: BigTableStore,
311        indexer_args: IndexerArgs,
312        client_args: ClientArgs,
313        ingestion_config: IngestionConfig,
314        committer: CommitterConfig,
315        config: IndexerConfig,
316        pipeline: PipelineLayer,
317        chain: Chain,
318        registry: &Registry,
319    ) -> Result<Self> {
320        let mut indexer = Indexer::new(
321            store,
322            indexer_args,
323            client_args,
324            ingestion_config.into(),
325            None,
326            registry,
327        )
328        .await?;
329
330        let global = config.total_max_rows_per_second.map(RateLimiter::new);
331        let base_rps = config.max_rows_per_second;
332
333        fn build_rate_limiter(
334            layer: &ConcurrentLayer,
335            base_rps: Option<u64>,
336            global: &Option<Arc<RateLimiter>>,
337        ) -> Arc<CompositeRateLimiter> {
338            let mut limiters = Vec::new();
339            if let Some(rps) = layer.max_rows_per_second.or(base_rps) {
340                limiters.push(RateLimiter::new(rps));
341            }
342            if let Some(g) = global {
343                limiters.push(g.clone());
344            }
345            Arc::new(CompositeRateLimiter::new(limiters))
346        }
347
348        let base = ConcurrentConfig {
349            committer,
350            pruner: None,
351            ..Default::default()
352        };
353
354        indexer
355            .concurrent_pipeline(
356                BigTableHandler::new(
357                    CheckpointsPipeline,
358                    &pipeline.checkpoints,
359                    build_rate_limiter(&pipeline.checkpoints, base_rps, &global),
360                ),
361                pipeline.checkpoints.finish(base.clone()),
362            )
363            .await?;
364        indexer
365            .concurrent_pipeline(
366                BigTableHandler::new(
367                    CheckpointsByDigestPipeline,
368                    &pipeline.checkpoints_by_digest,
369                    build_rate_limiter(&pipeline.checkpoints_by_digest, base_rps, &global),
370                ),
371                pipeline.checkpoints_by_digest.finish(base.clone()),
372            )
373            .await?;
374        indexer
375            .concurrent_pipeline(
376                BigTableHandler::new(
377                    TransactionsPipeline,
378                    &pipeline.transactions,
379                    build_rate_limiter(&pipeline.transactions, base_rps, &global),
380                ),
381                pipeline.transactions.finish(base.clone()),
382            )
383            .await?;
384        indexer
385            .concurrent_pipeline(
386                BigTableHandler::new(
387                    ObjectsPipeline,
388                    &pipeline.objects,
389                    build_rate_limiter(&pipeline.objects, base_rps, &global),
390                ),
391                pipeline.objects.finish(base.clone()),
392            )
393            .await?;
394        indexer
395            .concurrent_pipeline(
396                BigTableHandler::new(
397                    EpochStartPipeline,
398                    &pipeline.epoch_start,
399                    build_rate_limiter(&pipeline.epoch_start, base_rps, &global),
400                ),
401                pipeline.epoch_start.finish(base.clone()),
402            )
403            .await?;
404        indexer
405            .concurrent_pipeline(
406                BigTableHandler::new(
407                    EpochEndPipeline,
408                    &pipeline.epoch_end,
409                    build_rate_limiter(&pipeline.epoch_end, base_rps, &global),
410                ),
411                pipeline.epoch_end.finish(base.clone()),
412            )
413            .await?;
414        indexer
415            .concurrent_pipeline(
416                BigTableHandler::new(
417                    ProtocolConfigsPipeline(chain),
418                    &pipeline.protocol_configs,
419                    build_rate_limiter(&pipeline.protocol_configs, base_rps, &global),
420                ),
421                pipeline.protocol_configs.finish(base.clone()),
422            )
423            .await?;
424        indexer
425            .concurrent_pipeline(
426                BigTableHandler::new(
427                    PackagesPipeline,
428                    &pipeline.packages,
429                    build_rate_limiter(&pipeline.packages, base_rps, &global),
430                ),
431                pipeline.packages.finish(base.clone()),
432            )
433            .await?;
434        indexer
435            .concurrent_pipeline(
436                BigTableHandler::new(
437                    PackagesByIdPipeline,
438                    &pipeline.packages_by_id,
439                    build_rate_limiter(&pipeline.packages_by_id, base_rps, &global),
440                ),
441                pipeline.packages_by_id.finish(base.clone()),
442            )
443            .await?;
444        indexer
445            .concurrent_pipeline(
446                BigTableHandler::new(
447                    PackagesByCheckpointPipeline,
448                    &pipeline.packages_by_checkpoint,
449                    build_rate_limiter(&pipeline.packages_by_checkpoint, base_rps, &global),
450                ),
451                pipeline.packages_by_checkpoint.finish(base.clone()),
452            )
453            .await?;
454        indexer
455            .concurrent_pipeline(
456                BigTableHandler::new(
457                    SystemPackagesPipeline,
458                    &pipeline.system_packages,
459                    build_rate_limiter(&pipeline.system_packages, base_rps, &global),
460                ),
461                pipeline.system_packages.finish(base.clone()),
462            )
463            .await?;
464
465        if write_legacy_data() {
466            indexer
467                .concurrent_pipeline(EpochLegacyPipeline, pipeline.epoch_legacy.finish(base))
468                .await?;
469        }
470
471        Ok(Self { indexer })
472    }
473
474    pub fn pipeline_names(&self) -> Vec<&'static str> {
475        self.indexer.pipelines().collect()
476    }
477}
