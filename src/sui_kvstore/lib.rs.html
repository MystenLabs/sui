25#[async_trait]
26pub trait KeyValueStoreReader {
27    async fn get_objects(&mut self, objects: &[ObjectKey]) -> Result<Vec<Object>>;
28    async fn get_transactions(
29        &mut self,
30        transactions: &[TransactionDigest],
31    ) -> Result<Vec<TransactionData>>;
32    async fn get_checkpoints(
33        &mut self,
34        sequence_numbers: &[CheckpointSequenceNumber],
35    ) -> Result<Vec<Checkpoint>>;
36    async fn get_checkpoint_by_digest(
37        &mut self,
38        digest: CheckpointDigest,
39    ) -> Result<Option<Checkpoint>>;
40    async fn get_latest_checkpoint(&mut self) -> Result<CheckpointSequenceNumber>;
41    async fn get_latest_checkpoint_summary(&mut self) -> Result<Option<CheckpointSummary>>;
42    async fn get_latest_object(&mut self, object_id: &ObjectID) -> Result<Option<Object>>;
43    async fn get_epoch(&mut self, epoch_id: EpochId) -> Result<Option<EpochInfo>>;
44    async fn get_latest_epoch(&mut self) -> Result<Option<EpochInfo>>;
45    async fn get_events_for_transactions(
46        &mut self,
47        keys: &[TransactionDigest],
48    ) -> Result<Vec<(TransactionDigest, TransactionEventsData)>>;
49}

51#[async_trait]
52pub trait KeyValueStoreWriter {
53    async fn save_objects(&mut self, objects: &[&Object], timestamp_ms: TimestampMs) -> Result<()>;
54    async fn save_transactions(&mut self, transactions: &[TransactionData]) -> Result<()>;
55    async fn save_checkpoint(&mut self, checkpoint: &CheckpointData) -> Result<()>;
56    async fn save_watermark(&mut self, watermark: CheckpointSequenceNumber) -> Result<()>;
57    async fn save_epoch(&mut self, epoch: EpochInfo) -> Result<()>;
58}

60#[derive(Clone, Debug, Serialize, Deserialize)]
61pub struct Checkpoint {

67#[derive(Clone, Debug, Serialize, Deserialize)]
68pub struct TransactionData {
69    pub transaction: Transaction,
70    pub effects: TransactionEffects,
71    pub events: Option<TransactionEvents>,
72    pub checkpoint_number: CheckpointSequenceNumber,
73    pub timestamp: u64,
74}

76/// Partial transaction and events for when we only need transaction content for events
