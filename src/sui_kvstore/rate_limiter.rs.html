4use std::num::NonZeroU32;
5use std::sync::Arc;
6
7use governor::Quota;
8use governor::RateLimiter as GovRateLimiter;
9
10/// A single token-bucket rate limiter that can be shared across pipelines via `Arc`.
11pub(crate) struct RateLimiter {
12    limiter: governor::DefaultDirectRateLimiter,
13    burst: NonZeroU32,
14}
15
16/// A composed rate limiter that acquires tokens from one or more underlying limiters.
17/// Used to enforce both per-pipeline and global rate limits with a single `acquire` call.
18pub(crate) struct CompositeRateLimiter {
19    limiters: Vec<Arc<RateLimiter>>,
20}
21
22impl RateLimiter {
23    pub(crate) fn new(max_rows_per_second: u64) -> Arc<Self> {
24        let burst =
25            NonZeroU32::new(max_rows_per_second as u32).expect("max_rows_per_second must be > 0");
26        let quota = Quota::per_second(burst);
27        Arc::new(Self {
28            limiter: GovRateLimiter::direct(quota),
29            burst,
30        })
31    }
32
33    async fn acquire(&self, count: usize) {
34        let burst = self.burst.get();
35        let mut remaining = count as u32;
36        while remaining > 0 {
37            let take = remaining.min(burst);
38            self.limiter
39                .until_n_ready(NonZeroU32::new(take).unwrap())
40                .await
41                .expect("take <= burst, so this cannot fail");
42            remaining -= take;
43        }
44    }
45}
46
47impl CompositeRateLimiter {
48    pub(crate) fn new(limiters: Vec<Arc<RateLimiter>>) -> Self {
49        Self { limiters }
50    }
51
52    #[cfg(test)]
53    pub(crate) fn noop() -> Self {
54        Self {
55            limiters: Vec::new(),
56        }
57    }
58
59    /// Acquire `count` tokens from every underlying limiter concurrently.
60    pub(crate) async fn acquire(&self, count: usize) {
61        futures::future::join_all(self.limiters.iter().map(|l| l.acquire(count))).await;
62    }
63}
