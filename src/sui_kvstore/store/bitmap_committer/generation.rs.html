22/// Aggregated row-flush completions returned by the writer after BigTable
23/// accepts row writes for a checkpoint generation.
24pub(super) struct RowsFlushed {
25    pub(super) checkpoint: u64,
26    pub(super) count: u64,
27}

29/// Handle / shard / writer → generation task (BOUNDED).
30///
31/// Bounded so that a slow generation task applies backpressure to its producers,
32/// which cascades up to the framework's adaptive ingestion controller.
33/// Unbounded would let commit and row-flush messages pile up without limit when
34/// the generation task falls behind.
35pub(super) enum Event {
36    /// `Handle::commit` registered a new checkpoint generation. This is sent
37    /// before shard fanout so the generation task observes commits in
38    /// sequential checkpoint order.
39    GenerationStarted {
40        watermark: CommitterWatermark,
41        framework_commit_time: Instant,
42    },
43    /// One shard has merged its input and sent every row flush it discovered to
44    /// the writer.
45    ShardFlushesScheduled {
46        checkpoint: u64,
47        rows_scheduled: u64,
48    },
49    /// The writer persisted row writes for these checkpoint generations.
50    /// These can race ahead of `ShardFlushesScheduled` for the same generation;
51    /// promotion waits for both counters to line up.
52    RowsFlushed { counts: Vec<RowsFlushed> },
53}
54
55pub(super) struct GenerationWorker {
56    pub(super) pipeline: &'static str,
57    pub(super) generation_rx: mpsc::Receiver<Event>,
58    pub(super) shard_seal_senders: Vec<mpsc::UnboundedSender<shard::Seal>>,
59    pub(super) watermark_commit_tx: mpsc::Sender<watermark::Commit>,
60    pub(super) is_sealed: fn(u64, CommitterWatermark) -> bool,
61    pub(super) initial_watermarks: BitmapInitialWatermarks,
62}
63
64struct GenerationState {
65    watermark: CommitterWatermark,
66    framework_commit_time: std::time::Instant,
67    bucket_start_cp: u64,
68    shards_scheduled: u16,
69    rows_scheduled: u64,
70    rows_flushed: u64,
71}
72
73struct GenerationLoopState {
74    // Bucket containing the most recent observed watermark's `tx_hi`.
75    current_bucket_id: BucketId,
76    // Replay floor for `current_bucket_id`: `init_watermark` resumes from
77    // just before this checkpoint so the active bucket is rebuilt after a
78    // crash. This can be the checkpoint that made the bucket active, even if
79    // the first transaction in that bucket appears in the next checkpoint.
80    current_bucket_start_cp: u64,
81    next_bucket_to_evict: BucketId,
82    generations: BTreeMap<u64, GenerationState>,
83}
84
85impl GenerationWorker {
86    pub(super) async fn run(mut self) -> Result<()> {
87        info!(self.pipeline, "Bitmap generation task started");
88
89        let Some(mut state) = self.initialize().await? else {
90            return Ok(());
91        };
92
93        while let Some(msg) = self.generation_rx.recv().await {
94            self.handle_generation_event(&mut state, msg);
95            self.try_promote(&mut state).await;
96        }
97
98        info!(self.pipeline, "Bitmap generation task exiting");
99        Ok(())
100    }
101
102    async fn initialize(&mut self) -> Result<Option<GenerationLoopState>> {
103        // The framework runs `init_watermark` before the first `commit()`, and
104        // `commit()` always sends `GenerationStarted` before any shard can send row
105        // flush accounting. Treat that first generation as the init barrier.
106        let Some(first_msg) = self.generation_rx.recv().await else {
107            info!(
108                self.pipeline,
109                "Bitmap generation task exiting before first generation"
110            );
111            return Ok(None);
112        };
113        let Event::GenerationStarted {
114            watermark,
115            framework_commit_time,
116        } = first_msg
117        else {
118            bail!(
119                "bitmap generation task for `{}` received accounting event \
120                 before first GenerationStarted",
121                self.pipeline,
122            );
123        };
124
125        let startup = self.initial_watermarks.get(self.pipeline)?;
126        let startup_watermark = startup.watermark;
127        let startup_bucket_start_cp = startup.bucket_start_cp;
128        info!(
129            self.pipeline,
130            startup_tx_hi = startup_watermark.tx_hi,
131            startup_cp_hi = startup_watermark.checkpoint_hi_inclusive,
132            startup_bucket_start_cp,
133            "Bitmap generation task initial watermark loaded"
134        );
135
136        let mut state =
137            GenerationLoopState::new(startup_watermark, startup_bucket_start_cp, self.is_sealed);
138        state.record_generation_started(watermark, framework_commit_time, self.is_sealed);
139        Ok(Some(state))
140    }
141
142    fn handle_generation_event(&self, state: &mut GenerationLoopState, msg: Event) {
143        match msg {
144            Event::GenerationStarted {
145                watermark,
146                framework_commit_time,
147            } => {
148                state.record_generation_started(watermark, framework_commit_time, self.is_sealed);
149            }
150            Event::ShardFlushesScheduled {
151                checkpoint,
152                rows_scheduled,
153            } => {
154                if let Some(entry) = state.generations.get_mut(&checkpoint) {
155                    entry.record_shard_flushes_scheduled(rows_scheduled);
156                } else {
157                    warn!(
158                        self.pipeline,
159                        checkpoint, "ShardFlushesScheduled for unknown checkpoint"
160                    );
161                }
162            }
163            Event::RowsFlushed { counts } => {
164                for flushed in counts {
165                    if let Some(entry) = state.generations.get_mut(&flushed.checkpoint) {
166                        entry.record_rows_flushed(flushed.count);
167                    } else {
168                        warn!(
169                            self.pipeline,
170                            checkpoint = flushed.checkpoint,
171                            "RowsFlushed for unknown checkpoint"
172                        );
173                    }
174                }
175            }
176        }
177    }
178
179    /// Walks generations in checkpoint order, dispatches a watermark for the newest
180    /// contiguous generation whose rows are durable, and drops every generation covered
181    /// by that request.
182    async fn try_promote(&self, state: &mut GenerationLoopState) {
183        let mut candidate_cp: Option<u64> = None;
184        for (&cp, entry) in state.generations.iter() {
185            if !entry.all_flushes_complete() {
186                break;
187            }
188            candidate_cp = Some(cp);
189        }
190
191        let Some(cp) = candidate_cp else {
192            return;
193        };
194        let (watermark, bucket_start_cp, framework_commit_time) = {
195            let entry = state
196                .generations
197                .get(&cp)
198                .expect("candidate generation exists");
199            (
200                entry.watermark,
201                entry.bucket_start_cp,
202                entry.framework_commit_time,
203            )
204        };
205
206        let previous_bucket_to_evict = state.next_bucket_to_evict;
207        while (self.is_sealed)(state.next_bucket_to_evict, watermark) {
208            state.next_bucket_to_evict += 1;
209        }
210        if state.next_bucket_to_evict > previous_bucket_to_evict {
211            for tx in &self.shard_seal_senders {
212                let _ = tx.send(shard::Seal {
213                    bucket_id_exclusive: state.next_bucket_to_evict,
214                });
215            }
216        }
217
218        let req = watermark::Commit {
219            watermark,
220            bucket_start_cp,
221            framework_commit_time,
222        };
223        if self.watermark_commit_tx.send(req).await.is_err() {
224            warn!(self.pipeline, "Watermark writer closed; cannot promote");
225            return;
226        }
227
228        while let Some((&front_cp, _)) = state.generations.first_key_value()
229            && front_cp <= cp
230        {
231            state.generations.pop_first();
232        }
233
234        debug!(
235            self.pipeline,
236            cp,
237            tx_hi = watermark.tx_hi,
238            bucket_start_cp,
239            "Watermark dispatched to writer"
240        );
241    }
242}
243
244impl GenerationState {
245    fn new(
246        watermark: CommitterWatermark,
247        framework_commit_time: std::time::Instant,
248        bucket_start_cp: u64,
249    ) -> Self {
250        Self {
251            watermark,
252            framework_commit_time,
253            bucket_start_cp,
254            shards_scheduled: 0,
255            rows_scheduled: 0,
256            rows_flushed: 0,
257        }
258    }
259
260    fn record_shard_flushes_scheduled(&mut self, rows_scheduled: u64) {
261        self.shards_scheduled += 1;
262        self.rows_scheduled += rows_scheduled;
263        debug_assert!(
264            self.shards_scheduled <= NUM_SHARDS as u16,
265            "generation scheduled more shards than exist"
266        );
267    }
268
269    fn record_rows_flushed(&mut self, count: u64) {
270        // Writer completions can race ahead of a shard's
271        // `ShardFlushesScheduled` message for the same generation because the
272        // shard sends row batches before it sends its scheduled count.
273        self.rows_flushed += count;
274    }
275
276    fn all_flushes_complete(&self) -> bool {
277        let all_shards_scheduled = self.shards_scheduled as usize == NUM_SHARDS;
278        if all_shards_scheduled {
279            debug_assert!(
280                self.rows_flushed <= self.rows_scheduled,
281                "generation flushed more rows than were scheduled"
282            );
283        }
284        all_shards_scheduled && self.rows_flushed == self.rows_scheduled
285    }
286}
287
288impl GenerationLoopState {
289    fn new(
290        startup_watermark: CommitterWatermark,
291        startup_bucket_start_cp: u64,
292        is_sealed: fn(u64, CommitterWatermark) -> bool,
293    ) -> Self {
294        let current_bucket_id = super::bucket_of(startup_watermark, is_sealed);
295        Self {
296            current_bucket_id,
297            current_bucket_start_cp: startup_bucket_start_cp,
298            next_bucket_to_evict: current_bucket_id,
299            generations: BTreeMap::new(),
300        }
301    }
302
303    fn record_generation_started(
304        &mut self,
305        watermark: CommitterWatermark,
306        framework_commit_time: std::time::Instant,
307        is_sealed: fn(u64, CommitterWatermark) -> bool,
308    ) {
309        let cp = watermark.checkpoint_hi_inclusive;
310        while is_sealed(self.current_bucket_id, watermark) {
311            self.current_bucket_id += 1;
312            self.current_bucket_start_cp = cp;
313        }
314        self.generations.insert(
315            cp,
316            GenerationState::new(
317                watermark,
318                framework_commit_time,
319                self.current_bucket_start_cp,
320            ),
321        );
322    }
323}

341    fn is_sealed(bucket_id: u64, watermark: CommitterWatermark) -> bool {
342        watermark.tx_hi >= (bucket_id + 1) * 10
343    }
344
345    fn watermark(cp: u64, tx_hi: u64) -> CommitterWatermark {
346        CommitterWatermark {
347            epoch_hi_inclusive: 0,
348            checkpoint_hi_inclusive: cp,
349            tx_hi,
350            timestamp_ms_hi_inclusive: cp * 1_000,
351        }
352    }
353
354    fn complete_generation(state: &mut GenerationState, rows: u64) {
355        state.record_rows_flushed(rows);
356        for _ in 0..NUM_SHARDS - 1 {
357            state.record_shard_flushes_scheduled(0);
358        }
359        state.record_shard_flushes_scheduled(rows);
360    }

362    fn worker(
363        watermark_commit_tx: mpsc::Sender<watermark::Commit>,
364        shard_seal_senders: Vec<mpsc::UnboundedSender<shard::Seal>>,
365    ) -> GenerationWorker {
366        let (_generation_tx, generation_rx) = mpsc::channel(1);
367        let mut init_results = HashMap::new();
368        init_results.insert(
369            PIPELINE.to_string(),
370            PipelineInitResult {
371                init: Some(InitWatermark {
372                    checkpoint_hi_inclusive: None,
373                    reader_lo: Some(0),
374                }),
375                bitmap: BitmapInitialWatermark {
376                    watermark: CommitterWatermark::default(),
377                    bucket_start_cp: 0,
378                },
379            },
380        );
381
382        GenerationWorker {
383            pipeline: PIPELINE,
384            generation_rx,
385            shard_seal_senders,
386            watermark_commit_tx,
387            is_sealed,
388            initial_watermarks: BitmapInitialWatermarks {
389                init_results: Arc::new(Mutex::new(init_results)),
390            },
391        }
392    }

442    #[tokio::test]
443    async fn promote_dispatches_latest_contiguous_watermark_and_bucket_seal() {
444        let (watermark_tx, mut watermark_rx) = mpsc::channel(1);
445        let (seal_tx, mut seal_rx) = seal_channel();
446        let worker = worker(watermark_tx, vec![seal_tx]);
447        let mut state = GenerationLoopState::new(CommitterWatermark::default(), 0, is_sealed);
448
449        state.record_generation_started(watermark(1, 5), Instant::now(), is_sealed);
450        state.record_generation_started(watermark(2, 25), Instant::now(), is_sealed);
451        state.record_generation_started(watermark(3, 35), Instant::now(), is_sealed);
452        complete_generation(state.generations.get_mut(&1).unwrap(), 1);
453        complete_generation(state.generations.get_mut(&2).unwrap(), 0);
454
455        worker.try_promote(&mut state).await;
456
457        let req = watermark_rx.try_recv().unwrap();
458        assert_eq!(req.watermark.checkpoint_hi_inclusive, 2);
459        assert_eq!(req.bucket_start_cp, 2);
460        let seal = seal_rx.try_recv().unwrap();
461        assert_eq!(seal.bucket_id_exclusive, 2);
462        assert!(!state.generations.contains_key(&1));
463        assert!(!state.generations.contains_key(&2));
464        assert!(state.generations.contains_key(&3));
465    }
