4//! Bitmap index committer.
5//!
6//! The bitmap pipelines use the framework's sequential mode: `commit()` calls
7//! arrive in checkpoint order, and commit `N + 1` is not delivered before
8//! commit `N` returns. This module depends on that ordering. At commit `N`, the
9//! process has seen every bitmap input through checkpoint `N`, so the watermark
10//! for `N` can be published once this module has durably written the row
11//! snapshots it scheduled for `N`.
12//!
13//! This committer exists because the generic sequential store path does not
14//! expose the controls bitmap indexing needs:
15//!
16//! - bitmap rows are long-lived accumulated buckets with deterministic boundaries,
17//!   not one framework commit's rows;
18//! - merge/serialize work must be parallelized across CPUs, while
19//!   the framework's `handler::batch()` is single-threaded.
20//!
21//! A generation is one framework commit, keyed by `checkpoint_hi_inclusive`.
22//! The generation task may send a watermark for checkpoint `N` only after all
23//! shards have reported their scheduled row count for `N` and the writer has
24//! reported that many durable row writes for `N`. Empty shards still report
25//! zero scheduled rows; otherwise a sparse commit could never advance.
26//!
27//! Actor/message flow:
28//!
29//! ```text
30//! framework commit
31//!        │
32//!        ▼
33//! ┌────────────────┐
34//! │ Handle::commit │
35//! └──┬──────────┬──┘
36//!    │          │
37//!    │          └─ shard::Merge x NUM_SHARDS ──▶ ┌────────────────────────────┐
38//!    │                                           │ ShardWorker x NUM_SHARDS   │
39//!    │                                           └──┬──────────────▲────┬─────┘
40//!    │                                              │              │    │
41//!    │                                 ShardFlushesScheduled       │    │writer::Batch
42//!    │                                              │              │    ▼
43//!    │                                              │              │ ┌───────────┐
44//!    │                                              │              │ │ Writer    │ ──▶ BigTable rows
45//!    │                                              │              │ └──────┬────┘     (retry/backoff)
46//!    │                                              │              │        │
47//!    │                                              │         shard::Seal   │RowsFlushed
48//!    │GenerationStarted                             │          (evict)      │
49//!    ▼                                              ▼              │        ▼
50//! ┌──────────────────────────────────────────────────────────────────────────┐
51//! │                            GenerationWorker                              │
52//! └────────────────────────────────────┬─────────────────────────────────────┘
53//!                                      │
54//!                         watermark::Commit
55//!                                      ▼
56//!                              ┌─────────────────┐
57//!                              │ WatermarkWriter │ ──▶ BigTable watermark
58//!                              └─────────────────┘       (retry/backoff)
59//! ```
60//!
61//! Bounded channels provide backpressure to the framework. Per-shard merge
62//! channels have capacity `1`; row batches, generation events, and watermark
63//! commits use the small capacities below. The only unbounded channel is
64//! generation-to-shard [`shard::Seal`]: seal messages are tiny and bucket-rate,
65//! and eviction is cleanup, while `GenerationWorker` is the accounting hub.
66//! Awaiting shard cleanup capacity there could block it from draining
67//! writer/shard progress.
68//!
69//! Restart safety comes from two pieces of state in the watermark row. The
70//! normal `tx_hi` tells the committer which buckets were already sealed before
71//! startup, so replayed rows for older buckets are ignored. The bitmap-only
72//! `bucket_start_cp` tells `init_watermark` how far to rewind the framework so
73//! the active bucket is rebuilt from replay.

110const COMMIT_RETRY_BACKOFF: Duration = Duration::from_millis(100);

112/// Number of hash shards across which accumulated rows are partitioned.
113/// Power of two for cheap masking. One tokio task owns each shard for the
114/// life of the process. It should be at least `num_cpus`, but making it larger
115/// can reduce each shard's active working set and improve cache affinity.
116pub(crate) const NUM_SHARDS: usize = 64;
117const SHARD_MASK: usize = NUM_SHARDS - 1;
118
119// Compile-time guarantee that `BitmapIndexValue::shard_id` (u8) can hold
120// every shard index.
121const _: () = assert!(NUM_SHARDS <= u8::MAX as usize + 1);
122
123/// Bounded capacity of each per-shard merge channel. Small — `Handle::commit`
124/// fills fast. When full, `Handle::commit` blocks on `.send()` →
125/// `Handler::commit` blocks → framework throttles.
126const SHARD_MERGE_CHANNEL_CAPACITY: usize = 1;
127
128/// Generation receives `1 + NUM_SHARDS + row-completion-batches` messages per
129/// framework commit. Keep enough room for a few commits of bookkeeping without
130/// letting this become an unbounded proxy for stalled watermark/row accounting.
131const GENERATION_CHANNEL_CAPACITY: usize = NUM_SHARDS * 8;
132
133/// The store connection's view of the committer after [`BitmapCommitter::spawn`]
134/// returns: the per-shard work senders. Task `JoinHandle`s live in the
135/// returned [`Service`], not here.
136#[derive(Clone)]
137pub(crate) struct Handle {
138    pipeline: &'static str,
139    shard_merge_senders: Vec<mpsc::Sender<shard::Merge>>,
140    generation_tx: mpsc::Sender<generation::Event>,
141    rows_written: Arc<AtomicU64>,
142}
143
144impl Handle {
145    pub(crate) fn take_rows_written(&self) -> usize {
146        self.rows_written.swap(0, Ordering::Relaxed) as usize
147    }
148
149    /// Fan out per-shard merge requests. Returns `Err` if any downstream
150    /// task has exited.
151    ///
152    /// `batch` is pre-partitioned by `shard_id` in `Handler::batch`: slot
153    /// `i` holds an `Arc` to shard `i`'s values. We clone (refcount bump)
154    /// and ship.
155    ///
156    /// Empty shards still receive a merge — every shard owns one
157    /// flush-scheduling unit per commit and must finish it. Skipping empty
158    /// shards would leave sparse-commit watermarks unable to promote.
159    ///
160    /// Backpressure: waits until every shard accepts its merge. A full shard
161    /// channel blocks this commit, which cascades up to the framework's writer.
162    pub(crate) async fn commit(
163        &self,
164        batch: Vec<Arc<Vec<BitmapIndexValue>>>,
165        watermark: CommitterWatermark,
166    ) -> Result<(), ()> {
167        debug_assert_eq!(
168            batch.len(),
169            NUM_SHARDS,
170            "Handler::batch must initialize all shard slots",
171        );
172
173        let framework_commit_time = Instant::now();
174        let checkpoint = watermark.checkpoint_hi_inclusive;
175        // This must precede shard fanout: generation state observes commits in
176        // the same cp order as the sequential framework calls `commit()`.
177        if self
178            .generation_tx
179            .send(generation::Event::GenerationStarted {
180                watermark,
181                framework_commit_time,
182            })
183            .await
184            .is_err()
185        {
186            warn!(
187                self.pipeline,
188                "Generation task closed during generation start"
189            );
190            return Err(());
191        }
192        let sends = self
193            .shard_merge_senders
194            .iter()
195            .zip_debug_eq(batch.into_iter())
196            .enumerate()
197            .map(|(shard_id, (tx, values))| async move {
198                let merge = shard::Merge { checkpoint, values };
199                tx.send(merge).await.map_err(|_| shard_id)
200            });
201        if let Err(shard_id) = try_join_all(sends).await {
202            warn!(
203                self.pipeline,
204                shard_id, "Shard closed during merge request send"
205            );
206            return Err(());
207        }
208
209        Ok(())
210    }
211}

213/// Configuration and dependencies for the async bitmap committer.
214/// Populate the fields and call [`Self::spawn`] to wire the background
215/// tasks.
216pub(crate) struct BitmapCommitter {
217    pub pipeline: &'static str,
218    pub table: &'static str,
219    pub column: &'static str,
220    pub is_sealed: fn(u64, CommitterWatermark) -> bool,
221    pub initial_watermarks: BitmapInitialWatermarks,
222    pub client: BigTableClient,
223    pub rate_limiter: Arc<CompositeRateLimiter>,
224    pub write_chunk_size: usize,
225    pub write_concurrency: usize,
226    pub metrics: Arc<BitmapIndexMetrics>,
227}
228
229pub(crate) type BitmapCommitterHandle = Handle;
230pub(crate) type BucketId = u64;
231pub(crate) type ShardId = u16;
232
233impl BitmapCommitter {
234    /// Wire the committer. Returns `(Handle, Service)`: the handle is the
235    /// store connection's send side, and the `Service` owns
236    /// every background task as a primary. The caller is expected to
237    /// `merge` the `Service` into the framework's indexer service so
238    /// panics propagate and shutdown is coordinated.
239    pub(crate) fn spawn(self) -> (Handle, Service) {
240        let Self {
241            pipeline,
242            table,
243            column,
244            is_sealed,
245            initial_watermarks,
246            client,
247            rate_limiter,
248            write_chunk_size,
249            write_concurrency,
250            metrics,
251        } = self;
252
253        // The generation task reads its initial watermark (`tx_hi` and
254        // `bucket_start_cp`) from store init state when it receives the first
255        // `GenerationStarted`. By then the framework's `init_watermark` call
256        // has returned and populated the entry for `pipeline`.
257        let (write_tx, write_rx) = mpsc::channel::<writer::Batch>(bounded_channel_capacity());
258        // Bounded so a slow generation task applies backpressure to its producers.
259        let (generation_tx, generation_rx) =
260            mpsc::channel::<generation::Event>(GENERATION_CHANNEL_CAPACITY);
261        // Generation task → dedicated BigTable watermark committer. Bounded so
262        // a wedged watermark writer eventually backpressures generation.
263        let (watermark_commit_tx, watermark_commit_rx) =
264            mpsc::channel::<watermark::Commit>(bounded_channel_capacity());
265        let rows_written = Arc::new(AtomicU64::new(0));
266
267        // Per-shard channels. Merge inputs are bounded so shard backlog
268        // backpressures framework ingestion. Eviction commands are unbounded:
269        // they are tiny bucket-rate messages, and eviction is cleanup, while
270        // GenerationWorker is the accounting hub. Awaiting shard cleanup
271        // capacity there could block it from draining writer/shard progress.
272        let mut shard_merge_senders: Vec<mpsc::Sender<shard::Merge>> =
273            Vec::with_capacity(NUM_SHARDS);
274        let mut shard_merge_receivers: Vec<mpsc::Receiver<shard::Merge>> =
275            Vec::with_capacity(NUM_SHARDS);
276        let mut shard_seal_senders: Vec<mpsc::UnboundedSender<shard::Seal>> =
277            Vec::with_capacity(NUM_SHARDS);
278        let mut shard_seal_receivers: Vec<mpsc::UnboundedReceiver<shard::Seal>> =
279            Vec::with_capacity(NUM_SHARDS);
280        for _ in 0..NUM_SHARDS {
281            let (itx, irx) = mpsc::channel::<shard::Merge>(SHARD_MERGE_CHANNEL_CAPACITY);
282            #[allow(clippy::disallowed_methods)]
283            let (etx, erx) = mpsc::unbounded_channel::<shard::Seal>();
284            shard_merge_senders.push(itx);
285            shard_merge_receivers.push(irx);
286            shard_seal_senders.push(etx);
287            shard_seal_receivers.push(erx);
288        }
289
290        // All tasks go into a single `Service` as primary tasks. Task errors
291        // and panics surface via tokio's `JoinSet` and are re-raised by the
292        // Service's `main()`.
293        let mut service = Service::new();
294        for (i, (merge_rx, seal_rx)) in shard_merge_receivers
295            .into_iter()
296            .zip_debug_eq(shard_seal_receivers)
297            .enumerate()
298        {
299            let shard_write_tx = write_tx.clone();
300            let generation_tx = generation_tx.clone();
301            let metrics = metrics.clone();
302            let initial_watermarks = initial_watermarks.clone();
303            let state = Shard::new(i as ShardId, table);
304            let worker = ShardWorker {
305                pipeline,
306                shard: state,
307                merge_rx,
308                seal_rx,
309                write_tx: shard_write_tx,
310                generation_tx,
311                initial_watermarks,
312                is_sealed,
313                startup_min_unsealed_bucket: Default::default(),
314                metrics,
315            };
316            service = service.spawn(async move { worker.run().await });
317        }
318
319        drop(write_tx);
320
321        service = service.spawn({
322            let client = client.clone();
323            let generation_tx = generation_tx.clone();
324            let metrics = metrics.clone();
325            let rows_written = rows_written.clone();
326            let worker = Writer {
327                pipeline,
328                table,
329                column,
330                client,
331                rate_limiter,
332                write_rx,
333                generation_tx,
334                write_chunk_size,
335                write_concurrency,
336                rows_written,
337                metrics,
338            };
339            async move {
340                worker.run().await;
341                Ok(())
342            }
343        });
344
345        service = service.spawn({
346            let metrics = metrics.clone();
347            let worker = WatermarkWriter {
348                pipeline,
349                client,
350                commit_rx: watermark_commit_rx,
351                metrics,
352            };
353            async move {
354                worker.run().await;
355                Ok(())
356            }
357        });
358
359        let handle_generation_tx = generation_tx.clone();
360        drop(generation_tx);
361
362        service = service.spawn({
363            let worker = GenerationWorker {
364                pipeline,
365                generation_rx,
366                shard_seal_senders,
367                watermark_commit_tx,
368                is_sealed,
369                initial_watermarks: initial_watermarks.clone(),
370            };
371            async move { worker.run().await }
372        });
373
374        let handle = Handle {
375            pipeline,
376            shard_merge_senders,
377            generation_tx: handle_generation_tx,
378            rows_written,
379        };
380        (handle, service)
381    }
382}

384/// Default capacity for ordinary bounded control/data channels. Small queues
385/// keep backpressure close to the stalled subsystem instead of retaining lots
386/// of already-produced work in memory.
387fn bounded_channel_capacity() -> usize {
388    num_cpus::get().saturating_div(2).max(4)
389}
390
391/// Smallest bucket not sealed by `watermark`. The processor's `is_sealed`
392/// closure decides which watermark dimension matters; this loop walks until
393/// the first unsealed bucket so the next bucket is the only one that may need
394/// replay.
395fn bucket_of(
396    watermark: CommitterWatermark,
397    is_sealed: fn(u64, CommitterWatermark) -> bool,
398) -> BucketId {
399    let mut b: BucketId = 0;
400    while is_sealed(b, watermark) {
401        b += 1;
402    }
403    b
404}
405
406/// Shard index for a row key. Needs to be deterministic within a single
407/// process (handler and shards must agree on routing); cross-process
408/// stability isn't required — shard state is entirely in-memory and
409/// rebuilt on restart from bucket-start replay.
410pub(crate) fn shard_for(key: &[u8]) -> usize {
411    let mut h = rustc_hash::FxHasher::default();
412    h.write(key);
413    (h.finish() as usize) & SHARD_MASK
414}

416#[cfg(test)]
417mod bucket_of_tests {
418    use super::*;
419
420    fn is_sealed(b: u64, watermark: CommitterWatermark) -> bool {
421        watermark.tx_hi >= (b + 1) * 10
422    }
423
424    fn wm(tx_hi: u64) -> CommitterWatermark {
425        CommitterWatermark {
426            tx_hi,
427            ..Default::default()
428        }
429    }
430
431    /// `is_sealed(b, wm)` true means bucket `b` is sealed, so `bucket_of`
432    /// returns the next bucket. Pin the boundary semantics: at `tx_hi`
433    /// exactly equal to `(b + 1) * 10`, bucket `b` is sealed.
434    #[test]
435    fn bucket_of_handles_boundary() {
436        assert_eq!(bucket_of(wm(0), is_sealed), 0);
437        assert_eq!(bucket_of(wm(9), is_sealed), 0);
438        assert_eq!(bucket_of(wm(10), is_sealed), 1, "tx_hi == 10 seals 0");
439        assert_eq!(bucket_of(wm(11), is_sealed), 1);
440        assert_eq!(bucket_of(wm(20), is_sealed), 2, "tx_hi == 20 seals 1");
441        assert_eq!(bucket_of(wm(25), is_sealed), 2);
442    }
