31const MAX_ROW_WRITE_RETRY_BACKOFF: Duration = Duration::from_secs(1);

33/// A single row write destined for BigTable, serialized on a shard task and
34/// shipped to the writer.
35pub(super) struct Row {
36    pub(super) row_key: Bytes,
37    pub(super) serialized: Bytes,
38    pub(super) max_ts_ms: u64,
39    /// Checkpoint generation that scheduled this row write. The writer
40    /// reports durable rows by this value, so generation accounting is a simple
41    /// counter.
42    pub(super) generation_cp: u64,
43}
44
45/// Bounded unit sent from shards to the writer. Shards serialize rows
46/// into batches so backpressure starts before a single shard can enqueue an
47/// unbounded number of individual row writes.
48pub(super) struct Batch {
49    pub(super) rows: Vec<Row>,
50}

87impl Writer {
88    pub(super) async fn run(self) {
89        let Self {
90            pipeline,
91            table,
92            column,
93            client,
94            rate_limiter,
95            write_rx,
96            generation_tx,
97            write_chunk_size,
98            write_concurrency,
99            rows_written,
100            metrics,
101        } = self;
102
103        info!(pipeline, "Bitmap row writer started");
104
105        let write_chunk_size = write_chunk_size.max(1);
106        let write_concurrency = write_concurrency.max(1);
107        let context = WriteContext {
108            pipeline,
109            table,
110            column,
111            client,
112            rate_limiter,
113            generation_tx,
114            rows_written,
115            metrics,
116        };
117
118        let chunks = futures::stream::unfold(write_rx, |mut write_rx| async move {
119            let batch = write_rx.recv().await?;
120            Some((batch, write_rx))
121        })
122        .flat_map(|batch| futures::stream::iter(batch.rows))
123        .ready_chunks(write_chunk_size);
124
125        let result = chunks
126            .try_for_each_spawned(write_concurrency, move |rows| {
127                let context = context.clone();
128                async move { context.write_chunk_retrying(Chunk::new(rows)).await }
129            })
130            .await;
131
132        if result.is_err() {
133            warn!(
134                pipeline,
135                "Bitmap row writer stopping after downstream channel closed"
136            );
137        }
138
139        info!(pipeline, "Bitmap row writer exiting");
140    }
141}
142
143impl WriteContext {
144    async fn write_chunk_retrying(self, mut chunk: Chunk) -> Result<(), ()> {
145        let mut backoff = row_write_backoff();
146
147        loop {
148            let result = self.attempt_write_chunk(chunk).await;
149
150            let rows_written: u64 = result.flushed.iter().map(|r| r.count).sum();
151            if rows_written != 0 {
152                self.rows_written.fetch_add(rows_written, Ordering::Relaxed);
153            }
154
155            if !result.flushed.is_empty()
156                && self
157                    .generation_tx
158                    .send(generation::Event::RowsFlushed {
159                        counts: result.flushed,
160                    })
161                    .await
162                    .is_err()
163            {
164                warn!(
165                    self.pipeline,
166                    "Generation task closed while reporting durable bitmap rows"
167                );
168                return Err(());
169            }
170
171            if result.failed.is_empty() {
172                return Ok(());
173            }
174
175            let failed = result.failed.len();
176            self.metrics.retry_rows.inc_by(failed as u64);
177            chunk = Chunk {
178                rows: result.failed,
179            };
180            let delay = backoff
181                .next_backoff()
182                .unwrap_or(MAX_ROW_WRITE_RETRY_BACKOFF);
183            warn!(
184                failed,
185                ?delay,
186                "Bitmap row write failed; retrying after backoff"
187            );
188            tokio::time::sleep(delay).await;
189        }
190    }
191
192    async fn attempt_write_chunk(&self, chunk: Chunk) -> Attempt {
193        self.rate_limiter.acquire(chunk.rows.len()).await;
194        let entries = make_entries(&chunk.rows, self.column);
195
196        let write_start = Instant::now();
197        let mut client = self.client.clone();
198        let result = client.write_entries(self.table, entries).await;
199        self.metrics
200            .write_chunk_latency
201            .observe(write_start.elapsed().as_secs_f64());
202
203        split_write_result(result, chunk)
204    }
205}

213fn row_write_backoff() -> ExponentialBackoff {
214    ExponentialBackoff {
215        initial_interval: COMMIT_RETRY_BACKOFF,
216        current_interval: COMMIT_RETRY_BACKOFF,
217        max_interval: MAX_ROW_WRITE_RETRY_BACKOFF,
218        max_elapsed_time: None,
219        ..Default::default()
220    }
221}
222
223fn split_write_result(result: anyhow::Result<()>, chunk: Chunk) -> Attempt {
224    match result {
225        Ok(()) => Attempt {
226            flushed: rows_flushed_by_generation(chunk.rows),
227            failed: Vec::new(),
228        },
229        Err(e) => {
230            if let Some(partial) = e.downcast_ref::<PartialWriteError>() {
231                let failed_keys: FxHashSet<&Bytes> =
232                    partial.failed_keys.iter().map(|f| &f.key).collect();
233                let Chunk { rows } = chunk;
234                let mut flushed_by_checkpoint = BTreeMap::<u64, u64>::new();
235                let mut failed = Vec::new();
236                for row in rows {
237                    if failed_keys.contains(&row.row_key) {
238                        failed.push(row);
239                    } else {
240                        *flushed_by_checkpoint.entry(row.generation_cp).or_default() += 1;
241                    }
242                }
243                Attempt {
244                    flushed: flushed_by_checkpoint
245                        .into_iter()
246                        .map(|(checkpoint, count)| generation::RowsFlushed { checkpoint, count })
247                        .collect(),
248                    failed,
249                }
250            } else {
251                Attempt {
252                    flushed: Vec::new(),
253                    failed: chunk.rows,
254                }
255            }
256        }
257    }
258}
259
260fn rows_flushed_by_generation(rows: Vec<Row>) -> Vec<generation::RowsFlushed> {
261    let mut flushed_by_checkpoint = BTreeMap::<u64, u64>::new();
262    for row in rows {
263        *flushed_by_checkpoint.entry(row.generation_cp).or_default() += 1;
264    }
265    flushed_by_checkpoint
266        .into_iter()
267        .map(|(checkpoint, count)| generation::RowsFlushed { checkpoint, count })
268        .collect()
269}
270
271fn make_entries(rows: &[Row], column: &'static str) -> Vec<Entry> {
272    rows.iter()
273        .map(|r| {
274            tables::make_entry(
275                r.row_key.clone(),
276                [(column, r.serialized.clone())],
277                Some(r.max_ts_ms),
278            )
279        })
280        .collect()
281}

293    fn row(key: &'static [u8], generation_cp: u64) -> Row {
294        Row {
295            row_key: Bytes::from_static(key),
296            serialized: Bytes::from_static(b"bitmap"),
297            max_ts_ms: generation_cp * 1_000,
298            generation_cp,
299        }
300    }
301
302    #[test]
303    fn rows_flushed_are_grouped_by_generation() {
304        let flushed = rows_flushed_by_generation(vec![row(b"a", 2), row(b"b", 1), row(b"c", 2)]);
305
306        let counts: Vec<_> = flushed
307            .into_iter()
308            .map(|r| (r.checkpoint, r.count))
309            .collect();
310        assert_eq!(counts, vec![(1, 1), (2, 2)]);
311    }
312
313    #[test]
314    fn partial_write_retries_only_failed_rows_and_counts_successes() {
315        let chunk = Chunk::new(vec![row(b"a", 1), row(b"b", 1), row(b"c", 2)]);
316        let result: anyhow::Result<()> = Err(PartialWriteError {
317            failed_keys: vec![MutationError {
318                key: Bytes::from_static(b"b"),
319                code: 8,
320                message: "injected".to_string(),
321            }],
322        }
323        .into());
324
325        let attempt = split_write_result(result, chunk);
326
327        assert_eq!(attempt.failed.len(), 1);
328        assert_eq!(attempt.failed[0].row_key, Bytes::from_static(b"b"));
329        let flushed: Vec<_> = attempt
330            .flushed
331            .into_iter()
332            .map(|r| (r.checkpoint, r.count))
333            .collect();
334        assert_eq!(flushed, vec![(1, 1), (2, 1)]);
335    }
336
337    #[test]
338    fn non_partial_write_error_retries_entire_chunk() {
339        let chunk = Chunk::new(vec![row(b"a", 1), row(b"b", 2)]);
340
341        let attempt = split_write_result(Err(anyhow!("rpc unavailable")), chunk);
342
343        assert!(attempt.flushed.is_empty());
344        assert_eq!(attempt.failed.len(), 2);
345    }

347    #[test]
348    fn row_timestamp_becomes_bigtable_cell_version() {
349        let rows = vec![Row {
350            row_key: Bytes::from_static(b"a"),
351            serialized: Bytes::from_static(b"bitmap"),
352            max_ts_ms: 123,
353            generation_cp: 1,
354        }];
355
356        let entries = make_entries(&rows, "bitmap");
357
358        let cell = match &entries[0].mutations[0].mutation {
359            Some(mutation::Mutation::SetCell(cell)) => cell,
360            other => panic!("expected SetCell mutation, got {other:?}"),
361        };
362        assert_eq!(cell.timestamp_micros, 123_000);
363        assert_eq!(cell.value, Bytes::from_static(b"bitmap"));
364    }
