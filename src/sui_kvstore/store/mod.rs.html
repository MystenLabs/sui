4//! Framework store facade backed by BigTable.
5//!
6//! Implements the `Store`, `Connection`, `ConcurrentStore`/`ConcurrentConnection`, and
7//! `SequentialStore`/`SequentialConnection` traits over the lower-level
8//! [`crate::bigtable::client`] RPC layer. Per-pipeline watermarks are stored
9//! in the `watermark_alt` table as:
10//! - `w` (BCS v0 `WatermarkV0`) — kept in sync for backward compatibility.
11//! - `v` (schema version, currently `1`) — marks the row as using the new schema.
12//! - `ehi` / `chi` / `th` / `tmhi` / `rl` / `ph` / `ptm` — one u64 BE cell per
13//!   [`WatermarkV1`] field. `chi` (checkpoint) is absent when the committer has not observed
14//!   a checkpoint yet.
15//! - `b` — bitmap-only bucket-start cell. See `tables::watermarks::col::BUCKET_START_CP`.
16//!
17//! All three watermark setters (committer, reader, pruner) enforce monotonicity via a
18//! BigTable CheckAndMutate CAS on the guarded column. Writes also happen to be idempotent
19//! on same-value retry (CAS fails silently).
20//!
21//! ## Sequential transactions
22//!
23//! BigTable has no multi-row transaction, so [`SequentialStore::transaction`] runs the
24//! closure inline and defers the committer-watermark write until the closure returns
25//! successfully. Inside a transaction, `set_committer_watermark` buffers the value on the
26//! connection; the transaction impl flushes it to BigTable after the closure succeeds.
27//!
28//! Bitmap pipelines suppress that deferred synchronous write and let the store-owned
29//! committer promote watermarks after row writes are durable. Its OR-based bitmap writes
30//! are idempotent under retry because each write contains cumulative row state. On restart,
31//! bitmap pipelines resume at the active bucket's replay floor (via the persisted `b`
32//! column surfaced through `init_watermark`), and the committer ignores replayed rows for
33//! buckets already sealed by the persisted `tx_hi`.

77/// Bitmap-pipeline initial watermark — the fields decoded from the watermark
78/// row during `init_watermark` that the framework's `InitWatermark` doesn't
79/// surface. `watermark` is the full persisted committer watermark (un-clamped:
80/// it drives the generation task's bucket identity, which must match
81/// `bucket_start_cp`). Processors decide which watermark dimension to gate on
82/// via [`BitmapIndexProcessor::is_sealed`].
83#[derive(Clone, Copy, Debug, Default)]
84pub struct BitmapInitialWatermark {
85    pub watermark: CommitterWatermark,
86    pub bucket_start_cp: u64,
87}

89/// A Store implementation backed by BigTable.
90#[derive(Clone)]
91pub struct BigTableStore {
92    client: BigTableClient,
93    /// Per-pipeline `init_watermark` results. Populated exactly once per
94    /// pipeline — on the first `init_watermark` call — and read-only
95    /// thereafter. Shared across every `Clone` / `connect()` of this store
96    /// so the framework's `init_watermark` call during pipeline registration
97    /// returns the same value as any earlier bootstrap call.
98    ///
99    /// This is an ordering contract, not a cache: callers of
100    /// [`BitmapInitialWatermarks::get`] must ensure `init_watermark` has
101    /// already run for the pipeline. A missing entry is a programmer error
102    /// surfaced as a task error rather than silently triggering a fresh BigTable read.
103    init_results: Arc<Mutex<HashMap<String, PipelineInitResult>>>,
104    /// Bitmap committer handles registered by pipeline name. Connections
105    /// clone a handle out of this map before awaiting channel sends.
106    bitmap_committers: Arc<Mutex<HashMap<String, BitmapCommitterHandle>>>,
107}

109/// A connection to BigTable for watermark operations and data writes.
110///
111/// While a [`SequentialStore::transaction`] is in flight, `set_committer_watermark` buffers
112/// its write in `pending_watermark` instead of hitting BigTable; the transaction impl
113/// flushes it after the transaction closure returns successfully.
114pub struct BigTableConnection<'a> {
115    client: BigTableClient,
116    pending_watermark: Option<(String, CommitterWatermark)>,
117    /// `true` while running under `SequentialStore::transaction`.
118    in_sequential_transaction: bool,
119    /// When `true`, the deferred watermark write inside
120    /// [`SequentialStore::transaction`] is skipped even if `pending_watermark`
121    /// is staged. Bitmap commits set this because their store-owned committer
122    /// promotes watermarks asynchronously after row writes are durable.
123    skip_watermark_write: bool,
124    /// Shared with the owning [`BigTableStore`]. `init_watermark` writes here on the
125    /// first call per `pipeline_task` and reads back on subsequent calls, so the
126    /// framework's implicit call during `sequential_pipeline` registration doesn't
127    /// duplicate an earlier bootstrap call's BigTable round-trip.
128    init_results: Arc<Mutex<HashMap<String, PipelineInitResult>>>,
129    /// Shared registry of store-owned bitmap committers.
130    bitmap_committers: Arc<Mutex<HashMap<String, BitmapCommitterHandle>>>,
131    _marker: std::marker::PhantomData<&'a ()>,
132}

186impl BigTableConnection<'_> {
187    /// Returns a mutable reference to the underlying BigTable client.
188    pub fn client(&mut self) -> &mut BigTableClient {
189        &mut self.client
190    }
191
192    /// Enqueue a bitmap batch into the store-owned committer and suppress the
193    /// framework's synchronous watermark write for this transaction.
194    pub(crate) async fn commit_bitmap_batch<P>(&mut self, batch: &BitmapBatch) -> Result<usize>
195    where
196        P: BitmapIndexProcessor + Send + Sync + 'static,
197    {
198        let Some((pipeline, watermark)) = self.pending_watermark.as_ref() else {
199            bail!("set_committer_watermark must be called before bitmap handler commit");
200        };
201        if pipeline != P::NAME {
202            bail!(
203                "bitmap handler `{}` saw staged watermark for `{pipeline}`",
204                P::NAME
205            );
206        }
207        let watermark = *watermark;
208
209        // Always suppress the framework's deferred-watermark write. Even
210        // empty-batch commits must flow through the async bitmap pipeline so
211        // their watermarks are promoted in order after any prior in-flight
212        // row writes become durable.
213        self.skip_watermark_write = true;
214
215        let committer = self
216            .bitmap_committers
217            .lock()
218            .unwrap()
219            .get(P::NAME)
220            .cloned()
221            .unwrap_or_else(|| panic!("bitmap committer for `{}` is not registered", P::NAME));
222
223        if committer
224            .commit(batch.clone_shards(), watermark)
225            .await
226            .is_err()
227        {
228            bail!("{}: bitmap merge loop exited; cannot continue", P::NAME);
229        }
230
231        // Bitmap row writes happen asynchronously after `commit()` returns, so
232        // this is a lagging drain of rows accepted by BigTable since the last
233        // framework commit for this pipeline.
234        Ok(committer.take_rows_written())
235    }
236
237    fn record_init_result(
238        &self,
239        pipeline_task: &str,
240        init: Option<InitWatermark>,
241        watermark: CommitterWatermark,
242        bucket_start_cp: u64,
243    ) {
244        self.init_results.lock().unwrap().insert(
245            pipeline_task.to_string(),
246            PipelineInitResult {
247                init,
248                bitmap: BitmapInitialWatermark {
249                    watermark,
250                    bucket_start_cp,
251                },
252            },
253        );
254    }
255
256    /// Read a watermark for read-side methods. Enforces the "hide if `checkpoint < reader_lo`
257    /// or `checkpoint == None`" rule and returns the unwrapped checkpoint for callers that
258    /// need it.
259    async fn get_watermark_for_read(
260        &mut self,
261        pipeline: &str,
262    ) -> Result<Option<(WatermarkV1, u64)>> {
263        let row = self.client.get_pipeline_watermark_rows(pipeline).await?;
264        let Some(watermark) = decode_v1(&row)? else {
265            return Ok(None);
266        };
267        let Some(checkpoint_hi_inclusive) = watermark
268            .checkpoint_hi_inclusive
269            .filter(|&cp| cp >= watermark.reader_lo)
270        else {
271            return Ok(None);
272        };
273        Ok(Some((watermark, checkpoint_hi_inclusive)))
274    }
275}

277impl BigTableStoreRuntimeBuilder {
278    pub(crate) fn with_bitmap_committer<P>(
279        mut self,
280        write_chunk_size: usize,
281        write_concurrency: usize,
282        rate_limiter: Arc<CompositeRateLimiter>,
283        registry: Option<&Registry>,
284    ) -> Self
285    where
286        P: BitmapIndexProcessor + Send + Sync + 'static,
287    {
288        let metrics = match registry {
289            Some(reg) => BitmapIndexMetrics::new(P::NAME, reg),
290            None => BitmapIndexMetrics::noop(),
291        };
292
293        let (handle, service) = BitmapCommitter {
294            pipeline: P::NAME,
295            table: P::TABLE,
296            column: P::COLUMN,
297            is_sealed: P::is_sealed,
298            initial_watermarks: self.store.bitmap_initial_watermarks(),
299            client: self.store.client.clone(),
300            rate_limiter,
301            write_chunk_size,
302            write_concurrency,
303            metrics,
304        }
305        .spawn();
306
307        assert!(
308            self.store
309                .bitmap_committers
310                .lock()
311                .unwrap()
312                .insert(P::NAME.to_string(), handle)
313                .is_none(),
314            "bitmap committer for pipeline `{}` registered more than once",
315            P::NAME,
316        );
317
318        self.service = self.service.merge(service);
319        self
320    }
321
322    pub(crate) fn into_service(self) -> Service {
323        self.service
324    }
325}

327impl BitmapInitialWatermarks {
328    /// Bitmap fields recorded during the first `init_watermark` call for
329    /// `pipeline_task`. Returns an error if `init_watermark` has not yet been
330    /// invoked for this pipeline — framework must drive `init_watermark` first.
331    pub(crate) fn get(&self, pipeline_task: &str) -> Result<BitmapInitialWatermark> {
332        self.init_results
333            .lock()
334            .unwrap()
335            .get(pipeline_task)
336            .map(|result| result.bitmap)
337            .ok_or_else(|| {
338                anyhow::anyhow!(
339                    "bitmap initial watermark requested for `{pipeline_task}` before \
340                     init_watermark; the indexer bootstrap must run init_watermark for \
341                     every bitmap pipeline before its first commit",
342                )
343            })
344    }
345}

369#[async_trait]
370impl SequentialStore for BigTableStore {
371    type SequentialConnection<'c> = BigTableConnection<'c>;
372
373    async fn transaction<'a, R, F>(&self, f: F) -> Result<R>
374    where
375        R: Send + 'a,
376        F: Send + 'a,
377        F: for<'r> FnOnce(&'r mut Self::Connection<'_>) -> ScopedBoxFuture<'a, 'r, Result<R>>,
378    {
379        let mut conn = self.connect().await?;
380        conn.in_sequential_transaction = true;
381        let result = f(&mut conn).await?;
382        // Closure returned `Ok` — now persist the staged watermark unless a
383        // store-owned side effect suppressed it. The flush goes through the
384        // CAS-guarded `set_committer_watermark` path (via the client helper),
385        // so concurrent writers cannot cause regressions.
386        if let Some((pipeline, watermark)) = conn.pending_watermark.take()
387            && !conn.skip_watermark_write
388        {
389            conn.client
390                .set_committer_watermark_cells(&pipeline, &watermark, None)
391                .await?;
392        }
393        Ok(result)
394    }
395}

397#[async_trait]
398impl Connection for BigTableConnection<'_> {
399    async fn init_watermark(
400        &mut self,
401        pipeline_task: &str,
402        checkpoint_hi_inclusive: Option<u64>,
403    ) -> Result<Option<InitWatermark>> {
404        // Re-entry for the same `pipeline_task` returns the previously computed
405        // answer without a BigTable round-trip. Also lets callers with a legitimate
406        // need for bitmap fields pre-seed via an explicit bootstrap call.
407        if let Some(existing) = self.init_results.lock().unwrap().get(pipeline_task) {
408            return Ok(existing.init);
409        }
410
411        // This initial read is to determine if we need to migrate from v0 to v1 and to pick
412        // up the bitmap `b` cell (if any) so bitmap pipelines resume at their bucket start.
413        let row = self
414            .client
415            .get_pipeline_watermark_rows(pipeline_task)
416            .await?;
417        let existing_v1 = decode_v1(&row)?;
418        let existing_v0 = decode_v0(&row)?;
419
420        // Case 1: row already in the v1 format → return its values, no write.
421        if let Some(wm) = existing_v1 {
422            let bucket_start_cp = wm.bucket_start_cp;
423            let init = InitWatermark {
424                checkpoint_hi_inclusive: clamp_to_bucket_start(
425                    wm.checkpoint_hi_inclusive,
426                    bucket_start_cp,
427                ),
428                reader_lo: Some(wm.reader_lo),
429            };
430            let watermark = CommitterWatermark {
431                epoch_hi_inclusive: wm.epoch_hi_inclusive,
432                checkpoint_hi_inclusive: wm.checkpoint_hi_inclusive.unwrap_or(0),
433                tx_hi: wm.tx_hi,
434                timestamp_ms_hi_inclusive: wm.timestamp_ms_hi_inclusive,
435            };
436            self.record_init_result(
437                pipeline_task,
438                Some(init),
439                watermark,
440                bucket_start_cp.unwrap_or(0),
441            );
442            return Ok(Some(init));
443        }
444
445        let initial = if let Some(v0) = existing_v0 {
446            // Case 2: v0-only row → bootstrap a v1 watermark from the v0 committer fields.
447            let reader_lo = v0.checkpoint_hi_inclusive + 1;
448            WatermarkV1 {
449                epoch_hi_inclusive: v0.epoch_hi_inclusive,
450                checkpoint_hi_inclusive: Some(v0.checkpoint_hi_inclusive),
451                tx_hi: v0.tx_hi,
452                timestamp_ms_hi_inclusive: v0.timestamp_ms_hi_inclusive,
453                reader_lo,
454                pruner_hi: reader_lo,
455                pruner_timestamp_ms: 0,
456                bucket_start_cp: None,
457            }
458        } else {
459            // Case 3: nothing exists → write a fresh row from the framework's input.
460            let reader_lo = checkpoint_hi_inclusive.map_or(0, |cp| cp + 1);
461            WatermarkV1 {
462                epoch_hi_inclusive: 0,
463                checkpoint_hi_inclusive,
464                tx_hi: 0,
465                timestamp_ms_hi_inclusive: 0,
466                reader_lo,
467                pruner_hi: reader_lo,
468                pruner_timestamp_ms: 0,
469                bucket_start_cp: None,
470            }
471        };
472
473        let write_happened = self
474            .client
475            .create_pipeline_watermark_if_absent(pipeline_task, &initial)
476            .await?;
477
478        let final_wm = if write_happened {
479            initial
480        } else {
481            let row = self
482                .client
483                .get_pipeline_watermark_rows(pipeline_task)
484                .await?;
485            decode_v1(&row)?.ok_or_else(|| {
486                anyhow::anyhow!(
487                    "watermark for pipeline {} missing after creation",
488                    pipeline_task
489                )
490            })?
491        };
492
493        let init = InitWatermark {
494            checkpoint_hi_inclusive: clamp_to_bucket_start(
495                final_wm.checkpoint_hi_inclusive,
496                final_wm.bucket_start_cp,
497            ),
498            reader_lo: Some(final_wm.reader_lo),
499        };
500        let watermark = CommitterWatermark {
501            epoch_hi_inclusive: final_wm.epoch_hi_inclusive,
502            checkpoint_hi_inclusive: final_wm.checkpoint_hi_inclusive.unwrap_or(0),
503            tx_hi: final_wm.tx_hi,
504            timestamp_ms_hi_inclusive: final_wm.timestamp_ms_hi_inclusive,
505        };
506        self.record_init_result(
507            pipeline_task,
508            Some(init),
509            watermark,
510            final_wm.bucket_start_cp.unwrap_or(0),
511        );
512        Ok(Some(init))
513    }
514
515    async fn accepts_chain_id(&mut self, pipeline_task: &str, chain_id: [u8; 32]) -> Result<bool> {
516        self.client.accepts_chain_id(pipeline_task, chain_id).await
517    }
518
519    async fn committer_watermark(
520        &mut self,
521        pipeline_task: &str,
522    ) -> Result<Option<CommitterWatermark>> {
523        Ok(self.get_watermark_for_read(pipeline_task).await?.map(
524            |(wm, checkpoint_hi_inclusive)| CommitterWatermark {
525                epoch_hi_inclusive: wm.epoch_hi_inclusive,
526                checkpoint_hi_inclusive,
527                tx_hi: wm.tx_hi,
528                timestamp_ms_hi_inclusive: wm.timestamp_ms_hi_inclusive,
529            },
530        ))
531    }
532
533    async fn set_committer_watermark(
534        &mut self,
535        pipeline_task: &str,
536        watermark: CommitterWatermark,
537    ) -> Result<bool> {
538        if self.in_sequential_transaction {
539            if let Some((prev, _)) = self.pending_watermark.as_ref()
540                && prev != pipeline_task
541            {
542                bail!(
543                    "set_committer_watermark called for '{pipeline_task}' \
544                    inside a transaction that already staged '{prev}'"
545                );
546            }
547            self.pending_watermark = Some((pipeline_task.to_string(), watermark));
548            return Ok(true);
549        }
550
551        self.client
552            .set_committer_watermark_cells(pipeline_task, &watermark, None)
553            .await
554    }
555}

557#[async_trait]
558impl ConcurrentConnection for BigTableConnection<'_> {
559    async fn reader_watermark(&mut self, pipeline: &str) -> Result<Option<ReaderWatermark>> {
560        Ok(self
561            .get_watermark_for_read(pipeline)
562            .await?
563            .map(|(wm, checkpoint_hi_inclusive)| ReaderWatermark {
564                checkpoint_hi_inclusive,
565                reader_lo: wm.reader_lo,
566            }))
567    }
568
569    async fn pruner_watermark(
570        &mut self,
571        pipeline: &'static str,
572        delay: Duration,
573    ) -> Result<Option<PrunerWatermark>> {
574        let Some((watermark, _)) = self.get_watermark_for_read(pipeline).await? else {
575            return Ok(None);
576        };
577        // Compute max(0, (pruner_timestamp + delay) - now). Use u128 to avoid overflow when
578        // summing the two operands, and saturating_sub so we never underflow when the wait
579        // period has already elapsed. saturating_sub is safe because callers treat anything
580        // < 1 the same.
581        let pruner_ready_ms = (watermark.pruner_timestamp_ms as u128) + delay.as_millis();
582        let now_ms = SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis();
583        let wait_for_ms = i64::try_from(pruner_ready_ms.saturating_sub(now_ms))?;
584        Ok(Some(PrunerWatermark {
585            wait_for_ms,
586            reader_lo: watermark.reader_lo,
587            pruner_hi: watermark.pruner_hi,
588        }))
589    }
590
591    async fn set_reader_watermark(
592        &mut self,
593        pipeline: &'static str,
594        reader_lo: u64,
595    ) -> Result<bool> {
596        let now_ms = SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis() as u64;
597        let cells = vec![
598            (col::READER_LO, u64_be(reader_lo)),
599            (col::PRUNER_TIMESTAMP_MS, u64_be(now_ms)),
600        ];
601        self.client
602            .cas_write_pipeline_watermark_cells(pipeline, col::READER_LO, reader_lo, cells)
603            .await
604    }
605
606    async fn set_pruner_watermark(
607        &mut self,
608        pipeline: &'static str,
609        pruner_hi: u64,
610    ) -> Result<bool> {
611        let cells = vec![(col::PRUNER_HI, u64_be(pruner_hi))];
612        self.client
613            .cas_write_pipeline_watermark_cells(pipeline, col::PRUNER_HI, pruner_hi, cells)
614            .await
615    }
616}

625/// If a bitmap `bucket_start_cp` is persisted, clamp the returned
626/// `checkpoint_hi_inclusive` back so the framework resumes at the start of
627/// the currently-active bucket. Non-bitmap pipelines never write `b`, so
628/// `bucket_start_cp` is `None` and the raw checkpoint passes through.
629fn clamp_to_bucket_start(
630    checkpoint_hi_inclusive: Option<u64>,
631    bucket_start_cp: Option<u64>,
632) -> Option<u64> {
633    match bucket_start_cp {
634        Some(0) => None,
635        Some(start) if start > 0 => Some(start - 1),
636        _ => checkpoint_hi_inclusive,
637    }
638}

655        let emulator = tokio::task::spawn_blocking(BigTableEmulator::start)
656            .await
657            .unwrap()
658            .unwrap();

660        let client = BigTableClient::new_local(emulator.host().to_string(), INSTANCE_ID.into())
661            .await
662            .unwrap();
663        (emulator, BigTableStore::new(client))
664    }
