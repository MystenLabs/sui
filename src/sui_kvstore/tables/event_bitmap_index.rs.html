4//! Roaring-bitmap inverted index keyed by packed event sequence number.
5//!
6//! Transactions have a global `tx_seq`, but events are only ordered within their
7//! containing transaction. We pack `(tx_seq, event_idx)` into one globally
8//! ordered `u64`:
9//!
10//! ```text
11//! event_seq = (tx_seq << EVENT_BITS) | event_idx
12//! ```
13//!
14//! `EVENT_BITS = 16` reserves 65,536 event slots per transaction. The protocol
15//! config's `max_num_event_emit` is currently 1,024, so this gives 64x headroom
16//! for future protocol changes while leaving 48 bits for `tx_seq` (~281T
17//! historical transactions, over 55k times the chain's current transaction
18//! count at the time this index was introduced). Even at a sustained 1M TPS,
19//! 48 tx bits provide roughly 8.9 years of transaction history before a schema
20//! bump would be needed.
21//!
22//! The packed namespace is sparse: most transactions use only a tiny prefix of
23//! their reserved event slots. Each row spans 268,435,456 packed event positions
24//! (`BUCKET_SIZE = 2^28`), which is exactly 4,096 transactions worth of event
25//! namespace at `EVENT_BITS = 16`, while leaving bucket-relative bit positions
26//! (max `2^28 - 1`) well within RoaringBitmap's `u32` limit.
27
28pub const NAME: &str = "event_bitmap_index";
29
30pub const SCHEMA_VERSION: u32 = 1;
31pub const BUCKET_ID_WIDTH: usize = 12;
32/// Number of packed `event_seq`s per bitmap bucket.
33pub const BUCKET_SIZE: u64 = 268_435_456;
34
35/// Number of low bits of `event_seq` reserved for the per-tx event index.
36pub const EVENT_BITS: u32 = 16;
37
38/// Maximum number of events per tx representable in a packed event_seq.
39pub const MAX_EVENTS_PER_TX: u32 = 1 << EVENT_BITS;
40
41/// Maximum tx_seq representable in a packed event_seq (48 bits at
42/// EVENT_BITS = 16). Anything larger would shift bits off the top.
43pub const MAX_TX_SEQ: u64 = u64::MAX >> EVENT_BITS;
44
45pub mod col {
46    pub const BITMAP: &str = "b";
47}
48
49/// Pack `(tx_seq, event_idx)` into a single globally-ordered u64 event_seq.
50///
51/// `event_idx` must be `< MAX_EVENTS_PER_TX` and `tx_seq` must be
52/// `<= MAX_TX_SEQ`. Caller is responsible for respecting the protocol-level
53/// limit on events-per-tx.
54#[inline]
55pub fn encode_event_seq(tx_seq: u64, event_idx: u32) -> u64 {
56    debug_assert!(event_idx < MAX_EVENTS_PER_TX);
57    debug_assert!(
58        tx_seq <= MAX_TX_SEQ,
59        "tx_seq {} exceeds {} bits and would lose data when shifted by EVENT_BITS={}",
60        tx_seq,
61        64 - EVENT_BITS,
62        EVENT_BITS,
63    );
64    (tx_seq << EVENT_BITS) | (event_idx as u64)
65}
66
67/// Inverse of [`encode_event_seq`]: unpack a packed event_seq back into
68/// `(tx_seq, event_idx)`.
69#[inline]
70pub fn decode_event_seq(event_seq: u64) -> (u64, u32) {
71    let tx_seq = event_seq >> EVENT_BITS;
72    let event_idx = (event_seq & (MAX_EVENTS_PER_TX as u64 - 1)) as u32;
73    (tx_seq, event_idx)
74}
75
76/// Lowest possible event_seq for a given tx_seq (idx 0).
77#[inline]
78pub fn event_seq_lo(tx_seq: u64) -> u64 {
79    tx_seq << EVENT_BITS
80}
81
82/// Encode a full row key for the event-keyed bitmap index.
83///
84/// Format: `v{version}#{dimension_key}#{bucket_id:012}`
85///
86/// Uses 12-digit zero-padded bucket ids because the event_seq namespace is
87/// ~2^16 larger than the tx_seq namespace, overflowing the 10-digit format
88/// used by the tx-keyed index.
89pub fn encode_row_key(version: u32, dimension_key: &[u8], bucket_id: u64) -> Vec<u8> {
90    let mut key = Vec::new();
91    encode_row_key_into(&mut key, version, dimension_key, bucket_id);
92    key
93}
94
95pub fn encode_row_key_into(out: &mut Vec<u8>, version: u32, dimension_key: &[u8], bucket_id: u64) {
96    super::encode_bitmap_row_key_into(out, version, BUCKET_ID_WIDTH, dimension_key, bucket_id);
97}

99#[cfg(test)]
100mod tests {
101    use super::*;
102
103    #[test]
104    fn test_encode_decode_event_seq_roundtrip() {
105        for (tx_seq, event_idx) in [
106            (0, 0),
107            (0, 1023),
108            (1, 0),
109            (1_000_000, 42),
110            (u64::MAX >> EVENT_BITS, MAX_EVENTS_PER_TX - 1),
111        ] {
112            let packed = encode_event_seq(tx_seq, event_idx);
113            assert_eq!(decode_event_seq(packed), (tx_seq, event_idx));
114        }
115    }
116
117    #[test]
118    fn test_event_seq_ordering() {
119        // Same tx, increasing idx → increasing event_seq
120        let a = encode_event_seq(100, 0);
121        let b = encode_event_seq(100, 5);
122        assert!(a < b);
123        // Different tx, same idx → tx_seq dominates
124        let c = encode_event_seq(101, 0);
125        assert!(b < c);
126    }
127
128    #[test]
129    fn test_encode_row_key_format() {
130        let dim_key = vec![0x01, 0xAA];
131        let key = encode_row_key(1, &dim_key, 42);
132        let key_str = String::from_utf8_lossy(&key);
133        assert!(key_str.starts_with("v1#"));
134        assert!(key_str.ends_with("#000000000042"));
135    }
136
137    #[test]
138    fn test_bucket_size_fits_u32() {
139        assert!(BUCKET_SIZE <= u32::MAX as u64);
140    }
141}
