4//! Packages-by-checkpoint table: checkpoint-ordered index of packages.
5//!
6//! Row key: `cp_sequence_number (8B) || original_id (32B) || package_version (8B)`
7//! Column: default (empty marker â€” metadata fetched from `packages` table)
8
9use anyhow::Result;
10use bytes::Bytes;
11
12use crate::tables::DEFAULT_COLUMN;
13
14pub const NAME: &str = "packages_by_checkpoint";
15
16pub fn encode_key(cp_sequence_number: u64, original_id: &[u8], package_version: u64) -> Vec<u8> {
17    let mut key = Vec::with_capacity(48);
18    key.extend_from_slice(&cp_sequence_number.to_be_bytes());
19    key.extend_from_slice(original_id);
20    key.extend_from_slice(&package_version.to_be_bytes());
21    key
22}
23
24pub fn encode() -> [(&'static str, Bytes); 1] {
25    [(DEFAULT_COLUMN, Bytes::new())]
26}
27
28/// Extract (cp_sequence_number, original_id, package_version) from a 48-byte key.
29pub fn decode_key(key: &[u8]) -> Result<(u64, Vec<u8>, u64)> {
30    anyhow::ensure!(key.len() == 48, "expected 48-byte key, got {}", key.len());
31    let cp = u64::from_be_bytes(key[..8].try_into()?);
32    let original_id = key[8..40].to_vec();
33    let version = u64::from_be_bytes(key[40..48].try_into()?);
34    Ok((cp, original_id, version))
35}
