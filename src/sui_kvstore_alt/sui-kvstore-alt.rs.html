4//! New kvstore binary using sui-indexer-alt-framework.
5//!
6//! This binary can run alongside the legacy kvstore binary during migration.
7//! Both write to the same BigTable tables and share the same watermark.
8
9use std::time::Duration;
10
11use anyhow::Result;
12use clap::Parser;
13use sui_indexer_alt_framework::ingestion::{ClientArgs, IngestionConfig};
14use sui_indexer_alt_framework::pipeline::CommitterConfig;
15use sui_indexer_alt_framework::pipeline::concurrent::ConcurrentConfig;
16use sui_indexer_alt_framework::{Indexer, IndexerArgs};
17use sui_indexer_alt_metrics::MetricsArgs;
18use sui_kvstore::{BigTableClient, BigTableStore, KvStorePipeline};
19use telemetry_subscribers::TelemetryConfig;
20use tracing::info;
21
22#[derive(Parser)]
23#[command(name = "sui-kvstore-alt")]
24#[command(about = "KVStore indexer using sui-indexer-alt-framework")]
25struct Args {
26    /// BigTable instance ID (e.g., "projects/myproject/instances/myinstance")
27    instance_id: String,
28
29    /// Number of concurrent checkpoint writes
30    #[arg(long, default_value = "10")]
31    write_concurrency: usize,
32
33    /// Interval between watermark updates
34    #[arg(long, default_value = "1m", value_parser = humantime::parse_duration)]
35    watermark_interval: Duration,
36
37    /// BigTable app profile ID
38    #[arg(long)]
39    app_profile_id: Option<String>,
40
41    #[command(flatten)]
42    metrics_args: MetricsArgs,
43
44    #[command(flatten)]
45    client_args: ClientArgs,
46
47    #[command(flatten)]
48    indexer_args: IndexerArgs,
49}
50
51#[tokio::main]
52async fn main() -> Result<()> {
53    let _guard = TelemetryConfig::new().with_env().init();
54
55    let args = Args::parse();
56
57    info!("Starting sui-kvstore-alt indexer");
58    info!(instance_id = %args.instance_id);
59
60    // Create BigTable client
61    let client = BigTableClient::new_remote(
62        args.instance_id,
63        false, // write mode
64        None,
65        "sui-kvstore-alt".to_string(),
66        None,
67        args.app_profile_id,
68    )
69    .await?;
70
71    // Create store
72    let store = BigTableStore::new(client);
73
74    // Set up metrics
75    let registry = prometheus::Registry::new_custom(Some("kvstore_alt".into()), None)?;
76    let metrics_service =
77        sui_indexer_alt_metrics::MetricsService::new(args.metrics_args, registry.clone());
78
79    // Create indexer
80    let mut indexer = Indexer::new(
81        store,
82        args.indexer_args,
83        args.client_args,
84        IngestionConfig::default(),
85        None,
86        &registry,
87    )
88    .await?;
89
90    // Register the kvstore pipeline
91    let config = ConcurrentConfig {
92        committer: CommitterConfig {
93            write_concurrency: args.write_concurrency,
94            watermark_interval_ms: args.watermark_interval.as_millis() as u64,
95            ..Default::default()
96        },
97        ..Default::default()
98    };
99    indexer.concurrent_pipeline(KvStorePipeline, config).await?;
100
101    info!("Indexer created");
102
103    // Run the indexer
104    let metrics_handle = metrics_service.run().await?;
105    let service = indexer.run().await?;
106    service.attach(metrics_handle).main().await?;
107
108    Ok(())
109}
