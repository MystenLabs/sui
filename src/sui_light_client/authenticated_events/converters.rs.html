10pub(super) fn proto_object_ref_to_sui_object_ref(
11    proto: &sui_rpc::proto::sui::rpc::v2::ObjectReference,
12) -> Result<sui_types::base_types::ObjectRef, ClientError> {
13    let object_id_str = proto
14        .object_id
15        .as_ref()
16        .ok_or_else(|| ClientError::InternalError("Missing object_id".to_string()))?;
17
18    let object_id = ObjectID::from_str(object_id_str)
19        .map_err(|e| ClientError::InternalError(format!("Invalid object_id: {}", e)))?;
20
21    let version = proto
22        .version
23        .ok_or_else(|| ClientError::InternalError("Missing version".to_string()))?
24        .into();
25
26    let digest_str = proto
27        .digest
28        .as_ref()
29        .ok_or_else(|| ClientError::InternalError("Missing digest".to_string()))?;
30
31    let digest = sui_types::digests::ObjectDigest::from_str(digest_str)
32        .map_err(|e| ClientError::InternalError(format!("Invalid digest: {}", e)))?;
33
34    Ok((object_id, version, digest))
35}

37pub(super) fn proto_ocs_inclusion_proof_to_light_client_proof(
38    proto: &ProtoOcsInclusionProof,
39) -> Result<OCSInclusionProof, ClientError> {
40    let merkle_proof_bytes = proto
41        .merkle_proof
42        .as_ref()
43        .ok_or_else(|| ClientError::InternalError("Missing merkle_proof".to_string()))?;
44
45    let merkle_proof: fastcrypto::merkle::MerkleProof = bcs::from_bytes(merkle_proof_bytes)?;
46
47    let leaf_index = proto
48        .leaf_index
49        .ok_or_else(|| ClientError::InternalError("Missing leaf_index".to_string()))?
50        as usize;
51
52    let tree_root_bytes = proto
53        .tree_root
54        .as_ref()
55        .ok_or_else(|| ClientError::InternalError("Missing tree_root".to_string()))?;
56
57    if tree_root_bytes.len() != 32 {
58        return Err(ClientError::InternalError(format!(
59            "Invalid tree_root length: {}",
60            tree_root_bytes.len()
61        )));
62    }
63    let mut tree_root_arr = [0u8; 32];
64    tree_root_arr.copy_from_slice(tree_root_bytes);
65    let tree_root = sui_types::digests::Digest::new(tree_root_arr);
66
67    Ok(OCSInclusionProof {
68        merkle_proof,
69        leaf_index,
70        tree_root,
71    })
72}
