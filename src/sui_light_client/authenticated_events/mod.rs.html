30/// A cryptographically verified event.
31///
32/// Each `AuthenticatedEvent` can be verified using the EventStreamHead's MMR
33/// (Merkle Mountain Range) commitment, ensuring its authenticity and inclusion
34/// at the specified checkpoint.
35#[derive(Debug, Clone)]
36pub struct AuthenticatedEvent {
37    /// The underlying Sui event data.
38    pub event: Event,
39    /// The checkpoint sequence number where this event was included.
40    pub checkpoint: u64,
41    /// The accumulator version when this event was committed.
42    pub accumulator_version: u64,
43    /// The transaction index within the checkpoint.
44    pub transaction_idx: u32,
45    /// The event index within the transaction.
46    pub event_idx: u32,
47}
48
49impl TryFrom<sui_rpc_api::grpc::alpha::event_service_proto::AuthenticatedEvent>
50    for AuthenticatedEvent
51{
52    type Error = ClientError;
53
54    fn try_from(
55        event: sui_rpc_api::grpc::alpha::event_service_proto::AuthenticatedEvent,
56    ) -> Result<Self, Self::Error> {
57        let proto_event = event
58            .event
59            .ok_or_else(|| ClientError::InternalError("Missing event data".to_string()))?;
60
61        let contents = proto_event
62            .contents
63            .ok_or_else(|| ClientError::InternalError("Missing event contents".to_string()))?;
64
65        let event_bytes = contents
66            .value
67            .ok_or_else(|| ClientError::InternalError("Missing event value".to_string()))?;
68
69        let package_id = proto_event
70            .package_id
71            .as_ref()
72            .ok_or_else(|| ClientError::InternalError("Missing package_id".to_string()))?;
73        let package_id = sui_types::base_types::ObjectID::from_hex_literal(package_id)
74            .map_err(|e| ClientError::InternalError(format!("Invalid package_id: {}", e)))?;
75
76        let module = proto_event
77            .module
78            .as_ref()
79            .ok_or_else(|| ClientError::InternalError("Missing module".to_string()))?;
80        let module = Identifier::new(module.as_str())
81            .map_err(|e| ClientError::InternalError(format!("Invalid module: {}", e)))?;
82
83        let sender = proto_event
84            .sender
85            .as_ref()
86            .ok_or_else(|| ClientError::InternalError("Missing sender".to_string()))?;
87        let sender = sender
88            .parse()
89            .map_err(|e| ClientError::InternalError(format!("Invalid sender: {}", e)))?;
90
91        let event_type = proto_event
92            .event_type
93            .as_ref()
94            .ok_or_else(|| ClientError::InternalError("Missing event_type".to_string()))?;
95        let type_tag: move_core_types::language_storage::StructTag = event_type
96            .parse()
97            .map_err(|e| ClientError::InternalError(format!("Invalid event_type: {}", e)))?;
98
99        let event_data = Event {
100            package_id,
101            transaction_module: module,
102            sender,
103            type_: type_tag,
104            contents: event_bytes.to_vec(),
105        };
106
107        let checkpoint = event
108            .checkpoint
109            .ok_or_else(|| ClientError::InternalError("Missing checkpoint".to_string()))?;
110        let transaction_idx = event
111            .transaction_idx
112            .ok_or_else(|| ClientError::InternalError("Missing transaction_idx".to_string()))?;
113        let event_idx = event
114            .event_idx
115            .ok_or_else(|| ClientError::InternalError("Missing event_idx".to_string()))?;
116        let accumulator_version = event
117            .accumulator_version
118            .ok_or_else(|| ClientError::InternalError("Missing accumulator_version".to_string()))?;
119
120        Ok(AuthenticatedEvent {
121            event: event_data,
122            checkpoint,
123            accumulator_version,
124            transaction_idx,
125            event_idx,
126        })
127    }
128}

130/// Configuration for the authenticated events client.
131///
132/// Controls streaming behavior (page size, polling, pagination) and RPC communication (timeouts).
133#[derive(Debug, Clone)]
134pub struct ClientConfig {
135    pub page_size: u32,
136    pub poll_interval: Duration,
137    pub max_pagination_iterations: usize,
138    pub rpc_timeout: Duration,
139}
140
141impl ClientConfig {
142    pub fn new(
143        page_size: u32,
144        poll_interval: Duration,
145        max_pagination_iterations: usize,
146        rpc_timeout: Duration,
147    ) -> Result<Self, String> {
148        if page_size == 0 {
149            return Err("page_size must be greater than 0".to_string());
150        }
151        if page_size > 1000 {
152            return Err("page_size must not exceed 1000 (server limit)".to_string());
153        }
154        if poll_interval.is_zero() {
155            return Err("poll_interval must be greater than 0".to_string());
156        }
157        if max_pagination_iterations == 0 {
158            return Err("max_pagination_iterations must be greater than 0".to_string());
159        }
160        if rpc_timeout.is_zero() {
161            return Err("rpc_timeout must be greater than 0".to_string());
162        }
163
164        Ok(Self {
165            page_size,
166            poll_interval,
167            max_pagination_iterations,
168            rpc_timeout,
169        })
170    }
171}
172
173impl Default for ClientConfig {
174    fn default() -> Self {
175        Self {
176            page_size: 1000,
177            poll_interval: Duration::from_secs(1),
178            max_pagination_iterations: 100,
179            rpc_timeout: Duration::from_secs(30),
180        }
181    }

184#[derive(Debug, Error)]
185pub enum ClientError {
186    #[error("Verification failed: {0}")]
187    VerificationError(String),
188
189    #[error("Internal error: {0}")]
190    InternalError(String),
191
192    #[error("RPC error: {0}")]
193    RpcError(#[from] tonic::Status),
194
195    #[error("Transport error: {0}")]
196    TransportError(#[from] tonic::transport::Error),
197}
198
199impl From<bcs::Error> for ClientError {
200    fn from(e: bcs::Error) -> Self {
201        ClientError::InternalError(format!("BCS deserialization failed: {}", e))
202    }
203}
204
205impl From<ProofError> for ClientError {
206    fn from(e: ProofError) -> Self {
207        ClientError::VerificationError(e.to_string())
208    }
209}
210
211impl ClientError {
212    pub(crate) fn is_terminal(&self) -> bool {
213        match self {
214            ClientError::RpcError(status) => !Self::is_retriable_grpc_code(status.code()),
215            ClientError::TransportError(_) => false,
216            _ => true,
217        }
218    }
219
220    fn is_retriable_grpc_code(code: tonic::Code) -> bool {
221        matches!(
222            code,
223            tonic::Code::Unavailable
224                | tonic::Code::DeadlineExceeded
225                | tonic::Code::ResourceExhausted
226                | tonic::Code::Aborted
227        )
228    }
229}

244    pub async fn new_with_config(
245        rpc_url: &str,
246        genesis_committee: Committee,
247        config: ClientConfig,
248    ) -> Result<Self, ClientError> {
249        let endpoint = Channel::from_shared(rpc_url.to_string())
250            .map_err(|e| ClientError::InternalError(format!("Invalid RPC URL: {}", e)))?
251            .connect_timeout(Duration::from_secs(5))
252            .timeout(config.rpc_timeout);
253
254        const MAX_RETRIES: u32 = 10;
255        let mut last_err = None;
256        for _ in 0..MAX_RETRIES {
257            match endpoint.connect().await {
258                Ok(ch) => {
259                    let event_service = EventServiceClient::new(ch.clone());
260                    let proof_service = ProofServiceClient::new(ch.clone());
261                    let ledger_service = LedgerServiceClient::new(ch);
262                    let epoch_cache = EpochCache::new(genesis_committee);
263
264                    return Ok(Self {
265                        event_service,
266                        proof_service,
267                        ledger_service,
268                        epoch_cache: Arc::new(tokio::sync::Mutex::new(epoch_cache)),
269                        config,
270                    });
271                }
272                Err(e) => {
273                    last_err = Some(e);
274                    tokio::time::sleep(Duration::from_millis(100)).await;
275                }
276            }
277        }
278        Err(last_err.unwrap().into())
279    }

300    /// Creates a stream of verified events starting from the latest position.
301    ///
302    /// # Arguments
303    ///
304    /// * `stream_id` - The address identifying the event stream (typically the package ID)
305    ///
306    /// # Returns
307    ///
308    /// A stream of `AuthenticatedEvent`s. Each event has been cryptographically verified against the
309    /// EventStreamHead's MMR (Merkle Mountain Range) commitment.
310    ///
311    /// # Error Handling
312    ///
313    /// The stream automatically handles transient errors by retrying:
314    /// - Network failures (`TransportError`)
315    /// - Temporary RPC errors (e.g., `Unavailable`, `DeadlineExceeded`)
316    ///
317    /// The stream terminates on terminal errors:
318    /// - `VerificationError`: Cryptographic verification failed (malicious or corrupted data)
319    /// - `InternalError`: Invalid state or deserialization failures
320    /// - Non-retriable RPC errors (e.g., `InvalidArgument`, `NotFound`)
321    ///
322    /// When the stream yields an `Err`, it's a terminal error and no more events will be produced.
323    /// The client should stop consuming the stream when an error is received.
324    ///
325    /// # Example
326    ///
327    /// ```no_run
328    /// # use std::sync::Arc;
329    /// # use sui_light_client::authenticated_events::AuthenticatedEventsClient;
330    /// # use sui_types::base_types::SuiAddress;
331    /// # async fn example(client: Arc<AuthenticatedEventsClient>, stream_id: SuiAddress) {
332    /// use futures::StreamExt;
333    ///
334    /// let mut stream = client.clone().stream_events(stream_id).await.unwrap();
335    ///
336    /// while let Some(result) = stream.next().await {
337    ///     match result {
338    ///         Ok(event) => {
339    ///             println!("Verified event at checkpoint {}", event.checkpoint);
340    ///         }
341    ///         Err(e) => {
342    ///             eprintln!("Terminal error: {:?}", e);
343    ///             break;
344    ///         }
345    ///     }
346    /// }
347    /// # }
348    /// ```
349    pub async fn stream_events(
350        self: Arc<Self>,
351        stream_id: SuiAddress,
352    ) -> Result<impl Stream<Item = Result<AuthenticatedEvent, ClientError>>, ClientError> {
353        let config = self.config.clone();
354        let stream_object_id = derive_event_stream_head_object_id(stream_id)
355            .map_err(|e| ClientError::InternalError(e.to_string()))?;
356
357        let result = self
358            .fetch_current_stream_head_and_verify(stream_object_id)
359            .await?;
360
361        let (verified_head, start_checkpoint) = match result {
362            Some((head, checkpoint)) => (Some(head), checkpoint + 1),
363            None => (None, 0),
364        };
365
366        stream::create_event_stream_with_head(
367            self,
368            stream_id,
369            stream_object_id,
370            start_checkpoint,
371            verified_head,
372            config,
373        )
374        .await
375    }

377    /// Creates a stream of verified events resuming from after a specific checkpoint where
378    /// event_stream_head was last modified.
379    ///
380    /// # Arguments
381    ///
382    /// * `stream_id` - The address identifying the event stream (typically the package ID)
383    /// * `last_verified_checkpoint` - The checkpoint to resume from (exclusive). Must be a checkpoint
384    ///   where the EventStreamHead was last modified (had events).
385    ///
386    /// # Returns
387    ///
388    /// A stream of `AuthenticatedEvent`s. Each event has been cryptographically verified against the
389    /// EventStreamHead's MMR (Merkle Mountain Range) commitment.
390    ///
391    /// # Setup Errors
392    ///
393    /// Returns an error during stream creation if:
394    /// - The EventStreamHead was not updated at `last_verified_checkpoint` (no events at that checkpoint)
395    /// - The checkpoint has been pruned
396    /// - Network or RPC communication errors during initial setup
397    ///
398    /// # Error Handling
399    ///
400    /// Once started, the stream automatically handles transient errors by retrying:
401    /// - Network failures (`TransportError`)
402    /// - Temporary RPC errors (e.g., `Unavailable`, `DeadlineExceeded`)
403    ///
404    /// The stream terminates on terminal errors:
405    /// - `VerificationError`: Cryptographic verification failed (malicious or corrupted data)
406    /// - `InternalError`: Invalid state or deserialization failures
407    /// - Non-retriable RPC errors (e.g., `InvalidArgument`, `NotFound`)
408    ///
409    /// When the stream yields an `Err`, it's a terminal error and no more events will be produced.
410    /// The client should stop consuming the stream when an error is received.
411    ///
412    /// # Example
413    ///
414    /// ```no_run
415    /// # use std::sync::Arc;
416    /// # use sui_light_client::authenticated_events::AuthenticatedEventsClient;
417    /// # use sui_types::base_types::SuiAddress;
418    /// # async fn example(client: Arc<AuthenticatedEventsClient>, stream_id: SuiAddress) {
419    /// use futures::StreamExt;
420    ///
421    /// let last_checkpoint = 42;
422    /// let mut stream = client.clone()
423    ///     .stream_events_from_checkpoint(stream_id, last_checkpoint)
424    ///     .await
425    ///     .unwrap();
426    ///
427    /// while let Some(result) = stream.next().await {
428    ///     match result {
429    ///         Ok(event) => {
430    ///             println!("Verified event at checkpoint {}", event.checkpoint);
431    ///         }
432    ///         Err(e) => {
433    ///             eprintln!("Terminal error: {:?}", e);
434    ///             break;
435    ///         }
436    ///     }
437    /// }
438    /// # }
439    /// ```
440    pub async fn stream_events_from_checkpoint(
441        self: Arc<Self>,
442        stream_id: SuiAddress,
443        last_verified_checkpoint: u64,
444    ) -> Result<impl Stream<Item = Result<AuthenticatedEvent, ClientError>>, ClientError> {
445        let stream_object_id = derive_event_stream_head_object_id(stream_id)
446            .map_err(|e| ClientError::InternalError(e.to_string()))?;
447
448        let (verified_head, start_checkpoint) = if last_verified_checkpoint == 0 {
449            (None, 0)
450        } else {
451            let verified_head = self
452                .fetch_and_verify_stream_head(stream_object_id, last_verified_checkpoint)
453                .await?;
454
455            (Some(verified_head), last_verified_checkpoint + 1)
456        };
457
458        let config = self.config.clone();
459        stream::create_event_stream_with_head(
460            self,
461            stream_id,
462            stream_object_id,
463            start_checkpoint,
464            verified_head,
465            config,
466        )
467        .await
468    }

470    async fn get_committee_for_checkpoint(
471        &self,
472        checkpoint: u64,
473    ) -> Result<Committee, ClientError> {
474        self.trust_ratchet_to_checkpoint(checkpoint).await?;
475
476        let epoch_cache = self.epoch_cache.lock().await;
477        let committee = epoch_cache
478            .get_committee_for_checkpoint(checkpoint)
479            .expect("Committee must exist after ensure_committee_for_checkpoint succeeded")
480            .clone();
481
482        Ok(committee)
483    }

485    // TODO: Move trust ratcheting to a shared module since it will be useful for all light clients,
486    // not just authenticated events.
487    async fn trust_ratchet_to_checkpoint(&self, checkpoint: u64) -> Result<(), ClientError> {
488        loop {
489            let (is_in_completed_epoch, current_epoch, current_committee, current_epoch_start) = {
490                let epoch_cache = self.epoch_cache.lock().await;
491                let is_in_completed_epoch =
492                    checkpoint < epoch_cache.current_epoch_start_checkpoint();
493                let current_epoch = epoch_cache.current_epoch();
494                let current_epoch_start = epoch_cache.current_epoch_start_checkpoint();
495                let current_committee = epoch_cache.current_committee().clone();
496                (
497                    is_in_completed_epoch,
498                    current_epoch,
499                    current_committee,
500                    current_epoch_start,
501                )
502            };
503
504            if is_in_completed_epoch {
505                return Ok(());
506            }
507
508            let result = self
509                .fetch_and_verify_next_epoch(current_epoch, &current_committee, checkpoint)
510                .await?;
511
512            let Some((end_of_epoch_checkpoint, next_committee)) = result else {
513                return Ok(());
514            };
515
516            let mut epoch_cache = self.epoch_cache.lock().await;
517            if epoch_cache.current_epoch() == current_epoch {
518                epoch_cache.apply_ratchet_update(
519                    current_epoch_start,
520                    end_of_epoch_checkpoint,
521                    current_committee,
522                    next_committee,
523                );
524            }
525        }
526    }

532    pub(crate) async fn fetch_and_verify_stream_head(
533        &self,
534        stream_object_id: sui_types::base_types::ObjectID,
535        checkpoint: u64,
536    ) -> Result<EventStreamHead, ClientError> {
537        let committee = self.get_committee_for_checkpoint(checkpoint).await?;
538
539        let mut proof_client = self.proof_service.clone();
540
541        let mut request =
542            sui_rpc_api::grpc::alpha::proof_service_proto::GetObjectInclusionProofRequest::default(
543            );
544        request.object_id = Some(stream_object_id.to_string());
545        request.checkpoint = Some(checkpoint);
546
547        let response = match proof_client.get_object_inclusion_proof(request).await {
548            Ok(resp) => resp.into_inner(),
549            Err(status) if status.code() == tonic::Code::FailedPrecondition => {
550                return Err(ClientError::InternalError(format!(
551                    "EventStreamHead was not updated at checkpoint {} (no events were emitted)",
552                    checkpoint
553                )));
554            }
555            Err(status) => return Err(ClientError::RpcError(status)),
556        };
557
558        let object_data_bytes = response
559            .object_data
560            .as_ref()
561            .ok_or_else(|| ClientError::InternalError("Missing object data".to_string()))?;
562
563        let object: sui_types::object::Object = bcs::from_bytes(object_data_bytes)?;
564        let stream_head = Self::extract_stream_head_from_object(&object)?;
565
566        self.verify_ocs_inclusion_proof(&committee, &response)?;
567
568        Ok(stream_head)
569    }

571    async fn fetch_current_stream_head_and_verify(
572        &self,
573        stream_object_id: sui_types::base_types::ObjectID,
574    ) -> Result<Option<(EventStreamHead, u64)>, ClientError> {
575        let mut ledger_client = self.ledger_service.clone();
576
577        let mut request = sui_rpc_api::proto::sui::rpc::v2::GetObjectRequest::default();
578        request.object_id = Some(stream_object_id.to_string());
579        request.read_mask = Some(FieldMask::from_paths(["bcs"]));
580
581        let response = match ledger_client.get_object(request).await {
582            Ok(r) => r.into_inner(),
583            Err(status) if status.code() == tonic::Code::NotFound => {
584                return Ok(None);
585            }
586            Err(status) => return Err(ClientError::RpcError(status)),
587        };
588
589        let proto_object = response
590            .object
591            .ok_or_else(|| ClientError::InternalError("Missing object in response".to_string()))?;
592
593        let bcs_data = proto_object
594            .bcs
595            .ok_or_else(|| ClientError::InternalError("Missing bcs data".to_string()))?;
596
597        let object_data_bytes = bcs_data
598            .value
599            .ok_or_else(|| ClientError::InternalError("Missing bcs value".to_string()))?;
600        let object: sui_types::object::Object = bcs::from_bytes(&object_data_bytes)?;
601        let stream_head = Self::extract_stream_head_from_object(&object)?;
602        let checkpoint = stream_head.checkpoint_seq;
603
604        let verified_head = self
605            .fetch_and_verify_stream_head(stream_object_id, checkpoint)
606            .await?;
607
608        Ok(Some((verified_head, checkpoint)))
609    }

611    async fn fetch_and_verify_next_epoch(
612        &self,
613        current_epoch: u64,
614        current_committee: &Committee,
615        to_checkpoint: u64,
616    ) -> Result<Option<(u64, Committee)>, ClientError> {
617        let mut ledger_client = self.ledger_service.clone();
618        let response = ledger_client
619            .get_epoch(GetEpochRequest::new(current_epoch))
620            .await;
621
622        let end_of_epoch_checkpoint_seq = match response {
623            Ok(resp) => {
624                let epoch_info =
625                    resp.into_inner()
626                        .epoch
627                        .ok_or(ClientError::InternalError(format!(
628                            "Failed to get last checkpoint of epoch {}: Missing epoch info",
629                            current_epoch
630                        )))?;
631                match epoch_info.last_checkpoint {
632                    Some(end) => end,
633                    None => return Ok(None),
634                }
635            }
636            Err(status) if status.code() == tonic::Code::NotFound => return Ok(None),
637            Err(status) => {
638                return Err(ClientError::InternalError(format!(
639                    "Failed to get last checkpoint of epoch {}: {}",
640                    current_epoch, status
641                )));
642            }
643        };
644
645        if to_checkpoint <= end_of_epoch_checkpoint_seq {
646            return Ok(None);
647        }
648
649        let checkpoint_response = ledger_client
650            .get_checkpoint(
651                GetCheckpointRequest::by_sequence_number(end_of_epoch_checkpoint_seq)
652                    .with_read_mask(FieldMask::from_paths(["summary", "signature", "contents"])),
653            )
654            .await
655            .map_err(|status| {
656                ClientError::InternalError(format!(
657                    "Failed to fetch checkpoint {}: {}",
658                    end_of_epoch_checkpoint_seq, status
659                ))
660            })?
661            .into_inner();
662
663        let proto_checkpoint = checkpoint_response
664            .checkpoint
665            .ok_or(ClientError::InternalError(
666                "Missing checkpoint in response".to_string(),
667            ))?;
668
669        let checkpoint: sui_types::full_checkpoint_content::Checkpoint =
670            (&proto_checkpoint).try_into().map_err(|e| {
671                ClientError::InternalError(format!("Failed to convert checkpoint: {:?}", e))
672            })?;
673
674        checkpoint
675            .summary
676            .verify_with_contents(current_committee, None)
677            .map_err(|e| {
678                ClientError::VerificationError(format!(
679                    "Failed to verify checkpoint {}: {}",
680                    end_of_epoch_checkpoint_seq, e
681                ))
682            })?;
683
684        let next_committee = extract_new_committee_info(&checkpoint.summary).map_err(|e| {
685            ClientError::VerificationError(format!(
686                "Failed to extract committee from checkpoint {}: {}",
687                end_of_epoch_checkpoint_seq, e
688            ))
689        })?;
690
691        Ok(Some((end_of_epoch_checkpoint_seq, next_committee)))
692    }
693
694    fn verify_ocs_inclusion_proof(
695        &self,
696        committee: &Committee,
697        response: &sui_rpc_api::grpc::alpha::proof_service_proto::GetObjectInclusionProofResponse,
698    ) -> Result<(), ClientError> {
699        let checkpoint_summary_bytes = response
700            .checkpoint_summary
701            .as_ref()
702            .ok_or_else(|| ClientError::InternalError("Missing checkpoint summary".to_string()))?;
703
704        let checkpoint_summary: sui_types::messages_checkpoint::CertifiedCheckpointSummary =
705            bcs::from_bytes(checkpoint_summary_bytes)?;
706
707        let object_ref_proto = response
708            .object_ref
709            .as_ref()
710            .ok_or_else(|| ClientError::InternalError("Missing object_ref".to_string()))?;
711
712        let inclusion_proof = response
713            .inclusion_proof
714            .as_ref()
715            .ok_or_else(|| ClientError::InternalError("Missing inclusion proof".to_string()))?;
716
717        let object_ref = converters::proto_object_ref_to_sui_object_ref(object_ref_proto)?;
718        let ocs_inclusion_proof =
719            converters::proto_ocs_inclusion_proof_to_light_client_proof(inclusion_proof)?;
720
721        let target = OCSTarget::new_inclusion_target(object_ref);
722
723        let proof = Proof {
724            targets: ProofTarget::ObjectCheckpointState(target),
725            checkpoint_summary,
726            proof_contents: ProofContents::ObjectCheckpointStateProof(OCSProof::Inclusion(
727                ocs_inclusion_proof,
728            )),
729        };
730
731        proof.verify(committee)?;
732
733        Ok(())
734    }
