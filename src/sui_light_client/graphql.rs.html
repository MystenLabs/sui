4use reqwest;
5use serde_json::{Value, json};
6
7use crate::config::Config;
8
9pub async fn query_last_checkpoint_of_epoch(config: &Config, epoch_id: u64) -> anyhow::Result<u64> {
10    // GraphQL query to get the last checkpoint of an epoch
11    let query = json!({
12        "query": "query ($epochID: Int) { epoch(id: $epochID) { checkpoints(last: 1) { nodes { sequenceNumber } } } }",
13        "variables": { "epochID": epoch_id }
14    });
15
16    // Submit the query by POSTing to the GraphQL endpoint
17    let client = reqwest::Client::new();
18    let resp = client
19        .post(config.graphql_url.as_ref().cloned().unwrap())
20        .header("Content-Type", "application/json")
21        .body(query.to_string())
22        .send()
23        .await
24        .expect("Cannot connect to graphql")
25        .text()
26        .await
27        .expect("Cannot parse response");
28
29    // Parse the JSON response to get the last checkpoint of the epoch
30    let v: Value = serde_json::from_str(resp.as_str()).expect("Incorrect JSON response");
31    let checkpoint_number = v["data"]["epoch"]["checkpoints"]["nodes"][0]["sequenceNumber"]
32        .as_u64()
33        .unwrap();
34
35    Ok(checkpoint_number)
36}
