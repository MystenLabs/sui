4use fastcrypto::hash::{Blake2b256, HashFunction};
5use move_core_types::u256::U256;
6use sui_types::accumulator_root::{EventCommitment, EventStreamHead, build_event_merkle_root};
7
8const U256_ZERO: U256 = U256::zero();
9
10fn hash_two_to_one_u256(left: U256, right: U256) -> U256 {
11    let mut concatenated = bcs::to_bytes(&left).expect("Failed to serialize left U256");
12    concatenated.extend_from_slice(&bcs::to_bytes(&right).expect("Failed to serialize right U256"));
13    let hash = Blake2b256::digest(&concatenated);
14    U256::from_le_bytes(&hash.digest)
15}
16
17fn add_to_stream(mmr: &mut Vec<U256>, new_val: U256) {
18    let mut i = 0;
19    let mut cur = new_val;
20
21    while i < mmr.len() {
22        let r = &mut mmr[i];
23        if *r == U256_ZERO {
24            *r = cur;
25            return;
26        } else {
27            cur = hash_two_to_one_u256(*r, cur);
28            *r = U256_ZERO;
29        }
30        i += 1;
31    }
32
33    // Vector length insufficient. Increase by 1.
34    mmr.push(cur);
35}
36
37// Returns the new stream head after applying the updates.
38// - head: the stream head to update.
39// - events: a list of events for each checkpoint.
40pub fn apply_stream_updates(
41    head: &EventStreamHead,
42    events: Vec<Vec<EventCommitment>>,
43) -> EventStreamHead {
44    let mut new_head = head.clone();
45    for cp_events in events {
46        // Verify that there are events in the checkpoint.
47        debug_assert!(!cp_events.is_empty());
48
49        // TODO: checkpoint_seq in EventCommitment is always 0, so we don't validate it
50
51        let digest = build_event_merkle_root(&cp_events);
52        let merkle_root = U256::from_le_bytes(&digest.into_inner());
53        add_to_stream(&mut new_head.mmr, merkle_root);
54        new_head.num_events += cp_events.len() as u64;
55    }
56    new_head
57}

66    #[test]
67    fn test_basic() {
68        let mut stream_head = EventStreamHead::new();
69        let new_val = U256::from(1u64);
70        add_to_stream(&mut stream_head.mmr, new_val);
71        assert_eq!(stream_head.mmr, vec![U256::from(1u64)]);
