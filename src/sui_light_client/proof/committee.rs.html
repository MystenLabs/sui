18/// The new committee to be verified.
19#[derive(Debug, Serialize, Deserialize, Clone)]
20pub struct CommitteeTarget {
21    pub committee: Committee,
22}
23
24impl ProofBuilder for CommitteeTarget {
25    fn construct(self, checkpoint: &CheckpointData) -> ProofResult<Proof> {
26        // Do a minimal check that the given checkpoint data is consistent with the committee
27        // Check we have the correct epoch
28        if checkpoint.checkpoint_summary.epoch() + 1 != self.committee.epoch {
29            return Err(ProofError::EpochMismatch);
30        }
31
32        // Check its an end of epoch checkpoint
33        if checkpoint.checkpoint_summary.end_of_epoch_data.is_none() {
34            return Err(ProofError::ExpectedEndOfEpochCheckpoint);
35        }
36
37        Ok(Proof {
38            targets: ProofTarget::Committee(self),
39            checkpoint_summary: checkpoint.checkpoint_summary.clone(),
40            proof_contents: ProofContents::CommitteeProof(CommitteeProof {}),
41        })
42    }
43}
44
45/// Note: The summary is enough to verify the committee.
46/// This is a placeholder for the committee proof.
47#[derive(Debug, Serialize, Deserialize, Clone)]
48pub struct CommitteeProof {}
49
50impl ProofContentsVerifier for CommitteeProof {
51    fn verify(self, targets: &ProofTarget, summary: &VerifiedCheckpoint) -> ProofResult<()> {
52        // Note: We just need to verify the new committee is the same as the one in the checkpoint
53        // summary as the summary is already verified.
54        match targets {
55            ProofTarget::Committee(target) => {
56                let new_committee = extract_new_committee_info(summary)?;
57                if new_committee != target.committee {
58                    return Err(ProofError::CommitteeMismatch);
59                }
60                Ok(())
61            }
62            _ => Err(ProofError::MismatchedTargetAndProofType),
63        }
64    }
65}
66
67/// Get the new committee from the end of epoch checkpoint summary.
68pub fn extract_new_committee_info(summary: &CertifiedCheckpointSummary) -> ProofResult<Committee> {
69    if let Some(next_epoch_committee) = summary.next_epoch_committee() {
70        let next_committee = next_epoch_committee.iter().cloned().collect();
71        let next_epoch = summary
72            .epoch()
73            .checked_add(1)
74            .ok_or(ProofError::EpochAddOverflow)?;
75        Ok(Committee::new(next_epoch, next_committee))
76    } else {
77        Err(ProofError::ExpectedEndOfEpochCheckpoint)
78    }
79}
