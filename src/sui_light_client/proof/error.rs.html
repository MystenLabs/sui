4use thiserror::Error;
5
6pub type ProofResult<T> = Result<T, ProofError>;
7
8/// Errors that can occur during proof construction and verification.
9#[derive(Debug, Error)]
10pub enum ProofError {
11    #[error("Given committee does not match the end of epoch committee")]
12    CommitteeMismatch,
13
14    #[error("Targets and proof type do not match")]
15    MismatchedTargetAndProofType,
16
17    #[error("No targets found")]
18    NoTargetsFound,
19
20    #[error("Transaction not found")]
21    TransactionNotFound,
22
23    #[error("All targets must refer to the same transaction")]
24    MultipleTransactionsNotSupported,
25
26    #[error("Object reference does not match the object")]
27    ObjectReferenceMismatch,
28
29    #[error("Object not found")]
30    ObjectNotFound,
31
32    #[error("Events digest does not match with effects")]
33    EventsDigestMismatch,
34
35    #[error("Event does not belong to the transaction")]
36    EventTransactionMismatch,
37
38    #[error("Event sequence number out of bounds")]
39    EventSequenceOutOfBounds,
40
41    #[error("Event contents do not match")]
42    EventContentsMismatch,
43
44    #[error("Events are missing from the transaction")]
45    EventsMissing,
46
47    #[error("Contents digest does not match the checkpoint summary")]
48    ContentsDigestMismatch,
49
50    #[error("Transaction digest does not match with execution digest")]
51    TransactionDigestMismatch,
52
53    #[error("Transaction digest not found in the checkpoint contents")]
54    TransactionDigestNotFound,
55
56    #[error("Epoch overflow when calculating next epoch")]
57    EpochAddOverflow,
58
59    #[error("Epoch mismatch between checkpoint and committee")]
60    EpochMismatch,
61
62    #[error("Expected end of epoch checkpoint")]
63    ExpectedEndOfEpochCheckpoint,
64
65    #[error("Checkpoint summary verification failed: {0}")]
66    SummaryVerificationFailed(String),
67
68    #[error("Invalid proof")]
69    InvalidProof,
70
71    #[error("Artifact digest mismatch")]
72    ArtifactDigestMismatch,
73
74    #[error("General error: {0}")]
75    GeneralError(String),
76}
