4use serde::{Deserialize, Serialize};
5
6use sui_types::{
7    event::{Event, EventID},
8    full_checkpoint_content::CheckpointData,
9};
10
11use crate::proof::{
12    base::{Proof, ProofBuilder, ProofContents, ProofTarget},
13    error::{ProofError, ProofResult},
14    transaction_proof::TransactionProof,
15};
16
17#[derive(Debug, Serialize, Deserialize, Clone)]
18pub struct EventsTarget {
19    pub events: Vec<(EventID, Event)>,
20}
21
22impl ProofBuilder for EventsTarget {
23    fn construct(self, checkpoint: &CheckpointData) -> ProofResult<Proof> {
24        let mut event_txs = self.events.iter().map(|(eid, _)| eid.tx_digest);
25
26        let target_tx = event_txs.next().ok_or(ProofError::NoTargetsFound)?;
27
28        // Check that all targets refer to the same transaction
29        if !event_txs.all(|tx| tx == target_tx) {
30            return Err(ProofError::MultipleTransactionsNotSupported);
31        }
32
33        let transaction_proof = TransactionProof::new(target_tx, checkpoint, true)?;
34
35        Ok(Proof {
36            targets: ProofTarget::Events(self),
37            checkpoint_summary: checkpoint.checkpoint_summary.clone(),
38            proof_contents: ProofContents::TransactionProof(transaction_proof),
39        })
40    }
41}
