4use clap::Parser;
5use move_binary_format::CompiledModule;
6use move_bytecode_source_map::utils::serialize_to_json_string;
7use move_cli::base;
8use move_disassembler::disassembler::Disassembler;
9use move_ir_types::location::Spanned;
10use move_package::BuildConfig;
11use std::fs::File;
12use std::io::{BufReader, Read};
13use std::path::Path;
14use std::path::PathBuf;
15
16#[derive(Parser)]
17#[group(id = "sui-move-disassemmble")]
18pub struct Disassemble {
19    /// Path to a .mv file to disassemble
20    #[clap(name = "module_path")]
21    module_path: PathBuf,
22
23    /// Whether to display the disassembly in raw Debug format
24    #[clap(long = "Xdebug")]
25    debug: bool,
26
27    #[clap(short = 'i', long = "interactive")]
28    interactive: bool,
29
30    /// Print the "bytecode map" (source map for disassembled bytecode)
31    #[clap(long = "bytecode-map")]
32    pub bytecode_map: bool,
33}
34
35impl Disassemble {
36    pub fn execute(
37        self,
38        package_path: Option<&Path>,
39        build_config: BuildConfig,
40    ) -> anyhow::Result<()> {
41        if base::reroot_path(Some(&self.module_path)).is_ok() {
42            // disassembling bytecode inside the source package that produced it--use the source info
43            let module_name = self
44                .module_path
45                .file_stem()
46                .expect("Bad module path")
47                .to_str()
48                .expect("Cannot convert module name to string")
49                .to_owned();
50            move_cli::base::disassemble::Disassemble {
51                interactive: self.interactive,
52                package_name: None,
53                module_or_script_name: module_name,
54                debug: self.debug,
55                bytecode_map: self.bytecode_map,
56            }
57            .execute(package_path, build_config)?;
58            return Ok(());
59        }
60
61        // disassembling a bytecode file with no source info
62        assert!(
63            Path::new(&self.module_path).exists(),
64            "Bad path to .mv file"
65        );
66
67        let mut bytes = Vec::new();
68        let mut file = BufReader::new(File::open(self.module_path)?);
69        file.read_to_end(&mut bytes)?;
70        // this deserialized a module to the max version of the bytecode but it's OK here because
71        // it's not run as part of the deterministic replicated state machine.
72        let module = CompiledModule::deserialize_with_defaults(&bytes)?;
73
74        if self.debug {
75            println!("{module:#?}");
76        } else {
77            let d = Disassembler::from_module(&module, Spanned::unsafe_no_loc(()).loc)?;
78            let (disassemble_string, bcode_map) = d.disassemble_with_source_map()?;
79            if self.bytecode_map {
80                println!("{}", serialize_to_json_string(&bcode_map)?);
81            }
82            println!("{}", disassemble_string);
83        }
84
85        Ok(())
86    }
87}
