4use clap::Parser;
5use move_cli::base::summary;
6use move_core_types::account_address::AccountAddress;
7use move_package::{BuildConfig, resolution::resolution_graph::ResolvedGraph};
8use std::{
9    collections::{BTreeMap, BTreeSet},
10    path::{Path, PathBuf},
11};
12use sui_types::{
13    base_types::ObjectID,
14    move_package::{TypeOrigin, UpgradeInfo},
15};
16
17#[derive(Parser)]
18#[group(id = "sui-move-summary")]
19pub struct Summary {
20    #[clap(flatten)]
21    pub summary: summary::Summary,
22    /// The object ID to summarize if `package-id` is present. The `--path` will be ignored if this field is used.
23    #[clap(long = "package-id", value_parser = ObjectID::from_hex_literal)]
24    pub package_id: Option<ObjectID>,
25}
26
27#[derive(serde::Serialize, serde::Deserialize, Clone, Default)]
28pub struct PackageSummaryMetadata {
29    #[serde(skip_serializing_if = "Option::is_none", default)]
30    pub root_package_id: Option<ObjectID>,
31    #[serde(skip_serializing_if = "Option::is_none", default)]
32    pub root_package_original_id: Option<ObjectID>,
33    #[serde(skip_serializing_if = "Option::is_none", default)]
34    pub root_package_version: Option<u64>,
35    // Mapping of original package ID to path to the package relative to the summary directory
36    // root.
37    #[serde(skip_serializing_if = "Option::is_none", default)]
38    pub dependencies: Option<BTreeMap<ObjectID, PathBuf>>,
39    // Mapping of original package ID to upgraded (on-chain) package ID.
40    #[serde(skip_serializing_if = "Option::is_none", default)]
41    pub linkage: Option<BTreeMap<ObjectID, UpgradeInfo>>,
42    // Mapping of original package ID to type origins for that package for each package.
43    #[serde(skip_serializing_if = "Option::is_none", default)]
44    pub type_origins: Option<BTreeMap<ObjectID, Vec<TypeOrigin>>>,
45}
46
47impl Summary {
48    pub fn execute(
49        self,
50        path: Option<&Path>,
51        build_config: BuildConfig,
52        sui_package_metadata: PackageSummaryMetadata,
53    ) -> anyhow::Result<()> {
54        self.summary.execute(
55            path,
56            build_config,
57            Some(&sui_package_metadata),
58            Some(Self::derive_ids),
59        )
60    }
61
62    fn derive_ids(resolved_graph: &mut ResolvedGraph) -> anyhow::Result<()> {
63        let root_pkg = resolved_graph
64            .package_table
65            .get_mut(&resolved_graph.root_package())
66            .unwrap();
67        let in_use_addrs = root_pkg
68            .resolved_table
69            .values()
70            .cloned()
71            .collect::<BTreeSet<_>>();
72        // Assign a unique address to each named address in the package deterministically.
73        // So start at 42 and increment until we find an address that is not in use (this should be
74        // always immediate in expectation).
75        let mut i = 42;
76        for (_, old_addr) in root_pkg.resolved_table.iter_mut() {
77            // If the named address is unset (0x0) then derive unique address for it.
78            if *old_addr == AccountAddress::ZERO {
79                loop {
80                    let random_addr = AccountAddress::from_suffix(i);
81                    i += 1;
82                    if !in_use_addrs.contains(&random_addr) {
83                        *old_addr = random_addr;
84                        break;
85                    }
86                }
87            }
88        }
89
90        let new_address_assignment = root_pkg.resolved_table.clone();
91
92        // NB: The root package is the has a global resolution of all named addresses so we are
93        // guaranteed to have all addresses in the `new_address_mapping`. If we can't find it
94        // that's an error.
95        let root_renaming = resolved_graph.root_renaming();
96        for (pkg_name, pkg) in resolved_graph.package_table.iter_mut() {
97            let package_root_renaming =
98                root_renaming.get(pkg_name).expect("Will always be present");
99            for (local_name, old_addr) in pkg.resolved_table.iter_mut() {
100                let root_name = package_root_renaming
101                    .get(local_name)
102                    .expect("Root renaming entry is present for every in-scope address");
103                let Some(new_addr) = new_address_assignment.get(root_name) else {
104                    anyhow::bail!(
105                        "IPE: Address {root_name} (local name = {local_name}) not found in new address mapping -- this shouldn't happen",
106                    );
107                };
108                *old_addr = *new_addr;
109            }
110        }
111        Ok(())
112    }
113}
