154    fn fn_info(units: &[AnnotatedCompiledModule]) -> FnInfoMap {
155        let mut fn_info_map = BTreeMap::new();
156        for u in units {
157            let mod_addr = u.named_module.address.into_inner();
158            let mod_is_test = u.attributes.is_test_or_test_only();
159            for (_, s, info) in &u.function_infos {
160                let fn_name = s.as_str().to_string();
161                let is_test = mod_is_test || info.attributes.is_test_or_test_only();
162                fn_info_map.insert(FnInfoKey { fn_name, mod_addr }, FnInfo { is_test });
163            }
164        }
165
166        fn_info_map
167    }

169    fn compile_package<W: Write>(
170        resolution_graph: &ResolvedGraph,
171        writer: &mut W,
172    ) -> anyhow::Result<(MoveCompiledPackage, FnInfoMap)> {
173        let build_plan = BuildPlan::create(resolution_graph)?;
174        let mut fn_info = None;
175        let compiled_pkg = build_plan.compile_with_driver(writer, |compiler| {
176            let (files, units_res) = compiler.build()?;
177            match units_res {
178                Ok((units, warning_diags)) => {
179                    decorate_warnings(warning_diags, Some(&files));
180                    fn_info = Some(Self::fn_info(&units));
181                    Ok((files, units))
182                }
183                Err(error_diags) => {
184                    // with errors present don't even try decorating warnings output to avoid
185                    // clutter
186                    assert!(!error_diags.is_empty());
187                    let diags_buf =
188                        report_diagnostics_to_buffer(&files, error_diags, /* color */ true);
189                    if let Err(err) = std::io::stderr().write_all(&diags_buf) {
190                        anyhow::bail!("Cannot output compiler diagnostics: {}", err);
191                    }
192                    anyhow::bail!("Compilation error");
193                }
194            }
195        })?;
196        Ok((compiled_pkg, fn_info.unwrap()))
197    }

199    /// Given a `path` and a `build_config`, build the package in that path, including its dependencies.
200    /// If we are building the Sui framework, we skip the check that the addresses should be 0
201    pub fn build(self, path: &Path) -> SuiResult<CompiledPackage> {
202        let print_diags_to_stderr = self.print_diags_to_stderr;
203        let run_bytecode_verifier = self.run_bytecode_verifier;
204        let chain_id = self.chain_id.clone();
205        let resolution_graph = self.resolution_graph(path, chain_id.clone())?;
206        build_from_resolution_graph(
207            resolution_graph,
208            run_bytecode_verifier,
209            print_diags_to_stderr,
210            chain_id,
211        )
212    }
213
214    pub fn resolution_graph(
215        mut self,
216        path: &Path,
217        chain_id: Option<String>,
218    ) -> SuiResult<ResolvedGraph> {
219        if let Some(err_msg) = set_sui_flavor(&mut self.config) {
220            return Err(SuiErrorKind::ModuleBuildFailure { error: err_msg }.into());
221        }
222
223        if self.print_diags_to_stderr {
224            self.config
225                .resolution_graph_for_package(path, chain_id, &mut std::io::stderr())
226        } else {
227            self.config
228                .resolution_graph_for_package(path, chain_id, &mut std::io::sink())
229        }
230        .map_err(|err| {
231            SuiErrorKind::ModuleBuildFailure {
232                error: format!("{:?}", err),
233            }
234            .into()
235        })
236    }
237}

239/// There may be additional information that needs to be displayed after diagnostics are reported
240/// (optionally report diagnostics themselves if files argument is provided).
241pub fn decorate_warnings(warning_diags: Diagnostics, files: Option<&MappedFiles>) {
242    let any_linter_warnings = warning_diags.any_with_prefix(LINT_WARNING_PREFIX);
243    let (filtered_diags_num, unique) =
244        warning_diags.filtered_source_diags_with_prefix(LINT_WARNING_PREFIX);
245    if let Some(f) = files {
246        report_warnings(f, warning_diags);
247    }
248    if any_linter_warnings {
249        eprintln!("Please report feedback on the linter warnings at https://forums.sui.io\n");
250    }
251    if filtered_diags_num > 0 {
252        eprintln!(
253            "Total number of linter warnings suppressed: {filtered_diags_num} (unique lints: {unique})"
254        );
255    }
256}

258/// Sets build config's default flavor to `Flavor::Sui`. Returns error message if the flavor was
259/// previously set to something else than `Flavor::Sui`.
260pub fn set_sui_flavor(build_config: &mut MoveBuildConfig) -> Option<String> {
261    use move_compiler::editions::Flavor;
262
263    let flavor = build_config.default_flavor.get_or_insert(Flavor::Sui);
264    if flavor != &Flavor::Sui {
265        return Some(format!(
266            "The flavor of the Move compiler cannot be overridden with anything but \
267                 \"{}\", but the default override was set to: \"{flavor}\"",
268            Flavor::Sui,
269        ));
270    }
271    None
272}

274pub fn build_from_resolution_graph(
275    resolution_graph: ResolvedGraph,
276    run_bytecode_verifier: bool,
277    print_diags_to_stderr: bool,
278    chain_id: Option<String>,
279) -> SuiResult<CompiledPackage> {
280    let (published_at, dependency_ids) = gather_published_ids(&resolution_graph, chain_id);
281
282    // collect bytecode dependencies as these are not returned as part of core
283    // `CompiledPackage`
284    let bytecode_deps = collect_bytecode_deps(&resolution_graph)?;
285
286    // compile!
287    let result = if print_diags_to_stderr {
288        BuildConfig::compile_package(&resolution_graph, &mut std::io::stderr())
289    } else {
290        BuildConfig::compile_package(&resolution_graph, &mut std::io::sink())
291    };
292
293    let (package, fn_info) = result.map_err(|error| SuiErrorKind::ModuleBuildFailure {
294        // Use [Debug] formatting to capture [anyhow] error context
295        error: format!("{:?}", error),
296    })?;
297
298    if run_bytecode_verifier {
299        verify_bytecode(&package, &fn_info)?;
300    }
301
302    Ok(CompiledPackage {
303        package,
304        published_at,
305        dependency_ids,
306        bytecode_deps,
307        dependency_graph: resolution_graph.graph,
308    })
309}

311/// Returns the bytecode deps from `resolution_graph` that have no source code
312fn collect_bytecode_deps(
313    resolution_graph: &ResolvedGraph,
314) -> SuiResult<Vec<(Symbol, CompiledModule)>> {
315    let mut bytecode_deps = vec![];
316    for (name, pkg) in resolution_graph.package_table.iter() {
317        if !pkg
318            .get_sources(&resolution_graph.build_options)
319            .unwrap()
320            .is_empty()
321        {
322            continue;
323        }
324        let modules = pkg.get_bytecodes_bytes().map_err(|error| {
325            SuiErrorKind::ModuleDeserializationFailure {
326                error: format!(
327                    "Deserializing bytecode dependency for package {}: {:?}",
328                    name, error
329                ),
330            }
331        })?;
332        for module in modules {
333            let module =
334                CompiledModule::deserialize_with_defaults(module.as_ref()).map_err(|error| {
335                    SuiErrorKind::ModuleDeserializationFailure {
336                        error: format!(
337                            "Deserializing bytecode dependency for package {}: {:?}",
338                            name, error
339                        ),
340                    }
341                })?;
342            bytecode_deps.push((*name, module));
343        }
344    }
345    Ok(bytecode_deps)
346}

348/// Check that the compiled modules in `package` are valid
349fn verify_bytecode(package: &MoveCompiledPackage, fn_info: &FnInfoMap) -> SuiResult<()> {
350    let compiled_modules = package.root_modules_map();
351    let verifier_config = ProtocolConfig::get_for_version(ProtocolVersion::MAX, Chain::Unknown)
352        .verifier_config(/* signing_limits */ None);
353
354    for m in compiled_modules.iter_modules() {
355        move_bytecode_verifier::verify_module_unmetered(m).map_err(|err| {
356            SuiErrorKind::ModuleVerificationFailure {
357                error: err.to_string(),
358            }
359        })?;
360        sui_bytecode_verifier::sui_verify_module_unmetered(m, fn_info, &verifier_config)?;
361    }
362    // TODO(https://github.com/MystenLabs/sui/issues/69): Run Move linker
363
364    Ok(())
365}

396    /// Return all of the bytecode modules in this package and the modules of its direct and transitive dependencies.

405    /// Return the bytecode modules in this package, topologically sorted in dependency order.
406    /// Optionally include dependencies that have not been published (are at address 0x0), if
407    /// `with_unpublished_deps` is true. This is the function to call if you would like to publish
408    /// or statically analyze the modules.
409    pub fn get_dependency_sorted_modules(
410        &self,
411        with_unpublished_deps: bool,
412    ) -> Vec<CompiledModule> {
413        let all_modules = Modules::new(self.get_modules_and_deps());
414
415        // SAFETY: package built successfully
416        let modules = all_modules.compute_topological_order().unwrap();
417
418        if with_unpublished_deps {
419            // For each transitive dependent module, if they are not to be published, they must have
420            // a non-zero address (meaning they are already published on-chain).
421            modules
422                .filter(|module| module.address() == &AccountAddress::ZERO)
423                .cloned()
424                .collect()
425        } else {
426            // Collect all module IDs from the current package to be published (module names are not
427            // sufficient as we may have modules with the same names in user code and in Sui
428            // framework which would result in the latter being pulled into a set of modules to be
429            // published).
430            let self_modules: HashSet<_> = self
431                .package
432                .root_modules_map()
433                .iter_modules()
434                .iter()
435                .map(|m| m.self_id())
436                .collect();
437
438            modules
439                .filter(|module| self_modules.contains(&module.self_id()))
440                .cloned()
441                .collect()
442        }
443    }

511    /// Generate layout schemas for all types declared by this package, as well as
512    /// all struct types passed into `entry` functions declared by modules in this package
513    /// (either directly or by reference).
514    /// These layout schemas can be consumed by clients (e.g., the TypeScript SDK) to enable
515    /// BCS serialization/deserialization of the package's objects, tx arguments, and events.
516    pub fn generate_struct_layouts(&self) -> Registry {
517        let pool = &mut normalized::RcPool::new();
518        let mut package_types = BTreeSet::new();
519        for m in self.get_modules() {
520            let normalized_m = normalized::Module::new(pool, m, /* include code */ false);
521            // 1. generate struct layouts for all declared types
522            'structs: for (name, s) in normalized_m.structs {
523                let mut dummy_type_parameters = Vec::new();
524                for t in &s.type_parameters {
525                    if t.is_phantom {
526                        // if all of t's type parameters are phantom, we can generate a type layout
527                        // we make this happen by creating a StructTag with dummy `type_params`, since the layout generator won't look at them.
528                        // we need to do this because SerdeLayoutBuilder will refuse to generate a layout for any open StructTag, but phantom types
529                        // cannot affect the layout of a struct, so we just use dummy values
530                        dummy_type_parameters.push(TypeTag::Signer)
531                    } else {
532                        // open type--do not attempt to generate a layout
533                        // TODO: handle generating layouts for open types?
534                        continue 'structs;
535                    }
536                }
537                debug_assert!(dummy_type_parameters.len() == s.type_parameters.len());
538                package_types.insert(StructTag {
539                    address: *m.address(),
540                    module: m.name().to_owned(),
541                    name: name.as_ident_str().to_owned(),
542                    type_params: dummy_type_parameters,
543                });
544            }
545            // 2. generate struct layouts for all parameters of `entry` funs
546            for (_name, f) in normalized_m.functions {
547                if f.is_entry {
548                    for t in &*f.parameters {
549                        let tag_opt = match &**t {
550                            Type::Address
551                            | Type::Bool
552                            | Type::Signer
553                            | Type::TypeParameter(_)
554                            | Type::U8
555                            | Type::U16
556                            | Type::U32
557                            | Type::U64
558                            | Type::U128
559                            | Type::U256
560                            | Type::Vector(_) => continue,
561                            Type::Reference(_, inner) => inner.to_struct_tag(pool),
562                            Type::Datatype(_) => t.to_struct_tag(pool),
563                        };
564                        if let Some(tag) = tag_opt {
565                            package_types.insert(tag);
566                        }
567                    }
568                }
569            }
570        }
571        let mut layout_builder = SerdeLayoutBuilder::new(self);
572        for typ in &package_types {
573            layout_builder.build_data_layout(typ).unwrap();
574        }
575        layout_builder.into_registry()
576    }

600    pub fn verify_unpublished_dependencies(
601        &self,
602        unpublished_deps: &BTreeSet<Symbol>,
603    ) -> SuiResult<()> {
604        if unpublished_deps.is_empty() {
605            return Ok(());
606        }
607
608        let errors = self
609            .package
610            .deps_compiled_units
611            .iter()
612            .filter_map(|(p, m)| {
613                if !unpublished_deps.contains(p) || m.unit.module.address() == &AccountAddress::ZERO
614                {
615                    return None;
616                }
617                Some(format!(
618                    " - {}::{} in dependency {}",
619                    m.unit.module.address(),
620                    m.unit.name,
621                    p
622                ))
623            })
624            .collect::<Vec<String>>();
625
626        if errors.is_empty() {
627            return Ok(());
628        }
629
630        let mut error_message = vec![];
631        error_message.push(
632            "The following modules in package dependencies set a non-zero self-address:".into(),
633        );
634        error_message.extend(errors);
635        error_message.push(
636            "If these packages really are unpublished, their self-addresses should be set \
637	     to \"0x0\" in the [addresses] section of the manifest when publishing. If they \
638	     are already published, ensure they specify the address in the `published-at` of \
639	     their Move.toml manifest."
640                .into(),
641        );
642
643        Err(SuiErrorKind::ModulePublishFailure {
644            error: error_message.join("\n"),
645        }
646        .into())
647    }

649    pub fn get_published_dependencies_ids(&self) -> Vec<ObjectID> {
650        self.dependency_ids.published.values().cloned().collect()
651    }

653    /// Find the map of packages that are immediate dependencies of the root modules, joined with
654    /// the set of bytecode dependencies.
655    pub fn find_immediate_deps_pkgs_to_keep(
656        &self,
657        with_unpublished_deps: bool,
658    ) -> Result<BTreeMap<Symbol, ObjectID>, anyhow::Error> {
659        // Start from the root modules (or all modules if with_unpublished_deps is true as we
660        // need to include modules with 0x0 address)
661        let root_modules: Vec<_> = if with_unpublished_deps {
662            self.package
663                .all_compiled_units_with_source()
664                .filter(|m| m.unit.address.into_inner() == AccountAddress::ZERO)
665                .map(|x| x.unit.clone())
666                .collect()
667        } else {
668            self.package
669                .root_modules()
670                .map(|x| x.unit.clone())
671                .collect()
672        };
673
674        // Find the immediate dependencies for each root module and store the package name
675        // in the pkgs_to_keep set. This basically prunes the packages that are not used
676        // based on the modules information.
677        let mut pkgs_to_keep: BTreeSet<Symbol> = BTreeSet::new();
678        let module_to_pkg_name: BTreeMap<_, _> = self
679            .package
680            .all_modules()
681            .map(|m| (m.unit.module.self_id(), m.unit.package_name))
682            .collect();
683
684        for module in &root_modules {
685            let immediate_deps = module.module.immediate_dependencies();
686            for dep in immediate_deps {
687                if let Some(pkg_name) = module_to_pkg_name.get(&dep) {
688                    let Some(pkg_name) = pkg_name else {
689                        bail!("Expected a package name but it's None")
690                    };
691                    pkgs_to_keep.insert(*pkg_name);
692                }
693            }
694        }
695
696        // If a package depends on another published package that has only bytecode without source
697        // code available, we need to include also that package as dep.
698        pkgs_to_keep.extend(self.bytecode_deps.iter().map(|(name, _)| *name));
699
700        // Finally, filter out packages that are published and exist in the manifest at the
701        // compilation time but are not referenced in the source code.
702        Ok(self
703            .dependency_ids
704            .clone()
705            .published
706            .into_iter()
707            .filter(|(pkg_name, _)| pkgs_to_keep.contains(pkg_name))
708            .collect())
709    }
710}

712/// Create a set of [Dependencies] from a [SystemPackagesVersion]; the dependencies are override git
713/// dependencies to the specific revision given by the [SystemPackagesVersion]
714///
715/// Skips "Deepbook" dependency.
716pub fn implicit_deps(packages: &SystemPackagesVersion) -> Dependencies {
717    let deps_to_skip = ["DeepBook".to_string()];
718    packages
719        .packages
720        .iter()
721        .filter(|package| !deps_to_skip.contains(&package.package_name))
722        .map(|package| {
723            (
724                package.package_name.clone().into(),
725                Dependency::Internal(InternalDependency {
726                    kind: DependencyKind::Git(GitInfo {
727                        git_url: SYSTEM_GIT_REPO.into(),
728                        git_rev: packages.git_revision.clone().into(),
729                        subdir: package.repo_path.clone().into(),
730                    }),
731                    subst: None,
732                    digest: None,
733                    dep_override: true,
734                }),
735            )
736        })
737        .collect()
738}

789pub struct PackageDependencies {
790    /// Set of published dependencies (name and address).
791    pub published: BTreeMap<Symbol, ObjectID>,
792    /// Set of unpublished dependencies (name).
793    pub unpublished: BTreeSet<Symbol>,
794    /// Set of dependencies with invalid `published-at` addresses.
795    pub invalid: BTreeMap<Symbol, String>,
796    /// Set of dependencies that have conflicting `published-at` addresses. The key refers to
797    /// the package, and the tuple refers to the address in the (Move.lock, Move.toml) respectively.
798    pub conflicting: BTreeMap<Symbol, (ObjectID, ObjectID)>,
799}

801/// Partition packages in `resolution_graph` into one of four groups:
802/// - The ID that the package itself is published at (if it is published)
803/// - The IDs of dependencies that have been published
804/// - The names of packages that have not been published on chain.
805/// - The names of packages that have a `published-at` field that isn't filled with a valid address.
806pub fn gather_published_ids(
807    resolution_graph: &ResolvedGraph,
808    chain_id: Option<String>,
809) -> (Result<ObjectID, PublishedAtError>, PackageDependencies) {
810    let root = resolution_graph.root_package();
811
812    let mut published = BTreeMap::new();
813    let mut unpublished = BTreeSet::new();
814    let mut invalid = BTreeMap::new();
815    let mut conflicting = BTreeMap::new();
816    let mut published_at = Err(PublishedAtError::NotPresent);
817
818    for (name, package) in &resolution_graph.package_table {
819        let property = resolve_published_id(package, chain_id.clone());
820        if name == &root {
821            // Separate out the root package as a special case
822            published_at = property;
823            continue;
824        }
825
826        match property {
827            Ok(id) => {
828                published.insert(*name, id);
829            }
830            Err(PublishedAtError::NotPresent) => {
831                unpublished.insert(*name);
832            }
833            Err(PublishedAtError::Invalid(value)) => {
834                invalid.insert(*name, value);
835            }
836            Err(PublishedAtError::Conflict {
837                id_lock,
838                id_manifest,
839            }) => {
840                conflicting.insert(*name, (id_lock, id_manifest));
841            }
842        };
843    }
844
845    (
846        published_at,
847        PackageDependencies {
848            published,
849            unpublished,
850            invalid,
851            conflicting,
852        },
853    )
854}

868pub fn check_unpublished_dependencies(unpublished: &BTreeSet<Symbol>) -> Result<(), SuiError> {
869    if unpublished.is_empty() {
870        return Ok(());
871    };

873    let mut error_messages = unpublished

918pub fn check_conflicting_addresses(
919    conflicting: &BTreeMap<Symbol, (ObjectID, ObjectID)>,
920    dump_bytecode_base64: bool,
921) -> Result<(), SuiError> {
922    if conflicting.is_empty() {
923        return Ok(());
924    }
925
926    let suffix = if conflicting.len() == 1 { "" } else { "es" };
927
928    let err_msg = format!("found the following conflicting published package address{suffix}:");
929    let suggestion_message =
930        "You may want to:
931 - delete the published-at address in the `Move.toml` if the `Move.lock` address is correct; OR
932 - update the `Move.lock` address to be the same as the `Move.toml`; OR
933 - check that your `sui active-env` corresponds to the chain on which the package is published (i.e., devnet, testnet, mainnet); OR
934 - contact the maintainer if this package is a dependency and request resolving the conflict.";
935
936    let conflicting_addresses_msg = conflicting
937        .iter()
938        .map(|(_, (id_lock, id_manifest))| {
939            format!(
940                "  `Move.toml` contains published-at address \
941                 {id_manifest} but `Move.lock` file contains published-at address {id_lock}."
942            )
943        })
944        .collect::<Vec<_>>()
945        .join("\n");
946
947    let error = format!("{err_msg}\n{conflicting_addresses_msg}\n{suggestion_message}");
948
949    let err = if dump_bytecode_base64 {
950        SuiErrorKind::ModuleBuildFailure { error }
951    } else {
952        SuiErrorKind::ModulePublishFailure { error }
953    };
954
955    Err(err.into())
