14const E_ADDRESS_PARSE_ERROR: u64 = 0;
15#[derive(Clone)]
16pub struct AddressFromBytesCostParams {
17    /// addresses are constant size, so base cost suffices
18    pub address_from_bytes_cost_base: InternalGas,
19}
20/***************************************************************************************************
21 * native fun from_bytes
22 * Implementation of the Move native function `address::from_bytes(bytes: vector<u8>)`
23 *   gas cost: address_from_bytes_cost_base                                        | addresses are constant size, so base cost suffices
24 **************************************************************************************************/
25pub fn from_bytes(
26    context: &mut NativeContext,
27    ty_args: Vec<Type>,
28    mut args: VecDeque<Value>,
29) -> PartialVMResult<NativeResult> {
30    debug_assert!(ty_args.is_empty());
31    debug_assert!(args.len() == 1);
32
33    let address_from_bytes_cost_params = get_extension!(context, NativesCostTable)?
34        .address_from_bytes_cost_params
35        .clone();
36
37    // Charge base fee
38    native_charge_gas_early_exit!(
39        context,
40        address_from_bytes_cost_params.address_from_bytes_cost_base
41    );
42
43    let addr_bytes = pop_arg!(args, Vec<u8>);
44    let cost = context.gas_used();
45
46    // Address parsing can fail if fed the incorrect number of bytes.
47    Ok(match AccountAddress::from_bytes(addr_bytes) {
48        Ok(addr) => NativeResult::ok(cost, smallvec![Value::address(addr)]),
49        Err(_) => NativeResult::err(cost, E_ADDRESS_PARSE_ERROR),
50    })
51}
52#[derive(Clone)]
53pub struct AddressToU256CostParams {
54    /// addresses and u256 are constant size, so base cost suffices
55    pub address_to_u256_cost_base: InternalGas,
56}
57/***************************************************************************************************
58 * native fun to_u256
59 * Implementation of the Move native function `address::to_u256(address): u256`
60 *   gas cost:  address_to_u256_cost_base                   | addresses and u256 are constant size, so base cost suffices
61 **************************************************************************************************/
62pub fn to_u256(
63    context: &mut NativeContext,
64    ty_args: Vec<Type>,
65    mut args: VecDeque<Value>,
66) -> PartialVMResult<NativeResult> {
67    debug_assert!(ty_args.is_empty());
68    debug_assert!(args.len() == 1);
69
70    let address_to_u256_cost_params = get_extension!(context, NativesCostTable)?
71        .address_to_u256_cost_params
72        .clone();
73
74    // Charge flat cost
75    native_charge_gas_early_exit!(
76        context,
77        address_to_u256_cost_params.address_to_u256_cost_base
78    );
79
80    let addr = pop_arg!(args, AccountAddress);
81    let mut addr_bytes_le = addr.to_vec();
82    addr_bytes_le.reverse();
83
84    // unwrap safe because we know addr_bytes_le is length 32
85    let u256_val = Value::u256(U256::from_le_bytes(&addr_bytes_le.try_into().unwrap()));
86    Ok(NativeResult::ok(context.gas_used(), smallvec![u256_val]))
87}

108        .address_from_u256_cost_params

112    native_charge_gas_early_exit!(

116
117    let u256 = pop_arg!(args, U256);
118    let mut u256_bytes = u256.to_le_bytes().to_vec();
119    u256_bytes.reverse();
120
121    // unwrap safe because we are passing a 32 byte slice
122    let addr_val = Value::address(AccountAddress::from_bytes(&u256_bytes[..]).unwrap());
123    Ok(NativeResult::ok(context.gas_used(), smallvec![addr_val]))
