51/***************************************************************************************************
52 * native fun secp256r1_ecrecover
53 * Implementation of the Move native function `secp256r1_ecrecover(signature: &vector<u8>, msg: &vector<u8>, hash: u8): vector<u8>`
54 * This function has two cost modes depending on the hash being set to `KECCAK256` or `SHA256`. The core formula is same but constants differ.
55 * If hash = 0, we use the `keccak256` cost constants, otherwise we use the `sha256` cost constants.
56 *   gas cost: ecdsa_r1_ecrecover_cost_base                    | covers various fixed costs in the oper
57 *              + ecdsa_r1_ecrecover_msg_cost_per_byte    * size_of(msg)        | covers cost of operating on each byte of `msg`
58 *              + ecdsa_r1_ecrecover_msg_cost_per_block   * num_blocks(msg)     | covers cost of operating on each block in `msg`
59 * Note: each block is of size `KECCAK256_BLOCK_SIZE` bytes for `keccak256` and `SHA256_BLOCK_SIZE` for `sha256`, and we round up.
60 *       `signature` is fixed size, so the cost is included in the base cost.
61 **************************************************************************************************/
62pub fn ecrecover(
63    context: &mut NativeContext,
64    ty_args: Vec<Type>,
65    mut args: VecDeque<Value>,
66) -> PartialVMResult<NativeResult> {
67    debug_assert!(ty_args.is_empty());
68    debug_assert!(args.len() == 3);
69
70    let hash = pop_arg!(args, u8);
71
72    // Load the cost parameters from the protocol config
73    let (ecdsa_r1_ecrecover_cost_params, crypto_invalid_arguments_cost) = {
74        let cost_table: &NativesCostTable = get_extension!(context)?;
75        (
76            cost_table.ecdsa_r1_ecrecover_cost_params.clone(),
77            cost_table.crypto_invalid_arguments_cost,
78        )
79    };
80
81    let (base_cost, cost_per_byte, cost_per_block, block_size) = match hash {
82        KECCAK256 => (
83            ecdsa_r1_ecrecover_cost_params.ecdsa_r1_ecrecover_keccak256_cost_base,
84            ecdsa_r1_ecrecover_cost_params.ecdsa_r1_ecrecover_keccak256_msg_cost_per_byte,
85            ecdsa_r1_ecrecover_cost_params.ecdsa_r1_ecrecover_keccak256_msg_cost_per_block,
86            KECCAK256_BLOCK_SIZE,
87        ),
88        SHA256 => (
89            ecdsa_r1_ecrecover_cost_params.ecdsa_r1_ecrecover_sha256_cost_base,
90            ecdsa_r1_ecrecover_cost_params.ecdsa_r1_ecrecover_sha256_msg_cost_per_byte,
91            ecdsa_r1_ecrecover_cost_params.ecdsa_r1_ecrecover_sha256_msg_cost_per_block,
92            SHA256_BLOCK_SIZE,
93        ),
94        _ => {
95            // Charge for failure but dont fail if we run out of gas otherwise the actual error is masked by OUT_OF_GAS error
96            context.charge_gas(crypto_invalid_arguments_cost);
97
98            return Ok(NativeResult::err(
99                context.gas_used(),
100                FAIL_TO_RECOVER_PUBKEY,
101            ));
102        }
103    };
104
105    // Charge the base cost for this oper
106    native_charge_gas_early_exit!(context, base_cost);
107
108    let msg = pop_arg!(args, VectorRef);
109    let signature = pop_arg!(args, VectorRef);
110
111    let msg_ref = msg.as_bytes_ref();
112    let signature_ref = signature.as_bytes_ref();
113
114    // Charge the arg size dependent costs
115    native_charge_gas_early_exit!(
116        context,
117        cost_per_byte * (msg_ref.len() as u64).into()
118            + cost_per_block * (msg_ref.len().div_ceil(block_size) as u64).into()
119    );
120
121    let cost = context.gas_used();
122
123    let Ok(sig) = <Secp256r1RecoverableSignature as ToFromBytes>::from_bytes(&signature_ref) else {
124        return Ok(NativeResult::err(cost, INVALID_SIGNATURE));
125    };
126
127    let pk = match hash {
128        KECCAK256 => sig.recover_with_hash::<Keccak256>(&msg_ref),
129        SHA256 => sig.recover_with_hash::<Sha256>(&msg_ref),
130        _ => Err(FastCryptoError::InvalidInput),
131    };
132
133    match pk {
134        Ok(pk) => Ok(NativeResult::ok(
135            cost,
136            smallvec![Value::vector_u8(pk.as_bytes().to_vec())],
137        )),
138        Err(_) => Ok(NativeResult::err(cost, FAIL_TO_RECOVER_PUBKEY)),
139    }
140}

158/***************************************************************************************************
159 * native fun secp256r1_verify
160 * Implementation of the Move native function `secp256r1_verify(signature: &vector<u8>, public_key: &vector<u8>, msg: &vector<u8>, hash: u8): bool`
161 * This function has two cost modes depending on the hash being set to `KECCAK256` or `SHA256`. The core formula is same but constants differ.
162 * If hash = 0, we use the `keccak256` cost constants, otherwise we use the `sha256` cost constants.
163 *   gas cost: ecdsa_r1_secp256r1_verify_cost_base                    | covers various fixed costs in the oper
164 *              + ecdsa_r1_secp256r1_verify_msg_cost_per_byte    * size_of(msg)        | covers cost of operating on each byte of `msg`
165 *              + ecdsa_r1_secp256r1_verify_msg_cost_per_block   * num_blocks(msg)     | covers cost of operating on each block in `msg`
166 * Note: each block is of size `KECCAK256_BLOCK_SIZE` bytes for `keccak256` and `SHA256_BLOCK_SIZE` for `sha256`, and we round up.
167 *       `signature` and `public_key` are fixed size, so their costs are included in the base cost.
168 **************************************************************************************************/
169pub fn secp256r1_verify(
170    context: &mut NativeContext,
171    ty_args: Vec<Type>,
172    mut args: VecDeque<Value>,
173) -> PartialVMResult<NativeResult> {
174    debug_assert!(ty_args.is_empty());
175    debug_assert!(args.len() == 4);
176    // Load the cost parameters from the protocol config
177    let (ecdsa_r1_secp256_r1_verify_cost_params, crypto_invalid_arguments_cost) = {
178        let cost_table: &NativesCostTable = get_extension!(context)?;
179        (
180            cost_table.ecdsa_r1_secp256_r1_verify_cost_params.clone(),
181            cost_table.crypto_invalid_arguments_cost,
182        )
183    };
184    let hash = pop_arg!(args, u8);
185    let (base_cost, cost_per_byte, cost_per_block, block_size) = match hash {
186        KECCAK256 => (
187            ecdsa_r1_secp256_r1_verify_cost_params.ecdsa_r1_secp256r1_verify_keccak256_cost_base,
188            ecdsa_r1_secp256_r1_verify_cost_params
189                .ecdsa_r1_secp256r1_verify_keccak256_msg_cost_per_byte,
190            ecdsa_r1_secp256_r1_verify_cost_params
191                .ecdsa_r1_secp256r1_verify_keccak256_msg_cost_per_block,
192            KECCAK256_BLOCK_SIZE,
193        ),
194        SHA256 => (
195            ecdsa_r1_secp256_r1_verify_cost_params.ecdsa_r1_secp256r1_verify_sha256_cost_base,
196            ecdsa_r1_secp256_r1_verify_cost_params
197                .ecdsa_r1_secp256r1_verify_sha256_msg_cost_per_byte,
198            ecdsa_r1_secp256_r1_verify_cost_params
199                .ecdsa_r1_secp256r1_verify_sha256_msg_cost_per_block,
200            SHA256_BLOCK_SIZE,
201        ),
202        _ => {
203            // Charge for failure but dont fail if we run out of gas otherwise the actual error is masked by OUT_OF_GAS error
204            context.charge_gas(crypto_invalid_arguments_cost);
205            return Ok(NativeResult::ok(
206                context.gas_used(),
207                smallvec![Value::bool(false)],
208            ));
209        }
210    };
211
212    // Charge the base cost for this oper
213    native_charge_gas_early_exit!(context, base_cost);
214
215    let msg = pop_arg!(args, VectorRef);
216    let public_key_bytes = pop_arg!(args, VectorRef);
217    let signature_bytes = pop_arg!(args, VectorRef);
218
219    let msg_ref = msg.as_bytes_ref();
220    let public_key_bytes_ref = public_key_bytes.as_bytes_ref();
221    let signature_bytes_ref = signature_bytes.as_bytes_ref();
222
223    // Charge the arg size dependent costs
224    native_charge_gas_early_exit!(
225        context,
226        cost_per_byte * (msg_ref.len() as u64).into()
227            + cost_per_block * (msg_ref.len().div_ceil(block_size) as u64).into()
228    );
229
230    let cost = context.gas_used();
231
232    let Ok(sig) = <Secp256r1Signature as ToFromBytes>::from_bytes(&signature_bytes_ref) else {
233        return Ok(NativeResult::ok(cost, smallvec![Value::bool(false)]));
234    };
235
236    let Ok(pk) = <Secp256r1PublicKey as ToFromBytes>::from_bytes(&public_key_bytes_ref) else {
237        return Ok(NativeResult::ok(cost, smallvec![Value::bool(false)]));
238    };
239
240    let result = match hash {
241        KECCAK256 => pk.verify_with_hash::<Keccak256>(&msg_ref, &sig).is_ok(),
242        SHA256 => pk.verify_with_hash::<Sha256>(&msg_ref, &sig).is_ok(),
243        _ => false,
244    };
245
246    Ok(NativeResult::ok(cost, smallvec![Value::bool(result)]))
