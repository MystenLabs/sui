3use crate::{NativesCostTable, get_extension};
4use fastcrypto::vrf::VRFProof;
5use fastcrypto::vrf::ecvrf::{ECVRFProof, ECVRFPublicKey};
6use move_binary_format::errors::PartialVMResult;
7use move_core_types::gas_algebra::InternalGas;
8use move_vm_runtime::{native_charge_gas_early_exit, native_functions::NativeContext};
9use move_vm_types::{
10    loaded_data::runtime_types::Type,
11    natives::function::NativeResult,
12    pop_arg,
13    values::{Value, VectorRef},
14};
15use smallvec::smallvec;
16use std::collections::VecDeque;
17
18pub const INVALID_ECVRF_HASH_LENGTH: u64 = 1;
19pub const INVALID_ECVRF_PUBLIC_KEY: u64 = 2;
20pub const INVALID_ECVRF_PROOF: u64 = 3;
21
22const ECVRF_SHA512_BLOCK_SIZE: usize = 128;
23
24#[derive(Clone)]
25pub struct EcvrfEcvrfVerifyCostParams {
26    /// Base cost for invoking the `ecvrf_verify`
27    pub ecvrf_ecvrf_verify_cost_base: InternalGas,
28    ///  Cost per byte of `alpha_string`
29    pub ecvrf_ecvrf_verify_alpha_string_cost_per_byte: InternalGas,
30    ///  Cost per block of `alpha_string` with block size = 128
31    pub ecvrf_ecvrf_verify_alpha_string_cost_per_block: InternalGas,
32}
33/***************************************************************************************************
34 * native fun ecvrf_verify
35 * Implementation of the Move native function `ecvrf_verify(hash: &vector<u8>, alpha_string: &vector<u8>, public_key: &vector<u8>, proof: &vector<u8>): bool`
36 *   gas cost: ecvrf_ecvrf_verify_cost_base                    | covers various fixed costs in the oper
37 *              + ecvrf_ecvrf_verify_alpha_string_cost_per_byte    * size_of(alpha_string)        | covers cost of operating on each byte of `alpha_string`
38 *              + ecvrf_ecvrf_verify_alpha_string_cost_per_block   * num_blocks(alpha_string)     | covers cost of operating on each block in `alpha_string`
39 * Note: each block is of size `ECVRF_SHA512_BLOCK_SIZE` bytes, and we round up.
40 *       `hash`, `proof`, and `public_key` are fixed size, so their costs are included in the base cost.
41 **************************************************************************************************/
42pub fn ecvrf_verify(
43    context: &mut NativeContext,
44    ty_args: Vec<Type>,
45    mut args: VecDeque<Value>,
46) -> PartialVMResult<NativeResult> {
47    debug_assert!(ty_args.is_empty());
48    debug_assert!(args.len() == 4);
49
50    // Load the cost parameters from the protocol config
51    let ecvrf_ecvrf_verify_cost_params = get_extension!(context, NativesCostTable)?
52        .ecvrf_ecvrf_verify_cost_params
53        .clone();
54    // Charge the base cost for this oper
55    native_charge_gas_early_exit!(
56        context,
57        ecvrf_ecvrf_verify_cost_params.ecvrf_ecvrf_verify_cost_base
58    );
59
60    let proof_bytes = pop_arg!(args, VectorRef);
61    let public_key_bytes = pop_arg!(args, VectorRef);
62    let alpha_string = pop_arg!(args, VectorRef);
63    let hash_bytes = pop_arg!(args, VectorRef);
64
65    let alpha_string_len = alpha_string.as_bytes_ref().len();
66    // Charge the arg size dependent costs
67    native_charge_gas_early_exit!(
68        context,
69        ecvrf_ecvrf_verify_cost_params.ecvrf_ecvrf_verify_alpha_string_cost_per_byte
70            * (alpha_string_len as u64).into()
71            + ecvrf_ecvrf_verify_cost_params.ecvrf_ecvrf_verify_alpha_string_cost_per_block
72                * (alpha_string_len.div_ceil(ECVRF_SHA512_BLOCK_SIZE) as u64).into()
73    );
74
75    let cost = context.gas_used();
76
77    let Ok(hash) = hash_bytes.as_bytes_ref().as_slice().try_into() else {
78        return Ok(NativeResult::err(cost, INVALID_ECVRF_HASH_LENGTH));
79    };
80
81    let Ok(public_key) =
82        bcs::from_bytes::<ECVRFPublicKey>(public_key_bytes.as_bytes_ref().as_slice())
83    else {
84        return Ok(NativeResult::err(cost, INVALID_ECVRF_PUBLIC_KEY));
85    };
86
87    let Ok(proof) = bcs::from_bytes::<ECVRFProof>(proof_bytes.as_bytes_ref().as_slice()) else {
88        return Ok(NativeResult::err(cost, INVALID_ECVRF_PROOF));
89    };
90
91    let result = proof.verify_output(alpha_string.as_bytes_ref().as_slice(), &public_key, &hash);
92    Ok(NativeResult::ok(
93        cost,
94        smallvec![Value::bool(result.is_ok())],
95    ))
96}
