19pub const NOT_SUPPORTED_ERROR: u64 = 0;
20pub const PARSE_ERROR: u64 = 1;
21pub const VERIFY_ERROR: u64 = 2;
22pub const INVALID_PCRS_ERROR: u64 = 3;

24// Gas related structs and functions.
25#[derive(Clone)]
26pub struct NitroAttestationCostParams {
27    pub parse_base_cost: Option<InternalGas>,
28    pub parse_cost_per_byte: Option<InternalGas>,
29    pub verify_base_cost: Option<InternalGas>,
30    pub verify_cost_per_cert: Option<InternalGas>,
31}

33macro_rules! native_charge_gas_early_exit_option {
34    ($native_context:ident, $cost:expr) => {{
35        use move_binary_format::errors::PartialVMError;
36        use move_core_types::vm_status::StatusCode;
37        native_charge_gas_early_exit!(
38            $native_context,
39            $cost.ok_or_else(|| {
40                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
41                    .with_message("Gas cost for nitro attestation is missing".to_string())
42            })?
43        );
44    }};
45}

47fn is_supported(context: &NativeContext) -> PartialVMResult<bool> {
48    Ok(get_extension!(context, ObjectRuntime)?
49        .protocol_config
50        .enable_nitro_attestation())
51}

53fn is_upgraded(context: &NativeContext) -> PartialVMResult<bool> {
54    Ok(get_extension!(context, ObjectRuntime)?
55        .protocol_config
56        .enable_nitro_attestation_upgraded_parsing())
57}

59pub fn load_nitro_attestation_internal(
60    context: &mut NativeContext,
61    ty_args: Vec<Type>,
62    mut args: VecDeque<Value>,
63) -> PartialVMResult<NativeResult> {
64    debug_assert!(ty_args.is_empty());
65    debug_assert!(args.len() == 2);
66
67    let cost = context.gas_used();
68    if !is_supported(context)? {
69        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
70    }
71
72    let current_timestamp = pop_arg!(args, u64);
73    let attestation_ref = pop_arg!(args, VectorRef);
74    let attestation_bytes = attestation_ref.as_bytes_ref();
75
76    let cost_params = get_extension!(context, NativesCostTable)?
77        .nitro_attestation_cost_params
78        .clone();
79
80    native_charge_gas_early_exit_option!(
81        context,
82        cost_params.parse_base_cost.and_then(|base_cost| cost_params
83            .parse_cost_per_byte
84            .map(|per_byte| base_cost + per_byte * (attestation_bytes.len() as u64).into()))
85    );
86
87    match parse_nitro_attestation(&attestation_bytes, is_upgraded(context)?) {
88        Ok((signature, signed_message, payload)) => {
89            let cert_chain_length = payload.get_cert_chain_length();
90            native_charge_gas_early_exit_option!(
91                context,
92                cost_params
93                    .verify_base_cost
94                    .and_then(|base_cost| cost_params
95                        .verify_cost_per_cert
96                        .map(|per_cert| base_cost + per_cert * (cert_chain_length as u64).into()))
97            );
98            match verify_nitro_attestation(&signature, &signed_message, &payload, current_timestamp)
99            {
100                Ok(()) => {
101                    let pcrs = if is_upgraded(context)? {
102                        to_indexed_struct(payload.pcr_map)?
103                    } else {
104                        to_indexed_struct_legacy(payload.pcr_vec)?
105                    };
106                    // Encapsulate as a lambda and call to allow us to capture any `Err` returns.
107                    // Could do this with `and_then` as well if desired.
108                    let result = || {
109                        Ok(Value::struct_(Struct::pack(vec![
110                            Value::vector_u8(payload.module_id.as_bytes().to_vec()),
111                            Value::u64(payload.timestamp),
112                            Value::vector_u8(payload.digest.as_bytes().to_vec()),
113                            pcrs,
114                            to_option_vector_u8(payload.public_key)?,
115                            to_option_vector_u8(payload.user_data)?,
116                            to_option_vector_u8(payload.nonce)?,
117                        ])))
118                    };
119                    NativeResult::map_partial_vm_result_one(context.gas_used(), result())
120                }
121                Err(_) => Ok(NativeResult::err(context.gas_used(), VERIFY_ERROR)),
122            }
123        }
124        Err(_) => Ok(NativeResult::err(context.gas_used(), PARSE_ERROR)),
125    }
126}
127// Build an Option<vector<u8>> value
128fn to_option_vector_u8(value: Option<Vec<u8>>) -> PartialVMResult<Value> {
129    match value {
130        // Some(<vector<u8>>) = { vector[ <vector<u8>> ] }
131        Some(vec) => Ok(Value::struct_(Struct::pack(vec![Vector::pack(
132            VectorSpecialization::Container,
133            vec![Value::vector_u8(vec)],
134        )?]))),
135        // None = { vector[ ] }
136        None => Ok(Value::struct_(Struct::pack(vec![Vector::empty(
137            VectorSpecialization::Container,
138        )?]))),
139    }
140}

142// Convert a map of index -> PCR to a vector of PCREntry struct with index
143// and value where the indices are [0, 1, 2, 3, 4, 8] since AWS currently
144// supports PCR0, PCR1, PCR2, PCR3, PCR4, PCR8.
145fn to_indexed_struct(pcrs: BTreeMap<u8, Vec<u8>>) -> PartialVMResult<Value> {
146    let mut sorted = pcrs.iter().collect::<Vec<_>>();
147    sorted.sort_by_key(|(key, _)| *key);
148    let mut indexed_struct = vec![];
149    for (index, pcr) in sorted.into_iter() {
150        indexed_struct.push(Value::struct_(Struct::pack(vec![
151            Value::u8(*index),
152            Value::vector_u8(pcr.to_vec()),
153        ])));
154    }
155    Vector::pack(VectorSpecialization::Container, indexed_struct)
156}
157
158// Convert a list of PCRs into a vector of PCREntry struct with index and value,
159// where the indices are [0, 1, 2, 3, 4, 8] since AWS currently supports PCR0,
160// PCR1, PCR2, PCR3, PCR4, PCR8.
161fn to_indexed_struct_legacy(pcrs: Vec<Vec<u8>>) -> PartialVMResult<Value> {
162    let indices = [0, 1, 2, 3, 4, 8];
163    if pcrs.len() != indices.len() {
164        return Err(PartialVMError::new(StatusCode::ABORTED).with_sub_status(INVALID_PCRS_ERROR));
165    };
166    let mut indexed_struct = vec![];
167    for (index, pcr) in pcrs.iter().enumerate() {
168        indexed_struct.push(Value::struct_(Struct::pack(vec![
169            Value::u8(indices[index]),
170            Value::vector_u8(pcr.to_vec()),
171        ])));
172    }
173    Vector::pack(VectorSpecialization::Container, indexed_struct)
174}
