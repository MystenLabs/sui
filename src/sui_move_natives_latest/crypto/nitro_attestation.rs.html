19pub const NOT_SUPPORTED_ERROR: u64 = 0;

33macro_rules! native_charge_gas_early_exit_option {
34    ($native_context:ident, $cost:expr) => {{
35        use move_binary_format::errors::PartialVMError;
36        use move_core_types::vm_status::StatusCode;
37        native_charge_gas_early_exit!(
38            $native_context,
39            $cost.ok_or_else(|| {
40                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
41                    .with_message("Gas cost for nitro attestation is missing".to_string())
42            })?
43        );
44    }};
45}

71pub fn load_nitro_attestation_internal(
72    context: &mut NativeContext,
73    ty_args: Vec<Type>,
74    mut args: VecDeque<Value>,
75) -> PartialVMResult<NativeResult> {
76    debug_assert!(ty_args.is_empty());
77    debug_assert!(args.len() == 2);
78
79    let cost = context.gas_used();
80    if !is_supported(context)? {
81        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
82    }
83
84    let current_timestamp = pop_arg!(args, u64);
85    let attestation_ref = pop_arg!(args, VectorRef);
86    let attestation_bytes = attestation_ref.as_bytes_ref();
87
88    let cost_params = get_extension!(context, NativesCostTable)?
89        .nitro_attestation_cost_params
90        .clone();
91
92    native_charge_gas_early_exit_option!(
93        context,
94        cost_params.parse_base_cost.and_then(|base_cost| cost_params
95            .parse_cost_per_byte
96            .map(|per_byte| base_cost + per_byte * (attestation_bytes.len() as u64).into()))
97    );
98
99    match parse_nitro_attestation(
100        &attestation_bytes,
101        is_upgraded(context)?,
102        is_all_nonzero_pcrs_included(context)?,
103        is_always_include_required_pcrs(context)?,
104    ) {
105        Ok((signature, signed_message, payload)) => {
106            let cert_chain_length = payload.get_cert_chain_length();
107            native_charge_gas_early_exit_option!(
108                context,
109                cost_params
110                    .verify_base_cost
111                    .and_then(|base_cost| cost_params
112                        .verify_cost_per_cert
113                        .map(|per_cert| base_cost + per_cert * (cert_chain_length as u64).into()))
114            );
115            match verify_nitro_attestation(&signature, &signed_message, &payload, current_timestamp)
116            {
117                Ok(()) => {
118                    let pcrs = if is_upgraded(context)? {
119                        to_indexed_struct(payload.pcr_map)?
120                    } else {
121                        to_indexed_struct_legacy(payload.pcr_vec)?
122                    };
123                    // Encapsulate as a lambda and call to allow us to capture any `Err` returns.
124                    // Could do this with `and_then` as well if desired.
125                    let result = || {
126                        Ok(Value::struct_(Struct::pack(vec![
127                            Value::vector_u8(payload.module_id.as_bytes().to_vec()),
128                            Value::u64(payload.timestamp),
129                            Value::vector_u8(payload.digest.as_bytes().to_vec()),
130                            pcrs,
131                            to_option_vector_u8(payload.public_key)?,
132                            to_option_vector_u8(payload.user_data)?,
133                            to_option_vector_u8(payload.nonce)?,
134                        ])))
135                    };
136                    NativeResult::map_partial_vm_result_one(context.gas_used(), result())
137                }
138                Err(_) => Ok(NativeResult::err(context.gas_used(), VERIFY_ERROR)),
139            }
140        }
141        Err(_) => Ok(NativeResult::err(context.gas_used(), PARSE_ERROR)),
142    }
143}
144// Build an Option<vector<u8>> value
145fn to_option_vector_u8(value: Option<Vec<u8>>) -> PartialVMResult<Value> {
146    match value {
147        // Some(<vector<u8>>) = { vector[ <vector<u8>> ] }
148        Some(vec) => Ok(Value::struct_(Struct::pack(vec![Vector::pack(
149            VectorSpecialization::Container,
150            vec![Value::vector_u8(vec)],
151        )?]))),
152        // None = { vector[ ] }
153        None => Ok(Value::struct_(Struct::pack(vec![Vector::empty(
154            VectorSpecialization::Container,
155        )?]))),
156    }
157}

159// Convert a map of index -> PCR to a vector of PCREntry struct with index
160// and value where the indices are [0, 1, 2, 3, 4, 8] since AWS currently
161// supports PCR0, PCR1, PCR2, PCR3, PCR4, PCR8.
162fn to_indexed_struct(pcrs: BTreeMap<u8, Vec<u8>>) -> PartialVMResult<Value> {
163    let mut sorted = pcrs.iter().collect::<Vec<_>>();
164    sorted.sort_by_key(|(key, _)| *key);
165    let mut indexed_struct = vec![];
166    for (index, pcr) in sorted.into_iter() {
167        indexed_struct.push(Value::struct_(Struct::pack(vec![
168            Value::u8(*index),
169            Value::vector_u8(pcr.to_vec()),
170        ])));
171    }
172    Vector::pack(VectorSpecialization::Container, indexed_struct)
173}
174
175// Convert a list of PCRs into a vector of PCREntry struct with index and value,
176// where the indices are [0, 1, 2, 3, 4, 8] since AWS currently supports PCR0,
177// PCR1, PCR2, PCR3, PCR4, PCR8.
178fn to_indexed_struct_legacy(pcrs: Vec<Vec<u8>>) -> PartialVMResult<Value> {
179    let indices = [0, 1, 2, 3, 4, 8];
180    if pcrs.len() != indices.len() {
181        return Err(PartialVMError::new(StatusCode::ABORTED).with_sub_status(INVALID_PCRS_ERROR));
182    };
183    let mut indexed_struct = vec![];
184    for (index, pcr) in pcrs.iter().enumerate() {
185        indexed_struct.push(Value::struct_(Struct::pack(vec![
186            Value::u8(indices[index]),
187            Value::vector_u8(pcr.to_vec()),
188        ])));
189    }
190    Vector::pack(VectorSpecialization::Container, indexed_struct)
191}
