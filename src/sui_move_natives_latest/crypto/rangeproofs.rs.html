23pub const NOT_SUPPORTED: u64 = 0;
24pub const INVALID_PROOF: u64 = 1;
25pub const INVALID_RANGE: u64 = 2;
26pub const INVALID_BATCH_SIZE: u64 = 3;
27
28/// Upper bound for batch size * range in bits
29pub const MAX_TOTAL_BITS: u64 = 512;
30
31/// Proofs with MAX_TOTAL_BITS = 512 will be exactly 864 bytes.
32const MAX_PROOF_SIZE: usize = 864;
33
34#[derive(Clone)]
35pub struct BulletproofsCostParams {
36    pub verify_bulletproofs_ristretto255_base_cost: Option<InternalGas>,
37    // The performance depends on the number of values/commitments * bits in range
38    pub verify_bulletproofs_ristretto255_cost_per_bit_and_commitment: Option<InternalGas>,
39}
40
41fn is_supported(context: &NativeContext) -> PartialVMResult<bool> {
42    Ok(get_extension!(context, ObjectRuntime)?
43        .protocol_config
44        .enable_verify_bulletproofs_ristretto255())
45}
46
47pub fn verify_bulletproofs_ristretto255(
48    context: &mut NativeContext,
49    _ty_args: Vec<Type>,
50    _args: VecDeque<Value>,
51) -> PartialVMResult<NativeResult> {
52    // Disabled: no longer called by the framework. The binding is retained only until the
53    // next execution version cut and can be removed as part of the next execution version cut.
54    Ok(NativeResult::err(context.gas_used(), NOT_SUPPORTED))
55}
56
57pub fn verify_bulletproofs_with_dst_ristretto255(
58    context: &mut NativeContext,
59    ty_args: Vec<Type>,
60    mut args: VecDeque<Value>,
61) -> PartialVMResult<NativeResult> {
62    debug_assert!(ty_args.is_empty());
63    debug_assert!(args.len() == 4);
64
65    if !is_supported(context)? {
66        return Ok(NativeResult::err(context.gas_used(), NOT_SUPPORTED));
67    }
68
69    let dst = pop_arg!(args, VectorRef).as_bytes_ref()?.to_vec();
70    let commitments = pop_arg!(args, VectorRef);
71    let range_bits = pop_arg!(args, u8);
72    let proof = pop_arg!(args, VectorRef);
73
74    // Load the cost parameters from the protocol config
75    let cost_parameters = get_extension!(context, NativesCostTable)?
76        .bulletproofs_cost_params
77        .clone();
78
79    // Charge the base cost for this operation
80    native_charge_gas_early_exit!(
81        context,
82        cost_parameters
83            .verify_bulletproofs_ristretto255_base_cost
84            .ok_or_else(|| partial_vm_error!(
85                UNKNOWN_INVARIANT_VIOLATION_ERROR,
86                "verify_bulletproofs_ristretto255_base_cost not available",
87            ))?
88    );
89
90    let proof_bytes = proof.as_bytes_ref()?;
91    if proof_bytes.len() > MAX_PROOF_SIZE {
92        return Ok(NativeResult::err(context.gas_used(), INVALID_PROOF));
93    }
94
95    let Ok(proof) = RangeProof::from_bytes(&proof_bytes) else {
96        return Ok(NativeResult::err(context.gas_used(), INVALID_PROOF));
97    };
98
99    let Ok(range) = range_from_bits(range_bits) else {
100        return Ok(NativeResult::err(context.gas_used(), INVALID_RANGE));
101    };
102
103    let vector_u8_type = Type::Vector(Box::new(Type::U8));
104    let length = commitments.len(&vector_u8_type)?.value_as::<u64>()?;
105
106    // The performance is linear in the product of length and range bits, so it is computed as base_cost + cost_per_bit * length * range_bits.
107    let total_bits = length * range_bits as u64;
108    if length == 0 || !length.is_power_of_two() || total_bits > MAX_TOTAL_BITS {
109        return Ok(NativeResult::err(context.gas_used(), INVALID_BATCH_SIZE));
110    }
111
112    // Charge the message dependent cost
113    native_charge_gas_early_exit!(
114        context,
115        cost_parameters
116            .verify_bulletproofs_ristretto255_cost_per_bit_and_commitment
117            .ok_or_else(|| partial_vm_error!(
118                UNKNOWN_INVARIANT_VIOLATION_ERROR,
119                "verify_bulletproofs_ristretto255_cost_per_bit_and_commitment not available",
120            ))?
121            * total_bits.into()
122    );
123
124    let commitments = (0..length)
125        .map(|i| {
126            commitments
127                .borrow_elem(i as usize, &vector_u8_type)
128                .and_then(|reference| reference.value_as::<VectorRef>())
129                .and_then(|v| Ok(v.as_bytes_ref()?.to_vec()))
130                .and_then(|v| {
131                    v.try_into()
132                        .map_err(|_| partial_vm_error!(INTERNAL_TYPE_ERROR))
133                })
134                .and_then(|b| {
135                    RistrettoPoint::from_trusted_byte_array(&b)
136                        .map_err(|_| partial_vm_error!(INTERNAL_TYPE_ERROR))
137                })
138                .map(PedersenCommitment)
139        })
140        .collect::<PartialVMResult<Vec<PedersenCommitment>>>()?;
141
142    let result = proof.verify_batch(&commitments, &range, &dst, &mut thread_rng());
143
144    Ok(NativeResult::ok(
145        context.gas_used(),
146        smallvec![Value::bool(result.is_ok())],
147    ))
148}

150fn range_from_bits(bits: u8) -> FastCryptoResult<Range> {
151    match bits {
152        8 => Ok(Range::Bits8),
153        16 => Ok(Range::Bits16),
154        32 => Ok(Range::Bits32),
155        64 => Ok(Range::Bits64),
156        _ => Err(InvalidInput),
157    }
158}
