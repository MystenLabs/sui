22const E_OVERFLOW: u64 = 0;
23const E_ADDRESS_BALANCE_NOT_ENABLED: u64 = 1;
24
25pub fn add_to_accumulator_address(
26    context: &mut NativeContext,
27    mut ty_args: Vec<Type>,
28    mut args: VecDeque<Value>,
29) -> PartialVMResult<NativeResult> {
30    debug_assert!(ty_args.len() == 1);
31    debug_assert!(args.len() == 3);
32
33    // TODO(address-balances): add specific cost for this
34    let event_emit_cost_params = context
35        .extensions_mut()
36        .get::<NativesCostTable>()?
37        .event_emit_cost_params
38        .clone();
39    native_charge_gas_early_exit!(context, event_emit_cost_params.event_emit_cost_base);
40
41    let ty_tag = context.type_to_type_tag(&ty_args.pop().unwrap())?;
42
43    let Some(value) = args.pop_back().unwrap().value_as::<Struct>().ok() else {
44        // TODO in the future this is guaranteed/checked via a custom verifier rule
45        debug_assert!(false);
46        return Err(
47            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
48                "Balance should be guaranteed under current implementation".to_owned(),
49            ),
50        );
51    };
52    let recipient = args
53        .pop_back()
54        .unwrap()
55        .value_as::<AccountAddress>()
56        .unwrap();
57    let accumulator: ObjectID = args
58        .pop_back()
59        .unwrap()
60        .value_as::<AccountAddress>()
61        .unwrap()
62        .into();
63
64    // TODO this will need to look at the layout of T when this is not guaranteed to be a Balance
65    let Some([amount]): Option<[Value; 1]> = value
66        .unpack()
67        .ok()
68        .and_then(|vs| vs.collect::<Vec<_>>().try_into().ok())
69    else {
70        debug_assert!(false);
71        return Err(
72            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
73                "Balance should be guaranteed under current implementation".to_owned(),
74            ),
75        );
76    };
77    let Some(amount) = amount.value_as::<u64>().ok() else {
78        debug_assert!(false);
79        return Err(
80            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
81                "Balance should be guaranteed under current implementation".to_owned(),
82            ),
83        );
84    };
85
86    let cost = context.gas_used();
87
88    let obj_runtime: &mut ObjectRuntime = context.extensions_mut().get_mut()?;
89
90    if !obj_runtime.protocol_config.enable_accumulators() {
91        return Ok(NativeResult::err(cost, E_ADDRESS_BALANCE_NOT_ENABLED));
92    }
93
94    obj_runtime.emit_accumulator_event(
95        accumulator,
96        MoveAccumulatorAction::Merge,
97        recipient,
98        ty_tag,
99        MoveAccumulatorValue::U64(amount),
100    )?;
101    Ok(NativeResult::ok(context.gas_used(), smallvec![]))
102}

104pub fn withdraw_from_accumulator_address(
105    context: &mut NativeContext,
106    mut ty_args: Vec<Type>,
107    mut args: VecDeque<Value>,
108) -> PartialVMResult<NativeResult> {
109    debug_assert!(ty_args.len() == 1);
110    debug_assert!(args.len() == 3);
111
112    // TODO(address-balances): add specific cost for this
113    // TODO(address-balances): determine storage cost for "Merge"
114    let event_emit_cost_params = context
115        .extensions_mut()
116        .get::<NativesCostTable>()?
117        .event_emit_cost_params
118        .clone();
119    native_charge_gas_early_exit!(context, event_emit_cost_params.event_emit_cost_base);
120
121    let ty_tag = context.type_to_type_tag(&ty_args.pop().unwrap())?;
122
123    let value = args.pop_back().unwrap().value_as::<U256>().unwrap();
124    let recipient = args
125        .pop_back()
126        .unwrap()
127        .value_as::<AccountAddress>()
128        .unwrap();
129    let accumulator: ObjectID = args
130        .pop_back()
131        .unwrap()
132        .value_as::<AccountAddress>()
133        .unwrap()
134        .into();
135
136    // TODO this will need to look at the layout of T when this is not guaranteed to be a Balance
137    let Ok(amount): Result<u64, _> = value.try_into() else {
138        return Ok(NativeResult::err(context.gas_used(), E_OVERFLOW));
139    };
140
141    let obj_runtime: &mut ObjectRuntime = context.extensions_mut().get_mut()?;
142    obj_runtime.emit_accumulator_event(
143        accumulator,
144        MoveAccumulatorAction::Split,
145        recipient,
146        ty_tag,
147        MoveAccumulatorValue::U64(amount),
148    )?;
149    // TODO this will need to look at the layout of T when this is not guaranteed to be a Balance
150    let withdrawn = Value::struct_(Struct::pack(vec![Value::u64(amount)]));
151    Ok(NativeResult::ok(context.gas_used(), smallvec![withdrawn]))
152}
