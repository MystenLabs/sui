17#[derive(Clone)]
18pub struct BorrowUidCostParams {
19    pub object_borrow_uid_cost_base: InternalGas,
20}
21/***************************************************************************************************
22 * native fun borrow_uid
23 * Implementation of the Move native function `borrow_uid<T: key>(obj: &T): &UID`
24 *   gas cost: object_borrow_uid_cost_base                | this is hard to calculate as it's very sensitive to `borrow_field` impl. Making it flat
25 **************************************************************************************************/
26pub fn borrow_uid(
27    context: &mut NativeContext,
28    ty_args: Vec<Type>,
29    mut args: VecDeque<Value>,
30) -> PartialVMResult<NativeResult> {
31    debug_assert!(ty_args.len() == 1);
32    debug_assert!(args.len() == 1);
33
34    let borrow_uid_cost_params = get_extension!(context, NativesCostTable)?
35        .borrow_uid_cost_params
36        .clone();
37
38    // Charge base fee
39    native_charge_gas_early_exit!(context, borrow_uid_cost_params.object_borrow_uid_cost_base);
40
41    let obj = pop_arg!(args, StructRef);
42    let id_field = obj.borrow_field(0)?;
43
44    Ok(NativeResult::ok(context.gas_used(), smallvec![id_field]))
45}

47#[derive(Clone)]
48pub struct DeleteImplCostParams {
49    pub object_delete_impl_cost_base: InternalGas,
50}
51/***************************************************************************************************
52 * native fun delete_impl
53 * Implementation of the Move native function `delete_impl(id: address)`
54 *   gas cost: cost_base                | this is a simple ID deletion
55 **************************************************************************************************/
56pub fn delete_impl(
57    context: &mut NativeContext,
58    ty_args: Vec<Type>,
59    mut args: VecDeque<Value>,
60) -> PartialVMResult<NativeResult> {
61    debug_assert!(ty_args.is_empty());
62    debug_assert!(args.len() == 1);
63
64    let delete_impl_cost_params = get_extension!(context, NativesCostTable)?
65        .delete_impl_cost_params
66        .clone();
67
68    // Charge base fee
69    native_charge_gas_early_exit!(
70        context,
71        delete_impl_cost_params.object_delete_impl_cost_base
72    );
73
74    // unwrap safe because the interface of native function guarantees it.
75    let uid_bytes = pop_arg!(args, AccountAddress);
76
77    let obj_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
78    obj_runtime.delete_id(uid_bytes.into())?;
79    Ok(NativeResult::ok(context.gas_used(), smallvec![]))
80}
