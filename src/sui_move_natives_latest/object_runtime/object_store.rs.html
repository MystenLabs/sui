84struct Inner<'a> {
85    // used for loading child objects
86    resolver: &'a dyn ChildObjectResolver,
87    // The version of the root object in ownership at the beginning of the transaction.
88    // If it was a child object, it resolves to the root parent's sequence number.
89    // Otherwise, it is just the sequence number at the beginning of the transaction.
90    root_version: BTreeMap<ObjectID, SequenceNumber>,
91    // A map from a wrapped object to the object it was contained in at the
92    // beginning of the transaction.
93    wrapped_object_containers: BTreeMap<ObjectID, ObjectID>,
94    // cached objects from the resolver. An object might be in this map but not in the store
95    // if it's existence was queried, but the value was not used.
96    cached_objects: BTreeMap<ObjectID, Option<Object>>,
97    // whether or not this TX is gas metered
98    is_metered: bool,
99    // Protocol config used to enforce limits
100    protocol_config: &'a ProtocolConfig,
101    // Metrics for reporting exceeded limits
102    metrics: Arc<LimitsMetrics>,
103    // Epoch ID for the current transaction. Used for receiving objects.
104    current_epoch_id: EpochId,
105}

107// maintains the runtime GlobalValues for child objects and manages the fetching of objects
108// from storage, through the `ChildObjectResolver`
109pub(super) struct ChildObjectStore<'a> {
110    // contains object resolver and object cache
111    // kept as a separate struct to deal with lifetime issues where the `store` is accessed
112    // at the same time as the `cached_objects` is populated
113    inner: Inner<'a>,
114    // Maps of populated GlobalValues, meaning the child object has been accessed in this
115    // transaction
116    store: BTreeMap<ObjectID, ChildObject>,
117    config_setting_cache: BTreeMap<ObjectID, ConfigSetting>,
118    // whether or not this TX is gas metered
119    is_metered: bool,
120}

130macro_rules! fetch_child_object_unbounded {
131    ($inner:ident, $parent:ident, $child:ident, $parents_root_version:expr, $had_parent_root_version:expr) => {{
132        let child_opt = $inner
133            .resolver
134            .read_child_object(&$parent, &$child, $parents_root_version)
135            .map_err(|msg| {
136                PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!("{msg}"))
137            })?;
138        if let Some(object) = child_opt {
139            // if there was no root version, guard against reading a child object. A newly
140            // created parent should not have a child in storage
141            if !$had_parent_root_version {
142                return Err(
143                    PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!(
144                        "A new parent {} should not have a child object {}.",
145                        $parent, $child
146                    )),
147                );
148            }
149            // guard against bugs in `read_child_object`: if it returns a child object such that
150            // C.parent != parent, we raise an invariant violation
151            match &object.owner {
152                Owner::ObjectOwner(id) => {
153                    if ObjectID::from(*id) != $parent {
154                        return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
155                            format!(
156                                "Bad owner for {}. Expected owner {} but found owner {}",
157                                $child, $parent, id
158                            ),
159                        ));
160                    }
161                }
162                Owner::AddressOwner(_)
163                | Owner::Immutable
164                | Owner::Shared { .. }
165                | Owner::ConsensusAddressOwner { .. } => {
166                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
167                        format!(
168                            "Bad owner for {}. \
169                            Expected an id owner {} but found an address, \
170                            immutable, or shared owner",
171                            $child, $parent
172                        ),
173                    ));
174                }
175            };
176            match &object.data {
177                Data::Package(_) => {
178                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
179                        format!(
180                            "Mismatched object type for {}. \
181                            Expected a Move object but found a Move package",
182                            $child
183                        ),
184                    ));
185                }
186                Data::Move(_) => Some(object),
187            }
188        } else {
189            None
190        }
191    }};
192}

194impl Inner<'_> {
195    fn receive_object_from_store(
196        &self,
197        owner: ObjectID,
198        child: ObjectID,
199        version: SequenceNumber,
200    ) -> PartialVMResult<CacheMetadata<LoadedWithMetadataResult<MoveObject>>> {
201        let child_opt = self
202            .resolver
203            .get_object_received_at_version(&owner, &child, version, self.current_epoch_id)
204            .map_err(|msg| {
205                PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!("{msg}"))
206            })?;
207        let (cache_info, obj_opt) = if let Some(object) = child_opt {
208            // guard against bugs in `receive_object_at_version`: if it returns a child object such that
209            // C.parent != parent, we raise an invariant violation since that should be checked by
210            // `receive_object_at_version`.
211            if object.owner != Owner::AddressOwner(owner.into()) {
212                return Err(
213                    PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!(
214                        "Bad owner for {child}. \
215                        Expected owner {owner} but found owner {}",
216                        object.owner
217                    )),
218                );
219            }
220            let loaded_metadata = DynamicallyLoadedObjectMetadata {
221                version,
222                digest: object.digest(),
223                storage_rebate: object.storage_rebate,
224                owner: object.owner.clone(),
225                previous_transaction: object.previous_transaction,
226            };
227
228            // `ChildObjectResolver::receive_object_at_version` should return the object at the
229            // version or nothing at all. If it returns an object with a different version, we
230            // should raise an invariant violation since it should be checked by
231            // `receive_object_at_version`.
232            if object.version() != version {
233                return Err(
234                    PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!(
235                        "Bad version for {child}. \
236                        Expected version {version} but found version {}",
237                        object.version()
238                    )),
239                );
240            }
241            match object.into_inner().data {
242                Data::Package(_) => {
243                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
244                        format!(
245                            "Mismatched object type for {child}. \
246                                Expected a Move object but found a Move package"
247                        ),
248                    ));
249                }
250                Data::Move(mo @ MoveObject { .. }) => (
251                    CacheInfo::Loaded(Some(mo.contents().len())),
252                    Some((mo, loaded_metadata)),
253                ),
254            }
255        } else {
256            (CacheInfo::Loaded(None), None)
257        };
258        Ok((cache_info, obj_opt))
259    }

261    #[allow(clippy::map_entry)]
262    fn get_or_fetch_object_from_store(
263        &mut self,
264        parent: ObjectID,
265        child: ObjectID,
266    ) -> PartialVMResult<CacheMetadata<Option<&MoveObject>>> {
267        let cached_objects_count = self.cached_objects.len() as u64;
268        let parents_root_version = self.root_version.get(&parent).copied();
269        let had_parent_root_version = parents_root_version.is_some();
270        // if not found, it must be new so it won't have any child objects, thus
271        // we can return SequenceNumber(0) as no child object will be found
272        let parents_root_version = parents_root_version.unwrap_or(SequenceNumber::new());
273        let cache_info = if let btree_map::Entry::Vacant(e) = self.cached_objects.entry(child) {
274            let obj_opt = fetch_child_object_unbounded!(
275                self,
276                parent,
277                child,
278                parents_root_version,
279                had_parent_root_version
280            );
281
282            if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
283                self.is_metered,
284                cached_objects_count,
285                self.protocol_config.object_runtime_max_num_cached_objects(),
286                self.protocol_config
287                    .object_runtime_max_num_cached_objects_system_tx(),
288                self.metrics.excessive_object_runtime_cached_objects
289            ) {
290                return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
291                    .with_message(format!(
292                        "Object runtime cached objects limit ({} entries) reached",
293                        lim
294                    ))
295                    .with_sub_status(
296                        VMMemoryLimitExceededSubStatusCode::OBJECT_RUNTIME_CACHE_LIMIT_EXCEEDED
297                            as u64,
298                    ));
299            };
300            let num_bytes_opt = match &obj_opt {
301                Some(obj) => {
302                    // unwrap safe because we only insert Move objects
303                    let move_obj = obj.data.try_as_move().unwrap();
304                    Some(move_obj.contents().len())
305                }
306                None => None,
307            };
308
309            e.insert(obj_opt);
310            CacheInfo::Loaded(num_bytes_opt)
311        } else {
312            CacheInfo::CachedObject
313        };
314        // unwraps are safe because it must be inserted and we only insert Move objects
315        let move_obj_opt = self
316            .cached_objects
317            .get(&child)
318            .unwrap()
319            .as_ref()
320            .map(|obj| obj.data.try_as_move().unwrap());
321        Ok((cache_info, move_obj_opt))
322    }

324    fn fetch_object_impl(
325        &mut self,
326        parent: ObjectID,
327        child: ObjectID,
328        child_ty_layout: &R::MoveTypeLayout,
329        child_ty_fully_annotated_layout: &A::MoveTypeLayout,
330        child_move_type: &MoveObjectType,
331    ) -> PartialVMResult<
332        ObjectResult<CacheMetadata<(MoveObjectType, GlobalValue, ObjectFingerprint)>>,
333    > {
334        // we copy the reference to the protocol config ahead of time for lifetime reasons
335        let protocol_config = self.protocol_config;
336        // retrieve the object from storage if it exists
337        let (cache_info, obj_opt) = self.get_or_fetch_object_from_store(parent, child)?;
338        let Some(obj) = obj_opt else {
339            return Ok(ObjectResult::Loaded((
340                cache_info,
341                (
342                    child_move_type.clone(),
343                    GlobalValue::none(),
344                    ObjectFingerprint::none(protocol_config),
345                ),
346            )));
347        };
348        // object exists, but the type does not match
349        if obj.type_() != child_move_type {
350            return Ok(ObjectResult::MismatchedType);
351        }
352        // deserialize the value
353        let obj_contents = obj.contents();
354        let v = match Value::simple_deserialize(obj_contents, child_ty_layout) {
355            Some(v) => v,
356            None => return Err(
357                PartialVMError::new(StatusCode::FAILED_TO_DESERIALIZE_RESOURCE).with_message(
358                    format!("Failed to deserialize object {child} with type {child_move_type}",),
359                ),
360            ),
361        };
362        // save a fingerprint
363        let fingerprint =
364            ObjectFingerprint::preexisting(protocol_config, &parent, child_move_type, &v).map_err(
365                |e| {
366                    PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
367                        format!("Failed to fingerprint value for object {child}. Error: {e}"),
368                    )
369                },
370            )?;
371        // generate a global value
372        let global_value =
373            match GlobalValue::cached(v) {
374                Ok(gv) => gv,
375                Err(e) => {
376                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
377                        format!("Object {child} did not deserialize to a struct Value. Error: {e}"),
378                    ));
379                }
380            };
381        // Find all UIDs inside of the value and update the object parent maps
382        let contained_uids =
383            get_all_uids(child_ty_fully_annotated_layout, obj_contents).map_err(|e| {
384                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
385                    .with_message(format!("Failed to find UIDs. ERROR: {e}"))
386            })?;
387        let parents_root_version = self.root_version.get(&parent).copied();
388        if let Some(v) = parents_root_version {
389            debug_assert!(contained_uids.contains(&child));
390            for id in contained_uids {
391                self.root_version.insert(id, v);
392                if id != child {
393                    let prev = self.wrapped_object_containers.insert(id, child);
394                    debug_assert!(prev.is_none())
395                }
396            }
397        }
398        Ok(ObjectResult::Loaded((
399            cache_info,
400            (child_move_type.clone(), global_value, fingerprint),
401        )))
402    }
403}
404
405fn deserialize_move_object(
406    obj: &MoveObject,
407    child_ty_layout: &R::MoveTypeLayout,
408    child_move_type: MoveObjectType,
409) -> PartialVMResult<ObjectResult<(MoveObjectType, Value)>> {
410    let child_id = obj.id();
411    // object exists, but the type does not match
412    if obj.type_() != &child_move_type {
413        return Ok(ObjectResult::MismatchedType);
414    }
415    let value = match Value::simple_deserialize(obj.contents(), child_ty_layout) {
416        Some(v) => v,
417        None => {
418            return Err(
419                PartialVMError::new(StatusCode::FAILED_TO_DESERIALIZE_RESOURCE).with_message(
420                    format!("Failed to deserialize object {child_id} with type {child_move_type}",),
421                ),
422            );
423        }
424    };
425    Ok(ObjectResult::Loaded((child_move_type, value)))
426}

428impl<'a> ChildObjectStore<'a> {
429    pub(super) fn new(
430        resolver: &'a dyn ChildObjectResolver,
431        root_version: BTreeMap<ObjectID, SequenceNumber>,
432        wrapped_object_containers: BTreeMap<ObjectID, ObjectID>,
433        is_metered: bool,
434        protocol_config: &'a ProtocolConfig,
435        metrics: Arc<LimitsMetrics>,
436        current_epoch_id: EpochId,
437    ) -> Self {
438        Self {
439            inner: Inner {
440                resolver,
441                root_version,
442                wrapped_object_containers,
443                cached_objects: BTreeMap::new(),
444                is_metered,
445                protocol_config,
446                metrics,
447                current_epoch_id,
448            },
449            store: BTreeMap::new(),
450            config_setting_cache: BTreeMap::new(),
451            is_metered,
452        }
453    }
454
455    pub(super) fn receive_object(
456        &mut self,
457        parent: ObjectID,
458        child: ObjectID,
459        child_version: SequenceNumber,
460        child_layout: &R::MoveTypeLayout,
461        child_fully_annotated_layout: &A::MoveTypeLayout,
462        child_move_type: MoveObjectType,
463    ) -> PartialVMResult<LoadedWithMetadataResult<ObjectResult<CacheMetadata<Value>>>> {
464        let (cache_info, Some((obj, obj_meta))) =
465            self.inner
466                .receive_object_from_store(parent, child, child_version)?
467        else {
468            return Ok(None);
469        };
470
471        Ok(Some(
472            match deserialize_move_object(&obj, child_layout, child_move_type)? {
473                ObjectResult::MismatchedType => (ObjectResult::MismatchedType, obj_meta),
474                ObjectResult::Loaded((_, v)) => {
475                    // Find all UIDs inside of the value and update the object parent maps with the contained
476                    // UIDs in the received value. They should all have an upper bound version as the receiving object.
477                    // Only do this if we successfully load the object though.
478                    let contained_uids = get_all_uids(child_fully_annotated_layout, obj.contents())
479                        .map_err(|e| {
480                            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
481                                .with_message(format!(
482                                    "Failed to find UIDs for receiving object. ERROR: {e}"
483                                ))
484                        })?;
485                    for id in contained_uids {
486                        self.inner.root_version.insert(id, child_version);
487                        if id != child {
488                            let prev = self.inner.wrapped_object_containers.insert(id, child);
489                            debug_assert!(prev.is_none())
490                        }
491                    }
492                    (ObjectResult::Loaded((cache_info, v)), obj_meta)
493                }
494            },
495        ))
496    }

498    pub(super) fn object_exists(
499        &mut self,
500        parent: ObjectID,
501        child: ObjectID,
502    ) -> PartialVMResult<CacheMetadata<bool>> {
503        if let Some(child_object) = self.store.get(&child) {
504            return child_object
505                .value
506                .exists()
507                .map(|exists| (CacheInfo::CachedValue, exists));
508        }
509        let (cache_info, obj_opt) = self.inner.get_or_fetch_object_from_store(parent, child)?;
510        Ok((cache_info, obj_opt.is_some()))
511    }
512
513    pub(super) fn object_exists_and_has_type(
514        &mut self,
515        parent: ObjectID,
516        child: ObjectID,
517        child_move_type: &MoveObjectType,
518    ) -> PartialVMResult<CacheMetadata<bool>> {
519        if let Some(child_object) = self.store.get(&child) {
520            // exists and has same type
521            return Ok((
522                CacheInfo::CachedValue,
523                child_object.value.exists()? && &child_object.ty == child_move_type,
524            ));
525        }
526        let (cache_info, obj_opt) = self.inner.get_or_fetch_object_from_store(parent, child)?;
527        Ok((
528            cache_info,
529            obj_opt.is_some_and(|obj| obj.type_() == child_move_type),
530        ))
531    }
532
533    pub(super) fn get_or_fetch_object(
534        &mut self,
535        parent: ObjectID,
536        child: ObjectID,
537        child_layout: &R::MoveTypeLayout,
538        child_fully_annotated_layout: &A::MoveTypeLayout,
539        child_move_type: MoveObjectType,
540    ) -> PartialVMResult<ObjectResult<CacheMetadata<&mut ChildObject>>> {
541        let store_entries_count = self.store.len() as u64;
542        let (cache_info, child_object) = match self.store.entry(child) {
543            btree_map::Entry::Vacant(e) => {
544                let (cache_info, (ty, value, fingerprint)) = match self.inner.fetch_object_impl(
545                    parent,
546                    child,
547                    child_layout,
548                    child_fully_annotated_layout,
549                    &child_move_type,
550                )? {
551                    ObjectResult::MismatchedType => return Ok(ObjectResult::MismatchedType),
552                    ObjectResult::Loaded(res) => res,
553                };
554
555                if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
556                    self.is_metered,
557                    store_entries_count,
558                    self.inner
559                        .protocol_config
560                        .object_runtime_max_num_store_entries(),
561                    self.inner
562                        .protocol_config
563                        .object_runtime_max_num_store_entries_system_tx(),
564                    self.inner.metrics.excessive_object_runtime_store_entries
565                ) {
566                    return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
567                        .with_message(format!(
568                            "Object runtime store limit ({} entries) reached",
569                            lim
570                        ))
571                        .with_sub_status(
572                            VMMemoryLimitExceededSubStatusCode::OBJECT_RUNTIME_STORE_LIMIT_EXCEEDED
573                                as u64,
574                        ));
575                };
576
577                debug_assert_eq!(
578                    ty, child_move_type,
579                    "Child object type mismatch. \
580                    Expected {child_move_type} but found {ty}"
581                );
582                (
583                    cache_info,
584                    e.insert(ChildObject {
585                        owner: parent,
586                        ty,
587                        value,
588                        fingerprint,
589                    }),
590                )
591            }
592            btree_map::Entry::Occupied(e) => {
593                let child_object = e.into_mut();
594                if child_object.ty != child_move_type {
595                    return Ok(ObjectResult::MismatchedType);
596                }
597                (CacheInfo::CachedValue, child_object)
598            }
599        };
600        Ok(ObjectResult::Loaded((cache_info, child_object)))
601    }

603    pub(super) fn add_object(
604        &mut self,
605        parent: ObjectID,
606        child: ObjectID,
607        child_move_type: MoveObjectType,
608        child_value: Value,
609    ) -> PartialVMResult<()> {
610        if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
611            self.is_metered,
612            self.store.len(),
613            self.inner
614                .protocol_config
615                .object_runtime_max_num_store_entries(),
616            self.inner
617                .protocol_config
618                .object_runtime_max_num_store_entries_system_tx(),
619            self.inner.metrics.excessive_object_runtime_store_entries
620        ) {
621            return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
622                .with_message(format!(
623                    "Object runtime store limit ({} entries) reached",
624                    lim
625                ))
626                .with_sub_status(
627                    VMMemoryLimitExceededSubStatusCode::OBJECT_RUNTIME_STORE_LIMIT_EXCEEDED as u64,
628                ));
629        };
630
631        let (mut value, fingerprint) = if let Some(ChildObject {
632            value, fingerprint, ..
633        }) = self.store.remove(&child)
634        {
635            if value.exists()? {
636                return Err(
637                    PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
638                        .with_message(
639                            "Duplicate addition of a child object. \
640                            The previous value cannot be dropped. Indicates possible duplication \
641                            of objects as an object was fetched more than once from two different \
642                            parents, yet was not removed from one first"
643                                .to_string(),
644                        ),
645                );
646            }
647            (value, fingerprint)
648        } else {
649            let fingerprint = ObjectFingerprint::none(self.inner.protocol_config);
650            (GlobalValue::none(), fingerprint)
651        };
652        if let Err((e, _)) = value.move_to(child_value) {
653            return Err(
654                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
655                    format!("Unable to set value for child {child}, with error {e}",),
656                ),
657            );
658        }
659        let child_object = ChildObject {
660            owner: parent,
661            ty: child_move_type,
662            value,
663            fingerprint,
664        };
665        self.store.insert(child, child_object);
666        Ok(())
667    }

669    pub(super) fn config_setting_unsequenced_read(
670        &mut self,
671        config_id: ObjectID,
672        name_df_id: ObjectID,
673        field_setting_layout: &R::MoveTypeLayout,
674        field_setting_object_type: &MoveObjectType,
675    ) -> PartialVMResult<ObjectResult<Option<Value>>> {
676        let parent = config_id;
677        let child = name_df_id;
678
679        let setting = match self.config_setting_cache.entry(child) {
680            btree_map::Entry::Vacant(e) => {
681                let child_move_type = field_setting_object_type;
682                let inner = &self.inner;
683                let obj_opt =
684                    fetch_child_object_unbounded!(inner, parent, child, SequenceNumber::MAX, true);
685                let Some(move_obj) = obj_opt.as_ref().map(|obj| obj.data.try_as_move().unwrap())
686                else {
687                    return Ok(ObjectResult::Loaded(None));
688                };
689                let Some(value) =
690                    Value::simple_deserialize(move_obj.contents(), field_setting_layout)
691                else {
692                    return Err(
693                        PartialVMError::new(StatusCode::FAILED_TO_DESERIALIZE_RESOURCE)
694                            .with_message(format!(
695                            "Failed to deserialize object {child} with type {field_setting_layout}",
696                        )),
697                    );
698                };
699                e.insert(ConfigSetting {
700                    config: parent,
701                    ty: child_move_type.clone(),
702                    value,
703                })
704            }
705            btree_map::Entry::Occupied(e) => {
706                let setting = e.into_mut();
707                if setting.ty != *field_setting_object_type {
708                    return Ok(ObjectResult::MismatchedType);
709                }
710                if setting.config != parent {
711                    return Err(
712                        PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
713                            .with_message(format!(
714                                "Parent for config setting changed. Potential hash collision?
715                                parent: {parent},
716                                child: {child},
717                                setting_value_object_type: {field_setting_object_type},
718                                setting: {setting:#?}"
719                            )),
720                    );
721                }
722                setting
723            }
724        };
725        let value = setting.value.copy_value().map_err(|e| {
726            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
727                format!("Failed to copy value for config setting {child}, with error {e}",),
728            )
729        })?;
730        Ok(ObjectResult::Loaded(Some(value)))
731    }

733    /// Used by test scenario to insert a config setting into the cache, which replicates the
734    /// behavior of a config already being in the object store.
735    pub(super) fn config_setting_cache_update(
736        &mut self,
737        config_id: ObjectID,
738        name_df_id: ObjectID,
739        setting_value_object_type: MoveObjectType,
740        value: Option<Value>,
741    ) {
742        let child_move_type = setting_value_object_type;
743        match value {
744            Some(value) => {
745                let setting = ConfigSetting {
746                    config: config_id,
747                    ty: child_move_type,
748                    value,
749                };
750                self.config_setting_cache.insert(name_df_id, setting);
751            }
752            None => {
753                self.config_setting_cache.remove(&name_df_id);
754            }
755        }
756    }

766    // retrieve the `Op` effects for the child objects
767    pub(super) fn take_effects(&mut self) -> PartialVMResult<ChildObjectEffects> {
768        if self.inner.protocol_config.minimize_child_object_mutations() {
769            let v1_effects = std::mem::take(&mut self.store)
770                .into_iter()
771                .map(|(id, child_object)| {
772                    let ChildObject {
773                        owner,
774                        ty,
775                        value,
776                        fingerprint,
777                    } = child_object;
778                    #[cfg(debug_assertions)]
779                    let dirty_flag_mutated = value.is_mutated();
780                    let final_value = value.into_value();
781                    let object_changed =
782                        fingerprint.object_has_changed(&owner, &ty, &final_value)?;
783                    // The old dirty flag was pessimistic in its tracking of mutations, meaning
784                    // it would mark mutations even if the value remained the same.
785                    // This means that if the object changed, the dirty flag must have been marked.
786                    // However, we can't guarantee the opposite.
787                    // object changed ==> dirty flag mutated
788                    #[cfg(debug_assertions)]
789                    debug_assert!(!object_changed || dirty_flag_mutated);
790                    let child_effect = ChildObjectEffectV1 {
791                        owner,
792                        ty,
793                        final_value,
794                        object_changed,
795                    };
796                    Ok((id, child_effect))
797                })
798                .collect::<PartialVMResult<_>>()?;
799            Ok(ChildObjectEffects::V1(v1_effects))
800        } else {
801            let v0_effects = std::mem::take(&mut self.store)
802                .into_iter()
803                .filter_map(|(id, child_object)| {
804                    let ChildObject {
805                        owner,
806                        ty,
807                        value,
808                        fingerprint: _fingerprint,
809                    } = child_object;
810                    let effect = value.into_effect()?;
811                    // should be disabled if the feature is disabled
812                    #[cfg(debug_assertions)]
813                    debug_assert!(_fingerprint.is_disabled());
814                    let child_effect = ChildObjectEffectV0 { owner, ty, effect };
815                    Some((id, child_effect))
816                })
817                .collect();
818            Ok(ChildObjectEffects::V0(v0_effects))
819        }
820    }

822    pub(super) fn all_active_objects(&self) -> impl Iterator<Item = ActiveChildObject<'_>> {
823        self.store.iter().map(|(id, child_object)| {
824            let copied_child_value = if child_object.value.exists().unwrap() {
825                Some(
826                    child_object
827                        .value
828                        .borrow_global()
829                        .unwrap()
830                        .value_as::<StructRef>()
831                        .unwrap()
832                        .read_ref()
833                        .unwrap(),
834                )
835            } else {
836                None
837            };
838            ActiveChildObject {
839                id,
840                owner: &child_object.owner,
841                ty: &child_object.ty,
842                copied_value: copied_child_value,
843            }
844        })
845    }
