17/***************************************************************************************************
18 * native fun is_feature_enabled
19 *
20 * Implementation of the Move native function `protocol_config::is_feature_enabled(
21 *      feature_flag_name: vector<u8>): bool`
22 *
23 * Checks if a protocol feature flag is enabled in the current protocol version.
24 *
25 * Gas cost: 0 (zero cost for framework-internal use)
26 **************************************************************************************************/
27pub fn is_feature_enabled(
28    context: &mut NativeContext,
29    ty_args: Vec<Type>,
30    mut args: VecDeque<Value>,
31) -> PartialVMResult<NativeResult> {
32    debug_assert!(ty_args.is_empty());
33    debug_assert!(args.len() == 1);
34
35    let feature_flag_name_bytes = pop_arg!(args, Vector);
36    let bytes = feature_flag_name_bytes.to_vec_u8()?;
37
38    let protocol_config = &get_extension!(context, ObjectRuntime)?.protocol_config;
39
40    let is_enabled = match String::from_utf8(bytes.to_vec()) {
41        Ok(s) => {
42            // Use the auto-generated lookup_feature method to find the feature flag
43            match protocol_config.lookup_feature(s) {
44                Some(value) => value,
45                None => {
46                    debug_assert!(false);
47                    // We don't distinguish between feature flags that are not present and feature flags that are present but disabled.
48                    // This is to handle the case where we accidentally shipped a framework upgrade that check a feature flag that is not present
49                    // in the binary yet.
50                    false
51                }
52            }
53        }
54        Err(_) => {
55            debug_assert!(false);
56            // Invalid UTF feature flags are treated as disabled feature flags.
57            false
58        }
59    };
60
61    Ok(NativeResult::ok(
62        context.gas_used(),
63        smallvec![Value::bool(is_enabled)],
64    ))
65}
