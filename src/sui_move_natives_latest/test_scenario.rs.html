61/// An in-memory test store is a thin wrapper around the in-memory storage in a mutex. The mutex
62/// allows this to be used by both the object runtime (for reading) and the test scenario (for
63/// writing) while hiding mutability.
64#[derive(Tid)]
65pub struct InMemoryTestStore(pub RefCell<InMemoryStorage>);
66impl<'a> NativeExtensionMarker<'a> for &'a InMemoryTestStore {}

68impl ChildObjectResolver for InMemoryTestStore {
69    fn read_child_object(
70        &self,
71        parent: &ObjectID,
72        child: &ObjectID,
73        child_version_upper_bound: SequenceNumber,
74    ) -> sui_types::error::SuiResult<Option<Object>> {
75        self.0
76            .borrow()
77            .read_child_object(parent, child, child_version_upper_bound)
78    }
79
80    fn get_object_received_at_version(
81        &self,
82        owner: &ObjectID,
83        receiving_object_id: &ObjectID,
84        receive_object_at_version: SequenceNumber,
85        epoch_id: sui_types::committee::EpochId,
86    ) -> sui_types::error::SuiResult<Option<Object>> {
87        self.0.borrow().get_object_received_at_version(
88            owner,
89            receiving_object_id,
90            receive_object_at_version,
91            epoch_id,
92        )
93    }
94}

96// This function updates the inventories based on the transfers and deletes that occurred in the
97// transaction
98// native fun end_transaction(): TransactionResult;
99pub fn end_transaction(
100    context: &mut NativeContext,
101    ty_args: Vec<Type>,
102    args: VecDeque<Value>,
103) -> PartialVMResult<NativeResult> {
104    safe_assert!(ty_args.is_empty());
105    safe_assert!(args.is_empty());
106    let object_runtime_ref: &mut ObjectRuntime = get_extension_mut!(context)?;
107    let taken_shared_or_imm: BTreeMap<_, _> = object_runtime_ref
108        .test_inventories
109        .taken
110        .iter()
111        .filter(|(_id, owner)| matches!(owner, Owner::Shared { .. } | Owner::Immutable))
112        .map(|(id, owner)| (*id, owner.clone()))
113        .collect();
114    // set to true if a shared or imm object was:
115    // - transferred in a way that changes it from its original shared/imm state
116    // - wraps the object
117    // if true, we will "abort"
118    let mut incorrect_shared_or_imm_handling = false;
119
120    // Handle the allocated tickets:
121    // * Remove all allocated_tickets in the test inventories.
122    // * For each allocated ticket, if the ticket's object ID is loaded, move it to `received`.
123    // * Otherwise re-insert the allocated ticket into the objects inventory, and mark it to be
124    //   removed from the backing storage (deferred due to needing to have access to `context` which
125    //   has outstanding references at this point).
126    let allocated_tickets =
127        std::mem::take(&mut object_runtime_ref.test_inventories.allocated_tickets);
128    let mut received = BTreeMap::new();
129    let mut unreceived = BTreeSet::new();
130    let loaded_runtime_objects = object_runtime_ref.loaded_runtime_objects();
131    for (id, (metadata, value)) in allocated_tickets {
132        if loaded_runtime_objects.contains_key(&id) {
133            received.insert(id, metadata);
134        } else {
135            unreceived.insert(id);
136            // This must be untouched since the allocated ticket is still live, so ok to re-insert.
137            object_runtime_ref
138                .test_inventories
139                .objects
140                .insert(id, value);
141        }
142    }
143
144    let object_runtime_state = object_runtime_ref.take_state();
145    // Determine writes and deletes
146    // We pass the received objects since they should be viewed as "loaded" for the purposes of
147    // calculating the effects of the transaction.
148    let results = object_runtime_state.finish(received, ChildObjectEffects::new());
149    let RuntimeResults {
150        writes,
151        user_events,
152        loaded_child_objects: _,
153        created_object_ids,
154        deleted_object_ids,
155        accumulator_events: _,
156        settlement_input_sui: _,
157        settlement_output_sui: _,
158    } = match results {
159        Ok(res) => res,
160        Err(_) => {
161            return Ok(NativeResult::err(
162                legacy_test_cost(),
163                E_COULD_NOT_GENERATE_EFFECTS,
164            ));
165        }
166    };
167    let object_runtime_ref: &mut ObjectRuntime = get_extension_mut!(context)?;
168    let all_active_child_objects_with_values = object_runtime_ref
169        .all_active_child_objects()
170        .filter(|child| child.copied_value.is_some())
171        .map(|child| *child.id)
172        .collect::<BTreeSet<_>>();
173    let inventories = &mut object_runtime_ref.test_inventories;
174    let mut new_object_values = IndexMap::new();
175    let mut transferred = vec![];
176    // cleanup inventories
177    // we will remove all changed objects
178    // - deleted objects need to be removed to mark deletions
179    // - written objects are removed and later replaced to mark new values and new owners
180    // - child objects will not be reflected in transfers, but need to be no longer retrievable
181    for id in deleted_object_ids
182        .iter()
183        .chain(writes.keys())
184        .chain(&all_active_child_objects_with_values)
185    {
186        for addr_inventory in inventories.address_inventories.values_mut() {
187            for s in addr_inventory.values_mut() {
188                s.shift_remove(id);
189            }
190        }
191        for s in &mut inventories.shared_inventory.values_mut() {
192            s.shift_remove(id);
193        }
194        for s in &mut inventories.immutable_inventory.values_mut() {
195            s.shift_remove(id);
196        }
197        inventories.taken.remove(id);
198    }
199
200    // handle transfers, inserting transferred/written objects into their respective inventory
201    let mut created = vec![];
202    let mut written = vec![];
203    for (id, (owner, ty, value)) in writes {
204        // write configs to cache
205        new_object_values.insert(id, (ty.clone(), value.copy_value()));
206        transferred.push((id, owner.clone()));
207        incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
208            || taken_shared_or_imm
209                .get(&id)
210                .map(|shared_or_imm_owner| shared_or_imm_owner != &owner)
211                .unwrap_or(/* not incorrect */ false);
212        if created_object_ids.contains(&id) {
213            created.push(id);
214        } else {
215            written.push(id);
216        }
217        match owner {
218            Owner::AddressOwner(a) => {
219                inventories
220                    .address_inventories
221                    .entry(a)
222                    .or_default()
223                    .entry(ty)
224                    .or_default()
225                    .insert(id);
226            }
227            Owner::ObjectOwner(_) => (),
228            Owner::Shared { .. } => {
229                inventories
230                    .shared_inventory
231                    .entry(ty)
232                    .or_default()
233                    .insert(id);
234            }
235            Owner::Immutable => {
236                inventories
237                    .immutable_inventory
238                    .entry(ty)
239                    .or_default()
240                    .insert(id);
241            }
242            Owner::ConsensusAddressOwner { owner, .. } => {
243                inventories
244                    .address_inventories
245                    .entry(owner)
246                    .or_default()
247                    .entry(ty)
248                    .or_default()
249                    .insert(id);
250            }
251            Owner::Party { .. } => {
252                // TODO(Party WIP)
253                todo!("Party WIP")
254            }
255        }
256    }
257
258    // For any unused allocated tickets, remove them from the store.
259    let store: &&InMemoryTestStore = get_extension!(context)?;
260    for id in unreceived {
261        if store.0.borrow_mut().remove_object(id).is_none() {
262            return Ok(NativeResult::err(
263                context.gas_used(),
264                E_UNABLE_TO_DEALLOCATE_RECEIVING_TICKET,
265            ));
266        }
267    }
268
269    // deletions already handled above, but we drop the delete kind for the effects
270    let mut deleted = vec![];
271    for id in deleted_object_ids {
272        // Mark as "incorrect" if a imm object was deleted. Allow shared objects to be deleted though.
273        incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
274            || taken_shared_or_imm
275                .get(&id)
276                .is_some_and(|owner| matches!(owner, Owner::Immutable));
277        deleted.push(id);
278    }
279    // find all wrapped objects
280    let mut all_wrapped = BTreeSet::new();
281    let object_runtime_ref: &ObjectRuntime = get_extension!(context)?;
282    find_all_wrapped_objects(
283        context,
284        &mut all_wrapped,
285        new_object_values
286            .iter()
287            .map(|(id, (ty, value))| (id, ty, value)),
288    )?;
289    find_all_wrapped_objects(
290        context,
291        &mut all_wrapped,
292        object_runtime_ref
293            .all_active_child_objects()
294            .filter_map(|child| Some((child.id, child.ty, child.copied_value?))),
295    )?;
296    // mark as "incorrect" if a shared/imm object was wrapped or is a child object
297    incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
298        || taken_shared_or_imm.keys().any(|id| {
299            all_wrapped.contains(id) || all_active_child_objects_with_values.contains(id)
300        });
301    // if incorrect handling, return with an 'abort'
302    if incorrect_shared_or_imm_handling {
303        return Ok(NativeResult::err(
304            legacy_test_cost(),
305            E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
306        ));
307    }
308
309    // mark all wrapped as deleted
310    for wrapped in all_wrapped {
311        deleted.push(wrapped)
312    }
313
314    // new input objects are remaining taken objects not written/deleted
315    let object_runtime_ref: &mut ObjectRuntime = get_extension_mut!(context)?;
316    let mut config_settings = vec![];
317    for child in object_runtime_ref.all_active_child_objects() {
318        let s: StructTag = child.ty.clone().into();
319        let is_setting = DynamicFieldInfo::is_dynamic_field(&s)
320            && matches!(&s.type_params[1], TypeTag::Struct(s) if config::is_setting(s));
321        if is_setting {
322            config_settings.push((
323                *child.owner,
324                *child.id,
325                child.ty.clone(),
326                child.copied_value,
327            ));
328        }
329    }
330    for (config, setting, ty, value) in config_settings {
331        object_runtime_ref.config_setting_cache_update(config, setting, ty, value)
332    }
333    object_runtime_ref.state.input_objects = object_runtime_ref
334        .test_inventories
335        .taken
336        .iter()
337        .map(|(id, owner)| (*id, owner.clone()))
338        .collect::<BTreeMap<_, _>>();
339    // update inventories
340    // check for bad updates to immutable values
341    for (id, (ty, value)) in new_object_values {
342        debug_assert!(!all_active_child_objects_with_values.contains(&id));
343        if let Some(prev_value) = object_runtime_ref
344            .test_inventories
345            .taken_immutable_values
346            .get(&ty)
347            .and_then(|values| values.get(&id))
348            && !value.equals(prev_value)?
349        {
350            return Ok(NativeResult::err(
351                legacy_test_cost(),
352                E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
353            ));
354        }
355        object_runtime_ref
356            .test_inventories
357            .objects
358            .insert(id, value);
359    }
360    // remove deleted
361    for id in &deleted {
362        object_runtime_ref.test_inventories.objects.remove(id);
363    }
364    // remove active child objects
365    for id in all_active_child_objects_with_values {
366        object_runtime_ref.test_inventories.objects.remove(&id);
367    }
368
369    let effects = transaction_effects(
370        created,
371        written,
372        deleted,
373        transferred,
374        user_events.len() as u64,
375        // TODO: do we need accumulator events here?
376    )?;
377    Ok(NativeResult::ok(legacy_test_cost(), smallvec![effects]))
378}

380// native fun take_from_address_by_id<T: key>(account: address, id: ID): T;
381pub fn take_from_address_by_id(
382    context: &mut NativeContext,
383    ty_args: Vec<Type>,
384    mut args: VecDeque<Value>,
385) -> PartialVMResult<NativeResult> {
386    let specified_ty = get_specified_ty(ty_args)?;
387    let id = pop_id(&mut args)?;
388    let account: SuiAddress = pop_arg!(args, AccountAddress).into();
389    pop_arg!(args, StructRef);
390    safe_assert!(args.is_empty());
391    let specified_obj_ty = object_type_of_type(context, &specified_ty)?;
392    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
393    let inventories = &mut object_runtime.test_inventories;
394    let res = take_from_inventory(
395        |x| {
396            inventories
397                .address_inventories
398                .get(&account)
399                .and_then(|inv| inv.get(&specified_obj_ty))
400                .map(|s| s.contains(x))
401                .unwrap_or(false)
402        },
403        &inventories.objects,
404        &mut inventories.taken,
405        &mut object_runtime.state.input_objects,
406        id,
407        Owner::AddressOwner(account),
408    );
409    Ok(match res {
410        Ok(value) => NativeResult::ok(legacy_test_cost(), smallvec![value]),
411        Err(native_err) => native_err,
412    })
413}

415// native fun ids_for_address<T: key>(account: address): vector<ID>;
416pub fn ids_for_address(
417    context: &mut NativeContext,
418    ty_args: Vec<Type>,
419    mut args: VecDeque<Value>,
420) -> PartialVMResult<NativeResult> {
421    let specified_ty = get_specified_ty(ty_args)?;
422    let account: SuiAddress = pop_arg!(args, AccountAddress).into();
423    safe_assert!(args.is_empty());
424    let specified_obj_ty = object_type_of_type(context, &specified_ty)?;
425    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
426    let inventories = &mut object_runtime.test_inventories;
427    let ids = inventories
428        .address_inventories
429        .get(&account)
430        .and_then(|inv| inv.get(&specified_obj_ty))
431        .map(|s| s.iter().map(|id| pack_id(*id)).collect::<Vec<Value>>())
432        .unwrap_or_default();
433    let ids_vector = safe_unwrap!(Vector::pack(VectorSpecialization::Container, ids));
434    Ok(NativeResult::ok(legacy_test_cost(), smallvec![ids_vector]))
435}

629    safe_assert!(ty_args.is_empty());
630    let id = pop_id(&mut args)?;
631    safe_assert!(args.is_empty());
632    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
633    let inventories = &mut object_runtime.test_inventories;
634    let was_taken = inventories
635        .taken
636        .get(&id)
637        .map(|owner| matches!(owner, Owner::Shared { .. }))
638        .unwrap_or(false);
639    Ok(NativeResult::ok(
640        legacy_test_cost(),
641        smallvec![Value::bool(was_taken)],
642    ))
643}

645pub fn allocate_receiving_ticket_for_object(
646    context: &mut NativeContext,
647    ty_args: Vec<Type>,
648    mut args: VecDeque<Value>,
649) -> PartialVMResult<NativeResult> {
650    let ty = get_specified_ty(ty_args)?;
651    let id = pop_id(&mut args)?;
652
653    let abilities = context.type_to_abilities(&ty)?;
654    let Some((tag, layout, _)) = get_tag_and_layouts(context, &ty)? else {
655        return Ok(NativeResult::err(
656            context.gas_used(),
657            E_UNABLE_TO_ALLOCATE_RECEIVING_TICKET,
658        ));
659    };
660    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
661    let object_version = SequenceNumber::new();
662    let inventories = &mut object_runtime.test_inventories;
663    if inventories.allocated_tickets.contains_key(&id) {
664        return Ok(NativeResult::err(
665            context.gas_used(),
666            E_RECEIVING_TICKET_ALREADY_ALLOCATED,
667        ));
668    }
669
670    let obj_value = safe_unwrap!(inventories.objects.remove(&id));
671    let Some(bytes) = obj_value.typed_serialize(&layout) else {
672        return Ok(NativeResult::err(
673            context.gas_used(),
674            E_UNABLE_TO_ALLOCATE_RECEIVING_TICKET,
675        ));
676    };
677    let has_public_transfer = abilities.has_store();
678    let move_object = safe_unwrap!(unsafe {
679        MoveObject::new_from_execution_with_limit(
680            tag.into(),
681            has_public_transfer,
682            object_version,
683            bytes,
684            250 * 1024,
685        )
686    });
687
688    let Some((owner, _)) = inventories
689        .address_inventories
690        .iter()
691        .find(|(_addr, objs)| objs.iter().any(|(_, ids)| ids.contains(&id)))
692    else {
693        return Ok(NativeResult::err(
694            context.gas_used(),
695            E_OBJECT_NOT_FOUND_CODE,
696        ));
697    };
698
699    inventories.allocated_tickets.insert(
700        id,
701        (
702            DynamicallyLoadedObjectMetadata {
703                version: SequenceNumber::new(),
704                digest: ObjectDigest::MIN,
705                owner: Owner::AddressOwner(*owner),
706                storage_rebate: 0,
707                previous_transaction: TransactionDigest::default(),
708            },
709            obj_value,
710        ),
711    );
712
713    let object = Object::new_move(
714        move_object,
715        Owner::AddressOwner(*owner),
716        TransactionDigest::default(),
717    );
718
719    // NB: Must be a `&&` reference since the extension stores a static ref to the object storage.
720    let store: &&InMemoryTestStore = get_extension!(context)?;
721    store.0.borrow_mut().insert_object(object);
722
723    Ok(NativeResult::ok(
724        legacy_test_cost(),
725        smallvec![Value::u64(object_version.value())],
726    ))
727}
728
729pub fn deallocate_receiving_ticket_for_object(
730    context: &mut NativeContext,
731    _ty_args: Vec<Type>,
732    mut args: VecDeque<Value>,
733) -> PartialVMResult<NativeResult> {
734    let id = pop_id(&mut args)?;
735
736    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
737    let inventories = &mut object_runtime.test_inventories;
738    // Deallocate the ticket -- we should never hit this scenario
739    let Some((_, value)) = inventories.allocated_tickets.remove(&id) else {
740        return Ok(NativeResult::err(
741            context.gas_used(),
742            E_UNABLE_TO_DEALLOCATE_RECEIVING_TICKET,
743        ));
744    };
745
746    // Insert the object value that we saved from earlier and put it back into the object set.
747    // This is fine since it can't have been touched.
748    inventories.objects.insert(id, value);
749
750    // Remove the object from storage. We should never hit this scenario either.
751    let store: &&InMemoryTestStore = get_extension!(context)?;
752    if store.0.borrow_mut().remove_object(id).is_none() {
753        return Ok(NativeResult::err(
754            context.gas_used(),
755            E_UNABLE_TO_DEALLOCATE_RECEIVING_TICKET,
756        ));
757    };
758
759    Ok(NativeResult::ok(legacy_test_cost(), smallvec![]))
760}

764fn take_from_inventory(
765    is_in_inventory: impl FnOnce(&ObjectID) -> bool,
766    objects: &BTreeMap<ObjectID, Value>,
767    taken: &mut BTreeMap<ObjectID, Owner>,
768    input_objects: &mut BTreeMap<ObjectID, Owner>,
769    id: ObjectID,
770    owner: Owner,
771) -> Result<Value, NativeResult> {
772    let is_taken = taken.contains_key(&id);
773    let obj = match objects.get(&id) {
774        Some(obj) if !is_taken && is_in_inventory(&id) => obj,
775        _ => {
776            return Err(NativeResult::err(
777                legacy_test_cost(),
778                E_OBJECT_NOT_FOUND_CODE,
779            ));
780        }
781    };
782    taken.insert(id, owner.clone());
783    input_objects.insert(id, owner);
784    Ok(obj.copy_value())
785}

859fn transaction_effects(
860    created: impl IntoIterator<Item = impl Into<AccountAddress>>,
861    written: impl IntoIterator<Item = impl Into<AccountAddress>>,
862    deleted: impl IntoIterator<Item = impl Into<AccountAddress>>,
863    transferred: impl IntoIterator<Item = (ObjectID, Owner)>,
864    num_events: u64,
865) -> PartialVMResult<Value> {
866    let mut transferred_to_account = vec![];
867    let mut transferred_to_object = vec![];
868    let mut shared = vec![];
869    let mut frozen = vec![];
870    for (id, owner) in transferred {
871        match owner {
872            Owner::AddressOwner(a) => {
873                transferred_to_account.push((pack_id(id), Value::address(a.into())))
874            }
875            Owner::ObjectOwner(o) => transferred_to_object.push((pack_id(id), pack_id(o))),
876            Owner::Shared { .. } => shared.push(id),
877            Owner::Immutable => frozen.push(id),
878            Owner::ConsensusAddressOwner { owner, .. } => {
879                transferred_to_account.push((pack_id(id), Value::address(owner.into())))
880            }
881            Owner::Party { .. } => {
882                // TODO(Party WIP)
883                todo!("Party WIP")
884            }
885        }
886    }
887
888    let created_field = pack_ids(created)?;
889    let written_field = pack_ids(written)?;
890    let deleted_field = pack_ids(deleted)?;
891    let transferred_to_account_field = pack_vec_map(transferred_to_account)?;
892    let transferred_to_object_field = pack_vec_map(transferred_to_object)?;
893    let shared_field = pack_ids(shared)?;
894    let frozen_field = pack_ids(frozen)?;
895    let num_events_field = Value::u64(num_events);
896    Ok(Value::struct_(values::Struct::pack(vec![
897        created_field,
898        written_field,
899        deleted_field,
900        transferred_to_account_field,
901        transferred_to_object_field,
902        shared_field,
903        frozen_field,
904        num_events_field,
905    ])))
906}

928fn find_all_wrapped_objects<'a, 'i>(
929    context: &NativeContext,
930    ids: &'i mut BTreeSet<ObjectID>,
931    new_object_values: impl IntoIterator<Item = (&'a ObjectID, &'a MoveObjectType, impl Borrow<Value>)>,
932) -> PartialVMResult<()> {
933    #[derive(Copy, Clone)]
934    enum LookingFor {
935        Wrapped,
936        Uid,
937        Address,
938    }
939
940    struct Traversal<'i, 'u> {
941        state: LookingFor,
942        ids: &'i mut BTreeSet<ObjectID>,
943        uid: &'u MoveStructLayout,
944    }
945
946    impl<'b, 'l> AV::Traversal<'b, 'l> for Traversal<'_, '_> {
947        type Error = AV::Error;
948
949        fn traverse_struct(
950            &mut self,
951            driver: &mut AV::StructDriver<'_, 'b, 'l>,
952        ) -> Result<(), Self::Error> {
953            match self.state {
954                // We're at the top-level of the traversal, looking for an object to recurse into.
955                // We can unconditionally switch to looking for UID fields at the level below,
956                // because we know that all the top-level values are objects.
957                LookingFor::Wrapped => {
958                    while driver
959                        .next_field(&mut Traversal {
960                            state: LookingFor::Uid,
961                            ids: self.ids,
962                            uid: self.uid,
963                        })?
964                        .is_some()
965                    {}
966                }
967
968                // We are looking for UID fields. If we find one (which we confirm by checking its
969                // layout), switch to looking for addresses in its sub-structure.
970                LookingFor::Uid => {
971                    while let Some(MoveFieldLayout { name: _, layout }) = driver.peek_field() {
972                        if matches!(layout, MoveTypeLayout::Struct(s) if s.as_ref() == self.uid) {
973                            driver.next_field(&mut Traversal {
974                                state: LookingFor::Address,
975                                ids: self.ids,
976                                uid: self.uid,
977                            })?;
978                        } else {
979                            driver.next_field(self)?;
980                        }
981                    }
982                }
983
984                // When looking for addresses, recurse through structs, as the address is nested
985                // within the UID.
986                LookingFor::Address => while driver.next_field(self)?.is_some() {},
987            }
988
989            Ok(())
990        }
991
992        fn traverse_address(
993            &mut self,
994            _: &AV::ValueDriver<'_, 'b, 'l>,
995            address: AccountAddress,
996        ) -> Result<(), Self::Error> {
997            // If we're looking for addresses, and we found one, then save it.
998            if matches!(self.state, LookingFor::Address) {
999                self.ids.insert(address.into());
1000            }
1001            Ok(())
1002        }
1003    }
1004
1005    let uid = UID::layout();
1006    for (_id, ty, value) in new_object_values {
1007        let type_tag = TypeTag::from(ty.clone());
1008        // NB: We can get the layout from the VM's cache since the types and modules
1009        // associated with all of these types must be in the type/module cache in the VM -- THIS IS
1010        // BECAUSE WE ARE IN TEST SCENARIO ONLY AND THIS MAY NOT GENERALLY HOLD IN A
1011        // MULTI-TRANSACTION SETTING.
1012        let Some(layout) = context.type_tag_to_type_layout(&type_tag) else {
1013            debug_assert!(false);
1014            continue;
1015        };
1016
1017        let Some(annotated_layout) = context.type_tag_to_annotated_type_layout(&type_tag) else {
1018            debug_assert!(false);
1019            continue;
1020        };
1021
1022        let blob = safe_unwrap!(value.borrow().typed_serialize(&layout));
1023        safe_unwrap!(MoveValue::visit_deserialize(
1024            &blob,
1025            &annotated_layout,
1026            &mut Traversal {
1027                state: LookingFor::Wrapped,
1028                ids,
1029                uid: &uid,
1030            },
1031        ));
1032    }
1033    Ok(())
1034}
