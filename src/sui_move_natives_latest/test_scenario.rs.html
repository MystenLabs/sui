53/// An in-memory test store is a thin wrapper around the in-memory storage in a mutex. The mutex
54/// allows this to be used by both the object runtime (for reading) and the test scenario (for
55/// writing) while hiding mutability.
56#[derive(Tid)]
57pub struct InMemoryTestStore(pub RefCell<InMemoryStorage>);
58impl<'a> NativeExtensionMarker<'a> for &'a InMemoryTestStore {}
59
60impl ChildObjectResolver for InMemoryTestStore {
61    fn read_child_object(
62        &self,
63        parent: &ObjectID,
64        child: &ObjectID,
65        child_version_upper_bound: SequenceNumber,
66    ) -> sui_types::error::SuiResult<Option<Object>> {
67        self.0
68            .borrow()
69            .read_child_object(parent, child, child_version_upper_bound)
70    }
71
72    fn get_object_received_at_version(
73        &self,
74        owner: &ObjectID,
75        receiving_object_id: &ObjectID,
76        receive_object_at_version: SequenceNumber,
77        epoch_id: sui_types::committee::EpochId,
78    ) -> sui_types::error::SuiResult<Option<Object>> {
79        self.0.borrow().get_object_received_at_version(
80            owner,
81            receiving_object_id,
82            receive_object_at_version,
83            epoch_id,
84        )
85    }
86}

88// This function updates the inventories based on the transfers and deletes that occurred in the
89// transaction
90// native fun end_transaction(): TransactionResult;
91pub fn end_transaction(
92    context: &mut NativeContext,
93    ty_args: Vec<Type>,
94    args: VecDeque<Value>,
95) -> PartialVMResult<NativeResult> {
96    assert!(ty_args.is_empty());
97    assert!(args.is_empty());
98    let object_runtime_ref: &mut ObjectRuntime = get_extension_mut!(context)?;
99    let taken_shared_or_imm: BTreeMap<_, _> = object_runtime_ref
100        .test_inventories
101        .taken
102        .iter()
103        .filter(|(_id, owner)| matches!(owner, Owner::Shared { .. } | Owner::Immutable))
104        .map(|(id, owner)| (*id, owner.clone()))
105        .collect();
106    // set to true if a shared or imm object was:
107    // - transferred in a way that changes it from its original shared/imm state
108    // - wraps the object
109    // if true, we will "abort"
110    let mut incorrect_shared_or_imm_handling = false;
111
112    // Handle the allocated tickets:
113    // * Remove all allocated_tickets in the test inventories.
114    // * For each allocated ticket, if the ticket's object ID is loaded, move it to `received`.
115    // * Otherwise re-insert the allocated ticket into the objects inventory, and mark it to be
116    //   removed from the backing storage (deferred due to needing to have access to `context` which
117    //   has outstanding references at this point).
118    let allocated_tickets =
119        std::mem::take(&mut object_runtime_ref.test_inventories.allocated_tickets);
120    let mut received = BTreeMap::new();
121    let mut unreceived = BTreeSet::new();
122    let loaded_runtime_objects = object_runtime_ref.loaded_runtime_objects();
123    for (id, (metadata, value)) in allocated_tickets {
124        if loaded_runtime_objects.contains_key(&id) {
125            received.insert(id, metadata);
126        } else {
127            unreceived.insert(id);
128            // This must be untouched since the allocated ticket is still live, so ok to re-insert.
129            object_runtime_ref
130                .test_inventories
131                .objects
132                .insert(id, value);
133        }
134    }
135
136    let object_runtime_state = object_runtime_ref.take_state();
137    // Determine writes and deletes
138    // We pass the received objects since they should be viewed as "loaded" for the purposes of
139    // calculating the effects of the transaction.
140    let results = object_runtime_state.finish(received, ChildObjectEffects::empty());
141    let RuntimeResults {
142        writes,
143        user_events,
144        loaded_child_objects: _,
145        created_object_ids,
146        deleted_object_ids,
147        accumulator_events: _,
148        settlement_input_sui: _,
149        settlement_output_sui: _,
150    } = match results {
151        Ok(res) => res,
152        Err(_) => {
153            return Ok(NativeResult::err(
154                legacy_test_cost(),
155                E_COULD_NOT_GENERATE_EFFECTS,
156            ));
157        }
158    };
159    let object_runtime_ref: &mut ObjectRuntime = get_extension_mut!(context)?;
160    let all_active_child_objects_with_values = object_runtime_ref
161        .all_active_child_objects()
162        .filter(|child| child.copied_value.is_some())
163        .map(|child| *child.id)
164        .collect::<BTreeSet<_>>();
165    let inventories = &mut object_runtime_ref.test_inventories;
166    let mut new_object_values = IndexMap::new();
167    let mut transferred = vec![];
168    // cleanup inventories
169    // we will remove all changed objects
170    // - deleted objects need to be removed to mark deletions
171    // - written objects are removed and later replaced to mark new values and new owners
172    // - child objects will not be reflected in transfers, but need to be no longer retrievable
173    for id in deleted_object_ids
174        .iter()
175        .chain(writes.keys())
176        .chain(&all_active_child_objects_with_values)
177    {
178        for addr_inventory in inventories.address_inventories.values_mut() {
179            for s in addr_inventory.values_mut() {
180                s.shift_remove(id);
181            }
182        }
183        for s in &mut inventories.shared_inventory.values_mut() {
184            s.shift_remove(id);
185        }
186        for s in &mut inventories.immutable_inventory.values_mut() {
187            s.shift_remove(id);
188        }
189        inventories.taken.remove(id);
190    }
191
192    // handle transfers, inserting transferred/written objects into their respective inventory
193    let mut created = vec![];
194    let mut written = vec![];
195    for (id, (owner, ty, value)) in writes {
196        // write configs to cache
197        new_object_values.insert(id, (ty.clone(), value.copy_value().unwrap()));
198        transferred.push((id, owner.clone()));
199        incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
200            || taken_shared_or_imm
201                .get(&id)
202                .map(|shared_or_imm_owner| shared_or_imm_owner != &owner)
203                .unwrap_or(/* not incorrect */ false);
204        if created_object_ids.contains(&id) {
205            created.push(id);
206        } else {
207            written.push(id);
208        }
209        match owner {
210            Owner::AddressOwner(a) => {
211                inventories
212                    .address_inventories
213                    .entry(a)
214                    .or_default()
215                    .entry(ty)
216                    .or_default()
217                    .insert(id);
218            }
219            Owner::ObjectOwner(_) => (),
220            Owner::Shared { .. } => {
221                inventories
222                    .shared_inventory
223                    .entry(ty)
224                    .or_default()
225                    .insert(id);
226            }
227            Owner::Immutable => {
228                inventories
229                    .immutable_inventory
230                    .entry(ty)
231                    .or_default()
232                    .insert(id);
233            }
234            Owner::ConsensusAddressOwner { owner, .. } => {
235                inventories
236                    .address_inventories
237                    .entry(owner)
238                    .or_default()
239                    .entry(ty)
240                    .or_default()
241                    .insert(id);
242            }
243        }
244    }
245
246    // For any unused allocated tickets, remove them from the store.
247    let store: &&InMemoryTestStore = get_extension!(context)?;
248    for id in unreceived {
249        if store.0.borrow_mut().remove_object(id).is_none() {
250            return Ok(NativeResult::err(
251                context.gas_used(),
252                E_UNABLE_TO_DEALLOCATE_RECEIVING_TICKET,
253            ));
254        }
255    }
256
257    // deletions already handled above, but we drop the delete kind for the effects
258    let mut deleted = vec![];
259    for id in deleted_object_ids {
260        // Mark as "incorrect" if a imm object was deleted. Allow shared objects to be deleted though.
261        incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
262            || taken_shared_or_imm
263                .get(&id)
264                .is_some_and(|owner| matches!(owner, Owner::Immutable));
265        deleted.push(id);
266    }
267    // find all wrapped objects
268    let mut all_wrapped = BTreeSet::new();
269    let object_runtime_ref: &ObjectRuntime = get_extension!(context)?;
270    find_all_wrapped_objects(
271        context,
272        &mut all_wrapped,
273        new_object_values
274            .iter()
275            .map(|(id, (ty, value))| (id, ty, value)),
276    );
277    find_all_wrapped_objects(
278        context,
279        &mut all_wrapped,
280        object_runtime_ref
281            .all_active_child_objects()
282            .filter_map(|child| Some((child.id, child.ty, child.copied_value?))),
283    );
284    // mark as "incorrect" if a shared/imm object was wrapped or is a child object
285    incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
286        || taken_shared_or_imm.keys().any(|id| {
287            all_wrapped.contains(id) || all_active_child_objects_with_values.contains(id)
288        });
289    // if incorrect handling, return with an 'abort'
290    if incorrect_shared_or_imm_handling {
291        return Ok(NativeResult::err(
292            legacy_test_cost(),
293            E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
294        ));
295    }
296
297    // mark all wrapped as deleted
298    for wrapped in all_wrapped {
299        deleted.push(wrapped)
300    }
301
302    // new input objects are remaining taken objects not written/deleted
303    let object_runtime_ref: &mut ObjectRuntime = get_extension_mut!(context)?;
304    let mut config_settings = vec![];
305    for child in object_runtime_ref.all_active_child_objects() {
306        let s: StructTag = child.ty.clone().into();
307        let is_setting = DynamicFieldInfo::is_dynamic_field(&s)
308            && matches!(&s.type_params[1], TypeTag::Struct(s) if config::is_setting(s));
309        if is_setting {
310            config_settings.push((
311                *child.owner,
312                *child.id,
313                child.ty.clone(),
314                child.copied_value,
315            ));
316        }
317    }
318    for (config, setting, ty, value) in config_settings {
319        object_runtime_ref.config_setting_cache_update(config, setting, ty, value)
320    }
321    object_runtime_ref.state.input_objects = object_runtime_ref
322        .test_inventories
323        .taken
324        .iter()
325        .map(|(id, owner)| (*id, owner.clone()))
326        .collect::<BTreeMap<_, _>>();
327    // update inventories
328    // check for bad updates to immutable values
329    for (id, (ty, value)) in new_object_values {
330        debug_assert!(!all_active_child_objects_with_values.contains(&id));
331        if let Some(prev_value) = object_runtime_ref
332            .test_inventories
333            .taken_immutable_values
334            .get(&ty)
335            .and_then(|values| values.get(&id))
336            && !value.equals(prev_value)?
337        {
338            return Ok(NativeResult::err(
339                legacy_test_cost(),
340                E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
341            ));
342        }
343        object_runtime_ref
344            .test_inventories
345            .objects
346            .insert(id, value);
347    }
348    // remove deleted
349    for id in &deleted {
350        object_runtime_ref.test_inventories.objects.remove(id);
351    }
352    // remove active child objects
353    for id in all_active_child_objects_with_values {
354        object_runtime_ref.test_inventories.objects.remove(&id);
355    }
356
357    let effects = transaction_effects(
358        created,
359        written,
360        deleted,
361        transferred,
362        user_events.len() as u64,
363        // TODO: do we need accumulator events here?
364    );
365    Ok(NativeResult::ok(legacy_test_cost(), smallvec![effects]))
366}

368// native fun take_from_address_by_id<T: key>(account: address, id: ID): T;
369pub fn take_from_address_by_id(
370    context: &mut NativeContext,
371    ty_args: Vec<Type>,
372    mut args: VecDeque<Value>,
373) -> PartialVMResult<NativeResult> {
374    let specified_ty = get_specified_ty(ty_args);
375    let id = pop_id(&mut args)?;
376    let account: SuiAddress = pop_arg!(args, AccountAddress).into();
377    pop_arg!(args, StructRef);
378    assert!(args.is_empty());
379    let specified_obj_ty = object_type_of_type(context, &specified_ty)?;
380    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
381    let inventories = &mut object_runtime.test_inventories;
382    let res = take_from_inventory(
383        |x| {
384            inventories
385                .address_inventories
386                .get(&account)
387                .and_then(|inv| inv.get(&specified_obj_ty))
388                .map(|s| s.contains(x))
389                .unwrap_or(false)
390        },
391        &inventories.objects,
392        &mut inventories.taken,
393        &mut object_runtime.state.input_objects,
394        id,
395        Owner::AddressOwner(account),
396    );
397    Ok(match res {
398        Ok(value) => NativeResult::ok(legacy_test_cost(), smallvec![value]),
399        Err(native_err) => native_err,
400    })
401}

403// native fun ids_for_address<T: key>(account: address): vector<ID>;
404pub fn ids_for_address(
405    context: &mut NativeContext,
406    ty_args: Vec<Type>,
407    mut args: VecDeque<Value>,
408) -> PartialVMResult<NativeResult> {
409    let specified_ty = get_specified_ty(ty_args);
410    let account: SuiAddress = pop_arg!(args, AccountAddress).into();
411    assert!(args.is_empty());
412    let specified_obj_ty = object_type_of_type(context, &specified_ty)?;
413    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
414    let inventories = &mut object_runtime.test_inventories;
415    let ids = inventories
416        .address_inventories
417        .get(&account)
418        .and_then(|inv| inv.get(&specified_obj_ty))
419        .map(|s| s.iter().map(|id| pack_id(*id)).collect::<Vec<Value>>())
420        .unwrap_or_default();
421    let ids_vector = Vector::pack(VectorSpecialization::Container, ids).unwrap();
422    Ok(NativeResult::ok(legacy_test_cost(), smallvec![ids_vector]))
423}

611// native fun was_taken_shared(id: ID): bool;
612pub fn was_taken_shared(
613    context: &mut NativeContext,
614    ty_args: Vec<Type>,
615    mut args: VecDeque<Value>,
616) -> PartialVMResult<NativeResult> {
617    assert!(ty_args.is_empty());
618    let id = pop_id(&mut args)?;
619    assert!(args.is_empty());
620    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
621    let inventories = &mut object_runtime.test_inventories;
622    let was_taken = inventories
623        .taken
624        .get(&id)
625        .map(|owner| matches!(owner, Owner::Shared { .. }))
626        .unwrap_or(false);
627    Ok(NativeResult::ok(
628        legacy_test_cost(),
629        smallvec![Value::bool(was_taken)],
630    ))
631}

633pub fn allocate_receiving_ticket_for_object(
634    context: &mut NativeContext,
635    ty_args: Vec<Type>,
636    mut args: VecDeque<Value>,
637) -> PartialVMResult<NativeResult> {
638    let ty = get_specified_ty(ty_args);
639    let id = pop_id(&mut args)?;
640
641    let abilities = context.type_to_abilities(&ty)?;
642    let Some((tag, layout, _)) = get_tag_and_layouts(context, &ty)? else {
643        return Ok(NativeResult::err(
644            context.gas_used(),
645            E_UNABLE_TO_ALLOCATE_RECEIVING_TICKET,
646        ));
647    };
648    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
649    let object_version = SequenceNumber::new();
650    let inventories = &mut object_runtime.test_inventories;
651    if inventories.allocated_tickets.contains_key(&id) {
652        return Ok(NativeResult::err(
653            context.gas_used(),
654            E_RECEIVING_TICKET_ALREADY_ALLOCATED,
655        ));
656    }
657
658    let obj_value = inventories.objects.remove(&id).unwrap();
659    let Some(bytes) = obj_value.typed_serialize(&layout) else {
660        return Ok(NativeResult::err(
661            context.gas_used(),
662            E_UNABLE_TO_ALLOCATE_RECEIVING_TICKET,
663        ));
664    };
665    let has_public_transfer = abilities.has_store();
666    let move_object = unsafe {
667        MoveObject::new_from_execution_with_limit(
668            tag.into(),
669            has_public_transfer,
670            object_version,
671            bytes,
672            250 * 1024,
673        )
674    }
675    .unwrap();
676
677    let Some((owner, _)) = inventories
678        .address_inventories
679        .iter()
680        .find(|(_addr, objs)| objs.iter().any(|(_, ids)| ids.contains(&id)))
681    else {
682        return Ok(NativeResult::err(
683            context.gas_used(),
684            E_OBJECT_NOT_FOUND_CODE,
685        ));
686    };
687
688    inventories.allocated_tickets.insert(
689        id,
690        (
691            DynamicallyLoadedObjectMetadata {
692                version: SequenceNumber::new(),
693                digest: ObjectDigest::MIN,
694                owner: Owner::AddressOwner(*owner),
695                storage_rebate: 0,
696                previous_transaction: TransactionDigest::default(),
697            },
698            obj_value,
699        ),
700    );
701
702    let object = Object::new_move(
703        move_object,
704        Owner::AddressOwner(*owner),
705        TransactionDigest::default(),
706    );
707
708    // NB: Must be a `&&` reference since the extension stores a static ref to the object storage.
709    let store: &&InMemoryTestStore = get_extension!(context)?;
710    store.0.borrow_mut().insert_object(object);
711
712    Ok(NativeResult::ok(
713        legacy_test_cost(),
714        smallvec![Value::u64(object_version.value())],
715    ))
716}

718pub fn deallocate_receiving_ticket_for_object(
719    context: &mut NativeContext,
720    _ty_args: Vec<Type>,
721    mut args: VecDeque<Value>,
722) -> PartialVMResult<NativeResult> {
723    let id = pop_id(&mut args)?;
724
725    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
726    let inventories = &mut object_runtime.test_inventories;
727    // Deallocate the ticket -- we should never hit this scenario
728    let Some((_, value)) = inventories.allocated_tickets.remove(&id) else {
729        return Ok(NativeResult::err(
730            context.gas_used(),
731            E_UNABLE_TO_DEALLOCATE_RECEIVING_TICKET,
732        ));
733    };
734
735    // Insert the object value that we saved from earlier and put it back into the object set.
736    // This is fine since it can't have been touched.
737    inventories.objects.insert(id, value);
738
739    // Remove the object from storage. We should never hit this scenario either.
740    let store: &&InMemoryTestStore = get_extension!(context)?;
741    if store.0.borrow_mut().remove_object(id).is_none() {
742        return Ok(NativeResult::err(
743            context.gas_used(),
744            E_UNABLE_TO_DEALLOCATE_RECEIVING_TICKET,
745        ));
746    };
747
748    Ok(NativeResult::ok(legacy_test_cost(), smallvec![]))
749}

753fn take_from_inventory(
754    is_in_inventory: impl FnOnce(&ObjectID) -> bool,
755    objects: &BTreeMap<ObjectID, Value>,
756    taken: &mut BTreeMap<ObjectID, Owner>,
757    input_objects: &mut BTreeMap<ObjectID, Owner>,
758    id: ObjectID,
759    owner: Owner,
760) -> Result<Value, NativeResult> {
761    let obj_opt = objects.get(&id);
762    let is_taken = taken.contains_key(&id);
763    if is_taken || !is_in_inventory(&id) || obj_opt.is_none() {
764        return Err(NativeResult::err(
765            legacy_test_cost(),
766            E_OBJECT_NOT_FOUND_CODE,
767        ));
768    }
769    taken.insert(id, owner.clone());
770    input_objects.insert(id, owner);
771    let obj = obj_opt.unwrap();
772    Ok(obj.copy_value().unwrap())
773}

813fn pop_id(args: &mut VecDeque<Value>) -> PartialVMResult<ObjectID> {
814    let v = match args.pop_back() {
815        None => {
816            return Err(PartialVMError::new(
817                StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR,
818            ));
819        }
820        Some(v) => v,
821    };
822    Ok(get_nth_struct_field(v, 0)?
823        .value_as::<AccountAddress>()?
824        .into())
825}

851fn transaction_effects(
852    created: impl IntoIterator<Item = impl Into<AccountAddress>>,
853    written: impl IntoIterator<Item = impl Into<AccountAddress>>,
854    deleted: impl IntoIterator<Item = impl Into<AccountAddress>>,
855    transferred: impl IntoIterator<Item = (ObjectID, Owner)>,
856    num_events: u64,
857) -> Value {
858    let mut transferred_to_account = vec![];
859    let mut transferred_to_object = vec![];
860    let mut shared = vec![];
861    let mut frozen = vec![];
862    for (id, owner) in transferred {
863        match owner {
864            Owner::AddressOwner(a) => {
865                transferred_to_account.push((pack_id(id), Value::address(a.into())))
866            }
867            Owner::ObjectOwner(o) => transferred_to_object.push((pack_id(id), pack_id(o))),
868            Owner::Shared { .. } => shared.push(id),
869            Owner::Immutable => frozen.push(id),
870            Owner::ConsensusAddressOwner { owner, .. } => {
871                transferred_to_account.push((pack_id(id), Value::address(owner.into())))
872            }
873        }
874    }
875
876    let created_field = pack_ids(created);
877    let written_field = pack_ids(written);
878    let deleted_field = pack_ids(deleted);
879    let transferred_to_account_field = pack_vec_map(transferred_to_account);
880    let transferred_to_object_field = pack_vec_map(transferred_to_object);
881    let shared_field = pack_ids(shared);
882    let frozen_field = pack_ids(frozen);
883    let num_events_field = Value::u64(num_events);
884    Value::struct_(values::Struct::pack(vec![
885        created_field,
886        written_field,
887        deleted_field,
888        transferred_to_account_field,
889        transferred_to_object_field,
890        shared_field,
891        frozen_field,
892        num_events_field,
893    ]))
894}

915fn find_all_wrapped_objects<'a, 'i>(
916    context: &NativeContext,
917    ids: &'i mut BTreeSet<ObjectID>,
918    new_object_values: impl IntoIterator<Item = (&'a ObjectID, &'a MoveObjectType, impl Borrow<Value>)>,
919) {
920    #[derive(Copy, Clone)]
921    enum LookingFor {
922        Wrapped,
923        Uid,
924        Address,
925    }
926
927    struct Traversal<'i, 'u> {
928        state: LookingFor,
929        ids: &'i mut BTreeSet<ObjectID>,
930        uid: &'u MoveStructLayout,
931    }
932
933    impl<'b, 'l> AV::Traversal<'b, 'l> for Traversal<'_, '_> {
934        type Error = AV::Error;
935
936        fn traverse_struct(
937            &mut self,
938            driver: &mut AV::StructDriver<'_, 'b, 'l>,
939        ) -> Result<(), Self::Error> {
940            match self.state {
941                // We're at the top-level of the traversal, looking for an object to recurse into.
942                // We can unconditionally switch to looking for UID fields at the level below,
943                // because we know that all the top-level values are objects.
944                LookingFor::Wrapped => {
945                    while driver
946                        .next_field(&mut Traversal {
947                            state: LookingFor::Uid,
948                            ids: self.ids,
949                            uid: self.uid,
950                        })?
951                        .is_some()
952                    {}
953                }
954
955                // We are looking for UID fields. If we find one (which we confirm by checking its
956                // layout), switch to looking for addresses in its sub-structure.
957                LookingFor::Uid => {
958                    while let Some(MoveFieldLayout { name: _, layout }) = driver.peek_field() {
959                        if matches!(layout, MoveTypeLayout::Struct(s) if s.as_ref() == self.uid) {
960                            driver.next_field(&mut Traversal {
961                                state: LookingFor::Address,
962                                ids: self.ids,
963                                uid: self.uid,
964                            })?;
965                        } else {
966                            driver.next_field(self)?;
967                        }
968                    }
969                }
970
971                // When looking for addresses, recurse through structs, as the address is nested
972                // within the UID.
973                LookingFor::Address => while driver.next_field(self)?.is_some() {},
974            }
975
976            Ok(())
977        }
978
979        fn traverse_address(
980            &mut self,
981            _: &AV::ValueDriver<'_, 'b, 'l>,
982            address: AccountAddress,
983        ) -> Result<(), Self::Error> {
984            // If we're looking for addresses, and we found one, then save it.
985            if matches!(self.state, LookingFor::Address) {
986                self.ids.insert(address.into());
987            }
988            Ok(())
989        }
990    }
991
992    let uid = UID::layout();
993    for (_id, ty, value) in new_object_values {
994        let type_tag = TypeTag::from(ty.clone());
995        // NB: We can get the layout from the VM's cache since the types and modules
996        // associated with all of these types must be in the type/module cache in the VM -- THIS IS
997        // BECAUSE WE ARE IN TEST SCENARIO ONLY AND THIS MAY NOT GENERALLY HOLD IN A
998        // MULTI-TRANSACTION SETTING.
999        let Some(layout) = context.type_tag_to_layout_for_test_scenario_only(&type_tag) else {
1000            debug_assert!(false);
1001            continue;
1002        };
1003
1004        let Some(annotated_layout) =
1005            context.type_tag_to_fully_annotated_layout_for_test_scenario_only(&type_tag)
1006        else {
1007            debug_assert!(false);
1008            continue;
1009        };
1010
1011        let blob = value.borrow().typed_serialize(&layout).unwrap();
1012        MoveValue::visit_deserialize(
1013            &blob,
1014            &annotated_layout,
1015            &mut Traversal {
1016                state: LookingFor::Wrapped,
1017                ids,
1018                uid: &uid,
1019            },
1020        )
1021        .unwrap();
1022    }
