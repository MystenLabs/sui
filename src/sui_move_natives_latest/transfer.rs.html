25const E_SHARED_NON_NEW_OBJECT: u64 = 0;
26const E_BCS_SERIALIZATION_FAILURE: u64 = 1;
27const E_RECEIVING_OBJECT_TYPE_MISMATCH: u64 = 2;
28// Represents both the case where the object does not exist and the case where the object is not
29// able to be accessed through the parent that is passed-in.
30const E_UNABLE_TO_RECEIVE_OBJECT: u64 = 3;
31// Operation not yet supported
32const E_NOT_SUPPORTED: u64 = 5;

34#[derive(Clone, Debug)]
35pub struct TransferReceiveObjectInternalCostParams {
36    pub transfer_receive_object_internal_cost_base: InternalGas,
37}
38/***************************************************************************************************
39* native fun receive_object_internal
40* Implementation of the Move native function `receive_object_internal<T: key>(parent: &mut UID, rec: Receiver<T>): T`
41*   gas cost: transfer_receive_object_internal_cost_base |  covers various fixed costs in the oper
42**************************************************************************************************/
43
44pub fn receive_object_internal(
45    context: &mut NativeContext,
46    mut ty_args: Vec<Type>,
47    mut args: VecDeque<Value>,
48) -> PartialVMResult<NativeResult> {
49    debug_assert!(ty_args.len() == 1);
50    debug_assert!(args.len() == 3);
51    let transfer_receive_object_internal_cost_params = get_extension!(context, NativesCostTable)?
52        .transfer_receive_object_internal_cost_params
53        .clone();
54    native_charge_gas_early_exit!(
55        context,
56        transfer_receive_object_internal_cost_params.transfer_receive_object_internal_cost_base
57    );
58    let child_ty = ty_args.pop().unwrap();
59    let child_receiver_sequence_number: SequenceNumber = pop_arg!(args, u64).into();
60    let child_receiver_object_id = args.pop_back().unwrap();
61    let parent = pop_arg!(args, AccountAddress).into();
62    assert!(args.is_empty());
63    let child_id: ObjectID = get_receiver_object_id(child_receiver_object_id.copy_value().unwrap())
64        .unwrap()
65        .value_as::<AccountAddress>()
66        .unwrap()
67        .into();
68    assert!(ty_args.is_empty());
69
70    let Some((tag, layout, annotated_layout)) = get_tag_and_layouts(context, &child_ty)? else {
71        return Ok(NativeResult::err(
72            context.gas_used(),
73            E_BCS_SERIALIZATION_FAILURE,
74        ));
75    };
76
77    let object_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
78    let (_cache_info, child) = match object_runtime.receive_object(
79        parent,
80        child_id,
81        child_receiver_sequence_number,
82        &layout,
83        &annotated_layout,
84        MoveObjectType::from(tag),
85    ) {
86        // NB: Loaded and doesn't exist and inauthenticated read should lead to the exact same error
87        Ok(None) => {
88            return Ok(NativeResult::err(
89                context.gas_used(),
90                E_UNABLE_TO_RECEIVE_OBJECT,
91            ));
92        }
93        Ok(Some(ObjectResult::Loaded(gv))) => gv,
94        Ok(Some(ObjectResult::MismatchedType)) => {
95            return Ok(NativeResult::err(
96                context.gas_used(),
97                E_RECEIVING_OBJECT_TYPE_MISMATCH,
98            ));
99        }
100        Err(x) => return Err(x),
101    };
102
103    Ok(NativeResult::ok(context.gas_used(), smallvec![child]))
104}

107pub struct TransferInternalCostParams {
108    pub transfer_transfer_internal_cost_base: InternalGas,
109}
110/***************************************************************************************************
111* native fun transfer_impl
112* Implementation of the Move native function `transfer_impl<T: key>(obj: T, recipient: address)`
113*   gas cost: transfer_transfer_internal_cost_base                  |  covers various fixed costs in the oper
114**************************************************************************************************/
115pub fn transfer_internal(
116    context: &mut NativeContext,
117    mut ty_args: Vec<Type>,
118    mut args: VecDeque<Value>,
119) -> PartialVMResult<NativeResult> {
120    debug_assert!(ty_args.len() == 1);
121    debug_assert!(args.len() == 2);
122
123    let transfer_transfer_internal_cost_params = get_extension!(context, NativesCostTable)?
124        .transfer_transfer_internal_cost_params
125        .clone();
126
127    native_charge_gas_early_exit!(
128        context,
129        transfer_transfer_internal_cost_params.transfer_transfer_internal_cost_base
130    );
131
132    let ty = ty_args.pop().unwrap();
133    let recipient = pop_arg!(args, AccountAddress);
134    let obj = args.pop_back().unwrap();
135
136    let owner = Owner::AddressOwner(recipient.into());
137    object_runtime_transfer(context, owner, ty, obj)?;
138    let cost = context.gas_used();
139    Ok(NativeResult::ok(cost, smallvec![]))
140}

147macro_rules! native_charge_gas_early_exit_option {
148    ($native_context:ident, $cost:expr) => {{
149        use move_binary_format::errors::PartialVMError;
150        use move_core_types::vm_status::StatusCode;
151        native_charge_gas_early_exit!(
152            $native_context,
153            $cost.ok_or_else(|| {
154                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
155                    .with_message("Gas cost for party is missing".to_string())
156            })?
157        );
158    }};
159}
160/***************************************************************************************************
161* native fun multi_partytransfer_impl
162* Implementation of the Move native function
163*   `party_transfer_impl<T: key>(obj: T, recipient: address)`
164*   gas cost: transfer_party_transfer_internal_cost_base                  |  covers various fixed costs in the oper
165**************************************************************************************************/
166pub fn party_transfer_internal(
167    context: &mut NativeContext,
168    mut ty_args: Vec<Type>,
169    mut args: VecDeque<Value>,
170) -> PartialVMResult<NativeResult> {
171    const NONE: u64 = 0;
172    const READ: u64 = 0b0001;
173    const WRITE: u64 = 0b0010;
174    const DELETE: u64 = 0b0100;
175    const TRANSFER: u64 = 0b1000;
176    const ALL: u64 = READ | WRITE | DELETE | TRANSFER;
177
178    debug_assert!(ty_args.len() == 1);
179    debug_assert!(args.len() == 4);
180
181    let is_supported = get_extension!(context, ObjectRuntime)?
182        .protocol_config
183        .enable_party_transfer();
184    if !is_supported {
185        let cost = context.gas_used();
186        return Ok(NativeResult::err(cost, E_NOT_SUPPORTED));
187    }
188
189    let transfer_party_transfer_internal_cost_params = get_extension!(context, NativesCostTable)?
190        .transfer_party_transfer_internal_cost_params
191        .clone();
192
193    native_charge_gas_early_exit_option!(
194        context,
195        transfer_party_transfer_internal_cost_params.transfer_party_transfer_internal_cost_base
196    );
197
198    let ty = ty_args.pop().unwrap();
199    let permissions = pop_arg!(args, Vec<u64>);
200    let addresses = pop_arg!(args, Vec<AccountAddress>);
201    let default_permissions = pop_arg!(args, u64);
202    let obj = args.pop_back().unwrap();
203    let Ok([permissions]): Result<[u64; 1], _> = permissions.try_into() else {
204        return Err(
205            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
206                .with_message("Party transfer only supports one party member".to_string()),
207        );
208    };
209    if permissions != ALL || default_permissions != NONE {
210        return Err(
211            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
212                "Party transfer only supports one party member with all permissions".to_string(),
213            ),
214        );
215    }
216    let Ok([address]): Result<[AccountAddress; 1], _> = addresses.try_into() else {
217        return Err(
218            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
219                .with_message("Party transfer only supports one party member".to_string()),
220        );
221    };
222
223    // Dummy version, to be filled with the correct initial version when the effects of the
224    // transaction are written to storage.
225    let owner = Owner::ConsensusAddressOwner {
226        start_version: SequenceNumber::new(),
227        owner: address.into(),
228    };
229    object_runtime_transfer(context, owner, ty, obj)?;
230    let cost = context.gas_used();
231    Ok(NativeResult::ok(cost, smallvec![]))
232}

269pub struct TransferShareObjectCostParams {
270    pub transfer_share_object_cost_base: InternalGas,
271}
272/***************************************************************************************************
273* native fun share_object
274* Implementation of the Move native function `share_object<T: key>(obj: T)`
275*   gas cost: transfer_share_object_cost_base                  |  covers various fixed costs in the oper
276**************************************************************************************************/
277pub fn share_object(
278    context: &mut NativeContext,
279    mut ty_args: Vec<Type>,
280    mut args: VecDeque<Value>,
281) -> PartialVMResult<NativeResult> {
282    debug_assert!(ty_args.len() == 1);
283    debug_assert!(args.len() == 1);
284
285    let transfer_share_object_cost_params = get_extension!(context, NativesCostTable)?
286        .transfer_share_object_cost_params
287        .clone();
288
289    native_charge_gas_early_exit!(
290        context,
291        transfer_share_object_cost_params.transfer_share_object_cost_base
292    );
293
294    let ty = ty_args.pop().unwrap();
295    let obj = args.pop_back().unwrap();
296    let transfer_result = object_runtime_transfer(
297        context,
298        // Dummy version, to be filled with the correct initial version when the effects of the
299        // transaction are written to storage.
300        Owner::Shared {
301            initial_shared_version: SequenceNumber::new(),
302        },
303        ty,
304        obj,
305    )?;
306    let cost = context.gas_used();
307    Ok(match transfer_result {
308        // New means the ID was created in this transaction
309        // SameOwner means the object was previously shared and was re-shared
310        TransferResult::New | TransferResult::SameOwner => NativeResult::ok(cost, smallvec![]),
311        TransferResult::OwnerChanged => NativeResult::err(cost, E_SHARED_NON_NEW_OBJECT),
312    })
313}
314
315fn object_runtime_transfer(
316    context: &mut NativeContext,
317    owner: Owner,
318    ty: Type,
319    obj: Value,
320) -> PartialVMResult<TransferResult> {
321    let object_type = match context.type_to_type_tag(&ty)? {
322        TypeTag::Struct(s) => MoveObjectType::from(*s),
323        _ => {
324            return Err(
325                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
326                    .with_message("Sui verifier guarantees this is a struct".to_string()),
327            );
328        }
329    };
330
331    let obj_runtime: &mut ObjectRuntime = get_extension_mut!(context)?;
332    obj_runtime.transfer(owner, object_type, obj, /* end of transaction */ false)
333}
