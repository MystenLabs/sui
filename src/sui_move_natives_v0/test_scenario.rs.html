42// This function updates the inventories based on the transfers and deletes that occurred in the
43// transaction
44// native fun end_transaction(): TransactionResult;
45pub fn end_transaction(
46    context: &mut NativeContext,
47    ty_args: Vec<Type>,
48    args: VecDeque<Value>,
49) -> PartialVMResult<NativeResult> {
50    assert!(ty_args.is_empty());
51    assert!(args.is_empty());
52    let object_runtime_ref: &mut ObjectRuntime = context.extensions_mut().get_mut();
53    let taken_shared_or_imm: BTreeMap<_, _> = object_runtime_ref
54        .test_inventories
55        .taken
56        .iter()
57        .filter(|(_id, owner)| matches!(owner, Owner::Shared { .. } | Owner::Immutable))
58        .map(|(id, owner)| (*id, owner.clone()))
59        .collect();
60    // set to true if a shared or imm object was:
61    // - transferred in a way that changes it from its original shared/imm state
62    // - wraps the object
63    // if true, we will "abort"
64    let mut incorrect_shared_or_imm_handling = false;
65
66    let object_runtime_state = object_runtime_ref.take_state();
67    // Determine writes and deletes
68    // We pass an empty map as we do not expose dynamic field objects in the system
69    let results = object_runtime_state.finish(
70        BTreeSet::new(),
71        BTreeSet::new(),
72        BTreeMap::new(),
73        BTreeMap::new(),
74        false,
75    );
76    let RuntimeResults {
77        writes,
78        deletions,
79        user_events,
80        loaded_child_objects: _,
81    } = match results {
82        Ok(res) => res,
83        Err(_) => {
84            return Ok(NativeResult::err(
85                legacy_test_cost(),
86                E_COULD_NOT_GENERATE_EFFECTS,
87            ));
88        }
89    };
90    let all_active_child_objects = object_runtime_ref
91        .all_active_child_objects()
92        .map(|(id, _, _)| *id)
93        .collect::<BTreeSet<_>>();
94    let inventories = &mut object_runtime_ref.test_inventories;
95    let mut new_object_values = LinkedHashMap::new();
96    let mut transferred = vec![];
97    // cleanup inventories
98    // we will remove all changed objects
99    // - deleted objects need to be removed to mark deletions
100    // - written objects are removed and later replaced to mark new values and new owners
101    // - child objects will not be reflected in transfers, but need to be no longer retrievable
102    for id in deletions
103        .keys()
104        .chain(writes.keys())
105        .chain(&all_active_child_objects)
106    {
107        for addr_inventory in inventories.address_inventories.values_mut() {
108            for s in addr_inventory.values_mut() {
109                s.remove(id);
110            }
111        }
112        for s in &mut inventories.shared_inventory.values_mut() {
113            s.remove(id);
114        }
115        for s in &mut inventories.immutable_inventory.values_mut() {
116            s.remove(id);
117        }
118        inventories.taken.remove(id);
119    }
120    // handle transfers, inserting transferred/written objects into their respective inventory
121    let mut created = vec![];
122    let mut written = vec![];
123    for (id, (kind, owner, ty, value)) in writes {
124        new_object_values.insert(id, (ty.clone(), value.copy_value().unwrap()));
125        transferred.push((id, owner.clone()));
126        incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
127            || taken_shared_or_imm
128                .get(&id)
129                .map(|shared_or_imm_owner| shared_or_imm_owner != &owner)
130                .unwrap_or(/* not incorrect */ false);
131        match kind {
132            WriteKind::Create => created.push(id),
133            WriteKind::Mutate | WriteKind::Unwrap => written.push(id),
134        }
135        match owner {
136            Owner::AddressOwner(a) => {
137                inventories
138                    .address_inventories
139                    .entry(a)
140                    .or_default()
141                    .entry(ty)
142                    .or_default()
143                    .insert(id, ());
144            }
145            Owner::ObjectOwner(_) => (),
146            Owner::Shared { .. } => {
147                inventories
148                    .shared_inventory
149                    .entry(ty)
150                    .or_default()
151                    .insert(id, ());
152            }
153            Owner::Immutable => {
154                inventories
155                    .immutable_inventory
156                    .entry(ty)
157                    .or_default()
158                    .insert(id, ());
159            }
160            Owner::ConsensusAddressOwner { .. } => {
161                unimplemented!("ConsensusAddressOwner does not exist for this execution version")
162            }
163        }
164    }
165    // deletions already handled above, but we drop the delete kind for the effects
166    let mut deleted = vec![];
167    for (id, _) in deletions {
168        incorrect_shared_or_imm_handling =
169            incorrect_shared_or_imm_handling || taken_shared_or_imm.contains_key(&id);
170        deleted.push(id);
171    }
172    // find all wrapped objects
173    let mut all_wrapped = BTreeSet::new();
174    let object_runtime_ref: &ObjectRuntime = context.extensions().get();
175    find_all_wrapped_objects(
176        context,
177        &mut all_wrapped,
178        new_object_values
179            .iter()
180            .map(|(id, (ty, value))| (id, ty, value)),
181    );
182    find_all_wrapped_objects(
183        context,
184        &mut all_wrapped,
185        object_runtime_ref.all_active_child_objects(),
186    );
187    // mark as "incorrect" if a shared/imm object was wrapped or is a child object
188    incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
189        || taken_shared_or_imm
190            .keys()
191            .any(|id| all_wrapped.contains(id) || all_active_child_objects.contains(id));
192    // if incorrect handling, return with an 'abort'
193    if incorrect_shared_or_imm_handling {
194        return Ok(NativeResult::err(
195            legacy_test_cost(),
196            E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
197        ));
198    }
199
200    // mark all wrapped as deleted
201    for wrapped in all_wrapped {
202        deleted.push(wrapped)
203    }
204
205    // new input objects are remaining taken objects not written/deleted
206    let object_runtime_ref: &mut ObjectRuntime = context.extensions_mut().get_mut();
207    object_runtime_ref.state.input_objects = object_runtime_ref
208        .test_inventories
209        .taken
210        .iter()
211        .map(|(id, owner)| (*id, owner.clone()))
212        .collect::<BTreeMap<_, _>>();
213    // update inventories
214    // check for bad updates to immutable values
215    for (id, (ty, value)) in new_object_values {
216        debug_assert!(!all_active_child_objects.contains(&id));
217        if let Some(prev_value) = object_runtime_ref
218            .test_inventories
219            .taken_immutable_values
220            .get(&ty)
221            .and_then(|values| values.get(&id))
222        {
223            if !value.equals(prev_value)? {
224                return Ok(NativeResult::err(
225                    legacy_test_cost(),
226                    E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
227                ));
228            }
229        }
230        object_runtime_ref
231            .test_inventories
232            .objects
233            .insert(id, value);
234    }
235    // remove deleted
236    for id in &deleted {
237        object_runtime_ref.test_inventories.objects.remove(id);
238    }
239    // remove active child objects
240    for id in all_active_child_objects {
241        object_runtime_ref.test_inventories.objects.remove(&id);
242    }
243
244    let effects = transaction_effects(
245        created,
246        written,
247        deleted,
248        transferred,
249        user_events.len() as u64,
250    );
251    Ok(NativeResult::ok(legacy_test_cost(), smallvec![effects]))
252}

254// native fun take_from_address_by_id<T: key>(account: address, id: ID): T;
255pub fn take_from_address_by_id(
256    context: &mut NativeContext,
257    ty_args: Vec<Type>,
258    mut args: VecDeque<Value>,
259) -> PartialVMResult<NativeResult> {
260    let specified_ty = get_specified_ty(ty_args);
261    let id = pop_id(&mut args)?;
262    let account: SuiAddress = pop_arg!(args, AccountAddress).into();
263    pop_arg!(args, StructRef);
264    assert!(args.is_empty());
265    let object_runtime: &mut ObjectRuntime = context.extensions_mut().get_mut();
266    let inventories = &mut object_runtime.test_inventories;
267    let res = take_from_inventory(
268        |x| {
269            inventories
270                .address_inventories
271                .get(&account)
272                .and_then(|inv| inv.get(&specified_ty))
273                .map(|s| s.contains_key(x))
274                .unwrap_or(false)
275        },
276        &inventories.objects,
277        &mut inventories.taken,
278        &mut object_runtime.state.input_objects,
279        id,
280        Owner::AddressOwner(account),
281    );
282    Ok(match res {
283        Ok(value) => NativeResult::ok(legacy_test_cost(), smallvec![value]),
284        Err(native_err) => native_err,
285    })
286}

512fn take_from_inventory(
513    is_in_inventory: impl FnOnce(&ObjectID) -> bool,
514    objects: &BTreeMap<ObjectID, Value>,
515    taken: &mut BTreeMap<ObjectID, Owner>,
516    input_objects: &mut BTreeMap<ObjectID, Owner>,
517    id: ObjectID,
518    owner: Owner,
519) -> Result<Value, NativeResult> {
520    let obj_opt = objects.get(&id);
521    let is_taken = taken.contains_key(&id);
522    if is_taken || !is_in_inventory(&id) || obj_opt.is_none() {
523        return Err(NativeResult::err(
524            legacy_test_cost(),
525            E_OBJECT_NOT_FOUND_CODE,
526        ));
527    }
528    taken.insert(id, owner.clone());
529    input_objects.insert(id, owner);
530    let obj = obj_opt.unwrap();
531    Ok(obj.copy_value().unwrap())
532}

534fn most_recent_at_ty(
535    taken: &BTreeMap<ObjectID, Owner>,
536    inv: &BTreeMap<Type, Set<ObjectID>>,
537    ty: Type,
538) -> Value {
539    pack_option(most_recent_at_ty_opt(taken, inv, ty))
540}
541
542fn most_recent_at_ty_opt(
543    taken: &BTreeMap<ObjectID, Owner>,
544    inv: &BTreeMap<Type, Set<ObjectID>>,
545    ty: Type,
546) -> Option<Value> {
547    let s = inv.get(&ty)?;
548    let most_recent_id = s.keys().rfind(|id| !taken.contains_key(id))?;
549    Some(pack_id(*most_recent_id))
550}

552fn get_specified_ty(mut ty_args: Vec<Type>) -> Type {
553    assert!(ty_args.len() == 1);
554    ty_args.pop().unwrap()
555}

557// helpers
558fn pop_id(args: &mut VecDeque<Value>) -> PartialVMResult<ObjectID> {
559    let v = match args.pop_back() {
560        None => {
561            return Err(PartialVMError::new(
562                StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR,
563            ))
564        }
565        Some(v) => v,
566    };
567    Ok(get_nth_struct_field(v, 0)?
568        .value_as::<AccountAddress>()?
569        .into())
570}
571
572fn pack_id(a: impl Into<AccountAddress>) -> Value {
573    Value::struct_(values::Struct::pack(vec![Value::address(a.into())]))
574}
575
576fn pack_ids(items: impl IntoIterator<Item = impl Into<AccountAddress>>) -> Value {
577    Value::vector_for_testing_only(items.into_iter().map(pack_id))
578}
579
580fn pack_vec_map(items: impl IntoIterator<Item = (Value, Value)>) -> Value {
581    Value::struct_(values::Struct::pack(vec![Value::vector_for_testing_only(
582        items
583            .into_iter()
584            .map(|(k, v)| Value::struct_(values::Struct::pack(vec![k, v]))),
585    )]))
586}
587
588fn transaction_effects(
589    created: impl IntoIterator<Item = impl Into<AccountAddress>>,
590    written: impl IntoIterator<Item = impl Into<AccountAddress>>,
591    deleted: impl IntoIterator<Item = impl Into<AccountAddress>>,
592    transferred: impl IntoIterator<Item = (ObjectID, Owner)>,
593    num_events: u64,
594) -> Value {
595    let mut transferred_to_account = vec![];
596    let mut transferred_to_object = vec![];
597    let mut shared = vec![];
598    let mut frozen = vec![];
599    for (id, owner) in transferred {
600        match owner {
601            Owner::AddressOwner(a) => {
602                transferred_to_account.push((pack_id(id), Value::address(a.into())))
603            }
604            Owner::ObjectOwner(o) => transferred_to_object.push((pack_id(id), pack_id(o))),
605            Owner::Shared { .. } => shared.push(id),
606            Owner::Immutable => frozen.push(id),
607            Owner::ConsensusAddressOwner { .. } => {
608                unimplemented!("ConsensusAddressOwner does not exist for this execution version")
609            }
610        }
611    }
612
613    let created_field = pack_ids(created);
614    let written_field = pack_ids(written);
615    let deleted_field = pack_ids(deleted);
616    let transferred_to_account_field = pack_vec_map(transferred_to_account);
617    let transferred_to_object_field = pack_vec_map(transferred_to_object);
618    let shared_field = pack_ids(shared);
619    let frozen_field = pack_ids(frozen);
620    let num_events_field = Value::u64(num_events);
621    Value::struct_(values::Struct::pack(vec![
622        created_field,
623        written_field,
624        deleted_field,
625        transferred_to_account_field,
626        transferred_to_object_field,
627        shared_field,
628        frozen_field,
629        num_events_field,
630    ]))
631}
632
633fn pack_option(opt: Option<Value>) -> Value {
634    let item = match opt {
635        Some(v) => vec![v],
636        None => vec![],
637    };
638    Value::struct_(values::Struct::pack(vec![Value::vector_for_testing_only(
639        item,
640    )]))
641}

643fn find_all_wrapped_objects<'a, 'i>(
644    context: &NativeContext,
645    ids: &'i mut BTreeSet<ObjectID>,
646    new_object_values: impl IntoIterator<Item = (&'a ObjectID, &'a Type, impl Borrow<Value>)>,
647) {
648    #[derive(Copy, Clone)]
649    enum LookingFor {
650        Wrapped,
651        Uid,
652        Address,
653    }
654
655    struct Traversal<'i, 'u> {
656        state: LookingFor,
657        ids: &'i mut BTreeSet<ObjectID>,
658        uid: &'u MoveStructLayout,
659    }
660
661    impl<'b, 'l> AV::Traversal<'b, 'l> for Traversal<'_, '_> {
662        type Error = AV::Error;
663
664        fn traverse_struct(
665            &mut self,
666            driver: &mut AV::StructDriver<'_, 'b, 'l>,
667        ) -> Result<(), Self::Error> {
668            match self.state {
669                // We're at the top-level of the traversal, looking for an object to recurse into.
670                // We can unconditionally switch to looking for UID fields at the level below,
671                // because we know that all the top-level values are objects.
672                LookingFor::Wrapped => {
673                    while driver
674                        .next_field(&mut Traversal {
675                            state: LookingFor::Uid,
676                            ids: self.ids,
677                            uid: self.uid,
678                        })?
679                        .is_some()
680                    {}
681                }
682
683                // We are looking for UID fields. If we find one (which we confirm by checking its
684                // layout), switch to looking for addresses in its sub-structure.
685                LookingFor::Uid => {
686                    while let Some(MoveFieldLayout { name: _, layout }) = driver.peek_field() {
687                        if matches!(layout, MoveTypeLayout::Struct(s) if s.as_ref() == self.uid) {
688                            driver.next_field(&mut Traversal {
689                                state: LookingFor::Address,
690                                ids: self.ids,
691                                uid: self.uid,
692                            })?;
693                        } else {
694                            driver.next_field(self)?;
695                        }
696                    }
697                }
698
699                // When looking for addresses, recurse through structs, as the address is nested
700                // within the UID.
701                LookingFor::Address => while driver.next_field(self)?.is_some() {},
702            }
703
704            Ok(())
705        }
706
707        fn traverse_address(
708            &mut self,
709            _: &AV::ValueDriver<'_, 'b, 'l>,
710            address: AccountAddress,
711        ) -> Result<(), Self::Error> {
712            // If we're looking for addresses, and we found one, then save it.
713            if matches!(self.state, LookingFor::Address) {
714                self.ids.insert(address.into());
715            }
716            Ok(())
717        }
718    }
719
720    let uid = UID::layout();
721    for (_id, ty, value) in new_object_values {
722        let Ok(Some(layout)) = context.type_to_type_layout(ty) else {
723            debug_assert!(false);
724            continue;
725        };
726
727        let Ok(Some(annotated_layout)) = context.type_to_fully_annotated_layout(ty) else {
728            debug_assert!(false);
729            continue;
730        };
731
732        let blob = value.borrow().simple_serialize(&layout).unwrap();
733        MoveValue::visit_deserialize(
734            &blob,
735            &annotated_layout,
736            &mut Traversal {
737                state: LookingFor::Wrapped,
738                ids,
739                uid: &uid,
740            },
741        )
742        .unwrap();
743    }
744}
