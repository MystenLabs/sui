753// ID { bytes: address }
754// Extract the first field of the struct to get the address bytes.
755pub fn get_receiver_object_id(object: Value) -> Result<Value, PartialVMError> {
756    get_nested_struct_field(object, &[0])
757}
758
759// Object { id: UID { id: ID { bytes: address } } .. }
760// Extract the first field of the struct 3 times to get the id bytes.
761pub fn get_object_id(object: Value) -> Result<Value, PartialVMError> {
762    get_nested_struct_field(object, &[0, 0, 0])
763}
764
765// Extract a field value that's nested inside value `v`. The offset of each nesting
766// is determined by `offsets`.
767pub fn get_nested_struct_field(mut v: Value, offsets: &[usize]) -> Result<Value, PartialVMError> {
768    for offset in offsets {
769        v = get_nth_struct_field(v, *offset)?;
770    }
771    Ok(v)
772}
773
774pub fn get_nth_struct_field(v: Value, n: usize) -> Result<Value, PartialVMError> {
775    let mut itr = v.value_as::<Struct>()?.unpack()?;
776    Ok(itr.nth(n).unwrap())
777}

779/// Returns the struct tag, non-annotated type layout, and fully annotated type layout of `ty`.
780pub(crate) fn get_tag_and_layouts(
781    context: &NativeContext,
782    ty: &Type,
783) -> PartialVMResult<Option<(StructTag, R::MoveTypeLayout, A::MoveTypeLayout)>> {
784    let tag = match context.type_to_type_tag(ty)? {
785        TypeTag::Struct(s) => s,
786        _ => {
787            return Err(
788                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
789                    .with_message("Sui verifier guarantees this is a struct".to_string()),
790            )
791        }
792    };
793    let Some(layout) = context.type_to_type_layout(ty)? else {
794        return Ok(None);
795    };
796    let Some(annotated_layout) = context.type_to_fully_annotated_layout(ty)? else {
797        return Ok(None);
798    };
799    Ok(Some((*tag, layout, annotated_layout)))
800}
