177    pub fn new(
178        object_resolver: &'a dyn ChildObjectResolver,
179        input_objects: BTreeMap<ObjectID, InputObject>,
180        is_metered: bool,
181        protocol_config: &ProtocolConfig,
182        metrics: Arc<LimitsMetrics>,
183        epoch_id: EpochId,
184    ) -> Self {
185        let mut input_object_owners = BTreeMap::new();
186        let mut root_version = BTreeMap::new();
187        for (id, input_object) in input_objects {
188            let InputObject {
189                contained_uids,
190                version,
191                owner,
192            } = input_object;
193            input_object_owners.insert(id, owner);
194            debug_assert!(contained_uids.contains(&id));
195            for contained_uid in contained_uids {
196                root_version.insert(contained_uid, version);
197            }
198        }
199        Self {
200            child_object_store: ChildObjectStore::new(
201                object_resolver,
202                root_version,
203                is_metered,
204                LocalProtocolConfig::new(protocol_config),
205                metrics.clone(),
206                epoch_id,
207            ),
208            test_inventories: TestInventories::new(),
209            state: ObjectRuntimeState {
210                input_objects: input_object_owners,
211                new_ids: Set::new(),
212                deleted_ids: Set::new(),
213                transfers: LinkedHashMap::new(),
214                events: vec![],
215                total_events_size: 0,
216                received: LinkedHashMap::new(),
217            },
218            is_metered,
219            local_config: LocalProtocolConfig::new(protocol_config),
220            metrics,
221        }
222    }

224    pub fn new_id(&mut self, id: ObjectID) -> PartialVMResult<()> {
225        // If metered, we use the metered limit (non system tx limit) as the hard limit
226        // This macro takes care of that
227        if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
228            self.is_metered,
229            self.state.new_ids.len(),
230            self.local_config.max_num_new_move_object_ids,
231            self.local_config.max_num_new_move_object_ids_system_tx,
232            self.metrics.excessive_new_move_object_ids
233        ) {
234            return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
235                .with_message(format!("Creating more than {} IDs is not allowed", lim))
236                .with_sub_status(
237                    VMMemoryLimitExceededSubStatusCode::NEW_ID_COUNT_LIMIT_EXCEEDED as u64,
238                ));
239        };
240
241        // remove from deleted_ids for the case in dynamic fields where the Field object was deleted
242        // and then re-added in a single transaction. In that case, we also skip adding it
243        // to new_ids.
244        if self.state.deleted_ids.remove(&id).is_none() {
245            // mark the id as new
246            self.state.new_ids.insert(id, ());
247        }
248        Ok(())
249    }
250
251    pub fn delete_id(&mut self, id: ObjectID) -> PartialVMResult<()> {
252        // This is defensive because `self.state.deleted_ids` may not indeed
253        // be called based on the `was_new` flag
254        // Metered transactions don't have limits for now
255
256        if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
257            self.is_metered,
258            self.state.deleted_ids.len(),
259            self.local_config.max_num_deleted_move_object_ids,
260            self.local_config.max_num_deleted_move_object_ids_system_tx,
261            self.metrics.excessive_deleted_move_object_ids
262        ) {
263            return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
264                .with_message(format!("Deleting more than {} IDs is not allowed", lim))
265                .with_sub_status(
266                    VMMemoryLimitExceededSubStatusCode::DELETED_ID_COUNT_LIMIT_EXCEEDED as u64,
267                ));
268        };
269
270        let was_new = self.state.new_ids.remove(&id).is_some();
271        if !was_new {
272            self.state.deleted_ids.insert(id, ());
273        }
274        Ok(())
275    }

277    pub fn transfer(
278        &mut self,
279        owner: Owner,
280        ty: Type,
281        obj: Value,
282    ) -> PartialVMResult<TransferResult> {
283        let id: ObjectID = get_object_id(obj.copy_value()?)?
284            .value_as::<AccountAddress>()?
285            .into();
286        // - An object is new if it is contained in the new ids or if it is one of the objects
287        //   created during genesis (the system state object or clock).
288        // - Otherwise, check the input objects for the previous owner
289        // - If it was not in the input objects, it must have been wrapped or must have been a
290        //   child object
291        let is_framework_obj = [
292            SUI_SYSTEM_STATE_OBJECT_ID,
293            SUI_CLOCK_OBJECT_ID,
294            SUI_AUTHENTICATOR_STATE_OBJECT_ID,
295        ]
296        .contains(&id);
297        let transfer_result = if self.state.new_ids.contains_key(&id) {
298            TransferResult::New
299        } else if is_framework_obj {
300            // framework objects are always created when they are transferred, but the id is
301            // hard-coded so it is not yet in new_ids
302            self.state.new_ids.insert(id, ());
303            TransferResult::New
304        } else if let Some(prev_owner) = self.state.input_objects.get(&id) {
305            match (&owner, prev_owner) {
306                // don't use == for dummy values in Shared owner
307                (Owner::Shared { .. }, Owner::Shared { .. }) => TransferResult::SameOwner,
308                (new, old) if new == old => TransferResult::SameOwner,
309                _ => TransferResult::OwnerChanged,
310            }
311        } else {
312            TransferResult::OwnerChanged
313        };
314
315        // Metered transactions don't have limits for now
316
317        if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
318            // TODO: is this not redundant? Metered TX implies framework obj cannot be transferred
319            self.is_metered && !is_framework_obj, // We have higher limits for unmetered transactions and framework obj
320            self.state.transfers.len(),
321            self.local_config.max_num_transferred_move_object_ids,
322            self.local_config
323                .max_num_transferred_move_object_ids_system_tx,
324            self.metrics.excessive_transferred_move_object_ids
325        ) {
326            return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
327                .with_message(format!("Transferring more than {} IDs is not allowed", lim))
328                .with_sub_status(
329                    VMMemoryLimitExceededSubStatusCode::TRANSFER_ID_COUNT_LIMIT_EXCEEDED as u64,
330                ));
331        };
332
333        self.state.transfers.insert(id, (owner, ty, obj));
334        Ok(transfer_result)
335    }

337    pub fn emit_event(&mut self, ty: Type, tag: StructTag, event: Value) -> PartialVMResult<()> {
338        if self.state.events.len() >= (self.local_config.max_num_event_emit as usize) {
339            return Err(max_event_error(self.local_config.max_num_event_emit));
340        }
341        self.state.events.push((ty, tag, event));
342        Ok(())
343    }

367    pub(super) fn receive_object(
368        &mut self,
369        parent: ObjectID,
370        child: ObjectID,
371        child_version: SequenceNumber,
372        child_ty: &Type,
373        child_layout: &R::MoveTypeLayout,
374        child_fully_annotated_layout: &MoveTypeLayout,
375        child_move_type: MoveObjectType,
376    ) -> PartialVMResult<Option<ObjectResult<Value>>> {
377        let Some((value, obj_meta)) = self.child_object_store.receive_object(
378            parent,
379            child,
380            child_version,
381            child_ty,
382            child_layout,
383            child_fully_annotated_layout,
384            child_move_type,
385        )?
386        else {
387            return Ok(None);
388        };
389        // NB: It is important that the object only be added to the received set after it has been
390        // fully authenticated and loaded.
391        if self.state.received.insert(child, obj_meta).is_some() {
392            // We should never hit this -- it means that we have received the same object twice which
393            // means we have a duplicated a receiving ticket somehow.
394            return Err(
395                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(format!(
396                    "Object {child} at version {child_version} already received. This can only happen \
397                    if multiple `Receiving` arguments exist for the same object in the transaction which is impossible."
398                )),
399            );
400        }
401        Ok(Some(value))
402    }

456    pub fn loaded_runtime_objects(&self) -> BTreeMap<ObjectID, DynamicallyLoadedObjectMetadata> {
457        // The loaded child objects, and the received objects, should be disjoint. If they are not,
458        // this is an error since it could lead to incorrect transaction dependency computations.
459        debug_assert!(self
460            .child_object_store
461            .cached_objects()
462            .keys()
463            .all(|id| !self.state.received.contains_key(id)));
464        self.child_object_store
465            .cached_objects()
466            .iter()
467            .filter_map(|(id, obj_opt)| {
468                obj_opt.as_ref().map(|obj| {
469                    (
470                        *id,
471                        DynamicallyLoadedObjectMetadata {
472                            version: obj.version(),
473                            digest: obj.digest(),
474                            storage_rebate: obj.storage_rebate,
475                            owner: obj.owner.clone(),
476                            previous_transaction: obj.previous_transaction,
477                        },
478                    )
479                })
480            })
481            .chain(
482                self.state
483                    .received
484                    .iter()
485                    .map(|(id, meta)| (*id, meta.clone())),
486            )
487            .collect()
488    }

500impl ObjectRuntimeState {
501    /// Update `state_view` with the effects of successfully executing a transaction:
502    /// - Given the effects `Op<Value>` of child objects, processes the changes in terms of
503    ///   object writes/deletes
504    /// - Process `transfers` and `input_objects` to determine whether the type of change
505    ///   (WriteKind) to the object
506    /// - Process `deleted_ids` with previously determined information to determine the
507    ///   DeleteKind
508    /// - Passes through user events
509    pub(crate) fn finish(
510        mut self,
511        loaded_child_objects: BTreeMap<ObjectID, DynamicallyLoadedObjectMetadata>,
512        child_object_effects: BTreeMap<ObjectID, ChildObjectEffect>,
513    ) -> Result<RuntimeResults, ExecutionError> {
514        let mut loaded_child_objects: BTreeMap<_, _> = loaded_child_objects
515            .into_iter()
516            .map(|(id, metadata)| {
517                (
518                    id,
519                    LoadedRuntimeObject {
520                        version: metadata.version,
521                        is_modified: false,
522                    },
523                )
524            })
525            .collect();
526        for (child, child_object_effect) in child_object_effects {
527            let ChildObjectEffect {
528                owner: parent,
529                ty,
530                effect,
531            } = child_object_effect;
532
533            if let Some(loaded_child) = loaded_child_objects.get_mut(&child) {
534                loaded_child.is_modified = true;
535            }
536
537            match effect {
538                // was modified, so mark it as mutated and transferred
539                Op::Modify(v) => {
540                    debug_assert!(!self.transfers.contains_key(&child));
541                    debug_assert!(!self.new_ids.contains_key(&child));
542                    debug_assert!(loaded_child_objects.contains_key(&child));
543                    self.transfers
544                        .insert(child, (Owner::ObjectOwner(parent.into()), ty, v));
545                }
546
547                Op::New(v) => {
548                    debug_assert!(!self.transfers.contains_key(&child));
549                    self.transfers
550                        .insert(child, (Owner::ObjectOwner(parent.into()), ty, v));
551                }
552
553                Op::Delete => {
554                    // was transferred so not actually deleted
555                    if self.transfers.contains_key(&child) {
556                        debug_assert!(!self.deleted_ids.contains_key(&child));
557                    }
558                    // ID was deleted too was deleted so mark as deleted
559                    if self.deleted_ids.contains_key(&child) {
560                        debug_assert!(!self.transfers.contains_key(&child));
561                        debug_assert!(!self.new_ids.contains_key(&child));
562                    }
563                }
564            }
565        }
566        let ObjectRuntimeState {
567            input_objects: _,
568            new_ids,
569            deleted_ids,
570            transfers,
571            events: user_events,
572            total_events_size: _,
573            received,
574        } = self;
575
576        // Check new owners from transfers, reports an error on cycles.
577        // TODO can we have cycles in the new system?
578        check_circular_ownership(
579            transfers
580                .iter()
581                .map(|(id, (owner, _, _))| (*id, owner.clone())),
582        )?;
583        // For both written_objects and deleted_ids, we need to mark the loaded child object as modified.
584        // These may not be covered in the child object effects if they are taken out in one PT command and then
585        // transferred/deleted in a different command. Marking them as modified will allow us properly determine their
586        // mutation category in effects.
587        // TODO: This could get error-prone quickly: what if we forgot to mark an object as modified? There may be a cleaner
588        // sulution.
589        let written_objects: LinkedHashMap<_, _> = transfers
590            .into_iter()
591            .map(|(id, (owner, type_, value))| {
592                if let Some(loaded_child) = loaded_child_objects.get_mut(&id) {
593                    loaded_child.is_modified = true;
594                }
595                (id, (owner, type_, value))
596            })
597            .collect();
598        for deleted_id in deleted_ids.keys() {
599            if let Some(loaded_child) = loaded_child_objects.get_mut(deleted_id) {
600                loaded_child.is_modified = true;
601            }
602        }
603
604        // Any received objects are viewed as modified. They had to be loaded in order to be
605        // received so they must be in the loaded_child_objects map otherwise it's an invariant
606        // violation.
607        for (received_object, _) in received.into_iter() {
608            match loaded_child_objects.get_mut(&received_object) {
609                Some(loaded_child) => {
610                    loaded_child.is_modified = true;
611                }
612                None => {
613                    return Err(ExecutionError::invariant_violation(format!(
614                        "Failed to find received UID {received_object} in loaded child objects."
615                    )))
616                }
617            }
618        }
619
620        Ok(RuntimeResults {
621            writes: written_objects,
622            user_events,
623            loaded_child_objects,

638fn check_circular_ownership(
639    transfers: impl IntoIterator<Item = (ObjectID, Owner)>,
640) -> Result<(), ExecutionError> {
641    let mut object_owner_map = BTreeMap::new();
642    for (id, recipient) in transfers {
643        object_owner_map.remove(&id);
644        match recipient {
645            Owner::AddressOwner(_) | Owner::Shared { .. } | Owner::Immutable => (),
646            Owner::ObjectOwner(new_owner) => {
647                let new_owner: ObjectID = new_owner.into();
648                let mut cur = new_owner;
649                loop {
650                    if cur == id {
651                        return Err(ExecutionError::from_kind(
652                            ExecutionErrorKind::CircularObjectOwnership { object: cur },
653                        ));
654                    }
655                    if let Some(parent) = object_owner_map.get(&cur) {
656                        cur = *parent;
657                    } else {
658                        break;
659                    }
660                }
661                object_owner_map.insert(id, new_owner);
662            }
663            Owner::ConsensusAddressOwner { .. } => {
664                unimplemented!("ConsensusAddressOwner does not exist for this execution version")
665            }
666        }
667    }
668    Ok(())
669}

671/// WARNING! This function assumes that the bcs bytes have already been validated,
672/// and it will give an invariant violation otherwise.
673/// In short, we are relying on the invariant that the bytes are valid for objects
674/// in storage.  We do not need this invariant for dev-inspect, as the programmable
675/// transaction execution will validate the bytes before we get to this point.
676pub fn get_all_uids(
677    fully_annotated_layout: &MoveTypeLayout,
678    bcs_bytes: &[u8],
679) -> Result<BTreeSet<ObjectID>, /* invariant violation */ String> {
680    let mut ids = BTreeSet::new();
681    struct UIDTraversal<'i>(&'i mut BTreeSet<ObjectID>);
682    struct UIDCollector<'i>(&'i mut BTreeSet<ObjectID>);
683
684    impl<'b, 'l> AV::Traversal<'b, 'l> for UIDTraversal<'_> {
685        type Error = AV::Error;
686
687        fn traverse_struct(
688            &mut self,
689            driver: &mut AV::StructDriver<'_, 'b, 'l>,
690        ) -> Result<(), Self::Error> {
691            if driver.struct_layout().type_ == UID::type_() {
692                while driver.next_field(&mut UIDCollector(self.0))?.is_some() {}
693            } else {
694                while driver.next_field(self)?.is_some() {}
695            }
696            Ok(())
697        }
698    }
699
700    impl<'b, 'l> AV::Traversal<'b, 'l> for UIDCollector<'_> {
701        type Error = AV::Error;
702        fn traverse_address(
703            &mut self,
704            _driver: &AV::ValueDriver<'_, 'b, 'l>,
705            value: AccountAddress,
706        ) -> Result<(), Self::Error> {
707            self.0.insert(value.into());
708            Ok(())
709        }
710    }
711
712    MoveValue::visit_deserialize(
713        bcs_bytes,
714        fully_annotated_layout,
715        &mut UIDTraversal(&mut ids),
716    )
717    .map_err(|e| format!("Failed to deserialize. {e}"))?;
718    Ok(ids)
719}
