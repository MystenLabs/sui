41// This function updates the inventories based on the transfers and deletes that occurred in the
42// transaction
43// native fun end_transaction(): TransactionResult;
44pub fn end_transaction(
45    context: &mut NativeContext,
46    ty_args: Vec<Type>,
47    args: VecDeque<Value>,
48) -> PartialVMResult<NativeResult> {
49    assert!(ty_args.is_empty());
50    assert!(args.is_empty());
51    let object_runtime_ref: &mut ObjectRuntime = context.extensions_mut().get_mut();
52    let taken_shared_or_imm: BTreeMap<_, _> = object_runtime_ref
53        .test_inventories
54        .taken
55        .iter()
56        .filter(|(_id, owner)| matches!(owner, Owner::Shared { .. } | Owner::Immutable))
57        .map(|(id, owner)| (*id, owner.clone()))
58        .collect();
59    // set to true if a shared or imm object was:
60    // - transferred in a way that changes it from its original shared/imm state
61    // - wraps the object
62    // if true, we will "abort"
63    let mut incorrect_shared_or_imm_handling = false;
64
65    let object_runtime_state = object_runtime_ref.take_state();
66    // Determine writes and deletes
67    // We pass an empty map as we do not expose dynamic field objects in the system
68    let results = object_runtime_state.finish(BTreeMap::new(), BTreeMap::new());
69    let RuntimeResults {
70        writes,
71        user_events,
72        loaded_child_objects: _,
73        created_object_ids,
74        deleted_object_ids,
75    } = match results {
76        Ok(res) => res,
77        Err(_) => {
78            return Ok(NativeResult::err(
79                legacy_test_cost(),
80                E_COULD_NOT_GENERATE_EFFECTS,
81            ));
82        }
83    };
84    let all_active_child_objects = object_runtime_ref
85        .all_active_child_objects()
86        .map(|(id, _, _)| *id)
87        .collect::<BTreeSet<_>>();
88    let inventories = &mut object_runtime_ref.test_inventories;
89    let mut new_object_values = LinkedHashMap::new();
90    let mut transferred = vec![];
91    // cleanup inventories
92    // we will remove all changed objects
93    // - deleted objects need to be removed to mark deletions
94    // - written objects are removed and later replaced to mark new values and new owners
95    // - child objects will not be reflected in transfers, but need to be no longer retrievable
96    for id in deleted_object_ids
97        .keys()
98        .chain(writes.keys())
99        .chain(&all_active_child_objects)
100    {
101        for addr_inventory in inventories.address_inventories.values_mut() {
102            for s in addr_inventory.values_mut() {
103                s.remove(id);
104            }
105        }
106        for s in &mut inventories.shared_inventory.values_mut() {
107            s.remove(id);
108        }
109        for s in &mut inventories.immutable_inventory.values_mut() {
110            s.remove(id);
111        }
112        inventories.taken.remove(id);
113    }
114    // handle transfers, inserting transferred/written objects into their respective inventory
115    let mut created = vec![];
116    let mut written = vec![];
117    for (id, (owner, ty, value)) in writes {
118        new_object_values.insert(id, (ty.clone(), value.copy_value().unwrap()));
119        transferred.push((id, owner.clone()));
120        incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
121            || taken_shared_or_imm
122                .get(&id)
123                .map(|shared_or_imm_owner| shared_or_imm_owner != &owner)
124                .unwrap_or(/* not incorrect */ false);
125        if created_object_ids.contains_key(&id) {
126            created.push(id);
127        } else {
128            written.push(id);
129        }
130        match owner {
131            Owner::AddressOwner(a) => {
132                inventories
133                    .address_inventories
134                    .entry(a)
135                    .or_default()
136                    .entry(ty)
137                    .or_default()
138                    .insert(id, ());
139            }
140            Owner::ObjectOwner(_) => (),
141            Owner::Shared { .. } => {
142                inventories
143                    .shared_inventory
144                    .entry(ty)
145                    .or_default()
146                    .insert(id, ());
147            }
148            Owner::Immutable => {
149                inventories
150                    .immutable_inventory
151                    .entry(ty)
152                    .or_default()
153                    .insert(id, ());
154            }
155            Owner::ConsensusAddressOwner { .. } => {
156                unimplemented!("ConsensusAddressOwner does not exist for this execution version")
157            }
158        }
159    }
160    // deletions already handled above, but we drop the delete kind for the effects
161    let mut deleted = vec![];
162    for (id, _) in deleted_object_ids {
163        incorrect_shared_or_imm_handling =
164            incorrect_shared_or_imm_handling || taken_shared_or_imm.contains_key(&id);
165        deleted.push(id);
166    }
167    // find all wrapped objects
168    let mut all_wrapped = BTreeSet::new();
169    let object_runtime_ref: &ObjectRuntime = context.extensions().get();
170    find_all_wrapped_objects(
171        context,
172        &mut all_wrapped,
173        new_object_values
174            .iter()
175            .map(|(id, (ty, value))| (id, ty, value)),
176    );
177    find_all_wrapped_objects(
178        context,
179        &mut all_wrapped,
180        object_runtime_ref.all_active_child_objects(),
181    );
182    // mark as "incorrect" if a shared/imm object was wrapped or is a child object
183    incorrect_shared_or_imm_handling = incorrect_shared_or_imm_handling
184        || taken_shared_or_imm
185            .keys()
186            .any(|id| all_wrapped.contains(id) || all_active_child_objects.contains(id));
187    // if incorrect handling, return with an 'abort'
188    if incorrect_shared_or_imm_handling {
189        return Ok(NativeResult::err(
190            legacy_test_cost(),
191            E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
192        ));
193    }
194
195    // mark all wrapped as deleted
196    for wrapped in all_wrapped {
197        deleted.push(wrapped)
198    }
199
200    // new input objects are remaining taken objects not written/deleted
201    let object_runtime_ref: &mut ObjectRuntime = context.extensions_mut().get_mut();
202    object_runtime_ref.state.input_objects = object_runtime_ref
203        .test_inventories
204        .taken
205        .iter()
206        .map(|(id, owner)| (*id, owner.clone()))
207        .collect::<BTreeMap<_, _>>();
208    // update inventories
209    // check for bad updates to immutable values
210    for (id, (ty, value)) in new_object_values {
211        debug_assert!(!all_active_child_objects.contains(&id));
212        if let Some(prev_value) = object_runtime_ref
213            .test_inventories
214            .taken_immutable_values
215            .get(&ty)
216            .and_then(|values| values.get(&id))
217        {
218            if !value.equals(prev_value)? {
219                return Ok(NativeResult::err(
220                    legacy_test_cost(),
221                    E_INVALID_SHARED_OR_IMMUTABLE_USAGE,
222                ));
223            }
224        }
225        object_runtime_ref
226            .test_inventories
227            .objects
228            .insert(id, value);
229    }
230    // remove deleted
231    for id in &deleted {
232        object_runtime_ref.test_inventories.objects.remove(id);
233    }
234    // remove active child objects
235    for id in all_active_child_objects {
236        object_runtime_ref.test_inventories.objects.remove(&id);
237    }
238
239    let effects = transaction_effects(
240        created,
241        written,
242        deleted,
243        transferred,
244        user_events.len() as u64,
245    );
246    Ok(NativeResult::ok(legacy_test_cost(), smallvec![effects]))
247}

249// native fun take_from_address_by_id<T: key>(account: address, id: ID): T;
250pub fn take_from_address_by_id(
251    context: &mut NativeContext,
252    ty_args: Vec<Type>,
253    mut args: VecDeque<Value>,
254) -> PartialVMResult<NativeResult> {
255    let specified_ty = get_specified_ty(ty_args);
256    let id = pop_id(&mut args)?;
257    let account: SuiAddress = pop_arg!(args, AccountAddress).into();
258    pop_arg!(args, StructRef);
259    assert!(args.is_empty());
260    let object_runtime: &mut ObjectRuntime = context.extensions_mut().get_mut();
261    let inventories = &mut object_runtime.test_inventories;
262    let res = take_from_inventory(
263        |x| {
264            inventories
265                .address_inventories
266                .get(&account)
267                .and_then(|inv| inv.get(&specified_ty))
268                .map(|s| s.contains_key(x))
269                .unwrap_or(false)
270        },
271        &inventories.objects,
272        &mut inventories.taken,
273        &mut object_runtime.state.input_objects,
274        id,
275        Owner::AddressOwner(account),
276    );
277    Ok(match res {
278        Ok(value) => NativeResult::ok(legacy_test_cost(), smallvec![value]),
279        Err(native_err) => native_err,
280    })
281}

507fn take_from_inventory(
508    is_in_inventory: impl FnOnce(&ObjectID) -> bool,
509    objects: &BTreeMap<ObjectID, Value>,
510    taken: &mut BTreeMap<ObjectID, Owner>,
511    input_objects: &mut BTreeMap<ObjectID, Owner>,
512    id: ObjectID,
513    owner: Owner,
514) -> Result<Value, NativeResult> {
515    let obj_opt = objects.get(&id);
516    let is_taken = taken.contains_key(&id);
517    if is_taken || !is_in_inventory(&id) || obj_opt.is_none() {
518        return Err(NativeResult::err(
519            legacy_test_cost(),
520            E_OBJECT_NOT_FOUND_CODE,
521        ));
522    }
523    taken.insert(id, owner.clone());
524    input_objects.insert(id, owner);
525    let obj = obj_opt.unwrap();
526    Ok(obj.copy_value().unwrap())
527}

529fn most_recent_at_ty(
530    taken: &BTreeMap<ObjectID, Owner>,
531    inv: &BTreeMap<Type, Set<ObjectID>>,
532    ty: Type,
533) -> Value {
534    pack_option(most_recent_at_ty_opt(taken, inv, ty))
535}
536
537fn most_recent_at_ty_opt(
538    taken: &BTreeMap<ObjectID, Owner>,
539    inv: &BTreeMap<Type, Set<ObjectID>>,
540    ty: Type,
541) -> Option<Value> {
542    let s = inv.get(&ty)?;
543    let most_recent_id = s.keys().filter(|id| !taken.contains_key(id)).next_back()?;
544    Some(pack_id(*most_recent_id))
545}

547fn get_specified_ty(mut ty_args: Vec<Type>) -> Type {
548    assert!(ty_args.len() == 1);
549    ty_args.pop().unwrap()
550}

552// helpers
553fn pop_id(args: &mut VecDeque<Value>) -> PartialVMResult<ObjectID> {
554    let v = match args.pop_back() {
555        None => {
556            return Err(PartialVMError::new(
557                StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR,
558            ))
559        }
560        Some(v) => v,
561    };
562    Ok(get_nth_struct_field(v, 0)?
563        .value_as::<AccountAddress>()?
564        .into())
565}
566
567fn pack_id(a: impl Into<AccountAddress>) -> Value {
568    Value::struct_(values::Struct::pack(vec![Value::address(a.into())]))
569}
570
571fn pack_ids(items: impl IntoIterator<Item = impl Into<AccountAddress>>) -> Value {
572    Value::vector_for_testing_only(items.into_iter().map(pack_id))
573}
574
575fn pack_vec_map(items: impl IntoIterator<Item = (Value, Value)>) -> Value {
576    Value::struct_(values::Struct::pack(vec![Value::vector_for_testing_only(
577        items
578            .into_iter()
579            .map(|(k, v)| Value::struct_(values::Struct::pack(vec![k, v]))),
580    )]))
581}
582
583fn transaction_effects(
584    created: impl IntoIterator<Item = impl Into<AccountAddress>>,
585    written: impl IntoIterator<Item = impl Into<AccountAddress>>,
586    deleted: impl IntoIterator<Item = impl Into<AccountAddress>>,
587    transferred: impl IntoIterator<Item = (ObjectID, Owner)>,
588    num_events: u64,
589) -> Value {
590    let mut transferred_to_account = vec![];
591    let mut transferred_to_object = vec![];
592    let mut shared = vec![];
593    let mut frozen = vec![];
594    for (id, owner) in transferred {
595        match owner {
596            Owner::AddressOwner(a) => {
597                transferred_to_account.push((pack_id(id), Value::address(a.into())))
598            }
599            Owner::ObjectOwner(o) => transferred_to_object.push((pack_id(id), pack_id(o))),
600            Owner::Shared { .. } => shared.push(id),
601            Owner::Immutable => frozen.push(id),
602            Owner::ConsensusAddressOwner { .. } => {
603                unimplemented!("ConsensusAddressOwner does not exist for this execution version")
604            }
605        }
606    }
607
608    let created_field = pack_ids(created);
609    let written_field = pack_ids(written);
610    let deleted_field = pack_ids(deleted);
611    let transferred_to_account_field = pack_vec_map(transferred_to_account);
612    let transferred_to_object_field = pack_vec_map(transferred_to_object);
613    let shared_field = pack_ids(shared);
614    let frozen_field = pack_ids(frozen);
615    let num_events_field = Value::u64(num_events);
616    Value::struct_(values::Struct::pack(vec![
617        created_field,
618        written_field,
619        deleted_field,
620        transferred_to_account_field,
621        transferred_to_object_field,
622        shared_field,
623        frozen_field,
624        num_events_field,
625    ]))
626}
627
628fn pack_option(opt: Option<Value>) -> Value {
629    let item = match opt {
630        Some(v) => vec![v],
631        None => vec![],
632    };
633    Value::struct_(values::Struct::pack(vec![Value::vector_for_testing_only(
634        item,
635    )]))
636}

638fn find_all_wrapped_objects<'a, 'i>(
639    context: &NativeContext,
640    ids: &'i mut BTreeSet<ObjectID>,
641    new_object_values: impl IntoIterator<Item = (&'a ObjectID, &'a Type, impl Borrow<Value>)>,
642) {
643    #[derive(Copy, Clone)]
644    enum LookingFor {
645        Wrapped,
646        Uid,
647        Address,
648    }
649
650    struct Traversal<'i, 'u> {
651        state: LookingFor,
652        ids: &'i mut BTreeSet<ObjectID>,
653        uid: &'u MoveStructLayout,
654    }
655
656    impl<'b, 'l> AV::Traversal<'b, 'l> for Traversal<'_, '_> {
657        type Error = AV::Error;
658
659        fn traverse_struct(
660            &mut self,
661            driver: &mut AV::StructDriver<'_, 'b, 'l>,
662        ) -> Result<(), Self::Error> {
663            match self.state {
664                // We're at the top-level of the traversal, looking for an object to recurse into.
665                // We can unconditionally switch to looking for UID fields at the level below,
666                // because we know that all the top-level values are objects.
667                LookingFor::Wrapped => {
668                    while driver
669                        .next_field(&mut Traversal {
670                            state: LookingFor::Uid,
671                            ids: self.ids,
672                            uid: self.uid,
673                        })?
674                        .is_some()
675                    {}
676                }
677
678                // We are looking for UID fields. If we find one (which we confirm by checking its
679                // layout), switch to looking for addresses in its sub-structure.
680                LookingFor::Uid => {
681                    while let Some(MoveFieldLayout { name: _, layout }) = driver.peek_field() {
682                        if matches!(layout, MoveTypeLayout::Struct(s) if s.as_ref() == self.uid) {
683                            driver.next_field(&mut Traversal {
684                                state: LookingFor::Address,
685                                ids: self.ids,
686                                uid: self.uid,
687                            })?;
688                        } else {
689                            driver.next_field(self)?;
690                        }
691                    }
692                }
693
694                // When looking for addresses, recurse through structs, as the address is nested
695                // within the UID.
696                LookingFor::Address => while driver.next_field(self)?.is_some() {},
697            }
698
699            Ok(())
700        }
701
702        fn traverse_address(
703            &mut self,
704            _: &AV::ValueDriver<'_, 'b, 'l>,
705            address: AccountAddress,
706        ) -> Result<(), Self::Error> {
707            // If we're looking for addresses, and we found one, then save it.
708            if matches!(self.state, LookingFor::Address) {
709                self.ids.insert(address.into());
710            }
711            Ok(())
712        }
713    }
714
715    let uid = UID::layout();
716    for (_id, ty, value) in new_object_values {
717        let Ok(Some(layout)) = context.type_to_type_layout(ty) else {
718            debug_assert!(false);
719            continue;
720        };
721
722        let Ok(Some(annotated_layout)) = context.type_to_fully_annotated_layout(ty) else {
723            debug_assert!(false);
724            continue;
725        };
726
727        let blob = value.borrow().simple_serialize(&layout).unwrap();
728        MoveValue::visit_deserialize(
729            &blob,
730            &annotated_layout,
731            &mut Traversal {
732                state: LookingFor::Wrapped,
733                ids,
734                uid: &uid,
735            },
736        )
737        .unwrap();
738    }
739}
