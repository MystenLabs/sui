4use move_binary_format::errors::PartialVMResult;
5use move_core_types::{account_address::AccountAddress, gas_algebra::InternalGas};
6use move_vm_runtime::{native_charge_gas_early_exit, native_functions::NativeContext};
7use move_vm_types::{
8    loaded_data::runtime_types::Type, natives::function::NativeResult, pop_arg, values::Value,
9};
10use smallvec::smallvec;
11use std::{collections::VecDeque, convert::TryFrom};
12use sui_types::base_types::{ObjectID, TransactionDigest};
13
14use crate::{object_runtime::ObjectRuntime, NativesCostTable};
15
16#[derive(Clone)]
17pub struct TxContextDeriveIdCostParams {
18    pub tx_context_derive_id_cost_base: InternalGas,
19}
20/***************************************************************************************************
21 * native fun derive_id
22 * Implementation of the Move native function `fun derive_id(tx_hash: vector<u8>, ids_created: u64): address`
23 *   gas cost: tx_context_derive_id_cost_base                | we operate on fixed size data structures
24 **************************************************************************************************/
25pub fn derive_id(
26    context: &mut NativeContext,
27    ty_args: Vec<Type>,
28    mut args: VecDeque<Value>,
29) -> PartialVMResult<NativeResult> {
30    debug_assert!(ty_args.is_empty());
31    debug_assert!(args.len() == 2);
32
33    let tx_context_derive_id_cost_params = context
34        .extensions_mut()
35        .get::<NativesCostTable>()
36        .tx_context_derive_id_cost_params
37        .clone();
38    native_charge_gas_early_exit!(
39        context,
40        tx_context_derive_id_cost_params.tx_context_derive_id_cost_base
41    );
42
43    let ids_created = pop_arg!(args, u64);
44    let tx_hash = pop_arg!(args, Vec<u8>);
45
46    // unwrap safe because all digests in Move are serialized from the Rust `TransactionDigest`
47    let digest = TransactionDigest::try_from(tx_hash.as_slice()).unwrap();
48    let address = AccountAddress::from(ObjectID::derive_id(digest, ids_created));
49    let obj_runtime: &mut ObjectRuntime = context.extensions_mut().get_mut();
50    obj_runtime.new_id(address.into())?;
51
52    Ok(NativeResult::ok(
53        context.gas_used(),
54        smallvec![Value::address(address)],
55    ))
56}
