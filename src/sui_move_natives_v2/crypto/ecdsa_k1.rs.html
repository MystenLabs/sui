50/***************************************************************************************************
51 * native fun secp256k1_ecrecover
52 * Implementation of the Move native function `secp256k1_ecrecover(signature: &vector<u8>, msg: &vector<u8>, hash: u8): vector<u8>`
53 * This function has two cost modes depending on the hash being set to `KECCAK256` or `SHA256`. The core formula is same but constants differ.
54 * If hash = 0, we use the `keccak256` cost constants, otherwise we use the `sha256` cost constants.
55 *   gas cost: ecdsa_k1_ecrecover_cost_base                    | covers various fixed costs in the oper
56 *              + ecdsa_k1_ecrecover_msg_cost_per_byte    * size_of(msg)        | covers cost of operating on each byte of `msg`
57 *              + ecdsa_k1_ecrecover_msg_cost_per_block   * num_blocks(msg)     | covers cost of operating on each block in `msg`
58 * Note: each block is of size `KECCAK256_BLOCK_SIZE` bytes for `keccak256` and `SHA256_BLOCK_SIZE` for `sha256`, and we round up.
59 *       `signature` is fixed size, so the cost is included in the base cost.
60 **************************************************************************************************/
61pub fn ecrecover(
62    context: &mut NativeContext,
63    ty_args: Vec<Type>,
64    mut args: VecDeque<Value>,
65) -> PartialVMResult<NativeResult> {
66    debug_assert!(ty_args.is_empty());
67    debug_assert!(args.len() == 3);
68
69    let hash = pop_arg!(args, u8);
70
71    // Load the cost parameters from the protocol config
72    let (ecdsa_k1_ecrecover_cost_params, crypto_invalid_arguments_cost) = {
73        let cost_table = &context.extensions().get::<NativesCostTable>();
74        (
75            cost_table.ecdsa_k1_ecrecover_cost_params.clone(),
76            cost_table.crypto_invalid_arguments_cost,
77        )
78    };
79    let (base_cost, cost_per_byte, cost_per_block, block_size) = match hash {
80        KECCAK256 => (
81            ecdsa_k1_ecrecover_cost_params.ecdsa_k1_ecrecover_keccak256_cost_base,
82            ecdsa_k1_ecrecover_cost_params.ecdsa_k1_ecrecover_keccak256_msg_cost_per_byte,
83            ecdsa_k1_ecrecover_cost_params.ecdsa_k1_ecrecover_keccak256_msg_cost_per_block,
84            KECCAK256_BLOCK_SIZE,
85        ),
86        SHA256 => (
87            ecdsa_k1_ecrecover_cost_params.ecdsa_k1_ecrecover_sha256_cost_base,
88            ecdsa_k1_ecrecover_cost_params.ecdsa_k1_ecrecover_sha256_msg_cost_per_byte,
89            ecdsa_k1_ecrecover_cost_params.ecdsa_k1_ecrecover_sha256_msg_cost_per_block,
90            SHA256_BLOCK_SIZE,
91        ),
92        _ => {
93            // Charge for failure but dont fail if we run out of gas otherwise the actual error is masked by OUT_OF_GAS error
94            context.charge_gas(crypto_invalid_arguments_cost);
95            return Ok(NativeResult::err(
96                context.gas_used(),
97                FAIL_TO_RECOVER_PUBKEY,
98            ));
99        }
100    };
101
102    // Charge the base cost for this oper
103    native_charge_gas_early_exit!(context, base_cost);
104
105    let msg = pop_arg!(args, VectorRef);
106    let signature = pop_arg!(args, VectorRef);
107
108    let msg_ref = msg.as_bytes_ref();
109    let signature_ref = signature.as_bytes_ref();
110
111    // Charge the arg size dependent costs
112    native_charge_gas_early_exit!(
113        context,
114        cost_per_byte * (msg_ref.len() as u64).into()
115            + cost_per_block * (msg_ref.len().div_ceil(block_size) as u64).into()
116    );
117
118    let cost = context.gas_used();
119
120    let Ok(sig) = <Secp256k1RecoverableSignature as ToFromBytes>::from_bytes(&signature_ref) else {
121        return Ok(NativeResult::err(cost, INVALID_SIGNATURE));
122    };
123
124    let pk = match hash {
125        KECCAK256 => sig.recover_with_hash::<Keccak256>(&msg_ref),
126        SHA256 => sig.recover_with_hash::<Sha256>(&msg_ref),
127        _ => Err(FastCryptoError::InvalidInput), // We should never reach here
128    };
129
130    match pk {
131        Ok(pk) => Ok(NativeResult::ok(
132            cost,
133            smallvec![Value::vector_u8(pk.as_bytes().to_vec())],
134        )),
135        Err(_) => Ok(NativeResult::err(cost, FAIL_TO_RECOVER_PUBKEY)),
136    }
137}

143pub fn decompress_pubkey(
144    context: &mut NativeContext,
145    ty_args: Vec<Type>,
146    mut args: VecDeque<Value>,
147) -> PartialVMResult<NativeResult> {
148    debug_assert!(ty_args.is_empty());
149    debug_assert!(args.len() == 1);
150
151    // Load the cost parameters from the protocol config
152    let ecdsa_k1_decompress_pubkey_cost_params = &context
153        .extensions()
154        .get::<NativesCostTable>()
155        .ecdsa_k1_decompress_pubkey_cost_params
156        .clone();
157    // Charge the base cost for this oper
158    native_charge_gas_early_exit!(
159        context,
160        ecdsa_k1_decompress_pubkey_cost_params.ecdsa_k1_decompress_pubkey_cost_base
161    );
162
163    let pubkey = pop_arg!(args, VectorRef);
164    let pubkey_ref = pubkey.as_bytes_ref();
165
166    let cost = context.gas_used();
167
168    match Secp256k1PublicKey::from_bytes(&pubkey_ref) {
169        Ok(pubkey) => {
170            let uncompressed = &pubkey.pubkey.serialize_uncompressed();
171            Ok(NativeResult::ok(
172                cost,
173                smallvec![Value::vector_u8(uncompressed.to_vec())],
174            ))
175        }
176        Err(_) => Ok(NativeResult::err(cost, INVALID_PUBKEY)),
177    }
178}

196/***************************************************************************************************
197 * native fun secp256k1_verify
198 * Implementation of the Move native function `secp256k1_verify(signature: &vector<u8>, public_key: &vector<u8>, msg: &vector<u8>, hash: u8): bool`
199 * This function has two cost modes depending on the hash being set to`KECCAK256` or `SHA256`. The core formula is same but constants differ.
200 * If hash = 0, we use the `keccak256` cost constants, otherwise we use the `sha256` cost constants.
201 *   gas cost: ecdsa_k1_secp256k1_verify_cost_base                    | covers various fixed costs in the oper
202 *              + ecdsa_k1_secp256k1_verify_msg_cost_per_byte    * size_of(msg)        | covers cost of operating on each byte of `msg`
203 *              + ecdsa_k1_secp256k1_verify_msg_cost_per_block   * num_blocks(msg)     | covers cost of operating on each block in `msg`
204 * Note: each block is of size `KECCAK256_BLOCK_SIZE` bytes for `keccak256` and `SHA256_BLOCK_SIZE` for `sha256`, and we round up.
205 *       `signature` and `public_key` are fixed size, so their costs are included in the base cost.
206 **************************************************************************************************/
207pub fn secp256k1_verify(
208    context: &mut NativeContext,
209    ty_args: Vec<Type>,
210    mut args: VecDeque<Value>,
211) -> PartialVMResult<NativeResult> {
212    debug_assert!(ty_args.is_empty());
213    debug_assert!(args.len() == 4);
214
215    let hash = pop_arg!(args, u8);
216
217    // Load the cost parameters from the protocol config
218    let (ecdsa_k1_secp256k1_verify_cost_params, crypto_invalid_arguments_cost) = {
219        let cost_table = &context.extensions().get::<NativesCostTable>();
220        (
221            cost_table.ecdsa_k1_secp256k1_verify_cost_params.clone(),
222            cost_table.crypto_invalid_arguments_cost,
223        )
224    };
225
226    let (base_cost, cost_per_byte, cost_per_block, block_size) = match hash {
227        KECCAK256 => (
228            ecdsa_k1_secp256k1_verify_cost_params.ecdsa_k1_secp256k1_verify_keccak256_cost_base,
229            ecdsa_k1_secp256k1_verify_cost_params
230                .ecdsa_k1_secp256k1_verify_keccak256_msg_cost_per_byte,
231            ecdsa_k1_secp256k1_verify_cost_params
232                .ecdsa_k1_secp256k1_verify_keccak256_msg_cost_per_block,
233            KECCAK256_BLOCK_SIZE,
234        ),
235        SHA256 => (
236            ecdsa_k1_secp256k1_verify_cost_params.ecdsa_k1_secp256k1_verify_sha256_cost_base,
237            ecdsa_k1_secp256k1_verify_cost_params
238                .ecdsa_k1_secp256k1_verify_sha256_msg_cost_per_byte,
239            ecdsa_k1_secp256k1_verify_cost_params
240                .ecdsa_k1_secp256k1_verify_sha256_msg_cost_per_block,
241            SHA256_BLOCK_SIZE,
242        ),
243        _ => {
244            // Charge for failure but dont fail if we run out of gas otherwise the actual error is masked by OUT_OF_GAS error
245            context.charge_gas(crypto_invalid_arguments_cost);
246
247            return Ok(NativeResult::ok(
248                context.gas_used(),
249                smallvec![Value::bool(false)],
250            ));
251        }
252    };
253    // Charge the base cost for this oper
254    native_charge_gas_early_exit!(context, base_cost);
255
256    let msg = pop_arg!(args, VectorRef);
257    let public_key_bytes = pop_arg!(args, VectorRef);
258    let signature_bytes = pop_arg!(args, VectorRef);
259
260    let msg_ref = msg.as_bytes_ref();
261    let public_key_bytes_ref = public_key_bytes.as_bytes_ref();
262    let signature_bytes_ref = signature_bytes.as_bytes_ref();
263
264    // Charge the arg size dependent costs
265    native_charge_gas_early_exit!(
266        context,
267        cost_per_byte * (msg_ref.len() as u64).into()
268            + cost_per_block * (msg_ref.len().div_ceil(block_size) as u64).into()
269    );
270
271    let cost = context.gas_used();
272
273    let Ok(sig) = <Secp256k1Signature as ToFromBytes>::from_bytes(&signature_bytes_ref) else {
274        return Ok(NativeResult::ok(cost, smallvec![Value::bool(false)]));
275    };
276
277    let Ok(pk) = <Secp256k1PublicKey as ToFromBytes>::from_bytes(&public_key_bytes_ref) else {
278        return Ok(NativeResult::ok(cost, smallvec![Value::bool(false)]));
279    };
280
281    let result = match hash {
282        KECCAK256 => pk.verify_with_hash::<Keccak256>(&msg_ref, &sig).is_ok(),
283        SHA256 => pk.verify_with_hash::<Sha256>(&msg_ref, &sig).is_ok(),
284        _ => false,
285    };
286
287    Ok(NativeResult::ok(cost, smallvec![Value::bool(result)]))
