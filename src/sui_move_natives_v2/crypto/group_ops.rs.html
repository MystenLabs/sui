83macro_rules! native_charge_gas_early_exit_option {
84    ($native_context:ident, $cost:expr) => {{
85        use move_binary_format::errors::PartialVMError;
86        use move_core_types::vm_status::StatusCode;
87        native_charge_gas_early_exit!(
88            $native_context,
89            $cost.ok_or_else(|| {
90                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
91                    .with_message("Gas cost for group ops is missing".to_string())
92            })?
93        );
94    }};
95}

97// Next should be aligned with the related Move modules.
98#[repr(u8)]
99enum Groups {
100    BLS12381Scalar = 0,
101    BLS12381G1 = 1,
102    BLS12381G2 = 2,
103    BLS12381GT = 3,
104}
105
106impl Groups {
107    fn from_u8(value: u8) -> Option<Self> {
108        match value {
109            0 => Some(Groups::BLS12381Scalar),
110            1 => Some(Groups::BLS12381G1),
111            2 => Some(Groups::BLS12381G2),
112            3 => Some(Groups::BLS12381GT),
113            _ => None,
114        }
115    }
116}

118fn parse_untrusted<G: ToFromByteArray<S> + FromTrustedByteArray<S>, const S: usize>(
119    e: &[u8],
120) -> FastCryptoResult<G> {
121    G::from_byte_array(e.try_into().map_err(|_| FastCryptoError::InvalidInput)?)
122}
123
124fn parse_trusted<G: ToFromByteArray<S> + FromTrustedByteArray<S>, const S: usize>(
125    e: &[u8],
126) -> FastCryptoResult<G> {
127    G::from_trusted_byte_array(e.try_into().map_err(|_| FastCryptoError::InvalidInput)?)
128}
129
130// Binary operations with 2 different types.
131fn binary_op_diff<
132    G1: ToFromByteArray<S1> + FromTrustedByteArray<S1>,
133    G2: ToFromByteArray<S2> + FromTrustedByteArray<S2>,
134    const S1: usize,
135    const S2: usize,
136>(
137    op: impl Fn(G1, G2) -> FastCryptoResult<G2>,
138    a1: &[u8],
139    a2: &[u8],
140) -> FastCryptoResult<Vec<u8>> {
141    let e1 = parse_trusted::<G1, S1>(a1)?;
142    let e2 = parse_trusted::<G2, S2>(a2)?;
143    let result = op(e1, e2)?;
144    Ok(result.to_byte_array().to_vec())
145}
146
147// Binary operations with the same type.
148fn binary_op<G: ToFromByteArray<S> + FromTrustedByteArray<S>, const S: usize>(
149    op: impl Fn(G, G) -> FastCryptoResult<G>,
150    a1: &[u8],
151    a2: &[u8],
152) -> FastCryptoResult<Vec<u8>> {
153    binary_op_diff::<G, G, S, S>(op, a1, a2)
154}

160/***************************************************************************************************
161 * native fun internal_validate
162 * Implementation of the Move native function `internal_validate(type: u8, bytes: &vector<u8>): bool`
163 *   gas cost: group_ops_decode_bls12381_X_cost where X is the requested type
164 **************************************************************************************************/
165
166pub fn internal_validate(
167    context: &mut NativeContext,
168    ty_args: Vec<Type>,
169    mut args: VecDeque<Value>,
170) -> PartialVMResult<NativeResult> {
171    debug_assert!(ty_args.is_empty());
172    debug_assert!(args.len() == 2);
173
174    let cost = context.gas_used();
175    if !is_supported(context) {
176        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
177    }
178
179    let bytes_ref = pop_arg!(args, VectorRef);
180    let bytes = bytes_ref.as_bytes_ref();
181    let group_type = pop_arg!(args, u8);
182
183    let cost_params = &context
184        .extensions()
185        .get::<NativesCostTable>()
186        .group_ops_cost_params
187        .clone();
188
189    let result = match Groups::from_u8(group_type) {
190        Some(Groups::BLS12381Scalar) => {
191            native_charge_gas_early_exit_option!(context, cost_params.bls12381_decode_scalar_cost);
192            parse_untrusted::<bls::Scalar, { bls::Scalar::BYTE_LENGTH }>(&bytes).is_ok()
193        }
194        Some(Groups::BLS12381G1) => {
195            native_charge_gas_early_exit_option!(context, cost_params.bls12381_decode_g1_cost);
196            parse_untrusted::<bls::G1Element, { bls::G1Element::BYTE_LENGTH }>(&bytes).is_ok()
197        }
198        Some(Groups::BLS12381G2) => {
199            native_charge_gas_early_exit_option!(context, cost_params.bls12381_decode_g2_cost);
200            parse_untrusted::<bls::G2Element, { bls::G2Element::BYTE_LENGTH }>(&bytes).is_ok()
201        }
202        _ => false,
203    };
204
205    Ok(NativeResult::ok(cost, smallvec![Value::bool(result)]))
206}

466/***************************************************************************************************
467 * native fun internal_hash_to
468 * Implementation of the Move native function `internal_hash_to(type: u8, m: &vector<u8>): vector<u8>`
469 *   gas cost: group_ops_bls12381_X_hash_to_base_cost + group_ops_bls12381_X_hash_to_cost_per_byte * |input|
470 *             where X is the requested type
471 **************************************************************************************************/
472pub fn internal_hash_to(
473    context: &mut NativeContext,
474    ty_args: Vec<Type>,
475    mut args: VecDeque<Value>,
476) -> PartialVMResult<NativeResult> {
477    debug_assert!(ty_args.is_empty());
478    debug_assert!(args.len() == 2);
479
480    let cost = context.gas_used();
481    if !is_supported(context) {
482        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
483    }
484
485    let m_ref = pop_arg!(args, VectorRef);
486    let m = m_ref.as_bytes_ref();
487    let group_type = pop_arg!(args, u8);
488
489    if m.is_empty() {
490        return Ok(NativeResult::err(cost, INVALID_INPUT_ERROR));
491    }
492
493    let cost_params = &context
494        .extensions()
495        .get::<NativesCostTable>()
496        .group_ops_cost_params
497        .clone();
498
499    let result = match Groups::from_u8(group_type) {
500        Some(Groups::BLS12381G1) => {
501            native_charge_gas_early_exit_option!(
502                context,
503                cost_params
504                    .bls12381_g1_hash_to_base_cost
505                    .and_then(|base_cost| cost_params
506                        .bls12381_g1_hash_to_cost_per_byte
507                        .map(|per_byte| base_cost + per_byte * (m.len() as u64).into()))
508            );
509            Ok(bls::G1Element::hash_to_group_element(&m)
510                .to_byte_array()
511                .to_vec())
512        }
513        Some(Groups::BLS12381G2) => {
514            native_charge_gas_early_exit_option!(
515                context,
516                cost_params
517                    .bls12381_g2_hash_to_base_cost
518                    .and_then(|base_cost| cost_params
519                        .bls12381_g2_hash_to_cost_per_byte
520                        .map(|per_byte| base_cost + per_byte * (m.len() as u64).into()))
521            );
522            Ok(bls::G2Element::hash_to_group_element(&m)
523                .to_byte_array()
524                .to_vec())
525        }
526        _ => Err(FastCryptoError::InvalidInput),
527    };
528
529    match result {
530        Ok(bytes) => Ok(NativeResult::ok(cost, smallvec![Value::vector_u8(bytes)])),
531        // Since all Element<G> are validated on construction, this error should never happen unless the requested type is wrong or inputs are invalid.
532        Err(_) => Ok(NativeResult::err(cost, INVALID_INPUT_ERROR)),
533    }
534}

536// Based on calculation from https://github.com/supranational/blst/blob/master/src/multi_scalar.c#L270
537fn msm_num_of_additions(n: u64) -> u64 {
538    debug_assert!(n > 0);
539    let wbits = (64 - n.leading_zeros() - 1) as u64;
540    let window_size = match wbits {
541        0 => 1,
542        1..=4 => 2,
543        5..=12 => wbits - 2,
544        _ => wbits - 3,
545    };
546    let num_of_windows = 255 / window_size + if 255 % window_size == 0 { 0 } else { 1 };
547    num_of_windows * (n + (1 << window_size) + 1)
548}

559fn multi_scalar_mul<G, const SCALAR_SIZE: usize, const POINT_SIZE: usize>(
560    context: &mut NativeContext,
561    scalar_decode_cost: Option<InternalGas>,
562    point_decode_cost: Option<InternalGas>,
563    base_cost: Option<InternalGas>,
564    base_cost_per_addition: Option<InternalGas>,
565    max_len: u32,
566    scalars: &[u8],
567    points: &[u8],
568) -> PartialVMResult<NativeResult>
569where
570    G: GroupElement
571        + ToFromByteArray<POINT_SIZE>
572        + FromTrustedByteArray<POINT_SIZE>
573        + MultiScalarMul,
574    G::ScalarType: ToFromByteArray<SCALAR_SIZE> + FromTrustedByteArray<SCALAR_SIZE>,
575{
576    if points.is_empty()
577        || scalars.is_empty()
578        || !scalars.len().is_multiple_of(SCALAR_SIZE)
579        || !points.len().is_multiple_of(POINT_SIZE)
580        || points.len() / POINT_SIZE != scalars.len() / SCALAR_SIZE
581    {
582        return Ok(NativeResult::err(context.gas_used(), INVALID_INPUT_ERROR));
583    }
584
585    if points.len() / POINT_SIZE > max_len as usize {
586        return Ok(NativeResult::err(context.gas_used(), INPUT_TOO_LONG_ERROR));
587    }
588
589    native_charge_gas_early_exit_option!(
590        context,
591        scalar_decode_cost.map(|cost| cost * ((scalars.len() / SCALAR_SIZE) as u64).into())
592    );
593    let scalars = scalars
594        .chunks(SCALAR_SIZE)
595        .map(parse_trusted::<G::ScalarType, { SCALAR_SIZE }>)
596        .collect::<Result<Vec<_>, _>>();
597
598    native_charge_gas_early_exit_option!(
599        context,
600        point_decode_cost.map(|cost| cost * ((points.len() / POINT_SIZE) as u64).into())
601    );
602    let points = points
603        .chunks(POINT_SIZE)
604        .map(parse_trusted::<G, { POINT_SIZE }>)
605        .collect::<Result<Vec<_>, _>>();
606
607    if let (Ok(scalars), Ok(points)) = (scalars, points) {
608        // Checked above that len()>0
609        let num_of_additions = msm_num_of_additions(scalars.len() as u64);
610        native_charge_gas_early_exit_option!(
611            context,
612            base_cost.and_then(|base| base_cost_per_addition
613                .map(|per_addition| base + per_addition * num_of_additions.into()))
614        );
615
616        let r = G::multi_scalar_mul(&scalars, &points)
617            .expect("Already checked the lengths of the vectors");
618        Ok(NativeResult::ok(
619            context.gas_used(),
620            smallvec![Value::vector_u8(r.to_byte_array().to_vec())],
621        ))
622    } else {
623        Ok(NativeResult::err(context.gas_used(), INVALID_INPUT_ERROR))
624    }
625}

627/***************************************************************************************************
628 * native fun internal_multi_scalar_mul
629 * Implementation of the Move native function `internal_multi_scalar_mul(type: u8, scalars: &vector<u8>, elements: &vector<u8>): vector<u8>`
630 *   gas cost: (bls12381_decode_scalar_cost + bls12381_decode_X_cost) * N + bls12381_X_msm_base_cost +
631 *             bls12381_X_msm_base_cost_per_input * num_of_additions(N)
632 **************************************************************************************************/
633pub fn internal_multi_scalar_mul(
634    context: &mut NativeContext,
635    ty_args: Vec<Type>,
636    mut args: VecDeque<Value>,
637) -> PartialVMResult<NativeResult> {
638    debug_assert!(ty_args.is_empty());
639    debug_assert!(args.len() == 3);
640
641    let cost = context.gas_used();
642    if !is_supported(context) {
643        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
644    }
645
646    let elements_ref = pop_arg!(args, VectorRef);
647    let elements = elements_ref.as_bytes_ref();
648    let scalars_ref = pop_arg!(args, VectorRef);
649    let scalars = scalars_ref.as_bytes_ref();
650    let group_type = pop_arg!(args, u8);
651
652    let cost_params = &context
653        .extensions()
654        .get::<NativesCostTable>()
655        .group_ops_cost_params
656        .clone();
657
658    let max_len = cost_params.bls12381_msm_max_len.ok_or_else(|| {
659        PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
660            .with_message("Max len for MSM is not set".to_string())
661    })?;
662
663    // TODO: can potentially improve performance when some of the points are the generator.
664    match Groups::from_u8(group_type) {
665        Some(Groups::BLS12381G1) => multi_scalar_mul::<
666            bls::G1Element,
667            { bls::Scalar::BYTE_LENGTH },
668            { bls::G1Element::BYTE_LENGTH },
669        >(
670            context,
671            cost_params.bls12381_decode_scalar_cost,
672            cost_params.bls12381_decode_g1_cost,
673            cost_params.bls12381_g1_msm_base_cost,
674            cost_params.bls12381_g1_msm_base_cost_per_input,
675            max_len,
676            scalars.as_ref(),
677            elements.as_ref(),
678        ),
679        Some(Groups::BLS12381G2) => multi_scalar_mul::<
680            bls::G2Element,
681            { bls::Scalar::BYTE_LENGTH },
682            { bls::G2Element::BYTE_LENGTH },
683        >(
684            context,
685            cost_params.bls12381_decode_scalar_cost,
686            cost_params.bls12381_decode_g2_cost,
687            cost_params.bls12381_g2_msm_base_cost,
688            cost_params.bls12381_g2_msm_base_cost_per_input,
689            max_len,
690            scalars.as_ref(),
691            elements.as_ref(),
692        ),
693        _ => Ok(NativeResult::err(cost, INVALID_INPUT_ERROR)),
694    }
695}

697/***************************************************************************************************
698 * native fun internal_pairing
699 * Implementation of the Move native function `internal_pairing(type:u8, e1: &vector<u8>, e2: &vector<u8>): vector<u8>`
700 *   gas cost: group_ops_bls12381_pairing_cost
701 **************************************************************************************************/
702pub fn internal_pairing(
703    context: &mut NativeContext,
704    ty_args: Vec<Type>,
705    mut args: VecDeque<Value>,
706) -> PartialVMResult<NativeResult> {
707    debug_assert!(ty_args.is_empty());
708    debug_assert!(args.len() == 3);
709
710    let cost = context.gas_used();
711    if !is_supported(context) {
712        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
713    }
714
715    let e2_ref = pop_arg!(args, VectorRef);
716    let e2 = e2_ref.as_bytes_ref();
717    let e1_ref = pop_arg!(args, VectorRef);
718    let e1 = e1_ref.as_bytes_ref();
719    let group_type = pop_arg!(args, u8);
720
721    let cost_params = &context
722        .extensions()
723        .get::<NativesCostTable>()
724        .group_ops_cost_params
725        .clone();
726
727    let result = match Groups::from_u8(group_type) {
728        Some(Groups::BLS12381G1) => {
729            native_charge_gas_early_exit_option!(context, cost_params.bls12381_pairing_cost);
730            parse_trusted::<bls::G1Element, { bls::G1Element::BYTE_LENGTH }>(&e1).and_then(|e1| {
731                parse_trusted::<bls::G2Element, { bls::G2Element::BYTE_LENGTH }>(&e2).map(|e2| {
732                    let e3 = e1.pairing(&e2);
733                    e3.to_byte_array().to_vec()
734                })
735            })
736        }
737        _ => Err(FastCryptoError::InvalidInput),
738    };
739
740    match result {
741        Ok(bytes) => Ok(NativeResult::ok(cost, smallvec![Value::vector_u8(bytes)])),
742        // Since all Element<G> are validated on construction, this error should never happen unless the requested type is wrong.
743        Err(_) => Ok(NativeResult::err(cost, INVALID_INPUT_ERROR)),
744    }
745}
