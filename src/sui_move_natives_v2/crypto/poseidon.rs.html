20
21pub const NON_CANONICAL_INPUT: u64 = 0;
22pub const NOT_SUPPORTED_ERROR: u64 = 1;
23
24fn is_supported(context: &NativeContext) -> bool {
25    context
26        .extensions()
27        .get::<ObjectRuntime>()
28        .protocol_config
29        .enable_poseidon()
30}
31
32#[derive(Clone)]
33pub struct PoseidonBN254CostParams {
34    /// Base cost for invoking the `poseidon_bn254` function
35    pub poseidon_bn254_cost_base: Option<InternalGas>,
36    /// Cost per block of `data`, where a block is 32 bytes
37    pub poseidon_bn254_data_cost_per_block: Option<InternalGas>,
38}
39
40/***************************************************************************************************
41 * native fun poseidon_bn254
42 * Implementation of the Move native function `poseidon::poseidon_bn254_internal(data: &vector<vector<u8>>): vector<u8>
43 *   gas cost: poseidon_bn254_cost_base                           | base cost for function call and fixed opers
44 *              + poseidon_bn254_data_cost_per_block * num_inputs | cost depends on number of inputs
45 **************************************************************************************************/
46pub fn poseidon_bn254_internal(
47    context: &mut NativeContext,
48    ty_args: Vec<Type>,
49    mut args: VecDeque<Value>,
50) -> PartialVMResult<NativeResult> {
51    let cost = context.gas_used();
52    if !is_supported(context) {
53        return Ok(NativeResult::err(cost, NOT_SUPPORTED_ERROR));
54    }
55
56    // Load the cost parameters from the protocol config
57    let cost_params = &context
58        .extensions()
59        .get::<NativesCostTable>()
60        .poseidon_bn254_cost_params
61        .clone();
62
63    // Charge the base cost for this operation
64    native_charge_gas_early_exit!(
65        context,
66        cost_params
67            .poseidon_bn254_cost_base
68            .ok_or_else(
69                || PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
70                    .with_message("Gas cost for poseidon_bn254 not available".to_string())
71            )?
72    );
73
74    debug_assert!(ty_args.is_empty());
75    debug_assert!(args.len() == 1);
76
77    // The input is a reference to a vector of vector<u8>'s
78    let inputs = pop_arg!(args, VectorRef);
79
80    let length = inputs
81        .len(&Type::Vector(Box::new(Type::U8)))?
82        .value_as::<u64>()?;
83
84    // Charge the msg dependent costs
85    native_charge_gas_early_exit!(
86        context,
87        cost_params
88            .poseidon_bn254_data_cost_per_block
89            .ok_or_else(
90                || PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
91                    .with_message("Gas cost for poseidon_bn254 not available".to_string())
92            )?
93            .mul(length.into())
94    );
95
96    // Read the input vector
97    let field_elements = (0..length)
98        .map(|i| {
99            let reference = inputs.borrow_elem(i as usize, &Type::Vector(Box::new(Type::U8)))?;
100            let value = reference.value_as::<VectorRef>()?.as_bytes_ref().clone();
101            Ok(value)
102        })
103        .collect::<Result<Vec<_>, _>>()?;
104
105    match poseidon_bytes(&field_elements) {
106        Ok(result) => Ok(NativeResult::ok(
107            context.gas_used(),
108            smallvec![Value::vector_u8(result)],
109        )),
110        // This is also checked in the poseidon_bn254 move function but to be sure we handle it here also.
111        Err(_) => Ok(NativeResult::err(context.gas_used(), NON_CANONICAL_INPUT)),
112    }
113}
