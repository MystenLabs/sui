29const E_KEY_DOES_NOT_EXIST: u64 = 1;
30const E_FIELD_TYPE_MISMATCH: u64 = 2;

33macro_rules! get_or_fetch_object {
34    ($context:ident, $ty_args:ident, $parent:ident, $child_id:ident, $ty_cost_per_byte:expr) => {{
35        let child_ty = $ty_args.pop().unwrap();
36        native_charge_gas_early_exit!(
37            $context,
38            $ty_cost_per_byte * u64::from(child_ty.size()).into()
39        );
40
41        assert!($ty_args.is_empty());
42        let (tag, layout, annotated_layout) = match crate::get_tag_and_layouts($context, &child_ty)?
43        {
44            Some(res) => res,
45            None => {
46                return Ok(NativeResult::err(
47                    $context.gas_used(),
48                    E_BCS_SERIALIZATION_FAILURE,
49                ))
50            }
51        };
52
53        let object_runtime: &mut ObjectRuntime = $context.extensions_mut().get_mut();
54        object_runtime.get_or_fetch_child_object(
55            $parent,
56            $child_id,
57            &child_ty,
58            &layout,
59            &annotated_layout,
60            MoveObjectType::from(tag),
61        )?
62    }};
63}

73/***************************************************************************************************
74 * native fun hash_type_and_key
75 * Implementation of the Move native function `hash_type_and_key<K: copy + drop + store>(parent: address, k: K): address`
76 *   gas cost: dynamic_field_hash_type_and_key_cost_base                            | covers various fixed costs in the oper
77 *              + dynamic_field_hash_type_and_key_type_cost_per_byte * size_of(K)   | covers cost of operating on the type `K`
78 *              + dynamic_field_hash_type_and_key_value_cost_per_byte * size_of(k)  | covers cost of operating on the value `k`
79 *              + dynamic_field_hash_type_and_key_type_tag_cost_per_byte * size_of(type_tag(k))    | covers cost of operating on the type tag of `K`
80 **************************************************************************************************/
81#[instrument(level = "trace", skip_all)]
82pub fn hash_type_and_key(
83    context: &mut NativeContext,
84    mut ty_args: Vec<Type>,
85    mut args: VecDeque<Value>,
86) -> PartialVMResult<NativeResult> {
87    assert_eq!(ty_args.len(), 1);
88    assert_eq!(args.len(), 2);
89
90    let dynamic_field_hash_type_and_key_cost_params = context
91        .extensions_mut()
92        .get::<NativesCostTable>()
93        .dynamic_field_hash_type_and_key_cost_params
94        .clone();
95
96    // Charge base fee
97    native_charge_gas_early_exit!(
98        context,
99        dynamic_field_hash_type_and_key_cost_params.dynamic_field_hash_type_and_key_cost_base
100    );
101
102    let k_ty = ty_args.pop().unwrap();
103    let k: Value = args.pop_back().unwrap();
104    let parent = pop_arg!(args, AccountAddress);
105
106    // Get size info for costing for derivations, serializations, etc
107    let k_ty_size = u64::from(k_ty.size());
108    let k_value_size = u64::from(k.legacy_size());
109    native_charge_gas_early_exit!(
110        context,
111        dynamic_field_hash_type_and_key_cost_params
112            .dynamic_field_hash_type_and_key_type_cost_per_byte
113            * k_ty_size.into()
114            + dynamic_field_hash_type_and_key_cost_params
115                .dynamic_field_hash_type_and_key_value_cost_per_byte
116                * k_value_size.into()
117    );
118
119    let k_tag = context.type_to_type_tag(&k_ty)?;
120    let k_tag_size = u64::from(k_tag.abstract_size_for_gas_metering());
121
122    native_charge_gas_early_exit!(
123        context,
124        dynamic_field_hash_type_and_key_cost_params
125            .dynamic_field_hash_type_and_key_type_tag_cost_per_byte
126            * k_tag_size.into()
127    );
128
129    let cost = context.gas_used();
130
131    let k_layout = match context.type_to_type_layout(&k_ty) {
132        Ok(Some(layout)) => layout,
133        _ => return Ok(NativeResult::err(cost, E_BCS_SERIALIZATION_FAILURE)),
134    };
135    let Some(k_bytes) = k.simple_serialize(&k_layout) else {
136        return Ok(NativeResult::err(cost, E_BCS_SERIALIZATION_FAILURE));
137    };
138    let Ok(id) = derive_dynamic_field_id(parent, &k_tag, &k_bytes) else {
139        return Ok(NativeResult::err(cost, E_BCS_SERIALIZATION_FAILURE));
140    };
141
142    Ok(NativeResult::ok(cost, smallvec![Value::address(id.into())]))
143}

153/***************************************************************************************************
154 * native fun add_child_object
155 * throws `E_KEY_ALREADY_EXISTS` if a child already exists with that ID
156 * Implementation of the Move native function `add_child_object<Child: key>(parent: address, child: Child)`
157 *   gas cost: dynamic_field_add_child_object_cost_base                    | covers various fixed costs in the oper
158 *              + dynamic_field_add_child_object_type_cost_per_byte * size_of(Child)        | covers cost of operating on the type `Child`
159 *              + dynamic_field_add_child_object_value_cost_per_byte * size_of(child)       | covers cost of operating on the value `child`
160 *              + dynamic_field_add_child_object_struct_tag_cost_per_byte * size_of(struct)tag(Child))  | covers cost of operating on the struct tag of `Child`
161 **************************************************************************************************/
162#[instrument(level = "trace", skip_all)]
163pub fn add_child_object(
164    context: &mut NativeContext,
165    mut ty_args: Vec<Type>,
166    mut args: VecDeque<Value>,
167) -> PartialVMResult<NativeResult> {
168    assert!(ty_args.len() == 1);
169    assert!(args.len() == 2);
170
171    let dynamic_field_add_child_object_cost_params = context
172        .extensions_mut()
173        .get::<NativesCostTable>()
174        .dynamic_field_add_child_object_cost_params
175        .clone();
176
177    // Charge base fee
178    native_charge_gas_early_exit!(
179        context,
180        dynamic_field_add_child_object_cost_params.dynamic_field_add_child_object_cost_base
181    );
182
183    let child = args.pop_back().unwrap();
184    let parent = pop_arg!(args, AccountAddress).into();
185    assert!(args.is_empty());
186
187    let child_value_size = u64::from(child.legacy_size());
188    // ID extraction step
189    native_charge_gas_early_exit!(
190        context,
191        dynamic_field_add_child_object_cost_params
192            .dynamic_field_add_child_object_value_cost_per_byte
193            * child_value_size.into()
194    );
195
196    // TODO remove this copy_value, which will require VM changes
197    let child_id = get_object_id(child.copy_value().unwrap())
198        .unwrap()
199        .value_as::<AccountAddress>()
200        .unwrap()
201        .into();
202    let child_ty = ty_args.pop().unwrap();
203    let child_type_size = u64::from(child_ty.size());
204
205    native_charge_gas_early_exit!(
206        context,
207        dynamic_field_add_child_object_cost_params
208            .dynamic_field_add_child_object_type_cost_per_byte
209            * child_type_size.into()
210    );
211
212    assert!(ty_args.is_empty());
213    let tag = match context.type_to_type_tag(&child_ty)? {
214        TypeTag::Struct(s) => *s,
215        _ => {
216            return Err(
217                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
218                    .with_message("Sui verifier guarantees this is a struct".to_string()),
219            )
220        }
221    };
222
223    let struct_tag_size = u64::from(tag.abstract_size_for_gas_metering());
224    native_charge_gas_early_exit!(
225        context,
226        dynamic_field_add_child_object_cost_params
227            .dynamic_field_add_child_object_struct_tag_cost_per_byte
228            * struct_tag_size.into()
229    );
230
231    let object_runtime: &mut ObjectRuntime = context.extensions_mut().get_mut();
232    object_runtime.add_child_object(
233        parent,
234        child_id,
235        &child_ty,
236        MoveObjectType::from(tag),
237        child,
238    )?;
239    Ok(NativeResult::ok(context.gas_used(), smallvec![]))
240}

249/***************************************************************************************************
250 * native fun borrow_child_object
251 * throws `E_KEY_DOES_NOT_EXIST` if a child does not exist with that ID at that type
252 * or throws `E_FIELD_TYPE_MISMATCH` if the type does not match (as the runtime does not distinguish different reference types)
253 * Implementation of the Move native function `borrow_child_object_mut<Child: key>(parent: &mut UID, id: address): &mut Child`
254 *   gas cost: dynamic_field_borrow_child_object_cost_base                    | covers various fixed costs in the oper
255 *              + dynamic_field_borrow_child_object_child_ref_cost_per_byte  * size_of(&Child)  | covers cost of fetching and returning `&Child`
256 *              + dynamic_field_borrow_child_object_type_cost_per_byte  * size_of(Child)        | covers cost of operating on type `Child`
257 **************************************************************************************************/
258#[instrument(level = "trace", skip_all)]
259pub fn borrow_child_object(
260    context: &mut NativeContext,
261    mut ty_args: Vec<Type>,
262    mut args: VecDeque<Value>,
263) -> PartialVMResult<NativeResult> {
264    assert!(ty_args.len() == 1);
265    assert!(args.len() == 2);
266
267    let dynamic_field_borrow_child_object_cost_params = context
268        .extensions_mut()
269        .get::<NativesCostTable>()
270        .dynamic_field_borrow_child_object_cost_params
271        .clone();
272    native_charge_gas_early_exit!(
273        context,
274        dynamic_field_borrow_child_object_cost_params.dynamic_field_borrow_child_object_cost_base
275    );
276
277    let child_id = pop_arg!(args, AccountAddress).into();
278
279    let parent_uid = pop_arg!(args, StructRef).read_ref().unwrap();
280    // UID { id: ID { bytes: address } }
281    let parent = get_nested_struct_field(parent_uid, &[0, 0])
282        .unwrap()
283        .value_as::<AccountAddress>()
284        .unwrap()
285        .into();
286
287    assert!(args.is_empty());
288    let global_value_result = get_or_fetch_object!(
289        context,
290        ty_args,
291        parent,
292        child_id,
293        dynamic_field_borrow_child_object_cost_params
294            .dynamic_field_borrow_child_object_type_cost_per_byte
295    );
296    let global_value = match global_value_result {
297        ObjectResult::MismatchedType => {
298            return Ok(NativeResult::err(context.gas_used(), E_FIELD_TYPE_MISMATCH))
299        }
300        ObjectResult::Loaded(gv) => gv,
301    };
302    if !global_value.exists()? {
303        return Ok(NativeResult::err(context.gas_used(), E_KEY_DOES_NOT_EXIST));
304    }
305    let child_ref = global_value.borrow_global().inspect_err(|err| {
306        assert!(err.major_status() != StatusCode::MISSING_DATA);
307    })?;
308
309    native_charge_gas_early_exit!(
310        context,
311        dynamic_field_borrow_child_object_cost_params
312            .dynamic_field_borrow_child_object_child_ref_cost_per_byte
313            * u64::from(child_ref.legacy_size()).into()
314    );
315
316    Ok(NativeResult::ok(context.gas_used(), smallvec![child_ref]))
317}

325/***************************************************************************************************
326 * native fun remove_child_object
327 * throws `E_KEY_DOES_NOT_EXIST` if a child does not exist with that ID at that type
328 * or throws `E_FIELD_TYPE_MISMATCH` if the type does not match
329 * Implementation of the Move native function `remove_child_object<Child: key>(parent: address, id: address): Child`
330 *   gas cost: dynamic_field_remove_child_object_cost_base                    | covers various fixed costs in the oper
331 *              + dynamic_field_remove_child_object_type_cost_per_byte * size_of(Child)      | covers cost of operating on type `Child`
332 *              + dynamic_field_remove_child_object_child_cost_per_byte  * size_of(child)     | covers cost of fetching and returning value of type `Child`
333 **************************************************************************************************/
334#[instrument(level = "trace", skip_all)]
335pub fn remove_child_object(
336    context: &mut NativeContext,
337    mut ty_args: Vec<Type>,
338    mut args: VecDeque<Value>,
339) -> PartialVMResult<NativeResult> {
340    assert!(ty_args.len() == 1);
341    assert!(args.len() == 2);
342
343    let dynamic_field_remove_child_object_cost_params = context
344        .extensions_mut()
345        .get::<NativesCostTable>()
346        .dynamic_field_remove_child_object_cost_params
347        .clone();
348    native_charge_gas_early_exit!(
349        context,
350        dynamic_field_remove_child_object_cost_params.dynamic_field_remove_child_object_cost_base
351    );
352
353    let child_id = pop_arg!(args, AccountAddress).into();
354    let parent = pop_arg!(args, AccountAddress).into();
355    assert!(args.is_empty());
356    let global_value_result = get_or_fetch_object!(
357        context,
358        ty_args,
359        parent,
360        child_id,
361        dynamic_field_remove_child_object_cost_params
362            .dynamic_field_remove_child_object_type_cost_per_byte
363    );
364    let global_value = match global_value_result {
365        ObjectResult::MismatchedType => {
366            return Ok(NativeResult::err(context.gas_used(), E_FIELD_TYPE_MISMATCH))
367        }
368        ObjectResult::Loaded(gv) => gv,
369    };
370    if !global_value.exists()? {
371        return Ok(NativeResult::err(context.gas_used(), E_KEY_DOES_NOT_EXIST));
372    }
373    let child = global_value.move_from().inspect_err(|err| {
374        assert!(err.major_status() != StatusCode::MISSING_DATA);
375    })?;
376
377    native_charge_gas_early_exit!(
378        context,
379        dynamic_field_remove_child_object_cost_params
380            .dynamic_field_remove_child_object_child_cost_per_byte
381            * u64::from(child.legacy_size()).into()
382    );
383
384    Ok(NativeResult::ok(context.gas_used(), smallvec![child]))
385}
