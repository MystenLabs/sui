906// ID { bytes: address }
907// Extract the first field of the struct to get the address bytes.
908pub fn get_receiver_object_id(object: Value) -> Result<Value, PartialVMError> {
909    get_nested_struct_field(object, &[0])
910}
911
912// Object { id: UID { id: ID { bytes: address } } .. }
913// Extract the first field of the struct 3 times to get the id bytes.
914pub fn get_object_id(object: Value) -> Result<Value, PartialVMError> {
915    get_nested_struct_field(object, &[0, 0, 0])
916}

918// Extract a field value that's nested inside value `v`. The offset of each nesting
919// is determined by `offsets`.
920pub fn get_nested_struct_field(mut v: Value, offsets: &[usize]) -> Result<Value, PartialVMError> {
921    for offset in offsets {
922        v = get_nth_struct_field(v, *offset)?;
923    }
924    Ok(v)
925}
926
927pub fn get_nth_struct_field(v: Value, n: usize) -> Result<Value, PartialVMError> {
928    let mut itr = v.value_as::<Struct>()?.unpack()?;
929    Ok(itr.nth(n).unwrap())
930}

932/// Returns the struct tag, non-annotated type layout, and fully annotated type layout of `ty`.
933pub(crate) fn get_tag_and_layouts(
934    context: &NativeContext,
935    ty: &Type,
936) -> PartialVMResult<Option<(StructTag, R::MoveTypeLayout, A::MoveTypeLayout)>> {
937    let tag = match context.type_to_type_tag(ty)? {
938        TypeTag::Struct(s) => s,
939        _ => {
940            return Err(
941                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
942                    .with_message("Sui verifier guarantees this is a struct".to_string()),
943            )
944        }
945    };
946    let Some(layout) = context.type_to_type_layout(ty)? else {
947        return Ok(None);
948    };
949    let Some(annotated_layout) = context.type_to_fully_annotated_layout(ty)? else {
950        return Ok(None);
951    };
952    Ok(Some((*tag, layout, annotated_layout)))
953}
