42struct Inner<'a> {
43    // used for loading child objects
44    resolver: &'a dyn ChildObjectResolver,
45    // The version of the root object in ownership at the beginning of the transaction.
46    // If it was a child object, it resolves to the root parent's sequence number.
47    // Otherwise, it is just the sequence number at the beginning of the transaction.
48    root_version: BTreeMap<ObjectID, SequenceNumber>,
49    // A map from a wrapped object to the object it was contained in at the
50    // beginning of the transaction.
51    wrapped_object_containers: BTreeMap<ObjectID, ObjectID>,
52    // cached objects from the resolver. An object might be in this map but not in the store
53    // if it's existence was queried, but the value was not used.
54    cached_objects: BTreeMap<ObjectID, Option<Object>>,
55    // whether or not this TX is gas metered
56    is_metered: bool,
57    // Protocol config used to enforce limits
58    protocol_config: &'a ProtocolConfig,
59    // Metrics for reporting exceeded limits
60    metrics: Arc<LimitsMetrics>,
61    // Epoch ID for the current transaction. Used for receiving objects.
62    current_epoch_id: EpochId,
63}

65// maintains the runtime GlobalValues for child objects and manages the fetching of objects
66// from storage, through the `ChildObjectResolver`
67pub(super) struct ChildObjectStore<'a> {
68    // contains object resolver and object cache
69    // kept as a separate struct to deal with lifetime issues where the `store` is accessed
70    // at the same time as the `cached_objects` is populated
71    inner: Inner<'a>,
72    // Maps of populated GlobalValues, meaning the child object has been accessed in this
73    // transaction
74    store: BTreeMap<ObjectID, ChildObject>,
75    // whether or not this TX is gas metered
76    is_metered: bool,
77}

79pub(crate) enum ObjectResult<V> {
80    // object exists but type does not match. Should result in an abort
81    MismatchedType,
82    Loaded(V),
83}
84
85type LoadedWithMetadataResult<V> = Option<(V, DynamicallyLoadedObjectMetadata)>;

87impl Inner<'_> {
88    fn receive_object_from_store(
89        &self,
90        owner: ObjectID,
91        child: ObjectID,
92        version: SequenceNumber,
93    ) -> PartialVMResult<LoadedWithMetadataResult<MoveObject>> {
94        let child_opt = self
95            .resolver
96            .get_object_received_at_version(&owner, &child, version, self.current_epoch_id)
97            .map_err(|msg| {
98                PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!("{msg}"))
99            })?;
100        let obj_opt = if let Some(object) = child_opt {
101            // guard against bugs in `receive_object_at_version`: if it returns a child object such that
102            // C.parent != parent, we raise an invariant violation since that should be checked by
103            // `receive_object_at_version`.
104            if object.owner != Owner::AddressOwner(owner.into()) {
105                return Err(
106                    PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!(
107                        "Bad owner for {child}. \
108                        Expected owner {owner} but found owner {}",
109                        object.owner
110                    )),
111                );
112            }
113            let loaded_metadata = DynamicallyLoadedObjectMetadata {
114                version,
115                digest: object.digest(),
116                storage_rebate: object.storage_rebate,
117                owner: object.owner.clone(),
118                previous_transaction: object.previous_transaction,
119            };
120
121            // `ChildObjectResolver::receive_object_at_version` should return the object at the
122            // version or nothing at all. If it returns an object with a different version, we
123            // should raise an invariant violation since it should be checked by
124            // `receive_object_at_version`.
125            if object.version() != version {
126                return Err(
127                    PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!(
128                        "Bad version for {child}. \
129                        Expected version {version} but found version {}",
130                        object.version()
131                    )),
132                );
133            }
134            match object.into_inner().data {
135                Data::Package(_) => {
136                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
137                        format!(
138                            "Mismatched object type for {child}. \
139                                Expected a Move object but found a Move package"
140                        ),
141                    ))
142                }
143                Data::Move(mo @ MoveObject { .. }) => Some((mo, loaded_metadata)),
144            }
145        } else {
146            None
147        };
148        Ok(obj_opt)
149    }

151    fn get_or_fetch_object_from_store(
152        &mut self,
153        parent: ObjectID,
154        child: ObjectID,
155    ) -> PartialVMResult<Option<&MoveObject>> {
156        let cached_objects_count = self.cached_objects.len() as u64;
157        let parents_root_version = self.root_version.get(&parent).copied();
158        let had_parent_root_version = parents_root_version.is_some();
159        // if not found, it must be new so it won't have any child objects, thus
160        // we can return SequenceNumber(0) as no child object will be found
161        let parents_root_version = parents_root_version.unwrap_or(SequenceNumber::new());
162        if let btree_map::Entry::Vacant(e) = self.cached_objects.entry(child) {
163            let child_opt = self
164                .resolver
165                .read_child_object(&parent, &child, parents_root_version)
166                .map_err(|msg| {
167                    PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(format!("{msg}"))
168                })?;
169            let obj_opt = if let Some(object) = child_opt {
170                // if there was no root version, guard against reading a child object. A newly
171                // created parent should not have a child in storage
172                if !had_parent_root_version {
173                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
174                        format!("A new parent {parent} should not have a child object {child}."),
175                    ));
176                }
177                // guard against bugs in `read_child_object`: if it returns a child object such that
178                // C.parent != parent, we raise an invariant violation
179                match &object.owner {
180                    Owner::ObjectOwner(id) => {
181                        if ObjectID::from(*id) != parent {
182                            return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
183                                format!("Bad owner for {child}. \
184                                Expected owner {parent} but found owner {id}")
185                            ))
186                        }
187                    }
188                    Owner::AddressOwner(_) | Owner::Immutable | Owner::Shared { .. } => {
189                        return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
190                            format!("Bad owner for {child}. \
191                            Expected an id owner {parent} but found an address, immutable, or shared owner")
192                        ))
193                    }
194                    Owner::ConsensusAddressOwner { .. } => {
195                        unimplemented!("ConsensusAddressOwner does not exist for this execution version")
196                    }
197                };
198                match object.data {
199                    Data::Package(_) => {
200                        return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
201                            format!(
202                                "Mismatched object type for {child}. \
203                                Expected a Move object but found a Move package"
204                            ),
205                        ))
206                    }
207                    Data::Move(_) => Some(object),
208                }
209            } else {
210                None
211            };
212
213            if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
214                self.is_metered,
215                cached_objects_count,
216                self.protocol_config.object_runtime_max_num_cached_objects(),
217                self.protocol_config
218                    .object_runtime_max_num_cached_objects_system_tx(),
219                self.metrics.excessive_object_runtime_cached_objects
220            ) {
221                return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
222                    .with_message(format!(
223                        "Object runtime cached objects limit ({} entries) reached",
224                        lim
225                    ))
226                    .with_sub_status(
227                        VMMemoryLimitExceededSubStatusCode::OBJECT_RUNTIME_CACHE_LIMIT_EXCEEDED
228                            as u64,
229                    ));
230            };
231
232            e.insert(obj_opt);
233        }
234        Ok(self
235            .cached_objects
236            .get(&child)
237            .unwrap()
238            .as_ref()
239            .map(|obj| {
240                obj.data
241                    .try_as_move()
242                    // unwrap safe because we only insert Move objects
243                    .unwrap()
244            }))
245    }

247    fn fetch_object_impl(
248        &mut self,
249        parent: ObjectID,
250        child: ObjectID,
251        child_ty: &Type,
252        child_ty_layout: &R::MoveTypeLayout,
253        child_ty_fully_annotated_layout: &A::MoveTypeLayout,
254        child_move_type: MoveObjectType,
255    ) -> PartialVMResult<ObjectResult<(Type, MoveObjectType, GlobalValue)>> {
256        let obj = match self.get_or_fetch_object_from_store(parent, child)? {
257            None => {
258                return Ok(ObjectResult::Loaded((
259                    child_ty.clone(),
260                    child_move_type,
261                    GlobalValue::none(),
262                )))
263            }
264            Some(obj) => obj,
265        };
266        // object exists, but the type does not match
267        if obj.type_() != &child_move_type {
268            return Ok(ObjectResult::MismatchedType);
269        }
270        // generate a GlobalValue
271        let obj_contents = obj.contents();
272        let v = match Value::simple_deserialize(obj_contents, child_ty_layout) {
273            Some(v) => v,
274            None => return Err(
275                PartialVMError::new(StatusCode::FAILED_TO_DESERIALIZE_RESOURCE).with_message(
276                    format!("Failed to deserialize object {child} with type {child_move_type}",),
277                ),
278            ),
279        };
280        let global_value =
281            match GlobalValue::cached(v) {
282                Ok(gv) => gv,
283                Err(e) => {
284                    return Err(PartialVMError::new(StatusCode::STORAGE_ERROR).with_message(
285                        format!("Object {child} did not deserialize to a struct Value. Error: {e}"),
286                    ))
287                }
288            };
289        // Find all UIDs inside of the value and update the object parent maps
290        let contained_uids =
291            get_all_uids(child_ty_fully_annotated_layout, obj_contents).map_err(|e| {
292                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
293                    .with_message(format!("Failed to find UIDs. ERROR: {e}"))
294            })?;
295        let parents_root_version = self.root_version.get(&parent).copied();
296        if let Some(v) = parents_root_version {
297            debug_assert!(contained_uids.contains(&child));
298            for id in contained_uids {
299                self.root_version.insert(id, v);
300                if id != child {
301                    let prev = self.wrapped_object_containers.insert(id, child);
302                    debug_assert!(prev.is_none())
303                }
304            }
305        }
306        Ok(ObjectResult::Loaded((
307            child_ty.clone(),
308            child_move_type,
309            global_value,
310        )))
311    }
312}
313
314fn deserialize_move_object(
315    obj: &MoveObject,
316    child_ty: &Type,
317    child_ty_layout: &R::MoveTypeLayout,
318    child_move_type: MoveObjectType,
319) -> PartialVMResult<ObjectResult<(Type, MoveObjectType, Value)>> {
320    let child_id = obj.id();
321    // object exists, but the type does not match
322    if obj.type_() != &child_move_type {
323        return Ok(ObjectResult::MismatchedType);
324    }
325    let value = match Value::simple_deserialize(obj.contents(), child_ty_layout) {
326        Some(v) => v,
327        None => {
328            return Err(
329                PartialVMError::new(StatusCode::FAILED_TO_DESERIALIZE_RESOURCE).with_message(
330                    format!("Failed to deserialize object {child_id} with type {child_move_type}",),
331                ),
332            )
333        }
334    };
335    Ok(ObjectResult::Loaded((
336        child_ty.clone(),
337        child_move_type,
338        value,
339    )))
340}

342impl<'a> ChildObjectStore<'a> {
343    pub(super) fn new(
344        resolver: &'a dyn ChildObjectResolver,
345        root_version: BTreeMap<ObjectID, SequenceNumber>,
346        wrapped_object_containers: BTreeMap<ObjectID, ObjectID>,
347        is_metered: bool,
348        protocol_config: &'a ProtocolConfig,
349        metrics: Arc<LimitsMetrics>,
350        current_epoch_id: EpochId,
351    ) -> Self {
352        Self {
353            inner: Inner {
354                resolver,
355                root_version,
356                wrapped_object_containers,
357                cached_objects: BTreeMap::new(),
358                is_metered,
359                protocol_config,
360                metrics,
361                current_epoch_id,
362            },
363            store: BTreeMap::new(),
364            is_metered,
365        }
366    }
367
368    pub(super) fn receive_object(
369        &mut self,
370        parent: ObjectID,
371        child: ObjectID,
372        child_version: SequenceNumber,
373        child_ty: &Type,
374        child_layout: &R::MoveTypeLayout,
375        child_fully_annotated_layout: &A::MoveTypeLayout,
376        child_move_type: MoveObjectType,
377    ) -> PartialVMResult<LoadedWithMetadataResult<ObjectResult<Value>>> {
378        let Some((obj, obj_meta)) =
379            self.inner
380                .receive_object_from_store(parent, child, child_version)?
381        else {
382            return Ok(None);
383        };
384
385        Ok(Some(
386            match deserialize_move_object(&obj, child_ty, child_layout, child_move_type)? {
387                ObjectResult::MismatchedType => (ObjectResult::MismatchedType, obj_meta),
388                ObjectResult::Loaded((_, _, v)) => {
389                    // Find all UIDs inside of the value and update the object parent maps with the contained
390                    // UIDs in the received value. They should all have an upper bound version as the receiving object.
391                    // Only do this if we successfully load the object though.
392                    let contained_uids = get_all_uids(child_fully_annotated_layout, obj.contents())
393                        .map_err(|e| {
394                            PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
395                                .with_message(format!(
396                                    "Failed to find UIDs for receiving object. ERROR: {e}"
397                                ))
398                        })?;
399                    for id in contained_uids {
400                        self.inner.root_version.insert(id, child_version);
401                        if id != child {
402                            let prev = self.inner.wrapped_object_containers.insert(id, child);
403                            debug_assert!(prev.is_none())
404                        }
405                    }
406                    (ObjectResult::Loaded(v), obj_meta)
407                }
408            },
409        ))
410    }

412    pub(super) fn object_exists(
413        &mut self,
414        parent: ObjectID,
415        child: ObjectID,
416    ) -> PartialVMResult<bool> {
417        if let Some(child_object) = self.store.get(&child) {
418            return child_object.value.exists();
419        }
420        Ok(self
421            .inner
422            .get_or_fetch_object_from_store(parent, child)?
423            .is_some())
424    }
425
426    pub(super) fn object_exists_and_has_type(
427        &mut self,
428        parent: ObjectID,
429        child: ObjectID,
430        child_move_type: &MoveObjectType,
431    ) -> PartialVMResult<bool> {
432        if let Some(child_object) = self.store.get(&child) {
433            // exists and has same type
434            return Ok(child_object.value.exists()? && &child_object.move_type == child_move_type);
435        }
436        Ok(self
437            .inner
438            .get_or_fetch_object_from_store(parent, child)?
439            .map(|move_obj| move_obj.type_() == child_move_type)
440            .unwrap_or(false))
441    }
442
443    pub(super) fn get_or_fetch_object(
444        &mut self,
445        parent: ObjectID,
446        child: ObjectID,
447        child_ty: &Type,
448        child_layout: &R::MoveTypeLayout,
449        child_fully_annotated_layout: &A::MoveTypeLayout,
450        child_move_type: MoveObjectType,
451    ) -> PartialVMResult<ObjectResult<&mut ChildObject>> {
452        let store_entries_count = self.store.len() as u64;
453        let child_object = match self.store.entry(child) {
454            btree_map::Entry::Vacant(e) => {
455                let (ty, move_type, value) = match self.inner.fetch_object_impl(
456                    parent,
457                    child,
458                    child_ty,
459                    child_layout,
460                    child_fully_annotated_layout,
461                    child_move_type,
462                )? {
463                    ObjectResult::MismatchedType => return Ok(ObjectResult::MismatchedType),
464                    ObjectResult::Loaded(res) => res,
465                };
466
467                if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
468                    self.is_metered,
469                    store_entries_count,
470                    self.inner
471                        .protocol_config
472                        .object_runtime_max_num_store_entries(),
473                    self.inner
474                        .protocol_config
475                        .object_runtime_max_num_store_entries_system_tx(),
476                    self.inner.metrics.excessive_object_runtime_store_entries
477                ) {
478                    return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
479                        .with_message(format!(
480                            "Object runtime store limit ({} entries) reached",
481                            lim
482                        ))
483                        .with_sub_status(
484                            VMMemoryLimitExceededSubStatusCode::OBJECT_RUNTIME_STORE_LIMIT_EXCEEDED
485                                as u64,
486                        ));
487                };
488
489                e.insert(ChildObject {
490                    owner: parent,
491                    ty,
492                    move_type,
493                    value,
494                })
495            }
496            btree_map::Entry::Occupied(e) => {
497                let child_object = e.into_mut();
498                if child_object.move_type != child_move_type {
499                    return Ok(ObjectResult::MismatchedType);
500                }
501                child_object
502            }
503        };
504        Ok(ObjectResult::Loaded(child_object))
505    }

507    pub(super) fn add_object(
508        &mut self,
509        parent: ObjectID,
510        child: ObjectID,
511        child_ty: &Type,
512        child_move_type: MoveObjectType,
513        child_value: Value,
514    ) -> PartialVMResult<()> {
515        if let LimitThresholdCrossed::Hard(_, lim) = check_limit_by_meter!(
516            self.is_metered,
517            self.store.len(),
518            self.inner
519                .protocol_config
520                .object_runtime_max_num_store_entries(),
521            self.inner
522                .protocol_config
523                .object_runtime_max_num_store_entries_system_tx(),
524            self.inner.metrics.excessive_object_runtime_store_entries
525        ) {
526            return Err(PartialVMError::new(StatusCode::MEMORY_LIMIT_EXCEEDED)
527                .with_message(format!(
528                    "Object runtime store limit ({} entries) reached",
529                    lim
530                ))
531                .with_sub_status(
532                    VMMemoryLimitExceededSubStatusCode::OBJECT_RUNTIME_STORE_LIMIT_EXCEEDED as u64,
533                ));
534        };
535
536        let mut value = if let Some(ChildObject { ty, value, .. }) = self.store.remove(&child) {
537            if value.exists()? {
538                return Err(
539                    PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
540                        .with_message(
541                            "Duplicate addition of a child object. \
542                            The previous value cannot be dropped. Indicates possible duplication \
543                            of objects as an object was fetched more than once from two different \
544                            parents, yet was not removed from one first"
545                                .to_string(),
546                        ),
547                );
548            }
549            if self.inner.protocol_config.loaded_child_object_format() {
550                // double check format did not change
551                if !self.inner.protocol_config.loaded_child_object_format_type() && child_ty != &ty
552                {
553                    let msg = format!("Type changed for child {child} when setting the value back");
554                    return Err(
555                        PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR)
556                            .with_message(msg),
557                    );
558                }
559                value
560            } else {
561                GlobalValue::none()
562            }
563        } else {
564            GlobalValue::none()
565        };
566        if let Err((e, _)) = value.move_to(child_value) {
567            return Err(
568                PartialVMError::new(StatusCode::UNKNOWN_INVARIANT_VIOLATION_ERROR).with_message(
569                    format!("Unable to set value for child {child}, with error {e}",),
570                ),
571            );
572        }
573        let child_object = ChildObject {
574            owner: parent,
575            ty: child_ty.clone(),
576            move_type: child_move_type,
577            value,
578        };
579        self.store.insert(child, child_object);
580        Ok(())
581    }

591    // retrieve the `Op` effects for the child objects
592    pub(super) fn take_effects(&mut self) -> BTreeMap<ObjectID, ChildObjectEffect> {
593        std::mem::take(&mut self.store)
594            .into_iter()
595            .filter_map(|(id, child_object)| {
596                let ChildObject {
597                    owner,
598                    ty,
599                    move_type: _,
600                    value,
601                } = child_object;
602                let effect = value.into_effect()?;
603                let child_effect = ChildObjectEffect { owner, ty, effect };
604                Some((id, child_effect))
605            })
606            .collect()
607    }
608
609    pub(super) fn all_active_objects(&self) -> impl Iterator<Item = (&ObjectID, &Type, Value)> {
610        self.store.iter().filter_map(|(id, child_object)| {
611            let child_exists = child_object.value.exists().unwrap();
612            if !child_exists {
613                None
614            } else {
615                let copied_child_value = child_object
616                    .value
617                    .borrow_global()
618                    .unwrap()
619                    .value_as::<StructRef>()
620                    .unwrap()
621                    .read_ref()
622                    .unwrap();
623                Some((id, &child_object.ty, copied_child_value))
624            }
625        })
626    }
