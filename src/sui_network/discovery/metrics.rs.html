4use crate::endpoint_manager::AddressSource;
5use prometheus::{
6    IntGauge, IntGaugeVec, Registry, register_int_gauge_vec_with_registry,
7    register_int_gauge_with_registry,
8};
9use std::sync::Arc;
10use tap::Pipe;
11
12#[derive(Clone)]
13pub(super) struct Metrics(Option<Arc<Inner>>);
14
15impl std::fmt::Debug for Metrics {
16    fn fmt(&self, fmt: &mut std::fmt::Formatter) -> std::fmt::Result {
17        fmt.debug_struct("Metrics").finish()
18    }
19}
20
21impl Metrics {
22    pub fn enabled(registry: &Registry) -> Self {
23        Metrics(Some(Inner::new(registry)))
24    }
25
26    pub fn disabled() -> Self {
27        Metrics(None)
28    }
29
30    /// Sets the gauge of distinct peers known to have an external address.
31    pub fn set_num_peers_with_external_address(&self, value: i64) {
32        if let Some(inner) = &self.0 {
33            inner.num_peers_with_external_address.set(value);
34        }
35    }
36
37    /// Records the currently-active P2P address source for a trusted peer as a
38    /// single per-peer gauge whose value is the source's `metric_code` (0 = no
39    /// address installed / all sources cleared).
40    pub fn set_active_p2p_address_source(&self, peer_id: &str, active: Option<AddressSource>) {
41        if let Some(inner) = &self.0 {
42            let code = active.map_or(0, AddressSource::metric_code);
43            inner
44                .active_p2p_address_source
45                .with_label_values(&[peer_id])
46                .set(code);
47        }
48    }
49}
50
51struct Inner {
52    num_peers_with_external_address: IntGauge,
53    active_p2p_address_source: IntGaugeVec,
54}
55
56impl Inner {
57    pub fn new(registry: &Registry) -> Arc<Self> {
58        Self {
59            num_peers_with_external_address: register_int_gauge_with_registry!(
60                "num_peers_with_external_address",
61                "Number of peers with an external address configured for discovery",
62                registry
63            )
64            .unwrap(),
65            active_p2p_address_source: register_int_gauge_vec_with_registry!(
66                "discovery_active_p2p_address_source",
67                "Active P2P address source per trusted peer, encoded as the gauge value: \
68                 0=none (no address installed), 1=admin, 2=config, 3=discovery, 4=seed, \
69                 5=chain (highest to lowest priority). One series per peer; `peer_id` is the \
70                 full hex peer id.",
71                &["peer_id"],
72                registry
73            )
74            .unwrap(),
75        }
76        .pipe(Arc::new)
77    }
78}
