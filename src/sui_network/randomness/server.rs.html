4use super::{Randomness, RandomnessMessage, SendSignaturesRequest};
5use anemo::{Request, Response};
6use tokio::sync::mpsc;
7
8pub(super) struct Server {
9    pub(super) sender: mpsc::WeakSender<RandomnessMessage>,
10}
11
12#[anemo::async_trait]
13impl Randomness for Server {
14    async fn send_signatures(
15        &self,
16        request: Request<SendSignaturesRequest>,
17    ) -> Result<Response<()>, anemo::rpc::Status> {
18        let sender = self
19            .sender
20            .upgrade()
21            .ok_or_else(|| anemo::rpc::Status::internal("shutting down"))?;
22        let peer_id = *request
23            .peer_id()
24            .ok_or_else(|| anemo::rpc::Status::internal("missing peer ID"))?;
25        let SendSignaturesRequest {
26            epoch,
27            round,
28            partial_sigs,
29            sig,
30        } = request.into_inner();
31        let _ = sender // throw away error, caller will retry
32            .send(RandomnessMessage::ReceiveSignatures(
33                peer_id,
34                epoch,
35                round,
36                partial_sigs,
37                sig,
38            ))
39            .await;
40        Ok(anemo::Response::new(()))
41    }
42}
