43// TODO: allow more flexible decimals
44const DECIMAL: u8 = 6;
45const METRICS_MULTIPLIER: f64 = 10u64.pow(DECIMAL as u32) as f64;
46const UPLOAD_FAILURE_RECOVER_SEC: u64 = 10;
47static STALE_OBJ_ERROR: OnceCell<String> = OnceCell::new();

49pub struct OracleNode {
50    upload_feeds: HashMap<String, HashMap<String, UploadFeedConfig>>,
51    gas_obj_id: ObjectID,
52    download_feeds: DownloadFeedConfigs,
53    wallet_ctx: WalletContext,
54    metrics: Arc<OracleMetrics>,
55}
56
57impl OracleNode {
58    pub fn new(
59        upload_feeds: HashMap<String, HashMap<String, UploadFeedConfig>>,
60        gas_obj_id: ObjectID,
61        download_feeds: DownloadFeedConfigs,
62        wallet_ctx: WalletContext,
63        registry: Registry,
64    ) -> Self {
65        Self {
66            upload_feeds,
67            gas_obj_id,
68            download_feeds,
69            wallet_ctx,
70            metrics: Arc::new(OracleMetrics::new(&registry)),
71        }
72    }
73
74    pub async fn run(mut self) -> anyhow::Result<()> {
75        info!("Starting OracleNode...");
76        let signer_address = self.wallet_ctx.active_address()?;
77        let client = Arc::new(self.wallet_ctx.get_client().await?);
78
79        let wallet_ctx = Arc::new(self.wallet_ctx);
80        DataProviderRunner::new(
81            self.upload_feeds,
82            self.gas_obj_id,
83            wallet_ctx,
84            client.clone(),
85            signer_address,
86            self.metrics.clone(),
87        )
88        .await
89        .spawn();
90
91        let (sender, mut receiver) = tokio::sync::mpsc::channel(1000);
92
93        // Spawn a reader thread if reader_interval is configured.
94        if let Some(read_interval) = self.download_feeds.read_interval {
95            tokio::spawn(
96                OnChainDataReader {
97                    client: client.clone(),
98                    read_interval,
99                    read_configs: self.download_feeds.read_feeds,
100                    metrics: self.metrics.clone(),
101                }
102                .start(sender.clone()),
103            );
104        }
105
106        while let Some((read_feed, object_id, value)) = receiver.recv().await {
107            info!(
108                read_feed,
109                ?object_id,
110                ?value,
111                "Received data from on chain reader."
112            );
113        }
114
115        Ok(())
116    }
117}

124impl DataProviderRunner {
125    pub async fn new(
126        upload_feeds: HashMap<String, HashMap<String, UploadFeedConfig>>,
127        gas_coin_id: ObjectID,
128        wallet_ctx: Arc<WalletContext>,
129        client: Arc<SuiClient>,
130        signer_address: SuiAddress,
131        metrics: Arc<OracleMetrics>,
132    ) -> Self {
133        let mut providers = vec![];
134        let mut staleness_tolerance = HashMap::new();
135        let mut oracle_object_args = HashMap::new();
136        let (sender, receiver) = tokio::sync::mpsc::channel(10000);
137        for (feed_name, upload_feed) in upload_feeds {
138            for (source_name, data_feed) in upload_feed {
139                staleness_tolerance.insert(
140                    make_onchain_feed_name(&feed_name, &source_name),
141                    data_feed.submission_interval,
142                );
143                let oracle_obj_id = data_feed.upload_parameters.write_data_provider_object_id;
144                let data_provider = DataProvider {
145                    feed_name: feed_name.clone(),
146                    source_name: source_name.clone(),
147                    upload_feed: Arc::new(data_feed),
148                    sender: sender.clone(),
149                    metrics: metrics.clone(),
150                };
151                providers.push(Arc::new(data_provider));
152                if let std::collections::hash_map::Entry::Vacant(e) =
153                    oracle_object_args.entry(oracle_obj_id)
154                {
155                    e.insert(
156                        get_object_arg(client.read_api(), oracle_obj_id, true)
157                            .await
158                            .unwrap(),
159                    );
160                }
161            }
162        }
163        info!("Staleness tolerance: {:?}", staleness_tolerance);
164
165        let gas_obj_ref = get_gas_obj_ref(client.read_api(), gas_coin_id, signer_address).await;
166        info!("Gas object: {:?}", gas_obj_ref);
167
168        let uploader = OnChainDataUploader {
169            wallet_ctx: wallet_ctx.clone(),
170            client: client.clone(),
171            receiver,
172            signer_address,
173            gas_obj_ref,
174            staleness_tolerance,
175            oracle_object_args,
176            metrics: metrics.clone(),
177        };
178        Self {
179            providers,
180            uploader,
181        }
182    }
183
184    pub fn spawn(mut self) {
185        for data_provider in self.providers {
186            tokio::spawn(async move {
187                data_provider.run().await;
188            });
189        }
190        tokio::spawn(async move {
191            self.uploader.run().await;
192        });
193    }
194}

196async fn get_gas_obj_ref(
197    read_api: &ReadApi,
198    gas_obj_id: ObjectID,
199    owner_address: SuiAddress,
200) -> ObjectRef {
201    loop {
202        match read_api
203            .get_object_with_options(gas_obj_id, SuiObjectDataOptions::default().with_owner())
204            .await
205            .map(|resp| resp.data)
206        {
207            Ok(Some(gas_obj)) => {
208                assert_eq!(
209                    gas_obj.owner,
210                    Some(Owner::AddressOwner(owner_address)),
211                    "Provided gas obj {:?} does not belong to {}",
212                    gas_obj,
213                    owner_address
214                );
215                return gas_obj.object_ref();
216            }
217            other => {
218                warn!("Can't get gas object: {:?}: {:?}", gas_obj_id, other);
219                tokio::time::sleep(Duration::from_secs(5)).await;
220            }
221        }
222    }

225struct DataProvider {
226    pub feed_name: String,
227    pub source_name: String,
228    pub upload_feed: Arc<UploadFeedConfig>,
229    pub sender: tokio::sync::mpsc::Sender<DataPoint>,
230    metrics: Arc<OracleMetrics>,
231}
232
233impl DataProvider {
234    pub async fn run(&self) {
235        info!(
236            feed_name = self.feed_name,
237            source_name = self.source_name,
238            "Starting DataProvider"
239        );
240        let mut interval = tokio::time::interval(self.upload_feed.submission_interval);
241        interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
242
243        loop {
244            interval.tick().await;
245            self.run_once().await;
246        }
247    }
248
249    async fn run_once(&self) {
250        debug!(
251            feed_name = self.feed_name,
252            source_name = self.source_name,
253            "Running data provider once."
254        );
255        let value = self.retrieve_from_data_source().await;
256        if value.is_err() {
257            error!(
258                feed_name = self.feed_name,
259                source_name = self.source_name,
260                "Failed to retrieve data from data source: {:?}",
261                value
262            );
263            self.metrics
264                .data_source_errors
265                .with_label_values(&[&self.feed_name, &self.source_name])
266                .inc();
267            return;
268        }
269
270        self.metrics
271            .data_source_successes
272            .with_label_values(&[&self.feed_name, &self.source_name])
273            .inc();
274
275        // TODO: allow more flexible multiplers and data types
276        let value = (value.unwrap() * METRICS_MULTIPLIER) as u64;
277        self.send_to_uploader(value).await;
278    }
279
280    async fn retrieve_from_data_source(&self) -> anyhow::Result<f64> {
281        // TODO: support websocket
282        let url = &self.upload_feed.data_source_config.url;
283        let json_path = &self.upload_feed.data_source_config.json_path;
284        let response = reqwest::Client::new().get(url).send().await?;
285
286        if !response.status().is_success() {
287            anyhow::bail!("Failed to fetch data: {:?}", response);
288        }
289
290        let json_blob: serde_json::Value = response.json().await.unwrap();
291        let data = jsonpath_lib::select(&json_blob, json_path)?;
292
293        if data.is_empty() {
294            anyhow::bail!(
295                "Failed to find data from json blob: {:?} with json path: {:?}",
296                json_blob,
297                json_path
298            );
299        }
300        // Assume there is one single value per request
301        match data[0].as_str() {
302            Some(value_str) => match value_str.parse::<f64>() {
303                Ok(value) => Ok(value),
304                Err(_) => anyhow::bail!(
305                    "Failed to parse data {:?} as f64 from json blob: {:?}",
306                    data[0],
307                    json_blob
308                ),
309            },
310            None => anyhow::bail!(
311                "Failed to parse data {:?} as string from json blob: {:?}",
312                data[0],
313                json_blob
314            ),
315        }
316    }
317
318    async fn send_to_uploader(&self, value: u64) {
319        let _ = self
320            .sender
321            .send(DataPoint {
322                feed_name: make_onchain_feed_name(&self.feed_name, &self.source_name),
323                upload_parameters: self.upload_feed.upload_parameters.clone(),
324                value,
325                retrieval_timestamp: SystemTime::now(),
326                retrieval_instant: Instant::now(),
327            })
328            .await
329            .tap_err(|err| error!("Failed to send data point to uploader: {:?}", err));
330    }
331}

333fn make_onchain_feed_name(feed_name: &str, source_name: &str) -> String {
334    format!(
335        "{}-{}",
336        feed_name.to_ascii_lowercase(),
337        source_name.to_ascii_lowercase()
338    )

341struct OnChainDataUploader {
342    wallet_ctx: Arc<WalletContext>,
343    client: Arc<SuiClient>,
344    receiver: tokio::sync::mpsc::Receiver<DataPoint>,
345    signer_address: SuiAddress,
346    gas_obj_ref: ObjectRef,
347    staleness_tolerance: HashMap<String, Duration>,
348    oracle_object_args: HashMap<ObjectID, ObjectArg>,
349    metrics: Arc<OracleMetrics>,
350}
351
352impl OnChainDataUploader {
353    async fn run(&mut self) {
354        info!("Starting OnChainDataUploader");
355        // The minimal latency is 1 second so we collect data every 1 second
356        let mut read_interval = tokio::time::interval(Duration::from_millis(500));
357        read_interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
358        loop {
359            read_interval.tick().await;
360            let data_points = self.collect().await;
361            if !data_points.is_empty()
362                && let Err(err) = self.upload(data_points).await
363            {
364                error!(
365                    "Upload failure: {err}. About to resting for {UPLOAD_FAILURE_RECOVER_SEC} sec."
366                );
367                tokio::time::sleep(Duration::from_secs(UPLOAD_FAILURE_RECOVER_SEC)).await;
368                self.gas_obj_ref = get_gas_obj_ref(
369                    self.client.read_api(),
370                    self.gas_obj_ref.0,
371                    self.signer_address,
372                )
373                .await;
374                error!("Updated gas object reference: {:?}", self.gas_obj_ref);
375            }
376        }
377    }
378
379    async fn collect(&mut self) -> Vec<DataPoint> {
380        let start = Instant::now();
381        let mut data_points = vec![];
382        while let Ok(Some(data_point)) =
383            tokio::time::timeout(Duration::from_millis(100), self.receiver.recv()).await
384        {
385            let feed_name = &data_point.feed_name;
386            debug!(
387                feed_name = data_point.feed_name,
388                value = data_point.value,
389                "Received data from data provider."
390            );
391            // TODO: for each source, at most take one value in each submission
392            let staleness_tolerance =
393                self.staleness_tolerance.get(feed_name).unwrap_or_else(|| {
394                    panic!("Bug, missing staleness tolerance for feed: {}", feed_name)
395                });
396            let duration_since = data_point.retrieval_instant.elapsed();
397            if duration_since > staleness_tolerance.add(Duration::from_secs(1)) {
398                warn!(
399                    feed_name,
400                    value = data_point.value,
401                    ?duration_since,
402                    ?staleness_tolerance,
403                    "Data is too stale, skipping."
404                );
405                self.metrics
406                    .data_staleness
407                    .with_label_values(&[feed_name])
408                    .inc();
409            } else {
410                data_points.push(data_point);
411            }
412
413            // One run only waits for 1 second
414            // But if we don't have any valid data points, we wait until we do.
415            if data_points.is_empty() && start.elapsed() >= Duration::from_millis(500) {
416                break;
417            }
418        }
419        debug!("Collected {} data points", data_points.len());
420        data_points
421    }

423    async fn upload(
424        &mut self,
425        data_points: Vec<DataPoint>,
426    ) -> anyhow::Result<SuiTransactionBlockEffects> {
427        let _scope = monitored_scope("Oracle::OnChainDataUploader::upload");
428        // TODO add more error handling & polling perhaps
429        let mut builder = ProgrammableTransactionBuilder::new();
430        let mut is_first = true;
431        for data_point in &data_points {
432            let package_id = data_point.upload_parameters.write_package_id;
433            let feed_name = &data_point.feed_name;
434            let oracle_obj_arg = *self
435                .oracle_object_args
436                .get(&data_point.upload_parameters.write_data_provider_object_id)
437                .unwrap_or_else(|| {
438                    panic!("Bug, missing oracle object arg for feed: {}", feed_name)
439                });
440            let duration_since_start = data_point.retrieval_instant.elapsed();
441            let data_point_ts: DateTime<Utc> =
442                DateTime::from(data_point.retrieval_timestamp + duration_since_start);
443
444            let mut arguments = if is_first {
445                vec![
446                    builder.input(CallArg::Object(oracle_obj_arg)).unwrap(),
447                    builder.input(CallArg::CLOCK_IMM).unwrap(),
448                ]
449            } else {
450                vec![Argument::Input(0), Argument::Input(1)]
451            };
452
453            let decimal = builder
454                .input(CallArg::Pure(bcs::to_bytes(&DECIMAL).unwrap()))
455                .unwrap();
456            let value = builder
457                .input(CallArg::Pure(bcs::to_bytes(&data_point.value).unwrap()))
458                .unwrap();
459
460            arguments.extend_from_slice(&[
461                builder
462                    .input(CallArg::Pure(bcs::to_bytes(&feed_name)?))
463                    .unwrap(),
464                builder.programmable_move_call(
465                    package_id,
466                    Identifier::from_str("decimal_value").unwrap(),
467                    Identifier::from_str("new").unwrap(),
468                    vec![],
469                    vec![value, decimal],
470                ),
471                builder
472                    .input(CallArg::Pure(bcs::to_bytes(&format!("{}", data_point_ts))?))
473                    .unwrap(),
474            ]);
475
476            builder.command(Command::move_call(
477                package_id,
478                Identifier::new(data_point.upload_parameters.write_module_name.clone()).unwrap(),
479                Identifier::new(data_point.upload_parameters.write_function_name.clone()).unwrap(),
480                // TODO: allow more generic data types
481                vec![
482                    parse_sui_type_tag(&format!("{package_id}::decimal_value::DecimalValue"))
483                        .unwrap(),
484                ],
485                arguments,
486            ));
487            is_first = false;
488        }
489        let pt = builder.finish();
490        let rgp = self
491            .client
492            .governance_api()
493            .get_reference_gas_price()
494            .await?;
495        let tx = TransactionData::new_programmable(
496            self.signer_address,
497            vec![self.gas_obj_ref],
498            pt,
499            // 15_000_000 is a heuristic number
500            15_000_000 * data_points.len() as u64,
501            rgp,
502        );
503
504        let signed_tx = self.wallet_ctx.sign_transaction(&tx).await;
505        let tx_digest = *signed_tx.digest();
506
507        let timer_start = Instant::now();
508        let response = self.execute(signed_tx).await?;
509        let time_spend_sec = timer_start.elapsed().as_secs_f32();
510
511        // We asked for effects.
512        // But is there a better way to handle this instead of panic?
513        let effects = response.effects.expect("Expect to see effects in response");
514
515        // It's critical to update the gas object reference for next transaction
516        self.gas_obj_ref = effects.gas_object().reference.to_object_ref();
517
518        let success = effects.status().is_ok();
519
520        // Update metrics
521        for data_point in &data_points {
522            if success {
523                self.metrics
524                    .upload_successes
525                    .with_label_values(&[&data_point.feed_name])
526                    .inc();
527                self.metrics
528                    .uploaded_values
529                    .with_label_values(&[&data_point.feed_name])
530                    .observe(data_point.value as f64);
531            } else {
532                self.metrics
533                    .upload_data_errors
534                    .with_label_values(&[&data_point.feed_name])
535                    .inc();
536            }
537        }
538
539        let gas_usage = effects.gas_cost_summary().gas_used();
540        let storage_rebate = effects.gas_cost_summary().storage_rebate;
541        let computation_cost = effects.gas_cost_summary().computation_cost;
542        let net_gas_usage = effects.gas_cost_summary().net_gas_usage();
543        self.metrics
544            .computation_gas_used
545            .observe(computation_cost as f64);
546        self.metrics.total_gas_cost.inc_by(gas_usage);
547        self.metrics.total_gas_rebate.inc_by(storage_rebate);
548
549        if success {
550            self.metrics
551                .total_data_points_uploaded
552                .inc_by(data_points.len() as u64);
553            info!(
554                ?tx_digest,
555                net_gas_usage,
556                time_spend_sec,
557                "Upload succeeded with {} data points",
558                data_points.len(),
559            );
560            Ok(effects)
561        } else {
562            error!(
563                ?tx_digest,
564                net_gas_usage,
565                "Upload failed with {} data points. Err: {:?}",
566                data_points.len(),
567                effects.status(),
568            );
569            anyhow::bail!("Failed to submit data on chain: {:?}", effects.status());
570        }
571    }

573    async fn execute(&mut self, tx: Transaction) -> anyhow::Result<SuiTransactionBlockResponse> {
574        let tx_digest = tx.digest();
575        let mut retry_attempts = 3;
576        loop {
577            match self
578                .client
579                .quorum_driver_api()
580                .execute_transaction_block(
581                    tx.clone(),
582                    SuiTransactionBlockResponseOptions::new().with_effects(),
583                    // TODO: after 1.4.0, we can simply use `WaitForEffectsCert` which is faster.
584                    // Some(sui_types::quorum_driver_types::ExecuteTransactionRequestType::WaitForEffectsCert),
585                    Some(sui_types::quorum_driver_types::ExecuteTransactionRequestType::WaitForLocalExecution),
586                )
587                .await {
588                Ok(response) => return Ok(response),
589                Err(sui_sdk::error::Error::RpcError(err)) => {
590                    // jsonrpsee translate every SuiError into jsonrpsee::core::Error, so we need to further distinguish 
591                    if err.to_string().contains(NON_RECOVERABLE_ERROR_MSG) {
592                        let stale_obj_error = STALE_OBJ_ERROR
593                            .get_or_init(||
594                                String::from(UserInputError::ObjectVersionUnavailableForConsumption { provided_obj_ref: random_object_ref(), current_version: 0.into() }.as_ref())
595                            );
596                        if err.to_string().contains(stale_obj_error) {
597                            error!(?tx_digest, "Failed to submit tx, it looks like gas object is stale : {:?}", err);
598                            let new_ref = get_gas_obj_ref(self.client.read_api(), self.gas_obj_ref.0, self.signer_address).await;
599                            self.gas_obj_ref = new_ref;
600                            info!("Gas object updated: {:?}", new_ref);
601                            anyhow::bail!("Gas object is stale, now updated to {:?}. tx_digest={:?}", new_ref, tx_digest);
602                        } else {
603                            error!(?tx_digest, "Failed to submit tx, with non recoverable error: {:?}", err);
604                            anyhow::bail!("Non-retryable error {:?}. tx_digest={:?}", err, tx_digest);
605                        }
606                    }
607                    // Likely retryable error?
608                    error!(?tx_digest, "Failed to submit tx, with (likely) recoverable error: {:?}. Remaining retry times: {}", err, retry_attempts);
609                    retry_attempts -= 1;
610                    if retry_attempts <= 0 {
611                        anyhow::bail!("Too many RPC errors: {}. tx_digest={:?}", err, tx_digest);
612                    }
613                }
614                // All other errors are unexpected
615                Err(err) => {
616                    error!(?tx_digest, "Failed to submit tx, with unexpected error: {:?}", err);
617                    anyhow::bail!("Unexpected error in tx submission {:?}. tx_digest={:?}", err, tx_digest);
618                }
619            }
620        }
621    }
622}

633struct OnChainDataReader {
634    pub client: Arc<SuiClient>,
635    // For now we share one read interval for all reads
636    pub read_interval: Duration,
637    pub read_configs: HashMap<String, ObjectID>,
638    metrics: Arc<OracleMetrics>,
639}
640
641impl OnChainDataReader {
642    pub async fn start(self, sender: tokio::sync::mpsc::Sender<(String, ObjectID, f64)>) {
643        info!(
644            "Starting on-chain data reader with interval {:?} and config: {:?}",
645            self.read_interval, self.read_configs
646        );
647        let mut read_interval = tokio::time::interval(self.read_interval);
648        read_interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
649        loop {
650            read_interval.tick().await;
651            for (feed_name, object_id) in &self.read_configs {
652                match self
653                    .client
654                    .read_api()
655                    .get_object_with_options(*object_id, SuiObjectDataOptions::default())
656                    .await
657                {
658                    Ok(SuiObjectResponse {
659                        data: Some(_data), ..
660                    }) => {
661                        // TODO parse value based on returned BCS
662                        let value = 5_f64;
663                        let _ = sender.send((feed_name.clone(), *object_id, value)).await;
664                        self.metrics
665                            .downloaded_values
666                            .with_label_values(&[feed_name])
667                            .observe(value * METRICS_MULTIPLIER);
668                        self.metrics
669                            .download_successes
670                            .with_label_values(&[feed_name, &object_id.to_string()])
671                            .inc();
672                    }
673                    other => {
674                        error!(
675                            read_feed = feed_name,
676                            ?object_id,
677                            "Failed to read data from on-chain: {:?}",
678                            other
679                        );
680                        self.metrics
681                            .download_data_errors
682                            .with_label_values(&[feed_name, &object_id.to_string()])
683                            .inc();
684                    }
685                }
686            }
687        }
688    }
689}

691async fn get_object_arg(
692    read_api: &ReadApi,
693    id: ObjectID,
694    is_mutable_ref: bool,
695) -> anyhow::Result<ObjectArg> {
696    let response = read_api
697        .get_object_with_options(id, SuiObjectDataOptions::bcs_lossless())
698        .await?;
699
700    let obj: Object = response.into_object()?.try_into()?;
701    let obj_ref = obj.compute_object_reference();
702    let owner = obj.owner.clone();
703    Ok(match owner {
704        Owner::Shared {
705            initial_shared_version,
706        }
707        | Owner::ConsensusAddressOwner {
708            start_version: initial_shared_version,
709            ..
710        } => ObjectArg::SharedObject {
711            id,
712            initial_shared_version,
713            mutability: if is_mutable_ref {
714                SharedObjectMutability::Mutable
715            } else {
716                SharedObjectMutability::Immutable
717            },
718        },
719        Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
720            ObjectArg::ImmOrOwnedObject(obj_ref)
721        }
722    })
723}
