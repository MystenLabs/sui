38// TODO: allow more flexible decimals
39const DECIMAL: u8 = 6;
40const METRICS_MULTIPLIER: f64 = 10u64.pow(DECIMAL as u32) as f64;
41const UPLOAD_FAILURE_RECOVER_SEC: u64 = 10;
42static STALE_OBJ_ERROR: OnceCell<String> = OnceCell::new();

44pub struct OracleNode {
45    upload_feeds: HashMap<String, HashMap<String, UploadFeedConfig>>,
46    gas_obj_id: ObjectID,
47    download_feeds: DownloadFeedConfigs,
48    wallet_ctx: WalletContext,
49    metrics: Arc<OracleMetrics>,
50}
51
52impl OracleNode {
53    pub fn new(
54        upload_feeds: HashMap<String, HashMap<String, UploadFeedConfig>>,
55        gas_obj_id: ObjectID,
56        download_feeds: DownloadFeedConfigs,
57        wallet_ctx: WalletContext,
58        registry: Registry,
59    ) -> Self {
60        Self {
61            upload_feeds,
62            gas_obj_id,
63            download_feeds,
64            wallet_ctx,
65            metrics: Arc::new(OracleMetrics::new(&registry)),
66        }
67    }
68
69    pub async fn run(mut self) -> anyhow::Result<()> {
70        info!("Starting OracleNode...");
71        let signer_address = self.wallet_ctx.active_address()?;
72        let client = self.wallet_ctx.grpc_client()?;
73
74        let wallet_ctx = Arc::new(self.wallet_ctx);
75        DataProviderRunner::new(
76            self.upload_feeds,
77            self.gas_obj_id,
78            wallet_ctx,
79            client.clone(),
80            signer_address,
81            self.metrics.clone(),
82        )
83        .await
84        .spawn();
85
86        let (sender, mut receiver) = tokio::sync::mpsc::channel(1000);
87
88        // Spawn a reader thread if reader_interval is configured.
89        if let Some(read_interval) = self.download_feeds.read_interval {
90            tokio::spawn(
91                OnChainDataReader {
92                    client: client.clone(),
93                    read_interval,
94                    read_configs: self.download_feeds.read_feeds,
95                    metrics: self.metrics.clone(),
96                }
97                .start(sender.clone()),
98            );
99        }
100
101        while let Some((read_feed, object_id, value)) = receiver.recv().await {
102            info!(
103                read_feed,
104                ?object_id,
105                ?value,
106                "Received data from on chain reader."
107            );
108        }
109
110        Ok(())
111    }
112}

114struct DataProviderRunner {
115    providers: Vec<Arc<DataProvider>>,
116    uploader: OnChainDataUploader,
117}

119impl DataProviderRunner {
120    pub async fn new(
121        upload_feeds: HashMap<String, HashMap<String, UploadFeedConfig>>,
122        gas_coin_id: ObjectID,
123        wallet_ctx: Arc<WalletContext>,
124        mut client: Client,
125        signer_address: SuiAddress,
126        metrics: Arc<OracleMetrics>,
127    ) -> Self {
128        let mut providers = vec![];
129        let mut staleness_tolerance = HashMap::new();
130        let mut oracle_object_args = HashMap::new();
131        let (sender, receiver) = tokio::sync::mpsc::channel(10000);
132        for (feed_name, upload_feed) in upload_feeds {
133            for (source_name, data_feed) in upload_feed {
134                staleness_tolerance.insert(
135                    make_onchain_feed_name(&feed_name, &source_name),
136                    data_feed.submission_interval,
137                );
138                let oracle_obj_id = data_feed.upload_parameters.write_data_provider_object_id;
139                let data_provider = DataProvider {
140                    feed_name: feed_name.clone(),
141                    source_name: source_name.clone(),
142                    upload_feed: Arc::new(data_feed),
143                    sender: sender.clone(),
144                    metrics: metrics.clone(),
145                };
146                providers.push(Arc::new(data_provider));
147                if let std::collections::hash_map::Entry::Vacant(e) =
148                    oracle_object_args.entry(oracle_obj_id)
149                {
150                    e.insert(
151                        get_object_arg(&mut client, oracle_obj_id, true)
152                            .await
153                            .unwrap(),
154                    );
155                }
156            }
157        }
158        info!("Staleness tolerance: {:?}", staleness_tolerance);
159
160        let gas_obj_ref = get_gas_obj_ref(client.clone(), gas_coin_id, signer_address).await;
161        info!("Gas object: {:?}", gas_obj_ref);
162
163        let uploader = OnChainDataUploader {
164            wallet_ctx: wallet_ctx.clone(),
165            client: client.clone(),
166            receiver,
167            signer_address,
168            gas_obj_ref,
169            staleness_tolerance,
170            oracle_object_args,
171            metrics: metrics.clone(),
172        };
173        Self {
174            providers,
175            uploader,
176        }
177    }
178
179    pub fn spawn(mut self) {
180        for data_provider in self.providers {
181            tokio::spawn(async move {
182                data_provider.run().await;
183            });
184        }
185        tokio::spawn(async move {
186            self.uploader.run().await;
187        });
188    }
189}
190
191async fn get_gas_obj_ref(
192    mut client: Client,
193    gas_obj_id: ObjectID,
194    owner_address: SuiAddress,
195) -> ObjectRef {
196    loop {
197        match client.get_object(gas_obj_id).await {
198            Ok(gas_obj) => {
199                assert_eq!(
200                    gas_obj.owner(),
201                    &Owner::AddressOwner(owner_address),
202                    "Provided gas obj {:?} does not belong to {}",
203                    gas_obj,
204                    owner_address
205                );
206                return gas_obj.compute_object_reference();
207            }
208            other => {
209                warn!("Can't get gas object: {:?}: {:?}", gas_obj_id, other);
210                tokio::time::sleep(Duration::from_secs(5)).await;
211            }
212        }
213    }

216struct DataProvider {
217    pub feed_name: String,
218    pub source_name: String,
219    pub upload_feed: Arc<UploadFeedConfig>,
220    pub sender: tokio::sync::mpsc::Sender<DataPoint>,
221    metrics: Arc<OracleMetrics>,
222}
223
224impl DataProvider {
225    pub async fn run(&self) {
226        info!(
227            feed_name = self.feed_name,
228            source_name = self.source_name,
229            "Starting DataProvider"
230        );
231        let mut interval = tokio::time::interval(self.upload_feed.submission_interval);
232        interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
233
234        loop {
235            interval.tick().await;
236            self.run_once().await;
237        }
238    }
239
240    async fn run_once(&self) {
241        debug!(
242            feed_name = self.feed_name,
243            source_name = self.source_name,
244            "Running data provider once."
245        );
246        let value = self.retrieve_from_data_source().await;
247        if value.is_err() {
248            error!(
249                feed_name = self.feed_name,
250                source_name = self.source_name,
251                "Failed to retrieve data from data source: {:?}",
252                value
253            );
254            self.metrics
255                .data_source_errors
256                .with_label_values(&[&self.feed_name, &self.source_name])
257                .inc();
258            return;
259        }
260
261        self.metrics
262            .data_source_successes
263            .with_label_values(&[&self.feed_name, &self.source_name])
264            .inc();
265
266        // TODO: allow more flexible multiplers and data types
267        let value = (value.unwrap() * METRICS_MULTIPLIER) as u64;
268        self.send_to_uploader(value).await;
269    }
270
271    async fn retrieve_from_data_source(&self) -> anyhow::Result<f64> {
272        // TODO: support websocket
273        let url = &self.upload_feed.data_source_config.url;
274        let json_path = &self.upload_feed.data_source_config.json_path;
275        let response = reqwest::Client::new().get(url).send().await?;
276
277        if !response.status().is_success() {
278            anyhow::bail!("Failed to fetch data: {:?}", response);
279        }
280
281        let json_blob: serde_json::Value = response.json().await.unwrap();
282        let data = jsonpath_lib::select(&json_blob, json_path)?;
283
284        if data.is_empty() {
285            anyhow::bail!(
286                "Failed to find data from json blob: {:?} with json path: {:?}",
287                json_blob,
288                json_path
289            );
290        }
291        // Assume there is one single value per request
292        match data[0].as_str() {
293            Some(value_str) => match value_str.parse::<f64>() {
294                Ok(value) => Ok(value),
295                Err(_) => anyhow::bail!(
296                    "Failed to parse data {:?} as f64 from json blob: {:?}",
297                    data[0],
298                    json_blob
299                ),
300            },
301            None => anyhow::bail!(
302                "Failed to parse data {:?} as string from json blob: {:?}",
303                data[0],
304                json_blob
305            ),
306        }
307    }
308
309    async fn send_to_uploader(&self, value: u64) {
310        let _ = self
311            .sender
312            .send(DataPoint {
313                feed_name: make_onchain_feed_name(&self.feed_name, &self.source_name),
314                upload_parameters: self.upload_feed.upload_parameters.clone(),
315                value,
316                retrieval_timestamp: SystemTime::now(),
317                retrieval_instant: Instant::now(),
318            })
319            .await
320            .tap_err(|err| error!("Failed to send data point to uploader: {:?}", err));
321    }
322}
323
324fn make_onchain_feed_name(feed_name: &str, source_name: &str) -> String {
325    format!(
326        "{}-{}",
327        feed_name.to_ascii_lowercase(),
328        source_name.to_ascii_lowercase()
329    )
330}

332struct OnChainDataUploader {
333    wallet_ctx: Arc<WalletContext>,
334    client: Client,
335    receiver: tokio::sync::mpsc::Receiver<DataPoint>,
336    signer_address: SuiAddress,
337    gas_obj_ref: ObjectRef,
338    staleness_tolerance: HashMap<String, Duration>,
339    oracle_object_args: HashMap<ObjectID, ObjectArg>,
340    metrics: Arc<OracleMetrics>,
341}
342
343impl OnChainDataUploader {
344    async fn run(&mut self) {
345        info!("Starting OnChainDataUploader");
346        // The minimal latency is 1 second so we collect data every 1 second
347        let mut read_interval = tokio::time::interval(Duration::from_millis(500));
348        read_interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
349        loop {
350            read_interval.tick().await;
351            let data_points = self.collect().await;
352            if !data_points.is_empty()
353                && let Err(err) = self.upload(data_points).await
354            {
355                error!(
356                    "Upload failure: {err}. About to resting for {UPLOAD_FAILURE_RECOVER_SEC} sec."
357                );
358                tokio::time::sleep(Duration::from_secs(UPLOAD_FAILURE_RECOVER_SEC)).await;
359                self.gas_obj_ref =
360                    get_gas_obj_ref(self.client.clone(), self.gas_obj_ref.0, self.signer_address)
361                        .await;
362                error!("Updated gas object reference: {:?}", self.gas_obj_ref);
363            }
364        }
365    }
366
367    async fn collect(&mut self) -> Vec<DataPoint> {
368        let start = Instant::now();
369        let mut data_points = vec![];
370        while let Ok(Some(data_point)) =
371            tokio::time::timeout(Duration::from_millis(100), self.receiver.recv()).await
372        {
373            let feed_name = &data_point.feed_name;
374            debug!(
375                feed_name = data_point.feed_name,
376                value = data_point.value,
377                "Received data from data provider."
378            );
379            // TODO: for each source, at most take one value in each submission
380            let staleness_tolerance =
381                self.staleness_tolerance.get(feed_name).unwrap_or_else(|| {
382                    panic!("Bug, missing staleness tolerance for feed: {}", feed_name)
383                });
384            let duration_since = data_point.retrieval_instant.elapsed();
385            if duration_since > staleness_tolerance.add(Duration::from_secs(1)) {
386                warn!(
387                    feed_name,
388                    value = data_point.value,
389                    ?duration_since,
390                    ?staleness_tolerance,
391                    "Data is too stale, skipping."
392                );
393                self.metrics
394                    .data_staleness
395                    .with_label_values(&[feed_name])
396                    .inc();
397            } else {
398                data_points.push(data_point);
399            }
400
401            // One run only waits for 1 second
402            // But if we don't have any valid data points, we wait until we do.
403            if data_points.is_empty() && start.elapsed() >= Duration::from_millis(500) {
404                break;
405            }
406        }
407        debug!("Collected {} data points", data_points.len());
408        data_points
409    }
410
411    async fn upload(&mut self, data_points: Vec<DataPoint>) -> anyhow::Result<TransactionEffects> {
412        let _scope = monitored_scope("Oracle::OnChainDataUploader::upload");
413        // TODO add more error handling & polling perhaps
414        let mut builder = ProgrammableTransactionBuilder::new();
415        let mut is_first = true;
416        for data_point in &data_points {
417            let package_id = data_point.upload_parameters.write_package_id;
418            let feed_name = &data_point.feed_name;
419            let oracle_obj_arg = *self
420                .oracle_object_args
421                .get(&data_point.upload_parameters.write_data_provider_object_id)
422                .unwrap_or_else(|| {
423                    panic!("Bug, missing oracle object arg for feed: {}", feed_name)
424                });
425            let duration_since_start = data_point.retrieval_instant.elapsed();
426            let data_point_ts: DateTime<Utc> =
427                DateTime::from(data_point.retrieval_timestamp + duration_since_start);
428
429            let mut arguments = if is_first {
430                vec![
431                    builder.input(CallArg::Object(oracle_obj_arg)).unwrap(),
432                    builder.input(CallArg::CLOCK_IMM).unwrap(),
433                ]
434            } else {
435                vec![Argument::Input(0), Argument::Input(1)]
436            };
437
438            let decimal = builder
439                .input(CallArg::Pure(bcs::to_bytes(&DECIMAL).unwrap()))
440                .unwrap();
441            let value = builder
442                .input(CallArg::Pure(bcs::to_bytes(&data_point.value).unwrap()))
443                .unwrap();
444
445            arguments.extend_from_slice(&[
446                builder
447                    .input(CallArg::Pure(bcs::to_bytes(&feed_name)?))
448                    .unwrap(),
449                builder.programmable_move_call(
450                    package_id,
451                    Identifier::from_str("decimal_value").unwrap(),
452                    Identifier::from_str("new").unwrap(),
453                    vec![],
454                    vec![value, decimal],
455                ),
456                builder
457                    .input(CallArg::Pure(bcs::to_bytes(&format!("{}", data_point_ts))?))
458                    .unwrap(),
459            ]);
460
461            builder.command(Command::move_call(
462                package_id,
463                Identifier::new(data_point.upload_parameters.write_module_name.clone()).unwrap(),
464                Identifier::new(data_point.upload_parameters.write_function_name.clone()).unwrap(),
465                // TODO: allow more generic data types
466                vec![
467                    parse_sui_type_tag(&format!("{package_id}::decimal_value::DecimalValue"))
468                        .unwrap(),
469                ],
470                arguments,
471            ));
472            is_first = false;
473        }
474        let pt = builder.finish();
475        let rgp = self.client.get_reference_gas_price().await?;
476        let tx = TransactionData::new_programmable(
477            self.signer_address,
478            vec![self.gas_obj_ref],
479            pt,
480            // 15_000_000 is a heuristic number
481            15_000_000 * data_points.len() as u64,
482            rgp,
483        );
484
485        let signed_tx = self.wallet_ctx.sign_transaction(&tx).await;
486        let tx_digest = *signed_tx.digest();
487
488        let timer_start = Instant::now();
489        let response = self.execute(signed_tx).await?;
490        let time_spend_sec = timer_start.elapsed().as_secs_f32();
491
492        // We asked for effects.
493        // But is there a better way to handle this instead of panic?
494        let effects = response.effects;
495
496        // It's critical to update the gas object reference for next transaction
497        self.gas_obj_ref = effects.gas_object().0;
498
499        let success = effects.status().is_ok();
500
501        // Update metrics
502        for data_point in &data_points {
503            if success {
504                self.metrics
505                    .upload_successes
506                    .with_label_values(&[&data_point.feed_name])
507                    .inc();
508                self.metrics
509                    .uploaded_values
510                    .with_label_values(&[&data_point.feed_name])
511                    .observe(data_point.value as f64);
512            } else {
513                self.metrics
514                    .upload_data_errors
515                    .with_label_values(&[&data_point.feed_name])
516                    .inc();
517            }
518        }
519
520        let gas_usage = effects.gas_cost_summary().gas_used();
521        let storage_rebate = effects.gas_cost_summary().storage_rebate;
522        let computation_cost = effects.gas_cost_summary().computation_cost;
523        let net_gas_usage = effects.gas_cost_summary().net_gas_usage();
524        self.metrics
525            .computation_gas_used
526            .observe(computation_cost as f64);
527        self.metrics.total_gas_cost.inc_by(gas_usage);
528        self.metrics.total_gas_rebate.inc_by(storage_rebate);
529
530        if success {
531            self.metrics
532                .total_data_points_uploaded
533                .inc_by(data_points.len() as u64);
534            info!(
535                ?tx_digest,
536                net_gas_usage,
537                time_spend_sec,
538                "Upload succeeded with {} data points",
539                data_points.len(),
540            );
541            Ok(effects)
542        } else {
543            error!(
544                ?tx_digest,
545                net_gas_usage,
546                "Upload failed with {} data points. Err: {:?}",
547                data_points.len(),
548                effects.status(),
549            );
550            anyhow::bail!("Failed to submit data on chain: {:?}", effects.status());
551        }
552    }
553
554    async fn execute(&mut self, tx: Transaction) -> anyhow::Result<ExecutedTransaction> {
555        let tx_digest = tx.digest();
556        let mut retry_attempts = 3;
557        loop {
558            match self
559                .client
560                .execute_transaction_and_wait_for_checkpoint(&tx)
561                .await
562            {
563                Ok(response) => return Ok(response),
564                Err(err) => {
565                    // jsonrpsee translate every SuiError into jsonrpsee::core::Error, so we need to further distinguish
566                    if err.to_string().contains(NON_RECOVERABLE_ERROR_MSG) {
567                        let stale_obj_error = STALE_OBJ_ERROR.get_or_init(|| {
568                            String::from(
569                                UserInputError::ObjectVersionUnavailableForConsumption {
570                                    provided_obj_ref: random_object_ref(),
571                                    current_version: 0.into(),
572                                }
573                                .as_ref(),
574                            )
575                        });
576                        if err.to_string().contains(stale_obj_error) {
577                            error!(
578                                ?tx_digest,
579                                "Failed to submit tx, it looks like gas object is stale : {:?}",
580                                err
581                            );
582                            let new_ref = get_gas_obj_ref(
583                                self.client.clone(),
584                                self.gas_obj_ref.0,
585                                self.signer_address,
586                            )
587                            .await;
588                            self.gas_obj_ref = new_ref;
589                            info!("Gas object updated: {:?}", new_ref);
590                            anyhow::bail!(
591                                "Gas object is stale, now updated to {:?}. tx_digest={:?}",
592                                new_ref,
593                                tx_digest
594                            );
595                        } else {
596                            error!(
597                                ?tx_digest,
598                                "Failed to submit tx, with non recoverable error: {:?}", err
599                            );
600                            anyhow::bail!(
601                                "Non-retryable error {:?}. tx_digest={:?}",
602                                err,
603                                tx_digest
604                            );
605                        }
606                    }
607                    // Likely retryable error?
608                    error!(
609                        ?tx_digest,
610                        "Failed to submit tx, with (likely) recoverable error: {:?}. Remaining retry times: {}",
611                        err,
612                        retry_attempts
613                    );
614                    retry_attempts -= 1;
615                    if retry_attempts <= 0 {
616                        anyhow::bail!("Too many RPC errors: {}. tx_digest={:?}", err, tx_digest);
617                    }
618                }
619            }
620        }
621    }
622}

625struct DataPoint {
626    feed_name: String,
627    upload_parameters: UploadParameters,
628    value: u64,
629    retrieval_timestamp: SystemTime,
630    retrieval_instant: Instant,
631}

633struct OnChainDataReader {
634    pub client: Client,
635    // For now we share one read interval for all reads
636    pub read_interval: Duration,
637    pub read_configs: HashMap<String, ObjectID>,
638    metrics: Arc<OracleMetrics>,
639}
640
641impl OnChainDataReader {
642    pub async fn start(mut self, sender: tokio::sync::mpsc::Sender<(String, ObjectID, f64)>) {
643        info!(
644            "Starting on-chain data reader with interval {:?} and config: {:?}",
645            self.read_interval, self.read_configs
646        );
647        let mut read_interval = tokio::time::interval(self.read_interval);
648        read_interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
649        loop {
650            read_interval.tick().await;
651            for (feed_name, object_id) in &self.read_configs {
652                match self.client.get_object(*object_id).await {
653                    Ok(_object) => {
654                        // TODO parse value based on returned BCS
655                        let value = 5_f64;
656                        let _ = sender.send((feed_name.clone(), *object_id, value)).await;
657                        self.metrics
658                            .downloaded_values
659                            .with_label_values(&[feed_name])
660                            .observe(value * METRICS_MULTIPLIER);
661                        self.metrics
662                            .download_successes
663                            .with_label_values(&[feed_name, &object_id.to_string()])
664                            .inc();
665                    }
666                    other => {
667                        error!(
668                            read_feed = feed_name,
669                            ?object_id,
670                            "Failed to read data from on-chain: {:?}",
671                            other
672                        );
673                        self.metrics
674                            .download_data_errors
675                            .with_label_values(&[feed_name, &object_id.to_string()])
676                            .inc();
677                    }
678                }
679            }
680        }
681    }
682}

684async fn get_object_arg(
685    client: &mut Client,
686    id: ObjectID,
687    is_mutable_ref: bool,
688) -> anyhow::Result<ObjectArg> {
689    let obj = client.get_object(id).await?;
690    let obj_ref = obj.compute_object_reference();
691    let owner = obj.owner.clone();
692    Ok(match owner {
693        Owner::Shared {
694            initial_shared_version,
695        }
696        | Owner::ConsensusAddressOwner {
697            start_version: initial_shared_version,
698            ..
699        } => ObjectArg::SharedObject {
700            id,
701            initial_shared_version,
702            mutability: if is_mutable_ref {
703                SharedObjectMutability::Mutable
704            } else {
705                SharedObjectMutability::Immutable
706            },
707        },
708        Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
709            ObjectArg::ImmOrOwnedObject(obj_ref)
710        }
711    })
712}
