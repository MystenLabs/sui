12use sui_sdk::wallet_context::WalletContext;

16/// Binds together the context for `find_environment` for helper functions
17struct EnvFinder<'a> {
18    explicit_env: Option<EnvironmentName>,
19    wallet: &'a WalletContext,
20    manifest_envs: IndexMap<EnvironmentName, EnvironmentID>,
21}
22
23/// Determine the correct environment to use for the package system based on
24///  - the path to a directory containing a Move.toml file
25///  - the `-e <env>` argument that was passed, if any
26///  - the CLI's active environment (`wallet`)
27pub async fn find_environment(
28    package_path: &Path,
29    explicit_env: Option<EnvironmentName>,
30    wallet: &WalletContext,
31) -> anyhow::Result<Environment> {
32    let manifest_envs = RootPackage::<SuiFlavor>::environments(package_path)?;
33    let finder = EnvFinder {
34        explicit_env,
35        wallet,
36        manifest_envs,
37    };
38
39    // use explicit environment if provided
40    if let Some(explicit) = finder.get_explicit()? {
41        return Ok(explicit);
42    }
43
44    // figure out an active environment to use
45    let active_env = finder.active_environment().await?;
46
47    // if the manifest has exactly that (name, chain ID) pair, use it
48    if let Some(exact) = finder.check_exact(&active_env)? {
49        return Ok(exact);
50    }
51
52    // if the manifest has exactly one entry with matching chain ID, use it
53    finder.get_unique_by_chain_id(active_env)
54}
55
56impl EnvFinder<'_> {
57    /// Return the explicitly passed environment if it exists. Fails if an environment was passed
58    /// explicitly but doesn't exist in the manifest.
59    fn get_explicit(&self) -> anyhow::Result<Option<Environment>> {
60        let Some(ref env_name) = self.explicit_env else {
61            return Ok(None);
62        };
63
64        let Some(env_id) = self.manifest_envs.get(env_name) else {
65            bail!("Environment `{env_name}` is not present in Move.toml");
66        };
67
68        Ok(Some(Environment::new(env_name.clone(), env_id.clone())))
69    }
70
71    /// Find the active environment. Checks the cache first and fails if the chain ID cannot be
72    /// determined (either from the cache or from the network or the manifest)
73    async fn active_environment(&self) -> anyhow::Result<Environment> {
74        let active_env = self.wallet.get_active_env()?.clone();
75        let chain_id = if let Some(chain_id) = active_env.chain_id {
76            // cached
77            chain_id
78        } else if let Ok(chain_id) = self.wallet.cache_chain_id().await {
79            // fetched
80            chain_id
81        } else if let Some(chain_id) = self.manifest_envs.get(&active_env.alias) {
82            // couldn't fetch but an environment with same name is in the manifest
83            chain_id.clone()
84        } else {
85            let mut s = String::new();
86            for e in self.manifest_envs.keys() {
87                s.push_str(&format!("\n\t-e {e}"))
88            }
89            bail!(
90                "Active environment `{}` does not correspond to any of environments defined for \
91                the package. Specify the environment by passing one of the following:{s}",
92                active_env.alias
93            );
94        };
95
96        Ok(Environment::new(active_env.alias, chain_id))
97    }
98
99    /// Check that the exact environment `active_env` is present in the manifest; returns an error
100    /// if the name is present but the chain ID differs; returns `None` if it's not present
101    fn check_exact(&self, active_env: &Environment) -> anyhow::Result<Option<Environment>> {
102        let Some(manifest_id) = self.manifest_envs.get(active_env.name()) else {
103            return Ok(None);
104        };
105
106        if manifest_id != active_env.id() {
107            bail!(
108                "Error: Environment `{active_env}` has chain ID `{chain_id}` in your CLI \
109                environment, but `Move.toml` expects `{active_env}` to have chain ID \
110                `{env_chain_id}`; this may indicate that `{active_env}` has been wiped or that you \
111                have a misconfigured CLI environment. If you want to ignore your CLI's chain ID and build for the `{active_env}` environment, you can pass `-e {active_env}`.",
112                active_env = active_env.name,
113                env_chain_id = manifest_id,
114                chain_id = active_env.id,
115            );
116        }
117
118        Ok(Some(active_env.clone()))
119    }
120
121    /// Check that there is exactly one entry of the manifest that matches the provided `chain_id`;
122    /// fails if there are either too many or too few
123    fn get_unique_by_chain_id(&self, env: Environment) -> anyhow::Result<Environment> {
124        let Environment {
125            name: active_env,
126            id: chain_id,
127        } = env;
128        let candidates: BTreeMap<&EnvironmentName, &EnvironmentID> = self
129            .manifest_envs
130            .iter()
131            .filter(|(_, v)| v == &&chain_id)
132            .collect();
133
134        if candidates.is_empty() {
135            // ephemeral case, no environment found with that name, we error
136            bail!(
137                "Your active environment `{active_env}` is not present in `Move.toml`, so you cannot \
138                publish to `{active_env}`.
139
140            - If you want to create a temporary publication on `{active_env}` and record the addresses \
141               in an ephemeral file, use the `test-publish` command instead.
142
143                sui client test-publish --help
144
145            - If you want to publish to `{active_env}` and record the addresses in the shared \
146            `Publications.toml` file, you will need to add the following to `Move.toml`:
147
148                [environments]
149                {active_env} = \"{chain_id}\""
150            );
151        }
152
153        if candidates.len() > 1 {
154            let mut s = String::new();
155            for e in candidates.keys() {
156                s.push_str(&format!("\n\t--build-env {e}"))
157            }
158            bail!(
159                "Found multiple environments defined in Move.toml with chain-id `{chain_id}`. Please pass one of the following:{s}"
160            );
161        }
162
163        let (env_name, env_id) = candidates
164            .first_key_value()
165            .expect("candidates has length 1");
166
167        eprintln!(
168            "Note: `Move.toml` does not define an `{active_env}` environment; building for `{env_name}` instead"
169        );
170
171        Ok(Environment::new(env_name.to_string(), env_id.to_string()))
172    }
173}
