98    // TODO this needs fixing, see todos
99    fn system_dependencies(
100        environment: EnvironmentID,
101    ) -> BTreeMap<PackageName, ReplacementDependency> {
102        let mut deps = BTreeMap::new();
103        let deps_to_skip = ["DeepBook".into()];
104        // TODO: we need to use packages for protocol version as well, so we need to fix this
105        let packages = latest_system_packages();
106        let sha = &packages.git_revision;
107        // filter out the packages that we want to skip
108        let pkgs = packages
109            .packages
110            .iter()
111            .filter(|package| !deps_to_skip.contains(&package.package_name));
112
113        let names = Self::system_deps_names_map();
114        for package in pkgs {
115            let repo = SYSTEM_GIT_REPO.to_string();
116            let dependency_info = ManifestDependencyInfo::Git(ManifestGitDependency {
117                repo: repo.clone(),
118                rev: Some(sha.clone()),
119                subdir: PathBuf::from(&package.repo_path),
120            });
121
122            let replacement_dep = ReplacementDependency {
123                dependency: Some(move_package_alt::schema::DefaultDependency {
124                    dependency_info,
125                    is_override: true,
126                    rename_from: None,
127                }),
128                addresses: None,
129                use_environment: None,
130            };
131
132            deps.insert(
133                names
134                    .get(
135                        &PackageName::new(package.package_name.clone())
136                            .expect("valid package name"),
137                    )
138                    .expect("package exists in the renaming table")
139                    .clone(),
140                replacement_dep,
141            );
142        }
143
144        deps
145    }
146
147    fn default_system_dependencies(
148        environment: EnvironmentID,
149    ) -> BTreeMap<PackageName, ReplacementDependency> {
150        let default_deps = Self::default_system_dep_names();
151
152        Self::system_dependencies(environment)
153            .into_iter()
154            .filter(|(name, _)| default_deps.contains(name))
155            .collect()
156    }
157}

159#[cfg(test)]
160mod tests {
161
162    use super::*;
163    use move_package_alt::package::RootPackage;
164    use move_package_alt::package::manifest::Manifest;
165    use move_package_alt::schema::{Environment, PackageName};
166
167    #[test]
168    fn test_implicit_deps() {
169        let implicit_deps = ["sui", "std", "sui_system", "bridge"];
170        let env = Environment::new("mainnet".into(), "35834a8a".into());
171
172        let deps = SuiFlavor::system_dependencies(env.id().into());
173
174        for i in implicit_deps {
175            assert!(
176                deps.contains_key(&PackageName::new(i).unwrap()),
177                "Dependency {} not found in implicit dependencies",
178                i
179            );
180            assert!(
181                !deps.contains_key(&PackageName::new("DeepBook").unwrap()),
182                "Dependency DeepBook should not be in the implicit dependencies"
183            );
184            assert!(
185                !deps.contains_key(&PackageName::new("deepbook").unwrap()),
186                "Dependency deepbook should not be in the implicit dependencies"
187            );
188        }
189    }
