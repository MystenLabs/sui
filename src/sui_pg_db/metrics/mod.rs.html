4use prometheus::IntCounter;
5use prometheus::Registry;
6use prometheus::register_int_counter_with_registry;
7use std::sync::Arc;
8
9pub struct PoolMetrics {
10    pub requested: IntCounter,
11    pub acquired: IntCounter,
12    pub unacquired_error: IntCounter,
13    pub unacquired_canceled: IntCounter,
14}
15
16impl PoolMetrics {
17    pub fn new(prefix: Option<&str>, registry: &Registry) -> anyhow::Result<Arc<Self>> {
18        let prefix = prefix.unwrap_or("db");
19        let name = |n| format!("{prefix}_{n}");
20        let pool_metrics = PoolMetrics {
21            requested: register_int_counter_with_registry!(
22                name("connections_requested"),
23                "Total requested connections from the pool",
24                registry,
25            )?,
26            acquired: register_int_counter_with_registry!(
27                name("connections_acquired"),
28                "Total requested connections from the pool that were acquired",
29                registry,
30            )?,
31            unacquired_error: register_int_counter_with_registry!(
32                name("connections_unacquired_error"),
33                "Total requested connections from the pool that were not acquired due to an error",
34                registry,
35            )?,
36            unacquired_canceled: register_int_counter_with_registry!(
37                name("connections_unacquired_canceled"),
38                "Total requested connections from the pool that were not acquired due to task cancellation",
39                registry,
40            )?,
41        };
42        Ok(Arc::new(pool_metrics))
43    }
44}
