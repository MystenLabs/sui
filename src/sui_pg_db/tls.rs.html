32/// A custom verifier that skips all server certificate verification. This mirrors libpq default
33/// behavior of not doing any server verification.
34#[derive(Debug)]
35pub(crate) struct SkipServerCertCheck;
36
37pub struct AsyncPgConnectionWithId {
38    pub pid: i32,
39    pub conn: AsyncPgConnection,
40}

62/// Implement the `ServerCertVerifier` trait for `SkipServerCertCheck` to always return valid. This
63/// skips all server certificate verification.
64impl ServerCertVerifier for SkipServerCertCheck {
65    fn verify_server_cert(
66        &self,
67        _end_entity: &CertificateDer<'_>,
68        _intermediates: &[CertificateDer<'_>],
69        _server_name: &ServerName<'_>,
70        _ocsp_response: &[u8],
71        _now: UnixTime,
72    ) -> Result<ServerCertVerified, rustls::Error> {
73        Ok(ServerCertVerified::assertion())
74    }
75
76    fn verify_tls12_signature(
77        &self,
78        _message: &[u8],
79        _cert: &CertificateDer<'_>,
80        _dss: &DigitallySignedStruct,
81    ) -> Result<HandshakeSignatureValid, rustls::Error> {
82        Ok(HandshakeSignatureValid::assertion())
83    }
84
85    fn verify_tls13_signature(
86        &self,
87        _message: &[u8],
88        _cert: &CertificateDer<'_>,
89        _dss: &DigitallySignedStruct,
90    ) -> Result<HandshakeSignatureValid, rustls::Error> {
91        Ok(HandshakeSignatureValid::assertion())
92    }
93
94    fn supported_verify_schemes(&self) -> Vec<rustls::SignatureScheme> {
95        rustls::client::WebPkiServerVerifier::builder_with_provider(
96            Arc::new(root_certs()),
97            Arc::new(rustls::crypto::ring::default_provider()),
98        )
99        .build()
100        .unwrap()
101        .supported_verify_schemes()
102    }
103}

105/// Establish a PostgreSQL connection with custom TLS configuration using tokio-postgres. The
106/// returned connection is compatible with diesel-async. This is needed because diesel-async does
107/// not expose TLS configuration. Even with a TLS connector, the actual usage is negotiated with the
108/// postgres server:
109/// - Client sends SSLRequest, server responds 'S' (TLS supported) or 'N' (TLS not supported)
110/// - If server supports TLS: encrypted connection using the provided config
111/// - If server doesn't support TLS: falls back to plaintext (when using sslmode=prefer default)
112/// - If sslmode=disable in URL: no TLS attempted regardless of connector
113pub(crate) async fn establish_tls_connection(
114    database_url: &str,
115    tls_config: ClientConfig,
116) -> ConnectionResult<AsyncPgConnectionWithId> {
117    let tls = MakeRustlsConnect::new(tls_config);
118    let (client, tokio_conn) = tokio_postgres::connect(database_url, tls)
119        .await
120        .map_err(|e| ConnectionError::BadConnection(e.to_string()))?;
121
122    // The `conn` object performs actual IO with the database, and tokio-postgres suggests spawning
123    // it off to run in the background. This will resolve only when the connection is closed,
124    // either because of a fatal error or because its associated Client has dropped and all
125    // outstanding work has completed.
126    let h_conn = tokio::spawn(tokio_conn);
127
128    // Users interact with the database through the client object. We convert it into an
129    // AsyncPgConnection so it can be compatible with diesel.
130    let mut conn = AsyncPgConnection::try_from(client).await?;
131
132    #[derive(QueryableByName)]
133    struct Pid {
134        #[diesel(sql_type = Integer)]
135        pid: i32,
136    }
137
138    let Pid { pid } = diesel::sql_query("SELECT pg_backend_pid() AS pid")
139        .get_result(&mut conn)
140        .await
141        .map_err(ConnectionError::CouldntSetupConfiguration)?;
142
143    info!(pid, "Database connection established");
144
145    tokio::spawn(async move {
146        match h_conn.await {
147            Ok(Err(e)) => {
148                error!(pid, "Database connection terminated with error: {e}");
149            }
150            Ok(Ok(())) => {
151                info!(pid, "Database connection terminated without error");
152            }
153            Err(e) => {
154                info!(pid, "Database connection task failed: {e}");
155            }
156        }
157    });
158
159    Ok(AsyncPgConnectionWithId { conn, pid })
160}

162/// Builds a TLS configuration from the provided DbArgs. If tls_verify_cert is false, disable server
163/// certificate verification. If tls_ca_cert_path is provided, add the custom CA certificate to the
164/// root certificates.
165pub(crate) fn build_tls_config(
166    tls_verify_cert: bool,
167    tls_ca_cert_path: Option<PathBuf>,
168) -> anyhow::Result<ClientConfig> {
169    let provider = Arc::new(rustls::crypto::ring::default_provider());
170    let config = ClientConfig::builder_with_provider(provider)
171        .with_protocol_versions(rustls::DEFAULT_VERSIONS)?;
172
173    if !tls_verify_cert {
174        return Ok(config
175            .dangerous()
176            .with_custom_certificate_verifier(Arc::new(SkipServerCertCheck))
177            .with_no_client_auth());
178    }
179
180    let mut root_certs = root_certs();
181
182    // Add custom CA certificate if provided
183    if let Some(ca_cert_path) = &tls_ca_cert_path {
184        let ca_cert_bytes = std::fs::read(ca_cert_path).with_context(|| {
185            format!(
186                "Failed to read CA certificate from {}",
187                ca_cert_path.display()
188            )
189        })?;
190
191        let certs = if ca_cert_bytes.starts_with(b"-----BEGIN CERTIFICATE-----") {
192            CertificateDer::pem_slice_iter(&ca_cert_bytes)
193                .collect::<Result<Vec<_>, _>>()
194                .with_context(|| {
195                    format!(
196                        "Failed to parse PEM certificates from {}",
197                        ca_cert_path.display()
198                    )
199                })?
200        } else {
201            // Assume DER format for binary files
202            vec![CertificateDer::from(ca_cert_bytes)]
203        };
204
205        // Add all certificates to the root store
206        for cert in certs {
207            root_certs
208                .add(cert)
209                .with_context(|| "Failed to add CA certificate to root store".to_string())?;
210        }
211    }
212
213    Ok(config
214        .with_root_certificates(root_certs)
215        .with_no_client_auth())
216}
217
218fn root_certs() -> RootCertStore {
219    RootCertStore {
220        roots: TLS_SERVER_ROOTS.to_vec(),
221    }
