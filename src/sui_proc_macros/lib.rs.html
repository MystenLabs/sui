16#[proc_macro_attribute]
17pub fn init_static_initializers(_args: TokenStream, item: TokenStream) -> TokenStream {
18    let mut input = parse_macro_input!(item as syn::ItemFn);
19
20    let body = &input.block;
21    input.block = syn::parse2(quote! {
22        {
23            // We have some lazily-initialized static state in the program. The initializers
24            // alter the thread-local hash container state any time they create a new hash
25            // container. Therefore, we need to ensure that these initializers are run in a
26            // separate thread before the first test thread is launched. Otherwise, they would
27            // run inside of the first test thread, but not subsequent ones.
28            //
29            // Note that none of this has any effect on process-level determinism. Without this
30            // code, we can still get the same test results from two processes started with the
31            // same seed.
32            //
33            // However, when using sim_test(check_determinism) or MSIM_TEST_CHECK_DETERMINISM=1,
34            // we want the same test invocation to be deterministic when run twice
35            // _in the same process_, so we need to take care of this. This will also
36            // be very important for being able to reproduce a failure that occurs in the Nth
37            // iteration of a multi-iteration test run.
38            std::thread::spawn(|| {
39                use sui_protocol_config::ProtocolConfig;
40                ::sui_simulator::telemetry_subscribers::init_for_testing();
41                ::sui_simulator::sui_types::execution_params::get_denied_certificates_for_sim_test();
42                ::sui_simulator::sui_framework::BuiltInFramework::all_package_ids();
43                ::sui_simulator::sui_types::gas::SuiGasStatus::new_unmetered();
44
45                // For reasons I can't understand, LruCache causes divergent behavior the second
46                // time one is constructed and inserted into, so construct one before the first
47                // test run for determinism.
48                let mut cache = ::sui_simulator::lru::LruCache::new(1.try_into().unwrap());
49                cache.put(1, 1);
50
51                {
52                    // Initialize the static initializers here:
53                    // https://github.com/move-language/move/blob/652badf6fd67e1d4cc2aa6dc69d63ad14083b673/language/tools/move-package/src/package_lock.rs#L12
54                    use std::path::PathBuf;
55                    use sui_simulator::sui_move_build::{BuildConfig, SuiPackageHooks};
56                    use sui_simulator::tempfile::TempDir;
57                    use sui_simulator::move_package::package_hooks::register_package_hooks;
58
59                    register_package_hooks(Box::new(SuiPackageHooks {}));
60                    let mut path = PathBuf::from(env!("SIMTEST_STATIC_INIT_MOVE"));
61                    let mut build_config = BuildConfig::new_for_testing();
62
63                    build_config.config.install_dir = Some(TempDir::new().unwrap().keep());
64                    let _all_module_bytes = build_config
65                        .build(&path)
66                        .unwrap()
67                        .get_package_bytes(/* with_unpublished_deps */ false);
68                }
69
70                use std::sync::Arc;
71
72                use ::sui_simulator::anemo_tower::callback::CallbackLayer;
73                use ::sui_simulator::anemo_tower::trace::DefaultMakeSpan;
74                use ::sui_simulator::anemo_tower::trace::DefaultOnFailure;
75                use ::sui_simulator::anemo_tower::trace::TraceLayer;
76                use ::sui_simulator::fastcrypto::traits::KeyPair;
77                use ::sui_simulator::mysten_network::metrics::MetricsMakeCallbackHandler;
78                use ::sui_simulator::mysten_network::metrics::NetworkMetrics;
79                use ::sui_simulator::rand_crate::rngs::{StdRng, OsRng};
80                use ::sui_simulator::rand::SeedableRng;
81                use ::sui_simulator::tower::ServiceBuilder;
82
83                // anemo uses x509-parser, which has many lazy static variables. start a network to
84                // initialize all that static state before the first test.
85                let rt = ::sui_simulator::runtime::Runtime::new();
86                rt.block_on(async move {
87                    use ::sui_simulator::anemo::{Network, Request};
88
89                    let make_network = |port: u16| {
90                        let registry = prometheus::Registry::new();
91                        let inbound_network_metrics =
92                            NetworkMetrics::new("sui", "inbound", &registry);
93                        let outbound_network_metrics =
94                            NetworkMetrics::new("sui", "outbound", &registry);
95
96                        let service = ServiceBuilder::new()
97                            .layer(
98                                TraceLayer::new_for_server_errors()
99                                    .make_span_with(DefaultMakeSpan::new().level(tracing::Level::INFO))
100                                    .on_failure(DefaultOnFailure::new().level(tracing::Level::WARN)),
101                            )
102                            .layer(CallbackLayer::new(MetricsMakeCallbackHandler::new(
103                                Arc::new(inbound_network_metrics),
104                                usize::MAX,
105                            )))
106                            .service(::sui_simulator::anemo::Router::new());
107
108                        let outbound_layer = ServiceBuilder::new()
109                            .layer(
110                                TraceLayer::new_for_client_and_server_errors()
111                                    .make_span_with(DefaultMakeSpan::new().level(tracing::Level::INFO))
112                                    .on_failure(DefaultOnFailure::new().level(tracing::Level::WARN)),
113                            )
114                            .layer(CallbackLayer::new(MetricsMakeCallbackHandler::new(
115                                Arc::new(outbound_network_metrics),
116                                usize::MAX,
117                            )))
118                            .into_inner();
119
120
121                        Network::bind(format!("127.0.0.1:{}", port))
122                            .server_name("static-init-network")
123                            .private_key(
124                                ::sui_simulator::fastcrypto::ed25519::Ed25519KeyPair::generate(&mut StdRng::from_rng(OsRng).unwrap())
125                                    .private()
126                                    .0
127                                    .to_bytes(),
128                            )
129                            .start(service)
130                            .unwrap()
131                    };
132                    let n1 = make_network(80);
133                    let n2 = make_network(81);
134
135                    let _peer = n1.connect(n2.local_addr()).await.unwrap();
136                });
137            }).join().unwrap();
138
139            #body
140        }
141    })
142    .expect("Parsing failure");
143
144    let result = quote! {
145        #input
146    };
147
148    result.into()
149}

151/// The sui_test macro will invoke either `#[msim::test]` or `#[tokio::test]`,
152/// depending on whether the simulator config var is enabled.
153///
154/// This should be used for tests that can meaningfully run in either environment.
155#[proc_macro_attribute]
156pub fn sui_test(args: TokenStream, item: TokenStream) -> TokenStream {
157    let input = parse_macro_input!(item as syn::ItemFn);
158    let arg_parser = Punctuated::<syn::Meta, Token![,]>::parse_terminated;
159    let args = arg_parser.parse(args).unwrap().into_iter();
160
161    let header = if cfg!(msim) {
162        quote! {
163            #[::sui_simulator::sim_test(crate = "sui_simulator", #(#args)* )]
164        }
165    } else {
166        quote! {
167            #[::tokio::test(#(#args)*)]
168        }
169    };
170
171    let result = quote! {
172        #header
173        #[::sui_macros::init_static_initializers]
174        #input
175    };
176
177    result.into()
178}

180/// The sim_test macro will invoke `#[msim::test]` if the simulator config var is enabled.
181///
182/// Otherwise, it will emit an ignored test - if forcibly run, the ignored test will panic.
183///
184/// This macro must be used in order to pass any simulator-specific arguments, such as
185/// `check_determinism`, which is not understood by tokio.
186#[proc_macro_attribute]
187pub fn sim_test(args: TokenStream, item: TokenStream) -> TokenStream {
188    let input = parse_macro_input!(item as syn::ItemFn);
189    let arg_parser = Punctuated::<syn::Meta, Token![,]>::parse_terminated;
190    let args = arg_parser.parse(args).unwrap().into_iter();
191
192    let ignore = input
193        .attrs
194        .iter()
195        .find(|attr| attr.path().is_ident("ignore"))
196        .map_or(quote! {}, |_| quote! { #[ignore] });
197
198    let result = if cfg!(msim) {
199        let sig = &input.sig;
200        let return_type = &sig.output;
201        let body = &input.block;
202        quote! {
203            #[::sui_simulator::sim_test(crate = "sui_simulator", #(#args),*)]
204            #[::sui_macros::init_static_initializers]
205            #ignore
206            #sig {
207                async fn body_fn() #return_type { #body }
208
209                let ret = body_fn().await;
210
211                ::sui_simulator::task::shutdown_all_nodes();
212
213                // all node handles should have been dropped after the above block exits, but task
214                // shutdown is asynchronous, so we need a brief delay before checking for leaks.
215                tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
216
217                assert_eq!(
218                    sui_simulator::NodeLeakDetector::get_current_node_count(),
219                    0,
220                    "SuiNode leak detected"
221                );
222
223                ret
224            }
225        }
226    } else {
227        let fn_name = &input.sig.ident;
228        let sig = &input.sig;
229        let body = &input.block;
230        quote! {
231            #[allow(clippy::needless_return)]
232            #[tokio::test]
233            #ignore
234            #sig {
235                if std::env::var("SUI_SKIP_SIMTESTS").is_ok() {
236                    println!("not running test {} in `cargo test`: SUI_SKIP_SIMTESTS is set", stringify!(#fn_name));
237
238                    struct Ret;
239
240                    impl From<Ret> for () {
241                        fn from(_ret: Ret) -> Self {
242                        }
243                    }
244
245                    impl<E> From<Ret> for Result<(), E> {
246                        fn from(_ret: Ret) -> Self {
247                            Ok(())
248                        }
249                    }
250
251                    return Ret.into();
252                }
253
254                #body
255            }
256        }
257    };
258
259    result.into()
260}

262#[proc_macro]
263pub fn checked_arithmetic(input: TokenStream) -> TokenStream {
264    let input_file = CheckArithmetic.fold_file(parse_macro_input!(input));
265
266    let output_items = input_file.items;
267
268    let output = quote! {
269        #(#output_items)*
270    };
271
272    TokenStream::from(output)
273}

275#[proc_macro_attribute]
276pub fn with_checked_arithmetic(_attr: TokenStream, item: TokenStream) -> TokenStream {
277    let input_item = parse_macro_input!(item as Item);
278    match input_item {
279        Item::Fn(input_fn) => {
280            let transformed_fn = CheckArithmetic.fold_item_fn(input_fn);
281            TokenStream::from(quote! { #transformed_fn })
282        }
283        Item::Impl(input_impl) => {
284            let transformed_impl = CheckArithmetic.fold_item_impl(input_impl);
285            TokenStream::from(quote! { #transformed_impl })
286        }
287        item => {
288            let transformed_impl = CheckArithmetic.fold_item(item);
289            TokenStream::from(quote! { #transformed_impl })
290        }
291    }
292}

294struct CheckArithmetic;
295
296impl CheckArithmetic {
297    fn maybe_skip_macro(&self, attrs: &mut Vec<Attribute>) -> bool {
298        if let Some(idx) = attrs
299            .iter()
300            .position(|attr| attr.path().is_ident("skip_checked_arithmetic"))
301        {
302            // Skip processing macro because it is annotated with
303            // #[skip_checked_arithmetic]
304            attrs.remove(idx);
305            true
306        } else {
307            false
308        }
309    }
310
311    fn process_macro_contents(
312        &mut self,
313        tokens: proc_macro2::TokenStream,
314    ) -> syn::Result<proc_macro2::TokenStream> {
315        // Parse the macro's contents as a comma-separated list of expressions.
316        let parser = Punctuated::<Expr, Token![,]>::parse_terminated;
317        let Ok(exprs) = parser.parse(tokens.clone().into()) else {
318            return Err(syn::Error::new_spanned(
319                tokens,
320                "could not process macro contents - use #[skip_checked_arithmetic] to skip this macro",
321            ));
322        };
323
324        // Fold each sub expression.
325        let folded_exprs = exprs
326            .into_iter()
327            .map(|expr| self.fold_expr(expr))
328            .collect::<Vec<_>>();
329
330        // Convert the folded expressions back into tokens and reconstruct the macro.
331        let mut folded_tokens = proc_macro2::TokenStream::new();
332        for (i, folded_expr) in folded_exprs.into_iter().enumerate() {
333            if i > 0 {
334                folded_tokens.extend(std::iter::once::<proc_macro2::TokenTree>(
335                    proc_macro2::Punct::new(',', proc_macro2::Spacing::Alone).into(),
336                ));
337            }
338            folded_expr.to_tokens(&mut folded_tokens);
339        }
340
341        Ok(folded_tokens)
342    }
343}

345impl Fold for CheckArithmetic {
346    fn fold_stmt(&mut self, stmt: Stmt) -> Stmt {
347        let stmt = fold_stmt(self, stmt);
348        if let Stmt::Macro(stmt_macro) = stmt {
349            let StmtMacro {
350                mut attrs,
351                mut mac,
352                semi_token,
353            } = stmt_macro;
354
355            if self.maybe_skip_macro(&mut attrs) {
356                Stmt::Macro(StmtMacro {
357                    attrs,
358                    mac,
359                    semi_token,
360                })
361            } else {
362                match self.process_macro_contents(mac.tokens.clone()) {
363                    Ok(folded_tokens) => {
364                        mac.tokens = folded_tokens;
365                        Stmt::Macro(StmtMacro {
366                            attrs,
367                            mac,
368                            semi_token,
369                        })
370                    }
371                    Err(error) => parse2(error.to_compile_error()).unwrap(),
372                }
373            }
374        } else {
375            stmt
376        }
377    }
378
379    fn fold_item_macro(&mut self, mut item_macro: ItemMacro) -> ItemMacro {
380        if !self.maybe_skip_macro(&mut item_macro.attrs) {
381            let err = syn::Error::new_spanned(
382                item_macro.to_token_stream(),
383                "cannot process macros - use #[skip_checked_arithmetic] to skip \
384                    processing this macro",
385            );
386
387            return parse2(err.to_compile_error()).unwrap();
388        }
389        fold_item_macro(self, item_macro)
390    }
391
392    fn fold_expr(&mut self, expr: Expr) -> Expr {
393        let span = expr.span();
394        let expr = fold_expr(self, expr);
395        let expr = match expr {
396            Expr::Macro(expr_macro) => {
397                let ExprMacro { mut attrs, mut mac } = expr_macro;
398
399                if self.maybe_skip_macro(&mut attrs) {
400                    return Expr::Macro(ExprMacro { attrs, mac });
401                } else {
402                    match self.process_macro_contents(mac.tokens.clone()) {
403                        Ok(folded_tokens) => {
404                            mac.tokens = folded_tokens;
405                            let expr_macro = Expr::Macro(ExprMacro { attrs, mac });
406                            quote!(#expr_macro)
407                        }
408                        Err(error) => {
409                            return Expr::Verbatim(error.to_compile_error());
410                        }
411                    }
412                }
413            }
414
415            Expr::Binary(expr_binary) => {
416                let ExprBinary {
417                    attrs,
418                    mut left,
419                    op,
420                    mut right,
421                } = expr_binary;
422
423                fn remove_parens(expr: &mut Expr) {
424                    if let Expr::Paren(paren) = expr {
425                        // i don't even think rust allows this, but just in case
426                        assert!(paren.attrs.is_empty(), "TODO: attrs on parenthesized");
427                        *expr = *paren.expr.clone();
428                    }
429                }
430
431                macro_rules! wrap_op {
432                    ($left: expr, $right: expr, $method: ident, $span: expr) => {{
433                        // Remove parens from exprs since both sides get assigned to tmp variables.
434                        // otherwise we get lint errors
435                        remove_parens(&mut $left);
436                        remove_parens(&mut $right);
437
438                        quote_spanned!($span => {
439                            // assign in one stmt in case either #left or #right contains
440                            // references to `left` or `right` symbols.
441                            let (left, right) = (#left, #right);
442                            left.$method(right)
443                                .unwrap_or_else(||
444                                    panic!(
445                                        "Overflow or underflow in {} {} + {}",
446                                        stringify!($method),
447                                        left,
448                                        right,
449                                    )
450                                )
451                        })
452                    }};
453                }
454
455                macro_rules! wrap_op_assign {
456                    ($left: expr, $right: expr, $method: ident, $span: expr) => {{
457                        // Remove parens from exprs since both sides get assigned to tmp variables.
458                        // otherwise we get lint errors
459                        remove_parens(&mut $left);
460                        remove_parens(&mut $right);
461
462                        quote_spanned!($span => {
463                            // assign in one stmt in case either #left or #right contains
464                            // references to `left` or `right` symbols.
465                            let (left, right) = (&mut #left, #right);
466                            *left = (*left).$method(right)
467                                .unwrap_or_else(||
468                                    panic!(
469                                        "Overflow or underflow in {} {} + {}",
470                                        stringify!($method),
471                                        *left,
472                                        right
473                                    )
474                                )
475                        })
476                    }};
477                }
478
479                match op {
480                    BinOp::Add(_) => {
481                        wrap_op!(left, right, checked_add, span)
482                    }
483                    BinOp::Sub(_) => {
484                        wrap_op!(left, right, checked_sub, span)
485                    }
486                    BinOp::Mul(_) => {
487                        wrap_op!(left, right, checked_mul, span)
488                    }
489                    BinOp::Div(_) => {
490                        wrap_op!(left, right, checked_div, span)
491                    }
492                    BinOp::Rem(_) => {
493                        wrap_op!(left, right, checked_rem, span)
494                    }
495                    BinOp::AddAssign(_) => {
496                        wrap_op_assign!(left, right, checked_add, span)
497                    }
498                    BinOp::SubAssign(_) => {
499                        wrap_op_assign!(left, right, checked_sub, span)
500                    }
501                    BinOp::MulAssign(_) => {
502                        wrap_op_assign!(left, right, checked_mul, span)
503                    }
504                    BinOp::DivAssign(_) => {
505                        wrap_op_assign!(left, right, checked_div, span)
506                    }
507                    BinOp::RemAssign(_) => {
508                        wrap_op_assign!(left, right, checked_rem, span)
509                    }
510                    _ => {
511                        let expr_binary = ExprBinary {
512                            attrs,
513                            left,
514                            op,
515                            right,
516                        };
517                        quote_spanned!(span => #expr_binary)
518                    }
519                }
520            }
521            Expr::Unary(expr_unary) => {
522                let op = &expr_unary.op;
523                let operand = &expr_unary.expr;
524                match op {
525                    UnOp::Neg(_) => {
526                        quote_spanned!(span => #operand.checked_neg().expect("Overflow or underflow in negation"))
527                    }
528                    _ => quote_spanned!(span => #expr_unary),
529                }
530            }
531            _ => quote_spanned!(span => #expr),
532        };
533
534        parse2(expr).unwrap()
535    }
536}

538/// This proc macro generates a function `order_to_variant_map` which returns a map
539/// of the position of each variant to the name of the variant.
540/// It is intended to catch changes in enum order when backward compat is required.
541/// ```rust,ignore
542///    /// Example for this enum
543///    #[derive(EnumVariantOrder)]
544///    pub enum MyEnum {
545///         A,
546///         B(u64),
547///         C{x: bool, y: i8},
548///     }
549///     let order_map = MyEnum::order_to_variant_map();
550///     assert!(order_map.get(0).unwrap() == "A");
551///     assert!(order_map.get(1).unwrap() == "B");
552///     assert!(order_map.get(2).unwrap() == "C");
553/// ```
554#[proc_macro_derive(EnumVariantOrder)]
555pub fn enum_variant_order_derive(input: TokenStream) -> TokenStream {
556    let ast = parse_macro_input!(input as DeriveInput);
557    let name = &ast.ident;
558
559    if let Data::Enum(DataEnum { variants, .. }) = ast.data {
560        let variant_entries = variants
561            .iter()
562            .enumerate()
563            .map(|(index, variant)| {
564                let variant_name = variant.ident.to_string();
565                quote! {
566                    map.insert( #index as u64, (#variant_name).to_string());
567                }
568            })
569            .collect::<Vec<_>>();
570
571        let deriv = quote! {
572            impl sui_enum_compat_util::EnumOrderMap for #name {
573                fn order_to_variant_map() -> std::collections::BTreeMap<u64, String > {
574                    let mut map = std::collections::BTreeMap::new();
575                    #(#variant_entries)*
576                    map
577                }
578            }
579        };
580
581        deriv.into()
582    } else {
583        panic!("EnumVariantOrder can only be used with enums.");
584    }
