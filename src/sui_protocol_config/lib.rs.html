279#[derive(Copy, Clone, Debug, Hash, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord)]
280pub struct ProtocolVersion(u64);
281
282impl ProtocolVersion {
283    // The minimum and maximum protocol version supported by this binary. Counterintuitively, this constant may
284    // change over time as support for old protocol versions is removed from the source. This
285    // ensures that when a new network (such as a testnet) is created, its genesis committee will
286    // use a protocol version that is actually supported by the binary.
287    pub const MIN: Self = Self(MIN_PROTOCOL_VERSION);
288
289    pub const MAX: Self = Self(MAX_PROTOCOL_VERSION);
290
291    #[cfg(not(msim))]
292    pub const MAX_ALLOWED: Self = Self::MAX;
293
294    // We create one additional "fake" version in simulator builds so that we can test upgrades.
295    #[cfg(msim)]
296    pub const MAX_ALLOWED: Self = Self(MAX_PROTOCOL_VERSION + 1);
297
298    pub fn new(v: u64) -> Self {
299        Self(v)
300    }
301
302    pub const fn as_u64(&self) -> u64 {
303        self.0
304    }
305
306    // For serde deserialization - we don't define a Default impl because there isn't a single
307    // universally appropriate default value.
308    pub fn max() -> Self {
309        Self::MAX
310    }
311
312    pub fn prev(self) -> Self {
313        Self(self.0.checked_sub(1).unwrap())
314    }
315}
316
317impl From<u64> for ProtocolVersion {
318    fn from(v: u64) -> Self {
319        Self::new(v)
320    }
321}
322
323impl std::ops::Sub<u64> for ProtocolVersion {
324    type Output = Self;
325    fn sub(self, rhs: u64) -> Self::Output {
326        Self::new(self.0 - rhs)
327    }
328}
329
330impl std::ops::Add<u64> for ProtocolVersion {
331    type Output = Self;
332    fn add(self, rhs: u64) -> Self::Output {
333        Self::new(self.0 + rhs)
334    }
335}

2407    pub fn get_for_version(version: ProtocolVersion, chain: Chain) -> Self {
2408        // ProtocolVersion can be deserialized so we need to check it here as well.
2409        assert!(
2410            version >= ProtocolVersion::MIN,
2411            "Network protocol version is {:?}, but the minimum supported version by the binary is {:?}. Please upgrade the binary.",
2412            version,
2413            ProtocolVersion::MIN.0,
2414        );
2415        assert!(
2416            version <= ProtocolVersion::MAX_ALLOWED,
2417            "Network protocol version is {:?}, but the maximum supported version by the binary is {:?}. Please upgrade the binary.",
2418            version,
2419            ProtocolVersion::MAX_ALLOWED.0,
2420        );
2421
2422        let mut ret = Self::get_for_version_impl(version, chain);
2423        ret.version = version;
2424
2425        ret = CONFIG_OVERRIDE.with(|ovr| {
2426            if let Some(override_fn) = &*ovr.borrow() {
2427                warn!(
2428                    "overriding ProtocolConfig settings with custom settings (you should not see this log outside of tests)"
2429                );
2430                override_fn(version, ret)
2431            } else {
2432                ret
2433            }
2434        });
2435
2436        if std::env::var("SUI_PROTOCOL_CONFIG_OVERRIDE_ENABLE").is_ok() {
2437            warn!(
2438                "overriding ProtocolConfig settings with custom settings; this may break non-local networks"
2439            );

2441                serde_env::from_env_with_prefix("SUI_PROTOCOL_CONFIG_OVERRIDE")
2442                    .expect("failed to parse ProtocolConfig override env variables");

4602/// Convenience function for comparing limit ranges
4603/// V::MAX must be at >= U::MAX and T::MAX
4604pub fn check_limit_in_range<T: Into<V>, U: Into<V>, V: PartialOrd + Into<u128>>(
4605    x: T,
4606    soft_limit: U,
4607    hard_limit: V,
4608) -> LimitThresholdCrossed {
4609    let x: V = x.into();
4610    let soft_limit: V = soft_limit.into();
4611
4612    debug_assert!(soft_limit <= hard_limit);
4613
4614    // It is important to preserve this comparison order because if soft_limit == hard_limit
4615    // we want LimitThresholdCrossed::Hard
4616    if x >= hard_limit {
4617        LimitThresholdCrossed::Hard(x.into(), hard_limit.into())
4618    } else if x < soft_limit {
4619        LimitThresholdCrossed::None
4620    } else {
4621        LimitThresholdCrossed::Soft(x.into(), soft_limit.into())
4622    }
4623}

4625#[macro_export]
4626macro_rules! check_limit {
4627    ($x:expr, $hard:expr) => {
4628        check_limit!($x, $hard, $hard)
4629    };
4630    ($x:expr, $soft:expr, $hard:expr) => {
4631        check_limit_in_range($x as u64, $soft, $hard)
4632    };
4633}

4635/// Used to check which limits were crossed if the TX is metered (not system tx)
4636/// Args are: is_metered, value_to_check, metered_limit, unmetered_limit
4637/// metered_limit is always less than or equal to unmetered_hard_limit
4638#[macro_export]
4639macro_rules! check_limit_by_meter {
4640    ($is_metered:expr, $x:expr, $metered_limit:expr, $unmetered_hard_limit:expr, $metric:expr) => {{
4641        // If this is metered, we use the metered_limit limit as the upper bound
4642        let (h, metered_str) = if $is_metered {
4643            ($metered_limit, "metered")
4644        } else {
4645            // Unmetered gets more headroom
4646            ($unmetered_hard_limit, "unmetered")
4647        };
4648        use sui_protocol_config::check_limit_in_range;
4649        let result = check_limit_in_range($x as u64, $metered_limit, h);
4650        match result {
4651            LimitThresholdCrossed::None => {}
4652            LimitThresholdCrossed::Soft(_, _) => {
4653                $metric.with_label_values(&[metered_str, "soft"]).inc();
4654            }
4655            LimitThresholdCrossed::Hard(_, _) => {
4656                $metric.with_label_values(&[metered_str, "hard"]).inc();
4657            }
4658        };
4659        result
4660    }};
4661}
