81#[derive(Hash, PartialEq, Eq, Debug, Clone)]
82pub struct AllowedPeer {

87/// Cache of `SuiAddress -> validator name` from the latest sui system state poll.
88/// Used by bridge/hashi resolvers to label peers by friendly validator name without

92/// SuiNodeProvider queries the sui blockchain and keeps a record of known validators based on the response from
93/// sui_getValidators.  The node name, public key and other info is extracted from the chain and stored in this
94/// data structure.  We pass this struct to the tls verifier and it depends on the state contained within.
95/// Handlers also use this data in an Extractor extension to check incoming clients on the http api against known keys.
96#[derive(Debug, Clone)]
97pub struct SuiNodeProvider {
98    sui_nodes: AllowedPeers,
99    bridge_nodes: AllowedPeers,
100    hashi_nodes: AllowedPeers,
101    static_nodes: AllowedPeers,
102    sui_validator_names: ValidatorNames,
103    rpc_url: String,
104    rpc_poll_interval: Duration,
105    /// Object ID of the `hashi::hashi::Hashi` shared object on the chain identified
106    /// by `rpc_url`. `None` disables the hashi resolver entirely.
107    hashi_object_id: Option<String>,
108}
109
110impl Allower for SuiNodeProvider {
111    fn allowed(&self, key: &Ed25519PublicKey) -> bool {
112        self.static_nodes.read().unwrap().contains_key(key)
113            || self.sui_nodes.read().unwrap().contains_key(key)
114            || self.bridge_nodes.read().unwrap().contains_key(key)
115            || self.hashi_nodes.read().unwrap().contains_key(key)
116    }
117}

119impl SuiNodeProvider {
120    pub fn new(
121        rpc_url: String,
122        rpc_poll_interval: Duration,
123        static_peers: Vec<AllowedPeer>,
124        hashi_object_id: Option<String>,
125    ) -> Self {
126        // build our hashmap with the static pub keys. we only do this one time at binary startup.
127        let static_nodes: HashMap<Ed25519PublicKey, AllowedPeer> = static_peers
128            .into_iter()
129            .map(|v| (v.public_key.clone(), v))
130            .collect();
131        let static_nodes = Arc::new(RwLock::new(static_nodes));
132        let sui_nodes = Arc::new(RwLock::new(HashMap::new()));
133        let bridge_nodes = Arc::new(RwLock::new(HashMap::new()));
134        let hashi_nodes = Arc::new(RwLock::new(HashMap::new()));
135        let sui_validator_names = Arc::new(RwLock::new(BTreeMap::new()));
136        Self {
137            sui_nodes,
138            bridge_nodes,
139            hashi_nodes,
140            static_nodes,
141            sui_validator_names,
142            rpc_url,
143            rpc_poll_interval,
144            hashi_object_id,
145        }
146    }

148    /// get is used to retrieve peer info in our handlers
149    pub fn get(&self, key: &Ed25519PublicKey) -> Option<AllowedPeer> {
150        debug!("look for {:?}", key);
151        // check static nodes first
152        if let Some(v) = self.static_nodes.read().unwrap().get(key) {
153            return Some(AllowedPeer {
154                name: v.name.to_owned(),
155                public_key: v.public_key.to_owned(),
156            });
157        }
158        // check sui validators
159        if let Some(v) = self.sui_nodes.read().unwrap().get(key) {
160            return Some(AllowedPeer {
161                name: v.name.to_owned(),
162                public_key: v.public_key.to_owned(),
163            });
164        }
165        // check bridge validators
166        if let Some(v) = self.bridge_nodes.read().unwrap().get(key) {
167            return Some(AllowedPeer {
168                name: v.name.to_owned(),
169                public_key: v.public_key.to_owned(),
170            });
171        }
172        // check hashi committee members
173        if let Some(v) = self.hashi_nodes.read().unwrap().get(key) {
174            return Some(AllowedPeer {
175                name: v.name.to_owned(),
176                public_key: v.public_key.to_owned(),
177            });
178        }
179        None
180    }

187    /// get_validators will retrieve known validators
188    async fn get_validators(url: String) -> Result<SuiSystemStateSummary> {
189        let rpc_method = "suix_getLatestSuiSystemState";
190        let observe = || {
191            let timer = JSON_RPC_DURATION
192                .with_label_values(&[rpc_method])
193                .start_timer();
194            || {
195                timer.observe_duration();
196            }
197        }();
198        let client = reqwest::Client::builder().build().unwrap();
199        let request = serde_json::json!({
200            "jsonrpc": "2.0",
201            "method":rpc_method,
202            "id":1,
203        });
204        let response = client
205            .post(url)
206            .header(reqwest::header::CONTENT_TYPE, "application/json")
207            .body(request.to_string())
208            .send()
209            .await
210            .with_context(|| {
211                JSON_RPC_STATE
212                    .with_label_values(&[rpc_method, "failed_get"])
213                    .inc();
214                observe();
215                "unable to perform json rpc"
216            })?;
217
218        let raw = response.bytes().await.with_context(|| {
219            JSON_RPC_STATE
220                .with_label_values(&[rpc_method, "failed_body_extract"])
221                .inc();
222            observe();
223            "unable to extract body bytes from json rpc"
224        })?;
225
226        #[derive(Debug, Deserialize)]
227        struct ResponseBody {
228            result: SuiSystemStateSummary,
229        }
230
231        let body: ResponseBody = match serde_json::from_slice(&raw) {
232            Ok(b) => b,
233            Err(error) => {
234                JSON_RPC_STATE
235                    .with_label_values(&[rpc_method, "failed_json_decode"])
236                    .inc();
237                observe();
238                bail!(
239                    "unable to decode json: {error} response from json rpc: {:?}",
240                    raw
241                )
242            }
243        };
244        JSON_RPC_STATE
245            .with_label_values(&[rpc_method, "success"])
246            .inc();
247        observe();
248        Ok(body.result)
249    }

305    async fn update_sui_validator_set(&self) {
306        match Self::get_validators(self.rpc_url.to_owned()).await {
307            Ok(summary) => {
308                // Snapshot the validator-address -> name map for downstream resolvers
309                // (bridge/hashi) before we hand `summary.active_validators` off to the
310                // network-key extractor.
311                let names: BTreeMap<SuiAddress, String> = summary
312                    .active_validators
313                    .iter()
314                    .map(|v| (v.sui_address, v.name.clone()))
315                    .collect();
316                {
317                    let mut nw = self.sui_validator_names.write().unwrap();
318                    *nw = names;
319                }
320
321                let validators = extract(summary);
322                let mut allow = self.sui_nodes.write().unwrap();
323                allow.clear();
324                allow.extend(validators);
325                info!(
326                    "{} sui validators managed to make it on the allow list",
327                    allow.len()
328                );
329            }
330            Err(error) => {
331                JSON_RPC_STATE
332                    .with_label_values(&["update_peer_count", "failed"])
333                    .inc();
334                error!("unable to refresh peer list: {error}");
335            }
336        };
337    }

339    async fn update_hashi_committee_set(&self, hashi_object_id: &str) {
340        let validator_names: BTreeMap<SuiAddress, String> =
341            self.sui_validator_names.read().unwrap().clone();
342
343        match resolve_hashi_committee(&self.rpc_url, hashi_object_id, &validator_names).await {
344            Ok(result) => {
345                HASHI_OBSERVED_EPOCH
346                    .with_label_values(&[hashi_object_id])
347                    .set(result.epoch as i64);
348                HASHI_ALLOWED_MEMBERS
349                    .with_label_values(&[hashi_object_id])
350                    .set(result.peers.len() as i64);
351                let mut allow = self.hashi_nodes.write().unwrap();
352                allow.clear();
353                allow.extend(result.peers);
354                info!(
355                    epoch = result.epoch,
356                    pending_epoch = ?result.pending_epoch,
357                    "{} hashi members on the allow list",
358                    allow.len(),
359                );
360            }
361            Err(error) => {
362                JSON_RPC_STATE
363                    .with_label_values(&["update_hashi_committee_set", "failed"])
364                    .inc();
365                error!("unable to refresh hashi peer list: {error:#}");
366            }
367        }
368    }

370    async fn update_bridge_validator_set(&self, metrics_keys: MetricsPubKeys) {
371        let sui_system = match Self::get_validators(self.rpc_url.to_owned()).await {
372            Ok(summary) => summary,
373            Err(error) => {
374                JSON_RPC_STATE
375                    .with_label_values(&["update_bridge_peer_count", "failed"])
376                    .inc();
377                error!("unable to get sui system state: {error}");
378                return;
379            }
380        };
381        match Self::get_bridge_validators(self.rpc_url.to_owned()).await {
382            Ok(summary) => {
383                let names = sui_system
384                    .active_validators
385                    .into_iter()
386                    .map(|v| (v.sui_address, v.name))
387                    .collect();
388                let validators = extract_bridge(summary, Arc::new(names), metrics_keys).await;
389                let mut allow = self.bridge_nodes.write().unwrap();
390                allow.clear();
391                allow.extend(validators);
392                info!(
393                    "{} bridge validators managed to make it on the allow list",
394                    allow.len()
395                );
396            }
397            Err(error) => {
398                JSON_RPC_STATE
399                    .with_label_values(&["update_bridge_peer_count", "failed"])
400                    .inc();
401                error!("unable to refresh sui bridge peer list: {error}");
402            }
403        };
404    }

406    /// poll_peer_list will act as a refresh interval for our cache
407    pub fn poll_peer_list(&self) {
408        info!("Started polling for peers using rpc: {}", self.rpc_url);
409
410        let rpc_poll_interval = self.rpc_poll_interval;
411        let cloned_self = self.clone();
412        let bridge_metrics_keys: MetricsPubKeys = Arc::new(RwLock::new(HashMap::new()));
413        tokio::spawn(async move {
414            let mut interval = tokio::time::interval(rpc_poll_interval);
415            interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
416
417            loop {
418                interval.tick().await;
419
420                // The sui validator set must update first; bridge and hashi resolvers
421                // read the cached `sui_validator_names` for friendly labeling.
422                cloned_self.update_sui_validator_set().await;
423                cloned_self
424                    .update_bridge_validator_set(bridge_metrics_keys.clone())
425                    .await;
426                if let Some(hashi_object_id) = cloned_self.hashi_object_id.as_deref() {
427                    cloned_self
428                        .update_hashi_committee_set(hashi_object_id)
429                        .await;
430                }
431            }
432        });
433    }

436/// extract will get the network pubkey bytes from a SuiValidatorSummary type.  This type comes from a
437/// full node rpc result.  See get_validators for details.  The key here, if extracted successfully, will
438/// ultimately be stored in the allow list and let us communicate with those actual peers via tls.
439fn extract(
440    summary: SuiSystemStateSummary,
441) -> impl Iterator<Item = (Ed25519PublicKey, AllowedPeer)> {
442    summary.active_validators.into_iter().filter_map(|vm| {
443        match Ed25519PublicKey::from_bytes(&vm.network_pubkey_bytes) {
444            Ok(public_key) => {
445                debug!(
446                    "adding public key {:?} for sui validator {:?}",
447                    public_key, vm.name
448                );
449                Some((
450                    public_key.clone(),
451                    AllowedPeer {
452                        name: vm.name,
453                        public_key,
454                    },
455                )) // scoped to filter_map
456            }
457            Err(error) => {
458                error!(
459                    "unable to decode public key for name: {:?} sui_address: {:?} error: {error}",
460                    vm.name, vm.sui_address
461                );
462                None // scoped to filter_map
463            }
464        }
465    })
466}

468async fn extract_bridge(
469    summary: BridgeSummary,
470    names: Arc<BTreeMap<SuiAddress, String>>,
471    metrics_keys: MetricsPubKeys,
472) -> Vec<(Ed25519PublicKey, AllowedPeer)> {
473    {
474        // Clean up the cache: retain only the metrics keys of the up-to-date bridge validator set
475        let mut metrics_keys_write = metrics_keys.write().unwrap();
476        metrics_keys_write.retain(|url, _| {
477            summary.committee.members.iter().any(|(_, cm)| {
478                String::from_utf8(cm.http_rest_url.clone()).ok().as_ref() == Some(url)
479            })
480        });
481    }
482
483    let client = reqwest::Client::builder()
484        .timeout(Duration::from_secs(10))
485        .build()
486        .unwrap();
487    let committee_members = summary.committee.members.clone();
488    let results: Vec<_> = stream::iter(committee_members)
489        .filter_map(|(_, cm)| {
490            let client = client.clone();
491            let metrics_keys = metrics_keys.clone();
492            let names = names.clone();
493            async move {
494                debug!(
495                    address =% cm.sui_address,
496                    "Extracting metrics public key for bridge node",
497                );
498
499                // Convert the Vec<u8> to a String and handle errors properly
500                let url_str = match String::from_utf8(cm.http_rest_url) {
501                    Ok(url) => url,
502                    Err(_) => {
503                        warn!(
504                            address =% cm.sui_address,
505                            "Invalid UTF-8 sequence in http_rest_url for bridge node ",
506                        );
507                        return None;
508                    }
509                };
510                // Parse the URL
511                let bridge_url = match Url::parse(&url_str) {
512                    Ok(url) => url,
513                    Err(_) => {
514                        warn!(url_str, "Unable to parse http_rest_url");
515                        return None;
516                    }
517                };
518
519                // Append "metrics_pub_key" to the path
520                let bridge_url = match append_path_segment(bridge_url, "metrics_pub_key") {
521                    Some(url) => url,
522                    None => {
523                        warn!(url_str, "Unable to append path segment to URL");
524                        return None;
525                    }
526                };
527
528                // Use the host portion of the http_rest_url as the "name"
529                let bridge_host = match bridge_url.host_str() {
530                    Some(host) => host,
531                    None => {
532                        warn!(url_str, "Hostname is missing from http_rest_url");
533                        return None;
534                    }
535                };
536                let bridge_name = names.get(&cm.sui_address).cloned().unwrap_or_else(|| {
537                    warn!(
538                        address =% cm.sui_address,
539                        "Bridge node not found in sui committee, using base URL as the name",
540                    );
541                    String::from(bridge_host)
542                });
543                let bridge_name = format!("bridge-{}", bridge_name);
544
545                let bridge_request_url = bridge_url.as_str();
546
547                let metrics_pub_key = match client.get(bridge_request_url).send().await {
548                    Ok(response) => {
549                        let raw = response.bytes().await.ok()?;
550                        let metrics_pub_key: String = match serde_json::from_slice(&raw) {
551                            Ok(key) => key,
552                            Err(error) => {
553                                warn!(?error, url_str, "Failed to deserialize response");
554                                return fallback_to_cached_key(
555                                    &metrics_keys,
556                                    &url_str,
557                                    &bridge_name,
558                                );
559                            }
560                        };
561                        let metrics_bytes = match Base64::decode(&metrics_pub_key) {
562                            Ok(pubkey_bytes) => pubkey_bytes,
563                            Err(error) => {
564                                warn!(
565                                    ?error,
566                                    bridge_name, "unable to decode public key for bridge node",
567                                );
568                                return None;
569                            }
570                        };
571                        match Ed25519PublicKey::from_bytes(&metrics_bytes) {
572                            Ok(pubkey) => {
573                                // Successfully fetched the key, update the cache
574                                let mut metrics_keys_write = metrics_keys.write().unwrap();
575                                metrics_keys_write.insert(url_str.clone(), pubkey.clone());
576                                debug!(
577                                    url_str,
578                                    public_key = ?pubkey,
579                                    "Successfully added bridge peer to metrics_keys"
580                                );
581                                pubkey
582                            }
583                            Err(error) => {
584                                warn!(
585                                    ?error,
586                                    bridge_request_url,
587                                    "unable to decode public key for bridge node",
588                                );
589                                return None;
590                            }
591                        }
592                    }
593                    Err(_) => {
594                        return fallback_to_cached_key(&metrics_keys, &url_str, &bridge_name);
595                    }
596                };
597                Some((
598                    metrics_pub_key.clone(),
599                    AllowedPeer {
600                        public_key: metrics_pub_key,
601                        name: bridge_name,
602                    },
603                ))
604            }
605        })
606        .collect()
607        .await;
608
609    results
610}

612fn fallback_to_cached_key(
613    metrics_keys: &MetricsPubKeys,
614    url_str: &str,
615    bridge_name: &str,
616) -> Option<(Ed25519PublicKey, AllowedPeer)> {
617    let metrics_keys_read = metrics_keys.read().unwrap();
618    if let Some(cached_key) = metrics_keys_read.get(url_str) {
619        debug!(
620            url_str,
621            "Using cached metrics public key after request failure"
622        );
623        Some((
624            cached_key.clone(),
625            AllowedPeer {
626                public_key: cached_key.clone(),
627                name: bridge_name.to_string(),
628            },
629        ))
630    } else {
631        warn!(
632            url_str,
633            "Failed to fetch public key and no cached key available"
634        );
635        None
636    }
637}

639fn append_path_segment(mut url: Url, segment: &str) -> Option<Url> {
640    url.path_segments_mut().ok()?.pop_if_empty().push(segment);
641    Some(url)

644// Hashi committee resolver.
645//
646// Reads on-chain state via sui-rpc gRPC, requesting the `json` rendering of
647// each Move object and path-walking into the fields we need — keeps the
648// resolver resilient to hashi adding sibling fields. See
649// `sui-types/src/object/rpc_visitor/mod.rs` for the JSON encoding rules
650// (UID/ID flattened to address strings, u64 as string, vector<u8> as base64,
651// Option<T> as null or bare T).
652
653fn json_struct(v: &JsonValue) -> Option<&std::collections::BTreeMap<String, JsonValue>> {
654    if let Some(JsonKind::StructValue(s)) = &v.kind {
655        Some(&s.fields)
656    } else {
657        None
658    }
659}

661fn json_string(v: &JsonValue) -> Option<&str> {
662    if let Some(JsonKind::StringValue(s)) = &v.kind {
663        Some(s)
664    } else {
665        None
666    }
667}
668
669fn json_list(v: &JsonValue) -> Option<&[JsonValue]> {
670    if let Some(JsonKind::ListValue(l)) = &v.kind {
671        Some(&l.values)
672    } else {
673        None
674    }
675}
676
677fn json_field<'a>(v: &'a JsonValue, key: &str) -> Option<&'a JsonValue> {
678    json_struct(v)?.get(key)
679}
680
681fn json_at<'a>(v: &'a JsonValue, path: &[&str]) -> Option<&'a JsonValue> {
682    let mut cur = v;
683    for key in path {
684        cur = json_field(cur, key)?;
685    }
686    Some(cur)
687}

689/// Parse a stringly-encoded u64 (Move JSON encodes u64 as a string to preserve precision).
690fn json_u64(v: &JsonValue, name: &str) -> Result<u64> {
691    json_string(v)
692        .with_context(|| format!("{name}: expected JSON string-encoded u64"))?
693        .parse::<u64>()
694        .with_context(|| format!("parsing {name} as u64"))
695}
696
697/// Parse a Move `Option<u64>`: JSON null = None, bare value = Some.
698fn json_option_u64(v: &JsonValue, name: &str) -> Result<Option<u64>> {
699    match &v.kind {
700        Some(JsonKind::NullValue(_)) | None => Ok(None),
701        _ => Ok(Some(json_u64(v, name)?)),
702    }
703}

714/// A single hashi member resolved from the on-chain `members` Bag. The
715/// `tls_public_key` may be empty for members that registered but haven't yet
716/// called `set_tls_public_key` — callers filter those out.
717#[derive(Debug, Clone, PartialEq, Eq)]
718pub(crate) struct ResolvedHashiMember {
719    pub validator_address: SuiAddress,
720    pub tls_public_key: Vec<u8>,
721}

723/// Output of `resolve_hashi_committee`: allowlist contents plus epoch info for
724/// observability metrics.
725#[derive(Debug)]
726struct HashiResolution {
727    epoch: u64,
728    pending_epoch: Option<u64>,
729    peers: Vec<(Ed25519PublicKey, AllowedPeer)>,
730}

732/// End-to-end resolve: gRPC reads + BCS decode -> peer allowlist entries.
733async fn resolve_hashi_committee(
734    rpc_url: &str,
735    hashi_object_id: &str,
736    validator_names: &BTreeMap<SuiAddress, String>,
737) -> Result<HashiResolution> {
738    let hashi_object_id = Address::from_str(hashi_object_id)
739        .with_context(|| format!("invalid hashi-object-id '{hashi_object_id}'"))?;
740    let mut client = SuiRpcClient::new(rpc_url.to_owned())
741        .with_context(|| format!("creating sui-rpc client for {rpc_url}"))?;
742
743    let snapshot = get_hashi_committee_snapshot(&mut client, hashi_object_id).await?;
744    debug!(
745        epoch = snapshot.epoch,
746        pending_epoch = ?snapshot.pending_epoch,
747        "fetched hashi committee snapshot"
748    );
749
750    // Union of validator_addresses across active and pending committees; the
751    // set tolerates the expected overlap during reconfig. A missing current
752    // Committee is not an error — at genesis the `committees` Bag is empty
753    // until the first `start_reconfig` runs.
754    let mut active_addrs: std::collections::HashSet<SuiAddress> =
755        match get_committee_validator_addresses(
756            &mut client,
757            snapshot.committees_bag_id,
758            snapshot.epoch,
759        )
760        .await
761        {
762            Ok(addrs) => addrs.into_iter().collect(),
763            Err(e) => {
764                debug!(
765                    epoch = snapshot.epoch,
766                    "no Committee at current epoch (pre-genesis or between reconfigs?): {e:#}"
767                );
768                std::collections::HashSet::new()
769            }
770        };
771    if let Some(next) = snapshot.pending_epoch {
772        match get_committee_validator_addresses(&mut client, snapshot.committees_bag_id, next).await
773        {
774            Ok(addrs) => active_addrs.extend(addrs),
775            Err(e) => warn!(
776                pending_epoch = next,
777                "could not fetch pending committee: {e:#}",
778            ),
779        }
780    }
781
782    // Fetch each member's MemberInfo concurrently. ~100 validators per chain,
783    // bounded concurrency keeps RPC load reasonable without serializing.
784    // sui_rpc::Client is cheap to clone — each clone shares the underlying
785    // tonic Channel so we don't open per-task connections.
786    let members: Vec<ResolvedHashiMember> = stream::iter(active_addrs)
787        .map(|addr| {
788            let mut client = client.clone();
789            let bag = snapshot.members_bag_id;
790            async move {
791                match get_hashi_member_info(&mut client, bag, addr).await {
792                    Ok(m) => Some(m),
793                    Err(e) => {
794                        warn!(addr =% addr, "could not fetch hashi MemberInfo: {e:#}");
795                        None
796                    }
797                }
798            }
799        })
800        .buffer_unordered(16)
801        .filter_map(|x| async move { x })
802        .collect()
803        .await;
804
805    let peers = extract_hashi(members, validator_names);
806
807    Ok(HashiResolution {
808        epoch: snapshot.epoch,
809        pending_epoch: snapshot.pending_epoch,
810        peers,
811    })
812}

814/// Filter to members with a valid 32-byte tls_public_key and build AllowedPeer
815/// entries labeled `hashi-<validator name>`.
816fn extract_hashi(
817    members: Vec<ResolvedHashiMember>,
818    validator_names: &BTreeMap<SuiAddress, String>,
819) -> Vec<(Ed25519PublicKey, AllowedPeer)> {
820    members
821        .into_iter()
822        .filter_map(|m| {
823            if m.tls_public_key.len() != 32 {
824                debug!(
825                    addr =% m.validator_address,
826                    "skipping hashi member with empty/invalid tls_public_key"
827                );
828                return None;
829            }
830            let pk = match Ed25519PublicKey::from_bytes(&m.tls_public_key) {
831                Ok(pk) => pk,
832                Err(error) => {
833                    warn!(
834                        addr =% m.validator_address,
835                        ?error,
836                        "invalid tls_public_key bytes for hashi member",
837                    );
838                    return None;
839                }
840            };
841            let name = validator_names
842                .get(&m.validator_address)
843                .cloned()
844                .unwrap_or_else(|| m.validator_address.to_string());
845            let labelled = format!("hashi-{name}");
846            debug!(
847                addr =% m.validator_address,
848                public_key = ?pk,
849                "adding hashi member to allow list as {labelled}",
850            );
851            Some((
852                pk.clone(),
853                AllowedPeer {
854                    name: labelled,
855                    public_key: pk,
856                },
857            ))
858        })
859        .collect()
860}

862async fn get_hashi_committee_snapshot(
863    client: &mut SuiRpcClient,
864    hashi_object_id: Address,
865) -> Result<CommitteeSetSnapshot> {
866    let rpc_method = "sui_rpc.LedgerService.GetObject:Hashi";
867    let _timer = JSON_RPC_DURATION
868        .with_label_values(&[rpc_method])
869        .start_timer();
870
871    let response = client
872        .ledger_client()
873        .get_object(
874            GetObjectRequest::new(&hashi_object_id)
875                .with_read_mask(FieldMask::from_paths([Object::path_builder().json()])),
876        )
877        .await
878        .with_context(|| {
879            JSON_RPC_STATE
880                .with_label_values(&[rpc_method, "failed_get"])
881                .inc();
882            format!("get_object failed for Hashi {hashi_object_id}")
883        })?;
884
885    let inner = response.into_inner();
886    let json = inner
887        .object_opt()
888        .and_then(|o| o.json_opt())
889        .with_context(|| {
890            JSON_RPC_STATE
891                .with_label_values(&[rpc_method, "missing_json"])
892                .inc();
893            format!("Hashi {hashi_object_id} response missing JSON rendering")
894        })?;
895
896    let cs = json_field(json, "committee_set").context("missing committee_set in Hashi JSON")?;
897    // Bag.id is a UID; the renderer flattens UID/ID to an address string.
898    let members_bag_id = json_at(cs, &["members", "id"])
899        .and_then(json_string)
900        .context("missing committee_set.members.id")?
901        .parse::<Address>()
902        .context("parsing members bag id as Address")?;
903    let committees_bag_id = json_at(cs, &["committees", "id"])
904        .and_then(json_string)
905        .context("missing committee_set.committees.id")?
906        .parse::<Address>()
907        .context("parsing committees bag id as Address")?;
908    let epoch = json_u64(
909        json_field(cs, "epoch").context("missing committee_set.epoch")?,
910        "committee_set.epoch",
911    )?;
912    let pending_epoch = json_option_u64(
913        json_field(cs, "pending_epoch_change")
914            .context("missing committee_set.pending_epoch_change")?,
915        "committee_set.pending_epoch_change",
916    )?;
917
918    JSON_RPC_STATE
919        .with_label_values(&[rpc_method, "success"])
920        .inc();
921
922    Ok(CommitteeSetSnapshot {
923        epoch,
924        pending_epoch,
925        members_bag_id,
926        committees_bag_id,
927    })
928}

930async fn get_committee_validator_addresses(
931    client: &mut SuiRpcClient,
932    committees_bag_id: Address,
933    epoch: u64,
934) -> Result<Vec<SuiAddress>> {
935    let rpc_method = "sui_rpc.LedgerService.GetObject:Committee";
936    let _timer = JSON_RPC_DURATION
937        .with_label_values(&[rpc_method])
938        .start_timer();
939
940    let field_id = committees_bag_id.derive_dynamic_child_id(
941        &TypeTag::U64,
942        &bcs::to_bytes(&epoch).expect("u64 always BCS-encodes"),
943    );
944
945    let response = client
946        .ledger_client()
947        .get_object(
948            GetObjectRequest::new(&field_id)
949                .with_read_mask(FieldMask::from_paths([Object::path_builder().json()])),
950        )
951        .await
952        .with_context(|| {
953            JSON_RPC_STATE
954                .with_label_values(&[rpc_method, "failed_get"])
955                .inc();
956            format!("get_object failed for Committee epoch={epoch} under {committees_bag_id}")
957        })?;
958
959    let inner = response.into_inner();
960    let json = inner
961        .object_opt()
962        .and_then(|o| o.json_opt())
963        .with_context(|| {
964            JSON_RPC_STATE
965                .with_label_values(&[rpc_method, "missing_json"])
966                .inc();
967            format!("Committee epoch={epoch} response missing JSON rendering")
968        })?;
969
970    let members = json_at(json, &["value", "members"])
971        .and_then(json_list)
972        .with_context(|| format!("Committee for epoch {epoch} missing value.members[]"))?;
973
974    let mut out = Vec::with_capacity(members.len());
975    for m in members {
976        let addr_str = json_field(m, "validator_address")
977            .and_then(json_string)
978            .context("missing CommitteeMember.validator_address")?;
979        let addr: SuiAddress = addr_str
980            .parse()
981            .with_context(|| format!("parsing CommitteeMember.validator_address: {addr_str}"))?;
982        out.push(addr);
983    }
984
985    JSON_RPC_STATE
986        .with_label_values(&[rpc_method, "success"])
987        .inc();
988    Ok(out)
989}

991async fn get_hashi_member_info(
992    client: &mut SuiRpcClient,
993    members_bag_id: Address,
994    validator_address: SuiAddress,
995) -> Result<ResolvedHashiMember> {
996    let rpc_method = "sui_rpc.LedgerService.GetObject:MemberInfo";
997    let _timer = JSON_RPC_DURATION
998        .with_label_values(&[rpc_method])
999        .start_timer();
1000
1001    let key = sui_address_to_sdk_address(validator_address);
1002    let field_id = members_bag_id.derive_dynamic_child_id(
1003        &TypeTag::Address,
1004        &bcs::to_bytes(&key).expect("Address always BCS-encodes"),
1005    );
1006
1007    let response = client
1008        .ledger_client()
1009        .get_object(
1010            GetObjectRequest::new(&field_id)
1011                .with_read_mask(FieldMask::from_paths([Object::path_builder().json()])),
1012        )
1013        .await
1014        .with_context(|| {
1015            JSON_RPC_STATE
1016                .with_label_values(&[rpc_method, "failed_get"])
1017                .inc();
1018            format!("get_object failed for MemberInfo {validator_address}")
1019        })?;
1020
1021    let inner = response.into_inner();
1022    let json = inner
1023        .object_opt()
1024        .and_then(|o| o.json_opt())
1025        .with_context(|| {
1026            JSON_RPC_STATE
1027                .with_label_values(&[rpc_method, "missing_json"])
1028                .inc();
1029            format!("MemberInfo {validator_address} response missing JSON rendering")
1030        })?;
1031
1032    let tls_public_key = match json_at(json, &["value", "tls_public_key"]).and_then(json_string) {
1033        Some(b64) => Base64::decode(b64)
1034            .with_context(|| format!("base64-decode tls_public_key for {validator_address}"))?,
1035        None => Vec::new(),
1036    };
1037
1038    JSON_RPC_STATE
1039        .with_label_values(&[rpc_method, "success"])
1040        .inc();
1041    Ok(ResolvedHashiMember {
1042        validator_address,
1043        tls_public_key,
1044    })
1045}

1340    #[test]
1341    fn extract_hashi_dedups_pubkey_collision() {
1342        // Two distinct validator_addresses with the same tls_public_key is a
1343        // pathological case (operators reusing keys); the second insertion into
1344        // the HashMap downstream of this fn wins. We just verify extract_hashi
1345        // itself emits both entries and lets the caller's HashMap dedup.
1346        let shared_key = fresh_pk_bytes();
1347        let members = vec![
1348            ResolvedHashiMember {
1349                validator_address: addr(0xAA),
1350                tls_public_key: shared_key.clone(),
1351            },
1352            ResolvedHashiMember {
1353                validator_address: addr(0xBB),
1354                tls_public_key: shared_key,
1355            },
1356        ];
1357        let peers = extract_hashi(members, &BTreeMap::new());
1358        assert_eq!(peers.len(), 2);
1359        // Same pubkey → both entries are dropped into the same HashMap key on
1360        // the consumer side; we confirm shared key here so the test fails loudly
1361        // if we ever silently change that behavior.
1362        assert_eq!(peers[0].0, peers[1].0);
1363    }
