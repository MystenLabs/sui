32/// These Display implementations provide alternate displays that are used to format info contained
33/// in these Structs when calling the CLI replay command with an additional provided flag.
34impl Display for Pretty<'_, FullPTB> {
35    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
36        let Pretty(full_ptb) = self;
37        let FullPTB { ptb, results } = full_ptb;
38
39        let ProgrammableTransaction { inputs, commands } = ptb;
40
41        // write input objects section
42        if !inputs.is_empty() {
43            let mut builder = TableBuilder::default();
44            for (i, input) in inputs.iter().enumerate() {
45                match input {
46                    Pure(v) => {
47                        if v.len() <= 16 {
48                            builder.push_record(vec![format!("{i:<3} Pure Arg          {:?}", v)]);
49                        } else {
50                            builder.push_record(vec![format!(
51                            "{i:<3} Pure Arg          [{}, {}, {}, {}, {}, {}, {}, {}, {}, {}, {}, {}, {}, {}, {}, ...]",
52                            v[0],
53                            v[1],
54                            v[2],
55                            v[3],
56                            v[4],
57                            v[5],
58                            v[6],
59                            v[7],
60                            v[8],
61                            v[9],
62                            v[10],
63                            v[11],
64                            v[12],
65                            v[13],
66                            v[14],
67                        )]);
68                        }
69                    }
70
71                    CallArg::Object(ObjectArg::ImmOrOwnedObject(o)) => {
72                        builder.push_record(vec![format!("{i:<3} Imm/Owned Object  ID: {}", o.0)]);
73                    }
74                    CallArg::Object(ObjectArg::SharedObject { id, .. }) => {
75                        builder.push_record(vec![format!("{i:<3} Shared Object     ID: {}", id)]);
76                    }
77                    CallArg::Object(ObjectArg::Receiving(o)) => {
78                        builder.push_record(vec![format!("{i:<3} Receiving Object  ID: {}", o.0)]);
79                    }
80                    CallArg::FundsWithdrawal(r) => {
81                        builder.push_record(vec![format!(
82                            "{i:<3} Funds Withdrawal reservation from {:?} and max amount {:?}",
83                            r.withdraw_from, r.reservation
84                        )]);
85                    }
86                };
87            }
88
89            let mut table = builder.build();
90            table.with(TablePanel::header("Input Objects"));
91            table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
92                1,
93                TableStyle::modern().get_horizontal(),
94            )]));
95            write!(f, "\n{}\n", table)?;
96        } else {
97            write!(f, "\n  No input objects for this transaction")?;
98        }
99
100        // write command results section
101        if !results.is_empty() {
102            write!(f, "\n\n")?;
103        }
104        for (i, result) in results.iter().enumerate() {
105            if i == results.len() - 1 {
106                write!(
107                    f,
108                    "╭───────────────────╮\n│ Command {i:<2} Output │\n╰───────────────────╯{}\n\n\n",
109                    Pretty(result)
110                )?
111            } else {
112                write!(
113                    f,
114                    "╭───────────────────╮\n│ Command {i:<2} Output │\n╰───────────────────╯{}\n",
115                    Pretty(result)
116                )?
117            }
118        }
119
120        // write ptb functions section
121        let mut builder = TableBuilder::default();
122        if !commands.is_empty() {
123            for (i, c) in commands.iter().enumerate() {
124                if i == commands.len() - 1 {
125                    builder.push_record(vec![format!("{i:<2} {}", Pretty(c))]);
126                } else {
127                    builder.push_record(vec![format!("{i:<2} {}\n", Pretty(c))]);
128                }
129            }
130
131            let mut table = builder.build();
132            table.with(TablePanel::header("Commands"));
133            table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
134                1,
135                TableStyle::modern().get_horizontal(),
136            )]));
137            write!(f, "\n{}\n", table)?;
138        } else {
139            write!(f, "\n  No commands for this transaction")?;
140        }
141
142        Ok(())
143    }
144}

146impl Display for Pretty<'_, Command> {
147    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
148        let Pretty(command) = self;
149        match command {
150            Command::MoveCall(p) => {
151                write!(f, "{}", Pretty(&**p))
152            }
153            Command::MakeMoveVec(ty_opt, elems) => {
154                write!(f, "MakeMoveVec:\n ┌")?;
155                if let Some(ty) = ty_opt {
156                    write!(f, "\n │ Type Tag: {ty}")?;
157                }
158                write!(f, "\n │ Arguments:\n │   ")?;
159                write_sep(f, elems.iter().map(Pretty), "\n │   ")?;
160                write!(f, "\n └")
161            }
162            Command::TransferObjects(objs, addr) => {
163                write!(f, "TransferObjects:\n ┌\n │ Arguments: \n │   ")?;
164                write_sep(f, objs.iter().map(Pretty), "\n │   ")?;
165                write!(f, "\n │ Address: {}\n └", Pretty(addr))
166            }
167            Command::SplitCoins(coin, amounts) => {
168                write!(
169                    f,
170                    "SplitCoins:\n ┌\n │ Coin: {}\n │ Amounts: \n │   ",
171                    Pretty(coin)
172                )?;
173                write_sep(f, amounts.iter().map(Pretty), "\n │   ")?;
174                write!(f, "\n └")
175            }
176            Command::MergeCoins(target, coins) => {
177                write!(
178                    f,
179                    "MergeCoins:\n ┌\n │ Target: {}\n │ Coins: \n │   ",
180                    Pretty(target)
181                )?;
182                write_sep(f, coins.iter().map(Pretty), "\n │   ")?;
183                write!(f, "\n └")
184            }
185            Command::Publish(_bytes, deps) => {
186                write!(f, "Publish:\n ┌\n │ Dependencies: \n │   ")?;
187                write_sep(f, deps, "\n │   ")?;
188                write!(f, "\n └")
189            }
190            Command::Upgrade(_bytes, deps, current_package_id, ticket) => {
191                write!(f, "Upgrade:\n ┌\n │ Dependencies: \n │   ")?;
192                write_sep(f, deps, "\n │   ")?;
193                write!(f, "\n │ Current Package ID: {current_package_id}")?;
194                write!(f, "\n │ Ticket: {}", Pretty(ticket))?;
195                write!(f, "\n └")
196            }
197        }
198    }
199}

201impl Display for Pretty<'_, ProgrammableMoveCall> {
202    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
203        let Pretty(move_call) = self;
204        let ProgrammableMoveCall {
205            package,
206            module,
207            function,
208            type_arguments,
209            arguments,
210        } = move_call;
211
212        write!(
213            f,
214            "MoveCall:\n ┌\n │ Function:  {} \n │ Module:    {}\n │ Package:   {}",
215            function, module, package
216        )?;
217
218        if !type_arguments.is_empty() {
219            write!(f, "\n │ Type Arguments: \n │   ")?;
220            write_sep(f, type_arguments, "\n │   ")?;
221        }
222        if !arguments.is_empty() {
223            write!(f, "\n │ Arguments: \n │   ")?;
224            write_sep(f, arguments.iter().map(Pretty), "\n │   ")?;
225        }
226
227        write!(f, "\n └")
228    }
229}

231impl Display for Pretty<'_, Argument> {
232    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
233        let Pretty(argument) = self;
234
235        let output = match argument {
236            Argument::GasCoin => "GasCoin".to_string(),
237            Argument::Input(i) => format!("Input  {}", i),
238            Argument::Result(i) => format!("Result {}", i),
239            Argument::NestedResult(j, k) => format!("Result {}: {}", j, k),
240        };
241        write!(f, "{}", output)
242    }
243}
244impl Display for Pretty<'_, ResolvedResults> {
245    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
246        let Pretty(ResolvedResults {
247            mutable_reference_outputs,
248            return_values,
249        }) = self;
250
251        let len_m_ref = mutable_reference_outputs.len();
252        let len_ret_vals = return_values.len();
253
254        if len_ret_vals > 0 {
255            write!(f, "\n Return Values:\n ──────────────")?;
256        }
257
258        for (i, value) in return_values.iter().enumerate() {
259            write!(f, "\n • Result {i:<2} ")?;
260            write!(f, "\n{:#}\n", value)?;
261        }
262
263        if len_m_ref > 0 {
264            write!(
265                f,
266                "\n Mutable Reference Outputs:\n ──────────────────────────"
267            )?;
268        }
269
270        for (arg, value) in mutable_reference_outputs {
271            write!(f, "\n • {} ", arg)?;
272            write!(f, "\n{:#}\n", value)?;
273        }
274
275        if len_ret_vals == 0 && len_m_ref == 0 {
276            write!(f, "\n No return values")?;
277        }
278
279        Ok(())
280    }
281}

283impl Display for Pretty<'_, TypeTag> {
284    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
285        let Pretty(type_tag) = self;
286        match type_tag {
287            TypeTag::Vector(v) => {
288                write!(f, "Vector of {}", Pretty(&**v))
289            }
290            TypeTag::Struct(s) => {
291                write!(f, "{}::{}", s.module, s.name)
292            }
293            _ => {
294                write!(f, "{}", type_tag)
295            }
296        }
297    }
298}

300fn resolve_to_layout(
301    type_tag: &TypeTag,
302    executor: &Arc<dyn Executor + Send + Sync>,
303    store_factory: &LocalExec,
304) -> MoveTypeLayout {
305    match type_tag {
306        TypeTag::Vector(inner) => {
307            MoveTypeLayout::Vector(Box::from(resolve_to_layout(inner, executor, store_factory)))
308        }
309        TypeTag::Struct(inner) => {
310            let mut layout_resolver = executor.type_layout_resolver(Box::new(store_factory));
311            layout_resolver
312                .get_annotated_layout(inner)
313                .unwrap()
314                .into_layout()
315        }
316        TypeTag::Bool => MoveTypeLayout::Bool,
317        TypeTag::U8 => MoveTypeLayout::U8,
318        TypeTag::U64 => MoveTypeLayout::U64,
319        TypeTag::U128 => MoveTypeLayout::U128,
320        TypeTag::Address => MoveTypeLayout::Address,
321        TypeTag::Signer => MoveTypeLayout::Signer,
322        TypeTag::U16 => MoveTypeLayout::U16,
323        TypeTag::U32 => MoveTypeLayout::U32,
324        TypeTag::U256 => MoveTypeLayout::U256,
325    }
326}

328fn resolve_value(
329    bytes: &[u8],
330    type_tag: &TypeTag,
331    executor: &Arc<dyn Executor + Send + Sync>,
332    store_factory: &LocalExec,
333) -> anyhow::Result<MoveValue> {
334    let layout = resolve_to_layout(type_tag, executor, store_factory);
335    BoundedVisitor::deserialize_value(bytes, &layout)
336}

338pub fn transform_command_results_to_annotated(
339    executor: &Arc<dyn Executor + Send + Sync>,
340    store_factory: &LocalExec,
341    results: Vec<ExecutionResult>,
342) -> anyhow::Result<Vec<ResolvedResults>> {
343    let mut output = Vec::new();
344    for (m_refs, return_vals) in results.iter() {
345        let mut m_refs_out = Vec::new();
346        let mut return_vals_out = Vec::new();
347        for (arg, bytes, tag) in m_refs {
348            m_refs_out.push((*arg, resolve_value(bytes, tag, executor, store_factory)?));
349        }
350        for (bytes, tag) in return_vals {
351            return_vals_out.push(resolve_value(bytes, tag, executor, store_factory)?);
352        }
353        output.push(ResolvedResults {
354            mutable_reference_outputs: m_refs_out,
355            return_values: return_vals_out,
356        });
357    }
358    Ok(output)
