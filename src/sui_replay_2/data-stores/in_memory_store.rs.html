4//! In-memory implementation of the replay interfaces: `TransactionStore`, `EpochStore`, and `ObjectStore`.
5//! The `InMemoryStore` provides fast in-memory lookups.
6//!
7//! This store is purely cache-based - it only returns data that has been explicitly stored in memory.
8//! For `TransactionStore`, `EpochStore`, and `ObjectStore`, missing data results in `Ok(None)`.
9//!
10//! # Usage Examples
11//!
12//! ```ignore
13//! use crate::data_stores::InMemoryStore;
14//! use crate::Node;
15//!
16//! // Create an in-memory store
17//! let store = InMemoryStore::new(Node::Mainnet);
18//!
19//! // Attempting to fetch data not in cache will return None
20//! let result = store.epoch_info(123); // Returns Ok(None) since cache is empty
21//!
22//! // Data must be explicitly added to cache before it can be retrieved
23//! // (This would typically be done by other parts of the system)
24//! ```
25

49/// In-memory store implementing the replay interfaces
50struct InMemoryStoreInner {
51    node: Node,
52    transaction_cache: BTreeMap<String, TransactionInfo>,
53    epoch_data_cache: BTreeMap<u64, EpochData>,
54    object_cache: BTreeMap<ObjectID, BTreeMap<u64, Object>>,
55    // The next 2 maps can be organized as flat maps or nested maps
56    // and the best solution depends on usage patterns.
57    // For now, we use flat maps but we may want to review in the future
58    /// Cache mapping (ObjectID, root_version) -> actual_version
59    /// Used for VersionQuery::RootVersion lookups
60    root_version_cache: BTreeMap<(ObjectID, u64), u64>,
61    /// Cache mapping (ObjectID, checkpoint) -> actual_version  
62    /// Used for VersionQuery::AtCheckpoint lookups
63    checkpoint_cache: BTreeMap<(ObjectID, u64), u64>,
64    /// Metrics: hit/miss counters for API calls
65    metrics: MemStoreMetrics,
66}
67
68// The RwLock is needed for 2 reasons:
69// 1. to make tokio happy
70// 2. to allow interior mutability for the cache
71pub struct InMemoryStore(RwLock<InMemoryStoreInner>);
72
73#[derive(Default)]
74struct MemStoreMetrics {
75    // transactions
76    txn_hit: AtomicU64,
77    txn_miss: AtomicU64,
78    txn_error: AtomicU64,
79    // epochs
80    epoch_hit: AtomicU64,
81    epoch_miss: AtomicU64,
82    epoch_error: AtomicU64,
83    // protocol config
84    proto_hit: AtomicU64,
85    proto_miss: AtomicU64,
86    proto_error: AtomicU64,
87    // objects by query kind
88    obj_version_hit: AtomicU64,
89    obj_version_miss: AtomicU64,
90    obj_version_error: AtomicU64,
91    obj_root_hit: AtomicU64,
92    obj_root_miss: AtomicU64,
93    obj_root_error: AtomicU64,
94    obj_checkpoint_hit: AtomicU64,
95    obj_checkpoint_miss: AtomicU64,
96    obj_checkpoint_error: AtomicU64,
97}

99impl InMemoryStore {
100    /// Create a new InMemoryStore with the given node
101    pub fn new(node: Node) -> Self {
102        Self(RwLock::new(InMemoryStoreInner {
103            node,
104            transaction_cache: BTreeMap::new(),
105            epoch_data_cache: BTreeMap::new(),
106            object_cache: BTreeMap::new(),
107            root_version_cache: BTreeMap::new(),
108            checkpoint_cache: BTreeMap::new(),
109            metrics: MemStoreMetrics::default(),
110        }))
111    }
112
113    /// Get the chain for this store
114    pub fn chain(&self) -> Chain {
115        self.0.read().unwrap().node.chain()
116    }
117
118    /// Get the node for this store
119    pub fn node(&self) -> Node {
120        self.0.read().unwrap().node.clone()
121    }
122
123    /// Clear all caches
124    pub fn clear_all_caches(&self) {
125        let mut inner = self.0.write().unwrap();
126        inner.transaction_cache.clear();
127        inner.epoch_data_cache.clear();
128        inner.object_cache.clear();
129        inner.root_version_cache.clear();
130        inner.checkpoint_cache.clear();
131    }
132
133    /// Get cache statistics
134    pub fn cache_stats(&self) -> CacheStats {
135        let inner = self.0.read().unwrap();
136        let object_cache_size = inner
137            .object_cache
138            .values()
139            .map(|versions| versions.len())
140            .sum();
141
142        let root_version_cache_size = inner.root_version_cache.len();
143
144        let checkpoint_cache_size = inner.checkpoint_cache.len();
145
146        CacheStats {
147            transaction_cache_size: inner.transaction_cache.len(),
148            epoch_data_cache_size: inner.epoch_data_cache.len(),
149            object_cache_size,
150            root_version_cache_size,
151            checkpoint_cache_size,
152        }
153    }
154
155    /// Add transaction data to the cache
156    pub fn add_transaction_data(&self, tx_digest: String, transaction_info: TransactionInfo) {
157        self.0
158            .write()
159            .unwrap()
160            .transaction_cache
161            .insert(tx_digest, transaction_info);
162    }
163
164    /// Add epoch data to the cache
165    pub fn add_epoch_data(&self, epoch: u64, epoch_data: EpochData) {
166        self.0
167            .write()
168            .unwrap()
169            .epoch_data_cache
170            .insert(epoch, epoch_data);
171    }
172
173    /// Add object data to the cache
174    pub fn add_object_data(&self, object_id: ObjectID, version: u64, object: Object) {
175        self.0
176            .write()
177            .unwrap()
178            .object_cache
179            .entry(object_id)
180            .or_default()
181            .insert(version, object);
182    }
183}

185/// Statistics about cache usage
186#[derive(Debug, Clone)]
187pub struct CacheStats {
188    pub transaction_cache_size: usize,
189    pub epoch_data_cache_size: usize,
190    pub object_cache_size: usize,
191    pub root_version_cache_size: usize,
192    pub checkpoint_cache_size: usize,
193}

195impl TransactionStore for InMemoryStore {
196    fn transaction_data_and_effects(
197        &self,
198        tx_digest: &str,
199    ) -> Result<Option<TransactionInfo>, Error> {
200        let inner = self.0.read().unwrap();
201        if let Some(cached) = inner.transaction_cache.get(tx_digest) {
202            inner.metrics.txn_hit.fetch_add(1, Ordering::Relaxed);
203            return Ok(Some(cached.clone()));
204        }
205        inner.metrics.txn_miss.fetch_add(1, Ordering::Relaxed);
206        Ok(None)
207    }
208}
209
210impl EpochStore for InMemoryStore {
211    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error> {
212        let inner = self.0.read().unwrap();
213        if let Some(cached) = inner.epoch_data_cache.get(&epoch) {
214            inner.metrics.epoch_hit.fetch_add(1, Ordering::Relaxed);
215            return Ok(Some(cached.clone()));
216        }
217        inner.metrics.epoch_miss.fetch_add(1, Ordering::Relaxed);
218        Ok(None)
219    }
220
221    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error> {
222        match self.epoch_info(epoch) {
223            Ok(Some(epoch_data)) => {
224                let inner = self.0.read().unwrap();
225                inner.metrics.proto_hit.fetch_add(1, Ordering::Relaxed);
226                Ok(Some(ProtocolConfig::get_for_version(
227                    ProtocolVersion::new(epoch_data.protocol_version),
228                    self.chain(),
229                )))
230            }
231            Ok(None) => {
232                let inner = self.0.read().unwrap();
233                inner.metrics.proto_miss.fetch_add(1, Ordering::Relaxed);
234                Ok(None)
235            }
236            Err(e) => {
237                let inner = self.0.read().unwrap();
238                inner.metrics.proto_error.fetch_add(1, Ordering::Relaxed);
239                Err(e)
240            }
241        }
242    }
243}

245impl ObjectStore for InMemoryStore {
246    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
247        let mut results = Vec::with_capacity(keys.len());
248        let inner = self.0.read().unwrap();
249
250        // Check caches for each key
251        for key in keys.iter() {
252            let (object_and_version, hit_ctr, miss_ctr) = match &key.version_query {
253                VersionQuery::Version(version) => {
254                    let res = if let Some(versions_map) = inner.object_cache.get(&key.object_id) {
255                        versions_map
256                            .get(version)
257                            .cloned()
258                            .map(|obj| (obj, *version))
259                    } else {
260                        None
261                    };
262                    (
263                        res,
264                        &inner.metrics.obj_version_hit,
265                        &inner.metrics.obj_version_miss,
266                    )
267                }
268                VersionQuery::RootVersion(max_version) => {
269                    let actual_version = inner
270                        .root_version_cache
271                        .get(&(key.object_id, *max_version))
272                        .copied();
273                    let res = actual_version.and_then(|actual_version| {
274                        inner
275                            .object_cache
276                            .get(&key.object_id)
277                            .and_then(|versions_map| versions_map.get(&actual_version))
278                            .cloned()
279                            .map(|obj| (obj, actual_version))
280                    });
281                    (
282                        res,
283                        &inner.metrics.obj_root_hit,
284                        &inner.metrics.obj_root_miss,
285                    )
286                }
287                VersionQuery::AtCheckpoint(checkpoint) => {
288                    let actual_version = inner
289                        .checkpoint_cache
290                        .get(&(key.object_id, *checkpoint))
291                        .copied();
292                    let res = actual_version.and_then(|actual_version| {
293                        inner
294                            .object_cache
295                            .get(&key.object_id)
296                            .and_then(|versions_map| versions_map.get(&actual_version))
297                            .cloned()
298                            .map(|obj| (obj, actual_version))
299                    });
300                    (
301                        res,
302                        &inner.metrics.obj_checkpoint_hit,
303                        &inner.metrics.obj_checkpoint_miss,
304                    )
305                }
306            };
307            if object_and_version.is_some() {
308                hit_ctr.fetch_add(1, Ordering::Relaxed);
309            } else {
310                miss_ctr.fetch_add(1, Ordering::Relaxed);
311            }
312            results.push(object_and_version);
313        }
314
315        Ok(results)
316    }
317}

345impl ObjectStoreWriter for InMemoryStore {
346    fn write_object(
347        &self,
348        key: &ObjectKey,
349        object: Object,
350        actual_version: u64,
351    ) -> Result<(), Error> {
352        let mut inner = self.0.write().unwrap();
353
354        // Always store the object at its actual version
355        inner
356            .object_cache
357            .entry(key.object_id)
358            .or_default()
359            .insert(actual_version, object);
360
361        // Handle version mappings based on query type
362        match &key.version_query {
363            VersionQuery::Version(_) => {
364                // No additional mapping needed for direct version queries
365            }
366            VersionQuery::RootVersion(max_version) => {
367                inner
368                    .root_version_cache
369                    .insert((key.object_id, *max_version), actual_version);
370            }
371            VersionQuery::AtCheckpoint(checkpoint) => {
372                inner
373                    .checkpoint_cache
374                    .insert((key.object_id, *checkpoint), actual_version);
375            }
376        }
377
378        Ok(())
379    }
380}

388impl StoreSummary for InMemoryStore {
389    fn summary<W: std::io::Write>(&self, w: &mut W) -> Result<()> {
390        let stats = self.cache_stats();
391        let inner = self.0.read().unwrap();
392        let m = &inner.metrics;
393        let txn_hit = m.txn_hit.load(Ordering::Relaxed);
394        let txn_miss = m.txn_miss.load(Ordering::Relaxed);
395        let txn_err = m.txn_error.load(Ordering::Relaxed);
396        let epoch_hit = m.epoch_hit.load(Ordering::Relaxed);
397        let epoch_miss = m.epoch_miss.load(Ordering::Relaxed);
398        let epoch_err = m.epoch_error.load(Ordering::Relaxed);
399        let proto_hit = m.proto_hit.load(Ordering::Relaxed);
400        let proto_miss = m.proto_miss.load(Ordering::Relaxed);
401        let proto_err = m.proto_error.load(Ordering::Relaxed);
402        let obj_v_hit = m.obj_version_hit.load(Ordering::Relaxed);
403        let obj_v_miss = m.obj_version_miss.load(Ordering::Relaxed);
404        let obj_v_err = m.obj_version_error.load(Ordering::Relaxed);
405        let obj_r_hit = m.obj_root_hit.load(Ordering::Relaxed);
406        let obj_r_miss = m.obj_root_miss.load(Ordering::Relaxed);
407        let obj_r_err = m.obj_root_error.load(Ordering::Relaxed);
408        let obj_c_hit = m.obj_checkpoint_hit.load(Ordering::Relaxed);
409        let obj_c_miss = m.obj_checkpoint_miss.load(Ordering::Relaxed);
410        let obj_c_err = m.obj_checkpoint_error.load(Ordering::Relaxed);
411        let obj_total_hit = obj_v_hit + obj_r_hit + obj_c_hit;
412        let obj_total_miss = obj_v_miss + obj_r_miss + obj_c_miss;
413        let obj_total_err = obj_v_err + obj_r_err + obj_c_err;
414        let total_hit = txn_hit + epoch_hit + proto_hit + obj_total_hit;
415        let total_miss = txn_miss + epoch_miss + proto_miss + obj_total_miss;
416        let total_err = txn_err + epoch_err + proto_err + obj_total_err;
417
418        writeln!(w, "InMemoryStore summary")?;
419        writeln!(w, "  Node: {:?}", self.node())?;
420        writeln!(w, "  Cache sizes:")?;
421        writeln!(
422            w,
423            "    Transactions: {} entries",
424            stats.transaction_cache_size
425        )?;
426        writeln!(w, "    Epochs: {} entries", stats.epoch_data_cache_size)?;
427        writeln!(w, "    Objects: {} versions", stats.object_cache_size)?;
428        writeln!(w, "    Root map: {} entries", stats.root_version_cache_size)?;
429        writeln!(
430            w,
431            "    Checkpoint map: {} entries",
432            stats.checkpoint_cache_size
433        )?;
434        writeln!(
435            w,
436            "  Overall:    hit={} miss={} error={}",
437            total_hit, total_miss, total_err
438        )?;
439        writeln!(w, "  Hits/Misses:")?;
440        writeln!(
441            w,
442            "    Transaction: hit={} miss={} error={}",
443            txn_hit, txn_miss, txn_err
444        )?;
445        writeln!(
446            w,
447            "    Epoch:       hit={} miss={} error={}",
448            epoch_hit, epoch_miss, epoch_err
449        )?;
450        writeln!(
451            w,
452            "    Protocol:    hit={} miss={} error={}",
453            proto_hit, proto_miss, proto_err
454        )?;
455        writeln!(
456            w,
457            "    Objects(all): hit={} miss={} error={}",
458            obj_total_hit, obj_total_miss, obj_total_err
459        )?;
460        writeln!(
461            w,
462            "      Version:     hit={} miss={} error={}",
463            obj_v_hit, obj_v_miss, obj_v_err
464        )?;
465        writeln!(
466            w,
467            "      RootVersion: hit={} miss={} error={}",
468            obj_r_hit, obj_r_miss, obj_r_err
469        )?;
470        writeln!(
471            w,
472            "      Checkpoint:  hit={} miss={} error={}",
473            obj_c_hit, obj_c_miss, obj_c_err
474        )?;
475        Ok(())
476    }
