11/// A read-through store that composes a primary (cache) and a secondary (source) store.
12/// It tries the primary first; on miss it reads from the secondary and writes back to the primary.
13pub struct ReadThroughStore<P, S>
14where
15    P: ReadWriteDataStore,
16    S: ReadDataStore,
17{
18    /// Primary store - supports both read and write (acts as cache)
19    primary: P,
20    /// Secondary store - read-only (acts as source of truth)
21    secondary: S,
22}
23
24impl<P, S> ReadThroughStore<P, S>
25where
26    P: ReadWriteDataStore,
27    S: ReadDataStore,
28{
29    /// Create a new read-through store with primary (cache) and secondary (source) stores
30    pub fn new(primary: P, secondary: S) -> Self {
31        Self { primary, secondary }
32    }
33}
34
35impl<P, S> TransactionStore for ReadThroughStore<P, S>
36where
37    P: ReadWriteDataStore,
38    S: ReadDataStore,
39{
40    fn transaction_data_and_effects(
41        &self,
42        tx_digest: &str,
43    ) -> Result<Option<TransactionInfo>, Error> {
44        match self.primary.transaction_data_and_effects(tx_digest)? {
45            Some(transaction_info) => Ok(Some(transaction_info)),
46            None => self
47                .secondary
48                .transaction_data_and_effects(tx_digest)?
49                .map_or(Ok(None), |info| {
50                    self.primary.write_transaction(tx_digest, info.clone())?;
51                    Ok(Some(info))
52                }),
53        }
54    }
55}

57impl<P, S> EpochStore for ReadThroughStore<P, S>
58where
59    P: ReadWriteDataStore,
60    S: ReadDataStore,
61{
62    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error> {
63        match self.primary.epoch_info(epoch)? {
64            Some(epoch_data) => Ok(Some(epoch_data)),
65            None => match self.secondary.epoch_info(epoch)? {
66                Some(epoch_data) => {
67                    self.primary.write_epoch_info(epoch, epoch_data.clone())?;
68                    Ok(Some(epoch_data))
69                }
70                None => Ok(None),
71            },
72        }
73    }
74
75    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error> {
76        match self.primary.protocol_config(epoch)? {
77            Some(config) => Ok(Some(config)),
78            None => self.secondary.protocol_config(epoch),
79        }
80    }
81}

83impl<P, S> ObjectStore for ReadThroughStore<P, S>
84where
85    P: ReadWriteDataStore,
86    S: ReadDataStore,
87{
88    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
89        let cached_objects = self.primary.get_objects(keys)?;
90
91        let mut keys_to_fetch = Vec::new();
92        let mut none_object_idx = Vec::new();
93        for (i, object) in cached_objects.iter().enumerate() {
94            if object.is_none() {
95                keys_to_fetch.push(keys[i].clone());
96                none_object_idx.push(i);
97            }
98        }
99
100        let mut objects = cached_objects;
101        if !keys_to_fetch.is_empty() {
102            let fetched_objects = self.secondary.get_objects(&keys_to_fetch)?;
103
104            // Sanity checks: the three vectors must align one-to-one
105            assert_eq!(none_object_idx.len(), keys_to_fetch.len());
106            assert_eq!(fetched_objects.len(), keys_to_fetch.len());
107
108            for ((idx, key), fetched) in none_object_idx
109                .iter()
110                .zip(keys_to_fetch.iter())
111                .zip(fetched_objects.iter())
112            {
113                // REVIEW: should we cache `None` to avoid repeated misses? Doing so would
114                // require API changes to represent cached-miss entries.
115                if let Some((object, actual_version)) = fetched {
116                    self.primary
117                        .write_object(key, object.clone(), *actual_version)?;
118                    objects[*idx] = Some((object.clone(), *actual_version));
119                }
120            }
121        }
122        Ok(objects)
123    }
124}

126impl<P, S> StoreSummary for ReadThroughStore<P, S>
127where
128    P: ReadWriteDataStore + StoreSummary,
129    S: ReadDataStore + StoreSummary,
130{
131    fn summary<W: std::io::Write>(&self, w: &mut W) -> Result<()> {
132        self.primary.summary(w)?;
133        self.secondary.summary(w)
134    }
135}

137impl<P, S> SetupStore for ReadThroughStore<P, S>
138where
139    P: ReadWriteDataStore + SetupStore,
140    S: ReadDataStore + SetupStore,
141{
142    fn setup(&self, _chain_id: Option<String>) -> Result<Option<String>, Error> {
143        // Call setup on secondary and pass the return value to primary if not an error
144        let chain_id = self.secondary.setup(None)?;
145        if let Some(ref chain_id) = chain_id {
146            self.primary.setup(Some(chain_id.clone()))?;
147        }
148        Ok(chain_id.clone())
149    }
