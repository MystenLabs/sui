4use crate::displays::Pretty;
5use std::fmt::{Display, Formatter};
6use sui_types::gas::GasUsageReport;
7use tabled::{
8    builder::Builder as TableBuilder,
9    settings::{Style as TableStyle, style::HorizontalLine},
10};
11
12impl Display for Pretty<'_, GasUsageReport> {
13    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
14        let Pretty(gas_report) = self;
15        display_info(f, gas_report)?;
16        per_object_storage_table(f, gas_report)?;
17        Ok(())
18    }
19}
20
21fn per_object_storage_table(f: &mut Formatter, gas_report: &GasUsageReport) -> std::fmt::Result {
22    let mut builder = TableBuilder::default();
23    builder.push_record(vec!["Object ID", "Bytes", "Old Rebate", "New Rebate"]);
24    for (object_id, per_obj_storage) in &gas_report.per_object_storage {
25        builder.push_record(vec![
26            object_id.to_string(),
27            per_obj_storage.new_size.to_string(),
28            per_obj_storage.storage_rebate.to_string(),
29            per_obj_storage.storage_cost.to_string(),
30        ]);
31    }
32    let mut table = builder.build();
33
34    table.with(TableStyle::rounded().horizontals([HorizontalLine::new(
35        1,
36        TableStyle::modern().get_horizontal(),
37    )]));
38    write!(f, "\n{}\n", table)
39}
40
41fn display_info(f: &mut Formatter<'_>, gas_report: &GasUsageReport) -> std::fmt::Result {
42    let mut builder = TableBuilder::default();
43    macro_rules! record {
44        ($($msg:expr),+) => {
45            builder.push_record(vec![$($msg.to_string()),+]);
46        };
47    }
48    record!("Gas Info");
49
50    record!("Computation Cost", gas_report.cost_summary.computation_cost);
51    record!("Storage Cost", gas_report.cost_summary.storage_cost);
52    record!("Storage Rebate", gas_report.cost_summary.storage_rebate);
53    record!(
54        "Non-Refundable Storage Fee",
55        gas_report.cost_summary.non_refundable_storage_fee
56    );
57    record!("Gas Used", gas_report.gas_used);
58    record!("Gas Budget", gas_report.gas_budget);
59    record!("Gas Price", gas_report.gas_price);
60    record!("Reference Gas Price", gas_report.reference_gas_price);
61    record!("Storage Gas Price", gas_report.storage_gas_price);
62    record!("Rebate Rate", gas_report.rebate_rate);
63
64    let mut table = builder.build();
65    table.with(TableStyle::rounded());
66
67    write!(f, "\n{}\n", table)
68}
