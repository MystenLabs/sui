4//! Execution module for replay.
5//! The call to the executor `execute_transaction_to_effects` is here
6//! and the logic to call it is pretty straightforward.
7//! `execute_transaction_to_effects` requires info from the `EpochStore`
8//! (epoch, protocol config, epoch start timestamp, rgp), from the `TransactionStore`
9//! as in transaction data and effects, and from the `ObjectStore` for dynamic loads
10//! (e.g. dynamic fields).
11//! This module also contains the traits used by execution to talk to
12//! the store (BackingPackageStore, ObjectStore, ChildObjectResolver)
13

50// Returned struct from execution. Contains all the data related to a transaction.
51// Transaction data and effects (both expected and actual) and the caches containing
52// the objects used during execution.
53pub struct TxnContextAndEffects {
54    pub txn_data: TransactionData,             // original transaction data
55    pub execution_effects: TransactionEffects, // effects of the replay execution
56    pub expected_effects: TransactionEffects,  // expected effects as found in the transaction data
57    pub gas_status: SuiGasStatus,              // gas status of the replay execution
58    pub object_cache: BTreeMap<ObjectID, BTreeMap<u64, Object>>, // object cache
59    pub inner_store: InnerTemporaryStore,      // temporary store used during execution
60    pub checkpoint: u64,                       // checkpoint where the transaction was included
61    pub protocol_version: u64,                 // protocol version used for execution
62}
63
64// Entry point. Executes a transaction.
65// Return all the information that can be used by a client
66// to verify execution.
67#[allow(clippy::type_complexity)]
68pub fn execute_transaction_to_effects(
69    txn: ReplayTransaction,
70    epoch_store: &dyn EpochStore,
71    object_store: &dyn ObjectStore,
72    trace_builder_opt: &mut Option<MoveTraceBuilder>,
73) -> Result<
74    (
75        Result<(), ExecutionError>, // transaction result
76        TxnContextAndEffects,       // data touched and changed during execution
77    ),
78    Error,
79> {
80    debug!(op = "execute_tx", phase = "start", "execution");
81    // TODO: Hook up...
82    let config_certificate_deny_set: HashSet<TransactionDigest> = HashSet::new();
83
84    let epoch = txn.epoch();
85    let digest = txn.digest();
86    let checkpoint = txn.checkpoint();
87    let _span = debug_span!("execute_tx", %digest, epoch, checkpoint).entered();
88    let input_objects = txn.get_input_objects_for_replay()?;
89    let ReplayTransaction {
90        digest,
91        checkpoint: _,
92        txn_data,
93        effects: expected_effects,
94        executor,
95        object_cache,
96    } = txn;
97
98    let protocol_config = &executor.protocol_config;
99    let epoch_data = epoch_store
100        .epoch_info(epoch)?
101        .ok_or_else(|| anyhow!(format!("Epoch {} not found", epoch)))?;
102    let epoch_start_timestamp = epoch_data.start_timestamp;
103    let gas_status = if txn_data.kind().is_system_tx() {
104        SuiGasStatus::new_unmetered()
105    } else {
106        SuiGasStatus::new(
107            txn_data.gas_data().budget,
108            txn_data.gas_data().price,
109            epoch_data.rgp,
110            protocol_config,
111        )
112        .expect("Failed to create gas status")
113    };
114    let store: ReplayStore<'_> = ReplayStore {
115        checkpoint,
116        store: object_store,
117        object_cache: RefCell::new(object_cache),
118    };
119    let input_objects = CheckedInputObjects::new_for_replay(input_objects);
120    // TODO(address-balances): Get withdraw status from effects.
121    let early_execution_error = get_early_execution_error(
122        &digest,
123        &input_objects,
124        &config_certificate_deny_set,
125        // TODO(address-balances): Support balance withdraw status for replay
126        &FundsWithdrawStatus::MaybeSufficient,
127    );
128    let execution_params = match early_execution_error {
129        Some(error) => ExecutionOrEarlyError::Err(error),
130        None => ExecutionOrEarlyError::Ok(()),
131    };
132    let (inner_store, gas_status, effects, _execution_timing, result) =
133        executor.executor.execute_transaction_to_effects(
134            &store,
135            protocol_config,
136            executor.metrics.clone(),
137            false, // expensive checks
138            execution_params,
139            &epoch,
140            epoch_start_timestamp,
141            input_objects,
142            txn_data.gas_data().clone(),
143            gas_status,
144            txn_data.kind().clone(),
145            txn_data.sender(),
146            digest,
147            trace_builder_opt,
148        );
149    let ReplayStore {
150        object_cache,
151        checkpoint: _,
152        store: _,
153    } = store;
154    let mut object_cache = object_cache.into_inner();
155
156    // Get created objects from transaction effects
157    for (object_ref, _owner) in effects.created() {
158        let object_id = object_ref.0;
159        // Look for the created object in inner_store's written objects
160        if let Some(object) = inner_store.written.get(&object_id) {
161            object_cache
162                .entry(object_id)
163                .or_default()
164                .insert(object.version().value(), object.clone());
165        } else {
166            // Return error if created object is not found in inner_store
167            return Err(anyhow!(
168                "Created object {} not found in inner_store written objects",
169                object_id
170            ));
171        }
172    }
173
174    debug!(op = "execute_tx", phase = "end", "execution");
175    Ok((
176        result,
177        TxnContextAndEffects {
178            txn_data,
179            execution_effects: effects,
180            expected_effects,
181            gas_status,
182            object_cache,
183            inner_store,
184            checkpoint,
185            protocol_version: protocol_config.version.as_u64(),
186        },
187    ))
188}

190impl ReplayExecutor {
191    pub fn new(protocol_config: ProtocolConfig) -> Result<Self, Error> {
192        let silent = true; // disable Move debug API
193        let executor = sui_execution::executor(&protocol_config, silent)
194            .context("Filed to create executor. ProtocolConfig inconsistency?")?;
195
196        let registry = prometheus::Registry::new();
197        let metrics = Arc::new(LimitsMetrics::new(&registry));
198
199        Ok(Self {
200            protocol_config,
201            executor,
202            metrics,
203        })
204    }
205}

211struct ReplayStore<'a> {
212    store: &'a dyn ObjectStore,
213    object_cache: RefCell<BTreeMap<ObjectID, BTreeMap<u64, Object>>>,
214    checkpoint: u64,
215}
216
217impl ReplayStore<'_> {
218    // utility function shared across traits functions
219    fn get_object_at_version(
220        &self,
221        object_id: &ObjectID,
222        version: VersionNumber,
223    ) -> Option<Object> {
224        // look up in the cache
225        if let Some(object) = self
226            .object_cache
227            .borrow()
228            .get(object_id)
229            .and_then(|versions| versions.get(&version.value()).cloned())
230        {
231            return Some(object);
232        }
233
234        // if not in the cache fetch it from the store
235        let object = self
236            .store
237            .get_objects(&[ObjectKey {
238                object_id: *object_id,
239                version_query: VersionQuery::Version(version.value()),
240            }])
241            .map_err(|e| SuiErrorKind::Storage(e.to_string()))
242            .ok()?
243            .into_iter()
244            .next()?
245            .map(|(obj, _version)| obj);
246        // add it to the cache
247        if let Some(obj) = &object {
248            self.object_cache
249                .borrow_mut()
250                .entry(obj.id())
251                .or_default()
252                .insert(obj.version().value(), obj.clone());
253        }
254
255        object
256    }
257}

259impl BackingPackageStore for ReplayStore<'_> {
260    // Look for a package in the object cache first.
261    // If not found, fetch it from the store, add to the cache, and return it.
262    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
263        trace!("get_package_object({})", package_id);
264        if let Some(versions) = self.object_cache.borrow().get(package_id) {
265            debug_assert!(
266                versions.len() == 1,
267                "Expected only one version in cache for package object {}",
268                package_id
269            );
270            return Ok(Some(PackageObject::new(
271                versions.values().next().unwrap().clone(),
272            )));
273        }
274        // If the package is not in the cache, fetch it from the store
275        let object_key = ObjectKey {
276            object_id: *package_id,
277            // Using AtCheckpoint to get the package version at the time of execution
278            version_query: VersionQuery::AtCheckpoint(self.checkpoint),
279        };
280        let package = self
281            .store
282            .get_objects(&[object_key])
283            .map_err(|e| SuiErrorKind::Storage(e.to_string()))?;
284        debug_assert!(
285            package.len() == 1,
286            "Expected one package object for {}",
287            package_id
288        );
289        let maybe = package.into_iter().next().and_then(|o| o);
290        if let Some((package, _version)) = maybe {
291            self.object_cache
292                .borrow_mut()
293                .entry(*package_id)
294                .or_default()
295                .insert(package.version().value(), package.clone());
296            Ok(Some(PackageObject::new(package)))
297        } else {
298            Ok(None)
299        }
300    }
301}

303impl sui_types::storage::ObjectStore for ReplayStore<'_> {
304    // Get an object by its ID. This translates to a query for the object
305    // at the checkpoint (mimic latest runtime behavior)
306    fn get_object(&self, object_id: &ObjectID) -> Option<Object> {
307        trace!("get_object({})", object_id);
308
309        // Check cache - use a scope to ensure the borrow is dropped before we fetch from store
310        {
311            let cache = self.object_cache.borrow();
312            if let Some(versions) = cache.get(object_id) {
313                return versions.last_key_value().map(|(_version, obj)| obj.clone());
314            }
315        } // Borrow dropped here
316
317        let fetched_object = self
318            .store
319            .get_objects(&[ObjectKey {
320                object_id: *object_id,
321                version_query: VersionQuery::AtCheckpoint(self.checkpoint),
322            }])
323            .map_err(|e| SuiErrorKind::Storage(e.to_string()))
324            .ok()?
325            .into_iter()
326            .next()?
327            .map(|(obj, _version)| obj)?;
328
329        // Add the fetched object to the cache
330        let mut cache = self.object_cache.borrow_mut();
331        cache
332            .entry(*object_id)
333            .or_default()
334            .insert(fetched_object.version().value(), fetched_object.clone());
335
336        Some(fetched_object)
337    }
338
339    // Get an object by its ID and version
340    fn get_object_by_key(&self, object_id: &ObjectID, version: VersionNumber) -> Option<Object> {
341        trace!("get_object_by_key({}, {})", object_id, version);
342        self.get_object_at_version(object_id, version)
343    }
344}

346impl ChildObjectResolver for ReplayStore<'_> {
347    // Load an `Object` at a root version. That is the version that is
348    // less than or equal to the given `child_version_upper_bound`.
349    fn read_child_object(
350        &self,
351        _parent: &ObjectID,
352        child: &ObjectID,
353        child_version_upper_bound: SequenceNumber,
354    ) -> SuiResult<Option<Object>> {
355        trace!(
356            "read_child_object({}, {}, {})",
357            _parent, child, child_version_upper_bound,
358        );
359        let object_key = ObjectKey {
360            object_id: *child,
361            version_query: VersionQuery::RootVersion(child_version_upper_bound.value()),
362        };
363        let object = self
364            .store
365            .get_objects(&[object_key])
366            .map_err(|e| SuiErrorKind::Storage(e.to_string()))?;
367        debug_assert!(object.len() == 1, "Expected one object for {}", child,);
368        let object = object
369            .into_iter()
370            .next()
371            .unwrap()
372            .map(|(obj, _version)| obj);
373
374        // Add object to cache if it exists and not already cached
375        if let Some(ref obj) = object {
376            self.object_cache
377                .borrow_mut()
378                .entry(obj.id())
379                .or_default()
380                .insert(obj.version().value(), obj.clone());
381        }
382        Ok(object)
383    }
384
385    // Load a receiving object. Results in a query at a specific version
386    // (`receive_object_at_version`).
387    fn get_object_received_at_version(
388        &self,
389        owner: &ObjectID,
390        receiving_object_id: &ObjectID,
391        receive_object_at_version: SequenceNumber,
392        epoch_id: EpochId,
393    ) -> SuiResult<Option<Object>> {
394        trace!(
395            "get_object_received_at_version({}, {}, {}, {})",
396            owner, receiving_object_id, receive_object_at_version, epoch_id
397        );
398        Ok(self.get_object_at_version(receiving_object_id, receive_object_at_version))
399    }
400}
