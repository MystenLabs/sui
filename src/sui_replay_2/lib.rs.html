49// Arguments to the replay tool.
50// It allows to replay a single transaction by digest or
51// a file containing multiple digests, one per line.
52// This may evolve to something very different in time and
53// it's not meant to be stable.
54// The options available are very convenient for the current
55// development cycle.
56#[derive(Parser, Clone, Debug)]
57#[clap(
58    name = "Sui Replay Tool",
59    about = "Replay executed transactions.",
60    rename_all = "kebab-case"
61)]
62pub struct Config {
63    #[command(subcommand)]
64    pub command: Option<Command>,
65    #[command(flatten)]
66    pub replay_stable: ReplayConfigStable,
67    #[command(flatten)]
68    pub replay_experimental: ReplayConfigExperimental,
69}

312/// Load replay configuration from ~/.sui/sui_config/replay.toml file.
313/// Returns default config (all fields set to None) if file cannot be found or read.
314pub fn load_config_file() -> Result<ReplayConfigStable> {
315    let config_dir = match sui_config_dir() {
316        Ok(dir) => dir,
317        Err(e) => {
318            eprintln!("Cannot locate replay config file: {e}");
319            return Ok(ReplayConfigStable::default());
320        }
321    };
322    let config_file_path = config_dir.join(CONFIG_FILE_NAME);
323
324    if !config_file_path.exists() {
325        return Ok(ReplayConfigStable::default());
326    }
327
328    let content = fs::read_to_string(&config_file_path).map_err(|e| {
329        anyhow!(
330            "Failed to read replay config file '{:?}': {}",
331            config_file_path,
332            e
333        )
334    })?;
335
336    let config: TOMLConfig = toml::from_str(&content).map_err(|e| {
337        anyhow!(
338            "Failed to parse replay config file '{:?}': {}",
339            config_file_path,
340            e
341        )
342    })?;
343
344    config.flags.ok_or_else(|| {
345        anyhow!(
346            "No flags section found in replay config file '{:?}'",
347            config_file_path
348        )
349    })
350}

352/// Merge CLI flags and config file flags into a single config. CLI flags take
353/// precedence over config file flags, which take precedence over defaults.
354pub fn merge_configs(
355    cli_config: ReplayConfigStable,
356    file_config: ReplayConfigStable,
357) -> ReplayConfigStableInternal {
358    let default_config = ReplayConfigStableInternal::default();
359    ReplayConfigStableInternal {
360        digest: cli_config.digest.or(file_config.digest),
361
362        digests_path: cli_config.digests_path.or(file_config.digests_path),
363
364        terminate_early: cli_config
365            .terminate_early
366            .or(file_config.terminate_early)
367            .unwrap_or(default_config.terminate_early),
368
369        trace: cli_config
370            .trace
371            .or(file_config.trace)
372            .unwrap_or(default_config.trace),
373
374        output_dir: cli_config.output_dir.or(file_config.output_dir),
375
376        show_effects: cli_config
377            .show_effects
378            .or(file_config.show_effects)
379            .unwrap_or(default_config.show_effects),
380
381        overwrite: cli_config
382            .overwrite
383            .or(file_config.overwrite)
384            .unwrap_or(default_config.overwrite),
385    }
386}

388pub async fn handle_replay_config(
389    stable_config: &ReplayConfigStableInternal,
390    experimental_config: &ReplayConfigExperimental,
391    version: &str,
392) -> Result<PathBuf> {
393    let ReplayConfigStableInternal {
394        digest,
395        digests_path,
396        terminate_early,
397        trace,
398        output_dir,
399        show_effects: _, // used in the caller
400        overwrite: overwrite_existing,
401    } = &stable_config;
402    let mut terminate_early = *terminate_early;
403
404    let ReplayConfigExperimental {
405        node,
406        verbose,
407        store_mode,
408        track_time,
409        cache_executor,
410    } = experimental_config;
411
412    let output_root_dir = if let Some(dir) = output_dir {
413        dir.to_path_buf()
414    } else {
415        // Default output directory is `<cur_dir>/.replay/<digest>`
416        let current_dir =
417            std::env::current_dir().map_err(|e| anyhow!("Failed to get current directory: {e}"))?;
418        current_dir.join(DEFAULT_OUTPUT_DIR)
419    };
420
421    // If trying to trace but the binary was not built with the tracing feature flag raise an error.
422    #[cfg(not(feature = "tracing"))]
423    if *trace {
424        bail!(
425            "Tracing is not enabled in this build. Please rebuild with the \
426            `tracing` feature (`--features tracing`) to use tracing in replay"
427        );
428    }
429
430    // If a file is specified it is read and the digest ignored.
431    // Once we decide on the options we want this is likely to change.
432    let digests = if let Some(digests_path) = digests_path {
433        // read digests from file
434        std::fs::read_to_string(digests_path.clone())
435            .map_err(|e| {
436                anyhow!(
437                    "Failed to read digests file {}: {e}",
438                    digests_path.display(),
439                )
440            })?
441            .lines()
442            .map(|s| s.trim().to_string())
443            .collect::<Vec<_>>()
444    } else if let Some(tx_digest) = digest {
445        // terminate early if a single digest is provided this way we get proper error messages from
446        terminate_early = true;
447        // single digest provided
448        vec![tx_digest.clone()]
449    } else {
450        bail!("either --digest or --digests-path must be provided");
451    };
452
453    debug!("Binary version: {version}");
454
455    // Build the selected data store and run replay
456    match store_mode {
457        StoreMode::GqlOnly => {
458            let gql_store = DataStore::new(node.clone(), version)
459                .map_err(|e| anyhow!("Failed to create data store: {:?}", e))?;
460            run_replay(
461                &gql_store,
462                &output_root_dir,
463                &digests,
464                node,
465                *overwrite_existing,
466                *trace,
467                *verbose,
468                terminate_early,
469                *track_time,
470                *cache_executor,
471            )
472            .await?;
473        }
474        StoreMode::FsThenGql => {
475            let fs_store = FileSystemStore::new(node.clone())
476                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
477            let gql_store = DataStore::new(node.clone(), version)
478                .map_err(|e| anyhow!("Failed to create data store: {:?}", e))?;
479            let store = ReadThroughStore::new(fs_store, gql_store);
480            run_replay(
481                &store,
482                &output_root_dir,
483                &digests,
484                node,
485                *overwrite_existing,
486                *trace,
487                *verbose,
488                terminate_early,
489                *track_time,
490                *cache_executor,
491            )
492            .await?;
493        }
494        StoreMode::FsOnly => {
495            let fs_store = FileSystemStore::new(node.clone())
496                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
497            run_replay(
498                &fs_store,
499                &output_root_dir,
500                &digests,
501                node,
502                *overwrite_existing,
503                *trace,
504                *verbose,
505                terminate_early,
506                *track_time,
507                *cache_executor,
508            )
509            .await?;
510        }
511        StoreMode::InmemFs => {
512            let fs_store = FileSystemStore::new(node.clone())
513                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
514            let in_memory_store = InMemoryStore::new(node.clone());
515            let store = ReadThroughStore::new(in_memory_store, fs_store);
516            run_replay(
517                &store,
518                &output_root_dir,
519                &digests,
520                node,
521                *overwrite_existing,
522                *trace,
523                *verbose,
524                terminate_early,
525                *track_time,
526                *cache_executor,
527            )
528            .await?;
529        }
530        StoreMode::InmemFsGql => {
531            let fs_store = FileSystemStore::new(node.clone())
532                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
533            let gql_store = DataStore::new(node.clone(), version)
534                .map_err(|e| anyhow!("Failed to create data store: {:?}", e))?;
535            let secondary_store = ReadThroughStore::new(fs_store, gql_store);
536            let in_memory_store = InMemoryStore::new(node.clone());
537            let store = ReadThroughStore::new(in_memory_store, secondary_store);
538            run_replay(
539                &store,
540                &output_root_dir,
541                &digests,
542                node,
543                *overwrite_existing,
544                *trace,
545                *verbose,
546                terminate_early,
547                *track_time,
548                *cache_executor,
549            )
550            .await?;
551        }
552    }
553
554    Ok(output_root_dir)
555}

557async fn run_replay<S>(
558    data_store: &S,
559    output_root_dir: &Path,
560    digests: &[String],
561    node: &Node,
562    overwrite_existing: bool,
563    trace: bool,
564    verbose: bool,
565    terminate_early: bool,
566    track_time: bool,
567    cache_executor: bool,
568) -> Result<()>
569where
570    S: ReadDataStore + StoreSummary + SetupStore,
571{
572    use crate::replay_txn::ExecutorProvider;
573    use std::time::Instant;
574
575    data_store.setup(None)?;
576    let mut total_metrics = TotalMetrics::new();
577    let mut executor_provider = ExecutorProvider::new(cache_executor);
578
579    for tx_digest in digests {
580        let tx_dir = output_root_dir.join(tx_digest);
581        let artifact_manager = ArtifactManager::new(&tx_dir, overwrite_existing)?;
582        let span = info_span!("replay", tx_digest = %tx_digest);
583
584        let tx_start = Instant::now();
585        let result = replay_transaction(
586            &artifact_manager,
587            tx_digest,
588            data_store,
589            node.network_name(),
590            trace,
591            &mut executor_provider,
592        )
593        .instrument(span)
594        .await;
595        let tx_total_ms = tx_start.elapsed().as_millis();
596
597        let success = result.is_ok();
598        let exec_ms = result.as_ref().ok().copied().unwrap_or(0);
599
600        total_metrics.add_transaction(success, tx_total_ms, exec_ms);
601
602        // Print per-transaction result
603        let status = if success { "OK" } else { "FAILED" };
604        if track_time {
605            println!(
606                "> Replayed txn {} ({}): exec_ms={}, total_ms={}",
607                tx_digest, status, exec_ms, tx_total_ms
608            );
609        } else {
610            println!("> Replayed txn {} ({})", tx_digest, status);
611        }
612
613        match result {
614            Err(e) if terminate_early => {
615                error!(tx_digest = %tx_digest, error = ?e, "Replay error; terminating early");
616                bail!("Replay terminated due to error: {}", e);
617            }
618            Err(e) => {
619                error!(tx_digest = %tx_digest, error = ?e, "Replay failed");
620            }
621            Ok(_) => {
622                info!(tx_digest = %tx_digest, "Replay succeeded");
623            }
624        }
625    }
626
627    if verbose {
628        let mut out = std::io::stdout().lock();
629        let _ = writeln!(out, "\nData store summary:");
630        if let Err(e) = data_store.summary(&mut out) {
631            warn!("Failed to write data store summary: {:?}", e);
632        }
633    }
634
635    if digests.len() > 1 {
636        println!(
637            "Replay run: tx_count={} success={} failure={} - exec_ms={}, total_ms={}",
638            total_metrics.tx_count,
639            total_metrics.success_count,
640            total_metrics.failure_count,
641            total_metrics.exec_ms,
642            total_metrics.total_ms
643        );
644    }
645
646    Ok(())
647}

649pub fn print_effects_or_fork<W: Write>(
650    digest: &str,
651    output_root: &Path,
652    show_effects: bool,
653    w: &mut W,
654) -> Result<()> {
655    let output_dir = output_root.join(digest);
656    let manager = ArtifactManager::new(&output_dir, false)?;
657    if manager.member(Artifact::ForkedTransactionEffects).exists() {
658        writeln!(w, "Transaction {digest} forked")?;
659        let forked_effects = manager
660            .member(Artifact::ForkedTransactionEffects)
661            .try_get_transaction_effects()
662            .transpose()?
663            .unwrap();
664        let expected_effects = manager
665            .member(Artifact::TransactionEffects)
666            .try_get_transaction_effects()
667            .transpose()?
668            .unwrap();
669        writeln!(
670            w,
671            "Forked Transaction Effects for {digest}\n{}",
672            diff_effects(&expected_effects, &forked_effects)
673        )?;
674    } else if show_effects {
675        let tx_effects = manager
676            .member(Artifact::TransactionEffects)
677            .try_get_transaction_effects()
678            .transpose()?
679            .unwrap();
680        writeln!(
681            w,
682            "{}",
683            SuiTransactionBlockEffects::try_from(tx_effects.clone())
684                .map_err(|e| anyhow!("Failed to convert effects: {e}"))?
685        )?;
686        manager
687            .member(Artifact::TransactionGasReport)
688            .try_get_gas_report()
689            .transpose()?
690            .map(|report| {
691                writeln!(
692                    w,
693                    "Transaction Gas Report for {digest}\n{}",
694                    Pretty(&report)
695                )
696                .unwrap()
697            })
698            .unwrap_or_else(|| {
699                writeln!(w, "No gas report available for transaction {digest}").unwrap();
700            });
701    }
702    Ok(())
703}

705/// Utility to diff `TransactionEffect` in a human readable format
706pub fn diff_effects(
707    expected_effect: &TransactionEffects,
708    txn_effects: &TransactionEffects,
709) -> String {
710    let expected = format!("{:#?}", expected_effect);
711    let result = format!("{:#?}", txn_effects);
712    let mut res = vec![];
713
714    let diff = TextDiff::from_lines(&expected, &result);
715    for change in diff.iter_all_changes() {
716        let sign = match change.tag() {
717            ChangeTag::Delete => "---",
718            ChangeTag::Insert => "+++",
719            ChangeTag::Equal => "   ",
720        };
721        res.push(format!("{}{}", sign, change));
722    }
