43// Arguments to the replay tool.
44// It allows to replay a single transaction by digest or
45// a file containing multiple digests, one per line.
46// This may evolve to something very different in time and
47// it's not meant to be stable.
48// The options available are very convenient for the current
49// development cycle.
50#[derive(Parser, Clone, Debug)]
51#[clap(
52    name = "Sui Replay Tool",
53    about = "Replay executed transactions.",
54    rename_all = "kebab-case"
55)]
56pub struct Config {
57    #[command(subcommand)]
58    pub command: Option<Command>,
59    #[command(flatten)]
60    pub replay_stable: ReplayConfigStable,
61    #[command(flatten)]
62    pub replay_experimental: ReplayConfigExperimental,
63}

248/// Load replay configuration from ~/.sui/sui_config/replay.toml file.
249/// Returns default config (all fields set to None) if file cannot be found or read.
250pub fn load_config_file() -> Result<ReplayConfigStable> {
251    let config_dir = match sui_config_dir() {
252        Ok(dir) => dir,
253        Err(e) => {
254            eprintln!("Cannot locate replay config file: {e}");
255            return Ok(ReplayConfigStable::default());
256        }
257    };
258    let config_file_path = config_dir.join(CONFIG_FILE_NAME);
259
260    if !config_file_path.exists() {
261        return Ok(ReplayConfigStable::default());
262    }
263
264    let content = fs::read_to_string(&config_file_path).map_err(|e| {
265        anyhow!(
266            "Failed to read replay config file '{:?}': {}",
267            config_file_path,
268            e
269        )
270    })?;
271
272    let config: TOMLConfig = toml::from_str(&content).map_err(|e| {
273        anyhow!(
274            "Failed to parse replay config file '{:?}': {}",
275            config_file_path,
276            e
277        )
278    })?;
279
280    config.flags.ok_or_else(|| {
281        anyhow!(
282            "No flags section found in replay config file '{:?}'",
283            config_file_path
284        )
285    })
286}

288/// Merge CLI flags and config file flags into a single config. CLI flags take
289/// precedence over config file flags, which take precedence over defaults.
290pub fn merge_configs(
291    cli_config: ReplayConfigStable,
292    file_config: ReplayConfigStable,
293) -> ReplayConfigStableInternal {
294    let default_config = ReplayConfigStableInternal::default();
295    ReplayConfigStableInternal {
296        digest: cli_config.digest.or(file_config.digest),
297
298        digests_path: cli_config.digests_path.or(file_config.digests_path),
299
300        terminate_early: cli_config
301            .terminate_early
302            .or(file_config.terminate_early)
303            .unwrap_or(default_config.terminate_early),
304
305        trace: cli_config
306            .trace
307            .or(file_config.trace)
308            .unwrap_or(default_config.trace),
309
310        output_dir: cli_config.output_dir.or(file_config.output_dir),
311
312        show_effects: cli_config
313            .show_effects
314            .or(file_config.show_effects)
315            .unwrap_or(default_config.show_effects),
316
317        overwrite: cli_config
318            .overwrite
319            .or(file_config.overwrite)
320            .unwrap_or(default_config.overwrite),
321    }
322}

324pub async fn handle_replay_config(
325    stable_config: &ReplayConfigStableInternal,
326    experimental_config: &ReplayConfigExperimental,
327    version: &str,
328) -> Result<PathBuf> {
329    let ReplayConfigStableInternal {
330        digest,
331        digests_path,
332        terminate_early,
333        trace,
334        output_dir,
335        show_effects: _, // used in the caller
336        overwrite: overwrite_existing,
337    } = &stable_config;
338    let mut terminate_early = *terminate_early;
339
340    let ReplayConfigExperimental {
341        node,
342        verbose,
343        store_mode,
344        track_time,
345        cache_executor,
346    } = experimental_config;
347
348    let output_root_dir = if let Some(dir) = output_dir {
349        dir.to_path_buf()
350    } else {
351        // Default output directory is `<cur_dir>/.replay/<digest>`
352        let current_dir =
353            std::env::current_dir().map_err(|e| anyhow!("Failed to get current directory: {e}"))?;
354        current_dir.join(DEFAULT_OUTPUT_DIR)
355    };
356
357    // If trying to trace but the binary was not built with the tracing feature flag raise an error.
358    #[cfg(not(feature = "tracing"))]
359    if *trace {
360        bail!(
361            "Tracing is not enabled in this build. Please rebuild with the \
362            `tracing` feature (`--features tracing`) to use tracing in replay"
363        );
364    }
365
366    // If a file is specified it is read and the digest ignored.
367    // Once we decide on the options we want this is likely to change.
368    let digests = if let Some(digests_path) = digests_path {
369        // read digests from file
370        std::fs::read_to_string(digests_path.clone())
371            .map_err(|e| {
372                anyhow!(
373                    "Failed to read digests file {}: {e}",
374                    digests_path.display(),
375                )
376            })?
377            .lines()
378            .map(|s| s.trim().to_string())
379            .collect::<Vec<_>>()
380    } else if let Some(tx_digest) = digest {
381        // terminate early if a single digest is provided this way we get proper error messages from
382        terminate_early = true;
383        // single digest provided
384        vec![tx_digest.clone()]
385    } else {
386        bail!("either --digest or --digests-path must be provided");
387    };
388
389    debug!("Binary version: {version}");
390
391    // Build the selected data store and run replay
392    match store_mode {
393        StoreMode::GqlOnly => {
394            let gql_store = DataStore::new(node.clone(), version)
395                .map_err(|e| anyhow!("Failed to create data store: {:?}", e))?;
396            run_replay(
397                &gql_store,
398                &output_root_dir,
399                &digests,
400                node,
401                *overwrite_existing,
402                *trace,
403                *verbose,
404                terminate_early,
405                *track_time,
406                *cache_executor,
407            )
408            .await?;
409        }
410        StoreMode::FsThenGql => {
411            let fs_store = FileSystemStore::new(node.clone())
412                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
413            let gql_store = DataStore::new(node.clone(), version)
414                .map_err(|e| anyhow!("Failed to create data store: {:?}", e))?;
415            let store = ReadThroughStore::new(fs_store, gql_store);
416            run_replay(
417                &store,
418                &output_root_dir,
419                &digests,
420                node,
421                *overwrite_existing,
422                *trace,
423                *verbose,
424                terminate_early,
425                *track_time,
426                *cache_executor,
427            )
428            .await?;
429        }
430        StoreMode::FsOnly => {
431            let fs_store = FileSystemStore::new(node.clone())
432                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
433            run_replay(
434                &fs_store,
435                &output_root_dir,
436                &digests,
437                node,
438                *overwrite_existing,
439                *trace,
440                *verbose,
441                terminate_early,
442                *track_time,
443                *cache_executor,
444            )
445            .await?;
446        }
447        StoreMode::InmemFs => {
448            let fs_store = FileSystemStore::new(node.clone())
449                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
450            let in_memory_store = InMemoryStore::new(node.clone());
451            let store = ReadThroughStore::new(in_memory_store, fs_store);
452            run_replay(
453                &store,
454                &output_root_dir,
455                &digests,
456                node,
457                *overwrite_existing,
458                *trace,
459                *verbose,
460                terminate_early,
461                *track_time,
462                *cache_executor,
463            )
464            .await?;
465        }
466        StoreMode::InmemFsGql => {
467            let fs_store = FileSystemStore::new(node.clone())
468                .map_err(|e| anyhow!("Failed to create file system store: {:?}", e))?;
469            let gql_store = DataStore::new(node.clone(), version)
470                .map_err(|e| anyhow!("Failed to create data store: {:?}", e))?;
471            let secondary_store = ReadThroughStore::new(fs_store, gql_store);
472            let in_memory_store = InMemoryStore::new(node.clone());
473            let store = ReadThroughStore::new(in_memory_store, secondary_store);
474            run_replay(
475                &store,
476                &output_root_dir,
477                &digests,
478                node,
479                *overwrite_existing,
480                *trace,
481                *verbose,
482                terminate_early,
483                *track_time,
484                *cache_executor,
485            )
486            .await?;
487        }
488    }
489
490    Ok(output_root_dir)
491}

493async fn run_replay<S>(
494    data_store: &S,
495    output_root_dir: &Path,
496    digests: &[String],
497    node: &Node,
498    overwrite_existing: bool,
499    trace: bool,
500    verbose: bool,
501    terminate_early: bool,
502    track_time: bool,
503    cache_executor: bool,
504) -> Result<()>
505where
506    S: ReadDataStore + StoreSummary + SetupStore,
507{
508    use crate::replay_txn::ExecutorProvider;
509    use std::time::Instant;
510
511    data_store.setup(None)?;
512    let mut total_metrics = TotalMetrics::new();
513    let mut executor_provider = ExecutorProvider::new(cache_executor);
514
515    let mp = MultiProgress::new();
516    let tx_spinner = mp.add(ProgressBar::new_spinner());
517    let progress_bar = mp.add(ProgressBar::new(digests.len() as u64));
518
519    tx_spinner.set_style(ProgressStyle::with_template("{spinner}: {msg}").unwrap());
520    tx_spinner.enable_steady_tick(Duration::from_millis(80));
521
522    for tx_digest in digests {
523        let tx_dir = output_root_dir.join(tx_digest);
524        let artifact_manager = ArtifactManager::new(&tx_dir, overwrite_existing)?;
525        let span = info_span!("replay", tx_digest = %tx_digest);
526
527        tx_spinner.set_message(format!("Executing transaction {}", tx_digest));
528
529        let tx_start = Instant::now();
530        let result = replay_transaction(
531            &artifact_manager,
532            tx_digest,
533            data_store,
534            node.network_name(),
535            trace,
536            &mut executor_provider,
537        )
538        .instrument(span)
539        .await;
540        let tx_total_ms = tx_start.elapsed().as_millis();
541
542        let success = result.is_ok();
543        let exec_ms = result.as_ref().ok().copied().unwrap_or(0);
544
545        total_metrics.add_transaction(success, tx_total_ms, exec_ms);
546
547        // Print per-transaction result
548        let status = if success { "OK" } else { "FAILED" };
549
550        let time_info = if track_time {
551            format!(
552                " ({}): exec_ms={}, total_ms={}",
553                status, exec_ms, tx_total_ms
554            )
555        } else {
556            "".to_owned()
557        };
558
559        tx_spinner.println(format!("Executed transaction {}{}", tx_digest, time_info));
560
561        match result {
562            Err(e) if terminate_early => {
563                error!(tx_digest = %tx_digest, error = ?e, "Replay error; terminating early");
564                bail!("Replay terminated due to error: {}", e);
565            }
566            Err(e) => {
567                error!(tx_digest = %tx_digest, error = ?e, "Replay failed");
568            }
569            Ok(_) => {}
570        }
571        progress_bar.inc(1);
572    }
573
574    tx_spinner.finish_and_clear();
575
576    if verbose {
577        let mut out = std::io::stdout().lock();
578        let _ = writeln!(out, "\nData store summary:");
579        if let Err(e) = data_store.summary(&mut out) {
580            warn!("Failed to write data store summary: {:?}", e);
581        }
582    }
583
584    if digests.len() > 1 {
585        println!(
586            "Replay run: tx_count={} success={} failure={} - exec_ms={}, total_ms={}",
587            total_metrics.tx_count,
588            total_metrics.success_count,
589            total_metrics.failure_count,
590            total_metrics.exec_ms,
591            total_metrics.total_ms
592        );
593    }
594
595    Ok(())
596}

598pub fn print_effects_or_fork<W: Write>(
599    digest: &str,
600    output_root: &Path,
601    show_effects: bool,
602    w: &mut W,
603) -> Result<()> {
604    let output_dir = output_root.join(digest);
605    let manager = ArtifactManager::new(&output_dir, false)?;
606    if manager.member(Artifact::ForkedTransactionEffects).exists() {
607        writeln!(w, "Transaction {digest} forked")?;
608        let forked_effects = manager
609            .member(Artifact::ForkedTransactionEffects)
610            .try_get_transaction_effects()
611            .transpose()?
612            .unwrap();
613        let expected_effects = manager
614            .member(Artifact::TransactionEffects)
615            .try_get_transaction_effects()
616            .transpose()?
617            .unwrap();
618        writeln!(
619            w,
620            "Forked Transaction Effects for {digest}\n{}",
621            diff_effects(&expected_effects, &forked_effects)
622        )?;
623    } else if show_effects {
624        let tx_effects = manager
625            .member(Artifact::TransactionEffects)
626            .try_get_transaction_effects()
627            .transpose()?
628            .unwrap();
629        writeln!(
630            w,
631            "{}",
632            SuiTransactionBlockEffects::try_from(tx_effects.clone())
633                .map_err(|e| anyhow!("Failed to convert effects: {e}"))?
634        )?;
635        manager
636            .member(Artifact::TransactionGasReport)
637            .try_get_gas_report()
638            .transpose()?
639            .map(|report| {
640                writeln!(
641                    w,
642                    "Transaction Gas Report for {digest}\n{}",
643                    Pretty(&report)
644                )
645                .unwrap()
646            })
647            .unwrap_or_else(|| {
648                writeln!(w, "No gas report available for transaction {digest}").unwrap();
649            });
650    }
651    Ok(())
652}

654/// Utility to diff `TransactionEffect` in a human readable format
655pub fn diff_effects(
656    expected_effect: &TransactionEffects,
657    txn_effects: &TransactionEffects,
658) -> String {
659    let expected = format!("{:#?}", expected_effect);
660    let result = format!("{:#?}", txn_effects);
661    let mut res = vec![];
662
663    let diff = TextDiff::from_lines(&expected, &result);
664    for change in diff.iter_all_changes() {
665        let sign = match change.tag() {
666            ChangeTag::Delete => "---",
667            ChangeTag::Insert => "+++",
668            ChangeTag::Equal => "   ",
669        };
670        res.push(format!("{}{}", sign, change));
671    }
672
673    res.join("")
674}
