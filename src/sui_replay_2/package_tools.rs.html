42    /// Extract the original package from the file system store
43    /// Returns the package object, and version number.
44    /// Expects exactly one version file to exist.
45    fn extract_original_package(&self) -> Result<(Object, u64)> {
46        let (_, version) = self.pkg_version_path()?;
47
48        // Use FileSystemStore to properly deserialize the object
49        let store = FileSystemStore::new(self.node.clone())?;
50        let key = ObjectKey {
51            object_id: self.package_id,
52            version_query: VersionQuery::Version(version),
53        };
54
55        let objects = store.get_objects(&[key])?;
56
57        // Unwraps are safe because we know there is exactly one package object
58        let (pkg, _) = objects.first().unwrap().clone().unwrap();
59
60        Ok((pkg, version))
61    }

63    fn pkg_dir_path(&self) -> Result<PathBuf> {
64        // Get chain ID from node_mapping.csv
65        let chain_id =
66            get_chain_id_from_mapping(&self.node).context("Failed to get chain ID for node")?;
67
68        Ok(FileSystemStore::base_path()?
69            .join(chain_id)
70            .join(OBJECTS_DIR)
71            .join(self.package_id.to_string()))
72    }

74    /// Get the implicit package version path and the version number itself.
75    /// This is the version corresponding to the name of the package
76    /// file in the cache.
77    fn pkg_version_path(&self) -> Result<(PathBuf, u64)> {
78        let pkg_dir = self.pkg_dir_path()?;
79
80        if !pkg_dir.exists() {
81            bail!(
82                "Package {} not found in cache at {:?}",
83                self.package_id,
84                pkg_dir
85            );
86        }
87
88        // Find numeric files in the directory (these are version files)
89        let mut version_files = Vec::new();
90        for entry in fs::read_dir(&pkg_dir)? {
91            let entry = entry?;
92            let filename = entry.file_name();
93            let Some(filename_str) = filename.to_str() else {
94                continue;
95            };
96
97            // Check if the filename is a number (version)
98            if let Ok(version) = filename_str.parse::<u64>() {
99                version_files.push((entry.path(), version));
100            }
101        }
102
103        if version_files.is_empty() {
104            bail!("No version file found in package directory {:?}", pkg_dir);
105        }
106
107        if version_files.len() > 1 {
108            bail!(
109                "Expected exactly one version file in package directory {:?}",
110                pkg_dir,
111            );
112        }
113        Ok(version_files.first().unwrap().clone())
114    }

116    /// Save a package object to a file
117    pub fn save_package_to_file(object: &Object, output_path: &PathBuf) -> Result<()> {
118        let bytes = bcs::to_bytes(object).context("Failed to serialize package object to BCS")?;
119
120        // Ensure parent directory exists
121        if let Some(parent) = output_path.parent() {
122            fs::create_dir_all(parent).context("Failed to create output directory")?;
123        }
124
125        fs::write(output_path, &bytes).context("Failed to write package binary to disk")?;
126
127        Ok(())
128    }

130    /// Replace the package in the cache with the given object
131    pub fn save_package_to_cache(
132        &self,
133        object: &Object,
134        version: SequenceNumber,
135    ) -> Result<PathBuf> {
136        let pkg_file = self.pkg_dir_path()?.join(version.value().to_string());
137
138        if !pkg_file.exists() {
139            bail!(
140                "Package {} version {} not found in cache",
141                self.package_id,
142                version
143            );
144        }
145
146        let bytes = bcs::to_bytes(object).context("Failed to serialize package object to BCS")?;
147
148        fs::write(&pkg_file, &bytes).context("Failed to write package to cache")?;
149
150        Ok(pkg_file)
151    }

162/// Read chain ID from node_mapping.csv file
163fn get_chain_id_from_mapping(node: &Node) -> Result<String> {
164    let mapping_file = FileSystemStore::base_path()?.join(NODE_MAPPING_FILE);
165
166    if !mapping_file.exists() {
167        bail!(
168            "Node mapping file not found at {:?}. Please ensure the replay data store is properly initialized.",
169            mapping_file
170        );
171    }
172
173    let file =
174        fs::File::open(&mapping_file).context(format!("Failed to open {}", NODE_MAPPING_FILE))?;
175
176    let mut rdr = csv::ReaderBuilder::new()
177        .has_headers(false)
178        .trim(csv::Trim::All)
179        .from_reader(file);
180
181    let node_name = node.network_name();
182
183    // Read the file and find the matching node
184    for result in rdr.records() {
185        let record = result.context(format!(
186            "Failed to read CSV record from {}",
187            NODE_MAPPING_FILE
188        ))?;
189
190        if record.len() != 2 {
191            bail!(
192                "Invalid format in {}. Expected 2 columns (node_name,chain_id), got {}",
193                NODE_MAPPING_FILE,
194                record.len()
195            );
196        }
197
198        let name = record[0].trim();
199        let chain_id = record[1].trim();
200
201        if name == node_name {
202            return Ok(chain_id.to_string());
203        }
204    }
205
206    bail!("Node '{}' not found in {}", node_name, NODE_MAPPING_FILE)
207}

225    /// Main entry point to rebuild a package
226    pub fn rebuild(&self) -> Result<()> {
227        // Step 1: Extract original package from file system store
228        let (original_object, extracted_version) = self.package_info.extract_original_package()?;
229
230        // Step 2: Get metadata from original package
231        let (original_package, tx_digest, version) =
232            self.extract_package_metadata(&original_object)?;
233
234        if extracted_version != version.value() {
235            bail!(
236                "Extracted version {} does not actual package match version {}",
237                extracted_version,
238                version
239            );
240        }
241
242        // Step 3: Compile the new source code with the original package ID
243        let compiled_modules = self.compile_package_with_id()?;
244
245        // Step 4: Reconstruct the package with new modules but original metadata
246        let rebuilt_package = self.rebuild_package(original_package, compiled_modules, version)?;
247
248        // Step 5: Create the object wrapper
249        let rebuilt_object = Object::new_from_package(rebuilt_package, tx_digest);
250
251        // Step 6: Serialize and save (either to file or cache)
252        match &self.output_path {
253            Some(output_path) => {
254                // Save to specified file
255                PackageInfo::save_package_to_file(&rebuilt_object, output_path)?;
256
257                // Step 7: Verify if source unchanged (for testing)
258                self.verify_rebuild(&original_object, &rebuilt_object)?;
259
260                println!(
261                    "Package rebuilt successfully and saved to: {:?}",
262                    output_path
263                );
264            }
265            None => {
266                // Replace in cache
267                let cache_path = self
268                    .package_info
269                    .save_package_to_cache(&rebuilt_object, version)?;
270
271                // Step 7: Verify if source unchanged (for testing)
272                self.verify_rebuild(&original_object, &rebuilt_object)?;
273
274                println!("Package rebuilt and updated in cache at: {:?}", cache_path);
275            }
276        }
277
278        Ok(())
279    }

281    /// Extract metadata from the original package object
282    fn extract_package_metadata(
283        &self,
284        object: &Object,
285    ) -> Result<(MovePackage, TransactionDigest, SequenceNumber)> {
286        let package = match &object.data {
287            Data::Package(pkg) => pkg.clone(),
288            _ => bail!("Object {} is not a package", self.package_info.package_id),
289        };
290
291        let tx_digest = object.previous_transaction;
292        let version = object.version();
293
294        Ok((package, tx_digest, version))
295    }

297    /// Compile the source package with the original package ID
298    fn compile_package_with_id(&self) -> Result<Vec<CompiledModule>> {
299        // Create build config (following build.rs pattern)
300        let config = MoveBuildConfig::default();
301
302        let envs = RootPackage::<SuiFlavor>::environments(&self.source_path)?;
303        let Some(env_id) = envs.get(&self.env) else {
304            todo!()
305        };
306        let environment = Environment {
307            name: self.env.clone(),
308            id: env_id.clone(),
309        };
310
311        // Create BuildConfig - simplified like in build.rs
312        let config = BuildConfig {
313            config,
314            run_bytecode_verifier: false, // We don't need verification for rebuilding
315            print_diags_to_stderr: true,  // Print diagnostics like build.rs does
316            environment,
317        };
318
319        // Build the package (same as build.rs does)
320        let compiled_package = config
321            .build(&self.source_path)
322            .context("Failed to build package")?;
323
324        // Get the compiled modules
325        let mut modules = compiled_package.get_modules().cloned().collect::<Vec<_>>();
326
327        // Update the package ID in the compiled modules to match the original
328        self.update_module_addresses(&mut modules, self.package_info.package_id)?;
329
330        Ok(modules)
331    }

333    /// Update the package address in compiled modules to match the target package ID
334    fn update_module_addresses(
335        &self,
336        modules: &mut [CompiledModule],
337        target_id: ObjectID,
338    ) -> Result<()> {
339        let target_address = AccountAddress::from(target_id);
340
341        for module in modules.iter_mut() {
342            // Get the self module handle to find which address index it uses
343            let self_handle_idx = module.self_module_handle_idx.0 as usize;
344            let self_handle = module
345                .module_handles
346                .get(self_handle_idx)
347                .ok_or_else(|| anyhow!("Invalid self module handle index"))?;
348
349            // Get the address index from the module handle
350            let address_idx = self_handle.address.0 as usize;
351
352            // Update the address in the address pool only if it's currently 0x0
353            if let Some(address) = module.address_identifiers.get_mut(address_idx) {
354                // Only replace if the current address is 0x0
355                if *address == AccountAddress::ZERO {
356                    *address = target_address;
357                }
358            } else {
359                bail!("Invalid address index {} in module", address_idx);
360            }
361        }
362
363        Ok(())
364    }

366    /// Build type origin table by merging existing origins with new ones
367    /// This follows the same logic as build_upgraded_type_origin_table in sui-types
368    fn build_merged_type_origin_table(
369        &self,
370        original_package: &MovePackage,
371        compiled_modules: &[CompiledModule],
372    ) -> Result<Vec<TypeOrigin>> {
373        let mut new_table = vec![];
374        let mut existing_table = original_package.type_origin_map();
375
376        // Process all types in the new compiled modules
377        for module in compiled_modules {
378            // Process struct definitions
379            for struct_def in module.struct_defs() {
380                let struct_handle = module.datatype_handle_at(struct_def.struct_handle);
381                let module_name = module.name().to_string();
382                let struct_name = module.identifier_at(struct_handle.name).to_string();
383                let type_key = (module_name.clone(), struct_name.clone());
384
385                // If type exists in original, preserve its origin; otherwise use current package ID
386                let origin_package = existing_table
387                    .remove(&type_key)
388                    .unwrap_or(self.package_info.package_id);
389                new_table.push(TypeOrigin {
390                    module_name,
391                    datatype_name: struct_name,
392                    package: origin_package,
393                });
394            }
395
396            // Process enum definitions
397            for enum_def in module.enum_defs() {
398                let enum_handle = module.datatype_handle_at(enum_def.enum_handle);
399                let module_name = module.name().to_string();
400                let enum_name = module.identifier_at(enum_handle.name).to_string();
401                let type_key = (module_name.clone(), enum_name.clone());
402
403                // If type exists in original, preserve its origin; otherwise use current package ID
404                let origin_package = existing_table
405                    .remove(&type_key)
406                    .unwrap_or(self.package_info.package_id);
407                new_table.push(TypeOrigin {
408                    module_name,
409                    datatype_name: enum_name,
410                    package: origin_package,
411                });
412            }
413        }
414
415        // Check if any types were removed (not allowed for compatibility)
416        if !existing_table.is_empty() {
417            let removed_types: Vec<String> = existing_table
418                .keys()
419                .map(|(module, datatype)| format!("{}::{}", module, datatype))
420                .collect();
421            bail!(
422                "Package rebuild would remove the following types, which breaks compatibility: {}",
423                removed_types.join(", ")
424            );
425        }
426
427        Ok(new_table)
428    }

430    /// Extract immediate dependencies from compiled modules
431    fn extract_immediate_dependencies(&self, modules: &[CompiledModule]) -> BTreeSet<ObjectID> {
432        let mut immediate_deps = BTreeSet::new();
433
434        for module in modules {
435            // Get all immediate dependencies (excluding self)
436            for dep in module.immediate_dependencies() {
437                let dep_id = ObjectID::from(*dep.address());
438                // Don't include self-references
439                if dep_id != self.package_info.package_id {
440                    immediate_deps.insert(dep_id);
441                }
442            }
443        }
444
445        immediate_deps
446    }

448    /// Compare dependencies between new modules and original package
449    fn compare_dependencies(
450        &self,
451        new_modules: &[CompiledModule],
452        original_package: &MovePackage,
453    ) -> Result<()> {
454        // Get immediate dependencies from new modules
455        let new_deps = self.extract_immediate_dependencies(new_modules);
456
457        // Get immediate dependencies from original package modules
458        let mut original_deps = BTreeSet::new();
459        for module_bytes in original_package.serialized_module_map().values() {
460            // Deserialize original module to extract dependencies
461            let original_module = CompiledModule::deserialize_with_defaults(module_bytes)
462                .context("Failed to deserialize original package module")?;
463
464            for dep in original_module.immediate_dependencies() {
465                let dep_id = ObjectID::from(*dep.address());
466                // Don't include self-references
467                if dep_id != self.package_info.package_id {
468                    original_deps.insert(dep_id);
469                }
470            }
471        }
472
473        // Compare dependency sets
474        let added_deps: Vec<_> = new_deps.difference(&original_deps).collect();
475        let removed_deps: Vec<_> = original_deps.difference(&new_deps).collect();
476
477        if !added_deps.is_empty() || !removed_deps.is_empty() {
478            let mut error_msg = String::from(
479                "Dependencies have changed, which is not supported for package rebuild:\n",
480            );
481
482            if !added_deps.is_empty() {
483                error_msg.push_str(&format!(
484                    "  Added dependencies: {}\n",
485                    added_deps
486                        .iter()
487                        .map(|d| format!("0x{}", d))
488                        .collect::<Vec<_>>()
489                        .join(", ")
490                ));
491            }
492
493            if !removed_deps.is_empty() {
494                error_msg.push_str(&format!(
495                    "  Removed dependencies: {}\n",
496                    removed_deps
497                        .iter()
498                        .map(|d| format!("0x{}", d))
499                        .collect::<Vec<_>>()
500                        .join(", ")
501                ));
502            }
503
504            error_msg.push_str("Package rebuild requires identical dependencies. ");
505
506            bail!(error_msg);
507        }
508
509        Ok(())
510    }

512    /// Build linkage table by validating dependencies and reusing original
513    fn build_new_linkage_table(
514        &self,
515        modules: &[CompiledModule],
516        original_package: &MovePackage,
517    ) -> Result<BTreeMap<ObjectID, UpgradeInfo>> {
518        // At this point we assume that dependencies would not change
519        self.compare_dependencies(modules, original_package)
520            .context("Dependency validation failed")?;
521
522        // If dependencies are identical, it's safe to reuse the original linkage table
523        let original_linkage = original_package.linkage_table().clone();
524
525        if !original_linkage.is_empty() {
526            println!(
527                "Reusing original linkage table with {} entries",
528                original_linkage.len()
529            );
530        }
531
532        Ok(original_linkage)
533    }

535    /// Rebuild the package with new modules but preserving original metadata
536    fn rebuild_package(
537        &self,
538        original_package: MovePackage,
539        compiled_modules: Vec<CompiledModule>,
540        version: SequenceNumber,
541    ) -> Result<MovePackage> {
542        // Convert compiled modules to the format needed by MovePackage
543        let mut module_map = BTreeMap::new();
544        for module in &compiled_modules {
545            let module_name = module.name().to_string();
546            // Use the proper serialization method for CompiledModule with version
547            let mut module_bytes = Vec::new();
548            module
549                .serialize_with_version(module.version, &mut module_bytes)
550                .context("Failed to serialize module")?;
551            module_map.insert(module_name, module_bytes);
552        }
553
554        // Build type origin table by merging existing origins with new ones
555        let type_origin_table = self
556            .build_merged_type_origin_table(&original_package, &compiled_modules)
557            .context("Failed to build type origin table")?;
558
559        // Build linkage table for dependencies
560        let linkage_table = self
561            .build_new_linkage_table(&compiled_modules, &original_package)
562            .context("Failed to build linkage table")?;
563
564        // Get protocol config for max package size
565        let protocol_config = ProtocolConfig::get_for_max_version_UNSAFE();
566        let max_package_size = protocol_config.max_move_package_size();
567
568        // Create a new MovePackage with updated modules and properly generated tables
569        let rebuilt_package = MovePackage::new(
570            self.package_info.package_id, // Use the package ID directly
571            version,
572            module_map,
573            max_package_size,
574            type_origin_table,
575            linkage_table,
576        )
577        .context("Failed to create new MovePackage")?;
578
579        Ok(rebuilt_package)
580    }

582    /// Verify that the rebuilt package matches the original (when source unchanged)
583    fn verify_rebuild(&self, original: &Object, rebuilt: &Object) -> Result<()> {
584        let original_bytes = bcs::to_bytes(original)?;
585        let rebuilt_bytes = bcs::to_bytes(rebuilt)?;
586
587        if original_bytes == rebuilt_bytes {
588            println!(" Verification passed: Rebuilt package matches the original");
589        } else {
590            println!(" Warning: Rebuilt package differs from original");
591            println!("  Original size: {} bytes", original_bytes.len());
592            println!("  Rebuilt size: {} bytes", rebuilt_bytes.len());
593
594            // Extract and compare type origin tables for debugging
595            if let (Data::Package(orig_pkg), Data::Package(rebuilt_pkg)) =
596                (&original.data, &rebuilt.data)
597            {
598                println!(
599                    "  Original type origins: {} entries",
600                    orig_pkg.type_origin_table().len()
601                );
602                println!(
603                    "  Rebuilt type origins: {} entries",
604                    rebuilt_pkg.type_origin_table().len()
605                );
606                println!(
607                    "  Original linkage table: {} entries",
608                    orig_pkg.linkage_table().len()
609                );
610                println!(
611                    "  Rebuilt linkage table: {} entries",
612                    rebuilt_pkg.linkage_table().len()
613                );
614            }
615        }
616
617        Ok(())
618    }

633/// Entry point for the extract-package command
634pub fn extract_package(node: Node, package_id: ObjectID, output_path: PathBuf) -> Result<()> {
635    // Create PackageInfo to handle extraction
636    let package_info = PackageInfo::new(node, package_id);
637
638    // Extract the package using the shared implementation
639    let (object, version) = package_info
640        .extract_original_package()
641        .context("Failed to extract package from cache")?;
642
643    // Save extracted package to the specified output path
644    PackageInfo::save_package_to_file(&object, &output_path)
645        .context("Failed to save extracted package")?;
646
647    println!(
648        "Successfully extracted package {} (version {}) to {}",
649        package_id,
650        version,
651        output_path.display()
652    );
653
654    Ok(())
655}

657/// Entry point for the overwrite-package command
658pub fn overwrite_package(node: Node, package_id: ObjectID, package_path: PathBuf) -> Result<()> {
659    // Read and validate the package file
660    let package_bytes = fs::read(&package_path)
661        .context(format!("Failed to read package file: {:?}", package_path))?;
662
663    // Validate it's a valid BCS-encoded Object
664    let _object: Object = bcs::from_bytes(&package_bytes)
665        .context("Invalid package file: not a valid BCS-encoded Object")?;
666
667    // Create PackageInfo to handle path resolution
668    let package_info = PackageInfo::new(node, package_id);
669
670    // Get the implicit package version path and version number
671    let (version_path, _) = package_info.pkg_version_path()?;
672
673    // Write the package to the target path
674    fs::write(&version_path, &package_bytes).context(format!(
675        "Failed to write package to cache: {:?}",
676        version_path
677    ))?;
678
679    println!(
680        "Package {} successfully overwritten in cache at: {:?}",
681        package_id, version_path
682    );
683
684    Ok(())
