39    /// Extract the original package from the file system store
40    /// Returns the package object, and version number.
41    /// Expects exactly one version file to exist.
42    fn extract_original_package(&self) -> Result<(Object, u64)> {
43        let (_, version) = self.pkg_version_path()?;
44
45        // Use FileSystemStore to properly deserialize the object
46        let store = FileSystemStore::new(self.node.clone())?;
47        let key = ObjectKey {
48            object_id: self.package_id,
49            version_query: VersionQuery::Version(version),
50        };
51
52        let objects = store.get_objects(&[key])?;
53
54        // Unwraps are safe because we know there is exactly one package object
55        let (pkg, _) = objects.first().unwrap().clone().unwrap();
56
57        Ok((pkg, version))
58    }
59
60    fn pkg_dir_path(&self) -> Result<PathBuf> {
61        // Get chain ID from node_mapping.csv
62        let chain_id =
63            get_chain_id_from_mapping(&self.node).context("Failed to get chain ID for node")?;
64
65        Ok(FileSystemStore::base_path()?
66            .join(chain_id)
67            .join(OBJECTS_DIR)
68            .join(self.package_id.to_string()))
69    }

71    /// Get the implicit package version path and the version number itself.
72    /// This is the version corresponding to the name of the package
73    /// file in the cache.
74    fn pkg_version_path(&self) -> Result<(PathBuf, u64)> {
75        let pkg_dir = self.pkg_dir_path()?;
76
77        if !pkg_dir.exists() {
78            bail!(
79                "Package {} not found in cache at {:?}",
80                self.package_id,
81                pkg_dir
82            );
83        }
84
85        // Find numeric files in the directory (these are version files)
86        let mut version_files = Vec::new();
87        for entry in fs::read_dir(&pkg_dir)? {
88            let entry = entry?;
89            let filename = entry.file_name();
90            let Some(filename_str) = filename.to_str() else {
91                continue;
92            };
93
94            // Check if the filename is a number (version)
95            if let Ok(version) = filename_str.parse::<u64>() {
96                version_files.push((entry.path(), version));
97            }
98        }
99
100        if version_files.is_empty() {
101            bail!("No version file found in package directory {:?}", pkg_dir);
102        }
103
104        if version_files.len() > 1 {
105            bail!(
106                "Expected exactly one version file in package directory {:?}",
107                pkg_dir,
108            );
109        }
110        Ok(version_files.first().unwrap().clone())
111    }

113    /// Save a package object to a file
114    pub fn save_package_to_file(object: &Object, output_path: &PathBuf) -> Result<()> {
115        let bytes = bcs::to_bytes(object).context("Failed to serialize package object to BCS")?;
116
117        // Ensure parent directory exists
118        if let Some(parent) = output_path.parent() {
119            fs::create_dir_all(parent).context("Failed to create output directory")?;
120        }
121
122        fs::write(output_path, &bytes).context("Failed to write package binary to disk")?;
123
124        Ok(())
125    }

127    /// Replace the package in the cache with the given object
128    pub fn save_package_to_cache(
129        &self,
130        object: &Object,
131        version: SequenceNumber,
132    ) -> Result<PathBuf> {
133        let pkg_file = self.pkg_dir_path()?.join(version.value().to_string());
134
135        if !pkg_file.exists() {
136            bail!(
137                "Package {} version {} not found in cache",
138                self.package_id,
139                version
140            );
141        }
142
143        let bytes = bcs::to_bytes(object).context("Failed to serialize package object to BCS")?;
144
145        fs::write(&pkg_file, &bytes).context("Failed to write package to cache")?;
146
147        Ok(pkg_file)
148    }

158/// Read chain ID from node_mapping.csv file
159fn get_chain_id_from_mapping(node: &Node) -> Result<String> {
160    let mapping_file = FileSystemStore::base_path()?.join(NODE_MAPPING_FILE);
161
162    if !mapping_file.exists() {
163        bail!(
164            "Node mapping file not found at {:?}. Please ensure the replay data store is properly initialized.",
165            mapping_file
166        );
167    }
168
169    let file =
170        fs::File::open(&mapping_file).context(format!("Failed to open {}", NODE_MAPPING_FILE))?;
171
172    let mut rdr = csv::ReaderBuilder::new()
173        .has_headers(false)
174        .trim(csv::Trim::All)
175        .from_reader(file);
176
177    let node_name = node.network_name();
178
179    // Read the file and find the matching node
180    for result in rdr.records() {
181        let record = result.context(format!(
182            "Failed to read CSV record from {}",
183            NODE_MAPPING_FILE
184        ))?;
185
186        if record.len() != 2 {
187            bail!(
188                "Invalid format in {}. Expected 2 columns (node_name,chain_id), got {}",
189                NODE_MAPPING_FILE,
190                record.len()
191            );
192        }
193
194        let name = record[0].trim();
195        let chain_id = record[1].trim();
196
197        if name == node_name {
198            return Ok(chain_id.to_string());
199        }
200    }
201
202    bail!("Node '{}' not found in {}", node_name, NODE_MAPPING_FILE)
203}

219    /// Main entry point to rebuild a package
220    pub fn rebuild(&self) -> Result<()> {
221        // Step 1: Extract original package from file system store
222        let (original_object, extracted_version) = self.package_info.extract_original_package()?;
223
224        // Step 2: Get metadata from original package
225        let (original_package, tx_digest, version) =
226            self.extract_package_metadata(&original_object)?;
227
228        if extracted_version != version.value() {
229            bail!(
230                "Extracted version {} does not actual package match version {}",
231                extracted_version,
232                version
233            );
234        }
235
236        // Step 3: Compile the new source code with the original package ID
237        let compiled_modules = self.compile_package_with_id()?;
238
239        // Step 4: Reconstruct the package with new modules but original metadata
240        let rebuilt_package = self.rebuild_package(original_package, compiled_modules, version)?;
241
242        // Step 5: Create the object wrapper
243        let rebuilt_object = Object::new_from_package(rebuilt_package, tx_digest);
244
245        // Step 6: Serialize and save (either to file or cache)
246        match &self.output_path {
247            Some(output_path) => {
248                // Save to specified file
249                PackageInfo::save_package_to_file(&rebuilt_object, output_path)?;
250
251                // Step 7: Verify if source unchanged (for testing)
252                self.verify_rebuild(&original_object, &rebuilt_object)?;
253
254                println!(
255                    "Package rebuilt successfully and saved to: {:?}",
256                    output_path
257                );
258            }
259            None => {
260                // Replace in cache
261                let cache_path = self
262                    .package_info
263                    .save_package_to_cache(&rebuilt_object, version)?;
264
265                // Step 7: Verify if source unchanged (for testing)
266                self.verify_rebuild(&original_object, &rebuilt_object)?;
267
268                println!("Package rebuilt and updated in cache at: {:?}", cache_path);
269            }
270        }
271
272        Ok(())
273    }

275    /// Extract metadata from the original package object
276    fn extract_package_metadata(
277        &self,
278        object: &Object,
279    ) -> Result<(MovePackage, TransactionDigest, SequenceNumber)> {
280        let package = match &object.data {
281            Data::Package(pkg) => pkg.clone(),
282            _ => bail!("Object {} is not a package", self.package_info.package_id),
283        };
284
285        let tx_digest = object.previous_transaction;
286        let version = object.version();
287
288        Ok((package, tx_digest, version))
289    }

291    /// Compile the source package with the original package ID
292    fn compile_package_with_id(&self) -> Result<Vec<CompiledModule>> {
293        // Register SuiPackageHooks to recognize Sui-specific manifest fields (matching sui move build)
294        // This prevents warnings about "published-at" and "version" fields in dependency Move.toml files
295        move_package::package_hooks::register_package_hooks(Box::new(SuiPackageHooks));
296
297        // Create build config (following build.rs pattern)
298        let build_config = MoveBuildConfig::default();
299
300        // Resolve lock file path to respect Move.lock
301        let mut config = resolve_lock_file_path(build_config, Some(&self.source_path))
302            .context("Failed to resolve lock file path")?;
303
304        // Set implicit dependencies after resolving lock file (matching build.rs)
305        config.implicit_dependencies = implicit_deps(latest_system_packages());
306
307        // Create BuildConfig - simplified like in build.rs
308        // We use chain_id from node for better dependency resolution
309        let chain_id = self.get_chain_id();
310        let config = BuildConfig {
311            config,
312            run_bytecode_verifier: false, // We don't need verification for rebuilding
313            print_diags_to_stderr: true,  // Print diagnostics like build.rs does
314            chain_id,
315        };
316
317        // Build the package (same as build.rs does)
318        let compiled_package = config
319            .build(&self.source_path)
320            .context("Failed to build package")?;
321
322        // Get the compiled modules
323        let mut modules = compiled_package.get_modules().cloned().collect::<Vec<_>>();
324
325        // Update the package ID in the compiled modules to match the original
326        self.update_module_addresses(&mut modules, self.package_info.package_id)?;
327
328        Ok(modules)
329    }

331    /// Update the package address in compiled modules to match the target package ID
332    fn update_module_addresses(
333        &self,
334        modules: &mut [CompiledModule],
335        target_id: ObjectID,
336    ) -> Result<()> {
337        let target_address = AccountAddress::from(target_id);
338
339        for module in modules.iter_mut() {
340            // Get the self module handle to find which address index it uses
341            let self_handle_idx = module.self_module_handle_idx.0 as usize;
342            let self_handle = module
343                .module_handles
344                .get(self_handle_idx)
345                .ok_or_else(|| anyhow!("Invalid self module handle index"))?;
346
347            // Get the address index from the module handle
348            let address_idx = self_handle.address.0 as usize;
349
350            // Update the address in the address pool only if it's currently 0x0
351            if let Some(address) = module.address_identifiers.get_mut(address_idx) {
352                // Only replace if the current address is 0x0
353                if *address == AccountAddress::ZERO {
354                    *address = target_address;
355                }
356            } else {
357                bail!("Invalid address index {} in module", address_idx);
358            }
359        }
360
361        Ok(())
362    }

364    /// Build type origin table by merging existing origins with new ones
365    /// This follows the same logic as build_upgraded_type_origin_table in sui-types
366    fn build_merged_type_origin_table(
367        &self,
368        original_package: &MovePackage,
369        compiled_modules: &[CompiledModule],
370    ) -> Result<Vec<TypeOrigin>> {
371        let mut new_table = vec![];
372        let mut existing_table = original_package.type_origin_map();
373
374        // Process all types in the new compiled modules
375        for module in compiled_modules {
376            // Process struct definitions
377            for struct_def in module.struct_defs() {
378                let struct_handle = module.datatype_handle_at(struct_def.struct_handle);
379                let module_name = module.name().to_string();
380                let struct_name = module.identifier_at(struct_handle.name).to_string();
381                let type_key = (module_name.clone(), struct_name.clone());
382
383                // If type exists in original, preserve its origin; otherwise use current package ID
384                let origin_package = existing_table
385                    .remove(&type_key)
386                    .unwrap_or(self.package_info.package_id);
387                new_table.push(TypeOrigin {
388                    module_name,
389                    datatype_name: struct_name,
390                    package: origin_package,
391                });
392            }
393
394            // Process enum definitions
395            for enum_def in module.enum_defs() {
396                let enum_handle = module.datatype_handle_at(enum_def.enum_handle);
397                let module_name = module.name().to_string();
398                let enum_name = module.identifier_at(enum_handle.name).to_string();
399                let type_key = (module_name.clone(), enum_name.clone());
400
401                // If type exists in original, preserve its origin; otherwise use current package ID
402                let origin_package = existing_table
403                    .remove(&type_key)
404                    .unwrap_or(self.package_info.package_id);
405                new_table.push(TypeOrigin {
406                    module_name,
407                    datatype_name: enum_name,
408                    package: origin_package,
409                });
410            }
411        }
412
413        // Check if any types were removed (not allowed for compatibility)
414        if !existing_table.is_empty() {
415            let removed_types: Vec<String> = existing_table
416                .keys()
417                .map(|(module, datatype)| format!("{}::{}", module, datatype))
418                .collect();
419            bail!(
420                "Package rebuild would remove the following types, which breaks compatibility: {}",
421                removed_types.join(", ")
422            );
423        }
424
425        Ok(new_table)
426    }

428    /// Extract immediate dependencies from compiled modules
429    fn extract_immediate_dependencies(&self, modules: &[CompiledModule]) -> BTreeSet<ObjectID> {
430        let mut immediate_deps = BTreeSet::new();
431
432        for module in modules {
433            // Get all immediate dependencies (excluding self)
434            for dep in module.immediate_dependencies() {
435                let dep_id = ObjectID::from(*dep.address());
436                // Don't include self-references
437                if dep_id != self.package_info.package_id {
438                    immediate_deps.insert(dep_id);
439                }
440            }
441        }
442
443        immediate_deps
444    }

446    /// Compare dependencies between new modules and original package
447    fn compare_dependencies(
448        &self,
449        new_modules: &[CompiledModule],
450        original_package: &MovePackage,
451    ) -> Result<()> {
452        // Get immediate dependencies from new modules
453        let new_deps = self.extract_immediate_dependencies(new_modules);
454
455        // Get immediate dependencies from original package modules
456        let mut original_deps = BTreeSet::new();
457        for module_bytes in original_package.serialized_module_map().values() {
458            // Deserialize original module to extract dependencies
459            let original_module = CompiledModule::deserialize_with_defaults(module_bytes)
460                .context("Failed to deserialize original package module")?;
461
462            for dep in original_module.immediate_dependencies() {
463                let dep_id = ObjectID::from(*dep.address());
464                // Don't include self-references
465                if dep_id != self.package_info.package_id {
466                    original_deps.insert(dep_id);
467                }
468            }
469        }
470
471        // Compare dependency sets
472        let added_deps: Vec<_> = new_deps.difference(&original_deps).collect();
473        let removed_deps: Vec<_> = original_deps.difference(&new_deps).collect();
474
475        if !added_deps.is_empty() || !removed_deps.is_empty() {
476            let mut error_msg = String::from(
477                "Dependencies have changed, which is not supported for package rebuild:\n",
478            );
479
480            if !added_deps.is_empty() {
481                error_msg.push_str(&format!(
482                    "  Added dependencies: {}\n",
483                    added_deps
484                        .iter()
485                        .map(|d| format!("0x{}", d))
486                        .collect::<Vec<_>>()
487                        .join(", ")
488                ));
489            }
490
491            if !removed_deps.is_empty() {
492                error_msg.push_str(&format!(
493                    "  Removed dependencies: {}\n",
494                    removed_deps
495                        .iter()
496                        .map(|d| format!("0x{}", d))
497                        .collect::<Vec<_>>()
498                        .join(", ")
499                ));
500            }
501
502            error_msg.push_str("Package rebuild requires identical dependencies. ");
503
504            bail!(error_msg);
505        }
506
507        Ok(())
508    }

510    /// Build linkage table by validating dependencies and reusing original
511    fn build_new_linkage_table(
512        &self,
513        modules: &[CompiledModule],
514        original_package: &MovePackage,
515    ) -> Result<BTreeMap<ObjectID, UpgradeInfo>> {
516        // At this point we assume that dependencies would not change
517        self.compare_dependencies(modules, original_package)
518            .context("Dependency validation failed")?;
519
520        // If dependencies are identical, it's safe to reuse the original linkage table
521        let original_linkage = original_package.linkage_table().clone();
522
523        if !original_linkage.is_empty() {
524            println!(
525                "Reusing original linkage table with {} entries",
526                original_linkage.len()
527            );
528        }
529
530        Ok(original_linkage)
531    }

533    /// Rebuild the package with new modules but preserving original metadata
534    fn rebuild_package(
535        &self,
536        original_package: MovePackage,
537        compiled_modules: Vec<CompiledModule>,
538        version: SequenceNumber,
539    ) -> Result<MovePackage> {
540        // Convert compiled modules to the format needed by MovePackage
541        let mut module_map = BTreeMap::new();
542        for module in &compiled_modules {
543            let module_name = module.name().to_string();
544            // Use the proper serialization method for CompiledModule with version
545            let mut module_bytes = Vec::new();
546            module
547                .serialize_with_version(module.version, &mut module_bytes)
548                .context("Failed to serialize module")?;
549            module_map.insert(module_name, module_bytes);
550        }
551
552        // Build type origin table by merging existing origins with new ones
553        let type_origin_table = self
554            .build_merged_type_origin_table(&original_package, &compiled_modules)
555            .context("Failed to build type origin table")?;
556
557        // Build linkage table for dependencies
558        let linkage_table = self
559            .build_new_linkage_table(&compiled_modules, &original_package)
560            .context("Failed to build linkage table")?;
561
562        // Get protocol config for max package size
563        let protocol_config = ProtocolConfig::get_for_max_version_UNSAFE();
564        let max_package_size = protocol_config.max_move_package_size();
565
566        // Create a new MovePackage with updated modules and properly generated tables
567        let rebuilt_package = MovePackage::new(
568            self.package_info.package_id, // Use the package ID directly
569            version,
570            module_map,
571            max_package_size,
572            type_origin_table,
573            linkage_table,
574        )
575        .context("Failed to create new MovePackage")?;
576
577        Ok(rebuilt_package)
578    }

586    /// Verify that the rebuilt package matches the original (when source unchanged)
587    fn verify_rebuild(&self, original: &Object, rebuilt: &Object) -> Result<()> {
588        let original_bytes = bcs::to_bytes(original)?;
589        let rebuilt_bytes = bcs::to_bytes(rebuilt)?;
590
591        if original_bytes == rebuilt_bytes {
592            println!("✓ Verification passed: Rebuilt package matches the original");
593        } else {
594            println!("⚠ Warning: Rebuilt package differs from original");
595            println!("  Original size: {} bytes", original_bytes.len());
596            println!("  Rebuilt size: {} bytes", rebuilt_bytes.len());
597
598            // Extract and compare type origin tables for debugging
599            if let (Data::Package(orig_pkg), Data::Package(rebuilt_pkg)) =
600                (&original.data, &rebuilt.data)
601            {
602                println!(
603                    "  Original type origins: {} entries",
604                    orig_pkg.type_origin_table().len()
605                );
606                println!(
607                    "  Rebuilt type origins: {} entries",
608                    rebuilt_pkg.type_origin_table().len()
609                );
610                println!(
611                    "  Original linkage table: {} entries",
612                    orig_pkg.linkage_table().len()
613                );
614                println!(
615                    "  Rebuilt linkage table: {} entries",
616                    rebuilt_pkg.linkage_table().len()
617                );
618            }
619        }
620
621        Ok(())
622    }

625/// Entry point for the package rebuild command
626pub fn rebuild_package(
627    node: Node,
628    package_id: ObjectID,
629    source_path: PathBuf,
630    output_path: Option<PathBuf>,
631) -> Result<()> {
632    let rebuilder = PackageRebuilder::new(node, package_id, source_path, output_path);
633    rebuilder.rebuild()
634}

636/// Entry point for the extract-package command
637pub fn extract_package(node: Node, package_id: ObjectID, output_path: PathBuf) -> Result<()> {
638    // Create PackageInfo to handle extraction
639    let package_info = PackageInfo::new(node, package_id);
640
641    // Extract the package using the shared implementation
642    let (object, version) = package_info
643        .extract_original_package()
644        .context("Failed to extract package from cache")?;
645
646    // Save extracted package to the specified output path
647    PackageInfo::save_package_to_file(&object, &output_path)
648        .context("Failed to save extracted package")?;
649
650    println!(
651        "Successfully extracted package {} (version {}) to {}",
652        package_id,
653        version,
654        output_path.display()
655    );
656
657    Ok(())
658}

660/// Entry point for the overwrite-package command
661pub fn overwrite_package(node: Node, package_id: ObjectID, package_path: PathBuf) -> Result<()> {
662    // Read and validate the package file
663    let package_bytes = fs::read(&package_path)
664        .context(format!("Failed to read package file: {:?}", package_path))?;
665
666    // Validate it's a valid BCS-encoded Object
667    let _object: Object = bcs::from_bytes(&package_bytes)
668        .context("Invalid package file: not a valid BCS-encoded Object")?;
669
670    // Create PackageInfo to handle path resolution
671    let package_info = PackageInfo::new(node, package_id);
672
673    // Get the implicit package version path and version number
674    let (version_path, _) = package_info.pkg_version_path()?;
675
676    // Write the package to the target path
677    fs::write(&version_path, &package_bytes).context(format!(
678        "Failed to write package to cache: {:?}",
679        version_path
680    ))?;
681
682    println!(
683        "Package {} successfully overwritten in cache at: {:?}",
684        package_id, version_path
685    );
686
687    Ok(())
