4//! Logical stores needed by the replay tool.
5//! Those stores are loosely modeled after the GQL schema in
6//! `crates/sui-indexer-alt-graphql/schema.graphql`.
7//! A `TransactionStore` is used to retrieve transaction data and effects by digest.
8//! An `EpochStore` is used to retrieve epoch information and protocol configuration.
9//! An `ObjectStore` is used to retrieve objects by their keys, with different query options.
10//!
11//! Data is usually retrieved by getting BCS-encoded data rather than navigating the
12//! GQL schema.
13//! Essentially the code uses the schema to retrieve the data, deserializes it into runtime
14//! structures, and then operates on those.
15//!
16//! A `DataStore` with reasonable defaults is provided for convenience (`data_store.rs`).
17//! Other styles of data stores are also provided in `data_stores` for different use cases.
18
19use anyhow::{Error, Result};
20use std::io::Write;
21use sui_types::{
22    base_types::ObjectID, effects::TransactionEffects, object::Object,
23    supported_protocol_versions::ProtocolConfig, transaction::TransactionData,
24};

30/// Transaction data with effects and checkpoint required to replay a transaction.
31#[derive(Clone, Debug)]
32pub struct TransactionInfo {
33    pub data: TransactionData,
34    pub effects: TransactionEffects,
35    pub checkpoint: u64,
36}
37
38/// A `TransactionStore` has to be able to retrieve transaction data for a given digest.
39/// To replay a transaction the data provided to
40/// `sui_execution::executor::Executor::execute_transaction_to_effects` must be available.
41/// Some of that data is not provided by the user. It is naturally available at runtime on a
42/// live system and later saved in effects and in the context of a checkpoint.
43pub trait TransactionStore {
44    /// Given a transaction digest, return transaction info including data, effects,
45    /// and the checkpoint that transaction was executed in.
46    /// Returns `None` if the transaction is not found.
47    fn transaction_data_and_effects(
48        &self,
49        tx_digest: &str,
50    ) -> Result<Option<TransactionInfo>, Error>;
51}
52
53/// Epoch data required to reaplay a transaction.
54#[derive(Clone, Debug)]
55pub struct EpochData {
56    pub epoch_id: u64,
57    pub protocol_version: u64,
58    pub rgp: u64,
59    pub start_timestamp: u64,
60}
61
62/// An `EpochStore` retrieves the epoch data and protocol configuration
63/// for a given epoch.
64/// Epoch data is collected by an indexer and it is not stored anywhere otherwise.
65/// This is a very small amount of information and could conceivably be saved locally
66/// and never hit a server.
67pub trait EpochStore {
68    /// Return the `EpochData` for a given epoch.
69    fn epoch_info(&self, epoch: u64) -> Result<Option<EpochData>, Error>;
70    /// Return the `ProtocolConfig` for a given epoch.
71    fn protocol_config(&self, epoch: u64) -> Result<Option<ProtocolConfig>, Error>;
72}
73
74/// Query for an object.
75/// Specifies an `ObjectID` and the "rule" to retrieve it.
76#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
77pub struct ObjectKey {
78    pub object_id: ObjectID,
79    pub version_query: VersionQuery,
80}
81
82/// Query options for an object.
83/// `Version` request an object at a specific version
84/// `RootVersion` request an object at a given version at most (<=)
85/// `AtCheckpoint` request an object at a given checkpoint. Useful for unknown `Version`.
86#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
87pub enum VersionQuery {
88    Version(u64),
89    RootVersion(u64),
90    AtCheckpoint(u64),
91}
92
93/// The `ObjectStore` trait is used to retrieve objects by their keys,
94/// with different query options.
95///
96/// This trait can execute a subset of what is allowed by
97/// `crates/sui-indexer-alt-graphql/schema.graphql::multiGetObjects`.
98/// That query likely allows more than what the replay tool needs, which is fairly limited in
99/// its usage.
100pub trait ObjectStore {
101    /// Retrieve objects by their keys, with different query options.
102    ///
103    /// If the object is not found, the element in the vector is `None`.
104    /// Otherwise each tuple contains:
105    /// - `Object`: The object data
106    /// - `u64`: The actual version of the object
107    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error>;
108}

110// ============================================================================
111// Set up trait
112// ============================================================================
113
114// This is a bit of a hack to allow the FileSystemStore to map network to chain id.
115// It is not exposed in a very simple and consistent way quite yet and something
116// we want to revisit in the future.
117
118/// A trait to set up the data store.
119/// This is used to setup internal state of the data store before starting the replay.
120/// At the moment is exclusively used by the FileSystemStore to map network to chain id.
121pub trait SetupStore {
122    /// Set up the data store.
123    /// Returns the chain identifier if available, or None if not available.
124    /// When `chain_id` is `None` this is a no-op, unless the given data store
125    /// has a way to fetch the chain id from the network.
126    /// When `chain_id` is `Some(chain_id)` the given data store should override
127    /// the map from network to chain id if it has one.
128    /// That is a meaningful operation only for the FileSystemStore.
129    fn setup(&self, chain_id: Option<String>) -> Result<Option<String>, Error>;
130}

132// ============================================================================
133// Write-back traits
134// ============================================================================
135
136/// Write-back trait for transaction data.
137/// Allows storing transaction data, effects, and checkpoint information.
138pub trait TransactionStoreWriter: TransactionStore {
139    /// Store transaction data, effects, and the checkpoint it was executed in.
140    fn write_transaction(
141        &self,
142        tx_digest: &str,
143        transaction_info: TransactionInfo,
144    ) -> Result<(), Error>;
145}
146
147/// Write-back trait for epoch data.
148/// Allows storing epoch information.
149pub trait EpochStoreWriter: EpochStore {
150    /// Store epoch data for a given epoch.
151    fn write_epoch_info(&self, epoch: u64, epoch_data: EpochData) -> Result<(), Error>;
152}
153
154/// Write-back trait for object data.
155/// Allows storing objects and their version mappings based on the query type.
156pub trait ObjectStoreWriter: ObjectStore {
157    /// Store object data based on the ObjectKey and actual version.
158    ///
159    /// Behavior depends on the VersionQuery in the key:
160    /// - `Version(v)`: Stores the object at version `v` (actual_version should equal `v`)
161    /// - `RootVersion(max_v)`: Stores a mapping from `max_v` to `actual_version` and
162    ///   the object at `actual_version`
163    /// - `AtCheckpoint(checkpoint)`: Stores a mapping from `checkpoint` to `actual_version` and
164    ///   the object at `actual_version`
165    fn write_object(
166        &self,
167        key: &ObjectKey,
168        object: Object,
169        actual_version: u64,
170    ) -> Result<(), Error>;
171}

177/// A reporting trait for data stores to print a usage/summary report.
178///
179/// Implementors are free to print any relevant statistics or configuration details.
180/// The writer allows callers to decide where summaries go (stdout, file, buffers, etc.).
181pub trait StoreSummary {
182    fn summary<W: Write>(&self, writer: &mut W) -> Result<()>;
183}

189/// Trait combining all read capabilities for a data store
190pub trait ReadDataStore: TransactionStore + EpochStore + ObjectStore {}
191
192/// Trait combining all read and write capabilities for a data store
193pub trait ReadWriteDataStore:
194    ReadDataStore + TransactionStoreWriter + EpochStoreWriter + ObjectStoreWriter
195{
196}
197
198// Blanket implementations for the read and write traits
199impl<T> ReadDataStore for T where T: TransactionStore + EpochStore + ObjectStore {}
200
201impl<T> ReadWriteDataStore for T where
202    T: ReadDataStore + TransactionStoreWriter + EpochStoreWriter + ObjectStoreWriter
203{
