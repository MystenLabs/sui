49/// Provides executors for transaction replay, with optional caching.
50/// When caching is enabled, executors are cached per protocol version to avoid recreation.
51/// When caching is disabled, a fresh executor is created for each transaction.
52pub struct ExecutorProvider {
53    cache: BTreeMap<u64, ReplayExecutor>, // u64 is protocol version
54    cache_enabled: bool,
55}
56
57impl ExecutorProvider {
58    pub fn new(cache_enabled: bool) -> Self {
59        Self {
60            cache: BTreeMap::new(),
61            cache_enabled,
62        }
63    }
64
65    /// Get or create an executor for the given epoch.
66    /// If caching is disabled, always creates a new executor.
67    /// If caching is enabled, reuses cached executors (by protocol version) or creates and caches new ones.
68    pub fn get_or_create(
69        &mut self,
70        epoch: u64,
71        epoch_store: &dyn EpochStore,
72    ) -> anyhow::Result<ReplayExecutor> {
73        let protocol_config = epoch_store
74            .protocol_config(epoch)?
75            .ok_or_else(|| anyhow!("Protocol config missing for epoch {}", epoch))?;
76
77        if !self.cache_enabled {
78            return ReplayExecutor::new(protocol_config);
79        }
80
81        let protocol_version = protocol_config.version.as_u64();
82
83        if let Some(executor) = self.cache.get(&protocol_version) {
84            return Ok(executor.clone());
85        }
86
87        let executor = ReplayExecutor::new(protocol_config)?;
88        self.cache.insert(protocol_version, executor.clone());
89        Ok(executor)
90    }
91}

93// `ReplayTransaction` contains all the data needed to replay a transaction.
94// The `object_cache` will contain all the objects and packages touched by the transaction.
95pub struct ReplayTransaction {
96    pub digest: TransactionDigest,
97    pub checkpoint: u64, // used for object queries
98    pub txn_data: TransactionData,
99    pub effects: TransactionEffects,
100    pub executor: ReplayExecutor,
101    // Objects and packages used by the transaction
102    pub object_cache: BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>,
103}

105//
106// Run a single transaction and print results to stdout
107//
108pub(crate) async fn replay_transaction<S: ReadDataStore>(
109    artifact_manager: &ArtifactManager<'_>,
110    tx_digest: &str,
111    data_store: &S,
112    network: String,
113    trace: bool,
114    executor_provider: &mut ExecutorProvider,
115) -> Result<u128> {
116    let _span = info_span!("replay_tx", tx_digest = %tx_digest).entered();
117    // load a `ReplayTransaction`
118    let replay_txn = match ReplayTransaction::load(
119        tx_digest,
120        data_store,
121        data_store,
122        data_store,
123        executor_provider,
124    ) {
125        Ok(replay_txn) => replay_txn,
126        Err(e) => {
127            bail!("Failed to load transaction {}: {:?}", tx_digest, e);
128        }
129    };
130
131    // replay the transaction
132    let mut trace_builder_opt = trace.then(MoveTraceBuilder::new);
133
134    let exec_t0 = Instant::now();
135    let (result, context_and_effects) =
136        execute_transaction_to_effects(replay_txn, data_store, data_store, &mut trace_builder_opt)?;
137    let exec_ms = exec_t0.elapsed().as_millis();
138
139    // TODO: make tracing better abstracted? different tracers?
140    if let Some(trace_builder) = trace_builder_opt {
141        save_trace_output(artifact_manager, trace_builder, &context_and_effects).map_err(|e| {
142            anyhow!(
143                "transaction {} failed to build a trace output path -> {:?}",
144                tx_digest,
145                e
146            )
147        })?;
148    }
149
150    // Save results
151    debug!(
152        tx_digest = %tx_digest,
153        result = ?result,
154        output_dir = %artifact_manager.base_path.display(),
155        "Executed transaction",
156    );
157
158    artifact_manager
159        .member(Artifact::TransactionData)
160        .serialize_artifact(&context_and_effects.txn_data)
161        .transpose()?
162        .unwrap();
163
164    artifact_manager
165        .member(Artifact::TransactionEffects)
166        .serialize_artifact(&context_and_effects.execution_effects)
167        .transpose()?
168        .unwrap();
169
170    artifact_manager
171        .member(Artifact::TransactionGasReport)
172        .serialize_artifact(&context_and_effects.gas_status.gas_usage_report())
173        .transpose()?
174        .unwrap();
175
176    // Save the replay cache summary
177    let cache_summary = ReplayCacheSummary::from_cache(
178        context_and_effects.expected_effects.executed_epoch(),
179        context_and_effects.checkpoint,
180        network.clone(),
181        context_and_effects.protocol_version,
182        &context_and_effects.object_cache,
183    );
184    artifact_manager
185        .member(Artifact::ReplayCacheSummary)
186        .serialize_artifact(&cache_summary)
187        .transpose()?
188        .unwrap();
189
190    // Save move call info if the transaction is a ProgrammableTransaction
191    if let sui_types::transaction::TransactionKind::ProgrammableTransaction(ptb) =
192        context_and_effects.txn_data.kind()
193    {
194        debug!(tx_digest = %tx_digest, "Extracting move call info for {} commands", ptb.commands.len());
195        match MoveCallInfo::from_transaction(ptb, &context_and_effects.object_cache) {
196            Ok(move_call_info) => {
197                let successful_extractions = move_call_info
198                    .command_signatures
199                    .iter()
200                    .filter(|s| s.is_some())
201                    .count();
202                debug!(tx_digest = %tx_digest, "Successfully extracted {} function signatures out of {} commands",
203                    successful_extractions, move_call_info.command_signatures.len());
204
205                artifact_manager
206                    .member(Artifact::MoveCallInfo)
207                    .serialize_artifact(&move_call_info)
208                    .transpose()?
209                    .unwrap();
210            }
211            Err(e) => {
212                warn!(tx_digest = %tx_digest, "Failed to extract move call info: {}", e);
213            }
214        }
215    }
216
217    verify_txn_and_save_effects(
218        artifact_manager,
219        &context_and_effects.expected_effects,
220        &context_and_effects.execution_effects,
221    )?;
222
223    Ok(exec_ms)
224}

226fn verify_txn_and_save_effects(
227    artifact_manager: &ArtifactManager<'_>,
228    expected_effects: &TransactionEffects,
229    effects: &TransactionEffects,
230) -> Result<()> {
231    // If replayed effects are different from the expected ones
232    // (obtained from the chain), save the forked effects and the expected effects
233    // so that they can be diffed in the output.
234    // If replayed and expected effects are the same, save the replayed effects
235    // and try removing the forked effects (if any) so that the output just shows
236    // the replayed effects rather than (now spurious) effects diff.
237    if effects != expected_effects {
238        error!(
239            tx_digest = %effects.transaction_digest(),
240            "Transaction effects do not match expected effects for transaction {}; saving forked effects",
241            effects.transaction_digest(),
242        );
243        artifact_manager
244            .member(Artifact::ForkedTransactionEffects)
245            .serialize_artifact(effects)
246            .transpose()?
247            .unwrap();
248        artifact_manager
249            .member(Artifact::TransactionEffects)
250            .serialize_artifact(expected_effects)
251            .transpose()?
252            .unwrap();
253    } else {
254        artifact_manager
255            .member(Artifact::TransactionEffects)
256            .serialize_artifact(effects)
257            .transpose()?
258            .unwrap();
259        artifact_manager
260            .member(Artifact::ForkedTransactionEffects)
261            .try_remove_artifact()?;
262    }
263    Ok(())
264}

266impl ReplayTransaction {
267    // Load a transaction and builds a `ReplayTransaction` instance.
268    pub fn load(
269        tx_digest: &str,
270        txn_store: &dyn TransactionStore,
271        epoch_store: &dyn EpochStore,
272        object_store: &dyn ObjectStore,
273        executor_provider: &mut ExecutorProvider,
274    ) -> Result<Self, Error> {
275        debug!(op = "load_tx", phase = "start", tx_digest = %tx_digest, "load transaction");
276
277        let digest = tx_digest
278            .parse()
279            .context(format!("Transaction digest malformed: {}", tx_digest))?;
280
281        //
282        // load transaction data and effects
283        let transaction_info = txn_store
284            .transaction_data_and_effects(tx_digest)?
285            .ok_or_else(|| anyhow!(format!("Transaction not found for digest: {}", tx_digest)))?;
286        let txn_data = transaction_info.data;
287        let effects = transaction_info.effects;
288        let checkpoint = transaction_info.checkpoint;
289
290        //
291        // load all objects and packages used by the transaction
292        let object_cache = load_transaction_objects(&txn_data, &effects, checkpoint, object_store)?;
293
294        //
295        // get or create the executor for this epoch
296        let epoch = effects.executed_epoch();
297        let executor = executor_provider.get_or_create(epoch, epoch_store)?;
298
299        debug!(op = "load_tx", phase = "end", tx_digest = %tx_digest, "load transaction");
300
301        Ok(Self {
302            digest,
303            checkpoint,
304            txn_data,
305            effects,
306            executor,
307            object_cache,
308        })
309    }

335    // Get `InputObjects` from a set of (ObjectId, version) pairs, where version is a u64.
336    // This is currently called from `execute_transaction_to_effects` but it could
337    // be computed for a `ReplayTransaction` and cached.
338    pub fn get_input_objects_for_replay(&self) -> Result<InputObjects, anyhow::Error> {
339        let _deleted_shared_info_map: BTreeMap<ObjectID, (TransactionDigest, SequenceNumber)> =
340            BTreeMap::new();
341        let mut resolved_input_objs = vec![];
342        let input_objects_kind = self.txn_data.input_objects().context(format!(
343            "Failed to get input objects from transaction {}",
344            self.digest,
345        ))?;
346        for kind in input_objects_kind.iter() {
347            match kind {
348                InputObjectKind::MovePackage(pkg_id) => {
349                    self
350                        .object_cache
351                        .get(pkg_id)
352                        .map(|pkgs| {
353                            debug_assert!(
354                                pkgs.len() == 1,
355                                "Expected only one version for package {}",
356                                pkg_id
357                            );
358                            let (_version, pkg) = pkgs.iter().next().unwrap();
359                            resolved_input_objs.push(ObjectReadResult {
360                                input_object_kind: *kind,
361                                object: ObjectReadResultKind::Object(pkg.clone()),
362                            })
363                        })
364                        .ok_or_else(|| anyhow::anyhow!(
365                            format!(
366                                "Package {} not found in transaction cache. Should have been loaded already",
367                                pkg_id,
368                            )
369                        ))?;
370                }
371                InputObjectKind::ImmOrOwnedMoveObject((obj_id, version, _digest)) => {
372                    let object = self
373                        .object_cache
374                        .get(obj_id)
375                        .ok_or_else(|| anyhow::anyhow!(
376                            format!(
377                                "Object id {}[{}] not found in transaction cache. Should have been loaded already",
378                                obj_id, version,
379                            )
380                        ))?
381                        .get(&version.value())
382                        .ok_or_else(|| anyhow::anyhow!(
383                            format!(
384                                "Object version {}[{}] not found in transaction cache. Should have been loaded already",
385                                obj_id, version,
386                            )
387                        ))?;
388                    let input_object_kind =
389                        InputObjectKind::ImmOrOwnedMoveObject(object.compute_object_reference());
390                    resolved_input_objs.push(ObjectReadResult {
391                        input_object_kind,
392                        object: ObjectReadResultKind::Object(object.clone()),
393                    });
394                }
395                InputObjectKind::SharedMoveObject {
396                    id,
397                    initial_shared_version,
398                    mutability,
399                } => {
400                    let input_object_kind = InputObjectKind::SharedMoveObject {
401                        id: *id,
402                        initial_shared_version: *initial_shared_version,
403                        mutability: *mutability,
404                    };
405                    let versions =
406                        self.object_cache
407                            .get(id)
408                            .ok_or_else(|| anyhow::anyhow!(
409                                format!(
410                                    "Shared Object id {} not found in transaction cache. Should have been loaded already",
411                                    id,
412                                )
413                            ))?;
414                    debug_assert!(
415                        versions.len() == 1,
416                        "Expected only one version for shared object {}",
417                        id
418                    );
419                    let (_version, obj) = versions.iter().next().unwrap();
420                    resolved_input_objs.push(ObjectReadResult {
421                        input_object_kind,
422                        object: ObjectReadResultKind::Object(obj.clone()),
423                    });
424                }
425            }
426        }
427        trace!("resolved input objects: {:#?}", resolved_input_objs);
428        Ok(InputObjects::new(resolved_input_objs))
429    }

432// Load the objects and packages used by the transaction.
433// Use data and effects to retrieve the objects and packages used.
434// This is the tricky part of replay.
435fn load_transaction_objects(
436    txn_data: &TransactionData,
437    effects: &TransactionEffects,
438    checkpoint: u64,
439    object_store: &dyn ObjectStore,
440) -> Result<BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>, Error> {
441    // collect all package ids required by the transaction
442    let mut packages = get_packages(txn_data)?;
443
444    // get the ids and versions of the input objects to load
445    // load the objects and collect the package ids of the type parameters
446    let object_keys = get_txn_object_keys(txn_data, effects)?;
447    let (mut object_cache, tp_pkgs) = load_objects(&object_keys, object_store)?;
448    packages.extend(&tp_pkgs);
449
450    // load the packages and add them to the object cache
451    let pkg_objects = load_packages(&packages, checkpoint, object_store)?;
452    for (object_id, versions) in pkg_objects {
453        match object_cache.entry(object_id) {
454            Entry::Occupied(mut entry) => {
455                // If the package already exists, extend its versions
456                entry.get_mut().extend(versions);
457            }
458            Entry::Vacant(entry) => {
459                // If the package does not exist, insert it
460                entry.insert(versions);
461            }
462        }
463    }
464
465    Ok(object_cache)
466}

468// Collect all packages in input.
469// For move calls is the package of the call.
470// For vector commands the packages of the type parameter.
471// For publish and upgrade commands, the packages of the dependencies.
472fn get_packages(txn_data: &TransactionData) -> Result<BTreeSet<ObjectID>, Error> {
473    let mut packages = BTreeSet::new();
474    if let TransactionKind::ProgrammableTransaction(ptb) = txn_data.kind() {
475        for cmd in &ptb.commands {
476            match cmd {
477                Command::MoveCall(move_call) => {
478                    packages.insert(move_call.package);
479                    for type_input in move_call.type_arguments.iter() {
480                        let typ = type_input
481                            .to_type_tag()
482                            .context(format!("Failed to read TypeTag: {:?}", type_input))?;
483                        packages_from_type_tag(&typ, &mut packages);
484                    }
485                }
486                Command::MakeMoveVec(type_input, _) => {
487                    if let Some(t) = type_input {
488                        let typ = t
489                            .to_type_tag()
490                            .context(format!("Failed to read TypeTag: {:?}", type_input))?;
491                        packages_from_type_tag(&typ, &mut packages);
492                    }
493                }
494                Command::Publish(_, deps) => {
495                    packages.extend(deps);
496                }
497                Command::Upgrade(_, deps, pkg_id, _) => {
498                    packages.extend(deps);
499                    packages.insert(*pkg_id);
500                }
501                Command::TransferObjects(_, _)
502                | Command::SplitCoins(_, _)
503                | Command::MergeCoins(_, _) => (),
504            }
505        }
506    }
507    Ok(packages)
508}

510// Load the given objects. Objects are loaded and returned as a map from ObjectID to a map of
511// version to Object.
512// Returns also the packages of the type parameters instantiated (e.g. `SUI` in `Coin<SUI>`).
513#[allow(clippy::type_complexity)]
514fn load_objects(
515    object_keys: &[ObjectKey],
516    object_store: &dyn ObjectStore,
517) -> Result<
518    (
519        BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>, // objets loaded
520        BTreeSet<ObjectID>,                                  // packages referenced
521    ),
522    Error,
523> {
524    let mut packages = BTreeSet::new();
525    let mut object_cache: BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>> = BTreeMap::new();
526    let objects = object_store.get_objects(object_keys)?;
527    for object in objects.into_iter() {
528        if object.is_none() {
529            // REVIEW: a `None` is simply ignored, is that correct?
530            continue;
531        }
532        let (object, _version) = object.unwrap();
533        let object_id = object.id();
534        if let Some(tag) = object.as_inner().struct_tag() {
535            packages_from_type_tag(&tag.into(), &mut packages);
536        }
537        let version = object.version().value();
538        object_cache
539            .entry(object_id)
540            .or_default()
541            .insert(version, object);
542    }
543    Ok((object_cache, packages))
544}

546// Load packages and dependencies.
547// This is a 2 steps process. We first load the packages and then collect all the
548// dependencies and query for all of them.
549// REVIEW: depending on what we do for system packages, we may not need to
550// query by checkpoint.
551// Non system package are immutable and may be queried with no version info.
552fn load_packages(
553    packages: &BTreeSet<ObjectID>,
554    checkpoint: u64,
555    object_store: &dyn ObjectStore,
556) -> Result<BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>, Error> {
557    let pkg_object_keys = packages
558        .iter()
559        .map(|pkg_id| ObjectKey {
560            object_id: *pkg_id,
561            version_query: VersionQuery::AtCheckpoint(checkpoint),
562        })
563        .collect::<Vec<_>>();
564    debug!(op = "load_packages", phase = "start", "load_packages");
565    let (objects, packages) = load_objects(&pkg_object_keys, object_store)?;
566    debug!(op = "load_packages", phase = "end", "load_packages");
567    debug_assert!(
568        packages.is_empty(),
569        "Packages should be empty from packages load, there is no type parameter in packages"
570    );
571    Ok(objects)
572}

574// Return the list of objects to load in terms of `ObjectKey` (query to execute)
575// Package objects are not included in the list and handled in `get_packages`.
576fn get_txn_object_keys(
577    txn_data: &TransactionData,
578    effects: &TransactionEffects,
579) -> Result<Vec<ObjectKey>, Error> {
580    let input_object_ids = get_input_ids(txn_data)?;
581    trace!("Input Object IDs: {:#?}", input_object_ids);
582    let effects_object_ids = get_effects_ids(effects)?;
583    trace!("Effects Object IDs: {:#?}", effects_object_ids);
584    // merge input and effects object ids; add the input ids to the effects ids if not present.
585    let mut effect_ids = effects_object_ids
586        .into_iter()
587        .map(|input| (input.object_id, input.version_query))
588        .collect::<BTreeMap<_, _>>();
589    for input_object in input_object_ids.into_iter() {
590        effect_ids
591            .entry(input_object.object_id)
592            .or_insert(input_object.version_query);
593    }
594    Ok(effect_ids
595        .into_iter()
596        .map(|(object_id, version_query)| ObjectKey {
597            object_id,
598            version_query,
599        })
600        .collect::<BTreeSet<ObjectKey>>()
601        .into_iter()
602        .collect::<Vec<_>>())
603}

605// Find all the object ids and versions that are defined in the transaction data.
606// That includes:
607// - the gas coins
608// -- all `CallArg::Object` to PTBs
609fn get_input_ids(txn_data: &TransactionData) -> Result<BTreeSet<ObjectKey>, Error> {
610    // grab all coins
611    let mut object_keys: BTreeSet<ObjectKey> = txn_data
612        .gas_data()
613        .payment
614        .iter()
615        .map(|(id, seq_num, _)| ObjectKey {
616            object_id: *id,
617            version_query: VersionQuery::Version(seq_num.value()),
618        })
619        .collect();
620    // grab all input objects whose version is defined in transaction data (e.g. owned, not shared)
621    if let TransactionKind::ProgrammableTransaction(ptb) = txn_data.kind() {
622        let input_object_keys = ptb
623            .inputs
624            .iter()
625            .filter_map(|input| {
626                if let CallArg::Object(call_arg) = input {
627                    match call_arg {
628                        ObjectArg::ImmOrOwnedObject((id, seq_num, _digest)) => Some(ObjectKey {
629                            object_id: *id,
630                            version_query: VersionQuery::Version(seq_num.value()),
631                        }),
632                        ObjectArg::SharedObject { .. } => {
633                            None // will be in transaction effects
634                        }
635                        ObjectArg::Receiving((id, seq_num, _digest)) => Some(ObjectKey {
636                            object_id: *id,
637                            version_query: VersionQuery::Version(seq_num.value()),
638                        }),
639                    }
640                } else {
641                    None
642                }
643            })
644            .collect::<Vec<_>>();
645        object_keys.extend(input_object_keys);
646    }
647    Ok(object_keys)
648}

650// Get the input shared objects and unchanged consensus objects from the transaction effects
651fn get_effects_ids(effects: &TransactionEffects) -> Result<BTreeSet<ObjectKey>, Error> {
652    let mut object_keys = effects
653        .input_consensus_objects()
654        .iter()
655        .map(|input_consensus_object| match input_consensus_object {
656            InputConsensusObject::MutateConsensusStreamEnded(object_id, version)
657            | InputConsensusObject::ReadConsensusStreamEnded(object_id, version)
658            | InputConsensusObject::Cancelled(object_id, version) => ObjectKey {
659                object_id: *object_id,
660                version_query: VersionQuery::Version(version.value()),
661            },
662            InputConsensusObject::Mutate((object_id, version, _digest))
663            | InputConsensusObject::ReadOnly((object_id, version, _digest)) => ObjectKey {
664                object_id: *object_id,
665                version_query: VersionQuery::Version(version.value()),
666            },
667        })
668        .collect::<BTreeSet<_>>();
669    effects
670        .unchanged_consensus_objects()
671        .iter()
672        .for_each(|(obj_id, kind)| match kind {
673            UnchangedConsensusKind::ReadOnlyRoot((ver, _digest)) => {
674                object_keys.insert(ObjectKey {
675                    object_id: *obj_id,
676                    version_query: VersionQuery::Version(ver.value()),
677                });
678            }
679            UnchangedConsensusKind::MutateConsensusStreamEnded(_)
680            | UnchangedConsensusKind::ReadConsensusStreamEnded(_)
681            | UnchangedConsensusKind::Cancelled(_)
682            | UnchangedConsensusKind::PerEpochConfig => {
683                trace!("Ignored `UnchangedConsensusKind`: {:?}", kind);
684            }
685        });
686    Ok(object_keys)
687}

689// get the package info from the type tag and insert the packages of the type tags (if any)
690// in `packages`
691fn packages_from_type_tag(typ: &TypeTag, packages: &mut BTreeSet<ObjectID>) {
692    match typ {
693        TypeTag::Struct(struct_tag) => {
694            packages.insert(struct_tag.address.into());
695            for ty in struct_tag.type_params.iter() {
696                packages_from_type_tag(ty, packages);
697            }
698        }
699        TypeTag::Vector(type_tag) => {
700            packages_from_type_tag(type_tag, packages);
701        }
702        TypeTag::Bool
703        | TypeTag::U8
704        | TypeTag::U64
705        | TypeTag::U128
706        | TypeTag::Address
707        | TypeTag::Signer
708        | TypeTag::U16
709        | TypeTag::U32
710        | TypeTag::U256 => (),
711    }
712}
