50/// Provides executors for transaction replay, with optional caching.
51/// When caching is enabled, executors are cached per protocol version to avoid recreation.
52/// When caching is disabled, a fresh executor is created for each transaction.
53pub struct ExecutorProvider {
54    cache: BTreeMap<u64, ReplayExecutor>, // u64 is protocol version
55    cache_enabled: bool,
56}
57
58impl ExecutorProvider {
59    pub fn new(cache_enabled: bool) -> Self {
60        Self {
61            cache: BTreeMap::new(),
62            cache_enabled,
63        }
64    }
65
66    /// Get or create an executor for the given epoch.
67    /// If caching is disabled, always creates a new executor.
68    /// If caching is enabled, reuses cached executors (by protocol version) or creates and caches new ones.
69    pub fn get_or_create(
70        &mut self,
71        epoch: u64,
72        epoch_store: &dyn EpochStore,
73    ) -> anyhow::Result<ReplayExecutor> {
74        let protocol_config = epoch_store
75            .protocol_config(epoch)?
76            .ok_or_else(|| anyhow!("Protocol config missing for epoch {}", epoch))?;
77
78        if !self.cache_enabled {
79            return ReplayExecutor::new(protocol_config);
80        }
81
82        let protocol_version = protocol_config.version.as_u64();
83
84        if let Some(executor) = self.cache.get(&protocol_version) {
85            return Ok(executor.clone());
86        }
87
88        let executor = ReplayExecutor::new(protocol_config)?;
89        self.cache.insert(protocol_version, executor.clone());
90        Ok(executor)
91    }
92}

94// `ReplayTransaction` contains all the data needed to replay a transaction.
95// The `object_cache` will contain all the objects and packages touched by the transaction.
96pub struct ReplayTransaction {
97    pub digest: TransactionDigest,
98    pub checkpoint: u64, // used for object queries
99    pub txn_data: TransactionData,
100    pub effects: TransactionEffects,
101    pub executor: ReplayExecutor,
102    // Objects and packages used by the transaction
103    pub object_cache: BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>,
104}

106//
107// Run a single transaction and print results to stdout
108//
109pub(crate) async fn replay_transaction<S: ReadDataStore>(
110    artifact_manager: &ArtifactManager<'_>,
111    tx_digest: &str,
112    data_store: &S,
113    network: String,
114    trace: bool,
115    executor_provider: &mut ExecutorProvider,
116) -> Result<u128> {
117    let _span = info_span!("replay_tx", tx_digest = %tx_digest).entered();
118    // load a `ReplayTransaction`
119    tx_metrics_reset();
120    let replay_txn = match ReplayTransaction::load(
121        tx_digest,
122        data_store,
123        data_store,
124        data_store,
125        executor_provider,
126    ) {
127        Ok(replay_txn) => replay_txn,
128        Err(e) => {
129            bail!("Failed to load transaction {}: {:?}", tx_digest, e);
130        }
131    };
132
133    // replay the transaction
134    let mut trace_builder_opt = trace.then(MoveTraceBuilder::new);
135
136    let exec_t0 = Instant::now();
137    let (result, context_and_effects) =
138        execute_transaction_to_effects(replay_txn, data_store, data_store, &mut trace_builder_opt)?;
139    let exec_ms = exec_t0.elapsed().as_millis();
140
141    // TODO: make tracing better abstracted? different tracers?
142    if let Some(trace_builder) = trace_builder_opt {
143        save_trace_output(artifact_manager, trace_builder, &context_and_effects).map_err(|e| {
144            anyhow!(
145                "transaction {} failed to build a trace output path -> {:?}",
146                tx_digest,
147                e
148            )
149        })?;
150    }
151
152    // Save results
153    info!(
154        tx_digest = %tx_digest,
155        result = ?result,
156        output_dir = %artifact_manager.base_path.display(),
157        "Executed transaction",
158    );
159
160    artifact_manager
161        .member(Artifact::TransactionData)
162        .serialize_artifact(&context_and_effects.txn_data)
163        .transpose()?
164        .unwrap();
165
166    artifact_manager
167        .member(Artifact::TransactionEffects)
168        .serialize_artifact(&context_and_effects.execution_effects)
169        .transpose()?
170        .unwrap();
171
172    artifact_manager
173        .member(Artifact::TransactionGasReport)
174        .serialize_artifact(&context_and_effects.gas_status.gas_usage_report())
175        .transpose()?
176        .unwrap();
177
178    // Save the replay cache summary
179    let cache_summary = ReplayCacheSummary::from_cache(
180        context_and_effects.expected_effects.executed_epoch(),
181        context_and_effects.checkpoint,
182        network.clone(),
183        context_and_effects.protocol_version,
184        &context_and_effects.object_cache,
185    );
186    artifact_manager
187        .member(Artifact::ReplayCacheSummary)
188        .serialize_artifact(&cache_summary)
189        .transpose()?
190        .unwrap();
191
192    // Save move call info if the transaction is a ProgrammableTransaction
193    if let sui_types::transaction::TransactionKind::ProgrammableTransaction(ptb) =
194        context_and_effects.txn_data.kind()
195    {
196        info!(tx_digest = %tx_digest, "Extracting move call info for {} commands", ptb.commands.len());
197        match MoveCallInfo::from_transaction(ptb, &context_and_effects.object_cache) {
198            Ok(move_call_info) => {
199                let successful_extractions = move_call_info
200                    .command_signatures
201                    .iter()
202                    .filter(|s| s.is_some())
203                    .count();
204                info!(tx_digest = %tx_digest, "Successfully extracted {} function signatures out of {} commands",
205                    successful_extractions, move_call_info.command_signatures.len());
206
207                artifact_manager
208                    .member(Artifact::MoveCallInfo)
209                    .serialize_artifact(&move_call_info)
210                    .transpose()?
211                    .unwrap();
212            }
213            Err(e) => {
214                info!(tx_digest = %tx_digest, "Failed to extract move call info: {}", e);
215            }
216        }
217    }
218
219    verify_txn_and_save_effects(
220        artifact_manager,
221        &context_and_effects.expected_effects,
222        &context_and_effects.execution_effects,
223    )?;
224
225    Ok(exec_ms)
226}

228fn verify_txn_and_save_effects(
229    artifact_manager: &ArtifactManager<'_>,
230    expected_effects: &TransactionEffects,
231    effects: &TransactionEffects,
232) -> Result<()> {
233    // If replayed effects are different from the expected ones
234    // (obtained from the chain), save the forked effects and the expected effects
235    // so that they can be diffed in the output.
236    // If replayed and expected effects are the same, save the replayed effects
237    // and try removing the forked effects (if any) so that the output just shows
238    // the replayed effects rather than (now spurious) effects diff.
239    if effects != expected_effects {
240        error!(
241            tx_digest = %effects.transaction_digest(),
242            "Transaction effects do not match expected effects for transaction {}; saving forked effects",
243            effects.transaction_digest(),
244        );
245        artifact_manager
246            .member(Artifact::ForkedTransactionEffects)
247            .serialize_artifact(effects)
248            .transpose()?
249            .unwrap();
250        artifact_manager
251            .member(Artifact::TransactionEffects)
252            .serialize_artifact(expected_effects)
253            .transpose()?
254            .unwrap();
255    } else {
256        artifact_manager
257            .member(Artifact::TransactionEffects)
258            .serialize_artifact(effects)
259            .transpose()?
260            .unwrap();
261        artifact_manager
262            .member(Artifact::ForkedTransactionEffects)
263            .try_remove_artifact()?;
264    }
265    Ok(())
266}

268impl ReplayTransaction {
269    // Load a transaction and builds a `ReplayTransaction` instance.
270    pub fn load(
271        tx_digest: &str,
272        txn_store: &dyn TransactionStore,
273        epoch_store: &dyn EpochStore,
274        object_store: &dyn ObjectStore,
275        executor_provider: &mut ExecutorProvider,
276    ) -> Result<Self, Error> {
277        debug!(op = "load_tx", phase = "start", tx_digest = %tx_digest, "load transaction");
278
279        let digest = tx_digest
280            .parse()
281            .context(format!("Transaction digest malformed: {}", tx_digest))?;
282
283        //
284        // load transaction data and effects
285        let transaction_info = txn_store
286            .transaction_data_and_effects(tx_digest)?
287            .ok_or_else(|| anyhow!(format!("Transaction not found for digest: {}", tx_digest)))?;
288        let txn_data = transaction_info.data;
289        let effects = transaction_info.effects;
290        let checkpoint = transaction_info.checkpoint;
291
292        //
293        // load all objects and packages used by the transaction
294        let object_cache = load_transaction_objects(&txn_data, &effects, checkpoint, object_store)?;
295
296        //
297        // get or create the executor for this epoch
298        let epoch = effects.executed_epoch();
299        let executor = executor_provider.get_or_create(epoch, epoch_store)?;
300
301        debug!(op = "load_tx", phase = "end", tx_digest = %tx_digest, "load transaction");
302
303        Ok(Self {
304            digest,
305            checkpoint,
306            txn_data,
307            effects,
308            executor,
309            object_cache,
310        })
311    }

337    // Get `InputObjects` from a set of (ObjectId, version) pairs, where version is a u64.
338    // This is currently called from `execute_transaction_to_effects` but it could
339    // be computed for a `ReplayTransaction` and cached.
340    pub fn get_input_objects_for_replay(&self) -> Result<InputObjects, anyhow::Error> {
341        let _deleted_shared_info_map: BTreeMap<ObjectID, (TransactionDigest, SequenceNumber)> =
342            BTreeMap::new();
343        let mut resolved_input_objs = vec![];
344        let input_objects_kind = self.txn_data.input_objects().context(format!(
345            "Failed to get input objects from transaction {}",
346            self.digest,
347        ))?;
348        for kind in input_objects_kind.iter() {
349            match kind {
350                InputObjectKind::MovePackage(pkg_id) => {
351                    self
352                        .object_cache
353                        .get(pkg_id)
354                        .map(|pkgs| {
355                            debug_assert!(
356                                pkgs.len() == 1,
357                                "Expected only one version for package {}",
358                                pkg_id
359                            );
360                            let (_version, pkg) = pkgs.iter().next().unwrap();
361                            resolved_input_objs.push(ObjectReadResult {
362                                input_object_kind: *kind,
363                                object: ObjectReadResultKind::Object(pkg.clone()),
364                            })
365                        })
366                        .ok_or_else(|| anyhow::anyhow!(
367                            format!(
368                                "Package {} not found in transaction cache. Should have been loaded already",
369                                pkg_id,
370                            )
371                        ))?;
372                }
373                InputObjectKind::ImmOrOwnedMoveObject((obj_id, version, _digest)) => {
374                    let object = self
375                        .object_cache
376                        .get(obj_id)
377                        .ok_or_else(|| anyhow::anyhow!(
378                            format!(
379                                "Object id {}[{}] not found in transaction cache. Should have been loaded already",
380                                obj_id, version,
381                            )
382                        ))?
383                        .get(&version.value())
384                        .ok_or_else(|| anyhow::anyhow!(
385                            format!(
386                                "Object version {}[{}] not found in transaction cache. Should have been loaded already",
387                                obj_id, version,
388                            )
389                        ))?;
390                    let input_object_kind =
391                        InputObjectKind::ImmOrOwnedMoveObject(object.compute_object_reference());
392                    resolved_input_objs.push(ObjectReadResult {
393                        input_object_kind,
394                        object: ObjectReadResultKind::Object(object.clone()),
395                    });
396                }
397                InputObjectKind::SharedMoveObject {
398                    id,
399                    initial_shared_version,
400                    mutability,
401                } => {
402                    let input_object_kind = InputObjectKind::SharedMoveObject {
403                        id: *id,
404                        initial_shared_version: *initial_shared_version,
405                        mutability: *mutability,
406                    };
407                    let versions =
408                        self.object_cache
409                            .get(id)
410                            .ok_or_else(|| anyhow::anyhow!(
411                                format!(
412                                    "Shared Object id {} not found in transaction cache. Should have been loaded already",
413                                    id,
414                                )
415                            ))?;
416                    debug_assert!(
417                        versions.len() == 1,
418                        "Expected only one version for shared object {}",
419                        id
420                    );
421                    let (_version, obj) = versions.iter().next().unwrap();
422                    resolved_input_objs.push(ObjectReadResult {
423                        input_object_kind,
424                        object: ObjectReadResultKind::Object(obj.clone()),
425                    });
426                }
427            }
428        }
429        trace!("resolved input objects: {:#?}", resolved_input_objs);
430        Ok(InputObjects::new(resolved_input_objs))
431    }

434// Load the objects and packages used by the transaction.
435// Use data and effects to retrieve the objects and packages used.
436// This is the tricky part of replay.
437fn load_transaction_objects(
438    txn_data: &TransactionData,
439    effects: &TransactionEffects,
440    checkpoint: u64,
441    object_store: &dyn ObjectStore,
442) -> Result<BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>, Error> {
443    // collect all package ids required by the transaction
444    let mut packages = get_packages(txn_data)?;
445
446    // get the ids and versions of the input objects to load
447    // load the objects and collect the package ids of the type parameters
448    let object_keys = get_txn_object_keys(txn_data, effects)?;
449    let (mut object_cache, tp_pkgs) = load_objects(&object_keys, object_store)?;
450    packages.extend(&tp_pkgs);
451
452    // load the packages and add them to the object cache
453    let pkg_objects = load_packages(&packages, checkpoint, object_store)?;
454    for (object_id, versions) in pkg_objects {
455        match object_cache.entry(object_id) {
456            Entry::Occupied(mut entry) => {
457                // If the package already exists, extend its versions
458                entry.get_mut().extend(versions);
459            }
460            Entry::Vacant(entry) => {
461                // If the package does not exist, insert it
462                entry.insert(versions);
463            }
464        }
465    }
466
467    Ok(object_cache)
468}

470// Collect all packages in input.
471// For move calls is the package of the call.
472// For vector commands the packages of the type parameter.
473// For publish and upgrade commands, the packages of the dependencies.
474fn get_packages(txn_data: &TransactionData) -> Result<BTreeSet<ObjectID>, Error> {
475    let mut packages = BTreeSet::new();
476    if let TransactionKind::ProgrammableTransaction(ptb) = txn_data.kind() {
477        for cmd in &ptb.commands {
478            match cmd {
479                Command::MoveCall(move_call) => {
480                    packages.insert(move_call.package);
481                    for type_input in move_call.type_arguments.iter() {
482                        let typ = type_input
483                            .to_type_tag()
484                            .context(format!("Failed to read TypeTag: {:?}", type_input))?;
485                        packages_from_type_tag(&typ, &mut packages);
486                    }
487                }
488                Command::MakeMoveVec(type_input, _) => {
489                    if let Some(t) = type_input {
490                        let typ = t
491                            .to_type_tag()
492                            .context(format!("Failed to read TypeTag: {:?}", type_input))?;
493                        packages_from_type_tag(&typ, &mut packages);
494                    }
495                }
496                Command::Publish(_, deps) => {
497                    packages.extend(deps);
498                }
499                Command::Upgrade(_, deps, pkg_id, _) => {
500                    packages.extend(deps);
501                    packages.insert(*pkg_id);
502                }
503                Command::TransferObjects(_, _)
504                | Command::SplitCoins(_, _)
505                | Command::MergeCoins(_, _) => (),
506            }
507        }
508    }
509    Ok(packages)
510}

512// Load the given objects. Objects are loaded and returned as a map from ObjectID to a map of
513// version to Object.
514// Returns also the packages of the type parameters instantiated (e.g. `SUI` in `Coin<SUI>`).
515#[allow(clippy::type_complexity)]
516fn load_objects(
517    object_keys: &[ObjectKey],
518    object_store: &dyn ObjectStore,
519) -> Result<
520    (
521        BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>, // objets loaded
522        BTreeSet<ObjectID>,                                  // packages referenced
523    ),
524    Error,
525> {
526    let mut packages = BTreeSet::new();
527    let mut object_cache: BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>> = BTreeMap::new();
528    let objects = object_store.get_objects(object_keys)?;
529    for object in objects.into_iter() {
530        if object.is_none() {
531            // REVIEW: a `None` is simply ignored, is that correct?
532            continue;
533        }
534        let (object, _version) = object.unwrap();
535        let object_id = object.id();
536        if let Some(tag) = object.as_inner().struct_tag() {
537            packages_from_type_tag(&tag.into(), &mut packages);
538        }
539        let version = object.version().value();
540        object_cache
541            .entry(object_id)
542            .or_default()
543            .insert(version, object);
544    }
545    Ok((object_cache, packages))
546}

548// Load packages and dependencies.
549// This is a 2 steps process. We first load the packages and then collect all the
550// dependencies and query for all of them.
551// REVIEW: depending on what we do for system packages, we may not need to
552// query by checkpoint.
553// Non system package are immutable and may be queried with no version info.
554fn load_packages(
555    packages: &BTreeSet<ObjectID>,
556    checkpoint: u64,
557    object_store: &dyn ObjectStore,
558) -> Result<BTreeMap<ObjectID, BTreeMap<ObjectVersion, Object>>, Error> {
559    let pkg_object_keys = packages
560        .iter()
561        .map(|pkg_id| ObjectKey {
562            object_id: *pkg_id,
563            version_query: VersionQuery::AtCheckpoint(checkpoint),
564        })
565        .collect::<Vec<_>>();
566    debug!(op = "load_packages", phase = "start", "load_packages");
567    let (objects, packages) = load_objects(&pkg_object_keys, object_store)?;
568    debug!(op = "load_packages", phase = "end", "load_packages");
569    debug_assert!(
570        packages.is_empty(),
571        "Packages should be empty from packages load, there is no type parameter in packages"
572    );
573    Ok(objects)
574}

576// Return the list of objects to load in terms of `ObjectKey` (query to execute)
577// Package objects are not included in the list and handled in `get_packages`.
578fn get_txn_object_keys(
579    txn_data: &TransactionData,
580    effects: &TransactionEffects,
581) -> Result<Vec<ObjectKey>, Error> {
582    let input_object_ids = get_input_ids(txn_data)?;
583    trace!("Input Object IDs: {:#?}", input_object_ids);
584    let effects_object_ids = get_effects_ids(effects)?;
585    trace!("Effects Object IDs: {:#?}", effects_object_ids);
586    // merge input and effects object ids; add the input ids to the effects ids if not present.
587    let mut effect_ids = effects_object_ids
588        .into_iter()
589        .map(|input| (input.object_id, input.version_query))
590        .collect::<BTreeMap<_, _>>();
591    for input_object in input_object_ids.into_iter() {
592        effect_ids
593            .entry(input_object.object_id)
594            .or_insert(input_object.version_query);
595    }
596    Ok(effect_ids
597        .into_iter()
598        .map(|(object_id, version_query)| ObjectKey {
599            object_id,
600            version_query,
601        })
602        .collect::<BTreeSet<ObjectKey>>()
603        .into_iter()
604        .collect::<Vec<_>>())
605}

607// Find all the object ids and versions that are defined in the transaction data.
608// That includes:
609// - the gas coins
610// -- all `CallArg::Object` to PTBs
611fn get_input_ids(txn_data: &TransactionData) -> Result<BTreeSet<ObjectKey>, Error> {
612    // grab all coins
613    let mut object_keys: BTreeSet<ObjectKey> = txn_data
614        .gas_data()
615        .payment
616        .iter()
617        .map(|(id, seq_num, _)| ObjectKey {
618            object_id: *id,
619            version_query: VersionQuery::Version(seq_num.value()),
620        })
621        .collect();
622    // grab all input objects whose version is defined in transaction data (e.g. owned, not shared)
623    if let TransactionKind::ProgrammableTransaction(ptb) = txn_data.kind() {
624        let input_object_keys = ptb
625            .inputs
626            .iter()
627            .filter_map(|input| {
628                if let CallArg::Object(call_arg) = input {
629                    match call_arg {
630                        ObjectArg::ImmOrOwnedObject((id, seq_num, _digest)) => Some(ObjectKey {
631                            object_id: *id,
632                            version_query: VersionQuery::Version(seq_num.value()),
633                        }),
634                        ObjectArg::SharedObject { .. } => {
635                            None // will be in transaction effects
636                        }
637                        ObjectArg::Receiving((id, seq_num, _digest)) => Some(ObjectKey {
638                            object_id: *id,
639                            version_query: VersionQuery::Version(seq_num.value()),
640                        }),
641                    }
642                } else {
643                    None
644                }
645            })
646            .collect::<Vec<_>>();
647        object_keys.extend(input_object_keys);
648    }
649    Ok(object_keys)
650}

652// Get the input shared objects and unchanged consensus objects from the transaction effects
653fn get_effects_ids(effects: &TransactionEffects) -> Result<BTreeSet<ObjectKey>, Error> {
654    let mut object_keys = effects
655        .input_consensus_objects()
656        .iter()
657        .map(|input_consensus_object| match input_consensus_object {
658            InputConsensusObject::MutateConsensusStreamEnded(object_id, version)
659            | InputConsensusObject::ReadConsensusStreamEnded(object_id, version)
660            | InputConsensusObject::Cancelled(object_id, version) => ObjectKey {
661                object_id: *object_id,
662                version_query: VersionQuery::Version(version.value()),
663            },
664            InputConsensusObject::Mutate((object_id, version, _digest))
665            | InputConsensusObject::ReadOnly((object_id, version, _digest)) => ObjectKey {
666                object_id: *object_id,
667                version_query: VersionQuery::Version(version.value()),
668            },
669        })
670        .collect::<BTreeSet<_>>();
671    effects
672        .unchanged_consensus_objects()
673        .iter()
674        .for_each(|(obj_id, kind)| match kind {
675            UnchangedConsensusKind::ReadOnlyRoot((ver, _digest)) => {
676                object_keys.insert(ObjectKey {
677                    object_id: *obj_id,
678                    version_query: VersionQuery::Version(ver.value()),
679                });
680            }
681            UnchangedConsensusKind::MutateConsensusStreamEnded(_)
682            | UnchangedConsensusKind::ReadConsensusStreamEnded(_)
683            | UnchangedConsensusKind::Cancelled(_)
684            | UnchangedConsensusKind::PerEpochConfig => {
685                trace!("Ignored `UnchangedConsensusKind`: {:?}", kind);
686            }
687        });
688    Ok(object_keys)
689}

691// get the package info from the type tag and insert the packages of the type tags (if any)
692// in `packages`
693fn packages_from_type_tag(typ: &TypeTag, packages: &mut BTreeSet<ObjectID>) {
694    match typ {
695        TypeTag::Struct(struct_tag) => {
696            packages.insert(struct_tag.address.into());
697            for ty in struct_tag.type_params.iter() {
698                packages_from_type_tag(ty, packages);
699            }
700        }
701        TypeTag::Vector(type_tag) => {
702            packages_from_type_tag(type_tag, packages);
703        }
704        TypeTag::Bool
705        | TypeTag::U8
706        | TypeTag::U64
707        | TypeTag::U128
708        | TypeTag::Address
709        | TypeTag::Signer
710        | TypeTag::U16
711        | TypeTag::U32
712        | TypeTag::U256 => (),
713    }
714}
