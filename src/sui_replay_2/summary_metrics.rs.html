4use std::sync::atomic::{AtomicU64, Ordering};
5
6/// Per-transaction metrics structure (shared atomic state)
7pub(crate) struct TxMetrics {
8    query_txn: (AtomicU64, AtomicU64),   // (ms, count)
9    query_objs: (AtomicU64, AtomicU64),  // (ms, count)
10    query_epoch: (AtomicU64, AtomicU64), // (ms, count)
11}
12
13impl TxMetrics {
14    const fn new() -> Self {
15        Self {
16            query_txn: (AtomicU64::new(0), AtomicU64::new(0)),
17            query_objs: (AtomicU64::new(0), AtomicU64::new(0)),
18            query_epoch: (AtomicU64::new(0), AtomicU64::new(0)),
19        }
20    }
21
22    fn reset(&self) {
23        self.query_txn.0.store(0, Ordering::Relaxed);
24        self.query_txn.1.store(0, Ordering::Relaxed);
25        self.query_objs.0.store(0, Ordering::Relaxed);
26        self.query_objs.1.store(0, Ordering::Relaxed);
27        self.query_epoch.0.store(0, Ordering::Relaxed);
28        self.query_epoch.1.store(0, Ordering::Relaxed);
29    }
30
31    fn add_txn(&self, ms: u128) {
32        self.query_txn.0.fetch_add(ms as u64, Ordering::Relaxed);
33        self.query_txn.1.fetch_add(1, Ordering::Relaxed);
34    }
35
36    fn add_objs(&self, ms: u128) {
37        self.query_objs.0.fetch_add(ms as u64, Ordering::Relaxed);
38        self.query_objs.1.fetch_add(1, Ordering::Relaxed);
39    }
40
41    fn add_epoch(&self, ms: u128) {
42        self.query_epoch.0.fetch_add(ms as u64, Ordering::Relaxed);
43        self.query_epoch.1.fetch_add(1, Ordering::Relaxed);
44    }
45}
46
47static TX_METRICS: TxMetrics = TxMetrics::new();
48
49// Reset per-transaction metrics (timers and query counts).
50pub(crate) fn tx_metrics_reset() {
51    TX_METRICS.reset();
52}
53
54// Add elapsed milliseconds to the transaction-data query timer and increment count.
55pub(crate) fn tx_metrics_add_txn(ms: u128) {
56    TX_METRICS.add_txn(ms);
57}
58
59// Add elapsed milliseconds to the objects query timer and increment count.
60pub(crate) fn tx_metrics_add_objs(ms: u128) {
61    TX_METRICS.add_objs(ms);
62}
63
64// Add elapsed milliseconds to the epoch query timer and increment count.
65pub(crate) fn tx_metrics_add_epoch(ms: u128) {
66    TX_METRICS.add_epoch(ms);
67}

69/// Accumulator for total metrics across all transactions in a replay run.
70#[derive(Debug, Default, Clone)]
71pub struct TotalMetrics {
72    pub tx_count: u64,
73    pub success_count: u64,
74    pub failure_count: u64,
75    pub total_ms: u128,
76    pub exec_ms: u128,
77}
78
79impl TotalMetrics {
80    pub fn new() -> Self {
81        Self::default()
82    }
83
84    /// Accumulate metrics from a single transaction replay.
85    pub fn add_transaction(&mut self, success: bool, total_ms: u128, exec_ms: u128) {
86        self.tx_count += 1;
87        if success {
88            self.success_count += 1;
89        } else {
90            self.failure_count += 1;
91        }
92        self.total_ms += total_ms;
93        self.exec_ms += exec_ms;
94    }
95}
