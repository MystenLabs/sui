4//! Tracing utilities.
5//! Mostly deals with directory/file saving and what gets saved in the trace output.
6
7use crate::{
8    artifacts::{Artifact, ArtifactManager},
9    execution::TxnContextAndEffects,
10};
11use anyhow::{Context, Error};
12use move_binary_format::CompiledModule;
13use move_bytecode_source_map::utils::serialize_to_json_string;
14use move_command_line_common::files::MOVE_BYTECODE_EXTENSION;
15use move_disassembler::disassembler::Disassembler;
16use move_ir_types::location::Spanned;
17use move_trace_format::format::MoveTraceBuilder;
18use std::fs;
19use sui_types::object::Data;
20
21const BCODE_DIR: &str = "bytecode";
22const SOURCE_DIR: &str = "source";
23
24/// Saves the trace and additional metadata needed to analyze the trace
25/// to a subderectory named after the transaction digest.
26pub fn save_trace_output(
27    artifact_manager: &ArtifactManager<'_>,
28    trace_builder: MoveTraceBuilder,
29    context_and_effects: &TxnContextAndEffects,
30) -> Result<(), Error> {
31    let trace = trace_builder.into_trace();
32    let trace_member = artifact_manager.member(Artifact::Trace);
33    trace_member
34        .serialize_move_trace(trace)
35        .transpose()?
36        .unwrap();
37
38    let TxnContextAndEffects {
39        txn_data: _,
40        execution_effects: _,
41        expected_effects: _,
42        gas_status: _,
43        object_cache,
44        inner_store: tmp_store,
45        checkpoint: _,
46        protocol_version: _,
47    } = context_and_effects;
48
49    // grab all packages from the transaction and save them locally for debug
50    let mut pkgs = object_cache
51        .values()
52        .flat_map(|versions| versions.values())
53        .filter_map(|obj| {
54            if let Data::Package(pkg) = &obj.data {
55                Some(pkg)
56            } else {
57                None
58            }
59        })
60        .collect::<Vec<_>>();
61    for obj in tmp_store.written.values() {
62        if let Data::Package(pkg) = &obj.data {
63            pkgs.push(pkg);
64        }
65    }
66    for pkg in pkgs {
67        let pkg_addr = format!("{:?}", pkg.id());
68        let pkg_dir = artifact_manager.base_path.join(&pkg_addr);
69        // TODO: have this use the artifact manager as well.
70        let bcode_pkg_dir = pkg_dir.join(BCODE_DIR);
71        fs::create_dir_all(&bcode_pkg_dir).context(format!(
72            "Failed to create bytecode output directory '{:?}'",
73            bcode_pkg_dir,
74        ))?;
75        fs::create_dir_all(&bcode_pkg_dir)
76            .context("Failed to create bytecode package directory")?;
77        for (mod_name, serialized_mod) in pkg.serialized_module_map() {
78            let compiled_mod =
79                CompiledModule::deserialize_with_defaults(serialized_mod).context(format!(
80                    "Failed to deserialize module {:?} in package {}",
81                    mod_name, &pkg_addr,
82                ))?;
83            let d = Disassembler::from_module(&compiled_mod, Spanned::unsafe_no_loc(()).loc)
84                .context(format!(
85                    "Failed to create disassembler for module {:?} in package {}",
86                    mod_name, &pkg_addr,
87                ))?;
88            let (disassemble_string, mut bcode_map) =
89                d.disassemble_with_source_map().context(format!(
90                    "Failed to disassemble module {:?} in package {}",
91                    mod_name, &pkg_addr,
92                ))?;
93            // need version ID here (for potentially upgraded package) rather than original ID
94            // (for the original version of the package), otherwise we won't be able to
95            // distinguish map content based on their module IDs
96            bcode_map.module_name.0 = pkg.id().into();
97            let bcode_map_json = serialize_to_json_string(&bcode_map).context(format!(
98                "Failed to serialize bytecode source map for module {:?} in package {}",
99                mod_name, &pkg_addr,
100            ))?;
101            fs::write(
102                bcode_pkg_dir.join(format!("{}.{}", mod_name, MOVE_BYTECODE_EXTENSION)),
103                disassemble_string,
104            )
105            .context(format!(
106                "Failed to write disassembled bytecode for module {:?} in package {}",
107                mod_name, &pkg_addr,
108            ))?;
109            fs::write(
110                bcode_pkg_dir.join(format!("{}.json", mod_name)),
111                bcode_map_json,
112            )
113            .context(format!(
114                "Failed to write bytecode source map for module {:?} in package {}",
115                mod_name, &pkg_addr,
116            ))?;
117        }
118        // create empty sources directory as a known placeholder for the users
119        // to put optional source files there
120        let src_pkg_dir = pkg_dir.join(SOURCE_DIR);
121        fs::create_dir_all(&src_pkg_dir).context(format!(
122            "Failed to create source output directory '{:?}'",
123            src_pkg_dir,
124        ))?;
125    }
126
127    Ok(())
128}
