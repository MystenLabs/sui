38/// Get an array of all AccountBalances for an AccountIdentifier and the BlockIdentifier
39/// at which the balance lookup was performed.
40/// [Mesh API Spec](https://docs.cdp.coinbase.com/api-reference/mesh/account/get-an-account-balance)
41pub async fn balance(
42    State(mut ctx): State<OnlineServerContext>,
43    Extension(env): Extension<SuiEnv>,
44    WithRejection(Json(request), _): WithRejection<Json<AccountBalanceRequest>, Error>,
45) -> Result<AccountBalanceResponse, Error> {
46    env.check_network_identifier(&request.network_identifier)?;
47    let address = request.account_identifier.address;
48    let currencies = &request.currencies;
49
50    let checkpoint = get_checkpoint(&mut ctx).await?;
51    let balances = get_balances(&mut ctx, &request, address, currencies.clone()).await?;
52
53    Ok(AccountBalanceResponse {
54        block_identifier: ctx.blocks().create_block_identifier(checkpoint).await?,
55        balances,
56    })
57}

59async fn get_checkpoint(ctx: &mut OnlineServerContext) -> Result<CheckpointSequenceNumber, Error> {
60    let request =
61        GetCheckpointRequest::latest().with_read_mask(FieldMask::from_paths(["sequence_number"]));
62
63    Ok(ctx
64        .client
65        .ledger_client()
66        .get_checkpoint(request)
67        .await?
68        .into_inner()
69        .checkpoint()
70        .sequence_number())
71}

73async fn get_balances(
74    ctx: &mut OnlineServerContext,
75    request: &AccountBalanceRequest,
76    address: SuiAddress,
77    currencies: Currencies,
78) -> Result<Vec<Amount>, Error> {
79    if let Some(sub_account) = &request.account_identifier.sub_account {
80        let account_type = sub_account.account_type.clone();
81        get_sub_account_balances(account_type, &mut ctx.client, address).await
82    } else if !currencies.0.is_empty() {
83        let balance_futures = currencies.0.iter().map(|currency| {
84            let coin_type = currency.metadata.clone().coin_type.clone();
85            let mut client = ctx.client.clone();
86            async move {
87                (
88                    currency.clone(),
89                    get_account_balances(&mut client, address, &coin_type).await,
90                )
91            }
92        });
93        let balances: Vec<(Currency, Result<i128, Error>)> = join_all(balance_futures).await;
94        let mut amounts = Vec::new();
95        for (currency, balance_result) in balances {
96            match balance_result {
97                Ok(value) => amounts.push(Amount::new(value, Some(currency))),
98                Err(_e) => {
99                    return Err(Error::InvalidInput(format!(
100                        "{:?}",
101                        currency.metadata.coin_type
102                    )));
103                }
104            }
105        }
106        Ok(amounts)
107    } else {
108        Err(Error::InvalidInput(
109            "Coin type is required for this request".to_string(),
110        ))
111    }
112}

114async fn get_account_balances(
115    client: &mut Client,
116    address: SuiAddress,
117    coin_type: &str,
118) -> Result<i128, Error> {
119    let request = GetBalanceRequest::default()
120        .with_owner(address.to_string())
121        .with_coin_type(coin_type.to_string());
122    let response = client.state_client().get_balance(request).await?;
123    Ok(response.into_inner().balance().balance() as i128)
124}

126struct EpochTimingInfo {
127    epoch: u64,
128    epoch_start_timestamp_ms: u64,
129    epoch_duration_ms: u64,
130}

132async fn get_epoch_timing(client: &mut Client) -> Result<EpochTimingInfo, Error> {
133    let request = GetEpochRequest::latest().with_read_mask(FieldMask::from_paths([
134        "epoch",
135        "system_state.epoch_start_timestamp_ms",
136        "system_state.parameters.epoch_duration_ms",
137    ]));
138    let response = client
139        .ledger_client()
140        .get_epoch(request)
141        .await?
142        .into_inner();
143    let epoch_info = response.epoch();
144    let system_state = epoch_info.system_state();
145    Ok(EpochTimingInfo {
146        epoch: epoch_info.epoch(),
147        epoch_start_timestamp_ms: system_state.epoch_start_timestamp_ms(),
148        epoch_duration_ms: system_state.parameters().epoch_duration_ms(),
149    })
150}

163/// Reads exchange rates for all active validator staking pools.
164pub(crate) async fn get_pool_exchange_rates(
165    client: &mut Client,

167    Ok(get_pool_exchange_rates_with_epoch(client).await?.0)

170/// Atomic snapshot of validator-set state from a single `GetEpochRequest::latest()`.
171///
172/// Splitting these reads across multiple RPCs allows an epoch transition to
173/// land between them: the caller could observe rate from epoch N, then read
174/// epoch N+1, and bind the transaction to N+1 with a stale N rate, silently
175/// violating AtMost caps and aborting AtLeast guards. Bundling rate, epoch,
176/// and the inactive-table id into one response eliminates that race.
177pub(crate) struct ValidatorSetSnapshot {
178    /// Active pool rates keyed by `pool.id` (canonical 0x-prefixed hex string).
179    pub active_rates: std::collections::HashMap<String, PoolRateInfo>,
180    /// Current epoch the snapshot was taken in.
181    pub epoch: u64,
182    /// `validators.inactive_validators.id` — UID of the
183    /// `Table<ID, ValidatorWrapper>` storing deactivated pools. `None` only if
184    /// the proto omitted the field.
185    pub inactive_validators_table_id: Option<String>,
186}
187
188/// Reads exchange rates and the epoch they're snapshotted in from a single
189/// `GetEpochRequest::latest()` response. Used by amount-sensitive operations
190/// (e.g. `MergeAndRedeemFungibleStakedSui::AtLeast`/`AtMost`) that must pin
191/// the rate quote to the same epoch the resulting transaction will be bound to.
192pub(crate) async fn get_pool_exchange_rates_with_epoch(
193    client: &mut Client,
194) -> Result<(std::collections::HashMap<String, PoolRateInfo>, u64), Error> {
195    let snap = get_validator_set_snapshot(client).await?;
196    Ok((snap.active_rates, snap.epoch))
197}

199/// Read the full validator-set snapshot atomically.
200pub(crate) async fn get_validator_set_snapshot(
201    client: &mut Client,
202) -> Result<ValidatorSetSnapshot, Error> {
203    let request = GetEpochRequest::latest().with_read_mask(FieldMask::from_paths([
204        "epoch",
205        "system_state.validators.active_validators",
206        "system_state.validators.inactive_validators",
207    ]));
208    let response = client
209        .ledger_client()
210        .get_epoch(request)
211        .await?
212        .into_inner();
213    let epoch_obj = response.epoch();
214    let epoch = epoch_obj.epoch();
215    let system_state = epoch_obj.system_state();
216    let validators_proto = system_state.validators();
217    let validators = validators_proto.active_validators();
218
219    let inactive_validators_table_id = validators_proto
220        .inactive_validators
221        .as_ref()
222        .and_then(|t| t.id.as_ref())
223        .cloned();
224
225    let mut rates = std::collections::HashMap::new();
226    for validator in validators {
227        let pool = validator.staking_pool();
228        let pool_id = pool.id().to_string();
229        let validator_address = Address::from_str(validator.address())
230            .map_err(|e| Error::DataError(format!("Invalid validator address: {}", e)))?;
231        let pool_extra_fields_id = pool
232            .extra_fields_opt()
233            .and_then(|t| t.id_opt())
234            .map(|s| s.to_string());
235        rates.insert(
236            pool_id,
237            PoolRateInfo {
238                sui_balance: pool.sui_balance(),
239                pool_token_balance: pool.pool_token_balance(),
240                validator_address,
241                pool_extra_fields_id,
242            },
243        );
244    }
245    Ok(ValidatorSetSnapshot {
246        active_rates: rates,
247        epoch,
248        inactive_validators_table_id,
249    })
250}

252async fn get_sub_account_balances(
253    account_type: SubAccountType,
254    client: &mut Client,
255    address: SuiAddress,
256) -> Result<Vec<Amount>, Error> {
257    let epoch_timing = get_epoch_timing(client).await?;
258    let current_epoch = epoch_timing.epoch;
259    let address = Address::from(address);
260
261    if matches!(account_type, SubAccountType::FungibleStakedSuiValue) {
262        return get_fungible_staked_sui_value(client, address, &epoch_timing).await;
263    }
264
265    let delegated_stakes = client.list_delegated_stake(&address).await?;
266
267    let amounts: Vec<SubBalance> = match account_type {
268        SubAccountType::Stake => delegated_stakes
269            .into_iter()
270            .filter(|stake| current_epoch >= stake.activation_epoch)
271            .map(|stake| SubBalance {
272                stake_id: stake.staked_sui_id,
273                validator: stake.validator_address,
274                value: stake.principal as i128,
275                activation_epoch: Some(stake.activation_epoch),
276            })
277            .collect(),
278        SubAccountType::PendingStake => delegated_stakes
279            .into_iter()
280            .filter(|stake| current_epoch < stake.activation_epoch)
281            .map(|stake| SubBalance {
282                stake_id: stake.staked_sui_id,
283                validator: stake.validator_address,
284                value: stake.principal as i128,
285                activation_epoch: Some(stake.activation_epoch),
286            })
287            .collect(),
288        SubAccountType::EstimatedReward => delegated_stakes
289            .into_iter()
290            .filter(|stake| current_epoch >= stake.activation_epoch)
291            .map(|stake| SubBalance {
292                stake_id: stake.staked_sui_id,
293                validator: stake.validator_address,
294                value: stake.rewards as i128,
295                activation_epoch: None,
296            })
297            .collect(),
298        SubAccountType::FungibleStakedSuiValue => unreachable!(),
299    };
300
301    let amount = if amounts.is_empty() {
302        Amount::new(0, None)
303    } else {
304        Amount::new_from_sub_balances(amounts)
305    };
306
307    Ok(vec![amount.with_epoch_timing(
308        epoch_timing.epoch,
309        epoch_timing.epoch_start_timestamp_ms,
310        epoch_timing.epoch_duration_ms,
311    )])
312}

314async fn get_fungible_staked_sui_value(
315    client: &mut Client,
316    address: Address,
317    epoch_timing: &EpochTimingInfo,
318) -> Result<Vec<Amount>, Error> {
319    use futures::TryStreamExt;
320
321    let list_request = ListOwnedObjectsRequest::default()
322        .with_owner(address.to_string())
323        .with_object_type("0x3::staking_pool::FungibleStakedSui".to_string())
324        .with_page_size(1000u32)
325        .with_read_mask(FieldMask::from_paths(["object_id", "contents"]));
326
327    let fss_objects: Vec<_> = client
328        .list_owned_objects(list_request)
329        .map_err(Error::from)
330        .try_collect()
331        .await?;
332
333    if fss_objects.is_empty() {
334        return Ok(vec![Amount::new(0, None).with_epoch_timing(
335            epoch_timing.epoch,
336            epoch_timing.epoch_start_timestamp_ms,
337            epoch_timing.epoch_duration_ms,
338        )]);
339    }
340
341    let pool_rates = get_pool_exchange_rates(client).await?;
342
343    let mut sub_balances = Vec::new();
344    for obj in &fss_objects {
345        let contents = obj
346            .contents
347            .as_ref()
348            .ok_or_else(|| Error::DataError("FungibleStakedSui missing contents".to_string()))?;
349        let fss: FungibleStakedSuiBcs = contents.deserialize().map_err(|e| {
350            Error::DataError(format!("Failed to deserialize FungibleStakedSui: {}", e))
351        })?;
352
353        let pool_id_str = fss.pool_id.to_string();
354        let rate = pool_rates.get(&pool_id_str).ok_or_else(|| {
355            Error::DataError(format!("No exchange rate found for pool {}", pool_id_str))
356        })?;
357
358        let sui_equivalent = if rate.pool_token_balance > 0 {
359            (fss.value as u128 * rate.sui_balance as u128 / rate.pool_token_balance as u128) as u64
360        } else {
361            fss.value
362        };
363
364        sub_balances.push(SubBalance {
365            stake_id: Address::from_str(obj.object_id())
366                .map_err(|e| Error::DataError(format!("Invalid FSS object_id: {}", e)))?,
367            validator: rate.validator_address,
368            value: sui_equivalent as i128,
369            activation_epoch: None,
370        });
371    }
372
373    let amount = if sub_balances.is_empty() {
374        Amount::new(0, None)
375    } else {
376        Amount::new_from_sub_balances(sub_balances)
377    };
378
379    Ok(vec![amount.with_epoch_timing(
380        epoch_timing.epoch,
381        epoch_timing.epoch_start_timestamp_ms,
382        epoch_timing.epoch_duration_ms,
383    )])
384}

386/// Get an array of all unspent coins for an AccountIdentifier and the BlockIdentifier at which the lookup was performed. .
387/// [Mesh API Spec](https://docs.cdp.coinbase.com/api-reference/mesh/account/get-an-account-unspent-coins)
388/// TODO This API is supposed to return coins of all types, not just SUI. It also has a 'currencies' parameter that we
389/// are igorning which can be used to filter the type of coins that are returned.
390pub async fn coins(
391    State(context): State<OnlineServerContext>,
392    Extension(env): Extension<SuiEnv>,
393    WithRejection(Json(request), _): WithRejection<Json<AccountCoinsRequest>, Error>,
394) -> Result<AccountCoinsResponse, Error> {
395    env.check_network_identifier(&request.network_identifier)?;
396
397    let coin_request = ListOwnedObjectsRequest::default()
398        .with_owner(request.account_identifier.address.to_string())
399        .with_object_type(StructTag::gas_coin().to_string())
400        .with_page_size(5000u32)
401        .with_read_mask(FieldMask::from_paths(["object_id", "version", "balance"]));
402
403    let coins = context
404        .client
405        .list_owned_objects(coin_request)
406        .map_err(Error::from)
407        .and_then(|object| async move {
408            Ok(Coin {
409                coin_identifier: CoinIdentifier {
410                    identifier: CoinID {
411                        id: ObjectID::from_hex_literal(object.object_id())
412                            .map_err(|e| Error::DataError(format!("Invalid object_id: {}", e)))?,
413                        version: SequenceNumber::from(object.version()),
414                    },
415                },
416                amount: Amount::new(object.balance() as i128, None),
417            })
418        })
419        .try_collect()
420        .await?;
421
422    Ok(AccountCoinsResponse {
423        block_identifier: context.blocks().current_block_identifier().await?,
424        coins,
425    })
