50#[async_trait]
51pub trait BlockProvider {
52    async fn get_block_by_index(&self, index: u64) -> Result<BlockResponse, Error>;
53    async fn get_block_by_hash(&self, hash: BlockHash) -> Result<BlockResponse, Error>;
54    async fn current_block(&self) -> Result<BlockResponse, Error>;
55    async fn genesis_block_identifier(&self) -> Result<BlockIdentifier, Error>;
56    async fn oldest_block_identifier(&self) -> Result<BlockIdentifier, Error>;
57    async fn current_block_identifier(&self) -> Result<BlockIdentifier, Error>;
58    async fn create_block_identifier(
59        &self,
60        checkpoint: CheckpointSequenceNumber,
61    ) -> Result<BlockIdentifier, Error>;
62}

70#[async_trait]
71impl BlockProvider for CheckpointBlockProvider {
72    async fn get_block_by_index(&self, index: u64) -> Result<BlockResponse, Error> {
73        let request = GetCheckpointRequest::by_sequence_number(index).with_read_mask(
74            FieldMask::from_paths([
75                "sequence_number",
76                "digest",
77                "summary.sequence_number",
78                "summary.previous_digest",
79                "summary.timestamp",
80                "transactions.digest",
81                "transactions.transaction.sender",
82                "transactions.transaction.gas_payment",
83                "transactions.transaction.kind",
84                "transactions.effects.gas_object",
85                "transactions.effects.gas_used",
86                "transactions.effects.status",
87                "transactions.balance_changes",
88                "transactions.events.events.event_type",
89                "transactions.events.events.json",
90            ]),
91        );
92
93        let mut client = self.client.clone();
94        let response = client
95            .ledger_client()
96            .get_checkpoint(request)
97            .await
98            .map_err(|e| Error::from(anyhow::anyhow!("Failed to get checkpoint: {}", e)))?
99            .into_inner();
100
101        let checkpoint = response
102            .checkpoint
103            .ok_or_else(|| Error::DataError("Checkpoint not found".to_string()))?;
104
105        self.create_block_response(checkpoint).await
106    }
107
108    async fn get_block_by_hash(&self, hash: BlockHash) -> Result<BlockResponse, Error> {
109        let mut request = GetCheckpointRequest::default().with_read_mask(FieldMask::from_paths([
110            "sequence_number",
111            "digest",
112            "summary.sequence_number",
113            "summary.previous_digest",
114            "summary.timestamp",
115            "transactions.digest",
116            "transactions.transaction.sender",
117            "transactions.transaction.gas_payment",
118            "transactions.transaction.kind",
119            "transactions.effects.gas_object",
120            "transactions.effects.gas_used",
121            "transactions.effects.status",
122            "transactions.balance_changes",
123            "transactions.events.events.event_type",
124            "transactions.events.events.json",
125        ]));
126        request.checkpoint_id = Some(get_checkpoint_request::CheckpointId::Digest(
127            hash.to_string(),
128        ));
129
130        let mut client = self.client.clone();
131        let response = client
132            .ledger_client()
133            .get_checkpoint(request)
134            .await?
135            .into_inner();
136        let checkpoint = response
137            .checkpoint
138            .ok_or_else(|| Error::DataError("Checkpoint not found".to_string()))?;
139
140        self.create_block_response(checkpoint).await
141    }
142
143    async fn current_block(&self) -> Result<BlockResponse, Error> {
144        let request = GetCheckpointRequest::latest()
145            .with_read_mask(FieldMask::from_paths(["sequence_number"]));
146
147        let mut client = self.client.clone();
148        let response = client
149            .ledger_client()
150            .get_checkpoint(request)
151            .await?
152            .into_inner();
153
154        let sequence_number = response.checkpoint().sequence_number();
155        self.get_block_by_index(sequence_number).await
156    }
157
158    async fn genesis_block_identifier(&self) -> Result<BlockIdentifier, Error> {
159        let response = self
160            .client
161            .clone()
162            .ledger_client()
163            .get_service_info(GetServiceInfoRequest::default())
164            .await?
165            .into_inner();
166        let chain_id = response
167            .chain_id
168            .ok_or_else(|| Error::DataError("Missing chain_id".to_string()))?;
169        let hash = CheckpointDigest::from_str(&chain_id)?;
170        Ok(BlockIdentifier { index: 0, hash })
171    }
172
173    async fn oldest_block_identifier(&self) -> Result<BlockIdentifier, Error> {
174        let response = self
175            .client
176            .clone()
177            .ledger_client()
178            .get_service_info(GetServiceInfoRequest::default())
179            .await?
180            .into_inner();
181        let lowest = response
182            .lowest_available_checkpoint
183            .ok_or_else(|| Error::DataError("Missing lowest_available_checkpoint".to_string()))?;
184        self.create_block_identifier(lowest).await
185    }
186
187    async fn current_block_identifier(&self) -> Result<BlockIdentifier, Error> {
188        let request = GetCheckpointRequest::latest()
189            .with_read_mask(FieldMask::from_paths(["sequence_number"]));
190
191        let response = self
192            .client
193            .clone()
194            .ledger_client()
195            .get_checkpoint(request)
196            .await?
197            .into_inner();
198
199        let checkpoint = response
200            .checkpoint
201            .ok_or_else(|| Error::DataError("Missing checkpoint".to_string()))?;
202
203        let sequence_number = checkpoint.sequence_number();
204
205        self.create_block_identifier(sequence_number).await
206    }
207
208    async fn create_block_identifier(
209        &self,
210        checkpoint: CheckpointSequenceNumber,
211    ) -> Result<BlockIdentifier, Error> {
212        self.create_block_identifier(checkpoint).await
213    }
214}

216impl CheckpointBlockProvider {
217    pub fn new(client: GrpcClient, coin_metadata_cache: CoinMetadataCache) -> Self {
218        Self {
219            client,
220            coin_metadata_cache,
221        }
222    }
223
224    async fn create_block_response(&self, checkpoint: Checkpoint) -> Result<BlockResponse, Error> {
225        let summary = checkpoint.summary();
226        let index = summary.sequence_number();
227        let hash = CheckpointDigest::from_str(checkpoint.digest())?;
228        // Genesis checkpoint (index 0) has no previous digest
229        let previous_hash = if index == 0 {
230            hash
231        } else {
232            CheckpointDigest::from_str(summary.previous_digest())?
233        };
234        let timestamp_ms = summary
235            .timestamp
236            .ok_or_else(|| Error::DataError("Checkpoint timestamp is missing".to_string()))
237            .and_then(|ts| {
238                DateTime::from_timestamp(ts.seconds, ts.nanos as u32)
239                    .ok_or_else(|| Error::DataError(format!("Invalid timestamp: {}", ts)))
240            })?
241            .timestamp_millis() as u64;
242
243        let transactions: Vec<Transaction> = stream::iter(checkpoint.transactions)
244            .map(|executed_tx| async move {
245                let digest = TransactionDigest::from_str(executed_tx.digest())?;
246                Ok::<_, Error>(Transaction {
247                    transaction_identifier: TransactionIdentifier { hash: digest },
248                    // This is async because it makes a GetCoinMetadata call if the coin metadata
249                    // isn't already cached.
250                    operations: Operations::try_from_executed_transaction(
251                        executed_tx,
252                        &self.coin_metadata_cache,
253                    )
254                    .await?,
255                    related_transactions: vec![],
256                    metadata: None,
257                })
258            })
259            // A checkpoint can have thousands of transactions so
260            // limit the amount of work a single rosetta request can generate
261            // concurrently to prevent resource starvation issues.
262            .buffer_unordered(10)
263            .try_collect()
264            .await?;
265
266        let parent_block_identifier = if index == 0 {
267            // Genesis block is its own parent
268            BlockIdentifier { index, hash }
269        } else {
270            BlockIdentifier {
271                index: index - 1,
272                hash: previous_hash,
273            }
274        };
275
276        Ok(BlockResponse {
277            block: Block {
278                block_identifier: BlockIdentifier { index, hash },
279                parent_block_identifier,
280                timestamp: timestamp_ms,
281                transactions,
282                metadata: None,
283            },
284            other_transactions: vec![],
285        })
286    }
287
288    async fn create_block_identifier(
289        &self,
290        seq_number: CheckpointSequenceNumber,
291    ) -> Result<BlockIdentifier, Error> {
292        let grpc_request = GetCheckpointRequest::by_sequence_number(seq_number)
293            .with_read_mask(FieldMask::from_paths(["sequence_number", "digest"]));
294        let mut client = self.client.clone();
295        let response = client
296            .ledger_client()
297            .get_checkpoint(grpc_request)
298            .await?
299            .into_inner();
300
301        let checkpoint = response.checkpoint();
302        let index = checkpoint.sequence_number();
303        let hash = checkpoint.digest();
304
305        Ok(BlockIdentifier {
306            index,
307            hash: CheckpointDigest::from_str(hash)?,
308        })
309    }
