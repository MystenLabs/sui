53impl SuiEnv {
54    pub fn check_network_identifier(
55        &self,
56        network_identifier: &NetworkIdentifier,
57    ) -> Result<(), Error> {
58        if &network_identifier.blockchain != "sui" {
59            return Err(Error::UnsupportedBlockchain(
60                network_identifier.blockchain.clone(),
61            ));
62        }
63        if &network_identifier.network != self {
64            return Err(Error::UnsupportedNetwork(network_identifier.network));
65        }
66        Ok(())
67    }
68}

267#[derive(Clone, Debug, Eq, PartialEq)]
268pub struct CoinID {
269    pub id: ObjectID,
270    pub version: SequenceNumber,
271}
272
273impl Serialize for CoinID {
274    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
275    where
276        S: Serializer,
277    {
278        format!("{}:{}", self.id, self.version.value()).serialize(serializer)
279    }
280}
281
282impl<'de> Deserialize<'de> for CoinID {
283    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
284    where
285        D: Deserializer<'de>,
286    {
287        let s = String::deserialize(deserializer)?;
288
289        let (id, version) = s.split_at(
290            s.find(':')
291                .ok_or_else(|| D::Error::custom(format!("Malformed Coin id [{s}].")))?,
292        );
293        let version = version.trim_start_matches(':');
294        let id = ObjectID::from_hex_literal(id).map_err(D::Error::custom)?;
295        let version = SequenceNumber::from_u64(u64::from_str(version).map_err(D::Error::custom)?);
296
297        Ok(Self { id, version })
298    }
299}

301#[test]
302fn test_coin_id_serde() {
303    let id = ObjectID::random();
304    let coin_id = CoinID {
305        id,
306        version: SequenceNumber::from_u64(10),
307    };
308    let s = serde_json::to_string(&coin_id).unwrap();
309    assert_eq!(format!("\"{}:{}\"", id, 10), s);
310
311    let deserialized: CoinID = serde_json::from_str(&s).unwrap();
312
313    assert_eq!(id, deserialized.id);
314    assert_eq!(SequenceNumber::from_u64(10), deserialized.version)
315}

346#[derive(Serialize, Deserialize, Debug)]
347pub struct PublicKey {
348    pub hex_bytes: Hex,
349    pub curve_type: CurveType,
350}
351
352impl From<SuiPublicKey> for PublicKey {
353    fn from(pk: SuiPublicKey) -> Self {
354        match pk {
355            SuiPublicKey::Ed25519(k) => PublicKey {
356                hex_bytes: Hex::from_bytes(&k.0),
357                curve_type: CurveType::Edwards25519,
358            },
359            SuiPublicKey::Secp256k1(k) => PublicKey {
360                hex_bytes: Hex::from_bytes(&k.0),
361                curve_type: CurveType::Secp256k1,
362            },
363            SuiPublicKey::Secp256r1(k) => PublicKey {
364                hex_bytes: Hex::from_bytes(&k.0),
365                curve_type: CurveType::Secp256r1,
366            },
367            SuiPublicKey::ZkLogin(k) => PublicKey {
368                hex_bytes: Hex::from_bytes(&k.0),
369                curve_type: CurveType::ZkLogin, // inaccurate but added for completeness.
370            },
371            SuiPublicKey::Passkey(k) => PublicKey {
372                hex_bytes: Hex::from_bytes(&k.0),
373                curve_type: CurveType::Secp256r1,
374            },
375        }
376    }
377}
378
379impl TryInto<SuiAddress> for PublicKey {
380    type Error = Error;
381
382    fn try_into(self) -> Result<SuiAddress, Self::Error> {
383        let key_bytes = self.hex_bytes.to_vec()?;
384        let pub_key = SuiPublicKey::try_from_bytes(self.curve_type.into(), &key_bytes)?;
385        Ok((&pub_key).into())
386    }
387}

398impl From<CurveType> for SignatureScheme {

430#[derive(Deserialize, Serialize, Copy, Clone, Debug, EnumIter, Eq, PartialEq)]
431pub enum OperationType {
432    // Balance changing operations from TransactionEffect
433    Gas,
434    SuiBalanceChange,
435    StakeReward,
436    StakePrinciple,
437    // sui-rosetta supported operation type
438    PaySui,
439    PayCoin,
440    Stake,
441    WithdrawStake,
442    // All other Sui transaction types, readonly
443    EpochChange,
444    Genesis,
445    ConsensusCommitPrologue,
446    ProgrammableTransaction,
447    AuthenticatorStateUpdate,
448    RandomnessStateUpdate,
449    EndOfEpochTransaction,
450    ProgrammableSystemTransaction,
451    Unknown,
452}
453
454impl From<&TransactionKind> for OperationType {
455    fn from(tx: &TransactionKind) -> Self {
456        match tx.kind.and_then(|k| Kind::try_from(k).ok()) {
457            Some(Kind::ProgrammableTransaction) => OperationType::ProgrammableTransaction,
458            Some(Kind::ChangeEpoch) => OperationType::EpochChange,
459            Some(Kind::Genesis) => OperationType::Genesis,
460            Some(Kind::ConsensusCommitPrologueV1)
461            | Some(Kind::ConsensusCommitPrologueV2)
462            | Some(Kind::ConsensusCommitPrologueV3)
463            | Some(Kind::ConsensusCommitPrologueV4) => OperationType::ConsensusCommitPrologue,
464            Some(Kind::AuthenticatorStateUpdate) => OperationType::AuthenticatorStateUpdate,
465            Some(Kind::RandomnessStateUpdate) => OperationType::RandomnessStateUpdate,
466            Some(Kind::EndOfEpoch) => OperationType::EndOfEpochTransaction,
467            Some(Kind::Unknown) | Some(_) | None => OperationType::Unknown,
468        }
469    }
470}
