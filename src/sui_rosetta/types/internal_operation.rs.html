32pub const MAX_GAS_COINS: usize = 255;
33const MAX_COMMAND_ARGS: usize = 511;
34const MAX_GAS_BUDGET: u64 = 50_000_000_000;
35/// Minimum gas-units a tx might need
36const START_GAS_UNITS: u64 = 1_000;
37
38pub struct TransactionObjectData {
39    pub gas_coins: Vec<ObjectRef>,
40    pub extra_gas_coins: Vec<ObjectRef>,
41    pub objects: Vec<ObjectRef>,
42    /// Refers to the sum of the `Coin<SUI>` balance of the coins participating in the transaction;
43    /// either as gas or as objects.
44    pub total_sui_balance: i128,
45    pub budget: u64,
46}

48#[async_trait]

50pub trait TryConstructTransaction {
51    async fn try_fetch_needed_objects(
52        self,
53        client: &SuiClient,
54        gas_price: Option<u64>,
55        budget: Option<u64>,
56    ) -> Result<TransactionObjectData, Error>;

59#[enum_dispatch(TryConstructTransaction)]
60#[derive(Serialize, Deserialize, Debug, Clone)]
61pub enum InternalOperation {
62    PaySui(PaySui),
63    PayCoin(PayCoin),
64    Stake(Stake),
65    WithdrawStake(WithdrawStake),
66}
67
68impl InternalOperation {
69    pub fn sender(&self) -> SuiAddress {
70        match self {
71            InternalOperation::PaySui(PaySui { sender, .. })
72            | InternalOperation::PayCoin(PayCoin { sender, .. })
73            | InternalOperation::Stake(Stake { sender, .. })
74            | InternalOperation::WithdrawStake(WithdrawStake { sender, .. }) => *sender,
75        }
76    }
77
78    /// Combine with ConstructionMetadata to form the TransactionData
79    pub fn try_into_data(self, metadata: ConstructionMetadata) -> Result<TransactionData, Error> {
80        let pt = match self {
81            Self::PaySui(PaySui {
82                recipients,
83                amounts,
84                ..
85            }) => pay_sui_pt(recipients, amounts, &metadata.extra_gas_coins)?,
86            Self::PayCoin(PayCoin {
87                recipients,
88                amounts,
89                ..
90            }) => {
91                let currency = &metadata
92                    .currency
93                    .ok_or(anyhow!("metadata.coin_type is needed to PayCoin"))?;
94                pay_coin_pt(recipients, amounts, &metadata.objects, currency)?
95            }
96            InternalOperation::Stake(Stake {
97                validator, amount, ..
98            }) => {
99                let (stake_all, amount) = match amount {
100                    Some(amount) => (false, amount),
101                    None => {
102                        if (metadata.total_coin_value - metadata.budget as i128) < 0 {
103                            return Err(anyhow!(
104                                "ConstructionMetadata malformed. total_coin_value - budget < 0"
105                            )
106                            .into());
107                        }
108                        (true, metadata.total_coin_value as u64 - metadata.budget)
109                    }
110                };
111                stake_pt(validator, amount, stake_all, &metadata.extra_gas_coins)?
112            }
113            InternalOperation::WithdrawStake(WithdrawStake { stake_ids, .. }) => {
114                let withdraw_all = stake_ids.is_empty();
115                withdraw_stake_pt(metadata.objects, withdraw_all)?
116            }
117        };
118
119        Ok(TransactionData::new_programmable(
120            metadata.sender,
121            metadata.gas_coins,
122            pt,
123            metadata.budget,
124            metadata.gas_price,
125        ))
126    }
127}

129async fn budget_from_dry_run(
130    client: &SuiClient,
131    pt: ProgrammableTransaction,
132    sender: SuiAddress,
133    gas_price: Option<u64>,
134) -> Result<u64, Error> {
135    let gas_price = match gas_price {
136        Some(p) => p,
137        None => client.governance_api().get_reference_gas_price().await? + 100, // make sure it works over epoch changes
138    };
139    // We don't want dry run to fail due to budget, so we leave coins empty and set MAX_GAS_BUDGET
140    let dry_run = client
141        .read_api()
142        .dry_run_transaction_block(TransactionData::new_programmable(
143            sender,
144            vec![],
145            pt.clone(),
146            MAX_GAS_BUDGET,
147            gas_price,
148        ))
149        .await?;
150    let effects = dry_run.effects;
151
152    if let SuiExecutionStatus::Failure { error } = effects.status() {
153        return Err(Error::TransactionDryRunError(error.to_string()));
154    }
155    // Update budget to be the result of the dry run
156    Ok(effects.gas_cost_summary().computation_cost + effects.gas_cost_summary().storage_cost)
157}

159async fn collect_coins_until_budget_met(
160    client: &SuiClient,
161    sender: SuiAddress,
162    pt: impl Fn(&[(ObjectID, SequenceNumber, ObjectDigest)]) -> anyhow::Result<ProgrammableTransaction>,
163    amount: u64,
164    gas_price: Option<u64>,
165) -> Result<TransactionObjectData, Error> {
166    let mut coins_stream = Box::pin(client.coin_read_api().get_coins_stream(sender, None));
167    // Fetch it once instead of fetching it again and again in the below loop.
168    let gas_price = match gas_price {
169        Some(p) => p,
170        None => client.governance_api().get_reference_gas_price().await? + 100, // make sure it works over epoch changes
171    };
172
173    let mut all_coins = vec![];
174    let mut gas_coins: Vec<_>;
175    let mut extra_gas_coins: Vec<_>;
176    let mut gathered = 0;
177    let mut budget = START_GAS_UNITS * gas_price;
178    // We need to dry-run in a loop, because depending on the amount of coins used the tx might
179    // differ slightly: (merge / no merge / number of merge-coins)
180    loop {
181        while let Some(coin) = coins_stream.next().await {
182            gathered += coin.balance;
183            all_coins.push(coin);
184            if gathered >= amount + budget {
185                break;
186            }
187        }
188        if gathered < amount + budget {
189            return Err(Error::InvalidInput(format!(
190                "Address {sender} does not have amount: {amount} + budget: {budget} balance. SUI balance: {gathered}."
191            )));
192        }
193
194        // The coins to merge should be used as transaction object inputs, as
195        // `TransactionData::new_programmable` used in `InternalOperation::try_into_data`,
196        // uses all coins passed as gas payment.
197        let mut iter = all_coins.iter().map(|c| c.object_ref());
198        gas_coins = iter.by_ref().take(MAX_GAS_COINS).collect();
199        extra_gas_coins = iter.collect();
200        let pt = pt(&extra_gas_coins)?;
201        budget = budget_from_dry_run(client, pt.clone(), sender, Some(gas_price)).await?;
202        // If we have already gathered the needed amount of coins we don't need to dry run again,
203        // as the transaction will be the same.
204        if budget + amount <= gathered {
205            break;
206        }
207    }
208
209    let total_sui_balance = all_coins.iter().map(|c| c.balance).sum::<u64>() as i128;
210    Ok(TransactionObjectData {
211        gas_coins,
212        extra_gas_coins,
213        objects: vec![],
214        total_sui_balance,
215        budget,
216    })
