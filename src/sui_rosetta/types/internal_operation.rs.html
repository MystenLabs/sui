51pub struct TransactionObjectData {
52    pub gas_coins: Vec<ObjectRef>,
53    /// For PaySui/Stake: extra gas coins to merge into gas
54    /// For PayCoin: payment coins of the specified type
55    /// For WithdrawStake: stake objects to withdraw
56    pub objects: Vec<ObjectRef>,
57    /// Party-owned (ConsensusAddress) version of objects
58    pub party_objects: Vec<(ObjectID, SequenceNumber)>,
59    /// Refers to the sum of the `Coin<SUI>` balance of the coins participating in the transaction;
60    /// either as gas or as objects.
61    pub total_sui_balance: i128,
62    pub budget: u64,
63    /// Amount to withdraw from address balance for payment
64    pub address_balance_withdrawal: u64,
65}

78#[enum_dispatch(TryConstructTransaction)]
79#[derive(Serialize, Deserialize, Debug, Clone)]
80pub enum InternalOperation {
81    PaySui(PaySui),
82    PayCoin(PayCoin),
83    Stake(Stake),
84    WithdrawStake(WithdrawStake),
85}
86
87impl InternalOperation {
88    pub fn sender(&self) -> SuiAddress {
89        match self {
90            InternalOperation::PaySui(PaySui { sender, .. })
91            | InternalOperation::PayCoin(PayCoin { sender, .. })
92            | InternalOperation::Stake(Stake { sender, .. })
93            | InternalOperation::WithdrawStake(WithdrawStake { sender, .. }) => *sender,
94        }
95    }
96
97    /// Combine with ConstructionMetadata to form the TransactionData
98    pub fn try_into_data(self, metadata: ConstructionMetadata) -> Result<TransactionData, Error> {
99        let use_addr_balance_gas = metadata.gas_coins.is_empty();
100        let withdrawal = metadata.address_balance_withdrawal;
101        let pt = match self {
102            Self::PaySui(PaySui {
103                sender,
104                recipients,
105                amounts,
106            }) => {
107                let coins = if !metadata.objects.is_empty() {
108                    &metadata.objects
109                } else {
110                    &metadata.extra_gas_coins
111                };
112                if use_addr_balance_gas {
113                    pay_sui_pt_ab_gas(
114                        sender,
115                        recipients,
116                        amounts,
117                        coins,
118                        &metadata.party_objects,
119                        withdrawal,
120                    )?
121                } else {
122                    pay_sui_pt_coin_gas(
123                        recipients,
124                        amounts,
125                        coins,
126                        &metadata.party_objects,
127                        withdrawal,
128                    )?
129                }
130            }
131            Self::PayCoin(PayCoin {
132                sender,
133                recipients,
134                amounts,
135                ..
136            }) => {
137                let currency = &metadata
138                    .currency
139                    .ok_or(anyhow!("metadata.coin_type is needed to PayCoin"))?;
140                pay_coin_pt(
141                    sender,
142                    recipients,
143                    amounts,
144                    &metadata.objects,
145                    &metadata.party_objects,
146                    withdrawal,
147                    currency,
148                )?
149            }
150            InternalOperation::Stake(Stake {
151                sender,
152                validator,
153                amount,
154            }) => {
155                let (stake_all, amount) = match amount {
156                    Some(amount) => (false, amount),
157                    None => {
158                        if (metadata.total_coin_value - metadata.budget as i128) < 0 {
159                            return Err(anyhow!(
160                                "ConstructionMetadata malformed. total_coin_value - budget < 0"
161                            )
162                            .into());
163                        }
164                        (true, metadata.total_coin_value as u64 - metadata.budget)
165                    }
166                };
167                let coins = if !metadata.objects.is_empty() {
168                    &metadata.objects
169                } else {
170                    &metadata.extra_gas_coins
171                };
172                if use_addr_balance_gas {
173                    stake_pt_ab_gas(
174                        sender,
175                        validator,
176                        amount,
177                        stake_all,
178                        coins,
179                        &metadata.party_objects,
180                        withdrawal,
181                    )?
182                } else {
183                    stake_pt_coin_gas(
184                        validator,
185                        amount,
186                        stake_all,
187                        coins,
188                        &metadata.party_objects,
189                        withdrawal,
190                    )?
191                }
192            }
193            InternalOperation::WithdrawStake(WithdrawStake { stake_ids, .. }) => {
194                let withdraw_all = stake_ids.is_empty();
195                withdraw_stake_pt(metadata.objects, withdraw_all)?
196            }
197        };
198
199        if metadata.gas_coins.is_empty() {
200            let chain_id_str = metadata
201                .chain_id
202                .ok_or(anyhow!("chain_id required for address-balance gas"))?;
203            let digest = CheckpointDigest::from_str(&chain_id_str)
204                .map_err(|e| anyhow!("invalid chain_id: {e}"))?;
205            let chain_id = ChainIdentifier::from(digest);
206            let epoch = metadata
207                .epoch
208                .ok_or(anyhow!("epoch required for address-balance gas"))?;
209            let nonce = rand::thread_rng().r#gen::<u32>();
210
211            Ok(TransactionData::new_programmable_with_address_balance_gas(
212                metadata.sender,
213                pt,
214                metadata.budget,
215                metadata.gas_price,
216                chain_id,
217                epoch,
218                nonce,
219            ))
220        } else {
221            Ok(TransactionData::new_programmable(
222                metadata.sender,
223                metadata.gas_coins,
224                pt,
225                metadata.budget,
226                metadata.gas_price,
227            ))
228        }
229    }
230}

232/// Withdraw from address balance as a Coin<T>.
233/// FundsWithdrawal → coin::redeem_funds → Coin<T>
234pub(crate) fn withdraw_coin_from_address_balance(
235    builder: &mut sui_types::programmable_transaction_builder::ProgrammableTransactionBuilder,
236    amount: u64,
237    type_tag: TypeTag,
238) -> anyhow::Result<Argument> {
239    let withdrawal_arg = builder.input(CallArg::FundsWithdrawal(
240        FundsWithdrawalArg::balance_from_sender(amount, type_tag.clone()),
241    ))?;
242
243    let coin = builder.command(Command::move_call(
244        SUI_FRAMEWORK_PACKAGE_ID,
245        Identifier::new("coin")?,
246        Identifier::new("redeem_funds")?,
247        vec![type_tag],
248        vec![withdrawal_arg],
249    ));
250
251    Ok(coin)
252}

254/// RPC auto-selects gas coins if empty, uses reference gas price if None, and estimates budget if None.
255/// Returns the resolved budget and gas coins used by the transaction.
256async fn simulate_transaction(
257    client: &mut Client,
258    pt: ProgrammableTransaction,
259    sender: SuiAddress,
260    gas_coins: Vec<ObjectRef>,
261    gas_price: Option<u64>,
262    budget: Option<u64>,
263) -> Result<(u64, Vec<Object>), Error> {
264    let ptb_proto: ProtoProgrammableTransaction = pt.into();
265    let mut transaction = Transaction::default()
266        .with_kind(
267            TransactionKind::default()
268                .with_programmable_transaction(ptb_proto)
269                .with_kind(transaction_kind::Kind::ProgrammableTransaction),
270        )
271        .with_sender(sender.to_string());
272
273    let mut gas_payment = GasPayment::default();
274    gas_payment.objects = gas_coins
275        .into_iter()
276        .map(|gas_ref| {
277            let mut obj_ref = ObjectReference::default();
278            obj_ref.object_id = Some(gas_ref.0.to_string());
279            obj_ref.version = Some(gas_ref.1.value());
280            obj_ref.digest = Some(gas_ref.2.to_string());
281            obj_ref
282        })
283        .collect();
284    gas_payment.budget = budget;
285    gas_payment.price = gas_price;
286    gas_payment.owner = Some(sender.to_string());
287    transaction.gas_payment = Some(gas_payment);
288
289    let request = SimulateTransactionRequest::default()
290        .with_transaction(transaction)
291        .with_read_mask(FieldMask::from_paths([
292            "transaction.effects.status",
293            "transaction.transaction.gas_payment",
294        ]))
295        .with_checks(TransactionChecks::Enabled)
296        .with_do_gas_selection(true);
297
298    let response = client
299        .execution_client()
300        .simulate_transaction(request)
301        .await?
302        .into_inner();
303
304    let executed_tx = response.transaction();
305    let effects = executed_tx.effects();
306    if !effects.status().success() {
307        return Err(Error::TransactionDryRunError(Box::new(
308            effects.status().error().clone(),
309        )));
310    }
311
312    let resolved_tx = executed_tx.transaction();
313    let gas_payment = resolved_tx.gas_payment();
314
315    // When gas_payment has no objects, the transaction uses address-balance gas.
316    // Skip the batch fetch and return empty gas coins to signal this.
317    let gas_objects = gas_payment.objects();
318    if gas_objects.is_empty() {
319        return Ok((gas_payment.budget(), vec![]));
320    }
321
322    let mut batch_request =
323        BatchGetObjectsRequest::default().with_read_mask(FieldMask::from_paths([
324            "object_id",
325            "version",
326            "digest",
327            "balance",
328        ]));
329
330    for obj_ref in gas_objects {
331        let get_request = GetObjectRequest::default()
332            .with_object_id(obj_ref.object_id().to_string())
333            .with_version(obj_ref.version());
334        batch_request.requests.push(get_request);
335    }
336
337    let batch_response = client
338        .ledger_client()
339        .batch_get_objects(batch_request)
340        .await?
341        .into_inner();
342
343    let mut gas_coins = Vec::new();
344    for result in batch_response.objects {
345        match result.result {
346            Some(get_object_result::Result::Object(obj)) => {
347                gas_coins.push(obj);
348            }
349            Some(get_object_result::Result::Error(err)) => {
350                return Err(Error::DataError(format!(
351                    "Failed to fetch gas coin object: {:?}",
352                    err
353                )));
354            }
355            None => {
356                return Err(Error::DataError(
357                    "Failed to fetch gas coin object: no result returned".to_string(),
358                ));
359            }
360            Some(_) => {
361                return Err(Error::DataError(
362                    "Failed to fetch gas coin object: unexpected result type".to_string(),
363                ));
364            }
365        }
366    }
367
368    Ok((gas_payment.budget(), gas_coins))
369}
