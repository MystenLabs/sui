4//! `MergeAndRedeemFungibleStakedSui` construction logic.
5//!
6//! Translates a user's `(validator, amount, redeem_mode)` request into a PTB
7//! that consumes the sender's `FungibleStakedSui` for the named validator's
8//! pool and produces liquid SUI.
9//!
10//! The semantics differ by mode:
11//!
12//! * `All` — redeem every FSS the sender owns for the pool.
13//! * `AtLeast(amount)` — choose the minimum pool-token count whose redemption
14//!   yields at least `amount` MIST. The PTB also installs a runtime
15//!   `balance::split` guard so a slight under-delivery aborts the transaction
16//!   on chain instead of silently shorting the user.
17//! * `AtMost(amount)` — choose the maximum pool-token count whose chain-side
18//!   upper bound `expected = floor(token * sui_balance / pool_token_balance)`
19//!   stays at or below `amount`. The pool's invariant
20//!   `actual <= expected` then guarantees the cap holds at execution time.
21//!   This is conservative: actual SUI may fall a few MIST short of the cap.
22//!
23//! Amount-sensitive plans (`AtLeast` / `AtMost`) bind the resulting transaction
24//! to the quote epoch via `TransactionExpiration`, so a stale quote cannot
25//! execute against a different epoch's exchange rate.
26

61/// Snapshot of the redeem-relevant state of a staking pool's
62/// `FungibleStakedSuiData` plus the latest exchange rate, sufficient to mirror
63/// the chain's payout formula in pure Rust.
64#[derive(Clone, Debug)]
65pub(crate) struct PoolRedeemData {
66    /// Latest `exchange_rate.sui_amount` (= pool.sui_balance within an epoch).
67    pub rate_sui: u64,
68    /// Latest `exchange_rate.pool_token_amount` (= pool.pool_token_balance).
69    pub rate_token: u64,
70    /// `FungibleStakedSuiData.principal.value()` — current Balance<SUI> in the
71    /// FSS sub-pool. Mutates within an epoch as other users convert/redeem.
72    pub fss_principal: u64,
73    /// `FungibleStakedSuiData.total_supply` — total outstanding FSS pool tokens.
74    pub fss_total_supply: u64,
75}

77/// `floor(a * b / c)` with checked u64 overflow. Move's `mul_div!` aborts on
78/// overflow; this helper returns `Err(DataError)` to match — silently truncating
79/// to `u64` would let the off-chain mirror disagree with the chain in pathological
80/// rate scenarios.
81fn mul_div_u64(a: u64, b: u64, c: u64) -> Result<u64, Error> {
82    if c == 0 {
83        return Err(Error::DataError("mul_div_u64: divide by zero".into()));
84    }
85    let v = (a as u128) * (b as u128) / (c as u128);
86    u64::try_from(v).map_err(|_| {
87        Error::DataError(format!(
88            "redeem math overflow: floor({a} * {b} / {c}) does not fit in u64"
89        ))
90    })
91}

93/// Mirror of `0x3::staking_pool::PoolTokenExchangeRate::get_sui_amount` at
94/// `staking_pool.move:638-646`. Returns the SUI cap for `token_amount` pool
95/// tokens given an exchange rate of `(rate_sui, rate_token)`.
96///
97/// The chain enforces the invariant
98/// `actual_redeemed <= floor(token * rate_sui / rate_token)` even after the
99/// per-redeemer split-floor accounting in `calculate_fungible_staked_sui_withdraw_amount`
100/// (see `staking_pool.move:264-268`), so any token count satisfying
101/// `expected_sui_amount(token) <= max_sui` will stay within the cap at
102/// execution time regardless of intra-epoch pool drift.
103pub(crate) fn expected_sui_amount(
104    rate_sui: u64,
105    rate_token: u64,
106    token_amount: u64,
107) -> Result<u64, Error> {
108    if rate_sui == 0 || rate_token == 0 {
109        return Ok(token_amount);
110    }
111    mul_div_u64(token_amount, rate_sui, rate_token)
112}

114/// Mirror of `0x3::staking_pool::calculate_fungible_staked_sui_withdraw_amount`
115/// at `staking_pool.move:231-271`. Returns the actual SUI the chain will
116/// deliver for `token_amount` FSS pool tokens given the supplied pool state.
117///
118/// The chain splits the payout into a principal share and a rewards share, each
119/// with an independent `floor`, so `actual <= expected` and the gap is bounded
120/// by ~2 MIST per redemption (one MIST per floor, scaled by `token/supply`).
121/// AtLeast selection must use this (not `expected_sui_amount`) to avoid picking
122/// a token count whose actual payout falls below the user's `min_sui`.
123///
124/// Returns `Err(DataError)` if any intermediate `mul_div` overflows u64 — the
125/// chain's `mul_div!` aborts on overflow, so the mirror surfaces the same
126/// failure mode rather than silently truncating.
127pub(crate) fn mirror_redeem_actual(data: &PoolRedeemData, token_amount: u64) -> Result<u64, Error> {
128    if data.fss_total_supply == 0 {
129        return Ok(0);
130    }
131
132    // total_sui = exchange_rate.get_sui_amount(total_supply)
133    let total_sui = if data.rate_sui == 0 || data.rate_token == 0 {
134        data.fss_total_supply
135    } else {
136        mul_div_u64(data.rate_sui, data.fss_total_supply, data.rate_token)?
137    };
138
139    let principal = data.fss_principal.min(total_sui);
140    let rewards = total_sui - principal;
141
142    let principal_out = mul_div_u64(token_amount, principal, data.fss_total_supply)?;
143    let rewards_out = mul_div_u64(token_amount, rewards, data.fss_total_supply)?;
144
145    principal_out
146        .checked_add(rewards_out)
147        .ok_or_else(|| Error::DataError("redeem math overflow: principal_out + rewards_out".into()))
148}

150/// Binary-search the smallest `token_amount` in `[1, total_tokens]` such that
151/// `mirror_redeem_actual(data, token_amount) >= target_sui`.
152///
153/// Returns `Ok(None)` if even redeeming `total_tokens` produces less than
154/// `target_sui` (caller reports "Insufficient FSS balance"). Returns
155/// `Err` only if the underlying mirror math would overflow u64.
156pub(crate) fn binary_search_at_least(
157    data: &PoolRedeemData,
158    total_tokens: u64,
159    target_sui: u64,
160) -> Result<Option<u64>, Error> {
161    if total_tokens == 0 {
162        return Ok(None);
163    }
164    if mirror_redeem_actual(data, total_tokens)? < target_sui {
165        return Ok(None);
166    }
167    let mut lo: u64 = 1;
168    let mut hi: u64 = total_tokens;
169    while lo < hi {
170        let mid = lo + (hi - lo) / 2;
171        if mirror_redeem_actual(data, mid)? >= target_sui {
172            hi = mid;
173        } else {
174            lo = mid + 1;
175        }
176    }
177    Ok(Some(lo))
178}

180/// Binary-search the largest `token_amount` in `[1, total_tokens]` such that
181/// `expected_sui_amount(rate_sui, rate_token, token_amount) <= max_sui`.
182///
183/// Returns `Ok(None)` if even one token would already exceed `max_sui`
184/// (caller reports "AtMost amount too small to redeem any tokens"). Returns
185/// `Err` only if the underlying mirror math would overflow u64.
186pub(crate) fn binary_search_at_most(
187    rate_sui: u64,
188    rate_token: u64,
189    total_tokens: u64,
190    max_sui: u64,
191) -> Result<Option<u64>, Error> {
192    if total_tokens == 0 {
193        return Ok(None);
194    }
195    if expected_sui_amount(rate_sui, rate_token, 1)? > max_sui {
196        return Ok(None);
197    }
198    let mut lo: u64 = 1;
199    let mut hi: u64 = total_tokens;
200    let mut ans: Option<u64> = None;
201    while lo <= hi {
202        let mid = lo + (hi - lo) / 2;
203        if expected_sui_amount(rate_sui, rate_token, mid)? <= max_sui {
204            ans = Some(mid);
205            if mid == u64::MAX {
206                break;
207            }
208            lo = mid + 1;
209        } else {
210            if mid == 0 {
211                break;
212            }
213            hi = mid - 1;
214        }
215    }
216    Ok(ans)
217}

223/// Information extracted from a single owned `FungibleStakedSui` object.

225struct OwnedFss {
226    object_ref: ObjectRef,
227    pool_id: ObjectID,
228    value: u64,
229}

231async fn list_owned_fss(client: &mut Client, sender: SuiAddress) -> Result<Vec<OwnedFss>, Error> {
232    let request = ListOwnedObjectsRequest::default()
233        .with_owner(sender.to_string())
234        .with_object_type("0x3::staking_pool::FungibleStakedSui".to_string())
235        .with_page_size(1000u32)
236        .with_read_mask(FieldMask::from_paths([
237            "object_id",
238            "version",
239            "digest",
240            "contents",
241        ]));
242    let objects: Vec<_> = client
243        .list_owned_objects(request)
244        .map_err(Error::from)
245        .try_collect()
246        .await?;
247
248    let mut out = Vec::with_capacity(objects.len());
249    for obj in &objects {
250        let contents = obj
251            .contents
252            .as_ref()
253            .ok_or_else(|| Error::DataError("FungibleStakedSui missing contents".to_string()))?;
254        let fss: FungibleStakedSuiBcs = contents.deserialize().map_err(|e| {
255            Error::DataError(format!("Failed to deserialize FungibleStakedSui: {e}"))
256        })?;
257        let pool_id = ObjectID::from_str(&fss.pool_id.to_string())
258            .map_err(|e| Error::DataError(format!("Invalid pool_id: {e}")))?;
259        let object_id = ObjectID::from_str(obj.object_id())
260            .map_err(|e| Error::DataError(format!("Invalid object_id: {e}")))?;
261        let object_ref: ObjectRef = (
262            object_id,
263            obj.version().into(),
264            obj.digest()
265                .parse()
266                .map_err(|e| Error::DataError(format!("Invalid digest: {e}")))?,
267        );
268        out.push(OwnedFss {
269            object_ref,
270            pool_id,
271            value: fss.value,
272        });
273    }
274    Ok(out)
275}

277/// Resolve `validator` to a pool, the sender's FSS in that pool, and the
278/// `(rate_sui, rate_token, epoch)` snapshot. Supports both active and inactive
279/// validator pools — the chain's `validator_set::redeem_fungible_staked_sui`
280/// (validator_set.move:346-364) routes through `staking_pool_mappings` for
281/// active pools and falls back to `inactive_validators[pool_id]` otherwise,
282/// so Rosetta must accept the same set.
283///
284/// FSS-first: enumerate sender's FSS, group by pool_id, then for each candidate
285/// pool look up the validator address (active fast path, inactive_validators
286/// dynamic field walk fallback). The first pool whose validator address matches
287/// `target_validator` is selected.
288///
289/// **Atomicity is load-bearing**: amount-sensitive plans bind the resulting
290/// transaction to `epoch`, so the rate must come from the same RPC response
291/// as the inactive-table lookup. Otherwise an epoch transition between the
292/// rate read and the bind_epoch read could leave the quote pinned to a stale
293/// rate, silently violating AtMost caps and aborting AtLeast guards.
294async fn resolve_pool_fss_and_rate(
295    client: &mut Client,
296    sender: SuiAddress,
297    target_validator: SuiAddress,
298) -> Result<PoolFssRateSnapshot, Error> {
299    let owned = list_owned_fss(client, sender).await?;
300    if owned.is_empty() {
301        return Err(Error::InvalidInput(format!(
302            "No FungibleStakedSui found for sender {sender}"
303        )));
304    }
305
306    let snapshot = crate::account::get_validator_set_snapshot(client).await?;
307    let target_validator_addr = sui_sdk_types::Address::from(target_validator);
308
309    // Group FSS by pool_id; deterministic ordering (BTreeMap on ObjectID).
310    let mut by_pool: std::collections::BTreeMap<ObjectID, Vec<&OwnedFss>> =
311        std::collections::BTreeMap::new();
312    for fss in &owned {
313        by_pool.entry(fss.pool_id).or_default().push(fss);
314    }
315
316    for (pool_id, fss_list) in &by_pool {
317        // Active fast path.
318        if let Some(rate_info) = snapshot.active_rates.get(&pool_id.to_string())
319            && rate_info.validator_address == target_validator_addr
320        {
321            return Ok(build_snapshot(fss_list, rate_info, snapshot.epoch));
322        }
323
324        // Inactive fallback.
325        if let Some(table_id_str) = snapshot.inactive_validators_table_id.as_deref() {
326            let table_id = ObjectID::from_str(table_id_str).map_err(|e| {
327                Error::DataError(format!("Invalid inactive_validators table id: {e}"))
328            })?;
329            if let Some(inactive) =
330                lookup_inactive_pool(client, table_id, *pool_id, target_validator).await?
331            {
332                let (rate_sui, rate_token) = fetch_pool_exchange_rate_at_epoch(
333                    client,
334                    inactive.exchange_rates_table_id,
335                    inactive.activation_epoch,
336                    inactive.deactivation_epoch,
337                    snapshot.epoch,
338                )
339                .await?;
340                let mut sorted = fss_list.clone();
341                sorted.sort_by_key(|f| f.object_ref.0);
342                let total_tokens: u64 = sorted.iter().map(|f| f.value).sum();
343                let fss_refs: Vec<ObjectRef> = sorted.iter().map(|f| f.object_ref).collect();
344                return Ok(PoolFssRateSnapshot {
345                    fss_refs,
346                    total_tokens,
347                    rate_sui,
348                    rate_token,
349                    epoch: snapshot.epoch,
350                    pool_extra_fields_id: Some(inactive.pool_extra_fields_id),
351                });
352            }
353        }
354    }
355
356    Err(Error::InvalidInput(format!(
357        "No FungibleStakedSui found for validator {target_validator}'s pool \
358         (sender holds {} FSS object(s) but none are for that validator's active \
359          or inactive pool)",
360        owned.len()
361    )))
362}

364fn build_snapshot(
365    fss_list: &[&OwnedFss],
366    rate_info: &crate::account::PoolRateInfo,
367    epoch: u64,
368) -> PoolFssRateSnapshot {
369    let mut sorted = fss_list.to_vec();
370    sorted.sort_by_key(|f| f.object_ref.0);
371    let total_tokens: u64 = sorted.iter().map(|f| f.value).sum();
372    let fss_refs: Vec<ObjectRef> = sorted.iter().map(|f| f.object_ref).collect();
373    let pool_extra_fields_id = rate_info
374        .pool_extra_fields_id
375        .as_deref()
376        .and_then(|s| ObjectID::from_str(s).ok());
377    PoolFssRateSnapshot {
378        fss_refs,
379        total_tokens,
380        rate_sui: rate_info.sui_balance,
381        rate_token: rate_info.pool_token_balance,
382        epoch,
383        pool_extra_fields_id,
384    }
385}

387/// Pool data extracted from an inactive validator's `ValidatorWrapper`.
388///
389/// The on-chain redeem path (`staking_pool.move:205`) calls
390/// `pool.pool_token_exchange_rate_at_epoch(ctx.epoch())` rather than reading
391/// `pool.sui_balance / pool_token_balance` directly. For inactive pools these
392/// live fields can drift from the exchange-rate snapshot when other users do
393/// regular `request_withdraw_stake` (StakedSui, not FSS) on the deactivated
394/// pool — `staking_pool.move:187-188` immediately processes the pending
395/// withdrawal for inactive/preactive pools, which mutates `sui_balance` and
396/// `pool_token_balance`. So we extract the table id and walk it the same way
397/// the chain does.
398struct InactivePoolData {
399    activation_epoch: Option<u64>,
400    deactivation_epoch: Option<u64>,
401    exchange_rates_table_id: ObjectID,
402    pool_extra_fields_id: ObjectID,
403}

405/// Walk `inactive_validators[pool_id]` → `ValidatorWrapper` → `Versioned` →
406/// `ValidatorV1` to find the validator with matching address and extract the
407/// pool's exchange-rate table id (plus activation/deactivation epochs needed
408/// for the walk-back) and the FSS data Bag id.
409///
410/// The chain stores `inactive_validators` as `Table<ID, ValidatorWrapper>`
411/// (validator_set.move:73). Each entry is a dynamic field at
412/// `derive(table.id, &TypeTag::ID, bcs(pool_id))`. The wrapper holds a
413/// `Versioned` whose latest value is the actual `ValidatorV1`, stored at
414/// `derive(versioned.id, &TypeTag::U64, bcs(version))`.
415///
416/// Returns `None` if the wrapper at the derived id resolves to a validator
417/// whose address doesn't match (caller can try the next candidate pool).
418async fn lookup_inactive_pool(
419    client: &mut Client,
420    inactive_table_id: ObjectID,
421    pool_id: ObjectID,
422    expected_validator: SuiAddress,
423) -> Result<Option<InactivePoolData>, Error> {
424    use sui_types::dynamic_field::{Field, derive_dynamic_field_id};
425    use sui_types::id::ID;
426    use sui_types::sui_system_state::ValidatorWrapper;
427    use sui_types::sui_system_state::sui_system_state_inner_v1::ValidatorV1;
428
429    let id_type = sui_types::TypeTag::from_str("0x2::object::ID")
430        .map_err(|e| Error::DataError(format!("ID TypeTag: {e}")))?;
431    let pool_id_bcs =
432        bcs::to_bytes(&pool_id).map_err(|e| Error::DataError(format!("pool_id bcs: {e}")))?;
433    let wrapper_field_id = derive_dynamic_field_id(inactive_table_id, &id_type, &pool_id_bcs)
434        .map_err(|e| Error::DataError(format!("derive inactive wrapper field id: {e}")))?;
435
436    let Some(wrapper_bytes) = try_read_object_contents(client, wrapper_field_id).await? else {
437        return Ok(None);
438    };
439    let wrapper_field: Field<ID, ValidatorWrapper> = bcs::from_bytes(&wrapper_bytes)
440        .map_err(|e| Error::DataError(format!("Field<ID,ValidatorWrapper> decode: {e}")))?;
441    let versioned_id = wrapper_field.value.inner.id.id.bytes;
442    let version = wrapper_field.value.inner.version;
443
444    let u64_type = sui_types::TypeTag::U64;
445    let version_bcs =
446        bcs::to_bytes(&version).map_err(|e| Error::DataError(format!("version bcs: {e}")))?;
447    let validator_field_id = derive_dynamic_field_id(versioned_id, &u64_type, &version_bcs)
448        .map_err(|e| Error::DataError(format!("derive ValidatorV1 field id: {e}")))?;
449
450    let validator_bytes = try_read_object_contents(client, validator_field_id)
451        .await?
452        .ok_or_else(|| {
453            Error::DataError(format!(
454                "Inactive validator wrapper at {wrapper_field_id} points at \
455                 missing ValidatorV1 at {validator_field_id} (version {version})"
456            ))
457        })?;
458    let validator_field: Field<u64, ValidatorV1> = bcs::from_bytes(&validator_bytes)
459        .map_err(|e| Error::DataError(format!("Field<u64,ValidatorV1> decode: {e}")))?;
460    let validator = validator_field.value;
461    if validator.metadata.sui_address != expected_validator {
462        return Ok(None);
463    }
464
465    Ok(Some(InactivePoolData {
466        activation_epoch: validator.staking_pool.activation_epoch,
467        deactivation_epoch: validator.staking_pool.deactivation_epoch,
468        exchange_rates_table_id: validator.staking_pool.exchange_rates.id,
469        pool_extra_fields_id: validator.staking_pool.extra_fields.id.id.bytes,
470    }))
471}

473/// Mirror of `staking_pool::pool_token_exchange_rate_at_epoch`
474/// (`staking_pool.move:587-608`). Returns `(sui_amount, pool_token_amount)`
475/// of the exchange rate the chain would use when redeeming at `current_epoch`.
476///
477/// For deactivated pools the lookup epoch is clamped to `deactivation_epoch`
478/// (rates aren't recorded after deactivation; the last recorded rate is the
479/// one the chain replays). The walk steps backward from the clamped epoch
480/// down to `activation_epoch` and returns the first `exchange_rates[epoch]`
481/// entry it finds. If none is found (or the pool is preactive) the chain
482/// returns `initial_exchange_rate()` which `get_sui_amount` then treats as
483/// 1:1 — we surface the same `(0, 0)` here so downstream `expected_sui_amount`
484/// applies the same fallback.
485async fn fetch_pool_exchange_rate_at_epoch(
486    client: &mut Client,
487    table_id: ObjectID,
488    activation_epoch: Option<u64>,
489    deactivation_epoch: Option<u64>,
490    current_epoch: u64,
491) -> Result<(u64, u64), Error> {
492    let candidates = walk_back_epochs(activation_epoch, deactivation_epoch, current_epoch);
493    for epoch in candidates {
494        if let Some((sui_amount, pool_token_amount)) =
495            try_read_exchange_rate(client, table_id, epoch).await?
496        {
497            return Ok((sui_amount, pool_token_amount));
498        }
499    }
500    Ok((0, 0))
501}

503/// Pure-function half of the walk-back: given the pool's activation /
504/// deactivation epochs and the current epoch, produce the ordered list of
505/// epochs to probe, matching `pool_token_exchange_rate_at_epoch` semantics.
506///
507/// * Preactive pool (activation > current, or never activated) → empty list:
508///   the on-chain `is_preactive_at_epoch` short-circuits to
509///   `initial_exchange_rate()`.
510/// * Otherwise clamp to `min(deactivation.unwrap_or(current), current)` and
511///   walk down to `activation`.
512fn walk_back_epochs(
513    activation_epoch: Option<u64>,
514    deactivation_epoch: Option<u64>,
515    current_epoch: u64,
516) -> Vec<u64> {
517    let Some(activation) = activation_epoch else {
518        return Vec::new();
519    };
520    if activation > current_epoch {
521        return Vec::new();
522    }
523    let clamped = deactivation_epoch
524        .map(|d| d.min(current_epoch))
525        .unwrap_or(current_epoch);
526    let start = clamped.max(activation);
527    (activation..=start).rev().collect()
528}

530/// BCS layout of `0x3::staking_pool::PoolTokenExchangeRate`. Mirrors the Move
531/// struct at `staking_pool.move:69-72`. We can't use `sui_types`'s
532/// `PoolTokenExchangeRate` directly because its fields are private; the BCS
533/// layout (positional u64 / u64) is the public interface.
534#[derive(Deserialize, Debug)]
535struct PoolTokenExchangeRateBcs {

540/// Direct dynamic-field lookup of `exchange_rates[epoch]`. Returns
541/// `Ok(None)` if the entry doesn't exist at the derived id (no rate was
542/// recorded for that epoch — caller walks back to the previous one).
543async fn try_read_exchange_rate(
544    client: &mut Client,
545    table_id: ObjectID,
546    epoch: u64,
547) -> Result<Option<(u64, u64)>, Error> {
548    use sui_types::dynamic_field::{Field, derive_dynamic_field_id};
549
550    let key_bcs = bcs::to_bytes(&epoch)
551        .map_err(|e| Error::DataError(format!("exchange_rates key bcs: {e}")))?;
552    let field_id = derive_dynamic_field_id(table_id, &sui_types::TypeTag::U64, &key_bcs)
553        .map_err(|e| Error::DataError(format!("derive exchange_rates[{epoch}] field id: {e}")))?;
554
555    let Some(bytes) = try_read_object_contents(client, field_id).await? else {
556        return Ok(None);
557    };
558    let field: Field<u64, PoolTokenExchangeRateBcs> = bcs::from_bytes(&bytes).map_err(|e| {
559        Error::DataError(format!(
560            "Field<u64,PoolTokenExchangeRate> decode at epoch {epoch}: {e}"
561        ))
562    })?;
563    Ok(Some((
564        field.value.sui_amount,
565        field.value.pool_token_amount,
566    )))
567}

569/// Fetch raw BCS contents of an object, returning `None` if the object does
570/// not exist (so callers can probe derived addresses without erroring).
571async fn try_read_object_contents(
572    client: &mut Client,
573    object_id: ObjectID,
574) -> Result<Option<Vec<u8>>, Error> {
575    let request = sui_rpc::proto::sui::rpc::v2::GetObjectRequest::default()
576        .with_object_id(object_id.to_string())
577        .with_read_mask(FieldMask::from_paths(["contents"]));
578    let response = match client.ledger_client().get_object(request).await {
579        Ok(r) => r.into_inner(),
580        Err(status) if status.code() == tonic::Code::NotFound => return Ok(None),
581        Err(e) => return Err(Error::from(e)),
582    };
583    let Some(obj) = response.object else {
584        return Ok(None);
585    };
586    let bytes = obj
587        .contents
588        .as_ref()
589        .and_then(|b| b.value.as_deref())
590        .map(|v| v.to_vec())
591        .unwrap_or_default();
592    if bytes.is_empty() {
593        Ok(None)
594    } else {
595        Ok(Some(bytes))
596    }
597}

613/// Subset of `0x3::staking_pool::FungibleStakedSuiData` we BCS-deserialize.
614/// Field order MUST match the Move source at `staking_pool.move:98-104`:
615///   `id: UID, total_supply: u64, principal: Balance<SUI>`
616/// where `UID` is a 32-byte address and `Balance<SUI>` is a `{ value: u64 }`
617/// struct (stored positionally as a u64).
618#[derive(Deserialize, Debug)]
619struct FungibleStakedSuiDataBcs {
620    _id: sui_sdk_types::Address,
621    total_supply: u64,
622    principal: BalanceValueBcs,
623}

625#[derive(Deserialize, Debug)]
626struct BalanceValueBcs {
627    value: u64,
628}

630/// Fetch the `FungibleStakedSuiData` stored at
631/// `pool.extra_fields[FungibleStakedSuiDataKey {}]`. Lists the bag's dynamic
632/// fields and finds the one whose `value_type` matches
633/// `<SUI_SYSTEM_PACKAGE>::staking_pool::FungibleStakedSuiData`, then BCS-
634/// deserializes the value into `(principal_value, total_supply)`.
635///
636/// We list rather than derive the field id locally to avoid coupling to the
637/// dynamic-field hashing rules (which would need a faithful Move-side
638/// `bcs::to_bytes(&FungibleStakedSuiDataKey {})` and TypeTag canonicalization).
639/// The bag normally holds a small number of entries (currently just FSS data),
640/// so the cost is bounded.
641async fn fetch_fss_data(client: &mut Client, bag_id: ObjectID) -> Result<(u64, u64), Error> {
642    let want_type = format!(
643        "{}::staking_pool::FungibleStakedSuiData",
644        SUI_SYSTEM_PACKAGE_ID
645    );
646
647    let request = ListDynamicFieldsRequest::default()
648        .with_parent(bag_id.to_string())
649        .with_page_size(50u32)
650        .with_read_mask(FieldMask::from_paths(["value_type", "value"]));
651    let dynamic_fields: Vec<_> = client
652        .list_dynamic_fields(request)
653        .map_err(Error::from)
654        .try_collect()
655        .await?;
656
657    for df in dynamic_fields {
658        let Some(value_type) = df.value_type.as_deref() else {
659            continue;
660        };
661        if !type_tags_match(value_type, &want_type) {
662            continue;
663        }
664        let bytes = df
665            .value
666            .as_ref()
667            .and_then(|b| b.value.as_deref())
668            .ok_or_else(|| {
669                Error::DataError("FungibleStakedSuiData entry has empty value".to_string())
670            })?;
671        let data: FungibleStakedSuiDataBcs = bcs::from_bytes(bytes)
672            .map_err(|e| Error::DataError(format!("FungibleStakedSuiData decode: {e}")))?;
673        return Ok((data.principal.value, data.total_supply));
674    }
675
676    Err(Error::DataError(format!(
677        "No FungibleStakedSuiData entry found in pool extra_fields bag {bag_id} \
678         (validator has no FSS yet — but sender holds FSS, this should not happen)"
679    )))
680}

682/// Compare two Move type tag strings ignoring address normalization differences
683/// (`0x3::...` vs `0x0000...3::...`). Both are valid canonical forms.
684fn type_tags_match(a: &str, b: &str) -> bool {
685    use sui_types::TypeTag;
686    match (TypeTag::from_str(a), TypeTag::from_str(b)) {
687        (Ok(ta), Ok(tb)) => ta == tb,
688        _ => a == b,
689    }
690}

692#[async_trait]
693impl TryConstructTransaction for MergeAndRedeemFungibleStakedSui {
694    async fn try_fetch_needed_objects(
695        self,
696        client: &mut Client,
697        gas_price: Option<u64>,
698        budget: Option<u64>,
699    ) -> Result<TransactionObjectData, Error> {
700        let Self {
701            sender,
702            validator,
703            amount,
704            redeem_mode,
705        } = self;
706
707        let snapshot = resolve_pool_fss_and_rate(client, sender, validator).await?;
708        let PoolFssRateSnapshot {
709            fss_refs,
710            total_tokens,
711            rate_sui,
712            rate_token,
713            epoch,
714            pool_extra_fields_id,
715        } = snapshot;
716
717        // AtLeast needs the FSS sub-pool's principal+total_supply to mirror the
718        // chain's split-floor payout formula and pick the minimum token count
719        // whose actual payout >= min_sui. Fetched lazily so All/AtMost paths
720        // skip the extra GetObject.
721        let fss_data = if matches!(redeem_mode, RedeemMode::AtLeast) {
722            let bag_id = pool_extra_fields_id.ok_or_else(|| {
723                Error::DataError(
724                    "Staking pool extra_fields.id missing from system state response — \
725                     cannot locate FungibleStakedSuiData for AtLeast mode"
726                        .to_string(),
727                )
728            })?;
729            let (principal, total_supply) = fetch_fss_data(client, bag_id).await?;
730            Some(PoolRedeemData {
731                rate_sui,
732                rate_token,
733                fss_principal: principal,
734                fss_total_supply: total_supply,
735            })
736        } else {
737            None
738        };
739
740        let plan = build_redeem_plan(
741            redeem_mode,
742            amount,
743            total_tokens,
744            rate_sui,
745            rate_token,
746            fss_data.as_ref(),
747        )?;
748        // Amount-sensitive plans bind to `epoch` so the chain executes against
749        // the same exchange rate the off-chain quote used. `epoch` here comes
750        // from the same RPC response as `(rate_sui, rate_token)`, so there is
751        // no race between rate read and epoch read.
752        let bind_epoch = match plan {
753            RedeemPlan::All => None,
754            RedeemPlan::AtLeast { .. } | RedeemPlan::AtMost { .. } => Some(epoch),
755        };
756
757        let pt = merge_and_redeem_fss_pt(sender, fss_refs.clone(), &plan)?;
758        let (budget, gas_coin_objs) =
759            simulate_transaction(client, pt, sender, vec![], gas_price, budget).await?;
760
761        let total_sui_balance = gas_coin_objs.iter().map(|c| c.balance()).sum::<u64>() as i128;
762        let gas_coins = gas_coin_objs
763            .iter()
764            .map(|obj| obj.object_reference().try_to_object_ref())
765            .collect::<Result<Vec<_>, _>>()?;
766
767        let redeem_token_amount = match plan {
768            RedeemPlan::All => None,
769            RedeemPlan::AtLeast { token_amount, .. } | RedeemPlan::AtMost { token_amount, .. } => {
770                token_amount
771            }
772        };
773
774        Ok(TransactionObjectData {
775            gas_coins,
776            objects: fss_refs,
777            party_objects: vec![],
778            total_sui_balance,
779            budget,
780            address_balance_withdrawal: 0,
781            fss_object_count: None,
782            redeem_token_amount,
783            redeem_plan: Some(plan),
784            bind_epoch,
785        })
786    }
787}

789/// Translate a user-supplied `(redeem_mode, amount)` pair into a `RedeemPlan`.
790///
791/// `AtLeast` mirrors the chain's split-floor payout formula
792/// (`calculate_fungible_staked_sui_withdraw_amount`) and binary-searches for
793/// the minimum token count whose actual payout is `>= min_sui`. `fss_data`
794/// must be `Some` for AtLeast — without it we'd fall back to the `expected`
795/// formula which can pick a token count whose actual falls 1-2 MIST short,
796/// causing the chain guard to abort needlessly.
797///
798/// `AtMost` searches by `expected` alone — the chain's `actual <= expected`
799/// invariant (`staking_pool.move:264-268`) keeps the cap safe without the FSS
800/// sub-pool data.
801pub(crate) fn build_redeem_plan(
802    redeem_mode: RedeemMode,
803    amount: Option<u64>,
804    total_tokens: u64,
805    rate_sui: u64,
806    rate_token: u64,
807    fss_data: Option<&PoolRedeemData>,
808) -> Result<RedeemPlan, Error> {
809    match redeem_mode {
810        RedeemMode::All => Ok(RedeemPlan::All),
811        RedeemMode::AtLeast => {
812            let min_sui = amount.ok_or_else(|| {
813                Error::InvalidInput("amount required for AtLeast mode".to_string())
814            })?;
815            if min_sui == 0 {
816                return Err(Error::InvalidInput(
817                    "AtLeast amount must be at least 1 MIST".to_string(),
818                ));
819            }
820            if rate_sui == 0 || rate_token == 0 {
821                return Err(Error::DataError(
822                    "Pool has zero exchange rate, cannot compute redeem plan".to_string(),
823                ));
824            }
825            let data = fss_data.ok_or_else(|| {
826                Error::DataError(
827                    "FungibleStakedSuiData required for AtLeast mode but not supplied".to_string(),
828                )
829            })?;
830            let token_amount =
831                binary_search_at_least(data, total_tokens, min_sui)?.ok_or_else(|| {
832                    Error::InvalidInput(format!(
833                        "Insufficient FSS balance: cannot deliver AtLeast {min_sui} SUI \
834                         from {total_tokens} pool tokens at current exchange rate",
835                    ))
836                })?;
837            Ok(RedeemPlan::AtLeast {
838                token_amount: full_redeem_or_split(token_amount, total_tokens),
839                min_sui,
840            })
841        }
842        RedeemMode::AtMost => {
843            let max_sui = amount.ok_or_else(|| {
844                Error::InvalidInput("amount required for AtMost mode".to_string())
845            })?;
846            if max_sui == 0 {
847                return Err(Error::InvalidInput(
848                    "AtMost amount must be at least 1 MIST".to_string(),
849                ));
850            }
851            if rate_sui == 0 || rate_token == 0 {
852                return Err(Error::DataError(
853                    "Pool has zero exchange rate, cannot compute redeem plan".to_string(),
854                ));
855            }
856            let token_amount = binary_search_at_most(rate_sui, rate_token, total_tokens, max_sui)?
857                .ok_or_else(|| {
858                    Error::InvalidInput(
859                    "AtMost amount too small: would redeem 0 pool tokens at current exchange rate"
860                        .to_string(),
861                )
862                })?;
863            Ok(RedeemPlan::AtMost {
864                token_amount: full_redeem_or_split(token_amount, total_tokens),
865                max_sui,
866            })
867        }
868    }
869}

871/// Decide whether a `token_amount` selected by binary search needs a
872/// `split_fungible_staked_sui` step. If the search picked exactly the total,
873/// the merged FSS is already the right size — splitting would just leave a
874/// zero-value FSS object as dust.
875fn full_redeem_or_split(token_amount: u64, total_tokens: u64) -> Option<u64> {
876    if token_amount == total_tokens {
877        None
878    } else {
879        Some(token_amount)
880    }
881}

883/// Build PTB for merging FSS and redeeming according to a `RedeemPlan`.
884///
885/// Common shape:
886/// 1. Merge all `fss_refs` into a single FSS object.
887/// 2. (Partial) `split_fungible_staked_sui` to a sub-FSS of `token_amount`.
888/// 3. `redeem_fungible_staked_sui` → `Balance<SUI>`.
889/// 4. (`AtLeast` only) `balance::split(&mut Balance<SUI>, min_sui)` → sub-balance,
890///    then `balance::join` it back. The split aborts on chain when the redeemed
891///    `Balance.value` is below `min_sui`, providing a runtime guard against
892///    pool drift between simulate and execution. The join restores the original
893///    balance so subsequent commands see a single, full `Balance<SUI>` again.
894/// 5. `coin::from_balance<SUI>` → `Coin<SUI>`.
895/// 6. `TransferObjects` → sender.
896///
897/// The `All` mode skips step 2 (redeems the merged FSS in full).
898/// The `AtMost` mode performs step 2 with a token count chosen so that the
899/// chain's `expected = floor(token * rate_sui / rate_token)` stays at or below
900/// the user-supplied `max_sui`; the on-chain invariant `actual <= expected`
901/// then guarantees the cap holds without an explicit guard.
902pub fn merge_and_redeem_fss_pt(
903    sender: SuiAddress,
904    fss_refs: Vec<ObjectRef>,
905    plan: &RedeemPlan,
906) -> anyhow::Result<ProgrammableTransaction> {
907    let mut builder = ProgrammableTransactionBuilder::new();
908
909    if fss_refs.is_empty() {
910        return Ok(builder.finish());
911    }
912
913    let system_state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
914
915    let merged_fss = builder.obj(ObjectArg::ImmOrOwnedObject(fss_refs[0]))?;
916    for fss_ref in &fss_refs[1..] {
917        let other = builder.obj(ObjectArg::ImmOrOwnedObject(*fss_ref))?;
918        builder.command(Command::move_call(
919            SUI_SYSTEM_PACKAGE_ID,
920            Identifier::new("staking_pool")?,
921            Identifier::new("join_fungible_staked_sui")?,
922            vec![],
923            vec![merged_fss, other],
924        ));
925    }
926
927    // None = redeem the merged FSS in full (no split). Splitting when
928    // token_amount equals the total would just leave a zero-value FSS as dust.
929    let split_token_amount = match plan {
930        RedeemPlan::All => None,
931        RedeemPlan::AtLeast { token_amount, .. } | RedeemPlan::AtMost { token_amount, .. } => {
932            *token_amount
933        }
934    };
935    let redeem_target = if let Some(token_amount) = split_token_amount {
936        let split_amount = builder.pure(token_amount)?;
937        builder.command(Command::move_call(
938            SUI_SYSTEM_PACKAGE_ID,
939            Identifier::new("staking_pool")?,
940            Identifier::new("split_fungible_staked_sui")?,
941            vec![],
942            vec![merged_fss, split_amount],
943        ))
944    } else {
945        merged_fss
946    };
947
948    let balance_result = builder.command(Command::move_call(
949        SUI_SYSTEM_PACKAGE_ID,
950        SUI_SYSTEM_MODULE_NAME.to_owned(),
951        Identifier::new("redeem_fungible_staked_sui")?,
952        vec![],
953        vec![system_state, redeem_target],
954    ));
955
956    let sui_type = sui_types::TypeTag::from_str("0x2::sui::SUI")?;
957
958    if let RedeemPlan::AtLeast { min_sui, .. } = plan {
959        let min_arg = builder.pure(*min_sui)?;
960        let split_balance = builder.command(Command::move_call(
961            SUI_FRAMEWORK_PACKAGE_ID,
962            Identifier::new("balance")?,
963            Identifier::new("split")?,
964            vec![sui_type.clone()],
965            vec![balance_result, min_arg],
966        ));
967        builder.command(Command::move_call(
968            SUI_FRAMEWORK_PACKAGE_ID,
969            Identifier::new("balance")?,
970            Identifier::new("join")?,
971            vec![sui_type.clone()],
972            vec![balance_result, split_balance],
973        ));
974    }
975
976    let coin_result = builder.command(Command::move_call(
977        SUI_FRAMEWORK_PACKAGE_ID,
978        Identifier::new("coin")?,
979        Identifier::new("from_balance")?,
980        vec![sui_type],
981        vec![balance_result],
982    ));
983
984    let sender_arg = builder.pure(sender)?;
985    builder.command(Command::TransferObjects(vec![coin_result], sender_arg));
986
987    Ok(builder.finish())
988}

1243        // Property check: mirror_redeem_actual <= expected_sui_amount across a
1244        // grid of plausible pool states, with the gap bounded by ~2 MIST.
1245        for rate_sui in [100u64, 199, 1024, 100_003] {
1246            for rate_token in [99u64, 100, 1023, 99_997] {
1247                let supply = rate_token;
1248                for principal_pct in [0u64, 30, 50, 80, 100] {
1249                    let principal = supply * principal_pct / 100;
1250                    let data = fss(rate_sui, rate_token, principal, supply);
1251                    for token in [1u64, 2, supply / 2, supply.saturating_sub(1), supply] {
1252                        if token == 0 || token > supply {
1253                            continue;
1254                        }
1255                        let expected = expected_sui_amount(rate_sui, rate_token, token).unwrap();
1256                        let actual = mirror_redeem_actual(&data, token).unwrap();
1257                        assert!(actual <= expected, "actual={actual} > expected={expected}");
1258                        assert!(
1259                            actual + 2 >= expected,
1260                            "gap > 2 MIST: expected={expected} actual={actual}"
1261                        );
1262                    }
1263                }
1264            }
1265        }
