21#[derive(Serialize, Deserialize, Clone, Debug)]
22pub struct PayCoin {
23    pub sender: SuiAddress,
24    pub recipients: Vec<SuiAddress>,
25    pub amounts: Vec<u64>,
26    pub currency: Currency,
27}

29#[async_trait]
30impl TryConstructTransaction for PayCoin {
31    async fn try_fetch_needed_objects(
32        self,
33        client: &SuiClient,
34        gas_price: Option<u64>,
35        budget: Option<u64>,
36    ) -> Result<TransactionObjectData, Error> {
37        let Self {
38            sender,
39            recipients,
40            amounts,
41            currency,
42        } = self;
43
44        let amount = amounts.iter().sum::<u64>();
45        let coin_objs: Vec<ObjectRef> = client
46            .coin_read_api()
47            .select_coins(
48                sender,
49                Some(currency.metadata.coin_type.clone()),
50                amount.into(),
51                vec![],
52            )
53            .await?
54            .iter()
55            .map(|coin| coin.object_ref())
56            .collect();
57
58        let budget = match budget {
59            Some(budget) => budget,
60            None => {
61                let pt = pay_coin_pt(recipients, amounts, &coin_objs, &currency)?;
62                budget_from_dry_run(client, pt, sender, gas_price).await?
63            }
64        };
65
66        let gas_coins = client
67            .coin_read_api()
68            .select_coins(sender, None, budget as u128, vec![])
69            .await?;
70        if gas_coins.len() > MAX_GAS_COINS {
71            return Err(SuiErrorKind::UserInputError {
72                error: UserInputError::SizeLimitExceeded {
73                    limit: "maximum number of gas payment objects".to_string(),
74                    value: MAX_GAS_COINS.to_string(),
75                },
76            }
77            .into());
78        }
79
80        let gas_coins_iter = gas_coins.into_iter();
81        let total_sui_balance = gas_coins_iter.clone().map(|c| c.balance).sum::<u64>() as i128;
82        let gas_coins = gas_coins_iter.map(|c| c.object_ref()).collect();
83
84        Ok(TransactionObjectData {
85            gas_coins,
86            extra_gas_coins: vec![],
87            objects: coin_objs,
88            total_sui_balance,
89            budget,
90        })
91    }
92}

94// If the transaction budget is not enough here, there is nothing we can do.
95// Merging gas inside the transaction works only when gas is used also for other purposes other than
96// transaction fees.
97pub fn pay_coin_pt(
98    recipients: Vec<SuiAddress>,
99    amounts: Vec<u64>,
100    coins: &[ObjectRef],
101    currency: &Currency,
102) -> anyhow::Result<ProgrammableTransaction> {
103    if recipients.len() != amounts.len() {
104        return Err(anyhow!("Amounts length does not match recipients"));
105    }
106    if coins.is_empty() {
107        return Err(anyhow!("Cannot PayCoin without any coins"));
108    }
109
110    let mut commands = 0;
111    let mut builder = ProgrammableTransactionBuilder::new();
112
113    let mut merged = coins
114        .chunks(MAX_COMMAND_ARGS)
115        .map(|chunk| -> anyhow::Result<Argument> {
116            let mut to_merge: Vec<Argument> = chunk
117                .iter()
118                .map(|&o| builder.obj(ObjectArg::ImmOrOwnedObject(o)))
119                .collect::<Result<Vec<Argument>, anyhow::Error>>()?;
120            let merge_into = to_merge.pop().expect("Already checked for non-zero length");
121            if !to_merge.is_empty() {
122                builder.command(Command::MergeCoins(merge_into, to_merge));
123                commands += 1;
124            }
125            Ok(merge_into)
126        })
127        .collect::<Result<Vec<_>, anyhow::Error>>()?;
128    // Accumulate all dust coins into a single one
129    let single_coin = merged
130        .pop()
131        .expect("Already checked for non-zero coins above");
132    if !merged.is_empty() {
133        builder.command(Command::MergeCoins(single_coin, merged));
134        commands += 1;
135    }
136
137    // We could optimally not split the last coin if the sum of the coins.balance given matches
138    // the amounts.sum. This would require changes in the ConstructionMetadata type, as information
139    // about the total-coin-value would be needed.
140    let amount_args = amounts
141        .into_iter()
142        .map(|v| builder.pure(v))
143        .collect::<Result<Vec<_>, anyhow::Error>>()?;
144    let split_command = commands;
145    builder.command(Command::SplitCoins(single_coin, amount_args));
146
147    recipients
148        .into_iter()
149        .enumerate()
150        .for_each(|(i, recipient)| {
151            builder.transfer_arg(recipient, Argument::NestedResult(split_command, i as u16));
152        });
153
154    // This is a workaround in order to have the currency info available during the process
155    // of constructing back the Operations object from the transaction data. A process that
156    // takes place upon the request to the construction's /parse endpoint. The pure value is
157    // not actually being used in any on-chain transaction execution and its sole purpose
158    // is to act as a bearer of the currency info between the various steps of the flow.
159    // See also the value is being later accessed within the operations.rs file's
160    // parse_programmable_transaction function.
161    let currency_string = serde_json::to_string(currency)?;
162    builder.pure(currency_string)?;
163    Ok(builder.finish())
164}
