23#[derive(Serialize, Deserialize, Clone, Debug)]
24pub struct PayCoin {
25    pub sender: SuiAddress,
26    pub recipients: Vec<SuiAddress>,
27    pub amounts: Vec<u64>,
28    pub currency: Currency,
29}

31#[async_trait]
32impl TryConstructTransaction for PayCoin {
33    async fn try_fetch_needed_objects(
34        self,
35        client: &mut Client,
36        gas_price: Option<u64>,
37        budget: Option<u64>,
38    ) -> Result<TransactionObjectData, Error> {
39        let Self {
40            sender,
41            recipients,
42            amounts,
43            currency,
44        } = self;
45
46        let amount = amounts.iter().sum::<u64>();
47        let coin_type = TypeTag::from_str(&currency.metadata.coin_type)
48            .map_err(|e| Error::DataError(format!("Invalid coin type: {}", e)))?;
49        let all_coins = client
50            .select_coins(&Address::from(sender), &coin_type, amount, &[])
51            .await?;
52
53        let (party_objects, non_party_objects): (Vec<_>, Vec<_>) = all_coins
54            .iter()
55            .partition(|obj| obj.owner().kind() == OwnerKind::ConsensusAddress);
56
57        let coins: Vec<ObjectRef> = non_party_objects
58            .iter()
59            .map(|obj: &&Object| obj.object_reference().try_to_object_ref())
60            .collect::<Result<Vec<_>, _>>()?;
61
62        let party_coins: Vec<(ObjectID, SequenceNumber)> = party_objects
63            .iter()
64            .map(|obj: &&Object| -> Result<_, Error> {
65                let id = ObjectID::from_str(obj.object_id())
66                    .map_err(|e| Error::DataError(format!("Invalid party object ID: {}", e)))?;
67                let start_version = SequenceNumber::from_u64(obj.owner().version());
68                Ok((id, start_version))
69            })
70            .collect::<Result<Vec<_>, _>>()?;
71
72        // If budget is provided, we still need to select gas coins
73        let pt = pay_coin_pt(recipients, amounts, &coins, &party_coins, &currency)?;
74        let (budget, gas_coin_objs) =
75            simulate_transaction(client, pt, sender, vec![], gas_price, budget).await?;
76
77        let total_sui_balance = gas_coin_objs.iter().map(|c| c.balance()).sum::<u64>() as i128;
78        let gas_coins = gas_coin_objs
79            .iter()
80            .map(|obj: &Object| obj.object_reference().try_to_object_ref())
81            .collect::<Result<Vec<_>, _>>()?;
82
83        Ok(TransactionObjectData {
84            gas_coins,
85            objects: coins,
86            party_objects: party_coins,
87            total_sui_balance,
88            budget,
89        })
90    }
91}

93// If the transaction budget is not enough here, there is nothing we can do.
94// Merging gas inside the transaction works only when gas is used also for other purposes other than
95// transaction fees.
96pub fn pay_coin_pt(
97    recipients: Vec<SuiAddress>,
98    amounts: Vec<u64>,
99    coins: &[ObjectRef],
100    party_coins: &[(ObjectID, SequenceNumber)],
101    currency: &Currency,
102) -> anyhow::Result<ProgrammableTransaction> {
103    if recipients.len() != amounts.len() {
104        return Err(anyhow!("Amounts length does not match recipients"));
105    }
106    if coins.is_empty() && party_coins.is_empty() {
107        return Err(anyhow!("Cannot PayCoin without any coins"));
108    }
109
110    let mut builder = ProgrammableTransactionBuilder::new();
111
112    let all_chunks: Vec<Vec<ObjectArg>> = coins
113        .chunks(MAX_COMMAND_ARGS)
114        .map(|chunk| {
115            chunk
116                .iter()
117                .map(|&o| ObjectArg::ImmOrOwnedObject(o))
118                .collect::<Vec<_>>()
119        })
120        .chain(party_coins.chunks(MAX_COMMAND_ARGS).map(|chunk| {
121            chunk
122                .iter()
123                .map(|&(id, initial_shared_version)| ObjectArg::SharedObject {
124                    id,
125                    initial_shared_version,
126                    mutability: sui_types::transaction::SharedObjectMutability::Mutable,
127                })
128                .collect::<Vec<_>>()
129        }))
130        .collect();
131
132    let mut commands = 0;
133    let mut merged: Vec<Argument> = all_chunks
134        .into_iter()
135        .map(|chunk| -> anyhow::Result<Argument> {
136            let mut to_merge: Vec<Argument> = chunk
137                .into_iter()
138                .map(|o| builder.obj(o))
139                .collect::<Result<Vec<Argument>, anyhow::Error>>()?;
140            let merge_into = to_merge
141                .pop()
142                .expect("chunks() guarantees non-empty chunks");
143            if !to_merge.is_empty() {
144                builder.command(Command::MergeCoins(merge_into, to_merge));
145                commands += 1;
146            }
147            Ok(merge_into)
148        })
149        .collect::<Result<Vec<_>, anyhow::Error>>()?;
150    // Accumulate all dust coins into a single one
151    let single_coin = merged
152        .pop()
153        .expect("At least one of coins or party_coins is non-empty");
154    if !merged.is_empty() {
155        builder.command(Command::MergeCoins(single_coin, merged));
156        commands += 1;
157    }
158
159    // We could optimally not split the last coin if the sum of the coins.balance given matches
160    // the amounts.sum. This would require changes in the ConstructionMetadata type, as information
161    // about the total-coin-value would be needed.
162    let amount_args = amounts
163        .into_iter()
164        .map(|v| builder.pure(v))
165        .collect::<Result<Vec<_>, anyhow::Error>>()?;
166    let split_command = commands;
167    builder.command(Command::SplitCoins(single_coin, amount_args));
168
169    recipients
170        .into_iter()
171        .enumerate()
172        .for_each(|(i, recipient)| {
173            builder.transfer_arg(recipient, Argument::NestedResult(split_command, i as u16));
174        });
175
176    // This is a workaround in order to have the currency info available during the process
177    // of constructing back the Operations object from the transaction data. A process that
178    // takes place upon the request to the construction's /parse endpoint. The pure value is
179    // not actually being used in any on-chain transaction execution and its sole purpose
180    // is to act as a bearer of the currency info between the various steps of the flow.
181    // See also the value is being later accessed within the operations.rs file's
182    // parse_programmable_transaction function.
183    let currency_string = serde_json::to_string(currency)?;
184    builder.pure(currency_string)?;
185    Ok(builder.finish())
186}
