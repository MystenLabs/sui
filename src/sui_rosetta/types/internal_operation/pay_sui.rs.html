19#[derive(Serialize, Deserialize, Clone, Debug)]
20pub struct PaySui {
21    pub sender: SuiAddress,
22    pub recipients: Vec<SuiAddress>,
23    pub amounts: Vec<u64>,
24}

26#[async_trait]
27impl TryConstructTransaction for PaySui {
28    async fn try_fetch_needed_objects(
29        self,
30        client: &SuiClient,
31        gas_price: Option<u64>,
32        budget: Option<u64>,
33    ) -> Result<TransactionObjectData, Error> {
34        let Self {
35            sender,
36            recipients,
37            amounts,
38        } = self;
39
40        let total_amount = amounts.iter().sum::<u64>();
41        if let Some(budget) = budget {
42            // We have a constant budget, so no need to dry-run
43            let all_coins = client
44                .coin_read_api()
45                .select_coins(sender, None, (total_amount + budget) as u128, vec![])
46                .await?;
47
48            let total_sui_balance = all_coins.iter().map(|c| c.balance).sum::<u64>() as i128;
49
50            let mut iter = all_coins.into_iter().map(|c| c.object_ref());
51            let gas_coins: Vec<_> = iter.by_ref().take(MAX_GAS_COINS).collect();
52            let extra_gas_coins: Vec<_> = iter.collect();
53
54            return Ok(TransactionObjectData {
55                gas_coins,
56                extra_gas_coins,
57                objects: vec![],
58                total_sui_balance,
59                budget,
60            });
61        };
62
63        let total_amount = amounts.iter().sum::<u64>();
64        let pay_sui_pt = |extra_gas_coins: &[ObjectRef]| {
65            pay_sui_pt(recipients.clone(), amounts.clone(), extra_gas_coins)
66        };
67        collect_coins_until_budget_met(client, sender, pay_sui_pt, total_amount, gas_price).await
68    }
69}

71/// Creates the `ProgrammableTransaction` for a pay-sui operation.
72/// In case pay-sui needs more than 255 gas-coins to be smashed, it tries to merge the surplus
73/// coins into the gas coin as regular transaction inputs - not gas-payment.
74/// This approach has the limit at around 1650 coins in total which triggers transaction-size
75/// limit (see also test_limit_many_small_coins test).
76pub fn pay_sui_pt(
77    recipients: Vec<SuiAddress>,
78    amounts: Vec<u64>,
79    coins_to_merge: &[ObjectRef],
80) -> anyhow::Result<ProgrammableTransaction> {
81    let mut builder = ProgrammableTransactionBuilder::new();
82    if !coins_to_merge.is_empty() {
83        // We need to merge the rest of the coins.
84        // Each merge has a limit of 511 arguments.
85        coins_to_merge
86            .chunks(MAX_COMMAND_ARGS)
87            .try_for_each(|chunk| -> anyhow::Result<()> {
88                let to_merge = chunk
89                    .iter()
90                    .map(|&o| builder.obj(ObjectArg::ImmOrOwnedObject(o)))
91                    .collect::<Result<Vec<Argument>, anyhow::Error>>()?;
92                builder.command(Command::MergeCoins(Argument::GasCoin, to_merge));
93                Ok(())
94            })?;
95    };
96    builder.pay_sui(recipients, amounts)?;
97    Ok(builder.finish())
98}
