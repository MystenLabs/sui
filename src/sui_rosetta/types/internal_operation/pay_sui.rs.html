31#[async_trait]
32impl TryConstructTransaction for PaySui {
33    async fn try_fetch_needed_objects(
34        self,
35        client: &mut Client,
36        gas_price: Option<u64>,
37        budget: Option<u64>,
38    ) -> Result<TransactionObjectData, Error> {
39        let Self {
40            sender,
41            recipients,
42            amounts,
43        } = self;
44
45        let balance_info = {
46            let request = GetBalanceRequest::default()
47                .with_owner(sender.to_string())
48                .with_coin_type("0x2::sui::SUI".to_string());
49            client
50                .state_client()
51                .get_balance(request)
52                .await?
53                .into_inner()
54        };
55        let address_balance = balance_info.balance().address_balance();
56
57        // PaySui needs enough SUI to cover both payment amount and gas. We select up to 1500
58        // coins (we observed ~1650 coins in a single transaction hits transaction size limits)
59        // and merge them all together, then split off the payment amount.
60        //
61        // This handles cases where the user has sufficient total balance but no single coin or
62        // simple combination covers payment + gas without merging/splitting. For example, with
63        // [40, 35, 25] SUI coins and needing to pay 50 + gas, no discrete set works - we must
64        // merge first to create a coin large enough to split appropriately.
65        //
66        // This approach is optimal because storage refunds from merging dust outweigh smashing
67        // costs by an order of magnitude, and ensures we can handle fragmented balances.
68        // Use the full Coin<SUI> struct tag
69
70        let all_coins = client
71            .select_up_to_n_largest_coins(
72                &Address::from(sender),
73                &StructTag::sui().into(),
74                1500,
75                &[],
76            )
77            .await?;
78
79        let coin_objects_total = all_coins.iter().map(|c| c.balance()).sum::<u64>();
80
81        // Separate party objects (ConsensusAddressOwner) from regular objects.
82        // Party objects cannot be used as gas but can be merged into the gas coin.
83        let (party_objects, non_party_objects): (Vec<_>, Vec<_>) = all_coins
84            .iter()
85            .partition(|obj| obj.owner().kind() == OwnerKind::ConsensusAddress);
86
87        let non_party_refs: Vec<ObjectRef> = non_party_objects
88            .iter()
89            .map(|obj: &&Object| obj.object_reference().try_to_object_ref())
90            .collect::<Result<Vec<_>, _>>()?;
91
92        let party_refs: Vec<(ObjectID, SequenceNumber)> = party_objects
93            .iter()
94            .map(|obj: &&Object| -> Result<_, Error> {
95                let id = ObjectID::from_str(obj.object_id())
96                    .map_err(|e| Error::DataError(format!("Invalid party object ID: {}", e)))?;
97                let start_version = SequenceNumber::from_u64(obj.owner().version());
98                Ok((id, start_version))
99            })
100            .collect::<Result<Vec<_>, _>>()?;
101
102        let total_payment: u64 = amounts.iter().sum();
103        let address_balance_withdrawl = total_payment.saturating_sub(coin_objects_total);
104
105        // Path A: merge coins, withdraw deficit from AB, pay. No GasCoin → AB gas.
106        let pt_ab_gas = pay_sui_pt_ab_gas(
107            sender,
108            recipients.clone(),
109            amounts.clone(),
110            &non_party_refs,
111            &party_refs,
112            address_balance_withdrawl,
113        )?;
114        let sim_result =
115            simulate_transaction(client, pt_ab_gas, sender, vec![], gas_price, budget).await;
116
117        match sim_result {
118            Ok((budget, gas_coin_objs)) if gas_coin_objs.is_empty() => {
119                // Path A succeeded with address-balance gas
120                let total_sui_balance = (coin_objects_total as i128) + (address_balance as i128);
121                Ok(TransactionObjectData {
122                    gas_coins: vec![],
123                    objects: non_party_refs,
124                    party_objects: party_refs,
125                    total_sui_balance,
126                    budget,
127                    address_balance_withdrawal: address_balance_withdrawl,
128                })
129            }
130            _ => {
131                // Path B: merge coins into GasCoin, pay from GasCoin, send remainder to address balance.
132                let address_balance_withdrawal = address_balance;
133                let total_sui_balance =
134                    (coin_objects_total as i128) + (address_balance_withdrawal as i128);
135
136                let mut gas_coin_iter = non_party_refs.iter().copied();
137                let gas_coins: Vec<ObjectRef> =
138                    gas_coin_iter.by_ref().take(MAX_GAS_COINS).collect();
139                let extra_coins: Vec<ObjectRef> = gas_coin_iter.collect();
140
141                // send_remainder is just a placeholder for simulation — the actual value
142                // is recomputed in try_into_data. We only need the send_funds commands
143                // present so the gas estimator accounts for them.
144                let pt_b = pay_sui_pt_coin_gas(
145                    recipients,
146                    amounts,
147                    &extra_coins,
148                    &party_refs,
149                    address_balance_withdrawal,
150                )?;
151                let (budget, gas_coin_objs) =
152                    simulate_transaction(client, pt_b, sender, gas_coins, gas_price, budget)
153                        .await?;
154
155                let gas_coins = gas_coin_objs
156                    .iter()
157                    .map(|obj| obj.object_reference().try_to_object_ref())
158                    .collect::<Result<Vec<_>, _>>()?;
159
160                Ok(TransactionObjectData {
161                    gas_coins,
162                    objects: extra_coins,
163                    party_objects: party_refs,
164                    total_sui_balance,
165                    budget,
166                    address_balance_withdrawal,
167                })
168            }
169        }
170    }
171}

173/// Path A: merge coins, optionally withdraw deficit from AB, pay, send remainder to address balance.
174/// No GasCoin reference → simulator auto-selects address-balance gas.
175pub(crate) fn pay_sui_pt_ab_gas(
176    sender: SuiAddress,
177    recipients: Vec<SuiAddress>,
178    amounts: Vec<u64>,
179    coins: &[ObjectRef],
180    party_coins: &[(ObjectID, SequenceNumber)],
181    address_balance_withdrawal: u64,
182) -> anyhow::Result<ProgrammableTransaction> {
183    let mut builder = ProgrammableTransactionBuilder::new();
184    let mut source: Option<Argument> = None;
185    // Track whether the source is a command result (vs input object).
186    // Command results need explicit transfer after SplitCoins; input objects
187    // are implicitly returned.
188    let mut source_is_command_result = false;
189
190    // Step 1: Merge all coin objects into one
191    if !coins.is_empty() || !party_coins.is_empty() {
192        let target = if let Some(&first) = coins.first() {
193            builder.obj(ObjectArg::ImmOrOwnedObject(first))?
194        } else {
195            let &(id, initial_shared_version) = &party_coins[0];
196            builder.obj(ObjectArg::SharedObject {
197                id,
198                initial_shared_version,
199                mutability: sui_types::transaction::SharedObjectMutability::Mutable,
200            })?
201        };
202
203        if coins.len() > 1 {
204            coins[1..]
205                .chunks(MAX_COMMAND_ARGS)
206                .try_for_each(|chunk| -> anyhow::Result<()> {
207                    let to_merge = chunk
208                        .iter()
209                        .map(|&o| builder.obj(ObjectArg::ImmOrOwnedObject(o)))
210                        .collect::<Result<Vec<Argument>, _>>()?;
211                    builder.command(Command::MergeCoins(target, to_merge));
212                    Ok(())
213                })?;
214        }
215
216        let party_skip = if coins.is_empty() { 1 } else { 0 };
217        let party_slice = &party_coins[party_skip..];
218        if !party_slice.is_empty() {
219            party_slice
220                .chunks(MAX_COMMAND_ARGS)
221                .try_for_each(|chunk| -> anyhow::Result<()> {
222                    let to_merge = chunk
223                        .iter()
224                        .map(|&(id, initial_shared_version)| {
225                            builder.obj(ObjectArg::SharedObject {
226                                id,
227                                initial_shared_version,
228                                mutability: sui_types::transaction::SharedObjectMutability::Mutable,
229                            })
230                        })
231                        .collect::<Result<Vec<Argument>, _>>()?;
232                    builder.command(Command::MergeCoins(target, to_merge));
233                    Ok(())
234                })?;
235        }
236
237        source = Some(target);
238    }
239
240    // Step 2: Withdraw deficit from address balance and merge into coin
241    if address_balance_withdrawal > 0 {
242        let withdrawal_coin = withdraw_coin_from_address_balance(
243            &mut builder,
244            address_balance_withdrawal,
245            GAS::type_tag(),
246        )?;
247        match source {
248            Some(target) => {
249                // Merge command result into input object — input is implicitly returned
250                builder.command(Command::MergeCoins(target, vec![withdrawal_coin]));
251            }
252            None => {
253                // No coin objects; source is entirely from AB (a command result)
254                source = Some(withdrawal_coin);
255                source_is_command_result = true;
256            }
257        }
258    }
259
260    let source =
261        source.ok_or_else(|| anyhow::anyhow!("No coins or address balance to pay from"))?;
262
263    // Step 3: Split payment amounts and transfer to recipients
264    let amount_args: Vec<Argument> = amounts
265        .iter()
266        .map(|&v| builder.pure(v))
267        .collect::<Result<Vec<_>, _>>()?;
268    let split_result = builder.command(Command::SplitCoins(source, amount_args));
269    let Argument::Result(split_idx) = split_result else {
270        anyhow::bail!("Expected Result argument from SplitCoins");
271    };
272
273    for (i, recipient) in recipients.into_iter().enumerate() {
274        builder.transfer_arg(recipient, Argument::NestedResult(split_idx, i as u16));
275    }
276
277    // Command results (from AB withdrawal) aren't implicitly returned like input objects.
278    // Transfer the remainder back to the sender.
279    if source_is_command_result {
280        builder.transfer_arg(sender, source);
281    }
282
283    Ok(builder.finish())
284}

286/// Path B: merge coins into GasCoin, pay from GasCoin.
287pub(crate) fn pay_sui_pt_coin_gas(
288    recipients: Vec<SuiAddress>,
289    amounts: Vec<u64>,
290    coins_to_merge: &[ObjectRef],
291    party_coins: &[(ObjectID, SequenceNumber)],
292    address_balance_withdrawal: u64,
293) -> anyhow::Result<ProgrammableTransaction> {
294    let mut builder = ProgrammableTransactionBuilder::new();
295
296    // Merge extra coins into GasCoin
297    if !coins_to_merge.is_empty() {
298        coins_to_merge
299            .chunks(MAX_COMMAND_ARGS)
300            .try_for_each(|chunk| -> anyhow::Result<()> {
301                let to_merge = chunk
302                    .iter()
303                    .map(|&o| builder.obj(ObjectArg::ImmOrOwnedObject(o)))
304                    .collect::<Result<Vec<Argument>, _>>()?;
305                builder.command(Command::MergeCoins(Argument::GasCoin, to_merge));
306                Ok(())
307            })?;
308    }
309
310    // Merge party coins into GasCoin
311    if !party_coins.is_empty() {
312        party_coins
313            .chunks(MAX_COMMAND_ARGS)
314            .try_for_each(|chunk| -> anyhow::Result<()> {
315                let to_merge = chunk
316                    .iter()
317                    .map(|&(id, initial_shared_version)| {
318                        builder.obj(ObjectArg::SharedObject {
319                            id,
320                            initial_shared_version,
321                            mutability: sui_types::transaction::SharedObjectMutability::Mutable,
322                        })
323                    })
324                    .collect::<Result<Vec<Argument>, _>>()?;
325                builder.command(Command::MergeCoins(Argument::GasCoin, to_merge));
326                Ok(())
327            })?;
328    }
329
330    // Withdraw address balance and merge into GasCoin
331    if address_balance_withdrawal > 0 {
332        let withdrawn_coin = withdraw_coin_from_address_balance(
333            &mut builder,
334            address_balance_withdrawal,
335            GAS::type_tag(),
336        )?;
337        builder.command(Command::MergeCoins(Argument::GasCoin, vec![withdrawn_coin]));
338    }
339
340    // Pay from GasCoin
341    builder.pay_sui(recipients, amounts)?;
342
343    Ok(builder.finish())
344}
