35#[async_trait]
36impl TryConstructTransaction for Stake {
37    async fn try_fetch_needed_objects(
38        self,
39        client: &mut Client,
40        gas_price: Option<u64>,
41        budget: Option<u64>,
42    ) -> Result<TransactionObjectData, Error> {
43        let Self {
44            sender,
45            validator,
46            amount,
47        } = self;
48
49        // Staking needs enough SUI to cover both the stake amount and gas. We select up to 1500
50        // coins (we observed ~1650 coins in a single transaction hits transaction size limits)
51        // and merge them all together, then split off the stake amount.
52        //
53        // This handles cases where the user has sufficient total balance but no single coin or
54        // simple combination covers stake + gas without merging/splitting. For example, with
55        // [8, 6, 4] SUI coins and wanting to stake 10 + gas, no discrete set works - we must
56        // merge first to create a coin large enough to split appropriately.
57        //
58        // This approach is optimal because storage refunds from merging dust outweigh smashing
59        // costs by an order of magnitude, and ensures we can handle fragmented balances.
60        // Use the full Coin<SUI> struct tag
61        let all_coins = client
62            .select_up_to_n_largest_coins(
63                &Address::from(sender),
64                &StructTag::sui().into(),
65                1500,
66                &[],
67            )
68            .await?;
69
70        let total_sui_balance = all_coins.iter().map(|c| c.balance()).sum::<u64>() as i128;
71
72        // Separate party objects (ConsensusAddressOwner) from regular objects.
73        // Party objects cannot be used as gas but can be merged into the gas coin using SharedObject.
74        let (party_objects, non_party_objects): (Vec<_>, Vec<_>) = all_coins
75            .iter()
76            .partition(|obj| obj.owner().kind() == OwnerKind::ConsensusAddress);
77
78        let mut iter = non_party_objects
79            .iter()
80            .map(|obj: &&Object| obj.object_reference().try_to_object_ref());
81        let gas_coins = iter
82            .by_ref()
83            .take(MAX_GAS_COINS)
84            .collect::<Result<Vec<_>, _>>()?;
85
86        let extra_gas_coins = iter.collect::<Result<Vec<_>, _>>()?;
87
88        let extra_party_coins: Vec<(ObjectID, SequenceNumber)> = party_objects
89            .iter()
90            .map(|obj: &&Object| -> Result<_, Error> {
91                let id = ObjectID::from_str(obj.object_id())
92                    .map_err(|e| Error::DataError(format!("Invalid party object ID: {}", e)))?;
93                let start_version = SequenceNumber::from_u64(obj.owner().version());
94                Ok((id, start_version))
95            })
96            .collect::<Result<Vec<_>, _>>()?;
97
98        // Always simulate to validate the transaction
99        // For stake_all (amount is None), simulate with minimal amount
100        // For specific amount, simulate with actual amount
101        let simulation_amount = amount.unwrap_or(1_000_000_000); // 1 SUI minimum staking threshold
102
103        let pt = stake_pt(
104            validator,
105            simulation_amount,
106            false,
107            &extra_gas_coins,
108            &extra_party_coins,
109        )?;
110        let (budget, _) =
111            simulate_transaction(client, pt, sender, gas_coins.clone(), gas_price, budget).await?;
112
113        Ok(TransactionObjectData {
114            gas_coins,
115            objects: extra_gas_coins,
116            party_objects: extra_party_coins,
117            total_sui_balance,
118            budget,
119        })
120    }
121}

123pub fn stake_pt(
124    validator: SuiAddress,
125    amount: u64,
126    stake_all: bool,
127    coins_to_merge: &[ObjectRef],
128    party_coins: &[(ObjectID, SequenceNumber)],
129) -> anyhow::Result<ProgrammableTransaction> {
130    let mut builder = ProgrammableTransactionBuilder::new();
131
132    // [WORKAROUND] - this is a hack to work out if the staking ops is for a selected amount or None amount (whole wallet).
133    // if amount is none, validator input will be created after the system object input
134    // TODO: Investigate whether using asimple input argument with relevant metadata, similar
135    // to PayCoinOperation, would work as well or even better. Would help with consistency.
136    let amount = builder.pure(amount)?;
137    let (validator, system_state) = if !stake_all {
138        let validator = builder.input(CallArg::Pure(bcs::to_bytes(&validator)?))?;
139        let state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
140        (validator, state)
141    } else {
142        let state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
143        let validator = builder.input(CallArg::Pure(bcs::to_bytes(&validator)?))?;
144        (validator, state)
145    };
146
147    if !coins_to_merge.is_empty() {
148        coins_to_merge
149            .chunks(MAX_COMMAND_ARGS)
150            .try_for_each(|chunk| -> anyhow::Result<()> {
151                let to_merge = chunk
152                    .iter()
153                    .map(|&o| builder.obj(ObjectArg::ImmOrOwnedObject(o)))
154                    .collect::<Result<Vec<Argument>, anyhow::Error>>()?;
155                builder.command(Command::MergeCoins(Argument::GasCoin, to_merge));
156                Ok(())
157            })?;
158    };
159
160    if !party_coins.is_empty() {
161        party_coins
162            .chunks(MAX_COMMAND_ARGS)
163            .try_for_each(|chunk| -> anyhow::Result<()> {
164                let to_merge = chunk
165                    .iter()
166                    .map(|&(id, initial_shared_version)| {
167                        builder.obj(ObjectArg::SharedObject {
168                            id,
169                            initial_shared_version,
170                            mutability: sui_types::transaction::SharedObjectMutability::Mutable,
171                        })
172                    })
173                    .collect::<Result<Vec<Argument>, anyhow::Error>>()?;
174                builder.command(Command::MergeCoins(Argument::GasCoin, to_merge));
175                Ok(())
176            })?;
177    };
178
179    // Theoretically, if stake_all is true, we could not use amount, and instead,
180    // directly use Argument::GasCoin here, but this is how this Operation has always worked.
181    // Changing this now would require editing other endpoints too.
182    let coin = builder.command(Command::SplitCoins(Argument::GasCoin, vec![amount]));
183
184    let arguments = vec![system_state, coin, validator];
185
186    builder.command(Command::move_call(
187        SUI_SYSTEM_PACKAGE_ID,
188        SUI_SYSTEM_MODULE_NAME.to_owned(),
189        ADD_STAKE_FUN_NAME.to_owned(),
190        vec![],
191        arguments,
192    ));
193    Ok(builder.finish())
194}
