21#[derive(Serialize, Deserialize, Clone, Debug)]
22pub struct WithdrawStake {
23    pub sender: SuiAddress,
24    #[serde(default, skip_serializing_if = "Vec::is_empty")]
25    pub stake_ids: Vec<ObjectID>,
26}
27
28#[async_trait]
29impl TryConstructTransaction for WithdrawStake {
30    async fn try_fetch_needed_objects(
31        self,
32        client: &SuiClient,
33        gas_price: Option<u64>,
34        budget: Option<u64>,
35    ) -> Result<TransactionObjectData, Error> {
36        let Self { sender, stake_ids } = self;
37
38        let withdraw_all = stake_ids.is_empty();
39        let stake_ids = if withdraw_all {
40            // unstake all
41            client
42                .governance_api()
43                .get_stakes(sender)
44                .await?
45                .into_iter()
46                .flat_map(|s| {
47                    s.stakes.into_iter().filter_map(|s| {
48                        if let StakeStatus::Active { .. } = s.status {
49                            Some(s.staked_sui_id)
50                        } else {
51                            None
52                        }
53                    })
54                })
55                .collect()
56        } else {
57            stake_ids
58        };
59
60        if stake_ids.is_empty() {
61            return Err(Error::InvalidInput("No active stake to withdraw".into()));
62        }
63
64        let responses = client
65            .read_api()
66            .multi_get_object_with_options(stake_ids, SuiObjectDataOptions::default())
67            .await?;
68        let stake_refs = responses
69            .into_iter()
70            .map(|stake| stake.into_object().map(|o| o.object_ref()))
71            .collect::<Result<Vec<_>, _>>()
72            .map_err(SuiError::from)?;
73
74        // dry run
75        let budget = match budget {
76            Some(budget) => budget,
77            None => {
78                let pt = withdraw_stake_pt(stake_refs.clone(), withdraw_all)?;
79                budget_from_dry_run(client, pt.clone(), sender, gas_price).await?
80            }
81        };
82
83        let gas_coins = client
84            .coin_read_api()
85            .select_coins(sender, None, budget as u128, vec![])
86            .await?;
87        if gas_coins.len() > MAX_GAS_COINS {
88            return Err(SuiErrorKind::UserInputError {
89                error: UserInputError::SizeLimitExceeded {
90                    limit: "maximum number of gas payment objects".to_string(),
91                    value: MAX_GAS_COINS.to_string(),
92                },
93            }
94            .into());
95        }
96
97        let gas_coins_iter = gas_coins.into_iter();
98        let total_sui_balance = gas_coins_iter.clone().map(|c| c.balance).sum::<u64>() as i128;
99        let gas_coins = gas_coins_iter.map(|c| c.object_ref()).collect();
100
101        Ok(TransactionObjectData {
102            gas_coins,
103            extra_gas_coins: vec![],
104            objects: stake_refs,
105            total_sui_balance,
106            budget,
107        })
108    }
109}

111pub fn withdraw_stake_pt(
112    stake_objs: Vec<ObjectRef>,
113    withdraw_all: bool,
114) -> anyhow::Result<ProgrammableTransaction> {
115    let mut builder = ProgrammableTransactionBuilder::new();
116
117    for stake_id in stake_objs {
118        // [WORKAROUND] - this is a hack to work out if the withdraw stake ops is for selected stake_ids or None (all stakes) using the index of the call args.
119        // if stake_ids is not empty, id input will be created after the system object input
120        // TODO: Investigate whether using asimple input argument with relevant metadata, similar
121        // to PayCoinOperation, would work as well or even better. Would help with consistency.
122        let (system_state, id) = if !withdraw_all {
123            let system_state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
124            let id = builder.obj(ObjectArg::ImmOrOwnedObject(stake_id))?;
125            (system_state, id)
126        } else {
127            let id = builder.obj(ObjectArg::ImmOrOwnedObject(stake_id))?;
128            let system_state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
129            (system_state, id)
130        };
131
132        let arguments = vec![system_state, id];
133        builder.command(Command::move_call(
134            SUI_SYSTEM_PACKAGE_ID,
135            SUI_SYSTEM_MODULE_NAME.to_owned(),
136            WITHDRAW_STAKE_FUN_NAME.to_owned(),
137            vec![],
138            arguments,
139        ));
140    }
141    Ok(builder.finish())
142}
