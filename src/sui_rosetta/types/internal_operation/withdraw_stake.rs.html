24#[derive(Serialize, Deserialize, Clone, Debug)]
25pub struct WithdrawStake {
26    pub sender: SuiAddress,
27    #[serde(default, skip_serializing_if = "Vec::is_empty")]
28    pub stake_ids: Vec<ObjectID>,
29}
30
31#[async_trait]
32impl TryConstructTransaction for WithdrawStake {
33    async fn try_fetch_needed_objects(
34        self,
35        client: &mut Client,
36        gas_price: Option<u64>,
37        budget: Option<u64>,
38    ) -> Result<TransactionObjectData, Error> {
39        let Self { sender, stake_ids } = self;
40
41        let withdraw_all = stake_ids.is_empty();
42        let stake_ids = if withdraw_all {
43            use futures::TryStreamExt;
44
45            let list_request = ListOwnedObjectsRequest::default()
46                .with_owner(sender.to_string())
47                .with_object_type("0x3::staking_pool::StakedSui".to_string())
48                .with_page_size(1000u32)
49                .with_read_mask(FieldMask::from_paths(["object_id"]));
50
51            client
52                .list_owned_objects(list_request)
53                .map_err(Error::from)
54                .and_then(|object| async move {
55                    let object_id = ObjectID::from_hex_literal(object.object_id())
56                        .map_err(|e| Error::InvalidInput(format!("Invalid object_id: {}", e)))?;
57                    Ok(object_id)
58                })
59                .try_collect::<Vec<_>>()
60                .await?
61        } else {
62            stake_ids
63        };
64
65        if stake_ids.is_empty() {
66            return Err(Error::InvalidInput("No active stake to withdraw".into()));
67        }
68
69        let mut stake_refs = Vec::new();
70        for stake_id in &stake_ids {
71            let stake_address: sui_sdk_types::Address = (*stake_id).into();
72            let request = GetObjectRequest::new(&stake_address)
73                .with_read_mask(FieldMask::from_paths(["object_id", "version", "digest"]));
74
75            let response = client
76                .ledger_client()
77                .get_object(request)
78                .await?
79                .into_inner();
80
81            if let Some(object) = response.object {
82                let object_id = object.object_id.ok_or_else(|| {
83                    Error::InvalidInput("Object ID missing in response".to_string())
84                })?;
85                let version = object.version.ok_or_else(|| {
86                    Error::InvalidInput("Version missing in response".to_string())
87                })?;
88                let digest = object
89                    .digest
90                    .ok_or_else(|| Error::InvalidInput("Digest missing in response".to_string()))?;
91
92                stake_refs.push((
93                    ObjectID::from_str(&object_id).map_err(|e| {
94                        Error::InvalidInput(format!("Failed to parse object ID: {}", e))
95                    })?,
96                    version.into(),
97                    digest.parse().map_err(|e| {
98                        Error::InvalidInput(format!("Failed to parse digest: {}", e))
99                    })?,
100                ));
101            } else {
102                return Err(Error::InvalidInput(format!(
103                    "Stake object {} not found",
104                    stake_id
105                )));
106            }
107        }
108
109        let pt = withdraw_stake_pt(stake_refs.clone(), withdraw_all)?;
110        let (budget, gas_coin_objs) =
111            simulate_transaction(client, pt, sender, vec![], gas_price, budget).await?;
112
113        let total_sui_balance = gas_coin_objs.iter().map(|c| c.balance()).sum::<u64>() as i128;
114        let gas_coins = gas_coin_objs
115            .iter()
116            .map(|obj| obj.object_reference().try_to_object_ref())
117            .collect::<Result<Vec<_>, _>>()?;
118
119        Ok(TransactionObjectData {
120            gas_coins,
121            objects: stake_refs,
122            party_objects: vec![],
123            total_sui_balance,
124            budget,
125        })
126    }
127}

129pub fn withdraw_stake_pt(
130    stake_objs: Vec<ObjectRef>,
131    withdraw_all: bool,
132) -> anyhow::Result<ProgrammableTransaction> {
133    let mut builder = ProgrammableTransactionBuilder::new();
134
135    for stake_id in stake_objs {
136        // [WORKAROUND] - this is a hack to work out if the withdraw stake ops is for selected stake_ids or None (all stakes) using the index of the call args.
137        // if stake_ids is not empty, id input will be created after the system object input
138        // TODO: Investigate whether using asimple input argument with relevant metadata, similar
139        // to PayCoinOperation, would work as well or even better. Would help with consistency.
140        let (system_state, id) = if !withdraw_all {
141            let system_state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
142            let id = builder.obj(ObjectArg::ImmOrOwnedObject(stake_id))?;
143            (system_state, id)
144        } else {
145            let id = builder.obj(ObjectArg::ImmOrOwnedObject(stake_id))?;
146            let system_state = builder.input(CallArg::SUI_SYSTEM_MUT)?;
147            (system_state, id)
148        };
149
150        let arguments = vec![system_state, id];
151        builder.command(Command::move_call(
152            SUI_SYSTEM_PACKAGE_ID,
153            SUI_SYSTEM_MODULE_NAME.to_owned(),
154            WITHDRAW_STAKE_FUN_NAME.to_owned(),
155            vec![],
156            arguments,
157        ));
158    }
159    Ok(builder.finish())
160}
