61/// Evaluated bitmap state carried between blocking chunks.
62///
63/// The inverted-index evaluator yields `(bucket_id, RoaringBitmap)`. Holding the
64/// bitmap plus an absolute remaining range lets us resume a dense bucket without
65/// flattening it into an unbounded `Vec<u64>` or rereading it from RocksDB.
66#[derive(Clone)]
67pub(super) struct PendingBitmapBucket {
68    bucket_id: u64,
69    bitmap: RoaringBitmap,
70    remaining: Range<u64>,
71}
72
73impl PendingBitmapBucket {
74    fn new(
75        bucket_id: u64,
76        bitmap: RoaringBitmap,
77        range: &Range<u64>,
78        bucket_size: u64,
79    ) -> Option<Self> {
80        let bucket_start = bucket_id * bucket_size;
81        let bucket_end = bucket_start + bucket_size;
82        let remaining = range.start.max(bucket_start)..range.end.min(bucket_end);
83        (!remaining.is_empty() && !bitmap.is_empty()).then_some(Self {
84            bucket_id,
85            bitmap,
86            remaining,
87        })
88    }
89
90    fn drain_into(
91        &mut self,
92        out: &mut Vec<u64>,
93        hit_limit: usize,
94        bucket_size: u64,
95        direction: ScanDirection,
96    ) {
97        while out.len() < hit_limit {
98            let Some(seq) = self.next_seq(bucket_size, direction) else {
99                break;
100            };
101            out.push(seq);
102            self.remaining =
103                remaining_range_after(self.remaining.clone(), seq, direction.is_ascending())
104                    .unwrap_or(0..0);
105        }
106    }
107
108    fn has_hits(&self, bucket_size: u64, direction: ScanDirection) -> bool {
109        self.next_seq(bucket_size, direction).is_some()
110    }
111
112    fn next_seq(&self, bucket_size: u64, direction: ScanDirection) -> Option<u64> {
113        if self.remaining.is_empty() {
114            return None;
115        }
116
117        let bucket_start = self.bucket_id * bucket_size;
118        let lo = (self.remaining.start - bucket_start) as u32;
119        let hi = (self.remaining.end - bucket_start) as u32;
120        let bit = if direction.is_ascending() {
121            self.bitmap.range(lo..hi).next()
122        } else {
123            self.bitmap.range(lo..hi).next_back()
124        }?;
125        Some(bucket_start + u64::from(bit))
126    }
127
128    #[cfg(test)]
129    fn remaining(&self) -> Range<u64> {
130        self.remaining.clone()
131    }
132}

161pub(super) fn drain_bitmap_hits_with_budget(
162    service: RpcService,
163    kind: LedgerBitmapKind,
164    bucket_size: u64,
165    query: BitmapQuery,
166    pending_bucket: Option<PendingBitmapBucket>,
167    range: Option<Range<u64>>,
168    direction: ScanDirection,
169    hit_limit: usize,
170    scan_budget: usize,
171    cancel: &CancellationToken,
172) -> Result<DrainedBitmapHits, RpcError> {
173    if hit_limit == 0 {
174        return Ok(DrainedBitmapHits {
175            items: Vec::new(),
176            pending_bucket,
177            next_range: range,
178            buckets_scanned: 0,
179            coalesced_frontier: None,
180            chunk_scan_limit_reached: false,
181        });
182    }
183
184    // The iterator evaluator in `sui-inverted-index` is synchronous, so the
185    // borrowed RocksDB iterators stay on this blocking task for the whole chunk.
186    // That avoids a runtime dependency in this path while keeping BigTable on
187    // the async stream evaluator.
188    let scan_budget_start = scan_budget;
189    let scan_budget = BitmapScanBudget::new(scan_budget_start);
190    let source = RpcIndexesBitmapSource::new(
191        &service,
192        kind,
193        bucket_size,
194        scan_budget.clone(),
195        cancel.clone(),
196    );
197    let state = drain_watermarked_buckets(
198        move |scan_range| {
199            eval_bitmap_query_bucket_iter(
200                source,
201                query,
202                scan_range,
203                bucket_size,
204                direction,
205                SkipPolicy {
206                    drain_probe_rows: std::num::NonZeroU32::new(BITMAP_DRAIN_PROBE_ROWS),
207                },
208            )
209        },
210        pending_bucket,
211        range,
212        bucket_size,
213        direction,
214        hit_limit,
215        cancel,
216    )?;
217    Ok(DrainedBitmapHits {
218        items: state.items,
219        pending_bucket: state.pending_bucket,
220        next_range: state.next_range,
221        buckets_scanned: scan_budget_start - scan_budget.remaining(),
222        coalesced_frontier: state.coalesced_frontier,
223        chunk_scan_limit_reached: state.chunk_scan_limit_reached,
224    })
225}

231/// Loop state from draining a watermarked-bucket iterator, independent of budget
232/// accounting (the caller adds `buckets_scanned` from its `BitmapScanBudget`).
233struct DrainLoopState {
234    items: Vec<u64>,
235    pending_bucket: Option<PendingBitmapBucket>,
236    next_range: Option<Range<u64>>,
237    coalesced_frontier: Option<u64>,
238    chunk_scan_limit_reached: bool,
239}
240
241/// Drain matching member ids until `hit_limit` items, the iterator ends, or it
242/// signals [`ScanStop::ScanLimit`]. `open_iter` builds the iterator over a scan
243/// range and is invoked at most once, lazily — only when a fresh scan beyond
244/// `pending_bucket` is needed — so resuming a dense pending bucket never opens
245/// a RocksDB iterator. Ascending evaluator frontiers are the first unscanned
246/// position and remain inclusive; descending frontiers retain their existing
247/// exclusive upper-bound semantics. The caller decides whether a scan-limit
248/// stop is the per-request cap or a per-chunk cap.
249fn drain_watermarked_buckets<I>(
250    open_iter: impl FnOnce(Range<u64>) -> I,
251    mut pending_bucket: Option<PendingBitmapBucket>,
252    mut next_range: Option<Range<u64>>,
253    bucket_size: u64,
254    direction: ScanDirection,
255    hit_limit: usize,
256    cancel: &CancellationToken,
257) -> Result<DrainLoopState, RpcError>
258where
259    I: Iterator<Item = WatermarkedBucketItem>,
260{
261    let mut open_iter = Some(open_iter);
262    let mut iter: Option<I> = None;
263    let mut iter_range: Option<Range<u64>> = None;
264    let mut out = Vec::new();
265    // Furthest progress watermark seen this chunk. The evaluator emits watermarks
266    // monotonically in scan direction, so last-seen is furthest.
267    let mut coalesced_frontier: Option<u64> = None;
268    let mut chunk_scan_limit_reached = false;
269    while out.len() < hit_limit {
270        if cancel.is_cancelled() {
271            return Err(cancelled());
272        }
273        if let Some(mut bucket) = pending_bucket.take() {
274            bucket.drain_into(&mut out, hit_limit, bucket_size, direction);
275            if bucket.has_hits(bucket_size, direction) {
276                pending_bucket = Some(bucket);
277                break;
278            }
279            continue;
280        }
281
282        let Some(scan_range) = next_range.clone() else {
283            break;
284        };
285        if iter.is_none() {
286            let open = open_iter
287                .take()
288                .expect("bitmap query iterator is only opened once");
289            iter_range = Some(scan_range.clone());
290            iter = Some(open(scan_range));
291        }
292
293        let iter_range = iter_range
294            .as_ref()
295            .expect("bitmap iterator range is set before polling");
296        match iter
297            .as_mut()
298            .expect("bitmap iterator is set before polling")
299            .next()
300        {
301            Some(Ok(Watermarked::Item((bucket_id, bitmap)))) => {
302                next_range =
303                    remaining_range_after_bucket(iter_range, bucket_id, bucket_size, direction);
304                pending_bucket =
305                    PendingBitmapBucket::new(bucket_id, bitmap, iter_range, bucket_size);
306            }
307            // Progress watermark: advance the chunk frontier without consuming
308            // item budget. Lets the caller emit a scan watermark when the chunk
309            // matches nothing across a sparse gap.
310            Some(Ok(Watermarked::Watermark(pos))) => {
311                coalesced_frontier = Some(pos);
312            }
313            Some(Err(ScanStop::ScanLimit { scan_frontier })) => {
314                // Budget stop: the terminal carries the merged resume frontier
315                // (the exact value the stopping round's beacon would have held).
316                // Fold it into the chunk frontier. In ascending scans this is
317                // the first unscanned position, so resume inclusively without
318                // item-consumption's `+ 1`. Descending continuation remains
319                // strictly below the frontier.
320                chunk_scan_limit_reached = true;
321                coalesced_frontier = Some(scan_frontier);
322                next_range = if direction.is_ascending() {
323                    let remaining = scan_frontier.max(iter_range.start)..iter_range.end;
324                    (!remaining.is_empty()).then_some(remaining)
325                } else {
326                    remaining_range_after(iter_range.clone(), scan_frontier, false)
327                };
328                break;
329            }
330            // Cancelled stream → gRPC Cancelled status.
331            Some(Err(ScanStop::Cancelled)) => {
332                return Err(RpcError::new(
333                    tonic::Code::Cancelled,
334                    ScanStop::Cancelled.to_string(),
335                ));
336            }
337            // Genuine fault → gRPC Internal, carrying the error unchanged.
338            Some(Err(ScanStop::Fault(inner))) => {
339                return Err(RpcError::new(tonic::Code::Internal, inner.to_string()));
340            }
341            None => {
342                next_range = None;
343                break;
344            }
345        }
346    }
347    Ok(DrainLoopState {
348        items: out,
349        pending_bucket,
350        next_range,
351        coalesced_frontier,
352        chunk_scan_limit_reached,
353    })
354}

356fn remaining_range_after_bucket(
357    range: &Range<u64>,
358    bucket_id: u64,
359    bucket_size: u64,
360    direction: ScanDirection,
361) -> Option<Range<u64>> {
362    let remaining = if direction.is_ascending() {
363        ((bucket_id + 1) * bucket_size).max(range.start)..range.end
364    } else {
365        range.start..(bucket_id * bucket_size).min(range.end)
366    };
367    (!remaining.is_empty()).then_some(remaining)
368}

392/// Inner bucket source: a stored RocksDB scan, or the synthesized dense
393/// tx-universe sequence — `IndexDimension::TxUniverse` is query-only and
394/// never reaches storage.
395enum BitmapBucketIter<'a> {
396    Stored(LedgerBitmapBucketIterator<'a>),
397    Universe(DenseUniverseBuckets),
398}
399
400struct RpcIndexesBitmapIterator<'a> {
401    scan_budget: BitmapScanBudget,
402    cancel: CancellationToken,
403    iter: Option<BitmapBucketIter<'a>>,
404    finished: bool,
405    initial_error: Option<LeafStop>,
406    /// This leaf has not yet charged a bucket. Its first bucket is reserved
407    /// (always allowed) so every leaf emits its first watermark; see
408    /// [`BitmapScanBudget::take_first`].
409    first: bool,
410}
411
412impl<'a> RpcIndexesBitmapIterator<'a> {
413    fn new(
414        service: &'a RpcService,
415        kind: LedgerBitmapKind,
416        bucket_size: u64,
417        scan_budget: BitmapScanBudget,
418        cancel: CancellationToken,
419        dimension_key: Vec<u8>,
420        range: Range<u64>,
421        direction: ScanDirection,
422    ) -> Self {
423        if range.is_empty() {
424            return Self {
425                scan_budget,
426                cancel,
427                iter: None,
428                finished: true,
429                initial_error: None,
430                first: true,
431            };
432        }
433
434        // The tx universe is dense (every tx_seq in range is real), so it is
435        // synthesized rather than read from storage. Gated on the tx kind: the
436        // key is only ever produced by the tx filter layer, and in event-space
437        // a dense universe would be semantically wrong.
438        if kind == LedgerBitmapKind::Transaction
439            && dimension_key.first() == Some(&IndexDimension::TxUniverse.tag_byte())
440        {
441            return Self {
442                scan_budget,
443                cancel,
444                finished: false,
445                iter: Some(BitmapBucketIter::Universe(dense_universe_buckets(
446                    range,
447                    bucket_size,
448                    direction,
449                ))),
450                initial_error: None,
451                first: true,
452            };
453        }
454
455        let start_bucket = range.start / bucket_size;
456        let end_bucket_exclusive = (range.end - 1) / bucket_size + 1;
457        let iter = service
458            .reader
459            .inner()
460            .indexes()
461            .ok_or_else(|| anyhow::anyhow!("rpc indexes are disabled"))
462            .and_then(|indexes| {
463                let descending = !direction.is_ascending();
464                let iter = match kind {
465                    LedgerBitmapKind::Transaction => indexes.transaction_bitmap_bucket_iter(
466                        dimension_key,
467                        start_bucket,
468                        end_bucket_exclusive,
469                        descending,
470                    ),
471                    LedgerBitmapKind::Event => indexes.event_bitmap_bucket_iter(
472                        dimension_key,
473                        start_bucket,
474                        end_bucket_exclusive,
475                        descending,
476                    ),
477                };
478                iter.map_err(|e| anyhow::anyhow!(e.to_string()))
479            });
480
481        let (iter, initial_error) = match iter {
482            Ok(iter) => (Some(BitmapBucketIter::Stored(iter)), None),
483            Err(e) => (None, Some(e.into())),
484        };
485        Self {
486            scan_budget,
487            cancel,
488            finished: false,
489            iter,
490            initial_error,
491            first: true,
492        }
493    }
494
495    fn read_next_bucket(&mut self) -> Option<Result<(u64, RoaringBitmap), LeafStop>> {
496        if self.cancel.is_cancelled() {
497            self.finished = true;
498            return Some(Err(LeafStop::Cancelled));
499        }
500        let Some(iter) = self.iter.as_mut() else {
501            self.finished = true;
502            return None;
503        };
504
505        let next = match iter {
506            BitmapBucketIter::Stored(iter) => match iter.next() {
507                Some(Ok(bucket)) => Some(Ok((bucket.bucket_id, bucket.bitmap))),
508                Some(Err(e)) => Some(Err(LeafStop::Fault(anyhow::anyhow!(e.to_string())))),
509                None => None,
510            },
511            BitmapBucketIter::Universe(iter) => iter.next().map(Ok),
512        };
513        match next {
514            Some(Ok(bucket)) => {
515                if self.first {
516                    // Reserved: a leaf's first bucket is always allowed so it
517                    // emits its first watermark even if sibling leaves already
518                    // drained the shared budget. Without it a starved leaf
519                    // never reports a position, leaving the merged floor pinned
520                    // to the request bound and risking a resume loop. See
521                    // [`BitmapScanBudget::take_first`].
522                    self.scan_budget.take_first();
523                    self.first = false;
524                } else if let Err(e) = self.scan_budget.take_one() {
525                    self.finished = true;
526                    return Some(Err(e));
527                }
528                Some(Ok(bucket))
529            }
530            None => {
531                self.finished = true;
532                None
533            }
534            Some(Err(e)) => {
535                self.finished = true;
536                Some(Err(e))
537            }
538        }
539    }
540}

542impl Iterator for RpcIndexesBitmapIterator<'_> {
543    type Item = Result<(u64, RoaringBitmap), LeafStop>;
544
545    fn next(&mut self) -> Option<Self::Item> {
546        if let Some(e) = self.initial_error.take() {
547            self.finished = true;
548            return Some(Err(e));
549        }
550        if self.finished {
551            return None;
552        }
553
554        self.read_next_bucket()
555    }
556}
557impl SeekableBucketIterator for RpcIndexesBitmapIterator<'_> {
558    fn seek_bucket(&mut self, bucket: u64) {
559        match self.iter.as_mut() {
560            Some(BitmapBucketIter::Stored(iter)) => iter.seek_bucket(bucket),
561            Some(BitmapBucketIter::Universe(iter)) => iter.seek_bucket(bucket),
562            None => {}
563        }
564    }

567#[derive(Clone)]
568struct BitmapScanBudget {
569    remaining: Rc<Cell<usize>>,
570}
571
572impl BitmapScanBudget {
573    fn new(scan_budget: usize) -> Self {
574        Self {
575            remaining: Rc::new(Cell::new(scan_budget)),
576        }
577    }
578
579    fn take_one(&self) -> Result<(), LeafStop> {
580        let remaining = self.remaining.get();
581        if remaining == 0 {
582            return Err(LeafStop::BudgetExhausted);
583        }
584        self.remaining.set(remaining - 1);
585        Ok(())
586    }
587
588    /// Charge a leaf's mandatory first bucket: decrements when it can but
589    /// always succeeds, so every leaf is guaranteed to emit its first
590    /// watermark regardless of how sibling leaves drained the shared pool.
591    /// Saturating at 0 only undercounts a first bucket taken after the pool
592    /// was already exhausted; the common `budget >> leaves` case stays exact.
593    fn take_first(&self) {
594        self.remaining.set(self.remaining.get().saturating_sub(1));
595    }
596
597    fn remaining(&self) -> usize {
598        self.remaining.get()
599    }
600}

680    fn drain(
681        events: Vec<WatermarkedBucketItem>,
682        pending: Option<PendingBitmapBucket>,
683        range: Option<Range<u64>>,
684        bucket_size: u64,
685        direction: ScanDirection,
686        hit_limit: usize,
687    ) -> DrainLoopState {
688        let cancel = CancellationToken::new();
689        drain_watermarked_buckets(
690            move |_range| events.into_iter(),
691            pending,
692            range,
693            bucket_size,
694            direction,
695            hit_limit,
696            &cancel,
697        )
698        .expect("drain succeeds")
699    }
