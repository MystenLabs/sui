17/// Chunk terminal for a scan that either exhausted its request scan budget
18/// (authoritative frontier watermark, built lazily) or ended at the resolved
19/// range boundary. Chunks only run over nonempty intervals, so the range
20/// terminal is never `interval_empty`.
21pub(crate) fn scan_limit_or_range<Frontier>(
22    request_scan_limit_reached: bool,
23    exhaustion: RangeExhaustion,
24    position: Position,
25    frontier_watermark: Frontier,
26) -> Result<ScanTerminal, RpcError>
27where
28    Frontier: FnOnce() -> Result<Watermark, RpcError>,
29{
30    if request_scan_limit_reached {
31        Ok(ScanTerminal::ScanLimit {
32            watermark: frontier_watermark()?,
33        })
34    } else {
35        Ok(ScanTerminal::from_range_exhaustion(
36            exhaustion, position, false,
37        ))
38    }
39}

41pub(crate) struct ScanChunkDone<State, Item> {
42    /// Frames to yield: rendered item frames plus, only when the chunk produced
43    /// no items, at most one scan watermark frame.
44    pub(crate) items: Vec<Item>,
45    /// Count of real item frames in `items` (excludes any scan watermark).
46    /// Drives the request item limit, not `items.len()`.
47    pub(crate) produced: usize,
48    pub(crate) next_state: Option<State>,
49    pub(crate) terminal: ScanTerminal,
50    pub(crate) remaining_scan_budget: usize,
51}
52
53pub(crate) struct ChunkArgs {
54    pub(crate) scan_budget: usize,
55    pub(crate) chunk_item_limit: usize,
56    pub(crate) remaining_request_item_limit: usize,
57    pub(crate) cancel: CancellationToken,
58}

60pub(super) fn cancelled() -> RpcError {
61    RpcError::new(tonic::Code::Cancelled, "request cancelled")
62}
63pub(crate) fn spawn_list_chunk<T, F>(
64    metrics: Option<ListStreamMetrics>,
65    work: F,
66) -> JoinHandle<Result<T, RpcError>>
67where
68    T: Send + 'static,
69    F: FnOnce(Option<&ListStreamMetrics>) -> Result<T, RpcError> + Send + 'static,
70{
71    match metrics {
72        Some(metrics) => {
73            let queue_timer = metrics.start_queue_timer();
74            tokio::task::spawn_blocking(move || {
75                queue_timer.stop_and_record();
76                let _work_timer = metrics.start_work_timer();
77                work(Some(&metrics))
78            })
79        }
80        None => tokio::task::spawn_blocking(move || work(None)),
81    }
82}

84/// Bridges the async gRPC stream to blocking RocksDB reads.
85///
86/// Each request has at most one blocking chunk worker in flight. A worker builds
87/// a whole chunk and returns it as a `Vec`, so the blocking task is never parked
88/// behind gRPC stream backpressure. Once a chunk completes, the next worker is
89/// scheduled before the async stream drains the completed chunk's buffered
90/// items, letting slow clients delay response delivery without holding a Rocks
91/// iterator or blocking thread open.
92pub(crate) struct ChunkedScan<State, Item, Spawn>
93where
94    State: Send + 'static,
95    Item: Send + 'static,
96    Spawn: FnMut(State, ChunkArgs) -> JoinHandle<Result<ScanChunkDone<State, Item>, RpcError>>
97        + Send
98        + 'static,
99{
100    current: Option<JoinHandle<Result<ScanChunkDone<State, Item>, RpcError>>>,
101    buffered: VecDeque<Item>,
102    spawn: Spawn,
103    produced: usize,
104    initial_scan_budget: usize,
105    scan_budget: usize,
106    terminal: Option<ScanTerminal>,
107    limit_items: usize,
108    chunk_max: usize,
109    cancel: CancellationToken,
110    _cancel_guard: DropGuard,
111}
112
113impl<State, Item, Spawn> ChunkedScan<State, Item, Spawn>
114where
115    State: Send + 'static,
116    Item: Send + 'static,
117    Spawn: FnMut(State, ChunkArgs) -> JoinHandle<Result<ScanChunkDone<State, Item>, RpcError>>
118        + Send
119        + 'static,
120{
121    pub(crate) fn new(
122        initial_state: State,
123        limit_items: usize,
124        chunk_max: usize,
125        scan_budget: usize,
126        mut spawn: Spawn,
127    ) -> Self {
128        assert!(limit_items > 0, "chunked scan limit must be nonzero");
129        assert!(chunk_max > 0, "chunked scan chunk size must be nonzero");
130        let initial_chunk_item_limit = chunk_max.min(limit_items);
131        let cancel = CancellationToken::new();
132        let current = Some(spawn(
133            initial_state,
134            ChunkArgs {
135                scan_budget,
136                chunk_item_limit: initial_chunk_item_limit,
137                remaining_request_item_limit: limit_items,
138                cancel: cancel.clone(),
139            },
140        ));
141        let _cancel_guard = cancel.clone().drop_guard();
142        Self {
143            current,
144            buffered: VecDeque::new(),
145            spawn,
146            produced: 0,
147            initial_scan_budget: scan_budget,
148            scan_budget,
149            terminal: None,
150            limit_items,
151            chunk_max,
152            cancel,
153            _cancel_guard,
154        }
155    }
156
157    pub(crate) async fn next_item(&mut self) -> Result<Option<Item>, RpcError> {
158        loop {
159            if let Some(item) = self.buffered.pop_front() {
160                return Ok(Some(item));
161            }
162
163            let Some(chunk) = self.current.take() else {
164                return Ok(None);
165            };
166
167            let done = chunk
168                .await
169                .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))??;
170            self.apply_done(done);
171        }
172    }
173
174    pub(crate) fn into_terminal(self) -> Option<ScanTerminal> {
175        self.terminal
176    }
177
178    /// Total real items produced across all chunks (excludes watermark frames).
179    pub(crate) fn produced(&self) -> usize {
180        self.produced
181    }
182
183    pub(crate) fn bitmap_buckets_evaluated(&self) -> usize {
184        self.initial_scan_budget - self.scan_budget
185    }
186
187    /// True when no buffered frame remains and no further chunk is in flight —
188    /// the next `next_item()` call would return `None`.
189    pub(crate) fn exhausted(&self) -> bool {
190        self.buffered.is_empty() && self.current.is_none()
191    }
192
193    #[cfg(test)]
194    pub(super) fn cancel_token(&self) -> CancellationToken {
195        self.cancel.clone()
196    }
197
198    fn apply_done(&mut self, done: ScanChunkDone<State, Item>) {
199        self.terminal = Some(done.terminal);
200        self.scan_budget = done.remaining_scan_budget;
201        self.produced = self.produced.saturating_add(done.produced);
202        self.buffered = done.items.into();
203        if self.produced < self.limit_items
204            && let Some(state) = done.next_state
205        {
206            let remaining_request_item_limit = self.limit_items - self.produced;
207            let chunk_item_limit = self.chunk_max.min(remaining_request_item_limit);
208            self.current = Some((self.spawn)(
209                state,
210                ChunkArgs {
211                    scan_budget: self.scan_budget,
212                    chunk_item_limit,
213                    remaining_request_item_limit,
214                    cancel: self.cancel.clone(),
215                },
216            ));
217        }
218    }
219}

235    #[tokio::test]
236    async fn chunked_scan_drains_vector_chunks_in_order() {
237        let calls = Arc::new(Mutex::new(Vec::new()));
238        let spawn_calls = calls.clone();
239        let mut scan = ChunkedScan::new(0usize, 5, 2, 10, move |state, args: ChunkArgs| {
240            spawn_calls.lock().unwrap().push((
241                state,
242                args.scan_budget,
243                args.chunk_item_limit,
244                args.remaining_request_item_limit,
245            ));
246            let scan_budget = args.scan_budget;
247            let _ = args.cancel;
248            tokio::task::spawn_blocking(move || {
249                let items = match state {
250                    0 => vec![0, 1],
251                    1 => vec![10, 11],
252                    2 => vec![20],
253                    _ => Vec::new(),
254                };
255                Ok(ScanChunkDone {
256                    produced: items.len(),
257                    items,
258                    next_state: (state < 2).then_some(state + 1),
259                    terminal: ScanTerminal::from_range_exhaustion(
260                        RangeExhaustion::CheckpointBound,
261                        Position::Transactions {
262                            checkpoint: 0,
263                            tx_seq: 0,
264                        },
265                        false,
266                    ),
267                    remaining_scan_budget: scan_budget - 1,
268                })
269            })
270        });
271
272        let mut items = Vec::new();
273        while let Some(item) = scan.next_item().await.unwrap() {
274            items.push(item);
275        }
276
277        assert_eq!(items, vec![0, 1, 10, 11, 20]);
278        assert_eq!(
279            scan.into_terminal(),
280            Some(ScanTerminal::NaturalRange {
281                end: NaturalRangeEnd::CheckpointBound,
282                position: Position::Transactions {
283                    checkpoint: 0,
284                    tx_seq: 0,
285                },
286                interval_empty: false,
287            })
288        );
289        assert_eq!(
290            *calls.lock().unwrap(),
291            vec![(0, 10, 2, 5), (1, 9, 2, 3), (2, 8, 1, 1)]
292        );
293    }

345    #[tokio::test]
346    async fn limit_reaching_final_item_exhausts_without_scheduling_another_chunk() {
347        use std::sync::atomic::AtomicUsize;
348        use std::sync::atomic::Ordering;
349
350        let spawn_count = Arc::new(AtomicUsize::new(0));
351        let spawn_count_for_worker = spawn_count.clone();
352        let mut scan = ChunkedScan::new(0usize, 2, 2, 10, move |_state, args: ChunkArgs| {
353            spawn_count_for_worker.fetch_add(1, Ordering::SeqCst);
354            tokio::task::spawn_blocking(move || {
355                Ok(ScanChunkDone {
356                    items: vec!["first", "final"],
357                    produced: 2,
358                    next_state: Some(1),
359                    terminal: ScanTerminal::ScanLimit {
360                        watermark: {
361                            let mut watermark = Watermark::default();
362                            watermark.cursor = Some(b"scan-frontier".to_vec().into());
363                            watermark
364                        },
365                    },
366                    remaining_scan_budget: args.scan_budget - 1,
367                })
368            })
369        });
370
371        assert_eq!(scan.next_item().await.unwrap(), Some("first"));
372        assert_eq!(scan.produced(), 2);
373        assert!(!scan.exhausted(), "the final item is still buffered");
374
375        assert_eq!(scan.next_item().await.unwrap(), Some("final"));
376        assert_eq!(scan.produced(), 2);
377        assert!(
378            scan.exhausted(),
379            "the item limit leaves no trailing frame or chunk in flight"
380        );
381        assert_eq!(scan.next_item().await.unwrap(), None);
382        let terminal = scan.into_terminal().expect("worker terminal is retained");
383        let ScanTerminal::ScanLimit { watermark } = terminal else {
384            panic!("expected scan-limit terminal");
385        };
386        assert!(
387            watermark.cursor.is_some(),
388            "scan-limit terminal must own a resume cursor"
389        );
390        assert_eq!(
391            spawn_count.load(Ordering::SeqCst),
392            1,
393            "next_state must not schedule another chunk at the item limit"
394        );
395    }

397    async fn spawn_list_chunk_observes_queue_and_work_on_success_and_error() {
398        let registry = Registry::new();
399        let metrics = ListApiMetrics::new(&registry).stream_metrics("list_checkpoints", "summary");
400
401        let success = spawn_list_chunk(Some(metrics.clone()), |_| Ok(()));
402        let error = spawn_list_chunk::<(), _>(Some(metrics), |_| {

415            .find(|family| family.name() == "list_chunk_seconds")
416            .expect("list chunk metric family");

418        for phase in ["queue", "work"] {
419            let metric = family
420                .get_metric()
421                .iter()
422                .find(|metric| {
423                    metric
424                        .get_label()
425                        .iter()
426                        .any(|label| label.name() == "phase" && label.value() == phase)
427                })
428                .unwrap_or_else(|| panic!("missing {phase} metric"));
429            assert_eq!(metric.get_histogram().get_sample_count(), 2, "{phase}");

433    #[tokio::test]
434    async fn chunked_scan_accumulates_bitmap_buckets_across_chunks() {
435        let mut scan = ChunkedScan::new(0usize, 1, 1, 10, move |state, args: ChunkArgs| {
436            tokio::task::spawn_blocking(move || {
437                let buckets_evaluated = if state == 0 { 2 } else { 3 };
438                Ok(ScanChunkDone::<usize, usize> {
439                    items: Vec::new(),
440                    produced: 0,
441                    next_state: (state == 0).then_some(1),
442                    terminal: ScanTerminal::from_range_exhaustion(
443                        RangeExhaustion::CheckpointBound,
444                        Position::Transactions {
445                            checkpoint: 0,
446                            tx_seq: 0,
447                        },
448                        false,
449                    ),
450                    remaining_scan_budget: args.scan_budget - buckets_evaluated,
451                })
452            })
453        });
454
455        assert_eq!(scan.next_item().await.unwrap(), None);
456        assert_eq!(scan.bitmap_buckets_evaluated(), 5);
457
458        let filtered_registry = Registry::new();
459        let mut filtered_metrics = ListRequestMetrics::new(
460            Some(
461                ListApiMetrics::new(&filtered_registry)
462                    .stream_metrics("list_checkpoints", "summary"),
463            ),
464            Instant::now(),
465        );
466        filtered_metrics.finish_success(QueryEndReason::LedgerTip, Some(5));
467        let filtered_family = filtered_registry
468            .gather()
469            .into_iter()
470            .find(|family| family.name() == "list_bitmap_buckets_evaluated")
471            .expect("filtered bitmap metric family");
472        assert_eq!(filtered_family.get_metric().len(), 1);
473        let filtered_histogram = filtered_family.get_metric()[0].get_histogram();
474        assert_eq!(filtered_histogram.get_sample_count(), 1);
475        assert_eq!(filtered_histogram.get_sample_sum(), 5.0);
476
477        let unfiltered_registry = Registry::new();
478        let mut unfiltered_metrics = ListRequestMetrics::new(
479            Some(
480                ListApiMetrics::new(&unfiltered_registry)
481                    .stream_metrics("list_checkpoints", "summary"),
482            ),
483            Instant::now(),
484        );
485        unfiltered_metrics.finish_success(QueryEndReason::LedgerTip, None);
486        let unfiltered_family = unfiltered_registry
487            .gather()
488            .into_iter()
489            .find(|family| family.name() == "list_bitmap_buckets_evaluated")
490            .expect("unfiltered bitmap metric family");
491        assert_eq!(
492            unfiltered_family.get_metric()[0]
493                .get_histogram()
494                .get_sample_count(),
495            0
496        );
497    }
