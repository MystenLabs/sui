4use crate::ErrorReason;
5use crate::RpcError;
6use crate::RpcService;
7use crate::error::CheckpointNotFoundError;
8use prost_types::FieldMask;
9use sui_rpc::field::FieldMaskTree;
10use sui_rpc::field::FieldMaskUtil;
11use sui_rpc::merge::Merge;
12use sui_rpc::proto::google::rpc::bad_request::FieldViolation;
13use sui_rpc::proto::sui::rpc::v2::Checkpoint;
14use sui_rpc::proto::sui::rpc::v2::Event;
15use sui_rpc::proto::sui::rpc::v2::ExecutedTransaction;
16use sui_rpc::proto::sui::rpc::v2::GetCheckpointRequest;
17use sui_rpc::proto::sui::rpc::v2::GetCheckpointResponse;
18use sui_rpc::proto::sui::rpc::v2::ObjectSet;
19use sui_rpc::proto::sui::rpc::v2::TransactionEvents;
20use sui_rpc::proto::sui::rpc::v2::get_checkpoint_request::CheckpointId;
21use sui_sdk_types::Digest;
22
23pub const READ_MASK_DEFAULT: &str = "sequence_number,digest";
24
25#[tracing::instrument(skip(service))]
26pub fn get_checkpoint(
27    service: &RpcService,
28    request: GetCheckpointRequest,
29) -> Result<GetCheckpointResponse, RpcError> {
30    let read_mask = {
31        let read_mask = request
32            .read_mask
33            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
34        read_mask.validate::<Checkpoint>().map_err(|path| {
35            FieldViolation::new("read_mask")
36                .with_description(format!("invalid read_mask path: {path}"))
37                .with_reason(ErrorReason::FieldInvalid)
38        })?;
39        FieldMaskTree::from(read_mask)
40    };
41
42    let verified_summary = match request.checkpoint_id {
43        Some(CheckpointId::SequenceNumber(s)) => service
44            .reader
45            .inner()
46            .get_checkpoint_by_sequence_number(s)
47            .ok_or(CheckpointNotFoundError::sequence_number(s))?,
48        Some(CheckpointId::Digest(digest)) => {
49            let digest = digest.parse::<Digest>().map_err(|e| {
50                FieldViolation::new("digest")
51                    .with_description(format!("invalid digest: {e}"))
52                    .with_reason(ErrorReason::FieldInvalid)
53            })?;
54
55            service
56                .reader
57                .inner()
58                .get_checkpoint_by_digest(&digest.into())
59                .ok_or(CheckpointNotFoundError::digest(digest))?
60        }
61        None => service.reader.inner().get_latest_checkpoint()?,
62        _ => service.reader.inner().get_latest_checkpoint()?,
63    };
64
65    let summary = verified_summary.data();
66    let signature = verified_summary.auth_sig();
67    let sequence_number = summary.sequence_number;
68    let timestamp_ms = summary.timestamp_ms;
69
70    let mut checkpoint = Checkpoint::default();
71
72    checkpoint.merge(summary, &read_mask);
73    checkpoint.merge(signature.clone(), &read_mask);
74
75    if read_mask.contains(Checkpoint::CONTENTS_FIELD.name)
76        || read_mask.contains(Checkpoint::TRANSACTIONS_FIELD.name)
77        || read_mask.contains(Checkpoint::OBJECTS_FIELD.name)
78    {
79        let core_contents = service
80            .reader
81            .inner()
82            .get_checkpoint_contents_by_sequence_number(sequence_number)
83            .ok_or(CheckpointNotFoundError::sequence_number(sequence_number))?;
84
85        if read_mask.contains(Checkpoint::CONTENTS_FIELD.name) {
86            checkpoint.merge(core_contents.clone(), &read_mask);
87        }
88
89        if read_mask.contains(Checkpoint::TRANSACTIONS_FIELD.name)
90            || read_mask.contains(Checkpoint::OBJECTS_FIELD.name)
91        {
92            let checkpoint_data = service
93                .reader
94                .inner()
95                .get_checkpoint_data(verified_summary, core_contents)?;
96
97            if let Some(submask) = read_mask
98                .subtree(Checkpoint::OBJECTS_FIELD)
99                .and_then(|submask| submask.subtree(ObjectSet::OBJECTS_FIELD))
100            {
101                let set = checkpoint_data
102                    .object_set
103                    .iter()
104                    .map(|o| sui_rpc::proto::sui::rpc::v2::Object::merge_from(o, &submask))
105                    .collect();
106                checkpoint.objects = Some(ObjectSet::default().with_objects(set));
107            }
108
109            if let Some(submask) = read_mask.subtree(Checkpoint::TRANSACTIONS_FIELD.name) {
110                checkpoint.transactions = checkpoint_data
111                    .transactions
112                    .into_iter()
113                    .map(|t| {
114                        let balance_changes = submask
115                            .contains(ExecutedTransaction::BALANCE_CHANGES_FIELD)
116                            .then(|| {
117                                service
118                                    .reader
119                                    .get_transaction_info(&t.transaction.digest())
120                                    .map(|info| {
121                                        info.balance_changes
122                                            .into_iter()
123                                            .map(sui_rpc::proto::sui::rpc::v2::BalanceChange::from)
124                                            .collect::<Vec<_>>()
125                                    })
126                            })
127                            .flatten()
128                            .unwrap_or_default();
129                        let mut transaction = ExecutedTransaction::merge_from(&t, &submask);
130                        transaction.checkpoint = submask
131                            .contains(ExecutedTransaction::CHECKPOINT_FIELD)
132                            .then_some(sequence_number);
133                        transaction.timestamp = submask
134                            .contains(ExecutedTransaction::TIMESTAMP_FIELD)
135                            .then(|| sui_rpc::proto::timestamp_ms_to_proto(timestamp_ms));
136                        transaction.balance_changes = balance_changes;
137
138                        if let Some(events_mask) =
139                            submask.subtree(ExecutedTransaction::EVENTS_FIELD.name)
140                            && let Some(event_mask) =
141                                events_mask.subtree(TransactionEvents::EVENTS_FIELD.name)
142                            && event_mask.contains(Event::JSON_FIELD.name)
143                            && let Some(events) = transaction.events.as_mut()
144                            && let Some(sdk_events) = &t.events
145                        {
146                            for (message, event) in events.events.iter_mut().zip(&sdk_events.data) {
147                                message.json = crate::grpc::v2::render_json(
148                                    service,
149                                    &event.type_,
150                                    &event.contents,
151                                )
152                                .map(Box::new);
153                            }
154                        }
155
156                        transaction
157                    })
158                    .collect();
159            }
160        }
161    }
162
163    Ok(GetCheckpointResponse::new(checkpoint))
164}
