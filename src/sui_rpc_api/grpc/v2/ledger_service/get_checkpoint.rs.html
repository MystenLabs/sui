4use crate::ErrorReason;
5use crate::RpcError;
6use crate::RpcService;
7use crate::error::CheckpointNotFoundError;
8use prost_types::FieldMask;
9use sui_rpc::field::FieldMaskTree;
10use sui_rpc::field::FieldMaskUtil;
11use sui_rpc::merge::Merge;
12use sui_rpc::proto::google::rpc::bad_request::FieldViolation;
13use sui_rpc::proto::sui::rpc::v2::Checkpoint;
14use sui_rpc::proto::sui::rpc::v2::Event;
15use sui_rpc::proto::sui::rpc::v2::ExecutedTransaction;
16use sui_rpc::proto::sui::rpc::v2::GetCheckpointRequest;
17use sui_rpc::proto::sui::rpc::v2::GetCheckpointResponse;
18use sui_rpc::proto::sui::rpc::v2::ObjectSet;
19use sui_rpc::proto::sui::rpc::v2::TransactionEvents;
20use sui_rpc::proto::sui::rpc::v2::get_checkpoint_request::CheckpointId;
21use sui_sdk_types::Digest;
22use sui_types::balance_change::derive_balance_changes_2;
23
24pub const READ_MASK_DEFAULT: &str = "sequence_number,digest";
25
26#[tracing::instrument(skip(service))]
27pub fn get_checkpoint(
28    service: &RpcService,
29    request: GetCheckpointRequest,
30) -> Result<GetCheckpointResponse, RpcError> {
31    let read_mask = {
32        let read_mask = request
33            .read_mask
34            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
35        read_mask.validate::<Checkpoint>().map_err(|path| {
36            FieldViolation::new("read_mask")
37                .with_description(format!("invalid read_mask path: {path}"))
38                .with_reason(ErrorReason::FieldInvalid)
39        })?;
40        FieldMaskTree::from(read_mask)
41    };
42
43    let verified_summary = match request.checkpoint_id {
44        Some(CheckpointId::SequenceNumber(s)) => service
45            .reader
46            .inner()
47            .get_checkpoint_by_sequence_number(s)
48            .ok_or(CheckpointNotFoundError::sequence_number(s))?,
49        Some(CheckpointId::Digest(digest)) => {
50            let digest = digest.parse::<Digest>().map_err(|e| {
51                FieldViolation::new("digest")
52                    .with_description(format!("invalid digest: {e}"))
53                    .with_reason(ErrorReason::FieldInvalid)
54            })?;
55
56            service
57                .reader
58                .inner()
59                .get_checkpoint_by_digest(&digest.into())
60                .ok_or(CheckpointNotFoundError::digest(digest))?
61        }
62        None => service.reader.inner().get_latest_checkpoint()?,
63        _ => service.reader.inner().get_latest_checkpoint()?,
64    };
65
66    let summary = verified_summary.data();
67    let signature = verified_summary.auth_sig();
68    let sequence_number = summary.sequence_number;
69    let timestamp_ms = summary.timestamp_ms;
70
71    let latest_checkpoint = service
72        .reader
73        .inner()
74        .get_latest_checkpoint()?
75        .sequence_number;
76    let lowest_available_checkpoint = service.reader.get_lowest_available_checkpoint()?;
77
78    if !(lowest_available_checkpoint..=latest_checkpoint).contains(&sequence_number) {
79        return Err(CheckpointNotFoundError::sequence_number(sequence_number).into());
80    }
81
82    let mut checkpoint = Checkpoint::default();
83
84    checkpoint.merge(summary, &read_mask);
85    checkpoint.merge(signature.clone(), &read_mask);
86
87    if read_mask.contains(Checkpoint::CONTENTS_FIELD.name)
88        || read_mask.contains(Checkpoint::TRANSACTIONS_FIELD.name)
89        || read_mask.contains(Checkpoint::OBJECTS_FIELD.name)
90    {
91        let core_contents = service
92            .reader
93            .inner()
94            .get_checkpoint_contents_by_sequence_number(sequence_number)
95            .ok_or(CheckpointNotFoundError::sequence_number(sequence_number))?;
96
97        if read_mask.contains(Checkpoint::CONTENTS_FIELD.name) {
98            checkpoint.merge(core_contents.clone(), &read_mask);
99        }
100
101        if read_mask.contains(Checkpoint::TRANSACTIONS_FIELD.name)
102            || read_mask.contains(Checkpoint::OBJECTS_FIELD.name)
103        {
104            let checkpoint_data = service
105                .reader
106                .inner()
107                .get_checkpoint_data(verified_summary, core_contents)?;
108
109            if let Some(submask) = read_mask
110                .subtree(Checkpoint::OBJECTS_FIELD)
111                .and_then(|submask| submask.subtree(ObjectSet::OBJECTS_FIELD))
112            {
113                let set = checkpoint_data
114                    .object_set
115                    .iter()
116                    .map(|o| sui_rpc::proto::sui::rpc::v2::Object::merge_from(o, &submask))
117                    .collect();
118                checkpoint.objects = Some(ObjectSet::default().with_objects(set));
119            }
120
121            if let Some(submask) = read_mask.subtree(Checkpoint::TRANSACTIONS_FIELD.name) {
122                checkpoint.transactions = checkpoint_data
123                    .transactions
124                    .into_iter()
125                    .map(|t| {
126                        let balance_changes =
127                            if submask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD) {
128                                derive_balance_changes_2(&t.effects, &checkpoint_data.object_set)
129                                    .into_iter()
130                                    .map(Into::into)
131                                    .collect()
132                            } else {
133                                Vec::new()
134                            };
135                        let mut transaction = ExecutedTransaction::merge_from(&t, &submask);
136                        transaction.checkpoint = submask
137                            .contains(ExecutedTransaction::CHECKPOINT_FIELD)
138                            .then_some(sequence_number);
139                        transaction.timestamp = submask
140                            .contains(ExecutedTransaction::TIMESTAMP_FIELD)
141                            .then(|| sui_rpc::proto::timestamp_ms_to_proto(timestamp_ms));
142                        transaction.balance_changes = balance_changes;
143
144                        if let Some(events_mask) =
145                            submask.subtree(ExecutedTransaction::EVENTS_FIELD.name)
146                            && let Some(event_mask) =
147                                events_mask.subtree(TransactionEvents::EVENTS_FIELD.name)
148                            && event_mask.contains(Event::JSON_FIELD.name)
149                            && let Some(events) = transaction.events.as_mut()
150                            && let Some(sdk_events) = &t.events
151                        {
152                            for (message, event) in events.events.iter_mut().zip(&sdk_events.data) {
153                                message.json = service
154                                    .render_json(&event.type_, &event.contents)
155                                    .map(Box::new);
156                            }
157                        }
158
159                        transaction
160                    })
161                    .collect();
162            }
163        }
164    }
165
166    Ok(GetCheckpointResponse::new(checkpoint))
167}
