4use crate::RpcError;
5use crate::RpcService;
6use sui_rpc::proto::sui::rpc::v2::GetServiceInfoResponse;
7use sui_rpc::proto::timestamp_ms_to_proto;
8use sui_sdk_types::Digest;
9use tap::Pipe;
10
11#[tracing::instrument(skip(service))]
12pub fn get_service_info(service: &RpcService) -> Result<GetServiceInfoResponse, RpcError> {
13    let latest_checkpoint = service.reader.inner().get_latest_checkpoint()?;
14    let lowest_available_checkpoint = service
15        .reader
16        .inner()
17        .get_lowest_available_checkpoint()?
18        .pipe(Some);
19    let lowest_available_checkpoint_objects = service
20        .reader
21        .inner()
22        .get_lowest_available_checkpoint_objects()?
23        .pipe(Some);
24
25    let mut message = GetServiceInfoResponse::default();
26    message.chain_id = Some(Digest::new(service.chain_id().as_bytes().to_owned()).to_string());
27    message.chain = Some(service.chain_id().chain().as_str().into());
28    message.epoch = Some(latest_checkpoint.epoch());
29    message.checkpoint_height = Some(latest_checkpoint.sequence_number);
30    message.timestamp = Some(timestamp_ms_to_proto(latest_checkpoint.timestamp_ms));
31    message.lowest_available_checkpoint = lowest_available_checkpoint;
32    message.lowest_available_checkpoint_objects = lowest_available_checkpoint_objects;
33    message.server = service.server_version().map(ToString::to_string);
34    Ok(message)
35}
