27pub const READ_MASK_DEFAULT: &str = "digest";
28
29#[tracing::instrument(skip(service))]
30pub fn get_transaction(
31    service: &RpcService,
32    request: GetTransactionRequest,
33) -> Result<GetTransactionResponse, RpcError> {
34    let transaction_digest = request
35        .digest
36        .ok_or_else(|| {
37            FieldViolation::new("digest")
38                .with_description("missing digest")
39                .with_reason(ErrorReason::FieldMissing)
40        })?
41        .parse::<Digest>()
42        .map_err(|e| {
43            FieldViolation::new("digest")
44                .with_description(format!("invalid digest: {e}"))
45                .with_reason(ErrorReason::FieldInvalid)
46        })?;
47
48    let read_mask = {
49        let read_mask = request
50            .read_mask
51            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
52        read_mask
53            .validate::<ExecutedTransaction>()
54            .map_err(|path| {
55                FieldViolation::new("read_mask")
56                    .with_description(format!("invalid read_mask path: {path}"))
57                    .with_reason(ErrorReason::FieldInvalid)
58            })?;
59        FieldMaskTree::from(read_mask)
60    };
61
62    let transaction_read = service.reader.get_transaction_read(transaction_digest)?;
63
64    let transaction = transaction_to_response(service, transaction_read, &read_mask);
65
66    Ok(GetTransactionResponse::new(transaction))
67}

69#[tracing::instrument(skip(service))]
70pub fn batch_get_transactions(
71    service: &RpcService,
72    BatchGetTransactionsRequest {
73        digests, read_mask, ..
74    }: BatchGetTransactionsRequest,
75) -> Result<BatchGetTransactionsResponse, RpcError> {
76    let read_mask = {
77        let read_mask = read_mask.unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
78        read_mask
79            .validate::<ExecutedTransaction>()
80            .map_err(|path| {
81                FieldViolation::new("read_mask")
82                    .with_description(format!("invalid read_mask path: {path}"))
83                    .with_reason(ErrorReason::FieldInvalid)
84            })?;
85        FieldMaskTree::from(read_mask)
86    };
87
88    let transactions = digests
89        .into_iter()
90        .enumerate()
91        .map(|(idx, digest)| {
92            let digest = digest.parse().map_err(|e| {
93                FieldViolation::new_at("digests", idx)
94                    .with_description(format!("invalid digest: {e}"))
95                    .with_reason(ErrorReason::FieldInvalid)
96            })?;
97
98            service
99                .reader
100                .get_transaction_read(digest)
101                .map(|transaction_read| {
102                    transaction_to_response(service, transaction_read, &read_mask)
103                })
104        })
105        .map(|result| match result {
106            Ok(transaction) => GetTransactionResult::new_transaction(transaction),
107            Err(error) => GetTransactionResult::new_error(error.into_status_proto()),
108        })
109        .collect();
110
111    Ok(BatchGetTransactionsResponse::new(transactions))
112}

114fn transaction_to_response(
115    service: &RpcService,
116    source: crate::reader::TransactionRead,
117    mask: &FieldMaskTree,
118) -> ExecutedTransaction {
119    let mut message = ExecutedTransaction::default();
120
121    if mask.contains(ExecutedTransaction::DIGEST_FIELD.name) {
122        message.digest = Some(source.digest.to_string());
123    }
124
125    if let Some(submask) = mask.subtree(ExecutedTransaction::TRANSACTION_FIELD.name) {
126        message.transaction = Some(Transaction::merge_from(source.transaction, &submask));
127    }
128
129    if let Some(submask) = mask.subtree(ExecutedTransaction::SIGNATURES_FIELD.name) {
130        message.signatures = source
131            .signatures
132            .into_iter()
133            .map(|s| UserSignature::merge_from(s, &submask))
134            .collect();
135    }
136
137    if let Some(submask) = mask.subtree(ExecutedTransaction::EFFECTS_FIELD.name) {
138        let mut effects = TransactionEffects::merge_from(&source.effects, &submask);
139
140        if submask.contains(TransactionEffects::UNCHANGED_LOADED_RUNTIME_OBJECTS_FIELD) {
141            effects.unchanged_loaded_runtime_objects = source
142                .unchanged_loaded_runtime_objects
143                .unwrap_or_default()
144                .iter()
145                .map(Into::into)
146                .collect();
147        }
148
149        if let Some(object_types) = source.object_types {
150            if submask.contains(TransactionEffects::CHANGED_OBJECTS_FIELD.name) {
151                for changed_object in effects.changed_objects.iter_mut() {
152                    let Ok(object_id) = changed_object.object_id().parse::<Address>() else {
153                        continue;
154                    };
155
156                    if let Some(ty) = object_types.get(&object_id.into()) {
157                        changed_object.object_type = Some(match ty {
158                            sui_types::base_types::ObjectType::Package => "package".to_owned(),
159                            sui_types::base_types::ObjectType::Struct(struct_tag) => {
160                                struct_tag.to_canonical_string(true)
161                            }
162                        });
163                    }
164                }
165            }
166
167            if submask.contains(TransactionEffects::UNCHANGED_CONSENSUS_OBJECTS_FIELD.name) {
168                for unchanged_consensus_object in effects.unchanged_consensus_objects.iter_mut() {
169                    let Ok(object_id) = unchanged_consensus_object.object_id().parse::<Address>()
170                    else {
171                        continue;
172                    };
173
174                    if let Some(ty) = object_types.get(&object_id.into()) {
175                        unchanged_consensus_object.object_type = Some(match ty {
176                            sui_types::base_types::ObjectType::Package => "package".to_owned(),
177                            sui_types::base_types::ObjectType::Struct(struct_tag) => {
178                                struct_tag.to_canonical_string(true)
179                            }
180                        });
181                    }
182                }
183            }
184        }
185
186        // Try to render clever error info
187        render_clever_error(service, &mut effects);
188
189        message.effects = Some(effects);
190    }
191
192    if let Some(submask) = mask.subtree(ExecutedTransaction::EVENTS_FIELD.name) {
193        message.events = source.events.map(|events| {
194            let mut message = TransactionEvents::merge_from(events.clone(), &submask);
195
196            if let Some(event_mask) = submask.subtree(TransactionEvents::EVENTS_FIELD.name)
197                && event_mask.contains(Event::JSON_FIELD.name)
198            {
199                for (message, event) in message.events.iter_mut().zip(&events.0) {
200                    message.json =
201                        struct_tag_sdk_to_core(event.type_.clone())
202                            .ok()
203                            .and_then(|struct_tag| {
204                                crate::grpc::v2::render_json(service, &struct_tag, &event.contents)
205                                    .map(Box::new)
206                            });
207                }
208            }
209
210            message
211        });
212    }
213
214    if mask.contains(ExecutedTransaction::CHECKPOINT_FIELD.name) {
215        message.checkpoint = source.checkpoint;
216    }
217
218    if mask.contains(ExecutedTransaction::TIMESTAMP_FIELD.name) {
219        message.timestamp = source.timestamp_ms.map(timestamp_ms_to_proto);
220    }
221
222    if mask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD.name) {
223        message.balance_changes = source
224            .balance_changes
225            .map(|balance_changes| balance_changes.into_iter().map(Into::into).collect())
226            .unwrap_or_default();
227    }
228
229    message
230}

232pub(crate) fn render_clever_error(service: &RpcService, effects: &mut TransactionEffects) {
233    use sui_rpc::proto::sui::rpc::v2::CleverError;
234    use sui_rpc::proto::sui::rpc::v2::MoveAbort;
235    use sui_rpc::proto::sui::rpc::v2::clever_error;
236    use sui_rpc::proto::sui::rpc::v2::execution_error::ErrorDetails;
237
238    let Some(move_abort) = effects
239        .status
240        .as_mut()
241        .and_then(|status| status.error.as_mut())
242        .and_then(|error| match &mut error.error_details {
243            Some(ErrorDetails::Abort(move_abort)) => Some(move_abort),
244            _ => None,
245        })
246    else {
247        return;
248    };
249
250    fn render(service: &RpcService, move_abort: &MoveAbort) -> Option<CleverError> {
251        let location = move_abort.location.as_ref()?;
252        let abort_code = move_abort.abort_code();
253        let package_id = location.package().parse::<sui_sdk_types::Address>().ok()?;
254        let module = location.module();
255
256        let package = {
257            let object = service.reader.inner().get_object(&package_id.into())?;
258            sui_package_resolver::Package::read_from_object(&object).ok()?
259        };
260
261        let clever_error = package.resolve_clever_error(module, abort_code)?;
262
263        let mut clever_error_message = CleverError::default();
264
265        match clever_error.error_info {
266            sui_package_resolver::ErrorConstants::None => {}
267            sui_package_resolver::ErrorConstants::Rendered {
268                identifier,
269                constant,
270            } => {
271                clever_error_message.constant_name = Some(identifier);
272                clever_error_message.value = Some(clever_error::Value::Rendered(constant));
273            }
274            sui_package_resolver::ErrorConstants::Raw { identifier, bytes } => {
275                clever_error_message.constant_name = Some(identifier);
276                clever_error_message.value = Some(clever_error::Value::Raw(bytes.into()));
277            }
278        }
279
280        clever_error_message.error_code = clever_error.error_code.map(Into::into);
281        clever_error_message.line_number = Some(clever_error.source_line_number.into());
282
283        Some(clever_error_message)
284    }
285
286    move_abort.clever_error = render(service, move_abort);
