19pub(super) fn validate_checkpoint_bounds(
20    start_checkpoint: Option<u64>,
21    end_checkpoint: Option<u64>,
22) -> Result<(), RpcError> {
23    let start = start_checkpoint.unwrap_or(0);
24    if let Some(end) = end_checkpoint
25        && end < start
26    {
27        return Err(FieldViolation::new("end_checkpoint")
28            .with_description("end_checkpoint must be greater than or equal to start_checkpoint")
29            .with_reason(ErrorReason::FieldInvalid)
30            .into());
31    }
32    Ok(())
33}

35pub(super) fn checkpoint_hi_exclusive(service: &RpcService) -> Result<u64, RpcError> {
36    let indexes = service
37        .reader
38        .inner()
39        .indexes()
40        .ok_or_else(|| RpcError::new(tonic::Code::Unavailable, "rpc indexes are disabled"))?;
41    let checkpoint = indexes
42        .get_highest_indexed_checkpoint_seq_number()
43        .map_err(storage_error)?
44        .ok_or_else(|| RpcError::new(tonic::Code::Unavailable, "rpc index is empty"))?;
45    checkpoint
46        .checked_add(1)
47        .ok_or_else(|| RpcError::new(tonic::Code::Internal, "checkpoint bound overflow"))
48}

50pub(super) fn checkpoint_to_tx_boundary(
51    service: &RpcService,
52    checkpoint: CheckpointSequenceNumber,
53) -> Result<u64, RpcError> {
54    if checkpoint == 0 {
55        return Ok(0);
56    }
57    service
58        .reader
59        .inner()
60        .get_checkpoint_by_sequence_number(checkpoint - 1)
61        .map(|checkpoint| checkpoint.data().network_total_transactions)
62        .ok_or_else(|| {
63            RpcError::new(
64                tonic::Code::Internal,
65                format!(
66                    "missing checkpoint {} while resolving tx boundary",
67                    checkpoint - 1
68                ),
69            )
70        })
71}

73pub(super) fn checkpoint_to_tx_range(
74    service: &RpcService,
75    checkpoint_range: Range<u64>,
76) -> Result<Range<u64>, RpcError> {
77    let start = checkpoint_to_tx_boundary(service, checkpoint_range.start)?;
78    let end = checkpoint_to_tx_boundary(service, checkpoint_range.end)?;
79    Ok(start..end)
80}

82/// The serving floor in tx-seq space: the first tx_sequence_number whose
83/// checkpoint contents are still retained (the first tx of the lowest available
84/// checkpoint). This is the perpetual-store pruning floor, NOT the index's own
85/// floor — the index prunes *after* the perpetual store, so it can still hold
86/// `tx_seq_digest` rows for transactions whose contents are already gone. Serving
87/// must be bounded by content availability, matching `get_transaction`.
88///
89/// `checkpoint_to_tx_boundary` reads the lowest available checkpoint's predecessor
90/// from `certified_checkpoints`, which is retained across pruning, so this resolves
91/// even at the floor.
92fn lowest_available_tx_seq(service: &RpcService) -> Result<u64, RpcError> {
93    let lowest_checkpoint = service.reader.get_lowest_available_checkpoint()?;
94    checkpoint_to_tx_boundary(service, lowest_checkpoint)
95}

97/// Enforce the pruning `floor` on a resolved scan's low end (`start`, tx-seq space).
98/// Returns the effective start: unchanged when at/above the floor; clamped up to
99/// the floor when the low end was open-ended; or `OutOfRange` when an explicitly
100/// requested low end (a `start_checkpoint` or `after` cursor) is below the floor —
101/// that data was pruned and is permanently gone.
102fn apply_tx_seq_floor(start: u64, explicit_lower: bool, floor: u64) -> Result<u64, RpcError> {
103    if start >= floor {
104        Ok(start)
105    } else if explicit_lower {
106        Err(out_of_range(floor))
107    } else {
108        Ok(floor)
109    }
110}

112/// The serving floor resolved onto a scan: the effective first scannable
113/// transaction and its containing checkpoint, after clamping a resolved
114/// scan's low end to the pruning floor.
115pub(super) struct ServingFloor {
116    pub(super) tx_seq: u64,
117    pub(super) checkpoint: u64,
118}
119
120/// Clamp a resolved scan's low end (`start_tx`, tx-seq space) to the serving
121/// floor. Returns the resolved floor only when it actually moved the low end
122/// (an implicit-genesis low); an explicitly requested low below the floor
123/// (`start_checkpoint` or `after` cursor) errors `OutOfRange` instead, and an
124/// untouched low returns `None`. Callers MUST reconcile their interval and
125/// watermark metadata from the returned floor, so no watermark ever claims
126/// pruned, never-scanned checkpoints: for a nonempty retained intersection,
127/// ascending scans raise their entry claim to `checkpoint` and descending
128/// scans pin their terminal to it; a floor that consumes the interval leaves
129/// the request-derived terminal boundary in place and marks the interval
130/// empty.
131pub(super) fn clamp_to_serving_floor(
132    service: &RpcService,
133    start_tx: u64,
134    start_checkpoint: Option<u64>,
135    options: &crate::ledger_history::query_options::QueryOptions,
136) -> Result<Option<ServingFloor>, RpcError> {
137    let explicit_lower = start_checkpoint.is_some() || options.has_after_cursor();
138    let floor = lowest_available_tx_seq(service)?;
139    let clamped = apply_tx_seq_floor(start_tx, explicit_lower, floor)?;
140    if clamped == start_tx {
141        return Ok(None);
142    }
143    Ok(Some(ServingFloor {
144        tx_seq: clamped,
145        checkpoint: tx_checkpoint(service, clamped)?,
146    }))
147}

149fn out_of_range(floor: u64) -> RpcError {
150    RpcError::new(
151        tonic::Code::OutOfRange,
152        format!("requested data below earliest available; lowest available tx_seq is {floor}"),
153    )
154}

156/// Checkpoint-space analogue of [`clamp_to_serving_floor`] for scans that
157/// walk checkpoint sequence numbers directly (the unfiltered
158/// `list_checkpoints` path). Returns the floor checkpoint only when it
159/// actually moves the resolved low end (an implicit-genesis low); an
160/// explicitly requested low below the floor (`start_checkpoint` or `after`
161/// cursor) errors `OutOfRange` — that data was pruned and is permanently
162/// gone — and an untouched low returns `None`.
163pub(super) fn clamp_checkpoints_to_serving_floor(
164    service: &RpcService,
165    start: u64,
166    start_checkpoint: Option<u64>,
167    options: &crate::ledger_history::query_options::QueryOptions,
168) -> Result<Option<u64>, RpcError> {
169    let explicit_lower = start_checkpoint.is_some() || options.has_after_cursor();
170    let floor = service.reader.get_lowest_available_checkpoint()?;
171    if start >= floor {
172        return Ok(None);
173    }
174    if explicit_lower {
175        return Err(RpcError::new(
176            tonic::Code::OutOfRange,
177            format!(
178                "requested data below earliest available; lowest available checkpoint is {floor}"
179            ),
180        ));
181    }
182    Ok(Some(floor))
183}

185pub(super) fn get_tx_seq_digest_multi(
186    service: &RpcService,
187    tx_seqs: &[u64],
188) -> Result<Vec<LedgerTxSeqDigest>, RpcError> {
189    if tx_seqs.is_empty() {
190        return Ok(Vec::new());
191    }
192
193    let indexes = service
194        .reader
195        .inner()
196        .indexes()
197        .ok_or_else(|| RpcError::new(tonic::Code::Unavailable, "rpc indexes are disabled"))?;
198    indexes
199        .ledger_tx_seq_digest_multi_get(tx_seqs)
200        .map_err(storage_error)?
201        .into_iter()
202        .zip_debug_eq(tx_seqs.iter().copied())
203        .map(|(row, tx_seq)| row.ok_or_else(|| missing_row_error(service, tx_seq)))
204        .collect()
205}

207pub(super) fn get_tx_seq_digest_rows(
208    service: &RpcService,
209    range: Range<u64>,
210    descending: bool,
211    row_limit: usize,
212) -> Result<Vec<LedgerTxSeqDigest>, RpcError> {
213    if range.is_empty() || row_limit == 0 {
214        return Ok(Vec::new());
215    }
216
217    let indexes = service
218        .reader
219        .inner()
220        .indexes()
221        .ok_or_else(|| RpcError::new(tonic::Code::Unavailable, "rpc indexes are disabled"))?;
222    let mut iter = indexes
223        .ledger_tx_seq_digest_iter(range.start, range.end, descending)
224        .map_err(storage_error)?;
225    let mut rows = Vec::with_capacity(row_limit);
226    let mut expected = if descending {
227        range.end - 1
228    } else {
229        range.start
230    };
231
232    while rows.len() < row_limit {
233        match iter.next() {
234            Some(result) => {
235                let row = result.map_err(storage_error)?;
236                if row.tx_sequence_number != expected {
237                    return Err(missing_row_error(service, expected));
238                }
239                rows.push(row);
240
241                if descending {
242                    if expected == range.start {
243                        break;
244                    }
245                    expected -= 1;
246                } else {
247                    expected += 1;
248                    if expected >= range.end {
249                        break;
250                    }
251                }
252            }
253            None => {
254                return Err(missing_row_error(service, expected));
255            }
256        }
257    }
258
259    Ok(rows)
260}

262/// Resolve a transaction sequence number to its containing checkpoint.
263pub(super) fn tx_checkpoint(service: &RpcService, tx_seq: u64) -> Result<u64, RpcError> {
264    let checkpoint = get_tx_seq_digest_multi(service, &[tx_seq])?
265        .into_iter()
266        .next()
267        .expect("multi-get of one tx_seq returns one row")
268        .checkpoint_number;
269    Ok(checkpoint)
270}

272/// Resolve a u64 scan frontier to the checkpoint that bounds the covered range.
273/// The watermark proves coverage up to `frontier - 1` ascending or `frontier`
274/// descending. Returns `None` only when the ascending frontier is at genesis,
275/// where nothing is yet covered.
276pub(super) fn sequence_frontier_checkpoint(
277    service: &RpcService,
278    frontier: u64,
279    ascending: bool,
280) -> Result<Option<u64>, RpcError> {
281    let lookup_position = if ascending {
282        match frontier.checked_sub(1) {
283            Some(position) => position,
284            None => return Ok(None),
285        }
286    } else {
287        frontier
288    };
289    tx_checkpoint(service, lookup_position).map(Some)
290}

292/// Classify a missing `tx_seq_digest` row encountered during a scan. The serving
293/// floor can advance after a range is resolved (a concurrent prune), so a row
294/// missing below the *current* floor was pruned mid-scan — permanently gone,
295/// surfaced as `OutOfRange`. A row missing at/above the floor is a genuine
296/// inconsistency (`Internal`). A failed floor lookup conservatively reports the
297/// latter.
298fn missing_row_error(service: &RpcService, tx_seq: u64) -> RpcError {
299    match lowest_available_tx_seq(service) {
300        Ok(floor) if tx_seq < floor => out_of_range(floor),
301        _ => missing_tx_seq_digest(tx_seq),
302    }
303}
304
305fn missing_tx_seq_digest(tx_seq: u64) -> RpcError {
306    RpcError::new(
307        tonic::Code::Internal,
308        format!("missing tx_seq_digest row for tx_seq {tx_seq}"),
309    )
310}

312pub(super) fn remaining_range_after(
313    range: Range<u64>,
314    position: u64,
315    ascending: bool,
316) -> Option<Range<u64>> {
317    let remaining = if ascending {
318        position.saturating_add(1)..range.end
319    } else {
320        range.start..position
321    };
322    (!remaining.is_empty()).then_some(remaining)
323}

329    #[test]
330    fn apply_tx_seq_floor_passes_through_at_or_above_floor() {
331        assert_eq!(apply_tx_seq_floor(10, false, 10).unwrap(), 10);
332        assert_eq!(apply_tx_seq_floor(15, true, 10).unwrap(), 15);
333    }
334
335    #[test]
336    fn apply_tx_seq_floor_clamps_open_ended_low_end() {
337        // No explicit low bound below the floor → clamp up to the floor.
338        assert_eq!(apply_tx_seq_floor(0, false, 10).unwrap(), 10);
339        assert_eq!(apply_tx_seq_floor(7, false, 10).unwrap(), 10);
340    }
341
342    #[test]
343    fn apply_tx_seq_floor_errors_on_explicit_below_floor() {
344        for start in [0u64, 9] {
345            let status = tonic::Status::from(apply_tx_seq_floor(start, true, 10).unwrap_err());
346            assert_eq!(status.code(), tonic::Code::OutOfRange);
347        }
348    }
349
350    #[test]
351    fn apply_tx_seq_floor_zero_floor_never_clamps_or_errors() {
352        // Empty/unpruned index (floor 0): every start is at/above the floor.
353        assert_eq!(apply_tx_seq_floor(0, true, 0).unwrap(), 0);
354        assert_eq!(apply_tx_seq_floor(0, false, 0).unwrap(), 0);
355        assert_eq!(apply_tx_seq_floor(123, true, 0).unwrap(), 123);
356    }
