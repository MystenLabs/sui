4use crate::{
5    ErrorReason, Result, RpcService,
6    grpc::v2::move_package_service::{
7        conversions::{convert_error, convert_module},
8        load_package,
9    },
10};
11use sui_rpc::proto::google::rpc::bad_request::FieldViolation;
12use sui_rpc::proto::sui::rpc::v2::{GetPackageRequest, GetPackageResponse, Package};
13
14#[tracing::instrument(skip(service))]
15pub fn get_package(service: &RpcService, request: GetPackageRequest) -> Result<GetPackageResponse> {
16    let package_id_str = request.package_id.as_ref().ok_or_else(|| {
17        FieldViolation::new("package_id")
18            .with_description("missing package_id")
19            .with_reason(ErrorReason::FieldMissing)
20    })?;
21
22    let package = load_package(service, package_id_str)?;
23    let package_id = package.id();
24
25    let resolved_package =
26        sui_package_resolver::Package::read_from_package(&package).map_err(convert_error)?;
27
28    let modules: Vec<_> = resolved_package
29        .modules()
30        .iter()
31        .map(|(module_name, resolver_module)| {
32            convert_module(module_name, resolver_module, &package_id)
33        })
34        .collect::<Result<Vec<_>>>()?;
35
36    let mut message = Package::default();
37    message.storage_id = Some(package_id.to_canonical_string(true));
38    message.original_id = Some(package.original_package_id().to_canonical_string(true));
39    message.version = Some(package.version().value());
40    message.modules = modules;
41
42    Ok(GetPackageResponse::new(message))
43}
