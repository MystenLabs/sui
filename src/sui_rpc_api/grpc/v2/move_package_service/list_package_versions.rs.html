4use super::load_package;
5use crate::{ErrorReason, Result, RpcError, RpcService};
6use bytes::Bytes;
7use sui_rpc::proto::google::rpc::bad_request::FieldViolation;
8use sui_rpc::proto::sui::rpc::v2::{
9    ListPackageVersionsRequest, ListPackageVersionsResponse, PackageVersion,
10};
11use sui_types::base_types::ObjectID;
12use tap::Pipe;
13
14#[derive(serde::Serialize, serde::Deserialize)]
15struct PageToken {
16    original_package_id: ObjectID,
17    version: u64,
18}
19
20#[tracing::instrument(skip(service))]
21pub fn list_package_versions(
22    service: &RpcService,
23    request: ListPackageVersionsRequest,
24) -> Result<ListPackageVersionsResponse> {
25    let package_id_str = request.package_id.as_ref().ok_or_else(|| {
26        FieldViolation::new("package_id")
27            .with_description("missing package_id")
28            .with_reason(ErrorReason::FieldMissing)
29    })?;
30
31    let package = load_package(service, package_id_str)?;
32    let original_package_id = package.original_package_id();
33
34    let page_size = request
35        .page_size
36        .map(|s| (s as usize).clamp(1, 10000))
37        .unwrap_or(1000);
38
39    let page_token = request
40        .page_token
41        .map(|token| decode_page_token(&token))
42        .transpose()?;
43
44    if let Some(token) = &page_token
45        && token.original_package_id != original_package_id
46    {
47        return Err(FieldViolation::new("page_token")
48            .with_description("page token package ID does not match request package ID")
49            .with_reason(ErrorReason::FieldInvalid)
50            .into());
51    }
52
53    let indexes = service
54        .reader
55        .inner()
56        .indexes()
57        .ok_or_else(RpcError::not_found)?;
58
59    let mut versions = vec![];
60    let iter = indexes
61        .package_versions_iter(original_package_id, page_token.map(|t| t.version))
62        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
63
64    for result in iter.take(page_size + 1) {
65        let (version, storage_id) =
66            result.map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
67
68        versions.push(PackageVersion::new(&storage_id.into(), version));
69    }
70
71    let next_page_token = if versions.len() > page_size {
72        // SAFETY: We've already verified that versions is greater than page_size, which is
73        // guaranteed to be >= 1.
74        versions.pop().unwrap().pipe(|v| {
75            v.version.map(|version| {
76                encode_page_token(PageToken {
77                    original_package_id,
78                    version,
79                })
80            })
81        })
82    } else {
83        None
84    };
85
86    Ok(ListPackageVersionsResponse::new(versions, next_page_token))
87}
88
89fn decode_page_token(page_token: &[u8]) -> Result<PageToken> {
90    bcs::from_bytes(page_token).map_err(|e| {
91        FieldViolation::new("page_token")
92            .with_description(format!("invalid page token encoding: {e}"))
93            .with_reason(ErrorReason::FieldInvalid)
94            .into()
95    })
96}
97
98fn encode_page_token(page_token: PageToken) -> Bytes {
99    bcs::to_bytes(&page_token).unwrap().into()
