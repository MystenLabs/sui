6use sui_rpc::proto::sui::rpc::v2::{Balance, GetBalanceRequest, GetBalanceResponse};
7use sui_sdk_types::Address;
8use sui_sdk_types::StructTag;
9use sui_types::base_types::SuiAddress;
10use sui_types::storage::BalanceInfo;
11use sui_types::sui_sdk_types_conversions::struct_tag_sdk_to_core;

13#[tracing::instrument(skip(service))]
14pub fn get_balance(service: &RpcService, request: GetBalanceRequest) -> Result<GetBalanceResponse> {
15    let indexes = service
16        .reader
17        .inner()
18        .indexes()
19        .ok_or_else(RpcError::not_found)?;
20
21    let owner = request
22        .owner
23        .as_ref()
24        .ok_or_else(|| {
25            FieldViolation::new("owner")
26                .with_description("missing owner")
27                .with_reason(ErrorReason::FieldMissing)
28        })?
29        .parse::<Address>()
30        .map_err(|e| {
31            FieldViolation::new("owner")
32                .with_description(format!("invalid owner: {e}"))
33                .with_reason(ErrorReason::FieldInvalid)
34        })?;
35    let owner = SuiAddress::from(owner);
36
37    let coin_type = request
38        .coin_type
39        .as_ref()
40        .ok_or_else(|| {
41            FieldViolation::new("coin_type")
42                .with_description("missing coin_type")
43                .with_reason(ErrorReason::FieldMissing)
44        })?
45        .parse::<StructTag>()
46        .map_err(|e| {
47            FieldViolation::new("coin_type")
48                .with_description(format!("invalid coin_type: {e}"))
49                .with_reason(ErrorReason::FieldInvalid)
50        })?;
51
52    let core_coin_type = struct_tag_sdk_to_core(coin_type.clone())?;
53
54    let balance_info = indexes
55        .get_balance(&owner, &core_coin_type)?
56        .unwrap_or_default(); // Use default (zero) if no balance found
57
58    let balance = render_balance(service, owner, core_coin_type, balance_info);
59
60    Ok(GetBalanceResponse::default().with_balance(balance))
61}
62
63pub(super) fn render_balance(
64    service: &RpcService,
65    owner: SuiAddress,
66    coin_type: move_core_types::language_storage::StructTag,
67    balance_info: BalanceInfo,
68) -> Balance {
69    let mut balance = Balance::default()
70        .with_coin_type(coin_type.to_canonical_string(true))
71        .with_balance(balance_info.balance);
72
73    if balance_info.balance == 0 {
74        return balance;
75    }
76
77    // When looking up an Address's balance for a particular coin type, there is a possibility that
78    // the value we read is "newer" (further ahead in time) than the summed balance we've read from
79    // the indexes.
80    //
81    // This inconsistency should in practice be hard to see (as its a race) but it can lead to
82    // slightly inconsistent responses:
83    //
84    // - If the Address balance is greater than it was at the checkpoint corrisponding to our index
85    // read, then we'll clamp the value returned to what was in the indexes and its possible that
86    // we under report the balance stored in `Coin<T>`s by saying that the address has no coin
87    // balance.
88    //
89    // - If the Address balance is less than it was at the checkpoint corrisponding to our index
90    // read, then we can possibly over-inflate the balance stored in `Coin<T>`s.
91    if let Some(address_balance) = service.reader.lookup_address_balance(owner, coin_type) {
92        balance.set_address_balance(address_balance);
93
94        match address_balance.cmp(&balance_info.balance) {
95            std::cmp::Ordering::Less => {
96                // If the AddressBalance is less than the total balance we read from the indexes,
97                // then the difference is attributed to coins
98                balance.set_coin_balance(balance_info.balance.saturating_sub(address_balance));
99            }
100            std::cmp::Ordering::Equal => {}
101            std::cmp::Ordering::Greater => {
102                // There is a potential race where the Address balance we read is newer than what
103                // we read from the indexes, and if its higher then lets just cap it based on what
104                // we have in the indexes.
105                balance.set_address_balance(balance_info.balance);
106            }
107        }
108    } else {
109        // If there is no AddressBalance for this coin type, then all the balance is attributed to
110        // coins
111        balance.set_coin_balance(balance_info.balance);
112    }
113
114    balance
115}
