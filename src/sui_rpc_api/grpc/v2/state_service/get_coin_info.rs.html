20const SUI_COIN_TREASURY: CoinTreasury = {
21    let mut treasury = CoinTreasury::const_default();
22    treasury.total_supply = Some(sui_types::gas_coin::TOTAL_SUPPLY_MIST);
23    treasury.supply_state = Some(RpcSupplyState::Fixed as i32);
24    treasury
25};

27#[derive(Debug)]
28pub struct CoinNotFoundError(StructTag);
29
30impl std::fmt::Display for CoinNotFoundError {
31    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
32        write!(f, "Coin type {} not found", self.0)
33    }
34}
35
36impl std::error::Error for CoinNotFoundError {}
37
38impl From<CoinNotFoundError> for crate::RpcError {
39    fn from(value: CoinNotFoundError) -> Self {
40        Self::new(tonic::Code::NotFound, value.to_string())
41    }
42}

44#[tracing::instrument(skip(service))]
45pub fn get_coin_info(
46    service: &RpcService,
47    request: GetCoinInfoRequest,
48) -> Result<GetCoinInfoResponse> {
49    let indexes = service
50        .reader
51        .inner()
52        .indexes()
53        .ok_or_else(RpcError::not_found)?;
54
55    let coin_type = request.coin_type().parse::<StructTag>().map_err(|e| {
56        RpcError::new(
57            tonic::Code::InvalidArgument,
58            format!("invalid coin_type: {e}"),
59        )
60    })?;
61
62    let core_coin_type = struct_tag_sdk_to_core(coin_type.clone())?;
63
64    match get_coin_info_from_registry(service, &coin_type, &core_coin_type) {
65        Ok(Some(response)) => Ok(response),
66        Ok(None) => get_coin_info_from_index(service, indexes, &coin_type, &core_coin_type)?
67            .ok_or_else(|| CoinNotFoundError(coin_type).into()),
68        Err(e) => Err(e),
69    }
70}

72fn get_coin_info_from_registry(
73    service: &RpcService,
74    coin_type: &StructTag,
75    core_coin_type: &move_core_types::language_storage::StructTag,
76) -> Result<Option<GetCoinInfoResponse>> {
77    let currency_id = Currency::derive_object_id(core_coin_type.clone().into()).map_err(|e| {
78        RpcError::new(
79            tonic::Code::Internal,
80            format!(
81                "Failed to derive Currency ID for coin type {}: {}",
82                core_coin_type, e
83            ),
84        )
85    })?;
86
87    let object_store = service.reader.inner();
88    let currency_obj = match object_store.get_object(&currency_id) {
89        Some(obj) => obj,
90        None => {
91            return Ok(None); // Coin not registered in CoinRegistry
92        }
93    };
94
95    let move_obj = currency_obj.data.try_as_move().ok_or_else(|| {
96        RpcError::new(
97            tonic::Code::Internal,
98            format!(
99                "Currency for coin type {} is not a Move object",
100                core_coin_type
101            ),
102        )
103    })?;
104
105    let currency = bcs::from_bytes::<Currency>(move_obj.contents()).map_err(|e| {
106        RpcError::new(
107            tonic::Code::Internal,
108            format!(
109                "Failed to deserialize Currency for coin type {}: {}",
110                core_coin_type, e
111            ),
112        )
113    })?;
114    let metadata = Some((&currency).into());
115
116    let treasury = if sui_types::gas_coin::GAS::is_gas(core_coin_type) {
117        Some(SUI_COIN_TREASURY)
118    } else {
119        match &currency.supply {
120            Some(SupplyState::Fixed(supply)) => {
121                let mut treasury = CoinTreasury::default();
122                treasury.id = currency
123                    .treasury_cap_id
124                    .map(|id| Address::from(id).to_string());
125                treasury.total_supply = Some(*supply);
126                treasury.supply_state = Some(RpcSupplyState::Fixed.into());
127                Some(treasury)
128            }
129            Some(SupplyState::BurnOnly(supply)) => {
130                let mut treasury = CoinTreasury::default();
131                treasury.id = currency
132                    .treasury_cap_id
133                    .map(|id| Address::from(id).to_string());
134                treasury.total_supply = Some(*supply);
135                treasury.supply_state = Some(RpcSupplyState::BurnOnly.into());
136                Some(treasury)
137            }
138            Some(SupplyState::Unknown) | None => {
139                // For unknown or missing supply state, look up the treasury cap object
140                currency
141                    .treasury_cap_id
142                    .or_else(|| {
143                        service
144                            .reader
145                            .inner()
146                            .indexes()
147                            .and_then(|indexes| indexes.get_coin_info(core_coin_type).ok()?)
148                            .and_then(|coin_info| coin_info.treasury_object_id)
149                    })
150                    .and_then(|id| get_treasury_cap_info(service, id))
151            }
152        }
153    };
154
155    // If registry has a definitive state, use it
156    let regulated_metadata = if !matches!(&currency.regulated, CurrencyRegulatedState::Unknown) {
157        Some((&currency.regulated).into())
158    } else {
159        // Registry has Unknown state, need to check for legacy RegulatedCoinMetadata
160        let indexes = match service.reader.inner().indexes() {
161            Some(indexes) => indexes,
162            None => {
163                // No indexes available, keep as Unknown
164                let mut response = GetCoinInfoResponse::default();
165                response.coin_type = Some(coin_type.to_string());
166                response.metadata = metadata;
167                response.treasury = treasury;
168                response.regulated_metadata = Some(CurrencyRegulatedState::Unknown.into());
169                return Ok(Some(response));
170            }
171        };
172
173        let coin_info = indexes.get_coin_info(core_coin_type).map_err(|e| {
174            RpcError::new(
175                tonic::Code::Internal,
176                format!("Failed to get coin info for {}: {}", core_coin_type, e),
177            )
178        })?;
179
180        let regulated_id = coin_info.and_then(|info| info.regulated_coin_metadata_object_id);
181
182        match regulated_id {
183            None => {
184                // No RegulatedCoinMetadata exists, coin is unregulated
185                Some(CurrencyRegulatedState::Unregulated.into())
186            }
187            Some(id) => object_store
188                .get_object(&id)
189                .map(RegulatedCoinMetadata::try_from)
190                .transpose()
191                .map_err(|_| {
192                    RpcError::new(
193                        tonic::Code::Internal,
194                        format!(
195                            "Unable to read object {} for coin type {} as RegulatedCoinMetadata",
196                            id, core_coin_type
197                        ),
198                    )
199                })?
200                .map(Into::into)
201                .or(Some(CurrencyRegulatedState::Unregulated.into())),
202        }
203    };
204
205    let mut response = GetCoinInfoResponse::default();
206    response.coin_type = Some(coin_type.to_string());
207    response.metadata = metadata;
208    response.treasury = treasury;
209    response.regulated_metadata = regulated_metadata;
210    Ok(Some(response))
211}

213fn get_coin_info_from_index(
214    service: &RpcService,
215    indexes: &dyn sui_types::storage::RpcIndexes,
216    coin_type: &StructTag,
217    core_coin_type: &move_core_types::language_storage::StructTag,
218) -> Result<Option<GetCoinInfoResponse>> {
219    let coin_info = match indexes.get_coin_info(core_coin_type)? {
220        Some(info) => info,
221        None => return Ok(None),
222    };
223
224    let metadata = if let Some(coin_metadata_object_id) = coin_info.coin_metadata_object_id {
225        service
226            .reader
227            .inner()
228            .get_object(&coin_metadata_object_id)
229            .map(sui_types::coin::CoinMetadata::try_from)
230            .transpose()
231            .map_err(|_| {
232                RpcError::new(
233                    tonic::Code::Internal,
234                    format!(
235                        "Unable to read object {} for coin type {} as CoinMetadata",
236                        coin_metadata_object_id, core_coin_type
237                    ),
238                )
239            })?
240            .map(Into::into)
241    } else {
242        None
243    };
244
245    let treasury = if let Some(treasury_object_id) = coin_info.treasury_object_id {
246        get_treasury_cap_info(service, treasury_object_id)
247    } else if sui_types::gas_coin::GAS::is_gas(core_coin_type) {
248        Some(SUI_COIN_TREASURY)
249    } else {
250        None
251    };
252
253    let regulated_metadata = if let Some(regulated_coin_metadata_object_id) =
254        coin_info.regulated_coin_metadata_object_id
255    {
256        service
257            .reader
258            .inner()
259            .get_object(&regulated_coin_metadata_object_id)
260            .map(RegulatedCoinMetadata::try_from)
261            .transpose()
262            .map_err(|_| {
263                RpcError::new(
264                    tonic::Code::Internal,
265                    format!(
266                        "Unable to read object {} for coin type {} as RegulatedCoinMetadata",
267                        regulated_coin_metadata_object_id, core_coin_type
268                    ),
269                )
270            })?
271            .map(Into::into)
272    } else {
273        Some(CurrencyRegulatedState::Unregulated.into())
274    };
275
276    {
277        let mut response = GetCoinInfoResponse::default();
278        response.coin_type = Some(coin_type.to_string());
279        response.metadata = metadata;
280        response.treasury = treasury;
281        response.regulated_metadata = regulated_metadata;
282        Ok(Some(response))
283    }
284}

286fn get_treasury_cap_info(
287    service: &RpcService,
288    treasury_object_id: SuiObjectID,
289) -> Option<CoinTreasury> {
290    let obj = service.reader.inner().get_object(&treasury_object_id)?;
291
292    // Treasury caps that are immutable, owned by 0x0, or consensus-owned by 0x0 indicate fixed supply
293    let supply_state = match &obj.owner {
294        Immutable => RpcSupplyState::Fixed,
295        AddressOwner(addr) if *addr == SuiAddress::ZERO => RpcSupplyState::Fixed,
296        ConsensusAddressOwner { owner, .. } if *owner == SuiAddress::ZERO => RpcSupplyState::Fixed,
297        _ => RpcSupplyState::Unknown,
298    };
299
300    sui_types::coin::TreasuryCap::try_from(obj)
301        .ok()
302        .map(|treasury| {
303            let mut coin_treasury: CoinTreasury = treasury.into();
304            coin_treasury.supply_state = Some(supply_state.into());
305            coin_treasury
306        })
307}
