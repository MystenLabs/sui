17#[tracing::instrument(skip(service))]
18pub fn list_balances(
19    service: &RpcService,
20    request: ListBalancesRequest,
21) -> Result<ListBalancesResponse> {
22    let indexes = service
23        .reader
24        .inner()
25        .indexes()
26        .ok_or_else(RpcError::not_found)?;
27
28    let owner: Address = request
29        .owner
30        .as_ref()
31        .ok_or_else(|| {
32            FieldViolation::new("owner")
33                .with_description("missing owner")
34                .with_reason(ErrorReason::FieldMissing)
35        })?
36        .parse()
37        .map_err(|e| {
38            FieldViolation::new("owner")
39                .with_description(format!("invalid owner: {e}"))
40                .with_reason(ErrorReason::FieldInvalid)
41        })?;
42
43    let page_size = request
44        .page_size
45        .map(|s| (s as usize).clamp(1, 1000))
46        .unwrap_or(50);
47    let page_token = request
48        .page_token
49        .map(|token| decode_page_token(&token))
50        .transpose()?;
51
52    if let Some(token) = &page_token
53        && token.owner != owner
54    {
55        return Err(FieldViolation::new("page_token")
56            .with_description("page token owner does not match request owner")
57            .with_reason(ErrorReason::FieldInvalid)
58            .into());
59    }
60
61    let mut balances = indexes
62        .balance_iter(
63            &owner.into(),
64            page_token.map(|t| (owner.into(), t.coin_type)),
65        )?
66        .take(page_size + 1)
67        .map(|result| result.map_err(|err| RpcError::new(tonic::Code::Internal, err.to_string())))
68        .collect::<Result<Vec<_>, _>>()?;
69
70    let next_page_token = if balances.len() > page_size {
71        // SAFETY: We've already verified that balances is greater than limit, which is
72        // guaranteed to be >= 1.
73        balances
74            .pop()
75            .unwrap()
76            .pipe(|(coin_type, _info)| encode_page_token(PageToken { owner, coin_type }))
77            .pipe(Some)
78    } else {
79        None
80    };
81
82    let mut response = ListBalancesResponse::default();
83    response.balances = balances.into_iter().map(balance_info_to_proto).collect();
84    response.next_page_token = next_page_token;
85    Ok(response)
86}
87
88fn decode_page_token(page_token: &[u8]) -> Result<PageToken> {
89    bcs::from_bytes(page_token).map_err(|e| {
90        FieldViolation::new("page_token")
91            .with_description(format!("invalid page token encoding: {e}"))
92            .with_reason(ErrorReason::FieldInvalid)
93            .into()
94    })
95}
96
97fn encode_page_token(page_token: PageToken) -> Bytes {
98    bcs::to_bytes(&page_token).unwrap().into()
99}
100
101fn balance_info_to_proto(
102    (coin_type, info): (move_core_types::language_storage::StructTag, BalanceInfo),
103) -> Balance {
104    let mut balance = Balance::default();
105    balance.coin_type = Some(coin_type.to_canonical_string(true));
106    balance.balance = Some(info.balance);
107    balance
108}
109
110#[derive(serde::Serialize, serde::Deserialize)]
111struct PageToken {
112    owner: Address,
113    coin_type: move_core_types::language_storage::StructTag,
114}
