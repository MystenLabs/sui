4use crate::ErrorReason;
5use crate::Result;
6use crate::RpcError;
7use crate::RpcService;
8use bytes::Bytes;
9use sui_rpc::proto::google::rpc::bad_request::FieldViolation;
10use sui_rpc::proto::sui::rpc::v2::ListBalancesRequest;
11use sui_rpc::proto::sui::rpc::v2::ListBalancesResponse;
12use sui_sdk_types::Address;
13use tap::Pipe;
14
15#[tracing::instrument(skip(service))]
16pub fn list_balances(
17    service: &RpcService,
18    request: ListBalancesRequest,
19) -> Result<ListBalancesResponse> {
20    let indexes = service
21        .reader
22        .inner()
23        .indexes()
24        .ok_or_else(RpcError::not_found)?;
25
26    let owner: Address = request
27        .owner
28        .as_ref()
29        .ok_or_else(|| {
30            FieldViolation::new("owner")
31                .with_description("missing owner")
32                .with_reason(ErrorReason::FieldMissing)
33        })?
34        .parse()
35        .map_err(|e| {
36            FieldViolation::new("owner")
37                .with_description(format!("invalid owner: {e}"))
38                .with_reason(ErrorReason::FieldInvalid)
39        })?;
40
41    let page_size = request
42        .page_size
43        .map(|s| (s as usize).clamp(1, 1000))
44        .unwrap_or(50);
45    let page_token = request
46        .page_token
47        .map(|token| decode_page_token(&token))
48        .transpose()?;
49
50    if let Some(token) = &page_token
51        && token.owner != owner
52    {
53        return Err(FieldViolation::new("page_token")
54            .with_description("page token owner does not match request owner")
55            .with_reason(ErrorReason::FieldInvalid)
56            .into());
57    }
58
59    let mut balances = indexes
60        .balance_iter(
61            &owner.into(),
62            page_token.map(|t| (owner.into(), t.coin_type)),
63        )?
64        .take(page_size + 1)
65        .map(|result| result.map_err(|err| RpcError::new(tonic::Code::Internal, err.to_string())))
66        .collect::<Result<Vec<_>, _>>()?;
67
68    let next_page_token = if balances.len() > page_size {
69        // SAFETY: We've already verified that balances is greater than limit, which is
70        // guaranteed to be >= 1.
71        balances
72            .pop()
73            .unwrap()
74            .pipe(|(coin_type, _info)| encode_page_token(PageToken { owner, coin_type }))
75            .pipe(Some)
76    } else {
77        None
78    };
79
80    let owner = sui_types::base_types::SuiAddress::from(owner);
81    let mut response = ListBalancesResponse::default();
82    response.balances = balances
83        .into_iter()
84        .map(|(coin_type, balance_info)| {
85            super::get_balance::render_balance(service, owner, coin_type, balance_info)
86        })
87        .collect();
88    response.next_page_token = next_page_token;
89    Ok(response)
90}
91
92fn decode_page_token(page_token: &[u8]) -> Result<PageToken> {
93    bcs::from_bytes(page_token).map_err(|e| {
94        FieldViolation::new("page_token")
95            .with_description(format!("invalid page token encoding: {e}"))
96            .with_reason(ErrorReason::FieldInvalid)
97            .into()
98    })
99}
100
101fn encode_page_token(page_token: PageToken) -> Bytes {
102    bcs::to_bytes(&page_token).unwrap().into()
103}
104
105#[derive(serde::Serialize, serde::Deserialize)]
106struct PageToken {
107    owner: Address,
108    coin_type: move_core_types::language_storage::StructTag,
109}
