20const MAX_PAGE_SIZE: usize = 1000;
21const DEFAULT_PAGE_SIZE: usize = 50;
22const MAX_PAGE_SIZE_BYTES: usize = 512 * 1024; // 512KiB
23const READ_MASK_DEFAULT: &str = "object_id,version,object_type";
24
25#[tracing::instrument(skip(service))]
26pub fn list_owned_objects(
27    service: &RpcService,
28    request: ListOwnedObjectsRequest,
29) -> Result<ListOwnedObjectsResponse> {
30    let indexes = service
31        .reader
32        .inner()
33        .indexes()
34        .ok_or_else(RpcError::not_found)?;
35
36    let owner: Address = request
37        .owner
38        .as_ref()
39        .ok_or_else(|| FieldViolation::new("owner").with_reason(ErrorReason::FieldMissing))?
40        .parse()
41        .map_err(|e| {
42            FieldViolation::new("owner")
43                .with_description(format!("invalid owner: {e}"))
44                .with_reason(ErrorReason::FieldInvalid)
45        })?;
46    let object_type = request
47        .object_type
48        .map(|s| s.parse())
49        .transpose()
50        .map_err(|e| {
51            FieldViolation::new("object_type")
52                .with_description(format!("invalid object_type: {e}"))
53                .with_reason(ErrorReason::FieldInvalid)
54        })?;
55
56    let page_size = request
57        .page_size
58        .map(|s| (s as usize).clamp(1, MAX_PAGE_SIZE))
59        .unwrap_or(DEFAULT_PAGE_SIZE);
60    let page_token = request
61        .page_token
62        .map(|token| decode_page_token(&token))
63        .transpose()?;
64    if let Some(token) = &page_token
65        && (token.owner != owner || token.object_type != object_type)
66    {
67        return Err(FieldViolation::new("page_token")
68            .with_description("invalid page_token")
69            .with_reason(ErrorReason::FieldInvalid)
70            .into());
71    }
72    let read_mask = {
73        let read_mask = request
74            .read_mask
75            .unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
76        read_mask.validate::<Object>().map_err(|path| {
77            FieldViolation::new("read_mask")
78                .with_description(format!("invalid read_mask path: {path}"))
79                .with_reason(ErrorReason::FieldInvalid)
80        })?;
81        FieldMaskTree::from(read_mask)
82    };
83
84    let should_load_object = should_load_object(&read_mask);
85    let mut iter = indexes.owned_objects_iter(
86        owner.into(),
87        object_type.clone(),
88        page_token.map(|t| t.inner),
89    )?;
90    let mut objects = Vec::with_capacity(page_size);
91    let mut size_bytes = 0;
92    while let Some(object_info) = iter
93        .next()
94        .transpose()
95        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?
96    {
97        let object = if should_load_object {
98            let Some(object) = service
99                .reader
100                .inner()
101                .get_object_by_key(&object_info.object_id, object_info.version)
102            else {
103                tracing::debug!(
104                    "unable to find object {}:{} while iterating through owned objects",
105                    object_info.object_id,
106                    object_info.version
107                );
108                continue;
109            };
110
111            service.render_object_to_proto(&object, &read_mask)
112        } else {
113            owned_object_to_proto(object_info, &read_mask)
114        };
115
116        size_bytes += object.encoded_len();
117        objects.push(object);
118
119        if objects.len() >= page_size || size_bytes >= MAX_PAGE_SIZE_BYTES {
120            break;
121        }
122    }
123
124    let next_page_token = iter
125        .next()
126        .transpose()
127        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?
128        .map(|cursor| {
129            encode_page_token(PageToken {
130                owner,
131                object_type,
132                inner: cursor,
133            })
134        });
135
136    let mut message = ListOwnedObjectsResponse::default();
137    message.objects = objects;
138    message.next_page_token = next_page_token;
139    Ok(message)
140}
141
142fn decode_page_token(page_token: &[u8]) -> Result<PageToken> {
143    bcs::from_bytes(page_token).map_err(|_| {
144        FieldViolation::new("page_token")
145            .with_description("invalid page_token")
146            .with_reason(ErrorReason::FieldInvalid)
147            .into()
148    })
149}
150
151fn encode_page_token(page_token: PageToken) -> Bytes {
152    bcs::to_bytes(&page_token).unwrap().into()
153}
154
155fn owned_object_to_proto(info: OwnedObjectInfo, mask: &FieldMaskTree) -> Object {
156    let mut message = Object::default();
157
158    if mask.contains(Object::OBJECT_ID_FIELD) {
159        message.object_id = Some(info.object_id.to_canonical_string(true));
160    }
161    if mask.contains(Object::VERSION_FIELD) {
162        message.version = Some(info.version.value());
163    }
164    if mask.contains(Object::OBJECT_TYPE_FIELD) {
165        message.object_type = Some(info.object_type.to_canonical_string(true));
166    }
167    if mask.contains(Object::BALANCE_FIELD) {
168        message.balance = info.balance;
169    }
170
171    message
172}

174#[derive(serde::Serialize, serde::Deserialize)]
175struct PageToken {
176    owner: Address,
177    object_type: Option<move_core_types::language_storage::StructTag>,
178    inner: OwnedObjectInfo,
179}
180
181fn should_load_object(mask: &FieldMaskTree) -> bool {
182    [
183        Object::BCS_FIELD,
184        Object::DIGEST_FIELD,
185        Object::OWNER_FIELD,
186        Object::HAS_PUBLIC_TRANSFER_FIELD,
187        Object::CONTENTS_FIELD,
188        // Object::PACKAGE_FIELD, owned objects can't be packages
189        Object::PREVIOUS_TRANSACTION_FIELD,
190        Object::STORAGE_REBATE_FIELD,
191        Object::JSON_FIELD,
192    ]
193    .into_iter()
194    .any(|field| mask.contains(field))
195}
