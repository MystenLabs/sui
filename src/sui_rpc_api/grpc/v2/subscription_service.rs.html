4//! v2 `SubscriptionService`: filtered, real-time streams of checkpoints,
5//! transactions, and events.
6//!
7//! Each Subscribe API pairs with the LedgerService List API of the same name:
8//! requests take the same DNF filter message and responses use identical cursor
9//! semantics, so clients can replay subscription gaps with the paired List API.
10//!
11//! A subscription behaves like an unbounded ascending scan. Every transaction
12//! and event frame carries a `watermark`; its payload is optional. Every
13//! checkpoint frame carries a scalar `cursor`; its checkpoint payload is
14//! optional, and progress-only checkpoint frames occur only on filtered
15//! streams. The first frame on a
16//! filtered subscription is a progress-only frame establishing the stream's
17//! start position. Further progress-only frames are emitted when a stream
18//! advances a configured number of checkpoints without a matching item (see
19//! `RpcConfig::subscription_watermark_interval`). Streams have no successful
20//! end: when the subscription actor drops a subscriber (lag or backpressure),
21//! the stream simply closes and the client reconnects, backfilling via List.

66    async fn subscribe_checkpoints(
67        &self,
68        request: tonic::Request<SubscribeCheckpointsRequest>,
69    ) -> Result<tonic::Response<BoxStream<SubscribeCheckpointsResponse>>, tonic::Status> {
70        let request = request.into_inner();
71        let read_mask = read_mask_defaults::validate_read_mask::<Checkpoint>(
72            request.read_mask,
73            read_mask_defaults::CHECKPOINT,
74        )?;
75        let query = compile_transaction_filter(self, request.filter.as_ref())?;
76        let (mut receiver, stream_metrics) = register(
77            self,
78            SubscriptionSpec {
79                kind: SubscriptionKind::Checkpoints,
80                query,
81            },
82        )
83        .await?;
84
85        let response = Box::pin(async_stream::stream! {
86            while let Some(update) = receiver.recv().await {
87                let mut response = SubscribeCheckpointsResponse::default();
88                let frame_kind = match update {
89                    SubscriptionUpdate::Matched(matched) => {
90                        let cp = matched.checkpoint.summary.sequence_number;
91                        response.cursor = Some(cp);
92                        response.checkpoint =
93                            Some(render_checkpoint_message(&matched.checkpoint, &read_mask));
94                        SubscriptionFrameKind::Payload
95                    }
96                    SubscriptionUpdate::WatermarkTick { checkpoint: cp, .. } => {
97                        response.cursor = Some(cp);
98                        SubscriptionFrameKind::Watermark
99                    }
100                };
101                stream_metrics.observe_frame(&response, frame_kind);
102                let yielded_at = Instant::now();
103                yield Ok(response);
104                stream_metrics.observe_yield_wait(yielded_at.elapsed());
105            }
106        });
107
108        Ok(tonic::Response::new(response))
109    }

111    async fn subscribe_transactions(
112        &self,
113        request: tonic::Request<SubscribeTransactionsRequest>,
114    ) -> Result<tonic::Response<BoxStream<SubscribeTransactionsResponse>>, tonic::Status> {
115        let request = request.into_inner();
116        let read_mask = read_mask_defaults::validate_read_mask::<ExecutedTransaction>(
117            request.read_mask,
118            read_mask_defaults::TRANSACTION,
119        )?;
120        let query = compile_transaction_filter(self, request.filter.as_ref())?;
121        let (mut receiver, stream_metrics) = register(
122            self,
123            SubscriptionSpec {
124                kind: SubscriptionKind::Transactions,
125                query,
126            },
127        )
128        .await?;
129
130        let service = self.clone();
131        let options = QueryOptions::subscription();
132        let response = Box::pin(async_stream::stream! {
133            let mut boundary: Option<u64> = None;
134            let mut entry_checkpoint = None;
135            while let Some(update) = receiver.recv().await {
136                match update {
137                    SubscriptionUpdate::Matched(matched) => {
138                        let checkpoint = &matched.checkpoint;
139                        let Some(indices) = matched
140                            .matches
141                            .transaction_indices(checkpoint.transactions.len() as u32)
142                        else {
143                            continue;
144                        };
145                        let cp = checkpoint.summary.sequence_number;
146                        let entry_checkpoint = *entry_checkpoint.get_or_insert(cp);
147                        let tx_hi = checkpoint.summary.data().network_total_transactions;
148                        let tx_lo = tx_hi - checkpoint.transactions.len() as u64;
149                        for i in indices {
150                            let tx_seq = tx_lo + i as u64;
151                            // An item never proves its own checkpoint
152                            // complete (list_transactions semantics).
153                            boundary = advance_covered_bound_before_checkpoint(
154                                boundary,
155                                cp,
156                                entry_checkpoint,
157                                &options,
158                            );
159                            let mut response = SubscribeTransactionsResponse::default();
160                            response.transaction = Some(render_transaction_message(
161                                &service,
162                                checkpoint,
163                                i as usize,
164                                &read_mask,
165                            ));
166                            response.watermark = Some(item_watermark(
167                                Position::Transactions {
168                                    checkpoint: cp,
169                                    tx_seq,
170                                },
171                                boundary,
172                            ));
173                            stream_metrics.observe_frame(
174                                &response,
175                                SubscriptionFrameKind::Payload,
176                            );
177                            let yielded_at = Instant::now();
178                            yield Ok(response);
179                            stream_metrics.observe_yield_wait(yielded_at.elapsed());
180                        }
181                    }
182                    SubscriptionUpdate::WatermarkTick { checkpoint: cp, tx_hi } => {
183                        record_watermark_tick_coverage(
184                            &mut boundary,
185                            &mut entry_checkpoint,
186                            cp,
187                            &options,
188                        );
189                        let mut response = SubscribeTransactionsResponse::default();
190                        response.watermark = Some(boundary_watermark(
191                            Position::Transactions {
192                                checkpoint: cp + 1,
193                                tx_seq: tx_hi,
194                            },
195                            boundary,
196                        ));
197                        stream_metrics.observe_frame(
198                            &response,
199                            SubscriptionFrameKind::Watermark,
200                        );
201                        let yielded_at = Instant::now();
202                        yield Ok(response);
203                        stream_metrics.observe_yield_wait(yielded_at.elapsed());
204                    }
205                }
206            }
207        });
208
209        Ok(tonic::Response::new(response))
210    }

212    async fn subscribe_events(
213        &self,
214        request: tonic::Request<SubscribeEventsRequest>,
215    ) -> Result<tonic::Response<BoxStream<SubscribeEventsResponse>>, tonic::Status> {
216        let request = request.into_inner();
217        let read_mask = read_mask_defaults::validate_read_mask::<ProtoEvent>(
218            request.read_mask,
219            read_mask_defaults::EVENT,
220        )?;
221        let query = compile_event_filter(self, request.filter.as_ref())?;
222        let (mut receiver, stream_metrics) = register(
223            self,
224            SubscriptionSpec {
225                kind: SubscriptionKind::Events,
226                query,
227            },
228        )
229        .await?;
230
231        let service = self.clone();
232        let options = QueryOptions::subscription();
233        let response = Box::pin(async_stream::stream! {
234            let mut boundary: Option<u64> = None;
235            let mut entry_checkpoint = None;
236            while let Some(update) = receiver.recv().await {
237                match update {
238                    SubscriptionUpdate::Matched(matched) => {
239                        let checkpoint = &matched.checkpoint;
240                        let Some(pairs) = matched.matches.event_indices(checkpoint) else {
241                            continue;
242                        };
243                        let cp = checkpoint.summary.sequence_number;
244                        let entry_checkpoint = *entry_checkpoint.get_or_insert(cp);
245                        let tx_hi = checkpoint.summary.data().network_total_transactions;
246                        let tx_lo = tx_hi - checkpoint.transactions.len() as u64;
247                        for (tx_idx, ev) in pairs {
248                            let tx = &checkpoint.transactions[tx_idx as usize];
249                            let tx_seq = tx_lo + tx_idx as u64;
250                            boundary = advance_covered_bound_before_checkpoint(
251                                boundary,
252                                cp,
253                                entry_checkpoint,
254                                &options,
255                            );
256                            let event = &tx
257                                .events
258                                .as_ref()
259                                .expect("matched event implies events")
260                                .data[ev as usize];
261                            let mut proto_event = service.render_event_to_proto(
262                                event,
263                                &read_mask,
264                                &checkpoint.object_set,
265                            );
266                            if read_mask.contains(ProtoEvent::CHECKPOINT_FIELD.name) {
267                                proto_event.checkpoint = Some(cp);
268                            }
269                            if read_mask.contains(ProtoEvent::TRANSACTION_DIGEST_FIELD.name) {
270                                proto_event.transaction_digest =
271                                    Some(tx.effects.transaction_digest().base58_encode());
272                            }
273                            if read_mask.contains(ProtoEvent::TRANSACTION_INDEX_FIELD.name) {
274                                proto_event.transaction_index = Some(tx_idx as u64);
275                            }
276                            if read_mask.contains(ProtoEvent::EVENT_INDEX_FIELD.name) {
277                                proto_event.event_index = Some(ev);
278                            }
279                            let mut response = SubscribeEventsResponse::default();
280                            response.event = Some(proto_event);
281                            response.watermark = Some(item_watermark(
282                                Position::Events {
283                                    checkpoint: cp,
284                                    tx_seq,
285                                    event_index: ev,
286                                },
287                                boundary,
288                            ));
289                            stream_metrics.observe_frame(
290                                &response,
291                                SubscriptionFrameKind::Payload,
292                            );
293                            let yielded_at = Instant::now();
294                            yield Ok(response);
295                            stream_metrics.observe_yield_wait(yielded_at.elapsed());
296                        }
297                    }
298                    SubscriptionUpdate::WatermarkTick { checkpoint: cp, tx_hi } => {
299                        record_watermark_tick_coverage(
300                            &mut boundary,
301                            &mut entry_checkpoint,
302                            cp,
303                            &options,
304                        );
305                        let mut response = SubscribeEventsResponse::default();
306                        response.watermark = Some(boundary_watermark(
307                            Position::Events {
308                                checkpoint: cp + 1,
309                                tx_seq: tx_hi,
310                                event_index: 0,
311                            },
312                            boundary,
313                        ));
314                        stream_metrics.observe_frame(
315                            &response,
316                            SubscriptionFrameKind::Watermark,
317                        );
318                        let yielded_at = Instant::now();
319                        yield Ok(response);
320                        stream_metrics.observe_yield_wait(yielded_at.elapsed());
321                    }
322                }
323            }
324        });
325
326        Ok(tonic::Response::new(response))
327    }

330/// Render a full `Checkpoint` message from live executed-checkpoint data,
331/// including the `transactions.balance_changes` special case that
332/// `merge_from` cannot fill (it needs the checkpoint's `ObjectSet`).
333fn render_checkpoint_message(
334    checkpoint: &sui_types::full_checkpoint_content::Checkpoint,
335    read_mask: &FieldMaskTree,
336) -> Checkpoint {
337    let mut checkpoint_message = Checkpoint::merge_from(checkpoint, read_mask);
338
339    if read_mask.contains("transactions.balance_changes") {
340        for (txn, effects) in checkpoint_message
341            .transactions_mut()
342            .iter_mut()
343            .zip_debug_eq(checkpoint.transactions.iter().map(|t| &t.effects))
344        {
345            *txn.balance_changes_mut() = derive_balance_changes_2(effects, &checkpoint.object_set)
346                .into_iter()
347                .map(Into::into)
348                .collect();
349        }
350    }
351
352    checkpoint_message
353}

355fn transaction_objects<'a>(
356    transaction: &'a sui_types::full_checkpoint_content::ExecutedTransaction,
357    checkpoint_objects: &'a sui_types::full_checkpoint_content::ObjectSet,
358) -> impl Iterator<Item = &'a sui_types::object::Object> + 'a {
359    sui_types::storage::get_transaction_object_set(
360        &transaction.transaction,
361        &transaction.effects,
362        &transaction.unchanged_loaded_runtime_objects,
363    )
364    .into_iter()
365    .filter_map(move |key| checkpoint_objects.get(&key))
366}

368/// Render one `ExecutedTransaction` message from live executed-checkpoint
369/// data. The nested-transaction `merge_from` does not set `checkpoint` /
370/// `timestamp` (the checkpoint-level merge does), so set them here, along
371/// with `balance_changes` which needs the checkpoint's `ObjectSet`.
372fn render_transaction_message(
373    service: &RpcService,
374    checkpoint: &sui_types::full_checkpoint_content::Checkpoint,
375    index: usize,
376    read_mask: &FieldMaskTree,
377) -> ExecutedTransaction {
378    let tx = &checkpoint.transactions[index];
379    let mut message = ExecutedTransaction::merge_from(tx, read_mask);
380
381    if let Some(object_mask) = read_mask
382        .subtree(ExecutedTransaction::OBJECTS_FIELD)
383        .and_then(|submask| submask.subtree(ProtoObjectSet::OBJECTS_FIELD))
384    {
385        message.objects = Some(
386            ProtoObjectSet::default().with_objects(
387                transaction_objects(tx, &checkpoint.object_set)
388                    .map(|object| {
389                        service.render_object_to_proto(object, &object_mask, &checkpoint.object_set)
390                    })
391                    .collect(),
392            ),
393        );
394    }
395
396    if read_mask.contains(ExecutedTransaction::CHECKPOINT_FIELD) {
397        message.checkpoint = Some(checkpoint.summary.sequence_number);
398    }
399    if read_mask.contains(ExecutedTransaction::TIMESTAMP_FIELD) {
400        message.timestamp = Some(sui_rpc::proto::timestamp_ms_to_proto(
401            checkpoint.summary.timestamp_ms,
402        ));
403    }
404    if read_mask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD) {
405        message.balance_changes = derive_balance_changes_2(&tx.effects, &checkpoint.object_set)
406            .into_iter()
407            .map(Into::into)
408            .collect();
409    }
410    if read_mask.contains(ExecutedTransaction::TRANSACTION_INDEX_FIELD) {
411        message.transaction_index = Some(index as u64);
412    }
413
414    message
415}

417/// Fold a subscription progress tick into the stream's checkpoint coverage.
418/// A filtered subscription's first tick names the checkpoint immediately before
419/// its entry checkpoint; later ticks name checkpoints the actor fully processed.
420/// Both establish a safe covered boundary.
421fn record_watermark_tick_coverage(
422    covered_checkpoint_bound: &mut Option<u64>,
423    entry_checkpoint: &mut Option<u64>,
424    checkpoint: u64,
425    options: &QueryOptions,
426) {
427    if entry_checkpoint.is_none() {
428        *entry_checkpoint = Some(checkpoint.saturating_add(1));
429    }
430    *covered_checkpoint_bound =
431        merge_covered_checkpoint_bound(*covered_checkpoint_bound, checkpoint, options);
432}

434fn compile_transaction_filter(
435    service: &RpcService,
436    filter: Option<&TransactionFilter>,
437) -> Result<Option<BitmapQuery>, RpcError> {
438    let max_literals = service.config.ledger_history().max_bitmap_filter_literals();
439    filter
440        .map(|filter| transaction_filter_to_query(filter, max_literals))
441        .transpose()
442}

449    filter

452}

454async fn register(
455    service: &RpcService,
456    spec: SubscriptionSpec,
457) -> Result<
458    (
459        mpsc::Receiver<SubscriptionUpdate>,
460        SubscriptionStreamMetrics,
461    ),
462    tonic::Status,
463> {
464    let handle = service
465        .subscription_service_handle
466        .as_ref()
467        .ok_or_else(|| tonic::Status::unimplemented("subscription service not enabled"))?;
468    let kind = spec.kind;
469    let receiver = handle
470        .register_subscription(spec)
471        .await
472        .ok_or_else(|| tonic::Status::unavailable("subscription service is unavailable"))?;
473    let stream_metrics = handle.stream_metrics(kind);
474    Ok((receiver, stream_metrics))
475}
