40pub fn simulate_transaction(
41    service: &RpcService,
42    request: SimulateTransactionRequest,
43) -> Result<SimulateTransactionResponse> {
44    let executor = service
45        .executor
46        .as_ref()
47        .ok_or_else(|| RpcError::new(tonic::Code::Unimplemented, "no transaction executor"))?;
48
49    let read_mask = request
50        .read_mask
51        .as_ref()
52        .map(FieldMaskTree::from_field_mask)
53        .unwrap_or_else(FieldMaskTree::new_wildcard);
54
55    let transaction_proto = request
56        .transaction
57        .as_ref()
58        .ok_or_else(|| FieldViolation::new("transaction").with_reason(ErrorReason::FieldMissing))?;
59
60    let checks = TransactionChecks::from(request.checks());
61
62    // TODO make this more efficient
63    let (reference_gas_price, protocol_config) = {
64        let system_state = service.reader.get_system_state_summary()?;
65        let protocol_config = ProtocolConfig::get_for_version_if_supported(
66            system_state.protocol_version.into(),
67            service.reader.inner().get_chain_identifier()?.chain(),
68        )
69        .ok_or_else(|| {
70            RpcError::new(
71                tonic::Code::Internal,
72                "unable to get current protocol config",
73            )
74        })?;
75
76        (system_state.reference_gas_price, protocol_config)
77    };
78
79    // Try to parse out a fully-formed transaction. If one wasn't provided then we will attempt to
80    // perform transaction resolution.
81    let mut transaction = match sui_sdk_types::Transaction::try_from(transaction_proto) {
82        Ok(transaction) => sui_types::transaction::TransactionData::try_from(transaction)?,
83
84        // If we weren't able to parse out a fully-formed transaction and the client provided BCS
85        // TransactionData, then we'll error out early since we're unable to perform resolution
86        // given a BCS payload
87        Err(e) if transaction_proto.bcs.is_some() => {
88            return Err(FieldViolation::new("transaction")
89                .with_description(format!("invalid transaction: {e}"))
90                .with_reason(ErrorReason::FieldInvalid)
91                .into());
92        }
93
94        // We weren't able to parse out a fully-formed transaction so we'll attempt to perform
95        // transaction resolution
96        _ => resolve::resolve_transaction(
97            service,
98            transaction_proto,
99            reference_gas_price,
100            &protocol_config,
101        )?,
102    };
103
104    let perform_gas_selection = request.do_gas_selection() && checks.enabled();
105    let simulation_result = 'simulate: {
106        // BCS transactions that are already in gasless shape (price=0, no payment) are simulated
107        // as-is — the caller pre-built the transaction, so gas selection is not needed and the
108        // priced flow must be skipped (it would fail with GasPriceUnderRGP for price=0).
109        let skip_gas_selection_for_bcs_gasless = perform_gas_selection
110            && request.transaction().bcs_opt().is_some()
111            && transaction.is_gasless_transaction();
112
113        if perform_gas_selection && !skip_gas_selection_for_bcs_gasless {
114            // If the caller didn't set a non-zero price and the tx passes the cheap structural +
115            // object-input gasless checks, try a gasless simulate first. Post-execution gasless
116            // requirements (all input Coins consumed, minimum transfer amounts) can only be
117            // verified by running the tx. If that fails, we discard the gasless variant and
118            // fall through to the priced flow. `payment` is already empty here, verified by
119            // is_gasless_candidate.
120            if is_gasless_candidate(&request, &transaction, &protocol_config, service)? {
121                let mut gasless_tx = transaction.clone();
122                gasless_tx.gas_data_mut().price = 0;
123                gasless_tx.gas_data_mut().budget = 0;
124                // All gassless txns have to have a correct `ValidDuring` TransactionExpiration.
125                set_valid_during_transaction_expiration(service, &mut gasless_tx)?;
126
127                let simulation_result = executor
128                    .simulate_transaction(gasless_tx.clone(), checks, false)
129                    .map_err(simulation_error_to_rpc_error)?;
130
131                if !is_gasless_post_execution_failure(simulation_result.effects.status()) {
132                    transaction = gasless_tx;
133                    break 'simulate simulation_result;
134                }
135            }
136
137            // Priced-flow budget estimation and gas selection.
138            // At this point, the budget on the transaction can be set to one of the following:
139            // - The budget from the request, if specified.
140            // - The total balance of all of the gas payment coins (clamped to the protocol
141            //   MAX_GAS_BUDGET) in the request if the budget was not
142            //   specified but the gas payment coins were specified.
143            // - Protocol MAX_GAS_BUDGET if the request did not specified neither gas payment or budget.
144            //
145            // If the request did not specify a budget, then simulate the transaction to get a budget estimate and
146            // overwrite the resolved budget with the more accurate estimate.
147            // When the request didn't specify a budget, the budget computed below covers
148            // computation + storage + safe-overhead, with the synthetic gas coin's storage
149            // cost subtracted (it doesn't exist at execution time). The cost of loading
150            // any additional payment objects is added either in `estimate_gas_budget_from_gas_cost`
151            // (when payment was specified) or incrementally inside `select_gas` (when gas
152            // selection picks the coins).
153            let budget_was_estimated = request.transaction().gas_payment().budget.is_none()
154                && request.transaction().bcs_opt().is_none();
155            if budget_was_estimated {
156                let mut estimation_transaction = transaction.clone();
157                estimation_transaction.gas_data_mut().payment = Vec::new();
158                estimation_transaction.gas_data_mut().budget = protocol_config.max_tx_gas();
159
160                let simulation_result = executor
161                    .simulate_transaction(
162                        estimation_transaction,
163                        TransactionChecks::Enabled,
164                        true, /* allow mock gas coin */
165                    )
166                    .map_err(simulation_error_to_rpc_error)?;
167
168                let estimate = estimate_gas_budget_from_gas_cost(
169                    simulation_result.effects.gas_cost_summary(),
170                    reference_gas_price,
171                    request.transaction().gas_payment().objects.len(),
172                    mock_gas_storage_cost(&simulation_result),
173                    &protocol_config,
174                );
175
176                // If the request specified gas payment, then transaction.gas_data().budget should have been
177                // resolved to the cumulative balance of those coins. We don't want to return a resolved transaction
178                // where the gas payment can't satisfy the budget, so validate that balance can actually cover the
179                // estimated budget.
180                let gas_balance = transaction.gas_data().budget;
181                if gas_balance < estimate {
182                    return Err(RpcError::new(
183                        tonic::Code::InvalidArgument,
184                        format!(
185                            "Insufficient gas balance to cover estimated transaction cost. \
186                            Available gas balance: {gas_balance} MIST. Estimated gas budget required: {estimate} MIST"
187                        ),
188                    ));
189                }
190                transaction.gas_data_mut().budget = estimate;
191            }
192
193            if transaction.gas_data().payment.is_empty() {
194                select_gas(
195                    service,
196                    &mut transaction,
197                    // Only adjust the budget for actually-selected coins when we just
198                    // computed the budget from estimation. A caller-supplied budget is
199                    // taken as-is.
200                    budget_was_estimated.then_some(reference_gas_price),
201                    &protocol_config,
202                )?;
203            }
204        }
205
206        executor
207            .simulate_transaction(transaction.clone(), checks, !perform_gas_selection)
208            .map_err(simulation_error_to_rpc_error)?
209    };
210
211    let SimulateTransactionResult {
212        effects,
213        events,
214        objects,
215        execution_result,
216        mock_gas_id,
217        unchanged_loaded_runtime_objects,
218        suggested_gas_price,
219    } = simulation_result;
220
221    if perform_gas_selection && mock_gas_id.is_some() {
222        // If we don't allow for using a mock coin, but we still did, return a server error
223        return Err(RpcError::new(
224            tonic::Code::Internal,
225            "unexpected mock gas coin used",
226        ));
227    }
228
229    let transaction = if let Some(submask) = read_mask.subtree("transaction") {
230        let mut message = ExecutedTransaction::default();
231        let transaction = sui_sdk_types::Transaction::try_from(transaction)?;
232
233        message.balance_changes =
234            if submask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD.name) {
235                derive_balance_changes_2(&effects, &objects)
236                    .into_iter()
237                    .map(Into::into)
238                    .collect()
239            } else {
240                vec![]
241            };
242
243        message.effects = submask
244            .subtree(ExecutedTransaction::EFFECTS_FIELD)
245            .map(|mask| {
246                service.render_effects_to_proto(
247                    &effects,
248                    &unchanged_loaded_runtime_objects,
249                    &objects,
250                    &mask,
251                )
252            });
253
254        message.events = submask
255            .subtree(ExecutedTransaction::EVENTS_FIELD.name)
256            .and_then(|mask| {
257                events.map(|events| service.render_events_to_proto(&events, &mask, &objects))
258            });
259
260        message.transaction = submask
261            .subtree(ExecutedTransaction::TRANSACTION_FIELD.name)
262            .map(|mask| Transaction::merge_from(transaction, &mask));
263
264        message.objects = submask
265            .subtree(
266                ExecutedTransaction::path_builder()
267                    .objects()
268                    .objects()
269                    .finish(),
270            )
271            .map(|mask| {
272                ObjectSet::default().with_objects(
273                    objects
274                        .iter()
275                        .map(|o| service.render_object_to_proto(o, &mask, &objects))
276                        .collect(),
277                )
278            });
279
280        Some(message)
281    } else {
282        None
283    };
284
285    let outputs = if read_mask.contains(SimulateTransactionResponse::COMMAND_OUTPUTS_FIELD) {
286        execution_result
287            .into_iter()
288            .flatten()
289            .map(|(reference_outputs, return_values)| {
290                let mut message = CommandResult::default();
291                message.return_values = return_values
292                    .into_iter()
293                    .map(|(bcs, ty)| to_command_output(service, None, bcs, ty))
294                    .collect();
295                message.mutated_by_ref = reference_outputs
296                    .into_iter()
297                    .map(|(arg, bcs, ty)| to_command_output(service, Some(arg), bcs, ty))
298                    .collect();
299                message
300            })
301            .collect()
302    } else {
303        Vec::new()
304    };
305
306    let mut response = SimulateTransactionResponse::default();
307    response.transaction = transaction;
308    response.command_outputs = outputs;
309    if read_mask.contains(SimulateTransactionResponse::SUGGESTED_GAS_PRICE_FIELD) {
310        response.suggested_gas_price = suggested_gas_price;
311    }
312    Ok(response)
313}

315fn simulation_error_to_rpc_error(error: SuiError) -> RpcError {
316    match error.as_inner() {
317        SuiErrorKind::UserInputError { .. } => {
318            RpcError::new(tonic::Code::InvalidArgument, error.to_string())
319        }
320        SuiErrorKind::UnsupportedFeatureError { .. } => {
321            RpcError::new(tonic::Code::InvalidArgument, error.to_string())
322        }
323        _ => RpcError::new(tonic::Code::Internal, error.to_string()),
324    }
325}

327fn to_command_output(
328    service: &RpcService,
329    arg: Option<sui_types::transaction::Argument>,
330    bcs: Vec<u8>,
331    ty: sui_types::TypeTag,
332) -> CommandOutput {
333    let json = service
334        .reader
335        .inner()
336        .get_type_layout(&ty)
337        .ok()
338        .flatten()
339        .and_then(|layout| {
340            let bound = service.config.max_json_move_value_size();
341            sui_types::object::rpc_visitor::proto::ProtoVisitor::new(bound)
342                .deserialize_value(&bcs, &layout)
343                .map_err(|e| tracing::debug!("unable to convert to JSON: {e}"))
344                .ok()
345                .map(Box::new)
346        });
347
348    let mut message = CommandOutput::default();
349    message.argument = arg.map(Into::into);
350    message.value = Some(Bcs::from(bcs).with_name(ty.to_canonical_string(true)));
351    message.json = json;
352    message
353}

355/// Estimate the gas budget for a transaction based on simulation results.
356///
357/// The estimation includes:
358/// 1. Base cost from gas_cost_summary (computation + storage costs), with the synthetic gas
359///    coin's storage cost subtracted (it doesn't exist at execution time).
360/// 2. Cost of loading additional gas payment objects beyond the synthetic gas coin already
361///    in the simulation. When the request didn't specify gas payment objects, this is 0 —
362///    `select_gas` will pick the actual coins (or address balance) and adjust the budget
363///    incrementally for each one.
364/// 3. Rounding up to the protocol gas rounding step (typically 1000 MIST).
365/// 4. Adding safe overhead buffer (1000 * reference_gas_price).
366/// 5. Clamping to max_tx_gas protocol limit.
367fn estimate_gas_budget_from_gas_cost(
368    gas_cost_summary: &sui_types::gas::GasCostSummary,
369    reference_gas_price: u64,
370    num_payment_objects_on_request: usize,
371    mock_gas_storage_cost: u64,
372    protocol_config: &ProtocolConfig,
373) -> u64 {
374    const GAS_SAFE_OVERHEAD: u64 = 1000;
375
376    // The simulation always loads a synthetic gas coin so that it can produce a gas cost
377    // summary even when the caller hasn't specified a gas payment. That coin's storage
378    // write is phantom — it is not written at execution time (real address-balance gas
379    // emits an accumulator event, real coin gas writes the user-provided coin instead) —
380    // so subtract its contribution from `storage_cost` before deriving the estimate.
381    let storage_cost = gas_cost_summary
382        .storage_cost
383        .saturating_sub(mock_gas_storage_cost);
384    let gas_used = gas_cost_summary
385        .computation_cost
386        .saturating_add(storage_cost);
387    let net_gas_usage = (gas_used as i64).saturating_sub(gas_cost_summary.storage_rebate as i64);
388    let base_estimate_mist = gas_cost_summary.computation_cost.max(if net_gas_usage < 0 {
389        0
390    } else {
391        net_gas_usage as u64
392    });
393
394    // Loading cost for additional gas coins beyond the synthetic gas coin already counted
395    // by the simulation. When the request did not specify any payment objects, the loading
396    // cost is added incrementally inside `select_gas` once the actual coins are known.
397    let extra_payment_objects = (num_payment_objects_on_request as u64).saturating_sub(1);
398    let gas_loading_cost_mist =
399        compute_gas_loading_cost_mist(extra_payment_objects, reference_gas_price, protocol_config);
400
401    let safe_overhead_mist = GAS_SAFE_OVERHEAD.saturating_mul(reference_gas_price);
402
403    base_estimate_mist
404        .saturating_add(gas_loading_cost_mist)
405        .saturating_add(safe_overhead_mist)
406        .min(protocol_config.max_tx_gas())
407}

409/// Cost in MIST of loading `extra_coins` gas-payment objects beyond the synthetic gas coin
410/// already accounted for by the estimation simulation. Mirrors the protocol's per-byte
411/// read cost, rounded up to the protocol gas rounding step in gas units before being
412/// converted to MIST.
413fn compute_gas_loading_cost_mist(
414    extra_coins: u64,
415    reference_gas_price: u64,
416    protocol_config: &ProtocolConfig,
417) -> u64 {
418    let units = extra_coins
419        .saturating_mul(GAS_COIN_SIZE_BYTES)
420        .saturating_mul(protocol_config.obj_access_cost_read_per_byte());
421    let rounded = if let Some(step) = protocol_config.gas_rounding_step_as_option() {
422        round_up_to_nearest(units, step)
423    } else {
424        units
425    };
426    rounded.saturating_mul(reference_gas_price)
427}
428
429/// Round up a value to the nearest multiple of `step` using saturating arithmetic.
430fn round_up_to_nearest(value: u64, step: u64) -> u64 {
431    let remainder = value % step;
432    if remainder == 0 {
433        value
434    } else {
435        value.saturating_add(step - remainder)
436    }
437}

439/// Storage cost (in MIST) of the synthetic gas coin that the simulator wrote during the
440/// estimation pass. The new `storage_rebate` on the written object equals the storage cost
441/// (see `track_storage_mutation` / `collect_storage_and_rebate`) so we can read it directly
442/// from the simulation's object set instead of re-deriving it from protocol parameters.
443/// Returns 0 when the simulation didn't use a mock gas coin.
444fn mock_gas_storage_cost(
445    simulation_result: &sui_types::transaction_executor::SimulateTransactionResult,
446) -> u64 {
447    let Some(mock_gas_id) = simulation_result.mock_gas_id else {
448        return 0;
449    };
450    // Both the input version (rebate 0, since the mock coin is fresh) and the written version
451    // (rebate set to the storage cost) end up in `objects`. The written version always carries
452    // the larger value, so taking the max is correct and avoids depending on iteration order.
453    simulation_result
454        .objects
455        .iter()
456        .filter(|o| o.id() == mock_gas_id)
457        .map(|o| o.storage_rebate)
458        .max()
459        .unwrap_or(0)
460}

462/// Populate a `ValidDuring` expiration covering the current epoch and the next one.
463fn set_valid_during_transaction_expiration(
464    service: &RpcService,
465    transaction: &mut sui_types::transaction::TransactionData,
466) -> Result<()> {
467    // Early return if the TransactionExpiration is already set to `ValidDuring`
468    if matches!(
469        transaction.expiration(),
470        TransactionExpiration::ValidDuring { .. }
471    ) {
472        return Ok(());
473    }
474
475    let current_epoch = service.reader.inner().get_latest_checkpoint()?.epoch();
476    *transaction.expiration_mut() = TransactionExpiration::ValidDuring {
477        min_epoch: Some(current_epoch),
478        max_epoch: Some(current_epoch.saturating_add(1)),
479        min_timestamp: None,
480        max_timestamp: None,
481        chain: service.chain_id,
482        nonce: rand::random(),
483    };
484    Ok(())
485}

487fn select_gas(
488    service: &RpcService,
489    transaction: &mut sui_types::transaction::TransactionData,
490    incremental_loading_rgp: Option<u64>,
491    protocol_config: &ProtocolConfig,
492) -> Result<()> {
493    use sui_types::accumulator_root::AccumulatorValue;
494    use sui_types::balance::Balance;
495    use sui_types::base_types::SequenceNumber;
496    use sui_types::coin_reservation::CoinReservationResolver;
497    use sui_types::coin_reservation::ParsedDigest;
498    use sui_types::coin_reservation::ParsedObjectRefWithdrawal;
499    use sui_types::gas_coin::GAS;
500    use sui_types::gas_coin::GasCoin;
501    use sui_types::transaction::Command;
502    use sui_types::transaction::TransactionDataAPI;
503
504    let reader = &service.reader;
505
506    let owner = transaction.gas_data().owner;
507    let budget = transaction.gas_data().budget;
508
509    let gas_coin_used = transaction
510        .kind()
511        .iter_commands()
512        .any(Command::is_gas_coin_used);
513    let address_balance = reader
514        .lookup_address_balance(owner, GAS::type_())
515        .map(|balance| {
516            // Sum up the explicit SUI reservations (excluding the implicit gas payment) for the
517            // `owner` so that we can deduct that from the available address balance. We use the
518            // estimation variant to avoid double-counting: the gas budget is what we're trying to
519            // satisfy, not a pre-existing reservation.
520            let coin_resolver = CoinReservationResolver::new(reader.inner().clone());
521
522            let reserved_sui = transaction
523                .process_funds_withdrawals_for_estimation(service.chain_id, &coin_resolver)
524                .ok()
525                .and_then(|withdrawals| {
526                    let sui_type = Balance::type_tag(GAS::type_tag());
527                    let sui_account_id = AccumulatorValue::get_field_id(owner, &sui_type).ok()?;
528                    withdrawals
529                        .get(&sui_account_id)
530                        .map(|(amount, _, _)| *amount)
531                })
532                .unwrap_or(0);
533
534            balance.saturating_sub(reserved_sui)
535        });
536
537    // If the gas coin isn't used and there is sufficient address balance budget to satisfy the
538    // required budget then we will use the `owner`s address balance to pay for gas. Otherwise we
539    // fallback to doing coin selection
540    let selected_gas_value = if !gas_coin_used
541        && let Some(address_balance) = address_balance
542        && address_balance >= budget
543    {
544        // We probably don't need to do this, but explicitly clear out the payment to force using
545        // Address balance
546        transaction.gas_data_mut().payment.clear();
547
548        if matches!(transaction.expiration(), TransactionExpiration::None) {
549            set_valid_during_transaction_expiration(service, transaction)?;
550        }
551
552        budget
553    } else {
554        let input_objects = transaction
555            .input_objects()
556            .map_err(anyhow::Error::from)?
557            .iter()
558            .flat_map(|obj| match obj {
559                sui_types::transaction::InputObjectKind::ImmOrOwnedMoveObject((id, _, _)) => {
560                    Some(*id)
561                }
562                _ => None,
563            })
564            .collect_vec();
565
566        let gas_coins = reader
567            .inner()
568            .indexes()
569            .ok_or_else(RpcError::not_found)?
570            .owned_objects_iter(owner, Some(GasCoin::type_()), None)?
571            .filter_ok(|info| !input_objects.contains(&info.object_id))
572            .filter_map_ok(|info| reader.inner().get_object(&info.object_id))
573            // filter for objects which are not ConsensusAddress owned,
574            // since only Address owned can be used for gas payments today
575            .filter_ok(|object| !object.is_consensus())
576            .filter_map_ok(|object| {
577                GasCoin::try_from(&object)
578                    .ok()
579                    .map(|coin| (object.compute_object_reference(), coin.value()))
580            })
581            .take(protocol_config.max_gas_payment_objects() as usize);
582
583        let mut selected_gas = vec![];
584        let mut selected_gas_value = 0;
585
586        for maybe_coin in gas_coins {
587            let (object_ref, value) =
588                maybe_coin.map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
589            selected_gas.push(object_ref);
590            selected_gas_value += value;
591        }
592
593        // When GasCoin is used and there's address balance, prepend a coin reservation
594        // to make all SUI in the account available (coins + address balance)
595        if protocol_config.enable_coin_reservation_obj_refs()
596            && gas_coin_used
597            && let Some(ab_value) = address_balance
598            && ab_value > 0
599        {
600            let current_epoch = service.reader.inner().get_latest_checkpoint()?.epoch();
601
602            let accumulator_obj_id =
603                AccumulatorValue::get_field_id(owner, &Balance::type_tag(GAS::type_tag()))
604                    .map_err(|e| {
605                        RpcError::new(
606                            tonic::Code::Internal,
607                            format!("Failed to get accumulator object ID: {e}"),
608                        )
609                    })?;
610
611            let reservation = ParsedObjectRefWithdrawal::new(
612                *accumulator_obj_id.inner(),
613                current_epoch,
614                ab_value,
615            );
616            let coin_reservation = reservation.encode(SequenceNumber::new(), service.chain_id);
617
618            // Prepend coin reservation to make address balance accessible via GasCoin
619            selected_gas.insert(0, coin_reservation);
620            selected_gas_value += ab_value;
621
622            if matches!(transaction.expiration(), TransactionExpiration::None) {
623                set_valid_during_transaction_expiration(service, transaction)?;
624            }
625        }
626
627        transaction.gas_data_mut().payment = selected_gas;
628
629        selected_gas_value
630    };
631
632    // When the caller asked us to top up the budget for the loading cost of the just-picked
633    // payment objects, do so before the balance check. The simulation already counted the
634    // synthetic gas coin's load, so charge for the rest. Coin reservations don't load real
635    // gas-coin objects and are excluded.
636    let final_budget = if let Some(rgp) = incremental_loading_rgp {
637        let real_coins = transaction
638            .gas_data()
639            .payment
640            .iter()
641            .filter(|obj_ref| !ParsedDigest::is_coin_reservation_digest(&obj_ref.2))
642            .count() as u64;
643        let extra_loading_mist =
644            compute_gas_loading_cost_mist(real_coins.saturating_sub(1), rgp, protocol_config);
645        let new_budget = budget
646            .saturating_add(extra_loading_mist)
647            .min(protocol_config.max_tx_gas());
648        transaction.gas_data_mut().budget = new_budget;
649        new_budget
650    } else {
651        budget
652    };
653
654    if selected_gas_value >= final_budget {
655        Ok(())
656    } else {
657        Err(RpcError::new(
658            tonic::Code::InvalidArgument,
659            format!(
660                "Unable to perform gas selection due to insufficient SUI \
661                balance (in address balance or coins) for account {owner} \
662                to satisfy required budget {final_budget}."
663            ),
664        ))
665    }
666}

668/// Returns true if the simulate request is eligible for auto gas_price=0 handling.
669///
670/// Requires: gasless enabled by protocol, caller did not set price (or set it to 0) and did not
671/// set gas payment objects, tx is a PTB that passes the structural gasless checks, and all loaded
672/// Move object inputs pass the runtime gasless input check (`Coin<T>` with `T` allowlisted,
673/// AddressOwner/ConsensusAddressOwner).
674fn is_gasless_candidate(
675    request: &SimulateTransactionRequest,
676    transaction: &sui_types::transaction::TransactionData,
677    protocol_config: &ProtocolConfig,
678    service: &RpcService,
679) -> Result<bool> {
680    if !protocol_config.enable_gasless() {
681        return Ok(false);
682    }
683    // When the caller passed a full BCS TransactionData, treat it as explicit — don't second-guess
684    // their gas choice. Only auto-switch in the unresolved/proto path.
685    if request.transaction().bcs_opt().is_some() {
686        return Ok(false);
687    }
688    // An explicit non-zero price opts out of gasless. price=0 is treated the same as unset: the
689    // caller is either asserting gasless intent or echoing the suggestion from a prior simulate.
690    if request
691        .transaction()
692        .gas_payment()
693        .price
694        .is_some_and(|p| p != 0)
695    {
696        return Ok(false);
697    }
698    if !request.transaction().gas_payment().objects.is_empty() {
699        return Ok(false);
700    }
701    let TransactionKind::ProgrammableTransaction(pt) = transaction.kind() else {
702        return Ok(false);
703    };
704    if pt.validate_gasless_transaction(protocol_config).is_err() {
705        return Ok(false);
706    }
707
708    // Load Move object inputs so we can run the runtime input check. Packages need not be loaded
709    // since check_gasless_object_inputs skips them.
710    let input_object_kinds = match transaction.input_objects() {
711        Ok(kinds) => kinds,
712        Err(_) => return Ok(false),
713    };
714    let mut loaded = Vec::with_capacity(input_object_kinds.len());
715    for kind in input_object_kinds {
716        match kind {
717            InputObjectKind::MovePackage(_) => continue,
718            InputObjectKind::ImmOrOwnedMoveObject(object_ref) => {
719                let Some(object) = service.reader.inner().get_object(&object_ref.0) else {
720                    return Ok(false);
721                };
722                loaded.push(ObjectReadResult::new(kind, object.into()));
723            }
724            InputObjectKind::SharedMoveObject { id, .. } => {
725                let Some(object) = service.reader.inner().get_object(&id) else {
726                    return Ok(false);
727                };
728                loaded.push(ObjectReadResult::new(kind, object.into()));
729            }
730        }
731    }
732    let input_objects = InputObjects::new(loaded);
733    Ok(
734        sui_transaction_checks::check_gasless_object_inputs(&input_objects, protocol_config)
735            .is_ok(),
736    )
737}

739/// The executor maps a post-execution gasless-requirements failure
740/// (`TemporaryStore::check_gasless_execution_requirements`) to
741/// `ExecutionErrorKind::InsufficientGas` on the effects (see
742/// `sui-execution/latest/sui-adapter/src/execution_engine.rs`). During a gasless simulate, that's
743/// the only way InsufficientGas can surface (gasless uses a large compute cap and ignores budget),
744/// so we treat it as the fallback trigger.
745fn is_gasless_post_execution_failure(status: &ExecutionStatus) -> bool {
746    matches!(
747        status,
748        ExecutionStatus::Failure(ExecutionFailure {
749            error: sui_types::execution_status::ExecutionErrorKind::InsufficientGas,
750            ..
751        })
752    )
753}
