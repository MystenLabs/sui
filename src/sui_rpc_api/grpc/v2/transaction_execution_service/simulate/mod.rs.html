36pub fn simulate_transaction(
37    service: &RpcService,
38    request: SimulateTransactionRequest,
39) -> Result<SimulateTransactionResponse> {
40    let executor = service
41        .executor
42        .as_ref()
43        .ok_or_else(|| RpcError::new(tonic::Code::Unimplemented, "no transaction executor"))?;
44
45    let read_mask = request
46        .read_mask
47        .as_ref()
48        .map(FieldMaskTree::from_field_mask)
49        .unwrap_or_else(FieldMaskTree::new_wildcard);
50
51    let transaction_proto = request
52        .transaction
53        .as_ref()
54        .ok_or_else(|| FieldViolation::new("transaction").with_reason(ErrorReason::FieldMissing))?;
55
56    let checks = TransactionChecks::from(request.checks());
57
58    // TODO make this more efficient
59    let (reference_gas_price, protocol_config) = {
60        let system_state = service.reader.get_system_state_summary()?;
61        let protocol_config = ProtocolConfig::get_for_version_if_supported(
62            system_state.protocol_version.into(),
63            service.reader.inner().get_chain_identifier()?.chain(),
64        )
65        .ok_or_else(|| {
66            RpcError::new(
67                tonic::Code::Internal,
68                "unable to get current protocol config",
69            )
70        })?;
71
72        (system_state.reference_gas_price, protocol_config)
73    };
74
75    // Try to parse out a fully-formed transaction. If one wasn't provided then we will attempt to
76    // perform transaction resolution.
77    let mut transaction = match sui_sdk_types::Transaction::try_from(transaction_proto) {
78        Ok(transaction) => sui_types::transaction::TransactionData::try_from(transaction)?,
79
80        // If we weren't able to parse out a fully-formed transaction and the client provided BCS
81        // TransactionData, then we'll error out early since we're unable to perform resolution
82        // given a BCS payload
83        Err(e) if transaction_proto.bcs.is_some() => {
84            return Err(FieldViolation::new("transaction")
85                .with_description(format!("invalid transaction: {e}"))
86                .with_reason(ErrorReason::FieldInvalid)
87                .into());
88        }
89
90        // We weren't able to parse out a fully-formed transaction so we'll attempt to perform
91        // transaction resolution
92        _ => resolve::resolve_transaction(
93            service,
94            transaction_proto,
95            reference_gas_price,
96            &protocol_config,
97        )?,
98    };
99
100    // Perform budgest estimation and gas selection if requested and if TransactionChecks are enabled (it
101    // makes no sense to do gas selection if checks are disabled because such a transaction can't
102    // ever be committed to the chain).
103    if request.do_gas_selection() && checks.enabled() {
104        // At this point, the budget on the transaction can be set to one of the following:
105        // - The budget from the request, if specified.
106        // - The total balance of all of the gas payment coins (clamped to the protocol
107        //   MAX_GAS_BUDGET) in the request if the budget was not
108        //   specified but the gas payment coins were specified.
109        // - Protocol MAX_GAS_BUDGET if the request did not specified neither gas payment or budget.
110        //
111        // If the request did not specify a budget, then simulate the transaction to get a budget estimate and
112        // overwrite the resolved budget with the more accurate estimate.
113        if request.transaction().gas_payment().budget.is_none()
114            && request.transaction().bcs_opt().is_none()
115        {
116            let simulation_result = executor
117                .simulate_transaction(transaction.clone(), TransactionChecks::Enabled)
118                .map_err(anyhow::Error::from)?;
119
120            let estimate = estimate_gas_budget_from_gas_cost(
121                simulation_result.effects.gas_cost_summary(),
122                reference_gas_price,
123            );
124
125            // If the request specified gas payment, then transaction.gas_data().budget should have been
126            // resolved to the cumulative balance of those coins. We don't want to return a resolved transaction
127            // where the gas payment can't satisfy the budget, so validate that balance can actually cover the
128            // estimated budget.
129            let gas_balance = transaction.gas_data().budget;
130            if gas_balance < estimate {
131                return Err(RpcError::new(
132                    tonic::Code::InvalidArgument,
133                    format!(
134                        "Insufficient gas balance to cover estimated transaction cost. \
135                        Available gas balance: {gas_balance} MIST. Estimated gas budget required: {estimate} MIST"
136                    ),
137                ));
138            }
139            transaction.gas_data_mut().budget = estimate;
140        }
141
142        if transaction.gas_data().payment.is_empty() {
143            let input_objects = transaction
144                .input_objects()
145                .map_err(anyhow::Error::from)?
146                .iter()
147                .flat_map(|obj| match obj {
148                    sui_types::transaction::InputObjectKind::ImmOrOwnedMoveObject((id, _, _)) => {
149                        Some(*id)
150                    }
151                    _ => None,
152                })
153                .collect_vec();
154            let gas_coins = select_gas(
155                &service.reader,
156                transaction.gas_data().owner,
157                transaction.gas_data().budget,
158                protocol_config.max_gas_payment_objects(),
159                &input_objects,
160            )?;
161            transaction.gas_data_mut().payment = gas_coins;
162        }
163    }
164
165    let SimulateTransactionResult {
166        effects,
167        events,
168        objects,
169        execution_result,
170        mock_gas_id: _,
171        unchanged_loaded_runtime_objects,
172    } = executor
173        .simulate_transaction(transaction.clone(), checks)
174        .map_err(anyhow::Error::from)?;
175
176    let transaction = if let Some(submask) = read_mask.subtree("transaction") {
177        let mut message = ExecutedTransaction::default();
178        let transaction = sui_sdk_types::Transaction::try_from(transaction)?;
179
180        message.balance_changes =
181            if submask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD.name) {
182                derive_balance_changes_2(&effects, &objects)
183                    .into_iter()
184                    .map(Into::into)
185                    .collect()
186            } else {
187                vec![]
188            };
189
190        message.effects = {
191            let effects = sui_sdk_types::TransactionEffects::try_from(effects)?;
192            submask
193                .subtree(ExecutedTransaction::EFFECTS_FIELD)
194                .map(|mask| {
195                    let mut effects = TransactionEffects::merge_from(&effects, &mask);
196
197                    if submask.contains(TransactionEffects::UNCHANGED_LOADED_RUNTIME_OBJECTS_FIELD)
198                    {
199                        effects.unchanged_loaded_runtime_objects = unchanged_loaded_runtime_objects
200                            .iter()
201                            .map(Into::into)
202                            .collect();
203                    }
204
205                    if mask.contains(TransactionEffects::CHANGED_OBJECTS_FIELD.name) {
206                        for changed_object in effects.changed_objects.iter_mut() {
207                            let Ok(object_id) = changed_object.object_id().parse::<ObjectID>()
208                            else {
209                                continue;
210                            };
211
212                            if let Some(object) = objects.iter().find(|o| o.id() == object_id) {
213                                changed_object.object_type = Some(match object.struct_tag() {
214                                    Some(struct_tag) => struct_tag.to_canonical_string(true),
215                                    None => "package".to_owned(),
216                                });
217                            }
218                        }
219                    }
220
221                    if mask.contains(TransactionEffects::UNCHANGED_CONSENSUS_OBJECTS_FIELD.name) {
222                        for unchanged_consensus_object in
223                            effects.unchanged_consensus_objects.iter_mut()
224                        {
225                            let Ok(object_id) =
226                                unchanged_consensus_object.object_id().parse::<ObjectID>()
227                            else {
228                                continue;
229                            };
230
231                            if let Some(object) = objects.iter().find(|o| o.id() == object_id) {
232                                unchanged_consensus_object.object_type =
233                                    Some(match object.struct_tag() {
234                                        Some(struct_tag) => struct_tag.to_canonical_string(true),
235                                        None => "package".to_owned(),
236                                    });
237                            }
238                        }
239                    }
240
241                    // Try to render clever error info
242                    super::super::ledger_service::render_clever_error(service, &mut effects);
243
244                    effects
245                })
246        };
247
248        message.events = submask
249            .subtree(ExecutedTransaction::EVENTS_FIELD.name)
250            .and_then(|mask| {
251                events.map(|events| {
252                    sui_sdk_types::TransactionEvents::try_from(events)
253                        .map(|events| TransactionEvents::merge_from(events, &mask))
254                })
255            })
256            .transpose()?;
257
258        message.transaction = submask
259            .subtree(ExecutedTransaction::TRANSACTION_FIELD.name)
260            .map(|mask| Transaction::merge_from(transaction, &mask));
261
262        message.objects = submask
263            .subtree(
264                ExecutedTransaction::path_builder()
265                    .objects()
266                    .objects()
267                    .finish(),
268            )
269            .map(|mask| {
270                ObjectSet::default().with_objects(
271                    objects
272                        .iter()
273                        .map(|o| Object::merge_from(o, &mask))
274                        .collect(),
275                )
276            });
277
278        Some(message)
279    } else {
280        None
281    };
282
283    let outputs = if read_mask.contains(SimulateTransactionResponse::COMMAND_OUTPUTS_FIELD) {
284        execution_result
285            .into_iter()
286            .flatten()
287            .map(|(reference_outputs, return_values)| {
288                let mut message = CommandResult::default();
289                message.return_values = return_values
290                    .into_iter()
291                    .map(|(bcs, ty)| to_command_output(service, None, bcs, ty))
292                    .collect();
293                message.mutated_by_ref = reference_outputs
294                    .into_iter()
295                    .map(|(arg, bcs, ty)| to_command_output(service, Some(arg), bcs, ty))
296                    .collect();
297                message
298            })
299            .collect()
300    } else {
301        Vec::new()
302    };
303
304    let mut response = SimulateTransactionResponse::default();
305    response.transaction = transaction;
306    response.command_outputs = outputs;
307    Ok(response)
308}

310fn to_command_output(
311    service: &RpcService,
312    arg: Option<sui_types::transaction::Argument>,
313    bcs: Vec<u8>,
314    ty: sui_types::TypeTag,
315) -> CommandOutput {
316    let json = service
317        .reader
318        .inner()
319        .get_type_layout(&ty)
320        .ok()
321        .flatten()
322        .and_then(|layout| {
323            sui_types::proto_value::ProtoVisitorBuilder::new(
324                service.config.max_json_move_value_size(),
325            )
326            .deserialize_value(&bcs, &layout)
327            .map_err(|e| tracing::debug!("unable to convert to JSON: {e}"))
328            .ok()
329            .map(Box::new)
330        });
331
332    let mut message = CommandOutput::default();
333    message.argument = arg.map(Into::into);
334    message.value = Some(Bcs::from(bcs).with_name(ty.to_canonical_string(true)));
335    message.json = json;
336    message
337}

339/// Estimate the gas budget using the gas_cost_summary from a previous DryRun
340///
341/// The estimated gas budget is computed as following:
342/// * the maximum between A and B, where:
343///   A = computation cost + GAS_SAFE_OVERHEAD * reference gas price
344///   B = computation cost + storage cost - storage rebate + GAS_SAFE_OVERHEAD * reference gas price
345///   overhead
346///
347/// This gas estimate is computed similarly as in the TypeScript SDK
348fn estimate_gas_budget_from_gas_cost(
349    gas_cost_summary: &sui_types::gas::GasCostSummary,
350    reference_gas_price: u64,
351) -> u64 {
352    const GAS_SAFE_OVERHEAD: u64 = 1000;
353
354    let safe_overhead = GAS_SAFE_OVERHEAD * reference_gas_price;
355    let computation_cost_with_overhead = gas_cost_summary.computation_cost + safe_overhead;
356
357    let gas_usage = gas_cost_summary.net_gas_usage() + safe_overhead as i64;
358    computation_cost_with_overhead.max(if gas_usage < 0 { 0 } else { gas_usage as u64 })
359}

361fn select_gas(
362    reader: &StateReader,
363    owner: SuiAddress,
364    budget: u64,
365    max_gas_payment_objects: u32,
366    input_objects: &[ObjectID],
367) -> Result<Vec<ObjectRef>> {
368    use sui_types::gas_coin::GasCoin;
369
370    let gas_coins = reader
371        .inner()
372        .indexes()
373        .ok_or_else(RpcError::not_found)?
374        .owned_objects_iter(owner, Some(GasCoin::type_()), None)?
375        .filter_ok(|info| !input_objects.contains(&info.object_id))
376        .filter_map_ok(|info| reader.inner().get_object(&info.object_id))
377        // filter for objects which are not ConsensusAddress owned,
378        // since only Address owned can be used for gas payments today
379        .filter_ok(|object| !object.is_consensus())
380        .filter_map_ok(|object| {
381            GasCoin::try_from(&object)
382                .ok()
383                .map(|coin| (object.compute_object_reference(), coin.value()))
384        })
385        .take(max_gas_payment_objects as usize);
386
387    let mut selected_gas = vec![];
388    let mut selected_gas_value = 0;
389
390    for maybe_coin in gas_coins {
391        let (object_ref, value) =
392            maybe_coin.map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
393        selected_gas.push(object_ref);
394        selected_gas_value += value;
395    }
396
397    if selected_gas_value >= budget {
398        Ok(selected_gas)
399    } else {
400        Err(RpcError::new(
401            tonic::Code::InvalidArgument,
402            format!(
403                "unable to select sufficient gas coins from account {owner} \
404                    to satisfy required budget {budget}"
405            ),
406        ))
407    }
