31pub fn simulate_transaction(
32    service: &RpcService,
33    request: SimulateTransactionRequest,
34) -> Result<SimulateTransactionResponse> {
35    let executor = service
36        .executor
37        .as_ref()
38        .ok_or_else(|| RpcError::new(tonic::Code::Unimplemented, "no transaction executor"))?;
39
40    let read_mask = request
41        .read_mask
42        .as_ref()
43        .map(FieldMaskTree::from_field_mask)
44        .unwrap_or_else(FieldMaskTree::new_wildcard);
45
46    let transaction_proto = request
47        .transaction
48        .as_ref()
49        .ok_or_else(|| FieldViolation::new("transaction").with_reason(ErrorReason::FieldMissing))?;
50
51    let checks = TransactionChecks::from(request.checks());
52
53    // TODO make this more efficient
54    let (reference_gas_price, protocol_config) = {
55        let system_state = service.reader.get_system_state_summary()?;
56        let protocol_config = ProtocolConfig::get_for_version_if_supported(
57            system_state.protocol_version.into(),
58            service.reader.inner().get_chain_identifier()?.chain(),
59        )
60        .ok_or_else(|| {
61            RpcError::new(
62                tonic::Code::Internal,
63                "unable to get current protocol config",
64            )
65        })?;
66
67        (system_state.reference_gas_price, protocol_config)
68    };
69
70    // Try to parse out a fully-formed transaction. If one wasn't provided then we will attempt to
71    // perform transaction resolution.
72    let mut transaction = match sui_sdk_types::Transaction::try_from(transaction_proto) {
73        Ok(transaction) => sui_types::transaction::TransactionData::try_from(transaction)?,
74
75        // If we weren't able to parse out a fully-formed transaction and the client provided BCS
76        // TransactionData, then we'll error out early since we're unable to perform resolution
77        // given a BCS payload
78        Err(e) if transaction_proto.bcs.is_some() => {
79            return Err(FieldViolation::new("transaction")
80                .with_description(format!("invalid transaction: {e}"))
81                .with_reason(ErrorReason::FieldInvalid)
82                .into());
83        }
84
85        // We weren't able to parse out a fully-formed transaction so we'll attempt to perform
86        // transaction resolution
87        _ => resolve::resolve_transaction(
88            service,
89            transaction_proto,
90            reference_gas_price,
91            &protocol_config,
92        )?,
93    };
94
95    // Perform budgest estimation and gas selection if requested and if TransactionChecks are enabled (it
96    // makes no sense to do gas selection if checks are disabled because such a transaction can't
97    // ever be committed to the chain).
98    if request.do_gas_selection() && checks.enabled() {
99        // At this point, the budget on the transaction can be set to one of the following:
100        // - The budget from the request, if specified.
101        // - The total balance of all of the gas payment coins (clamped to the protocol
102        //   MAX_GAS_BUDGET) in the request if the budget was not
103        //   specified but the gas payment coins were specified.
104        // - Protocol MAX_GAS_BUDGET if the request did not specified neither gas payment or budget.
105        //
106        // If the request did not specify a budget, then simulate the transaction to get a budget estimate and
107        // overwrite the resolved budget with the more accurate estimate.
108        if request.transaction().gas_payment().budget.is_none()
109            && request.transaction().bcs_opt().is_none()
110        {
111            let mut estimation_transaction = transaction.clone();
112            estimation_transaction.gas_data_mut().payment = Vec::new();
113            estimation_transaction.gas_data_mut().budget = protocol_config.max_tx_gas();
114
115            let simulation_result = executor
116                .simulate_transaction(
117                    estimation_transaction,
118                    TransactionChecks::Enabled,
119                    true, /* allow mock gas coin */
120                )
121                .map_err(anyhow::Error::from)?;
122
123            let estimate = estimate_gas_budget_from_gas_cost(
124                simulation_result.effects.gas_cost_summary(),
125                reference_gas_price,
126                request.transaction().gas_payment().objects.len(),
127                &protocol_config,
128            );
129
130            // If the request specified gas payment, then transaction.gas_data().budget should have been
131            // resolved to the cumulative balance of those coins. We don't want to return a resolved transaction
132            // where the gas payment can't satisfy the budget, so validate that balance can actually cover the
133            // estimated budget.
134            let gas_balance = transaction.gas_data().budget;
135            if gas_balance < estimate {
136                return Err(RpcError::new(
137                    tonic::Code::InvalidArgument,
138                    format!(
139                        "Insufficient gas balance to cover estimated transaction cost. \
140                        Available gas balance: {gas_balance} MIST. Estimated gas budget required: {estimate} MIST"
141                    ),
142                ));
143            }
144            transaction.gas_data_mut().budget = estimate;
145        }
146
147        if transaction.gas_data().payment.is_empty() {
148            select_gas(
149                service,
150                &mut transaction,
151                protocol_config.max_gas_payment_objects(),
152            )?;
153        }
154    }
155
156    let allow_mock_gas_coin = checks.disabled() || !request.do_gas_selection();
157
158    let SimulateTransactionResult {
159        effects,
160        events,
161        objects,
162        execution_result,
163        mock_gas_id,
164        unchanged_loaded_runtime_objects,
165        suggested_gas_price,
166    } = executor
167        .simulate_transaction(transaction.clone(), checks, allow_mock_gas_coin)
168        .map_err(anyhow::Error::from)?;
169
170    if !allow_mock_gas_coin && mock_gas_id.is_some() {
171        // If we don't allow for using a mock coin, but we still did, return a server error
172        return Err(RpcError::new(
173            tonic::Code::Internal,
174            "unexpected mock gas coin used",
175        ));
176    }
177
178    let transaction = if let Some(submask) = read_mask.subtree("transaction") {
179        let mut message = ExecutedTransaction::default();
180        let transaction = sui_sdk_types::Transaction::try_from(transaction)?;
181
182        message.balance_changes =
183            if submask.contains(ExecutedTransaction::BALANCE_CHANGES_FIELD.name) {
184                derive_balance_changes_2(&effects, &objects)
185                    .into_iter()
186                    .map(Into::into)
187                    .collect()
188            } else {
189                vec![]
190            };
191
192        message.effects = submask
193            .subtree(ExecutedTransaction::EFFECTS_FIELD)
194            .map(|mask| {
195                service.render_effects_to_proto(
196                    &effects,
197                    &unchanged_loaded_runtime_objects,
198                    &objects,
199                    &mask,
200                )
201            });
202
203        message.events = submask
204            .subtree(ExecutedTransaction::EVENTS_FIELD.name)
205            .and_then(|mask| events.map(|events| service.render_events_to_proto(&events, &mask)));
206
207        message.transaction = submask
208            .subtree(ExecutedTransaction::TRANSACTION_FIELD.name)
209            .map(|mask| Transaction::merge_from(transaction, &mask));
210
211        message.objects = submask
212            .subtree(
213                ExecutedTransaction::path_builder()
214                    .objects()
215                    .objects()
216                    .finish(),
217            )
218            .map(|mask| {
219                ObjectSet::default().with_objects(
220                    objects
221                        .iter()
222                        .map(|o| service.render_object_to_proto(o, &mask))
223                        .collect(),
224                )
225            });
226
227        Some(message)
228    } else {
229        None
230    };
231
232    let outputs = if read_mask.contains(SimulateTransactionResponse::COMMAND_OUTPUTS_FIELD) {
233        execution_result
234            .into_iter()
235            .flatten()
236            .map(|(reference_outputs, return_values)| {
237                let mut message = CommandResult::default();
238                message.return_values = return_values
239                    .into_iter()
240                    .map(|(bcs, ty)| to_command_output(service, None, bcs, ty))
241                    .collect();
242                message.mutated_by_ref = reference_outputs
243                    .into_iter()
244                    .map(|(arg, bcs, ty)| to_command_output(service, Some(arg), bcs, ty))
245                    .collect();
246                message
247            })
248            .collect()
249    } else {
250        Vec::new()
251    };
252
253    let mut response = SimulateTransactionResponse::default();
254    response.transaction = transaction;
255    response.command_outputs = outputs;
256    if read_mask.contains(SimulateTransactionResponse::SUGGESTED_GAS_PRICE_FIELD) {
257        response.suggested_gas_price = suggested_gas_price;
258    }
259    Ok(response)
260}

262fn to_command_output(
263    service: &RpcService,
264    arg: Option<sui_types::transaction::Argument>,
265    bcs: Vec<u8>,
266    ty: sui_types::TypeTag,
267) -> CommandOutput {
268    let json = service
269        .reader
270        .inner()
271        .get_type_layout(&ty)
272        .ok()
273        .flatten()
274        .and_then(|layout| {
275            sui_types::proto_value::ProtoVisitor::new(service.config.max_json_move_value_size())
276                .deserialize_value(&bcs, &layout)
277                .map_err(|e| tracing::debug!("unable to convert to JSON: {e}"))
278                .ok()
279                .map(Box::new)
280        });
281
282    let mut message = CommandOutput::default();
283    message.argument = arg.map(Into::into);
284    message.value = Some(Bcs::from(bcs).with_name(ty.to_canonical_string(true)));
285    message.json = json;
286    message
287}

289/// Estimate the gas budget for a transaction based on simulation results.
290///
291/// The estimation includes:
292/// 1. Base cost from gas_cost_summary (computation + storage costs)
293/// 2. Cost of loading gas payment objects (which weren't loaded during simulation)
294/// 3. Rounding up to the protocol gas rounding step (typically 1000 MIST)
295/// 4. Adding safe overhead buffer (1000 * reference_gas_price)
296/// 5. Clamping to max_tx_gas protocol limit
297fn estimate_gas_budget_from_gas_cost(
298    gas_cost_summary: &sui_types::gas::GasCostSummary,
299    reference_gas_price: u64,
300    num_payment_objects_on_request: usize,
301    protocol_config: &ProtocolConfig,
302) -> u64 {
303    const GAS_SAFE_OVERHEAD: u64 = 1000;
304
305    // Calculate base estimate from gas cost summary (in MIST)
306    let gas_usage = gas_cost_summary.net_gas_usage();
307    let base_estimate_mist =
308        gas_cost_summary
309            .computation_cost
310            .max(if gas_usage < 0 { 0 } else { gas_usage as u64 });
311
312    // Calculate cost of loading gas payment objects.
313    // Subtract 1 because the simulation already loaded one ephemeral gas coin.
314    let num_payment_objects_for_estimation = {
315        let total = if num_payment_objects_on_request == 0 {
316            protocol_config.max_gas_payment_objects() as u64
317        } else {
318            num_payment_objects_on_request as u64
319        };
320        total.saturating_sub(1)
321    };
322
323    // Calculate gas loading cost in gas units
324    let gas_loading_cost_units = num_payment_objects_for_estimation
325        .saturating_mul(GAS_COIN_SIZE_BYTES)
326        .saturating_mul(protocol_config.obj_access_cost_read_per_byte());
327
328    // Round up to the nearest gas rounding step (in gas units)
329    let rounded_gas_loading_cost_units =
330        if let Some(step) = protocol_config.gas_rounding_step_as_option() {
331            round_up_to_nearest(gas_loading_cost_units, step)
332        } else {
333            gas_loading_cost_units
334        };
335
336    // Convert gas loading cost to MIST
337    let gas_loading_cost_mist = rounded_gas_loading_cost_units.saturating_mul(reference_gas_price);
338
339    // Calculate safe overhead buffer in MIST
340    let safe_overhead_mist = GAS_SAFE_OVERHEAD.saturating_mul(reference_gas_price);
341
342    // Add all together: base (MIST) + loading (MIST) + overhead (MIST)
343    let estimate_mist = base_estimate_mist
344        .saturating_add(gas_loading_cost_mist)
345        .saturating_add(safe_overhead_mist);
346
347    // Clamp to max_tx_gas to ensure we don't exceed the protocol limit
348    estimate_mist.min(protocol_config.max_tx_gas())
349}
350
351/// Round up a value to the nearest multiple of `step` using saturating arithmetic.
352fn round_up_to_nearest(value: u64, step: u64) -> u64 {
353    let remainder = value % step;
354    if remainder == 0 {
355        value
356    } else {
357        value.saturating_add(step - remainder)
358    }
359}

361fn select_gas(
362    service: &RpcService,
363    transaction: &mut sui_types::transaction::TransactionData,
364    max_gas_payment_objects: u32,
365) -> Result<()> {
366    use sui_types::gas_coin::GAS;
367    use sui_types::gas_coin::GasCoin;
368    use sui_types::transaction::Command;
369    use sui_types::transaction::Reservation;
370    use sui_types::transaction::TransactionDataAPI;
371    use sui_types::transaction::TransactionExpiration;
372    use sui_types::transaction::WithdrawalTypeArg;
373
374    let reader = &service.reader;
375
376    let owner = transaction.gas_data().owner;
377    let budget = transaction.gas_data().budget;
378
379    let gas_coin_used = transaction
380        .kind()
381        .iter_commands()
382        .any(Command::is_gas_coin_used);
383    let address_balance = reader
384        .lookup_address_balance(owner, GAS::type_())
385        .map(|balance| {
386            // Sum up the total SUI reservations for the `owner` so that we can deduct that from the
387            // available address balance for determining if an account as sufficient funds.
388            let reserved_sui = transaction
389                .get_funds_withdrawals()
390                .into_iter()
391                .filter_map(|w| {
392                    // Skip if this withdrawal isn't for the gas owner
393                    if w.owner_for_withdrawal(&*transaction) != owner {
394                        return None;
395                    }
396
397                    // Skip if this withdrawal isn't for SUI
398                    let WithdrawalTypeArg::Balance(coin_type) = &w.type_arg;
399                    if !GAS::is_gas_type(coin_type) {
400                        return None;
401                    }
402
403                    match w.reservation {
404                        Reservation::MaxAmountU64(value) => Some(value),
405                    }
406                })
407                .sum::<u64>();
408
409            balance.saturating_sub(reserved_sui)
410        });
411
412    // If the gas coin isn't used and there is sufficient address balance budget to satisfy the
413    // required budget then we will use the `owner`s address balance to pay for gas. Otherwise we
414    // fallback to doing coin selection
415    let selected_gas_value = if !gas_coin_used
416        && let Some(address_balance) = address_balance
417        && address_balance >= budget
418    {
419        // We probably don't need to do this, but explicitly clear out the payment to force using
420        // Address balance
421        transaction.gas_data_mut().payment.clear();
422
423        let current_epoch = service.reader.inner().get_latest_checkpoint()?.epoch();
424
425        *transaction.expiration_mut() = TransactionExpiration::ValidDuring {
426            min_epoch: Some(current_epoch),
427            max_epoch: Some(current_epoch.saturating_add(1)),
428            min_timestamp: None,
429            max_timestamp: None,
430            chain: service.chain_id,
431            nonce: rand::random(), // generate a random nonce to use
432        };
433
434        budget
435    } else {
436        let input_objects = transaction
437            .input_objects()
438            .map_err(anyhow::Error::from)?
439            .iter()
440            .flat_map(|obj| match obj {
441                sui_types::transaction::InputObjectKind::ImmOrOwnedMoveObject((id, _, _)) => {
442                    Some(*id)
443                }
444                _ => None,
445            })
446            .collect_vec();
447
448        let gas_coins = reader
449            .inner()
450            .indexes()
451            .ok_or_else(RpcError::not_found)?
452            .owned_objects_iter(owner, Some(GasCoin::type_()), None)?
453            .filter_ok(|info| !input_objects.contains(&info.object_id))
454            .filter_map_ok(|info| reader.inner().get_object(&info.object_id))
455            // filter for objects which are not ConsensusAddress owned,
456            // since only Address owned can be used for gas payments today
457            .filter_ok(|object| !object.is_consensus())
458            .filter_map_ok(|object| {
459                GasCoin::try_from(&object)
460                    .ok()
461                    .map(|coin| (object.compute_object_reference(), coin.value()))
462            })
463            .take(max_gas_payment_objects as usize);
464
465        let mut selected_gas = vec![];
466        let mut selected_gas_value = 0;
467
468        for maybe_coin in gas_coins {
469            let (object_ref, value) =
470                maybe_coin.map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
471            selected_gas.push(object_ref);
472            selected_gas_value += value;
473        }
474
475        transaction.gas_data_mut().payment = selected_gas;
476
477        selected_gas_value
478    };
479
480    if selected_gas_value >= budget {
481        Ok(())
482    } else {
483        Err(RpcError::new(
484            tonic::Code::InvalidArgument,
485            format!(
486                "Unable to perform gas selection due to insufficient SUI \
487                balance (in address balance or coins) for account {owner} \
488                to satisfy required budget {budget}."
489            ),
490        ))
491    }
492}
