53/// Evaluated bitmap state carried between blocking chunks.
54///
55/// The inverted-index evaluator yields `(bucket_id, RoaringBitmap)`. Holding the
56/// bitmap plus an absolute remaining range lets us resume a dense bucket without
57/// flattening it into an unbounded `Vec<u64>` or rereading it from RocksDB.
58#[derive(Clone)]
59pub(super) struct PendingBitmapBucket {
60    bucket_id: u64,
61    bitmap: RoaringBitmap,
62    remaining: Range<u64>,
63}
64
65impl PendingBitmapBucket {
66    fn new(
67        bucket_id: u64,
68        bitmap: RoaringBitmap,
69        range: &Range<u64>,
70        bucket_size: u64,
71    ) -> Option<Self> {
72        let bucket_start = bucket_id * bucket_size;
73        let bucket_end = bucket_start + bucket_size;
74        let remaining = range.start.max(bucket_start)..range.end.min(bucket_end);
75        (!remaining.is_empty() && !bitmap.is_empty()).then_some(Self {
76            bucket_id,
77            bitmap,
78            remaining,
79        })
80    }
81
82    fn drain_into(
83        &mut self,
84        out: &mut Vec<u64>,
85        hit_limit: usize,
86        bucket_size: u64,
87        direction: ScanDirection,
88    ) {
89        while out.len() < hit_limit {
90            let Some(seq) = self.next_seq(bucket_size, direction) else {
91                break;
92            };
93            out.push(seq);
94            self.remaining =
95                remaining_range_after(self.remaining.clone(), seq, direction.is_ascending())
96                    .unwrap_or(0..0);
97        }
98    }
99
100    fn has_hits(&self, bucket_size: u64, direction: ScanDirection) -> bool {
101        self.next_seq(bucket_size, direction).is_some()
102    }
103
104    fn next_seq(&self, bucket_size: u64, direction: ScanDirection) -> Option<u64> {
105        if self.remaining.is_empty() {
106            return None;
107        }
108
109        let bucket_start = self.bucket_id * bucket_size;
110        let lo = (self.remaining.start - bucket_start) as u32;
111        let hi = (self.remaining.end - bucket_start) as u32;
112        let bit = if direction.is_ascending() {
113            self.bitmap.range(lo..hi).next()
114        } else {
115            self.bitmap.range(lo..hi).next_back()
116        }?;
117        Some(bucket_start + u64::from(bit))
118    }
119
120    #[cfg(test)]
121    fn remaining(&self) -> Range<u64> {
122        self.remaining.clone()
123    }
124}

153pub(super) fn drain_bitmap_hits_with_budget(
154    service: RpcService,
155    kind: LedgerBitmapKind,
156    bucket_size: u64,
157    query: BitmapQuery,
158    pending_bucket: Option<PendingBitmapBucket>,
159    range: Option<Range<u64>>,
160    direction: ScanDirection,
161    hit_limit: usize,
162    scan_budget: usize,
163    cancel: &CancellationToken,
164) -> Result<DrainedBitmapHits, RpcError> {
165    if hit_limit == 0 {
166        return Ok(DrainedBitmapHits {
167            items: Vec::new(),
168            pending_bucket,
169            next_range: range,
170            buckets_scanned: 0,
171            coalesced_frontier: None,
172            scan_limit_hit: false,
173        });
174    }
175
176    // The iterator evaluator in `sui-inverted-index` is synchronous, so the
177    // borrowed RocksDB iterators stay on this blocking task for the whole chunk.
178    // That avoids a runtime dependency in this path while keeping BigTable on
179    // the async stream evaluator.
180    let scan_budget_start = scan_budget;
181    let scan_budget = BitmapScanBudget::new(scan_budget_start);
182    let source = RpcIndexesBitmapSource::new(
183        &service,
184        kind,
185        bucket_size,
186        scan_budget.clone(),
187        cancel.clone(),
188    );
189    let state = drain_watermarked_buckets(
190        move |scan_range| {
191            eval_bitmap_query_bucket_iter(source, query, scan_range, bucket_size, direction)
192        },
193        pending_bucket,
194        range,
195        bucket_size,
196        direction,
197        hit_limit,
198        cancel,
199    )?;
200    Ok(DrainedBitmapHits {
201        items: state.items,
202        pending_bucket: state.pending_bucket,
203        next_range: state.next_range,
204        buckets_scanned: scan_budget_start - scan_budget.remaining(),
205        coalesced_frontier: state.coalesced_frontier,
206        scan_limit_hit: state.scan_limit_hit,
207    })
208}

210/// Item type of the iterative bitmap evaluator: a matching bucket, a progress
211/// watermark, or a terminal [`BitmapScanError`].
212type WatermarkedBucketItem = BitmapScanResult<Watermarked<(u64, RoaringBitmap)>>;
213
214/// Loop state from draining a watermarked-bucket iterator, independent of budget
215/// accounting (the caller adds `buckets_scanned` from its `BitmapScanBudget`).
216struct DrainLoopState {
217    items: Vec<u64>,
218    pending_bucket: Option<PendingBitmapBucket>,
219    next_range: Option<Range<u64>>,
220    coalesced_frontier: Option<u64>,
221    scan_limit_hit: bool,
222}
223
224/// Drain matching member ids until `hit_limit` items, the iterator ends, or it
225/// signals [`BitmapScanError::ScanLimit`]. `open_iter` builds the iterator over
226/// a scan range and is invoked at most once, lazily — only when a fresh scan
227/// beyond `pending_bucket` is needed — so resuming a dense pending bucket never
228/// opens a RocksDB iterator. On scan-limit exhaustion the resume range is
229/// anchored past the coalesced frontier; the caller decides whether that is the
230/// per-request cap (terminal `ScanLimit`) or a per-chunk cap (resume in the next
231/// chunk).
232fn drain_watermarked_buckets<I>(
233    open_iter: impl FnOnce(Range<u64>) -> I,
234    mut pending_bucket: Option<PendingBitmapBucket>,
235    mut next_range: Option<Range<u64>>,
236    bucket_size: u64,
237    direction: ScanDirection,
238    hit_limit: usize,
239    cancel: &CancellationToken,
240) -> Result<DrainLoopState, RpcError>
241where
242    I: Iterator<Item = WatermarkedBucketItem>,
243{
244    let mut open_iter = Some(open_iter);
245    let mut iter: Option<I> = None;
246    let mut iter_range: Option<Range<u64>> = None;
247    let mut out = Vec::new();
248    // Furthest progress watermark seen this chunk. The evaluator emits watermarks
249    // monotonically in scan direction, so last-seen is furthest.
250    let mut coalesced_frontier: Option<u64> = None;
251    let mut scan_limit_hit = false;
252    while out.len() < hit_limit {
253        if cancel.is_cancelled() {
254            return Err(cancelled());
255        }
256        if let Some(mut bucket) = pending_bucket.take() {
257            bucket.drain_into(&mut out, hit_limit, bucket_size, direction);
258            if bucket.has_hits(bucket_size, direction) {
259                pending_bucket = Some(bucket);
260                break;
261            }
262            continue;
263        }
264
265        let Some(scan_range) = next_range.clone() else {
266            break;
267        };
268        if iter.is_none() {
269            let open = open_iter
270                .take()
271                .expect("bitmap query iterator is only opened once");
272            iter_range = Some(scan_range.clone());
273            iter = Some(open(scan_range));
274        }
275
276        let iter_range = iter_range
277            .as_ref()
278            .expect("bitmap iterator range is set before polling");
279        match iter
280            .as_mut()
281            .expect("bitmap iterator is set before polling")
282            .next()
283        {
284            Some(Ok(Watermarked::Item((bucket_id, bitmap)))) => {
285                next_range =
286                    remaining_range_after_bucket(iter_range, bucket_id, bucket_size, direction);
287                pending_bucket =
288                    PendingBitmapBucket::new(bucket_id, bitmap, iter_range, bucket_size);
289            }
290            // Progress watermark: advance the chunk frontier without consuming
291            // item budget. Lets the caller emit a scan watermark when the chunk
292            // matches nothing across a sparse gap.
293            Some(Ok(Watermarked::Watermark(pos))) => {
294                coalesced_frontier = Some(pos);
295            }
296            Some(Err(BitmapScanError::ScanLimit)) => {
297                // Scan-limit is a graceful stop, not an error: the evaluator
298                // emits the frontier watermark before the error, so
299                // `coalesced_frontier` already holds the resume point. Anchor the
300                // resume range past the frontier; the caller decides whether this
301                // is the per-request cap (terminal `ScanLimit`) or a per-chunk cap
302                // (resume here in the next chunk).
303                scan_limit_hit = true;
304                next_range = coalesced_frontier.and_then(|f| {
305                    remaining_range_after(iter_range.clone(), f, direction.is_ascending())
306                });
307                break;
308            }
309            // Cancelled stream → gRPC Cancelled status.
310            Some(Err(BitmapScanError::Cancelled)) => {
311                return Err(RpcError::new(
312                    tonic::Code::Cancelled,
313                    BitmapScanError::Cancelled.to_string(),
314                ));
315            }
316            // Genuine fault → gRPC Internal, carrying the error unchanged.
317            Some(Err(BitmapScanError::Source(inner))) => {
318                return Err(RpcError::new(tonic::Code::Internal, inner.to_string()));
319            }
320            None => {
321                next_range = None;
322                break;
323            }
324        }
325    }
326    Ok(DrainLoopState {
327        items: out,
328        pending_bucket,
329        next_range,
330        coalesced_frontier,
331        scan_limit_hit,
332    })
333}

335fn remaining_range_after_bucket(
336    range: &Range<u64>,
337    bucket_id: u64,
338    bucket_size: u64,
339    direction: ScanDirection,
340) -> Option<Range<u64>> {
341    let remaining = if direction.is_ascending() {
342        ((bucket_id + 1) * bucket_size).max(range.start)..range.end
343    } else {
344        range.start..(bucket_id * bucket_size).min(range.end)
345    };
346    (!remaining.is_empty()).then_some(remaining)
347}

371/// Inner bucket source: a stored RocksDB scan, or the synthesized dense
372/// tx-universe sequence — `IndexDimension::TxUniverse` is query-only and
373/// never reaches storage.
374enum BitmapBucketIter<'a> {
375    Stored(LedgerBitmapBucketIterator<'a>),
376    Universe(Box<dyn Iterator<Item = (u64, RoaringBitmap)> + Send + 'a>),
377}
378
379struct RpcIndexesBitmapIterator<'a> {
380    scan_budget: BitmapScanBudget,
381    cancel: CancellationToken,
382    iter: Option<BitmapBucketIter<'a>>,
383    finished: bool,
384    initial_error: Option<BitmapScanError>,
385    /// This leaf has not yet charged a bucket. Its first bucket is reserved
386    /// (always allowed) so every leaf emits its first watermark; see
387    /// [`BitmapScanBudget::take_first`].
388    first: bool,
389}
390
391impl<'a> RpcIndexesBitmapIterator<'a> {
392    fn new(
393        service: &'a RpcService,
394        kind: LedgerBitmapKind,
395        bucket_size: u64,
396        scan_budget: BitmapScanBudget,
397        cancel: CancellationToken,
398        dimension_key: Vec<u8>,
399        range: Range<u64>,
400        direction: ScanDirection,
401    ) -> Self {
402        if range.is_empty() {
403            return Self {
404                scan_budget,
405                cancel,
406                iter: None,
407                finished: true,
408                initial_error: None,
409                first: true,
410            };
411        }
412
413        // The tx universe is dense (every tx_seq in range is real), so it is
414        // synthesized rather than read from storage. Gated on the tx kind: the
415        // key is only ever produced by the tx filter layer, and in event-space
416        // a dense universe would be semantically wrong.
417        if kind == LedgerBitmapKind::Transaction
418            && dimension_key.first() == Some(&IndexDimension::TxUniverse.tag_byte())
419        {
420            return Self {
421                scan_budget,
422                cancel,
423                finished: false,
424                iter: Some(BitmapBucketIter::Universe(Box::new(
425                    dense_universe_buckets(range, bucket_size, direction),
426                ))),
427                initial_error: None,
428                first: true,
429            };
430        }
431
432        let start_bucket = range.start / bucket_size;
433        let end_bucket_exclusive = (range.end - 1) / bucket_size + 1;
434        let iter = service
435            .reader
436            .inner()
437            .indexes()
438            .ok_or_else(|| anyhow::anyhow!("rpc indexes are disabled"))
439            .and_then(|indexes| {
440                let descending = !direction.is_ascending();
441                let iter = match kind {
442                    LedgerBitmapKind::Transaction => indexes.transaction_bitmap_bucket_iter(
443                        dimension_key,
444                        start_bucket,
445                        end_bucket_exclusive,
446                        descending,
447                    ),
448                    LedgerBitmapKind::Event => indexes.event_bitmap_bucket_iter(
449                        dimension_key,
450                        start_bucket,
451                        end_bucket_exclusive,
452                        descending,
453                    ),
454                };
455                iter.map_err(|e| anyhow::anyhow!(e.to_string()))
456            });
457
458        let (iter, initial_error) = match iter {
459            Ok(iter) => (Some(BitmapBucketIter::Stored(iter)), None),
460            Err(e) => (None, Some(e.into())),
461        };
462        Self {
463            scan_budget,
464            cancel,
465            finished: false,
466            iter,
467            initial_error,
468            first: true,
469        }
470    }
471
472    fn read_next_bucket(&mut self) -> Option<BitmapScanResult<(u64, RoaringBitmap)>> {
473        if self.cancel.is_cancelled() {
474            self.finished = true;
475            return Some(Err(BitmapScanError::Cancelled));
476        }
477        let Some(iter) = self.iter.as_mut() else {
478            self.finished = true;
479            return None;
480        };
481
482        let next = match iter {
483            BitmapBucketIter::Stored(iter) => match iter.next() {
484                Some(Ok(bucket)) => Some(Ok((bucket.bucket_id, bucket.bitmap))),
485                Some(Err(e)) => Some(Err(BitmapScanError::Source(anyhow::anyhow!(e.to_string())))),
486                None => None,
487            },
488            BitmapBucketIter::Universe(iter) => iter.next().map(Ok),
489        };
490        match next {
491            Some(Ok(bucket)) => {
492                if self.first {
493                    // Reserved: a leaf's first bucket is always allowed so it
494                    // emits its first watermark even if sibling leaves already
495                    // drained the shared budget. Without it a starved leaf
496                    // never reports a watermark, its merge slot stays `None`,
497                    // and the scan ends with a cursorless `ScanLimit` — a
498                    // client livelock. See [`BitmapScanBudget::take_first`].
499                    self.scan_budget.take_first();
500                    self.first = false;
501                } else if let Err(e) = self.scan_budget.take_one() {
502                    self.finished = true;
503                    return Some(Err(e));
504                }
505                Some(Ok(bucket))
506            }
507            None => {
508                self.finished = true;
509                None
510            }
511            Some(Err(e)) => {
512                self.finished = true;
513                Some(Err(e))
514            }
515        }
516    }
517}

519impl Iterator for RpcIndexesBitmapIterator<'_> {
520    type Item = BitmapScanResult<(u64, RoaringBitmap)>;
521
522    fn next(&mut self) -> Option<Self::Item> {
523        if let Some(e) = self.initial_error.take() {
524            self.finished = true;
525            return Some(Err(e));
526        }
527        if self.finished {
528            return None;
529        }
530
531        self.read_next_bucket()
532    }
533}

535#[derive(Clone)]
536struct BitmapScanBudget {
537    remaining: Rc<Cell<usize>>,
538}
539
540impl BitmapScanBudget {
541    fn new(scan_budget: usize) -> Self {
542        Self {
543            remaining: Rc::new(Cell::new(scan_budget)),
544        }
545    }
546
547    fn take_one(&self) -> BitmapScanResult<()> {
548        let remaining = self.remaining.get();
549        if remaining == 0 {
550            return Err(BitmapScanError::ScanLimit);
551        }
552        self.remaining.set(remaining - 1);
553        Ok(())
554    }
555
556    /// Charge a leaf's mandatory first bucket: decrements when it can but
557    /// always succeeds, so every leaf is guaranteed to emit its first
558    /// watermark regardless of how sibling leaves drained the shared pool.
559    /// Saturating at 0 only undercounts a first bucket taken after the pool
560    /// was already exhausted; the common `budget >> leaves` case stays exact.
561    fn take_first(&self) {
562        self.remaining.set(self.remaining.get().saturating_sub(1));
563    }
564
565    fn remaining(&self) -> usize {
566        self.remaining.get()
567    }
568}
