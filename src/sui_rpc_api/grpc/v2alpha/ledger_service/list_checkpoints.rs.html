71pub(crate) async fn list_checkpoints(
72    service: RpcService,
73    request: ListCheckpointsRequest,
74) -> Result<ListCheckpointsStream, RpcError> {
75    ensure_ledger_history_enabled(&service)?;
76    let started = Instant::now();
77    let start_checkpoint = request.start_checkpoint;
78    let end_checkpoint = request.end_checkpoint;
79    let filter = request.filter;
80    let request_options = request.options;
81    let filtered = filter.is_some();
82    validate_checkpoint_bounds(start_checkpoint, end_checkpoint)?;
83    let read_mask = validate_read_mask(request.read_mask)?;
84    let ledger_history = service.config.ledger_history();
85    let endpoint = ledger_history.list_checkpoints();
86    let bitmap_bucket_scan_budget = ledger_history.bitmap_bucket_scan_budget();
87    let chunk_bucket_scan_budget = ledger_history.chunk_bucket_scan_budget();
88    let max_bitmap_filter_literals = ledger_history.max_bitmap_filter_literals();
89    let options = QueryOptions::from_proto(
90        request_options.as_ref(),
91        endpoint.default_limit_items,
92        endpoint.max_limit_items,
93        QueryType::Checkpoints,
94    )?;
95    let limit_items = options.limit_items;
96    let ordering = options.ordering;
97    let filter_query = filter
98        .as_ref()
99        .map(|filter| transaction_filter_to_query(filter, max_bitmap_filter_literals))
100        .transpose()?;
101
102    let initial_state = CheckpointScanState::Init {
103        start_checkpoint,
104        end_checkpoint,
105        filter_query,
106    };
107
108    let terminal_options = options.clone();
109    Ok(async_stream::try_stream! {
110        let mut scan = ChunkedScan::new(
111            initial_state,
112            limit_items,
113            endpoint.chunk_max,
114            bitmap_bucket_scan_budget,
115            move |state, args: ChunkArgs| {
116                spawn_checkpoint_chunk(
117                    service.clone(),
118                    state,
119                    read_mask.clone(),
120                    options.clone(),
121                    args.scan_budget,
122                    chunk_bucket_scan_budget,
123                    args.chunk_item_limit,
124                    args.remaining_request_item_limit,
125                    args.cancel,
126                )
127            },
128        );
129
130        while let Some(response) = scan.next_item().await? {
131            yield response;
132        }
133
134        let emitted = scan.produced();
135        let terminal = scan.into_terminal().expect("query emits terminal state");
136        let reason = query_end(emitted, limit_items, terminal.reason);
137        if reached_range_end(reason) {
138            yield watermark_response(terminal_boundary_watermark(
139                &terminal_options,
140                terminal.end_checkpoint,
141                terminal.end_position,
142            ));
143        }
144        yield end_response(reason);
145        info!(
146            filtered,
147            limit_items,
148            ?ordering,
149            emitted,
150            ?reason,
151            elapsed_ms = started.elapsed().as_millis(),
152            "list_checkpoints: done"
153        );
154    }
155    .boxed())
156}

158fn spawn_checkpoint_chunk(
159    service: RpcService,
160    state: CheckpointScanState,
161    read_mask: FieldMaskTree,
162    options: QueryOptions,
163    scan_budget: usize,
164    chunk_scan_budget: usize,
165    chunk_item_limit: usize,
166    remaining_request_item_limit: usize,
167    cancel: CancellationToken,
168) -> JoinHandle<Result<CheckpointChunkDone, RpcError>> {
169    tokio::task::spawn_blocking(move || {
170        next_checkpoint_chunk(
171            service,
172            state,
173            read_mask,
174            options,
175            scan_budget,
176            chunk_scan_budget,
177            chunk_item_limit,
178            remaining_request_item_limit,
179            &cancel,
180        )
181    })
182}

184#[derive(Clone)]
185enum CheckpointScanState {
186    Init {
187        start_checkpoint: Option<u64>,
188        end_checkpoint: Option<u64>,
189        filter_query: Option<BitmapQuery>,
190    },
191    Unfiltered {
192        range: Range<u64>,
193        end_reason: QueryEndReason,
194        end_checkpoint: u64,
195        end_position: u64,
196    },
197    Filtered {
198        query: BitmapQuery,
199        tx_range: Option<Range<u64>>,
200        pending_bucket: Option<PendingBitmapBucket>,
201        buffered_cp_seqs: VecDeque<u64>,
202        last_cp_seq: Option<u64>,
203        end_reason: QueryEndReason,
204        end_checkpoint: u64,
205        end_position: u64,
206    },
207}

209type CheckpointChunkDone = ScanChunkDone<CheckpointScanState, ListCheckpointsResponse>;

211fn next_checkpoint_chunk(
212    service: RpcService,
213    state: CheckpointScanState,
214    read_mask: FieldMaskTree,
215    options: QueryOptions,
216    scan_budget: usize,
217    chunk_scan_budget: usize,
218    chunk_item_limit: usize,
219    remaining_request_item_limit: usize,
220    cancel: &CancellationToken,
221) -> Result<CheckpointChunkDone, RpcError> {
222    match state {
223        CheckpointScanState::Init {
224            start_checkpoint,
225            end_checkpoint,
226            filter_query,
227        } => {
228            let checkpoint_range = CheckpointRange::from_request(
229                start_checkpoint,
230                end_checkpoint,
231                checkpoint_hi_exclusive(&service)?,
232            )?;
233            let cp_range = resolve_cp_range(checkpoint_range, &options);
234            if cancel.is_cancelled() {
235                return Err(cancelled());
236            }
237            let terminal = ChunkTerminal {
238                reason: cp_range.end_reason,
239                end_checkpoint: cp_range.end_checkpoint,
240                end_position: cp_range.end_position,
241            };
242            let range = cp_range.range;
243            if range.is_empty() {
244                return Ok(CheckpointChunkDone {
245                    items: Vec::new(),
246                    produced: 0,
247                    next_state: None,
248                    terminal,
249                    remaining_scan_budget: scan_budget,
250                });
251            }
252            let state = if let Some(query) = filter_query {
253                let mut tx_range = checkpoint_to_tx_range(&service, range)?;
254                if !tx_range.is_empty() {
255                    let explicit_lower = start_checkpoint.is_some() || options.has_after_cursor();
256                    let floor = lowest_available_tx_seq(&service)?;
257                    tx_range.start = apply_tx_seq_floor(tx_range.start, explicit_lower, floor)?;
258                }
259                if tx_range.is_empty() {
260                    return Ok(CheckpointChunkDone {
261                        items: Vec::new(),
262                        produced: 0,
263                        next_state: None,
264                        terminal,
265                        remaining_scan_budget: scan_budget,
266                    });
267                }
268                CheckpointScanState::Filtered {
269                    query,
270                    tx_range: Some(tx_range),
271                    pending_bucket: None,
272                    buffered_cp_seqs: VecDeque::new(),
273                    last_cp_seq: None,
274                    end_reason: terminal.reason,
275                    end_checkpoint: terminal.end_checkpoint,
276                    end_position: terminal.end_position,
277                }
278            } else {
279                CheckpointScanState::Unfiltered {
280                    range,
281                    end_reason: terminal.reason,
282                    end_checkpoint: terminal.end_checkpoint,
283                    end_position: terminal.end_position,
284                }
285            };
286            next_checkpoint_chunk(
287                service,
288                state,
289                read_mask,
290                options,
291                scan_budget,
292                chunk_scan_budget,
293                chunk_item_limit,
294                remaining_request_item_limit,
295                cancel,
296            )
297        }
298        CheckpointScanState::Unfiltered {
299            range,
300            end_reason,
301            end_checkpoint,
302            end_position,
303        } => next_unfiltered_checkpoint_chunk(
304            service,
305            range,
306            end_reason,
307            end_checkpoint,
308            end_position,
309            read_mask,
310            options,
311            scan_budget,
312            chunk_item_limit,
313            cancel,
314        ),
315        CheckpointScanState::Filtered {
316            query,
317            tx_range,
318            pending_bucket,
319            buffered_cp_seqs,
320            last_cp_seq,
321            end_reason,
322            end_checkpoint,
323            end_position,
324        } => next_filtered_checkpoint_chunk(
325            service,
326            query,
327            tx_range,
328            pending_bucket,
329            buffered_cp_seqs,
330            last_cp_seq,
331            end_reason,
332            end_checkpoint,
333            end_position,
334            read_mask,
335            options,
336            scan_budget,
337            chunk_scan_budget,
338            chunk_item_limit,
339            remaining_request_item_limit,
340            cancel,
341        ),
342    }
343}

345fn next_unfiltered_checkpoint_chunk(
346    service: RpcService,
347    range: Range<u64>,
348    end_reason: QueryEndReason,
349    end_checkpoint: u64,
350    end_position: u64,
351    read_mask: FieldMaskTree,
352    options: QueryOptions,
353    scan_budget: usize,
354    chunk_item_limit: usize,
355    cancel: &CancellationToken,
356) -> Result<CheckpointChunkDone, RpcError> {
357    let ascending = options.is_ascending();
358    let seqs = checkpoint_seqs_for_range(range.clone(), ascending, chunk_item_limit);
359    let next_state = seqs
360        .last()
361        .and_then(|cp_seq| remaining_range_after(range, *cp_seq, ascending))
362        .map(|range| CheckpointScanState::Unfiltered {
363            range,
364            end_reason,
365            end_checkpoint,
366            end_position,
367        });
368    if cancel.is_cancelled() {
369        return Err(cancelled());
370    }
371    let items = render_checkpoint_seqs(&service, seqs, &read_mask, &options, cancel)?;
372    let produced = items.len();
373    Ok(CheckpointChunkDone {
374        items,
375        produced,
376        next_state,
377        terminal: ChunkTerminal {
378            reason: end_reason,
379            end_checkpoint,
380            end_position,
381        },
382        remaining_scan_budget: scan_budget,
383    })
384}

386fn next_filtered_checkpoint_chunk(
387    service: RpcService,
388    query: BitmapQuery,
389    mut tx_range: Option<Range<u64>>,
390    mut pending_bucket: Option<PendingBitmapBucket>,
391    mut buffered_cp_seqs: VecDeque<u64>,
392    mut last_cp_seq: Option<u64>,
393    end_reason: QueryEndReason,
394    end_checkpoint: u64,
395    end_position: u64,
396    read_mask: FieldMaskTree,
397    options: QueryOptions,
398    scan_budget: usize,
399    chunk_scan_budget: usize,
400    chunk_item_limit: usize,
401    remaining_request_item_limit: usize,
402    cancel: &CancellationToken,
403) -> Result<CheckpointChunkDone, RpcError> {
404    let ascending = options.is_ascending();
405    let item_limit = chunk_item_limit.min(remaining_request_item_limit);
406    let mut remaining_scan_budget = scan_budget;
407    let mut cp_seqs = buffered_cp_seqs
408        .drain(..item_limit.min(buffered_cp_seqs.len()))
409        .collect::<Vec<_>>();
410    let mut scan_limited = false;
411    let mut frontier: Option<u64> = None;
412    // Per-chunk bucket cap, bounding this chunk's total scan across all drain
413    // iterations so a sparse scan yields incremental scan watermarks instead of
414    // one at the per-request limit. Stays <= remaining_scan_budget (both shrink
415    // by the same amount each iteration).
416    let mut chunk_scan_budget = remaining_scan_budget.min(chunk_scan_budget);
417
418    while cp_seqs.len() < item_limit
419        && (pending_bucket.is_some() || tx_range.is_some())
420        && chunk_scan_budget > 0
421    {
422        if cancel.is_cancelled() {
423            return Err(cancelled());
424        }
425        let tx_hit_limit =
426            remaining_request_item_limit.saturating_sub(cp_seqs.len() + buffered_cp_seqs.len());
427        if tx_hit_limit == 0 {
428            break;
429        }
430        let hits = drain_bitmap_hits_with_budget(
431            service.clone(),
432            LedgerBitmapKind::Transaction,
433            TX_BITMAP_BUCKET_SIZE,
434            query.clone(),
435            pending_bucket,
436            tx_range,
437            options.scan_direction(),
438            tx_hit_limit,
439            chunk_scan_budget,
440            cancel,
441        )?;
442        remaining_scan_budget -= hits.buckets_scanned;
443        chunk_scan_budget -= hits.buckets_scanned;
444        pending_bucket = hits.pending_bucket;
445        tx_range = hits.next_range;
446        if hits.scan_limit_hit {
447            scan_limited = true;
448            frontier = hits.coalesced_frontier;
449            break;
450        }
451        if hits.items.is_empty() {
452            break;
453        }
454
455        let tx_seqs = hits.items;
456        if cancel.is_cancelled() {
457            return Err(cancelled());
458        }
459        let rows = get_tx_seq_digest_multi(&service, &tx_seqs)?;
460        for row in rows {
461            let cp_seq = row.checkpoint_number;
462            if last_cp_seq == Some(cp_seq) {
463                continue;
464            }
465
466            last_cp_seq = Some(cp_seq);
467            if cp_seqs.len() < item_limit {
468                cp_seqs.push(cp_seq);
469            } else if cp_seqs.len() + buffered_cp_seqs.len() < remaining_request_item_limit {
470                buffered_cp_seqs.push_back(cp_seq);
471            } else {
472                break;
473            }
474        }
475    }
476
477    if cp_seqs.len() + buffered_cp_seqs.len() == remaining_request_item_limit {
478        pending_bucket = None;
479        tx_range = None;
480    }
481
482    // Only the per-request budget (or a scan-limit with no resume point) ends
483    // the query; a per-chunk cap resumes in the next chunk. The scan watermark
484    // below carries the resume point when this chunk surfaced no new checkpoints.
485    let request_exhausted = scan_limited
486        && (remaining_scan_budget == 0
487            || (tx_range.is_none() && pending_bucket.is_none() && buffered_cp_seqs.is_empty()));
488    let next_state = if request_exhausted {
489        None
490    } else {
491        (!buffered_cp_seqs.is_empty() || pending_bucket.is_some() || tx_range.is_some()).then_some(
492            CheckpointScanState::Filtered {
493                query,
494                tx_range,
495                pending_bucket,
496                buffered_cp_seqs,
497                last_cp_seq,
498                end_reason,
499                end_checkpoint,
500                end_position,
501            },
502        )
503    };
504    let reason = if request_exhausted {
505        QueryEndReason::ScanLimit
506    } else {
507        end_reason
508    };
509    let scan_watermark = scan_checkpoint_watermark(
510        &service,
511        &options,
512        scan_limited,
513        cp_seqs.is_empty(),
514        frontier,
515        ascending,
516    )?;
517
518    if cancel.is_cancelled() {
519        return Err(cancelled());
520    }
521    let mut items = render_checkpoint_seqs(&service, cp_seqs, &read_mask, &options, cancel)?;
522    let produced = items.len();
523    if let Some(watermark) = scan_watermark {
524        items.push(watermark);
525    }
526    Ok(CheckpointChunkDone {
527        items,
528        produced,
529        next_state,
530        terminal: ChunkTerminal {
531            reason,
532            end_checkpoint,
533            end_position,
534        },
535        remaining_scan_budget,
536    })
537}

539/// Scan watermark for a filtered checkpoint chunk that surfaced no new
540/// checkpoints before the scan budget ran out. The filter scans the transaction
541/// bitmap, so the frontier is a `tx_sequence_number`; resolve it to its
542/// checkpoint and emit a checkpoint-space watermark there.
543fn scan_checkpoint_watermark(
544    service: &RpcService,
545    options: &QueryOptions,
546    scan_limited: bool,
547    no_items: bool,
548    frontier: Option<u64>,
549    ascending: bool,
550) -> Result<Option<ListCheckpointsResponse>, RpcError> {
551    if !(scan_limited && no_items) {
552        return Ok(None);
553    }
554    let Some(frontier) = frontier else {
555        return Ok(None);
556    };
557    let Some(cp) = resolve_frontier_checkpoint(service, frontier, ascending, |p| p)? else {
558        return Ok(None);
559    };
560    // The frontier lands partway through checkpoint `cp`, so `cp` itself is not
561    // yet proven complete — exclude it from the boundary (`cp ∓ 1`). Contrast
562    // the item path, which feeds an emitted (hence complete) cp_seq straight
563    // into `advance_checkpoint_boundary`.
564    let boundary = advance_boundary_excluding_cp(None, cp, options);
565    // Checkpoint cursors live in checkpoint space: position == checkpoint.
566    let cursor_cp = boundary_cursor_cp(cp, options.scan_direction());
567    let watermark = boundary_watermark(options, cursor_cp, cp, boundary);
568    Ok(Some(watermark_response(watermark)))
569}

571fn checkpoint_seqs_for_range(
572    range: Range<u64>,
573    ascending: bool,
574    checkpoint_limit: usize,
575) -> Vec<u64> {
576    if ascending {
577        range.take(checkpoint_limit).collect()
578    } else {
579        range.rev().take(checkpoint_limit).collect()
580    }
581}

583fn render_checkpoint_seqs(
584    service: &RpcService,
585    seqs: Vec<u64>,
586    read_mask: &FieldMaskTree,
587    options: &QueryOptions,
588    cancel: &CancellationToken,
589) -> Result<Vec<ListCheckpointsResponse>, RpcError> {
590    let mut items = Vec::with_capacity(seqs.len());
591    // Per-chunk running boundary; monotonic across chunks because seqs are
592    // emitted in scan order.
593    let mut checkpoint_boundary: Option<u64> = None;
594    for cp_seq in seqs {
595        if cancel.is_cancelled() {
596            return Err(cancelled());
597        }
598        checkpoint_boundary = advance_checkpoint_boundary(checkpoint_boundary, cp_seq, options);
599        items.push(render_checkpoint_seq(
600            service,
601            cp_seq,
602            read_mask,
603            options,
604            checkpoint_boundary,
605        )?);
606    }
607    Ok(items)
608}
609
610fn render_checkpoint_seq(
611    service: &RpcService,
612    cp_seq: u64,
613    read_mask: &FieldMaskTree,
614    options: &QueryOptions,
615    checkpoint_boundary: Option<u64>,
616) -> Result<ListCheckpointsResponse, RpcError> {
617    let read_mask = read_mask.to_field_mask();
618    let mut request = GetCheckpointRequest::default();
619    request.checkpoint_id = Some(CheckpointId::SequenceNumber(cp_seq));
620    request.read_mask = Some(read_mask);
621    let checkpoint = get_checkpoint(service, request)?
622        .checkpoint
623        .ok_or_else(|| {
624            RpcError::new(
625                tonic::Code::Internal,
626                format!("get_checkpoint returned no checkpoint for {cp_seq}"),
627            )
628        })?;
629    let watermark = item_watermark(options, cp_seq, cp_seq, checkpoint_boundary);
630    Ok(response_for(watermark, checkpoint))
631}
632
633fn validate_read_mask(read_mask: Option<FieldMask>) -> Result<FieldMaskTree, RpcError> {
634    let read_mask = read_mask.unwrap_or_else(|| FieldMask::from_str(READ_MASK_DEFAULT));
635    read_mask.validate::<Checkpoint>().map_err(|path| {
636        FieldViolation::new("read_mask")
637            .with_description(format!("invalid read_mask path: {path}"))
638            .with_reason(ErrorReason::FieldInvalid)
639    })?;
640    Ok(FieldMaskTree::from(read_mask))
641}
642
643fn resolve_cp_range(checkpoint_range: CheckpointRange, options: &QueryOptions) -> ResolvedRange {
644    let cp_range = checkpoint_range.resolve(options);
645    let range = cp_range.range.clone();
646    options.apply_cursor_bounds(cp_range.with_range(range, options.ordering))
647}
648
649fn response_for(watermark: Watermark, message: Checkpoint) -> ListCheckpointsResponse {
650    let mut item = CheckpointItem::default();
651    item.watermark = Some(watermark);
652    item.checkpoint = Some(message);
653
654    let mut response = ListCheckpointsResponse::default();
655    response.response = Some(list_checkpoints_response::Response::Item(item));
656    response
657}
658
659fn watermark_response(watermark: Watermark) -> ListCheckpointsResponse {
660    let mut response = ListCheckpointsResponse::default();
661    response.response = Some(list_checkpoints_response::Response::Watermark(watermark));
662    response
663}

665fn end_response(reason: QueryEndReason) -> ListCheckpointsResponse {
666    let mut end = QueryEnd::default();
667    end.reason = reason as i32;
668
669    let mut response = ListCheckpointsResponse::default();
670    response.response = Some(list_checkpoints_response::Response::End(end));
671    response
672}
