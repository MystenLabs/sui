24#[tracing::instrument(skip(service))]
25pub(super) fn get_checkpoint_object_proof(
26    service: &RpcService,
27    request: GetCheckpointObjectProofRequest,
28) -> Result<GetCheckpointObjectProofResponse, RpcError> {
29    let object_id = parse_object_id(&request)?;
30    let checkpoint_seq = request.checkpoint.ok_or_else(|| {
31        RpcError::new(
32            tonic::Code::InvalidArgument,
33            "missing checkpoint".to_string(),
34        )
35    })?;
36    validate_checkpoint_bounds(service, checkpoint_seq)?;
37
38    let checkpoint_data = load_checkpoint(service, checkpoint_seq)?;
39    let ocs_leaves = build_ocs_leaves(&checkpoint_data, checkpoint_seq)?;
40
41    let checkpoint_summary = bcs::to_bytes(&checkpoint_data.summary)
42        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
43
44    let proof = match ocs_leaves.binary_search_by_key(&object_id, |r| r.0) {
45        Ok(leaf_index) => build_inclusion_proof(service, &ocs_leaves, leaf_index)?,
46        Err(_) => build_non_inclusion_proof(&ocs_leaves, object_id)?,
47    };
48
49    let mut response = GetCheckpointObjectProofResponse::default();
50    response.checkpoint_summary = Some(checkpoint_summary.into());
51    response.proof = Some(proof);
52    Ok(response)
53}

55fn parse_object_id(request: &GetCheckpointObjectProofRequest) -> Result<ObjectID, RpcError> {
56    let object_id_str = request.object_id.as_ref().ok_or_else(|| {
57        RpcError::new(
58            tonic::Code::InvalidArgument,
59            "missing object_id".to_string(),
60        )
61    })?;
62    if object_id_str.trim().is_empty() {
63        return Err(RpcError::new(
64            tonic::Code::InvalidArgument,
65            "object_id cannot be empty".to_string(),
66        ));
67    }
68    ObjectID::from_str(object_id_str).map_err(|e| {
69        RpcError::new(
70            tonic::Code::InvalidArgument,
71            format!("invalid object_id: {e}"),
72        )
73    })
74}

76fn validate_checkpoint_bounds(service: &RpcService, checkpoint_seq: u64) -> Result<(), RpcError> {
77    let reader = service.reader.inner();
78    let indexes = reader.indexes().ok_or_else(RpcError::not_found)?;
79    let highest_indexed = indexes
80        .get_highest_indexed_checkpoint_seq_number()
81        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?
82        .unwrap_or(0);
83    let lowest_available = reader
84        .get_lowest_available_checkpoint_objects()
85        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
86
87    if checkpoint_seq < lowest_available {
88        return Err(RpcError::new(
89            tonic::Code::NotFound,
90            format!(
91                "requested checkpoint {checkpoint_seq} has been pruned; lowest available checkpoint \
92                 is {lowest_available}",
93            ),
94        ));
95    }
96    if checkpoint_seq > highest_indexed {
97        return Err(RpcError::new(
98            tonic::Code::NotFound,
99            format!(
100                "requested checkpoint {checkpoint_seq} is not yet indexed; highest indexed \
101                 checkpoint is {highest_indexed}",
102            ),
103        ));
104    }
105    Ok(())
106}

108fn load_checkpoint(service: &RpcService, checkpoint_seq: u64) -> Result<FullCheckpoint, RpcError> {
109    let reader = service.reader.inner();
110    let checkpoint = reader
111        .get_checkpoint_by_sequence_number(checkpoint_seq)
112        .ok_or_else(|| {
113            RpcError::new(
114                tonic::Code::NotFound,
115                format!("checkpoint {checkpoint_seq} not found"),
116            )
117        })?;
118    let checkpoint_contents = reader
119        .get_checkpoint_contents_by_sequence_number(checkpoint_seq)
120        .ok_or_else(|| {
121            RpcError::new(
122                tonic::Code::NotFound,
123                format!("checkpoint contents for {checkpoint_seq} not found"),
124            )
125        })?;
126    reader
127        .get_checkpoint_data(checkpoint, checkpoint_contents)
128        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))
129}

131/// Build the OCS Merkle leaves for `checkpoint`. Leaves are
132/// `(ObjectID, SequenceNumber, ObjectDigest)` tuples from the checkpoint's
133/// modified-object set, sorted by ObjectID (the order
134/// [`CheckpointArtifacts::object_states`] already returns them in, since the
135/// underlying map is a `BTreeMap` keyed by `ObjectID`).
136fn build_ocs_leaves(
137    checkpoint: &FullCheckpoint,
138    checkpoint_seq: u64,
139) -> Result<Vec<ObjectRef>, RpcError> {
140    let effects_refs: Vec<&TransactionEffects> = checkpoint
141        .transactions
142        .iter()
143        .map(|tx| &tx.effects)
144        .collect();
145
146    // V1 effects predate the OCS commitment, so checkpoints that contain any
147    // V1 effects cannot be authenticated against a checkpoint-artifacts
148    // digest. Reject up front rather than producing a proof that wouldn't
149    // verify.
150    if effects_refs
151        .iter()
152        .any(|effects| matches!(effects, TransactionEffects::V1(_)))
153    {
154        return Err(RpcError::new(
155            tonic::Code::FailedPrecondition,
156            format!(
157                "object proofs are not supported for checkpoint {checkpoint_seq} because it \
158                 contains TransactionEffectsV1",
159            ),
160        ));
161    }
162
163    let artifacts = CheckpointArtifacts::from(effects_refs.as_slice());
164    let object_states = artifacts
165        .object_states()
166        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
167
168    Ok(object_states
169        .iter()
170        .map(|(id, (version, digest))| (*id, *version, *digest))
171        .collect())
172}

174fn build_inclusion_proof(
175    service: &RpcService,
176    ocs_leaves: &[ObjectRef],
177    leaf_index: usize,
178) -> Result<get_checkpoint_object_proof_response::Proof, RpcError> {
179    let object_ref = ocs_leaves[leaf_index];
180    let sdk_leaves = sdk_object_refs(ocs_leaves);
181    let tree = MerkleTree::build_from_unserialized(&sdk_leaves)
182        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
183    let merkle_proof = tree
184        .get_proof(leaf_index)
185        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
186    let tree_root = sui_sdk_types::Digest::new(tree.root().bytes());
187
188    let sdk_proof = SdkOcsInclusionProof {
189        merkle_proof,
190        leaf_index: leaf_index as u64,
191        tree_root,
192    };
193
194    let mut proto_proof: ProtoOcsInclusionProof = (&sdk_proof).into();
195    proto_proof.object_ref = Some(sdk_leaves[leaf_index].clone().into());
196
197    // Carry the live object's BCS-encoded `Object` next to the inclusion
198    // proof so verifiers don't need a separate round-trip. Deletions and
199    // wraps are signalled by the leaf digest sentinel and have no live
200    // object to return.
201    if object_ref.2.is_alive() {
202        let reader = service.reader.inner();
203        let object = reader
204            .get_object_by_key(&object_ref.0, object_ref.1)
205            .ok_or_else(|| {
206                RpcError::new(
207                    tonic::Code::NotFound,
208                    format!(
209                        "object {} not found at version {}",
210                        object_ref.0, object_ref.1
211                    ),
212                )
213            })?;
214        debug_assert_eq!(
215            object.compute_object_reference().2,
216            object_ref.2,
217            "object digest mismatch for object {} at version {}",
218            object_ref.0,
219            object_ref.1
220        );
221        let object_data = bcs::to_bytes(&object)
222            .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
223        proto_proof.object_data = Some(object_data.into());
224    }
225
226    Ok(get_checkpoint_object_proof_response::Proof::Inclusion(
227        proto_proof,
228    ))
229}

231fn build_non_inclusion_proof(
232    ocs_leaves: &[ObjectRef],
233    object_id: ObjectID,
234) -> Result<get_checkpoint_object_proof_response::Proof, RpcError> {
235    let sdk_leaves = sdk_object_refs(ocs_leaves);
236    let tree = MerkleTree::build_from_unserialized(&sdk_leaves)
237        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
238    // The target only needs the object id to be correct: the non-inclusion
239    // search keys on the leaf's BCS sort key, and (object_id, _, _) sorts
240    // exactly where any leaf with that id would sort regardless of the
241    // version/digest bytes that follow.
242    let target = SdkObjectReference::new(object_id.into(), 0, sui_sdk_types::Digest::new([0; 32]));
243    let non_inclusion = tree
244        .compute_non_inclusion_proof(&sdk_leaves, &target)
245        .map_err(|e| RpcError::new(tonic::Code::Internal, e.to_string()))?;
246    let tree_root = sui_sdk_types::Digest::new(tree.root().bytes());
247
248    let sdk_proof = SdkOcsNonInclusionProof {
249        non_inclusion_proof: non_inclusion,
250        tree_root,
251    };
252    let proto_proof: ProtoOcsNonInclusionProof = (&sdk_proof).into();
253    Ok(get_checkpoint_object_proof_response::Proof::NonInclusion(
254        proto_proof,
255    ))
256}
257
258fn sdk_object_refs(refs: &[ObjectRef]) -> Vec<SdkObjectReference> {
259    refs.iter()
260        .map(|(id, version, digest)| {
261            SdkObjectReference::new((*id).into(), version.value(), (*digest).into())
262        })
263        .collect()
