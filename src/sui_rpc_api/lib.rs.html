31#[derive(Clone)]
32pub struct ServerVersion {
33    pub bin: &'static str,
34    pub version: &'static str,
35}
36
37impl ServerVersion {
38    pub fn new(bin: &'static str, version: &'static str) -> Self {
39        Self { bin, version }
40    }
41}
42
43impl std::fmt::Display for ServerVersion {
44    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
45        f.write_str(self.bin)?;
46        f.write_str("/")?;
47        f.write_str(self.version)
48    }
49}

51#[derive(Clone)]
52pub struct RpcService {
53    reader: StateReader,
54    executor: Option<Arc<dyn TransactionExecutor>>,
55    subscription_service_handle: Option<SubscriptionServiceHandle>,
56    chain_id: sui_types::digests::ChainIdentifier,
57    server_version: Option<ServerVersion>,
58    metrics: Option<Arc<RpcMetrics>>,
59    config: Config,
60}
61
62impl RpcService {
63    pub fn new(reader: Arc<dyn RpcStateReader>) -> Self {
64        let chain_id = reader.get_chain_identifier().unwrap();
65        Self {
66            reader: StateReader::new(reader),
67            executor: None,
68            subscription_service_handle: None,
69            chain_id,
70            server_version: None,
71            metrics: None,
72            config: Config::default(),
73        }
74    }
75
76    pub fn with_server_version(&mut self, server_version: ServerVersion) -> &mut Self {
77        self.server_version = Some(server_version);
78        self
79    }
80
81    pub fn with_config(&mut self, config: Config) {
82        self.config = config;
83    }
84
85    pub fn with_executor(&mut self, executor: Arc<dyn TransactionExecutor + Send + Sync>) {
86        self.executor = Some(executor);
87    }
88
89    pub fn with_subscription_service(
90        &mut self,
91        subscription_service_handle: SubscriptionServiceHandle,
92    ) {
93        self.subscription_service_handle = Some(subscription_service_handle);
94    }
95
96    pub fn with_metrics(&mut self, metrics: RpcMetrics) {
97        self.metrics = Some(Arc::new(metrics));
98    }
99
100    pub fn chain_id(&self) -> sui_types::digests::ChainIdentifier {
101        self.chain_id
102    }

104    pub fn server_version(&self) -> Option<&ServerVersion> {

106    }

108    pub async fn into_router(self) -> axum::Router {
109        let metrics = self.metrics.clone();
110
111        let router = {
112            let ledger_service =
113                sui_rpc::proto::sui::rpc::v2::ledger_service_server::LedgerServiceServer::new(
114                    self.clone(),
115                )
116                .send_compressed(tonic::codec::CompressionEncoding::Zstd);
117            let transaction_execution_service = sui_rpc::proto::sui::rpc::v2::transaction_execution_service_server::TransactionExecutionServiceServer::new(self.clone())
118                .send_compressed(tonic::codec::CompressionEncoding::Zstd);
119            let state_service =
120                sui_rpc::proto::sui::rpc::v2::state_service_server::StateServiceServer::new(
121                    self.clone(),
122                )
123                .send_compressed(tonic::codec::CompressionEncoding::Zstd);
124            let signature_verification_service = sui_rpc::proto::sui::rpc::v2::signature_verification_service_server::SignatureVerificationServiceServer::new(self.clone())
125                .send_compressed(tonic::codec::CompressionEncoding::Zstd);
126            let move_package_service = sui_rpc::proto::sui::rpc::v2::move_package_service_server::MovePackageServiceServer::new(self.clone())
127                .send_compressed(tonic::codec::CompressionEncoding::Zstd);
128            let name_service =
129                sui_rpc::proto::sui::rpc::v2::name_service_server::NameServiceServer::new(
130                    self.clone(),
131                )
132                .send_compressed(tonic::codec::CompressionEncoding::Zstd);
133
134            let event_service_alpha =
135                crate::grpc::alpha::event_service_proto::event_service_server::EventServiceServer::new(
136                    self.clone(),
137                );
138
139            let (health_reporter, health_service) = tonic_health::server::health_reporter();
140
141            let reflection_v1 = tonic_reflection::server::Builder::configure()
142                .register_encoded_file_descriptor_set(
143                    crate::proto::google::protobuf::FILE_DESCRIPTOR_SET,
144                )
145                .register_encoded_file_descriptor_set(
146                    crate::proto::google::rpc::FILE_DESCRIPTOR_SET,
147                )
148                .register_encoded_file_descriptor_set(
149                    sui_rpc::proto::sui::rpc::v2::FILE_DESCRIPTOR_SET,
150                )
151                .register_encoded_file_descriptor_set(tonic_health::pb::FILE_DESCRIPTOR_SET)
152                .build_v1()
153                .unwrap();
154
155            let reflection_v1alpha = tonic_reflection::server::Builder::configure()
156                .register_encoded_file_descriptor_set(
157                    crate::proto::google::protobuf::FILE_DESCRIPTOR_SET,
158                )
159                .register_encoded_file_descriptor_set(
160                    crate::proto::google::rpc::FILE_DESCRIPTOR_SET,
161                )
162                .register_encoded_file_descriptor_set(
163                    sui_rpc::proto::sui::rpc::v2::FILE_DESCRIPTOR_SET,
164                )
165                .register_encoded_file_descriptor_set(tonic_health::pb::FILE_DESCRIPTOR_SET)
166                .build_v1alpha()
167                .unwrap();
168
169            fn service_name<S: tonic::server::NamedService>(_service: &S) -> &'static str {
170                S::NAME
171            }
172
173            for service_name in [
174                service_name(&ledger_service),
175                service_name(&transaction_execution_service),
176                service_name(&state_service),
177                service_name(&signature_verification_service),
178                service_name(&move_package_service),
179                service_name(&name_service),
180                service_name(&event_service_alpha),
181                service_name(&reflection_v1),
182                service_name(&reflection_v1alpha),
183            ] {
184                health_reporter
185                    .set_service_status(service_name, tonic_health::ServingStatus::Serving)
186                    .await;
187            }
188
189            let mut services = grpc::Services::new()
190                // V2
191                .add_service(ledger_service)
192                .add_service(transaction_execution_service)
193                .add_service(state_service)
194                .add_service(signature_verification_service)
195                .add_service(move_package_service)
196                .add_service(name_service)
197                // alpha
198                .add_service(event_service_alpha)
199                // Reflection
200                .add_service(reflection_v1)
201                .add_service(reflection_v1alpha);
202
203            if self.subscription_service_handle.is_some() {
204                let subscription_service =
205sui_rpc::proto::sui::rpc::v2::subscription_service_server::SubscriptionServiceServer::new(self.clone());
206                health_reporter
207                    .set_service_status(
208                        service_name(&subscription_service),
209                        tonic_health::ServingStatus::Serving,
210                    )
211                    .await;
212
213                services = services.add_service(subscription_service);
214            }
215
216            services.add_service(health_service).into_router()
217        };
218
219        let health_endpoint = axum::Router::new()
220            .route("/health", axum::routing::get(service::health::health))
221            .with_state(self.clone());
222
223        router
224            .merge(health_endpoint)
225            .layer(axum::middleware::map_response_with_state(
226                self,
227                response::append_info_headers,
228            ))
229            .pipe(|router| {
230                if let Some(metrics) = metrics {
231                    router.layer(CallbackLayer::new(
232                        metrics::RpcMetricsMakeCallbackHandler::new(metrics),
233                    ))
234                } else {
235                    router
236                }
237            })
238    }
239
240    pub async fn start_service(self, socket_address: std::net::SocketAddr) {
241        let listener = tokio::net::TcpListener::bind(socket_address).await.unwrap();
242        axum::serve(listener, self.into_router().await)
243            .await
244            .unwrap();
245    }
246}
