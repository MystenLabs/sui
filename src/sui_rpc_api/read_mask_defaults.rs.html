37/// Validate an optional request `read_mask` against message type `M`, falling
38/// back to `default` when the request omits one, and return it as a
39/// `FieldMaskTree`. Shared by every read-mask-bearing handler (get, list, and
40/// subscribe) so validation and the "no mask -> default" rule stay identical
41/// across them. `default` is an explicit argument because one message type can
42/// carry different per-endpoint defaults (e.g. `ExecutedTransaction` uses
43/// `TRANSACTION` for get/list but `EXECUTE_TRANSACTION` for execution).
44pub fn validate_read_mask<M: MessageFields>(
45    read_mask: Option<FieldMask>,
46    default: &str,
47) -> Result<FieldMaskTree, RpcError> {
48    let read_mask = read_mask.unwrap_or_else(|| FieldMask::from_str(default));
49    read_mask.validate::<M>().map_err(|path| {
50        FieldViolation::new("read_mask")
51            .with_description(format!("invalid read_mask path: {path}"))
52            .with_reason(ErrorReason::FieldInvalid)
53    })?;
54    Ok(FieldMaskTree::from(read_mask))
55}
