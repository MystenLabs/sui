88    #[tracing::instrument(skip(self))]
89    pub fn get_transaction(
90        &self,
91        digest: sui_sdk_types::Digest,
92    ) -> crate::Result<(
93        sui_sdk_types::SignedTransaction,
94        sui_types::effects::TransactionEffects,
95        Option<sui_types::effects::TransactionEvents>,
96    )> {
97        use sui_types::effects::TransactionEffectsAPI;
98
99        let transaction_digest = digest.into();
100
101        let transaction = (*self
102            .inner()
103            .get_transaction(&transaction_digest)
104            .ok_or(TransactionNotFoundError(digest))?)
105        .clone()
106        .into_inner();
107        let effects = self
108            .inner()
109            .get_transaction_effects(&transaction_digest)
110            .ok_or(TransactionNotFoundError(digest))?;
111        let events = if effects.events_digest().is_some() {
112            self.inner()
113                .get_events(effects.transaction_digest())
114                .ok_or(TransactionNotFoundError(digest))?
115                .pipe(Some)
116        } else {
117            None
118        };
119
120        Ok((transaction.try_into()?, effects, events))
121    }

135    #[tracing::instrument(skip(self))]
136    pub fn get_transaction_read(
137        &self,
138        digest: sui_sdk_types::Digest,
139    ) -> crate::Result<TransactionRead> {
140        let (
141            SignedTransaction {
142                transaction,
143                signatures,
144            },
145            effects,
146            events,
147        ) = self.get_transaction(digest)?;
148
149        let (checkpoint, balance_changes, object_types) =
150            if let Some(info) = self.get_transaction_info(&(digest.into())) {
151                (
152                    Some(info.checkpoint),
153                    Some(info.balance_changes),
154                    Some(info.object_types),
155                )
156            } else {
157                let checkpoint = self.inner().get_transaction_checkpoint(&(digest.into()));
158                (checkpoint, None, None)
159            };
160        let timestamp_ms = if let Some(checkpoint) = checkpoint {
161            self.inner()
162                .get_checkpoint_by_sequence_number(checkpoint)
163                .map(|checkpoint| checkpoint.timestamp_ms)
164        } else {
165            None
166        };
167
168        let unchanged_loaded_runtime_objects = self
169            .inner()
170            .get_unchanged_loaded_runtime_objects(&(digest.into()));
171
172        Ok(TransactionRead {
173            digest: transaction.digest(),
174            transaction,
175            signatures,
176            effects,
177            events,
178            checkpoint,
179            timestamp_ms,
180            balance_changes,
181            object_types,
182            unchanged_loaded_runtime_objects,
183        })
184    }

204    pub fn lookup_address_balance(
205        &self,
206        owner: sui_types::base_types::SuiAddress,
207        coin_type: move_core_types::language_storage::StructTag,
208    ) -> Option<u64> {
209        use sui_types::MoveTypeTagTraitGeneric;
210        use sui_types::SUI_ACCUMULATOR_ROOT_OBJECT_ID;
211        use sui_types::accumulator_root::AccumulatorKey;
212        use sui_types::dynamic_field::DynamicFieldKey;
213
214        let balance_type = sui_types::balance::Balance::type_tag(coin_type.into());
215
216        let key = AccumulatorKey { owner };
217        let key_type_tag = AccumulatorKey::get_type_tag(&[balance_type]);
218
219        DynamicFieldKey(SUI_ACCUMULATOR_ROOT_OBJECT_ID, key, key_type_tag)
220            .into_unbounded_id()
221            .unwrap()
222            .load_object(self.inner())
223            .and_then(|o| o.load_value::<u128>().ok())
224            .map(|balance| balance as u64)
225    }

242pub struct CheckpointTransactionsIter {
243    reader: StateReader,
244    direction: Direction,
245
246    next_cursor: Option<(CheckpointSequenceNumber, Option<usize>)>,
247    checkpoint: Option<(
248        sui_types::messages_checkpoint::CheckpointSummary,
249        sui_types::messages_checkpoint::CheckpointContents,
250    )>,
251}
252
253impl CheckpointTransactionsIter {
254    #[allow(unused)]
255    pub fn new(
256        reader: StateReader,
257        direction: Direction,
258        start: (CheckpointSequenceNumber, Option<usize>),
259    ) -> Self {
260        Self {
261            reader,
262            direction,
263            next_cursor: Some(start),
264            checkpoint: None,
265        }
266    }
267}
268
269impl Iterator for CheckpointTransactionsIter {
270    type Item = Result<(CursorInfo, sui_types::digests::TransactionDigest)>;
271
272    fn next(&mut self) -> Option<Self::Item> {
273        loop {
274            let (current_checkpoint, transaction_index) = self.next_cursor?;
275
276            let (checkpoint, contents) = if let Some(checkpoint) = &self.checkpoint {
277                if checkpoint.0.sequence_number != current_checkpoint {
278                    self.checkpoint = None;
279                    continue;
280                } else {
281                    checkpoint
282                }
283            } else {
284                let checkpoint = self
285                    .reader
286                    .inner()
287                    .get_checkpoint_by_sequence_number(current_checkpoint)?;
288                let contents = self
289                    .reader
290                    .inner()
291                    .get_checkpoint_contents_by_sequence_number(checkpoint.sequence_number)?;
292
293                self.checkpoint = Some((checkpoint.into_inner().into_data(), contents));
294                self.checkpoint.as_ref().unwrap()
295            };
296
297            let index = transaction_index
298                .map(|idx| idx.clamp(0, contents.size().saturating_sub(1)))
299                .unwrap_or_else(|| match self.direction {
300                    Direction::Ascending => 0,
301                    Direction::Descending => contents.size().saturating_sub(1),
302                });
303
304            self.next_cursor = {
305                let next_index = match self.direction {
306                    Direction::Ascending => {
307                        let next_index = index + 1;
308                        if next_index >= contents.size() {
309                            None
310                        } else {
311                            Some(next_index)
312                        }
313                    }
314                    Direction::Descending => index.checked_sub(1),
315                };
316
317                let next_checkpoint = if next_index.is_some() {
318                    Some(current_checkpoint)
319                } else {
320                    match self.direction {
321                        Direction::Ascending => current_checkpoint.checked_add(1),
322                        Direction::Descending => current_checkpoint.checked_sub(1),
323                    }
324                };
325
326                next_checkpoint.map(|checkpoint| (checkpoint, next_index))
327            };
328
329            if contents.size() == 0 {
330                continue;
331            }
332
333            let digest = contents.inner()[index].transaction;
334
335            let cursor_info = CursorInfo {
336                checkpoint: checkpoint.sequence_number,
337                timestamp_ms: checkpoint.timestamp_ms,
338                index: index as u64,
339                next_cursor: self.next_cursor,
340            };
341
342            return Some(Ok((cursor_info, digest)));
343        }
344    }
345}

347#[allow(unused)]
348pub struct CursorInfo {
349    pub checkpoint: CheckpointSequenceNumber,
350    pub timestamp_ms: u64,
351    #[allow(unused)]

358pub struct CheckpointIter {
359    reader: StateReader,
360    direction: Direction,
361
362    next_cursor: Option<CheckpointSequenceNumber>,
363}
364
365impl CheckpointIter {
366    #[allow(unused)]
367    pub fn new(reader: StateReader, direction: Direction, start: CheckpointSequenceNumber) -> Self {
368        Self {
369            reader,
370            direction,
371            next_cursor: Some(start),
372        }
373    }
374}
375
376impl Iterator for CheckpointIter {
377    type Item = Result<(
378        sui_types::messages_checkpoint::CertifiedCheckpointSummary,
379        sui_types::messages_checkpoint::CheckpointContents,
380    )>;
381
382    fn next(&mut self) -> Option<Self::Item> {
383        let current_checkpoint = self.next_cursor?;
384
385        let checkpoint = self
386            .reader
387            .inner()
388            .get_checkpoint_by_sequence_number(current_checkpoint)?
389            .into_inner();
390        let contents = self
391            .reader
392            .inner()
393            .get_checkpoint_contents_by_sequence_number(checkpoint.sequence_number)?;
394
395        self.next_cursor = match self.direction {
396            Direction::Ascending => current_checkpoint.checked_add(1),
397            Direction::Descending => current_checkpoint.checked_sub(1),
398        };
399
400        Some(Ok((checkpoint, contents)))
401    }
402}

404#[derive(Debug)]
405pub struct TransactionNotFoundError(pub sui_sdk_types::Digest);
406
407impl std::fmt::Display for TransactionNotFoundError {
408    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
409        write!(f, "Transaction {} not found", self.0)
410    }
411}
412
413impl std::error::Error for TransactionNotFoundError {}
414
415impl From<TransactionNotFoundError> for crate::RpcError {
416    fn from(value: TransactionNotFoundError) -> Self {
417        Self::new(tonic::Code::NotFound, value.to_string())
418    }
