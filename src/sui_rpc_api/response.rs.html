4use axum::{
5    extract::State,
6    http::HeaderMap,
7    response::{IntoResponse, Response},
8};
9
10use crate::RpcService;
11use sui_rpc::headers::{
12    X_SUI_CHAIN, X_SUI_CHAIN_ID, X_SUI_CHECKPOINT_HEIGHT, X_SUI_EPOCH,
13    X_SUI_LOWEST_AVAILABLE_CHECKPOINT, X_SUI_LOWEST_AVAILABLE_CHECKPOINT_OBJECTS, X_SUI_TIMESTAMP,
14    X_SUI_TIMESTAMP_MS,
15};
16
17pub async fn append_info_headers(
18    State(state): State<RpcService>,
19    response: Response,
20) -> impl IntoResponse {
21    let mut headers = HeaderMap::new();
22
23    if let Ok(chain_id) = state.chain_id().to_string().try_into() {
24        headers.insert(X_SUI_CHAIN_ID, chain_id);
25    }
26
27    if let Ok(chain) = state.chain_id().chain().as_str().try_into() {
28        headers.insert(X_SUI_CHAIN, chain);
29    }
30
31    if let Ok(latest_checkpoint) = state.reader.inner().get_latest_checkpoint() {
32        headers.insert(X_SUI_EPOCH, latest_checkpoint.epoch().into());
33        headers.insert(
34            X_SUI_CHECKPOINT_HEIGHT,
35            latest_checkpoint.sequence_number.into(),
36        );
37        headers.insert(X_SUI_TIMESTAMP_MS, latest_checkpoint.timestamp_ms.into());
38
39        headers.insert(
40            X_SUI_TIMESTAMP,
41            crate::proto::timestamp_ms_to_proto(latest_checkpoint.timestamp_ms)
42                .to_string()
43                .try_into()
44                .expect("timestamp is a valid HeaderValue"),
45        );
46    }
47
48    if let Ok(lowest_available_checkpoint) = state.reader.get_lowest_available_checkpoint() {
49        headers.insert(
50            X_SUI_LOWEST_AVAILABLE_CHECKPOINT,
51            lowest_available_checkpoint.into(),
52        );
53
54        // Still set the deprecated header
55        headers.insert(
56            X_SUI_LOWEST_AVAILABLE_CHECKPOINT_OBJECTS,
57            lowest_available_checkpoint.into(),
58        );
59    }
60
61    if let Some(server_version) = state
62        .server_version()
63        .and_then(|version| version.to_string().try_into().ok())
64    {
65        headers.insert(axum::http::header::SERVER, server_version);
66    }
67
68    (headers, response)
69}
