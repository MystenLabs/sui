17const CHECKPOINT_MAILBOX_SIZE: usize = 1024;
18const MAILBOX_SIZE: usize = 128;
19const SUBSCRIPTION_CHANNEL_SIZE: usize = 256;
20const MAX_SUBSCRIBERS: usize = 1024;
21
22/// Poll interval while waiting for the index to catch up to a checkpoint
23/// before delivering it (see [`IndexedCheckpointFn`]).
24const INDEX_WAIT_POLL_INTERVAL: Duration = Duration::from_millis(10);
25/// Upper bound on how long delivery of a single checkpoint waits for the
26/// index. A healthy index catches up in milliseconds; the bound just keeps
27/// a stalled indexer from wedging the subscription stream forever.
28const INDEX_WAIT_TIMEOUT: Duration = Duration::from_secs(10);
29
30/// Reads the highest checkpoint the index has committed (the embedded
31/// rpc-store's live-cohort watermark), or `None` if it has indexed nothing
32/// yet. When supplied to [`SubscriptionService::build`], the service holds a
33/// checkpoint back until the index has committed it, so a client that
34/// observes a checkpoint (e.g. via `execute_transaction_and_wait_for_checkpoint`)
35/// can immediately read that checkpoint's indexed state.
36pub type IndexedCheckpointFn = Arc<dyn Fn() -> Option<u64> + Send + Sync>;
37
38struct SubscriptionRequest {
39    sender: oneshot::Sender<mpsc::Receiver<Arc<Checkpoint>>>,
40}
41
42#[derive(Clone)]
43pub struct SubscriptionServiceHandle {
44    sender: mpsc::Sender<SubscriptionRequest>,
45}
46
47impl SubscriptionServiceHandle {
48    pub async fn register_subscription(&self) -> Option<mpsc::Receiver<Arc<Checkpoint>>> {
49        let (sender, receiver) = oneshot::channel();
50        let request = SubscriptionRequest { sender };
51        self.sender.send(request).await.ok()?;
52
53        receiver.await.ok()
54    }
55}

57pub struct SubscriptionService {
58    // Broadcast receiver for `Checkpoint`s published by the Checkpoint Executor.
59    //
60    // The executor publishes non-blocking, so a slow service can fall behind
61    // and observe `RecvError::Lagged`; checkpoints delivered between lags arrive
62    // in-order.
63    checkpoint_mailbox: broadcast::Receiver<Arc<Checkpoint>>,
64    mailbox: mpsc::Receiver<SubscriptionRequest>,
65    subscribers: Vec<mpsc::Sender<Arc<Checkpoint>>>,
66
67    // When set, delivery of a checkpoint waits until the index has committed
68    // it (see [`IndexedCheckpointFn`]). `None` preserves the immediate-delivery
69    // behavior used with the legacy synchronously-committed index.
70    indexed_checkpoint: Option<IndexedCheckpointFn>,
71
72    metrics: SubscriptionMetrics,
73}
74
75impl SubscriptionService {
76    pub fn build(
77        registry: &prometheus::Registry,
78        indexed_checkpoint: Option<IndexedCheckpointFn>,
79    ) -> (
80        broadcast::Sender<Arc<Checkpoint>>,
81        SubscriptionServiceHandle,
82    ) {
83        let metrics = SubscriptionMetrics::new(registry);
84        let (checkpoint_sender, checkpoint_mailbox) = broadcast::channel(CHECKPOINT_MAILBOX_SIZE);
85        let (subscription_request_sender, mailbox) = mpsc::channel(MAILBOX_SIZE);
86
87        tokio::spawn(
88            Self {
89                checkpoint_mailbox,
90                mailbox,
91                subscribers: Vec::new(),
92                indexed_checkpoint,
93                metrics,
94            }
95            .start(),
96        );
97
98        (
99            checkpoint_sender,
100            SubscriptionServiceHandle {
101                sender: subscription_request_sender,
102            },
103        )
104    }
105
106    async fn start(mut self) {
107        // Start main loop.
108        loop {
109            tokio::select! {
110                result = self.checkpoint_mailbox.recv() => {
111                    match result {
112                        Ok(checkpoint) => self.handle_checkpoint(checkpoint).await,
113                        Err(broadcast::error::RecvError::Lagged(skipped)) => {
114                            self.handle_lag(skipped);
115                        }
116                        // Once the executor drops the sender this yields `Closed`
117                        // and we can terminate the event loop.
118                        Err(broadcast::error::RecvError::Closed) => break,
119                    }
120                },
121                maybe_message = self.mailbox.recv() => {
122                    // Once all handles to our mailbox have been dropped this
123                    // will yield `None` and we can terminate the event loop
124                    if let Some(message) = maybe_message {
125                        self.handle_message(message);
126                    } else {
127                        break;
128                    }
129                },
130            }
131        }
132
133        info!("RPC Subscription Services ended");
134    }
135
136    async fn handle_checkpoint(&mut self, checkpoint: Arc<Checkpoint>) {
137        // Check that we recieved checkpoints in-order. The broadcast stream
138        // preserves send order, and any gap surfaces separately as `Lagged`
139        // (see `handle_lag`), so reaching here out-of-order indicates an
140        // executor bug.
141        {
142            let last_sequence_number = self.metrics.last_recieved_checkpoint.get();
143            let sequence_number = *checkpoint.summary.sequence_number() as i64;
144
145            if last_sequence_number != 0 && (last_sequence_number + 1) != sequence_number {
146                panic!(
147                    "recieved checkpoint out-of-order. expected checkpoint {}, recieved {}",
148                    last_sequence_number + 1,
149                    sequence_number
150                );
151            }
152
153            // Update the metric marking the latest checkpoint we've seen
154            self.metrics.last_recieved_checkpoint.set(sequence_number);
155        }
156
157        // Hold the checkpoint back until the index has committed it, so a
158        // client that observes this checkpoint can immediately read its
159        // indexed state. No-op unless an index gate was configured.
160        self.wait_until_indexed(*checkpoint.summary.sequence_number())
161            .await;
162
163        // Try to send the latest checkpoint to all subscribers. If a subscriber's channel is full
164        // then they are likely too slow so we drop them.
165        self.subscribers.retain(|subscriber| {
166            match subscriber.try_send(Arc::clone(&checkpoint)) {
167                Ok(()) => {
168                    trace!("successfully enqueued checkpont for subscriber");
169                    true // Retain this subscriber
170                }
171                Err(e) => {
172                    // It does not matter what the error is - channel full or closed, we drop the subscriber.
173                    trace!("unable to enqueue checkpoint for subscriber: {e}");
174                    self.metrics.inflight_subscribers.dec();
175                    false // Drop this subscriber
176                }
177            }
178        });
179    }
180
181    /// Block until the index has committed `sequence_number`, polling the
182    /// configured [`IndexedCheckpointFn`]. Returns immediately when no gate is
183    /// configured or the index is already caught up. Gives up after
184    /// [`INDEX_WAIT_TIMEOUT`] -- a stalled indexer should not wedge delivery
185    /// forever -- delivering a (possibly not-yet-indexed) checkpoint rather
186    /// than stalling the stream.
187    async fn wait_until_indexed(&self, sequence_number: u64) {
188        let Some(indexed) = &self.indexed_checkpoint else {
189            return;
190        };
191
192        if indexed().is_some_and(|hi| hi >= sequence_number) {
193            return;
194        }
195
196        let deadline = Instant::now() + INDEX_WAIT_TIMEOUT;
197        loop {
198            sleep(INDEX_WAIT_POLL_INTERVAL).await;
199            if indexed().is_some_and(|hi| hi >= sequence_number) {
200                return;
201            }
202            if Instant::now() >= deadline {
203                warn!(
204                    checkpoint = sequence_number,
205                    "index did not catch up within {INDEX_WAIT_TIMEOUT:?}; \
206                     delivering checkpoint anyway"
207                );
208                return;
209            }
210        }
211    }
212
213    /// Drop every in-progress subscription after the service fell behind the
214    /// broadcast stream. Having missed `skipped` checkpoints we can no longer
215    /// deliver an in-order, gap-free stream to any subscriber, and clients
216    /// already tolerate connection breaks and reconnect, so tearing them all
217    /// down is cheaper than trying to resynchronize each one.
218    fn handle_lag(&mut self, skipped: u64) {
219        warn!(
220            skipped,
221            "subscription service lagged behind the checkpoint stream; \
222             dropping all in-progress subscriptions"
223        );
224        let dropped = self.subscribers.len() as i64;
225        self.subscribers.clear();
226        self.metrics.inflight_subscribers.sub(dropped);
227        // The next delivered checkpoint jumps ahead by `skipped`; reset the
228        // in-order tracker so it is not mistaken for an out-of-order delivery.
229        self.metrics.last_recieved_checkpoint.set(0);
230    }
231
232    fn handle_message(&mut self, request: SubscriptionRequest) {
233        // Check if we've reached the limit to the number of subscribers we can have at one time.
234        if self.subscribers.len() >= MAX_SUBSCRIBERS {
235            trace!(
236                "failed to register new subscriber: hit maximum number of subscribers {}",
237                MAX_SUBSCRIBERS
238            );
239            return;
240        }
241
242        let (sender, reciever) = mpsc::channel(SUBSCRIPTION_CHANNEL_SIZE);
243        match request.sender.send(reciever) {
244            Ok(()) => {
245                trace!("successfully registered new subscriber");
246                self.metrics.inflight_subscribers.inc();
247                self.subscribers.push(sender);
248            }
249            Err(e) => {
250                trace!("failed to register new subscriber: {e:?}");
251            }
252        }
253    }
254}

257mod tests {
258    use std::sync::atomic::AtomicU64;
259    use std::sync::atomic::Ordering;

261    use sui_types::test_checkpoint_data_builder::TestCheckpointBuilder;

265    fn test_service() -> SubscriptionService {
266        test_service_with_index(None)

269    fn test_service_with_index(
270        indexed_checkpoint: Option<IndexedCheckpointFn>,
271    ) -> SubscriptionService {
272        let (_checkpoint_sender, checkpoint_mailbox) = broadcast::channel(16);
273        let (_request_sender, mailbox) = mpsc::channel(16);
274        SubscriptionService {
275            checkpoint_mailbox,
276            mailbox,
277            subscribers: Vec::new(),
278            indexed_checkpoint,
279            metrics: SubscriptionMetrics::new(&prometheus::Registry::new()),
280        }
281    }
282
283    fn checkpoint(sequence_number: u64) -> Arc<Checkpoint> {
284        Arc::new(TestCheckpointBuilder::new(sequence_number).build_checkpoint())
285    }
286
287    /// Register a subscriber the same way `handle_message` would, returning the
288    /// receiving half a client would hold.
289    fn add_subscriber(service: &mut SubscriptionService) -> mpsc::Receiver<Arc<Checkpoint>> {
290        let (sender, receiver) = mpsc::channel(SUBSCRIPTION_CHANNEL_SIZE);
291        service.metrics.inflight_subscribers.inc();
292        service.subscribers.push(sender);
293        receiver
294    }
295
296    #[tokio::test]
297    async fn handle_checkpoint_fans_out_in_order() {
298        let mut service = test_service();
299        let mut receiver = add_subscriber(&mut service);
300
301        service.handle_checkpoint(checkpoint(1)).await;
302        service.handle_checkpoint(checkpoint(2)).await;
303
304        assert_eq!(*receiver.recv().await.unwrap().summary.sequence_number(), 1);
305        assert_eq!(*receiver.recv().await.unwrap().summary.sequence_number(), 2);
306        assert_eq!(service.subscribers.len(), 1);
307    }
308
309    #[tokio::test]
310    async fn handle_checkpoint_drops_departed_subscriber() {
311        let mut service = test_service();
312        let receiver = add_subscriber(&mut service);
313        drop(receiver);
314
315        service.handle_checkpoint(checkpoint(1)).await;
316
317        assert!(service.subscribers.is_empty());
318        assert_eq!(service.metrics.inflight_subscribers.get(), 0);
319    }

321    #[tokio::test]
322    async fn handle_checkpoint_waits_for_index_before_delivering() {
323        // The index reports it has committed through checkpoint 4; checkpoint 5
324        // is not yet indexed.
325        let indexed = Arc::new(AtomicU64::new(4));
326        let gate = indexed.clone();
327        let mut service =
328            test_service_with_index(Some(Arc::new(move || Some(gate.load(Ordering::SeqCst)))));
329        let mut receiver = add_subscriber(&mut service);
330
331        // Delivery of checkpoint 5 blocks until the index catches up to it.
332        let mut deliver = std::pin::pin!(service.handle_checkpoint(checkpoint(5)));
333        assert!(
334            futures::poll!(&mut deliver).is_pending(),
335            "delivery should block while checkpoint 5 is unindexed"
336        );
337        assert!(receiver.try_recv().is_err());
338
339        // Once the index reaches 5, delivery completes.
340        indexed.store(5, Ordering::SeqCst);
341        deliver.await;
342        assert_eq!(*receiver.recv().await.unwrap().summary.sequence_number(), 5);
343    }
344
345    #[tokio::test]
346    async fn handle_lag_drops_all_subscribers_and_resets_tracker() {
347        let mut service = test_service();
348        let mut receiver_1 = add_subscriber(&mut service);
349        let mut receiver_2 = add_subscriber(&mut service);
350
351        service.handle_checkpoint(checkpoint(5)).await;
352        assert_eq!(service.metrics.last_recieved_checkpoint.get(), 5);
353
354        service.handle_lag(10);
355
356        assert!(service.subscribers.is_empty());
357        assert_eq!(service.metrics.inflight_subscribers.get(), 0);
358        // Both subscriptions are torn down, so the client streams close.
359        assert!(receiver_1.recv().await.is_some()); // checkpoint 5, then closed
360        assert!(receiver_1.recv().await.is_none());
361        assert!(receiver_2.recv().await.is_some());
362        assert!(receiver_2.recv().await.is_none());
363        // The tracker is reset so the next, jumped-ahead checkpoint is not
364        // mistaken for an out-of-order delivery (which would panic).
365        assert_eq!(service.metrics.last_recieved_checkpoint.get(), 0);
366
367        let mut receiver_3 = add_subscriber(&mut service);
368        service.handle_checkpoint(checkpoint(100)).await;
369        assert_eq!(
370            *receiver_3.recv().await.unwrap().summary.sequence_number(),
371            100
372        );
373    }
