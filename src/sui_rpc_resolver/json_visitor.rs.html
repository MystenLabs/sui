4//! A visitor implementation that constructs JSON values from BCS bytes.
5//!
6//! This visitor traverses BCS-encoded Move data and builds a `serde_json::Value`
7//! representation. Note that this approach loads the entire JSON structure into
8//! memory, which may have significant memory implications for large objects or
9//! collections. It should not be used in memory-constrained contexts like RPC
10//! handlers where the size of the data is unbounded.
11
12use move_core_types::annotated_value::MoveStruct;
13use move_core_types::annotated_value::MoveTypeLayout;
14use move_core_types::annotated_value::MoveValue;
15use move_core_types::annotated_visitor as AV;
16use move_core_types::language_storage::TypeTag;
17use serde_json::Map;
18use serde_json::Value;
19use sui_package_resolver::PackageStore;
20use sui_package_resolver::Resolver;
21use sui_package_resolver::error::Error as ResolverError;
22use sui_types::event::Event;
23use sui_types::object::option_visitor as OV;
24use sui_types::object::rpc_visitor as RV;

26/// Error type for JSON visitor operations
27#[derive(thiserror::Error, Debug)]
28pub enum Error {
29    #[error("Unexpected type")]
30    UnexpectedType,
31
32    #[error(transparent)]
33    Visitor(#[from] AV::Error),
34}
35
36/// Error type for deserialization operations that involve both type resolution and BCS deserialization.
37#[derive(thiserror::Error, Debug)]
38pub enum DeserializationError {
39    /// Failed to fetch type layout from the package resolver.
40    #[error("Failed to fetch type layout: {0}")]
41    LayoutFetch(#[from] ResolverError),
42
43    /// Failed to deserialize BCS data to JSON.
44    #[error("Failed to deserialize BCS data: {0}")]
45    Deserialization(#[from] anyhow::Error),
46}
47
48/// A visitor that constructs JSON values from BCS bytes.
49///
50/// Number representation:
51/// - u8, u16, u32 are represented as JSON numbers
52/// - u64, u128, u256 are represented as strings to avoid precision loss
53///
54/// Special types:
55/// - Addresses use full 64-character hex format with "0x" prefix
56/// - Byte vectors (`Vec<u8>`) are Base64-encoded strings
57pub struct JsonVisitor;
58
59struct JsonWriter;
60
61impl JsonVisitor {
62    /// Deserialize BCS bytes as JSON using the provided type layout.
63    pub fn deserialize_value(bytes: &[u8], layout: &MoveTypeLayout) -> anyhow::Result<Value> {
64        let mut visitor = RV::RpcVisitor::new(JsonWriter);
65        Ok(MoveValue::visit_deserialize(bytes, layout, &mut visitor)?)
66    }
67
68    /// Deserialize BCS bytes as a JSON object representing a struct.
69    pub fn deserialize_struct(
70        bytes: &[u8],
71        layout: &move_core_types::annotated_value::MoveStructLayout,
72    ) -> anyhow::Result<Value> {
73        let mut visitor = RV::RpcVisitor::new(JsonWriter);
74        Ok(MoveStruct::visit_deserialize(bytes, layout, &mut visitor)?)
75    }
76
77    /// Deserialize a single event to JSON using type resolution.
78    ///
79    /// This function:
80    /// 1. Resolves the type layout for the event's type
81    /// 2. Deserializes the BCS-encoded event contents to JSON
82    ///
83    /// If you need to deserialize multiple events, use
84    /// [`deserialize_events`](Self::deserialize_events) instead, which processes
85    /// events concurrently for better performance.
86    pub async fn deserialize_event<S>(
87        event: &Event,
88        resolver: &Resolver<S>,
89    ) -> Result<Value, DeserializationError>
90    where
91        S: PackageStore,
92    {
93        let type_tag = TypeTag::Struct(Box::new(event.type_.clone()));
94        let layout = resolver.type_layout(type_tag).await?;
95        Ok(Self::deserialize_value(&event.contents, &layout)?)
96    }
97
98    /// Deserialize multiple events to JSON concurrently.
99    ///
100    /// This function processes all events in parallel for better performance.
101    ///
102    /// If multiple events are from the same package, use
103    /// a `Resolver` with a cached `PackageStore` (e.g., `RpcPackageStore::with_cache()`)
104    /// to avoid fetching the same package multiple times.
105    pub async fn deserialize_events<S>(
106        events: &[Event],
107        resolver: &Resolver<S>,
108    ) -> Result<Vec<Value>, DeserializationError>
109    where
110        S: PackageStore,
111    {
112        use futures::future::try_join_all;
113
114        let futures = events
115            .iter()
116            .map(|event| Self::deserialize_event(event, resolver));
117        try_join_all(futures).await
118    }
119}
120
121impl RV::Writer for JsonWriter {
122    type Value = Value;
123    type Error = Error;
124
125    type Vec = Vec<Value>;
126    type Map = Map<String, Value>;
127
128    type Nested<'a> = Self;
129
130    fn nest(&mut self) -> Result<Self, Error> {
131        Ok(Self)
132    }
133
134    fn write_null(&mut self) -> Result<Value, Error> {
135        Ok(Value::Null)
136    }
137
138    fn write_bool(&mut self, value: bool) -> Result<Value, Error> {
139        Ok(Value::Bool(value))
140    }
141
142    fn write_number(&mut self, value: u32) -> Result<Value, Error> {
143        Ok(Value::Number(value.into()))
144    }
145
146    fn write_str(&mut self, value: String) -> Result<Value, Error> {
147        Ok(Value::String(value))
148    }
149
150    fn write_vec(&mut self, value: Self::Vec) -> Result<Value, Error> {
151        Ok(Value::Array(value))
152    }
153
154    fn write_map(&mut self, value: Self::Map) -> Result<Value, Error> {
155        Ok(Value::Object(value))
156    }
157
158    fn vec_push_element(&mut self, vec: &mut Vec<Value>, val: Value) -> Result<(), Error> {
159        vec.push(val);
160        Ok(())
161    }
162
163    fn map_push_field(
164        &mut self,
165        map: &mut Map<String, Value>,
166        key: String,
167        val: Value,
168    ) -> Result<(), Error> {
169        map.insert(key, val);
170        Ok(())
171    }
172}
173
174impl From<RV::Error> for Error {
175    fn from(RV::Error: RV::Error) -> Self {
176        Error::UnexpectedType
177    }
178}
179
180impl From<OV::Error> for Error {
181    fn from(OV::Error: OV::Error) -> Self {
182        Error::UnexpectedType
183    }
184}
