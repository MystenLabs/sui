4//! Package store implementation that fetches packages from a Sui fullnode via RPC.
5//!
6//! This provides a convenient way to resolve package types
7//! by fetching package data directly from the network.
8
9use async_trait::async_trait;
10use move_core_types::account_address::AccountAddress;
11use std::sync::Arc;
12use sui_package_resolver::{Package, PackageStore, PackageStoreWithLruCache, error::Error};
13use sui_rpc_api::Client;
14use sui_types::base_types::ObjectID;
15
16/// A PackageStore implementation that fetches packages from a Sui fullnode via RPC.
17#[derive(Clone)]
18pub struct RpcPackageStore {
19    client: Client,
20}
21
22impl RpcPackageStore {
23    /// Create a new RpcPackageStore with the given RPC URL.
24    ///
25    /// # Example
26    /// ```ignore
27    /// let store = RpcPackageStore::new("https://fullnode.testnet.sui.io:443");
28    /// ```
29    pub fn new(rpc_url: &str) -> Self {
30        let client = Client::new(rpc_url).expect("Failed to create RPC client - invalid URL");
31        Self { client }
32    }
33
34    /// Add an LRU cache layer to this package store for improved performance.
35    ///
36    /// This is particularly useful when processing many events or objects from
37    /// the same packages, as it avoids redundant package fetches over the network.
38    ///
39    /// # Example
40    /// ```ignore
41    /// use sui_rpc_resolver::package_store::RpcPackageStore;
42    /// use sui_package_resolver::Resolver;
43    ///
44    /// let store = RpcPackageStore::new("https://fullnode.testnet.sui.io:443");
45    /// let cached_store = store.with_cache();
46    /// let resolver = Resolver::new(cached_store);
47    /// ```
48    pub fn with_cache(self) -> PackageStoreWithLruCache<Self> {
49        PackageStoreWithLruCache::new(self)
50    }
51}
52
53#[async_trait]
54impl PackageStore for RpcPackageStore {
55    async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>, Error> {
56        // Fetch the object from the RPC client (client needs to be cloned because get_object takes ownership)
57        let object = self
58            .client
59            .clone()
60            .get_object(ObjectID::from(id))
61            .await
62            .map_err(|_| Error::PackageNotFound(id))?;
63
64        // Convert the object to a Package
65        Ok(Arc::new(Package::read_from_object(&object)?))
66    }
67}
