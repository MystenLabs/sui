4//! Package store implementation that fetches packages from a Sui fullnode via RPC.
5//!
6//! This provides a convenient way to resolve package types
7//! by fetching package data directly from the network.
8
9use std::sync::Arc;
10
11use async_trait::async_trait;
12use move_core_types::account_address::AccountAddress;
13use sui_package_resolver::Package;
14use sui_package_resolver::PackageStore;
15use sui_package_resolver::PackageStoreWithLruCache;
16use sui_package_resolver::error::Error;
17use sui_rpc_api::Client;
18use sui_types::base_types::ObjectID;
19
20/// A PackageStore implementation that fetches packages from a Sui fullnode via RPC.
21#[derive(Clone)]
22pub struct RpcPackageStore {
23    client: Client,
24}
25
26impl RpcPackageStore {
27    /// Create a new RpcPackageStore with the given RPC URL.
28    ///
29    /// # Example
30    /// ```ignore
31    /// let store = RpcPackageStore::new("https://fullnode.testnet.sui.io:443");
32    /// ```
33    pub fn new(rpc_url: &str) -> Self {
34        let client = Client::new(rpc_url).expect("Failed to create RPC client - invalid URL");
35        Self { client }
36    }
37
38    /// Add an LRU cache layer to this package store for improved performance.
39    ///
40    /// This is particularly useful when processing many events or objects from
41    /// the same packages, as it avoids redundant package fetches over the network.
42    ///
43    /// # Example
44    /// ```ignore
45    /// use sui_rpc_resolver::package_store::RpcPackageStore;
46    /// use sui_package_resolver::Resolver;
47    ///
48    /// let store = RpcPackageStore::new("https://fullnode.testnet.sui.io:443");
49    /// let cached_store = store.with_cache();
50    /// let resolver = Resolver::new(cached_store);
51    /// ```
52    pub fn with_cache(self) -> PackageStoreWithLruCache<Self> {
53        PackageStoreWithLruCache::new(self)
54    }
55}
56
57#[async_trait]
58impl PackageStore for RpcPackageStore {
59    async fn fetch(&self, id: AccountAddress) -> Result<Arc<Package>, Error> {
60        // Fetch the object from the RPC client (client needs to be cloned because get_object takes ownership)
61        let object = self
62            .client
63            .clone()
64            .get_object(ObjectID::from(id))
65            .await
66            .map_err(|_| Error::PackageNotFound(id))?;
67
68        // Convert the object to a Package
69        Ok(Arc::new(Package::read_from_object(&object)?))
70    }
71}
