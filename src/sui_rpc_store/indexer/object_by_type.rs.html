4//! Sequential pipeline that populates the
5//! [`schema::object_by_type`](crate::schema::object_by_type)
6//! index, using the same `Delete`-then-`Put` diff pattern as
7//! [`object_by_owner`](crate::indexer::object_by_owner).
8
9use std::sync::Arc;
10
11use async_trait::async_trait;
12use sui_consistent_store::Batch;
13use sui_consistent_store::Restore;
14use sui_indexer_alt_framework::pipeline::Processor;
15use sui_indexer_alt_framework::pipeline::sequential;
16use sui_types::full_checkpoint_content::Checkpoint;
17use sui_types::object::Object;
18
19use crate::RpcStoreSchema;
20use crate::indexer::Schema;
21use crate::indexer::Store;
22use crate::indexer::checkpoint_input_objects;
23use crate::indexer::checkpoint_output_objects;
24use crate::schema::object_by_type;
25use crate::schema::primitives::U64Varint;
26
27/// Pipeline marker for `object_by_type`.
28pub struct ObjectByType;
29
30pub enum Row {
31    Delete(object_by_type::Key),
32    Put(object_by_type::Key, U64Varint),
33}
34
35#[async_trait]
36impl Processor for ObjectByType {
37    const NAME: &'static str = "object_by_type";
38    type Value = Row;
39
40    async fn process(&self, checkpoint: &Arc<Checkpoint>) -> anyhow::Result<Vec<Row>> {
41        let mut rows = Vec::new();
42        for (_, (input, _)) in checkpoint_input_objects(checkpoint)? {
43            if let Some((key, _)) = object_by_type::store(input) {
44                rows.push(Row::Delete(key));
45            }
46        }
47        for (_, (output, _)) in checkpoint_output_objects(checkpoint)? {
48            if let Some((key, version)) = object_by_type::store(output) {
49                rows.push(Row::Put(key, version));
50            }
51        }
52        Ok(rows)
53    }
54}
55
56impl Restore for ObjectByType {
57    type Schema = RpcStoreSchema;
58
59    fn restore(
60        &self,
61        schema: &Self::Schema,
62        object: &Object,
63        batch: &mut Batch,
64    ) -> anyhow::Result<()> {
65        if let Some((key, value)) = object_by_type::store(object) {
66            batch.put(&schema.object_by_type, &key, &value)?;
67        }
68        Ok(())
69    }
70}
71
72#[async_trait]
73impl sequential::Handler for ObjectByType {
74    type Store = Store;
75    type Batch = Vec<Row>;
76
77    fn batch(&self, batch: &mut Self::Batch, values: std::vec::IntoIter<Row>) {
78        batch.extend(values);
79    }
80
81    async fn commit<'a>(
82        &self,
83        batch: &Self::Batch,
84        conn: &mut sui_consistent_store::Connection<'a, Schema>,
85    ) -> anyhow::Result<usize> {
86        let cf = &conn.store.schema().object_by_type;
87        for row in batch {
88            match row {
89                Row::Delete(key) => {
90                    conn.batch.delete(cf, key)?;
91                }
92                Row::Put(key, value) => {
93                    conn.batch.put(cf, key, value)?;
94                }
95            }
96        }
97        Ok(batch.len())
98    }
99}

119
120    #[test]
121    fn restore_indexes_object_by_struct_tag() {
122        let dir = tempfile::tempdir().unwrap();
123        let (db, schema) = Db::open::<RpcStoreSchema>(dir.path(), DbOptions::default()).unwrap();
124
125        let object =
126            Object::with_id_owner_for_testing(ObjectID::from_single_byte(11), SuiAddress::ZERO);
127        let type_ = object.type_().unwrap().clone().into();
128
129        let mut batch = db.batch();
130        ObjectByType.restore(&schema, &object, &mut batch).unwrap();
131        batch.commit().unwrap();
132
133        let rows: Vec<_> = schema
134            .iter_objects_of_type(&TypeFilter::Type(type_))
135            .unwrap()
136            .collect::<Result<Vec<_>, _>>()
137            .unwrap();
138        assert_eq!(rows.len(), 1);
139        assert_eq!(rows[0].0.object_id, object.id());
140    }
