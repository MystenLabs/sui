4//! `(dimension_key, bucket)` → `BitmapBlob`.
5//!
6//! Same wire shape as [`super::transaction_bitmap`]
7//! but indexes packed-event-seq space — each set bit identifies a
8//! single event by `(tx_seq << EVENT_BITS) | event_idx`.
9//!
10//! The merge operator is identical in structure to the
11//! transaction-bitmap one (union + optimize). The per-bucket
12//! compaction filter translates the `tx_seq` pruning floor from
13//! [`pruning_watermark::tx_seq_floor`](super::pruning_watermark::tx_seq_floor)
14//! into packed-event-seq space (`tx_seq << EVENT_BITS`) and drops
15//! buckets that fit entirely below it.
16

36pub const NAME: &str = "event_bitmap";
37
38/// Number of consecutive `packed_event_seq` values represented by
39/// one bucket. Sized so each bucket covers
40/// `EVENT_BUCKET_SIZE >> EVENT_BITS = 4096` consecutive
41/// transactions worth of events.
42pub const EVENT_BUCKET_SIZE: u64 = 1 << 28;
43
44const _: () = assert!(EVENT_BUCKET_SIZE <= u32::MAX as u64);
45
46#[derive(Debug, Clone, PartialEq, Eq, Hash)]
47pub struct Key {
48    pub dimension_key: Vec<u8>,
49    pub bucket: u64,
50}
51
52pub type Value = Protobuf<BitmapBlob>;
53
54impl Encode for Key {
55    fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
56        buf.put_slice(&self.dimension_key);
57        buf.put_slice(&self.bucket.to_be_bytes());
58        Ok(())
59    }
60}
61
62impl Decode for Key {
63    fn decode<B: Buf>(buf: &mut B) -> Result<Self, DecodeError> {
64        if buf.remaining() < 8 {
65            return Err(DecodeError::msg(format!(
66                "{NAME} Key too short: {} bytes",
67                buf.remaining(),
68            )));
69        }
70        let dim_len = buf.remaining() - 8;
71        let dim_bytes = buf.copy_to_bytes(dim_len);
72        let bucket = buf.get_u64();
73        Ok(Key {
74            dimension_key: dim_bytes.to_vec(),
75            bucket,
76        })
77    }
78}

80/// CF options: install the bitmap-union merge operator and a
81/// per-bucket compaction filter that drops buckets whose entire
82/// packed-event-seq range sits below the pruning floor.
83pub fn options(resolver: &sui_consistent_store::CfOptionsResolver) -> rocksdb::Options {
84    let mut opts = resolver.options(NAME);
85    opts.set_merge_operator_associative("event_bitmap_merge", merge);
86    let floor = tx_seq_floor().clone();
87    opts.set_compaction_filter("event_bitmap_pruning", move |_level, key, _value| {
88        let tx_seq_pruned = floor.load(Ordering::Relaxed);
89        if should_remove_bucket(key, tx_seq_pruned) {
90            rocksdb::CompactionDecision::Remove
91        } else {
92            rocksdb::CompactionDecision::Keep
93        }
94    });
95    opts
96}

98/// Pure logic of the compaction filter.
99///
100/// Translates the `tx_seq` floor into packed-event-seq space and
101/// asks whether every packed event the bucket could hold is
102/// strictly below the translated floor. Kept on any malformed
103/// input — silent data loss is worse than a stuck row.
104pub(crate) fn should_remove_bucket(key: &[u8], tx_seq_pruned_exclusive: u64) -> bool {
105    if key.len() < 8 {
106        return false;
107    }
108    let bucket_id_bytes: [u8; 8] = key[key.len() - 8..].try_into().expect("slice length");
109    let bucket_id = u64::from_be_bytes(bucket_id_bytes);
110    let packed_floor = packed_pruning_floor(tx_seq_pruned_exclusive);
111    bucket_id
112        .checked_add(1)
113        .and_then(|b| b.checked_mul(EVENT_BUCKET_SIZE))
114        .is_some_and(|highest_plus_one| highest_plus_one <= packed_floor)
115}
116
117/// Translate the `tx_seq` floor into packed-event-seq space.
118///
119/// `packed_event_seq = tx_seq << EVENT_BITS`. For
120/// `tx_seq >= 2^(64 - EVENT_BITS)` the shift would overflow a
121/// `u64`; we saturate to `u64::MAX`, which represents "every
122/// possible event has been pruned" — the conservative direction
123/// for a removal decision.
124fn packed_pruning_floor(tx_seq_pruned_exclusive: u64) -> u64 {
125    const OVERFLOW_THRESHOLD: u64 = 1u64 << (64 - EVENT_BITS);
126    if tx_seq_pruned_exclusive < OVERFLOW_THRESHOLD {
127        tx_seq_pruned_exclusive << EVENT_BITS
128    } else {
129        u64::MAX
130    }
131}

133/// The bucket that owns a given packed event sequence.
134pub fn bucket_of(packed: u64) -> u64 {
135    packed / EVENT_BUCKET_SIZE
136}
137
138/// The bit position within a bucket for a given packed event
139/// sequence. The cast is safe because `EVENT_BUCKET_SIZE`
140/// fits in a `u32` (enforced at compile time above).
141pub fn bit_of(packed: u64) -> u32 {
142    (packed % EVENT_BUCKET_SIZE) as u32
143}
144
145/// Build a `(Key, Value)` pair that adds the event identified by
146/// `(tx_seq, event_idx)` to the bitmap for its dimension and
147/// bucket. The merge operator unions this single-bit operand
148/// with whatever's already on disk.
149pub fn store_match(dimension_key: Vec<u8>, tx_seq: u64, event_idx: u32) -> (Key, Value) {
150    let packed = encode_event_seq(tx_seq, event_idx);
151    let mut bitmap = RoaringBitmap::new();
152    bitmap.insert(bit_of(packed));
153    store_bitmap(dimension_key, bucket_of(packed), bitmap)
154}
155
156/// Build a `(Key, Value)` pair that stages the given bitmap as a
157/// merge operand against the existing on-disk bitmap. Useful for
158/// pipelines that batch many events into one bucket per dimension
159/// before writing.
160pub fn store_bitmap(dimension_key: Vec<u8>, bucket: u64, bitmap: RoaringBitmap) -> (Key, Value) {
161    (
162        Key {
163            dimension_key,
164            bucket,
165        },
166        Protobuf(BitmapBlob {
167            data: serialize_bitmap(&bitmap).into(),
168        }),
169    )
170}

172impl<R: Reader> super::RpcStoreSchema<R> {
173    /// Look up the event bitmap for `(dimension_key, bucket)` and
174    /// return it deserialized.
175    pub fn get_event_bitmap(
176        &self,
177        dimension_key: Vec<u8>,
178        bucket: u64,
179    ) -> Result<Option<RoaringBitmap>, Error> {
180        let Some(stored) = self.event_bitmap.get(&Key {
181            dimension_key,
182            bucket,
183        })?
184        else {
185            return Ok(None);
186        };
187        let bytes = stored.into_inner().data;
188        let bitmap = RoaringBitmap::deserialize_from(bytes.as_ref())
189            .map_err(|e| DecodeError::with_source("deserialize RoaringBitmap", e))?;
190        Ok(Some(bitmap))
191    }
192
193    /// Iterate every bucket recorded against `dimension_key`, in
194    /// ascending bucket order.
195    pub fn iter_event_bitmap_buckets(
196        &self,
197        dimension_key: Vec<u8>,
198    ) -> Result<Iter<'_, Key, Value>, Error> {
199        self.event_bitmap
200            .iter_prefix(&DimensionPrefix(dimension_key))
201    }
202}

204/// Prefix encoder for "all buckets recorded against
205/// `dimension_key`". Encodes as the raw dimension bytes — the
206/// leading bytes of every `Key` whose `dimension_key` matches.
207pub struct DimensionPrefix(pub Vec<u8>);
208
209impl Encode for DimensionPrefix {
210    fn encode_into<B: BufMut>(&self, buf: &mut B) -> Result<(), EncodeError> {
211        buf.put_slice(&self.0);
212        Ok(())
213    }
214}

216/// Serialize a roaring bitmap for on-disk storage. Run-encodes
217/// dense containers first so a bucket that matches many
218/// consecutive packed event sequences compresses well.
219fn serialize_bitmap(bitmap: &RoaringBitmap) -> Vec<u8> {
220    let mut buf = Vec::with_capacity(bitmap.serialized_size());
221    bitmap
222        .serialize_into(&mut buf)
223        .expect("RoaringBitmap::serialize_into on Vec cannot fail");
224    buf
225}
226
227/// Associative merge: union every operand bitmap with the
228/// existing on-disk bitmap, then optimize the accumulator before
229/// re-serializing.
230///
231/// Encode / decode failures panic — this CF is written only by
232/// the crate's `store_*` helpers, so a parse failure indicates
233/// corruption rather than a recoverable situation.
234fn merge(
235    _key: &[u8],
236    existing_val: Option<&[u8]>,
237    operands: &rocksdb::MergeOperands,
238) -> Option<Vec<u8>> {
239    let mut acc = match existing_val {
240        Some(bytes) => decode_bitmap(bytes),
241        None => RoaringBitmap::new(),
242    };
243
244    for operand in operands {
245        let bitmap = decode_bitmap(operand);
246        acc |= bitmap;
247    }
248
249    acc.optimize();
250    Some(encode_bitmap_blob(&acc))
251}
252
253fn decode_bitmap(bytes: &[u8]) -> RoaringBitmap {
254    let blob = BitmapBlob::decode(bytes).expect("decode BitmapBlob");
255    RoaringBitmap::deserialize_from(blob.data.as_ref()).expect("deserialize RoaringBitmap")
256}
257
258fn encode_bitmap_blob(bitmap: &RoaringBitmap) -> Vec<u8> {
259    let blob = BitmapBlob {
260        data: serialize_bitmap(bitmap).into(),
261    };
262    blob.encode_to_vec()
263}

281    #[test]
282    fn pack_bucket_and_bit_math() {
283        // tx_seq=0, event_idx=0 → packed 0 → bucket 0 / bit 0.
284        let p = encode_event_seq(0, 0);
285        assert_eq!(p, 0);
286        assert_eq!(bucket_of(p), 0);
287        assert_eq!(bit_of(p), 0);
288
289        // tx_seq=1, event_idx=0 → packed `1 << 16` = 65_536.
290        let p = encode_event_seq(1, 0);
291        assert_eq!(p, 1 << EVENT_BITS);
292        assert_eq!(bucket_of(p), 0);
293        assert_eq!(bit_of(p), 1 << EVENT_BITS);
294
295        // The first packed value of the next bucket sits at the
296        // boundary `EVENT_BUCKET_SIZE` — that's
297        // `EVENT_BUCKET_SIZE >> EVENT_BITS = 4096` transactions in.
298        let first_in_next_bucket = encode_event_seq(EVENT_BUCKET_SIZE >> EVENT_BITS, 0);
299        assert_eq!(first_in_next_bucket, EVENT_BUCKET_SIZE);
300        assert_eq!(bucket_of(first_in_next_bucket), 1);
301        assert_eq!(bit_of(first_in_next_bucket), 0);
302    }

386    #[test]
387    fn should_remove_bucket_drops_only_fully_pruned_ranges() {
388        let dim = b"emitting_module:coin";
389        let bucket_0_key = Key {
390            dimension_key: dim.to_vec(),
391            bucket: 0,
392        }
393        .encode()
394        .unwrap();
395
396        // Floor 0 → nothing pruned.
397        assert!(!should_remove_bucket(&bucket_0_key, 0));
398
399        // EVENT_BUCKET_SIZE in packed-event-seq space corresponds
400        // to `EVENT_BUCKET_SIZE >> EVENT_BITS` transactions —
401        // anything below that tx_seq floor keeps bucket 0 alive.
402        let txs_per_bucket = EVENT_BUCKET_SIZE >> EVENT_BITS;
403        assert!(!should_remove_bucket(&bucket_0_key, txs_per_bucket - 1));
404        // At the tx_seq floor that translates to exactly
405        // EVENT_BUCKET_SIZE in packed space, bucket 0 becomes
406        // fully pruned.
407        assert!(should_remove_bucket(&bucket_0_key, txs_per_bucket));
408
409        // Bucket 5 needs floor past 6 * EVENT_BUCKET_SIZE in
410        // packed space, i.e. tx_seq past 6 * txs_per_bucket.
411        let bucket_5_key = Key {
412            dimension_key: dim.to_vec(),
413            bucket: 5,
414        }
415        .encode()
416        .unwrap();
417        assert!(!should_remove_bucket(&bucket_5_key, 6 * txs_per_bucket - 1));
418        assert!(should_remove_bucket(&bucket_5_key, 6 * txs_per_bucket));
419
420        // Key too short → kept.
421        assert!(!should_remove_bucket(&[0u8; 4], u64::MAX));
422    }

424    #[test]
425    fn packed_pruning_floor_saturates_on_overflow() {
426        assert_eq!(packed_pruning_floor(0), 0);
427        assert_eq!(packed_pruning_floor(1), 1u64 << EVENT_BITS);
428        // Just below the overflow threshold.
429        let just_below = (1u64 << (64 - EVENT_BITS)) - 1;
430        assert_eq!(packed_pruning_floor(just_below), just_below << EVENT_BITS,);
431        // At the threshold — `tx_seq << EVENT_BITS` would
432        // overflow, so we saturate.
433        assert_eq!(packed_pruning_floor(1u64 << (64 - EVENT_BITS)), u64::MAX);
434        assert_eq!(packed_pruning_floor(u64::MAX), u64::MAX);
435    }

438    fn iter_walks_buckets_for_one_dimension_in_order() {
439        let (_dir, db, schema) = fresh_db();
440        let dim = b"emitting_module:coin".to_vec();
441        let other = b"emitting_module:nft".to_vec();
442        // Three events whose packed seqs land in distinct

446        let tx_seqs = [0u64, txs_per_bucket + 5, 3 * txs_per_bucket + 9];

462            .unwrap()
463            .map(|res| res.unwrap().0.bucket)
