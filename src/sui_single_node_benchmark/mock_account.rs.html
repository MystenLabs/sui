4use futures::stream::FuturesUnordered;
5use std::collections::BTreeMap;
6use std::sync::Arc;
7use sui_types::base_types::{ObjectRef, SuiAddress};
8use sui_types::crypto::{AccountKeyPair, get_account_key_pair};
9use sui_types::object::Object;
10
11#[derive(Clone)]
12pub struct Account {
13    pub sender: SuiAddress,
14    pub keypair: Arc<AccountKeyPair>,
15    pub gas_objects: Arc<Vec<ObjectRef>>,
16}
17
18/// Generate \num_accounts accounts and for each account generate \gas_object_num_per_account gas objects.
19/// Return all accounts along with a flattened list of all gas objects as genesis objects.
20pub async fn batch_create_account_and_gas(
21    num_accounts: u64,
22    gas_object_num_per_account: u64,
23) -> (BTreeMap<SuiAddress, Account>, Vec<Object>) {
24    let tasks: FuturesUnordered<_> = (0..num_accounts)
25        .map(|_| {
26            tokio::spawn(async move {
27                let (sender, keypair) = get_account_key_pair();
28                let objects = (0..gas_object_num_per_account)
29                    .map(|_| Object::with_owner_for_testing(sender))
30                    .collect::<Vec<_>>();
31                (sender, keypair, objects)
32            })
33        })
34        .collect();
35    let mut accounts = BTreeMap::new();
36    let mut genesis_gas_objects = vec![];
37    for task in tasks {
38        let (sender, keypair, gas_objects) = task.await.unwrap();
39        let gas_object_refs: Vec<_> = gas_objects
40            .iter()
41            .map(|o| o.compute_object_reference())
42            .collect();
43        accounts.insert(
44            sender,
45            Account {
46                sender,
47                keypair: Arc::new(keypair),
48                gas_objects: Arc::new(gas_object_refs),
49            },
50        );
51        genesis_gas_objects.extend(gas_objects);
52    }
53    (accounts, genesis_gas_objects)
54}
