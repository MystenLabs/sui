22#[derive(Clone)]
23pub(crate) struct InMemoryObjectStore {
24    objects: Arc<RwLock<HashMap<ObjectID, Object>>>,
25    package_cache: Arc<PackageObjectCache>,
26    num_object_reads: Arc<AtomicU64>,
27}
28
29impl InMemoryObjectStore {
30    pub(crate) fn new(objects: HashMap<ObjectID, Object>) -> Self {
31        Self {
32            objects: Arc::new(RwLock::new(objects)),
33            package_cache: PackageObjectCache::new(),
34            num_object_reads: Arc::new(AtomicU64::new(0)),
35        }
36    }
37
38    pub(crate) fn get_num_object_reads(&self) -> u64 {
39        self.num_object_reads.get()
40    }
41
42    // TODO: This function is out-of-sync with read_objects_for_execution from transaction_input_loader.rs.
43    // For instance, it does not support the use of deleted shared objects.
44    // We will need a trait to unify the these functions. (similarly the one in simulacrum)
45    pub(crate) fn read_objects_for_execution(
46        &self,
47        tx_key: &TransactionKey,
48        assigned_versions: &AssignedVersions,
49        input_object_kinds: &[InputObjectKind],
50    ) -> SuiResult<InputObjects> {
51        let shared_version_assignments: HashMap<_, _> =
52            assigned_versions.iter().map(|(k, v)| (*k, *v)).collect();
53        let mut input_objects = Vec::new();
54        for kind in input_object_kinds {
55            let obj: Option<Object> = match kind {
56                InputObjectKind::MovePackage(id) => self.get_package_object(id)?.map(|o| o.into()),
57                InputObjectKind::ImmOrOwnedMoveObject(objref) => {
58                    self.get_object_by_key(&objref.0, objref.1)
59                }
60
61                InputObjectKind::SharedMoveObject {
62                    id,
63                    initial_shared_version,
64                    ..
65                } => {
66                    let version = shared_version_assignments.get(&(*id, *initial_shared_version)).unwrap_or_else(|| {
67                        panic!("Shared object version should have been assigned. key: {tx_key:?}, obj id: {id:?}")
68                    });
69
70                    self.get_object_by_key(id, *version)
71                }
72            };
73
74            input_objects.push(ObjectReadResult::new(
75                *kind,
76                obj.ok_or_else(|| kind.object_not_found_error())?.into(),
77            ));
78        }
79
80        Ok(input_objects.into())
81    }
82
83    pub(crate) fn commit_objects(&self, inner_temp_store: InnerTemporaryStore) {
84        let mut objects = self.objects.write().unwrap();
85        for (object_id, _) in inner_temp_store.mutable_inputs {
86            if !inner_temp_store.written.contains_key(&object_id) {
87                objects.remove(&object_id);
88            }
89        }
90        for (object_id, object) in inner_temp_store.written {
91            objects.insert(object_id, object);
92        }
93    }
94}
95
96impl ObjectStore for InMemoryObjectStore {
97    fn get_object(&self, object_id: &ObjectID) -> Option<Object> {
98        self.num_object_reads.inc_by(1);
99        self.objects.read().unwrap().get(object_id).cloned()
100    }
101
102    fn get_object_by_key(&self, object_id: &ObjectID, version: VersionNumber) -> Option<Object> {
103        self.get_object(object_id).and_then(|o| {
104            if o.version() == version {
105                Some(o.clone())
106            } else {
107                None
108            }
109        })
110    }
111}
112
113impl BackingPackageStore for InMemoryObjectStore {
114    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
115        self.package_cache.get_package_object(package_id, self)
116    }
117}

119impl ChildObjectResolver for InMemoryObjectStore {
120    fn read_child_object(
121        &self,
122        parent: &ObjectID,
123        child: &ObjectID,
124        child_version_upper_bound: SequenceNumber,
125    ) -> SuiResult<Option<Object>> {
126        Ok(self.get_object(child).and_then(|o| {
127            if o.version() <= child_version_upper_bound
128                && o.owner == Owner::ObjectOwner((*parent).into())
129            {
130                Some(o.clone())
131            } else {
132                None
133            }
134        }))
135    }
136
137    fn get_object_received_at_version(
138        &self,
139        _owner: &ObjectID,
140        _receiving_object_id: &ObjectID,
141        _receive_object_at_version: SequenceNumber,
142        _epoch_id: EpochId,
143    ) -> SuiResult<Option<Object>> {
144        unimplemented!()
145    }
146}
