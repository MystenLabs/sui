46    pub(crate) async fn new(genesis_objects: &[Object], component: Component) -> Self {
47        let validator = TestAuthorityBuilder::new()
48            .disable_indexer()
49            .with_starting_objects(genesis_objects)
50            // This is needed to properly run checkpoint executor.
51            .insert_genesis_checkpoint()
52            .build()
53            .await;
54        let epoch_store = validator.epoch_store_for_testing().clone();
55        let consensus_mode = match component {
56            Component::ValidatorWithFakeConsensus => ConsensusMode::DirectSequencing,
57            _ => ConsensusMode::Noop,
58        };
59        let consensus_adapter = Arc::new(ConsensusAdapter::new(
60            Arc::new(MockConsensusClient::new(
61                Arc::downgrade(&validator),
62                consensus_mode,
63            )),
64            validator.checkpoint_store.clone(),
65            validator.name,
66            Arc::new(ConnectionMonitorStatusForTests {}),
67            100_000,
68            100_000,
69            None,
70            None,
71            ConsensusAdapterMetrics::new_test(),
72            epoch_store.protocol_config().clone(),
73        ));
74        // TODO: for validator benchmarking purposes, we should allow for traffic control
75        // to be configurable and introduce traffic control benchmarks to test
76        // against different policies
77        let validator_service = Arc::new(ValidatorService::new_for_tests(
78            validator,
79            consensus_adapter,
80            Arc::new(ValidatorServiceMetrics::new_for_tests()),
81        ));
82        Self {
83            validator_service,
84            epoch_store,
85        }
86    }

96    /// Publish a package, returns the package object and the updated gas object.
97    pub async fn publish_package(
98        &self,
99        publish_data: PublishData,
100        sender: SuiAddress,
101        keypair: &AccountKeyPair,
102        gas: ObjectRef,
103    ) -> (ObjectRef, ObjectRef) {
104        let tx_builder = TestTransactionBuilder::new(sender, gas, DEFAULT_VALIDATOR_GAS_PRICE)
105            .publish_with_data(publish_data);
106        let transaction = tx_builder.build_and_sign(keypair);
107        let effects = self.execute_raw_transaction(transaction).await;
108        let package = effects
109            .all_changed_objects()
110            .into_iter()
111            .filter_map(|(oref, owner, _)| owner.is_immutable().then_some(oref))
112            .next()
113            .unwrap();
114        let updated_gas = effects.gas_object().0;
115        (package, updated_gas)
116    }

118    pub async fn execute_raw_transaction(&self, transaction: Transaction) -> TransactionEffects {
119        let executable = VerifiedExecutableTransaction::new_from_quorum_execution(
120            VerifiedTransaction::new_unchecked(transaction),
121            0,
122        );
123        let effects = self
124            .get_validator()
125            .try_execute_immediately(
126                &executable,
127                ExecutionEnv::new().with_scheduling_source(SchedulingSource::NonFastPath),
128                &self.epoch_store,
129            )
130            .await
131            .unwrap()
132            .0;
133        assert!(effects.status().is_ok());
134        effects
135    }

150    pub async fn execute_certificate(
151        &self,
152        cert: CertifiedTransaction,
153        assigned_versions: &AssignedVersions,
154        component: Component,
155    ) -> TransactionEffects {
156        let effects = match component {
157            Component::Baseline => {
158                let cert = VerifiedExecutableTransaction::new_from_certificate(
159                    VerifiedCertificate::new_unchecked(cert),
160                );
161                self.get_validator()
162                    .try_execute_immediately(
163                        &cert,
164                        ExecutionEnv::new().with_assigned_versions(assigned_versions.clone()),
165                        &self.epoch_store,
166                    )
167                    .await
168                    .unwrap()
169                    .0
170            }
171            Component::WithTxManager => {
172                let cert = VerifiedCertificate::new_unchecked(cert);
173                if cert.is_consensus_tx() {
174                    // For shared objects transactions, `execute_certificate` won't enqueue it because
175                    // it expects consensus to do so. However we don't have consensus, hence the manual enqueue.
176                    self.get_validator().execution_scheduler().enqueue(
177                        vec![(
178                            VerifiedExecutableTransaction::new_from_certificate(cert.clone())
179                                .into(),
180                            ExecutionEnv::new().with_assigned_versions(assigned_versions.clone()),
181                        )],
182                        &self.epoch_store,
183                    );
184                }
185                self.get_validator()
186                    .wait_for_certificate_execution(&cert, &self.epoch_store)
187                    .await
188                    .unwrap()
189            }
190            Component::ValidatorWithoutConsensus | Component::ValidatorWithFakeConsensus => {
191                let response = self
192                    .validator_service
193                    .execute_certificate_for_testing(cert)
194                    .await
195                    .unwrap()
196                    .into_inner();
197                response.signed_effects.into_data()
198            }
199            Component::TxnSigning | Component::CheckpointExecutor | Component::ExecutionOnly => {
200                unreachable!()
201            }
202        };
203        assert!(effects.status().is_ok());
204        effects
205    }

207    pub(crate) async fn execute_transaction_in_memory(
208        &self,
209        store: InMemoryObjectStore,
210        transaction: CertifiedTransaction,
211        assigned_versions: &AssignedVersions,
212    ) -> TransactionEffects {
213        let input_objects = transaction.transaction_data().input_objects().unwrap();
214        let objects = store
215            .read_objects_for_execution(&transaction.key(), assigned_versions, &input_objects)
216            .unwrap();
217
218        let executable = VerifiedExecutableTransaction::new_from_certificate(
219            VerifiedCertificate::new_unchecked(transaction),
220        );
221        let (gas_status, input_objects) = sui_transaction_checks::check_certificate_input(
222            &executable,
223            objects,
224            self.epoch_store.protocol_config(),
225            self.epoch_store.reference_gas_price(),
226        )
227        .unwrap();
228        let (kind, signer, gas_data) = executable.transaction_data().execution_parts();
229        let (inner_temp_store, _, effects, _timings, _) =
230            self.epoch_store.executor().execute_transaction_to_effects(
231                &store,
232                self.epoch_store.protocol_config(),
233                self.get_validator().metrics.limits_metrics.clone(),
234                false,
235                ExecutionOrEarlyError::Ok(()),
236                &self.epoch_store.epoch(),
237                0,
238                input_objects,
239                gas_data,
240                gas_status,
241                kind,
242                signer,
243                *executable.digest(),
244                &mut None,
245            );
246        assert!(effects.status().is_ok());
247        store.commit_objects(inner_temp_store);
248        effects
249    }

259    pub(crate) async fn build_checkpoints(
260        &self,
261        transactions: Vec<CertifiedTransaction>,
262        mut all_effects: BTreeMap<TransactionDigest, TransactionEffects>,
263        checkpoint_size: usize,
264    ) -> Vec<(VerifiedCheckpoint, VerifiedCheckpointContents)> {
265        let mut builder = MockCheckpointBuilder::new(
266            self.get_validator()
267                .get_checkpoint_store()
268                .get_latest_certified_checkpoint()
269                .unwrap()
270                .unwrap(),
271        );
272        let mut checkpoints = vec![];
273        for transaction in transactions {
274            let effects = all_effects.remove(transaction.digest()).unwrap();
275            builder.push_transaction(
276                VerifiedTransaction::new_unchecked(transaction.into_unsigned()),
277                effects,
278            );
279            if builder.size() == checkpoint_size {
280                let (checkpoint, _, full_contents) = builder.build(self, 0);
281                checkpoints.push((checkpoint, full_contents));
282            }
283        }
284        if builder.size() > 0 {
285            let (checkpoint, _, full_contents) = builder.build(self, 0);
286            checkpoints.push((checkpoint, full_contents));
287        }
288        checkpoints
289    }

303    pub(crate) fn create_in_memory_store(&self) -> InMemoryObjectStore {
304        let objects: HashMap<_, _> = self
305            .get_validator()
306            .get_global_state_hash_store()
307            .iter_cached_live_object_set_for_testing(false)
308            .map(|o| match o {
309                LiveObject::Normal(object) => (object.id(), object),
310                LiveObject::Wrapped(_) => unreachable!(),
311            })
312            .collect();
313        InMemoryObjectStore::new(objects)
314    }

316    pub(crate) async fn assigned_shared_object_versions(
317        &self,
318        transactions: &[CertifiedTransaction],
319    ) -> AssignedTxAndVersions {
320        let transactions: Vec<_> = transactions
321            .iter()
322            .map(|tx| {
323                VerifiedExecutableTransaction::new_from_certificate(
324                    VerifiedCertificate::new_unchecked(tx.clone()),
325                )
326            })
327            .collect();
328        let assignables: Vec<_> = transactions.iter().map(Schedulable::Transaction).collect();
329        self.epoch_store
330            .assign_shared_object_versions_idempotent(
331                self.get_validator().get_object_cache_reader().as_ref(),
332                assignables.iter(),
333            )
334            .unwrap()
335    }
