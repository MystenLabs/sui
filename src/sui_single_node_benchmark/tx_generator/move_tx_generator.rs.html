4use crate::mock_account::Account;
5use crate::tx_generator::TxGenerator;
6use move_core_types::identifier::Identifier;

11use sui_types::transaction::{

15pub struct MoveTxGenerator {
16    move_package: ObjectID,
17    num_transfers: u64,
18    use_native_transfer: bool,
19    computation: u8,
20    root_objects: HashMap<SuiAddress, ObjectRef>,
21    shared_objects: Vec<(ObjectID, SequenceNumber)>,
22    num_mints: u16,
23    nft_size: u16,
24    use_batch_mint: bool,
25}
26
27impl MoveTxGenerator {
28    pub fn new(
29        move_package: ObjectID,
30        num_transfers: u64,
31        use_native_transfer: bool,
32        computation: u8,
33        root_objects: HashMap<SuiAddress, ObjectRef>,
34        shared_objects: Vec<(ObjectID, SequenceNumber)>,
35        num_mints: u16,
36        nft_size: u16,
37        use_batch_mint: bool,
38    ) -> Self {
39        Self {
40            move_package,
41            num_transfers,
42            use_native_transfer,
43            computation,
44            root_objects,
45            shared_objects,
46            num_mints,
47            nft_size,
48            use_batch_mint,
49        }
50    }
51}
52
53impl TxGenerator for MoveTxGenerator {
54    fn generate_tx(&self, account: Account) -> Transaction {
55        let pt = {
56            let mut builder = ProgrammableTransactionBuilder::new();
57            // Step 1: transfer `num_transfers` objects.
58            // First object in the gas_objects is the gas object and we are not transferring it.
59            for i in 1..=self.num_transfers {
60                let object = account.gas_objects[i as usize];
61                if self.use_native_transfer {
62                    builder
63                        .transfer_object(account.sender, FullObjectRef::from_fastpath_ref(object))
64                        .unwrap();
65                } else {
66                    builder
67                        .move_call(
68                            self.move_package,
69                            Identifier::new("benchmark").unwrap(),
70                            Identifier::new("transfer_coin").unwrap(),
71                            vec![],
72                            vec![CallArg::Object(ObjectArg::ImmOrOwnedObject(object))],
73                        )
74                        .unwrap();
75                }
76            }
77            for shared_object in &self.shared_objects {
78                builder
79                    .move_call(
80                        self.move_package,
81                        Identifier::new("benchmark").unwrap(),
82                        Identifier::new("increment_shared_counter").unwrap(),
83                        vec![],
84                        vec![CallArg::Object(ObjectArg::SharedObject {
85                            id: shared_object.0,
86                            initial_shared_version: shared_object.1,
87                            mutability: SharedObjectMutability::Mutable,
88                        })],
89                    )
90                    .unwrap();
91            }
92
93            if !self.root_objects.is_empty() {
94                // Step 2: Read all dynamic fields from the root object.
95                let root_object = self.root_objects.get(&account.sender).unwrap();
96                let root_object_arg = builder
97                    .obj(ObjectArg::ImmOrOwnedObject(*root_object))
98                    .unwrap();
99                builder.programmable_move_call(
100                    self.move_package,
101                    Identifier::new("benchmark").unwrap(),
102                    Identifier::new("read_dynamic_fields").unwrap(),
103                    vec![],
104                    vec![root_object_arg],
105                );
106            }
107
108            if self.computation > 0 {
109                // Step 3: Run some computation.
110                let computation_arg = builder.pure(self.computation as u64 * 100).unwrap();
111                builder.programmable_move_call(
112                    self.move_package,
113                    Identifier::new("benchmark").unwrap(),
114                    Identifier::new("run_computation").unwrap(),
115                    vec![],
116                    vec![computation_arg],
117                );
118            }
119            if self.num_mints > 0 {
120                // Step 4: Mint some NFTs
121                let mut contents = Vec::new();
122                assert!(self.nft_size >= 32, "NFT size must be at least 32 bytes");
123                for _ in 0..self.nft_size - 32 {
124                    contents.push(7u8)
125                }
126                if self.use_batch_mint {
127                    // create a vector of sender addresses to pass to batch_mint
128                    let mut recipients = Vec::new();
129                    for _ in 0..self.num_mints {
130                        recipients.push(account.sender)
131                    }
132                    let args = vec![
133                        builder.pure(recipients).unwrap(),
134                        builder.pure(contents).unwrap(),
135                    ];
136                    builder.programmable_move_call(
137                        self.move_package,
138                        Identifier::new("benchmark").unwrap(),
139                        Identifier::new("batch_mint").unwrap(),
140                        vec![],
141                        args,
142                    );
143                } else {
144                    // create PTB with a command that transfers each
145                    for _ in 0..self.num_mints {
146                        let args = vec![
147                            builder.pure(account.sender).unwrap(),
148                            builder.pure(contents.clone()).unwrap(),
149                        ];
150                        builder.programmable_move_call(
151                            self.move_package,
152                            Identifier::new("benchmark").unwrap(),
153                            Identifier::new("mint_one").unwrap(),
154                            vec![],
155                            args,
156                        );
157                    }
158                }
159            }
160            builder.finish()
161        };
162        TestTransactionBuilder::new(
163            account.sender,
164            account.gas_objects[0],
165            DEFAULT_VALIDATOR_GAS_PRICE,
166        )
167        .programmable(pt)
168        .build_and_sign(account.keypair.as_ref())
169    }
170
171    fn name(&self) -> &'static str {
172        "Programmable Move Transaction Generator"
173    }
174}
