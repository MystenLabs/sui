44/// The following describes the format of an object file (*.obj) used for persisting live sui objects.
45/// The maximum size per .obj file is 128MB. State snapshot will be taken at the end of every epoch.
46/// Live object set is split into and stored across multiple hash buckets. The hashing function used
47/// for bucketing objects is the same as the one used to build the accumulator tree for computing
48/// state root hash. Buckets are further subdivided into partitions. A partition is a smallest storage
49/// unit which holds a subset of objects in one bucket. Each partition is a single *.obj file where
50/// objects are appended to in an append-only fashion. A new partition is created once the size of
51/// current one reaches the max size i.e. 128MB. Partitions allow a single hash bucket to be consumed
52/// in parallel. Partition files are optionally compressed with the zstd compression format. Partition
53/// filenames follows the format <bucket_number>_<partition_number>.obj. Object references for hash
54/// There is one single ref file per hash bucket. Object references are written in an append-only manner
55/// as well. Finally, the MANIFEST file contains per file metadata of every file in the snapshot directory.
56/// current one reaches the max size i.e. 64MB. Partitions allow a single hash bucket to be consumed
57/// in parallel. Partition files are compressed with the zstd compression format.
58/// State Snapshot Directory Layout
59///  - snapshot/
60///     - epoch_0/
61///        - 1_1.obj
62///        - 1_2.obj
63///        - 1_3.obj
64///        - 2_1.obj
65///        - ...
66///        - 1000_1.obj
67///        - REFERENCE-1
68///        - REFERENCE-2
69///        - ...
70///        - REFERENCE-1000
71///        - MANIFEST
72///     - epoch_1/
73///       - 1_1.obj
74///       - ...
75///
76/// Object File Disk Format
77///┌──────────────────────────────┐
78///│  magic(0x00B7EC75) <4 byte>  │
79///├──────────────────────────────┤
80///│ ┌──────────────────────────┐ │
81///│ │         Object 1         │ │
82///│ ├──────────────────────────┤ │
83///│ │          ...             │ │
84///│ ├──────────────────────────┤ │
85///│ │         Object N         │ │
86///│ └──────────────────────────┘ │
87///└──────────────────────────────┘
88/// Object
89///┌───────────────┬───────────────────┬──────────────┐
90///│ len <uvarint> │ encoding <1 byte> │ data <bytes> │
91///└───────────────┴───────────────────┴──────────────┘
92///
93/// REFERENCE File Disk Format
94///┌──────────────────────────────┐
95///│  magic(0x5EFE5E11) <4 byte>  │
96///├──────────────────────────────┤
97///│ ┌──────────────────────────┐ │
98///│ │         ObjectRef 1      │ │
99///│ ├──────────────────────────┤ │
100///│ │          ...             │ │
101///│ ├──────────────────────────┤ │
102///│ │         ObjectRef N      │ │
103///│ └──────────────────────────┘ │
104///└──────────────────────────────┘
105/// ObjectRef (ObjectID, SequenceNumber, ObjectDigest)
106///┌───────────────┬───────────────────┬──────────────┐
107///│         data (<(address_len + 8 + 32) bytes>)    │
108///└───────────────┴───────────────────┴──────────────┘
109///
110/// MANIFEST File Disk Format
111///┌──────────────────────────────┐
112///│  magic(0x00C0FFEE) <4 byte>  │
113///├──────────────────────────────┤
114///│   serialized manifest        │
115///├──────────────────────────────┤
116///│      sha3 <32 bytes>         │
117///└──────────────────────────────┘
118const OBJECT_FILE_MAGIC: u32 = 0x00B7EC75;
119const REFERENCE_FILE_MAGIC: u32 = 0xDEADBEEF;
120const MANIFEST_FILE_MAGIC: u32 = 0x00C0FFEE;
121const MAGIC_BYTES: usize = 4;
122const SNAPSHOT_VERSION_BYTES: usize = 1;
123const ADDRESS_LENGTH_BYTES: usize = 8;
124const PADDING_BYTES: usize = 3;
125const MANIFEST_FILE_HEADER_BYTES: usize =
126    MAGIC_BYTES + SNAPSHOT_VERSION_BYTES + ADDRESS_LENGTH_BYTES + PADDING_BYTES;
127const FILE_MAX_BYTES: usize = 128 * 1024 * 1024;
128const OBJECT_ID_BYTES: usize = ObjectID::LENGTH;
129const SEQUENCE_NUM_BYTES: usize = 8;
130const OBJECT_DIGEST_BYTES: usize = 32;
131const OBJECT_REF_BYTES: usize = OBJECT_ID_BYTES + SEQUENCE_NUM_BYTES + OBJECT_DIGEST_BYTES;
132const FILE_TYPE_BYTES: usize = 1;
133const BUCKET_BYTES: usize = 4;
134const BUCKET_PARTITION_BYTES: usize = 4;
135const COMPRESSION_TYPE_BYTES: usize = 1;
136const FILE_METADATA_BYTES: usize =
137    FILE_TYPE_BYTES + BUCKET_BYTES + BUCKET_PARTITION_BYTES + COMPRESSION_TYPE_BYTES + SHA3_BYTES;

139#[derive(
140    Copy, Clone, Debug, Eq, PartialEq, Serialize, Deserialize, TryFromPrimitive, IntoPrimitive,
141)]
142#[repr(u8)]
143pub enum FileType {
144    Object = 0,
145    Reference,
146}
147
148#[derive(Debug, Clone, Serialize, Deserialize, Eq, PartialEq)]
149pub struct FileMetadata {
150    pub file_type: FileType,
151    pub bucket_num: u32,
152    pub part_num: u32,
153    pub file_compression: FileCompression,
154    pub sha3_digest: [u8; 32],
155}
156
157impl FileMetadata {
158    pub fn file_path(&self, dir_path: &Path) -> Path {
159        match self.file_type {
160            FileType::Object => {
161                dir_path.child(&*format!("{}_{}.obj", self.bucket_num, self.part_num))
162            }
163            FileType::Reference => {
164                dir_path.child(&*format!("{}_{}.ref", self.bucket_num, self.part_num))
165            }
166        }
167    }
168    pub fn local_file_path(&self, root_path: &std::path::Path, dir_path: &Path) -> Result<PathBuf> {
169        path_to_filesystem(root_path.to_path_buf(), &self.file_path(dir_path))
170    }
171}

173#[derive(Debug, Serialize, Deserialize, Eq, PartialEq)]
174pub struct ManifestV1 {
175    pub snapshot_version: u8,
176    pub address_length: u64,
177    pub file_metadata: Vec<FileMetadata>,
178    pub epoch: u64,
179}
180
181#[derive(Debug, Serialize, Deserialize, Eq, PartialEq)]
182pub enum Manifest {
183    V1(ManifestV1),
184}
185
186impl Manifest {
187    pub fn snapshot_version(&self) -> u8 {
188        match self {
189            Self::V1(manifest) => manifest.snapshot_version,
190        }
191    }
192    pub fn address_length(&self) -> u64 {
193        match self {
194            Self::V1(manifest) => manifest.address_length,
195        }
196    }
197    pub fn file_metadata(&self) -> &Vec<FileMetadata> {
198        match self {
199            Self::V1(manifest) => &manifest.file_metadata,
200        }
201    }
202    pub fn epoch(&self) -> u64 {
203        match self {
204            Self::V1(manifest) => manifest.epoch,
205        }
206    }
207}

209pub fn create_file_metadata(
210    file_path: &std::path::Path,
211    file_compression: FileCompression,
212    file_type: FileType,
213    bucket_num: u32,
214    part_num: u32,
215) -> Result<FileMetadata> {
216    file_compression.compress(file_path)?;
217    let sha3_digest = compute_sha3_checksum(file_path)?;
218    let file_metadata = FileMetadata {
219        file_type,
220        bucket_num,
221        part_num,
222        file_compression,
223        sha3_digest,
224    };
225    Ok(file_metadata)
226}

228pub async fn setup_db_state(
229    epoch: u64,
230    accumulator: GlobalStateHash,
231    perpetual_db: Arc<AuthorityPerpetualTables>,
232    checkpoint_store: Arc<CheckpointStore>,
233    committee_store: Arc<CommitteeStore>,
234    chain: Chain,
235    verify: bool,
236    num_live_objects: u64,
237    m: MultiProgress,
238) -> Result<()> {
239    // This function should be called once state accumulator based hash verification
240    // is complete and live object set state is downloaded to local store
241    let system_state_object = get_sui_system_state(&perpetual_db)?;
242    let new_epoch_start_state = system_state_object.into_epoch_start_state();
243    let next_epoch_committee = new_epoch_start_state.get_sui_committee();
244    let root_digest: ECMHLiveObjectSetDigest = accumulator.digest().into();
245    let last_checkpoint = checkpoint_store
246        .get_epoch_last_checkpoint(epoch)
247        .expect("Error loading last checkpoint for current epoch")
248        .expect("Could not load last checkpoint for current epoch");
249    let flags = EpochFlag::default_for_no_config();
250    let epoch_start_configuration = EpochStartConfiguration::new(
251        new_epoch_start_state,
252        *last_checkpoint.digest(),
253        &perpetual_db,
254        flags,
255    )
256    .unwrap();
257    perpetual_db.set_epoch_start_configuration(&epoch_start_configuration)?;
258    perpetual_db.insert_root_state_hash(epoch, last_checkpoint.sequence_number, accumulator)?;
259    perpetual_db.set_highest_pruned_checkpoint_without_wb(last_checkpoint.sequence_number)?;
260    committee_store.insert_new_committee(&next_epoch_committee)?;
261    checkpoint_store.update_highest_executed_checkpoint(&last_checkpoint)?;
262
263    if verify {
264        let simplified_unwrap_then_delete = match (chain, epoch) {
265            (Chain::Mainnet, ep) if ep >= 87 => true,
266            (Chain::Mainnet, ep) if ep < 87 => false,
267            (Chain::Testnet, ep) if ep >= 50 => true,
268            (Chain::Testnet, ep) if ep < 50 => false,
269            _ => panic!("Unsupported chain"),
270        };
271        let include_tombstones = !simplified_unwrap_then_delete;
272        let iter = perpetual_db.iter_live_object_set(include_tombstones);
273        let local_digest = ECMHLiveObjectSetDigest::from(
274            accumulate_live_object_iter(Box::new(iter), m.clone(), num_live_objects)
275                .await
276                .digest(),
277        );
278        assert_eq!(
279            root_digest, local_digest,
280            "End of epoch {} root state digest {} does not match \
281                local root state hash {} after restoring db from formal snapshot",
282            epoch, root_digest.digest, local_digest.digest,
283        );
284        println!("DB live object state verification completed successfully!");
285    }
286
287    Ok(())
288}

290pub async fn accumulate_live_object_iter(
291    iter: Box<dyn Iterator<Item = LiveObject> + '_>,
292    m: MultiProgress,
293    num_live_objects: u64,
294) -> GlobalStateHash {
295    // Monitor progress of live object accumulation
296    let accum_progress_bar = m.add(ProgressBar::new(num_live_objects).with_style(
297        ProgressStyle::with_template("[{elapsed_precise}] {wide_bar} {pos}/{len} ({msg})").unwrap(),
298    ));
299    let accum_counter = Arc::new(AtomicU64::new(0));
300    let cloned_accum_counter = accum_counter.clone();
301    let cloned_progress_bar = accum_progress_bar.clone();
302    let handle = tokio::spawn(async move {
303        let a_instant = Instant::now();
304        loop {
305            if cloned_progress_bar.is_finished() {
306                break;
307            }
308            let num_accumulated = cloned_accum_counter.load(Ordering::Relaxed);
309            assert!(
310                num_accumulated <= num_live_objects,
311                "Accumulated more objects (at least {num_accumulated}) than expected ({num_live_objects})"
312            );
313            let accumulations_per_sec = num_accumulated as f64 / a_instant.elapsed().as_secs_f64();
314            cloned_progress_bar.set_position(num_accumulated);
315            cloned_progress_bar.set_message(format!(
316                "DB live obj accumulations per sec: {}",
317                accumulations_per_sec
318            ));
319            tokio::time::sleep(Duration::from_secs(1)).await;
320        }
321    });
322
323    // Accumulate live objects
324    let mut acc = GlobalStateHash::default();
325    for live_object in iter {
326        match live_object {
327            LiveObject::Normal(object) => {
328                acc.insert(object.compute_object_reference().2);
329            }
330            LiveObject::Wrapped(key) => {
331                acc.insert(
332                    bcs::to_bytes(&WrappedObject::new(key.0, key.1))
333                        .expect("Failed to serialize WrappedObject"),
334                );
335            }
336        }
337        accum_counter.fetch_add(1, Ordering::Relaxed);
338    }
339    accum_progress_bar.finish_with_message("DB live object accumulation completed");
340    handle
341        .await
342        .expect("Failed to join live object accumulation progress monitor");
343    acc
