61    async fn copy_file_with_retry<S: ObjectStoreGetExt, D: ObjectStorePutExt>(
62        src: &Path,
63        dest: &Path,
64        src_store: &S,
65        dest_store: &D,
66        max_retries: usize,
67    ) -> Result<()> {
68        let mut attempts = 0;
69        let max_attempts = max_retries + 1;
70        loop {
71            attempts += 1;
72            match src_store.get_bytes(src).await {
73                Ok(bytes) => {
74                    if bytes.is_empty() {
75                        tracing::warn!("Not copying empty file: {:?}", src);
76                        return Ok(());
77                    }
78                    match dest_store.put_bytes(dest, bytes).await {
79                        Ok(_) => return Ok(()),
80                        Err(e) => {
81                            if attempts >= max_attempts {
82                                return Err(anyhow::anyhow!(
83                                    "Failed to write {} after {} attempts: {}",
84                                    dest,
85                                    attempts,
86                                    e
87                                ));
88                            }
89                            tracing::warn!(
90                                "Failed to write {} (attempt {}/{}): {}, retrying in {}ms",
91                                dest,
92                                attempts,
93                                max_attempts,
94                                e,
95                                1000 * attempts
96                            );
97                            tokio::time::sleep(Duration::from_millis(1000 * attempts as u64)).await;
98                        }
99                    }
100                }
101                Err(e) => {
102                    if attempts >= max_attempts {
103                        return Err(anyhow::anyhow!(
104                            "Failed to download {} after {} attempts: {}",
105                            src,
106                            attempts,
107                            e
108                        ));
109                    }
110                    tracing::warn!(
111                        "Failed to download {} (attempt {}/{}): {}, retrying in {}ms",
112                        src,
113                        attempts,
114                        max_attempts,
115                        e,
116                        1000 * attempts
117                    );
118                    tokio::time::sleep(Duration::from_millis(1000 * attempts as u64)).await;
119                }
120            }
121        }
122    }

124    pub async fn new(
125        epoch: u64,
126        remote_store_config: &ObjectStoreConfig,
127        local_store_config: &ObjectStoreConfig,
128        download_concurrency: NonZeroUsize,
129        m: MultiProgress,
130        skip_reset_local_store: bool,
131        max_retries: usize,
132        num_parallel_chunks: usize,
133    ) -> Result<Self> {
134        let remote_object_store = if remote_store_config.no_sign_request {
135            remote_store_config.make_http()?
136        } else {
137            remote_store_config.make().map(Arc::new)?
138        };
139        let local_object_store: Arc<dyn ObjectStorePutExt> =
140            local_store_config.make().map(Arc::new)?;
141        let local_object_store_list: Arc<dyn ObjectStoreListExt> =
142            local_store_config.make().map(Arc::new)?;
143        let local_staging_dir_root = local_store_config
144            .directory
145            .as_ref()
146            .context("No directory specified")?
147            .clone();
148
149        let local_epoch_dir_name = format!("epoch_{}", epoch);
150        let local_epoch_dir_path = Path::from(local_epoch_dir_name.clone());
151
152        if !skip_reset_local_store {
153            let local_epoch_dir_absolute_path = local_staging_dir_root.join(&local_epoch_dir_name);
154            if local_epoch_dir_absolute_path.exists() {
155                fs::remove_dir_all(&local_epoch_dir_absolute_path)?;
156            }
157            fs::create_dir_all(&local_epoch_dir_absolute_path)?;
158        }
159
160        // Try to download MANIFEST from standard location first, then archive
161        let standard_epoch_dir = Path::from(format!("epoch_{}", epoch));
162        let archive_epoch_dir = Path::from(format!("archive/epoch_{}", epoch));
163
164        let standard_manifest_path = standard_epoch_dir.child("MANIFEST");
165        let archive_manifest_path = archive_epoch_dir.child("MANIFEST");
166
167        // We always download to local epoch dir's MANIFEST
168        let local_manifest_path = local_epoch_dir_path.child("MANIFEST");
169
170        let (remote_epoch_prefix, manifest_download_result) = match Self::copy_file_with_retry(
171            &standard_manifest_path,
172            &local_manifest_path,
173            &remote_object_store,
174            &local_object_store,
175            max_retries,
176        )
177        .await
178        {
179            Ok(_) => (standard_epoch_dir, Ok(())),
180            Err(_) => {
181                // Try archive
182                match Self::copy_file_with_retry(
183                    &archive_manifest_path,
184                    &local_manifest_path,
185                    &remote_object_store,
186                    &local_object_store,
187                    max_retries,
188                )
189                .await
190                {
191                    Ok(_) => (archive_epoch_dir, Ok(())),
192                    Err(e) => (standard_epoch_dir, Err(e)), // Return standard dir but error
193                }
194            }
195        };
196
197        manifest_download_result?;
198
199        let manifest = Self::read_manifest(path_to_filesystem(
200            local_staging_dir_root.clone(),
201            &local_manifest_path,
202        )?)?;
203        let snapshot_version = manifest.snapshot_version();
204        if snapshot_version != 1u8 {
205            return Err(anyhow!("Unexpected snapshot version: {}", snapshot_version));
206        }
207        if manifest.address_length() as usize > ObjectID::LENGTH {
208            return Err(anyhow!(
209                "Max possible address length is: {}",
210                ObjectID::LENGTH
211            ));
212        }
213        if manifest.epoch() != epoch {
214            return Err(anyhow!("Download manifest is not for epoch: {}", epoch,));
215        }
216        let mut object_files = BTreeMap::new();
217        let mut ref_files = BTreeMap::new();
218        for file_metadata in manifest.file_metadata() {
219            match file_metadata.file_type {
220                FileType::Object => {
221                    let entry = object_files
222                        .entry(file_metadata.bucket_num)
223                        .or_insert_with(BTreeMap::new);
224                    entry.insert(file_metadata.part_num, file_metadata.clone());
225                }
226                FileType::Reference => {
227                    let entry = ref_files
228                        .entry(file_metadata.bucket_num)
229                        .or_insert_with(BTreeMap::new);
230                    entry.insert(file_metadata.part_num, file_metadata.clone());
231                }
232            }
233        }
234
235        let mut src_files = Vec::new();
236        let mut dest_files = Vec::new();
237
238        let existing_files = if skip_reset_local_store {
239            let mut existing = std::collections::HashSet::new();
240            let mut list_stream = local_object_store_list
241                .list_objects(Some(&local_epoch_dir_path))
242                .await;
243            while let Some(Ok(meta)) = list_stream.next().await {
244                existing.insert(meta.location);
245            }
246            Some(existing)
247        } else {
248            None
249        };
250
251        for entry in ref_files.values() {
252            for file_metadata in entry.values() {
253                let dest = file_metadata.file_path(&local_epoch_dir_path);
254                if existing_files
255                    .as_ref()
256                    .is_some_and(|existing| existing.contains(&dest))
257                {
258                    continue;
259                }
260                src_files.push(file_metadata.file_path(&remote_epoch_prefix));
261                dest_files.push(dest);
262            }
263        }
264
265        let progress_bar = m.add(
266            ProgressBar::new(src_files.len() as u64).with_style(
267                ProgressStyle::with_template(
268                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} missing .ref files done ({msg})",
269                )
270                .unwrap(),
271            ),
272        );
273        copy_files(
274            &src_files,
275            &dest_files,
276            &remote_object_store,
277            &local_object_store,
278            download_concurrency,
279            Some(progress_bar.clone()),
280        )
281        .await?;
282        progress_bar.finish_with_message("Missing ref files download complete");
283        Ok(StateSnapshotReaderV1 {
284            epoch,
285            local_staging_dir_root,
286            remote_object_store,
287            local_object_store,
288            ref_files,
289            object_files,
290            m,
291            concurrency: download_concurrency.get(),
292            num_parallel_chunks,
293            max_retries,
294            remote_epoch_prefix,
295        })
296    }

298    pub async fn read(
299        &mut self,
300        perpetual_db: Arc<AuthorityPerpetualTables>,
301        abort_registration: AbortRegistration,
302        sender: Option<tokio::sync::mpsc::Sender<(GlobalStateHash, u64)>>,
303    ) -> Result<()> {
304        // This computes and stores the sha3 digest of object references in REFERENCE file for each
305        // bucket partition. When downloading objects, we will match sha3 digest of object references
306        // per *.obj file against this. We do this so during restore we can pre fetch object
307        // references and start building state accumulator and fail early if the state root hash
308        // doesn't match but we still need to ensure that objects match references exactly.
309        let (sha3_digests, num_part_files) = self.compute_checksum().await?;
310        let accum_handle =
311            sender.map(|sender| self.spawn_accumulation_tasks(sender, num_part_files));
312        self.sync_live_objects(perpetual_db.clone(), abort_registration, sha3_digests)
313            .await?;
314        if let Some(handle) = accum_handle {
315            handle.await?;
316        }
317        Ok(())
318    }

320    pub async fn compute_checksum(
321        &mut self,
322    ) -> Result<(Arc<Mutex<BTreeMap<u32, BTreeMap<u32, [u8; 32]>>>>, usize), anyhow::Error> {
323        let sha3_digests: Arc<Mutex<DigestByBucketAndPartition>> =
324            Arc::new(Mutex::new(BTreeMap::new()));
325
326        let num_part_files = self
327            .ref_files
328            .values()
329            .map(|part_files| part_files.len())
330            .sum::<usize>();
331
332        // Generate checksums
333        info!("Computing checksums");
334        let checksum_progress_bar = self.m.add(
335            ProgressBar::new(num_part_files as u64).with_style(
336                ProgressStyle::with_template(
337                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} ref files checksummed ({msg})",
338                )
339                .unwrap(),
340            ),
341        );
342
343        let ref_files_iter = self.ref_files.clone().into_iter();
344        futures::stream::iter(ref_files_iter)
345            .flat_map(|(bucket, part_files)| {
346                futures::stream::iter(
347                    part_files
348                        .into_iter()
349                        .map(move |(part, part_file)| (bucket, part, part_file)),
350                )
351            })
352            .try_for_each_spawned(self.concurrency, |(bucket, part, _part_file)| {
353                let sha3_digests = sha3_digests.clone();
354                let object_files = self.object_files.clone();
355                let bar = checksum_progress_bar.clone();
356                let this = self.clone();
357
358                async move {
359                    let ref_iter = this.ref_iter(bucket, part)?;
360                    let mut hasher = Sha3_256::default();
361                    let mut empty = true;
362
363                    object_files
364                        .get(&bucket)
365                        .context(format!("No bucket exists for: {bucket}"))?
366                        .get(&part)
367                        .context(format!("No part exists for bucket: {bucket}, part: {part}"))?;
368
369                    for object_ref in ref_iter {
370                        hasher.update(object_ref.2.inner());
371                        empty = false;
372                    }
373
374                    if !empty {
375                        let mut digests = sha3_digests.lock().await;
376                        digests
377                            .entry(bucket)
378                            .or_insert(BTreeMap::new())
379                            .entry(part)
380                            .or_insert(hasher.finalize().digest);
381                    }
382
383                    bar.inc(1);
384                    bar.set_message(format!("Bucket: {}, Part: {}", bucket, part));
385                    Ok::<(), anyhow::Error>(())
386                }
387            })
388            .await?;
389        checksum_progress_bar.finish_with_message("Checksumming complete");
390        Ok((sha3_digests, num_part_files))
391    }

393    fn spawn_accumulation_tasks(
394        &self,
395        sender: tokio::sync::mpsc::Sender<(GlobalStateHash, u64)>,
396        num_part_files: usize,
397    ) -> JoinHandle<()> {
398        // Spawn accumulation progress bar
399        let concurrency = self.concurrency;
400        let accum_counter = Arc::new(AtomicU64::new(0));
401        let cloned_accum_counter = accum_counter.clone();
402        let accum_progress_bar = self.m.add(
403             ProgressBar::new(num_part_files as u64).with_style(
404                 ProgressStyle::with_template(
405                     "[{elapsed_precise}] {wide_bar} {pos} out of {len} ref files accumulated from snapshot ({msg})",
406                 )
407                 .unwrap(),
408             ),
409         );
410        let cloned_accum_progress_bar = accum_progress_bar.clone();
411        tokio::spawn(async move {
412            let a_instant = Instant::now();
413            loop {
414                if cloned_accum_progress_bar.is_finished() {
415                    break;
416                }
417                let num_partitions = cloned_accum_counter.load(Ordering::Relaxed);
418                let total_partitions_per_sec =
419                    num_partitions as f64 / a_instant.elapsed().as_secs_f64();
420                cloned_accum_progress_bar.set_position(num_partitions);
421                cloned_accum_progress_bar.set_message(format!(
422                    "file partitions per sec: {}",
423                    total_partitions_per_sec
424                ));
425                tokio::time::sleep(Duration::from_secs(1)).await;
426            }
427        });
428
429        // spawn accumualation task
430        let ref_files = self.ref_files.clone();
431        let epoch_dir = self.epoch_dir();
432        let local_staging_dir_root = self.local_staging_dir_root.clone();
433        tokio::task::spawn(async move {
434            let local_staging_dir_root_clone = local_staging_dir_root.clone();
435            let epoch_dir_clone = epoch_dir.clone();
436            for (bucket, part_files) in ref_files.clone().iter() {
437                futures::stream::iter(part_files.iter())
438                    .map(|(part, _part_files)| {
439                        // TODO depending on concurrency limit here, we may be
440                        // materializing too many refs into memory at once.
441                        // This is only done because ObjectRefIter is not Send
442                        let obj_digests = {
443                            let file_metadata = ref_files
444                                .get(bucket)
445                                .expect("No ref files found for bucket: {bucket_num}")
446                                .get(part)
447                                .expect(
448                                    "No ref files found for bucket: {bucket_num}, part: {part_num}",
449                                );
450                            ObjectRefIter::new(
451                                file_metadata,
452                                local_staging_dir_root_clone.clone(),
453                                epoch_dir_clone.clone(),
454                            )
455                            .expect("Failed to create object ref iter")
456                        }
457                        .map(|obj_ref| obj_ref.2)
458                        .collect::<Vec<ObjectDigest>>();
459                        let sender_clone = sender.clone();
460                        tokio::spawn(async move {
461                            let mut partial_acc = GlobalStateHash::default();
462                            let num_objects = obj_digests.len();
463                            partial_acc.insert_all(obj_digests);
464                            sender_clone
465                                .send((partial_acc, num_objects as u64))
466                                .await
467                                .expect("Unable to send accumulator from snapshot reader");
468                        })
469                    })
470                    .boxed()
471                    .buffer_unordered(concurrency)
472                    .for_each(|result| {
473                        result.expect("Failed to generate partial accumulator");
474                        accum_counter.fetch_add(1, Ordering::Relaxed);
475                        futures::future::ready(())
476                    })
477                    .await;
478            }
479            accum_progress_bar.finish_with_message("Accumulation complete");
480        })
481    }

483    async fn sync_live_objects(
484        &self,
485        perpetual_db: Arc<AuthorityPerpetualTables>,
486        abort_registration: AbortRegistration,
487        sha3_digests: Arc<Mutex<DigestByBucketAndPartition>>,
488    ) -> Result<(), anyhow::Error> {
489        let epoch_dir = self.remote_epoch_prefix.clone();
490        let concurrency = self.concurrency;
491        let num_parallel_chunks = self.num_parallel_chunks;
492        let remote_object_store = self.remote_object_store.clone();
493        let input_files: Vec<_> = self
494            .object_files
495            .iter()
496            .flat_map(|(bucket, parts)| {
497                parts
498                    .clone()
499                    .into_iter()
500                    .map(|entry| (bucket, entry))
501                    .collect::<Vec<_>>()
502            })
503            .collect();
504        let obj_progress_bar = self.m.add(
505            ProgressBar::new(input_files.len() as u64).with_style(
506                ProgressStyle::with_template(
507                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} .obj files done ({msg})",
508                )
509                .unwrap(),
510            ),
511        );
512        let obj_progress_bar_clone = obj_progress_bar.clone();
513        let instant = Instant::now();
514        let downloaded_bytes = Arc::new(AtomicUsize::new(0));
515
516        let ret = Abortable::new(
517            async move {
518                futures::stream::iter(input_files.iter())
519                    .map(|(bucket, (part_num, file_metadata))| {
520                        let epoch_dir_clone = epoch_dir.clone();
521                        let remote_object_store_clone = remote_object_store.clone();
522                        let sha3_digests_clone = sha3_digests.clone();
523                        async move {
524                            // Download object file with retries
525                            let (bytes, sha3_digest) = download_bytes(
526                                remote_object_store_clone,
527                                file_metadata,
528                                epoch_dir_clone,
529                                sha3_digests_clone,
530                                bucket,
531                                part_num,
532                                None,
533                            )
534                            .await;
535                            Ok::<(Bytes, FileMetadata, [u8; 32]), anyhow::Error>((
536                                bytes,
537                                (*file_metadata).clone(),
538                                sha3_digest,
539                            ))
540                        }
541                    })
542                    .boxed()
543                    .buffer_unordered(concurrency)
544                    .try_for_each(|(bytes, file_metadata, sha3_digest)| {
545                        let perpetual_db = perpetual_db.clone();
546                        let obj_progress_bar_clone = obj_progress_bar_clone.clone();
547                        let downloaded_bytes = downloaded_bytes.clone();
548                        async move {
549                            let bytes_len = bytes.len();
550                            let objects: Vec<LiveObject> =
551                                LiveObjectIter::new(&file_metadata, bytes)?.collect();
552                            AuthorityStore::bulk_insert_live_objects(
553                                perpetual_db,
554                                objects,
555                                &sha3_digest,
556                                num_parallel_chunks,
557                            )
558                            .await?;
559                            downloaded_bytes.fetch_add(bytes_len, Ordering::Relaxed);
560                            obj_progress_bar_clone.inc(1);
561                            obj_progress_bar_clone.set_message(format!(
562                                "Download speed: {} MiB/s",
563                                downloaded_bytes.load(Ordering::Relaxed) as f64
564                                    / (1024 * 1024) as f64
565                                    / instant.elapsed().as_secs_f64(),
566                            ));
567                            Ok(())
568                        }
569                    })
570                    .await
571            },
572            abort_registration,
573        )
574        .await?;
575        obj_progress_bar.finish_with_message("Objects download complete");
576        ret
577    }

632    fn read_manifest(path: PathBuf) -> anyhow::Result<Manifest> {
633        let manifest_file = File::open(path)?;
634        let manifest_file_size = manifest_file.metadata()?.len() as usize;
635        let mut manifest_reader = BufReader::new(manifest_file);
636        manifest_reader.rewind()?;
637        let magic = manifest_reader.read_u32::<BigEndian>()?;
638        if magic != MANIFEST_FILE_MAGIC {
639            return Err(anyhow!("Unexpected magic byte: {}", magic));
640        }
641        manifest_reader.seek(SeekFrom::End(-(SHA3_BYTES as i64)))?;
642        let mut sha3_digest = [0u8; SHA3_BYTES];
643        manifest_reader.read_exact(&mut sha3_digest)?;
644        manifest_reader.rewind()?;
645        let mut content_buf = vec![0u8; manifest_file_size - SHA3_BYTES];
646        manifest_reader.read_exact(&mut content_buf)?;
647        let mut hasher = Sha3_256::default();
648        hasher.update(&content_buf);
649        let computed_digest = hasher.finalize().digest;
650        if computed_digest != sha3_digest {
651            return Err(anyhow!(
652                "Checksum: {:?} don't match: {:?}",
653                computed_digest,
654                sha3_digest
655            ));
656        }
657        manifest_reader.rewind()?;
658        manifest_reader.seek(SeekFrom::Start(MAGIC_BYTES as u64))?;
659        let manifest = bcs::from_bytes(&content_buf[MAGIC_BYTES..])?;
660        Ok(manifest)
661    }
662
663    pub fn get_multi_progress(&self) -> MultiProgress {
664        self.m.clone()
665    }
666}

668pub async fn download_bytes(
669    remote_object_store: Arc<dyn ObjectStoreGetExt>,
670    file_metadata: &FileMetadata,
671    epoch_dir: Path,
672    sha3_digests: Sha3DigestType,
673    bucket: &&u32,
674    part_num: &u32,
675    max_timeout_secs: Option<u64>,
676) -> (Bytes, [u8; 32]) {
677    const DOWNLOAD_TIMEOUT: Duration = Duration::from_secs(300); // 5 minutes for large files
678    const INITIAL_BACKOFF: Duration = Duration::from_secs(3);
679
680    let backoff_cap = Duration::from_secs(max_timeout_secs.unwrap_or(60));
681    let mut backoff = INITIAL_BACKOFF;
682    let mut attempts = 0usize;
683    let file_path = file_metadata.file_path(&epoch_dir);
684    let bytes = loop {
685        debug!(
686            "Downloading obj file: {:?} (attempt {}, timeout {:?})",
687            file_path, attempts, DOWNLOAD_TIMEOUT
688        );
689
690        match tokio::time::timeout(DOWNLOAD_TIMEOUT, remote_object_store.get_bytes(&file_path))
691            .await
692        {
693            Ok(Ok(bytes)) => {
694                break bytes;
695            }
696            Ok(Err(err)) => {
697                error!(
698                    "Failed to download {}: {} (attempt {})",
699                    file_path, err, attempts,
700                );
701            }
702            Err(_) => {
703                error!(
704                    "Download timed out for {} after {:?} (attempt {})",
705                    file_path, DOWNLOAD_TIMEOUT, attempts,
706                );
707            }
708        }
709
710        attempts += 1;
711        debug!("Retrying {} in {:?}...", file_path, backoff);
712        tokio::time::sleep(backoff).await;
713
714        // Exponential backoff with 1.5x multiplier, capped at backoff_cap
715        backoff += backoff / 2;
716        backoff = std::cmp::min(backoff, backoff_cap);
717    };
718
719    let sha3_digest = sha3_digests.lock().await;
720    let bucket_map = sha3_digest
721        .get(bucket)
722        .expect("Bucket not in digest map")
723        .clone();
724    let sha3_digest = *bucket_map
725        .get(part_num)
726        .expect("sha3 digest not in bucket map");
727    (bytes, sha3_digest)
728}

730/// An iterator over all object refs in a .ref file.
731pub struct ObjectRefIter {
732    reader: Box<dyn Read>,
733}
734
735impl ObjectRefIter {
736    pub fn new(file_metadata: &FileMetadata, root_path: PathBuf, dir_path: Path) -> Result<Self> {
737        let file_path = file_metadata.local_file_path(&root_path, &dir_path)?;
738        let mut reader = file_metadata.file_compression.decompress(&file_path)?;
739        let magic = reader.read_u32::<BigEndian>()?;
740        if magic != REFERENCE_FILE_MAGIC {
741            Err(anyhow!(
742                "Unexpected magic string in REFERENCE file: {:?}",
743                magic
744            ))
745        } else {
746            Ok(ObjectRefIter { reader })
747        }
748    }
749
750    fn next_ref(&mut self) -> Result<ObjectRef> {
751        let mut buf = [0u8; OBJECT_REF_BYTES];
752        self.reader.read_exact(&mut buf)?;
753        let object_id = &buf[0..OBJECT_ID_BYTES];
754        let sequence_number = &buf[OBJECT_ID_BYTES..OBJECT_ID_BYTES + SEQUENCE_NUM_BYTES]
755            .reader()
756            .read_u64::<BigEndian>()?;
757        let sha3_digest = &buf[OBJECT_ID_BYTES + SEQUENCE_NUM_BYTES..OBJECT_REF_BYTES];
758        let object_ref: ObjectRef = (
759            ObjectID::from_bytes(object_id)?,
760            SequenceNumber::from_u64(*sequence_number),
761            ObjectDigest::try_from(sha3_digest)?,
762        );
763        Ok(object_ref)
764    }
765}
766
767impl Iterator for ObjectRefIter {
768    type Item = ObjectRef;
769    fn next(&mut self) -> Option<Self::Item> {
770        self.next_ref().ok()
771    }
772}

774/// An iterator over all objects in a *.obj file.
775pub struct LiveObjectIter {
776    reader: Box<dyn Read>,
777}
778
779impl LiveObjectIter {
780    pub fn new(file_metadata: &FileMetadata, bytes: Bytes) -> Result<Self> {
781        let mut reader = file_metadata.file_compression.bytes_decompress(bytes)?;
782        let magic = reader.read_u32::<BigEndian>()?;
783        if magic != OBJECT_FILE_MAGIC {
784            Err(anyhow!(
785                "Unexpected magic string in object file: {:?}",
786                magic
787            ))
788        } else {
789            Ok(LiveObjectIter { reader })
790        }
791    }
792
793    fn next_object(&mut self) -> Result<LiveObject> {
794        let len = self.reader.read_varint::<u64>()? as usize;
795        if len == 0 {
796            return Err(anyhow!("Invalid object length of 0 in file"));
797        }
798        let encoding = self.reader.read_u8()?;
799        let mut data = vec![0u8; len];
800        self.reader.read_exact(&mut data)?;
801        let blob = Blob {
802            data,
803            encoding: BlobEncoding::try_from(encoding)?,
804        };
805        blob.decode()
806    }
807}
808
809impl Iterator for LiveObjectIter {
810    type Item = LiveObject;
811    fn next(&mut self) -> Option<Self::Item> {
812        self.next_object().ok()
813    }
