59    async fn copy_file_with_retry<S: ObjectStoreGetExt, D: ObjectStorePutExt>(
60        src: &Path,
61        dest: &Path,
62        src_store: &S,
63        dest_store: &D,
64        max_retries: usize,
65    ) -> Result<()> {
66        let mut attempts = 0;
67        let max_attempts = max_retries + 1;
68        loop {
69            attempts += 1;
70            match copy_file(src, dest, src_store, dest_store).await {
71                Ok(()) => return Ok(()),
72                Err(e) if attempts >= max_attempts => {
73                    return Err(anyhow::anyhow!(
74                        "Failed to download {} after {} attempts: {}",
75                        src,
76                        attempts,
77                        e
78                    ));
79                }
80                Err(e) => {
81                    tracing::warn!(
82                        "Failed to download {} (attempt {}/{}): {}, retrying in {}ms",
83                        src,
84                        attempts,
85                        max_attempts,
86                        e,
87                        1000 * attempts
88                    );
89                    tokio::time::sleep(Duration::from_millis(1000 * attempts as u64)).await;
90                }
91            }
92        }
93    }

95    pub async fn new(
96        epoch: u64,
97        remote_store_config: &ObjectStoreConfig,
98        local_store_config: &ObjectStoreConfig,
99        download_concurrency: NonZeroUsize,
100        m: MultiProgress,
101        skip_reset_local_store: bool,
102        max_retries: usize,
103    ) -> Result<Self> {
104        let epoch_dir = format!("epoch_{}", epoch);
105        let remote_object_store = if remote_store_config.no_sign_request {
106            remote_store_config.make_http()?
107        } else {
108            remote_store_config.make().map(Arc::new)?
109        };
110        let local_object_store: Arc<dyn ObjectStorePutExt> =
111            local_store_config.make().map(Arc::new)?;
112        let local_object_store_list: Arc<dyn ObjectStoreListExt> =
113            local_store_config.make().map(Arc::new)?;
114        let local_staging_dir_root = local_store_config
115            .directory
116            .as_ref()
117            .context("No directory specified")?
118            .clone();
119        if !skip_reset_local_store {
120            let local_epoch_dir_path = local_staging_dir_root.join(&epoch_dir);
121            if local_epoch_dir_path.exists() {
122                fs::remove_dir_all(&local_epoch_dir_path)?;
123            }
124            fs::create_dir_all(&local_epoch_dir_path)?;
125        }
126        // Download MANIFEST first
127        let manifest_file_path = Path::from(epoch_dir.clone()).child("MANIFEST");
128        Self::copy_file_with_retry(
129            &manifest_file_path,
130            &manifest_file_path,
131            &remote_object_store,
132            &local_object_store,
133            max_retries,
134        )
135        .await?;
136        let manifest = Self::read_manifest(path_to_filesystem(
137            local_staging_dir_root.clone(),
138            &manifest_file_path,
139        )?)?;
140        let snapshot_version = manifest.snapshot_version();
141        if snapshot_version != 1u8 {
142            return Err(anyhow!("Unexpected snapshot version: {}", snapshot_version));
143        }
144        if manifest.address_length() as usize > ObjectID::LENGTH {
145            return Err(anyhow!(
146                "Max possible address length is: {}",
147                ObjectID::LENGTH
148            ));
149        }
150        if manifest.epoch() != epoch {
151            return Err(anyhow!("Download manifest is not for epoch: {}", epoch,));
152        }
153        let mut object_files = BTreeMap::new();
154        let mut ref_files = BTreeMap::new();
155        for file_metadata in manifest.file_metadata() {
156            match file_metadata.file_type {
157                FileType::Object => {
158                    let entry = object_files
159                        .entry(file_metadata.bucket_num)
160                        .or_insert_with(BTreeMap::new);
161                    entry.insert(file_metadata.part_num, file_metadata.clone());
162                }
163                FileType::Reference => {
164                    let entry = ref_files
165                        .entry(file_metadata.bucket_num)
166                        .or_insert_with(BTreeMap::new);
167                    entry.insert(file_metadata.part_num, file_metadata.clone());
168                }
169            }
170        }
171        let epoch_dir_path = Path::from(epoch_dir);
172        let files: Vec<Path> = ref_files
173            .values()
174            .flat_map(|entry| {
175                let files: Vec<_> = entry
176                    .values()
177                    .map(|file_metadata| file_metadata.file_path(&epoch_dir_path))
178                    .collect();
179                files
180            })
181            .collect();
182
183        let files_to_download = if skip_reset_local_store {
184            let mut list_stream = local_object_store_list
185                .list_objects(Some(&epoch_dir_path))
186                .await;
187            let mut existing_files = std::collections::HashSet::new();
188            while let Some(Ok(meta)) = list_stream.next().await {
189                existing_files.insert(meta.location);
190            }
191            let mut missing_files = Vec::new();
192            for file in &files {
193                if !existing_files.contains(file) {
194                    missing_files.push(file.clone());
195                }
196            }
197            missing_files
198        } else {
199            files
200        };
201        let progress_bar = m.add(
202            ProgressBar::new(files_to_download.len() as u64).with_style(
203                ProgressStyle::with_template(
204                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} missing .ref files done ({msg})",
205                )
206                .unwrap(),
207            ),
208        );
209        copy_files(
210            &files_to_download,
211            &files_to_download,
212            &remote_object_store,
213            &local_object_store,
214            download_concurrency,
215            Some(progress_bar.clone()),
216        )
217        .await?;
218        progress_bar.finish_with_message("Missing ref files download complete");
219        Ok(StateSnapshotReaderV1 {
220            epoch,
221            local_staging_dir_root,
222            remote_object_store,
223            local_object_store,
224            ref_files,
225            object_files,
226            m,
227            concurrency: download_concurrency.get(),
228            max_retries,
229        })
230    }

232    pub async fn read(
233        &mut self,
234        perpetual_db: &AuthorityPerpetualTables,
235        abort_registration: AbortRegistration,
236        sender: Option<tokio::sync::mpsc::Sender<(GlobalStateHash, u64)>>,
237    ) -> Result<()> {
238        // This computes and stores the sha3 digest of object references in REFERENCE file for each
239        // bucket partition. When downloading objects, we will match sha3 digest of object references
240        // per *.obj file against this. We do this so during restore we can pre fetch object
241        // references and start building state accumulator and fail early if the state root hash
242        // doesn't match but we still need to ensure that objects match references exactly.
243        let (sha3_digests, num_part_files) = self.compute_checksum().await?;
244        let accum_handle =
245            sender.map(|sender| self.spawn_accumulation_tasks(sender, num_part_files));
246        self.sync_live_objects(perpetual_db, abort_registration, sha3_digests)
247            .await?;
248        if let Some(handle) = accum_handle {
249            handle.await?;
250        }
251        Ok(())
252    }

254    pub async fn compute_checksum(
255        &mut self,
256    ) -> Result<(Arc<Mutex<BTreeMap<u32, BTreeMap<u32, [u8; 32]>>>>, usize), anyhow::Error> {
257        let sha3_digests: Arc<Mutex<DigestByBucketAndPartition>> =
258            Arc::new(Mutex::new(BTreeMap::new()));
259
260        let num_part_files = self
261            .ref_files
262            .values()
263            .map(|part_files| part_files.len())
264            .sum::<usize>();
265
266        // Generate checksums
267        info!("Computing checksums");
268        let checksum_progress_bar = self.m.add(
269            ProgressBar::new(num_part_files as u64).with_style(
270                ProgressStyle::with_template(
271                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} ref files checksummed ({msg})",
272                )
273                .unwrap(),
274            ),
275        );
276
277        let ref_files_iter = self.ref_files.clone().into_iter();
278        futures::stream::iter(ref_files_iter)
279            .flat_map(|(bucket, part_files)| {
280                futures::stream::iter(
281                    part_files
282                        .into_iter()
283                        .map(move |(part, part_file)| (bucket, part, part_file)),
284                )
285            })
286            .try_for_each_spawned(self.concurrency, |(bucket, part, _part_file)| {
287                let sha3_digests = sha3_digests.clone();
288                let object_files = self.object_files.clone();
289                let bar = checksum_progress_bar.clone();
290                let this = self.clone();
291
292                async move {
293                    let ref_iter = this.ref_iter(bucket, part)?;
294                    let mut hasher = Sha3_256::default();
295                    let mut empty = true;
296
297                    object_files
298                        .get(&bucket)
299                        .context(format!("No bucket exists for: {bucket}"))?
300                        .get(&part)
301                        .context(format!("No part exists for bucket: {bucket}, part: {part}"))?;
302
303                    for object_ref in ref_iter {
304                        hasher.update(object_ref.2.inner());
305                        empty = false;
306                    }
307
308                    if !empty {
309                        let mut digests = sha3_digests.lock().await;
310                        digests
311                            .entry(bucket)
312                            .or_insert(BTreeMap::new())
313                            .entry(part)
314                            .or_insert(hasher.finalize().digest);
315                    }
316
317                    bar.inc(1);
318                    bar.set_message(format!("Bucket: {}, Part: {}", bucket, part));
319                    Ok::<(), anyhow::Error>(())
320                }
321            })
322            .await?;
323        checksum_progress_bar.finish_with_message("Checksumming complete");
324        Ok((sha3_digests, num_part_files))
325    }

327    fn spawn_accumulation_tasks(
328        &self,
329        sender: tokio::sync::mpsc::Sender<(GlobalStateHash, u64)>,
330        num_part_files: usize,
331    ) -> JoinHandle<()> {
332        // Spawn accumulation progress bar
333        let concurrency = self.concurrency;
334        let accum_counter = Arc::new(AtomicU64::new(0));
335        let cloned_accum_counter = accum_counter.clone();
336        let accum_progress_bar = self.m.add(
337             ProgressBar::new(num_part_files as u64).with_style(
338                 ProgressStyle::with_template(
339                     "[{elapsed_precise}] {wide_bar} {pos} out of {len} ref files accumulated from snapshot ({msg})",
340                 )
341                 .unwrap(),
342             ),
343         );
344        let cloned_accum_progress_bar = accum_progress_bar.clone();
345        tokio::spawn(async move {
346            let a_instant = Instant::now();
347            loop {
348                if cloned_accum_progress_bar.is_finished() {
349                    break;
350                }
351                let num_partitions = cloned_accum_counter.load(Ordering::Relaxed);
352                let total_partitions_per_sec =
353                    num_partitions as f64 / a_instant.elapsed().as_secs_f64();
354                cloned_accum_progress_bar.set_position(num_partitions);
355                cloned_accum_progress_bar.set_message(format!(
356                    "file partitions per sec: {}",
357                    total_partitions_per_sec
358                ));
359                tokio::time::sleep(Duration::from_secs(1)).await;
360            }
361        });
362
363        // spawn accumualation task
364        let ref_files = self.ref_files.clone();
365        let epoch_dir = self.epoch_dir();
366        let local_staging_dir_root = self.local_staging_dir_root.clone();
367        tokio::task::spawn(async move {
368            let local_staging_dir_root_clone = local_staging_dir_root.clone();
369            let epoch_dir_clone = epoch_dir.clone();
370            for (bucket, part_files) in ref_files.clone().iter() {
371                futures::stream::iter(part_files.iter())
372                    .map(|(part, _part_files)| {
373                        // TODO depending on concurrency limit here, we may be
374                        // materializing too many refs into memory at once.
375                        // This is only done because ObjectRefIter is not Send
376                        let obj_digests = {
377                            let file_metadata = ref_files
378                                .get(bucket)
379                                .expect("No ref files found for bucket: {bucket_num}")
380                                .get(part)
381                                .expect(
382                                    "No ref files found for bucket: {bucket_num}, part: {part_num}",
383                                );
384                            ObjectRefIter::new(
385                                file_metadata,
386                                local_staging_dir_root_clone.clone(),
387                                epoch_dir_clone.clone(),
388                            )
389                            .expect("Failed to create object ref iter")
390                        }
391                        .map(|obj_ref| obj_ref.2)
392                        .collect::<Vec<ObjectDigest>>();
393                        let sender_clone = sender.clone();
394                        tokio::spawn(async move {
395                            let mut partial_acc = GlobalStateHash::default();
396                            let num_objects = obj_digests.len();
397                            partial_acc.insert_all(obj_digests);
398                            sender_clone
399                                .send((partial_acc, num_objects as u64))
400                                .await
401                                .expect("Unable to send accumulator from snapshot reader");
402                        })
403                    })
404                    .boxed()
405                    .buffer_unordered(concurrency)
406                    .for_each(|result| {
407                        result.expect("Failed to generate partial accumulator");
408                        accum_counter.fetch_add(1, Ordering::Relaxed);
409                        futures::future::ready(())
410                    })
411                    .await;
412            }
413            accum_progress_bar.finish_with_message("Accumulation complete");
414        })
415    }

417    async fn sync_live_objects(
418        &self,
419        perpetual_db: &AuthorityPerpetualTables,
420        abort_registration: AbortRegistration,
421        sha3_digests: Arc<Mutex<DigestByBucketAndPartition>>,
422    ) -> Result<(), anyhow::Error> {
423        let epoch_dir = self.epoch_dir();
424        let concurrency = self.concurrency;
425        let remote_object_store = self.remote_object_store.clone();
426        let input_files: Vec<_> = self
427            .object_files
428            .iter()
429            .flat_map(|(bucket, parts)| {
430                parts
431                    .clone()
432                    .into_iter()
433                    .map(|entry| (bucket, entry))
434                    .collect::<Vec<_>>()
435            })
436            .collect();
437        let obj_progress_bar = self.m.add(
438            ProgressBar::new(input_files.len() as u64).with_style(
439                ProgressStyle::with_template(
440                    "[{elapsed_precise}] {wide_bar} {pos} out of {len} .obj files done ({msg})",
441                )
442                .unwrap(),
443            ),
444        );
445        let obj_progress_bar_clone = obj_progress_bar.clone();
446        let instant = Instant::now();
447        let downloaded_bytes = AtomicUsize::new(0);
448
449        let ret = Abortable::new(
450            async move {
451                futures::stream::iter(input_files.iter())
452                    .map(|(bucket, (part_num, file_metadata))| {
453                        let epoch_dir_clone = epoch_dir.clone();
454                        let remote_object_store_clone = remote_object_store.clone();
455                        let sha3_digests_clone = sha3_digests.clone();
456                        async move {
457                            // Download object file with retries
458                            let (bytes, sha3_digest) = download_bytes(
459                                remote_object_store_clone,
460                                file_metadata,
461                                epoch_dir_clone,
462                                sha3_digests_clone,
463                                bucket,
464                                part_num,
465                                None,
466                            )
467                            .await;
468                            Ok::<(Bytes, FileMetadata, [u8; 32]), anyhow::Error>((
469                                bytes,
470                                (*file_metadata).clone(),
471                                sha3_digest,
472                            ))
473                        }
474                    })
475                    .boxed()
476                    .buffer_unordered(concurrency)
477                    .try_for_each(|(bytes, file_metadata, sha3_digest)| {
478                        let bytes_len = bytes.len();
479                        let result: Result<(), anyhow::Error> =
480                            LiveObjectIter::new(&file_metadata, bytes).map(|obj_iter| {
481                                AuthorityStore::bulk_insert_live_objects(
482                                    perpetual_db,
483                                    obj_iter,
484                                    &sha3_digest,
485                                )
486                                .expect("Failed to insert live objects");
487                            });
488                        downloaded_bytes.fetch_add(bytes_len, Ordering::Relaxed);
489                        obj_progress_bar_clone.inc(1);
490                        obj_progress_bar_clone.set_message(format!(
491                            "Download speed: {} MiB/s",
492                            downloaded_bytes.load(Ordering::Relaxed) as f64
493                                / (1024 * 1024) as f64
494                                / instant.elapsed().as_secs_f64(),
495                        ));
496                        futures::future::ready(result)
497                    })
498                    .await
499            },
500            abort_registration,
501        )
502        .await?;
503        obj_progress_bar.finish_with_message("Objects download complete");
504        ret
505    }

560    fn read_manifest(path: PathBuf) -> anyhow::Result<Manifest> {
561        let manifest_file = File::open(path)?;
562        let manifest_file_size = manifest_file.metadata()?.len() as usize;
563        let mut manifest_reader = BufReader::new(manifest_file);
564        manifest_reader.rewind()?;
565        let magic = manifest_reader.read_u32::<BigEndian>()?;
566        if magic != MANIFEST_FILE_MAGIC {
567            return Err(anyhow!("Unexpected magic byte: {}", magic));
568        }
569        manifest_reader.seek(SeekFrom::End(-(SHA3_BYTES as i64)))?;
570        let mut sha3_digest = [0u8; SHA3_BYTES];
571        manifest_reader.read_exact(&mut sha3_digest)?;
572        manifest_reader.rewind()?;
573        let mut content_buf = vec![0u8; manifest_file_size - SHA3_BYTES];
574        manifest_reader.read_exact(&mut content_buf)?;
575        let mut hasher = Sha3_256::default();
576        hasher.update(&content_buf);
577        let computed_digest = hasher.finalize().digest;
578        if computed_digest != sha3_digest {
579            return Err(anyhow!(
580                "Checksum: {:?} don't match: {:?}",
581                computed_digest,
582                sha3_digest
583            ));
584        }
585        manifest_reader.rewind()?;
586        manifest_reader.seek(SeekFrom::Start(MAGIC_BYTES as u64))?;
587        let manifest = bcs::from_bytes(&content_buf[MAGIC_BYTES..])?;
588        Ok(manifest)
589    }

596pub async fn download_bytes(
597    remote_object_store: Arc<dyn ObjectStoreGetExt>,
598    file_metadata: &FileMetadata,
599    epoch_dir: Path,
600    sha3_digests: Sha3DigestType,
601    bucket: &&u32,
602    part_num: &u32,
603    max_timeout_secs: Option<u64>,
604) -> (Bytes, [u8; 32]) {
605    let max_timeout = Duration::from_secs(max_timeout_secs.unwrap_or(60));
606    let mut timeout = Duration::from_secs(2);
607    timeout += timeout / 2;
608    timeout = std::cmp::min(max_timeout, timeout);
609    let mut attempts = 0usize;
610    let file_path = file_metadata.file_path(&epoch_dir);
611    let bytes = loop {
612        match remote_object_store.get_bytes(&file_path).await {
613            Ok(bytes) => {
614                break bytes;
615            }
616            Err(err) => {
617                error!(
618                    "Obj {} .get failed (attempt {}): {}",
619                    file_metadata.file_path(&epoch_dir),
620                    attempts,
621                    err,
622                );
623                if timeout > max_timeout {
624                    panic!("Failed to get obj file after {} attempts", attempts);
625                } else {
626                    attempts += 1;
627                    tokio::time::sleep(timeout).await;
628                    timeout += timeout / 2;
629                    continue;
630                }
631            }
632        }
633    };
634    let sha3_digest = sha3_digests.lock().await;
635    let bucket_map = sha3_digest
636        .get(bucket)
637        .expect("Bucket not in digest map")
638        .clone();
639    let sha3_digest = *bucket_map
640        .get(part_num)
641        .expect("sha3 digest not in bucket map");
642    (bytes, sha3_digest)
643}

645/// An iterator over all object refs in a .ref file.
646pub struct ObjectRefIter {
647    reader: Box<dyn Read>,
648}
649
650impl ObjectRefIter {
651    pub fn new(file_metadata: &FileMetadata, root_path: PathBuf, dir_path: Path) -> Result<Self> {
652        let file_path = file_metadata.local_file_path(&root_path, &dir_path)?;
653        let mut reader = file_metadata.file_compression.decompress(&file_path)?;
654        let magic = reader.read_u32::<BigEndian>()?;
655        if magic != REFERENCE_FILE_MAGIC {
656            Err(anyhow!(
657                "Unexpected magic string in REFERENCE file: {:?}",
658                magic
659            ))
660        } else {
661            Ok(ObjectRefIter { reader })
662        }
663    }
664
665    fn next_ref(&mut self) -> Result<ObjectRef> {
666        let mut buf = [0u8; OBJECT_REF_BYTES];
667        self.reader.read_exact(&mut buf)?;
668        let object_id = &buf[0..OBJECT_ID_BYTES];
669        let sequence_number = &buf[OBJECT_ID_BYTES..OBJECT_ID_BYTES + SEQUENCE_NUM_BYTES]
670            .reader()
671            .read_u64::<BigEndian>()?;
672        let sha3_digest = &buf[OBJECT_ID_BYTES + SEQUENCE_NUM_BYTES..OBJECT_REF_BYTES];
673        let object_ref: ObjectRef = (
674            ObjectID::from_bytes(object_id)?,
675            SequenceNumber::from_u64(*sequence_number),
676            ObjectDigest::try_from(sha3_digest)?,
677        );
678        Ok(object_ref)
679    }
680}
681
682impl Iterator for ObjectRefIter {
683    type Item = ObjectRef;
684    fn next(&mut self) -> Option<Self::Item> {
685        self.next_ref().ok()
686    }
687}

689/// An iterator over all objects in a *.obj file.
690pub struct LiveObjectIter {
691    reader: Box<dyn Read>,
692}
693
694impl LiveObjectIter {
695    pub fn new(file_metadata: &FileMetadata, bytes: Bytes) -> Result<Self> {
696        let mut reader = file_metadata.file_compression.bytes_decompress(bytes)?;
697        let magic = reader.read_u32::<BigEndian>()?;
698        if magic != OBJECT_FILE_MAGIC {
699            Err(anyhow!(
700                "Unexpected magic string in object file: {:?}",
701                magic
702            ))
703        } else {
704            Ok(LiveObjectIter { reader })
705        }
706    }
707
708    fn next_object(&mut self) -> Result<LiveObject> {
709        let len = self.reader.read_varint::<u64>()? as usize;
710        if len == 0 {
711            return Err(anyhow!("Invalid object length of 0 in file"));
712        }
713        let encoding = self.reader.read_u8()?;
714        let mut data = vec![0u8; len];
715        self.reader.read_exact(&mut data)?;
716        let blob = Blob {
717            data,
718            encoding: BlobEncoding::try_from(encoding)?,
719        };
720        blob.decode()
721    }
722}
723
724impl Iterator for LiveObjectIter {
725    type Item = LiveObject;
726    fn next(&mut self) -> Option<Self::Item> {
727        self.next_object().ok()
728    }
729}
