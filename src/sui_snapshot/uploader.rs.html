29pub struct StateSnapshotUploaderMetrics {
30    pub first_missing_state_snapshot_epoch: IntGauge,
31    pub state_snapshot_upload_err: IntCounter,
32}
33
34impl StateSnapshotUploaderMetrics {
35    pub fn new(registry: &Registry) -> Arc<Self> {
36        let this = Self {
37            first_missing_state_snapshot_epoch: register_int_gauge_with_registry!(
38                "first_missing_state_snapshot_epoch",
39                "First epoch for which we have no state snapshot in remote store",
40                registry
41            )
42            .unwrap(),
43            state_snapshot_upload_err: register_int_counter_with_registry!(
44                "state_snapshot_upload_err",
45                "Track upload errors we can alert on",
46                registry
47            )
48            .unwrap(),
49        };
50        Arc::new(this)
51    }
52}

54pub struct StateSnapshotUploader {
55    /// Directory path on local disk where db checkpoints are stored
56    db_checkpoint_path: PathBuf,
57    /// Store on local disk where db checkpoints are written to
58    db_checkpoint_store: Arc<DynObjectStore>,
59    /// Checkpoint store; needed to fetch epoch state commitments for verification
60    checkpoint_store: Arc<CheckpointStore>,
61    /// Directory path on local disk where state snapshots are staged for upload
62    staging_path: PathBuf,
63    /// Store on local disk where state snapshots are staged for upload
64    staging_store: Arc<DynObjectStore>,
65    /// Remote store i.e. S3, GCS, etc where state snapshots are uploaded to
66    snapshot_store: Arc<DynObjectStore>,
67    /// Time interval to check for presence of new db checkpoint
68    interval: Duration,
69    metrics: Arc<StateSnapshotUploaderMetrics>,
70    /// The chain identifier is derived from the genesis checkpoint and used to identify the
71    /// network.
72    chain_identifier: ChainIdentifier,
73}
74
75impl StateSnapshotUploader {
76    pub fn new(
77        db_checkpoint_path: &std::path::Path,
78        staging_path: &std::path::Path,
79        snapshot_store_config: ObjectStoreConfig,
80        interval_s: u64,
81        registry: &Registry,
82        checkpoint_store: Arc<CheckpointStore>,
83        chain_identifier: ChainIdentifier,
84    ) -> Result<Arc<Self>> {
85        let db_checkpoint_store_config = ObjectStoreConfig {
86            object_store: Some(ObjectStoreType::File),
87            directory: Some(db_checkpoint_path.to_path_buf()),
88            ..Default::default()
89        };
90        let staging_store_config = ObjectStoreConfig {
91            object_store: Some(ObjectStoreType::File),
92            directory: Some(staging_path.to_path_buf()),
93            ..Default::default()
94        };
95        Ok(Arc::new(StateSnapshotUploader {
96            db_checkpoint_path: db_checkpoint_path.to_path_buf(),
97            db_checkpoint_store: db_checkpoint_store_config.make()?,
98            checkpoint_store,
99            staging_path: staging_path.to_path_buf(),
100            staging_store: staging_store_config.make()?,
101            snapshot_store: snapshot_store_config.make()?,
102            interval: Duration::from_secs(interval_s),
103            metrics: StateSnapshotUploaderMetrics::new(registry),
104            chain_identifier,
105        }))
106    }
107
108    pub fn start(self: Arc<Self>) -> tokio::sync::broadcast::Sender<()> {
109        let (kill_sender, _kill_receiver) = tokio::sync::broadcast::channel::<()>(1);
110        tokio::task::spawn(Self::run_upload_loop(self.clone(), kill_sender.subscribe()));
111        tokio::task::spawn(run_manifest_update_loop(
112            self.snapshot_store.clone(),
113            kill_sender.subscribe(),
114        ));
115        kill_sender
116    }
117
118    async fn upload_state_snapshot_to_object_store(&self, missing_epochs: Vec<u64>) -> Result<()> {
119        let last_missing_epoch = missing_epochs.last().cloned().unwrap_or(0);
120        let local_checkpoints_by_epoch =
121            find_all_dirs_with_epoch_prefix(&self.db_checkpoint_store, None).await?;
122        let mut dirs: Vec<_> = local_checkpoints_by_epoch.iter().collect();
123        dirs.sort_by_key(|(epoch_num, _path)| *epoch_num);
124        for (epoch, db_path) in dirs {
125            if missing_epochs.contains(epoch) || *epoch >= last_missing_epoch {
126                info!("Starting state snapshot creation for epoch: {}", *epoch);
127                let state_snapshot_writer = StateSnapshotWriterV1::new_from_store(
128                    &self.staging_path,
129                    &self.staging_store,
130                    &self.snapshot_store,
131                    FileCompression::Zstd,
132                    NonZeroUsize::new(20).unwrap(),
133                )
134                .await?;
135                let db = Arc::new(AuthorityPerpetualTables::open(
136                    &path_to_filesystem(self.db_checkpoint_path.clone(), &db_path.child("store"))?,
137                    None,
138                    None,
139                ));
140                let commitments = self
141                    .checkpoint_store
142                    .get_epoch_state_commitments(*epoch)
143                    .expect("Expected last checkpoint of epoch to have end of epoch data")
144                    .expect("Expected end of epoch data to be present");
145                let state_hash_commitment = match commitments
146                    .last()
147                    .expect("Expected at least one commitment")
148                    .clone()
149                {
150                    ECMHLiveObjectSetDigest(digest) => digest,
151                    _ => return Err(anyhow::anyhow!("Expected ECMHLiveObjectSetDigest")),
152                };
153                state_snapshot_writer
154                    .write(*epoch, db, state_hash_commitment, self.chain_identifier)
155                    .await?;
156                info!("State snapshot creation successful for epoch: {}", *epoch);
157                // Drop marker in the output directory that upload completed successfully
158                let bytes = Bytes::from_static(b"success");
159                let success_marker = db_path.child(SUCCESS_MARKER);
160                put(&self.snapshot_store, &success_marker, bytes.clone()).await?;
161                let bytes = Bytes::from_static(b"success");
162                let state_snapshot_completed_marker =
163                    db_path.child(STATE_SNAPSHOT_COMPLETED_MARKER);
164                put(
165                    &self.db_checkpoint_store.clone(),
166                    &state_snapshot_completed_marker,
167                    bytes.clone(),
168                )
169                .await?;
170                info!("State snapshot completed for epoch: {epoch}");
171            } else {
172                let bytes = Bytes::from_static(b"success");
173                let state_snapshot_completed_marker =
174                    db_path.child(STATE_SNAPSHOT_COMPLETED_MARKER);
175                put(
176                    &self.db_checkpoint_store.clone(),
177                    &state_snapshot_completed_marker,
178                    bytes.clone(),
179                )
180                .await?;
181                info!("State snapshot skipped for epoch: {epoch}");
182            }
183        }
184        Ok(())
185    }

187    async fn run_upload_loop(
188        self: Arc<Self>,
189        mut recv: tokio::sync::broadcast::Receiver<()>,
190    ) -> Result<()> {
191        let mut interval = tokio::time::interval(self.interval);
192        info!("State snapshot uploader loop started");
193        loop {
194            tokio::select! {
195                _now = interval.tick() => {
196                    let missing_epochs = self.get_missing_epochs().await;
197                    match missing_epochs {
198                        Ok(epochs) => {
199                            let first_missing_epoch = epochs.first().cloned().unwrap_or(0);
200                            self.metrics.first_missing_state_snapshot_epoch.set(first_missing_epoch as i64);
201                            if let Err(err) = self.upload_state_snapshot_to_object_store(epochs).await {
202                                self.metrics.state_snapshot_upload_err.inc();
203                                error!("Failed to upload state snapshot to remote store with err: {:?}", err);
204                            } else {
205                                debug!("Successfully completed snapshot upload loop");
206                            }
207                        }
208                        Err(err) => {
209                            error!("Failed to find missing state snapshot in remote store: {:?}", err);
210                        }
211                    }
212                },
213                _ = recv.recv() => break,
214            }
215        }
216        Ok(())
217    }
218
219    async fn get_missing_epochs(&self) -> Result<Vec<u64>> {
220        let missing_epochs = find_missing_epochs_dirs(&self.snapshot_store, SUCCESS_MARKER).await?;
221        Ok(missing_epochs.to_vec())
222    }
