43/// LiveObjectSetWriterV1 writes live object set. It creates multiple *.obj files and *.ref file
44struct LiveObjectSetWriterV1 {
45    dir_path: PathBuf,
46    bucket_num: u32,
47    current_part_num: u32,
48    wbuf: BufWriter<File>,
49    ref_wbuf: BufWriter<File>,
50    n: usize,
51    files: Vec<FileMetadata>,
52    sender: Option<Sender<FileMetadata>>,
53    file_compression: FileCompression,
54}
55
56impl LiveObjectSetWriterV1 {
57    fn new(
58        dir_path: PathBuf,
59        bucket_num: u32,
60        file_compression: FileCompression,
61        sender: Sender<FileMetadata>,
62    ) -> Result<Self> {
63        let part_num = 1;
64        let (n, obj_file) = Self::object_file(dir_path.clone(), bucket_num, part_num)?;
65        let ref_file = Self::ref_file(dir_path.clone(), bucket_num, part_num)?;
66        Ok(LiveObjectSetWriterV1 {
67            dir_path,
68            bucket_num,
69            current_part_num: part_num,
70            wbuf: BufWriter::new(obj_file),
71            ref_wbuf: BufWriter::new(ref_file),
72            n,
73            files: vec![],
74            sender: Some(sender),
75            file_compression,
76        })
77    }
78    pub fn write(&mut self, object: &LiveObject) -> Result<()> {
79        let object_reference = object.object_reference();
80        self.write_object(object)?;
81        self.write_object_ref(&object_reference)?;
82        Ok(())
83    }
84    pub fn done(mut self) -> Result<Vec<FileMetadata>> {
85        self.finalize()?;
86        self.finalize_ref()?;
87        self.sender = None;
88        Ok(self.files.clone())
89    }
90    fn object_file(dir_path: PathBuf, bucket_num: u32, part_num: u32) -> Result<(usize, File)> {
91        let next_part_file_path = dir_path.join(format!("{bucket_num}_{part_num}.obj"));
92        let next_part_file_tmp_path = dir_path.join(format!("{bucket_num}_{part_num}.obj.tmp"));
93        let mut f = File::create(next_part_file_tmp_path.clone())?;
94        let mut metab = [0u8; MAGIC_BYTES];
95        BigEndian::write_u32(&mut metab, OBJECT_FILE_MAGIC);
96        f.rewind()?;
97        let n = f.write(&metab)?;
98        drop(f);
99        fs::rename(next_part_file_tmp_path, next_part_file_path.clone())?;
100        let mut f = OpenOptions::new().append(true).open(next_part_file_path)?;
101        f.seek(SeekFrom::Start(n as u64))?;
102        Ok((n, f))
103    }
104    fn ref_file(dir_path: PathBuf, bucket_num: u32, part_num: u32) -> Result<File> {
105        let ref_path = dir_path.join(format!("{bucket_num}_{part_num}.ref"));
106        let ref_tmp_path = dir_path.join(format!("{bucket_num}_{part_num}.ref.tmp"));
107        let mut f = File::create(ref_tmp_path.clone())?;
108        f.rewind()?;
109        let mut metab = [0u8; MAGIC_BYTES];
110        BigEndian::write_u32(&mut metab, REFERENCE_FILE_MAGIC);
111        let n = f.write(&metab)?;
112        drop(f);
113        fs::rename(ref_tmp_path, ref_path.clone())?;
114        let mut f = OpenOptions::new().append(true).open(ref_path)?;
115        f.seek(SeekFrom::Start(n as u64))?;
116        Ok(f)
117    }
118    fn finalize(&mut self) -> Result<()> {
119        self.wbuf.flush()?;
120        self.wbuf.get_ref().sync_data()?;
121        let off = self.wbuf.get_ref().stream_position()?;
122        self.wbuf.get_ref().set_len(off)?;
123        let file_path = self
124            .dir_path
125            .join(format!("{}_{}.obj", self.bucket_num, self.current_part_num));
126        let file_metadata = create_file_metadata(
127            &file_path,
128            self.file_compression,
129            FileType::Object,
130            self.bucket_num,
131            self.current_part_num,
132        )?;
133        self.files.push(file_metadata.clone());
134        if let Some(sender) = &self.sender {
135            sender.blocking_send(file_metadata)?;
136        }
137        Ok(())
138    }
139    fn finalize_ref(&mut self) -> Result<()> {
140        self.ref_wbuf.flush()?;
141        self.ref_wbuf.get_ref().sync_data()?;
142        let off = self.ref_wbuf.get_ref().stream_position()?;
143        self.ref_wbuf.get_ref().set_len(off)?;
144        let file_path = self
145            .dir_path
146            .join(format!("{}_{}.ref", self.bucket_num, self.current_part_num));
147        let file_metadata = create_file_metadata(
148            &file_path,
149            self.file_compression,
150            FileType::Reference,
151            self.bucket_num,
152            self.current_part_num,
153        )?;
154        self.files.push(file_metadata.clone());
155        if let Some(sender) = &self.sender {
156            sender.blocking_send(file_metadata)?;
157        }
158        Ok(())
159    }
160    fn cut(&mut self) -> Result<()> {
161        self.finalize()?;
162        let (n, f) = Self::object_file(
163            self.dir_path.clone(),
164            self.bucket_num,
165            self.current_part_num + 1,
166        )?;
167        self.n = n;
168        self.wbuf = BufWriter::new(f);
169        Ok(())
170    }
171    fn cut_reference_file(&mut self) -> Result<()> {
172        self.finalize_ref()?;
173        let f = Self::ref_file(
174            self.dir_path.clone(),
175            self.bucket_num,
176            self.current_part_num + 1,
177        )?;
178        self.ref_wbuf = BufWriter::new(f);
179        Ok(())
180    }
181    fn write_object(&mut self, object: &LiveObject) -> Result<()> {
182        let blob = Blob::encode(object, BlobEncoding::Bcs)?;
183        let mut blob_size = blob.data.len().required_space();
184        blob_size += BLOB_ENCODING_BYTES;
185        blob_size += blob.data.len();
186        let cut_new_part_file = (self.n + blob_size) > FILE_MAX_BYTES;
187        if cut_new_part_file {
188            self.cut()?;
189            self.cut_reference_file()?;
190            self.current_part_num += 1;
191        }
192        self.n += blob.write(&mut self.wbuf)?;
193        Ok(())
194    }
195    fn write_object_ref(&mut self, object_ref: &ObjectRef) -> Result<()> {
196        let mut buf = [0u8; OBJECT_REF_BYTES];
197        buf[0..ObjectID::LENGTH].copy_from_slice(object_ref.0.as_ref());
198        BigEndian::write_u64(
199            &mut buf[ObjectID::LENGTH..OBJECT_REF_BYTES],
200            object_ref.1.value(),
201        );
202        buf[ObjectID::LENGTH + SEQUENCE_NUM_BYTES..OBJECT_REF_BYTES]
203            .copy_from_slice(object_ref.2.as_ref());
204        self.ref_wbuf.write_all(&buf)?;
205        Ok(())
206    }
207}

209/// StateSnapshotWriterV1 writes snapshot files to a local staging dir and simultaneously uploads them
210/// to a remote object store
211pub struct StateSnapshotWriterV1 {
212    local_staging_dir: PathBuf,
213    file_compression: FileCompression,
214    remote_object_store: Arc<DynObjectStore>,
215    local_staging_store: Arc<DynObjectStore>,
216    concurrency: usize,
217}

219impl StateSnapshotWriterV1 {
220    pub async fn new_from_store(
221        local_staging_path: &std::path::Path,
222        local_staging_store: &Arc<DynObjectStore>,
223        remote_object_store: &Arc<DynObjectStore>,
224        file_compression: FileCompression,
225        concurrency: NonZeroUsize,
226    ) -> Result<Self> {
227        Ok(StateSnapshotWriterV1 {
228            file_compression,
229            local_staging_dir: local_staging_path.to_path_buf(),
230            remote_object_store: remote_object_store.clone(),
231            local_staging_store: local_staging_store.clone(),
232            concurrency: concurrency.get(),
233        })
234    }

236    pub async fn new(
237        local_store_config: &ObjectStoreConfig,
238        remote_store_config: &ObjectStoreConfig,
239        file_compression: FileCompression,
240        concurrency: NonZeroUsize,
241    ) -> Result<Self> {
242        let remote_object_store = remote_store_config.make()?;
243        let local_staging_store = local_store_config.make()?;
244        let local_staging_dir = local_store_config
245            .directory
246            .as_ref()
247            .context("No local directory specified")?
248            .clone();
249        Ok(StateSnapshotWriterV1 {
250            local_staging_dir,
251            file_compression,
252            remote_object_store,
253            local_staging_store,
254            concurrency: concurrency.get(),
255        })
256    }

258    pub async fn write(
259        self,
260        epoch: u64,
261        perpetual_db: Arc<AuthorityPerpetualTables>,
262        root_state_hash: ECMHLiveObjectSetDigest,
263        chain_identifier: ChainIdentifier,
264    ) -> Result<()> {
265        let system_state_object = get_sui_system_state(&perpetual_db)?;
266
267        let protocol_version = system_state_object.protocol_version();
268        let protocol_config = ProtocolConfig::get_for_version(
269            ProtocolVersion::new(protocol_version),
270            chain_identifier.chain(),
271        );
272        let include_wrapped_tombstone = !protocol_config.simplified_unwrap_then_delete();
273        self.write_internal(
274            epoch,
275            include_wrapped_tombstone,
276            perpetual_db,
277            root_state_hash,
278        )
279        .await
280    }

282    pub(crate) async fn write_internal(
283        mut self,
284        epoch: u64,
285        include_wrapped_tombstone: bool,
286        perpetual_db: Arc<AuthorityPerpetualTables>,
287        root_state_hash: ECMHLiveObjectSetDigest,
288    ) -> Result<()> {
289        self.setup_epoch_dir(epoch).await?;
290
291        let manifest_file_path = self.epoch_dir(epoch).child("MANIFEST");
292        let local_staging_dir = self.local_staging_dir.clone();
293        let local_object_store = self.local_staging_store.clone();
294        let remote_object_store = self.remote_object_store.clone();
295
296        let (sender, receiver) = mpsc::channel::<FileMetadata>(1000);
297        let upload_handle = self.start_upload(epoch, receiver)?;
298        let write_handler = tokio::task::spawn_blocking(move || {
299            self.write_live_object_set(
300                epoch,
301                perpetual_db,
302                sender,
303                Self::bucket_func,
304                include_wrapped_tombstone,
305                root_state_hash,
306            )
307        });
308        write_handler.await?.context(format!(
309            "Failed to write state snapshot for epoch: {}",
310            &epoch
311        ))?;
312
313        upload_handle.await?.context(format!(
314            "Failed to upload state snapshot for epoch: {}",
315            &epoch
316        ))?;
317
318        Self::sync_file_to_remote(
319            local_staging_dir,
320            manifest_file_path,
321            local_object_store,
322            remote_object_store,
323        )
324        .await?;
325        Ok(())
326    }
327
328    fn start_upload(
329        &self,
330        epoch: u64,
331        receiver: Receiver<FileMetadata>,
332    ) -> Result<JoinHandle<Result<Vec<()>, anyhow::Error>>> {
333        let remote_object_store = self.remote_object_store.clone();
334        let local_staging_store = self.local_staging_store.clone();
335        let local_dir_path = self.local_staging_dir.clone();
336        let epoch_dir = self.epoch_dir(epoch);
337        let upload_concurrency = self.concurrency;
338        let join_handle = tokio::spawn(async move {
339            let results: Vec<Result<(), anyhow::Error>> = ReceiverStream::new(receiver)
340                .map(|file_metadata| {
341                    let file_path = file_metadata.file_path(&epoch_dir);
342                    let remote_object_store = remote_object_store.clone();
343                    let local_object_store = local_staging_store.clone();
344                    let local_dir_path = local_dir_path.clone();
345                    async move {
346                        Self::sync_file_to_remote(
347                            local_dir_path.clone(),
348                            file_path.clone(),
349                            local_object_store.clone(),
350                            remote_object_store.clone(),
351                        )
352                        .await?;
353                        Ok(())
354                    }
355                })
356                .boxed()
357                .buffer_unordered(upload_concurrency)
358                .collect()
359                .await;
360            results
361                .into_iter()
362                .collect::<Result<Vec<()>, anyhow::Error>>()
363        });
364        Ok(join_handle)
365    }

367    fn write_live_object_set<F>(
368        &mut self,
369        epoch: u64,
370        perpetual_db: Arc<AuthorityPerpetualTables>,
371        sender: Sender<FileMetadata>,
372        bucket_func: F,
373        include_wrapped_tombstone: bool,
374        root_state_hash: ECMHLiveObjectSetDigest,
375    ) -> Result<()>
376    where
377        F: Fn(&LiveObject) -> u32,
378    {
379        let mut object_writers: HashMap<u32, LiveObjectSetWriterV1> = HashMap::new();
380        let local_staging_dir_path =
381            path_to_filesystem(self.local_staging_dir.clone(), &self.epoch_dir(epoch))?;
382        let mut acc = GlobalStateHash::default();
383        for object in perpetual_db.iter_live_object_set(include_wrapped_tombstone) {
384            GlobalStateHasher::accumulate_live_object(&mut acc, &object);
385            let bucket_num = bucket_func(&object);
386            if let Vacant(entry) = object_writers.entry(bucket_num) {
387                entry.insert(LiveObjectSetWriterV1::new(
388                    local_staging_dir_path.clone(),
389                    bucket_num,
390                    self.file_compression,
391                    sender.clone(),
392                )?);
393            }
394            let writer = object_writers
395                .get_mut(&bucket_num)
396                .context("Unexpected missing bucket writer")?;
397            writer.write(&object)?;
398        }
399        assert_eq!(
400            ECMHLiveObjectSetDigest::from(acc.digest()),
401            root_state_hash,
402            "Root state hash mismatch!"
403        );
404        let mut files = vec![];
405        for (_, writer) in object_writers.into_iter() {
406            files.extend(writer.done()?);
407        }
408        self.write_manifest(epoch, files)?;
409        Ok(())
410    }
411
412    fn write_manifest(&mut self, epoch: u64, file_metadata: Vec<FileMetadata>) -> Result<()> {
413        let (f, manifest_file_path) = self.manifest_file(epoch)?;
414        let mut wbuf = BufWriter::new(f);
415        let manifest: Manifest = Manifest::V1(ManifestV1 {
416            snapshot_version: 1,
417            address_length: ObjectID::LENGTH as u64,
418            file_metadata,
419            epoch,
420        });
421        let serialized_manifest = bcs::to_bytes(&manifest)?;
422        wbuf.write_all(&serialized_manifest)?;
423        wbuf.flush()?;
424        wbuf.get_ref().sync_data()?;
425        let sha3_digest = compute_sha3_checksum(&manifest_file_path)?;
426        wbuf.write_all(&sha3_digest)?;
427        wbuf.flush()?;
428        wbuf.get_ref().sync_data()?;
429        let off = wbuf.get_ref().stream_position()?;
430        wbuf.get_ref().set_len(off)?;
431        Ok(())
432    }

434    fn manifest_file(&mut self, epoch: u64) -> Result<(File, PathBuf)> {
435        let manifest_file_path = path_to_filesystem(
436            self.local_staging_dir.clone(),
437            &self.epoch_dir(epoch).child("MANIFEST"),
438        )?;
439        let manifest_file_tmp_path = path_to_filesystem(
440            self.local_staging_dir.clone(),
441            &self.epoch_dir(epoch).child("MANIFEST.tmp"),
442        )?;
443        let mut f = File::create(manifest_file_tmp_path.clone())?;
444        let mut metab = vec![0u8; MAGIC_BYTES];
445        BigEndian::write_u32(&mut metab, MANIFEST_FILE_MAGIC);
446        f.rewind()?;
447        f.write_all(&metab)?;
448        drop(f);
449        fs::rename(manifest_file_tmp_path, manifest_file_path.clone())?;
450        let mut f = OpenOptions::new()
451            .append(true)
452            .open(manifest_file_path.clone())?;
453        f.seek(SeekFrom::Start(MAGIC_BYTES as u64))?;
454        Ok((f, manifest_file_path))
455    }

462    fn epoch_dir(&self, epoch: u64) -> Path {

466    async fn setup_epoch_dir(&self, epoch: u64) -> Result<()> {
467        let epoch_dir = self.epoch_dir(epoch);
468        // Delete remote epoch dir if it exists
469        delete_recursively(
470            &epoch_dir,
471            &self.remote_object_store,
472            NonZeroUsize::new(self.concurrency).unwrap(),
473        )
474        .await?;
475        // Delete local staging epoch dir if it exists
476        let local_epoch_dir_path = self.local_staging_dir.join(format!("epoch_{}", epoch));
477        if local_epoch_dir_path.exists() {
478            fs::remove_dir_all(&local_epoch_dir_path)?;
479        }
480        fs::create_dir_all(&local_epoch_dir_path)?;
481        Ok(())
482    }

484    async fn sync_file_to_remote(
485        local_path: PathBuf,
486        path: Path,
487        from: Arc<DynObjectStore>,
488        to: Arc<DynObjectStore>,
489    ) -> Result<()> {
490        debug!("Syncing snapshot file to remote: {:?}", path);
491        copy_file(&path, &path, &from, &to).await?;
492        fs::remove_file(path_to_filesystem(local_path, &path)?)?;
493        Ok(())
494    }
