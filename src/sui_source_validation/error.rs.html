4use std::fmt;
5
6use move_core_types::account_address::AccountAddress;
7use move_symbol_pool::Symbol;
8use sui_package_management::PublishedAtError;
9use sui_types::{base_types::ObjectID, error::SuiObjectResponseError};
10
11#[derive(Debug, thiserror::Error)]
12pub struct AggregateError(pub(crate) Vec<Error>);
13
14#[derive(Debug, thiserror::Error)]
15pub enum Error {
16    #[error("Cannot check local module for {package}: {message}")]
17    CannotCheckLocalModules { package: Symbol, message: String },
18
19    #[error("Could not read a dependency's on-chain object: {0}")]
20    DependencyObjectReadFailure(String),
21
22    #[error("On-chain package {0} is empty")]
23    EmptyOnChainPackage(AccountAddress),
24
25    #[error("Invalid module {name} with error: {message}")]
26    InvalidModuleFailure { name: String, message: String },
27
28    #[error("Local version of dependency {address}::{module} was not found.")]
29    LocalDependencyNotFound {
30        address: AccountAddress,
31        module: Symbol,
32    },
33
34    #[error("Source package depends on {0} which is not in the linkage table.")]
35    MissingDependencyInLinkageTable(AccountAddress),
36
37    #[error("On-chain package depends on {0} which is not a source dependency.")]
38    MissingDependencyInSourcePackage(AccountAddress),
39
40    #[error(
41        "Local dependency did not match its on-chain version at {address}::{package}::{module}"
42    )]
43    ModuleBytecodeMismatch {
44        address: AccountAddress,
45        package: Symbol,
46        module: Symbol,
47    },
48
49    #[error("Dependency ID contains a Sui object, not a Move package: {0}")]
50    ObjectFoundWhenPackageExpected(ObjectID),
51
52    #[error("Could not deserialize on-chain dependency {address}::{module}.")]
53    OnChainDependencyDeserializationError {
54        address: AccountAddress,
55        module: Symbol,
56    },
57
58    #[error("On-chain version of dependency {package}::{module} was not found.")]
59    OnChainDependencyNotFound { package: Symbol, module: Symbol },
60
61    #[error("{0}. Please supply an explicit on-chain address for the package")]
62    PublishedAt(#[from] PublishedAtError),
63
64    #[error("Dependency object does not exist or was deleted: {0:?}")]
65    SuiObjectRefFailure(SuiObjectResponseError),
66
67    #[error("On-chain address cannot be zero")]
68    ZeroOnChainAddresSpecifiedFailure,
69}
70
71impl fmt::Display for AggregateError {
72    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
73        let Self(errors) = self;
74        match &errors[..] {
75            [] => unreachable!("Aggregate error with no errors"),
76            [error] => write!(f, "{}", error)?,
77            errors => {
78                writeln!(f, "Multiple source verification errors found:")?;
79                for error in errors {
80                    write!(f, "\n- {}", error)?;
81                }
82                return Ok(());
83            }
84        };
85        Ok(())
86    }
87}
88
89impl From<Error> for AggregateError {
90    fn from(error: Error) -> Self {
91        Self(vec![error])
92    }
93}
