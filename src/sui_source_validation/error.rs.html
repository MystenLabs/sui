4use std::fmt;
5
6use move_core_types::account_address::AccountAddress;
7use move_symbol_pool::Symbol;
8use sui_json_rpc_types::SuiRawMoveObject;
9use sui_package_management::PublishedAtError;
10use sui_sdk::error::Error as SdkError;
11use sui_types::{base_types::ObjectID, error::SuiObjectResponseError};
12
13#[derive(Debug, thiserror::Error)]
14pub struct AggregateError(pub(crate) Vec<Error>);
15
16#[derive(Debug, thiserror::Error)]
17pub enum Error {
18    #[error("Cannot check local module for {package}: {message}")]
19    CannotCheckLocalModules { package: Symbol, message: String },
20
21    #[error("Could not read a dependency's on-chain object: {0:?}")]
22    DependencyObjectReadFailure(SdkError),
23
24    #[error("On-chain package {0} is empty")]
25    EmptyOnChainPackage(AccountAddress),
26
27    #[error("Invalid module {name} with error: {message}")]
28    InvalidModuleFailure { name: String, message: String },
29
30    #[error("Local version of dependency {address}::{module} was not found.")]
31    LocalDependencyNotFound {
32        address: AccountAddress,
33        module: Symbol,
34    },
35
36    #[error("Source package depends on {0} which is not in the linkage table.")]
37    MissingDependencyInLinkageTable(AccountAddress),
38
39    #[error("On-chain package depends on {0} which is not a source dependency.")]
40    MissingDependencyInSourcePackage(AccountAddress),
41
42    #[error(
43        "Local dependency did not match its on-chain version at {address}::{package}::{module}"
44    )]
45    ModuleBytecodeMismatch {
46        address: AccountAddress,
47        package: Symbol,
48        module: Symbol,
49    },
50
51    #[error("Dependency ID contains a Sui object, not a Move package: {0}")]
52    ObjectFoundWhenPackageExpected(ObjectID, SuiRawMoveObject),
53
54    #[error("Could not deserialize on-chain dependency {address}::{module}.")]
55    OnChainDependencyDeserializationError {
56        address: AccountAddress,
57        module: Symbol,
58    },
59
60    #[error("On-chain version of dependency {package}::{module} was not found.")]
61    OnChainDependencyNotFound { package: Symbol, module: Symbol },
62
63    #[error("{0}. Please supply an explicit on-chain address for the package")]
64    PublishedAt(#[from] PublishedAtError),
65
66    #[error("Dependency object does not exist or was deleted: {0:?}")]
67    SuiObjectRefFailure(SuiObjectResponseError),
68
69    #[error("On-chain address cannot be zero")]
70    ZeroOnChainAddresSpecifiedFailure,
71}
72
73impl fmt::Display for AggregateError {
74    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
75        let Self(errors) = self;
76        match &errors[..] {
77            [] => unreachable!("Aggregate error with no errors"),
78            [error] => write!(f, "{}", error)?,
79            errors => {
80                writeln!(f, "Multiple source verification errors found:")?;
81                for error in errors {
82                    write!(f, "\n- {}", error)?;
83                }
84                return Ok(());
85            }
86        };
87        Ok(())
88    }
89}
90
91impl From<Error> for AggregateError {
92    fn from(error: Error) -> Self {
93        Self(vec![error])
94    }
95}
