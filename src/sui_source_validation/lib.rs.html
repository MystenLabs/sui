24/// Details of what to verify
25pub enum ValidationMode {
26    /// Validate only the dependencies
27    Deps,
28
29    /// Validate the root package, and its linkage.
30    Root {
31        /// Additionally validate the dependencies, and make sure the runtime and storage IDs in
32        /// dependency source code matches the root package's on-chain linkage table.
33        deps: bool,
34
35        /// Look for the root package on-chain at the specified address, rather than the address in
36        /// its manifest.
37        at: Option<AccountAddress>,
38    },
39}
40
41pub struct BytecodeSourceVerifier<'a> {
42    rpc_client: &'a ReadApi,
43}

45/// Map package addresses and module names to package names and bytecode.
46type LocalModules = HashMap<(AccountAddress, Symbol), (Symbol, CompiledModule)>;
47
48#[derive(Default)]
49struct OnChainRepresentation {
50    /// Storage IDs from the root package's on-chain linkage table. This will only be present if
51    /// root package verification was requested, in which case the keys from this mapping must
52    /// match the source package's dependencies.
53    on_chain_dependencies: Option<HashSet<AccountAddress>>,
54
55    /// Map package addresses and module names to bytecode (package names are gone in the on-chain
56    /// representation).
57    modules: HashMap<(AccountAddress, Symbol), CompiledModule>,
58}
59
60impl ValidationMode {
61    /// Only verify that source dependencies match their on-chain versions.
62    pub fn deps() -> Self {
63        Self::Deps
64    }
65
66    /// Only verify that the root package matches its on-chain version (requires that the root
67    /// package is published with its address available in the manifest).
68    pub fn root() -> Self {
69        Self::Root {
70            deps: false,
71            at: None,
72        }
73    }
74
75    /// Only verify that the root package matches its on-chain version, but override the location
76    /// to look for the root package to `address`.
77    pub fn root_at(address: AccountAddress) -> Self {
78        Self::Root {
79            deps: false,
80            at: Some(address),
81        }
82    }
83
84    /// Verify both the root package and its dependencies (requires that the root package is
85    /// published with its address available in the manifest).
86    pub fn root_and_deps() -> Self {
87        Self::Root {
88            deps: true,
89            at: None,
90        }
91    }
92
93    /// Verify both the root package and its dependencies, but override the location to look for
94    /// the root package to `address`.
95    pub fn root_and_deps_at(address: AccountAddress) -> Self {
96        Self::Root {
97            deps: true,
98            at: Some(address),
99        }
100    }
101
102    /// Should we verify dependencies?
103    fn verify_deps(&self) -> bool {
104        matches!(self, Self::Deps | Self::Root { deps: true, .. })
105    }
106
107    /// If the root package needs to be verified, what address should it be fetched from?
108    #[allow(clippy::result_large_err)]
109    fn root_address(&self, package: &CompiledPackage) -> Result<Option<AccountAddress>, Error> {
110        match self {
111            Self::Root { at: Some(addr), .. } => Ok(Some(*addr)),
112            Self::Root { at: None, .. } => Ok(Some(*package.published_at.clone()?)),
113            Self::Deps => Ok(None),
114        }
115    }
116
117    /// All the on-chain addresses that we need to fetch to build on-chain addresses.
118    #[allow(clippy::result_large_err)]
119    fn on_chain_addresses(&self, package: &CompiledPackage) -> Result<Vec<AccountAddress>, Error> {
120        let mut addrs = vec![];
121
122        if let Some(addr) = self.root_address(package)? {
123            addrs.push(addr);
124        }
125
126        if self.verify_deps() {
127            addrs.extend(dependency_addresses(package));
128        }
129
130        Ok(addrs)
131    }
132
133    /// On-chain representation of the package and dependencies compiled to `package`, including
134    /// linkage information.
135    async fn on_chain(
136        &self,
137        package: &CompiledPackage,
138        verifier: &BytecodeSourceVerifier<'_>,
139    ) -> Result<OnChainRepresentation, AggregateError> {
140        let mut on_chain = OnChainRepresentation::default();
141        let mut errs: Vec<Error> = vec![];
142
143        let root = self.root_address(package)?;
144        let addrs = self.on_chain_addresses(package)?;
145
146        let resps =
147            future::join_all(addrs.iter().copied().map(|a| verifier.pkg_for_address(a))).await;
148
149        for (storage_id, pkg) in addrs.into_iter().zip(resps) {
150            let SuiRawMovePackage {
151                module_map,
152                linkage_table,
153                ..
154            } = pkg?;
155
156            let mut modules = module_map
157                .into_iter()
158                .map(|(name, bytes)| {
159                    let Ok(module) = CompiledModule::deserialize_with_defaults(&bytes) else {
160                        return Err(Error::OnChainDependencyDeserializationError {
161                            address: storage_id,
162                            module: name.into(),
163                        });
164                    };
165
166                    Ok::<_, Error>((Symbol::from(name), module))
167                })
168                .peekable();
169
170            let runtime_id = match modules.peek() {
171                Some(Ok((_, module))) => *module.self_id().address(),
172
173                Some(Err(_)) => {
174                    // SAFETY: The error type does not implement `Clone` so we need to take the
175                    // error by value. We do that by calling `next` to take the value we just
176                    // peeked, which we know is an error type.
177                    errs.push(modules.next().unwrap().unwrap_err());
178                    continue;
179                }
180
181                None => {
182                    errs.push(Error::EmptyOnChainPackage(storage_id));
183                    continue;
184                }
185            };
186
187            for module in modules {
188                match module {
189                    Ok((name, module)) => {
190                        on_chain.modules.insert((runtime_id, name), module);
191                    }
192
193                    Err(e) => {
194                        errs.push(e);
195                        continue;
196                    }
197                }
198            }
199
200            if root.is_some_and(|r| r == storage_id) {
201                on_chain.on_chain_dependencies = Some(HashSet::from_iter(
202                    linkage_table.into_values().map(|info| *info.upgraded_id),
203                ));
204            }
205        }
206
207        Ok(on_chain)
208    }
209
210    /// Local representation of the modules in `package`. If the validation mode requires verifying
211    /// dependencies, then the dependencies' modules are also included in the output.
212    ///
213    /// For the purposes of this function, a module is considered a dependency if it is from a
214    /// different source package, and that source package has already been published. Conversely, a
215    /// module that is from a different source package, but that has not been published is
216    /// considered part of the root package.
217    ///
218    /// If the validation mode requires verifying the root package at a specific address, then the
219    /// modules from the root package will be expected at address `0x0` and this address will be
220    /// substituted with the specified address.
221    #[allow(clippy::result_large_err)]
222    fn local(&self, package: &CompiledPackage) -> Result<LocalModules, Error> {
223        let sui_package = package;
224        let package = &package.package;
225        let root_package = package.compiled_package_info.package_name;
226        let mut map = LocalModules::new();
227
228        if self.verify_deps() {
229            let deps_compiled_units =
230                units_for_toolchain(&package.deps_compiled_units).map_err(|e| {
231                    Error::CannotCheckLocalModules {
232                        package: package.compiled_package_info.package_name,
233                        message: e.to_string(),
234                    }
235                })?;
236
237            // only keep modules that are actually used
238            let deps_compiled_units: Vec<_> = deps_compiled_units
239                .into_iter()
240                .filter(|pkg| sui_package.dependency_ids.published.contains_key(&pkg.0))
241                .collect();
242
243            for (package, local_unit) in deps_compiled_units {
244                let m = &local_unit.unit;
245                let module = m.name;
246                let address = m.address.into_inner();
247
248                // Skip modules with on 0x0 because they are treated as part of the root package,
249                // even if they are a source dependency.
250                if address == AccountAddress::ZERO {
251                    continue;
252                }
253
254                map.insert((address, module), (package, m.module.clone()));
255            }
256
257            // Include bytecode dependencies.
258            for (package, module) in sui_package.bytecode_deps.iter() {
259                let address = *module.address();
260                if address == AccountAddress::ZERO {
261                    continue;
262                }
263
264                map.insert(
265                    (address, Symbol::from(module.name().as_str())),
266                    (*package, module.clone()),
267                );
268            }
269        }
270
271        let Self::Root { at, .. } = self else {
272            return Ok(map);
273        };
274
275        // Potentially rebuild according to the toolchain that the package was originally built
276        // with.
277        let root_compiled_units = units_for_toolchain(
278            &package
279                .root_compiled_units
280                .iter()
281                .map(|u| ("root".into(), u.clone()))
282                .collect(),
283        )
284        .map_err(|e| Error::CannotCheckLocalModules {
285            package: package.compiled_package_info.package_name,
286            message: e.to_string(),
287        })?;
288
289        // Add the root modules, potentially remapping 0x0 if we have been supplied an address to
290        // substitute with.
291        for (_, local_unit) in root_compiled_units {
292            let m = &local_unit.unit;
293            let module = m.name;
294            let address = m.address.into_inner();
295
296            let (address, compiled_module) = if let Some(root_address) = at {
297                (*root_address, substitute_root_address(m, *root_address)?)
298            } else if address == AccountAddress::ZERO {
299                return Err(Error::InvalidModuleFailure {
300                    name: module.to_string(),
301                    message: "Can't verify unpublished source".to_string(),
302                });
303            } else {
304                (address, m.module.clone())
305            };
306
307            map.insert((address, module), (root_package, compiled_module));
308        }
309
310        // If we have a root address to substitute, we need to find unpublished dependencies that
311        // would have gone into the root package as well.
312        if let Some(root_address) = at {
313            for (package, local_unit) in &package.deps_compiled_units {
314                let m = &local_unit.unit;
315                let module = m.name;
316                let address = m.address.into_inner();
317
318                if address != AccountAddress::ZERO {
319                    continue;
320                }
321
322                map.insert(
323                    (*root_address, module),
324                    (*package, substitute_root_address(m, *root_address)?),
325                );
326            }
327        }
328
329        Ok(map)
330    }
331}

333impl<'a> BytecodeSourceVerifier<'a> {
334    pub fn new(rpc_client: &'a ReadApi) -> Self {
335        BytecodeSourceVerifier { rpc_client }
336    }
337
338    /// Verify that the `compiled_package` matches its on-chain representation.
339    ///
340    /// See [`ValidationMode`] for more details on what is verified.
341    pub async fn verify(
342        &self,
343        package: &CompiledPackage,
344        mode: ValidationMode,
345    ) -> Result<(), AggregateError> {
346        if matches!(
347            mode,
348            ValidationMode::Root {
349                at: Some(AccountAddress::ZERO),
350                ..
351            }
352        ) {
353            return Err(Error::ZeroOnChainAddresSpecifiedFailure.into());
354        }
355
356        let local = mode.local(package)?;
357        let mut chain = mode.on_chain(package, self).await?;
358        let mut errs = vec![];
359
360        // Check that the transitive dependencies listed on chain match the dependencies listed in
361        // source code. Ignore 0x0 becaus this signifies an unpublished dependency.
362        if let Some(on_chain_deps) = &mut chain.on_chain_dependencies {
363            for dependency_id in dependency_addresses(package) {
364                if dependency_id != AccountAddress::ZERO && !on_chain_deps.remove(&dependency_id) {
365                    errs.push(Error::MissingDependencyInLinkageTable(dependency_id));
366                }
367            }
368        }
369
370        for on_chain_dep_id in chain.on_chain_dependencies.take().into_iter().flatten() {
371            errs.push(Error::MissingDependencyInSourcePackage(on_chain_dep_id));
372        }
373
374        // Check that the contents of bytecode matches between modules.
375        for ((address, module), (package, local_module)) in local {
376            let Some(on_chain_module) = chain.modules.remove(&(address, module)) else {
377                errs.push(Error::OnChainDependencyNotFound { package, module });
378                continue;
379            };
380
381            if local_module != on_chain_module {
382                errs.push(Error::ModuleBytecodeMismatch {
383                    address,
384                    package,
385                    module,
386                })
387            }
388        }
389
390        for (address, module) in chain.modules.into_keys() {
391            errs.push(Error::LocalDependencyNotFound { address, module });
392        }
393
394        if !errs.is_empty() {
395            return Err(AggregateError(errs));
396        }
397
398        Ok(())
399    }

401    async fn pkg_for_address(&self, addr: AccountAddress) -> Result<SuiRawMovePackage, Error> {
402        // Move packages are specified with an AccountAddress, but are
403        // fetched from a sui network via sui_getObject, which takes an object ID
404        let obj_id = ObjectID::from(addr);
405
406        // fetch the Sui object at the address specified for the package in the local resolution table
407        // if future packages with a large set of dependency packages prove too slow to verify,
408        // batched object fetching should be added to the ReadApi & used here
409        let obj_read = self
410            .rpc_client
411            .get_object_with_options(obj_id, SuiObjectDataOptions::new().with_bcs())
412            .await
413            .map_err(Error::DependencyObjectReadFailure)?;
414
415        let obj = obj_read
416            .into_object()
417            .map_err(Error::SuiObjectRefFailure)?
418            .bcs
419            .ok_or_else(|| {
420                Error::DependencyObjectReadFailure(SdkError::DataError(
421                    "Bcs field is not found".to_string(),
422                ))
423            })?;
424
425        match obj {
426            SuiRawData::Package(pkg) => Ok(pkg),
427            SuiRawData::MoveObject(move_obj) => {
428                Err(Error::ObjectFoundWhenPackageExpected(obj_id, move_obj))
429            }
430        }
431    }
432}

434#[allow(clippy::result_large_err)]
435fn substitute_root_address(
436    named_module: &NamedCompiledModule,
437    root: AccountAddress,
438) -> Result<CompiledModule, Error> {
439    let mut module = named_module.module.clone();
440    let address_idx = module.self_handle().address;
441
442    let Some(addr) = module.address_identifiers.get_mut(address_idx.0 as usize) else {
443        return Err(Error::InvalidModuleFailure {
444            name: named_module.name.to_string(),
445            message: "Self address field missing".into(),
446        });
447    };
448
449    if *addr != AccountAddress::ZERO {
450        return Err(Error::InvalidModuleFailure {
451            name: named_module.name.to_string(),
452            message: "Self address already populated".to_string(),
453        });
454    }
455
456    *addr = root;
457    Ok(module)
458}

460/// The on-chain addresses for a source package's dependencies
461fn dependency_addresses(package: &CompiledPackage) -> impl Iterator<Item = AccountAddress> + '_ {
462    package.dependency_ids.published.values().map(|id| **id)
