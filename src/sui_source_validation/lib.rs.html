24/// Details of what to verify
25pub enum ValidationMode {
26    /// Validate only the dependencies
27    Deps,
28
29    /// Validate the root package, and its linkage.
30    Root {
31        /// Additionally validate the dependencies, and make sure the runtime and storage IDs in
32        /// dependency source code matches the root package's on-chain linkage table.
33        deps: bool,
34
35        /// Look for the root package on-chain at the specified address, rather than the address in
36        /// its manifest.
37        at: Option<AccountAddress>,
38    },
39}
40
41pub struct BytecodeSourceVerifier<'a> {
42    rpc_client: &'a Client,
43}

45/// Map package addresses and module names to package names and bytecode.
46type LocalModules = HashMap<(AccountAddress, Symbol), (Symbol, CompiledModule)>;
47
48#[derive(Default)]
49struct OnChainRepresentation {
50    /// Storage IDs from the root package's on-chain linkage table. This will only be present if
51    /// root package verification was requested, in which case the keys from this mapping must
52    /// match the source package's dependencies.
53    on_chain_dependencies: Option<HashSet<AccountAddress>>,
54
55    /// Map package addresses and module names to bytecode (package names are gone in the on-chain
56    /// representation).
57    modules: HashMap<(AccountAddress, Symbol), CompiledModule>,
58}
59
60impl ValidationMode {
61    /// Only verify that source dependencies match their on-chain versions.
62    pub fn deps() -> Self {
63        Self::Deps
64    }
65
66    /// Only verify that the root package matches its on-chain version (requires that the root
67    /// package is published with its address available in the manifest).
68    pub fn root() -> Self {
69        Self::Root {
70            deps: false,
71            at: None,
72        }
73    }
74
75    /// Only verify that the root package matches its on-chain version, but override the location
76    /// to look for the root package to `address`.
77    pub fn root_at(address: AccountAddress) -> Self {
78        Self::Root {
79            deps: false,
80            at: Some(address),
81        }
82    }
83
84    /// Verify both the root package and its dependencies (requires that the root package is
85    /// published with its address available in the manifest).
86    pub fn root_and_deps() -> Self {
87        Self::Root {
88            deps: true,
89            at: None,
90        }
91    }
92
93    /// Verify both the root package and its dependencies, but override the location to look for
94    /// the root package to `address`.
95    pub fn root_and_deps_at(address: AccountAddress) -> Self {
96        Self::Root {
97            deps: true,
98            at: Some(address),
99        }
100    }
101
102    /// Should we verify dependencies?
103    fn verify_deps(&self) -> bool {
104        matches!(self, Self::Deps | Self::Root { deps: true, .. })
105    }
106
107    /// If the root package needs to be verified, what address should it be fetched from?
108    #[allow(clippy::result_large_err)]
109    fn root_address(&self, package: &CompiledPackage) -> Result<Option<AccountAddress>, Error> {
110        match self {
111            Self::Root { at: Some(addr), .. } => Ok(Some(*addr)),
112            Self::Root { at: None, .. } => Ok(package.published_at.map(AccountAddress::from)),
113            Self::Deps => Ok(None),
114        }
115    }
116
117    /// All the on-chain addresses that we need to fetch to build on-chain addresses.
118    #[allow(clippy::result_large_err)]
119    fn on_chain_addresses(&self, package: &CompiledPackage) -> Result<Vec<AccountAddress>, Error> {
120        let mut addrs = vec![];
121
122        if let Some(addr) = self.root_address(package)? {
123            addrs.push(addr);
124        }
125
126        if self.verify_deps() {
127            addrs.extend(dependency_addresses(package));
128        }
129
130        Ok(addrs)
131    }
132
133    /// On-chain representation of the package and dependencies compiled to `package`, including
134    /// linkage information.
135    async fn on_chain(
136        &self,
137        package: &CompiledPackage,
138        verifier: &BytecodeSourceVerifier<'_>,
139    ) -> Result<OnChainRepresentation, AggregateError> {
140        let mut on_chain = OnChainRepresentation::default();
141        let mut errs: Vec<Error> = vec![];
142
143        let root = self.root_address(package)?;
144        let addrs = self.on_chain_addresses(package)?;
145
146        let resps =
147            future::join_all(addrs.iter().copied().map(|a| verifier.pkg_for_address(a))).await;
148
149        for (storage_id, pkg) in addrs.into_iter().zip(resps) {
150            let pkg = pkg?;
151
152            let module_map = pkg.serialized_module_map();
153            let linkage_table = pkg.linkage_table();
154
155            let mut modules = module_map
156                .iter()
157                .map(|(name, bytes)| {
158                    let Ok(module) = CompiledModule::deserialize_with_defaults(bytes) else {
159                        return Err(Error::OnChainDependencyDeserializationError {
160                            address: storage_id,
161                            module: name.as_str().into(),
162                        });
163                    };
164
165                    Ok::<_, Error>((Symbol::from(name.as_str()), module))
166                })
167                .peekable();
168
169            let runtime_id = match modules.peek() {
170                Some(Ok((_, module))) => *module.self_id().address(),
171
172                Some(Err(_)) => {
173                    // SAFETY: The error type does not implement `Clone` so we need to take the
174                    // error by value. We do that by calling `next` to take the value we just
175                    // peeked, which we know is an error type.
176                    errs.push(modules.next().unwrap().unwrap_err());
177                    continue;
178                }
179
180                None => {
181                    errs.push(Error::EmptyOnChainPackage(storage_id));
182                    continue;
183                }
184            };
185
186            for module in modules {
187                match module {
188                    Ok((name, module)) => {
189                        on_chain.modules.insert((runtime_id, name), module);
190                    }
191
192                    Err(e) => {
193                        errs.push(e);
194                        continue;
195                    }
196                }
197            }
198
199            if root.is_some_and(|r| r == storage_id) {
200                on_chain.on_chain_dependencies = Some(HashSet::from_iter(
201                    linkage_table
202                        .clone()
203                        .into_values()
204                        .map(|info| *info.upgraded_id),
205                ));
206            }
207        }
208
209        Ok(on_chain)
210    }
211
212    /// Local representation of the modules in `package`. If the validation mode requires verifying
213    /// dependencies, then the dependencies' modules are also included in the output.
214    ///
215    /// For the purposes of this function, a module is considered a dependency if it is from a
216    /// different source package, and that source package has already been published. Conversely, a
217    /// module that is from a different source package, but that has not been published is
218    /// considered part of the root package.
219    ///
220    /// If the validation mode requires verifying the root package at a specific address, then the
221    /// modules from the root package will be expected at address `0x0` and this address will be
222    /// substituted with the specified address.
223    #[allow(clippy::result_large_err)]
224    fn local(&self, package: &CompiledPackage, env: &Environment) -> Result<LocalModules, Error> {
225        let sui_package = package;
226        let package = &package.package;
227        let root_package = package.compiled_package_info.package_name;
228        let mut map = LocalModules::new();
229
230        if self.verify_deps() {
231            let deps_compiled_units = units_for_toolchain(&package.deps_compiled_units, env)
232                .map_err(|e| Error::CannotCheckLocalModules {
233                    package: package.compiled_package_info.package_name,
234                    message: e.to_string(),
235                })?;
236
237            // TODO: pkg-alt does this still work correctly given that pkg names might have
238            // changed?
239            // only keep modules that are actually used
240            let deps_compiled_units: Vec<_> = deps_compiled_units
241                .into_iter()
242                .filter(|pkg| {
243                    sui_package
244                        .dependency_ids
245                        .published
246                        .contains_key(&Symbol::from(pkg.0.as_str()))
247                })
248                .collect();
249
250            for (package, local_unit) in deps_compiled_units {
251                let m = &local_unit.unit;
252                let module = m.name;
253                let address = m.address.into_inner();
254                let package = Symbol::from(package.as_str());
255
256                // Skip modules with on 0x0 because they are treated as part of the root package,
257                // even if they are a source dependency.
258                if address == AccountAddress::ZERO {
259                    continue;
260                }
261
262                map.insert((address, module), (package, m.module.clone()));
263            }
264        }
265
266        let Self::Root { at, .. } = self else {
267            return Ok(map);
268        };
269
270        // Potentially rebuild according to the toolchain that the package was originally built
271        // with.
272        let root_compiled_units = units_for_toolchain(
273            &package
274                .root_compiled_units
275                .iter()
276                .map(|u| ("root".into(), u.clone()))
277                .collect(),
278            env,
279        )
280        .map_err(|e| Error::CannotCheckLocalModules {
281            package: package.compiled_package_info.package_name,
282            message: e.to_string(),
283        })?;
284
285        // Add the root modules, potentially remapping 0x0 if we have been supplied an address to
286        // substitute with.
287        for (_, local_unit) in root_compiled_units {
288            let m = &local_unit.unit;
289            let module = m.name;
290            let address = m.address.into_inner();
291
292            let (address, compiled_module) = if let Some(root_address) = at {
293                (*root_address, substitute_root_address(m, *root_address)?)
294            } else if address == AccountAddress::ZERO {
295                return Err(Error::InvalidModuleFailure {
296                    name: module.to_string(),
297                    message: "Can't verify unpublished source".to_string(),
298                });
299            } else {
300                (address, m.module.clone())
301            };
302
303            map.insert((address, module), (root_package, compiled_module));
304        }
305
306        // If we have a root address to substitute, we need to find unpublished dependencies that
307        // would have gone into the root package as well.
308        if let Some(root_address) = at {
309            for (package, local_unit) in &package.deps_compiled_units {
310                let m = &local_unit.unit;
311                let module = m.name;
312                let address = m.address.into_inner();
313
314                if address != AccountAddress::ZERO {
315                    continue;
316                }
317
318                map.insert(
319                    (*root_address, module),
320                    (*package, substitute_root_address(m, *root_address)?),
321                );
322            }
323        }
324
325        Ok(map)
326    }
327}

329impl<'a> BytecodeSourceVerifier<'a> {
330    pub fn new(rpc_client: &'a Client) -> Self {
331        BytecodeSourceVerifier { rpc_client }
332    }
333
334    /// Verify that the `compiled_package` matches its on-chain representation.
335    ///
336    /// See [`ValidationMode`] for more details on what is verified.
337    pub async fn verify(
338        &self,
339        package: &CompiledPackage,
340        mode: ValidationMode,
341        env: &Environment,
342    ) -> Result<(), AggregateError> {
343        if matches!(
344            mode,
345            ValidationMode::Root {
346                at: Some(AccountAddress::ZERO),
347                ..
348            }
349        ) {
350            return Err(Error::ZeroOnChainAddresSpecifiedFailure.into());
351        }
352
353        let local = mode.local(package, env)?;
354        let mut chain = mode.on_chain(package, self).await?;
355        let mut errs = vec![];
356
357        // Check that the transitive dependencies listed on chain match the dependencies listed in
358        // source code. Ignore 0x0 becaus this signifies an unpublished dependency.
359        if let Some(on_chain_deps) = &mut chain.on_chain_dependencies {
360            for dependency_id in dependency_addresses(package) {
361                if dependency_id != AccountAddress::ZERO && !on_chain_deps.remove(&dependency_id) {
362                    errs.push(Error::MissingDependencyInLinkageTable(dependency_id));
363                }
364            }
365        }
366
367        for on_chain_dep_id in chain.on_chain_dependencies.take().into_iter().flatten() {
368            errs.push(Error::MissingDependencyInSourcePackage(on_chain_dep_id));
369        }
370
371        // Check that the contents of bytecode matches between modules.
372        for ((address, module), (package, local_module)) in local {
373            let Some(on_chain_module) = chain.modules.remove(&(address, module)) else {
374                errs.push(Error::OnChainDependencyNotFound { package, module });
375                continue;
376            };
377
378            if local_module != on_chain_module {
379                errs.push(Error::ModuleBytecodeMismatch {
380                    address,
381                    package,
382                    module,
383                })
384            }
385        }
386
387        for (address, module) in chain.modules.into_keys() {
388            errs.push(Error::LocalDependencyNotFound { address, module });
389        }
390
391        if !errs.is_empty() {
392            return Err(AggregateError(errs));
393        }
394
395        Ok(())
396    }

398    async fn pkg_for_address(&self, addr: AccountAddress) -> Result<MovePackage, Error> {
399        // Move packages are specified with an AccountAddress, but are
400        // fetched from a sui network via sui_getObject, which takes an object ID
401        let obj_id = ObjectID::from(addr);
402
403        // fetch the Sui object at the address specified for the package in the local resolution table
404        // if future packages with a large set of dependency packages prove too slow to verify,
405        // batched object fetching should be added to the ReadApi & used here
406        let obj = self
407            .rpc_client
408            .clone()
409            .get_object(obj_id)
410            .await
411            .map_err(|e| Error::DependencyObjectReadFailure(e.message().to_owned()))?;
412
413        match &obj.data {
414            sui_types::object::Data::Package(pkg) => Ok(pkg.to_owned()),
415            sui_types::object::Data::Move(_) => Err(Error::ObjectFoundWhenPackageExpected(obj_id)),
416        }
417    }
418}

420#[allow(clippy::result_large_err)]
421fn substitute_root_address(
422    named_module: &NamedCompiledModule,
423    root: AccountAddress,
424) -> Result<CompiledModule, Error> {
425    let mut module = named_module.module.clone();
426    let address_idx = module.self_handle().address;
427
428    let Some(addr) = module.address_identifiers.get_mut(address_idx.0 as usize) else {
429        return Err(Error::InvalidModuleFailure {
430            name: named_module.name.to_string(),
431            message: "Self address field missing".into(),
432        });
433    };
434
435    if *addr != AccountAddress::ZERO {
436        return Err(Error::InvalidModuleFailure {
437            name: named_module.name.to_string(),
438            message: "Self address already populated".to_string(),
439        });
440    }
441
442    *addr = root;
443    Ok(module)
444}

446/// The on-chain addresses for a source package's dependencies
447fn dependency_addresses(package: &CompiledPackage) -> impl Iterator<Item = AccountAddress> + '_ {
448    package.dependency_ids.published.values().map(|id| **id)
