14/// The result of rebuilding the source package with the resolved toolchain binary: the compiled
15/// modules (root package, self-address `0x0`) and the storage ids of its linkage dependencies.
16pub struct GeneratedPackage {
17    pub modules: Vec<CompiledModule>,
18    pub dependencies: Vec<ObjectID>,
19}
20
21/// The JSON printed by `sui move build --dump-bytecode-as-base64`. `digest` is intentionally
22/// ignored — module-by-module comparison localizes mismatches better than a whole-package digest.
23#[derive(Deserialize)]
24struct DumpOutput {
25    modules: Vec<String>,
26    dependencies: Vec<ObjectID>,
27}
28
29/// Rebuild the package at `source_path` with `binary` and return its compiled modules and linkage
30/// dependencies via `--dump-bytecode-as-base64`.
31///
32/// The flag set is adapted to what `binary` supports (probed from its `--help`): older binaries lack
33/// `--no-tree-shaking` (their output includes all dependencies) and the `--build-env` selector.
34///
35/// `--with-unpublished-dependencies` is never passed. Up to v1.66 it means "emit only the modules
36/// still at `0x0`", which is empty for any package that has been published — precisely the packages
37/// being verified. From v1.70 it is a no-op unless the package bundles unpublished dependencies, a
38/// discouraged feature. Omitting it yields the package's modules on every release.
39///
40/// The edition and flavor recorded in the publication are deliberately *not* forced. They only apply
41/// when the manifest omits them, in which case the original build used this very binary's built-in
42/// defaults — which it will use again. Forcing the recorded value instead breaks packages whose
43/// record has since drifted: Pyth's `Move.lock` claims edition 2024 while its sources are legacy
44/// edition, so passing `--default-move-edition 2024` fails to compile a package that builds fine.
45pub fn dump(
46    binary: &Path,
47    source_path: &Path,
48    build_env: &str,
49    client_config: Option<&Path>,
50) -> Result<GeneratedPackage, Error> {
51    let help = help_text(binary);
52    if !help.contains("dump-bytecode-as-base64") {
53        return Err(Error::BuildSubprocess {
54            command: format!("{} move build --help", binary.display()),
55            stderr: "this sui version does not support --dump-bytecode-as-base64; \
56                     it predates source verification support"
57                .to_string(),
58        });
59    }
60
61    let mut args: Vec<String> = vec![
62        "move".into(),
63        "build".into(),
64        "--dump-bytecode-as-base64".into(),
65        "-p".into(),
66        source_path.display().to_string(),
67    ];
68
69    if help.contains("no-tree-shaking") {
70        args.push("--no-tree-shaking".into());
71    }
72    // The environment selector picks which environment's dependencies to resolve; without it the
73    // build cannot decide between e.g. testnet and mainnet. Only releases that support it need it:
74    // earlier ones have a single dependency set.
75    if help.contains("build-env") {
76        args.push("--build-env".into());
77        args.push(build_env.to_string());
78    }
79    let mut command = Command::new(binary);
80    command.args(&args);
81
82    // Releases that cannot disable tree-shaking must reach the network during the dump, and they
83    // locate the wallet through `SUI_CONFIG_DIR`. Without this the subprocess would consult the
84    // user's default configuration rather than the network being verified against.
85    if let Some(dir) = client_config.and_then(|path| path.parent()) {
86        command.env("SUI_CONFIG_DIR", dir);
87    }
88
89    let output = command.output().map_err(|e| Error::BuildSubprocess {
90        command: display_command(binary, &args),
91        stderr: format!("could not run the build: {e}"),
92    })?;
93
94    if !output.status.success() {
95        // Build diagnostics are not consistently written to stderr — dependency-resolution errors,
96        // for instance, are printed to stdout — so surface both.
97        let mut diagnostics = String::from_utf8_lossy(&output.stderr).into_owned();
98        diagnostics.push_str(&String::from_utf8_lossy(&output.stdout));
99        return Err(Error::BuildSubprocess {
100            command: display_command(binary, &args),
101            stderr: diagnostics,
102        });
103    }
104
105    parse_dump(&output.stdout)
106}

108/// Deserialize and decode the dump JSON into a [`GeneratedPackage`].
109fn parse_dump(stdout: &[u8]) -> Result<GeneratedPackage, Error> {
110    let parse_err = |message: String| Error::BuildOutputParse { message };
111
112    let dump: DumpOutput = serde_json::from_slice(stdout)
113        .map_err(|e| parse_err(format!("invalid build output JSON: {e}")))?;
114
115    let modules = dump
116        .modules
117        .iter()
118        .map(|b64| {
119            let bytes = Base64::decode(b64)
120                .map_err(|e| parse_err(format!("module is not valid base64: {e}")))?;
121            CompiledModule::deserialize_with_defaults(&bytes)
122                .map_err(|e| parse_err(format!("could not deserialize module bytecode: {e}")))
123        })
124        .collect::<Result<Vec<_>, _>>()?;
125
126    Ok(GeneratedPackage {
127        modules,
128        dependencies: dump.dependencies,
129    })
130}

132/// Capture `binary move build --help` (stdout and stderr combined) for flag probing.
133fn help_text(binary: &Path) -> String {
134    let Ok(output) = Command::new(binary)
135        .args(["move", "build", "--help"])
136        .output()
137    else {
138        return String::new();
139    };
140    let mut text = String::from_utf8_lossy(&output.stdout).into_owned();
141    text.push_str(&String::from_utf8_lossy(&output.stderr));
142    text
143}
144
145fn display_command(binary: &Path, args: &[String]) -> String {
146    format!("{} {}", binary.display(), args.join(" "))
