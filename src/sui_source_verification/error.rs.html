4use std::fmt;
5
6use move_core_types::account_address::AccountAddress;
7use sui_types::base_types::ObjectID;
8
9/// One or more [`Error`]s from a verification run. Comparison errors are collected so that all
10/// mismatches are reported together; fatal setup errors are reported singly.
11#[derive(Debug, thiserror::Error)]
12pub struct AggregateError(pub(crate) Vec<Error>);
13
14#[derive(Debug, thiserror::Error)]
15pub enum Error {
16    //
17    // Fatal setup errors (abort before comparison)
18    //
19    #[error(
20        "The package records no toolchain version. \
21         Pass --toolchain-version to specify the version to rebuild it with."
22    )]
23    ToolchainVersionNotFound,
24
25    #[error(
26        "The recorded toolchain sui {version} cannot verify this package: {reason}. \
27         Passing --toolchain-version {alternative} may work, since compiler output rarely changes \
28         between adjacent releases."
29    )]
30    UnsupportedToolchain {
31        version: String,
32        reason: String,
33        alternative: String,
34    },
35
36    #[error(
37        "Failed to download sui toolchain version {version}: {message}\n\
38         Pass --toolchain-version to rebuild with a different toolchain."
39    )]
40    BinaryDownload { version: String, message: String },
41
42    #[error(
43        "The build command failed.\nCommand: {command}\n--- stderr ---\n{stderr}\n\
44         Pass --toolchain-version to rebuild with a different toolchain."
45    )]
46    BuildSubprocess { command: String, stderr: String },
47
48    #[error("Could not parse the output of the build command: {message}")]
49    BuildOutputParse { message: String },
50
51    #[error("Could not read the on-chain package: {0}")]
52    PackageReadFailure(String),
53
54    #[error("Object at {0} is a Move object, not a package")]
55    ObjectFoundWhenPackageExpected(ObjectID),
56
57    #[error("On-chain package {0} is empty")]
58    EmptyOnChainPackage(AccountAddress),
59
60    #[error("Could not deserialize on-chain module {address}::{module}")]
61    OnChainModuleDeserialization {
62        address: AccountAddress,
63        module: String,
64    },
65
66    #[error(
67        "The on-chain package's original id ({on_chain}) does not match the one the source records \
68         ({recorded}); the package at the recorded address is not the one the source describes."
69    )]
70    OriginalIdMismatch {
71        recorded: AccountAddress,
72        on_chain: AccountAddress,
73    },
74
75    //
76    // Comparison errors (aggregated)
77    //
78    #[error("Invalid module {name}: {message}")]
79    InvalidModule { name: String, message: String },
80
81    #[error("Module {module} bytecode does not match its on-chain version")]
82    ModuleBytecodeMismatch { module: String },
83
84    #[error("Module {module} is produced by the source but is not present on-chain")]
85    SourceModuleNotOnChain { module: String },
86
87    #[error("Module {module} is present on-chain but is not produced by the source")]
88    OnChainModuleNotInSource { module: String },
89
90    #[error(
91        "Dependency {original} is linked at a different version: on-chain {on_chain}, source {in_source}"
92    )]
93    LinkageVersionMismatch {
94        original: AccountAddress,
95        on_chain: AccountAddress,
96        in_source: AccountAddress,
97    },
98
99    #[error("Source depends on package {0} which could not be found on-chain")]
100    SourceDependencyNotOnChain(AccountAddress),
101
102    #[error(
103        "Dependency `{dependency}` is pinned to `{rev}`, which is not a commit hash, so its \
104         contents can change over time. This package cannot be rebuilt reproducibly; pin its \
105         dependencies to commit hashes."
106    )]
107    NonReproducibleDependency { dependency: String, rev: String },
108}
109
110impl fmt::Display for AggregateError {
111    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
112        let Self(errors) = self;
113        match &errors[..] {
114            [] => unreachable!("Aggregate error with no errors"),
115            [error] => write!(f, "{}", error),
116            errors => {
117                writeln!(f, "Multiple source verification errors found:")?;
118                for error in errors {
119                    write!(f, "\n- {}", error)?;
120                }
121                Ok(())
122            }
123        }
124    }
125}
126
127impl From<Error> for AggregateError {
128    fn from(error: Error) -> Self {
129        Self(vec![error])
130    }
131}
