15/// The on-chain representation of a package: its modules (keyed by name), its original (runtime)
16/// package id, and its linkage table reduced to `original id -> storage id`.
17pub struct OnChainPackage {
18    pub original_id: AccountAddress,
19    pub modules: BTreeMap<String, CompiledModule>,
20    pub linkage: BTreeMap<AccountAddress, AccountAddress>,
21}
22
23/// Fetch and decode the on-chain package at `on_chain_id`.
24pub async fn fetch(client: &Client, on_chain_id: ObjectID) -> Result<OnChainPackage, Error> {
25    let pkg = pkg_for_address(client, on_chain_id).await?;
26
27    let mut modules = BTreeMap::new();
28    for (name, bytes) in pkg.serialized_module_map() {
29        let module = CompiledModule::deserialize_with_defaults(bytes).map_err(|_| {
30            Error::OnChainModuleDeserialization {
31                address: on_chain_id.into(),
32                module: name.as_str().into(),
33            }
34        })?;
35        modules.insert(name.as_str().to_string(), module);
36    }
37
38    if modules.is_empty() {
39        return Err(Error::EmptyOnChainPackage(on_chain_id.into()));
40    }
41
42    let linkage = pkg
43        .linkage_table()
44        .iter()
45        .map(|(original, info)| ((*original).into(), info.upgraded_id.into()))
46        .collect();
47
48    Ok(OnChainPackage {
49        original_id: pkg.original_package_id().into(),
50        modules,
51        linkage,
52    })
53}
54
55/// Fetch the object at `id`, requiring it to be a Move package.
56pub(crate) async fn pkg_for_address(client: &Client, id: ObjectID) -> Result<MovePackage, Error> {
57    let obj = client
58        .clone()
59        .get_object(id)
60        .await
61        .map_err(|e| Error::PackageReadFailure(e.to_string()))?;
62
63    match &obj.data {
64        Data::Package(pkg) => Ok(pkg.clone()),
65        Data::Move(_) => Err(Error::ObjectFoundWhenPackageExpected(id)),
66    }
67}
