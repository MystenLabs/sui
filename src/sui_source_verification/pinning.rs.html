4use std::path::Path;
5
6use move_package_alt::SourcePackageLayout;
7
8/// A dependency pinned to a revision that can move, such as a branch or a tag.
9#[derive(Debug, PartialEq, Eq, PartialOrd, Ord)]
10pub struct MovingRevision {
11    pub dependency: String,
12    pub rev: String,
13}
14
15/// Find the dependencies of the package at `root_path` that are pinned to a moving revision.
16///
17/// Legacy lockfiles record the manifest's `rev` verbatim, so a dependency on a branch such as
18/// `framework/mainnet` resolves to whatever that branch points at when the package is rebuilt,
19/// rather than to what it pointed at when the package was published. Such a package cannot be
20/// rebuilt reproducibly. Lockfiles written by the current package system always pin commit hashes.
21pub fn moving_revisions(root_path: &Path) -> Vec<MovingRevision> {
22    let path = root_path.join(SourcePackageLayout::Lock.path());
23    let Ok(contents) = std::fs::read_to_string(path) else {
24        return vec![];
25    };
26    moving_revisions_in(&contents)
27}
28
29/// The moving revisions recorded in the lockfile `contents`.
30fn moving_revisions_in(contents: &str) -> Vec<MovingRevision> {
31    let Ok(lock) = contents.parse::<toml::Value>() else {
32        return vec![];
33    };
34
35    let mut found = vec![];
36    collect(&lock, None, &mut found);
37    found.sort();
38    found.dedup();
39    found
40}
41
42/// Walk `value`, recording any table with a `source.rev` that is not a commit hash. Lockfiles have
43/// labelled dependencies with `id` and, in older versions, `name`; `key` is the name the table was
44/// found under, used when it carries neither.
45fn collect(value: &toml::Value, key: Option<&str>, found: &mut Vec<MovingRevision>) {
46    if let Some(elements) = value.as_array() {
47        for element in elements {
48            collect(element, key, found);
49        }
50        return;
51    }
52
53    let Some(table) = value.as_table() else {
54        return;
55    };
56
57    if let Some(rev) = table
58        .get("source")
59        .and_then(|source| source.get("rev"))
60        .and_then(|rev| rev.as_str())
61        && !is_commit_hash(rev)
62    {
63        let dependency = table
64            .get("id")
65            .or_else(|| table.get("name"))
66            .and_then(|name| name.as_str())
67            .or(key)
68            .unwrap_or("<unknown>");
69
70        found.push(MovingRevision {
71            dependency: dependency.to_string(),
72            rev: rev.to_string(),
73        });
74    }
75
76    for (name, child) in table {
77        collect(child, Some(name), found);
78    }
79}
80
81/// Whether `rev` looks like a git commit hash (pinned) rather than a branch or tag (moving).
82///
83/// Lockfiles historically recorded truncated hashes as short as ten hex digits, so any all-hex
84/// string of at least seven characters (git's shortest abbreviation) is treated as a pinned commit;
85/// a hex run that long is overwhelmingly a commit rather than a branch name.
86fn is_commit_hash(rev: &str) -> bool {
87    (7..=40).contains(&rev.len()) && rev.bytes().all(|b| b.is_ascii_hexdigit())
88}

90#[cfg(test)]
91mod tests {
92    use super::*;
93
94    /// A legacy lockfile records the manifest's `rev` verbatim, so branches survive into it.
95    #[test]
96    fn legacy_lock_reports_branches_but_not_hashes() {
97        let lock = r#"
98            [move]
99            version = 3
100
101            [[move.package]]
102            id = "Sui"
103            source = { git = "https://github.com/MystenLabs/sui.git", rev = "framework/mainnet", subdir = "x" }
104
105            [[move.package]]
106            id = "MoveStdlib"
107            source = { git = "https://github.com/MystenLabs/sui.git", rev = "041c5f2bae2fe52079e44b70514333532d69f4e6" }
108        "#;
109
110        assert_eq!(
111            moving_revisions_in(lock),
112            vec![MovingRevision {
113                dependency: "Sui".to_string(),
114                rev: "framework/mainnet".to_string(),
115            }]
116        );
117    }
118
119    /// The current package system pins commit hashes per environment, so nothing is reported.
120    #[test]
121    fn pinned_lock_reports_nothing() {
122        let lock = r#"
123            [pinned.mainnet.token]
124            source = { git = "https://github.com/MystenLabs/deepbookv3.git", rev = "5e82e2dd1ea7d47957855ddc66f835585d6fe091" }
125        "#;
126
127        assert!(moving_revisions_in(lock).is_empty());
128    }
129
130    /// Older lockfiles label dependencies with `name` rather than `id`.
131    #[test]
132    fn legacy_lock_uses_the_name_field() {
133        let lock = r#"
134            [[move.package]]
135            name = "Sui"
136            source = { git = "https://github.com/MystenLabs/sui.git", rev = "framework/mainnet" }
137        "#;
138
139        assert_eq!(
140            moving_revisions_in(lock),
141            vec![MovingRevision {
142                dependency: "Sui".to_string(),
143                rev: "framework/mainnet".to_string(),
144            }]
145        );
146    }
147
148    /// Truncated commit hashes (historically as short as ten hex digits) still pin a commit.
149    #[test]
150    fn truncated_hash_is_not_reported() {
151        let lock = r#"
152            [[move.package]]
153            id = "Sui"
154            source = { git = "https://github.com/MystenLabs/sui.git", rev = "041c5f2bae" }
155        "#;
156
157        assert!(moving_revisions_in(lock).is_empty());
158    }
159
160    /// A dependency with neither `id` nor `name` is named after the table it was found under.
161    #[test]
162    fn dependency_is_named_after_its_table_when_unlabelled() {
163        let lock = r#"
164            [pinned.mainnet.token]
165            source = { git = "https://example.com/x.git", rev = "main" }
166        "#;
167
168        assert_eq!(
169            moving_revisions_in(lock),
170            vec![MovingRevision {
171                dependency: "token".to_string(),
172                rev: "main".to_string(),
173            }]
174        );
175    }
