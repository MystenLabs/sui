4use crate::object_store::ObjectStoreGetExt;
5use crate::object_store::util::path_to_filesystem;
6use anyhow::{Context, Result, anyhow};
7use async_trait::async_trait;
8use bytes::Bytes;
9use object_store::path::Path;
10use std::fs::File;
11use std::io::Read;
12use std::path::PathBuf;
13use std::{fmt, fs};
14
15pub struct LocalStorage {
16    root: PathBuf,
17}
18
19impl LocalStorage {
20    pub fn new(directory: &std::path::Path) -> Result<Self> {
21        let path = fs::canonicalize(directory).context(anyhow!("Unable to canonicalize"))?;
22        fs::create_dir_all(&path).context(anyhow!(
23            "Failed to create local directory: {}",
24            path.display()
25        ))?;
26        Ok(LocalStorage { root: path })
27    }
28}
29
30impl fmt::Display for LocalStorage {
31    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
32        write!(f, "local:{}", self.root.display())
33    }
34}
35
36#[async_trait]
37impl ObjectStoreGetExt for LocalStorage {
38    async fn get_bytes(&self, location: &Path) -> Result<Bytes> {
39        let path_to_filesystem = path_to_filesystem(self.root.clone(), location)?;
40        let handle = tokio::task::spawn_blocking(move || {
41            let mut f = File::open(path_to_filesystem)
42                .map_err(|e| anyhow!("Failed to open file with error: {}", e.to_string()))?;
43            let mut buf = vec![];
44            f.read_to_end(&mut buf)
45                .context(anyhow!("Failed to read file"))?;
46            Ok(buf.into())
47        });
48        handle.await?
49    }
50}
