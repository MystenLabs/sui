29pub async fn run(
30    run_duration: Duration,
31    epoch_duration: Duration,
32    package_paths: Vec<PathBuf>,
33) -> SurfStatistics {
34    let cluster = TestClusterBuilder::new()
35        .with_num_validators(VALIDATOR_COUNT)
36        .with_epoch_duration_ms(epoch_duration.as_millis() as u64)
37        .with_accounts(vec![
38            AccountConfig {
39                address: None,
40                gas_amounts: vec![DEFAULT_GAS_AMOUNT; GAS_OBJECT_COUNT],
41            };
42            ACCOUNT_NUM
43        ])
44        .build()
45        .await;
46    info!(
47        "Started cluster with {} validators and epoch duration of {:?}ms",
48        VALIDATOR_COUNT,
49        epoch_duration.as_millis()
50    );
51    run_with_test_cluster(run_duration, package_paths, cluster.into(), 0).await
52}

54pub async fn run_with_test_cluster(
55    run_duration: Duration,
56    package_paths: Vec<PathBuf>,
57    cluster: Arc<TestCluster>,
58    // Skips the first N accounts, for use in case this is running concurrently with other
59    // processes that also need gas.
60    skip_accounts: usize,
61) -> SurfStatistics {
62    run_with_test_cluster_and_strategy(
63        SurfStrategy::default(),
64        run_duration,
65        package_paths,
66        cluster,
67        skip_accounts,
68    )
69    .await
70}

72pub async fn run_with_test_cluster_and_strategy(
73    surf_strategy: SurfStrategy,
74    run_duration: Duration,
75    package_paths: Vec<PathBuf>,
76    cluster: Arc<TestCluster>,
77    // Skips the first N accounts, for use in case this is running concurrently with other
78    // processes that also need gas.
79    skip_accounts: usize,
80) -> SurfStatistics {
81    let seed = rand::thread_rng().r#gen::<u64>();
82    info!("Initial Seed: {:?}", seed);
83    let mut rng = StdRng::seed_from_u64(seed);
84    let (exit_sender, exit_rcv) = watch::channel(());
85
86    let mut tasks = SurferTask::create_surfer_tasks(
87        cluster.clone(),
88        rng.r#gen::<u64>(),
89        exit_rcv,
90        skip_accounts,
91        surf_strategy,
92    )
93    .await;
94    info!("Created {} surfer tasks", tasks.len());
95
96    for path in &package_paths {
97        tasks
98            .choose_mut(&mut rng)
99            .unwrap()
100            .state
101            .publish_package(path)
102            .await;
103    }
104
105    let mut handles = vec![];
106    for task in tasks {
107        handles.push(tokio::task::spawn(task.surf()));
108    }
109    tokio::time::sleep(run_duration).await;
110    exit_sender.send(()).unwrap();
111    let all_stats: Result<Vec<_>, _> = join_all(handles).await.into_iter().collect();
112    SurfStatistics::aggregate(all_stats.unwrap())
113
114    // TODO: Right now it will panic here complaining about dropping a tokio runtime
115    // inside of another tokio runtime. Reason unclear.
116}
