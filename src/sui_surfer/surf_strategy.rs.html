4use std::time::Duration;
5
6use move_binary_format::normalized;
7use move_core_types::language_storage::StructTag;
8use rand::{Rng, seq::SliceRandom};
9use sui_types::{
10    base_types::ObjectRef,
11    transaction::{CallArg, ObjectArg, SharedObjectMutability},
12};
13use tokio::time::Instant;
14use tracing::debug;
15
16use crate::surfer_state::{EntryFunction, SurferState};
17
18enum InputObjectPassKind {
19    Value,
20    ByRef,
21    MutRef,
22}
23
24type Type = normalized::Type<normalized::ArcIdentifier>;
25
26#[derive(Clone, Default)]
27pub struct SurfStrategy {
28    min_tx_interval: Duration,
29}
30
31impl SurfStrategy {
32    pub fn new(min_tx_interval: Duration) -> Self {
33        Self { min_tx_interval }
34    }
35
36    /// Given a state and a list of callable Move entry functions,
37    /// explore them for a while, and eventually return. This function may
38    /// not return in some situations, so its important to call it with a
39    /// timeout or select! to ensure the task doesn't block forever.
40    pub async fn surf_for_a_while(
41        &mut self,
42        state: &mut SurferState,
43        mut entry_functions: Vec<EntryFunction>,
44    ) {
45        entry_functions.shuffle(&mut state.rng);
46        for entry in entry_functions {
47            let next_tx_time = Instant::now() + self.min_tx_interval;
48            let Some(args) = Self::choose_function_call_args(state, entry.parameters).await else {
49                debug!(
50                    "Failed to choose arguments for Move function {:?}::{:?}",
51                    entry.module, entry.function
52                );
53                continue;
54            };
55            state
56                .execute_move_transaction(entry.package, entry.module, entry.function, args)
57                .await;
58            tokio::time::sleep_until(next_tx_time).await;
59        }
60    }
61
62    async fn choose_function_call_args(
63        state: &mut SurferState,
64        params: Vec<Type>,
65    ) -> Option<Vec<CallArg>> {
66        let mut args = vec![];
67        let mut chosen_owned_objects = vec![];
68        let mut failed = false;
69        for param in params {
70            let arg = match param {
71                Type::Bool => CallArg::Pure(bcs::to_bytes(&state.rng.r#gen::<bool>()).unwrap()),
72                Type::U8 => CallArg::Pure(bcs::to_bytes(&state.rng.r#gen::<u8>()).unwrap()),
73                Type::U16 => CallArg::Pure(bcs::to_bytes(&state.rng.r#gen::<u16>()).unwrap()),
74                Type::U32 => CallArg::Pure(bcs::to_bytes(&state.rng.r#gen::<u32>()).unwrap()),
75                Type::U64 => CallArg::Pure(bcs::to_bytes(&state.rng.r#gen::<u64>()).unwrap()),
76                Type::U128 => CallArg::Pure(bcs::to_bytes(&state.rng.r#gen::<u128>()).unwrap()),
77                Type::Address => CallArg::Pure(
78                    bcs::to_bytes(&state.cluster.get_addresses().choose(&mut state.rng)).unwrap(),
79                ),
80                ty @ Type::Datatype(_) => {
81                    match Self::choose_object_call_arg(
82                        state,
83                        InputObjectPassKind::Value,
84                        ty,
85                        &mut chosen_owned_objects,
86                    )
87                    .await
88                    {
89                        Some(arg) => arg,
90                        None => {
91                            failed = true;
92                            break;
93                        }
94                    }
95                }
96                Type::Reference(mut_, ty) => {
97                    let kind = if mut_ {
98                        InputObjectPassKind::MutRef
99                    } else {
100                        InputObjectPassKind::ByRef
101                    };
102                    match Self::choose_object_call_arg(state, kind, *ty, &mut chosen_owned_objects)
103                        .await
104                    {
105                        Some(arg) => arg,
106                        None => {
107                            failed = true;
108                            break;
109                        }
110                    }
111                }
112                Type::U256 | Type::Signer | Type::Vector(_) | Type::TypeParameter(_) => {
113                    failed = true;
114                    break;
115                }
116            };
117            args.push(arg);
118        }
119        if failed {
120            for (struct_tag, obj_ref) in chosen_owned_objects {
121                state
122                    .owned_objects
123                    .get_mut(&struct_tag)
124                    .unwrap()
125                    .insert(obj_ref);
126            }
127            None
128        } else {
129            Some(args)
130        }
131    }
132
133    async fn choose_object_call_arg(
134        state: &mut SurferState,
135        kind: InputObjectPassKind,
136        arg_type: Type,
137        chosen_owned_objects: &mut Vec<(StructTag, ObjectRef)>,
138    ) -> Option<CallArg> {
139        let pool = state.pool.read().await;
140        let type_tag = match arg_type {
141            Type::Datatype(dt) => dt.to_struct_tag(&*pool),
142            _ => {
143                return None;
144            }
145        };
146        drop(pool);
147        let owned = state.matching_owned_objects_count(&type_tag);
148        let shared = state.matching_shared_objects_count(&type_tag).await;
149        let immutable = state.matching_immutable_objects_count(&type_tag).await;
150
151        let total_matching_count = match kind {
152            InputObjectPassKind::Value => owned,
153            InputObjectPassKind::MutRef => owned + shared,
154            InputObjectPassKind::ByRef => owned + shared + immutable,
155        };
156        if total_matching_count == 0 {
157            return None;
158        }
159        let mut n = state.rng.gen_range(0..total_matching_count);
160        if n < owned {
161            let obj_ref = state.choose_nth_owned_object(&type_tag, n);
162            chosen_owned_objects.push((type_tag, obj_ref));
163            return Some(CallArg::Object(ObjectArg::ImmOrOwnedObject(obj_ref)));
164        }
165        n -= owned;
166        if n < shared {
167            let (id, initial_shared_version) = state.choose_nth_shared_object(&type_tag, n).await;
168            return Some(CallArg::Object(ObjectArg::SharedObject {
169                id,
170                initial_shared_version,
171                mutability: if matches!(kind, InputObjectPassKind::MutRef) {
172                    SharedObjectMutability::Mutable
173                } else {
174                    SharedObjectMutability::Immutable
175                },
176            }));
177        }
178        n -= shared;
179        let obj_ref = state.choose_nth_immutable_object(&type_tag, n).await;
180        Some(CallArg::Object(ObjectArg::ImmOrOwnedObject(obj_ref)))
181    }
182}
