11pub(crate) const GA_API_SECRET: &str = "zeq-aYEzS0aGdRJ8kNZTEg";
12pub(crate) const GA_EVENT_NAME: &str = "node_telemetry_event";

20pub(crate) const UNKNOWN_STRING: &str = "UNKNOWN";

22#[derive(Debug, Serialize, Deserialize)]
23struct TelemetryEvent {
24    name: String,
25    params: BTreeMap<String, String>,
26}
27
28// The payload needs to meet this requirement in
29// https://developers.google.com/analytics/devguides/collection/protocol/ga4/reference?client_type=gtag#payload_post_body
30#[derive(Debug, Serialize, Deserialize)]
31struct TelemetryPayload {
32    client_id: String,
33    events: Vec<TelemetryEvent>,
34}
35
36#[derive(Debug, Serialize, Deserialize)]
37struct IpResponse {
38    ip: String,
39}

41pub async fn send_telemetry_event(state: Arc<AuthorityState>, is_validator: bool) {
42    let git_rev = env!("CARGO_PKG_VERSION").to_string();
43    let ip_address = get_ip().await;
44    let chain_identifier = state.get_chain_identifier().to_string();
45    let since_the_epoch = SystemTime::now()
46        .duration_since(UNIX_EPOCH)
47        .expect("Now should be later than epoch!");
48    let telemetry_event = TelemetryEvent {
49        name: GA_EVENT_NAME.into(),
50        params: BTreeMap::from([
51            ("chain_identifier".into(), chain_identifier),
52            ("node_address".into(), ip_address),
53            (
54                "node_type".into(),
55                if is_validator {
56                    "validator".into()
57                } else {
58                    "full_node".into()
59                },
60            ),
61            ("git_rev".into(), git_rev),
62            (
63                "seconds_since_epoch".into(),
64                since_the_epoch.as_secs().to_string(),
65            ),
66        ]),
67    };
68
69    let telemetry_payload = TelemetryPayload {
70        client_id: HARDCODED_CLIENT_ID.into(),
71        events: vec![telemetry_event],
72    };
73
74    send_telemetry_event_impl(telemetry_payload).await
75}
76
77async fn get_ip() -> String {
78    let resp = reqwest::get(IPLOOKUP_URL).await;
79    match resp {
80        Ok(json) => match json.json::<IpResponse>().await {
81            Ok(ip_json) => ip_json.ip,
82            Err(_) => UNKNOWN_STRING.into(),
83        },
84        Err(_) => UNKNOWN_STRING.into(),
85    }
86}

88async fn send_telemetry_event_impl(telemetry_payload: TelemetryPayload) {
89    let client = reqwest::Client::new();
90    let response_result = client
91        .post(format!(
92            "{}?&measurement_id={}&api_secret={}",
93            GA_URL, GA_MEASUREMENT_ID, GA_API_SECRET
94        ))
95        .json::<TelemetryPayload>(&telemetry_payload)
96        .send()
97        .await;
98
99    match response_result {
100        Ok(response) => {
101            let status = response.status().as_u16();
102            if (200..299).contains(&status) {
103                trace!("SUCCESS: Sent telemetry event: {:?}", &telemetry_payload,);
104            } else {
105                trace!(
106                    "FAIL: Sending telemetry event failed with status: {} and response {:?}.",
107                    response.status(),
108                    response
109                );
110            }
111        }
112        Err(error) => {
113            trace!(
114                "FAIL: Sending telemetry event failed with error: {:?}",
115                error
116            );
117        }
118    }
119}
