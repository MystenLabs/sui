389async fn check_locked_object(
390    sui_client: &Arc<SuiClient>,
391    committee: Arc<BTreeMap<AuthorityPublicKeyBytes, u64>>,
392    id: ObjectID,
393    rescue: bool,
394) -> anyhow::Result<()> {
395    let clients = Arc::new(make_clients(sui_client).await?);
396    let output = get_object(id, None, None, clients.clone()).await?;
397    let output = GroupedObjectOutput::new(output, committee);
398    if output.fully_locked {
399        println!("Object {} is fully locked.", id);
400        return Ok(());
401    }
402    let top_record = output.voting_power.first().unwrap();
403    let top_record_stake = top_record.1;
404    let top_record = top_record.0.clone().unwrap();
405    if top_record.4.is_none() {
406        println!(
407            "Object {} does not seem to be locked by majority of validators (unlocked stake: {})",
408            id, top_record_stake
409        );
410        return Ok(());
411    }
412
413    let tx_digest = top_record.2;
414    if !rescue {
415        println!("Object {} is rescueable, top tx: {:?}", id, tx_digest);
416        return Ok(());
417    }
418    println!("Object {} is rescueable, trying tx {}", id, tx_digest);
419    let validator = output
420        .grouped_results
421        .get(&Some(top_record))
422        .unwrap()
423        .first()
424        .unwrap();
425    let client = &clients.get(validator).unwrap().1;
426    let tx = client
427        .handle_transaction_info_request(TransactionInfoRequest {
428            transaction_digest: tx_digest,
429        })
430        .await?
431        .transaction;
432    let res = sui_client
433        .quorum_driver_api()
434        .execute_transaction_block(
435            Transaction::new(tx),
436            SuiTransactionBlockResponseOptions::full_content(),
437            None,
438        )
439        .await;
440    match res {
441        Ok(_) => {
442            println!("Transaction executed successfully ({:?})", tx_digest);
443        }
444        Err(e) => {
445            println!("Failed to execute transaction ({:?}): {:?}", tx_digest, e);
446        }
447    }
448    Ok(())
449}

451impl ToolCommand {
452    #[allow(clippy::format_in_format_args)]
453    pub async fn execute(self, tracing_handle: TracingHandle) -> Result<(), anyhow::Error> {
454        match self {
455            ToolCommand::ScanConsensusCommits {
456                db_path,
457                start_commit,
458                end_commit,
459            } => {
460                let rocks_db_store = RocksDBStore::new(&db_path);
461
462                let start_commit = start_commit.unwrap_or(0);
463                let end_commit = end_commit.unwrap_or(u32::MAX);
464
465                let commits = rocks_db_store
466                    .scan_commits(CommitRange::new(start_commit..=end_commit))
467                    .unwrap();
468                println!("found {} consensus commits", commits.len());
469
470                for commit in commits {
471                    let inner = &*commit;
472                    let block_refs = inner.blocks();
473                    let blocks = rocks_db_store.read_blocks(block_refs).unwrap();
474
475                    for block in blocks.iter().flatten() {
476                        let data = block.transactions_data();
477                        println!(
478                            "\"index\": \"{}\", \"leader\": \"{}\", \"blocks\": \"{:#?}\", {} txs",
479                            inner.index(),
480                            inner.leader(),
481                            inner.blocks(),
482                            data.len()
483                        );
484                        for txns in &data {
485                            let tx: ConsensusTransaction = bcs::from_bytes(txns).unwrap();
486                            println!("\t{:?}", tx.key());
487                        }
488                    }
489                }
490            }
491            ToolCommand::LockedObject {
492                id,
493                fullnode_rpc_url,
494                rescue,
495                address,
496            } => {
497                let sui_client =
498                    Arc::new(SuiClientBuilder::default().build(fullnode_rpc_url).await?);
499                let committee = Arc::new(
500                    sui_client
501                        .governance_api()
502                        .get_committee_info(None)
503                        .await?
504                        .validators
505                        .into_iter()
506                        .collect::<BTreeMap<_, _>>(),
507                );
508                let object_ids = match id {
509                    Some(id) => vec![id],
510                    None => {
511                        let address = address.expect("Either id or address must be provided");
512                        sui_client
513                            .coin_read_api()
514                            .get_coins_stream(address, None)
515                            .map(|c| c.coin_object_id)
516                            .collect()
517                            .await
518                    }
519                };
520                for ids in object_ids.chunks(30) {
521                    let mut tasks = vec![];
522                    for id in ids {
523                        tasks.push(check_locked_object(
524                            &sui_client,
525                            committee.clone(),
526                            *id,
527                            rescue,
528                        ))
529                    }
530                    join_all(tasks)
531                        .await
532                        .into_iter()
533                        .collect::<Result<Vec<_>, _>>()?;
534                }
535            }
536            ToolCommand::FetchObject {
537                id,
538                validator,
539                version,
540                fullnode_rpc_url,
541                verbosity,
542                concise_no_header,
543            } => {
544                let sui_client =
545                    Arc::new(SuiClientBuilder::default().build(fullnode_rpc_url).await?);
546                let clients = Arc::new(make_clients(&sui_client).await?);
547                let output = get_object(id, version, validator, clients).await?;
548
549                match verbosity {
550                    Verbosity::Grouped => {
551                        let committee = Arc::new(
552                            sui_client
553                                .governance_api()
554                                .get_committee_info(None)
555                                .await?
556                                .validators
557                                .into_iter()
558                                .collect::<BTreeMap<_, _>>(),
559                        );
560                        println!("{}", GroupedObjectOutput::new(output, committee));
561                    }
562                    Verbosity::Verbose => {
563                        println!("{}", VerboseObjectOutput(output));
564                    }
565                    Verbosity::Concise => {
566                        if !concise_no_header {
567                            println!("{}", ConciseObjectOutput::header());
568                        }
569                        println!("{}", ConciseObjectOutput(output));
570                    }
571                }
572            }
573            ToolCommand::FetchTransaction {
574                digest,
575                show_input_tx,
576                fullnode_rpc_url,
577            } => {
578                print!(
579                    "{}",
580                    get_transaction_block(digest, show_input_tx, fullnode_rpc_url).await?
581                );
582            }
583            #[cfg(not(tidehunter))]
584            ToolCommand::DbTool { db_path, cmd } => {
585                let path = PathBuf::from(db_path);
586                match cmd {
587                    Some(c) => execute_db_tool_command(path, c).await?,
588                    None => print_db_all_tables(path)?,
589                }
590            }
591            ToolCommand::DumpPackages {
592                rpc_url,
593                output_dir,
594                before_checkpoint,
595                verbose,
596            } => {
597                if !verbose {
598                    tracing_handle
599                        .update_log("off")
600                        .expect("Failed to update log level");
601                }
602
603                sui_package_dump::dump(rpc_url, output_dir, before_checkpoint).await?;
604            }
605            ToolCommand::DumpValidators { genesis, concise } => {
606                let genesis = Genesis::load(genesis).unwrap();
607                if !concise {
608                    println!("{:#?}", genesis.validator_set_for_tooling());
609                } else {
610                    for (i, val_info) in genesis.validator_set_for_tooling().iter().enumerate() {
611                        let metadata = val_info.verified_metadata();
612                        println!(
613                            "#{:<2} {:<20} {:?} {:?} {}",
614                            i,
615                            metadata.name,
616                            metadata.sui_pubkey_bytes().concise(),
617                            metadata.net_address,
618                            anemo::PeerId(metadata.network_pubkey.0.to_bytes()),
619                        )
620                    }
621                }
622            }
623            ToolCommand::DumpGenesis { genesis } => {
624                let genesis = Genesis::load(genesis)?;
625                println!("{:#?}", genesis);
626            }
627            ToolCommand::FetchCheckpoint {
628                sequence_number,
629                fullnode_rpc_url,
630            } => {
631                let sui_client =
632                    Arc::new(SuiClientBuilder::default().build(fullnode_rpc_url).await?);
633                let clients = make_clients(&sui_client).await?;
634
635                for (name, (_, client)) in clients {
636                    let resp = client
637                        .handle_checkpoint(CheckpointRequest {
638                            sequence_number,
639                            request_content: true,
640                        })
641                        .await
642                        .unwrap();
643                    let CheckpointResponse {
644                        checkpoint,
645                        contents,
646                    } = resp;
647
648                    let summary = checkpoint.clone().unwrap().data().clone();
649                    // write summary to file
650                    let mut file = std::fs::File::create("/tmp/ckpt_summary")
651                        .expect("Failed to create /tmp/summary");
652                    let bytes =
653                        bcs::to_bytes(&summary).expect("Failed to serialize summary to BCS");
654                    use std::io::Write;
655                    file.write_all(&bytes)
656                        .expect("Failed to write summary to /tmp/ckpt_summary");
657
658                    println!("Validator: {:?}\n", name.concise());
659                    println!("Checkpoint: {:?}\n", checkpoint);
660                    println!("Content: {:?}\n", contents);
661                }
662            }
663            ToolCommand::Anemo { args } => {
664                let config = crate::make_anemo_config();
665                anemo_cli::run(config, args).await
666            }
