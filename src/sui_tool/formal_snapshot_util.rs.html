4use anyhow::Result;
5use anyhow::anyhow;
6use futures::{StreamExt, TryStreamExt};
7use std::sync::Arc;
8use std::sync::atomic::{AtomicU64, Ordering};
9use sui_data_ingestion_core::{CheckpointReader, create_remote_store_client};
10use sui_types::messages_checkpoint::{CheckpointSequenceNumber, VerifiedCheckpoint};
11use sui_types::storage::WriteStore;
12
13pub(crate) async fn read_summaries_for_list_no_verify<S>(
14    ingestion_url: String,
15    concurrency: usize,
16    store: S,
17    checkpoints: Vec<CheckpointSequenceNumber>,
18    checkpoint_counter: Arc<AtomicU64>,
19) -> Result<()>
20where
21    S: WriteStore + Clone,
22{
23    let client = create_remote_store_client(ingestion_url, vec![], 60)?;
24    futures::stream::iter(checkpoints)
25        .map(|sq| CheckpointReader::fetch_from_object_store(&client, sq))
26        .buffer_unordered(concurrency)
27        .try_for_each(|checkpoint| {
28            let result = store
29                .insert_checkpoint(&VerifiedCheckpoint::new_unchecked(
30                    checkpoint.0.checkpoint_summary.clone(),
31                ))
32                .map_err(|e| anyhow!("Failed to insert checkpoint: {e}"));
33            checkpoint_counter.fetch_add(1, Ordering::Relaxed);
34            futures::future::ready(result)
35        })
36        .await
37}
