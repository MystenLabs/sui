48    // Entry point for all checks related to gas.
49    // Called on both signing and execution.
50    // On success the gas part of the transaction (gas data and gas coins)
51    // is verified and good to go
52    fn get_gas_status(
53        objects: &InputObjects,
54        gas: &[ObjectRef],
55        protocol_config: &ProtocolConfig,
56        reference_gas_price: u64,
57        transaction: &TransactionData,
58        gas_ownership_checks: bool,
59    ) -> SuiResult<SuiGasStatus> {
60        if transaction.kind().is_system_tx() {
61            Ok(SuiGasStatus::new_unmetered())
62        } else {
63            check_gas(
64                objects,
65                protocol_config,
66                reference_gas_price,
67                gas,
68                transaction,
69                gas_ownership_checks,
70            )
71        }
72    }

74    #[instrument(level = "trace", skip_all)]
75    pub fn check_transaction_input(
76        protocol_config: &ProtocolConfig,
77        reference_gas_price: u64,
78        transaction: &TransactionData,
79        input_objects: InputObjects,
80        receiving_objects: &ReceivingObjects,
81        metrics: &Arc<BytecodeVerifierMetrics>,
82        verifier_signing_config: &VerifierSigningConfig,
83    ) -> SuiResult<(SuiGasStatus, CheckedInputObjects)> {
84        let gas_status = check_transaction_input_inner(
85            protocol_config,
86            reference_gas_price,
87            transaction,
88            &input_objects,
89            &[],
90        )?;
91        check_receiving_objects(&input_objects, receiving_objects)?;
92        // Runs verifier, which could be expensive.
93        check_non_system_packages_to_be_published(
94            transaction,
95            protocol_config,
96            metrics,
97            verifier_signing_config,
98        )?;
99
100        Ok((gas_status, input_objects.into_checked()))
101    }

103    pub fn check_transaction_input_with_given_gas(
104        protocol_config: &ProtocolConfig,
105        reference_gas_price: u64,
106        transaction: &TransactionData,
107        mut input_objects: InputObjects,
108        receiving_objects: ReceivingObjects,
109        gas_object: Object,
110        metrics: &Arc<BytecodeVerifierMetrics>,
111        verifier_signing_config: &VerifierSigningConfig,
112    ) -> SuiResult<(SuiGasStatus, CheckedInputObjects)> {
113        let gas_object_ref = gas_object.compute_object_reference();
114        input_objects.push(ObjectReadResult::new_from_gas_object(&gas_object));
115
116        let gas_status = check_transaction_input_inner(
117            protocol_config,
118            reference_gas_price,
119            transaction,
120            &input_objects,
121            &[gas_object_ref],
122        )?;
123        check_receiving_objects(&input_objects, &receiving_objects)?;
124        // Runs verifier, which could be expensive.
125        check_non_system_packages_to_be_published(
126            transaction,
127            protocol_config,
128            metrics,
129            verifier_signing_config,
130        )?;
131
132        Ok((gas_status, input_objects.into_checked()))
133    }

139    #[instrument(level = "trace", skip_all)]
140    pub fn check_certificate_input(
141        cert: &VerifiedExecutableTransaction,
142        input_objects: InputObjects,
143        protocol_config: &ProtocolConfig,
144        reference_gas_price: u64,
145    ) -> SuiResult<(SuiGasStatus, CheckedInputObjects)> {
146        let transaction = cert.data().transaction_data();
147        let gas_status = check_transaction_input_inner(
148            protocol_config,
149            reference_gas_price,
150            transaction,
151            &input_objects,
152            &[],
153        )?;
154        // NB: We do not check receiving objects when executing. Only at signing time do we check.
155        // NB: move verifier is only checked at signing time, not at execution.
156
157        Ok((gas_status, input_objects.into_checked()))
158    }

160    /// WARNING! This should only be used for the dev-inspect transaction. This transaction type
161    /// bypasses many of the normal object checks
162    pub fn check_dev_inspect_input(
163        config: &ProtocolConfig,
164        transaction: &TransactionData,
165        input_objects: InputObjects,
166        // TODO: check ReceivingObjects for dev inspect?
167        _receiving_objects: ReceivingObjects,
168        reference_gas_price: u64,
169    ) -> SuiResult<(SuiGasStatus, CheckedInputObjects)> {
170        let kind = transaction.kind();
171        kind.validity_check(config)?;
172        if kind.is_system_tx() {
173            return Err(UserInputError::Unsupported(format!(
174                "Transaction kind {} is not supported in dev-inspect",
175                kind
176            ))
177            .into());
178        }
179        let mut used_objects: HashSet<SuiAddress> = HashSet::new();
180        for input_object in input_objects.iter() {
181            let Some(object) = input_object.as_object() else {
182                // object was deleted
183                continue;
184            };
185
186            if !object.is_immutable() {
187                fp_ensure!(
188                    used_objects.insert(object.id().into()),
189                    UserInputError::MutableObjectUsedMoreThanOnce {
190                        object_id: object.id()
191                    }
192                    .into()
193                );
194            }
195        }
196
197        let gas_status = get_gas_status(
198            &input_objects,
199            &transaction.gas_data().payment, //gas,
200            config,
201            reference_gas_price,
202            transaction,
203            false, // gas_ownership_checks - false means mostly transaction level checks
204        )?;
205
206        Ok((gas_status, input_objects.into_checked()))
207    }

209    // Common checks performed for transactions and certificates.
210    fn check_transaction_input_inner(
211        protocol_config: &ProtocolConfig,
212        reference_gas_price: u64,
213        transaction: &TransactionData,
214        input_objects: &InputObjects,
215        // Overrides the gas objects in the transaction.
216        gas_override: &[ObjectRef],
217    ) -> SuiResult<SuiGasStatus> {
218        let gas = if gas_override.is_empty() {
219            transaction.gas()
220        } else {
221            gas_override
222        };
223
224        let gas_status = get_gas_status(
225            input_objects,
226            gas,
227            protocol_config,
228            reference_gas_price,
229            transaction,
230            true, // gas_ownership_checks
231        )?;
232        check_objects(transaction, input_objects)?;
233
234        Ok(gas_status)
235    }

237    fn check_receiving_objects(
238        input_objects: &InputObjects,
239        receiving_objects: &ReceivingObjects,
240    ) -> Result<(), SuiError> {
241        let mut objects_in_txn: HashSet<_> = input_objects
242            .object_kinds()
243            .map(|x| x.object_id())
244            .collect();
245
246        // Since we're at signing we check that every object reference that we are receiving is the
247        // most recent version of that object. If it's been received at the version specified we
248        // let it through to allow the transaction to run and fail to unlock any other objects in
249        // the transaction. Otherwise, we return an error.
250        //
251        // If there are any object IDs in common (either between receiving objects and input
252        // objects) we return an error.
253        for ReceivingObjectReadResult {
254            object_ref: (object_id, version, object_digest),
255            object,
256        } in receiving_objects.iter()
257        {
258            fp_ensure!(
259                *version < SequenceNumber::MAX,
260                UserInputError::InvalidSequenceNumber.into()
261            );
262
263            let Some(object) = object.as_object() else {
264                // object was previously received
265                continue;
266            };
267
268            if !(object.owner.is_address_owned()
269                && object.version() == *version
270                && object.digest() == *object_digest)
271            {
272                // Version mismatch
273                fp_ensure!(
274                    object.version() == *version,
275                    UserInputError::ObjectVersionUnavailableForConsumption {
276                        provided_obj_ref: (*object_id, *version, *object_digest),
277                        current_version: object.version(),
278                    }
279                    .into()
280                );
281
282                // Tried to receive a package
283                fp_ensure!(
284                    !object.is_package(),
285                    UserInputError::MovePackageAsObject {
286                        object_id: *object_id
287                    }
288                    .into()
289                );
290
291                // Digest mismatch
292                let expected_digest = object.digest();
293                fp_ensure!(
294                    expected_digest == *object_digest,
295                    UserInputError::InvalidObjectDigest {
296                        object_id: *object_id,
297                        expected_digest
298                    }
299                    .into()
300                );
301
302                match object.owner {
303                    Owner::AddressOwner(_) => {
304                        debug_assert!(
305                            false,
306                            "Receiving object {:?} is invalid but we expect it should be valid. {:?}",
307                            (*object_id, *version, *object_id),
308                            object
309                        );
310                        error!(
311                            "Receiving object {:?} is invalid but we expect it should be valid. {:?}",
312                            (*object_id, *version, *object_id),
313                            object
314                        );
315                        // We should never get here, but if for some reason we do just default to
316                        // object not found and reject signing the transaction.
317                        fp_bail!(
318                            UserInputError::ObjectNotFound {
319                                object_id: *object_id,
320                                version: Some(*version),
321                            }
322                            .into()
323                        )
324                    }
325                    Owner::ObjectOwner(owner) => {
326                        fp_bail!(
327                            UserInputError::InvalidChildObjectArgument {
328                                child_id: object.id(),
329                                parent_id: owner.into(),
330                            }
331                            .into()
332                        )
333                    }
334                    Owner::Shared { .. } | Owner::ConsensusAddressOwner { .. } => {
335                        fp_bail!(UserInputError::NotSharedObjectError.into())
336                    }
337                    Owner::Immutable => fp_bail!(
338                        UserInputError::MutableParameterExpected {
339                            object_id: *object_id
340                        }
341                        .into()
342                    ),
343                };
344            }
345
346            fp_ensure!(
347                !objects_in_txn.contains(object_id),
348                UserInputError::DuplicateObjectRefInput.into()
349            );
350
351            objects_in_txn.insert(*object_id);
352        }
353        Ok(())
354    }

356    /// Check transaction gas data/info and gas coins consistency.
357    /// Return the gas status to be used for the lifecycle of the transaction.
358    #[instrument(level = "trace", skip_all)]
359    fn check_gas(
360        objects: &InputObjects,
361        protocol_config: &ProtocolConfig,
362        reference_gas_price: u64,
363        gas: &[ObjectRef],
364        transaction: &TransactionData,
365        gas_ownership_checks: bool,
366    ) -> SuiResult<SuiGasStatus> {
367        let gas_budget = transaction.gas_budget();
368        let gas_price = transaction.gas_price();
369        let gas_paid_from_address_balance = transaction.is_gas_paid_from_address_balance();
370
371        let gas_status =
372            SuiGasStatus::new(gas_budget, gas_price, reference_gas_price, protocol_config)?;
373
374        // check balance and coins consistency
375        // load all gas coins
376        let objects: BTreeMap<_, _> = objects.iter().map(|o| (o.id(), o)).collect();
377        let mut gas_objects = vec![];
378        for obj_ref in gas {
379            let obj = objects.get(&obj_ref.0);
380            let obj = *obj.ok_or(UserInputError::ObjectNotFound {
381                object_id: obj_ref.0,
382                version: Some(obj_ref.1),
383            })?;
384            gas_objects.push(obj);
385        }
386
387        // Skip gas balance check for address balance payments
388        // We reserve gas budget in advance
389        if !gas_paid_from_address_balance {
390            if gas_ownership_checks {
391                gas_status.check_gas_objects(&gas_objects)?;
392            }
393            gas_status.check_gas_data(&gas_objects, gas_budget)?;
394        }
395        Ok(gas_status)
396    }

398    /// Check all the objects used in the transaction against the database, and ensure
399    /// that they are all the correct version and number.
400    #[instrument(level = "trace", skip_all)]
401    fn check_objects(transaction: &TransactionData, objects: &InputObjects) -> UserInputResult<()> {
402        // We require that mutable objects cannot show up more than once.
403        let mut used_objects: HashSet<SuiAddress> = HashSet::new();
404        for object in objects.iter() {
405            if object.is_mutable() {
406                fp_ensure!(
407                    used_objects.insert(object.id().into()),
408                    UserInputError::MutableObjectUsedMoreThanOnce {
409                        object_id: object.id()
410                    }
411                );
412            }
413        }
414
415        if !transaction.is_genesis_tx() && objects.is_empty() {
416            return Err(UserInputError::ObjectInputArityViolation);
417        }
418
419        let gas_coins: HashSet<ObjectID> =
420            HashSet::from_iter(transaction.gas().iter().map(|obj_ref| obj_ref.0));
421        for object in objects.iter() {
422            let input_object_kind = object.input_object_kind;
423
424            match &object.object {
425                ObjectReadResultKind::Object(object) => {
426                    // For Gas Object, we check the object is owned by gas owner
427                    let owner_address = if gas_coins.contains(&object.id()) {
428                        transaction.gas_owner()
429                    } else {
430                        transaction.sender()
431                    };
432                    // Check if the object contents match the type of lock we need for
433                    // this object.
434                    let system_transaction = transaction.is_system_tx();
435                    check_one_object(
436                        &owner_address,
437                        input_object_kind,
438                        object,
439                        system_transaction,
440                    )?;
441                }
442                // We skip checking a removed consensus object because it no longer exists.
443                ObjectReadResultKind::ObjectConsensusStreamEnded(_, _) => (),
444                // We skip checking shared objects from cancelled transactions since we are not reading it.
445                ObjectReadResultKind::CancelledTransactionSharedObject(_) => (),
446            }
447        }
448
449        Ok(())
450    }

452    /// Check one object against a reference
453    fn check_one_object(
454        owner: &SuiAddress,
455        object_kind: InputObjectKind,
456        object: &Object,
457        system_transaction: bool,
458    ) -> UserInputResult {
459        match object_kind {
460            InputObjectKind::MovePackage(package_id) => {
461                fp_ensure!(
462                    object.data.try_as_package().is_some(),
463                    UserInputError::MoveObjectAsPackage {
464                        object_id: package_id
465                    }
466                );
467            }
468            InputObjectKind::ImmOrOwnedMoveObject((object_id, sequence_number, object_digest)) => {
469                fp_ensure!(
470                    !object.is_package(),
471                    UserInputError::MovePackageAsObject { object_id }
472                );
473                fp_ensure!(
474                    sequence_number < SequenceNumber::MAX,
475                    UserInputError::InvalidSequenceNumber
476                );
477
478                // This is an invariant - we just load the object with the given ID and version.
479                assert_eq!(
480                    object.version(),
481                    sequence_number,
482                    "The fetched object version {} does not match the requested version {}, object id: {}",
483                    object.version(),
484                    sequence_number,
485                    object.id(),
486                );
487
488                // Check the digest matches - user could give a mismatched ObjectDigest
489                let expected_digest = object.digest();
490                fp_ensure!(
491                    expected_digest == object_digest,
492                    UserInputError::InvalidObjectDigest {
493                        object_id,
494                        expected_digest
495                    }
496                );
497
498                match object.owner {
499                    Owner::Immutable => {
500                        // Nothing else to check for Immutable.
501                    }
502                    Owner::AddressOwner(actual_owner) => {
503                        // Check the owner is correct.
504                        fp_ensure!(
505                            owner == &actual_owner,
506                            UserInputError::IncorrectUserSignature {
507                                error: format!(
508                                    "Object {object_id:?} is owned by account address {actual_owner:?}, but given owner/signer address is {owner:?}"
509                                ),
510                            }
511                        );
512                    }
513                    Owner::ObjectOwner(owner) => {
514                        return Err(UserInputError::InvalidChildObjectArgument {
515                            child_id: object.id(),
516                            parent_id: owner.into(),
517                        });
518                    }
519                    Owner::Shared { .. } | Owner::ConsensusAddressOwner { .. } => {
520                        // This object is a mutable consensus object. However the transaction
521                        // specifies it as an owned object. This is inconsistent.
522                        return Err(UserInputError::NotOwnedObjectError);
523                    }
524                };
525            }
526            InputObjectKind::SharedMoveObject {
527                id: object_id,
528                initial_shared_version: input_initial_shared_version,
529                mutability,
530            } => {
531                fp_ensure!(
532                    object.version() < SequenceNumber::MAX,
533                    UserInputError::InvalidSequenceNumber
534                );
535
536                if object_id.is_system_object() {
537                    // System transactions can access system objects without further validation
538                    // (e.g., AuthenticatorStateUpdate uses a placeholder initial_shared_version).
539                    if system_transaction {
540                        return Ok(());
541                    }
542
543                    match (object_id, mutability) {
544                        // System objects that can be taken mutably
545                        (SUI_SYSTEM_STATE_OBJECT_ID, _)
546                        | (SUI_ADDRESS_ALIAS_STATE_OBJECT_ID, _)
547                        | (SUI_COIN_REGISTRY_OBJECT_ID, _)
548                        | (SUI_DENY_LIST_OBJECT_ID, _)
549                        | (SUI_BRIDGE_OBJECT_ID, _)
550
551                        // System objects that can only be taken immutably
552                        | (SUI_CLOCK_OBJECT_ID, SharedObjectMutability::Immutable)
553                        | (SUI_RANDOMNESS_STATE_OBJECT_ID, SharedObjectMutability::Immutable)
554                        | (SUI_ACCUMULATOR_ROOT_OBJECT_ID, SharedObjectMutability::Immutable) => (),
555
556                        // All other system objects: cannot be used as input at all
557                        _ => {
558                            return Err(UserInputError::ImmutableParameterExpectedError {
559                                object_id,
560                            });
561                        }
562                    }
563                }
564
565                match &object.owner {
566                    Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
567                        // When someone locks an object as shared it must be shared already.
568                        return Err(UserInputError::NotSharedObjectError);
569                    }
570                    Owner::Shared {
571                        initial_shared_version: actual_initial_shared_version,
572                    } => {
573                        fp_ensure!(
574                            input_initial_shared_version == *actual_initial_shared_version,
575                            UserInputError::SharedObjectStartingVersionMismatch
576                        )
577                    }
578                    Owner::ConsensusAddressOwner {
579                        start_version: actual_initial_shared_version,
580                        owner: actual_owner,
581                    } => {
582                        fp_ensure!(
583                            input_initial_shared_version == *actual_initial_shared_version,
584                            UserInputError::SharedObjectStartingVersionMismatch
585                        );
586                        // Check the owner is correct.
587                        fp_ensure!(
588                            owner == actual_owner,
589                            UserInputError::IncorrectUserSignature {
590                                error: format!(
591                                    "Object {object_id:?} is owned by account address {actual_owner:?}, but given owner/signer address is {owner:?}"
592                                ),
593                            }
594                        )
595                    }
596                }
597            }
598        };
599        Ok(())
600    }

602    /// Check package verification timeout
603    #[instrument(level = "trace", skip_all)]
604    pub fn check_non_system_packages_to_be_published(
605        transaction: &TransactionData,
606        protocol_config: &ProtocolConfig,
607        metrics: &Arc<BytecodeVerifierMetrics>,
608        verifier_signing_config: &VerifierSigningConfig,
609    ) -> UserInputResult<()> {
610        // Only meter non-system programmable transaction blocks
611        if transaction.is_system_tx() {
612            return Ok(());
613        }
614
615        let TransactionKind::ProgrammableTransaction(pt) = transaction.kind() else {
616            return Ok(());
617        };
618
619        // Use the same verifier and meter for all packages, custom configured for signing.
620        let signing_limits = Some(verifier_signing_config.limits_for_signing());
621        let mut verifier = sui_execution::verifier(protocol_config, signing_limits, metrics);
622        let mut meter = verifier.meter(verifier_signing_config.meter_config_for_signing());
623
624        // Measure time for verifying all packages in the PTB
625        let shared_meter_verifier_timer = metrics
626            .verifier_runtime_per_ptb_success_latency
627            .start_timer();
628
629        let verifier_status = pt
630            .non_system_packages_to_be_published()
631            .try_for_each(|module_bytes| {
632                verifier.meter_module_bytes(protocol_config, module_bytes, meter.as_mut())
633            })
634            .map_err(|e| UserInputError::PackageVerificationTimeout { err: e.to_string() });
635
636        match verifier_status {
637            Ok(_) => {
638                // Success: stop and record the success timer
639                shared_meter_verifier_timer.stop_and_record();
640            }
641            Err(err) => {
642                // Failure: redirect the success timers output to the failure timer and
643                // discard the success timer
644                metrics
645                    .verifier_runtime_per_ptb_timeout_latency
646                    .observe(shared_meter_verifier_timer.stop_and_discard());
647                return Err(err);
648            }
649        };
650
651        Ok(())
652    }
