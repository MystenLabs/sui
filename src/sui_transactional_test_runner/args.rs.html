308impl<ExtraValueArgs: ParsableValue, ExtraRunArgs: Parser> clap::FromArgMatches
309    for SuiSubcommand<ExtraValueArgs, ExtraRunArgs>
310{
311    fn from_arg_matches(matches: &clap::ArgMatches) -> Result<Self, clap::Error> {
312        Ok(match matches.subcommand() {
313            Some(("view-object", matches)) => {
314                SuiSubcommand::ViewObject(ViewObjectCommand::from_arg_matches(matches)?)
315            }
316            Some(("transfer-object", matches)) => {
317                SuiSubcommand::TransferObject(TransferObjectCommand::from_arg_matches(matches)?)
318            }
319            Some(("consensus-commit-prologue", matches)) => SuiSubcommand::ConsensusCommitPrologue(
320                ConsensusCommitPrologueCommand::from_arg_matches(matches)?,
321            ),
322            Some(("programmable", matches)) => SuiSubcommand::ProgrammableTransaction(
323                ProgrammableTransactionCommand::from_arg_matches(matches)?,
324            ),
325            Some(("upgrade", matches)) => {
326                SuiSubcommand::UpgradePackage(UpgradePackageCommand::from_arg_matches(matches)?)
327            }
328            Some(("stage-package", matches)) => {
329                SuiSubcommand::StagePackage(StagePackageCommand::from_arg_matches(matches)?)
330            }
331            Some(("set-address", matches)) => {
332                SuiSubcommand::SetAddress(SetAddressCommand::from_arg_matches(matches)?)
333            }
334            Some(("create-checkpoint", matches)) => {
335                SuiSubcommand::CreateCheckpoint(CreateCheckpointCommand::from_arg_matches(matches)?)
336            }
337            Some(("advance-epoch", matches)) => {
338                SuiSubcommand::AdvanceEpoch(AdvanceEpochCommand::from_arg_matches(matches)?)
339            }
340            Some(("advance-clock", matches)) => {
341                SuiSubcommand::AdvanceClock(AdvanceClockCommand::from_arg_matches(matches)?)
342            }
343            Some(("set-random-state", matches)) => {
344                SuiSubcommand::SetRandomState(SetRandomStateCommand::from_arg_matches(matches)?)
345            }
346            Some(("authenticator-state-update", matches)) => {
347                SuiSubcommand::AuthenticatorStateUpdate(
348                    AuthenticatorStateUpdateCommand::from_arg_matches(matches)?,
349                )
350            }
351            Some(("view-checkpoint", _)) => SuiSubcommand::ViewCheckpoint,
352            Some(("run-graphql", matches)) => {
353                SuiSubcommand::RunGraphql(RunGraphqlCommand::from_arg_matches(matches)?)
354            }
355            Some(("run-jsonrpc", matches)) => {
356                SuiSubcommand::RunJsonRpc(RunJsonRpcCommand::from_arg_matches(matches)?)
357            }
358            Some(("bench", matches)) => SuiSubcommand::Bench(
359                RunCommand::from_arg_matches(matches)?,
360                ExtraRunArgs::from_arg_matches(matches)?,
361            ),
362            _ => {
363                return Err(clap::Error::raw(
364                    clap::error::ErrorKind::InvalidSubcommand,
365                    "Invalid submcommand",
366                ));
367            }
368        })
369    }
370
371    fn update_from_arg_matches(&mut self, matches: &clap::ArgMatches) -> Result<(), clap::Error> {
372        *self = Self::from_arg_matches(matches)?;
373        Ok(())
374    }
375}

436impl SuiExtraValueArgs {
437    fn parse_object_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
438        parser: &mut MoveCLParser<'a, ValueToken, I>,
439    ) -> anyhow::Result<Self> {
440        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "object")?;
441        Ok(SuiExtraValueArgs::Object(fake_id, version))
442    }
443
444    fn parse_receiving_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
445        parser: &mut MoveCLParser<'a, ValueToken, I>,
446    ) -> anyhow::Result<Self> {
447        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "receiving")?;
448        Ok(SuiExtraValueArgs::Receiving(fake_id, version))
449    }
450
451    fn parse_read_shared_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
452        parser: &mut MoveCLParser<'a, ValueToken, I>,
453    ) -> anyhow::Result<Self> {
454        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "immshared")?;
455        Ok(SuiExtraValueArgs::ImmShared(fake_id, version))
456    }
457
458    fn parse_non_exlucsive_write_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
459        parser: &mut MoveCLParser<'a, ValueToken, I>,
460    ) -> anyhow::Result<Self> {
461        let (fake_id, version) = Self::parse_receiving_or_object_value(parser, "nonexclusive")?;
462        Ok(SuiExtraValueArgs::NonExclusiveWrite(fake_id, version))
463    }
464
465    fn parse_digest_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
466        parser: &mut MoveCLParser<'a, ValueToken, I>,
467    ) -> anyhow::Result<Self> {
468        let contents = parser.advance(ValueToken::Ident)?;
469        ensure!(contents == "digest");
470        parser.advance(ValueToken::LParen)?;
471        let package = parser.advance(ValueToken::Ident)?;
472        parser.advance(ValueToken::RParen)?;
473        Ok(SuiExtraValueArgs::Digest(package.to_owned()))
474    }
475
476    fn parse_withdraw_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
477        parser: &mut MoveCLParser<'a, ValueToken, I>,
478    ) -> anyhow::Result<Self> {
479        let contents = parser.advance(ValueToken::Ident)?;
480        ensure!(contents == "withdraw");
481
482        // Format: withdraw<Type>(amount)
483        parser.advance(ValueToken::LAngle)?;
484
485        // Parse type - collect all tokens until we hit the matching RAngle
486        // Need to track nesting level for types like Balance<Coin<SUI>>
487        let mut type_parts = Vec::new();
488        let mut angle_bracket_depth = 1; // We already consumed the opening <
489        loop {
490            let (tok, s) = match parser.peek() {
491                Some(v) => v,
492                None => bail!("Unexpected end of input while parsing withdraw type"),
493            };
494            match tok {
495                ValueToken::Whitespace => {
496                    parser.advance(ValueToken::Whitespace)?;
497                    // Skip whitespace
498                }
499                ValueToken::Ident => {
500                    parser.advance(ValueToken::Ident)?;
501                    type_parts.push(s.to_string());
502                }
503                ValueToken::ColonColon => {
504                    parser.advance(ValueToken::ColonColon)?;
505                    type_parts.push("::".to_string());
506                }
507                ValueToken::LAngle => {
508                    parser.advance(ValueToken::LAngle)?;
509                    type_parts.push("<".to_string());
510                    angle_bracket_depth += 1;
511                }
512                ValueToken::RAngle => {
513                    parser.advance(ValueToken::RAngle)?;
514                    angle_bracket_depth -= 1;
515                    if angle_bracket_depth == 0 {
516                        // This is the closing > for withdraw<Type>
517                        break;
518                    }
519                    type_parts.push(">".to_string());
520                }
521                ValueToken::Comma => {
522                    parser.advance(ValueToken::Comma)?;
523                    type_parts.push(",".to_string());
524                }
525                _ => bail!("Unexpected token {:?} while parsing withdraw type", tok),
526            }
527        }
528
529        let type_str = type_parts.join("");
530
531        // Parse the type from the type string
532        let type_tokens: Vec<_> = TypeToken::tokenize(&type_str)?.into_iter().collect();
533        let mut type_parser = move_core_types::parsing::parser::Parser::new(type_tokens);
534        let parsed_type = type_parser.parse_type()?;
535
536        // Now parse (amount)
537        parser.advance(ValueToken::LParen)?;
538        let amount_str = parser.advance(ValueToken::Number)?;
539        let (amount, _) = parse_u64(amount_str)?;
540        parser.advance(ValueToken::RParen)?;
541
542        Ok(SuiExtraValueArgs::Withdraw(amount, parsed_type))
543    }
544
545    fn parse_receiving_or_object_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
546        parser: &mut MoveCLParser<'a, ValueToken, I>,
547        ident_name: &str,
548    ) -> anyhow::Result<(FakeID, Option<SequenceNumber>)> {
549        let contents = parser.advance(ValueToken::Ident)?;
550        ensure!(contents == ident_name);
551        parser.advance(ValueToken::LParen)?;
552        let i_str = parser.advance(ValueToken::Number)?;
553        let (i, _) = parse_u256(i_str)?;
554        let fake_id = if let Some(ValueToken::Comma) = parser.peek_tok() {
555            parser.advance(ValueToken::Comma)?;
556            let j_str = parser.advance(ValueToken::Number)?;
557            let (j, _) = parse_u64(j_str)?;
558            if i > U256::from(u64::MAX) {
559                bail!("Object ID too large")
560            }
561            FakeID::Enumerated(i.unchecked_as_u64(), j)
562        } else {
563            let mut u256_bytes = i.to_le_bytes().to_vec();
564            u256_bytes.reverse();
565            let address: SuiAddress = SuiAddress::from_bytes(&u256_bytes).unwrap();
566            FakeID::Known(address.into())
567        };
568        parser.advance(ValueToken::RParen)?;
569        let version = if let Some(ValueToken::AtSign) = parser.peek_tok() {
570            parser.advance(ValueToken::AtSign)?;
571            let v_str = parser.advance(ValueToken::Number)?;
572            let (v, _) = parse_u64(v_str)?;
573            Some(SequenceNumber::from_u64(v))
574        } else {
575            None
576        };
577        Ok((fake_id, version))
578    }
579}

584            SuiValue::MoveValue(v) => v,
585            SuiValue::Object(_, _) => panic!("unexpected nested Sui object in args"),
586            SuiValue::ObjVec(_) => panic!("unexpected nested Sui object vector in args"),
587            SuiValue::Digest(_) => panic!("unexpected nested Sui package digest in args"),
588            SuiValue::Receiving(_, _) => panic!("unexpected nested Sui receiving object in args"),
589            SuiValue::ImmShared(_, _) => panic!("unexpected nested Sui shared object in args"),
590            SuiValue::NonExclusiveWrite(_, _) => {
591                panic!("unexpected nested Sui non-exclusive write object in args")
592            }
593            SuiValue::Withdraw(_, _) => {
594                panic!("unexpected nested Sui withdraw reservation in args")
595            }
596        }
597    }
598
599    fn assert_object(self) -> (FakeID, Option<SequenceNumber>) {
600        match self {
601            SuiValue::MoveValue(_) => panic!("unexpected nested non-object value in args"),
602            SuiValue::Object(id, version) => (id, version),
603            SuiValue::ObjVec(_) => panic!("unexpected nested Sui object vector in args"),
604            SuiValue::Digest(_) => panic!("unexpected nested Sui package digest in args"),
605            SuiValue::Receiving(_, _) => panic!("unexpected nested Sui receiving object in args"),
606            SuiValue::ImmShared(_, _) => panic!("unexpected nested Sui shared object in args"),
607            SuiValue::NonExclusiveWrite(_, _) => {
608                panic!("unexpected nested Sui non-exclusive write object in args")
609            }
610            SuiValue::Withdraw(_, _) => {
611                panic!("unexpected nested Sui withdraw reservation in args")
612            }
613        }
614    }

616    fn resolve_object(
617        fake_id: FakeID,
618        version: Option<SequenceNumber>,
619        test_adapter: &SuiTestAdapter,
620    ) -> anyhow::Result<Object> {
621        let id = match test_adapter.fake_to_real_object_id(fake_id) {
622            Some(id) => id,
623            None => bail!("INVALID TEST. Unknown object, object({})", fake_id),
624        };
625        let obj_res = if let Some(v) = version {
626            sui_types::storage::ObjectStore::get_object_by_key(&*test_adapter.executor, &id, v)
627        } else {
628            sui_types::storage::ObjectStore::get_object(&*test_adapter.executor, &id)
629        };
630        let obj = match obj_res {
631            Some(obj) => obj,
632            None => bail!("INVALID TEST. Could not load object argument {}", id),
633        };
634        Ok(obj)
635    }

672    fn shared_arg_impl(
673        fake_id: FakeID,
674        version: Option<SequenceNumber>,
675        test_adapter: &SuiTestAdapter,
676        mutability: SharedObjectMutability,
677    ) -> anyhow::Result<ObjectArg> {
678        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
679        let id = obj.id();
680        if let Owner::Shared {
681            initial_shared_version,
682        } = obj.owner
683        {
684            Ok(ObjectArg::SharedObject {
685                id,
686                initial_shared_version,
687                mutability,
688            })
689        } else {
690            bail!("{fake_id} is not a shared object.")
691        }
692    }
693
694    fn object_arg(
695        fake_id: FakeID,
696        version: Option<SequenceNumber>,
697        test_adapter: &SuiTestAdapter,
698    ) -> anyhow::Result<ObjectArg> {
699        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
700        let id = obj.id();
701        match obj.owner {
702            Owner::Shared {
703                initial_shared_version,
704            }
705            | Owner::ConsensusAddressOwner {
706                start_version: initial_shared_version,
707                ..
708            } => Ok(ObjectArg::SharedObject {
709                id,
710                initial_shared_version,
711                mutability: SharedObjectMutability::Mutable,
712            }),
713            Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
714                let obj_ref = obj.compute_object_reference();
715                Ok(ObjectArg::ImmOrOwnedObject(obj_ref))
716            }
717        }
718    }

720    pub(crate) fn into_call_arg(self, test_adapter: &SuiTestAdapter) -> anyhow::Result<CallArg> {
721        Ok(match self {
722            SuiValue::Object(fake_id, version) => {
723                CallArg::Object(Self::object_arg(fake_id, version, test_adapter)?)
724            }
725            SuiValue::MoveValue(v) => CallArg::Pure(v.simple_serialize().unwrap()),
726            SuiValue::Receiving(fake_id, version) => {
727                CallArg::Object(Self::receiving_arg(fake_id, version, test_adapter)?)
728            }
729            SuiValue::ImmShared(fake_id, version) => {
730                CallArg::Object(Self::read_shared_arg(fake_id, version, test_adapter)?)
731            }
732            SuiValue::NonExclusiveWrite(fake_id, version) => CallArg::Object(
733                Self::non_exclusive_write_arg(fake_id, version, test_adapter)?,
734            ),
735            SuiValue::ObjVec(_) => bail!("obj vec is not supported as an input"),
736            SuiValue::Digest(pkg) => {
737                let pkg = Symbol::from(pkg);
738                let Some(staged) = test_adapter.staged_modules.get(&pkg) else {
739                    bail!("Unbound staged package '{pkg}'")
740                };
741                CallArg::Pure(bcs::to_bytes(&staged.digest).unwrap())
742            }
743            SuiValue::Withdraw(amount, type_tag) => {
744                // Check if the type is Balance<T> and extract the inner type T
745                // For now, we only support Balance type for withdraw
746                let inner_type =
747                    Balance::maybe_get_balance_type_param(&type_tag).ok_or_else(|| {
748                        anyhow::anyhow!(
749                            "Withdraw only supports Balance<T> types, got: {}",
750                            type_tag
751                        )
752                    })?;
753                let inner_type_input = TypeInput::from(inner_type);
754                CallArg::FundsWithdrawal(FundsWithdrawalArg::balance_from_sender(
755                    amount,
756                    inner_type_input,
757                ))
758            }
759        })
760    }
761
762    pub(crate) fn into_argument(
763        self,
764        builder: &mut ProgrammableTransactionBuilder,
765        test_adapter: &SuiTestAdapter,
766    ) -> anyhow::Result<Argument> {
767        match self {
768            SuiValue::ObjVec(vec) => builder.make_obj_vec(
769                vec.iter()
770                    .map(|(fake_id, version)| Self::object_arg(*fake_id, *version, test_adapter))
771                    .collect::<Result<Vec<ObjectArg>, _>>()?,
772            ),
773            value => {
774                let call_arg = value.into_call_arg(test_adapter)?;
775                builder.input(call_arg)
776            }
777        }
778    }
779}

781impl ParsableValue for SuiExtraValueArgs {
782    type ConcreteValue = SuiValue;
783
784    fn parse_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
785        parser: &mut MoveCLParser<'a, ValueToken, I>,
786    ) -> Option<anyhow::Result<Self>> {
787        match parser.peek()? {
788            (ValueToken::Ident, "object") => Some(Self::parse_object_value(parser)),
789            (ValueToken::Ident, "digest") => Some(Self::parse_digest_value(parser)),
790            (ValueToken::Ident, "receiving") => Some(Self::parse_receiving_value(parser)),
791            (ValueToken::Ident, "immshared") => Some(Self::parse_read_shared_value(parser)),
792            (ValueToken::Ident, "nonexclusive") => {
793                Some(Self::parse_non_exlucsive_write_value(parser))
794            }
795            (ValueToken::Ident, "withdraw") => Some(Self::parse_withdraw_value(parser)),
796            _ => None,
797        }
798    }
799
800    fn move_value_into_concrete(v: MoveValue) -> anyhow::Result<Self::ConcreteValue> {
801        Ok(SuiValue::MoveValue(v))
802    }
803
804    fn concrete_vector(elems: Vec<Self::ConcreteValue>) -> anyhow::Result<Self::ConcreteValue> {
805        if !elems.is_empty() && matches!(elems[0], SuiValue::Object(_, _)) {
806            Ok(SuiValue::ObjVec(
807                elems.into_iter().map(SuiValue::assert_object).collect(),
808            ))
809        } else {
810            Ok(SuiValue::MoveValue(MoveValue::Vector(
811                elems.into_iter().map(SuiValue::assert_move_value).collect(),
812            )))
813        }
814    }
815
816    fn concrete_struct(values: Vec<Self::ConcreteValue>) -> anyhow::Result<Self::ConcreteValue> {
817        Ok(SuiValue::MoveValue(MoveValue::Struct(MoveStruct(
818            values.into_iter().map(|v| v.assert_move_value()).collect(),
819        ))))
820    }
821
822    fn into_concrete_value(
823        self,
824        mapping: &impl Fn(&str) -> Option<move_core_types::account_address::AccountAddress>,
825    ) -> anyhow::Result<Self::ConcreteValue> {
826        match self {
827            SuiExtraValueArgs::Object(id, version) => Ok(SuiValue::Object(id, version)),
828            SuiExtraValueArgs::Digest(pkg) => Ok(SuiValue::Digest(pkg)),
829            SuiExtraValueArgs::Receiving(id, version) => Ok(SuiValue::Receiving(id, version)),
830            SuiExtraValueArgs::ImmShared(id, version) => Ok(SuiValue::ImmShared(id, version)),
831            SuiExtraValueArgs::NonExclusiveWrite(id, version) => {
832                Ok(SuiValue::NonExclusiveWrite(id, version))
833            }
834            SuiExtraValueArgs::Withdraw(amount, parsed_type) => {
835                let type_tag = parsed_type.into_type_tag(mapping)?;
836                Ok(SuiValue::Withdraw(amount, type_tag))
837            }
838        }
839    }

842fn parse_fake_id(s: &str) -> anyhow::Result<FakeID> {
843    Ok(if let Some((s1, s2)) = s.split_once(',') {
844        let (i, _) = parse_u64(s1)?;
845        let (j, _) = parse_u64(s2)?;
846        FakeID::Enumerated(i, j)
847    } else {
848        let (i, _) = parse_u256(s)?;
849        let mut u256_bytes = i.to_le_bytes().to_vec();
850        u256_bytes.reverse();
851        let address: SuiAddress = SuiAddress::from_bytes(&u256_bytes).unwrap();
852        FakeID::Known(address.into())
853    })
854}

863        ),
864    })
