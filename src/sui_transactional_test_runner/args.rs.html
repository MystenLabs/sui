315    fn from_arg_matches(matches: &clap::ArgMatches) -> Result<Self, clap::Error> {
316        Ok(match matches.subcommand() {

320            Some(("transfer-object", matches)) => {
321                SuiSubcommand::TransferObject(TransferObjectCommand::from_arg_matches(matches)?)
322            }
323            Some(("consensus-commit-prologue", matches)) => SuiSubcommand::ConsensusCommitPrologue(
324                ConsensusCommitPrologueCommand::from_arg_matches(matches)?,
325            ),
326            Some(("programmable", matches)) => SuiSubcommand::ProgrammableTransaction(
327                ProgrammableTransactionCommand::from_arg_matches(matches)?,
328            ),
329            Some(("upgrade", matches)) => {
330                SuiSubcommand::UpgradePackage(UpgradePackageCommand::from_arg_matches(matches)?)
331            }
332            Some(("stage-package", matches)) => {
333                SuiSubcommand::StagePackage(StagePackageCommand::from_arg_matches(matches)?)
334            }
335            Some(("set-address", matches)) => {
336                SuiSubcommand::SetAddress(SetAddressCommand::from_arg_matches(matches)?)
337            }
338            Some(("create-checkpoint", matches)) => {
339                SuiSubcommand::CreateCheckpoint(CreateCheckpointCommand::from_arg_matches(matches)?)
340            }
341            Some(("advance-epoch", matches)) => {
342                SuiSubcommand::AdvanceEpoch(AdvanceEpochCommand::from_arg_matches(matches)?)
343            }
344            Some(("advance-clock", matches)) => {
345                SuiSubcommand::AdvanceClock(AdvanceClockCommand::from_arg_matches(matches)?)
346            }
347            Some(("set-random-state", matches)) => {
348                SuiSubcommand::SetRandomState(SetRandomStateCommand::from_arg_matches(matches)?)
349            }
350            Some(("authenticator-state-update", matches)) => {
351                SuiSubcommand::AuthenticatorStateUpdate(
352                    AuthenticatorStateUpdateCommand::from_arg_matches(matches)?,
353                )
354            }
355            Some(("view-checkpoint", _)) => SuiSubcommand::ViewCheckpoint,
356            Some(("run-graphql", matches)) => {
357                SuiSubcommand::RunGraphql(RunGraphqlCommand::from_arg_matches(matches)?)
358            }
359            Some(("run-jsonrpc", matches)) => {
360                SuiSubcommand::RunJsonRpc(RunJsonRpcCommand::from_arg_matches(matches)?)
361            }
362            Some(("bench", matches)) => SuiSubcommand::Bench(
363                RunCommand::from_arg_matches(matches)?,
364                ExtraRunArgs::from_arg_matches(matches)?,
365            ),
366            _ => {
367                return Err(clap::Error::raw(
368                    clap::error::ErrorKind::InvalidSubcommand,
369                    "Invalid submcommand",
370                ));
371            }
372        })
373    }
374
375    fn update_from_arg_matches(&mut self, matches: &clap::ArgMatches) -> Result<(), clap::Error> {
376        *self = Self::from_arg_matches(matches)?;
377        Ok(())
378    }

494
495    fn parse_digest_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
496        parser: &mut MoveCLParser<'a, ValueToken, I>,
497    ) -> anyhow::Result<Self> {
498        let contents = parser.advance(ValueToken::Ident)?;
499        ensure!(contents == "digest");
500        parser.advance(ValueToken::LParen)?;
501        let package = parser.advance(ValueToken::Ident)?;
502        parser.advance(ValueToken::RParen)?;
503        Ok(SuiExtraValueArgs::Digest(package.to_owned()))
504    }
505
506    fn parse_withdraw_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
507        parser: &mut MoveCLParser<'a, ValueToken, I>,
508    ) -> anyhow::Result<Self> {
509        let contents = parser.advance(ValueToken::Ident)?;
510        ensure!(contents == "withdraw");
511
512        // Format: withdraw<Type>(amount)
513        parser.advance(ValueToken::LAngle)?;
514
515        // Parse type - collect all tokens until we hit the matching RAngle
516        // Need to track nesting level for types like Balance<Coin<SUI>>
517        let mut type_parts = Vec::new();
518        let mut angle_bracket_depth = 1; // We already consumed the opening <
519        loop {
520            let (tok, s) = match parser.peek() {
521                Some(v) => v,
522                None => bail!("Unexpected end of input while parsing withdraw type"),
523            };
524            match tok {
525                ValueToken::Whitespace => {
526                    parser.advance(ValueToken::Whitespace)?;
527                    // Skip whitespace
528                }
529                ValueToken::Ident => {
530                    parser.advance(ValueToken::Ident)?;
531                    type_parts.push(s.to_string());
532                }
533                ValueToken::ColonColon => {
534                    parser.advance(ValueToken::ColonColon)?;
535                    type_parts.push("::".to_string());
536                }
537                ValueToken::LAngle => {
538                    parser.advance(ValueToken::LAngle)?;
539                    type_parts.push("<".to_string());
540                    angle_bracket_depth += 1;
541                }
542                ValueToken::RAngle => {
543                    parser.advance(ValueToken::RAngle)?;
544                    angle_bracket_depth -= 1;
545                    if angle_bracket_depth == 0 {
546                        // This is the closing > for withdraw<Type>
547                        break;
548                    }
549                    type_parts.push(">".to_string());
550                }
551                ValueToken::Comma => {
552                    parser.advance(ValueToken::Comma)?;
553                    type_parts.push(",".to_string());
554                }
555                _ => bail!("Unexpected token {:?} while parsing withdraw type", tok),
556            }
557        }
558
559        let type_str = type_parts.join("");
560
561        // Parse the type from the type string
562        let type_tokens: Vec<_> = TypeToken::tokenize(&type_str)?.into_iter().collect();
563        let mut type_parser = move_core_types::parsing::parser::Parser::new(type_tokens);
564        let parsed_type = type_parser.parse_type()?;
565
566        // Now parse (amount)
567        parser.advance(ValueToken::LParen)?;
568        let amount_str = parser.advance(ValueToken::Number)?;
569        let (amount, _) = parse_u64(amount_str)?;
570        parser.advance(ValueToken::RParen)?;
571
572        Ok(SuiExtraValueArgs::Withdraw(amount, parsed_type))
573    }
574
575    fn parse_receiving_or_object_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
576        parser: &mut MoveCLParser<'a, ValueToken, I>,
577        ident_name: &str,
578    ) -> anyhow::Result<(FakeID, Option<SequenceNumber>)> {
579        let contents = parser.advance(ValueToken::Ident)?;
580        ensure!(contents == ident_name);
581        parser.advance(ValueToken::LParen)?;
582        let i_str = parser.advance(ValueToken::Number)?;
583        let (i, _) = parse_u256(i_str)?;
584        let fake_id = if let Some(ValueToken::Comma) = parser.peek_tok() {
585            parser.advance(ValueToken::Comma)?;
586            let j_str = parser.advance(ValueToken::Number)?;
587            let (j, _) = parse_u64(j_str)?;
588            if i > U256::from(u64::MAX) {
589                bail!("Object ID too large")
590            }
591            FakeID::Enumerated(i.unchecked_as_u64(), j)
592        } else {
593            let mut u256_bytes = i.to_le_bytes().to_vec();
594            u256_bytes.reverse();
595            let address: SuiAddress = SuiAddress::from_bytes(&u256_bytes).unwrap();
596            FakeID::Known(address.into())
597        };
598        parser.advance(ValueToken::RParen)?;
599        let version = if let Some(ValueToken::AtSign) = parser.peek_tok() {
600            parser.advance(ValueToken::AtSign)?;
601            let v_str = parser.advance(ValueToken::Number)?;
602            let (v, _) = parse_u64(v_str)?;
603            Some(SequenceNumber::from_u64(v))
604        } else {
605            None
606        };
607        Ok((fake_id, version))
608    }
609}

611impl SuiValue {
612    fn assert_move_value(self) -> MoveValue {
613        match self {
614            SuiValue::MoveValue(v) => v,
615            SuiValue::Object(_, _) => panic!("unexpected nested Sui object in args"),
616            SuiValue::ObjVec(_) => panic!("unexpected nested Sui object vector in args"),
617            SuiValue::Digest(_) => panic!("unexpected nested Sui package digest in args"),
618            SuiValue::Receiving(_, _) => panic!("unexpected nested Sui receiving object in args"),
619            SuiValue::Owned(_, _) => panic!("unexpected nested Sui owned object in args"),
620            SuiValue::Shared(_, _, _) => panic!("unexpected nested Sui shared object in args"),
621            SuiValue::Withdraw(_, _) => {
622                panic!("unexpected nested Sui withdraw reservation in args")
623            }
624        }
625    }
626
627    fn assert_object(self) -> (FakeID, Option<SequenceNumber>) {
628        match self {
629            SuiValue::MoveValue(_) => panic!("unexpected nested non-object value in args"),
630            SuiValue::Object(id, version) => (id, version),
631            SuiValue::ObjVec(_) => panic!("unexpected nested Sui object vector in args"),
632            SuiValue::Digest(_) => panic!("unexpected nested Sui package digest in args"),
633            SuiValue::Receiving(_, _) => panic!("unexpected nested Sui receiving object in args"),
634            SuiValue::Owned(_, _) => panic!("unexpected nested Sui owned object in args"),
635            SuiValue::Shared(_, _, _) => panic!("unexpected nested Sui shared object in args"),
636            SuiValue::Withdraw(_, _) => {
637                panic!("unexpected nested Sui withdraw reservation in args")
638            }
639        }
640    }
641
642    fn resolve_object(
643        fake_id: FakeID,
644        version: Option<SequenceNumber>,
645        test_adapter: &SuiTestAdapter,
646    ) -> anyhow::Result<Object> {
647        let id = match test_adapter.fake_to_real_object_id(fake_id) {
648            Some(id) => id,
649            None => bail!("INVALID TEST. Unknown object, object({})", fake_id),
650        };
651        let obj_res = if let Some(v) = version {
652            sui_types::storage::ObjectStore::get_object_by_key(&*test_adapter.executor, &id, v)
653        } else {
654            sui_types::storage::ObjectStore::get_object(&*test_adapter.executor, &id)
655        };
656        let obj = match obj_res {
657            Some(obj) => obj,
658            None => bail!("INVALID TEST. Could not load object argument {}", id),
659        };
660        Ok(obj)
661    }
662
663    fn receiving_arg(
664        fake_id: FakeID,
665        version: Option<SequenceNumber>,
666        test_adapter: &SuiTestAdapter,
667    ) -> anyhow::Result<ObjectArg> {
668        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
669        Ok(ObjectArg::Receiving(obj.compute_object_reference()))
670    }
671
672    fn owned_arg(
673        fake_id: FakeID,
674        version: Option<SequenceNumber>,
675        test_adapter: &SuiTestAdapter,
676    ) -> anyhow::Result<ObjectArg> {
677        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
678        let obj_ref = obj.compute_object_reference();
679        Ok(ObjectArg::ImmOrOwnedObject(obj_ref))
680    }
681
682    fn shared_arg(
683        fake_id: FakeID,
684        version: Option<SequenceNumber>,
685        test_adapter: &SuiTestAdapter,
686        mutability: SharedObjectMutability,
687    ) -> anyhow::Result<ObjectArg> {
688        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
689        let id = obj.id();
690        let initial_shared_version = match obj.owner {
691            Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
692                SequenceNumber::from_u64(0)
693            }
694            Owner::Shared {
695                initial_shared_version,
696            } => initial_shared_version,
697            Owner::ConsensusAddressOwner { start_version, .. } => start_version,
698        };
699        Ok(ObjectArg::SharedObject {
700            id,
701            initial_shared_version,
702            mutability,
703        })
704    }
705
706    fn object_arg(
707        fake_id: FakeID,
708        version: Option<SequenceNumber>,
709        test_adapter: &SuiTestAdapter,
710    ) -> anyhow::Result<ObjectArg> {
711        let obj = Self::resolve_object(fake_id, version, test_adapter)?;
712        let id = obj.id();
713        match obj.owner {
714            Owner::Shared {
715                initial_shared_version,
716            }
717            | Owner::ConsensusAddressOwner {
718                start_version: initial_shared_version,
719                ..
720            } => Ok(ObjectArg::SharedObject {
721                id,
722                initial_shared_version,
723                mutability: SharedObjectMutability::Mutable,
724            }),
725            Owner::AddressOwner(_) | Owner::ObjectOwner(_) | Owner::Immutable => {
726                let obj_ref = obj.compute_object_reference();
727                Ok(ObjectArg::ImmOrOwnedObject(obj_ref))
728            }
729        }
730    }
731
732    pub(crate) fn into_call_arg(self, test_adapter: &SuiTestAdapter) -> anyhow::Result<CallArg> {
733        Ok(match self {
734            SuiValue::Object(fake_id, version) => {
735                CallArg::Object(Self::object_arg(fake_id, version, test_adapter)?)
736            }
737            SuiValue::MoveValue(v) => CallArg::Pure(v.simple_serialize().unwrap()),
738            SuiValue::Receiving(fake_id, version) => {
739                CallArg::Object(Self::receiving_arg(fake_id, version, test_adapter)?)
740            }
741            SuiValue::Owned(fake_id, version) => {
742                CallArg::Object(Self::owned_arg(fake_id, version, test_adapter)?)
743            }
744            SuiValue::Shared(mutability, fake_id, version) => CallArg::Object(Self::shared_arg(
745                fake_id,
746                version,
747                test_adapter,
748                mutability,
749            )?),
750            SuiValue::ObjVec(_) => bail!("obj vec is not supported as an input"),
751            SuiValue::Digest(pkg) => {
752                let pkg = Symbol::from(pkg);
753                let Some(staged) = test_adapter.staged_modules.get(&pkg) else {
754                    bail!("Unbound staged package '{pkg}'")
755                };
756                CallArg::Pure(bcs::to_bytes(&staged.digest).unwrap())
757            }
758            SuiValue::Withdraw(amount, type_tag) => {
759                // Check if the type is Balance<T> and extract the inner type T
760                // For now, we only support Balance type for withdraw
761                let inner_type =
762                    Balance::maybe_get_balance_type_param(&type_tag).ok_or_else(|| {
763                        anyhow::anyhow!(
764                            "Withdraw only supports Balance<T> types, got: {}",
765                            type_tag
766                        )
767                    })?;
768                CallArg::FundsWithdrawal(FundsWithdrawalArg::balance_from_sender(
769                    amount, inner_type,
770                ))
771            }
772        })
773    }
774
775    pub(crate) fn into_argument(
776        self,
777        builder: &mut ProgrammableTransactionBuilder,
778        test_adapter: &SuiTestAdapter,
779    ) -> anyhow::Result<Argument> {
780        match self {
781            SuiValue::ObjVec(vec) => builder.make_obj_vec(
782                vec.iter()
783                    .map(|(fake_id, version)| Self::object_arg(*fake_id, *version, test_adapter))
784                    .collect::<Result<Vec<ObjectArg>, _>>()?,
785            ),
786            value => {
787                let call_arg = value.into_call_arg(test_adapter)?;
788                builder.input(call_arg)
789            }
790        }
791    }
792}

794impl ParsableValue for SuiExtraValueArgs {
795    type ConcreteValue = SuiValue;
796
797    fn parse_value<'a, I: Iterator<Item = (ValueToken, &'a str)>>(
798        parser: &mut MoveCLParser<'a, ValueToken, I>,
799    ) -> Option<anyhow::Result<Self>> {
800        match parser.peek()? {
801            (ValueToken::Ident, "object") => Some(Self::parse_object_value(parser)),
802            (ValueToken::Ident, "digest") => Some(Self::parse_digest_value(parser)),
803            (ValueToken::Ident, "receiving") => Some(Self::parse_receiving_value(parser)),
804            (ValueToken::Ident, "owned") => Some(Self::parse_owned_object_value(parser)),
805            (ValueToken::Ident, "mutshared") => Some(Self::parse_mut_shared_value(parser)),
806            (ValueToken::Ident, "immshared") => Some(Self::parse_read_shared_value(parser)),
807            (ValueToken::Ident, "nonexclusive") => {
808                Some(Self::parse_non_exlucsive_write_value(parser))
809            }
810            (ValueToken::Ident, "withdraw") => Some(Self::parse_withdraw_value(parser)),
811            _ => None,
812        }
813    }
814
815    fn move_value_into_concrete(v: MoveValue) -> anyhow::Result<Self::ConcreteValue> {
816        Ok(SuiValue::MoveValue(v))
817    }
818
819    fn concrete_vector(elems: Vec<Self::ConcreteValue>) -> anyhow::Result<Self::ConcreteValue> {
820        if !elems.is_empty() && matches!(elems[0], SuiValue::Object(_, _)) {
821            Ok(SuiValue::ObjVec(
822                elems.into_iter().map(SuiValue::assert_object).collect(),
823            ))
824        } else {
825            Ok(SuiValue::MoveValue(MoveValue::Vector(
826                elems.into_iter().map(SuiValue::assert_move_value).collect(),
827            )))
828        }
829    }
830
831    fn concrete_struct(values: Vec<Self::ConcreteValue>) -> anyhow::Result<Self::ConcreteValue> {
832        Ok(SuiValue::MoveValue(MoveValue::Struct(MoveStruct(
833            values.into_iter().map(|v| v.assert_move_value()).collect(),
834        ))))
835    }
836
837    fn into_concrete_value(
838        self,
839        mapping: &impl Fn(&str) -> Option<move_core_types::account_address::AccountAddress>,
840    ) -> anyhow::Result<Self::ConcreteValue> {
841        match self {
842            SuiExtraValueArgs::Object(id, version) => Ok(SuiValue::Object(id, version)),
843            SuiExtraValueArgs::Digest(pkg) => Ok(SuiValue::Digest(pkg)),
844            SuiExtraValueArgs::Receiving(id, version) => Ok(SuiValue::Receiving(id, version)),
845            SuiExtraValueArgs::Owned(id, version) => Ok(SuiValue::Owned(id, version)),
846            SuiExtraValueArgs::Shared(mutability, id, version) => {
847                Ok(SuiValue::Shared(mutability, id, version))
848            }
849            SuiExtraValueArgs::Withdraw(amount, parsed_type) => {
850                let type_tag = parsed_type.into_type_tag(mapping)?;
851                Ok(SuiValue::Withdraw(amount, type_tag))
852            }
853        }
854    }
855}

857fn parse_fake_id(s: &str) -> anyhow::Result<FakeID> {
858    Ok(if let Some((s1, s2)) = s.split_once(',') {
859        let (i, _) = parse_u64(s1)?;
860        let (j, _) = parse_u64(s2)?;
861        FakeID::Enumerated(i, j)
862    } else {
863        let (i, _) = parse_u256(s)?;
864        let mut u256_bytes = i.to_le_bytes().to_vec();
865        u256_bytes.reverse();
866        let address: SuiAddress = SuiAddress::from_bytes(&u256_bytes).unwrap();
867        FakeID::Known(address.into())
868    })
869}

871fn parse_policy(x: &str) -> anyhow::Result<u8> {
872    Ok(match x {
873        "compatible" => UpgradePolicy::COMPATIBLE,
874        "additive" => UpgradePolicy::ADDITIVE,
875        "dep_only" => UpgradePolicy::DEP_ONLY,
876        _ => bail!(
877            "Invalid upgrade policy {x}. Policy must be one of 'compatible', 'additive', or 'dep_only'"
878        ),
879    })
