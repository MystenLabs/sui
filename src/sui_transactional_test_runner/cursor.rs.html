30#[derive(Debug, Clone, PartialEq)]
31pub enum Value {
32    // Scalars
33    Bytes(Vec<u8>),
34    ObjectId(ObjectID),
35    String(String),
36    U8(u8),
37    U16(u16),
38    U32(u32),
39    U64(u64),
40    U128(u128),
41    U256(U256),
42
43    // Types
44    Struct(StructTag),
45
46    // Aggregates
47    List(Vec<Value>),
48    Tuple(Vec<Value>),
49}
50
51enum Encoding {
52    Bcs,
53    Bin,
54}
55
56enum Suffix {
57    U8,
58    U16,
59    U32,
60    U64,
61    U128,
62    U256,
63}
64
65impl Serialize for Value {
66    fn serialize<S>(&self, serializer: S) -> std::result::Result<S::Ok, S::Error>
67    where
68        S: serde::Serializer,
69    {
70        match self {
71            Value::Bytes(bs) => bs.serialize(serializer),
72            Value::ObjectId(id) => id.serialize(serializer),
73            Value::String(s) => s.serialize(serializer),
74            Value::U8(n) => n.serialize(serializer),
75            Value::U16(n) => n.serialize(serializer),
76            Value::U32(n) => n.serialize(serializer),
77            Value::U64(n) => n.serialize(serializer),
78            Value::U128(n) => n.serialize(serializer),
79            Value::U256(n) => n.serialize(serializer),
80
81            Value::Struct(tag) => tag.serialize(serializer),
82
83            Value::List(values) => {
84                let mut seq = serializer.serialize_seq(Some(values.len()))?;
85                for value in values {
86                    seq.serialize_element(value)?;
87                }
88                seq.end()
89            }
90
91            Value::Tuple(values) => {
92                let mut tuple = serializer.serialize_tuple(values.len())?;
93                for value in values {
94                    tuple.serialize_element(value)?;
95                }
96                tuple.end()
97            }
98        }
99    }
100}

102/// Parse a string representing an encoded cursor. Cursors can either be BCS-encoded or bincoded,
103/// and can contain:
104///
105/// - Object IDs (e.g., `0x1234`)
106/// - Numbers (e.g., `42`, `1000u16`, `255u8`)
107/// - Strings (e.g., `'hello world'`, `'it\'s working')
108/// - Struct tags (e.g., `0x2::coin::Coin<0x2::sui::SUI>>`)
109/// - Nested tuples (e.g., `(42, 'hello')`)
110/// - Nested lists (e.g., `[1, 2, 3]`, `[]`)
111/// - Nested encoded bytes (e.g., `bcs(0x1234)` or `bin(0x1234)`)
112pub fn parse(mut input: &str) -> Result<Vec<u8>> {
113    encoded(&mut input)
114}
115
116fn bytes(input: &mut &str) -> Result<Value> {
117    Ok(Value::Bytes(encoded(input)?))
118}
119
120fn encoded(input: &mut &str) -> Result<Vec<u8>> {
121    let enc = encode(input)?;
122    let val = value(input)?;
123
124    Ok(match enc {
125        Encoding::Bcs => bcs::to_bytes(&val).map_err(|e| CE::from_external_error(input, e))?,
126        Encoding::Bin => bincode::encode_to_vec(BorrowCompat(&val), bincode_config())
127            .map_err(|e| CE::from_external_error(input, e))?,
128    })
129}
130
131fn bincode_config() -> bincode::config::Configuration<BigEndian, Fixint> {
132    bincode::config::standard()
133        .with_big_endian()
134        .with_fixed_int_encoding()
135}
136
137fn encode(input: &mut &str) -> Result<Encoding> {
138    let ident = alt(("bin".map(|_| Encoding::Bin), "bcs".map(|_| Encoding::Bcs)));
139    preceded(multispace0, ident).parse_next(input)
140}
141
142fn value(input: &mut &str) -> Result<Value> {
143    // Consume preceding whitespace and then try to parse each variant of value in turn until one
144    // succeeds. Ordering is relevant when handling grammars that share common prefixes (for
145    // example, `struct_` and `id` both start with an address).
146    preceded(
147        multispace0,
148        alt((struct_, id, number, string, bytes, tuple, list)),
149    )
150    .parse_next(input)
151}
152
153fn struct_(input: &mut &str) -> Result<Value> {
154    struct_tag(input).map(Value::Struct)
155}
156
157fn struct_tag(input: &mut &str) -> Result<StructTag> {
158    let (address, _, module, _, name, type_params) = seq!(
159        address,
160        "::",
161        identifier,
162        "::",
163        identifier,
164        opt(type_params)
165    )
166    .parse_next(input)?;
167
168    Ok(StructTag {
169        address,
170        module,
171        name,
172        type_params: type_params.unwrap_or_default(),
173    })
174}
175
176fn identifier(input: &mut &str) -> Result<Identifier> {
177    let content = seq!((
178        one_of(('a'..='z', 'A'..='Z', '_')),
179        repeat::<_, _, String, _, _>(0.., one_of(('a'..='z', 'A'..='Z', '0'..='9', '_')))
180    ))
181    .take()
182    .parse_next(input)?;
183
184    Identifier::from_str(content).map_err(|_| CE::new())
185}
186
187fn type_params(input: &mut &str) -> Result<Vec<TypeTag>> {
188    Ok(seq!(
189        "<",
190        separated(
191            0..,
192            preceded(multispace0, type_tag),
193            preceded(multispace0, ",")
194        ),
195        preceded(multispace0, ">")
196    )
197    .parse_next(input)?
198    .1)
199}
200
201fn type_tag(input: &mut &str) -> Result<TypeTag> {
202    alt((
203        "bool".map(|_| TypeTag::Bool),
204        "u8".map(|_| TypeTag::U8),
205        "u16".map(|_| TypeTag::U16),
206        "u32".map(|_| TypeTag::U32),
207        "u64".map(|_| TypeTag::U64),
208        "u128".map(|_| TypeTag::U128),
209        "u256".map(|_| TypeTag::U256),
210        "address".map(|_| TypeTag::Address),
211        vector,
212        struct_tag.map(|s| TypeTag::Struct(Box::new(s))),
213    ))
214    .parse_next(input)
215}
216
217fn vector(input: &mut &str) -> Result<TypeTag> {
218    let (_, _, inner, _) = seq!(
219        "vector",
220        preceded(multispace0, "<"),
221        preceded(multispace0, type_tag),
222        preceded(multispace0, ">")
223    )
224    .parse_next(input)?;
225    Ok(TypeTag::Vector(Box::new(inner)))
226}
227
228fn id(input: &mut &str) -> Result<Value> {
229    Ok(Value::ObjectId(address(input).map(Into::into)?))
230}
231
232fn address(input: &mut &str) -> Result<AccountAddress> {
233    let addr = ("0x", hex_digit1).take().parse_next(input)?;
234    AccountAddress::from_str(addr).map_err(|e| CE::from_external_error(input, e))
235}
236
237fn number(input: &mut &str) -> Result<Value> {
238    let num = take_while(1.., |c: char| c.is_ascii_digit()).parse_next(input)?;
239    let suffix = opt(suffix).parse_next(input)?.unwrap_or(Suffix::U64);
240
241    Ok(match suffix {
242        Suffix::U8 => Value::U8(num.parse().map_err(|e| CE::from_external_error(input, e))?),
243        Suffix::U16 => Value::U16(num.parse().map_err(|e| CE::from_external_error(input, e))?),
244        Suffix::U32 => Value::U32(num.parse().map_err(|e| CE::from_external_error(input, e))?),
245        Suffix::U64 => Value::U64(num.parse().map_err(|e| CE::from_external_error(input, e))?),
246        Suffix::U128 => Value::U128(num.parse().map_err(|e| CE::from_external_error(input, e))?),
247        Suffix::U256 => Value::U256(num.parse().map_err(|e| CE::from_external_error(input, e))?),
248    })
249}
250
251fn suffix(input: &mut &str) -> Result<Suffix> {
252    alt((
253        "u8".map(|_| Suffix::U8),
254        "u16".map(|_| Suffix::U16),
255        "u32".map(|_| Suffix::U32),
256        "u64".map(|_| Suffix::U64),
257        "u128".map(|_| Suffix::U128),
258        "u256".map(|_| Suffix::U256),
259    ))
260    .parse_next(input)
261}
262
263fn string(input: &mut &str) -> Result<Value> {
264    let (_, content, _) = seq!(
265        "'",
266        repeat(
267            0..,
268            alt(("\\'".map(|_| '\''), "\\\\".map(|_| '\\'), none_of('\'')))
269        ),
270        "'"
271    )
272    .parse_next(input)?;
273
274    Ok(Value::String(content))
275}
276
277fn tuple(input: &mut &str) -> Result<Value> {
278    let (_, values, _) = seq!(
279        preceded(multispace0, "("),
280        separated(1.., value, preceded(multispace0, ",")),
281        preceded(multispace0, ")"),
282    )
283    .parse_next(input)?;
284
285    Ok(Value::Tuple(values))
286}
287
288fn list(input: &mut &str) -> Result<Value> {
289    let (_, values, _) = seq!(
290        preceded(multispace0, "["),
291        separated(0.., value, preceded(multispace0, ",")),
292        preceded(multispace0, "]"),
293    )
294    .parse_next(input)?;
295
296    Ok(Value::List(values))
297}

299#[cfg(test)]
300mod tests {
301    use move_core_types::language_storage::StructTag;

303    use super::*;
304
305    fn expect<T: Serialize>(value: T) -> (Vec<u8>, Vec<u8>) {
306        let bcs = bcs::to_bytes(&value).unwrap();
307        let bin = bincode::encode_to_vec(BorrowCompat(&value), bincode_config()).unwrap();
308        (bcs, bin)
309    }
310
311    #[test]
312    fn test_object_id() {
313        let value = ObjectID::from_str("0x1234").unwrap();
314        let (bcs, bin) = expect(value);
315
316        assert_eq!(parse("bcs(0x1234)").unwrap(), bcs);
317        assert_eq!(parse("bin(0x1234)").unwrap(), bin);
318    }
319
320    #[test]
321    fn test_number() {
322        let value = 42u64;
323        let (bcs, bin) = expect(value);
324
325        assert_eq!(parse("bcs(42)").unwrap(), bcs);
326        assert_eq!(parse("bin(42)").unwrap(), bin);
327    }
328    #[test]
329    fn test_number_with_suffix() {
330        let u8_value = 255u8;
331        let (bcs_u8, bin_u8) = expect(u8_value);
332        assert_eq!(parse("bcs(255u8)").unwrap(), bcs_u8);
333        assert_eq!(parse("bin(255u8)").unwrap(), bin_u8);
334
335        let u16_value = 1000u16;
336        let (bcs_u16, bin_u16) = expect(u16_value);
337        assert_eq!(parse("bcs(1000u16)").unwrap(), bcs_u16);
338        assert_eq!(parse("bin(1000u16)").unwrap(), bin_u16);
339
340        let u128_value = 1_000_000u128;
341        let (bcs_u128, bin_u128) = expect(u128_value);
342        assert_eq!(parse("bcs(1000000u128)").unwrap(), bcs_u128);
343        assert_eq!(parse("bin(1000000u128)").unwrap(), bin_u128);
344    }
345    #[test]
346    fn test_string() {
347        let value = "hello world";
