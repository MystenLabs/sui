31#[derive(Debug, Clone, PartialEq)]
32pub enum Value {
33    // Scalars
34    Bytes(Vec<u8>),
35    Digest(Digest),
36    ObjectId(ObjectID),
37    String(String),
38    U8(u8),
39    U16(u16),
40    U32(u32),
41    U64(u64),
42    U128(u128),
43    U256(U256),
44
45    // Types
46    Struct(StructTag),
47
48    // Aggregates
49    List(Vec<Value>),
50    Tuple(Vec<Value>),
51}
52
53enum Encoding {
54    Bcs,
55    Bin,
56}
57
58enum Suffix {
59    U8,
60    U16,
61    U32,
62    U64,
63    U128,
64    U256,
65}

67impl Serialize for Value {
68    fn serialize<S>(&self, serializer: S) -> std::result::Result<S::Ok, S::Error>
69    where
70        S: serde::Serializer,
71    {
72        match self {
73            Value::Bytes(bs) => bs.serialize(serializer),
74            Value::Digest(bs) => bs.serialize(serializer),
75            Value::ObjectId(id) => id.serialize(serializer),
76            Value::String(s) => s.serialize(serializer),
77            Value::U8(n) => n.serialize(serializer),
78            Value::U16(n) => n.serialize(serializer),
79            Value::U32(n) => n.serialize(serializer),
80            Value::U64(n) => n.serialize(serializer),
81            Value::U128(n) => n.serialize(serializer),
82            Value::U256(n) => n.serialize(serializer),
83
84            Value::Struct(tag) => tag.serialize(serializer),
85
86            Value::List(values) => {
87                let mut seq = serializer.serialize_seq(Some(values.len()))?;
88                for value in values {
89                    seq.serialize_element(value)?;
90                }
91                seq.end()
92            }
93
94            Value::Tuple(values) => {
95                let mut tuple = serializer.serialize_tuple(values.len())?;
96                for value in values {
97                    tuple.serialize_element(value)?;
98                }
99                tuple.end()
100            }
101        }
102    }
103}

105/// Parse a string representing an encoded cursor. Cursors can either be BCS-encoded or bincoded,
106/// and can contain:
107///
108/// - Object IDs (e.g., `0x1234`)
109/// - Numbers (e.g., `42`, `1000u16`, `255u8`)
110/// - Strings (e.g., `'hello world'`, `'it\'s working')
111/// - Digests (e.g., `digest(123...xyz)`)
112/// - Struct tags (e.g., `0x2::coin::Coin<0x2::sui::SUI>>`)
113/// - Nested tuples (e.g., `(42, 'hello')`)
114/// - Nested lists (e.g., `[1, 2, 3]`, `[]`)
115/// - Nested encoded bytes (e.g., `bcs(0x1234)` or `bin(0x1234)`)
116pub fn parse(mut input: &str) -> Result<Vec<u8>> {
117    encoded(&mut input)
118}
119
120fn bytes(input: &mut &str) -> Result<Value> {
121    Ok(Value::Bytes(encoded(input)?))
122}
123
124fn encoded(input: &mut &str) -> Result<Vec<u8>> {
125    let enc = encode(input)?;
126    let val = value(input)?;
127
128    Ok(match enc {
129        Encoding::Bcs => bcs::to_bytes(&val).map_err(|e| CE::from_external_error(input, e))?,
130        Encoding::Bin => bincode::encode_to_vec(BorrowCompat(&val), bincode_config())
131            .map_err(|e| CE::from_external_error(input, e))?,
132    })
133}
134
135fn bincode_config() -> bincode::config::Configuration<BigEndian, Fixint> {
136    bincode::config::standard()
137        .with_big_endian()
138        .with_fixed_int_encoding()
139}
140
141fn encode(input: &mut &str) -> Result<Encoding> {
142    let ident = alt(("bcs".map(|_| Encoding::Bcs), "bin".map(|_| Encoding::Bin)));
143    preceded(multispace0, ident).parse_next(input)
144}
145
146fn value(input: &mut &str) -> Result<Value> {
147    // Consume preceding whitespace and then try to parse each variant of value in turn until one
148    // succeeds. Ordering is relevant when handling grammars that share common prefixes (for
149    // example, `struct_` and `id` both start with an address).
150    preceded(
151        multispace0,
152        alt((struct_, id, number, string, digest, bytes, tuple, list)),
153    )
154    .parse_next(input)
155}
156
157fn struct_(input: &mut &str) -> Result<Value> {
158    struct_tag(input).map(Value::Struct)
159}
160
161fn struct_tag(input: &mut &str) -> Result<StructTag> {
162    let (address, _, module, _, name, type_params) = seq!(
163        address,
164        "::",
165        identifier,
166        "::",
167        identifier,
168        opt(type_params)
169    )
170    .parse_next(input)?;
171
172    Ok(StructTag {
173        address,
174        module,
175        name,
176        type_params: type_params.unwrap_or_default(),
177    })
178}
179
180fn identifier(input: &mut &str) -> Result<Identifier> {
181    let content = seq!((
182        one_of(('a'..='z', 'A'..='Z', '_')),
183        repeat::<_, _, String, _, _>(0.., one_of(('a'..='z', 'A'..='Z', '0'..='9', '_')))
184    ))
185    .take()
186    .parse_next(input)?;
187
188    Identifier::from_str(content).map_err(|_| CE::new())
189}
190
191fn type_params(input: &mut &str) -> Result<Vec<TypeTag>> {
192    Ok(seq!(
193        "<",
194        separated(
195            0..,
196            preceded(multispace0, type_tag),
197            preceded(multispace0, ",")
198        ),
199        preceded(multispace0, ">")
200    )
201    .parse_next(input)?
202    .1)
203}
204
205fn type_tag(input: &mut &str) -> Result<TypeTag> {
206    alt((
207        "bool".map(|_| TypeTag::Bool),
208        "u8".map(|_| TypeTag::U8),
209        "u16".map(|_| TypeTag::U16),
210        "u32".map(|_| TypeTag::U32),
211        "u64".map(|_| TypeTag::U64),
212        "u128".map(|_| TypeTag::U128),
213        "u256".map(|_| TypeTag::U256),
214        "address".map(|_| TypeTag::Address),
215        vector,
216        struct_tag.map(|s| TypeTag::Struct(Box::new(s))),
217    ))
218    .parse_next(input)
219}
220
221fn vector(input: &mut &str) -> Result<TypeTag> {
222    let (_, _, inner, _) = seq!(
223        "vector",
224        preceded(multispace0, "<"),
225        preceded(multispace0, type_tag),
226        preceded(multispace0, ">")
227    )
228    .parse_next(input)?;
229    Ok(TypeTag::Vector(Box::new(inner)))
230}
231
232fn id(input: &mut &str) -> Result<Value> {
233    Ok(Value::ObjectId(address(input).map(Into::into)?))
234}
235
236fn address(input: &mut &str) -> Result<AccountAddress> {
237    let addr = ("0x", hex_digit1).take().parse_next(input)?;
238    AccountAddress::from_str(addr).map_err(|e| CE::from_external_error(input, e))
239}
240
241fn number(input: &mut &str) -> Result<Value> {
242    let num = take_while(1.., |c: char| c.is_ascii_digit()).parse_next(input)?;
243    let suffix = opt(suffix).parse_next(input)?.unwrap_or(Suffix::U64);
244
245    Ok(match suffix {
246        Suffix::U8 => Value::U8(num.parse().map_err(|e| CE::from_external_error(input, e))?),
247        Suffix::U16 => Value::U16(num.parse().map_err(|e| CE::from_external_error(input, e))?),
248        Suffix::U32 => Value::U32(num.parse().map_err(|e| CE::from_external_error(input, e))?),
249        Suffix::U64 => Value::U64(num.parse().map_err(|e| CE::from_external_error(input, e))?),
250        Suffix::U128 => Value::U128(num.parse().map_err(|e| CE::from_external_error(input, e))?),
251        Suffix::U256 => Value::U256(num.parse().map_err(|e| CE::from_external_error(input, e))?),
252    })
253}
254
255fn suffix(input: &mut &str) -> Result<Suffix> {
256    alt((
257        "u8".map(|_| Suffix::U8),
258        "u16".map(|_| Suffix::U16),
259        "u32".map(|_| Suffix::U32),
260        "u64".map(|_| Suffix::U64),
261        "u128".map(|_| Suffix::U128),
262        "u256".map(|_| Suffix::U256),
263    ))
264    .parse_next(input)
265}
266
267fn string(input: &mut &str) -> Result<Value> {
268    let (_, content, _) = seq!(
269        "'",
270        repeat(
271            0..,
272            alt(("\\'".map(|_| '\''), "\\\\".map(|_| '\\'), none_of('\'')))
273        ),
274        "'"
275    )
276    .parse_next(input)?;
277
278    Ok(Value::String(content))
279}
280
281fn digest(input: &mut &str) -> Result<Value> {
282    fn is_base58(c: char) -> bool {
283        c.is_ascii_alphanumeric() && !matches!(c, '0' | 'O' | 'I' | 'l')
284    }
285
286    let (_, _, s, _) = seq!(
287        "digest",
288        preceded(multispace0, "("),
289        preceded(multispace0, take_while(0.., is_base58)),
290        preceded(multispace0, ")")
291    )
292    .parse_next(input)?;
293
294    Ok(Value::Digest(
295        Base58::decode(s)
296            .map_err(|e| CE::from_external_error(input, e))?
297            .try_into()
298            .map_err(|e| CE::from_external_error(input, e))?,
299    ))
300}
301
302fn tuple(input: &mut &str) -> Result<Value> {
303    let (_, values, _) = seq!(
304        preceded(multispace0, "("),
305        separated(1.., value, preceded(multispace0, ",")),
306        preceded(multispace0, ")"),
307    )
308    .parse_next(input)?;
309
310    Ok(Value::Tuple(values))
311}
312
313fn list(input: &mut &str) -> Result<Value> {
314    let (_, values, _) = seq!(
315        preceded(multispace0, "["),
316        separated(0.., value, preceded(multispace0, ",")),
317        preceded(multispace0, "]"),
318    )
319    .parse_next(input)?;
320
321    Ok(Value::List(values))
322}

330    fn expect<T: Serialize>(value: T) -> (Vec<u8>, Vec<u8>) {
331        let bcs = bcs::to_bytes(&value).unwrap();
332        let bin = bincode::encode_to_vec(BorrowCompat(&value), bincode_config()).unwrap();
333        (bcs, bin)
334    }
335
336    #[test]
337    fn test_object_id() {
338        let value = ObjectID::from_str("0x1234").unwrap();
339        let (bcs, bin) = expect(value);
340
341        assert_eq!(parse("bcs(0x1234)").unwrap(), bcs);
342        assert_eq!(parse("bin(0x1234)").unwrap(), bin);
343    }
344
345    #[test]
346    fn test_number() {
347        let value = 42u64;
348        let (bcs, bin) = expect(value);
349
350        assert_eq!(parse("bcs(42)").unwrap(), bcs);
351        assert_eq!(parse("bin(42)").unwrap(), bin);
352    }
353
354    #[test]
355    fn test_number_with_suffix() {
356        let u8_value = 255u8;

358        assert_eq!(parse("bcs(255u8)").unwrap(), bcs_u8);
359        assert_eq!(parse("bin(255u8)").unwrap(), bin_u8);
360
361        let u16_value = 1000u16;
