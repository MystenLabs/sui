4//! Utility for printing a `ProgrammableTransaction` in the format for test cases.
5
6use core::panic;
7use std::{collections::BTreeMap, fmt};
8
9use move_core_types::account_address::AccountAddress;
10use sui_types::{
11    BRIDGE_ADDRESS, DEEPBOOK_ADDRESS, MOVE_STDLIB_ADDRESS, SUI_FRAMEWORK_ADDRESS,
12    SUI_SYSTEM_ADDRESS,
13    transaction::{Argument, Command, ProgrammableMoveCall},
14    type_input::{StructInput, TypeInput},
15};
16
17use crate::programmable_transaction_test_parser::token::{
18    GAS_COIN, INPUT, MAKE_MOVE_VEC, MERGE_COINS, NESTED_RESULT, RESULT, SPLIT_COINS,
19    TRANSFER_OBJECTS,
20};
21
22pub fn commands<W: fmt::Write>(
23    named_addresses: &BTreeMap<AccountAddress, String>,
24    buf: &mut W,
25    cs: &[Command],
26) -> fmt::Result {
27    for (i, c) in cs.iter().enumerate() {
28        write!(buf, "//> {i}: ")?;
29        command(named_addresses, buf, c)?;
30        writeln!(buf, ";")?;
31    }
32    Ok(())
33}
34
35pub fn command<W: fmt::Write>(
36    named_addresses: &BTreeMap<AccountAddress, String>,
37    buf: &mut W,
38    command: &Command,
39) -> fmt::Result {
40    match command {
41        Command::MoveCall(m) => {
42            let ProgrammableMoveCall {
43                package,
44                module,
45                function,
46                type_arguments: tys,
47                arguments: args,
48            } = &**m;
49            write!(
50                buf,
51                "{}::{}::{}",
52                address_str(named_addresses, **package),
53                module,
54                function
55            )?;
56            type_arguments(named_addresses, buf, tys)?;
57            write!(buf, "(")?;
58            arguments(buf, args)?;
59            write!(buf, ")")
60        }
61        Command::TransferObjects(objs, r) => {
62            write!(buf, "{}([", TRANSFER_OBJECTS)?;
63            arguments(buf, objs)?;
64            write!(buf, "], ")?;
65            argument(buf, r)?;
66            write!(buf, ")")
67        }
68        Command::SplitCoins(c, amts) => {
69            write!(buf, "{}(", SPLIT_COINS)?;
70            argument(buf, c)?;
71            write!(buf, ", [")?;
72            arguments(buf, amts)?;
73            write!(buf, "])")
74        }
75        Command::MergeCoins(c, coins) => {
76            write!(buf, "{}(", MERGE_COINS)?;
77            argument(buf, c)?;
78            write!(buf, ", [")?;
79            arguments(buf, coins)?;
80            write!(buf, "])")
81        }
82        Command::MakeMoveVec(t_opt, args) => {
83            write!(buf, "{}", MAKE_MOVE_VEC)?;
84            type_arguments(named_addresses, buf, t_opt.as_slice())?;
85            write!(buf, "([")?;
86            arguments(buf, args)?;
87            write!(buf, "])")
88        }
89        Command::Publish(_, _) | Command::Upgrade(_, _, _, _) => panic!(
90            "Publish and Upgrade are not supported. In transactional adapter tests, you cannot \
91            publish or upgrade via bytes"
92        ),
93    }
94}

96pub fn arguments<W: fmt::Write>(buf: &mut W, args: &[Argument]) -> fmt::Result {
97    for (i, a) in args.iter().enumerate() {
98        if i > 0 {
99            write!(buf, ", ")?;
100        }
101        argument(buf, a)?;
102    }
103    Ok(())
104}
105
106pub fn argument<W: fmt::Write>(buf: &mut W, arg: &Argument) -> fmt::Result {
107    match arg {
108        Argument::GasCoin => write!(buf, "{}", GAS_COIN),
109        Argument::Input(i) => write!(buf, "{}({})", INPUT, i),
110        Argument::Result(i) => write!(buf, "{}({})", RESULT, i),
111        Argument::NestedResult(i, j) => write!(buf, "{}({}, {})", NESTED_RESULT, i, j),
112    }
113}

115pub fn type_<W: fmt::Write>(
116    named_addresses: &BTreeMap<AccountAddress, String>,
117    buf: &mut W,
118    ty: &TypeInput,
119) -> fmt::Result {
120    match ty {
121        TypeInput::U8 => write!(buf, "u8"),
122        TypeInput::U16 => write!(buf, "u16"),
123        TypeInput::U32 => write!(buf, "u32"),
124        TypeInput::U64 => write!(buf, "u64"),
125        TypeInput::U128 => write!(buf, "u128"),
126        TypeInput::U256 => write!(buf, "u256"),
127        TypeInput::Address => write!(buf, "address"),
128        TypeInput::Signer => write!(buf, "signer"),
129        TypeInput::Bool => write!(buf, "bool"),
130        TypeInput::Struct(s) => struct_type_(named_addresses, buf, s),
131        TypeInput::Vector(ty) => {
132            write!(buf, "vector<")?;
133            type_(named_addresses, buf, ty)?;
134            write!(buf, ">")
135        }
136    }
137}
138
139pub fn struct_type_<W: fmt::Write>(
140    named_addresses: &BTreeMap<AccountAddress, String>,
141    buf: &mut W,
142    s: &StructInput,
143) -> fmt::Result {
144    let StructInput {
145        address,
146        module,
147        name,
148        type_params,
149    } = s;
150    write!(
151        buf,
152        "{}::{}::{}",
153        address_str(named_addresses, *address),
154        module,
155        name
156    )?;
157    type_arguments(named_addresses, buf, type_params)?;
158    Ok(())
159}
160
161pub fn type_arguments<W: fmt::Write>(
162    named_addresses: &BTreeMap<AccountAddress, String>,
163    buf: &mut W,
164    tys: &[TypeInput],
165) -> fmt::Result {
166    if !tys.is_empty() {
167        write!(buf, "<")?;
168        for (i, ty) in tys.iter().enumerate() {
169            if i > 0 {
170                write!(buf, ", ")?;
171            }
172            type_(named_addresses, buf, ty)?;
173        }
174        write!(buf, ">")?;
175    }
176    Ok(())
177}
178
179pub fn address_str(
180    named_addresses: &BTreeMap<AccountAddress, String>,
181    address: AccountAddress,
182) -> String {
183    match address {
184        SUI_FRAMEWORK_ADDRESS => "sui".to_owned(),
185        MOVE_STDLIB_ADDRESS => "std".to_owned(),
186        SUI_SYSTEM_ADDRESS => "sui_system".to_owned(),
187        BRIDGE_ADDRESS => "bridge".to_owned(),
188        DEEPBOOK_ADDRESS => "deepbook".to_owned(),
189        _ => match named_addresses.get(&address) {
190            Some(name) => name.clone(),
191            _ => panic!("Unknown address: {}", address),
192        },
193    }
194}
