103    pub fn new_sim_replica_with_protocol_version_and_accounts<R>(
104        mut rng: R,
105        chain_start_timestamp_ms: u64,
106        protocol_version: ProtocolVersion,
107        account_configs: Vec<AccountConfig>,
108        key_pair_wrappers: Vec<KeyPairWrapper>,
109        reference_gas_price: Option<u64>,
110        path: Option<PathBuf>,
111        enable_accumulators: bool,
112        enable_authenticated_event_streams: bool,
113    ) -> (Simulacrum<R, Self>, PersistedStoreInnerReadOnlyWrapper)
114    where
115        R: rand::RngCore + rand::CryptoRng,
116    {
117        let path: PathBuf = path.unwrap_or(tempdir().unwrap().keep());
118
119        let mut builder = ConfigBuilder::new_with_temp_dir()
120            .rng(&mut rng)
121            .with_chain_start_timestamp_ms(chain_start_timestamp_ms)
122            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
123            .with_protocol_version(protocol_version)
124            .with_accounts(account_configs);
125
126        if !key_pair_wrappers.is_empty() {
127            builder = builder.deterministic_committee_validators(key_pair_wrappers)
128        };
129        if let Some(reference_gas_price) = reference_gas_price {
130            builder = builder.with_reference_gas_price(reference_gas_price)
131        };
132
133        let _override_guard = if enable_accumulators || enable_authenticated_event_streams {
134            Some(ProtocolConfig::apply_overrides_for_testing(
135                move |_, cfg| {
136                    let mut c = cfg;
137                    if enable_accumulators {
138                        c.enable_accumulators_for_testing();
139                    }
140                    if enable_authenticated_event_streams {
141                        c.enable_authenticated_event_streams_for_testing();
142                    }
143                    c
144                },
145            ))
146        } else {
147            None
148        };
149
150        let config = builder.build();
151
152        let genesis = &config.genesis;
153
154        let store = PersistedStore::new(genesis, path);
155        let read_only_wrapper = store.read_replica();
156        (
157            Simulacrum::new_with_network_config_store(&config, rng, store),
158            read_only_wrapper,
159        )
160    }

298    fn insert_checkpoint(&mut self, checkpoint: VerifiedCheckpoint) {
299        self.read_write
300            .checkpoint_digest_to_sequence_number
301            .insert(checkpoint.digest(), checkpoint.sequence_number())
302            .expect("Fatal: DB write failed");
303        self.read_write
304            .checkpoints
305            .insert(checkpoint.sequence_number(), checkpoint.serializable_ref())
306            .expect("Fatal: DB write failed");
307    }

309    fn insert_checkpoint_contents(&mut self, contents: CheckpointContents) {
310        self.read_write
311            .checkpoint_contents
312            .insert(contents.digest(), &contents)
313            .expect("Fatal: DB write failed");
314    }
315
316    fn insert_committee(&mut self, committee: Committee) {
317        let epoch = committee.epoch as usize;
318
319        let mut committees = self
320            .read_write
321            .epoch_to_committee
322            .get(&())
323            .expect("Fatal: DB read failed")
324            .unwrap_or_default();
325
326        if committees.get(epoch).is_some() {
327            return;
328        }
329
330        if committees.len() == epoch {
331            committees.push(committee);
332        } else {
333            panic!("committee was inserted into EpochCommitteeMap out of order");
334        }
335        self.read_write
336            .epoch_to_committee
337            .insert(&(), &committees)
338            .expect("Fatal: DB write failed");
339    }
340
341    fn insert_executed_transaction(
342        &mut self,
343        transaction: VerifiedTransaction,
344        effects: TransactionEffects,
345        events: TransactionEvents,
346        written_objects: BTreeMap<ObjectID, Object>,
347    ) {
348        let deleted_objects = effects.deleted();
349        let tx_digest = *effects.transaction_digest();
350        self.insert_transaction(transaction);
351        self.insert_transaction_effects(effects);
352        self.insert_events(&tx_digest, events);
353        self.update_objects(written_objects, deleted_objects);
354    }
355
356    fn insert_transaction(&mut self, transaction: VerifiedTransaction) {
357        self.read_write
358            .transactions
359            .insert(transaction.digest(), transaction.serializable_ref())
360            .expect("Fatal: DB write failed");
361    }
362
363    fn insert_transaction_effects(&mut self, effects: TransactionEffects) {
364        self.read_write
365            .effects
366            .insert(effects.transaction_digest(), &effects)
367            .expect("Fatal: DB write failed");
368    }
369
370    fn insert_events(&mut self, tx_digest: &TransactionDigest, events: TransactionEvents) {
371        self.read_write
372            .events
373            .insert(tx_digest, &events)
374            .expect("Fatal: DB write failed");
375    }
376
377    fn update_objects(
378        &mut self,
379        written_objects: BTreeMap<ObjectID, Object>,
380        deleted_objects: Vec<(ObjectID, SequenceNumber, ObjectDigest)>,
381    ) {
382        for (object_id, _, _) in deleted_objects {
383            self.read_write
384                .live_objects
385                .remove(&object_id)
386                .expect("Fatal: DB write failed");
387        }
388
389        for (object_id, object) in written_objects {
390            let version = object.version();
391            self.read_write
392                .live_objects
393                .insert(&object_id, &version)
394                .expect("Fatal: DB write failed");
395            let mut q = self
396                .read_write
397                .objects
398                .get(&object_id)
399                .expect("Fatal: DB read failed")
400                .unwrap_or_default();
401            q.insert(version, object);
402            self.read_write
403                .objects
404                .insert(&object_id, &q)
405                .expect("Fatal: DB write failed");
406        }
407    }

423impl ChildObjectResolver for PersistedStore {
424    fn read_child_object(
425        &self,
426        parent: &ObjectID,
427        child: &ObjectID,
428        child_version_upper_bound: SequenceNumber,
429    ) -> sui_types::error::SuiResult<Option<Object>> {
430        let child_object = match SimulatorStore::get_object(self, child) {
431            None => return Ok(None),
432            Some(obj) => obj,
433        };
434
435        let parent = *parent;
436        if child_object.owner != Owner::ObjectOwner(parent.into()) {
437            return Err(SuiErrorKind::InvalidChildObjectAccess {
438                object: *child,
439                given_parent: parent,
440                actual_owner: child_object.owner.clone(),
441            }
442            .into());
443        }
444
445        if child_object.version() > child_version_upper_bound {
446            return Err(SuiErrorKind::UnsupportedFeatureError {
447                error: "TODO InMemoryStorage::read_child_object does not yet support bounded reads"
448                    .to_owned(),
449            }
450            .into());
451        }
452
453        Ok(Some(child_object))
454    }
455
456    fn get_object_received_at_version(
457        &self,
458        owner: &ObjectID,
459        receiving_object_id: &ObjectID,
460        receive_object_at_version: SequenceNumber,
461        _epoch_id: EpochId,
462    ) -> sui_types::error::SuiResult<Option<Object>> {
463        let recv_object = match SimulatorStore::get_object(self, receiving_object_id) {
464            None => return Ok(None),
465            Some(obj) => obj,
466        };
467        if recv_object.owner != Owner::AddressOwner((*owner).into()) {
468            return Ok(None);
469        }
470
471        if recv_object.version() != receive_object_at_version {
472            return Ok(None);
473        }
474        Ok(Some(recv_object))
475    }
476}
