103    pub fn new_sim_replica_with_protocol_version_and_accounts<R>(
104        mut rng: R,
105        chain_start_timestamp_ms: u64,
106        protocol_config: &ProtocolConfig,
107        account_configs: Vec<AccountConfig>,
108        key_pair_wrappers: Vec<KeyPairWrapper>,
109        reference_gas_price: Option<u64>,
110        path: Option<PathBuf>,
111    ) -> (Simulacrum<R, Self>, PersistedStoreInnerReadOnlyWrapper)
112    where
113        R: rand::RngCore + rand::CryptoRng,
114    {
115        let leaked: &'static ProtocolConfig = Box::leak(Box::new(protocol_config.clone()));
116        let _override_guard = ProtocolConfig::apply_overrides_for_testing(|_, _| leaked.clone());
117        let path: PathBuf = path.unwrap_or(tempdir().unwrap().keep());
118
119        let mut builder = ConfigBuilder::new_with_temp_dir()
120            .rng(&mut rng)
121            .with_chain_start_timestamp_ms(chain_start_timestamp_ms)
122            .deterministic_committee_size(NonZeroUsize::new(1).unwrap())
123            .with_protocol_version(protocol_config.version)
124            .with_accounts(account_configs);
125
126        if !key_pair_wrappers.is_empty() {
127            builder = builder.deterministic_committee_validators(key_pair_wrappers)
128        };
129        if let Some(reference_gas_price) = reference_gas_price {
130            builder = builder.with_reference_gas_price(reference_gas_price)
131        };
132
133        let config = builder.build();
134
135        let genesis = &config.genesis;
136
137        let store = PersistedStore::new(genesis, path);
138        let read_only_wrapper = store.read_replica();
139        (
140            Simulacrum::new_with_network_config_store(&config, rng, store),
141            read_only_wrapper,
142        )
143    }

277    fn insert_checkpoint(&mut self, checkpoint: VerifiedCheckpoint) {
278        self.read_write
279            .checkpoint_digest_to_sequence_number
280            .insert(checkpoint.digest(), checkpoint.sequence_number())
281            .expect("Fatal: DB write failed");

288    fn insert_checkpoint_contents(&mut self, contents: CheckpointContents) {

293    }
294
295    fn insert_committee(&mut self, committee: Committee) {
296        let epoch = committee.epoch as usize;
297
298        let mut committees = self
299            .read_write
300            .epoch_to_committee
301            .get(&())
302            .expect("Fatal: DB read failed")
303            .unwrap_or_default();
304
305        if committees.get(epoch).is_some() {
306            return;
307        }
308
309        if committees.len() == epoch {
310            committees.push(committee);
311        } else {
312            panic!("committee was inserted into EpochCommitteeMap out of order");
313        }
314        self.read_write
315            .epoch_to_committee
316            .insert(&(), &committees)
317            .expect("Fatal: DB write failed");
318    }
319
320    fn insert_executed_transaction(
321        &mut self,
322        transaction: VerifiedTransaction,
323        effects: TransactionEffects,
324        events: TransactionEvents,
325        written_objects: BTreeMap<ObjectID, Object>,
326    ) {
327        let deleted_objects = effects.deleted();
328        let tx_digest = *effects.transaction_digest();
329        self.insert_transaction(transaction);
330        self.insert_transaction_effects(effects);
331        self.insert_events(&tx_digest, events);
332        self.update_objects(written_objects, deleted_objects);
333    }

337            .transactions

339            .expect("Fatal: DB write failed");
340    }

342    fn insert_transaction_effects(&mut self, effects: TransactionEffects) {
343        self.read_write
344            .effects
345            .insert(effects.transaction_digest(), &effects)
346            .expect("Fatal: DB write failed");
347    }
348
349    fn insert_events(&mut self, tx_digest: &TransactionDigest, events: TransactionEvents) {
350        self.read_write
351            .events
352            .insert(tx_digest, &events)
353            .expect("Fatal: DB write failed");
354    }
355
356    fn update_objects(
357        &mut self,
358        written_objects: BTreeMap<ObjectID, Object>,
359        deleted_objects: Vec<(ObjectID, SequenceNumber, ObjectDigest)>,
360    ) {
361        for (object_id, _, _) in deleted_objects {
362            self.read_write
363                .live_objects
364                .remove(&object_id)
365                .expect("Fatal: DB write failed");
366        }
367
368        for (object_id, object) in written_objects {
369            let version = object.version();
370            self.read_write
371                .live_objects
372                .insert(&object_id, &version)
373                .expect("Fatal: DB write failed");
374            let mut q = self
375                .read_write
376                .objects
377                .get(&object_id)
378                .expect("Fatal: DB read failed")
379                .unwrap_or_default();
380            q.insert(version, object);
381            self.read_write
382                .objects
383                .insert(&object_id, &q)
384                .expect("Fatal: DB write failed");
385        }
386    }

402impl ChildObjectResolver for PersistedStore {
403    fn read_child_object(
404        &self,
405        parent: &ObjectID,
406        child: &ObjectID,
407        child_version_upper_bound: SequenceNumber,
408    ) -> sui_types::error::SuiResult<Option<Object>> {
409        let child_object = match SimulatorStore::get_object(self, child) {
410            None => return Ok(None),
411            Some(obj) => obj,
412        };
413
414        let parent = *parent;
415        if child_object.owner != Owner::ObjectOwner(parent.into()) {
416            return Err(SuiErrorKind::InvalidChildObjectAccess {
417                object: *child,
418                given_parent: parent,
419                actual_owner: child_object.owner.clone(),
420            }
421            .into());
422        }
423
424        if child_object.version() > child_version_upper_bound {
425            return Err(SuiErrorKind::UnsupportedFeatureError {
426                error: "TODO InMemoryStorage::read_child_object does not yet support bounded reads"
427                    .to_owned(),
428            }
429            .into());
430        }
431
432        Ok(Some(child_object))
433    }
434
435    fn get_object_received_at_version(
436        &self,
437        owner: &ObjectID,
438        receiving_object_id: &ObjectID,
439        receive_object_at_version: SequenceNumber,
440        _epoch_id: EpochId,
441    ) -> sui_types::error::SuiResult<Option<Object>> {
442        let recv_object = match SimulatorStore::get_object(self, receiving_object_id) {
443            None => return Ok(None),
444            Some(obj) => obj,
445        };
446        if recv_object.owner != Owner::AddressOwner((*owner).into()) {
447            return Ok(None);
448        }
449
450        if recv_object.version() != receive_object_at_version {
451            return Ok(None);
452        }
453        Ok(Some(recv_object))
454    }
