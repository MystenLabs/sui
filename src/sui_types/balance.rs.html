16pub const SUI_MODULE_NAME: &IdentStr = ident_str!("sui");
17pub const BALANCE_MODULE_NAME: &IdentStr = ident_str!("balance");
18pub const BALANCE_STRUCT_NAME: &IdentStr = ident_str!("Balance");
19pub const BALANCE_CREATE_REWARDS_FUNCTION_NAME: &IdentStr = ident_str!("create_staking_rewards");
20pub const BALANCE_DESTROY_REBATES_FUNCTION_NAME: &IdentStr = ident_str!("destroy_storage_rebates");

22#[serde_as]
23#[derive(Debug, Serialize, Deserialize, Clone, Eq, PartialEq, JsonSchema)]
24pub struct Supply {
25    #[schemars(with = "BigInt<u64>")]
26    #[serde_as(as = "Readable<BigInt<u64>, _>")]
27    pub value: u64,

30#[derive(Debug, Serialize, Deserialize, Clone, JsonSchema, Eq, PartialEq)]
31pub struct Balance {
32    value: u64,
33}
34
35impl Balance {
36    pub fn new(value: u64) -> Self {
37        Self { value }
38    }
39
40    pub fn type_(type_param: TypeTag) -> StructTag {
41        StructTag {
42            address: SUI_FRAMEWORK_ADDRESS,
43            module: BALANCE_MODULE_NAME.to_owned(),
44            name: BALANCE_STRUCT_NAME.to_owned(),
45            type_params: vec![type_param],
46        }
47    }
48
49    pub fn type_tag(type_param: TypeTag) -> TypeTag {
50        TypeTag::Struct(Box::new(Self::type_(type_param)))
51    }
52
53    pub fn is_balance(s: &StructTag) -> bool {
54        s.address == SUI_FRAMEWORK_ADDRESS
55            && s.module.as_ident_str() == BALANCE_MODULE_NAME
56            && s.name.as_ident_str() == BALANCE_STRUCT_NAME
57    }
58
59    pub fn is_balance_type(type_param: &TypeTag) -> bool {
60        if let TypeTag::Struct(struct_tag) = type_param {
61            Self::is_balance(struct_tag)
62        } else {
63            false
64        }
65    }
66
67    /// If the given type is `Balance<T>`, return `Some(T)`.
68    pub fn maybe_get_balance_type_param(ty: &TypeTag) -> Option<TypeTag> {
69        if let TypeTag::Struct(struct_tag) = ty
70            && Self::is_balance(struct_tag)
71        {
72            assert_eq!(struct_tag.type_params.len(), 1);
73            return Some(struct_tag.type_params[0].clone());
74        }
75        None
76    }
77
78    pub fn withdraw(&mut self, amount: u64) -> Result<(), ExecutionError> {
79        fp_ensure!(
80            self.value >= amount,
81            ExecutionError::new_with_source(
82                ExecutionErrorKind::InsufficientCoinBalance,
83                format!("balance: {} required: {}", self.value, amount)
84            )
85        );
86        self.value -= amount;
87        Ok(())
88    }
89
90    pub fn deposit_for_safe_mode(&mut self, amount: u64) {
91        self.value += amount;
92    }
93
94    pub fn value(&self) -> u64 {
95        self.value
96    }
97
98    pub fn to_bcs_bytes(&self) -> Vec<u8> {
99        bcs::to_bytes(&self).unwrap()
100    }
101
102    pub fn layout(type_param: TypeTag) -> MoveStructLayout {
103        MoveStructLayout {
104            type_: Self::type_(type_param),
105            fields: vec![MoveFieldLayout::new(
106                ident_str!("value").to_owned(),
107                MoveTypeLayout::U64,
108            )],
109        }
110    }
111
112    /// Check if a struct layout represents a `Balance<T>` type with the expected field structure.
113    pub fn is_balance_layout(struct_layout: &MoveStructLayout) -> bool {
114        let ty = &struct_layout.type_;
115
116        if !Self::is_balance(ty) {
117            return false;
118        }
119
120        if ty.type_params.len() != 1 {
121            return false;
122        }
123
124        if struct_layout.fields.len() != 1 {
125            return false;
126        }
127
128        let Some(field) = struct_layout.fields.first() else {
129            return false;
130        };
131
132        if field.name.as_str() != "value" {
133            return false;
134        }
135
136        if !matches!(field.layout, MoveTypeLayout::U64) {
137            return false;
138        }
139
140        true
141    }
142}
