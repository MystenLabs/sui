25/// Rust representation of the Move type 0x2::deny_list::DenyList.
26/// It has a bag that contains the deny lists for different system types.
27/// At creation, there is only one type (at key 0), which is the Coin type.
28/// We also take advantage of the dynamic nature of Bag to add more types in the future,
29/// as well as making changes to the deny lists for existing types.
30#[derive(Debug, Serialize, Deserialize, Clone)]
31pub struct DenyList {
32    pub id: UID,
33    pub lists: Bag,
34}

36/// Rust representation of the Move type 0x2::deny_list::PerTypeDenyList.
37/// denied_count is a table that stores the number of denied addresses for each coin template type.
38/// It can be used as a quick check to see if an address is denied for any coin template type.
39/// denied_addresses is a table that stores all the addresses that are denied for each coin template type.
40/// The key to the table is the coin template type in string form: package_id::module_name::struct_name.
41#[derive(Debug, Serialize, Deserialize, Clone)]
42pub struct PerTypeDenyList {
43    pub id: UID,
44    // Table<address, u64>
45    pub denied_count: Table,
46    // Table<vec<u8>, VecSet<address>>
47    pub denied_addresses: Table,
48}

50/// Checks coin denylist v1 at signing time.
51/// It checks that none of the coin types in the transaction are denied for the sender.
52pub fn check_coin_deny_list_v1(
53    sender: SuiAddress,
54    input_objects: &CheckedInputObjects,
55    receiving_objects: &ReceivingObjects,
56    funds_withdraw_types: BTreeSet<String>,
57    object_store: &dyn ObjectStore,
58) -> UserInputResult {
59    let mut coin_types =
60        input_object_coin_types_for_denylist_check(input_objects, receiving_objects);
61    coin_types.extend(funds_withdraw_types);
62
63    let Some(deny_list) = get_coin_deny_list(object_store) else {
64        // TODO: This is where we should fire an invariant violation metric.
65        if cfg!(debug_assertions) {
66            panic!("Failed to get the coin deny list");
67        } else {
68            return Ok(());
69        }
70    };
71    check_deny_list_v1_impl(deny_list, sender, coin_types, object_store)
72}

74/// Returns all unique coin types in canonical string form from the input objects and receiving objects.
75/// It filters out SUI coins since it's known that it's not a regulated coin.
76pub(crate) fn input_object_coin_types_for_denylist_check(
77    input_objects: &CheckedInputObjects,
78    receiving_objects: &ReceivingObjects,
79) -> BTreeSet<String> {
80    let all_objects = input_objects
81        .inner()
82        .iter_objects()
83        .chain(receiving_objects.iter_objects());
84    all_objects
85        .filter_map(|obj| {
86            if obj.is_gas_coin() {
87                None
88            } else {
89                obj.coin_type_maybe()
90                    .map(|type_tag| type_tag.to_canonical_string(false))
91            }
92        })
93        .collect()
94}

96fn check_deny_list_v1_impl(
97    deny_list: PerTypeDenyList,
98    address: SuiAddress,
99    coin_types: BTreeSet<String>,
100    object_store: &dyn ObjectStore,
101) -> UserInputResult {
102    let Ok(count) = get_dynamic_field_from_store::<SuiAddress, u64>(
103        object_store,
104        deny_list.denied_count.id,
105        &address,
106    ) else {
107        return Ok(());
108    };
109    if count == 0 {
110        return Ok(());
111    }
112    for coin_type in coin_types {
113        let Ok(denied_addresses) = get_dynamic_field_from_store::<Vec<u8>, VecSet<SuiAddress>>(
114            object_store,
115            deny_list.denied_addresses.id,
116            &coin_type.clone().into_bytes(),
117        ) else {
118            continue;
119        };
120        let denied_addresses: BTreeSet<_> = denied_addresses.contents.into_iter().collect();
121        if denied_addresses.contains(&address) {
122            debug!(
123                "Address {} is denied for coin package {:?}",
124                address, coin_type
125            );
126            return Err(UserInputError::AddressDeniedForCoin { address, coin_type });
127        }
128    }
129    Ok(())
130}

144pub fn get_deny_list_root_object(object_store: &dyn ObjectStore) -> Option<Object> {
145    // TODO: We should return error if this is not found.
146    match object_store.get_object(&SUI_DENY_LIST_OBJECT_ID) {
147        Some(obj) => Some(obj),
148        None => {
149            error!("Deny list object not found");
150            None
151        }
152    }
153}
154
155pub fn get_coin_deny_list(object_store: &dyn ObjectStore) -> Option<PerTypeDenyList> {
156    get_deny_list_root_object(object_store).and_then(|obj| {
157        let deny_list: DenyList = obj
158            .to_rust()
159            .expect("DenyList object type must be consistent");
160        match get_dynamic_field_from_store(
161            object_store,
162            *deny_list.lists.id.object_id(),
163            &DENY_LIST_COIN_TYPE_INDEX,
164        ) {
165            Ok(deny_list) => Some(deny_list),
166            Err(err) => {
167                error!("Failed to get deny list inner state: {}", err);
168                None
169            }
170        }
171    })
172}

174pub fn get_deny_list_obj_initial_shared_version(
175    object_store: &dyn ObjectStore,
176) -> Option<SequenceNumber> {
177    get_deny_list_root_object(object_store).map(|obj| match obj.owner {
178        Owner::Shared {
179            initial_shared_version,
180        } => initial_shared_version,
181        _ => unreachable!("Deny list object must be shared"),
182    })
