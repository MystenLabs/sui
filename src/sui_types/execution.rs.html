21/// A type containing all of the information needed to work in execution with an object whose
22/// consensus stream is ended, and when committing the execution effects of the transaction.
23/// This holds:
24/// 0. The object ID.
25/// 1. The version.
26/// 2. Whether the object appeared as mutable (or owned) in the transaction, or as read-only.
27/// 3. The transaction digest of the previous transaction that used this object mutably or
28///    took it by value.
29pub type ConsensusStreamEndedInfo = (
30    ObjectID,
31    SequenceNumber,
32    SharedObjectMutability,
33    TransactionDigest,
34);
35
36/// A sequence of information about removed consensus objects in the transaction's inputs.
37pub type ConsensusStreamEndedObjects = Vec<ConsensusStreamEndedInfo>;

71/// Used by sui-execution v1 and above, to capture the execution results from Move.
72/// The results represent the primitive information that can then be used to construct
73/// both transaction effects V1 and V2.
74#[derive(Debug, Default)]
75pub struct ExecutionResultsV2 {
76    /// All objects written regardless of whether they were mutated, created, or unwrapped.
77    pub written_objects: BTreeMap<ObjectID, Object>,
78    /// All objects that existed prior to this transaction, and are modified in this transaction.
79    /// This includes any type of modification, including mutated, wrapped and deleted objects.
80    pub modified_objects: BTreeSet<ObjectID>,
81    /// All object IDs created in this transaction.
82    pub created_object_ids: BTreeSet<ObjectID>,
83    /// All object IDs deleted in this transaction.
84    /// No object ID should be in both created_object_ids and deleted_object_ids.
85    pub deleted_object_ids: BTreeSet<ObjectID>,
86    /// All Move events emitted in this transaction.
87    pub user_events: Vec<Event>,
88    /// All accumulator events emitted in this transaction.
89    pub accumulator_events: Vec<AccumulatorEvent>,
90
91    /// Used to track SUI conservation in settlement transactions. Settlement transactions
92    /// gather up withdraws and deposits from other transactions, and record them to accumulator
93    /// fields. The settlement transaction records the total amount of SUI being disbursed here,
94    /// so that we can verify that the amount stored in the fields at the end of the transaction
95    /// is correct.
96    pub settlement_input_sui: u64,
97    pub settlement_output_sui: u64,
98}

105impl ExecutionResultsV2 {
106    pub fn drop_writes(&mut self) {
107        self.written_objects.clear();
108        self.modified_objects.clear();
109        self.created_object_ids.clear();
110        self.deleted_object_ids.clear();
111        self.user_events.clear();
112        self.accumulator_events.clear();
113    }
114
115    pub fn merge_results(&mut self, new_results: Self) {
116        self.written_objects.extend(new_results.written_objects);
117        self.modified_objects.extend(new_results.modified_objects);
118        self.created_object_ids
119            .extend(new_results.created_object_ids);
120        self.deleted_object_ids
121            .extend(new_results.deleted_object_ids);
122        self.user_events.extend(new_results.user_events);
123        self.accumulator_events
124            .extend(new_results.accumulator_events);
125        self.settlement_input_sui += new_results.settlement_input_sui;
126        self.settlement_output_sui += new_results.settlement_output_sui;
127    }
128
129    pub fn update_version_and_previous_tx(
130        &mut self,
131        lamport_version: SequenceNumber,
132        prev_tx: TransactionDigest,
133        input_objects: &BTreeMap<ObjectID, Object>,
134        reshare_at_initial_version: bool,
135    ) {
136        for (id, obj) in self.written_objects.iter_mut() {
137            // TODO: We can now get rid of the following logic by passing in lamport version
138            // into the execution layer, and create new objects using the lamport version directly.
139
140            // Update the version for the written object.
141            match &mut obj.data {
142                Data::Move(obj) => {
143                    // Move objects all get the transaction's lamport timestamp
144                    obj.increment_version_to(lamport_version);
145                }
146
147                Data::Package(pkg) => {
148                    // Modified packages get their version incremented (this is a special case that
149                    // only applies to system packages).  All other packages can only be created,
150                    // and they are left alone.
151                    if self.modified_objects.contains(id) {
152                        debug_assert!(is_system_package(*id));
153                        pkg.increment_version();
154                    }
155                }
156            }
157
158            // Record the version that the shared object was created at in its owner field.  Note,
159            // this only works because shared objects must be created as shared (not created as
160            // owned in one transaction and later converted to shared in another).
161            if let Owner::Shared {
162                initial_shared_version,
163            } = &mut obj.owner
164            {
165                if self.created_object_ids.contains(id) {
166                    assert_eq!(
167                        *initial_shared_version,
168                        SequenceNumber::new(),
169                        "Initial version should be blank before this point for {id:?}",
170                    );
171                    *initial_shared_version = lamport_version;
172                }
173
174                // Update initial_shared_version for reshared objects
175                if reshare_at_initial_version
176                    && let Some(Owner::Shared {
177                        initial_shared_version: previous_initial_shared_version,
178                    }) = input_objects.get(id).map(|obj| &obj.owner)
179                {
180                    debug_assert!(!self.created_object_ids.contains(id));
181                    debug_assert!(!self.deleted_object_ids.contains(id));
182                    debug_assert!(
183                        *initial_shared_version == SequenceNumber::new()
184                            || *initial_shared_version == *previous_initial_shared_version
185                    );
186
187                    *initial_shared_version = *previous_initial_shared_version;
188                }
189            }
190
191            // Record start version for ConsensusAddressOwner objects.
192            if let Owner::ConsensusAddressOwner {
193                start_version,
194                owner,
195            } = &mut obj.owner
196            {
197                debug_assert!(!self.deleted_object_ids.contains(id));
198
199                if let Some(Owner::ConsensusAddressOwner {
200                    start_version: previous_start_version,
201                    owner: previous_owner,
202                }) = input_objects.get(id).map(|obj| &obj.owner)
203                {
204                    if owner == previous_owner {
205                        // Assign existing start_version in case a ConsensusAddressOwner object was
206                        // transferred to the same owner.
207                        *start_version = *previous_start_version;
208                    } else {
209                        // If owner changes, we need to begin a new stream.
210                        *start_version = lamport_version;
211                    }
212                } else {
213                    // ConsensusAddressOwner object was created, transferred from another Owner
214                    // type, or unwrapped, so we begin a new stream.
215                    *start_version = lamport_version;
216                }
217            }
218
219            obj.previous_transaction = prev_tx;
220        }
221    }
222}

224#[derive(Debug, PartialEq, Eq, Hash, PartialOrd, Ord, Clone, Serialize, Deserialize)]
225pub enum ExecutionTimeObservationKey {
226    // Contains all the fields from `ProgrammableMoveCall` besides `arguments`.
227    MoveEntryPoint {
228        /// The package containing the module and function.
229        package: ObjectID,
230        /// The specific module in the package containing the function.
231        module: String,
232        /// The function to be called.
233        function: String,
234        /// The type arguments to the function.
235        /// NOTE: This field is currently not populated.
236        type_arguments: Vec<TypeInput>,
237    },
238    TransferObjects,
239    SplitCoins,
240    MergeCoins,
241    Publish, // special case: should not be used; we only use hard-coded estimate for this
242    MakeMoveVec,
243    Upgrade,
244}
245
246impl std::fmt::Display for ExecutionTimeObservationKey {
247    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
248        match self {
249            ExecutionTimeObservationKey::MoveEntryPoint {
250                module, function, ..
251            } => {
252                write!(f, "{}:{}", module, function)
253            }
254            ExecutionTimeObservationKey::TransferObjects => write!(f, "TransferObjects"),
255            ExecutionTimeObservationKey::SplitCoins => write!(f, "SplitCoins"),
256            ExecutionTimeObservationKey::MergeCoins => write!(f, "MergeCoins"),
257            ExecutionTimeObservationKey::Publish => write!(f, "Publish"),
258            ExecutionTimeObservationKey::MakeMoveVec => write!(f, "MakeMoveVec"),
259            ExecutionTimeObservationKey::Upgrade => write!(f, "Upgrade"),
260        }
261    }
262}
263
264impl ExecutionTimeObservationKey {
265    pub fn is_move_call(&self) -> bool {
266        matches!(self, ExecutionTimeObservationKey::MoveEntryPoint { .. })
267    }
268
269    pub fn from_command(command: &Command) -> Self {
270        match command {
271            Command::MoveCall(call) => ExecutionTimeObservationKey::MoveEntryPoint {
272                package: call.package,
273                module: call.module.clone(),
274                function: call.function.clone(),
275                type_arguments: vec![],
276            },
277            Command::TransferObjects(_, _) => ExecutionTimeObservationKey::TransferObjects,
278            Command::SplitCoins(_, _) => ExecutionTimeObservationKey::SplitCoins,
279            Command::MergeCoins(_, _) => ExecutionTimeObservationKey::MergeCoins,
280            Command::Publish(_, _) => ExecutionTimeObservationKey::Publish,
281            Command::MakeMoveVec(_, _) => ExecutionTimeObservationKey::MakeMoveVec,
282            Command::Upgrade(_, _, _, _) => ExecutionTimeObservationKey::Upgrade,
283        }
284    }
285
286    pub fn default_duration(&self) -> Duration {
287        match self {
288            ExecutionTimeObservationKey::MoveEntryPoint { .. } => Duration::from_millis(1),
289            ExecutionTimeObservationKey::TransferObjects => Duration::from_millis(1),
290            ExecutionTimeObservationKey::SplitCoins => Duration::from_millis(1),
291            ExecutionTimeObservationKey::MergeCoins => Duration::from_millis(1),
292            ExecutionTimeObservationKey::Publish => Duration::from_millis(3),
293            ExecutionTimeObservationKey::MakeMoveVec => Duration::from_millis(1),
294            ExecutionTimeObservationKey::Upgrade => Duration::from_millis(3),
295        }
296    }
297}

300pub enum ExecutionTiming {
301    Success(Duration),
302    Abort(Duration),
303}
304
305impl ExecutionTiming {
306    pub fn is_abort(&self) -> bool {
307        matches!(self, ExecutionTiming::Abort(_))
308    }
309
310    pub fn duration(&self) -> Duration {
311        match self {
312            ExecutionTiming::Success(duration) => *duration,
313            ExecutionTiming::Abort(duration) => *duration,
314        }
315    }

320/// Captures the output of executing a transaction in the execution driver.
321pub enum ExecutionOutput<T> {
322    /// The expected typical path - transaction executed successfully.
323    Success(T),
324    /// Validator has halted at epoch end or epoch mismatch. This is a valid state that should
325    /// be handled gracefully.
326    EpochEnded,
327    /// Execution failed with an error. This should never happen - we use fatal! when encountered.
328    Fatal(SuiError),
329    /// Execution should be retried later due to unsatisfied constraints such as insufficient object
330    /// balance withdrawals that require waiting for the balance to reach a deterministic amount.
331    /// When this happens, the transaction is auto-rescheduled from AuthorityState.
332    RetryLater,
333}
334
335impl<T> ExecutionOutput<T> {
336    /// Unwraps the ExecutionOutput, panicking if it's not Success.
337    /// This is primarily for test code.
338    pub fn unwrap(self) -> T {
339        match self {
340            ExecutionOutput::Success(value) => value,
341            ExecutionOutput::EpochEnded => {
342                panic!("called `ExecutionOutput::unwrap()` on `EpochEnded`")
343            }
344            ExecutionOutput::Fatal(e) => {
345                panic!("called `ExecutionOutput::unwrap()` on `Fatal`: {e}")
346            }
347            ExecutionOutput::RetryLater => {
348                panic!("called `ExecutionOutput::unwrap()` on `RetryLater`")
349            }
350        }
351    }
352
353    /// Expect the execution output to be an error (i.e. not Success).
354    pub fn unwrap_err<S>(self) -> ExecutionOutput<S> {
355        match self {
356            Self::Success(_) => {
357                panic!("called `ExecutionOutput::unwrap_err()` on `Success`")
358            }
359            Self::EpochEnded => ExecutionOutput::EpochEnded,
360            Self::Fatal(e) => ExecutionOutput::Fatal(e),
361            Self::RetryLater => ExecutionOutput::RetryLater,
362        }
363    }
