38impl ChildObjectResolver for InMemoryStorage {
39    fn read_child_object(
40        &self,
41        parent: &ObjectID,
42        child: &ObjectID,
43        child_version_upper_bound: SequenceNumber,
44    ) -> SuiResult<Option<Object>> {
45        let child_object = match self.persistent.get(child).cloned() {
46            None => return Ok(None),
47            Some(obj) => obj,
48        };
49        let parent = *parent;
50        if child_object.owner != Owner::ObjectOwner(parent.into()) {
51            return Err(SuiErrorKind::InvalidChildObjectAccess {
52                object: *child,
53                given_parent: parent,
54                actual_owner: child_object.owner.clone(),
55            }
56            .into());
57        }
58        if child_object.version() > child_version_upper_bound {
59            return Err(SuiErrorKind::UnsupportedFeatureError {
60                error: "TODO InMemoryStorage::read_child_object does not yet support bounded reads"
61                    .to_owned(),
62            }
63            .into());
64        }
65        Ok(Some(child_object))
66    }
67
68    fn get_object_received_at_version(
69        &self,
70        owner: &ObjectID,
71        receiving_object_id: &ObjectID,
72        receive_object_at_version: SequenceNumber,
73        _epoch_id: EpochId,
74    ) -> SuiResult<Option<Object>> {
75        let recv_object = match self.persistent.get(receiving_object_id).cloned() {
76            None => return Ok(None),
77            Some(obj) => obj,
78        };
79        if recv_object.owner != Owner::AddressOwner((*owner).into()) {
80            return Ok(None);
81        }
82
83        if recv_object.version() != receive_object_at_version {
84            return Ok(None);
85        }
86        Ok(Some(recv_object))
87    }
88}

112impl ObjectStore for InMemoryStorage {
113    fn get_object(&self, object_id: &ObjectID) -> Option<Object> {
114        self.persistent.get(object_id).cloned()
115    }
116
117    fn get_object_by_key(&self, object_id: &ObjectID, version: VersionNumber) -> Option<Object> {
118        self.persistent
119            .get(object_id)
120            .and_then(|obj| {
121                if obj.version() == version {
122                    Some(obj)
123                } else {
124                    None
125                }
126            })
127            .cloned()
128    }
129}

159impl InMemoryStorage {
160    pub fn new(objects: Vec<Object>) -> Self {
161        let mut persistent = BTreeMap::new();
162        for o in objects {
163            persistent.insert(o.id(), o);
164        }
165        Self { persistent }
166    }
167
168    pub fn read_input_objects_for_transaction(&self, transaction: &Transaction) -> InputObjects {
169        let mut input_objects = Vec::new();
170        for kind in transaction.transaction_data().input_objects().unwrap() {
171            let obj: Object = match kind {
172                InputObjectKind::MovePackage(id)
173                | InputObjectKind::ImmOrOwnedMoveObject((id, _, _))
174                | InputObjectKind::SharedMoveObject { id, .. } => {
175                    self.get_object(&id).unwrap().clone()
176                }
177            };
178
179            input_objects.push(ObjectReadResult::new(kind, obj.into()));
180        }
181        input_objects.into()
182    }
183
184    pub fn get_object(&self, id: &ObjectID) -> Option<&Object> {
185        self.persistent.get(id)
186    }
187
188    pub fn get_objects(&self, objects: &[ObjectID]) -> Vec<Option<&Object>> {
189        let mut result = Vec::new();
190        for id in objects {
191            result.push(self.get_object(id));
192        }
193        result
194    }
195
196    pub fn insert_object(&mut self, object: Object) {
197        let id = object.id();
198        self.persistent.insert(id, object);
199    }
200
201    pub fn remove_object(&mut self, object_id: ObjectID) -> Option<Object> {
202        self.persistent.remove(&object_id)
203    }
204
205    pub fn objects(&self) -> &BTreeMap<ObjectID, Object> {
206        &self.persistent
207    }
208
209    pub fn into_inner(self) -> BTreeMap<ObjectID, Object> {
210        self.persistent
211    }
212
213    pub fn finish(&mut self, written: WrittenObjects) {
214        for (_id, new_object) in written {
215            debug_assert!(new_object.id() == _id);
216            self.insert_object(new_object);
217        }
218    }
