41impl InnerTemporaryStore {
42    pub fn get_output_keys(&self, effects: &TransactionEffects) -> Vec<InputKey> {
43        let mut output_keys: Vec<_> = self
44            .written
45            .iter()
46            .map(|(id, obj)| {
47                if obj.is_package() {
48                    InputKey::Package { id: *id }
49                } else {
50                    InputKey::VersionedObject {
51                        id: obj.full_id(),
52                        version: obj.version(),
53                    }
54                }
55            })
56            .collect();
57
58        // Add all stream-ended consensus objects to the outputkeys that then get sent to notify_commit
59        let deleted_output_keys = effects
60            .deleted()
61            .into_iter()
62            .chain(effects.transferred_from_consensus())
63            .chain(effects.consensus_owner_changed())
64            .map(|oref| (oref.0, oref.1))
65            .filter_map(|(id, seq)| {
66                self.input_objects
67                    .get(&id)
68                    .and_then(|obj| obj.is_consensus().then_some((obj.full_id(), seq)))
69            })
70            .map(|(full_id, seq)| InputKey::VersionedObject {
71                id: full_id,
72                version: seq,
73            });
74        output_keys.extend(deleted_output_keys);
75
76        // For any previously stream-ended consensus objects that appeared mutably in the transaction,
77        // synthesize a notification for the next version of the object.
78        let smeared_version = self.lamport_version;
79        let deleted_accessed_objects = effects.stream_ended_mutably_accessed_consensus_objects();
80        for object_id in deleted_accessed_objects.into_iter() {
81            let id = self
82                .input_objects
83                .get(&object_id)
84                .map(|obj| obj.full_id())
85                .unwrap_or_else(|| {
86                    let start_version = self.stream_ended_consensus_objects.get(&object_id)
87                        .expect("stream-ended object must be in either input_objects or stream_ended_consensus_objects");
88                    FullObjectID::new(object_id, Some(*start_version))
89                });
90            let key = InputKey::VersionedObject {
91                id,
92                version: smeared_version,
93            };
94            output_keys.push(key);
95        }
96
97        output_keys
98    }
99}

101pub struct TemporaryModuleResolver<'a, R> {
102    temp_store: &'a InnerTemporaryStore,
103    fallback: R,
104}
105
106impl<'a, R> TemporaryModuleResolver<'a, R> {
107    pub fn new(temp_store: &'a InnerTemporaryStore, fallback: R) -> Self {
108        Self {
109            temp_store,
110            fallback,
111        }
112    }
113}
114
115impl<R> GetModule for TemporaryModuleResolver<'_, R>
116where
117    R: GetModule<Item = Arc<CompiledModule>, Error = anyhow::Error>,
118{
119    type Error = anyhow::Error;
120    type Item = Arc<CompiledModule>;
121
122    fn get_module_by_id(&self, id: &ModuleId) -> anyhow::Result<Option<Self::Item>, Self::Error> {
123        let obj = self.temp_store.written.get(&ObjectID::from(*id.address()));
124        if let Some(o) = obj
125            && let Some(p) = o.data.try_as_package()
126        {
127            return Ok(Some(Arc::new(p.deserialize_module(
128                &id.name().into(),
129                &self.temp_store.binary_config,
130            )?)));
131        }
132        self.fallback.get_module_by_id(id)
133    }
134}

136impl BackingPackageStore for InnerTemporaryStore {
137    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
138        Ok(self
139            .written
140            .get(package_id)
141            .cloned()
142            .map(PackageObject::new))
143    }
144}

146pub struct PackageStoreWithFallback<P, F> {
147    primary: P,
148    fallback: F,
149}
150
151impl<P, F> PackageStoreWithFallback<P, F> {
152    pub fn new(primary: P, fallback: F) -> Self {
153        Self { primary, fallback }
154    }
155}
156
157impl<P, F> BackingPackageStore for PackageStoreWithFallback<P, F>
158where
159    P: BackingPackageStore,
160    F: BackingPackageStore,
161{
162    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
163        if let Some(package) = self.primary.get_package_object(package_id)? {
164            Ok(Some(package))
165        } else {
166            self.fallback.get_package_object(package_id)
167        }
168    }
