27#[derive(Debug, Clone)]
28pub struct InnerTemporaryStore {
29    pub input_objects: ObjectMap,
30    pub stream_ended_consensus_objects: BTreeMap<ObjectID, SequenceNumber /* start_version */>,
31    pub mutable_inputs: BTreeMap<ObjectID, (VersionDigest, Owner)>,
32    // All the written objects' sequence number should have been updated to the lamport version.
33    pub written: WrittenObjects,
34    pub loaded_runtime_objects: BTreeMap<ObjectID, DynamicallyLoadedObjectMetadata>,
35    pub events: TransactionEvents,
36    pub accumulator_events: Vec<AccumulatorEvent>,
37    pub binary_config: BinaryConfig,
38    pub runtime_packages_loaded_from_db: BTreeMap<ObjectID, PackageObject>,
39    pub lamport_version: SequenceNumber,
40    /// For each accumulator account, tracks the max running net withdraws during this transaction.
41    /// For instance, if the funds accumulator events looke like this for an account:
42    /// - Split(100)
43    /// - Merge(100)
44    /// - Split(100)
45    ///
46    /// Then the accumulator_running_max_withdraws for this account will be 100,
47    /// because at any given moment, the net withdraws is at most 100.
48    pub accumulator_running_max_withdraws: BTreeMap<AccumulatorObjId, u128>,
49}

51pub struct TemporaryModuleResolver<'a, R> {
52    temp_store: &'a InnerTemporaryStore,
53    fallback: R,
54}
55
56impl<'a, R> TemporaryModuleResolver<'a, R> {
57    pub fn new(temp_store: &'a InnerTemporaryStore, fallback: R) -> Self {
58        Self {
59            temp_store,
60            fallback,
61        }
62    }
63}
64
65impl<R> GetModule for TemporaryModuleResolver<'_, R>
66where
67    R: GetModule<Item = Arc<CompiledModule>, Error = anyhow::Error>,
68{
69    type Error = anyhow::Error;
70    type Item = Arc<CompiledModule>;
71
72    fn get_module_by_id(&self, id: &ModuleId) -> anyhow::Result<Option<Self::Item>, Self::Error> {
73        let obj = self.temp_store.written.get(&ObjectID::from(*id.address()));
74        if let Some(o) = obj
75            && let Some(p) = o.data.try_as_package()
76        {
77            return Ok(Some(Arc::new(p.deserialize_module(
78                &id.name().into(),
79                &self.temp_store.binary_config,
80            )?)));
81        }
82        self.fallback.get_module_by_id(id)
83    }
84}

86impl BackingPackageStore for InnerTemporaryStore {
87    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
88        Ok(self
89            .written
90            .get(package_id)
91            .cloned()
92            .map(PackageObject::new))
93    }
94}

96pub struct PackageStoreWithFallback<P, F> {
97    primary: P,
98    fallback: F,
99}
100
101impl<P, F> PackageStoreWithFallback<P, F> {
102    pub fn new(primary: P, fallback: F) -> Self {
103        Self { primary, fallback }
104    }
105}
106
107impl<P, F> BackingPackageStore for PackageStoreWithFallback<P, F>
108where
109    P: BackingPackageStore,
110    F: BackingPackageStore,
111{
112    fn get_package_object(&self, package_id: &ObjectID) -> SuiResult<Option<PackageObject>> {
113        if let Some(package) = self.primary.get_package_object(package_id)? {
114            Ok(Some(package))
115        } else {
116            self.fallback.get_package_object(package_id)
117        }
118    }
