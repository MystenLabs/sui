18// Cache up to this many verified certs. We will need to tune this number in the future - a decent
19// guess to start with is that it should be 10-20 times larger than peak transactions per second,
20// on the assumption that we should see most certs twice within about 10-20 seconds at most:
21// Once via RPC, once via consensus.
22const VERIFIED_CERTIFICATE_CACHE_SIZE: usize = 100_000;
23
24pub struct VerifiedDigestCache<D> {
25    inner: RwLock<LruCache<D, ()>>,
26    cache_hits_counter: IntCounter,
27    cache_misses_counter: IntCounter,
28    cache_evictions_counter: IntCounter,
29}
30
31impl<D: Hash + Eq + Copy> VerifiedDigestCache<D> {
32    pub fn new(
33        cache_hits_counter: IntCounter,
34        cache_misses_counter: IntCounter,
35        cache_evictions_counter: IntCounter,
36    ) -> Self {
37        Self {
38            inner: RwLock::new(LruCache::new(
39                std::num::NonZeroUsize::new(VERIFIED_CERTIFICATE_CACHE_SIZE).unwrap(),
40            )),
41            cache_hits_counter,
42            cache_misses_counter,
43            cache_evictions_counter,
44        }
45    }
46
47    pub fn is_cached(&self, digest: &D) -> bool {
48        let inner = self.inner.read();
49        if inner.contains(digest) {
50            self.cache_hits_counter.inc();
51            true
52        } else {
53            self.cache_misses_counter.inc();
54            false
55        }
56    }
57
58    pub fn cache_digest(&self, digest: D) {
59        let mut inner = self.inner.write();
60        if let Some(old) = inner.push(digest, ())
61            && old.0 != digest
62        {
63            self.cache_evictions_counter.inc();
64        }
65    }
66
67    pub fn cache_digests(&self, digests: Vec<D>) {
68        let mut inner = self.inner.write();
69        digests.into_iter().for_each(|d| {
70            if let Some(old) = inner.push(d, ())
71                && old.0 != d
72            {
73                self.cache_evictions_counter.inc();
74            }
75        });
76    }
77
78    pub fn is_verified<F, G>(&self, digest: D, verify_callback: F, uncached_checks: G) -> SuiResult
79    where
80        F: FnOnce() -> SuiResult,
81        G: FnOnce() -> SuiResult,
82    {
83        if !self.is_cached(&digest) {
84            verify_callback()?;
85            self.cache_digest(digest);
86        } else {
87            // Checks that are required to be performed outside the cache.
88            uncached_checks()?;
89        }
90        Ok(())
91    }
92
93    pub fn clear(&self) {
94        let mut inner = self.inner.write();
95        inner.clear();
96    }
97
98    // Initialize an empty cache when the cache is not needed (in testing scenarios, graphql and rosetta initialization).
99    pub fn new_empty() -> Self {
100        Self::new(
101            IntCounter::new("test_cache_hits", "test cache hits").unwrap(),
102            IntCounter::new("test_cache_misses", "test cache misses").unwrap(),
103            IntCounter::new("test_cache_evictions", "test cache evictions").unwrap(),
104        )
105    }
106}

108/// Does crypto validation for a transaction which may be user-provided, or may be from a checkpoint.
109pub fn verify_sender_signed_data_message_signatures(
110    txn: &SenderSignedData,
111    current_epoch: EpochId,
112    verify_params: &VerifyParams,
113    zklogin_inputs_cache: Arc<VerifiedDigestCache<ZKLoginInputsDigest>>,
114) -> SuiResult {
115    let intent_message = txn.intent_message();
116    assert_eq!(intent_message.intent, Intent::sui_transaction());
117
118    // 1. System transactions do not require signatures. User-submitted transactions are verified not to
119    // be system transactions before this point
120    if intent_message.value.is_system_tx() {
121        return Ok(());
122    }
123
124    // 2. One signature per signer is required.
125    let signers: NonEmpty<_> = txn.intent_message().value.signers();
126    fp_ensure!(
127        txn.inner().tx_signatures.len() == signers.len(),
128        SuiErrorKind::SignerSignatureNumberMismatch {
129            actual: txn.inner().tx_signatures.len(),
130            expected: signers.len()
131        }
132        .into()
133    );
134
135    // 3. Each signer must provide a signature.
136    let present_sigs = txn.get_signer_sig_mapping(verify_params.verify_legacy_zklogin_address)?;
137    for s in signers {
138        if !present_sigs.contains_key(&s) {
139            return Err(SuiErrorKind::SignerSignatureAbsent {
140                expected: s.to_string(),
141                actual: present_sigs.keys().map(|s| s.to_string()).collect(),
142            }
143            .into());
144        }
145    }
146
147    // 4. Every signature must be valid.
148    for (signer, signature) in present_sigs {
149        signature.verify_authenticator(
150            intent_message,
151            signer,
152            current_epoch,
153            verify_params,
154            zklogin_inputs_cache.clone(),
155        )?;
156    }
157    Ok(())
158}
