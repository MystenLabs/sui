64    /// Get the highest verified checkpint. This is the highest checkpoint summary that has been

161    // Fetch all checkpoint data
162    // TODO fix return type to not be anyhow
163    fn get_checkpoint_data(
164        &self,
165        checkpoint: VerifiedCheckpoint,
166        checkpoint_contents: CheckpointContents,
167    ) -> anyhow::Result<Checkpoint> {
168        use crate::effects::TransactionEffectsAPI;
169        use std::collections::HashMap;
170
171        let transaction_digests = checkpoint_contents
172            .iter()
173            .map(|execution_digests| execution_digests.transaction)
174            .collect::<Vec<_>>();
175        let txns = self
176            .multi_get_transactions(&transaction_digests)
177            .into_iter()
178            .map(|maybe_transaction| {
179                maybe_transaction.ok_or_else(|| anyhow::anyhow!("missing transaction"))
180            })
181            .collect::<anyhow::Result<Vec<_>>>()?;
182
183        let effects = self
184            .multi_get_transaction_effects(&transaction_digests)
185            .into_iter()
186            .map(|maybe_effects| maybe_effects.ok_or_else(|| anyhow::anyhow!("missing effects")))
187            .collect::<anyhow::Result<Vec<_>>>()?;
188
189        let event_tx_digests = effects
190            .iter()
191            .flat_map(|fx| fx.events_digest().map(|_| fx.transaction_digest()).copied())
192            .collect::<Vec<_>>();
193
194        let mut events = self
195            .multi_get_events(&event_tx_digests)
196            .into_iter()
197            .zip(event_tx_digests)
198            .map(|(maybe_event, tx_digest)| {
199                maybe_event
200                    .ok_or_else(|| anyhow::anyhow!("missing event for tx {tx_digest}"))
201                    .map(|event| (tx_digest, event))
202            })
203            .collect::<anyhow::Result<HashMap<_, _>>>()?;
204
205        let mut transactions = Vec::with_capacity(txns.len());
206        for (tx, fx) in txns.into_iter().zip(effects) {
207            let events = fx.events_digest().map(|_event_digest| {
208                events
209                    .remove(fx.transaction_digest())
210                    .expect("event was already checked to be present")
211            });
212
213            let transaction = ExecutedTransaction {
214                transaction: tx.transaction_data().clone(),
215                signatures: tx.tx_signatures().to_vec(),
216                effects: fx.clone(),
217                events,
218                unchanged_loaded_runtime_objects: self
219                    .get_unchanged_loaded_runtime_objects(tx.digest())
220                    //TODO Do we throw an error or just stub in an empty vector?
221                    .unwrap_or_default(),
222            };
223            transactions.push(transaction);
224        }
225
226        let object_set = {
227            let refs = transactions
228                .iter()
229                .flat_map(|tx| {
230                    crate::storage::get_transaction_object_set(
231                        &tx.transaction,
232                        &tx.effects,
233                        &tx.unchanged_loaded_runtime_objects,
234                    )
235                })
236                .collect::<BTreeSet<_>>()
237                .into_iter()
238                .collect::<Vec<_>>();
239
240            let objects = self.multi_get_objects_by_key(&refs);
241
242            let mut object_set = ObjectSet::default();
243            for (idx, object) in objects.into_iter().enumerate() {
244                object_set.insert(object.ok_or_else(|| {
245                    crate::storage::error::Error::custom(format!(
246                        "unabled to load object {:?}",
247                        refs[idx]
248                    ))
249                })?);
250            }
251            object_set
252        };
253
254        let checkpoint_data = Checkpoint {
255            summary: checkpoint.into(),
256            contents: checkpoint_contents,
257            transactions,
258            object_set,
259        };
260
261        Ok(checkpoint_data)
262    }
263}

598/// Trait used to provide functionality to the REST API service.
599///
600/// It extends both ObjectStore and ReadStore by adding functionality that may require more
601/// detailed underlying databases or indexes to support.
602pub trait RpcStateReader: ObjectStore + ReadStore + Send + Sync {
603    /// Lowest available checkpoint for which object data can be requested.
604    ///
605    /// Specifically this is the lowest checkpoint for which input/output object data will be
606    /// available.
607    fn get_lowest_available_checkpoint_objects(&self) -> Result<CheckpointSequenceNumber>;
608
609    fn get_chain_identifier(&self) -> Result<ChainIdentifier>;
610
611    // Get a handle to an instance of the RpcIndexes
612    fn indexes(&self) -> Option<&dyn RpcIndexes>;
613
614    fn get_type_layout(&self, type_tag: &TypeTag) -> Result<Option<MoveTypeLayout>> {
615        match type_tag {
616            TypeTag::Bool => Ok(Some(MoveTypeLayout::Bool)),
617            TypeTag::U8 => Ok(Some(MoveTypeLayout::U8)),
618            TypeTag::U64 => Ok(Some(MoveTypeLayout::U64)),
619            TypeTag::U128 => Ok(Some(MoveTypeLayout::U128)),
620            TypeTag::Address => Ok(Some(MoveTypeLayout::Address)),
621            TypeTag::Signer => Ok(Some(MoveTypeLayout::Signer)),
622            TypeTag::Vector(type_tag) => Ok(self
623                .get_type_layout(type_tag)?
624                .map(|layout| MoveTypeLayout::Vector(Box::new(layout)))),
625            TypeTag::Struct(struct_tag) => self.get_struct_layout(struct_tag),
626            TypeTag::U16 => Ok(Some(MoveTypeLayout::U16)),
627            TypeTag::U32 => Ok(Some(MoveTypeLayout::U32)),
628            TypeTag::U256 => Ok(Some(MoveTypeLayout::U256)),
629        }
630    }
631    fn get_struct_layout(&self, type_tag: &StructTag) -> Result<Option<MoveTypeLayout>>;
632}

734pub struct TransactionInfo {
735    pub checkpoint: u64,
736    pub balance_changes: Vec<BalanceChange>,
737    pub object_types: HashMap<ObjectID, ObjectType>,
738}
739
740impl TransactionInfo {
741    pub fn new(
742        _transaction: &TransactionData,
743        effects: &TransactionEffects,
744        input_objects: &[Object],
745        output_objects: &[Object],
746        checkpoint: u64,
747    ) -> TransactionInfo {
748        let balance_changes = derive_balance_changes(effects, input_objects, output_objects);
749
750        let object_types = input_objects
751            .iter()
752            .chain(output_objects)
753            .map(|object| (object.id(), ObjectType::from(object)))
754            .collect();
755
756        TransactionInfo {
757            checkpoint,
758            balance_changes,
759            object_types,
760        }
761    }
762}
