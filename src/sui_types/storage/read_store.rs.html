36pub trait ReadStore: ObjectStore {
37    //
38    // Committee Getters
39    //

41    fn get_committee(&self, epoch: EpochId) -> Option<Arc<Committee>>;
42
43    //
44    // Checkpoint Getters
45    //
46
47    /// Get the latest available checkpoint. This is the latest executed checkpoint.
48    ///
49    /// All transactions, effects, objects and events are guaranteed to be available for the
50    /// returned checkpoint.
51    fn get_latest_checkpoint(&self) -> Result<VerifiedCheckpoint>;
52
53    /// Get the latest available checkpoint sequence number. This is the sequence number of the latest executed checkpoint.
54    fn get_latest_checkpoint_sequence_number(&self) -> Result<CheckpointSequenceNumber> {
55        let latest_checkpoint = self.get_latest_checkpoint()?;
56        Ok(*latest_checkpoint.sequence_number())
57    }
58
59    /// Get the epoch of the latest checkpoint
60    fn get_latest_epoch_id(&self) -> Result<EpochId> {
61        let latest_checkpoint = self.get_latest_checkpoint()?;
62        Ok(latest_checkpoint.epoch())
63    }
64
65    /// Get the highest verified checkpint. This is the highest checkpoint summary that has been
66    /// verified, generally by state-sync. Only the checkpoint header is guaranteed to be present in
67    /// the store.
68    fn get_highest_verified_checkpoint(&self) -> Result<VerifiedCheckpoint>;
69
70    /// Get the highest synced checkpint. This is the highest checkpoint that has been synced from
71    /// state-synce. The checkpoint header, contents, transactions, and effects of this checkpoint
72    /// are guaranteed to be present in the store
73    fn get_highest_synced_checkpoint(&self) -> Result<VerifiedCheckpoint>;
74
75    /// Lowest available checkpoint for which transaction and checkpoint data can be requested.
76    ///
77    /// Specifically this is the lowest checkpoint for which the following data can be requested:
78    ///  - checkpoints
79    ///  - transactions
80    ///  - effects
81    ///  - events

158    /// "full" checkpoints include: header, contents, transactions, effects.
159    /// sequence_number is optional since we can always query it using the digest.
160    /// However if it is provided, we can avoid an extra db lookup.

167    // Fetch all checkpoint data
168    fn get_checkpoint_data(
169        &self,
170        checkpoint: VerifiedCheckpoint,
171        checkpoint_contents: CheckpointContents,
172    ) -> Result<Checkpoint> {
173        use crate::effects::TransactionEffectsAPI;
174        use crate::storage::error::Error;
175        use std::collections::HashMap;
176
177        let transaction_digests = checkpoint_contents
178            .iter()
179            .map(|execution_digests| execution_digests.transaction)
180            .collect::<Vec<_>>();
181        let txns = self
182            .multi_get_transactions(&transaction_digests)
183            .into_iter()
184            .map(|maybe_transaction| {
185                maybe_transaction.ok_or_else(|| Error::missing("missing transaction"))
186            })
187            .collect::<Result<Vec<_>>>()?;
188
189        let effects = self
190            .multi_get_transaction_effects(&transaction_digests)
191            .into_iter()
192            .map(|maybe_effects| maybe_effects.ok_or_else(|| Error::missing("missing effects")))
193            .collect::<Result<Vec<_>>>()?;
194
195        let event_tx_digests = effects
196            .iter()
197            .flat_map(|fx| fx.events_digest().map(|_| fx.transaction_digest()).copied())
198            .collect::<Vec<_>>();
199
200        let mut events = self
201            .multi_get_events(&event_tx_digests)
202            .into_iter()
203            .zip(event_tx_digests)
204            .map(|(maybe_event, tx_digest)| {
205                maybe_event
206                    .ok_or_else(|| Error::missing(format!("missing event for tx {tx_digest}")))
207                    .map(|event| (tx_digest, event))
208            })
209            .collect::<Result<HashMap<_, _>>>()?;
210
211        let mut transactions = Vec::with_capacity(txns.len());
212        for (tx, fx) in txns.into_iter().zip(effects) {
213            let events = fx.events_digest().map(|_event_digest| {
214                events
215                    .remove(fx.transaction_digest())
216                    .expect("event was already checked to be present")
217            });
218
219            let transaction = ExecutedTransaction {
220                transaction: tx.transaction_data().clone(),
221                signatures: tx.tx_signatures().to_vec(),
222                effects: fx.clone(),
223                events,
224                unchanged_loaded_runtime_objects: self
225                    .get_unchanged_loaded_runtime_objects(tx.digest())
226                    //TODO Do we throw an error or just stub in an empty vector?
227                    .unwrap_or_default(),
228            };
229            transactions.push(transaction);
230        }
231
232        let object_set = {
233            let refs = transactions
234                .iter()
235                .flat_map(|tx| {
236                    crate::storage::get_transaction_object_set(
237                        &tx.transaction,
238                        &tx.effects,
239                        &tx.unchanged_loaded_runtime_objects,
240                    )
241                })
242                .collect::<BTreeSet<_>>()
243                .into_iter()
244                .collect::<Vec<_>>();
245
246            let objects = self.multi_get_objects_by_key(&refs);
247
248            let mut object_set = ObjectSet::default();
249            for (idx, object) in objects.into_iter().enumerate() {
250                object_set.insert(object.ok_or_else(|| {
251                    Error::missing(format!("unable to load object {:?}", refs[idx]))
252                })?);
253            }
254            object_set
255        };
256
257        let checkpoint_data = Checkpoint {
258            summary: checkpoint.into(),
259            contents: checkpoint_contents,
260            transactions,
261            object_set,
262        };
263
264        Ok(checkpoint_data)
265    }
266}

622/// Trait used to provide functionality to the REST API service.
623///
624/// It extends both ObjectStore and ReadStore by adding functionality that may require more
625/// detailed underlying databases or indexes to support.
626pub trait RpcStateReader: ObjectStore + ReadStore + Send + Sync {
627    /// Lowest available checkpoint for which object data can be requested.
628    ///
629    /// Specifically this is the lowest checkpoint for which input/output object data will be
630    /// available.
631    fn get_lowest_available_checkpoint_objects(&self) -> Result<CheckpointSequenceNumber>;
632
633    fn get_chain_identifier(&self) -> Result<ChainIdentifier>;
634
635    // Get a handle to an instance of the RpcIndexes
636    fn indexes(&self) -> Option<&dyn RpcIndexes>;
637
638    fn get_type_layout(&self, type_tag: &TypeTag) -> Result<Option<MoveTypeLayout>> {
639        match type_tag {
640            TypeTag::Bool => Ok(Some(MoveTypeLayout::Bool)),
641            TypeTag::U8 => Ok(Some(MoveTypeLayout::U8)),
642            TypeTag::U64 => Ok(Some(MoveTypeLayout::U64)),
643            TypeTag::U128 => Ok(Some(MoveTypeLayout::U128)),
644            TypeTag::Address => Ok(Some(MoveTypeLayout::Address)),
645            TypeTag::Signer => Ok(Some(MoveTypeLayout::Signer)),
646            TypeTag::Vector(type_tag) => Ok(self
647                .get_type_layout(type_tag)?
648                .map(|layout| MoveTypeLayout::Vector(Box::new(layout)))),
649            TypeTag::Struct(struct_tag) => self.get_struct_layout(struct_tag),
650            TypeTag::U16 => Ok(Some(MoveTypeLayout::U16)),
651            TypeTag::U32 => Ok(Some(MoveTypeLayout::U32)),
652            TypeTag::U256 => Ok(Some(MoveTypeLayout::U256)),
653        }
654    }
655    fn get_struct_layout(&self, type_tag: &StructTag) -> Result<Option<MoveTypeLayout>>;
656}

758    pub checkpoint: u64,
759    pub balance_changes: Vec<BalanceChange>,
760    pub object_types: HashMap<ObjectID, ObjectType>,
761}
762
763impl TransactionInfo {
764    pub fn new(
765        _transaction: &TransactionData,
766        effects: &TransactionEffects,
767        input_objects: &[Object],
768        output_objects: &[Object],
769        checkpoint: u64,
770    ) -> TransactionInfo {
771        let balance_changes = derive_balance_changes(effects, input_objects, output_objects);
772
773        let object_types = input_objects
774            .iter()
775            .chain(output_objects)
776            .map(|object| (object.id(), ObjectType::from(object)))
777            .collect();
778
779        TransactionInfo {
780            checkpoint,
781            balance_changes,
782            object_types,
783        }
784    }
785}
