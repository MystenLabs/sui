35/// This type captures the minimum amount of information from SuiSystemState needed by a validator
36/// to run the protocol. This allows us to decouple from the actual SuiSystemState type, and hence
37/// do not need to evolve it when we upgrade the SuiSystemState type.
38/// Evolving EpochStartSystemState is also a lot easier in that we could add optional fields
39/// and fill them with None for older versions. When we absolutely must delete fields, we could
40/// also add new db tables to store the new version. This is OK because we only store one copy of
41/// this as part of EpochStartConfiguration for the most recent epoch in the db.
42#[derive(Serialize, Deserialize, Debug, Eq, PartialEq)]
43#[enum_dispatch(EpochStartSystemStateTrait)]
44pub enum EpochStartSystemState {
45    V1(EpochStartSystemStateV1),
46}
47
48impl EpochStartSystemState {
49    pub fn new_v1(
50        epoch: EpochId,
51        protocol_version: u64,
52        reference_gas_price: u64,
53        safe_mode: bool,
54        epoch_start_timestamp_ms: u64,
55        epoch_duration_ms: u64,
56        active_validators: Vec<EpochStartValidatorInfoV1>,
57    ) -> Self {
58        Self::V1(EpochStartSystemStateV1 {
59            epoch,
60            protocol_version,
61            reference_gas_price,
62            safe_mode,
63            epoch_start_timestamp_ms,
64            epoch_duration_ms,
65            active_validators,
66        })
67    }
68
69    pub fn new_for_testing_with_epoch(epoch: EpochId) -> Self {
70        Self::V1(EpochStartSystemStateV1::new_for_testing_with_epoch(epoch))
71    }
72
73    pub fn new_at_next_epoch_for_testing(&self) -> Self {
74        // Only need to support the latest version for testing.
75        match self {
76            Self::V1(state) => Self::V1(EpochStartSystemStateV1 {
77                epoch: state.epoch + 1,
78                protocol_version: state.protocol_version,
79                reference_gas_price: state.reference_gas_price,
80                safe_mode: state.safe_mode,
81                epoch_start_timestamp_ms: state.epoch_start_timestamp_ms,
82                epoch_duration_ms: state.epoch_duration_ms,
83                active_validators: state.active_validators.clone(),
84            }),
85        }
86    }
87}

118impl EpochStartSystemStateTrait for EpochStartSystemStateV1 {
119    fn epoch(&self) -> EpochId {
120        self.epoch
121    }
122
123    fn protocol_version(&self) -> ProtocolVersion {
124        ProtocolVersion::new(self.protocol_version)
125    }
126
127    fn reference_gas_price(&self) -> u64 {
128        self.reference_gas_price
129    }
130
131    fn safe_mode(&self) -> bool {
132        self.safe_mode
133    }
134
135    fn epoch_start_timestamp_ms(&self) -> u64 {
136        self.epoch_start_timestamp_ms
137    }
138
139    fn epoch_duration_ms(&self) -> u64 {
140        self.epoch_duration_ms
141    }
142
143    fn get_validator_addresses(&self) -> Vec<SuiAddress> {
144        self.active_validators
145            .iter()
146            .map(|validator| validator.sui_address)
147            .collect()
148    }
149
150    fn get_sui_committee_with_network_metadata(&self) -> CommitteeWithNetworkMetadata {
151        let validators = self
152            .active_validators
153            .iter()
154            .map(|validator| {
155                (
156                    validator.authority_name(),
157                    (
158                        validator.voting_power,
159                        NetworkMetadata {
160                            network_address: validator.sui_net_address.clone(),
161                            narwhal_primary_address: validator.narwhal_primary_address.clone(),
162                            network_public_key: Some(validator.narwhal_network_pubkey.clone()),
163                        },
164                    ),
165                )
166            })
167            .collect();
168
169        CommitteeWithNetworkMetadata::new(self.epoch, validators)
170    }
171
172    fn get_sui_committee(&self) -> Committee {
173        let voting_rights = self
174            .active_validators
175            .iter()
176            .map(|validator| (validator.authority_name(), validator.voting_power))
177            .collect();
178        Committee::new(self.epoch, voting_rights)
179    }
180
181    fn get_consensus_committee(&self) -> ConsensusCommittee {
182        let mut authorities = vec![];
183        for validator in self.active_validators.iter() {
184            authorities.push(Authority {
185                stake: validator.voting_power as consensus_config::Stake,
186                // TODO(mysticeti): Add EpochStartValidatorInfoV2 with new field for mysticeti address.
187                address: validator.narwhal_primary_address.clone(),
188                hostname: validator.hostname.clone(),
189                authority_key: consensus_config::AuthorityPublicKey::new(
190                    validator.protocol_pubkey.clone(),
191                ),
192                protocol_key: consensus_config::ProtocolPublicKey::new(
193                    validator.narwhal_worker_pubkey.clone(),
194                ),
195                network_key: consensus_config::NetworkPublicKey::new(
196                    validator.narwhal_network_pubkey.clone(),
197                ),
198            });
199        }
200
201        // Sort the authorities by their protocol (public) key in ascending order, same as the order
202        // in the Sui committee returned from get_sui_committee().
203        authorities.sort_by(|a1, a2| a1.authority_key.cmp(&a2.authority_key));
204
205        for ((i, mysticeti_authority), sui_authority_name) in authorities
206            .iter()
207            .enumerate()
208            .zip(self.get_sui_committee().names())
209        {
210            if sui_authority_name.0 != mysticeti_authority.authority_key.to_bytes() {
211                error!(
212                    "Mismatched authority order between Sui and Mysticeti! Index {}, Mysticeti authority {:?}\nSui authority name {}",
213                    i, mysticeti_authority, sui_authority_name
214                );
215            }
216        }
217
218        ConsensusCommittee::new(self.epoch as consensus_config::Epoch, authorities)
219    }
220
221    fn get_validator_as_p2p_peers(
222        &self,
223        excluding_self: AuthorityName,
224    ) -> Vec<(PeerId, Multiaddr)> {
225        self.active_validators
226            .iter()
227            .filter(|validator| validator.authority_name() != excluding_self)
228            .map(|validator| {
229                let peer_id = PeerId(validator.narwhal_network_pubkey.0.to_bytes());
230                (peer_id, validator.p2p_address.clone())
231            })
232            .collect()
233    }
234
235    fn get_authority_names_to_peer_ids(&self) -> HashMap<AuthorityName, PeerId> {
236        self.active_validators
237            .iter()
238            .map(|validator| {
239                let name = validator.authority_name();
240                let peer_id = PeerId(validator.narwhal_network_pubkey.0.to_bytes());
241
242                (name, peer_id)
243            })
244            .collect()
245    }
246
247    fn get_authority_names_to_hostnames(&self) -> HashMap<AuthorityName, String> {
248        self.active_validators
249            .iter()
250            .map(|validator| {
251                let name = validator.authority_name();
252                let hostname = validator.hostname.clone();
253
254                (name, hostname)
255            })
256            .collect()
257    }
258}

280#[cfg(test)]
281mod test {
282    use crate::base_types::SuiAddress;
283    use crate::committee::CommitteeTrait;
284    use crate::crypto::{AuthorityKeyPair, NetworkKeyPair, get_key_pair};
285    use crate::sui_system_state::epoch_start_sui_system_state::{
286        EpochStartSystemStateTrait, EpochStartSystemStateV1, EpochStartValidatorInfoV1,
287    };
288    use fastcrypto::traits::KeyPair;
289    use mysten_network::Multiaddr;
290    use rand::thread_rng;
291    use sui_protocol_config::ProtocolVersion;
292
293    #[test]
294    fn test_sui_and_mysticeti_committee_are_same() {
295        // GIVEN
296        let mut active_validators = vec![];
297
298        for i in 0..10 {
299            let (sui_address, protocol_key): (SuiAddress, AuthorityKeyPair) = get_key_pair();
300            let narwhal_network_key = NetworkKeyPair::generate(&mut thread_rng());
301
302            active_validators.push(EpochStartValidatorInfoV1 {
303                sui_address,
304                protocol_pubkey: protocol_key.public().clone(),
305                narwhal_network_pubkey: narwhal_network_key.public().clone(),
306                narwhal_worker_pubkey: narwhal_network_key.public().clone(),
307                sui_net_address: Multiaddr::empty(),
308                p2p_address: Multiaddr::empty(),
309                narwhal_primary_address: Multiaddr::empty(),
310                narwhal_worker_address: Multiaddr::empty(),
311                voting_power: 1_000,
312                hostname: format!("host-{i}").to_string(),
313            })
314        }
315
316        let state = EpochStartSystemStateV1 {
317            epoch: 10,
318            protocol_version: ProtocolVersion::MAX.as_u64(),
319            reference_gas_price: 0,
320            safe_mode: false,
321            epoch_start_timestamp_ms: 0,
322            epoch_duration_ms: 0,
323            active_validators,
324        };
325
326        // WHEN
327        let sui_committee = state.get_sui_committee();
328        let consensus_committee = state.get_consensus_committee();
329
330        // THEN
331        // assert the validators details
332        assert_eq!(sui_committee.num_members(), 10);
333        assert_eq!(sui_committee.num_members(), consensus_committee.size());
334        assert_eq!(
335            sui_committee.validity_threshold(),
336            consensus_committee.validity_threshold()
337        );
338        assert_eq!(
339            sui_committee.quorum_threshold(),
340            consensus_committee.quorum_threshold()
341        );
342        assert_eq!(state.epoch, consensus_committee.epoch());
343
344        for (authority_index, consensus_authority) in consensus_committee.authorities() {
345            let sui_authority_name = sui_committee
346                .authority_by_index(authority_index.value() as u32)
347                .unwrap();
348
349            assert_eq!(
350                consensus_authority.authority_key.to_bytes(),
351                sui_authority_name.0,
352                "Mysten & SUI committee member of same index correspond to different public key"
353            );
354            assert_eq!(
355                consensus_authority.stake,
356                sui_committee.weight(sui_authority_name),
357                "Mysten & SUI committee member stake differs"
358            );
359        }
360    }
