123impl ValidatorMetadataV1 {
124    /// Verify validator metadata and return a verified version (on success) or error code (on failure)
125    pub fn verify(&self, metadata_v2: bool) -> Result<VerifiedValidatorMetadataV1, u64> {
126        let protocol_pubkey = AuthorityPublicKey::from_bytes(self.protocol_pubkey_bytes.as_ref())
127            .map_err(|_| E_METADATA_INVALID_PUBKEY)?;
128
129        // Verify proof of possession for the protocol key
130        let pop = AuthoritySignature::from_bytes(self.proof_of_possession_bytes.as_ref())
131            .map_err(|_| E_METADATA_INVALID_POP)?;
132        verify_proof_of_possession(&pop, &protocol_pubkey, self.sui_address)
133            .map_err(|_| E_METADATA_INVALID_POP)?;
134
135        let network_pubkey = NetworkPublicKey::from_bytes(self.network_pubkey_bytes.as_ref())
136            .map_err(|_| E_METADATA_INVALID_NET_PUBKEY)?;
137        let worker_pubkey = NetworkPublicKey::from_bytes(self.worker_pubkey_bytes.as_ref())
138            .map_err(|_| E_METADATA_INVALID_WORKER_PUBKEY)?;
139        if worker_pubkey == network_pubkey {
140            return Err(E_METADATA_INVALID_WORKER_PUBKEY);
141        }
142
143        let net_address = Multiaddr::try_from(self.net_address.clone())
144            .map_err(|_| E_METADATA_INVALID_NET_ADDR)?;
145
146        // Ensure p2p, primary, and worker addresses are both Multiaddr's and valid anemo addresses
147        let p2p_address = Multiaddr::try_from(self.p2p_address.clone())
148            .map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
149        p2p_address
150            .to_anemo_address()
151            .map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
152
153        let primary_address = Multiaddr::try_from(self.primary_address.clone())
154            .map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
155        primary_address
156            .to_anemo_address()
157            .map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
158
159        let worker_address = Multiaddr::try_from(self.worker_address.clone())
160            .map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
161        worker_address
162            .to_anemo_address()
163            .map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
164
165        let next_epoch_protocol_pubkey = match self.next_epoch_protocol_pubkey_bytes.clone() {
166            None => Ok::<Option<AuthorityPublicKey>, u64>(None),
167            Some(bytes) => Ok(Some(
168                AuthorityPublicKey::from_bytes(bytes.as_ref())
169                    .map_err(|_| E_METADATA_INVALID_PUBKEY)?,
170            )),
171        }?;
172
173        let next_epoch_pop = match self.next_epoch_proof_of_possession.clone() {
174            None => Ok::<Option<AuthoritySignature>, u64>(None),
175            Some(bytes) => Ok(Some(
176                AuthoritySignature::from_bytes(bytes.as_ref())
177                    .map_err(|_| E_METADATA_INVALID_POP)?,
178            )),
179        }?;
180        // Verify proof of possession for the next epoch protocol key
181        if let Some(ref next_epoch_protocol_pubkey) = next_epoch_protocol_pubkey {
182            match next_epoch_pop {
183                Some(next_epoch_pop) => {
184                    verify_proof_of_possession(
185                        &next_epoch_pop,
186                        next_epoch_protocol_pubkey,
187                        self.sui_address,
188                    )
189                    .map_err(|_| E_METADATA_INVALID_POP)?;
190                }
191                None => {
192                    return Err(E_METADATA_INVALID_POP);
193                }
194            }
195        }
196
197        let next_epoch_network_pubkey = match self.next_epoch_network_pubkey_bytes.clone() {
198            None => Ok::<Option<NetworkPublicKey>, u64>(None),
199            Some(bytes) => Ok(Some(
200                NetworkPublicKey::from_bytes(bytes.as_ref())
201                    .map_err(|_| E_METADATA_INVALID_NET_PUBKEY)?,
202            )),
203        }?;
204
205        let next_epoch_worker_pubkey: Option<NetworkPublicKey> =
206            match self.next_epoch_worker_pubkey_bytes.clone() {
207                None => Ok::<Option<NetworkPublicKey>, u64>(None),
208                Some(bytes) => Ok(Some(
209                    NetworkPublicKey::from_bytes(bytes.as_ref())
210                        .map_err(|_| E_METADATA_INVALID_WORKER_PUBKEY)?,
211                )),
212            }?;
213        if metadata_v2 {
214            match (&next_epoch_network_pubkey, &next_epoch_worker_pubkey) {
215                (Some(next_epoch_network_pubkey), Some(next_epoch_worker_pubkey)) => {
216                    if next_epoch_network_pubkey == next_epoch_worker_pubkey {
217                        return Err(E_METADATA_INVALID_WORKER_PUBKEY);
218                    }
219                }
220                (Some(next_epoch_network_pubkey), None) => {
221                    if next_epoch_network_pubkey == &worker_pubkey {
222                        return Err(E_METADATA_INVALID_NET_PUBKEY);
223                    }
224                }
225                (None, Some(next_epoch_worker_pubkey)) => {
226                    if next_epoch_worker_pubkey == &network_pubkey {
227                        return Err(E_METADATA_INVALID_WORKER_PUBKEY);
228                    }
229                }
230                (None, None) => (),
231            }
232        } else {
233            // TODO: Clean up this after metadata v2 is fully deployed.
234            if next_epoch_network_pubkey.is_some()
235                && next_epoch_network_pubkey == next_epoch_worker_pubkey
236            {
237                return Err(E_METADATA_INVALID_WORKER_PUBKEY);
238            }
239        }
240
241        let next_epoch_net_address = match self.next_epoch_net_address.clone() {
242            None => Ok::<Option<Multiaddr>, u64>(None),
243            Some(address) => Ok(Some(
244                Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_NET_ADDR)?,
245            )),
246        }?;
247
248        let next_epoch_p2p_address = match self.next_epoch_p2p_address.clone() {
249            None => Ok::<Option<Multiaddr>, u64>(None),
250            Some(address) => {
251                let address =
252                    Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
253                address
254                    .to_anemo_address()
255                    .map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
256
257                Ok(Some(address))
258            }
259        }?;
260
261        let next_epoch_primary_address = match self.next_epoch_primary_address.clone() {
262            None => Ok::<Option<Multiaddr>, u64>(None),
263            Some(address) => {
264                let address =
265                    Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
266                address
267                    .to_anemo_address()
268                    .map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
269
270                Ok(Some(address))
271            }
272        }?;
273
274        let next_epoch_worker_address = match self.next_epoch_worker_address.clone() {
275            None => Ok::<Option<Multiaddr>, u64>(None),
276            Some(address) => {
277                let address =
278                    Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
279                address
280                    .to_anemo_address()
281                    .map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
282
283                Ok(Some(address))
284            }
285        }?;
286
287        Ok(VerifiedValidatorMetadataV1 {
288            sui_address: self.sui_address,
289            protocol_pubkey,
290            network_pubkey,
291            worker_pubkey,
292            proof_of_possession_bytes: self.proof_of_possession_bytes.clone(),
293            name: self.name.clone(),
294            description: self.description.clone(),
295            image_url: self.image_url.clone(),
296            project_url: self.project_url.clone(),
297            net_address,
298            p2p_address,
299            primary_address,
300            worker_address,
301            next_epoch_protocol_pubkey,
302            next_epoch_proof_of_possession: self.next_epoch_proof_of_possession.clone(),
303            next_epoch_network_pubkey,
304            next_epoch_worker_pubkey,
305            next_epoch_net_address,
306            next_epoch_p2p_address,
307            next_epoch_primary_address,
308            next_epoch_worker_address,
309        })
310    }
311}

562    fn advance_epoch_safe_mode(&mut self, params: &AdvanceEpochParams) {
563        self.epoch = params.epoch;

566            .deposit_for_safe_mode(params.storage_charge);
567        self.safe_mode_storage_rebates += params.storage_rebate;
568        self.safe_mode_computation_rewards
569            .deposit_for_safe_mode(params.computation_charge);
570        self.safe_mode_non_refundable_storage_fee += params.non_refundable_storage_fee;
571        self.epoch_start_timestamp_ms = params.epoch_start_timestamp_ms;
572        self.protocol_version = params.next_protocol_version.as_u64();
573    }

575    fn get_current_epoch_committee(&self) -> CommitteeWithNetworkMetadata {
576        let validators = self
577            .validators
578            .active_validators
579            .iter()
580            .map(|validator| {
581                let verified_metadata = validator.verified_metadata();
582                let name = verified_metadata.sui_pubkey_bytes();
583                (
584                    name,
585                    (
586                        validator.voting_power,
587                        NetworkMetadata {
588                            network_address: verified_metadata.net_address.clone(),
589                            narwhal_primary_address: verified_metadata.primary_address.clone(),
590                            network_public_key: Some(verified_metadata.network_pubkey.clone()),
591                        },
592                    ),
593                )
594            })
595            .collect();
596        CommitteeWithNetworkMetadata::new(self.epoch, validators)
597    }

603        let table_id = self.validators.pending_active_validators.contents.id;
604        let table_size = self.validators.pending_active_validators.contents.size;
605        let validators: Vec<ValidatorV1> =
606            get_validators_from_table_vec(object_store, table_id, table_size)?;
607        Ok(validators
608            .into_iter()
609            .map(|v| v.into_sui_validator_summary())
610            .collect())
611    }
