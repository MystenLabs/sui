123impl ValidatorMetadataV1 {
124    /// Verify validator metadata and return a verified version (on success) or error code (on failure)
125    pub fn verify(&self) -> Result<VerifiedValidatorMetadataV1, u64> {
126        let protocol_pubkey = AuthorityPublicKey::from_bytes(self.protocol_pubkey_bytes.as_ref())
127            .map_err(|_| E_METADATA_INVALID_PUBKEY)?;
128
129        // Verify proof of possession for the protocol key
130        let pop = AuthoritySignature::from_bytes(self.proof_of_possession_bytes.as_ref())
131            .map_err(|_| E_METADATA_INVALID_POP)?;
132        verify_proof_of_possession(&pop, &protocol_pubkey, self.sui_address)
133            .map_err(|_| E_METADATA_INVALID_POP)?;
134
135        let network_pubkey = NetworkPublicKey::from_bytes(self.network_pubkey_bytes.as_ref())
136            .map_err(|_| E_METADATA_INVALID_NET_PUBKEY)?;
137        let worker_pubkey = NetworkPublicKey::from_bytes(self.worker_pubkey_bytes.as_ref())
138            .map_err(|_| E_METADATA_INVALID_WORKER_PUBKEY)?;
139        if worker_pubkey == network_pubkey {
140            return Err(E_METADATA_INVALID_WORKER_PUBKEY);
141        }
142
143        let net_address = Multiaddr::try_from(self.net_address.clone())
144            .map_err(|_| E_METADATA_INVALID_NET_ADDR)?;
145
146        // Ensure p2p, primary, and worker addresses are both Multiaddr's and valid anemo addresses
147        let p2p_address = Multiaddr::try_from(self.p2p_address.clone())
148            .map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
149        p2p_address
150            .to_anemo_address()
151            .map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
152
153        let primary_address = Multiaddr::try_from(self.primary_address.clone())
154            .map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
155        primary_address
156            .to_anemo_address()
157            .map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
158
159        let worker_address = Multiaddr::try_from(self.worker_address.clone())
160            .map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
161        worker_address
162            .to_anemo_address()
163            .map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
164
165        let next_epoch_protocol_pubkey = match self.next_epoch_protocol_pubkey_bytes.clone() {
166            None => Ok::<Option<AuthorityPublicKey>, u64>(None),
167            Some(bytes) => Ok(Some(
168                AuthorityPublicKey::from_bytes(bytes.as_ref())
169                    .map_err(|_| E_METADATA_INVALID_PUBKEY)?,
170            )),
171        }?;
172
173        let next_epoch_pop = match self.next_epoch_proof_of_possession.clone() {
174            None => Ok::<Option<AuthoritySignature>, u64>(None),
175            Some(bytes) => Ok(Some(
176                AuthoritySignature::from_bytes(bytes.as_ref())
177                    .map_err(|_| E_METADATA_INVALID_POP)?,
178            )),
179        }?;
180        // Verify proof of possession for the next epoch protocol key
181        if let Some(ref next_epoch_protocol_pubkey) = next_epoch_protocol_pubkey {
182            match next_epoch_pop {
183                Some(next_epoch_pop) => {
184                    verify_proof_of_possession(
185                        &next_epoch_pop,
186                        next_epoch_protocol_pubkey,
187                        self.sui_address,
188                    )
189                    .map_err(|_| E_METADATA_INVALID_POP)?;
190                }
191                None => {
192                    return Err(E_METADATA_INVALID_POP);
193                }
194            }
195        }
196
197        let next_epoch_network_pubkey = match self.next_epoch_network_pubkey_bytes.clone() {
198            None => Ok::<Option<NetworkPublicKey>, u64>(None),
199            Some(bytes) => Ok(Some(
200                NetworkPublicKey::from_bytes(bytes.as_ref())
201                    .map_err(|_| E_METADATA_INVALID_NET_PUBKEY)?,
202            )),
203        }?;
204
205        let next_epoch_worker_pubkey: Option<NetworkPublicKey> =
206            match self.next_epoch_worker_pubkey_bytes.clone() {
207                None => Ok::<Option<NetworkPublicKey>, u64>(None),
208                Some(bytes) => Ok(Some(
209                    NetworkPublicKey::from_bytes(bytes.as_ref())
210                        .map_err(|_| E_METADATA_INVALID_WORKER_PUBKEY)?,
211                )),
212            }?;
213        if next_epoch_network_pubkey.is_some()
214            && next_epoch_network_pubkey == next_epoch_worker_pubkey
215        {
216            return Err(E_METADATA_INVALID_WORKER_PUBKEY);
217        }
218
219        let next_epoch_net_address = match self.next_epoch_net_address.clone() {
220            None => Ok::<Option<Multiaddr>, u64>(None),
221            Some(address) => Ok(Some(
222                Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_NET_ADDR)?,
223            )),
224        }?;
225
226        let next_epoch_p2p_address = match self.next_epoch_p2p_address.clone() {
227            None => Ok::<Option<Multiaddr>, u64>(None),
228            Some(address) => {
229                let address =
230                    Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
231                address
232                    .to_anemo_address()
233                    .map_err(|_| E_METADATA_INVALID_P2P_ADDR)?;
234
235                Ok(Some(address))
236            }
237        }?;
238
239        let next_epoch_primary_address = match self.next_epoch_primary_address.clone() {
240            None => Ok::<Option<Multiaddr>, u64>(None),
241            Some(address) => {
242                let address =
243                    Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
244                address
245                    .to_anemo_address()
246                    .map_err(|_| E_METADATA_INVALID_PRIMARY_ADDR)?;
247
248                Ok(Some(address))
249            }
250        }?;
251
252        let next_epoch_worker_address = match self.next_epoch_worker_address.clone() {
253            None => Ok::<Option<Multiaddr>, u64>(None),
254            Some(address) => {
255                let address =
256                    Multiaddr::try_from(address).map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
257                address
258                    .to_anemo_address()
259                    .map_err(|_| E_METADATA_INVALID_WORKER_ADDR)?;
260
261                Ok(Some(address))
262            }
263        }?;
264
265        Ok(VerifiedValidatorMetadataV1 {
266            sui_address: self.sui_address,
267            protocol_pubkey,
268            network_pubkey,
269            worker_pubkey,
270            proof_of_possession_bytes: self.proof_of_possession_bytes.clone(),
271            name: self.name.clone(),
272            description: self.description.clone(),
273            image_url: self.image_url.clone(),
274            project_url: self.project_url.clone(),
275            net_address,
276            p2p_address,
277            primary_address,
278            worker_address,
279            next_epoch_protocol_pubkey,
280            next_epoch_proof_of_possession: self.next_epoch_proof_of_possession.clone(),
281            next_epoch_network_pubkey,
282            next_epoch_worker_pubkey,
283            next_epoch_net_address,
284            next_epoch_p2p_address,
285            next_epoch_primary_address,
286            next_epoch_worker_address,
287        })
288    }
289}

536    fn advance_epoch_safe_mode(&mut self, params: &AdvanceEpochParams) {
537        self.epoch = params.epoch;
538        self.safe_mode = true;
539        self.safe_mode_storage_rewards
540            .deposit_for_safe_mode(params.storage_charge);
541        self.safe_mode_storage_rebates += params.storage_rebate;
542        self.safe_mode_computation_rewards
543            .deposit_for_safe_mode(params.computation_charge);
544        self.safe_mode_non_refundable_storage_fee += params.non_refundable_storage_fee;
545        self.epoch_start_timestamp_ms = params.epoch_start_timestamp_ms;
546        self.protocol_version = params.next_protocol_version.as_u64();
547    }

549    fn get_current_epoch_committee(&self) -> CommitteeWithNetworkMetadata {
550        let validators = self
551            .validators
552            .active_validators
553            .iter()
554            .map(|validator| {
555                let verified_metadata = validator.verified_metadata();
556                let name = verified_metadata.sui_pubkey_bytes();
557                (
558                    name,
559                    (
560                        validator.voting_power,
561                        NetworkMetadata {
562                            network_address: verified_metadata.net_address.clone(),
563                            narwhal_primary_address: verified_metadata.primary_address.clone(),
564                            network_public_key: Some(verified_metadata.network_pubkey.clone()),
565                        },
566                    ),
567                )
568            })
569            .collect();
570        CommitteeWithNetworkMetadata::new(self.epoch, validators)
571    }
572
573    fn get_pending_active_validators<S: ObjectStore + ?Sized>(
574        &self,
575        object_store: &S,
576    ) -> Result<Vec<SuiValidatorSummary>, SuiError> {
577        let table_id = self.validators.pending_active_validators.contents.id;
578        let table_size = self.validators.pending_active_validators.contents.size;
579        let validators: Vec<ValidatorV1> =
580            get_validators_from_table_vec(object_store, table_id, table_size)?;
581        Ok(validators
582            .into_iter()
583            .map(|v| v.into_sui_validator_summary())
584            .collect())
585    }

587    fn into_epoch_start_state(self) -> EpochStartSystemState {
588        EpochStartSystemState::new_v1(
589            self.epoch,
590            self.protocol_version,
591            self.reference_gas_price,
592            self.safe_mode,
593            self.epoch_start_timestamp_ms,
594            self.parameters.epoch_duration_ms,
595            self.validators
596                .active_validators
597                .iter()
598                .map(|validator| {
599                    let metadata = validator.verified_metadata();
600                    EpochStartValidatorInfoV1 {
601                        sui_address: metadata.sui_address,
