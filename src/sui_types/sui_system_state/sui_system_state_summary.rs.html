24/// This is the JSON-RPC type for the SUI system state object.
25/// It flattens all fields to make them top-level fields such that it as minimum
26/// dependencies to the internal data structures of the SUI system state type.
27
28#[serde_as]
29#[derive(Debug, Serialize, Deserialize, Clone, JsonSchema)]
30#[serde(rename_all = "camelCase")]
31pub struct SuiSystemStateSummary {
32    /// The current epoch ID, starting from 0.
33    #[schemars(with = "BigInt<u64>")]
34    #[serde_as(as = "Readable<BigInt<u64>, _>")]
35    pub epoch: u64,
36    /// The current protocol version, starting from 1.
37    #[schemars(with = "BigInt<u64>")]
38    #[serde_as(as = "Readable<BigInt<u64>, _>")]
39    pub protocol_version: u64,
40    /// The current version of the system state data structure type.
41    #[schemars(with = "BigInt<u64>")]
42    #[serde_as(as = "Readable<BigInt<u64>, _>")]
43    pub system_state_version: u64,
44    /// The storage rebates of all the objects on-chain stored in the storage fund.
45    #[schemars(with = "BigInt<u64>")]
46    #[serde_as(as = "Readable<BigInt<u64>, _>")]
47    pub storage_fund_total_object_storage_rebates: u64,
48    /// The non-refundable portion of the storage fund coming from storage reinvestment, non-refundable
49    /// storage rebates and any leftover staking rewards.
50    #[schemars(with = "BigInt<u64>")]
51    #[serde_as(as = "Readable<BigInt<u64>, _>")]
52    pub storage_fund_non_refundable_balance: u64,
53    /// The reference gas price for the current epoch.
54    #[schemars(with = "BigInt<u64>")]
55    #[serde_as(as = "Readable<BigInt<u64>, _>")]
56    pub reference_gas_price: u64,
57    /// Whether the system is running in a downgraded safe mode due to a non-recoverable bug.
58    /// This is set whenever we failed to execute advance_epoch, and ended up executing advance_epoch_safe_mode.
59    /// It can be reset once we are able to successfully execute advance_epoch.
60    pub safe_mode: bool,
61    /// Amount of storage rewards accumulated (and not yet distributed) during safe mode.
62    #[schemars(with = "BigInt<u64>")]
63    #[serde_as(as = "Readable<BigInt<u64>, _>")]
64    pub safe_mode_storage_rewards: u64,
65    /// Amount of computation rewards accumulated (and not yet distributed) during safe mode.
66    #[schemars(with = "BigInt<u64>")]
67    #[serde_as(as = "Readable<BigInt<u64>, _>")]
68    pub safe_mode_computation_rewards: u64,
69    /// Amount of storage rebates accumulated (and not yet burned) during safe mode.
70    #[schemars(with = "BigInt<u64>")]
71    #[serde_as(as = "Readable<BigInt<u64>, _>")]
72    pub safe_mode_storage_rebates: u64,
73    /// Amount of non-refundable storage fee accumulated during safe mode.
74    #[schemars(with = "BigInt<u64>")]
75    #[serde_as(as = "Readable<BigInt<u64>, _>")]
76    pub safe_mode_non_refundable_storage_fee: u64,
77    /// Unix timestamp of the current epoch start
78    #[schemars(with = "BigInt<u64>")]
79    #[serde_as(as = "Readable<BigInt<u64>, _>")]
80    pub epoch_start_timestamp_ms: u64,
81
82    // System parameters
83    /// The duration of an epoch, in milliseconds.
84    #[schemars(with = "BigInt<u64>")]
85    #[serde_as(as = "Readable<BigInt<u64>, _>")]
86    pub epoch_duration_ms: u64,
87
88    /// The starting epoch in which stake subsidies start being paid out
89    #[schemars(with = "BigInt<u64>")]
90    #[serde_as(as = "Readable<BigInt<u64>, _>")]
91    pub stake_subsidy_start_epoch: u64,
92
93    /// Maximum number of active validators at any moment.
94    /// We do not allow the number of validators in any epoch to go above this.
95    #[schemars(with = "BigInt<u64>")]
96    #[serde_as(as = "Readable<BigInt<u64>, _>")]
97    pub max_validator_count: u64,
98
99    /// Lower-bound on the amount of stake required to become a validator.
100    #[schemars(with = "BigInt<u64>")]
101    #[serde_as(as = "Readable<BigInt<u64>, _>")]
102    pub min_validator_joining_stake: u64,
103
104    /// Validators with stake amount below `validator_low_stake_threshold` are considered to
105    /// have low stake and will be escorted out of the validator set after being below this
106    /// threshold for more than `validator_low_stake_grace_period` number of epochs.
107    #[schemars(with = "BigInt<u64>")]
108    #[serde_as(as = "Readable<BigInt<u64>, _>")]
109    pub validator_low_stake_threshold: u64,
110
111    /// Validators with stake below `validator_very_low_stake_threshold` will be removed
112    /// immediately at epoch change, no grace period.
113    #[schemars(with = "BigInt<u64>")]
114    #[serde_as(as = "Readable<BigInt<u64>, _>")]
115    pub validator_very_low_stake_threshold: u64,
116
117    /// A validator can have stake below `validator_low_stake_threshold`
118    /// for this many epochs before being kicked out.
119    #[schemars(with = "BigInt<u64>")]
120    #[serde_as(as = "Readable<BigInt<u64>, _>")]
121    pub validator_low_stake_grace_period: u64,
122
123    // Stake subsidy information
124    /// Balance of SUI set aside for stake subsidies that will be drawn down over time.
125    #[schemars(with = "BigInt<u64>")]
126    #[serde_as(as = "Readable<BigInt<u64>, _>")]
127    pub stake_subsidy_balance: u64,
128    /// This counter may be different from the current epoch number if
129    /// in some epochs we decide to skip the subsidy.
130    #[schemars(with = "BigInt<u64>")]
131    #[serde_as(as = "Readable<BigInt<u64>, _>")]
132    pub stake_subsidy_distribution_counter: u64,
133    /// The amount of stake subsidy to be drawn down per epoch.
134    /// This amount decays and decreases over time.
135    #[schemars(with = "BigInt<u64>")]
136    #[serde_as(as = "Readable<BigInt<u64>, _>")]
137    pub stake_subsidy_current_distribution_amount: u64,
138    /// Number of distributions to occur before the distribution amount decays.
139    #[schemars(with = "BigInt<u64>")]
140    #[serde_as(as = "Readable<BigInt<u64>, _>")]
141    pub stake_subsidy_period_length: u64,
142    /// The rate at which the distribution amount decays at the end of each
143    /// period. Expressed in basis points.
144    pub stake_subsidy_decrease_rate: u16,
145
146    // Validator set
147    /// Total amount of stake from all active validators at the beginning of the epoch.
148    #[schemars(with = "BigInt<u64>")]
149    #[serde_as(as = "Readable<BigInt<u64>, _>")]
150    pub total_stake: u64,
151    /// The list of active validators in the current epoch.
152    pub active_validators: Vec<SuiValidatorSummary>,
153    /// ID of the object that contains the list of new validators that will join at the end of the epoch.
154    pub pending_active_validators_id: ObjectID,
155    /// Number of new validators that will join at the end of the epoch.
156    #[schemars(with = "BigInt<u64>")]
157    #[serde_as(as = "Readable<BigInt<u64>, _>")]
158    pub pending_active_validators_size: u64,
159    /// Removal requests from the validators. Each element is an index
160    /// pointing to `active_validators`.
161    #[schemars(with = "Vec<BigInt<u64>>")]
162    #[serde_as(as = "Vec<Readable<BigInt<u64>, _>>")]
163    pub pending_removals: Vec<u64>,
164    /// ID of the object that maps from staking pool's ID to the sui address of a validator.
165    pub staking_pool_mappings_id: ObjectID,
166    /// Number of staking pool mappings.
167    #[schemars(with = "BigInt<u64>")]
168    #[serde_as(as = "Readable<BigInt<u64>, _>")]
169    pub staking_pool_mappings_size: u64,
170    /// ID of the object that maps from a staking pool ID to the inactive validator that has that pool as its staking pool.
171    pub inactive_pools_id: ObjectID,
172    /// Number of inactive staking pools.
173    #[schemars(with = "BigInt<u64>")]
174    #[serde_as(as = "Readable<BigInt<u64>, _>")]
175    pub inactive_pools_size: u64,
176    /// ID of the object that stores preactive validators, mapping their addresses to their `Validator` structs.
177    pub validator_candidates_id: ObjectID,
178    /// Number of preactive validators.
179    #[schemars(with = "BigInt<u64>")]
180    #[serde_as(as = "Readable<BigInt<u64>, _>")]
181    pub validator_candidates_size: u64,
182    /// Map storing the number of epochs for which each validator has been below the low stake threshold.
183    #[schemars(with = "Vec<(SuiAddress, BigInt<u64>)>")]
184    #[serde_as(as = "Vec<(_, Readable<BigInt<u64>, _>)>")]
185    pub at_risk_validators: Vec<(SuiAddress, u64)>,
186    /// A map storing the records of validator reporting each other.
187    pub validator_report_records: Vec<(SuiAddress, Vec<SuiAddress>)>,
188}

190impl SuiSystemStateSummary {
191    pub fn get_sui_committee_for_benchmarking(&self) -> CommitteeWithNetworkMetadata {
192        let validators = self
193            .active_validators
194            .iter()
195            .map(|validator| {
196                let name = AuthorityName::from_bytes(&validator.protocol_pubkey_bytes).unwrap();
197                (
198                    name,
199                    (
200                        validator.voting_power,
201                        NetworkMetadata {
202                            network_address: Multiaddr::try_from(validator.net_address.clone())
203                                .unwrap(),
204                            narwhal_primary_address: Multiaddr::try_from(
205                                validator.primary_address.clone(),
206                            )
207                            .unwrap(),
208                            network_public_key: NetworkPublicKey::from_bytes(
209                                &validator.network_pubkey_bytes,
210                            )
211                            .ok(),
212                        },
213                    ),
214                )
215            })
216            .collect();
217        CommitteeWithNetworkMetadata::new(self.epoch, validators)
218    }
219
220    pub fn get_committee_authority_names_to_hostnames(&self) -> HashMap<AuthorityName, String> {
221        self.active_validators
222            .iter()
223            .map(|validator| {
224                let name = AuthorityName::from_bytes(&validator.protocol_pubkey_bytes).unwrap();
225                let hostname = validator.name.clone();
226
227                (name, hostname)
228            })
229            .collect()
230    }
231}

431/// Given the staking pool id of a validator, return the validator's `SuiValidatorSummary`,
432/// works for validator candidates, active validators, as well as inactive validators.
433pub fn get_validator_by_pool_id<S>(
434    object_store: &S,
435    system_state: &SuiSystemState,
436    system_state_summary: &SuiSystemStateSummary,
437    pool_id: ObjectID,
438) -> Result<SuiValidatorSummary, SuiError>
439where
440    S: ObjectStore + ?Sized,
441{
442    // First try to find in active validator set.
443    let active_validator = system_state_summary
444        .active_validators
445        .iter()
446        .find(|v| v.staking_pool_id == pool_id);
447    if let Some(active) = active_validator {
448        return Ok(active.clone());
449    }
450    // Then try to find in pending active validator set.
451    let pending_active_validators = system_state.get_pending_active_validators(object_store)?;
452    let pending_active = pending_active_validators
453        .iter()
454        .find(|v| v.staking_pool_id == pool_id);
455    if let Some(pending) = pending_active {
456        return Ok(pending.clone());
457    }
458    // After that try to find in inactive pools.
459    let inactive_table_id = system_state_summary.inactive_pools_id;
460    if let Ok(inactive) =
461        get_validator_from_table(&object_store, inactive_table_id, &ID::new(pool_id))
462    {
463        return Ok(inactive);
464    }
465    // Finally look up the candidates pool.
466    let candidate_address: SuiAddress = get_dynamic_field_from_store(
467        &object_store,
468        system_state_summary.staking_pool_mappings_id,
469        &ID::new(pool_id),
470    )
471    .map_err(|err| {
472        SuiErrorKind::SuiSystemStateReadError(format!(
473            "Failed to load candidate address from pool mappings: {:?}",
474            err
475        ))
476    })?;
477    let candidate_table_id = system_state_summary.validator_candidates_id;
478    get_validator_from_table(&object_store, candidate_table_id, &candidate_address)
479}
