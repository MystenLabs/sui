17/// Trait to define the interface for how the gRPC service interacts with a  QuorumDriver or a
18/// simulated transaction executor.
19#[async_trait::async_trait]
20pub trait TransactionExecutor: Send + Sync {
21    async fn execute_transaction(
22        &self,
23        request: ExecuteTransactionRequestV3,
24        client_addr: Option<std::net::SocketAddr>,
25    ) -> Result<ExecuteTransactionResponseV3, TransactionSubmissionError>;
26
27    fn simulate_transaction(
28        &self,
29        transaction: TransactionData,
30        checks: TransactionChecks,
31        allow_mock_gas_coin: bool,
32    ) -> Result<SimulateTransactionResult, SuiError>;
33}
34
35pub struct SimulateTransactionResult {
36    pub effects: TransactionEffects,
37    pub events: Option<TransactionEvents>,
38    pub objects: ObjectSet,
39    pub execution_result: Result<Vec<ExecutionResult>, ExecutionError>,
40    pub mock_gas_id: Option<ObjectID>,
41    pub unchanged_loaded_runtime_objects: Vec<ObjectKey>,
42    pub suggested_gas_price: Option<u64>,
43}
44
45#[derive(Default, Debug, Copy, Clone)]
46pub enum TransactionChecks {
47    #[default]
48    Enabled,
49    Disabled,
50}
51
52impl TransactionChecks {
53    pub fn disabled(self) -> bool {
54        matches!(self, Self::Disabled)
55    }
56
57    pub fn enabled(self) -> bool {
58        matches!(self, Self::Enabled)
59    }
60}
