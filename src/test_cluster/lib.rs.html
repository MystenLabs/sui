340    /// Ask 2f+1 validators to close epoch actively, and wait for the entire network to reach the next
341    /// epoch. This requires waiting for both the fullnode and all validators to reach the next epoch.
342    pub async fn trigger_reconfiguration(&self) {
343        info!("Starting reconfiguration");
344        let start = Instant::now();
345
346        // Close epoch on 2f+1 validators.
347        let cur_committee = self
348            .fullnode_handle
349            .sui_node
350            .with(|node| node.state().clone_committee_for_testing());
351        let mut cur_stake = 0;
352        for node in self.swarm.active_validators() {
353            node.get_node_handle()
354                .unwrap()
355                .with_async(|node| async {
356                    node.close_epoch_for_testing().await.unwrap_or_else(|_| {
357                        fatal!(
358                            "Failed to close epoch for validator {:?}",
359                            node.state().name
360                        );
361                    });
362                    cur_stake += cur_committee.weight(&node.state().name);
363                })
364                .await;
365            if cur_stake >= cur_committee.quorum_threshold() {
366                break;
367            }
368        }
369        info!("close_epoch complete after {:?}", start.elapsed());
370
371        self.wait_for_epoch(Some(cur_committee.epoch + 1)).await;
372        self.wait_for_epoch_all_nodes(cur_committee.epoch + 1).await;
373
374        info!("reconfiguration complete after {:?}", start.elapsed());
375    }

377    /// To detect whether the network has reached such state, we use the fullnode as the
378    /// source of truth, since a fullnode only does epoch transition when the network has
379    /// done so.
380    /// If target_epoch is specified, wait until the cluster reaches that epoch.
381    /// If target_epoch is None, wait until the cluster reaches the next epoch.
382    /// Note that this function does not guarantee that every node is at the target epoch.
383    pub async fn wait_for_epoch(&self, target_epoch: Option<EpochId>) -> SuiSystemState {
384        self.wait_for_epoch_with_timeout(target_epoch, Duration::from_secs(60))
385            .await
386    }

388    pub async fn wait_for_epoch_on_node(
389        &self,
390        handle: &SuiNodeHandle,
391        target_epoch: Option<EpochId>,
392        timeout_dur: Duration,
393    ) -> SuiSystemState {
394        let mut epoch_rx = handle.with(|node| node.subscribe_to_epoch_change());
395
396        let mut state = None;
397        timeout(timeout_dur, async {
398            let epoch = handle.with(|node| node.state().epoch_store_for_testing().epoch());
399            if Some(epoch) == target_epoch {
400                return handle.with(|node| node.state().get_sui_system_state_object_for_testing().unwrap());
401            }
402            while let Ok(system_state) = epoch_rx.recv().await {
403                info!("received epoch {}", system_state.epoch());
404                state = Some(system_state.clone());
405                match target_epoch {
406                    Some(target_epoch) if system_state.epoch() >= target_epoch => {
407                        return system_state;
408                    }
409                    None => {
410                        return system_state;
411                    }
412                    _ => (),
413                }
414            }
415            unreachable!("Broken reconfig channel");
416        })
417        .await
418        .unwrap_or_else(|_| {
419            error!("Timed out waiting for cluster to reach epoch {target_epoch:?}");
420            if let Some(state) = state {
421                panic!("Timed out waiting for cluster to reach epoch {target_epoch:?}. Current epoch: {}", state.epoch());
422            }
423            panic!("Timed out waiting for cluster to target epoch {target_epoch:?}")
424        })
425    }

436    pub async fn wait_for_epoch_all_nodes(&self, target_epoch: EpochId) {
437        let handles: Vec<_> = self
438            .swarm
439            .all_nodes()
440            .map(|node| node.get_node_handle().unwrap())
441            .collect();
442        let tasks: Vec<_> = handles
443            .iter()
444            .map(|handle| {
445                handle.with_async(|node| async {
446                    let mut retries = 0;
447                    loop {
448                        let epoch = node.state().epoch_store_for_testing().epoch();
449                        if epoch == target_epoch {
450                            if let Some(agg) = node.clone_authority_aggregator() {
451                                // This is a fullnode, we need to wait for its auth aggregator to reconfigure as well.
452                                if agg.committee.epoch() == target_epoch {
453                                    break;
454                                }
455                            } else {
456                                // This is a validator, we don't need to check the auth aggregator.
457                                break;
458                            }
459                        }
460                        tokio::time::sleep(Duration::from_secs(1)).await;
461                        retries += 1;
462                        if retries % 5 == 0 {
463                            tracing::warn!(validator=?node.state().name.concise(), "Waiting for {:?} seconds to reach epoch {:?}. Currently at epoch {:?}", retries, target_epoch, epoch);
464                        }
465                    }
466                })
467            })
468            .collect();
469
470        timeout(Duration::from_secs(40), join_all(tasks))
471            .await
472            .expect("timed out waiting for reconfiguration to complete");
473    }

476    /// supported versions.
477    /// Note that we don't restart the fullnode here, and it is assumed that the fulnode supports

491            self.start_node(&authority).await;
492            info!("Restarted validator {}", authority);
493        }
494    }

515    pub async fn wait_for_authenticator_state_update(&self) {
516        timeout(
517            Duration::from_secs(60),
518            self.fullnode_handle.sui_node.with_async(|node| async move {
519                let mut txns = node.state().subscription_handler.subscribe_transactions(
520                    TransactionFilter::ChangedObject(ObjectID::from_hex_literal("0x7").unwrap()),
521                );
522                let state = node.state();
523
524                while let Some(tx) = txns.next().await {
525                    let digest = *tx.transaction_digest();
526                    let tx = state
527                        .get_transaction_cache_reader()
528                        .get_transaction_block(&digest)
529                        .unwrap();
530                    match &tx.data().intent_message().value.kind() {
531                        TransactionKind::EndOfEpochTransaction(_) => (),
532                        TransactionKind::AuthenticatorStateUpdate(_) => break,
533                        _ => panic!("{:?}", tx),
534                    }
535                }
536            }),
537        )
538        .await
539        .expect("Timed out waiting for authenticator state update");
540    }

607    /// Different from `execute_transaction` which returns RPC effects types, this function
608    /// returns raw effects, events and extra objects returned by the validators,
609    /// aggregated manually (without authority aggregator).
610    /// It also does not check whether the transaction is executed successfully.
611    /// In order to keep the fullnode up-to-date so that latter queries can read consistent
612    /// results, it calls execute_transaction_may_fail again which goes through fullnode.
613    /// This is less efficient and verbose, but can be used if more details are needed
614    /// from the execution results, and if the transaction is expected to fail.
615    pub async fn execute_transaction_return_raw_effects(
616        &self,
617        tx: Transaction,
618    ) -> anyhow::Result<(TransactionEffects, TransactionEvents)> {
619        let results = self
620            .submit_transaction_to_validators(tx.clone(), &self.get_validator_pubkeys())
621            .await?;
622        self.wallet.execute_transaction_may_fail(tx).await.unwrap();
623        Ok(results)
624    }

644    /// Execute a transaction on specified list of validators, and bypassing authority aggregator.
645    /// This allows us to obtain the return value directly from validators, so that we can access more
646    /// information directly such as the original effects, events and extra objects returned.
647    /// This also allows us to control which validator to send certificates to, which is useful in
648    /// some tests.
649    pub async fn submit_transaction_to_validators(
650        &self,
651        tx: Transaction,
652        pubkeys: &[AuthorityName],
653    ) -> anyhow::Result<(TransactionEffects, TransactionEvents)> {
654        let agg = self.authority_aggregator();
655        let certificate = agg
656            .process_transaction(tx, None)
657            .await?
658            .into_cert_for_testing();
659        let replies = loop {
660            let futures: Vec<_> = agg
661                .authority_clients
662                .iter()
663                .filter_map(|(name, client)| {
664                    if pubkeys.contains(name) {
665                        Some(client)
666                    } else {
667                        None
668                    }
669                })
670                .map(|client| {
671                    let cert = certificate.clone();
672                    async move { client.handle_certificate_v2(cert, None).await }
673                })
674                .collect();
675
676            let replies: Vec<_> = futures::future::join_all(futures)
677                .await
678                .into_iter()
679                .filter(|result| match result {
680                    Err(e) => !e.to_string().contains("deadline has elapsed"),
681                    _ => true,
682                })
683                .collect();
684
685            if !replies.is_empty() {
686                break replies;
687            }
688        };
689        let replies: SuiResult<Vec<_>> = replies.into_iter().collect();
690        let replies = replies?;
691        let mut all_effects = HashMap::new();
692        let mut all_events = HashMap::new();
693        for reply in replies {
694            let effects = reply.signed_effects.into_data();
695            all_effects.insert(effects.digest(), effects);
696            all_events.insert(reply.events.digest(), reply.events);
697            // reply.fastpath_input_objects is unused.
698        }
699        assert_eq!(all_effects.len(), 1);
700        assert_eq!(all_events.len(), 1);
701        Ok((
702            all_effects.into_values().next().unwrap(),
703            all_events.into_values().next().unwrap(),
704        ))
705    }

762pub struct RandomNodeRestarter {
763    test_cluster: Arc<TestCluster>,
764
765    // How frequently should we kill nodes
766    kill_interval: Uniform<Duration>,
767    // How long should we wait before restarting them.
768    restart_delay: Uniform<Duration>,
769
770    task_handle: Mutex<Option<JoinHandle<()>>>,
771}
772
773impl RandomNodeRestarter {
774    fn new(test_cluster: Arc<TestCluster>) -> Self {
775        Self {
776            test_cluster,
777            kill_interval: Uniform::new(Duration::from_secs(10), Duration::from_secs(11)),
778            restart_delay: Uniform::new(Duration::from_secs(1), Duration::from_secs(2)),
779            task_handle: Default::default(),
780        }
781    }
782
783    pub fn with_kill_interval_secs(mut self, a: u64, b: u64) -> Self {
784        self.kill_interval = Uniform::new(Duration::from_secs(a), Duration::from_secs(b));
785        self
786    }
787
788    pub fn with_restart_delay_secs(mut self, a: u64, b: u64) -> Self {
789        self.restart_delay = Uniform::new(Duration::from_secs(a), Duration::from_secs(b));
790        self
791    }
792
793    pub fn run(&self) {
794        let test_cluster = self.test_cluster.clone();
795        let kill_interval = self.kill_interval;
796        let restart_delay = self.restart_delay;
797        let validators = self.test_cluster.get_validator_pubkeys();
798        let mut task_handle = self.task_handle.lock().unwrap();
799        assert!(task_handle.is_none());
800        task_handle.replace(tokio::task::spawn(async move {
801            loop {
802                let delay = kill_interval.sample(&mut OsRng);
803                info!("Sleeping {delay:?} before killing a validator");
804                sleep(delay).await;
805
806                let validator = validators.choose(&mut OsRng).unwrap();
807                info!("Killing validator {:?}", validator.concise());
808                test_cluster.stop_node(validator);
809
810                let delay = restart_delay.sample(&mut OsRng);
811                info!("Sleeping {delay:?} before restarting");
812                sleep(delay).await;
813                info!("Starting validator {:?}", validator.concise());
814                test_cluster.start_node(validator).await;
815            }
816        }));
817    }
818}

820impl Drop for RandomNodeRestarter {
821    fn drop(&mut self) {
822        if let Some(handle) = self.task_handle.lock().unwrap().take() {
823            handle.abort();
824        }
825    }

1325        let mut swarm = builder.build();

1331        let wallet_path = dir.join(SUI_CLIENT_CONFIG);
1332        let keystore_path = dir.join(SUI_KEYSTORE_FILENAME);
1333
1334        swarm.config().save(network_path)?;
1335        let mut keystore = Keystore::from(FileBasedKeystore::load_or_create(&keystore_path)?);
1336        for key in &swarm.config().account_keys {
1337            keystore
1338                .import(None, SuiKeyPair::Ed25519(key.copy()))
1339                .await?;
1340        }
1341
1342        let active_address = keystore.addresses().first().cloned();
1343
1344        // Create wallet config with stated authorities port

1346            keystore: Keystore::from(FileBasedKeystore::load_or_create(&keystore_path)?),
1347            external_keys: None,
1348            envs: Default::default(),
1349            active_address,
1350            active_env: Default::default(),
1351        }
1352        .save(wallet_path)?;
1353
1354        // Return network handle
1355        Ok(swarm)
1356    }
