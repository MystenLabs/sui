21static UNIVERSE_SIZE: Lazy<usize> = Lazy::new(|| {
22    use std::env;
23
24    match env::var("UNIVERSE_SIZE") {
25        Ok(s) => match s.parse::<usize>() {
26            Ok(val) => val,
27            Err(err) => {
28                panic!("Could not parse universe size, aborting: {:?}", err);
29            }
30        },
31        Err(env::VarError::NotPresent) => 30,
32        Err(err) => {
33            panic!(
34                "Could not read universe size from the environment, aborting: {:?}",
35                err
36            );
37        }
38    }
39});

49/// Represents any sort of transaction that can be done in an account universe.
50pub trait AUTransactionGen: fmt::Debug {
51    /// Applies this transaction onto the universe, updating balances within the universe as
52    /// necessary. Returns a signed transaction that can be run on the VM and the execution status.
53    fn apply(
54        &self,
55        universe: &mut AccountUniverse,
56        exec: &mut Executor,
57    ) -> (Transaction, ExecutionResult);
58
59    /// Creates an arced version of this transaction, suitable for dynamic dispatch.
60    fn arced(self) -> Arc<dyn AUTransactionGen>
61    where
62        Self: 'static + Sized,
63    {
64        Arc::new(self)
65    }
66}
67
68impl AUTransactionGen for Arc<dyn AUTransactionGen> {
69    fn apply(
70        &self,
71        universe: &mut AccountUniverse,
72        exec: &mut Executor,
73    ) -> (Transaction, ExecutionResult) {
74        (**self).apply(universe, exec)
75    }
76}

78/// Returns a [`Strategy`] that provides a variety of balances (or transfer amounts) over a roughly
79/// logarithmic distribution.
80pub fn log_balance_strategy(min_balance: u64, max_balance: u64) -> impl Strategy<Value = u64> {
81    // The logarithmic distribution is modeled by uniformly picking from ranges of powers of 2.
82    assert!(max_balance >= min_balance, "minimum to make sense");
83    let mut strategies = vec![];
84    // Balances below and around the minimum are interesting but don't cover *every* power of 2,
85    // just those starting from the minimum.
86    let mut lower_bound: u64 = 0;
87    let mut upper_bound: u64 = min_balance;
88    loop {
89        strategies.push(lower_bound..upper_bound);
90        if upper_bound >= max_balance {
91            break;
92        }
93        lower_bound = upper_bound;
94        upper_bound = (upper_bound * 2).min(max_balance);
95    }
96    Union::new(strategies)
97}

99/// Run these transactions and verify the expected output.
100pub fn run_and_assert_universe(
101    universe: AccountUniverseGen,
102    transaction_gens: Vec<impl AUTransactionGen + Clone>,
103    executor: &mut Executor,
104) -> Result<(), TestCaseError> {
105    let mut universe = universe.setup(executor);
106    let (transactions, expected_values): (Vec<_>, Vec<_>) = transaction_gens
107        .iter()
108        .map(|transaction_gen| transaction_gen.clone().apply(&mut universe, executor))
109        .unzip();
110    let outputs = executor.execute_transactions(transactions);
111    prop_assert_eq!(outputs.len(), expected_values.len());
112
113    for (idx, (output, expected)) in outputs.iter().zip(&expected_values).enumerate() {
114        prop_assert!(
115            output == expected,
116            "unexpected status for transaction {} expected {:#?} but got {:#?}",
117            idx,
118            expected,
119            output
120        );
121    }
122    assert_accounts_match(&universe, executor)
123}

125pub fn assert_accounts_match(
126    universe: &AccountUniverse,
127    executor: &Executor,
128) -> Result<(), TestCaseError> {
129    let state = executor.state.clone();
130    let backing_package_store = state.get_backing_package_store();
131    let object_store = state.get_object_store();
132    let epoch_store = state.load_epoch_store_one_call_per_task();
133    let mut layout_resolver = epoch_store
134        .executor()
135        .type_layout_resolver(Box::new(backing_package_store.as_ref()));
136    for (idx, account) in universe.accounts().iter().enumerate() {
137        for (balance_idx, acc_object) in account.current_coins.iter().enumerate() {
138            let object = object_store.get_object(&acc_object.id()).unwrap();
139            let total_sui_value =
140                object.get_total_sui(layout_resolver.as_mut()).unwrap() - object.storage_rebate;
141            let account_balance_i = account.current_balances[balance_idx];
142            prop_assert_eq!(
143                account_balance_i,
144                total_sui_value,
145                "account {} should have correct balance {} for object {} but got {}",
146                idx,
147                total_sui_value,
148                acc_object.id(),
149                account_balance_i
150            );
151        }
152    }
153    Ok(())
154}
