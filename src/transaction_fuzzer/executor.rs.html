24pub type ExecutionResult = Result<ExecutionStatus, SuiError>;
25
26fn build_test_modules(test_dir: &str) -> (Vec<u8>, Vec<Vec<u8>>) {
27    let mut path = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
28    path.extend(["data", test_dir]);
29    let with_unpublished_deps = false;
30    let config = BuildConfig::new_for_testing();
31    let package = config.build(&path).unwrap();
32    (
33        package.get_package_digest(with_unpublished_deps).to_vec(),
34        package.get_package_bytes(with_unpublished_deps),
35    )
36}

38// We want to look for either panics (in which case we won't hit this) or invariant violations in
39// which case we want to panic.
40pub fn assert_is_acceptable_result(result: &ExecutionResult) {
41    if let Ok(
42        e @ ExecutionStatus::Failure {
43            error: ExecutionFailureStatus::InvariantViolation,
44            command: _,
45        },
46    ) = result
47    {
48        panic!("Invariant violation: {e:#?}")
49    }
50}

52#[derive(Clone)]
53pub struct Executor {
54    pub state: Arc<AuthorityState>,
55    pub rt: Arc<Runtime>,
56}
57
58impl Debug for Executor {
59    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
60        f.debug_struct("Executor").finish()
61    }
62}
63
64impl Default for Executor {
65    fn default() -> Self {
66        Self::new()
67    }
68}
69
70impl Executor {
71    pub fn new() -> Self {
72        let rt = Runtime::new().unwrap();
73        let state = rt.block_on(TestAuthorityBuilder::new().build());
74        Self {
75            state,
76            rt: Arc::new(rt),
77        }
78    }
79
80    pub fn new_with_rgp(rgp: u64) -> Self {
81        let rt = Runtime::new().unwrap();
82        let state = rt.block_on(
83            TestAuthorityBuilder::new()
84                .with_reference_gas_price(rgp)
85                .build(),
86        );
87        Self {
88            state,
89            rt: Arc::new(rt),
90        }
91    }
92
93    pub fn get_reference_gas_price(&self) -> u64 {
94        self.state.reference_gas_price_for_testing().unwrap()
95    }
96
97    pub fn add_object(&mut self, object: Object) {
98        self.rt.block_on(self.state.insert_genesis_object(object));
99    }
100
101    pub fn add_objects(&mut self, objects: &[Object]) {
102        self.rt.block_on(self.state.insert_genesis_objects(objects));
103    }
104
105    pub fn execute_transaction(&mut self, txn: Transaction) -> ExecutionResult {
106        self.rt
107            .block_on(submit_and_execute(&self.state, txn))
108            .map(|(_, effects)| effects.into_data().status().clone())
109    }
110
111    pub fn publish(
112        &mut self,
113        package_name: &str,
114        dep_ids: Vec<ObjectID>,
115        account: &mut AccountCurrent,
116    ) -> TransactionEffects {
117        let (_, modules) = build_test_modules(package_name);
118        // let gas_obj_ref = account.current_coins.last().unwrap().compute_object_reference();
119        let gas_object = account.new_gas_object(self);
120        let data = TransactionData::new_module(
121            account.initial_data.account.address,
122            gas_object.compute_object_reference(),
123            modules,
124            dep_ids,
125            PUBLISH_BUDGET,
126            1000,
127        );
128        let txn = to_sender_signed_transaction(data, &account.initial_data.account.key);
129        let effects = self
130            .rt
131            .block_on(submit_and_execute(&self.state, txn))
132            .unwrap()
133            .1
134            .into_data();
135
136        assert!(
137            matches!(effects.status(), ExecutionStatus::Success),
138            "{:?}",
139            effects.status()
140        );
141        effects
142    }
143
144    pub fn execute_transactions(
145        &mut self,
146        txn: impl IntoIterator<Item = Transaction>,
147    ) -> Vec<ExecutionResult> {
148        txn.into_iter()
149            .map(|txn| self.execute_transaction(txn))
150            .collect()
151    }
152}
