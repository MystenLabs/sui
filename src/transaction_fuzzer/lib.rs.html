27fn new_gas_coin_with_balance_and_owner(balance: u64, owner: Owner) -> Object {
28    Object::new_move(
29        MoveObject::new_gas_coin(OBJECT_START_VERSION, ObjectID::random(), balance),
30        owner,
31        TransactionDigest::genesis_marker(),
32    )
33}

35/// Given a list of gas coin owners, generate random gas data and gas coins
36/// with the given owners. When `allow_address_balance` is true, randomly
37/// generates address-balance gas payments (empty payment vector) instead.
38fn generate_random_gas_data(
39    seed: [u8; 32],
40    gas_coin_owners: Vec<Owner>, // arbitrarily generated owners, can be shared or immutable or obj-owned too
41    owned_by_sender: bool,       // whether to set owned gas coins to be owned by the sender
42    force_all_sender_owned: bool, // whether to force ALL gas coins to be owned by the sender
43    allow_address_balance: bool, // whether to sometimes generate address-balance gas payments
44) -> GasDataWithObjects {
45    let (sender, sender_key): (SuiAddress, AccountKeyPair) = get_key_pair();
46    let mut rng = StdRng::from_seed(seed);
47
48    // When allowed, randomly choose address-balance gas payment (empty payment vector)
49    if allow_address_balance && rng.gen_bool(0.5) {
50        return GasDataWithObjects {
51            gas_data: GasData {
52                payment: vec![],
53                owner: sender,
54                price: rng
55                    .gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_gas_price()),
56                budget: rng
57                    .gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_tx_gas()),
58            },
59            objects: vec![],
60            sender_key,
61        };
62    }
63
64    let mut gas_objects = vec![];
65    let mut object_refs = vec![];
66
67    let max_gas_balance = TOTAL_SUPPLY_MIST;
68
69    let total_gas_balance = rng.gen_range(0..=max_gas_balance);
70    let mut remaining_gas_balance = total_gas_balance;
71    let num_gas_objects = gas_coin_owners.len();
72    let gas_coin_owners = gas_coin_owners
73        .iter()
74        .map(|o| {
75            if force_all_sender_owned {
76                Owner::AddressOwner(sender)
77            } else if owned_by_sender {
78                match o {
79                    Owner::ObjectOwner(_) | Owner::AddressOwner(_) => Owner::AddressOwner(sender),
80                    _ => o.clone(),
81                }
82            } else {
83                o.clone()
84            }
85        })
86        .collect::<Vec<_>>();
87    for owner in gas_coin_owners.iter().take(num_gas_objects - 1) {
88        let gas_balance = rng.gen_range(0..=remaining_gas_balance);
89        let gas_object = new_gas_coin_with_balance_and_owner(gas_balance, owner.clone());
90        remaining_gas_balance -= gas_balance;
91        object_refs.push(gas_object.compute_object_reference());
92        gas_objects.push(gas_object);
93    }
94    // Put the remaining balance in the last gas object.
95    let last_gas_object = new_gas_coin_with_balance_and_owner(
96        remaining_gas_balance,
97        gas_coin_owners[num_gas_objects - 1].clone(),
98    );
99    object_refs.push(last_gas_object.compute_object_reference());
100    gas_objects.push(last_gas_object);
101
102    assert_eq!(gas_objects.len(), num_gas_objects);
103    assert_eq!(
104        gas_objects
105            .iter()
106            .map(|o| o.data.try_as_move().unwrap().get_coin_value_unsafe())
107            .sum::<u64>(),
108        total_gas_balance
109    );
110
111    GasDataWithObjects {
112        gas_data: GasData {
113            payment: object_refs,
114            owner: sender,
115            price: rng.gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_gas_price()),
116            budget: rng.gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_tx_gas()),
117        },
118        objects: gas_objects,
119        sender_key,
120    }
121}

131#[derive(Debug, Default)]
132pub struct GasDataGenConfig {
133    pub max_num_gas_objects: usize,
134    pub owned_by_sender: bool,
135    /// When true, ALL generated gas coins are forced to `AddressOwner(sender)`,
136    /// regardless of the randomly generated owner type.
137    pub force_all_sender_owned: bool,
138    /// When true, the strategy will sometimes generate address-balance gas
139    /// payments (empty `payment` vector, no gas coin objects).
140    pub allow_address_balance: bool,
141}
142
143impl GasDataGenConfig {
144    pub fn owned_by_sender_or_immut() -> Self {
145        Self {
146            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
147                .max_gas_payment_objects() as usize,
148            owned_by_sender: true,
149            force_all_sender_owned: false,
150            allow_address_balance: false,
151        }
152    }
153
154    pub fn any_owner() -> Self {
155        Self {
156            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
157                .max_gas_payment_objects() as usize,
158            owned_by_sender: false,
159            force_all_sender_owned: false,
160            allow_address_balance: false,
161        }
162    }
163
164    /// All gas coins are `AddressOwner(sender)`. Use this for execution modes
165    /// that skip ownership checks (e.g. dev-inspect with `skip_checks=true`)
166    /// where non-sender-owned gas would panic during execution.
167    pub fn sender_owned_only() -> Self {
168        Self {
169            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
170                .max_gas_payment_objects() as usize,
171            owned_by_sender: true,
172            force_all_sender_owned: true,
173            allow_address_balance: false,
174        }
175    }
176
177    /// Mix of coin-based and address-balance gas payments.
178    /// Address-balance payments use an empty `payment` vector and no gas coin objects.
179    pub fn with_address_balance() -> Self {
180        Self {
181            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
182                .max_gas_payment_objects() as usize,
183            owned_by_sender: true,
184            force_all_sender_owned: false,
185            allow_address_balance: true,
186        }
187    }
188}

190impl proptest::arbitrary::Arbitrary for GasDataWithObjects {
191    type Parameters = GasDataGenConfig;
192    type Strategy = BoxedStrategy<Self>;
193
194    fn arbitrary_with(params: Self::Parameters) -> Self::Strategy {
195        (
196            any::<[u8; 32]>(),
197            vec(any::<Owner>(), 1..=params.max_num_gas_objects),
198        )
199            .prop_map(move |(seed, owners)| {
200                generate_random_gas_data(
201                    seed,
202                    owners,
203                    params.owned_by_sender,
204                    params.force_all_sender_owned,
205                    params.allow_address_balance,
206                )
207            })
208            .boxed()
209    }
210}

218/// Run a proptest test with give number of test cases, a strategy for something and a test function testing that something
219/// with an `Arc<AuthorityState>`.
220pub fn run_proptest<D>(
221    num_test_cases: u32,
222    strategy: impl Strategy<Value = D>,
223    test_fn: impl Fn(D, Executor) -> Result<(), TestCaseError>,
224) where
225    D: Debug + 'static,
226{
227    run_proptest_with_executor(num_test_cases, Executor::new(), strategy, test_fn)
228}

230/// Same as `run_proptest` but uses a fullnode executor (required for dry-run and dev-inspect).
231pub fn run_proptest_with_fullnode<D>(
232    num_test_cases: u32,
233    strategy: impl Strategy<Value = D>,
234    test_fn: impl Fn(D, Executor) -> Result<(), TestCaseError>,
235) where
236    D: Debug + 'static,
237{
238    run_proptest_with_executor(num_test_cases, Executor::new_fullnode(), strategy, test_fn)
239}
240
241fn run_proptest_with_executor<D>(
242    num_test_cases: u32,
243    executor: Executor,
244    strategy: impl Strategy<Value = D>,
245    test_fn: impl Fn(D, Executor) -> Result<(), TestCaseError>,
246) where
247    D: Debug + 'static,
248{
249    let mut runner = TestRunner::new(ProptestConfig {
250        cases: num_test_cases,
251        ..Default::default()
252    });
253    let strategy_with_authority = strategy.prop_map(|data| TestData {
254        data,
255        executor: executor.clone(),
256    });
257    let result = runner.run(&strategy_with_authority, |test_data| {
258        test_fn(test_data.data, test_data.executor)
259    });
260    if result.is_err() {
261        panic!("test failed: {:?}", result);
262    }
