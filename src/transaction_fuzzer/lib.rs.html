27fn new_gas_coin_with_balance_and_owner(balance: u64, owner: Owner) -> Object {
28    Object::new_move(
29        MoveObject::new_gas_coin(OBJECT_START_VERSION, ObjectID::random(), balance),
30        owner,
31        TransactionDigest::genesis_marker(),
32    )
33}

35/// Given a list of gas coin owners, generate random gas data and gas coins
36/// with the given owners.
37fn generate_random_gas_data(
38    seed: [u8; 32],
39    gas_coin_owners: Vec<Owner>, // arbitrarily generated owners, can be shared or immutable or obj-owned too
40    owned_by_sender: bool,       // whether to set owned gas coins to be owned by the sender
41) -> GasDataWithObjects {
42    let (sender, sender_key): (SuiAddress, AccountKeyPair) = get_key_pair();
43    let mut rng = StdRng::from_seed(seed);
44    let mut gas_objects = vec![];
45    let mut object_refs = vec![];
46
47    let max_gas_balance = TOTAL_SUPPLY_MIST;
48
49    let total_gas_balance = rng.gen_range(0..=max_gas_balance);
50    let mut remaining_gas_balance = total_gas_balance;
51    let num_gas_objects = gas_coin_owners.len();
52    let gas_coin_owners = gas_coin_owners
53        .iter()
54        .map(|o| match o {
55            Owner::ObjectOwner(_) | Owner::AddressOwner(_) if owned_by_sender => {
56                Owner::AddressOwner(sender)
57            }
58            _ => o.clone(),
59        })
60        .collect::<Vec<_>>();
61    for owner in gas_coin_owners.iter().take(num_gas_objects - 1) {
62        let gas_balance = rng.gen_range(0..=remaining_gas_balance);
63        let gas_object = new_gas_coin_with_balance_and_owner(gas_balance, owner.clone());
64        remaining_gas_balance -= gas_balance;
65        object_refs.push(gas_object.compute_object_reference());
66        gas_objects.push(gas_object);
67    }
68    // Put the remaining balance in the last gas object.
69    let last_gas_object = new_gas_coin_with_balance_and_owner(
70        remaining_gas_balance,
71        gas_coin_owners[num_gas_objects - 1].clone(),
72    );
73    object_refs.push(last_gas_object.compute_object_reference());
74    gas_objects.push(last_gas_object);
75
76    assert_eq!(gas_objects.len(), num_gas_objects);
77    assert_eq!(
78        gas_objects
79            .iter()
80            .map(|o| o.data.try_as_move().unwrap().get_coin_value_unsafe())
81            .sum::<u64>(),
82        total_gas_balance
83    );
84
85    GasDataWithObjects {
86        gas_data: GasData {
87            payment: object_refs,
88            owner: sender,
89            price: rng.gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_gas_price()),
90            budget: rng.gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_tx_gas()),
91        },
92        objects: gas_objects,
93        sender_key,
94    }
95}

97/// Need to have a wrapper struct here so we can implement Arbitrary for it.
98#[derive(Debug)]
99pub struct GasDataWithObjects {
100    pub gas_data: GasData,
101    pub sender_key: AccountKeyPair,
102    pub objects: Vec<Object>,
103}

105#[derive(Debug, Default)]
106pub struct GasDataGenConfig {
107    pub max_num_gas_objects: usize,
108    pub owned_by_sender: bool,
109}
110
111impl GasDataGenConfig {
112    pub fn owned_by_sender_or_immut() -> Self {
113        Self {
114            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
115                .max_gas_payment_objects() as usize,
116            owned_by_sender: true,
117        }
118    }
119
120    pub fn any_owner() -> Self {
121        Self {
122            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
123                .max_gas_payment_objects() as usize,
124            owned_by_sender: false,
125        }
126    }
127}
128
129impl proptest::arbitrary::Arbitrary for GasDataWithObjects {
130    type Parameters = GasDataGenConfig;
131    type Strategy = BoxedStrategy<Self>;
132
133    fn arbitrary_with(params: Self::Parameters) -> Self::Strategy {
134        (
135            any::<[u8; 32]>(),
136            vec(any::<Owner>(), 1..=params.max_num_gas_objects),
137        )
138            .prop_map(move |(seed, owners)| {
139                generate_random_gas_data(seed, owners, params.owned_by_sender)
140            })
141            .boxed()
142    }
143}

145#[derive(Clone, Debug)]
146pub struct TestData<D> {
147    pub data: D,
148    pub executor: Executor,
149}

151/// Run a proptest test with give number of test cases, a strategy for something and a test function testing that something
152/// with an `Arc<AuthorityState>`.
153pub fn run_proptest<D>(
154    num_test_cases: u32,
155    strategy: impl Strategy<Value = D>,
156    test_fn: impl Fn(D, Executor) -> Result<(), TestCaseError>,
157) where
158    D: Debug + 'static,
159{
160    let mut runner = TestRunner::new(ProptestConfig {
161        cases: num_test_cases,
162        ..Default::default()
163    });
164    let executor = Executor::new();
165    let strategy_with_authority = strategy.prop_map(|data| TestData {
166        data,
167        executor: executor.clone(),
168    });
169    let result = runner.run(&strategy_with_authority, |test_data| {
170        test_fn(test_data.data, test_data.executor)
171    });
172    if result.is_err() {
173        panic!("test failed: {:?}", result);
174    }
175}
