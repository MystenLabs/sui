26fn new_gas_coin_with_balance_and_owner(balance: u64, owner: Owner) -> Object {
27    Object::new_move(
28        MoveObject::new_gas_coin(OBJECT_START_VERSION, ObjectID::random(), balance),
29        owner,
30        TransactionDigest::genesis_marker(),
31    )
32}

34/// Given a list of gas coin owners, generate random gas data and gas coins
35/// with the given owners.
36fn generate_random_gas_data(
37    seed: [u8; 32],
38    gas_coin_owners: Vec<Owner>, // arbitrarily generated owners, can be shared or immutable or obj-owned too
39    owned_by_sender: bool,       // whether to set owned gas coins to be owned by the sender
40) -> GasDataWithObjects {
41    let (sender, sender_key): (SuiAddress, AccountKeyPair) = get_key_pair();
42    let mut rng = StdRng::from_seed(seed);
43    let mut gas_objects = vec![];
44    let mut object_refs = vec![];
45
46    let max_gas_balance = TOTAL_SUPPLY_MIST;
47
48    let total_gas_balance = rng.gen_range(0..=max_gas_balance);
49    let mut remaining_gas_balance = total_gas_balance;
50    let num_gas_objects = gas_coin_owners.len();
51    let gas_coin_owners = gas_coin_owners
52        .iter()
53        .map(|o| match o {
54            Owner::ObjectOwner(_) | Owner::AddressOwner(_) if owned_by_sender => {
55                Owner::AddressOwner(sender)
56            }
57            _ => o.clone(),
58        })
59        .collect::<Vec<_>>();
60    for owner in gas_coin_owners.iter().take(num_gas_objects - 1) {
61        let gas_balance = rng.gen_range(0..=remaining_gas_balance);
62        let gas_object = new_gas_coin_with_balance_and_owner(gas_balance, owner.clone());
63        remaining_gas_balance -= gas_balance;
64        object_refs.push(gas_object.compute_object_reference());
65        gas_objects.push(gas_object);
66    }
67    // Put the remaining balance in the last gas object.
68    let last_gas_object = new_gas_coin_with_balance_and_owner(
69        remaining_gas_balance,
70        gas_coin_owners[num_gas_objects - 1].clone(),
71    );
72    object_refs.push(last_gas_object.compute_object_reference());
73    gas_objects.push(last_gas_object);
74
75    assert_eq!(gas_objects.len(), num_gas_objects);
76    assert_eq!(
77        gas_objects
78            .iter()
79            .map(|o| o.data.try_as_move().unwrap().get_coin_value_unsafe())
80            .sum::<u64>(),
81        total_gas_balance
82    );
83
84    GasDataWithObjects {
85        gas_data: GasData {
86            payment: object_refs,
87            owner: sender,
88            price: rng.gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_gas_price()),
89            budget: rng.gen_range(0..=ProtocolConfig::get_for_max_version_UNSAFE().max_tx_gas()),
90        },
91        objects: gas_objects,
92        sender_key,
93    }
94}

96/// Need to have a wrapper struct here so we can implement Arbitrary for it.
97#[derive(Debug)]
98pub struct GasDataWithObjects {
99    pub gas_data: GasData,
100    pub sender_key: AccountKeyPair,
101    pub objects: Vec<Object>,
102}

104#[derive(Debug, Default)]
105pub struct GasDataGenConfig {
106    pub max_num_gas_objects: usize,
107    pub owned_by_sender: bool,
108}
109
110impl GasDataGenConfig {
111    pub fn owned_by_sender_or_immut() -> Self {
112        Self {
113            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
114                .max_gas_payment_objects() as usize,
115            owned_by_sender: true,
116        }
117    }
118
119    pub fn any_owner() -> Self {
120        Self {
121            max_num_gas_objects: ProtocolConfig::get_for_max_version_UNSAFE()
122                .max_gas_payment_objects() as usize,
123            owned_by_sender: false,
124        }
125    }
126}
127
128impl proptest::arbitrary::Arbitrary for GasDataWithObjects {
129    type Parameters = GasDataGenConfig;
130    type Strategy = BoxedStrategy<Self>;
131
132    fn arbitrary_with(params: Self::Parameters) -> Self::Strategy {
133        (
134            any::<[u8; 32]>(),
135            vec(any::<Owner>(), 1..=params.max_num_gas_objects),
136        )
137            .prop_map(move |(seed, owners)| {
138                generate_random_gas_data(seed, owners, params.owned_by_sender)
139            })
140            .boxed()
141    }
142}

144#[derive(Clone, Debug)]
145pub struct TestData<D> {
146    pub data: D,
147    pub executor: Executor,
148}

150/// Run a proptest test with give number of test cases, a strategy for something and a test function testing that something
151/// with an `Arc<AuthorityState>`.
152pub fn run_proptest<D>(
153    num_test_cases: u32,
154    strategy: impl Strategy<Value = D>,
155    test_fn: impl Fn(D, Executor) -> Result<(), TestCaseError>,
156) where
157    D: Debug + 'static,
158{
159    let mut runner = TestRunner::new(ProptestConfig {
160        cases: num_test_cases,
161        ..Default::default()
162    });
163    let executor = Executor::new();
164    let strategy_with_authority = strategy.prop_map(|data| TestData {
165        data,
166        executor: executor.clone(),
167    });
168    let result = runner.run(&strategy_with_authority, |test_data| {
169        test_fn(test_data.data, test_data.executor)
170    });
171    if result.is_err() {
172        panic!("test failed: {:?}", result);
173    }
174}
