19const MAX_NUM_GAS_OBJS: usize = 1024_usize;

21pub fn gen_valid_during_expiration() -> impl Strategy<Value = TransactionExpiration> {
22    (
23        proptest::option::of(any::<u64>()),
24        proptest::option::of(any::<u64>()),
25        proptest::option::of(any::<u64>()),
26        proptest::option::of(any::<u64>()),
27        any::<u32>(),
28    )
29        .prop_map(
30            |(min_epoch, max_epoch, min_timestamp, max_timestamp, nonce)| {
31                TransactionExpiration::ValidDuring {
32                    min_epoch,
33                    max_epoch,
34                    min_timestamp,
35                    max_timestamp,
36                    chain: ChainIdentifier::random(),
37                    nonce,
38                }
39            },
40        )
41}

43pub fn gen_transaction_expiration_with_bound(
44    max_epoch: u64,
45) -> impl Strategy<Value = TransactionExpiration> {
46    prop_oneof![
47        Just(TransactionExpiration::None),
48        (0u64..=max_epoch).prop_map(TransactionExpiration::Epoch),
49        gen_valid_during_expiration(),
50    ]
51}
52
53pub fn gen_transaction_expiration() -> impl Strategy<Value = TransactionExpiration> {
54    prop_oneof![
55        Just(TransactionExpiration::None),
56        (0u64..=u64::MAX).prop_map(TransactionExpiration::Epoch),
57        gen_valid_during_expiration(),
58    ]
59}

61pub fn gen_object_ref() -> impl Strategy<Value = ObjectRef> {
62    (
63        any::<AccountAddress>(),
64        any::<SequenceNumber>(),
65        any::<[u8; 32]>(),
66    )
67        .prop_map(move |(addr, seq, seed)| {
68            (ObjectID::from_address(addr), seq, ObjectDigest::new(seed))
69        })
70}
71
72pub fn gen_gas_data(sender: SuiAddress) -> impl Strategy<Value = GasData> {
73    (
74        vec(gen_object_ref(), 0..MAX_NUM_GAS_OBJS),
75        gas_price_selection_strategy(),
76        gas_budget_selection_strategy(),
77    )
78        .prop_map(move |(obj_refs, price, budget)| GasData {
79            payment: obj_refs,
80            owner: sender,
81            price,
82            budget,
83        })
84}
85
86pub fn gen_transaction_kind() -> impl Strategy<Value = TransactionKind> {
87    (vec(gen_type_tag(), 0..10))
88        .prop_map(pt_for_tags)
89        .prop_map(TransactionKind::ProgrammableTransaction)
90}
91
92pub fn transaction_data_gen(sender: SuiAddress) -> impl Strategy<Value = TransactionData> {
93    TransactionDataGenBuilder::new(sender)
94        .kind(gen_transaction_kind())
95        .gas_data(gen_gas_data(sender))
96        .expiration(gen_transaction_expiration())
97        .finish()
98}

100pub struct TransactionDataGenBuilder<
101    K: Strategy<Value = TransactionKind>,
102    G: Strategy<Value = GasData>,
103    E: Strategy<Value = TransactionExpiration>,
104> {
105    pub kind: Option<K>,
106    pub sender: SuiAddress,
107    pub gas_data: Option<G>,
108    pub expiration: Option<E>,
109}
110
111impl<
112    K: Strategy<Value = TransactionKind>,
113    G: Strategy<Value = GasData>,
114    E: Strategy<Value = TransactionExpiration>,
115> TransactionDataGenBuilder<K, G, E>
116{
117    pub fn new(sender: SuiAddress) -> Self {
118        Self {
119            kind: None,
120            sender,
121            gas_data: None,
122            expiration: None,
123        }
124    }
125
126    pub fn kind(mut self, kind: K) -> Self {
127        self.kind = Some(kind);
128        self
129    }
130
131    pub fn gas_data(mut self, gas_data: G) -> Self {
132        self.gas_data = Some(gas_data);
133        self
134    }
135
136    pub fn expiration(mut self, expiration: E) -> Self {
137        self.expiration = Some(expiration);
138        self
139    }
140
141    pub fn finish(self) -> impl Strategy<Value = TransactionData> {
142        (
143            self.kind.expect("kind must be set"),
144            Just(self.sender),
145            self.gas_data.expect("gas_data must be set"),
146            self.expiration.expect("expiration must be set"),
147        )
148            .prop_map(|(kind, sender, gas_data, expiration)| TransactionDataV1 {
149                kind,
150                sender,
151                gas_data,
152                expiration,
153            })
154            .prop_map(TransactionData::V1)
155    }
156}
