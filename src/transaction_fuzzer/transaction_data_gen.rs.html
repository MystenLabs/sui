19const MAX_NUM_GAS_OBJS: usize = 1024_usize;
20
21pub fn gen_transaction_expiration_with_bound(
22    max_epoch: u64,
23) -> impl Strategy<Value = TransactionExpiration> {
24    prop_oneof![
25        Just(TransactionExpiration::None),
26        (0u64..=max_epoch).prop_map(TransactionExpiration::Epoch),
27    ]
28}
29
30pub fn gen_transaction_expiration() -> impl Strategy<Value = TransactionExpiration> {
31    prop_oneof![
32        Just(TransactionExpiration::None),
33        (0u64..=u64::MAX).prop_map(TransactionExpiration::Epoch),
34    ]
35}
36
37pub fn gen_object_ref() -> impl Strategy<Value = ObjectRef> {
38    (
39        any::<AccountAddress>(),
40        any::<SequenceNumber>(),
41        any::<[u8; 32]>(),
42    )
43        .prop_map(move |(addr, seq, seed)| {
44            (ObjectID::from_address(addr), seq, ObjectDigest::new(seed))
45        })
46}
47
48pub fn gen_gas_data(sender: SuiAddress) -> impl Strategy<Value = GasData> {
49    (
50        vec(gen_object_ref(), 0..MAX_NUM_GAS_OBJS),
51        gas_price_selection_strategy(),
52        gas_budget_selection_strategy(),
53    )
54        .prop_map(move |(obj_refs, price, budget)| GasData {
55            payment: obj_refs,
56            owner: sender,
57            price,
58            budget,
59        })
60}
61
62pub fn gen_transaction_kind() -> impl Strategy<Value = TransactionKind> {
63    (vec(gen_type_tag(), 0..10))
64        .prop_map(pt_for_tags)
65        .prop_map(TransactionKind::ProgrammableTransaction)
66}
67
68pub fn transaction_data_gen(sender: SuiAddress) -> impl Strategy<Value = TransactionData> {
69    TransactionDataGenBuilder::new(sender)
70        .kind(gen_transaction_kind())
71        .gas_data(gen_gas_data(sender))
72        .expiration(gen_transaction_expiration())
73        .finish()
74}
75
76pub struct TransactionDataGenBuilder<
77    K: Strategy<Value = TransactionKind>,
78    G: Strategy<Value = GasData>,
79    E: Strategy<Value = TransactionExpiration>,
80> {
81    pub kind: Option<K>,
82    pub sender: SuiAddress,
83    pub gas_data: Option<G>,
84    pub expiration: Option<E>,
85}
86
87impl<
88    K: Strategy<Value = TransactionKind>,
89    G: Strategy<Value = GasData>,
90    E: Strategy<Value = TransactionExpiration>,
91> TransactionDataGenBuilder<K, G, E>
92{
93    pub fn new(sender: SuiAddress) -> Self {
94        Self {
95            kind: None,
96            sender,
97            gas_data: None,
98            expiration: None,
99        }
100    }
101
102    pub fn kind(mut self, kind: K) -> Self {
103        self.kind = Some(kind);
104        self
105    }
106
107    pub fn gas_data(mut self, gas_data: G) -> Self {
108        self.gas_data = Some(gas_data);
109        self
110    }
111
112    pub fn expiration(mut self, expiration: E) -> Self {
113        self.expiration = Some(expiration);
114        self
115    }
116
117    pub fn finish(self) -> impl Strategy<Value = TransactionData> {
118        (
119            self.kind.expect("kind must be set"),
120            Just(self.sender),
121            self.gas_data.expect("gas_data must be set"),
122            self.expiration.expect("expiration must be set"),
123        )
124            .prop_map(|(kind, sender, gas_data, expiration)| TransactionDataV1 {
125                kind,
126                sender,
127                gas_data,
128                expiration,
129            })
130            .prop_map(TransactionData::V1)
131    }
132}
