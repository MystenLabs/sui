14/// An iterator over all key-value pairs in a data map.
15pub struct SafeIter<'a, K, V> {
16    cf_name: String,
17    db_iter: rocksdb::DBRawIteratorWithThreadMode<'a, DBWithThreadMode<MultiThreaded>>,
18    _phantom: PhantomData<(K, V)>,
19    direction: Direction,
20    is_initialized: bool,
21    _timer: Option<HistogramTimer>,
22    _perf_ctx: Option<RocksDBPerfContext>,
23    bytes_scanned: Option<Histogram>,
24    keys_scanned: Option<Histogram>,
25    db_metrics: Option<Arc<DBMetrics>>,
26    bytes_scanned_counter: usize,
27    keys_returned_counter: usize,
28}
29
30impl<'a, K: DeserializeOwned, V: DeserializeOwned> SafeIter<'a, K, V> {
31    pub(super) fn new(
32        cf_name: String,
33        db_iter: rocksdb::DBRawIteratorWithThreadMode<'a, DBWithThreadMode<MultiThreaded>>,
34        _timer: Option<HistogramTimer>,
35        _perf_ctx: Option<RocksDBPerfContext>,
36        bytes_scanned: Option<Histogram>,
37        keys_scanned: Option<Histogram>,
38        db_metrics: Option<Arc<DBMetrics>>,
39    ) -> Self {
40        Self {
41            cf_name,
42            db_iter,
43            _phantom: PhantomData,
44            direction: Direction::Forward,
45            is_initialized: false,
46            _timer,
47            _perf_ctx,
48            bytes_scanned,
49            keys_scanned,
50            db_metrics,
51            bytes_scanned_counter: 0,
52            keys_returned_counter: 0,
53        }
54    }
55}
56
57impl<K: DeserializeOwned, V: DeserializeOwned> Iterator for SafeIter<'_, K, V> {
58    type Item = Result<(K, V), TypedStoreError>;
59
60    fn next(&mut self) -> Option<Self::Item> {
61        // Implicitly set iterator to the first entry in the column family if it hasn't been initialized
62        // used for backward compatibility
63        if !self.is_initialized {
64            self.db_iter.seek_to_first();
65            self.is_initialized = true;
66        }
67        if self.db_iter.valid() {
68            let config = bincode::DefaultOptions::new()
69                .with_big_endian()
70                .with_fixint_encoding();
71            let raw_key = self
72                .db_iter
73                .key()
74                .expect("Valid iterator failed to get key");
75            let raw_value = self
76                .db_iter
77                .value()
78                .expect("Valid iterator failed to get value");
79            self.bytes_scanned_counter += raw_key.len() + raw_value.len();
80            self.keys_returned_counter += 1;
81            let key = config.deserialize(raw_key).ok();
82            let value = bcs::from_bytes(raw_value).ok();
83            match self.direction {
84                Direction::Forward => self.db_iter.next(),
85                Direction::Reverse => self.db_iter.prev(),
86            }
87            key.and_then(|k| value.map(|v| Ok((k, v))))
88        } else {
89            match self.db_iter.status() {
90                Ok(_) => None,
91                Err(err) => Some(Err(TypedStoreError::RocksDBError(format!("{err}")))),
92            }
93        }
94    }
95}

97impl<K, V> Drop for SafeIter<'_, K, V> {
98    fn drop(&mut self) {
99        if let Some(bytes_scanned) = self.bytes_scanned.take() {
100            bytes_scanned.observe(self.bytes_scanned_counter as f64);
101        }
102        if let Some(keys_scanned) = self.keys_scanned.take() {
103            keys_scanned.observe(self.keys_returned_counter as f64);
104        }
105        if let Some(db_metrics) = self.db_metrics.take() {
106            db_metrics
107                .read_perf_ctx_metrics
108                .report_metrics(&self.cf_name);
109        }
110    }
111}

113/// An iterator with a reverted direction to the original. The `RevIter`
114/// is hosting an iteration which is consuming in the opposing direction.
115/// It's not possible to do further manipulation (ex re-reverse) to the
116/// iterator.
117pub struct SafeRevIter<'a, K, V> {
118    iter: SafeIter<'a, K, V>,
119}
120
121impl<'a, K, V> SafeRevIter<'a, K, V> {
122    pub(crate) fn new(mut iter: SafeIter<'a, K, V>, upper_bound: Option<Vec<u8>>) -> Self {
123        iter.is_initialized = true;
124        iter.direction = Direction::Reverse;
125        match upper_bound {
126            None => iter.db_iter.seek_to_last(),
127            Some(key) => iter.db_iter.seek_for_prev(&key),
128        }
129        Self { iter }
130    }
131}
132
133impl<K: DeserializeOwned, V: DeserializeOwned> Iterator for SafeRevIter<'_, K, V> {
134    type Item = Result<(K, V), TypedStoreError>;
135
136    /// Will give the next item backwards
137    fn next(&mut self) -> Option<Self::Item> {
138        self.iter.next()
139    }
140}
